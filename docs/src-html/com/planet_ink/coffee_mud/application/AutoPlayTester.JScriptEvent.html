051        public LinkedList<String> bufferFill() throws IOException
052        {
053                int c;
054                final StringBuffer buf=new StringBuffer("");
055                int lastc=0;
056
057                try
058                {
059                        while((c=in.read()) >=0)
060                        {
061                                if(c==13 || c==10)
062                                {
063                                        if((c==13 && lastc != 10)
064                                        ||(c==10 && lastc != 13))
065                                        {
066                                                inbuffer.add(globalReactionary(buf.toString()));
067                                                buf.setLength(0);
068                                        }
069                                }
070                                else
071                                        buf.append((char)c);
072                                lastc=c;
073                        }
074                }
075                catch(final Exception e)
076                {
077
078                }
079                if(buf.length()>0)
080                        inbuffer.add(globalReactionary(buf.toString()));
081                return inbuffer;
082        }

101        public String[] waitFor(String regEx, int num) throws IOException
102        {
103                final long waitUntil = System.currentTimeMillis() + (60 * 1000);
104                final StringBuilder buildUp=new StringBuilder("");
105                final Pattern p=Pattern.compile(regEx);
106                while(System.currentTimeMillis() < waitUntil)
107                {
108                        bufferFill();
109                        if(inbuffer.size()==0)
110                        {
111                                s_sleep(100);
112                        }
113                        else
114                        {
115                                final String s=inbuffer.removeFirst();
116                                outbuffer.add(s);
117                                if(buildUp.length()>0)
118                                        buildUp.append(" ");
119                                buildUp.append(s);
120                                while(outbuffer.size() > 1000)
121                                        outbuffer.removeFirst();
122                                Matcher m=p.matcher(s);
123                                if(!m.matches())
124                                        m=p.matcher(buildUp.toString());
125                                if(m.matches())
126                                {
127                                        if(m.groupCount()>=num)
128                                        {
129                                                final String[] set=new String[num];
130                                                for(int i=0;i<num;i++)
131                                                        set[i]=m.group(i+1);
132                                                return set;
133                                        }
134                                        return new String[]{s};
135                                }
136                        }
137                }
138                throw new IOException("wait for "+regEx+" timed out.");
139        }

149        public boolean login()
150        {
151                try
152                {
153                        sock=new Socket(host,port);
154                        sock.setSoTimeout(100);
155                        in = new BufferedReader(new InputStreamReader(sock.getInputStream()));
156                        out = new BufferedWriter(new OutputStreamWriter(sock.getOutputStream()));

167        public String getJavaScript(String filename)
168        {
169                final StringBuilder js=new StringBuilder("");
170                try
171                {
172                        final BufferedReader br=new BufferedReader(new FileReader(filename));
173                        String s=br.readLine();
174                        while(s!=null)
175                        {
176                                if(s.trim().startsWith("//include "))
177                                        js.append(getJavaScript(s.trim().substring(10)));
178                                else
179                                        js.append(s).append("\n");
180                                s=br.readLine();
181                        }
182                        br.close();
183                }
184                catch(final Exception e)
185                {
186                        e.printStackTrace();
187                        System.exit(-1);
188                }
189                return js.toString();
190        }

192        public void run()
193        {
194                System.out.println("Executing: "+filename);
195                final String js=getJavaScript(filename);
196
197                final Context cx = Context.enter();
198                try
199                {
200                        final JScriptEvent scope = new JScriptEvent(this);
201                        cx.initStandardObjects(scope);
202                        scope.defineFunctionProperties(JScriptEvent.functions, JScriptEvent.class,
203                                                                                   ScriptableObject.DONTENUM);
204                        cx.evaluateString(scope, js.toString(),"<cmd>", 1, null);
205                }
206                catch(final Exception e)
207                {
208                        System.err.println("JSCRIPT Error: "+e.getMessage());
209                }
210                Context.exit();
211        }
