077        public static void fatalStartupError(Thread t, int type)
078        {
079                String errorInternal=null;
080                switch(type)
081                {
082                case 1:
083                        errorInternal="ERROR: initHost() will not run without properties. Exiting.";
084                        break;
085                case 2:
086                        break;
087                case 3:
088                        break;
089                case 4:
090                        errorInternal="Fatal exception. Exiting.";
091                        break;
092                case 5:
093                        errorInternal="OffLine Server did not start. Exiting.";
094                        break;
095                default:
096                        break;
097                }
098                System.out.println(errorInternal);
099                CMLib.killThread(t,500,1);
100        }

102        private static boolean initHost(Thread t)
103        {
104
105                if (!isOK)
106                {
107                        CMLib.killThread(t,500,1);
108                        return false;
109                }
110
111                while (!serverIsRunning && isOK)
112                {
113                        CMLib.s_sleep(1000);
114                }
115                if (!isOK)
116                {
117                        fatalStartupError(t,5);
118                        return false;
119                }
120
121                for(int i=0;i<mudThreads.size();i++)
122                        mudThreads.elementAt(i).acceptConnections=true;
123                System.out.println("Initialization complete.");
124                return true;
125        }

127        private void closeSocks(Socket sock, BufferedReader in, PrintWriter out)
128        {
129                try
130                {
131                        if(sock!=null)
132                        {
133                                if(out!=null)
134                                        out.flush();
135                                sock.shutdownInput();
136                                sock.shutdownOutput();
137                                if(out!=null)
138                                        out.close();
139                                sock.close();
140                        }
141                        in=null;
142                        out=null;
143                        sock=null;
144                }
145                catch(final IOException e)
146                {
147                }
148        }

150        public StringBuffer getFile(String fileName)
151        {
152                StringBuffer offLineText=(StringBuffer)Resources.getResource(fileName);
153                if(offLineText==null)
154                {
155                        offLineText=new StringBuffer("");
156                        FileInputStream fin = null;
157                        try
158                        {
159                                fin = new FileInputStream(fileName);
160                                while(fin.available()>0)
161                                        offLineText.append((char)fin.read());
162                                Resources.submitResource(fileName,offLineText);
163                        }
164                        catch (final Exception e)
165                        {
166                                e.printStackTrace();
167                        }
168                        finally
169                        {
170                                try
171                                {
172                                        if ( fin != null )
173                                        {
174                                                fin.close();
175                                                fin = null;
176                                        }
177                                }
178                                catch( final IOException ignore )
179                                {
180
181                                }
182                        }
183                }
184                return offLineText;
185        }

187        @Override
188        public void acceptConnection(Socket sock)
189        throws SocketException, IOException
190{
191                sock.setSoLinger(true,3);
192                state=1;
193
194                if (acceptConnections)
195                {
196                        String address="unknown";
197                        try
198                        {
199                                address=sock.getInetAddress().getHostAddress().trim();
200                        }
201                        catch(final Exception e)
202                        {
203                        }
204                        System.out.println("Connection from "+address+": "+port);
205                        // now see if they are banned!
206                        int proceed=0;
207
208                        int numAtThisAddress=0;
209                        final long ConnectionWindow=(180*1000);
210                        final long LastConnectionDelay=(5*60*1000);
211                        boolean anyAtThisAddress=false;
212                        final int maxAtThisAddress=6;
213                        try
214                        {
215                                for(int a=accessed.size()-1;a>=0;a--)
216                                {
217                                        if((((Long)accessed.elementAt(a,2)).longValue()+LastConnectionDelay)<System.currentTimeMillis())
218                                                accessed.removeElementAt(a);
219                                        else
220                                        if(((String)accessed.elementAt(a,1)).trim().equalsIgnoreCase(address))
221                                        {
222                                                anyAtThisAddress=true;
223                                                if((((Long)accessed.elementAt(a,2)).longValue()+ConnectionWindow)>System.currentTimeMillis())
224                                                        numAtThisAddress++;
225                                        }
226                                }
227                                if(autoblocked.contains(address.toUpperCase()))
228                                {
229                                        if(!anyAtThisAddress)
230                                                autoblocked.remove(address.toUpperCase());
231                                        else
232                                                proceed=2;
233                                }
234                                else
235                                if(numAtThisAddress>=maxAtThisAddress)
236                                {
237                                        autoblocked.addElement(address.toUpperCase());
238                                        proceed=2;
239                                }
240                        }
241                        catch(final java.lang.ArrayIndexOutOfBoundsException e)
242                        {
243                        }
244
245                        accessed.addElement(address,Long.valueOf(System.currentTimeMillis()));
246                        if(proceed!=0)
247                        {
248                                System.out.println("Blocking a connection from "+address+" on port "+port);
249                                final PrintWriter out = new PrintWriter(sock.getOutputStream());
250                                out.println("\n\rOFFLINE: Blocked\n\r");
251                                out.flush();
252                                if(proceed==2)
253                                        out.println("\n\rYour address has been blocked temporarily due to excessive invalid connections.  Please try back in " + (LastConnectionDelay/60000) + " minutes, and not before.\n\r\n\r");
254                                else
255                                        out.println("\n\rYou are unwelcome.  No one likes you here. Go away.\n\r\n\r");
256                                out.flush();
257                                out.close();
258                                sock = null;
259                        }
260                        else
261                        {
262                                state=2;
263                                final String fileName="resources"+File.separator+"text"+File.separator+"down.txt";
264                                final StringBuffer offLineText=getFile(fileName);
265                                try
266                                {
267                                        sock.setSoTimeout(300);
268                                        final OutputStream rawout=sock.getOutputStream();
269                                        final InputStream rawin=sock.getInputStream();
270                                        rawout.write('\n');
271                                        rawout.write('\n');
272                                        rawout.flush();
273
274                                        //out = new PrintWriter(new BufferedWriter(new OutputStreamWriter(rawout, "UTF-8")));
275                                        //in = new BufferedReader(new InputStreamReader(rawin, "UTF-8"));
276                                        BufferedReader in;
277                                        PrintWriter out;
278                                        out = new PrintWriter(new OutputStreamWriter(rawout,"iso-8859-1"));
279                                        in = new BufferedReader(new InputStreamReader(rawin,"iso-8859-1"));
280
281                                        if(offLineText!=null)
282                                                out.print(offLineText);
283                                        out.flush();
284                                        CMLib.s_sleep(250);
285                                        closeSocks(sock,in,out);
286                                }
287                                catch(final SocketException e)
288                                {
289                                }
290                                catch(final IOException e)
291                                {
292                                }
293                                closeSocks(sock,null,null);
294                                sock=null;
295                        }
296                }
297                else
298                {
299                        final String fileName="resources"+File.separator+"text"+File.separator+"offline.txt";
300                        final StringBuffer rejectText=getFile(fileName);
301                        final PrintWriter out = new PrintWriter(sock.getOutputStream());
302                        out.flush();
303                        out.println(rejectText);
304                        out.flush();
305                        out.close();
306                        CMLib.s_sleep(250);
307                        sock = null;
308                }
309        }

311        @Override
312        public void run()
313        {
314                final int q_len = 6;
315                Socket sock=null;
316                serverIsRunning = false;
317                CMLib.initialize(); // forces this thread to HAVE a library
318                Resources.initialize();
319                if (!isOK)
320                {
321                        System.err.println("Cancelling MUD server on port "+port);
322                        return;
323                }
324
325                InetAddress bindAddr = null;
326
327                if (bind.length() > 0)
328                {
329                        try
330                        {
331                                bindAddr = InetAddress.getByName(bind);
332                        }
333                        catch (final UnknownHostException e)
334                        {
335                                System.err.println("ERROR: MUD Server could not bind to address " + bind);
336                        }
337                }
338
339                try
340                {
341                        servsock=new ServerSocket(port, q_len);
342                        System.out.println("Off-Line Server started on port: "+port);
343                        if (bindAddr != null)
344                                System.out.println("Off-Line Server bound to: "+bindAddr.toString());
345                        serverIsRunning = true;
346
347                        while(true)
348                        {
349                                try
350                                {
351                                        state=0;
352                                        sock=servsock.accept();
353                                        acceptConnection(sock);
354                                }
355                                catch(final Exception t)
356                                {
357                                        if((!(t instanceof java.net.SocketException))
358                                        ||(t.getMessage()==null)
359                                        ||(t.getMessage().toLowerCase().indexOf("socket closed")<0))
360                                        {
361                                                t.printStackTrace(System.err);
362                                        }
363                                }
364                        }
365                }
366                catch(final Exception t)
367                {
368                        t.printStackTrace(System.err);
369                        if (!serverIsRunning)
370                                isOK = false;
371                }
372
373                System.out.println("Off-Line Server cleaning up.");
374
375                try
376                {
377                        if(servsock!=null)
378                                servsock.close();
379                        if(sock!=null)
380                                sock.close();
381                }
382                catch(final IOException e)
383                {
384                }
385
386                System.out.println("Off-Line Server on port "+port+" stopped!");
387        }

434        public static void main(String a[])
435        {
436                CMProps page=null;
437                CMLib.initialize(); // forces this thread to HAVE a library
438
439                String nameID="";
440                String iniFile="coffeemud.ini";
441                if(a.length>0)
442                {
443                        for (final String element : a)
444                                nameID+=" "+element;
445                        nameID=nameID.trim();
446                        final List<String> V=CMParms.cleanParameterList(nameID);
447                        for(int v=0;v<V.size();v++)
448                        {
449                                final String s=V.get(v);
450                                if(s.toUpperCase().startsWith("BOOT=")&&(s.length()>5))
451                                {
452                                        iniFile=s.substring(5);
453                                        V.remove(v);
454                                        v--;
455                                }
456                        }
457                        nameID=CMParms.combine(V,0);
458                }
459                if(nameID.length()==0)
460                        nameID="Unnamed CoffeeMud";
461                try
462                {
463                        while(true)
464                        {
465                                page=CMProps.loadPropPage(iniFile);
466                                if ((page==null)||(!page.isLoaded()))
467                                {
468                                        System.out.println("ERROR: Unable to read ini file: '"+iniFile+"'.");
469                                        System.exit(-1);
470                                        return;
471                                }
472
473                                isOK = true;
474                                bind=page.getStr("BIND");
475
476                                System.out.println();
477                                System.out.println("CoffeeMud Off-Line");
478                                System.out.println("(C) 2000-2018 Bo Zimmerman");
479                                System.out.println("http://www.coffeemud.org");
480
481                                if(OffLine.isOK)
482                                {
483                                        mudThreads=new Vector<OffLine>();
484                                        String ports=page.getProperty("PORT");
485                                        int pdex=ports.indexOf(',');
486                                        while(pdex>0)
487                                        {
488                                                final OffLine mud=new OffLine();
489                                                mud.acceptConnections=false;
490                                                mud.port=CMath.s_int(ports.substring(0,pdex));
491                                                ports=ports.substring(pdex+1);
492                                                mud.start();
493                                                mudThreads.addElement(mud);
494                                                pdex=ports.indexOf(',');
495                                        }
496                                        final OffLine mud=new OffLine();
497                                        mud.acceptConnections=false;
498                                        mud.port=CMath.s_int(ports);
499                                        mud.start();
500                                        mudThreads.addElement(mud);
501                                }
502
503                                final StringBuffer str=new StringBuffer("");
504                                for(int m=0;m<mudThreads.size();m++)
505                                {
506                                        final MudHost mud=mudThreads.elementAt(m);
507                                        str.append(" "+mud.getPort());
508                                }
509                                ports=str.toString();
510
511                                if(initHost(Thread.currentThread()))
512                                        mudThreads.firstElement().join();
513
514                                System.gc();
515                                System.runFinalization();
516
517                        }
518                }
519                catch(final InterruptedException e)
520                {
521                        e.printStackTrace();
522                }
523        }
