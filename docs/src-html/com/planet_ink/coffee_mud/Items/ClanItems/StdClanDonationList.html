037public class StdClanDonationList extends StdClanItem
038{
039
040        private Item lastItem=null;
041
042        public StdClanDonationList()
043        {
044                super();
045                setName("a donation list");
046                basePhyStats.setWeight(1);
047                setDisplayText("an list is setting here.");
048                setDescription("");
049                setClanItemType(ClanItem.ClanItemType.DONATIONJOURNAL);
050                CMLib.flags().setReadable(this,true);
051                secretIdentity="";
052                baseGoldValue=1;
053                material=RawMaterial.RESOURCE_PAPER;
054        }
055
056        @Override
057        public boolean okMessage(final Environmental myHost, final CMMsg msg)
058        {
059                if((((ClanItem)this).clanID().length()>0)
060                &&(CMLib.flags().isGettable(this))
061                &&(msg.target()==this)
062                &&(owner() instanceof Room))
063                {
064                        final Clan C=CMLib.clans().getClan(clanID());
065                        if((C!=null)&&(C.getDonation().length()>0))
066                        {
067                                final Room R=CMLib.map().getRoom(C.getDonation());
068                                if(R==owner())
069                                {
070                                        CMLib.flags().setGettable(this,false);
071                                        text();
072                                }
073                        }
074                }
075                return super.okMessage(myHost,msg);
076        }
077        
078        private String makeDonationDescription(final MOB source, final Environmental target, final String verb)
079        {
080                if(source != null)
081                {
082                        final Room R=source.location();
083                        if((R!=null)&&(R.getArea() != null))
084                        {
085                                return source.name()+" "+verb+" "+target.name()+" at "+R.getArea().getTimeObj().getShortTimeDescription()+".";
086                        }
087                }
088                return "";
089        }
090
091        private String getDonationKey(final MOB source)
092        {
093                if(source != null)
094                {
095                        return System.currentTimeMillis()+"/"+source.Name()+"/"+Math.random();
096                }
097                return System.currentTimeMillis()+"/UNK/"+Math.random();
098        }
099        
100        @Override
101        public void executeMsg(final Environmental myHost, final CMMsg msg)
102        {
103                if(((ClanItem)this).clanID().length()>0)
104                {
105                        if((msg.target()==this)
106                        &&(msg.targetMinor()==CMMsg.TYP_READ))
107                        {
108                                final MOB mob=msg.source();
109                                if(CMLib.flags().canBeSeenBy(this,mob))
110                                {
111                                        final StringBuffer text=new StringBuffer("");
112                                        final List<PlayerData> V=CMLib.database().DBReadPlayerData(clanID(),"DONATIONS");
113                                        final Vector<Object[]> sorted=new Vector<Object[]>();
114                                        String key=null;
115                                        int x=0;
116                                        long val=0;
117                                        DatabaseEngine.PlayerData set=null;
118                                        while(V.size()>0)
119                                        {
120                                                set=V.get(0);
121                                                key=set.key();
122                                                x=key.indexOf('/');
123                                                if(x>0)
124                                                {
125                                                        val=CMath.s_long(key.substring(0,x));
126                                                        boolean did=false;
127                                                        for(int i=0;i<sorted.size();i++)
128                                                        {
129                                                                if(((Long)sorted.elementAt(i)[0]).longValue()>val)
130                                                                {
131                                                                        did=true;
132                                                                        final Object[] O=new Object[2];
133                                                                        O[0]=Long.valueOf(val);
134                                                                        O[1]=set.xml();
135                                                                        sorted.insertElementAt(O,i);
136                                                                }
137                                                        }
138                                                        if(!did)
139                                                        {
140                                                                final Object[] O=new Object[2];
141                                                                O[0]=Long.valueOf(val);
142                                                                O[1]=set.xml();
143                                                                sorted.addElement(O);
144                                                        }
145                                                }
146                                                V.remove(0);
147                                        }
148                                        for(int i=0;i<sorted.size();i++)
149                                                text.append(((String)sorted.elementAt(i)[1])+"\n\r");
150
151                                        if(text.length()>0)
152                                                mob.tell(L("It says '@x1'.",text.toString()));
153                                        else
154                                                mob.tell(L("There is nothing written on @x1.",name()));
155                                }
156                                else
157                                        mob.tell(L("You can't see that!"));
158                                return;
159                        }
160                        else
161                        if((msg.target() instanceof Item)
162                        &&(msg.tool() instanceof Ability)
163                        &&(msg.target()!=lastItem)
164                        &&(msg.tool().ID().equalsIgnoreCase("Spell_ClanDonate"))
165                        &&(msg.sourceMinor()!=CMMsg.TYP_TEACH))
166                        {
167                                lastItem=(Item)msg.target();
168                                CMLib.database().DBCreatePlayerData(clanID(),"DONATIONS",getDonationKey(msg.source()),makeDonationDescription(msg.source(),msg.target(),"donated"));
169                        }
170                        else
171                        if((msg.targetMinor()==CMMsg.TYP_GET)
172                        &&(msg.target() instanceof Item)
173                        &&(!msg.targetMajor(CMMsg.MASK_INTERMSG)))
174                                CMLib.database().DBCreatePlayerData(clanID(),"DONATIONS",getDonationKey(msg.source()),makeDonationDescription(msg.source(),msg.target(),"gets"));
175                        else
176                        if(((msg.targetMinor()==CMMsg.TYP_PUSH)||(msg.targetMinor()==CMMsg.TYP_PULL))
177                        &&(msg.target() instanceof Item)
178                        &&(!msg.targetMajor(CMMsg.MASK_INTERMSG)))
179                                CMLib.database().DBCreatePlayerData(clanID(),"DONATIONS",getDonationKey(msg.source()),makeDonationDescription(msg.source(),msg.target(),"moves"));
180                        else
181                        if((msg.targetMinor()==CMMsg.TYP_DROP)
182                        &&(msg.target() instanceof Item)
183                        &&(!msg.targetMajor(CMMsg.MASK_INTERMSG)))
184                                CMLib.database().DBCreatePlayerData(clanID(),"DONATIONS",getDonationKey(msg.source()),makeDonationDescription(msg.source(),msg.target(),"drops"));
185                }
186                super.executeMsg(myHost,msg);
187        }
