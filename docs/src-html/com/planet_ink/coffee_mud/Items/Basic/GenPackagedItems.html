038public class GenPackagedItems extends GenItem implements PackagedItems
039{
040        @Override
041        public String ID()
042        {
043                return "GenPackagedItems";
044        }
045
046        public GenPackagedItems()
047        {
048                super();
049                setName("item");
050                basePhyStats.setWeight(150);
051                setDisplayText("");
052                setDescription("");
053                baseGoldValue = 5;
054                basePhyStats().setLevel(1);
055                setMaterial(RawMaterial.RESOURCE_MEAT);
056                recoverPhyStats();
057        }
058
059        @Override
060        protected boolean abilityImbuesMagic()
061        {
062                return false;
063        }
064
065        @Override
066        public String name()
067        {
068                if(numberOfItemsInPackage()!=1)
069                        return L("a package of @x1 @x2",""+numberOfItemsInPackage(), CMLib.english().makePlural(Name().trim()));
070                else
071                        return L("a package of @x1 @x2",""+numberOfItemsInPackage(), Name().trim());
072        }
073
074        @Override
075        public String displayText()
076        {
077                return L("a package of @x1 @x2(s) sit here.",""+numberOfItemsInPackage(), CMLib.english().makePlural(Name().trim()));
078        }
079
080        @Override
081        public int numberOfItemsInPackage()
082        {
083                return basePhyStats().ability();
084        }
085
086        @Override
087        public void setNumberOfItemsInPackage(int number)
088        {
089                basePhyStats().setAbility(number);
090                phyStats().setAbility(number);
091        }
092
093        protected byte[]        readableText    = null;
094
095        @Override
096        public String readableText()
097        {
098                return readableText == null ? "" : CMLib.encoder().decompressString(readableText);
099        }
100
101        @Override
102        public void setReadableText(String text)
103        {
104                readableText = (text.trim().length() == 0) ? null : CMLib.encoder().compressString(text);
105        }
106
107        @Override
108        public boolean packageMe(Item I, int number)
109        {
110                if ((I == null) 
111                || (!CMLib.utensils().disInvokeEffects(I)) 
112                || (I.amDestroyed()))
113                        return false;
114                name = CMLib.english().removeArticleLead(I.Name());
115                displayText = "";
116                if(I.description().trim().length()==0)
117                        setDescription(L("The contents of the stack appears as follows: ") + I.name());
118                else
119                        setDescription(L("The contents of the stack appears as follows: ") + "\n\r"+I.description());
120                basePhyStats().setLevel(I.basePhyStats().level());
121                basePhyStats().setWeight(I.basePhyStats().weight() * number);
122                basePhyStats().setHeight(I.basePhyStats().height());
123                setMaterial(I.material());
124                setBaseValue(I.baseGoldValue() * number);
125                final StringBuffer itemstr = new StringBuffer("");
126                itemstr.append("<PAKITEM>");
127                itemstr.append(CMLib.xml().convertXMLtoTag("PICLASS", CMClass.classID(I)));
128                itemstr.append(CMLib.xml().convertXMLtoTag("PIDATA", CMLib.coffeeMaker().getPropertiesStr(I, true)));
129                itemstr.append("</PAKITEM>");
130                setPackageText(itemstr.toString());
131                setNumberOfItemsInPackage(number);
132                recoverPhyStats();
133                return true;
134        }
135
136        @Override
137        public boolean isPackagable(List<Item> V)
138        {
139                if (V == null)
140                        return false;
141                if (V.size() == 0)
142                        return false;
143                for (int v1 = 0; v1 < V.size(); v1++)
144                {
145                        final Item I = V.get(v1);
146                        for (int v2 = v1 + 1; v2 < V.size(); v2++)
147                        {
148                                if (!V.get(v2).sameAs(I))
149                                        return false;
150                        }
151                }
152                return true;
153        }
154
155        @Override
156        public boolean areAllItemsTheSame()
157        {
158                return true;
159        }
160
161        @Override
162        public Item peekFirstItem()
163        {
164                if (packageText().length() == 0)
165                        return null;
166                final List<XMLLibrary.XMLTag> buf = CMLib.xml().parseAllXML(packageText());
167                if (buf == null)
168                {
169                        Log.errOut("Packaged", "Error parsing 'PAKITEM'.");
170                        return null;
171                }
172                final XMLTag iblk = CMLib.xml().getPieceFromPieces(buf, "PAKITEM");
173                if ((iblk == null) || (iblk.contents() == null))
174                {
175                        Log.errOut("Packaged", "Error parsing 'PAKITEM'.");
176                        return null;
177                }
178                final String itemi = iblk.getValFromPieces( "PICLASS");
179                final Environmental newOne = CMClass.getItem(itemi);
180                final List<XMLLibrary.XMLTag> idat = iblk.getContentsFromPieces( "PIDATA");
181                if ((idat == null) || (newOne == null) || (!(newOne instanceof Item)))
182                {
183                        Log.errOut("Packaged", "Error parsing 'PAKITEM' data.");
184                        return null;
185                }
186                CMLib.coffeeMaker().setPropertiesStr(newOne, idat, true);
187                return (Item) newOne;
188        }
189
190        @Override
191        public List<Item> unPackage(int number)
192        {
193                final List<Item> V = new Vector<Item>();
194                if (number >= numberOfItemsInPackage())
195                        number = numberOfItemsInPackage();
196                if (number <= 0)
197                        return V;
198                final int itemWeight = basePhyStats().weight() / numberOfItemsInPackage();
199                final int itemValue = baseGoldValue() / numberOfItemsInPackage();
200                final Item I = peekFirstItem();
201                if (I == null)
202                        return V;
203                I.recoverPhyStats();
204                for (int i = 0; i < number; i++)
205                        V.add((Item) I.copyOf());
206                setNumberOfItemsInPackage(numberOfItemsInPackage() - number);
207                if (numberOfItemsInPackage() <= 0)
208                {
209                        destroy();
210                        return V;
211                }
212                basePhyStats().setWeight(itemWeight * number);
213                setBaseValue(itemValue * number);
214                recoverPhyStats();
215                return V;
216        }
