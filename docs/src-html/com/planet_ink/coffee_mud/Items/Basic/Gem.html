035public class Gem extends StdItem
036{
037        @Override
038        public String ID()
039        {
040                return "Gem";
041        }
042
043        public final static int QUARTZ                  = 0;
044        public final static int AZURITE                 = 1;
045        public final static int BLOODSTONE              = 2;
046        public final static int JADE                    = 3;
047        public final static int DIAMOND                 = 4;
048        public final static int RUBY                    = 5;
049        public final static int OPAL                    = 6;
050        public final static int TOPAZ                   = 7;
051        public final static int SAPPHIRE                = 8;
052        public final static int ONYX                    = 9;
053        public final static int PEARL                   = 10;
054        public final static int EMERALD                 = 11;
055        public final static int AMETHYST                = 12;
056        private int lastLevel=-1;
057
058        public Gem()
059        {
060                super();
061
062                final Random randomizer = new Random(System.currentTimeMillis());
063                final int ringType = Math.abs(randomizer.nextInt() % 12);
064                basePhyStats.setLevel(ringType);
065                recoverPhyStats();
066        }
067
068        @Override
069        public void recoverPhyStats()
070        {
071                super.recoverPhyStats();
072                if(lastLevel!=phyStats().level())
073                {
074                        lastLevel=phyStats().level();
075                        setItemDescription(phyStats.level());
076                }
077        }
078
