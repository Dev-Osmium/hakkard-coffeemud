063        @Override
064        public boolean okMessage(final Environmental myHost, final CMMsg msg)
065        {
066                if(msg.amITarget(this))
067                {
068                        switch(msg.targetMinor())
069                        {
070                        case CMMsg.TYP_WRITE:
071                        {
072                                final String adminReq=getAdminReq().trim();
073                                final boolean admin=((adminReq.length()>0)&&CMLib.masking().maskCheck(adminReq,msg.source(),true))
074                                                                ||CMSecurity.isJournalAccessAllowed(msg.source(),Name());
075                                if((!CMLib.masking().maskCheck(getWriteReq(),msg.source(),true))
076                                &&(!admin)
077                                &&(!(CMSecurity.isAllowed(msg.source(),msg.source().location(),CMSecurity.SecFlag.JOURNALS))))
078                                {
079                                        msg.source().tell(L("You are not allowed to write on @x1",name()));
080                                        return false;
081                                }
082                                return true;
083                        }
084                        case CMMsg.TYP_REWRITE:
085                        {
086                                final MOB mob=msg.source();
087                                final String adminReq=getAdminReq().trim();
088                                final boolean admin=((adminReq.length()>0)&&CMLib.masking().maskCheck(adminReq,msg.source(),true))
089                                                                ||CMSecurity.isJournalAccessAllowed(msg.source(),Name());
090                                if((!CMLib.masking().maskCheck(getWriteReq(),msg.source(),true))
091                                &&(!admin)
092                                &&(!(CMSecurity.isAllowed(msg.source(),msg.source().location(),CMSecurity.SecFlag.JOURNALS))))
093                                {
094                                        msg.source().tell(L("You are not allowed to write on @x1",name()));
095                                        return false;
096                                }
097                                if(msg.targetMinor()==CMMsg.TYP_REWRITE)
098                                {
099                                        String entryStr;
100                                        if((msg.targetMessage()!=null)&&(msg.targetMessage().startsWith("DELETE ")))
101                                                entryStr=msg.targetMessage().substring(7).trim();
102                                        else
103                                        if((msg.targetMessage()!=null)&&(msg.targetMessage().startsWith("EDIT ")))
104                                        {
105                                                int x=msg.targetMessage().indexOf(' ',5);
106                                                if(x>5)
107                                                        entryStr=msg.targetMessage().substring(5,x).trim();
108                                                else
109                                                        entryStr=msg.targetMessage().substring(5).trim();
110                                        }
111                                        else
112                                                entryStr=msg.targetMessage();
113                                        if(!CMath.isInteger(entryStr))
114                                        {
115                                                mob.tell(L("The journal does not have an entry #@x1.",entryStr));
116                                                return false;
117                                        }
118                                        int which=CMath.s_int(entryStr);
119                                        final List<JournalEntry> journal2;
120                                        String[] tos=new String[0];
121                                        if(getReadFilter().length()>0)
122                                                tos=CMParms.parseCommas(this.getReadFilter(), true).toArray(new String[0]);
123                                        if(!getSortBy().toUpperCase().startsWith("CREAT"))
124                                                journal2=CMLib.database().DBReadJournalMsgsByUpdateDate(Name(), true, 100000,tos);
125                                        else
126                                                journal2=CMLib.database().DBReadJournalMsgsByCreateDate(Name(), true, 100000, tos);
127                                        if((which<1)||(which>journal2.size()))
128                                        {
129                                                mob.tell(L("The journal does not have an entry #@x1.",""+which));
130                                                return false;
131                                        }
132                                        else
133                                        {
134                                                final JournalEntry read=journal2.get(which-1);
135                                                if((!read.from().equalsIgnoreCase(mob.Name()))
136                                                &&(!read.to().equalsIgnoreCase(mob.Name()))
137                                                &&(!(CMSecurity.isAllowed(msg.source(),msg.source().location(),CMSecurity.SecFlag.JOURNALS)))
138                                                &&(!admin))
139                                                {
140                                                        mob.tell(L("You are not allowed to alter or remove entry #@x1.",""+which));
141                                                        return false;
142                                                }
143                                        }
144                                }
145                                return true;
146                        }
147                        }
148                }
149                return super.okMessage(myHost,msg);
150        }

152        protected boolean completeTransfer(final MOB mob, final CMMsg msg, String journal, JournalEntry entry2)
153        {
154                String realName=null;
155                for(final Enumeration<JournalsLibrary.CommandJournal> e=CMLib.journals().commandJournals();e.hasMoreElements();)
156                {
157                        final JournalsLibrary.CommandJournal CMJ=e.nextElement();
158                        if(journal.equalsIgnoreCase(CMJ.NAME())
159                        ||journal.equalsIgnoreCase(CMJ.NAME()+"S"))
160                        {
161                                realName=CMJ.JOURNAL_NAME();
162                                break;
163                        }
164                }
165                if(realName==null)
166                        realName=CMLib.database().DBGetRealJournalName(journal);
167                if(realName==null)
168                        realName=CMLib.database().DBGetRealJournalName(journal.toUpperCase());
169                if(realName!=null)
170                {
171                        CMLib.database().DBDeleteJournal(Name(),entry2.key());
172                        CMLib.database().DBWriteJournal(realName,
173                                                                                          entry2.from(),
174                                                                                          entry2.to(),
175                                                                                          entry2.subj(),
176                                                                                          entry2.msg());
177                        msg.setValue(-1);
178                        mob.tell(L("Message transferred."));
179                        return true;
180                }
181                else
182                {
183                        JournalsLibrary.CommandJournal cmJournalAlias=this.getMyCommandJournal();
184                        if(journal.equalsIgnoreCase("ALL"))
185                        {
186                                if(entry2.to().equalsIgnoreCase("ALL"))
187                                        mob.tell(L("Message already assigned to ALL."));
188                                else
189                                        realName="ALL";
190                        }
191                        else
192                        if(journal.equalsIgnoreCase("FROM"))
193                        {
194                                if(entry2.to().equalsIgnoreCase(entry2.from()))
195                                        mob.tell(L("Message already assigned to @x1.",entry2.from()));
196                                else
197                                if(entry2.from().equalsIgnoreCase(mob.Name()))
198                                        mob.tell(L("Message already accessible to @x1.",mob.Name()));
199                                else
200                                        realName=entry2.from();
201                        }
202                        else
203                        if(CMLib.players().playerExists(CMStrings.capitalizeAndLower(journal)))
204                                realName=CMStrings.capitalizeAndLower(journal);
205                        else
206                        if(cmJournalAlias!=null)
207                        {
208                                String otherToStr=cmJournalAlias.getFlag(CommandJournalFlags.ASSIGN);
209                                List<String> otherTos = CMParms.parseAny(otherToStr, ':', true);
210                                if(otherTos.contains(journal.toUpperCase().trim()))
211                                        realName=journal.toUpperCase().trim();
212                        }
213                        if(realName == null)
214                                mob.tell(L("The journal or user '@x1' does not presently exist.",journal));
215                        else
216                        {
217                                entry2.to(realName);
218                                CMLib.database().DBUpdateJournal(Name(), entry2);
219                                msg.setValue(-1);
220                                mob.tell(L("Message transferred."));
221                                return true;
222                        }
223                }
224                return false;
225        }

227        public JournalsLibrary.CommandJournal getMyCommandJournal()
228        {
229                JournalsLibrary.CommandJournal cmJournalAlias=null;
230                for(final Enumeration<JournalsLibrary.CommandJournal> e=CMLib.journals().commandJournals();e.hasMoreElements();)
231                {
232                        final JournalsLibrary.CommandJournal CMJ=e.nextElement();
233                        if(Name().equalsIgnoreCase(CMJ.NAME())
234                        ||Name().equalsIgnoreCase(CMJ.JOURNAL_NAME()))
235                                cmJournalAlias=CMJ;
236                }
237                return cmJournalAlias;
238        }

531                                        {

537                                        int which=CMath.s_int(entryStr);

576                        {
577                                try
578                                {
579                                        final Room R=mob.location();
580                                        final String adminReq=getAdminReq().trim();
581                                        final boolean admin=((adminReq.length()>0)&&CMLib.masking().maskCheck(adminReq,mob,true))
582                                                                        ||CMSecurity.isJournalAccessAllowed(mob,Name());
583                                        if((msg.targetMessage().toUpperCase().startsWith("DEL"))
584                                           &&(CMSecurity.isAllowed(mob,R,CMSecurity.SecFlag.JOURNALS)||admin)
585                                           &&(!mob.isMonster()))
586                                        {
587                                                if(mob.session().confirm(L("Delete all journal entries? Are you sure (y/N)?"),"N"))
588                                                        CMLib.database().DBDeleteJournal(name(),null);
589                                        }
590                                        else
591                                        if(!mob.isMonster())
592                                        {
593                                                String to="ALL";
594                                                String subject;
595                                                final String message;
596                                                if(msg.targetMessage().length()>1)
597                                                {
598                                                        message=msg.targetMessage();
599                                                        subject=CMStrings.ellipse(message, 40);
600                                                }
601                                                else
602                                                {
603                                                        if(CMath.s_bool(getParm("PRIVATE")))
604                                                                to=mob.Name();
605                                                        else
606                                                        if(CMath.s_bool(getParm("MAILBOX"))
607                                                        ||mob.session().confirm(L("Is this a private message (y/N)?"),"N"))
608                                                        {
609                                                                to=mob.session().prompt(L("To whom:"));
610                                                                if(((!to.toUpperCase().trim().startsWith("MASK=")
611                                                                                ||(!CMSecurity.isAllowed(mob,R,CMSecurity.SecFlag.JOURNALS)&&(!admin))))
612                                                                &&(!CMLib.players().playerExists(to)))
613                                                                {
614                                                                        mob.tell(L("I'm sorry, there is no such user."));
615                                                                        return;
616                                                                }
617                                                        }
618                                                        subject=mob.session().prompt(L("Enter a subject: "));
619                                                        if(subject.trim().length()==0)
620                                                        {
621                                                                mob.tell(L("Aborted."));
622                                                                return;
623                                                        }
624                                                        if((subject.toUpperCase().startsWith("MOTD")||subject.toUpperCase().startsWith("MOTM")||subject.toUpperCase().startsWith("MOTY"))
625                                                           &&(!admin)
626                                                           &&(!(CMSecurity.isAllowed(mob,R,CMSecurity.SecFlag.JOURNALS))))
627                                                                subject=subject.substring(4);
628                                                        message=mob.session().prompt(L("Enter your message\n\r: "));
629                                                        if(message.trim().length()==0)
630                                                        {
631                                                                mob.tell(L("Aborted."));
632                                                                return;
633                                                        }
634                                                }
635                                                if(message.startsWith("<cmvp>")
636                                                &&(!admin)
637                                                &&(!(CMSecurity.isAllowed(mob,R,CMSecurity.SecFlag.JOURNALS))))
638                                                {
639                                                        mob.tell(L("Illegal code, aborted."));
640                                                        return;
641                                                }
642                                                CMLib.database().DBWriteJournal(Name(),mob.Name(),to,subject,message);
643                                                if((R!=null)&&(msg.targetMessage().length()<=1))
644                                                        R.send(mob, ((CMMsg)msg.copyOf()).modify(CMMsg.MSG_WROTE, L("Journal entry added."), CMMsg.MSG_WROTE, subject+"\n\r"+message, -1, null));
645                                                else
646                                                        mob.tell(L("Journal entry added."));
647                                        }
648                                        return;
649                                }
650                                catch(final IOException e)
651                                {
652                                        Log.errOut("JournalItem",e.getMessage());
653                                }
654                                return;
655                        }
656                        default:
657                                break;
658                        }
659                }
660                else
661                if((msg.targetMinor()==CMMsg.TYP_ENTER)
662                &&(owner instanceof Room)
663                &&(msg.source().playerStats()!=null)
664                &&(CMLib.masking().maskCheck(getReadReq(),msg.source(),true)))
665                {
666
667                        final long[] newestDate=CMLib.database().DBJournalLatestDateNewerThan(Name(),msg.source().Name(),msg.source().playerStats().getLastDateTime());
668                        if((newestDate[0]>0)
669                        &&((newestDate[0]!=lastDateRead[0])||(msg.source()!=lastReadTo)))
670                        {
671                                lastReadTo=msg.source();
672                                lastDateRead=newestDate;
673                                msg.addTrailerMsg(CMClass.getMsg(msg.source(),null,null,CMMsg.MSG_OK_VISUAL,L("@x1 has @x2 new messages.",name(),""+newestDate[1]),CMMsg.NO_EFFECT,null,CMMsg.NO_EFFECT,null));
674                        }
675                }
676                else
677                if(((msg.targetMinor()==CMMsg.TYP_LOOK)||(msg.targetMinor()==CMMsg.TYP_EXAMINE))
678                &&(msg.target() instanceof Room)
679                &&(msg.source()==owner)
680                &&(msg.source().playerStats()!=null)
681                &&(CMLib.masking().maskCheck(getReadReq(),msg.source(),true)))
682                {
683                        final long[] newestDate=CMLib.database().DBJournalLatestDateNewerThan(Name(),msg.source().Name(),msg.source().playerStats().getLastDateTime());
684                        if((newestDate[0]>0)
685                        &&((newestDate[0]!=lastDateRead[0])||(msg.source()!=lastReadTo)))
686                        {
687                                lastReadTo=msg.source();
688                                lastDateRead=newestDate;
689                                msg.addTrailerMsg(CMClass.getMsg(msg.source(),null,null,CMMsg.MSG_OK_VISUAL,L("@x1 has @x2 new messages.",name(),""+newestDate[1]),CMMsg.NO_EFFECT,null,CMMsg.NO_EFFECT,null));
690                        }
691                }
692                super.executeMsg(myHost,msg);
693        }

695        public JournalEntry DBRead(MOB reader, String journal, int which, long lastTimeDate, boolean newOnly, boolean all)
696        {
697                final List<JournalEntry> journalEntries;
698                final boolean useCreateDate=getSortBy().toUpperCase().startsWith("CREAT");
699                String[] tos=new String[0];
700                if(getReadFilter().length()>0)
701                        tos=CMParms.parseCommas(this.getReadFilter(), true).toArray(new String[0]);
702                if(useCreateDate)
703                        journalEntries=CMLib.database().DBReadJournalMsgsByCreateDate(journal, true, 100000, tos);
704                else
705                        journalEntries=CMLib.database().DBReadJournalMsgsByUpdateDate(journal, true, 100000, tos);
706                final StringBuffer buf=new StringBuffer("");
707                final boolean shortFormat=readableText().toUpperCase().indexOf("SHORTLIST")>=0;
708                if((which<0)||(journalEntries==null)||(which>=journalEntries.size()))
709                {
710                        buf.append("#\n\r "+CMStrings.padRight("#",6)
711                                           +((shortFormat)?"":""
712                                           +CMStrings.padRight(L("From"),11)
713                                           +CMStrings.padRight(L("To"),11))
714                                           +CMStrings.padRight(L("Date"),20)
715                                           +"Subject\n\r");
716                        buf.append("-------------------------------------------------------------------------\n\r");
717                        if(journalEntries==null)
718                        {
719                                final JournalEntry fakeEntry = (JournalEntry)CMClass.getCommon("DefaultJournalEntry");
720                                fakeEntry.key("");
721                                fakeEntry.from("");
722                                fakeEntry.to("");
723                                fakeEntry.subj("");
724                                fakeEntry.derivedBuildMessage(buf);
725                                return fakeEntry;
726                        }
727                }
728
729                JournalEntry fakeEntry = (JournalEntry)CMClass.getCommon("DefaultJournalEntry");
730                if((which<0)||(which>=journalEntries.size()))
731                {
732                        if(journalEntries.size()>0)
733                        {
734                                final JournalEntry entry=journalEntries.get(0);
735                                fakeEntry = entry.copyOf();
736                        }
737                        int numToAdd=CMProps.getIntVar(CMProps.Int.JOURNALLIMIT);
738                        if((numToAdd==0)||(all))
739                                numToAdd=Integer.MAX_VALUE;
740                        final ArrayList<Integer> finalEntries = new ArrayList<Integer>();
741                        for(int j=journalEntries.size()-1;j>=0;j--)
742                        {
743                                final JournalEntry entry=journalEntries.get(j);
744                                final String from=entry.from();
745                                final String to=entry.to();
746                                // message is 5, but dont matter.
747                                final long compdate=entry.update();
748                                boolean mayRead=(to.equals("ALL")
749                                                                ||to.equalsIgnoreCase(reader.Name())
750                                                                ||from.equalsIgnoreCase(reader.Name()));
751                                if((to.toUpperCase().trim().startsWith("MASK="))&&(CMLib.masking().maskCheck(to.trim().substring(5),reader,true)))
752                                        mayRead=true;
753                                if(mayRead)
754                                {
755                                        if((compdate<=lastTimeDate)&&(newOnly))
756                                                continue;
757                                        if (numToAdd == 0)
758                                                break;
759                                        numToAdd--;
760                                        finalEntries.add(Integer.valueOf(j));
761                                        fakeEntry = entry.copyOf();
762                                }
763                        }
764                        final ArrayList<CharSequence> selections=new ArrayList<CharSequence>();
765                        for(int j=finalEntries.size()-1;j>=0;j--)
766                        {
767                                final Integer J = finalEntries.get(j);
768                                final JournalEntry entry=journalEntries.get(J.intValue());
769                                final String from=entry.from();
770                                final long date=entry.date();
771                                String to=entry.to();
772                                final String subject=entry.subj();
773                                // message is 5, but dont matter.
774                                final long compdate=entry.update();
775                                final StringBuffer selection=new StringBuffer("");
776                                if(compdate>lastTimeDate)
777                                        selection.append("*");
778                                else
779                                        selection.append(" ");
780                                if(to.toUpperCase().trim().startsWith("MASK="))
781                                        to=CMLib.masking().maskDesc(to.trim().substring(5),true);
782                                selection.append("^<JRNL \""+CMStrings.removeColors(name())+"\"^>"+CMStrings.padRightPreserve((J.intValue()+1)+"",4)+"^</JRNL^>) "
783                                                           +((shortFormat)?"":""
784                                                           +CMStrings.padRight(from,10)+" "
785                                                           +CMStrings.padRight(to,10)+" ")
786                                                           +CMStrings.padRight(CMLib.time().date2String(date),19)+" "
787                                                           +CMStrings.padRight(subject,25+(shortFormat?22:0))+"\n\r");
788                                selections.add(selection);
789                        }
790                        boolean notify=false;
791                        for(int v=0;v<selections.size();v++)
792                        {
793                                if(!(selections.get(v) instanceof StringBuffer))
794                                {
795                                        notify=true;
796                                        continue;
797                                }
798                                buf.append((StringBuffer)selections.get(v));
799                        }
800                        if(notify)
801                                buf.append(L("\n\rUse READ ALL [JOURNAL] to see missing entries."));
802                }
803                else
804                {
805                        final JournalEntry entry=journalEntries.get(which);
806                        final String from=entry.from();
807                        final long date=entry.date();
808                        String to=entry.to();
809                        final String subject=entry.subj();
810                        String message=entry.msg();
811
812                        fakeEntry = entry.copyOf();
813
814                        boolean allMine=(to.equalsIgnoreCase(reader.Name())
815                                                        ||from.equalsIgnoreCase(reader.Name()));
816                        if((to.toUpperCase().trim().startsWith("MASK="))&&(CMLib.masking().maskCheck(to.trim().substring(5),reader,true)))
817                        {
818                                allMine=true;
819                                to=CMLib.masking().maskDesc(to.trim().substring(5),true);
820                        }
821                        if(allMine)
822                                buf.append("*");
823                        else
824                                buf.append(" ");
825                        try
826                        {
827                                if(message.startsWith("<cmvp>"))
828                                        message=new String(CMLib.webMacroFilter().virtualPageFilter(message.substring(6).getBytes()));
829                        }
830                        catch(final HTTPRedirectException e)
831                        {
832                        }
833
834                        if((allMine)||(to.equals("ALL")))
835                        {
836                                buf.append("\n\r^<JRNL \""+CMStrings.removeColors(name())+"\"^>"+CMStrings.padRightPreserve((which+1)+"",4)+"^</JRNL^>)\n\r"
837                                                   +"FROM: "+from
838                                                   +"\n\rTO  : "+to
839                                                   +"\n\rDATE: "+CMLib.time().date2String(date)
840                                                   +"\n\rSUBJ: "+subject
841                                                   +"\n\r"+message);
842                        }
843                }
844                fakeEntry.derivedBuildMessage(buf);
845                return fakeEntry;
846        }

851        private String getParm(String parmName)
852        {
853                if(readableText().length()==0)
854                        return "";
855                final Map<String,String> h;
856                h=CMParms.parseEQParms(readableText().toUpperCase(), JOURNAL_PARMS_LIST);
857                if((parmName.equals("FILTER"))&&(h.containsKey("FILTER")))
858                {
859                        Map<String,String> h2=CMParms.parseEQParms(readableText(), JOURNAL_PARMS_LIST);
860                        for(String key : h2.keySet())
861                        {
862                                if(key.equalsIgnoreCase("FILTER"))
863                                {
864                                        h.put("FILTER", h2.get(key));
865                                }
866                        }
867                }
868                String req=h.get(parmName.toUpperCase().trim());
869                if(req==null)
870                        req="";
871                return req;
872        }
