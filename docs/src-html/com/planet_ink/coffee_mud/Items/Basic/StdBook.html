065        @Override
066        public boolean okMessage(final Environmental myHost, final CMMsg msg)
067        {
068                if(msg.amITarget(this))
069                {
070                        switch(msg.targetMinor())
071                        {
072                        case CMMsg.TYP_WRITE:
073                        case CMMsg.TYP_REWRITE:
074                                // the order that these things are checked in should
075                                // be holy, and etched in stone.
076                                int num=numBehaviors();
077                                MsgListener N=null;
078                                for(int b=0;b<num;b++)
079                                {
080                                        N=fetchBehavior(b);
081                                        if((N!=null)&&(!N.okMessage(this,msg)))
082                                                return false;
083                                }
084                                num=numScripts();
085                                for(int s=0;s<num;s++)
086                                {
087                                        N=fetchScript(s);
088                                        if((N!=null)&&(!N.okMessage(this,msg)))
089                                                return false;
090                                }
091                                num=numEffects();
092                                for(int i=0;i<num;i++)
093                                {
094                                        N=fetchEffect(i);
095                                        if((N!=null)&&(!N.okMessage(this,msg)))
096                                                return false;
097                                }
098                                break;
099                        default:
100                                break;
101                        }
102
103                        switch(msg.targetMinor())
104                        {
105                        case CMMsg.TYP_WRITE:
106                        {
107                                final String adminReq=getAdminReq().trim();
108                                final boolean admin=(adminReq.length()>0)&&CMLib.masking().maskCheck(adminReq,msg.source(),true);
109                                if((!CMLib.masking().maskCheck(getWriteReq(),msg.source(),true))
110                                &&(!admin)
111                                &&(!(CMSecurity.isAllowed(msg.source(),msg.source().location(),CMSecurity.SecFlag.JOURNALS))))
112                                {
113                                        msg.source().tell(L("You are not allowed to write on @x1",name()));
114                                        return false;
115                                }
116                                if(this.getMaxPages() != 0)
117                                {
118                                        final int numPages = this.getChapterCount("ALL");
119                                        if(numPages >= this.getMaxPages())
120                                        {
121                                                msg.source().tell(L("All the available pages in @x1 are used.",name()));
122                                                return false;
123                                        }
124                                }
125                                return true;
126                        }
127                        case CMMsg.TYP_REWRITE:
128                        {
129                                final String adminReq=getAdminReq().trim();
130                                final boolean admin=(adminReq.length()>0)&&CMLib.masking().maskCheck(adminReq,msg.source(),true);
131                                if((!CMLib.masking().maskCheck(getWriteReq(),msg.source(),true))
132                                &&(!admin)
133                                &&(!(CMSecurity.isAllowed(msg.source(),msg.source().location(),CMSecurity.SecFlag.JOURNALS))))
134                                {
135                                        msg.source().tell(L("You are not allowed to write on @x1",name()));
136                                        return false;
137                                }
138                                if(CMath.isInteger(msg.targetMessage()))
139                                {
140                                        final int msgNum=CMath.s_int(msg.targetMessage());
141                                        if((msgNum <1)||(msgNum>this.getChapterCount("ALL")))
142                                        {
143                                                msg.source().tell(L("There is no Chapter @x1",""+(msgNum)));
144                                                return false;
145                                        }
146                                        if((!admin)
147                                        &&(!(CMSecurity.isAllowed(msg.source(),msg.source().location(),CMSecurity.SecFlag.JOURNALS))))
148                                        {
149                                                final JournalEntry entry = this.readChaptersByCreateDate().get(msgNum-1);
150                                                if(!entry.from().equalsIgnoreCase(msg.source().Name()))
151                                                {
152                                                        msg.source().tell(L("You need permission to edit that chapter."));
153                                                        return false;
154                                                }
155                                        }
156                                }
157                                else
158                                if(msg.targetMessage().toUpperCase().startsWith("EDIT "))
159                                {
160                                        final int x=msg.targetMessage().indexOf(' ',5);
161                                        if(x>5)
162                                        {
163                                                final int msgNum=CMath.s_int(msg.targetMessage().substring(5,x).trim());
164                                                if((msgNum <1)||(msgNum>this.getChapterCount("ALL")))
165                                                {
166                                                        msg.source().tell(L("There is no Chapter @x1",""+(msgNum)));
167                                                        return false;
168                                                }
169                                                if((!admin)
170                                                &&(!(CMSecurity.isAllowed(msg.source(),msg.source().location(),CMSecurity.SecFlag.JOURNALS))))
171                                                {
172                                                        final JournalEntry entry = this.readChaptersByCreateDate().get(msgNum-1);
173                                                        if(!entry.from().equalsIgnoreCase(msg.source().Name()))
174                                                        {
175                                                                msg.source().tell(L("You need permission to edit that chapter."));
176                                                                return false;
177                                                        }
178                                                }
179                                        }
180                                }
181                                else
182                                if(msg.targetMessage().toUpperCase().startsWith("DELETE "))
183                                {
184                                        final String entryStr=msg.targetMessage().substring(7).trim();
185                                        final int entryNum=CMath.s_int(entryStr);
186                                        final int numEntries = this.getChapterCount("ALL");
187                                        if((entryNum < 1)||(numEntries>numEntries)||(!CMath.isInteger(entryStr)))
188                                        {
189                                                msg.source().tell(L("There is no Chapter @x1",""+(entryNum)));
190                                                return false;
191                                        }
192                                        if((!admin)
193                                        &&(!(CMSecurity.isAllowed(msg.source(),msg.source().location(),CMSecurity.SecFlag.JOURNALS))))
194                                        {
195                                                final JournalEntry entry = this.readChaptersByCreateDate().get(entryNum-1);
196                                                if(!entry.from().equalsIgnoreCase(msg.source().Name()))
197                                                {
198                                                        msg.source().tell(L("You need permission to remove that chapter."));
199                                                        return false;
200                                                }
201                                        }
202                                }
203                                return true;
204                        }
205                        }
206                }
207                return super.okMessage(myHost,msg);
208        }

210        @Override
211        public void executeMsg(final Environmental myHost, final CMMsg msg)
212        {
213                final MOB mob=msg.source();
214                if(msg.amITarget(this))
215                {
216                        switch(msg.targetMinor())
217                        {
218                        case CMMsg.TYP_READ:
219                        case CMMsg.TYP_WRITE:
220                        case CMMsg.TYP_REWRITE:
221                                // the order that these things are checked in should
222                                // be holy, and etched in stone.
223                                if(numBehaviors()>0)
224                                {
225                                        eachBehavior(new EachApplicable<Behavior>()
226                                        {
227                                                @Override
228                                                public final void apply(final Behavior B)
229                                                {
230                                                        B.executeMsg(me,msg);
231                                                }
232                                        });
233                                }
234                                if(numScripts()>0)
235                                {
236                                        eachScript(new EachApplicable<ScriptingEngine>()
237                                        {
238                                                @Override
239                                                public final void apply(final ScriptingEngine S)
240                                                {
241                                                        S.executeMsg(me,msg);
242                                                }
243                                        });
244                                }
245                                if(numEffects()>0)
246                                {
247                                        eachEffect(new EachApplicable<Ability>()
248                                        {
249                                                @Override
250                                                public final void apply(final Ability A)
251                                                {
252                                                        A.executeMsg(me, msg);
253                                                }
254                                        });
255                                }
256                                break;
257                        default:
258                                break;
259                        }
260                        switch(msg.targetMinor())
261                        {
262                        case CMMsg.TYP_READ:
263                        {
264                                final Room R=mob.location();
265                                if(!CMLib.flags().canBeSeenBy(this,mob))
266                                        mob.tell(L("You can't see that!"));
267                                else
268                                if((!mob.isMonster())
269                                &&(mob.playerStats()!=null))
270                                {
271                                        final String adminReq=getAdminReq().trim();
272                                        final boolean admin=(adminReq.length()>0)&&CMLib.masking().maskCheck(adminReq,mob,true);
273                                        final long lastTime=mob.playerStats().getLastDateTime();
274                                        if((admin)&&(!CMLib.masking().maskCheck(getReadReq(),mob,true)))
275                                        {
276                                                mob.tell(L("You are not allowed to read @x1.",name()));
277                                                return;
278                                        }
279                                        int which=-1;
280                                        boolean newOnly=false;
281                                        boolean all=false;
282                                        final Vector<String> parse=CMParms.parse(msg.targetMessage());
283                                        for(int v=0;v<parse.size();v++)
284                                        {
285                                                final String s=parse.elementAt(v);
286                                                if(CMath.s_long(s)>0)
287                                                        which=CMath.s_int(msg.targetMessage());
288                                                else
289                                                if(s.equalsIgnoreCase("NEW"))
290                                                        newOnly=true;
291                                                else
292                                                if(s.equalsIgnoreCase("ALL")||s.equalsIgnoreCase("OLD"))
293                                                        all=true;
294                                        }
295                                        final Triad<String,String,StringBuffer> read=DBRead(mob,which-1,lastTime, newOnly, all);
296                                        final StringBuffer entry=read.third;
297                                        if(entry.charAt(0)=='#')
298                                        {
299                                                which=-1;
300                                                entry.setCharAt(0,' ');
301                                        }
302                                        if((entry.charAt(0)=='*')
303                                           ||(admin)
304                                           ||(CMSecurity.isAllowed(mob,R,CMSecurity.SecFlag.JOURNALS)))
305                                                entry.setCharAt(0,' ');
306                                        else
307                                        if((newOnly)&&(msg.value()>0))
308                                                return;
309                                        final CMMsg readMsg=CMClass.getMsg(msg.source(), msg.target(), msg.tool(),
310                                                         CMMsg.MSG_WASREAD|CMMsg.MASK_ALWAYS, L("It says '@x1'.\n\r",entry.toString()),
311                                                         CMMsg.MSG_WASREAD|CMMsg.MASK_ALWAYS, entry.toString(),
312                                                         CMMsg.NO_EFFECT, null);
313                                        //mob.tell(entry.toString()+"\n\r");
314                                        if((entry.toString().trim().length()>0)
315                                        &&(which>0)
316                                        &&(CMLib.masking().maskCheck(getWriteReq(),mob,true)
317                                                ||(admin)
318                                                ||(CMSecurity.isAllowed(msg.source(),msg.source().location(),CMSecurity.SecFlag.JOURNALS))))
319                                        {
320                                        }
321                                        else
322                                        if(which<0)
323                                        {
324                                                readMsg.setSourceMessage(readMsg.sourceMessage()+description());
325                                        }
326                                        msg.addTrailerMsg(readMsg);
327                                        return;
328                                }
329                                return;
330                        }
331                        case CMMsg.TYP_WRITE:
332                        case CMMsg.TYP_REWRITE:
333                        {
334                                final MOB M=mob;
335                                final String[] subject=new String[1];
336                                final String[] message=new String[1];
337                                final String editKey[] = new String[1];
338                                final int maxCharsPerPage = this.getMaxCharsPerPage() > 0 ? this.getMaxCharsPerPage() : Integer.MAX_VALUE;
339                                if(!mob.isMonster())
340                                {
341                                        final Runnable addComplete=new Runnable()
342                                        {
343                                                final String to="ALL";
344                                                final MOB mob=M;
345                                                @Override
346                                                public void run()
347                                                {
348                                                        final Room R=mob.location();
349                                                        final String adminReq=getAdminReq().trim();
350                                                        final boolean admin=(adminReq.length()>0)&&CMLib.masking().maskCheck(adminReq,mob,true);
351                                                        if(message[0].startsWith("<cmvp>")
352                                                        &&(!admin)
353                                                        &&(!(CMSecurity.isAllowed(mob,R,CMSecurity.SecFlag.JOURNALS))))
354                                                        {
355                                                                mob.tell(L("Illegal code, aborted."));
356                                                                return;
357                                                        }
358                                                        if(editKey[0]==null)
359                                                        {
360                                                                if(message[0].length()>maxCharsPerPage)
361                                                                        mob.tell(L("That won't fit on the pages for this chapter.  The limit is @x1.",""+maxCharsPerPage));
362                                                                else
363                                                                {
364                                                                        addNewChapter(mob.Name(),to,subject[0],message[0]);
365                                                                        if((R!=null)&&(msg.targetMessage().length()<=1))
366                                                                                R.send(mob, ((CMMsg)msg.copyOf()).modify(CMMsg.MSG_WROTE, L("Chapter added."), CMMsg.MSG_WROTE, subject+"\n\r"+message, -1, null));
367                                                                        else
368                                                                                mob.tell(L("Chapter added."));
369                                                                }
370                                                        }
371                                                        else
372                                                        {
373                                                                if(message[0].length()>maxCharsPerPage)
374                                                                        mob.tell(L("That won't fit on the pages for this chapter.  The limit is @x1.",""+maxCharsPerPage));
375                                                                else
376                                                                {
377                                                                        editOldChapter(mob.Name(),to,editKey[0],subject[0],message[0]);
378                                                                        if((R!=null)&&(msg.targetMessage().length()<=1))
379                                                                                R.send(mob, ((CMMsg)msg.copyOf()).modify(CMMsg.MSG_WROTE, L("Chapter modified."), CMMsg.MSG_WROTE, subject+"\n\r"+message, -1, null));
380                                                                        else
381                                                                                mob.tell(L("Chapter modified."));
382                                                                }
383                                                        }
384                                                }
385                                        };
386                                        if((msg.targetMinor()==CMMsg.TYP_REWRITE)
387                                        &&(msg.targetMessage().toUpperCase().startsWith("DELETE ")))
388                                        {
389                                                final int entryNum=CMath.s_int(msg.targetMessage().substring(7).trim());
390                                                final JournalEntry entry = this.readChaptersByCreateDate().get(entryNum-1);
391                                                delOldChapter(mob.Name(),"ALL",entry.key());
392                                                mob.tell(L("Chapter removed."));
393                                        }
394                                        else
395                                        if((msg.targetMinor()==CMMsg.TYP_REWRITE)
396                                        &&(msg.targetMessage().toUpperCase().startsWith("EDIT ")))
397                                        {
398                                                int x=msg.targetMessage().indexOf(' ',5);
399                                                if(x>5)
400                                                {
401                                                        final int msgNum=CMath.s_int(msg.targetMessage().substring(5,x).trim());
402                                                        subject[0]=L("Chapter @x1",""+(msgNum));
403                                                        message[0]=msg.targetMessage().substring(x+1).trim();
404                                                        final JournalEntry entry = this.readChaptersByCreateDate().get(msgNum-1);
405                                                        editKey[0] = entry.key();
406                                                        if(message[0].startsWith("::"))
407                                                        {
408                                                                x=message[0].indexOf("::",2);
409                                                                if(x>1)
410                                                                {
411                                                                        subject[0]=message[0].substring(2,x);
412                                                                        message[0]=message[0].substring(x+2);
413                                                                }
414                                                        }
415                                                        addComplete.run();
416                                                }
417                                        }
418                                        else
419                                        if((msg.targetMessage().length()>1)
420                                        &&(msg.targetMinor()==CMMsg.TYP_WRITE)
421                                        &&(!CMath.isInteger(msg.targetMessage())))
422                                        {
423                                                message[0]=msg.targetMessage();
424                                                final int numMessages=getChapterCount("ALL");
425                                                subject[0]=L("Chapter @x1",""+(numMessages+1));
426                                                if(message[0].startsWith("::"))
427                                                {
428                                                        final int x=message[0].indexOf("::",2);
429                                                        if(x>1)
430                                                        {
431                                                                subject[0]=message[0].substring(2,x);
432                                                                message[0]=message[0].substring(x+2);
433                                                        }
434                                                }
435                                                addComplete.run();
436                                        }
437                                        else
438                                        {
439                                                final InputCallback contentCallBack = new InputCallback(InputCallback.Type.PROMPT,"",0)
440                                                {
441                                                        final MOB mob=M;
442                                                        final Runnable completer=addComplete;
443                                                        @Override
444                                                        public void showPrompt()
445                                                        {
446                                                                if((subject[0]!=null)&&(subject[0].length()>0))
447                                                                        mob.session().promptPrint(L("Enter the name of the chapter ("+subject[0]+"): "));
448                                                                else
449                                                                        mob.session().promptPrint(L("Enter the name of the chapter: "));
450                                                        }
451
452                                                        @Override
453                                                        public void timedOut()
454                                                        {
455                                                        }
456
457                                                        @Override
458                                                        public void callBack()
459                                                        {
460                                                                String subj=this.input.trim();
461                                                                if((subj.trim().length()==0)
462                                                                &&((subject[0]==null)||(subject[0].length()==0)))
463                                                                        subj=subject[0];
464                                                                if((subj == null) || (subj.trim().length()==0))
465                                                                {
466                                                                        mob.tell(L("Aborted."));
467                                                                        return;
468                                                                }
469                                                                subject[0]=subj;
470                                                                final String messageTitle="The contents of this chapter";
471                                                                mob.session().println(L("\n\rEnter the contents of this chapter:"));
472                                                                final List<String> vbuf=new ArrayList<String>();
473                                                                if(message[0]!=null)
474                                                                        vbuf.addAll(CMParms.parseAny(message[0],"\\n",false));
475                                                                CMLib.journals().makeMessageASync(mob, messageTitle, vbuf, true, new JournalsLibrary.MsgMkrCallback()
476                                                                {
477                                                                        @Override
478                                                                        public void callBack(final MOB mob, final Session sess, final MsgMkrResolution res)
479                                                                        {
480                                                                                if(res ==JournalsLibrary.MsgMkrResolution.CANCELFILE)
481                                                                                {
482                                                                                        mob.tell(L("Aborted."));
483                                                                                        return;
484                                                                                }
485                                                                                message[0]=CMParms.combineWith(vbuf, "\\n");
486                                                                                completer.run();
487                                                                        }
488                                                                });
489                                                        }
490                                                };
491                                                if((CMath.isInteger(msg.targetMessage()))
492                                                &&(msg.targetMinor()==CMMsg.TYP_REWRITE))
493                                                {
494                                                        try
495                                                        {
496                                                                final JournalEntry entry = this.readChaptersByCreateDate().get(CMath.s_int(msg.targetMessage())-1);
497                                                                if(entry != null)
498                                                                {
499                                                                        subject[0]=entry.subj();
500                                                                        message[0]=entry.msg();
501                                                                        editKey[0] = entry.key();
502                                                                        msg.setTargetMessage("");
503                                                                }
504                                                        }
505                                                        catch(final Exception e)
506                                                        {
507                                                        }
508                                                }
509                                                mob.session().prompt(contentCallBack);
510                                        }
511                                }
512                                return;
513                        }
514                        default:
515                                break;
516                        }
517                }
518                super.executeMsg(myHost,msg);
519        }

561        public Triad<String,String,StringBuffer> DBRead(final MOB readerMOB, final int which, final long lastTimeDate, final boolean newOnly, final boolean all)
562        {
563                final StringBuffer buf=new StringBuffer("");
564                final Triad<String,String,StringBuffer> reply=new Triad<String,String,StringBuffer>("","",new StringBuffer(""));
565                final List<JournalEntry> journal=this.readChaptersByCreateDate();
566                if((which<0)||(journal==null)||(which>=journal.size()))
567                {
568                        buf.append(getTOCHeader());
569                        buf.append("-------------------------------------------------------------------------\n\r");
570                        if(journal==null)
571                        {
572                                reply.first="";
573                                reply.second="";
574                                reply.third = buf;
575                                return reply;
576                        }
577                }
578
579                if((which<0)||(which>=journal.size()))
580                {
581                        if(journal.size()>0)
582                        {
583                                reply.first = journal.get(0).from();
584                                reply.second = journal.get(0).subj();
585                        }
586                        final ArrayList<Object> selections=new ArrayList<Object>();
587                        for(int j=0;j<journal.size();j++)
588                        {
589                                final JournalEntry entry=journal.get(j);
590                                final String from=entry.from();
591                                final String to=entry.to();
592                                final String subject=entry.subj();
593                                final StringBuffer selection=new StringBuffer("");
594                                if(to.equals("ALL")
595                                ||((readerMOB!=null)&&to.equalsIgnoreCase(readerMOB.Name()))
596                                ||((readerMOB!=null)&&from.equalsIgnoreCase(readerMOB.Name()))
597                                ||((readerMOB!=null)&&to.toUpperCase().trim().startsWith("MASK=")&&CMLib.masking().maskCheck(to.trim().substring(5),readerMOB,true)))
598                                {
599                                        //if(CMath.s_long(compdate)>lastTimeDate)
600                                        //    selection.append("*");
601                                        //else
602                                        //if(newOnly)
603                                        //    continue;
604                                        //else
605                                        //    selection.append(" ");
606                                        selection.append(subject+"\n\r");
607                                        selections.add(selection);
608                                }
609                        }
610                        int numToAdd=CMProps.getIntVar(CMProps.Int.JOURNALLIMIT);
611                        if((numToAdd==0)||(all))
612                                numToAdd=Integer.MAX_VALUE;
613                        for(int v=selections.size()-1;v>=0;v--)
614                        {
615                                if (numToAdd == 0)
616                                {
617                                        selections.set(v,"");
618                                        continue;
619                                }
620                                final StringBuffer str = (StringBuffer) selections.get(v);
621                                if ((newOnly) && (str.charAt(0) != '*'))
622                                {
623                                        selections.set(v,"");
624                                        continue;
625                                }
626                                numToAdd--;
627                        }
628                        boolean notify=false;
629                        for(int v=0;v<selections.size();v++)
630                        {
631                                if(!(selections.get(v) instanceof StringBuffer))
632                                {
633                                        notify=true;
634                                        continue;
635                                }
636                                buf.append((StringBuffer)selections.get(v));
637                        }
638                        if(notify)
639                                buf.append(L("\n\rUse READ ALL [BOOK] to see missing chapters."));
640                }
641                else
642                {
643                        final JournalEntry entry=journal.get(which);
644                        final String from=entry.from();
645                        final String to=entry.to();
646                        final String subject=entry.subj();
647                        String message=entry.msg();
648
649                        reply.first = entry.from();
650                        reply.second = entry.subj();
651
652                        //String compdate=(String)entry.elementAt(6);
653                        final boolean mineAble;
654                        if(readerMOB!=null)
655                                mineAble=to.equalsIgnoreCase(readerMOB.Name())
656                                                        ||(to.toUpperCase().trim().startsWith("MASK=")&&(CMLib.masking().maskCheck(to.trim().substring(5),readerMOB,true)))
657                                                        ||from.equalsIgnoreCase(readerMOB.Name());
658                        else
659                                mineAble=true;
660
661                        if(mineAble)
662                                buf.append("*");
663                        else
664                                buf.append(" ");
665                        try
666                        {
667                                if(message.startsWith("<cmvp>"))
668                                        message=new String(CMLib.webMacroFilter().virtualPageFilter(message.substring(6).getBytes()));
669                        }
670                        catch(final HTTPRedirectException e)
671                        {
672                        }
673
674                        if(to.equals("ALL")||mineAble)
675                                buf.append("\n\r"+subject
676                                                   +"\n\r"+message);
677                }
678                reply.third = buf;
679                return reply;
680        }
