110        public MapRoom[][] rebuildGrid(Hashtable<Room,MapRoom> areaMap)
111        {
112                // build grid!
113                int xoffset=0;
114                int yoffset=0;
115                for(final Enumeration<MapRoom> e=areaMap.elements();e.hasMoreElements();)
116                {
117                        final MapRoom mr=e.nextElement();
118                        if(mr.x<xoffset)
119                                xoffset=mr.x;
120                        if(mr.y<yoffset)
121                                yoffset=mr.y;
122                }
123
124                xoffset=xoffset*-1;
125                yoffset=yoffset*-1;
126
127                int Xbound=0;
128                int Ybound=0;
129                for(final Enumeration<MapRoom> e=areaMap.elements();e.hasMoreElements();)
130                {
131                        final MapRoom room=e.nextElement();
132                        room.x=room.x+xoffset;
133                        if(room.x>Xbound)
134                                Xbound=room.x;
135                        room.y=room.y+yoffset;
136                        if(room.y>Ybound)
137                                Ybound=room.y;
138                }
139                final MapRoom[][] grid=new MapRoom[Xbound+1][Ybound+1];
140                for(final Enumeration<MapRoom> e=areaMap.elements();e.hasMoreElements();)
141                {
142                        final MapRoom room=e.nextElement();
143                        grid[room.x][room.y]=room;
144                }
145
146                int numStragglers=0;
147                int somethingsLeft=0;
148                // now clear out stragglers.
149                for(int x=0;x<grid.length;x++)
150                {
151                        for(int y=0;y<grid[0].length;y++)
152                        {
153                                final MapRoom room=grid[x][y];
154                                if(room!=null)
155                                {
156                                        if(
157                                                ((x>0)&&(grid[x-1][y]!=null))
158                                        ||((y>0)&&(grid[x][y-1]!=null))
159                                        ||((x>0)&&(y>0)&&(grid[x-1][y-1]!=null))
160                                        ||((y<(grid[0].length-1))&&(grid[x][y+1]!=null))
161                                        ||((x<(grid.length-1))&&(grid[x+1][y]!=null))
162                                        ||((y<(grid[0].length-1))&&(x<(grid.length-1))&&(grid[x+1][y+1]!=null))
163                                           )
164                                                somethingsLeft++;
165                                        else
166                                        {
167                                                room.x=0;
168                                                room.y=0;
169                                                areaMap.remove(room.r);
170                                                grid[x][y]=null;
171                                                numStragglers++;
172                                        }
173                                }
174                        }
175                }
176                if((numStragglers>0)&&(somethingsLeft>0))
177                        return rebuildGrid(areaMap);
178
179                return grid;
180        }

197                        if((DOWN!=null)
198                        &&(DOWN.roomID().length()==0)
199                        &&(DOWN instanceof GridLocale))

201                                final List<Room> V=((GridLocale)DOWN).getAllRooms();
202                                for(int v=0;v<V.size();v++)
203                                        mapRooms.remove(V.get(v));
204                        }

208        public Hashtable<Room,MapRoom> makeMapRooms(int width)
209        {
210                final List<String> mapAreas=CMParms.parseSemicolons(getMapArea(),true);
211                final Hashtable<Room,MapRoom> mapRooms=new Hashtable<Room,MapRoom>();
212                for(int a=0;a<mapAreas.size();a++)
213                {
214                        final Area A=CMLib.map().getArea(mapAreas.get(a));
215                        if(A!=null)
216                        {
217                                for(final Enumeration<Room> r=A.getCompleteMap();r.hasMoreElements();)
218                                {
219                                        final Room R=r.nextElement();
220                                        final MapRoom mr=new MapRoom();
221                                        mr.r=R;
222                                        mapRooms.put(R,mr);
223                                }
224                        }
225                }
226                clearTheSkys(mapRooms);
227                return mapRooms;
228        }

230        public StringBuffer[][] finishMapMaking(int width)
231        {
232                final Hashtable<Room,MapRoom> mapRooms=makeMapRooms(width);
233                StringBuffer[][] map=new StringBuffer[0][0];
234                if(mapRooms.size()>0)
235                {
236                        placeRooms(mapRooms);
237                        final MapRoom[][] grid=rebuildGrid(mapRooms);
238                        if((grid.length==0)||(grid[0].length==0))
239                                return map;
240                        final int numXSquares=(int)Math.round(Math.floor(CMath.div(width-6,8)));
241                        final int numYSquares=((numXSquares/2)+1);
242                        final int xsize=grid.length/numXSquares;
243                        final int ysize=grid[0].length/numYSquares;
244                        if((xsize<0)||(ysize<0))
245                        {
246                                Log.errOut("StdMap","Error finishing " + xsize +"/"+ ysize+"/"+width);
247                                return map;
248                        }
249
250                        map=new StringBuffer[xsize+1][ysize+1];
251                        for(int y=0;y<grid[0].length;y++)
252                        {
253                                final int ycoord=y/numYSquares;
254                                int lastX=-1;
255                                String line1="";
256                                String line2="";
257                                String line3="";
258                                String line4="";
259                                String line5="";
260                                String line6="";
261                                int xcoord=-1;
262                                for(int x=0;x<grid.length;x++)
263                                {
264                                        xcoord=x/9;
265                                        if(xcoord!=lastX)
266                                        {
267                                                if(lastX>=0)
268                                                {
269                                                        if(map[lastX][ycoord]==null)
270                                                                map[lastX][ycoord]=new StringBuffer("");
271                                                        map[lastX][ycoord].append(line1+"\n\r"+line2+"\n\r"+line3+"\n\r"+line4+"\n\r"+line5+"\n\r"+line6+"\n\r");
272                                                }
273                                                lastX=xcoord;
274                                                line1="";
275                                                line2="";
276                                                line3="";
277                                                line4="";
278                                                line5="";
279                                                line6="";
280                                        }
281                                        final MapRoom room=grid[x][y];
282                                        if(room==null)
283                                        {
284                                                line1+="        ";
285                                                line2+="        ";
286                                                line3+="        ";
287                                                line4+="        ";
288                                                line5+="        ";
289                                                line6+="        ";
290                                        }
291                                        else
292                                        {
293                                                final String paddedName=CMStrings.padRight(room.r.displayText().trim(),30);
294                                                line1+="---"+dirChar(Directions.NORTH,grid,x,y,'-')+"----";
295                                                line2+="|"+paddedName.substring(0,6)+"|";
296                                                line3+=dirChar(Directions.WEST,grid,x,y,'|')+paddedName.substring(6,12)+dirChar(Directions.EAST,grid,x,y,'|');
297                                                line4+="|"+paddedName.substring(12,18)+"|";
298                                                line5+="|"+paddedName.substring(18,24)+"|";
299                                                line6+="---"+dirChar(Directions.SOUTH,grid,x,y,'-')+"----";
300                                        }
301                                }
302                                if(xcoord>=0)
303                                {
304                                        if(map[xcoord][ycoord]==null)
305                                                map[xcoord][ycoord]=new StringBuffer("");
306                                        map[xcoord][ycoord].append(line1+"\n\r"+line2+"\n\r"+line3+"\n\r"+line4+"\n\r"+line5+"\n\r"+line6+"\n\r");
307                                }
308                        }
309                }
310                for(int x=0;x<map.length;x++)
311                {
312                        for(int y=0;y<map[x].length;y++)
313                        {
314                                if(map[x][y]==null)
315                                        map[x][y]=new StringBuffer("");
316                        }
317                }
318                return map;
319        }

345        public char dirChar(int dirCode, MapRoom[][] grid, int x, int y, char wall)
346        {
347                final MapRoom room=grid[x][y];
348                if(room==null)
349                        return ' ';
350                switch(dirCode)
351                {
352                case Directions.NORTH:
353                        y=y-1;
354                        break;
355                case Directions.SOUTH:
356                        y=y+1;
357                        break;
358                case Directions.WEST:
359                        x=x-1;
360                        break;
361                case Directions.EAST:
362                        x=x+1;
363                        break;
364                default:
365                }
366                MapRoom nextRoom=null;
367                try
368                {       
369                        nextRoom=grid[x][y];    
370                }
371                catch(final Exception t)
372                {
373                }
374                if(nextRoom==null)
375                {
376                        if((room.r!=null)&&(room.r.getRoomInDir(dirCode)==null))
377                                return wall;
378                        return ' ';
379                }
380                dirCode=-1;
381                for(int d=Directions.NUM_DIRECTIONS()-1;d>=0;d--)
382                {
383                        if(d==Directions.GATE)
384                                continue;
385                        if(room.r.getRoomInDir(d)==nextRoom.r)
386                        {
387                                dirCode=d;
388                                break;
389                        }
390                }
391                if(dirCode<0)
392                {
393                        if((room.r!=null)&&(room.r.getRoomInDir(dirCode)==null))
394                                return wall;
395                        return ' ';
396                }
397                switch(dirCode)
398                {
399                case Directions.NORTH:
400                        return '^';
401                case Directions.SOUTH:
402                        return 'v';
403                case Directions.EAST:
404                        return '>';
405                case Directions.WEST:
406                        return '<';
407                case Directions.UP:
408                        return 'U';
409                case Directions.DOWN:
410                        return 'D';
411                }
412                return ' ';
413        }

431        public final static int CLUSTERSIZE=3;

433        public boolean isEmptyCluster(Map<Room,MapRoom> processed, int x, int y)
434        {
435                for(final Iterator<MapRoom> e=processed.values().iterator();e.hasNext();)
436                {
437                        final MapRoom room=e.next();
438                        if((((room.x>x-CLUSTERSIZE)&&(room.x<x+CLUSTERSIZE))
439                        &&((room.y>y-CLUSTERSIZE)&&(room.y<y+CLUSTERSIZE)))
440                        ||((room.x==x)&&(room.y==y)))
441                                return false;
442                }
443                return true;
444        }

446        public void findEmptyCluster(Map<Room,MapRoom> processed, List<Integer> XY)
447        {
448                final int x=XY.get(0).intValue();
449                final int y=XY.get(1).intValue();
450                int spacing=CLUSTERSIZE;
451                while(true)
452                {
453                        for(int i=0;i<8;i++)
454                        {
455                                int yadjust=0;
456                                int xadjust=0;
457                                switch(i)
458                                {
459                                case 0:
460                                        xadjust = 1;
461                                        break;
462                                case 1:
463                                        xadjust = 1;
464                                        yadjust = 1;
465                                        break;
466                                case 2:
467                                        yadjust = 1;
468                                        break;
469                                case 3:
470                                        xadjust = 1;
471                                        xadjust = -1;
472                                        break;
473                                case 4:
474                                        xadjust = -1;
475                                        break;
476                                case 5:
477                                        xadjust = -1;
478                                        yadjust = -1;
479                                        break;
480                                case 6:
481                                        yadjust = -1;
482                                        break;
483                                case 7:
484                                        yadjust = -1;
485                                        xadjust = 1;
486                                        break;
487                                }
488                                if(isEmptyCluster(processed,x+(spacing*xadjust),y+(spacing*yadjust)))
489                                {
490                                        XY.set(0,Integer.valueOf(x+(spacing*xadjust)));
491                                        XY.set(1,Integer.valueOf(y+(spacing*yadjust)));
492                                        return;
493                                }
494                        }
495                        spacing+=1;
496                }
497        }

507        public boolean okToPlace(MapRoom room)
508        {
509                if(room==null)
510                        return false;
511
512                if(room.r.domainType()==Room.DOMAIN_OUTDOORS_AIR)
513                        return false;
514
515                if((phyStats().level()<1)&&((room.r.domainType()&Room.INDOORS)==Room.INDOORS))
516                        return false;
517
518                boolean ok=false;
519                for(int d=0;d<Directions.NUM_DIRECTIONS()-1;d++)
520                {
521                        final Exit E=room.r.getExitInDir(d);
522                        if(E!=null)
523                        {
524                                if(phyStats().level()<2)
525                                {
526                                        if((!CMLib.flags().isHidden(E))&&(!CMLib.flags().isInvisible(E))&&(!E.defaultsLocked()))
527                                                ok=true;
528                                }
529                                else
530                                        ok=true;
531                        }
532                }
533                return ok;
534        }

552        public void placeRooms(final Map<Room,MapRoom> areaMap)
553        {
554                if(areaMap==null)
555                        return;
556                if(areaMap.size()==0)
557                        return;
558
559                for(final Iterator<MapRoom> e=areaMap.values().iterator();e.hasNext();)
560                {
561                        final MapRoom room=e.next();
562                        room.x=0;
563                        room.y=0;
564                        for(int d=0;d<Directions.NUM_DIRECTIONS()-1;d++)
565                        {
566                                final Room dir=room.r.getRoomInDir(d);
567                                if(dir!=null)
568                                {
569                                        final MapRoom rm=getRoom(areaMap,dir);
570                                        if(rm!=null)
571                                                rm.positionedAlready=false;
572                                }
573                        }
574                }
575
576                final Map<Room,MapRoom> processed=new HashMap<Room,MapRoom>();
577                boolean doneSomething=true;
578                while((areaMap.size()>processed.size())&&(doneSomething))
579                {
580                        doneSomething=false;
581                        for(final Iterator<MapRoom> e=areaMap.values().iterator();e.hasNext();)
582                        {
583                                final MapRoom room=e.next();
584                                if((processed.get(room.r)==null)&&(okToPlace(room)))
585                                {
586                                        placeRoom(room,areaMap,0,0,processed,true,true,0);
587                                        doneSomething=true;
588                                }
589                        }
590                }
591        }

593        @Override
594        public void executeMsg(final Environmental myHost, final CMMsg msg)
595        {
596                final MOB mob=msg.source();
597                if(!msg.amITarget(this))
598                        super.executeMsg(myHost,msg);
599                else
600                switch(msg.targetMinor())
601                {
602                case CMMsg.TYP_READ:
603                        if(CMLib.flags().canBeSeenBy(this,mob))
604                        {
605                                int cols = 80;
606                                if(mob.session()!=null)
607                                {
608                                        cols=mob.session().getWrap();
609                                        if(cols==0)
610                                                cols=254;
611                                        else
612                                                cols += 2;
613                                }
614                                final StringBuffer map[][]=getMyMappedRoom(cols);
615                                if((isReadable())
616                                &&(map!=null)
617                                &&(map.length>0)
618                                &&(map[0].length>0)
619                                &&(!mob.isMonster())
620                                &&(msg.targetMessage()!=null))
621                                {
622                                        int x=0;
623                                        int y=0;
624                                        String sec="A0";
625                                        if((msg.targetMessage().length()>0)&&(!msg.targetMessage().startsWith("<S-NAME>")))
626                                        {
627                                                sec="";
628                                                boolean trans=false;
629                                                for(int i=0;i<msg.targetMessage().length();i++)
630                                                {
631                                                        final char c=msg.targetMessage().charAt(i);
632                                                        if(Character.isDigit(c))
633                                                        {
634                                                                trans=true;
635                                                                y=(y*10)+CMath.s_int(""+c);
636                                                        }
637                                                        else
638                                                        if(Character.isLetter(c)&&(!trans))
639                                                        {
640                                                                x=(x*10)+(Character.toUpperCase(c)-'A');
641                                                                sec=sec+Character.toUpperCase(c);
642                                                        }
643                                                }
644                                                sec+=y;
645                                        }
646                                        if((x>=map.length)||(y>=map[0].length))
647                                        {
648                                                x=0;
649                                                y=0;
650                                                sec="A0";
651                                        }
652                                        if((map.length>1)||(map[0].length>1))
653                                        {
654                                                if(name().length()>0)
655                                                        mob.session().safeRawPrintln(L("Section: @x1",sec));
656                                        }
657                                        mob.session().rawPrint(map[x][y].toString());
658                                        if((map.length>1)||(map[0].length>1))
659                                        {
660                                                String letsec="A";
661                                                if(map.length>25)
662                                                {
663                                                        for(int l=0;l<map.length/26;l++)
664                                                                letsec+='Z';
665                                                }
666                                                letsec=letsec.substring(0,letsec.length()-1)+((char)((('A')+map.length%26)-1));
667                                                if(name().length()>0)
668                                                        mob.session().safeRawPrintln(L("(@x1) Use 'READ SEC MAPNAME' to read sections A0 through @x2@x3 (A-@x4, 0-@x5).",sec,letsec,""+(map[0].length-1),letsec,""+(map[0].length-1)));
669                                        }
670                                }
671                                else
672                                        mob.tell(L("@x1 appears to be blank.",name()));
673                        }
674                        else
675                                mob.tell(L("You can't see that!"));
676                        return;
677                }
678                super.executeMsg(myHost,msg);
679        }

681        public void placeRoom(MapRoom room,
682                                                  Map<Room,MapRoom> areaMap,
683                                                  int favoredX,
684                                                  int favoredY,
685                                                  Map<Room,MapRoom> processed,
686                                                  boolean doNotDefer,
687                                                  boolean passTwo,
688                                                  int depth)
689        {
690                if(room==null)
691                        return;
692                if(depth>500)
693                        return;
694                final MapRoom anythingAt=getProcessedRoomAt(processed,favoredX,favoredY);
695                if(anythingAt!=null)
696                {
697                        // maybe someone else will take care of it?
698                        if(!doNotDefer)
699                        {

717                        }

731
732                // once done, is never undone.  A room is

737                {

741
742                        if((nextRoom!=null)
743                        &&(processed.get(nextRoom.r)==null)
744                        &&(passTwo||((d!=Directions.UP)&&(d!=Directions.DOWN))))

746                                int newFavoredX=room.x;
747                                int newFavoredY=room.y;
748                                switch(d)
749                                {

754                                        case Directions.EAST:
755                                                newFavoredX++; break;
756                                        case Directions.WEST:
757                                                newFavoredX--; break;
758                                        case Directions.UP:
759                                                if(!anythingThatDirection(room,Directions.NORTH))
760                                                        newFavoredY--;
761                                                else
762                                                if(!anythingThatDirection(room,Directions.WEST))
763                                                        newFavoredX--;
764                                                else

766                                                        newFavoredX++;
767                                                else
768                                                if(!anythingThatDirection(room,Directions.SOUTH))
769                                                        newFavoredY++;
770                                                break;
771                                        case Directions.DOWN:
772                                                if(!anythingThatDirection(room,Directions.SOUTH))
773                                                        newFavoredY++;
774                                                else
775                                                if(!anythingThatDirection(room,Directions.EAST))
776                                                        newFavoredX++;
777                                                else
778                                                if(!anythingThatDirection(room,Directions.WEST))
779                                                        newFavoredX--;
780                                                else
781                                                if(!anythingThatDirection(room,Directions.NORTH))
782                                                        newFavoredY--;
783                                                break;
784                                }
785                                nextRoom.positionedAlready=true;
786                                placeRoom(nextRoom,areaMap,newFavoredX,newFavoredY,processed,false,passTwo,depth+1);
787                        }
788                }
