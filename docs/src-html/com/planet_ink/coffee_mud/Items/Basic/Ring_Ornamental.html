058        protected int lastLevel=-1;
059
060        public Ring_Ornamental()
061        {
062                super();
063
064                final int ringType = CMLib.dice().roll(1,14,-1);
065                this.basePhyStats.setLevel(ringType);
066                setItemDescription(this.basePhyStats.level());
067                lastLevel=ringType;
068                recoverPhyStats();
069        }
070
071        @Override
072        public void recoverPhyStats()
073        {
074                if(lastLevel!=basePhyStats().level())
075                {
076                        setItemDescription(basePhyStats().level());
077                        lastLevel=basePhyStats().level();
078                }
079                super.recoverPhyStats();
080        }
