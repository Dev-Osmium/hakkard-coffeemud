036public class GenCoins extends GenItem implements Coins
037{
038        @Override
039        public String ID()
040        {
041                return "GenCoins";
042        }
043
044        @Override
045        public int value()
046        {
047                return phyStats().ability();
048        }
049
050        protected String currency="";
051        protected double denomination=1.0;
052
053        public GenCoins()
054        {
055                super();
056                myContainer=null;
057                setMaterial(RawMaterial.RESOURCE_GOLD);
058                setNumberOfCoins(100);
059                setCurrency("");
060                setDenomination(CMLib.beanCounter().getLowestDenomination(""));
061                setDescription("");
062        }
063
064        @Override
065        public String Name()
066        {
067                return CMLib.beanCounter().getDenominationName(getCurrency(),getDenomination(),getNumberOfCoins());
068        }
069
070        @Override
071        public String displayText()
072        {
073                return CMLib.beanCounter().getDenominationName(getCurrency(),getDenomination(),getNumberOfCoins())+((getNumberOfCoins()==1)?" lies here.":" lie here.");
074        }
075
076        @Override
077        protected boolean abilityImbuesMagic()
078        {
079                return false;
080        }
081
082        public void setDynamicMaterial()
083        {
084                if((CMLib.english().containsString(name(),"note"))
085                ||(CMLib.english().containsString(name(),"bill"))
086                ||(CMLib.english().containsString(name(),"dollar")))
087                        setMaterial(RawMaterial.RESOURCE_PAPER);
088                else
089                {
090                        final RawMaterial.CODES codes = RawMaterial.CODES.instance();
091                        for(int s=0;s<codes.total();s++)
092                        {
093                                if(CMLib.english().containsString(name(),codes.name(s)))
094                                {
095                                        setMaterial(codes.get(s));
096                                        break;
097                                }
098                        }
099                }
100                setDescription(CMLib.beanCounter().getConvertableDescription(getCurrency(),getDenomination()));
101        }
102
103        @Override
104        public long getNumberOfCoins()
105        {
106                return phyStats().ability();
107        }
108
109        @Override
110        public void setNumberOfCoins(long number)
111        {
112                if(number<Integer.MAX_VALUE)
113                        basePhyStats().setAbility((int)number);
114                else
115                        basePhyStats().setAbility(Integer.MAX_VALUE);
116                recoverPhyStats();
117        }
118
119        @Override
120        public double getDenomination()
121        {
122                return denomination;
123        }
124
125        @Override
126        public void setDenomination(double valuePerCoin)
127        {
128                if(valuePerCoin==0.0)
129                        valuePerCoin=1.0;
130                denomination=valuePerCoin;
131                setDynamicMaterial();
132        }
133
134        @Override
135        public double getTotalValue()
136        {
137                return CMath.mul(getDenomination(),getNumberOfCoins());
138        }
139
140        @Override
141        public String getCurrency()
142        {
143                return currency;
144        }
145
146        @Override
147        public void setCurrency(String named)
148        {
149                currency=named;
150                setDynamicMaterial();
151        }
152
153        @Override
154        public boolean isGeneric()
155        {
156                return true;
157        }
158
159        @Override
160        public void recoverPhyStats()
161        {
162                if(((material&RawMaterial.MATERIAL_MASK)!=RawMaterial.MATERIAL_CLOTH)
163                &&((material&RawMaterial.MATERIAL_MASK)!=RawMaterial.MATERIAL_PAPER))
164                        basePhyStats.setWeight((int)Math.round((basePhyStats().ability()/100.0)));
165                basePhyStats.copyInto(phyStats);
166                // import not to sup this, otherwise 'ability' makes it magical!
167                for(final Enumeration<Ability> a=effects();a.hasMoreElements();)
168                {
169                        final Ability A=a.nextElement();
170                        if(A!=null)
171                                A.affectPhyStats(this,phyStats);
172                }
173        }
174
175        @Override
176        public boolean putCoinsBack()
177        {
178                if(amDestroyed())
179                        return false;
180                Coins alternative=null;
181                if(owner() instanceof Room)
182                {
183                        final Room R=(Room)owner();
184                        for(int i=0;i<R.numItems();i++)
185                        {
186                                final Item I=R.getItem(i);
187                                if((I!=null)
188                                &&(I!=this)
189                                &&(I instanceof Coins)
190                                &&(((Coins)I).getDenomination()==getDenomination())
191                                &&((Coins)I).getCurrency().equals(getCurrency())
192                                &&(I.container()==container()))
193                                {
194                                        alternative=(Coins)I;
195                                        break;
196                                }
197                        }
198                }
199                else
200                if(owner() instanceof MOB)
201                {
202                        final MOB M=(MOB)owner();
203                        for(int i=0;i<M.numItems();i++)
204                        {
205                                final Item I=M.getItem(i);
206                                if((I!=null)
207                                &&(I!=this)
208                                &&(I instanceof Coins)
209                                &&(((Coins)I).getDenomination()==getDenomination())
210                                &&((Coins)I).getCurrency().equals(getCurrency())
211                                &&(I.container()==container()))
212                                {
213                                        alternative=(Coins)I;
214                                        break;
215                                }
216                        }
217                }
218                if((alternative!=null)&&(alternative!=this))
219                {
220                        alternative.setNumberOfCoins(alternative.getNumberOfCoins()+getNumberOfCoins());
221                        destroy();
222                        return true;
223                }
224                return false;
225        }
226
227        private final static String[] MYCODES={"NUMCOINS","CURRENCY","DENOM"};
228        @Override
229        public String getStat(String code)
230        {
231                if(CMLib.coffeeMaker().getGenItemCodeNum(code)>=0)
232                        return CMLib.coffeeMaker().getGenItemStat(this,code);
233                switch(getCodeNum(code))
234                {
235                case 0:
236                        return "" + getNumberOfCoins();
237                case 1:
238                        return "" + getCurrency();
239                case 2:
240                        return "" + getDenomination();
241                default:
242                        return CMProps.getStatCodeExtensionValue(getStatCodes(), xtraValues, code);
243                }
244        }
245
246        @Override
247        public void setStat(String code, String val)
248        {
249                if(CMLib.coffeeMaker().getGenItemCodeNum(code)>=0)
250                        CMLib.coffeeMaker().setGenItemStat(this,code,val);
251                else
252                switch(getCodeNum(code))
253                {
254                case 0:
255                        setNumberOfCoins(CMath.s_parseIntExpression(val));
256                        break;
257                case 1:
258                        setCurrency(val);
259                        break;
260                case 2:
261                        setDenomination(CMath.s_double(val));
262                        break;
263                default:
264                        CMProps.setStatCodeExtensionValue(getStatCodes(), xtraValues, code, val);
265                        break;
266                }
267        }
268
269        @Override
270        protected int getCodeNum(final String code)
271        {
272                for(int i=0;i<MYCODES.length;i++)
273                {
274                        if(code.equalsIgnoreCase(MYCODES[i]))
275                                return i;
276                }
277                return -1;
278        }
279
280        private static String[] codes=null;
281
282        @Override
283        public String[] getStatCodes()
284        {
285                if(codes!=null)
286                        return codes;
287                final String[] MYCODES=CMProps.getStatCodesList(GenCoins.MYCODES,this);
288                final String[] superCodes=CMParms.toStringArray(GenericBuilder.GenItemCode.values());
289                codes=new String[superCodes.length+MYCODES.length];
290                int i=0;
291                for(;i<superCodes.length;i++)
292                        codes[i]=superCodes[i];
293                for(int x=0;x<MYCODES.length;i++,x++)
294                        codes[i]=MYCODES[x];
295                return codes;
296        }
297
298        @Override
299        public boolean sameAs(final Environmental E)
300        {
301                if(!(E instanceof GenCoins))
302                        return false;
303                final String[] codes=getStatCodes();
304                for(int i=0;i<codes.length;i++)
305                {
306                        if(!E.getStat(codes[i]).equals(getStat(codes[i])))
307                                return false;
308                }
309                return true;
310        }
