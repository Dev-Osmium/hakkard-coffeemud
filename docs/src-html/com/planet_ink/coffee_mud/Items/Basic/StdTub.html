134        protected int getExtraLiquidResourceType()
135        {
136                final List<Item> V=getContents();
137                for(int v=0;v<V.size();v++)
138                        if((V.get(v) instanceof Drink)
139                        &&((V.get(v).material()&RawMaterial.MATERIAL_MASK)==RawMaterial.MATERIAL_LIQUID))
140                                return V.get(v).material();
141                return -1;
142        }

144        @Override
145        public boolean containsDrink()
146        {
147                if((!CMLib.flags().isGettable(this))
148                &&(owner()!=null)
149                &&(owner() instanceof Room)
150                &&(((Room)owner()).getArea()!=null)
151                &&(((Room)owner()).getArea().getClimateObj().weatherType((Room)owner())==Climate.WEATHER_DROUGHT))
152                        return false;
153                if(liquidRemaining()<1)
154                        return (getExtraLiquidResourceType()>0);
155                return true;
156        }

263        @Override
264        public boolean okMessage(final Environmental myHost, final CMMsg msg)
265        {
266                if(!super.okMessage(myHost,msg))
267                        return false;
268                if(msg.amITarget(this))
269                {
270                        final MOB mob=msg.source();
271                        switch(msg.targetMinor())
272                        {
273                        case CMMsg.TYP_DRINK:
274                                if((mob.isMine(this))||(phyStats().weight()>1000)||(!CMLib.flags().isGettable(this)))
275                                {
276                                        if(!containsDrink())
277                                        {
278                                                mob.tell(L("@x1 is empty.",name()));
279                                                return false;
280                                        }
281                                        if((liquidType()==RawMaterial.RESOURCE_SALTWATER)
282                                        ||(liquidType()==RawMaterial.RESOURCE_LAMPOIL))
283                                        {
284                                                mob.tell(L("You don't want to be drinking @x1.",RawMaterial.CODES.NAME(liquidType()).toLowerCase()));
285                                                return false;
286                                        }
287                                        return true;
288                                }
289                                mob.tell(L("You don't have that."));
290                                return false;
291                        case CMMsg.TYP_FILL:
292                                if((liquidRemaining()>=amountOfLiquidHeld)
293                                &&(liquidHeld()<500000))
294                                {
295                                        mob.tell(L("@x1 is full.",name()));
296                                        return false;
297                                }
298                                if((msg.tool()!=msg.target())
299                                &&(msg.tool() instanceof Drink))
300                                {
301                                        final Drink thePuddle=(Drink)msg.tool();
302                                        if(!thePuddle.containsDrink())
303                                        {
304                                                mob.tell(L("@x1 is empty.",thePuddle.name()));
305                                                return false;
306                                        }
307                                        if((liquidRemaining()>0)&&(liquidType()!=thePuddle.liquidType()))
308                                        {
309                                                mob.tell(L("There is still some @x1 left in @x2.  You must empty it before you can fill it with @x3.",RawMaterial.CODES.NAME(liquidType()).toLowerCase(),name(),RawMaterial.CODES.NAME(thePuddle.liquidType()).toLowerCase()));
310                                                return false;
311
312                                        }
313                                        return true;
314                                }
315                                mob.tell(L("You can't fill @x1 from that.",name()));
316                                return false;
317                        default:
318                                break;
319                        }
320                }
321                return true;
322        }

324        @Override
325        public int amountTakenToFillMe(Drink theSource)
326        {
327                int amountToTake=amountOfLiquidHeld-amountOfLiquidRemaining;
328                if(amountOfLiquidHeld>=500000)
329                        amountToTake=theSource.liquidRemaining();
330                if(amountToTake>theSource.liquidRemaining())
331                        amountToTake=theSource.liquidRemaining();
332                return amountToTake;
333        }

335        @Override
336        public void executeMsg(final Environmental myHost, final CMMsg msg)
337        {
338                if((msg.source().riding()==this)
339                &&(CMLib.commands().isHygienicMessage(msg, 0, PlayerStats.HYGIENE_WATERCLEAN)))
340                {
341                        final int extraRsc=getExtraLiquidResourceType();
342                        if((amountOfLiquidRemaining>1)
343                        &&(this.liquidType==RawMaterial.RESOURCE_FRESHWATER)
344                        &&((extraRsc<0)||(extraRsc==RawMaterial.RESOURCE_FRESHWATER)))
345                        {
346                                CMLib.commands().handleHygienicMessage(msg, 0, PlayerStats.HYGIENE_WATERCLEAN);
347                        }
348                }
349
350                if(msg.amITarget(this))
351                {
352                        final MOB mob=msg.source();
353                        switch(msg.targetMinor())
354                        {
355                        case CMMsg.TYP_DRINK:
356                                amountOfLiquidRemaining-=amountOfThirstQuenched;
357                                final boolean thirsty=mob.curState().getThirst()<=0;
358                                final boolean full=!mob.curState().adjThirst(amountOfThirstQuenched,mob.maxState().maxThirst(mob.baseWeight()));
359                                if(thirsty)
360                                        mob.tell(L("You are no longer thirsty."));
361                                else
362                                if(full)
363                                        mob.tell(L("You have drunk all you can."));
364                                if(disappearsAfterDrinking)
365                                {
366                                        destroy();
367                                        return;
368                                }
369                                break;
370                        case CMMsg.TYP_FILL:
371                                if((msg.tool() instanceof Drink))
372                                {
373                                        final Drink thePuddle=(Drink)msg.tool();
374                                        final int amountToTake=amountTakenToFillMe(thePuddle);
375                                        thePuddle.setLiquidRemaining(thePuddle.liquidRemaining()-amountToTake);
376                                        if(amountOfLiquidRemaining<=0)
377                                                setLiquidType(thePuddle.liquidType());
378                                        if( ( (long)amountOfLiquidRemaining + (long)amountToTake ) <= Integer.MAX_VALUE )
379                                                amountOfLiquidRemaining+=amountToTake;
380                                        if(amountOfLiquidRemaining>amountOfLiquidHeld)
381                                                amountOfLiquidRemaining=amountOfLiquidHeld;
382                                        if((amountOfLiquidRemaining<=0)&&(disappearsAfterDrinking))
383                                        {
384                                                destroy();
385                                                return;
386                                        }
387                                }
388                                break;
389                        default:
390                                break;
391                        }
392                }
393                super.executeMsg(myHost,msg);
394        }
