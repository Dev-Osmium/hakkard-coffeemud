059        @Override
060        public boolean okMessage(final Environmental myHost, final CMMsg msg)
061        {
062                if(!super.okMessage(myHost, msg))
063                        return false;
064                
065                if(msg.amITarget( this ) 
066                && (msg.targetMinor()==CMMsg.TYP_READ))
067                {
068                        
069                        if((msg.targetMessage()!=null)
070                        && (recipeLines.length>0)
071                        &&(CMath.isInteger(msg.targetMessage())))
072                        {
073                                int x=CMath.s_int(msg.targetMessage());
074                                int z=this.getRecipeCodeLines().length;
075                                if(x<1)
076                                {
077                                        msg.source().tell(L("Which recipe?"));
078                                        return false;
079                                }
080                                else
081                                if(x>z)
082                                {
083                                        if(z==1)
084                                                msg.source().tell(L("There is only 1 recipe."));
085                                        else
086                                                msg.source().tell(L("There are only 1 recipes.",""+z));
087                                        return false;
088                                }
089                        }
090                        else
091                        if(this.commonSkillID.trim().length()>0)
092                        {
093                                final Ability A=CMClass.getAbility(this.commonSkillID);
094                                if(A==null)
095                                        this.commonSkillID="";
096                                else
097                                        msg.source().tell(L("This book is only for @x1 recipes.",A.name()));
098                        }
099                }
100                if(msg.amITarget( this ) 
101                && (msg.targetMinor()==CMMsg.TYP_WRITE))
102                {
103                        final List<String> recipes=CMParms.parseAny(msg.targetMessage(), "\n\r", true);
104                        if(recipes.size()>0)
105                        {
106                                int max=this.getTotalRecipePages();
107                                int num=this.getRecipeCodeLines().length;
108                                if((max-num)==0)
109                                {
110                                        msg.source().tell(L("There are no more pages in this recipe book."));
111                                        return false;
112                                }
113                                else
114                                if((max-num)<recipes.size())
115                                {
116                                        if((max-num)==1)
117                                                msg.source().tell(L("There is only one more page in this recipe book."));
118                                        else
119                                                msg.source().tell(L("There are only @x1 more pages in this recipe book.",""+(max-num)));
120                                        return false;
121                                }
122                                final Ability A=CMClass.getAbility( getCommonSkillID() );
123                                if(!(A instanceof CraftorAbility))
124                                {
125                                        msg.source().tell(L("This recipe book is un-prepped."));
126                                        return false;
127                                }
128                                for(String r : recipes)
129                                {
130                                        List<String> V=CMParms.parseTabs(r+" ", false);
131                                        if(V.size()<2)
132                                        {
133                                                msg.source().tell(L("Your recipes are clearly bad."));
134                                                return false;
135                                        }
136                                        final CraftorAbility C=(CraftorAbility)A;
137                                        final String components = C.getDecodedComponentsDescription( msg.source(), V );
138                                        if(components.equals("?"))
139                                        {
140                                                msg.source().tell(L("Your recipes are clearly bad."));
141                                                return false;
142                                        }
143                                        else
144                                        {
145                                                if(this.getRecipeCodeLines().length>0)
146                                                {
147                                                        boolean found=false;
148                                                        for(int i=0;i<this.getRecipeCodeLines().length;i++)
149                                                        {
150                                                                if(CMParms.parseTabs(this.getRecipeCodeLines()[i]+" ", false).size() == V.size())
151                                                                {
152                                                                        found=true;
153                                                                        break;
154                                                                }
155                                                        }
156                                                        if(!found)
157                                                        {
158                                                                msg.source().tell(L("Your recipes might be bad?"));
159                                                                return false;
160                                                        }
161                                                }
162                                        }
163                                }
164                        }
165                        else
166                        {
167                                msg.source().tell(L("Write what recipe?"));
168                                return false;
169                        }
170                }
171                return true;
172        }

174        protected void executeMsg(final CMMsg msg)
175        {
176                // the order that these things are checked in should
177                // be holy, and etched in stone.
178                if(numBehaviors()>0)
179                {
180                        eachBehavior(new EachApplicable<Behavior>()
181                        { 
182                                @Override
183                                public final void apply(final Behavior B)
184                                {
185                                        B.executeMsg(me,msg);
186                                } 
187                        });
188                }
189                if(numScripts()>0)
190                {
191                        eachScript(new EachApplicable<ScriptingEngine>()
192                        { 
193                                @Override
194                                public final void apply(final ScriptingEngine S)
195                                {
196                                        S.executeMsg(me,msg);
197                                } 
198                        });
199                }
200                if(numEffects()>0)
201                {
202                        eachEffect(new EachApplicable<Ability>()
203                        { 
204                                @Override
205                                public final void apply(final Ability A)
206                                {
207                                        A.executeMsg(me, msg);
208                                }
209                        });
210                }
211        }

213        @Override
214        public void executeMsg(final Environmental myHost, final CMMsg msg)
215        {
216                if(msg.amITarget( this ) 
217                && (msg.targetMinor()==CMMsg.TYP_READ) 
218                && (recipeLines.length>0))
219                {
220                        this.executeMsg(msg);
221                        if((msg.targetMessage()!=null)
222                        &&(CMath.isInteger(msg.targetMessage())))
223                        {
224                                final StringBuilder str = new StringBuilder("");
225                                int x=CMath.s_int(msg.targetMessage());
226                                if((x>0)&&(x<=this.getRecipeCodeLines().length))
227                                {
228                                        final Ability A=CMClass.getAbility( getCommonSkillID() );
229                                        if(A!=null)
230                                                str.append( L("The following recipe is for the @x1 skill:\n\r",A.name()));
231                                        if(A instanceof CraftorAbility)
232                                        {
233                                                final CraftorAbility C=(CraftorAbility)A;
234                                                final List<String> V=CMParms.parseTabs( this.getRecipeCodeLines()[x-1]+" ", false );
235                                                final Pair<String,Integer> nameAndLevel = C.getDecodedItemNameAndLevel( V );
236                                                final String components = C.getDecodedComponentsDescription( msg.source(), V );
237                                                final String name=CMStrings.replaceAll( nameAndLevel.first, "% ", "");
238        
239                                                str.append( name).append(", level "+nameAndLevel.second);
240                                                if(CMath.s_int(components)>0)
241                                                        str.append( L(", which requires @x1 standard components.\n\r",components));
242                                                else
243                                                        str.append( L(", which requires: @x1.\n\r",components));
244                                        }
245                                }
246                                CMMsg readMsg=CMClass.getMsg(msg.source(), msg.target(), msg.tool(), 
247                                                 CMMsg.MSG_WASREAD|CMMsg.MASK_ALWAYS, L("It says '@x1'.",str.toString()),
248                                                 CMMsg.MSG_WASREAD|CMMsg.MASK_ALWAYS, this.getRecipeCodeLines()[x-1]+" ", 
249                                                 CMMsg.NO_EFFECT, null);
250                                msg.addTrailerMsg(readMsg);
251                        }
252                        else
253                        {
254                                final StringBuilder str = new StringBuilder("");
255                                final Ability A=CMClass.getAbility( getCommonSkillID() );
256                                if(getTotalRecipePages() > 1)
257                                {
258                                        str.append( L("@x1 contains @x2 recipe(s)/schematic(s) out of @x3 total entries.\n\r",name(),""+recipeLines.length,""+getTotalRecipePages()));
259                                        if(A!=null)
260                                                str.append( L("The following recipes are for the @x1 skill:\n\r",A.name()));
261                                }
262                                else
263                                {
264                                        if(A!=null)
265                                                str.append( L("The following recipe is for the @x1 skill:\n\r",A.name()));
266                                }
267                                StringBuilder rawCodes=new StringBuilder("");
268                                if(A instanceof CraftorAbility)
269                                {
270                                        final CraftorAbility C=(CraftorAbility)A;
271                                        int lineNum=1;
272                                        for(final String line : recipeLines)
273                                        {
274                                                rawCodes.append(line).append("\n\r");
275                                                final List<String> V=CMParms.parseTabs( line+" ", false );
276                                                final Pair<String,Integer> nameAndLevel = C.getDecodedItemNameAndLevel( V );
277                                                final String components = C.getDecodedComponentsDescription( msg.source(), V );
278                                                final String name=CMStrings.replaceAll( nameAndLevel.first, "% ", "");
279        
280                                                str.append(lineNum).append(") ").append( name).append(", level "+nameAndLevel.second);
281                                                if(CMath.s_int(components)>0)
282                                                        str.append( L(", which requires @x1 standard components.\n\r",components));
283                                                else
284                                                        str.append( L(", which requires: @x1.\n\r",components));
285                                                lineNum++;
286                                        }
287                                }
288                                String writing=str.substring( 0, str.length()-2 );
289                                CMMsg readMsg=CMClass.getMsg(msg.source(), msg.target(), msg.tool(), 
290                                                 CMMsg.MSG_WASREAD|CMMsg.MASK_ALWAYS, L("It says '@x1'.",writing),
291                                                 CMMsg.MSG_WASREAD|CMMsg.MASK_ALWAYS, rawCodes.toString()+" ", 
292                                                 CMMsg.NO_EFFECT, null);
293                                msg.addTrailerMsg(readMsg);
294                        }
295                        return;
296                }
297                if(msg.amITarget( this ) 
298                && (msg.targetMinor()==CMMsg.TYP_WRITE))
299                {
300                        this.executeMsg(msg);
301                        final List<String> recipes=CMParms.parseAny(msg.targetMessage(), "\n\r", true);
302                        if(recipes.size()>0)
303                        {
304                                final int max=this.getTotalRecipePages();
305                                final int num=this.getRecipeCodeLines().length;
306                                for(int i=recipes.size()-1;i>=0;i--)
307                                {
308                                        if(recipes.get(i).trim().length()==0)
309                                                recipes.remove(i);
310                                }
311                                if((max-num)>=recipes.size())
312                                {
313                                        int startPos = this.recipeLines.length;
314                                        String[] newRecipeLines = Arrays.copyOf(this.recipeLines, this.recipeLines.length + recipes.size());
315                                        for(int i=0;i<recipes.size();i++)
316                                                newRecipeLines[i+startPos]=recipes.get(i);
317                                        this.setRecipeCodeLines(newRecipeLines);
318                                }
319                        }
320                        return;
321                }
322                super.executeMsg( myHost, msg );
323        }

331        @Override
332        public void recoverPhyStats()
333        {
334                CMLib.flags().setReadable(this,true);
335                super.recoverPhyStats();
336                if(this.getTotalRecipePages()==1)
337                {
338                        if(replaceName==null)
339                        {
340                                replaceName="";
341                                final int x=Name().indexOf( '%' );
342                                if((recipeLines!=null)&&(recipeLines.length==1)&&(this.getCommonSkillID().length()>0))
343                                {
344                                        final Ability A=CMClass.getAbility(this.getCommonSkillID());
345                                        if(A instanceof CraftorAbility)
346                                        {
347                                                final List<String> V=CMParms.parseTabs( recipeLines[0], false );
348                                                final Pair<String,Integer> nameAndLevel = ((CraftorAbility)A).getDecodedItemNameAndLevel( V );
349                                                String itemName=CMStrings.replaceAll( nameAndLevel.first, "% ","");
350                                                itemName=CMStrings.replaceAll( itemName, " % ","");
351                                                if(x>=0)
352                                                        replaceName=CMStrings.replaceAll( Name(), "%", itemName );
353                                                else
354                                                if(Name().indexOf(" of ")<0)
355                                                        replaceName=Name()+" of "+itemName;
356                                        }
357                                }
358                                else
359                                if(x>=0)
360                                {
361                                        replaceName=CMStrings.replaceAll( Name(), "%", "" );
362                                }
363                        }
364                        if((replaceName!=null)&&(replaceName.length()>0)&&(phyStats().newName()==null))
365                                phyStats().setName(replaceName);
366                }
367        }

407        private final static String[] MYCODES={"SKILLID","RECIPES","NUMRECIPES"};
408
409        @Override
410        public String getStat(String code)
411        {
412                if(CMLib.coffeeMaker().getGenItemCodeNum(code)>=0)
413                        return CMLib.coffeeMaker().getGenItemStat(this,code);
414                switch(getCodeNum(code))
415                {
416                case 0: 
417                        return ""+getCommonSkillID();
418                case 1:
419                        {
420                                final StringBuilder str=new StringBuilder("");
421                                for(final String s : recipeLines)
422                                        str.append(s).append("\n");
423                                if(str.length()==0)
424                                        return "";
425                                final String recipeStr = str.toString();
426                                return recipeStr.substring(0,recipeStr.length()-1);
427                        }
428                case 2: 
429                        return ""+this.getTotalRecipePages();
430                default:
431                        return CMProps.getStatCodeExtensionValue(getStatCodes(), xtraValues, code);
432                }
433        }
434
435        @Override
436        public void setStat(String code, String val)
437        {
438                if(CMLib.coffeeMaker().getGenItemCodeNum(code)>=0)
439                        CMLib.coffeeMaker().setGenItemStat(this,code,val);
440                else
441                switch(getCodeNum(code))
442                {
443                case 0:
444                        setCommonSkillID(val);
445                        break;
446                case 1:
447                        setRecipeCodeLines(CMParms.parseAny(val, '\n', true).toArray(new String[0]));
448                        break;
449                case 2:
450                        final int x = CMath.s_int(val);
451                        setTotalRecipePages(x > 0 ? x : 1);
452                        break;
453                default:
454                        CMProps.setStatCodeExtensionValue(getStatCodes(), xtraValues, code, val);
455                        break;
456                }
457        }

459        @Override
460        protected int getCodeNum(final String code)
461        {
462                for(int i=0;i<MYCODES.length;i++)
463                {
464                        if(code.equalsIgnoreCase(MYCODES[i]))
465                                return i;
466                }
467                return -1;
468        }
469
470        private static String[] codes=null;
471
472        @Override
473        public String[] getStatCodes()
474        {
475                if(codes!=null)
476                        return codes;
477                final String[] MYCODES=CMProps.getStatCodesList(GenRecipe.MYCODES,this);
478                final String[] superCodes=CMParms.toStringArray(GenericBuilder.GenItemCode.values());
479                codes=new String[superCodes.length+MYCODES.length];
480                int i=0;
481                for(;i<superCodes.length;i++)
482                        codes[i]=superCodes[i];
483                for(int x=0;x<MYCODES.length;i++,x++)
484                        codes[i]=MYCODES[x];
485                return codes;
486        }
487
488        @Override
489        public boolean sameAs(final Environmental E)
490        {
491                if(!(E instanceof GenRecipe))
492                        return false;
493                final String[] codes=getStatCodes();
494                for(int i=0;i<codes.length;i++)
495                {
496                        if(!E.getStat(codes[i]).equals(getStat(codes[i])))
497                                return false;
498                }
499                return true;
500        }
