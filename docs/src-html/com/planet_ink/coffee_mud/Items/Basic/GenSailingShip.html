160        protected void announceToDeck(final CMMsg msg)
161        {
162                MOB mob = null;
163                final MOB msgSrc = msg.source();
164                try
165                {
166                        final Area A=this.getShipArea();
167                        if(A!=null)
168                        {
169                                Room mobR = null;
170                                for(final Enumeration<Room> r=A.getProperMap();r.hasMoreElements();)
171                                {
172                                        final Room R=r.nextElement();
173                                        if((R!=null) && ((R.domainType()&Room.INDOORS)==0))
174                                        {
175                                                mobR=R;
176                                                break;
177                                        }
178                                }
179                                if(mobR!=null)
180                                {
181                                        mob = CMClass.getFactoryMOB(name(),phyStats().level(),mobR);
182                                        msg.setSource(mob);
183                                        for(final Enumeration<Room> r=A.getProperMap();r.hasMoreElements();)
184                                        {
185                                                final Room R=r.nextElement();
186                                                if((R!=null) && ((R.domainType()&Room.INDOORS)==0) && (R.okMessage(mob, msg)))
187                                                {
188                                                        if(R == mobR)
189                                                                R.send(mob, msg); // this lets the source know, i guess
190                                                        else
191                                                                R.sendOthers(mob, msg); // this lets the source know, i guess
192                                                }
193                                        }
194                                }
195                        }
196                }
197                finally
198                {
199                        msg.setSource(msgSrc);
200                        if(mob != null)
201                                mob.destroy();
202                }

316        @Override
317        public boolean okMessage(final Environmental myHost, final CMMsg msg)
318        {
319                switch(msg.sourceMinor())
320                {
321                case CMMsg.TYP_HUH:
322                case CMMsg.TYP_COMMANDFAIL:
323                case CMMsg.TYP_COMMAND:
324                        break;
325                default:
326                        if(!confirmAreaMessage(msg, true))
327                                return false;
328                        break;
329                }
330                
331                if((msg.target() == this)
332                &&(msg.tool()!=null)
333                &&(msg.tool().ID().equals("AWaterCurrent")))
334                {
335                        if(anchorDown)
336                                return false;
337                }
338                else
339                if((msg.sourceMinor()==CMMsg.TYP_HUH)
340                &&(msg.targetMessage()!=null)
341                &&(area == CMLib.map().areaLocation(msg.source())))
342                {
343                        final List<String> cmds=CMParms.parse(msg.targetMessage());
344                        if(cmds.size()==0)
345                                return true;
346                        final String word=cmds.get(0).toUpperCase();
347                        final String secondWord=(cmds.size()>1) ? cmds.get(1).toUpperCase() : "";
348                        SailingCommand cmd=null;
349                        if(secondWord.length()>0)
350                                cmd = (SailingCommand)CMath.s_valueOf(SailingCommand.class, word+"_"+secondWord);
351                        if(cmd == null)
352                                cmd = (SailingCommand)CMath.s_valueOf(SailingCommand.class, word);
353                        if(cmd != null)
354                        {
355                                switch(cmd)
356                                {
357                                case TARGET:
358                                {
359                                        if(cmds.size()==1)
360                                        {
361                                                msg.source().tell(L("You must specify another ship to target."));
362                                                return false;
363                                        }
364                                        final Room thisRoom = (Room)owner();
365                                        if(thisRoom==null)
366                                        {
367                                                msg.source().tell(L("This ship is nowhere to be found!"));
368                                                return false;
369                                        }
370                                        String rest = CMParms.combine(cmds,1);
371                                        Boolean result = startAttack(msg.source(),thisRoom,rest);
372                                        if(result  == Boolean.TRUE)
373                                        {
374                                                if(this.targetedShip != null)
375                                                        msg.source().tell(L("You are now targeting @x1.",this.targetedShip.Name()));
376                                                msg.source().tell(getTargetedShipInfo());
377                                        }
378                                        else
379                                        if(result  == Boolean.FALSE)
380                                                return false;
381                                        else
382                                        {
383                                                msg.source().tell(L("You don't see '@x1' here to target",rest));
384                                                return false;
385                                        }
386                                        break;
387                                }
388                                case TENDER:
389                                {
390                                        if(cmds.size()==1)
391                                        {
392                                                msg.source().tell(L("You must specify another ship to offer aboard."));
393                                                return false;
394                                        }
395                                        final Room thisRoom = (Room)owner();
396                                        if(thisRoom==null)
397                                        {
398                                                msg.source().tell(L("This ship is nowhere to be found!"));
399                                                return false;
400                                        }
401                                        if(this.targetedShip!=null)
402                                        {
403                                                msg.source().tell(L("Not while you are in combat!"));
404                                                return false;
405                                        }
406                                        String rest = CMParms.combine(cmds,1);
407                                        final Item I=thisRoom.findItem(rest);
408                                        if((I instanceof GenSailingShip)&&(I!=this)&&(CMLib.flags().canBeSeenBy(I, msg.source())))
409                                        {
410                                                GenSailingShip otherShip = (GenSailingShip)I;
411                                                if(otherShip.targetedShip != null)
412                                                {
413                                                        msg.source().tell(L("Not while @x1 is in in combat!",otherShip.Name()));
414                                                        return false;
415                                                }
416                                                final MOB mob = CMClass.getFactoryMOB(name(),phyStats().level(),thisRoom);
417                                                try
418                                                {
419                                                        mob.setRiding(this);
420                                                        mob.basePhyStats().setDisposition(mob.basePhyStats().disposition()|PhyStats.IS_SWIMMING);
421                                                        mob.phyStats().setDisposition(mob.phyStats().disposition()|PhyStats.IS_SWIMMING);
422                                                        if(otherShip.tenderShip == this)
423                                                        {
424                                                                if(thisRoom.show(mob, otherShip, CMMsg.MSG_NOISYMOVEMENT, L("<S-NAME> connect(s) her gangplank with <T-NAME>")))
425                                                                {
426                                                                        this.tenderShip = otherShip;
427                                                                        final BoardableShip myArea=(BoardableShip)this.getShipArea();
428                                                                        final BoardableShip hisArea=(BoardableShip)otherShip.getShipArea();
429                                                                        
430                                                                        final Room hisExitRoom = hisArea.unDock(false);
431                                                                        final Room myExitRoom = myArea.unDock(false);
432                                                                        myArea.dockHere(hisExitRoom);
433                                                                        hisArea.dockHere(myExitRoom);
434                                                                }
435                                                        }
436                                                        else
437                                                        {
438                                                                if(thisRoom.show(mob, otherShip, CMMsg.MSG_NOISYMOVEMENT, L("<S-NAME> extend(s) her gangplank toward(s) <T-NAME>")))
439                                                                        this.tenderShip = otherShip;
440                                                        }
441                                                }
442                                                finally
443                                                {
444                                                        mob.destroy();
445                                                }
446                                        }
447                                        else
448                                        {
449                                                msg.source().tell(L("You don't see the ship '@x1' here to board",rest));
450                                                return false;
451                                        }
452                                        break;
453                                }
454                                case SINK:
455                                {
456                                        if(!CMSecurity.isAllowedEverywhere(msg.source(), CMSecurity.SecFlag.CMDROOMS))
457                                                return true;
458                                        CMMsg damageMsg=CMClass.getMsg(msg.source(), this, CMMsg.MSG_DAMAGE, "SINK!!!");
459                                        damageMsg.setValue(99999);
460                                        this.executeMsg(this, damageMsg);
461                                        return false;
462                                }
463                                case AIM:
464                                {
465                                        final Room thisRoom = (Room)owner();
466                                        if(thisRoom==null)
467                                        {
468                                                msg.source().tell(L("This ship is nowhere to be found!"));
469                                                return false;
470                                        }
471                                        if((!this.amInTacticalMode())
472                                        ||(this.targetedShip==null)
473                                        ||(!thisRoom.isContent((Item)this.targetedShip)))
474                                        {
475                                                msg.source().tell(L("You ship must be targeting an enemy to aim weapons."));
476                                                return false;
477                                        }
478                                        if(cmds.size()<3)
479                                        {
480                                                msg.source().tell(L("Aim what weapon how far ahead?"));
481                                                msg.source().tell(getTargetedShipInfo());
482                                                return false;
483                                        }
484                                        String leadStr = cmds.remove(cmds.size()-1);
485                                        String weaponStr = CMParms.combine(cmds,1);
486                                        final Room mobR=msg.source().location();
487                                        if((!CMath.isInteger(leadStr))||(CMath.s_int(leadStr)<0))
488                                        {
489                                                if(this.targetedShip!=null)
490                                                        msg.source().tell(L("'@x1' is not a valid distance ahead of @x2 to fire.",leadStr,this.targetedShip.name()));
491                                                else
492                                                        msg.source().tell(L("'@x1' is not a valid distance.",leadStr));
493                                                return false;
494                                        }
495                                        if(mobR!=null)
496                                        {
497                                                Item I=mobR.findItem(null, weaponStr);
498                                                if((I==null)||(!CMLib.flags().canBeSeenBy(I,msg.source())))
499                                                {
500                                                        msg.source().tell(L("You don't see any siege weapon called '@x1' here.",leadStr));
501                                                        return false;
502                                                }
503                                                if(!CMLib.combat().isAShipSiegeWeapon(I))
504                                                {
505                                                        msg.source().tell(L("@x1 is not a useable siege weapon.",leadStr));
506                                                        return false;
507                                                }
508                                                AmmunitionWeapon weapon=(AmmunitionWeapon)I;
509                                                int distance = weapon.maxRange();
510                                                int[] targetCoords = new int[2];
511                                                int leadAmt=0;
512                                                if(this.targetedShip instanceof GenSailingShip)
513                                                {
514                                                        targetCoords = ((GenSailingShip)this.targetedShip).getMyCoords();
515                                                        int direction = ((GenSailingShip)this.targetedShip).directionFacing;
516                                                        if(targetCoords == null)
517                                                        {
518                                                                msg.source().tell(L("You ship must be targeting an enemy to aim weapons."));
519                                                                return false;
520                                                        }
521                                                        distance = this.getTacticalDistance(this.targetedShip);
522                                                        leadAmt = CMath.s_int(leadStr);
523                                                        for(int i=0;i<leadAmt;i++)
524                                                                targetCoords = Directions.adjustXYByDirections(targetCoords[0], targetCoords[1], direction);
525                                                }
526                                                if((weapon.maxRange() < distance)||(weapon.minRange() > distance))
527                                                {
528                                                        msg.source().tell(L("Your target is presently at distance of @x1, but this weapons range is @x2 to @x3.",
529                                                                                                ""+distance,""+weapon.minRange(),""+weapon.maxRange()));
530                                                        return false;
531                                                }
532                                                if(weapon.requiresAmmunition() 
533                                                && (weapon.ammunitionCapacity() > 0)
534                                                && (weapon.ammunitionRemaining() == 0))
535                                                {
536                                                        msg.source().tell(L("@x1 needs to be LOADed first.",weapon.Name()));
537                                                        return false;
538                                                }
539                                                String timeToFire=""+(CMLib.threads().msToNextTick((Tickable)CMLib.combat(), Tickable.TICKID_SUPPORT|Tickable.TICKID_SOLITARYMASK) / 1000);
540                                                String msgStr=L("<S-NAME> aim(s) <O-NAME> at <T-NAME> (@x1).",""+leadAmt);
541                                                if(msg.source().isMonster() && aimings.containsFirst(weapon))
542                                                {
543                                                        msg.source().tell(L("@x1 is already aimed.",weapon.Name()));
544                                                        return false;
545                                                }
546                                                CMMsg msg2=CMClass.getMsg(msg.source(), targetedShip, weapon, CMMsg.MSG_NOISYMOVEMENT, msgStr);
547                                                if(mobR.okMessage(msg.source(), msg2))
548                                                {
549                                                        this.aimings.removeFirst(weapon);
550                                                        this.aimings.add(new Pair<Weapon,int[]>(weapon,targetCoords));
551                                                        mobR.send(msg.source(), msg2);
552                                                        msg.source().tell(L("@x1 is now aimed and will be fired in @x2 seconds.",I.name(),timeToFire));
553                                                }
554                                        }
555                                        return false;
556                                }
557                                case RAISE:
558                                {
559                                        if(!securityCheck(msg.source()))
560                                        {
561                                                msg.source().tell(L("The captain does not permit you."));
562                                                return false;
563                                        }
564                                        final Room R=CMLib.map().roomLocation(this);
565                                        final Room targetR=msg.source().location();
566                                        if((R!=null)&&(targetR!=null))
567                                        {
568                                                if(((targetR.domainType()&Room.INDOORS)==0)
569                                                && (targetR.domainType()!=Room.DOMAIN_OUTDOORS_AIR)
570                                                &&(msg.source().isPlayer()))
571                                                {
572                                                        msg.source().tell(L("You must be on deck to raise a tendered ship."));
573                                                        return false;
574                                                }
575                                                String rest = CMParms.combine(cmds,1);
576                                                final Item I=R.findItem(rest);
577                                                if((I!=this)&&(CMLib.flags().canBeSeenBy(I, msg.source())))
578                                                {
579                                                        if((I instanceof Rideable)
580                                                        &&(((Rideable)I).mobileRideBasis()))
581                                                        {
582                                                                if(smallTenderRequests.contains(I))
583                                                                {
584                                                                        final MOB riderM=getBestRider(R,(Rideable)I);
585                                                                        if(((riderM==null)||(R.show(riderM, R, CMMsg.MSG_LEAVE, null)))
586                                                                        &&(R.show(msg.source(), I, CMMsg.MSG_NOISYMOVEMENT, L("<S-NAME> raise(s) <T-NAME> up onto @x1.",name())))
587                                                                        &&((riderM==null)||(R.show(riderM, targetR, CMMsg.MSG_ENTER, null))))
588                                                                        {
589                                                                                smallTenderRequests.remove(I);
590                                                                                targetR.moveItemTo(I, Expire.Never, Move.Followers);
591                                                                        }
592                                                                        return false;
593                                                                }
594                                                                else
595                                                                {
596                                                                        msg.source().tell(L("You can only raise @x1 once it has tendered itself to this one.",I.name()));
597                                                                        return false;
598                                                                }
599                                                        }
600                                                        else
601                                                        {
602                                                                msg.source().tell(L("You don't think @x1 is a suitable boat.",I.name()));
603                                                                return false;
