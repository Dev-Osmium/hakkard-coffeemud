037public class GenCaged extends GenItem implements CagedAnimal
038{
039        @Override
040        public String ID()
041        {
042                return "GenCaged";
043        }
044
045        public GenCaged()
046        {
047                super();
048                setName("a caged creature");
049                basePhyStats.setWeight(150);
050                setDisplayText("a caged creature sits here.");
051                setDescription("");
052                baseGoldValue=5;
053                basePhyStats().setLevel(1);
054                setMaterial(RawMaterial.RESOURCE_MEAT);
055                recoverPhyStats();
056        }
057
058        protected byte[]        readableText=null;
059
060        @Override
061        public String readableText()
062        {
063                return readableText==null?"":CMLib.encoder().decompressString(readableText);
064        }
065
066        @Override
067        public void setReadableText(String text)
068        {
069                readableText=(text.trim().length()==0)?null:CMLib.encoder().compressString(text);
070        }
071
072        @Override
073        public void recoverPhyStats()
074        {
075                super.recoverPhyStats();
076                phyStats().setSensesMask(phyStats.sensesMask() | PhyStats.SENSE_ITEMNOWISH |PhyStats.SENSE_ITEMNORUIN);
077        }
078        
079        @Override
080        public boolean cageMe(MOB M)
081        {
082                if(M==null)
083                        return false;
084                if(!M.isMonster())
085                        return false;
086                name=M.Name();
087                displayText=M.displayText();
088                setDescription(M.description());
089                basePhyStats().setLevel(M.basePhyStats().level());
090                basePhyStats().setWeight(M.basePhyStats().weight());
091                basePhyStats().setHeight(M.basePhyStats().height());
092                final StringBuffer itemstr=new StringBuffer("");
093                itemstr.append("<MOBITEM>");
094                itemstr.append(CMLib.xml().convertXMLtoTag("MICLASS",CMClass.classID(M)));
095                itemstr.append(CMLib.xml().convertXMLtoTag("MISTART",CMLib.map().getExtendedRoomID(M.getStartRoom())));
096                itemstr.append(CMLib.xml().convertXMLtoTag("MIDATA",CMLib.coffeeMaker().getPropertiesStr(M,true)));
097                itemstr.append("</MOBITEM>");
098                setCageText(itemstr.toString());
099                recoverPhyStats();
100                return true;
101        }
102
103        @Override
104        public void destroy()
105        {
106                if((CMSecurity.isDebugging(CMSecurity.DbgFlag.MISSINGKIDS))
107                &&(fetchEffect("Age")!=null)
108                &&CMath.isInteger(fetchEffect("Age").text())
109                &&(CMath.s_int(fetchEffect("Age").text())>Short.MAX_VALUE))
110                        Log.debugOut("MISSKIDS",new Exception(Name()+" went missing form "+CMLib.map().getDescriptiveExtendedRoomID(CMLib.map().roomLocation(this))));
111                super.destroy();
112        }
113
114        @Override
115        public void executeMsg(final Environmental myHost, final CMMsg msg)
116        {
117                if((msg.amITarget(this)
118                        ||((msg.tool()==this)&&(msg.target()==container())&&(container()!=null)))
119                &&((getCageFlagsBitmap()&CagedAnimal.CAGEFLAG_TO_MOB_PROGRAMMATICALLY)==0)
120                &&((msg.targetMinor()==CMMsg.TYP_GET)||(msg.targetMinor()==CMMsg.TYP_DROP)))
121                {
122                        final MOB M=unCageMe();
123                        if((M!=null)&&(msg.source().location()!=null))
124                                M.bringToLife(msg.source().location(),true);
125                        destroy();
126                        return;
127                }
128                super.executeMsg(myHost,msg);
129        }
130
131        @Override
132        public MOB unCageMe()
133        {
134                MOB M=null;
135                if(cageText().length()==0)
136                        return M;
137                final List<XMLLibrary.XMLTag> buf=CMLib.xml().parseAllXML(cageText());
138                if(buf==null)
139                {
140                        Log.errOut("Caged","Error parsing 'MOBITEM'.");
141                        return M;
142                }
143                final XMLTag iblk=CMLib.xml().getPieceFromPieces(buf,"MOBITEM");
144                if((iblk==null)||(iblk.contents()==null))
145                {
146                        Log.errOut("Caged","Error parsing 'MOBITEM'.");
147                        return M;
148                }
149                final String itemi=iblk.getValFromPieces("MICLASS");
150                final String startr=iblk.getValFromPieces("MISTART");
151                final Environmental newOne=CMClass.getMOB(itemi);
152                final List<XMLLibrary.XMLTag> idat=iblk.getContentsFromPieces("MIDATA");
153                if((idat==null)||(newOne==null)||(!(newOne instanceof MOB)))
154                {
155                        Log.errOut("Caged","Error parsing 'MOBITEM' data.");
156                        return M;
157                }
158                CMLib.coffeeMaker().setPropertiesStr(newOne,idat,true);
159                M=(MOB)newOne;
160                M.basePhyStats().setRejuv(PhyStats.NO_REJUV);
161                M.setStartRoom(null);
162                if(M.isGeneric())
163                        CMLib.coffeeMaker().resetGenMOB(M,M.text());
164                if((startr.length()>0)&&(!startr.equalsIgnoreCase("null")))
165                {
166                        final Room R=CMLib.map().getRoom(startr);
167                        if(R!=null)
168                                M.setStartRoom(R);
169                }
170                return M;
171        }
172
173        @Override
174        public String cageText()
175        {
176                return CMLib.xml().restoreAngleBrackets(readableText());
177        }
178
179        @Override
180        public void setCageText(String text)
181        {
182                setReadableText(CMLib.xml().parseOutAngleBrackets(text));
183                CMLib.flags().setReadable(this,false);
184        }
185        
186        @Override
187        public int getCageFlagsBitmap()
188        {
189                return basePhyStats().ability();
190        }
191
192        @Override
193        public void setCageFlagsBitmap(int bitmap)
194        {
195                basePhyStats.setAbility(bitmap);
196                phyStats.setAbility(bitmap);
197        }
