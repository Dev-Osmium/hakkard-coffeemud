045        public GenPackagedVariety()
046        {
047                super();
048                setName("a package of things");
049                basePhyStats.setWeight(150);
050                setDisplayText("a package sits here.");
051                setDescription("");
052                baseGoldValue=5;
053                basePhyStats().setLevel(1);
054                setMaterial(RawMaterial.RESOURCE_WOOD);
055                recoverPhyStats();
056        }

073        @Override
074        public boolean packageMe(Item I, int number)
075        {
076                if((I==null)
077                ||(I.amDestroyed()))
078                        return false;
079                CMLib.utensils().disInvokeEffects(I);
080                final String name = (number < 2) ? I.Name() : I.Name() + " (x"+number+")";
081                if(description().length()==0)
082                        setDescription("The contents of the package appears as follows:\n\r"+name);
083                else
084                        setDescription(description()+"\n\r"+name);
085                basePhyStats().setWeight(basePhyStats().weight() + (I.basePhyStats().weight()*number));
086                if(I.basePhyStats().height() > basePhyStats().height())
087                        basePhyStats().setHeight(I.basePhyStats().height());
088                if(I.basePhyStats().level() > basePhyStats().level())
089                        basePhyStats().setLevel(I.basePhyStats().level());
090                setBaseValue(baseGoldValue() + (I.baseGoldValue()*number));
091                final StringBuffer itemstr=new StringBuffer("");
092                itemstr.append("<PAKITEM>");
093                itemstr.append(CMLib.xml().convertXMLtoTag("PINUM",""+number));
094                itemstr.append(CMLib.xml().convertXMLtoTag("PICLASS",CMClass.classID(I)));
095                itemstr.append(CMLib.xml().convertXMLtoTag("PIDATA",CMLib.coffeeMaker().getPropertiesStr(I,true)));
096                itemstr.append("</PAKITEM>");
097                setNumberOfItemsInPackage(this.numberOfItemsInPackage() + number);
098                setPackageText(packageText() + itemstr.toString());
099                recoverPhyStats();
100                return true;
101        }

103        @Override
104        public void executeMsg(final Environmental myHost, final CMMsg msg)
105        {
106                if((msg.amITarget(this)
107                        ||((msg.tool()==this)&&(msg.target() instanceof Container)))
108                &&((getPackageFlagsBitmap()&PACKAGE_FLAG_TO_ITEMS_PROGRAMMATICALLY)==0)
109                &&((msg.targetMinor()==CMMsg.TYP_GET)||(msg.targetMinor()==CMMsg.TYP_DROP)))
110                {
111                        ItemPossessor possessor = owner();
112                        if((msg.targetMinor()==CMMsg.TYP_DROP)&&(msg.target() instanceof Room))
113                                possessor=(Room)msg.target();
114                        List<Item> items = unPackage(Integer.MAX_VALUE);
115                        for(Item I : items)
116                                possessor.addItem(I, ItemPossessor.Expire.Player_Drop);
117                        destroy();
118                        return;
119                }
120                super.executeMsg(myHost,msg);
121        }

133        @Override
134        public Item peekFirstItem()
135        {
136                if(packageText().length()==0)
137                        return null;
138                final List<XMLLibrary.XMLTag> buf=CMLib.xml().parseAllXML(packageText());
139                if(buf==null)
140                {
141                        Log.errOut("Packaged","Error parsing 'PAKITEM'.");
142                        return null;
143                }
144                final XMLTag iblk=CMLib.xml().getPieceFromPieces(buf,"PAKITEM");
145                if((iblk==null)||(iblk.contents()==null))
146                {
147                        Log.errOut("Packaged","Error parsing 'PAKITEM'.");
148                        return null;
149                }
150                final String itemi=iblk.getValFromPieces("PICLASS");
151                final Environmental newOne=CMClass.getItem(itemi);
152                final List<XMLLibrary.XMLTag> idat=iblk.getContentsFromPieces("PIDATA");
153                if((idat==null)||(newOne==null)||(!(newOne instanceof Item)))
154                {
155                        Log.errOut("Packaged","Error parsing 'PAKITEM' data.");
156                        return null;
157                }
158                CMLib.coffeeMaker().setPropertiesStr(newOne,idat,true);
159                return (Item)newOne;
160        }

168        @Override
169        public List<Item> unPackage(int number)
170        {
171                final List<Item> V=new Vector<Item>();
172                int numberInPackage=numberOfItemsInPackage();
173                if(number>=numberInPackage)
174                        number=numberInPackage;
175                if(number<=0)
176                        return V;
177                Item firstItem = null;
178                final List<XMLLibrary.XMLTag> buf=CMLib.xml().parseAllXML(packageText());
179                StringBuilder newXml=new StringBuilder("");
180                if(buf!=null)
181                {
182                        for(int p=0;p<buf.size();p++)
183                        {
184                                final XMLTag iblk=buf.get(p);
185                                if((iblk!=null)&&(iblk.contents()!=null)&&(iblk.tag().equals("PAKITEM")))
186                                {
187                                        if(number<=0)
188                                                newXml.append("<PAKITEM>"+iblk.value()+"</PAKITEM>");
189                                        else
190                                        {
191                                                final int numOfThese = iblk.getIntFromPieces("PINUM");
192                                                final String itemi=iblk.getValFromPieces("PICLASS");
193                                                final Environmental newOne=CMClass.getItem(itemi);
194                                                final List<XMLLibrary.XMLTag> idat=iblk.getContentsFromPieces("PIDATA");
195                                                if((idat!=null)&&(newOne!=null)&&(newOne instanceof Item))
196                                                {
197                                                        Item I=(Item)newOne;
198                                                        CMLib.coffeeMaker().setPropertiesStr(newOne,idat,true);
199                                                        for(int i=0;i<numOfThese;i++)
200                                                        {
201                                                                if(number<=0)
202                                                                {
203                                                                        newXml.append("<PAKITEM>");
204                                                                        newXml.append(CMLib.xml().convertXMLtoTag("PINUM",""+(numOfThese - i)));
205                                                                        newXml.append(CMLib.xml().convertXMLtoTag("PICLASS",itemi));
206                                                                        newXml.append(CMLib.xml().convertXMLtoTag("PIDATA",iblk.getValFromPieces("PIDATA")));
207                                                                        newXml.append("</PAKITEM>");
208                                                                }
209                                                                else
210                                                                {
211                                                                        I=(Item)newOne.copyOf();
212                                                                        if(basePhyStats().weight() > I.basePhyStats().weight())
213                                                                                basePhyStats().setWeight(basePhyStats().weight() - I.basePhyStats().weight());
214                                                                        if(baseGoldValue() > I.baseGoldValue())
215                                                                                setBaseValue(baseGoldValue() - I.baseGoldValue());
216                                                                        if(firstItem instanceof Container)
217                                                                                I.setContainer((Container)firstItem);
218                                                                        if(firstItem == null)
219                                                                                firstItem = I;
220                                                                        number--;
221                                                                        numberInPackage--;
222                                                                        V.add(I);
223                                                                }
224                                                        }
225                                                }
226                                        }
227                                }
228                        }
229                }
230                setNumberOfItemsInPackage(numberInPackage);
231                if(numberInPackage<=0)
232                {
233                        destroy();
234                        return V;
235                }
236                recoverPhyStats();
237                return V;
238        }

250                CMLib.flags().setReadable(this,false);
251        }
252        
253        @Override
254        public int numberOfItemsInPackage()
255        {
256                if(this.numberOfItemsInPackage <= 0)
257                {
258                        final List<XMLLibrary.XMLTag> buf=CMLib.xml().parseAllXML(packageText());
259                        if(buf!=null)
260                        {
261                                for(int p=0;p<buf.size();p++)
262                                {
263                                        final XMLTag iblk=buf.get(p);
264                                        if((iblk!=null)&&(iblk.contents()!=null))
265                                                this.numberOfItemsInPackage += iblk.getIntFromPieces("PINUM");
266                                }
267                        }
268                }
269                return numberOfItemsInPackage;
270        }
