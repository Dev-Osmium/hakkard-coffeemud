035public class StdCage extends StdContainer
036{
037        @Override
038        public String ID()
039        {
040                return "StdCage";
041        }
042
043        public StdCage()
044        {
045                super();
046                setName("a cage");
047                setDisplayText("a cage sits here.");
048                setDescription("It\\`s of solid wood construction with metal bracings.  The door has a key hole.");
049                capacity=1000;
050                setContainTypes(Container.CONTAIN_BODIES|Container.CONTAIN_CAGED);
051                material=RawMaterial.RESOURCE_OAK;
052                baseGoldValue=15;
053                basePhyStats().setWeight(25);
054                recoverPhyStats();
055        }
056
057        @Override
058        public boolean tick(final Tickable ticking, final int tickID)
059        {
060                if((tickID==Tickable.TICKID_EXIT_REOPEN)&&(isOpen()))
061                {
062                        final Room R=CMLib.map().roomLocation(this);
063                        if((R!=null)&&(owner() instanceof Room)&&(CMLib.flags().isInTheGame(this,true)))
064                        {
065                                final List<Item> mobContents=getContents();
066                                for (final Item item : mobContents)
067                                {
068                                        final Environmental E=item;
069                                        if(E instanceof CagedAnimal)
070                                        {
071                                                final MOB M=((CagedAnimal)E).unCageMe();
072                                                if(M!=null)
073                                                        M.bringToLife(R,true);
074                                                R.show(M,null,this,CMMsg.MSG_OK_ACTION,L("<S-NAME> escapes from <O-NAME>!"));
075                                                E.destroy();
076                                        }
077                                }
078                        }
079                }
080                return super.tick(ticking,tickID);
081        }
082
083        @Override
084        public void executeMsg(final Environmental myHost, final CMMsg msg)
085        {
086                if(msg.amITarget(this))
087                {
088                        switch(msg.targetMinor())
089                        {
090                        case CMMsg.TYP_CLOSE:
091                                if((hasALid)&&(isOpen))
092                                {
093                                        if(CMLib.threads().isTicking(this,Tickable.TICKID_EXIT_REOPEN))
094                                                CMLib.threads().deleteTick(this,Tickable.TICKID_EXIT_REOPEN);
095                                }
096                                break;
097                        case CMMsg.TYP_OPEN:
098                                if((hasALid)&&(!isOpen)&&(!isLocked))
099                                {
100                                        if((owner() instanceof Room)
101                                        &&(!CMLib.threads().isTicking(this,Tickable.TICKID_EXIT_REOPEN)))
102                                                CMLib.threads().startTickDown(this,Tickable.TICKID_EXIT_REOPEN,30);
103                                }
104                                break;
105                        case CMMsg.TYP_LOOK:
106                        case CMMsg.TYP_EXAMINE:
107                        {
108                                synchronized(this)
109                                {
110                                        if(!isOpen)
111                                        {
112                                                isOpen=true;
113                                                super.executeMsg(myHost,msg);
114                                                isOpen=false;
115                                                return;
116                                        }
117                                }
118                                break;
119                        }
120                        }
121                }
122                super.executeMsg(myHost,msg);
123        }
124}
