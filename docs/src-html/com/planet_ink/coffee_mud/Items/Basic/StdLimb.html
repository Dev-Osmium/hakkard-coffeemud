036public class StdLimb extends StdItem implements FalseLimb
037{
038        @Override
039        public String ID()
040        {
041                return "StdLimb";
042        }
043
044        protected String        raceID                  = "";
045        protected int           partnum                 = -1;
046        protected long          wearplace               = -1;
047        protected short         layer                   = -10;
048        protected short         layerAttributes = 0;
049
050        public StdLimb()
051        {
052                super();
053                setName("a false limb");
054                basePhyStats.setWeight(1);
055                setDisplayText("a false limb is here.");
056                setDescription("Looks like a false limb.");
057                properWornBitmap=Wearable.WORN_HELD|Wearable.WORN_FLOATING_NEARBY;
058                wornLogicalAnd=false;
059                baseGoldValue=10;
060                material=RawMaterial.RESOURCE_MEAT;
061                recoverPhyStats();
062        }
063
064        @Override
065        public void setName(String name)
066        {
067                super.setName(name);
068                wearplace=-1;
069                partnum=-1;
070        }
071
072        @Override
073        public boolean okMessage(final Environmental host, final CMMsg msg)
074        {
075                if(!super.okMessage(host, msg))
076                        return false;
077                if((msg.target()==this)
078                &&(msg.targetMinor()==CMMsg.TYP_REMOVE)
079                &&(!CMath.bset(msg.sourceMajor(),CMMsg.MASK_ALWAYS))
080                &&(owner() instanceof MOB))
081                {
082                        final MOB mob=(MOB)owner();
083                        final Wearable.CODES codes = Wearable.CODES.instance();
084                        for(int w=0;w<codes.total();w++)
085                        {
086                                if((amWearingAt(codes.get(w)))
087                                &&(codes.get(w)!=Wearable.IN_INVENTORY)
088                                &&(codes.dependency_masks()[w]>0))
089                                {
090                                        for(int w2=0;w2<codes.total();w2++)
091                                        {
092                                                if(!amWearingAt(codes.get(w2))
093                                                &&(codes.get(w2)!=Wearable.IN_INVENTORY)
094                                                &&(CMath.bset(codes.dependency_masks()[w], codes.get(w2))))
095                                                {
096                                                        final Item I=mob.fetchFirstWornItem(codes.get(w2));
097                                                        if((I!=null)&&(I!=this))
098                                                        {
099                                                                msg.source().tell(mob,I,null,L("You'll need to remove <T-NAMESELF> first."));
100                                                                return false;
101                                                        }
102                                                }
103                                        }
104                                }
105                        }
106                }
107                return true;
108        }
109
110        @Override
111        public void setSecretIdentity(String id)
112        {
113                super.setSecretIdentity(id);
114                wearplace=-1;
115                partnum=-1;
116        }
117
118        @Override
119        public int getBodyPartCode()
120        {
121                if(partnum>=0)
122                        return partnum;
123                for(int i=0;i<Race.BODYPARTSTR.length;i++)
124                {
125                        if((name().toUpperCase().endsWith(Race.BODYPARTSTR[i]))
126                        ||(rawSecretIdentity().toUpperCase().endsWith(Race.BODYPARTSTR[i])))
127                        {
128                                partnum=i;
129                                break;
130                        }
131                }
132                return partnum;
133        }
134
135        @Override
136        public void setBodyPartCode(int partNumber)
137        {
138                partnum=partNumber;
139        }
140
141        @Override
142        public long getWearLocations()
143        {
144                if(wearplace>=0)
145                        return wearplace;
146                final int num=getBodyPartCode();
147                if(num<0)
148                        wearplace=0;
149                else
150                        wearplace=Race.BODY_WEARVECTOR[num];
151                return wearplace;
152        }
153
154        @Override
155        public void setWearLocations(long wearPlace)
156        {
157                wearplace=wearPlace;
158        }
159
160        @Override
161        public SizeDeviation getSizingDeviation(MOB mob)
162        {
163                return SizeDeviation.FITS;
164        }
165        
166        @Override
167        public void affectCharStats(final MOB affected, final CharStats affectableStats)
168        {
169                super.affectCharStats(affected,affectableStats);
170
171                if((!amWearingAt(Wearable.IN_INVENTORY))
172                &&(!amWearingAt(Wearable.WORN_HELD))
173                &&(!amWearingAt(Wearable.WORN_FLOATING_NEARBY))
174                &&(!amWearingAt(Wearable.WORN_WIELD)))
175                {
176                        if(affected.charStats().getBodyPart(getBodyPartCode())<affected.charStats().getMyRace().bodyMask()[getBodyPartCode()])
177                                affectableStats.alterBodypart(getBodyPartCode(),phyStats().ability());
178                        else
179                                setRawWornCode(0);
180                }
181        }
182
183        @Override
184        public boolean canWear(MOB mob, long where)
185        {
186                if(where==Wearable.WORN_HELD)
187                        return super.canWear(mob,where);
188                if(where==Wearable.WORN_FLOATING_NEARBY)
189                        return false;
190                if(getBodyPartCode()<0)
191                        return false;
192                if((where!=0)&&(where!=getWearLocations()))
193                        return false;
194                final int numRacialTotal=mob.charStats().getMyRace().bodyMask()[getBodyPartCode()];
195                int numWorkingParts=mob.charStats().getBodyPart(getBodyPartCode());
196                // now add in other NON-FUNCTIONAL limb things worn
197                // FUNCTIONAL limbs are already included in numWorkingParts
198                for(int i=0;i<mob.numItems();i++)
199                {
200                        final Item I=mob.getItem(i);
201                        if((I instanceof StdLimb)
202                        &&(((StdLimb)I).getBodyPartCode()==getBodyPartCode())
203                        &&(I.amWearingAt(getWearLocations()))
204                        &&(I.container()==null))
205                                numWorkingParts++;
206                }
207                if(numWorkingParts>=numRacialTotal)
208                        return false;
209                return true;
210        }
211
212        @Override
213        public boolean fitsOn(long wornCode)
214        {
215                if(wornCode==Wearable.WORN_HELD)
216                        return super.fitsOn(wornCode);
217                if(wornCode==Wearable.WORN_FLOATING_NEARBY)
218                        return false;
219                if(wornCode<=0)
220                        return true;
221                return getWearLocations()==wornCode;
222        }
223
224        protected boolean canWearComplete(MOB mob)
225        {
226                if(getBodyPartCode()<0)
227                {
228                        mob.tell(L("This limb looks malformed."));
229                        return false;
230                }
231                if(!canWear(mob,0))
232                {
233                        mob.tell(L("You don't have any empty sockets to wear @x1 on.",name()));
234                        return false;
235                }
236                return true;
237        }

267        }

275                        this.raceID=raceID;
