034public class StdSmokable extends StdContainer implements Light
035{
036        @Override
037        public String ID()
038        {
039                return "StdSmokable";
040        }
041
042        protected boolean lit=false;
043        protected long puffTicks=30000/CMProps.getTickMillis();
044        protected int baseDuration=200;
045        protected int durationTicks=200;
046        protected boolean destroyedWhenBurnedOut=true;
047        protected boolean goesOutInTheRain=true;
048
049        public StdSmokable()
050        {
051                super();
052                setName("a cigar");
053                setDisplayText("a cigar has been left here.");
054                setDescription("Woven of fine leaf, it looks like a fine smoke!");
055
056                capacity=0;
057                containType=Container.CONTAIN_SMOKEABLES;
058                properWornBitmap=Wearable.WORN_MOUTH;
059                setMaterial(RawMaterial.RESOURCE_PIPEWEED);
060                wornLogicalAnd=false;
061                baseGoldValue=5;
062                recoverPhyStats();
063        }

107        @Override
108        public boolean okMessage(final Environmental myHost, final CMMsg msg)
109        {
110                final MOB mob=msg.source();
111
112                if(!msg.amITarget(this))
113                        return super.okMessage(myHost,msg);
114                switch(msg.targetMinor())
115                {
116                case CMMsg.TYP_WEAR:
117                        if(capacity>0)
118                        {
119                                if(hasContent())
120                                        durationTicks=baseDuration;
121                                else
122                                        durationTicks=0;
123                        }
124                        if(durationTicks==0)
125                        {
126                                mob.tell(L("@x1 looks empty.",name()));
127                                return false;
128                        }
129                        final Room room=mob.location();
130                        if(room!=null)
131                        {
132                                if(((LightSource.inTheRain(room)&&(goesOutInTheRain()))
133                                        ||(LightSource.inTheWater(msg.source(),room)))
134                                &&(durationTicks>0)
135                                &&(mob.isMine(this)))
136                                {
137                                        mob.tell(L("It's too wet to light @x1 here.",name()));
138                                        return false;
139                                }
140                        }
141                        msg.modify(msg.source(),msg.target(),msg.tool(),
142                                                msg.sourceCode(),L("<S-NAME> light(s) up <T-NAME>."),
143                                                msg.targetCode(),L("<S-NAME> light(s) up <T-NAME>."),
144                                                msg.othersCode(),L("<S-NAME> light(s) up <T-NAME>."));
145                        return super.okMessage(myHost,msg);
146                case CMMsg.TYP_EXTINGUISH:
147                        if((durationTicks==0)||(!isLit()))
148                        {
149                                mob.tell(L("@x1 is not lit!",name()));
150                                return false;
151                        }
152                        return true;
153                }
154                return super.okMessage(myHost,msg);
155        }
156
157        @Override
158        public boolean tick(final Tickable ticking, final int tickID)
159        {
160                if((tickID==Tickable.TICKID_LIGHT_FLICKERS)
161                &&(isLit())
162                &&(tickStatus==Tickable.STATUS_NOT)
163                &&(owner()!=null))
164                {
165                        tickStatus=Tickable.STATUS_ALIVE;
166                        if(((--durationTicks)>0)&&(!destroyed))
167                        {
168                                if(((durationTicks%puffTicks)==0)
169                                &&(owner() instanceof MOB)
170                                && amBeingWornProperly())
171                                {
172                                        final MOB mob=(MOB)owner();
173                                        if((mob.location()!=null)
174                                        &&(CMLib.flags().isAliveAwakeMobile(mob,true)))
175                                        {
176                                                tickStatus=Tickable.STATUS_WEATHER;
177                                                mob.location().show(mob,this,this,CMMsg.MSG_HANDS,L("<S-NAME> puff(s) on <T-NAME>."));
178                                                if((CMLib.dice().roll(1,1000,0)==1)&&(!CMSecurity.isDisabled(CMSecurity.DisFlag.AUTODISEASE)))
179                                                {
180                                                        final Ability A=CMClass.getAbility("Disease_Cancer");
181                                                        if((A!=null)&&(!CMSecurity.isAbilityDisabled(A.ID())))
182                                                                A.invoke(mob,mob,true,0);
183                                                }
184                                        }
185                                }
186                                tickStatus=Tickable.STATUS_NOT;
187                                return true;
188                        }
189                        if(owner() instanceof Room)
190                        {
191                                tickStatus=Tickable.STATUS_CLASS;
192                                final Room R=(Room)owner;
193                                if(R.numInhabitants()>0)
194                                        R.showHappens(CMMsg.MSG_OK_VISUAL,L("@x1 burns out.",name()));
195                                if(destroyedWhenBurnedOut())
196                                        destroy();
197                                R.recoverRoomStats();
198                        }
199                        else
200                        if(owner() instanceof MOB)
201                        {
202                                tickStatus=Tickable.STATUS_DEAD;
203                                final MOB M=(MOB)owner();
204                                M.tell(M,null,this,L("<O-NAME> burns out."));
205                                durationTicks=0;
206                                if(destroyedWhenBurnedOut())
207                                        destroy();
208                                M.recoverPhyStats();
209                                M.recoverCharStats();
210                                M.recoverMaxState();
211                                M.recoverPhyStats();
212                                if(M.location()!=null)
213                                        M.location().recoverRoomStats();
214                        }
215                        light(false);
216                        durationTicks=0;
217                }
218                tickStatus=Tickable.STATUS_NOT;
219                return false;
220        }
221
222        public void getAddictedTo(MOB mob, Item item)
223        {
224                Ability A=mob.fetchEffect("Addictions");
225                if(A==null)
226                {
227                        A=CMClass.getAbility("Addictions");
228                        if(A!=null)
229                                A.invoke(mob,item,true,0);
230                }
231        }

233        @Override
234        public void executeMsg(final Environmental myHost, final CMMsg msg)
235        {
236                final MOB mob=msg.source();
237                if(mob==null)
238                        return;
239                final Room room=mob.location();
240                if(room==null)
241                        return;
242                if(((LightSource.inTheRain(room)&&goesOutInTheRain())
243                   ||(LightSource.inTheWater(msg.source(),room)))
244                &&(isLit())
245                &&(durationTicks>0)
246                &&(mob.isMine(this))
247                &&((!CMLib.flags().isInFlight(mob))
248                   ||(LightSource.inTheRain(room))
249                   ||((room.domainType()!=Room.DOMAIN_OUTDOORS_WATERSURFACE)&&(room.domainType()!=Room.DOMAIN_INDOORS_WATERSURFACE))))
250                {
251                        if(LightSource.inTheWater(msg.source(),room))
252                                mob.tell(L("The water makes @x1 go out.",name()));
253                        else
254                                mob.tell(L("The rain makes @x1 go out.",name()));
255                        durationTicks=1;
256                        tick(this,Tickable.TICKID_LIGHT_FLICKERS);
257                }
258
259                if(msg.amITarget(this))
260                        switch(msg.targetMinor())
261                        {
262                        case CMMsg.TYP_EXTINGUISH:
263                                if(isLit())
264                                {
265                                        light(false);
266                                        CMLib.threads().deleteTick(this,Tickable.TICKID_LIGHT_FLICKERS);
267                                        recoverPhyStats();
268                                        room.recoverRoomStats();
269                                }
270                                break;
271                        case CMMsg.TYP_WEAR:
272                                if(durationTicks>0)
273                                {
274                                        if(capacity>0)
275                                        {
276                                                final List<Item> V=getContents();
277                                                Item I=null;
278                                                for(int v=0;v<V.size();v++)
279                                                {
280                                                        I=V.get(v);
281                                                        if(CMLib.dice().roll(1,100,0)==1)
282                                                                getAddictedTo(msg.source(),I);
283                                                        I.destroy();
284                                                }
285                                        }
286                                        else
287                                        if(CMLib.dice().roll(1,100,0)==1)
288                                                getAddictedTo(msg.source(),this);
289
290                                        light(true);
291                                        CMLib.threads().startTickDown(this,Tickable.TICKID_LIGHT_FLICKERS,1);
292                                        recoverPhyStats();
293                                        room.recoverRoomStats();
294                                }
295                                break;
296                        }
297                super.executeMsg(myHost,msg);
298                if((msg.tool()==this)
299                &&(msg.sourceMinor()==CMMsg.TYP_THROW))
300                {
301                        msg.source().recoverPhyStats();
302                        if(!CMath.bset(msg.sourceMajor(),CMMsg.MASK_OPTIMIZE))
303                        {
304                                if(msg.source().location()!=null)
305                                        msg.source().location().recoverRoomStats();
306                                final Room R=CMLib.map().roomLocation(msg.target());
307                                if((R!=null)&&(R!=msg.source().location()))
308                                        R.recoverRoomStats();
309                        }
310                }
311                else
312                if(msg.amITarget(this))
313                {
314                        switch(msg.targetMinor())
315                        {
316                        case CMMsg.TYP_DROP:
317                        case CMMsg.TYP_GET:
318                        case CMMsg.TYP_PULL:
319                        case CMMsg.TYP_PUSH:
320                        case CMMsg.TYP_REMOVE:
321                                if(!CMath.bset(msg.targetMajor(),CMMsg.MASK_OPTIMIZE))
322                                {
323                                        msg.source().recoverPhyStats();
324                                        if(msg.source().location()!=null)
325                                                msg.source().location().recoverRoomStats();
326                                        final Room R=CMLib.map().roomLocation(msg.tool());
327                                        if((R!=null)&&(R!=msg.source().location()))
328                                                R.recoverRoomStats();
329                                }
330                                break;
331                        }
332                }
333        }
