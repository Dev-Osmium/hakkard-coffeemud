075        {

481        @Override
482        public void recoverPhyStats()
483        {
484                super.recoverPhyStats();
485                if(rideBasis==Rideable.RIDEABLE_AIR)
486                {
487                        if((!subjectToWearAndTear())||(usesRemaining()>0))
488                                phyStats().setDisposition(phyStats().disposition()|PhyStats.IS_FLYING);
489                        else
490                                phyStats().setDisposition(phyStats().disposition()|PhyStats.IS_FALLING);
491                }
492                else
493                if(rideBasis==Rideable.RIDEABLE_WATER)
494                {
495                        if((!subjectToWearAndTear())||(usesRemaining()>0))
496                                phyStats().setDisposition(phyStats().disposition()|PhyStats.IS_SWIMMING);
497                        else
498                                phyStats().setDisposition(phyStats().disposition()|PhyStats.IS_FALLING);
499                }
500        }

502        @Override
503        public void affectPhyStats(final Physical affected, final PhyStats affectableStats)
504        {
505                super.affectPhyStats(affected,affectableStats);
506                if(affected instanceof MOB)
507                {
508                        final MOB mob=(MOB)affected;
509                        if(!CMLib.flags().isWithSeenContents(this))
510                                affectableStats.setDisposition(affectableStats.disposition()|PhyStats.IS_NOT_SEEN);
511                        if((mob.isInCombat())&&(mob.rangeToTarget()==0)&&(amRiding(mob)))
512                        {
513                                affectableStats.setAttackAdjustment(affectableStats.attackAdjustment()-mob.basePhyStats().attackAdjustment());
514                                affectableStats.setDamage(affectableStats.damage()-mob.basePhyStats().damage());
515                        }
516                        if((rideBasis()==Rideable.RIDEABLE_LADDER)
517                        &&(amRiding(mob)))
518                        {
519                                affectableStats.setDisposition(affectableStats.disposition()|PhyStats.IS_CLIMBING);
520                                affectableStats.setSpeed(affectableStats.speed()/2);
521                        }
522                }
523        }

525        @Override
526        public String displayText(final MOB mob)
527        {
528                if((numRiders()>0)
529                &&(stateStringSubject(this).length()>0)
530                &&(displayText!=null)
531                &&(displayText.length()>0)
532                &&CMLib.flags().isWithSeenContents(this))
533                {
534                        final StringBuffer sendBack=new StringBuffer(name(mob));
535                        sendBack.append(" "+stateStringSubject(this)+" ");
536                        for(int r=0;r<numRiders();r++)
537                        {
538                                final Rider rider=fetchRider(r);
539                                if(rider!=null)
540                                {
541                                        if(r>0)
542                                        {
543                                                sendBack.append(", ");
544                                                if(r==numRiders()-1)
545                                                        sendBack.append("and ");
546                                        }
547                                        sendBack.append(rider.name(mob));
548                                }
549
550                        }
551                        return sendBack.toString();
552                }
553                return super.displayText(mob);
554        }

562        @Override
563        public boolean okMessage(final Environmental myHost, final CMMsg msg)
564        {
565                if(!super.okMessage(myHost,msg))
566                        return false;
567                switch(msg.targetMinor())
568                {
569                case CMMsg.TYP_ADVANCE:
570                        if((rideBasis()==Rideable.RIDEABLE_LADDER)
571                        &&(amRiding(msg.source())))
572                        {
573                                msg.source().tell(L("You cannot advance while @x1 @x2!",stateString(msg.source()),name(msg.source())));
574                                return false;
575                        }
576                        break;
577                case CMMsg.TYP_RETREAT:
578                        if((rideBasis()==Rideable.RIDEABLE_LADDER)
579                        &&(amRiding(msg.source())))
580                        {
581                                msg.source().tell(L("You cannot retreat while @x1 @x2!",stateString(msg.source()),name(msg.source())));
582                                return false;
583                        }
584                        break;
585                case CMMsg.TYP_DISMOUNT:
586                        if(msg.amITarget(this))
587                        {
588                                if(msg.tool() instanceof Rider)
589                                {
590                                        if(!amRiding((Rider)msg.tool()))
591                                        {
592                                                msg.source().tell(L("@x1 is not @x2 @x3!",msg.tool().name(),stateString((Rider)msg.tool()),name(msg.source())));
593                                                if(((Rider)msg.tool()).riding()==this)
594                                                        ((Rider)msg.tool()).setRiding(null);
595                                                return false;
596                                        }
597                                }
598                                else
599                                if(!amRiding(msg.source()))
600                                {
601                                        msg.source().tell(L("You are not @x1 @x2!",stateString(msg.source()),name(msg.source())));
602                                        if(msg.source().riding()==this)
603                                                msg.source().setRiding(null);
604                                        return false;
605                                }
606                                // protects from standard item rejection
607                                return true;
608                        }
609                        break;
610                case CMMsg.TYP_SIT:
611                        if(amRiding(msg.source()))
612                        {
613                                if((rideBasis()==Rideable.RIDEABLE_SLEEP) && (CMLib.flags().isSleeping(msg.source())))
614                                        return true;
615                                msg.source().tell(L("You are @x1 @x2!",stateString(msg.source()),name(msg.source())));
616                                msg.source().setRiding(this);
617                                return false;
618                        }
619                        else
620                        if((riding()!=msg.source())
621                        &&((rideBasis()==Rideable.RIDEABLE_SIT)
622                        ||(rideBasis()==Rideable.RIDEABLE_ENTERIN)
623                        ||(rideBasis()==Rideable.RIDEABLE_TABLE)
624                        ||(rideBasis()==Rideable.RIDEABLE_SLEEP)))
625                        {
626                                if(msg.amITarget(this)
627                                &&(numRiders()>=riderCapacity())
628                                &&(!amRiding(msg.source())))
629                                {
630                                        // for items
631                                        msg.source().tell(L("@x1 is full.",name(msg.source())));
632                                        // for mobs
633                                        // msg.source().tell(L("No more can fit on @x1.",name(msg.source())));
634                                        return false;
635                                }
636                                return true;
637                        }
638                        else
639                        if(msg.amITarget(this))
640                        {
641                                msg.source().tell(L("You cannot sit on @x1.",name(msg.source())));
642                                return false;
643                        }
644                        break;
645                case CMMsg.TYP_SLEEP:
646                        if((amRiding(msg.source()))
647                        &&(((!msg.amITarget(this))&&(msg.target()!=null))
648                           ||((rideBasis()!=Rideable.RIDEABLE_SLEEP)&&(rideBasis()!=Rideable.RIDEABLE_ENTERIN))))
649                        {
650                                msg.source().tell(L("You are @x1 @x2!",stateString(msg.source()),name(msg.source())));
651                                msg.source().setRiding(this);
652                                return false;
653                        }
654                        else
655                        if((riding()!=msg.source())
656                        &&((rideBasis()==Rideable.RIDEABLE_SLEEP)
657                        ||(rideBasis()==Rideable.RIDEABLE_ENTERIN)))
658                        {
659                                if(msg.amITarget(this)
660                                &&(numRiders()>=riderCapacity())
661                                &&(!amRiding(msg.source())))
662                                {
663                                        // for items
664                                        msg.source().tell(L("@x1 is full.",name(msg.source())));
665                                        // for mobs
666                                        // msg.source().tell(L("No more can fit on @x1.",name(msg.source())));
667                                        return false;
668                                }
669                                return true;
670                        }
671                        else
672                        if(msg.amITarget(this))
673                        {
674                                msg.source().tell(L("You cannot lie down on @x1.",name(msg.source())));
675                                return false;
676                        }
677                        break;
678                case CMMsg.TYP_MOUNT:
679                {
680                        if(amRiding(msg.source()))
681                        {
682                                msg.source().tell(null,msg.source(),null,L("<T-NAME> <T-IS-ARE> @x1 @x2!",stateString(msg.source()),name(msg.source())));
683                                msg.source().setRiding(this);
684                                return false;
685                        }
686                        if((riding()==msg.target())&&(msg.tool() instanceof Item))
687                        {
688                                msg.source().tell(null,msg.source(),null,L("<T-NAME> <T-IS-ARE> already @x1 @x2!",stateString(msg.source()),name(msg.source())));
689                                return false;
690                        }
691                        if(msg.amITarget(this))
692                        {
693                                final Rider whoWantsToRide=(msg.tool() instanceof Rider)?(Rider)msg.tool():msg.source();
694                                if(amRiding(whoWantsToRide))
695                                {
696                                        msg.source().tell(null,whoWantsToRide,null,L("<T-NAME> <T-IS-ARE> @x1 @x2!",stateString(msg.source()),name(msg.source())));
697                                        whoWantsToRide.setRiding(this);
698                                        return false;
699                                }
700                                if((msg.tool() instanceof MOB)
701                                &&(!CMLib.flags().isBoundOrHeld((MOB)msg.tool())))
702                                {
703                                        msg.source().tell(L("@x1 won't let you do that.",((MOB)msg.tool()).name(msg.source())));
704                                        return false;
705                                }
706                                else
707                                if(riding()==whoWantsToRide)
708                                {
709                                        if(msg.tool() instanceof Physical)
710                                                msg.source().tell(L("@x1 can not be mounted to @x2!",((Physical)msg.tool()).name(msg.source()),name(msg.source())));
711                                        else
712                                                msg.source().tell(L("@x1 can not be mounted to @x2!",msg.tool().name(),name(msg.source())));
713                                        return false;
714                                }
715                                else
716                                if(msg.tool() instanceof Rideable)
717                                {
718                                        msg.source().tell(L("@x1 is not allowed on @x2.",((Rideable)msg.tool()).name(msg.source()),name(msg.source())));
719                                        return false;
720                                }
721                                if(msg.tool()==null)
722                                        switch(rideBasis())
723                                        {
724                                        case Rideable.RIDEABLE_ENTERIN:
725                                        case Rideable.RIDEABLE_SIT:
726                                        case Rideable.RIDEABLE_SLEEP:
727                                                msg.source().tell(L("@x1 can not be mounted in this way.",name(msg.source())));
728                                                return false;
729                                        default:
730                                                break;
731                                        }
732                                if((numRiders()>=riderCapacity())
733                                &&(!amRiding(whoWantsToRide)))
734                                {
735                                        // for items
736                                        msg.source().tell(L("@x1 is full.",name(msg.source())));
737                                        // for mobs
738                                        // msg.source().tell(L("No more can fit on @x1.",name(msg.source())));
739                                        return false;
740                                }
741                                // protects from standard item rejection
742                                return true;
743                        }
744                        break;
745                }
746                case CMMsg.TYP_ENTER:
747                        if(amRiding(msg.source())
748                        &&(msg.target() instanceof Room))
749                        {
750                                final Room sourceRoom=msg.source().location();
751                                final Room targetRoom=(Room)msg.target();
752                                if((sourceRoom!=null)&&(!msg.amITarget(sourceRoom)))
753                                {
754                                        boolean ok=((targetRoom.domainType()&Room.INDOORS)==0)
755                                                                ||(targetRoom.maxRange()>4);
756                                        switch(rideBasis)
757                                        {
758                                        case Rideable.RIDEABLE_LAND:
759                                        case Rideable.RIDEABLE_WAGON:
760                                                if((targetRoom.domainType()==Room.DOMAIN_OUTDOORS_AIR)
761                                                ||(targetRoom.domainType()==Room.DOMAIN_INDOORS_AIR)
762                                                ||CMLib.flags().isWateryRoom(targetRoom))
763                                                        ok=false;
764                                                if((rideBasis==Rideable.RIDEABLE_WAGON)
765                                                &&((riding()==null)
766                                                   ||(!(riding() instanceof MOB))
767                                                   ||(((MOB)riding()).basePhyStats().weight()<(basePhyStats().weight()/5))))
768                                                {
769                                                        msg.source().tell(L("@x1 doesn't seem to be moving.",name(msg.source())));
770                                                        return false;
771                                                }
772                                                break;
773                                        case Rideable.RIDEABLE_AIR:
774                                                break;
775                                        case Rideable.RIDEABLE_LADDER:
776                                                ok=true;
777                                                break;
778                                        case Rideable.RIDEABLE_WATER:
779                                                if((!CMLib.flags().isWaterySurfaceRoom(sourceRoom))
780                                                &&(!CMLib.flags().isWaterySurfaceRoom(targetRoom)))
781                                                        ok=false;
782                                                else
783                                                        ok=true;
784                                                if((targetRoom.domainType()==Room.DOMAIN_INDOORS_AIR)
785                                                ||(targetRoom.domainType()==Room.DOMAIN_OUTDOORS_AIR)
786                                                ||(CMLib.flags().isUnderWateryRoom(targetRoom)))
787                                                        ok=false;
788                                                break;
789                                        }
790                                        if(!ok)
791                                        {
792                                                msg.source().tell(L("You cannot ride @x1 that way.",name(msg.source())));
793                                                return false;
794                                        }
795                                        if(CMLib.flags().isSitting(msg.source()))
796                                        {
797                                                msg.source().tell(L("You cannot crawl while @x1 @x2.",stateString(msg.source()),name(msg.source())));
798                                                return false;
799                                        }
800                                }
801                        }
802                        break;
803                case CMMsg.TYP_GIVE:
804                        if(msg.target() instanceof MOB)
805                        {
806                                final MOB tmob=(MOB)msg.target();
807                                if((amRiding(tmob))&&(!amRiding(msg.source())))
808                                {
809                                        if(rideBasis()==Rideable.RIDEABLE_ENTERIN)
810                                                msg.source().tell(msg.source(),tmob,null,L("<T-NAME> must exit first."));
811                                        else
812                                                msg.source().tell(msg.source(),tmob,null,L("<T-NAME> must disembark first."));
813                                        return false;
814                                }
815                        }
816                        break;
817                case CMMsg.TYP_BUY:
818                case CMMsg.TYP_BID:
819                case CMMsg.TYP_SELL:
820                        if((amRiding(msg.source()))
821                        &&(rideBasis()!=Rideable.RIDEABLE_TABLE)
822                        &&(rideBasis()!=Rideable.RIDEABLE_SIT))
823                        {
824                                msg.source().tell(L("You can not do that while @x1 @x2.",stateString(msg.source()),name(msg.source())));
825                                return false;
826                        }
827                        return super.okMessage(myHost,msg);
828                }
829                if((msg.sourceMajor(CMMsg.MASK_HANDS))
830                &&(amRiding(msg.source()))
831                &&((msg.sourceMessage()!=null)||(msg.othersMessage()!=null))
832                &&(msg.target()!=this)
833                &&(msg.tool()!=this)
834                &&((!CMLib.utensils().reachableItem(msg.source(),msg.target()))
835                        || (!CMLib.utensils().reachableItem(msg.source(),msg.tool()))
836                        || ((msg.sourceMinor()==CMMsg.TYP_GIVE)&&(msg.target() instanceof MOB)&&(msg.target()!=this)&&(!amRiding((MOB)msg.target()))))
837                &&(!((msg.sourceMinor()==CMMsg.TYP_GIVE)&&(msg.target() instanceof MOB)&&(amRiding((MOB)msg.target()))&&(CMLib.flags().isStanding(msg.source())))))
838                {
839                        // some of the above applies to genrideable items only
840                        msg.source().tell(L("You can not do that while @x1 @x2.",stateString(msg.source()),name(msg.source())));
841                        return false;
842                }
843                return super.okMessage(myHost,msg);
844        }

846        @Override
847        public void executeMsg(final Environmental myHost, final CMMsg msg)
848        {
849                super.executeMsg(myHost,msg);
850                switch(msg.targetMinor())
851                {
852                case CMMsg.TYP_LOOK:
853                case CMMsg.TYP_EXAMINE:
854                        if(msg.target()==this)
855                        {
856                                if((numRiders()>0)
857                                &&(CMLib.flags().canBeSeenBy(this,msg.source())))
858                                        msg.addTrailerMsg(CMClass.getMsg(msg.source(),null,null,CMMsg.MSG_OK_VISUAL,displayText(),CMMsg.NO_EFFECT,null,CMMsg.NO_EFFECT,null));
859                                if((this.subjectToWearAndTear())
860                                && (this.rideBasis() == Rideable.RIDEABLE_WATER)
861                                && (CMath.bset(material(), RawMaterial.MATERIAL_WOODEN)))
862                                {
863                                        // this is for the small rideable boats
864                                        final StringBuilder visualCondition = new StringBuilder("");
865                                        if(this.subjectToWearAndTear() && (usesRemaining() <= 100))
866                                        {
867                                                final double pct=(CMath.div(usesRemaining(),100.0));
868                                                GenSailingShip.appendCondition(visualCondition,pct,CMStrings.capitalizeFirstLetter(name(msg.source())));
869                                        }
870                                        msg.addTrailerRunnable(new Runnable()
871                                        {
872                                                @Override
873                                                public void run()
874                                                {
875                                                        msg.source().tell(visualCondition.toString());
876                                                }
877                                        });
878                                }
879                        }
880                        break;
881                case CMMsg.TYP_DISMOUNT:
882                        if(msg.tool() instanceof Rider)
883                        {
884                                ((Rider)msg.tool()).setRiding(null);
885                                if(msg.source().location()!=null)
886                                        msg.source().location().recoverRoomStats();
887                        }
888                        else
889                        if(amRiding(msg.source()))
890                        {
891                                msg.source().setRiding(null);
892                                if(msg.source().location()!=null)
893                                        msg.source().location().recoverRoomStats();
894                        }
895                        break;
896                case CMMsg.TYP_ENTER:
897                case CMMsg.TYP_LEAVE:
898                case CMMsg.TYP_FLEE:
899                        if((rideBasis()==Rideable.RIDEABLE_LADDER)
900                        &&(amRiding(msg.source())))
901                        {
902                                msg.source().setRiding(null);
903                                if(msg.source().location()!=null)
904                                        msg.source().location().recoverRoomStats();
905                        }
906                        break;
907                case CMMsg.TYP_MOUNT:
908                case CMMsg.TYP_SIT:
909                case CMMsg.TYP_SLEEP:
910                        if(msg.amITarget(this))
911                        {
912                                if(msg.tool() instanceof Rider)
913                                {
914                                        final Rider R = (Rider)msg.tool();
915                                        R.setRiding(this);
916                                        if(msg.tool() instanceof MOB)
917                                        switch(rideBasis())
918                                        {
919                                        case Rideable.RIDEABLE_SIT:
920                                        case Rideable.RIDEABLE_ENTERIN:
921                                                R.basePhyStats().setDisposition(R.basePhyStats().disposition()|PhyStats.IS_SITTING);
922                                                break;
923                                        case Rideable.RIDEABLE_SLEEP:
924                                                R.basePhyStats().setDisposition(R.basePhyStats().disposition()|PhyStats.IS_SLEEPING);
925                                                break;
926                                        default:
927                                                break;
928                                        }
929                                        if(msg.source().location()!=null)
930                                                msg.source().location().recoverRoomStats();
931                                }
932                                else
933                                if(!amRiding(msg.source()))
934                                {
935                                        msg.source().setRiding(this);
936                                        if(msg.source().location()!=null)
937                                                msg.source().location().recoverRoomStats();
938                                }
939                        }
940                        break;
941                case CMMsg.TYP_WEAPONATTACK:
942                        if(msg.target()==this)
943                        {
944                                Weapon weapon=null;
945                                if((msg.tool() instanceof Weapon))
946                                        weapon=(Weapon)msg.tool();
947                                if((weapon!=null)&&(msg.source().riding()!=null))
948                                {
949                                        final boolean isHit=msg.value()>0;
950                                        if(isHit && CMLib.combat().isAShipSiegeWeapon(weapon)
951                                        && (((AmmunitionWeapon)weapon).ammunitionCapacity() > 1))
952                                        {
953                                                int shotsRemaining = ((AmmunitionWeapon)weapon).ammunitionRemaining() + 1;
954                                                ((AmmunitionWeapon)weapon).setAmmoRemaining(0);
955                                                final ArrayList<Pair<MOB,Room>> targets = new ArrayList<Pair<MOB,Room>>(5);
956                                                final Room R=CMLib.map().roomLocation(this);
957                                                if((R!=null)&&((R.domainType()&Room.INDOORS)==0))
958                                                {
959                                                        for(final Enumeration<MOB> m=R.inhabitants();m.hasMoreElements();)
960                                                        {
961                                                                final MOB M=m.nextElement();
962                                                                if((M!=null)&&(M.riding()==this))
963                                                                        targets.add(new Pair<MOB,Room>(M,R));
964                                                        }
965                                                }
966                                                final int chanceToHit = targets.size() * 20;
967                                                final Room oldRoom=msg.source().location();
968                                                try
969                                                {
970                                                        while(shotsRemaining-- > 0)
971                                                        {
972                                                                final Pair<MOB,Room> randomPair = (targets.size()>0)? targets.get(CMLib.dice().roll(1,targets.size(),-1)) : null;
973                                                                if((CMLib.dice().rollPercentage() < chanceToHit)&&(randomPair != null))
974                                                                {
975                                                                        msg.source().setLocation(randomPair.second);
976                                                                        final double pctLoss = CMath.div(msg.value(), phyStats().level());
977                                                                        final int pointsLost = (int)Math.round(pctLoss * msg.source().maxState().getHitPoints());
978                                                                        CMLib.combat().postWeaponDamage(msg.source(), randomPair.first, weapon, pointsLost);
979                                                                }
980                                                                else
981                                                                if(randomPair != null)
982                                                                {
983                                                                        msg.source().setLocation(randomPair.second);
984                                                                        CMLib.combat().postWeaponAttackResult(msg.source(), randomPair.first, weapon, false);
985                                                                }
986                                                                else
987                                                                if(R!=null)
988                                                                {
989                                                                        final CMMsg missMsg=CMClass.getMsg(msg.source(), msg.target(), weapon, CMMsg.MASK_ALWAYS|CMMsg.MSG_ATTACKMISS, weapon.missString());
990                                                                        if(R.okMessage(msg.source(), missMsg))
991                                                                                R.send(msg.source(), missMsg);
992                                                                }
993                                                        }
994                                                }
995                                                finally
996                                                {
997                                                        msg.source().setLocation(oldRoom);
998                                                }
999                                        }
1000                                        else
1001                                                CMLib.combat().postShipWeaponAttackResult(msg.source(), msg.source().riding(), this, weapon, isHit);
1002                                }
1003                        }
1004                        break;
1005                case CMMsg.TYP_DAMAGE:
1006                        if((msg.target()==this) && (msg.value() > 0))
1007                        {
1008                                int level = phyStats().level();
1009                                if(level < 10)
1010                                        level = 10;
1011                                final double pctLoss = CMath.div(msg.value(), level) * 10.0; // siege weapons against rideables is harsh
1012                                final int pointsLost = (int)Math.round(pctLoss * level);
1013                                if(pointsLost > 0)
1014                                {
1015                                        final int weaponType = (msg.tool() instanceof Weapon) ? ((Weapon)msg.tool()).weaponDamageType() : Weapon.TYPE_BASHING;
1016                                        final String hitWord = CMLib.combat().standardHitWord(weaponType, pctLoss);
1017                                        final String msgStr = (msg.targetMessage() == null) ? L("<O-NAME> fired from <S-NAME> hits and @x1 the ship.",hitWord) : msg.targetMessage();
1018                                        final CMMsg deckHitMsg=CMClass.getMsg(msg.source(), this, msg.tool(),CMMsg.MSG_OK_ACTION, msgStr);
1019                                        final Room targetRoom=CMLib.map().roomLocation(this);
1020                                        if(targetRoom.okMessage(msg.source(), deckHitMsg))
1021                                                targetRoom.send(msg.source(), deckHitMsg);
1022                                        final CMMsg underdeckHitMsg=CMClass.getMsg(msg.source(), this, msg.tool(),CMMsg.MSG_OK_ACTION, L("Something hits and @x1 the ship.",hitWord));
1023                                        if(targetRoom.okMessage(msg.source(), underdeckHitMsg))
1024                                                targetRoom.send(msg.source(), underdeckHitMsg);
1025                                        if(pointsLost >= this.usesRemaining())
1026                                        {
1027                                                this.setUsesRemaining(0);
1028                                                this.recoverPhyStats(); // takes away the swimmability!
1029                                                final Room shipR=CMLib.map().roomLocation(this);
1030                                                if(shipR!=null)
1031                                                {
1032                                                        CMLib.tracking().makeSink(this, shipR, false);
1033                                                        final String sinkString = L("<T-NAME> start(s) sinking!");
1034                                                        shipR.show(msg.source(), this, CMMsg.MSG_OK_ACTION, sinkString);
1035                                                }
1036                                                if(!CMLib.leveler().postExperienceToAllAboard(msg.source().riding(), 500))
1037                                                        CMLib.leveler().postExperience(msg.source(), null, null, 500, false);
1038                                        }
1039                                        else
1040                                        {
1041                                                this.setUsesRemaining(this.usesRemaining() - pointsLost);
1042                                        }
1043                                }
1044                        }
1045                        break;
1046                }
1047                switch(msg.sourceMinor())
1048                {
1049                case CMMsg.TYP_STAND:
1050                case CMMsg.TYP_QUIT:
1051                case CMMsg.TYP_PANIC:
1052                case CMMsg.TYP_DEATH:
1053                        if(amRiding(msg.source()))
1054                        {
1055                                msg.source().setRiding(null);
1056                                if(msg.source().location()!=null)
1057                                        msg.source().location().recoverRoomStats();
1058                        }
1059                        break;
1060                }
1061        }
