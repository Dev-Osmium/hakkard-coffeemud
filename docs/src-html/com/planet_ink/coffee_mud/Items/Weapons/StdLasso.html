035public class StdLasso extends StdWeapon
036{
037        @Override
038        public String ID()
039        {
040                return "StdLasso";
041        }
042
043        public StdLasso()
044        {
045                super();
046                setName("a lasso");
047                setDisplayText("a lasso has been left here.");
048                setDescription("Its a rope with a big stiff loop on one end!");
049                basePhyStats().setAbility(0);
050                basePhyStats().setLevel(0);
051                basePhyStats.setWeight(1);
052                basePhyStats().setAttackAdjustment(0);
053                basePhyStats().setDamage(0);
054                baseGoldValue=10;
055                recoverPhyStats();
056                minRange=1;
057                maxRange=1;
058                weaponDamageType=Weapon.TYPE_NATURAL;
059                material=RawMaterial.RESOURCE_HEMP;
060                weaponClassification=Weapon.CLASS_THROWN;
061                setRawLogicalAnd(true);
062        }
063
064        protected MOB lastBinder = null;
065
066        protected void untieIfPoss(final MOB binder, final MOB boundM)
067        {
068                final Room R=boundM.location();
069                if(R==null)
070                        return;
071                if((CMLib.flags().canBeSeenBy(boundM, binder))
072                &&(CMLib.flags().isAliveAwakeMobileUnbound(binder, true)))
073                {
074                        final Ability A=boundM.fetchEffect("Thief_Bind");
075                        if((A==null)||(A.invoker()!=binder))
076                        {
077                                binder.tell(L("You can't figure out the knots."));
078                        }
079                        else
080                        {
081                                final CMMsg msg2=CMClass.getMsg(binder,boundM,null,CMMsg.MSG_NOISYMOVEMENT,L("<S-NAME> untie(s) <T-NAME>."));
082                                if(R.okMessage(binder, msg2))
083                                {
084                                        if((CMLib.flags().isAnimalIntelligence(boundM))&&(boundM.getVictim()==binder))
085                                                boundM.makePeace(true);
086                                        A.unInvoke();
087                                        if((CMLib.flags().isAnimalIntelligence(boundM))&&(boundM.getVictim()==binder))
088                                                boundM.makePeace(true);
089                                        if(!CMLib.flags().isBound(boundM))
090                                        {
091                                                msg2.addTrailerMsg(CMClass.getMsg(boundM,this,CMMsg.MASK_ALWAYS|CMMsg.MSG_DROP,null));
092                                                msg2.addTrailerMsg(CMClass.getMsg(binder,this,CMMsg.MASK_ALWAYS|CMMsg.MSG_GET,null));
093                                        }
094                                        R.send(binder, msg2);
095                                }
096                        }
097                }
098                else
099                        binder.tell(L("You don't see that here."));
100        }
101        
102        @Override
103        public boolean okMessage(final Environmental myHost, final CMMsg msg)
104        {
105                if(!super.okMessage(myHost, msg))
106                        return false;
107                if((msg.tool()==this)
108                &&(msg.targetMinor()==CMMsg.TYP_DAMAGE)
109                &&(msg.value()>0)
110                &&(msg.target() !=null)
111                &&(msg.target() instanceof MOB)
112                &&(weaponClassification()==Weapon.CLASS_THROWN))
113                {
114                        msg.setValue(0);
115                        final MOB targetMOB=(MOB)msg.target();
116                        if((CMLib.flags().isAnimalIntelligence(targetMOB)))
117                        {
118                                final Set<MOB> allMyFam=msg.source().getGroupMembers(new HashSet<MOB>());
119                                boolean anyCombatants=false;
120                                for(final MOB M : allMyFam)
121                                {
122                                        if(M.isInCombat())
123                                                anyCombatants=true;
124                                }
125                                if(!anyCombatants)
126                                {
127                                        if(CMLib.law().doesHavePriviledgesHere(msg.source(), msg.source().location()))
128                                                targetMOB.makePeace(false);
129                                        msg.addTrailerRunnable(new Runnable()
130                                        {
131                                                @Override
132                                                public void run()
133                                                {
134                                                        for(final MOB M : allMyFam)
135                                                        {
136                                                                if(M.getVictim()==targetMOB)
137                                                                        M.makePeace(true);
138                                                        }
139                                                }
140                                        });
141                                }
142                        }
143                }
144                else
145                if(((msg.sourceMinor()==CMMsg.TYP_HUH)||(msg.sourceMinor()==CMMsg.TYP_COMMANDFAIL))
146                &&(msg.source()==lastBinder)
147                &&(owner() instanceof MOB)
148                &&(owner() != msg.source())
149                &&(msg.targetMessage()!=null)
150                &&(msg.targetMessage().length()>0)
151                &&(CMLib.flags().isBound(owner())))
152                {
153                        String tmsg=msg.targetMessage().toLowerCase();
154                        if(tmsg.startsWith("get ")||tmsg.startsWith("take "))
155                        {
156                                final MOB boundM=(MOB)owner();
157                                final Room R=boundM.location();
158                                if(R!=null)
159                                {
160                                        tmsg=tmsg.substring(4).trim();
161                                        List<String> rest=CMParms.parse(tmsg);
162                                        String itemName=tmsg;
163                                        int x=rest.indexOf("from");
164                                        if((x>0)&&(x<rest.size()-1))
165                                        {
166                                                itemName=CMParms.combine(rest, 0, x);
167                                                String mobName=CMParms.combine(rest,x+1);
168                                                final MOB M=R.fetchInhabitant(mobName);
169                                                if((M!=null)
170                                                &&(M==boundM)
171                                                &&(CMLib.english().containsString(name(), itemName)))
172                                                {
173                                                        untieIfPoss(msg.source(), boundM);
174                                                        return false;
175                                                }
176                                                return true;
177                                        }
178                                        if(CMLib.english().containsString(name(), itemName))
179                                        {
180                                                untieIfPoss(msg.source(), boundM);
181                                                return false;
182                                        }
183                                        return true;
184                                }
185                                else
186                                        lastBinder=null;
187                        }
188                        else
189                        if(tmsg.startsWith("untie "))
190                        {
191                                final MOB boundM=(MOB)owner();
192                                tmsg=tmsg.substring(6).trim();
193                                final Room R=boundM.location();
194                                if(R!=null)
195                                {
196                                        final MOB M=R.fetchInhabitant(tmsg);
197                                        if((M!=null)
198                                        &&(M==boundM))
199                                        {
200                                                untieIfPoss(msg.source(), boundM);
201                                                return false;
202                                        }
203                                }
204                                else
205                                        lastBinder=null;
206                        }
207                }
208                return true;
209        }

211        @Override
212        public void executeMsg(final Environmental myHost, final CMMsg msg)
213        {
214                if((msg.tool()==this)
215                &&(msg.targetMinor()==CMMsg.TYP_WEAPONATTACK)
216                &&(weaponClassification()==Weapon.CLASS_THROWN))
217                        return;
218                        //msg.addTrailerMsg(CMClass.getMsg(msg.source(),this,CMMsg.MSG_DROP,null));
219                else
220                if((msg.tool()==this)
221                &&(msg.targetMinor()==CMMsg.TYP_DAMAGE)
222                &&(msg.target() !=null)
223                &&(msg.target() instanceof MOB)
224                &&(weaponClassification()==Weapon.CLASS_THROWN))
225                {
226                        unWear();
227                        msg.addTrailerMsg(CMClass.getMsg(msg.source(),this,CMMsg.MASK_ALWAYS|CMMsg.MSG_DROP,null));
228                        msg.addTrailerMsg(CMClass.getMsg((MOB)msg.target(),this,CMMsg.MASK_ALWAYS|CMMsg.MSG_GET,null));
229                        msg.addTrailerMsg(CMClass.getMsg(msg.source(),msg.target(),this,CMMsg.MASK_ALWAYS|CMMsg.TYP_GENERAL,null));
230                }
231                else
232                if((msg.tool()==this)
233                &&(msg.target() instanceof MOB)
234                &&(msg.targetMinor()==CMMsg.TYP_GENERAL)
235                &&(((MOB)msg.target()).isMine(this))
236                &&(msg.sourceMessage()==null))
237                {
238                        final MOB tmob=(MOB)msg.target();
239                        Ability A=CMClass.getAbility("Thief_Bind");
240                        if(A!=null)
241                        {
242                                A.setAffectedOne(this);
243                                A.invoke(msg.source(),tmob,true,phyStats().level());
244                                A=tmob.fetchEffect("Thief_Bind");
245                                if(A!=null)
246                                {
247                                        lastBinder=A.invoker();
248                                        if(lastBinder==null)
249                                                lastBinder=msg.source();
250                                }
251                        }
252                }
253                else
254                        super.executeMsg(myHost,msg);
255        }
