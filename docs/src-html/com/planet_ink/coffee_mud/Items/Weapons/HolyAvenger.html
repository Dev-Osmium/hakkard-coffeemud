033public class HolyAvenger extends TwoHandedSword
034{
035        @Override
036        public String ID()
037        {
038                return "HolyAvenger";
039        }
040
041        public HolyAvenger()
042        {
043                super();
044
045                setName("the holy avenger");
046                setDisplayText("a beautiful two-handed sword has been left here");
047                setDescription("A two-handed sword crafted with a careful hand, and inscribed with several holy symbols.");
048                secretIdentity="The Holy Avenger!  A good-only Paladin sword that casts dispel evil on its victims";
049                basePhyStats().setAbility(0);
050                basePhyStats().setLevel(20);
051                basePhyStats().setWeight(25);
052                basePhyStats().setAttackAdjustment(25);
053                basePhyStats().setDamage(18);
054                basePhyStats().setDisposition(basePhyStats().disposition()|PhyStats.IS_GOOD|PhyStats.IS_BONUS);
055                baseGoldValue=15500;
056                recoverPhyStats();
057                material=RawMaterial.RESOURCE_MITHRIL;
058                weaponDamageType=TYPE_SLASHING;
059                setRawLogicalAnd(true);
060        }
061
062        @Override
063        public boolean okMessage(final Environmental myHost, final CMMsg msg)
064        {
065                if(!super.okMessage(myHost,msg))
066                        return false;
067
068                final MOB mob=msg.source();
069                if(mob.location()==null)
070                        return true;
071
072                if(msg.amITarget(this))
073                switch(msg.targetMinor())
074                {
075                case CMMsg.TYP_HOLD:
076                case CMMsg.TYP_WEAR:
077                case CMMsg.TYP_WIELD:
078                case CMMsg.TYP_GET:
079                        if((!msg.source().charStats().getCurrentClass().ID().equals("Paladin"))
080                        ||(!CMLib.flags().isGood(msg.source())))
081                        {
082                                unWear();
083                                mob.location().show(mob,null,CMMsg.MSG_OK_ACTION,L("@x1 flashes and flies out of <S-HIS-HER> hands!",name()));
084                                if(msg.source().isMine(this))
085                                        CMLib.commands().postDrop(msg.source(),this,true,false,false);
086                                return false;
087                        }
088                        break;
089                default:
090                        break;
091                }
092                return true;
093        }
094
095        @Override
096        public void executeMsg(final Environmental myHost, final CMMsg msg)
097        {
098                super.executeMsg(myHost,msg);
099                if((msg.source().location()!=null)
100                &&(msg.targetMinor()==CMMsg.TYP_DAMAGE)
101                &&((msg.value())>0)
102                &&(msg.tool()==this)
103                &&(msg.target() instanceof MOB)
104                &&(!((MOB)msg.target()).amDead())
105                &&(CMLib.flags().isEvil((MOB)msg.target())))
106                {
107                        final CMMsg msg2=CMClass.getMsg(msg.source(),msg.target(),new HolyAvenger(),CMMsg.MSG_OK_ACTION,CMMsg.MSK_MALICIOUS_MOVE|CMMsg.TYP_UNDEAD,CMMsg.MSG_NOISYMOVEMENT,null);
108                        if(msg.source().location().okMessage(msg.source(),msg2))
109                        {
110                                msg.source().location().send(msg.source(), msg2);
111                                int damage=CMLib.dice().roll(1,15,0);
112                                if(msg.value()>0)
113                                        damage=damage/2;
114                                msg.addTrailerMsg(CMClass.getMsg(msg.source(),msg.target(),CMMsg.MSG_OK_ACTION,L("@x1 dispels evil within <T-NAME> and @x2 <T-HIM-HER>>!",name(),CMLib.combat().standardHitWord(Weapon.TYPE_BURSTING,damage))));
115                                final CMMsg msg3=CMClass.getMsg(msg.source(),msg.target(),null,CMMsg.MSG_OK_VISUAL,CMMsg.MSG_DAMAGE,CMMsg.NO_EFFECT,null);
116                                msg3.setValue(damage);
117                                msg.addTrailerMsg(msg3);
118                        }
119                }
120        }
121
122}
