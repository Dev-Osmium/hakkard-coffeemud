033public class StdBoffWeapon extends StdWeapon
034{
035        @Override
036        public String ID()
037        {
038                return "StdBoffWeapon";
039        }
040
041        public StdBoffWeapon()
042        {
043                super();
044
045                setName("boff weapon");
046                setDisplayText(" sits here.");
047                setDescription("This is a not so deadly looking weapon.");
048                wornLogicalAnd=false;
049                properWornBitmap=Wearable.WORN_HELD|Wearable.WORN_WIELD;
050                basePhyStats().setAttackAdjustment(0);
051                basePhyStats().setDamage(0);
052                basePhyStats().setAbility(0);
053                baseGoldValue=15;
054                weaponDamageType=Weapon.TYPE_BASHING;
055                weaponClassification=Weapon.CLASS_BLUNT;
056                material=RawMaterial.RESOURCE_STEEL;
057                setUsesRemaining(100);
058                recoverPhyStats();
059        }
060
061        @Override
062        public String hitString(int damageAmount)
063        {
064                String word="boff(s)";
065                switch(CMLib.dice().roll(1,7,-1))
066                {
067                case 0:
068                        word = "puff(s)";
069                        break;
070                case 1:
071                        word = "boff(s)";
072                        break;
073                case 2:
074                        word = "poof(s)";
075                        break;
076                case 3:
077                        word = "bambam(s)";
078                        break;
079                case 4:
080                        word = "whack(s)";
081                        break;
082                case 5:
083                        word = "smoosh(es)";
084                        break;
085                case 6:
086                        word = "kabloom(s)";
087                        break;
088                }
089                final boolean showDamn=CMProps.getVar(CMProps.Str.SHOWDAMAGE).equalsIgnoreCase("YES");
090                switch(weaponClassification())
091                {
092                case Weapon.CLASS_RANGED:
093                        return "<S-NAME> fire(s) "+name()+" at <T-NAMESELF> and "+word+((showDamn)?" ("+damageAmount+")":"")+" <T-HIM-HER>."+CMLib.protocol().msp("arrow.wav",20);
094                case Weapon.CLASS_THROWN:
095                        return "<S-NAME> throw(s) "+name()+" at <T-NAMESELF> and "+word+((showDamn)?" ("+damageAmount+")":"")+" <T-HIM-HER>."+CMLib.protocol().msp("arrow.wav",20);
096                default:
097                        return "<S-NAME> "+word+((showDamn)?" ("+damageAmount+")":"")+" <T-NAMESELF> with "+name()+"."+CMLib.protocol().msp("punch"+CMLib.dice().roll(1,7,0)+".wav",20);
098                }
099        }
100}
