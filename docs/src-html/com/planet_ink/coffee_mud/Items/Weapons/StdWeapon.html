052        public StdWeapon()
053        {
054                super();
055

065                material=RawMaterial.RESOURCE_STEEL;
066                setUsesRemaining(100);

094        @Override
095        public String secretIdentity()
096        {
097                String id=super.secretIdentity();
098                final int timsLevel = CMLib.itemBuilder().timsLevelCalculator(this);
099                if(phyStats().ability()>0)
100                        id=name()+" +"+phyStats().ability()+((id.length()>0)?"\n":"")+id;
101                else
102                if(phyStats().ability()<0)
103                        id=name()+" "+phyStats().ability()+((id.length()>0)?"\n":"")+id;
104                if(timsLevel != phyStats.level())
105                        id += " (Power level: "+timsLevel+")";
106                return id+L("\n\rAttack: @x1, Damage: @x2",""+phyStats().attackAdjustment(),""+phyStats().damage());
107        }

109        @Override
110        public void affectPhyStats(final Physical affected, final PhyStats affectableStats)
111        {
112                super.affectPhyStats(affected,affectableStats);
113                if(amWearingAt(Wearable.WORN_WIELD))
114                {
115                        if(phyStats().attackAdjustment()!=0)
116                                affectableStats.setAttackAdjustment(affectableStats.attackAdjustment()+(phyStats().attackAdjustment()));
117                        if(phyStats().damage()!=0)
118                                affectableStats.setDamage(affectableStats.damage()+phyStats().damage());
119                }
120        }

122        @Override
123        public void recoverPhyStats()
124        {
125                super.recoverPhyStats();
126                if(phyStats().damage()!=0)
127                {
128                        final int ability=super.wornLogicalAnd ? (phyStats().ability()*CMath.numberOfSetBits(super.properWornBitmap)) : phyStats().ability();
129                        phyStats().setDamage(phyStats().damage()+(ability*2));
130                        phyStats().setAttackAdjustment(phyStats().attackAdjustment()+(ability*10));
131                }
132                if((subjectToWearAndTear())&&(usesRemaining()<100))
133                        phyStats().setDamage(((int)Math.round(CMath.mul(phyStats().damage(),CMath.div(usesRemaining(),100)))));
134        }

136        @Override
137        public void executeMsg(final Environmental myHost, final CMMsg msg)
138        {
139                super.executeMsg(myHost,msg);
140
141                if(msg.amITarget(this))
142                {
143                        switch(msg.targetMinor())
144                        {
145                        case CMMsg.TYP_LOOK:
146                        case CMMsg.TYP_EXAMINE:
147                                if(CMLib.flags().canBeSeenBy(this,msg.source()))
148                                {
149                                        if(requiresAmmunition())
150                                                msg.source().tell(L("@x1 remaining: @x2/@x3.",ammunitionType(),""+ammunitionRemaining(),""+ammunitionCapacity()));
151                                        if((subjectToWearAndTear())&&(usesRemaining()<100))
152                                                msg.source().tell(weaponHealth());
153                                }
154                                break;
155                        case CMMsg.TYP_RELOAD:
156                                if(msg.tool() instanceof Ammunition)
157                                {
158                                        boolean recover=false;
159                                        final Ammunition I=(Ammunition)msg.tool();
160                                        int howMuchToTake=ammunitionCapacity();
161                                        if(I.ammunitionRemaining()<howMuchToTake)
162                                                howMuchToTake=I.ammunitionRemaining();
163                                        if(this.ammunitionCapacity() - this.ammunitionRemaining() < howMuchToTake)
164                                                howMuchToTake=this.ammunitionCapacity() - this.ammunitionRemaining();
165                                        setAmmoRemaining(this.ammunitionRemaining() + howMuchToTake);
166                                        I.setAmmoRemaining(I.ammunitionRemaining()-howMuchToTake);
167                                        final LinkedList<Ability> removeThese=new LinkedList<Ability>();
168                                        for(final Enumeration<Ability> a=effects();a.hasMoreElements();)
169                                        {
170                                                final Ability A=a.nextElement();
171                                                if((A!=null)&&(!A.isSavable())&&(A.invoker()==null))
172                                                        removeThese.add(A);
173                                        }
174                                        for(final Ability A : removeThese)
175                                                delEffect(A);
176                                        for(final Enumeration<Ability> a=I.effects();a.hasMoreElements();)
177                                        {
178                                                Ability A=a.nextElement();
179                                                if((A!=null)&&(A.isSavable())&&(fetchEffect(A.ID())==null))
180                                                {
181                                                        A=(Ability)A.copyOf();
182                                                        A.setInvoker(null);
183                                                        A.setSavable(false);
184                                                        addEffect(A);
185                                                        recover=true;
186                                                }
187                                        }
188                                        if(I.ammunitionRemaining()<=0)
189                                                I.destroy();
190                                        if(recover)
191                                                recoverOwner();
192                                }
193                                break;
194                        case CMMsg.TYP_UNLOAD:
195                                if(msg.tool() instanceof Ammunition)
196                                {
197                                        final Ammunition ammo=(Ammunition)msg.tool();
198                                        for(final Enumeration<Ability> a=effects();a.hasMoreElements();)
199                                        {
200                                                final Ability A=a.nextElement();
201                                                if((A!=null)&&(!A.isSavable())&&(A.invoker()==null))
202                                                {
203                                                        final Ability ammoA=(Ability)A.copyOf();
204                                                        ammo.addNonUninvokableEffect(ammoA);
205                                                }
206                                        }
207                                        setAmmoRemaining(0);
208                                        final Room R=msg.source().location();
209                                        if(R!=null)
210                                        {
211                                                R.addItem(ammo, ItemPossessor.Expire.Player_Drop);
212                                                CMLib.commands().postGet(msg.source(), null, ammo, true);
213                                        }
214                                }
215                                break;
216                        }
217                }
218                else
219                if((msg.tool()==this)
220                &&(msg.targetMinor()==CMMsg.TYP_WEAPONATTACK)
221                &&(weaponClassification()==Weapon.CLASS_THROWN))
222                        msg.addTrailerMsg(CMClass.getMsg(msg.source(),this,CMMsg.MSG_DROP,null));
223
224                if((msg.targetMinor()==CMMsg.TYP_DAMAGE)
225                &&(msg.tool()==this)
226                &&(amWearingAt(Wearable.WORN_WIELD))
227                &&(weaponClassification()!=Weapon.CLASS_NATURAL)
228                &&(weaponDamageType()!=Weapon.TYPE_NATURAL)
229                &&(msg.target() instanceof MOB)
230                &&((msg.value())>0)
231                &&(owner() instanceof MOB)
232                &&(msg.amISource((MOB)owner())))
233                {
234                        final MOB ownerM=(MOB)owner();
235                        final int hurt=(msg.value());
236                        final MOB tmob=(MOB)msg.target();
237                        if((hurt>(tmob.maxState().getHitPoints()/10)||(hurt>50))
238                        &&(tmob.curState().getHitPoints()>hurt))
239                        {
240                                if((!tmob.isMonster())
241                                   &&(CMLib.dice().rollPercentage()==1)
242                                   &&(CMLib.dice().rollPercentage()>(tmob.charStats().getStat(CharStats.STAT_CONSTITUTION)*4))
243                                   &&(!CMSecurity.isDisabled(CMSecurity.DisFlag.AUTODISEASE)))
244                                {
245                                        Ability A=null;
246                                        if(subjectToWearAndTear()
247                                        &&(usesRemaining()<25)
248                                        &&((material()&RawMaterial.MATERIAL_MASK)==RawMaterial.MATERIAL_METAL))
249                                        {
250                                                if(CMLib.dice().rollPercentage()>50)
251                                                        A=CMClass.getAbility("Disease_Lockjaw");
252                                                else
253                                                        A=CMClass.getAbility("Disease_Tetanus");
254                                                if((A!=null)&&(tmob.fetchEffect(A.ID())==null)&&(!CMSecurity.isAbilityDisabled(A.ID())))
255                                                        A.invoke(msg.source(),tmob,true,phyStats().level());
256                                        }
257                                }
258                        }
259
260                        if((subjectToWearAndTear())
261                        &&(CMLib.dice().rollPercentage()==1)
262                        &&(msg.source().rangeToTarget()==0)
263                        &&(CMLib.dice().rollPercentage()>((phyStats().level()/2)+(10*phyStats().ability())+(CMLib.flags().isABonusItems(this)?20:0)))
264                        &&((material()&RawMaterial.MATERIAL_MASK)!=RawMaterial.MATERIAL_ENERGY)
265                        &&((material()&RawMaterial.MATERIAL_MASK)!=RawMaterial.MATERIAL_GAS))
266                        {
267                                CMLib.combat().postItemDamage(ownerM, this, null, 1, CMMsg.TYP_JUSTICE, null);
268                        }
269                }
270        }

272        @Override
273        public boolean okMessage(final Environmental myHost, final CMMsg msg)
274        {
275                if(!super.okMessage(myHost,msg))
276                        return false;
277
278                if((msg.targetMinor()==CMMsg.TYP_WEAPONATTACK)
279                &&(msg.tool()==this)
280                &&(requiresAmmunition())
281                &&(ammunitionCapacity()>0))
282                {
283                        if(ammunitionRemaining()>ammunitionCapacity())
284                                setAmmoRemaining(ammunitionCapacity());
285                        if(ammunitionRemaining()<=0)
286                        {
287                                if(lastReloadTime != msg.source().lastTickedDateTime())
288                                {
289                                        if(msg.source().isMonster() && (owner() instanceof Room))
290                                                ((Room)owner()).showHappens(CMMsg.MSG_OK_VISUAL, L("@x1 is out of @x2.",name(),ammunitionType()));
291                                        else
292                                                msg.source().tell(L("@x1 is out of @x2.",name(),ammunitionType()));
293                                        if((msg.source().isMine(this))
294                                        &&(msg.source().location()!=null)
295                                        &&(CMLib.flags().isAliveAwakeMobile(msg.source(),true)))
296                                        {
297                                                lastReloadTime=msg.source().lastTickedDateTime();
298                                                if((!msg.source().isMonster())||inventoryAmmoCheck(msg.source()))
299                                                        msg.source().enqueCommand(CMParms.parse("LOAD ALL \"$"+name()+"$\""), 0, 0);
300                                                else
301                                                        msg.source().enqueCommand(CMParms.parse("REMOVE \"$"+name()+"$\""), 0, 0);
302                                        }
303                                }
304                                return false;
305                        }
306                        else
307                                setUsesRemaining(usesRemaining()-1);
308                }
309                return true;
310        }
311
312        protected boolean inventoryAmmoCheck(final MOB M)
313        {
314                if(M==null)
315                        return false;
316                for(int i=0;i<M.numItems();i++)
317                {
318                        final Item I=M.getItem(i);
319                        if((I instanceof Ammunition)
320                        &&(((Ammunition)I).ammunitionType().equalsIgnoreCase(ammunitionType())))
321                                return true;
322                }
323                return false;
324        }

334        protected String weaponHealth()
335        {
336                final int[] condSet = new int[]{95, 85, 75, 50, 25, 10, 5, 0};
337                int ordinal=0;
338                for(int i=0;i<condSet.length;i++)
339                {
340                        if(usesRemaining()>=condSet[i])
341                        {
342                                ordinal=i;
343                                break;
344                        }
345                }
346                final String message;
347                switch(weaponClassification())
348                {
349                case Weapon.CLASS_AXE:
350                case Weapon.CLASS_DAGGER:
351                case Weapon.CLASS_EDGED:
352                case Weapon.CLASS_POLEARM:
353                case Weapon.CLASS_SWORD:
354                        message=CMProps.getListFileChoiceFromIndexedList(CMProps.ListFile.WEAPON_CONDITION_EDGED, ordinal);
355                        break;
356                default:
357                        message=CMProps.getListFileChoiceFromIndexedList(CMProps.ListFile.WEAPON_CONDITION_OTHER, ordinal);
358                        break;
359                }
360                return L(message,name(),""+usesRemaining());
361        }

431        @Override
432        public void setAmmoRemaining(int amount)
433        {
434                final int oldAmount=ammunitionRemaining();
435                if(amount==Integer.MAX_VALUE)
436                        amount=20;
437                setUsesRemaining(amount);
438                final ItemPossessor myOwner=owner;
439                if((oldAmount>0)
440                &&(amount==0)
441                &&(myOwner instanceof MOB)
442                &&(ammunitionCapacity()>0))
443                {
444                        boolean recover=false;
445                        for(final Enumeration<Ability> a=effects();a.hasMoreElements();)
446                        {
447                                final Ability A=a.nextElement();
448                                if((A!=null)&&(!A.isSavable())&&(A.invoker()==null))
449                                {
450                                        recover=true;
451                                        delEffect(A);
452                                }
453                        }
454                        if(recover)
455                                recoverOwner();
456                }
457        }

486        }

488        public void recoverOwner()
489        {
490                final ItemPossessor myOwner=owner;
491                if(myOwner instanceof MOB)
492                {
493                        ((MOB)myOwner).recoverCharStats();
494                        ((MOB)myOwner).recoverMaxState();
495                        ((MOB)myOwner).recoverPhyStats();
496                }
497                else
498                if(myOwner!=null)
499                        myOwner.recoverPhyStats();
500        }
