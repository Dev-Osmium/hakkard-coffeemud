064        @Override
065        public String secretIdentity()
066        {
067                final String tx=StdScroll.makeSecretIdentity("super pill",super.secretIdentity(),"",getSpells());
068                String id=readableText;
069                int x=id.toUpperCase().indexOf("ARM");
070                for(final StringBuffer ID=new StringBuffer(id);((x>0)&&(x<id.length()));x++)
071                {
072                        if(id.charAt(x)=='-')
073                        {
074                                ID.setCharAt(x,'+');
075                                id=ID.toString();
076                                break;
077                        }
078                        else
079                        if(id.charAt(x)=='+')
080                        {
081                                ID.setCharAt(x,'-');
082                                id=ID.toString();
083                                break;
084                        }
085                        else
086                        if(Character.isDigit(id.charAt(x)))
087                                break;
088                }
089                x=id.toUpperCase().indexOf("DIS");
090                if(x>=0)
091                {
092                        final long val=CMParms.getParmPlus(id,"dis");
093                        final int y=id.indexOf(""+val,x);
094                        if((val!=0)&&(y>x))
095                        {
096                                final StringBuffer middle=new StringBuffer("");
097                                for(int num=0;num<PhyStats.IS_VERBS.length;num++)
098                                {
099                                        if(CMath.bset(val,CMath.pow(2,num)))
100                                                middle.append(PhyStats.IS_VERBS[num]+" ");
101                                }
102                                id=id.substring(0,x)+middle.toString().trim()+id.substring(y+((""+val).length()));
103                        }
104                }
105                x=id.toUpperCase().indexOf("SEN");
106                if(x>=0)
107                {
108                        final long val=CMParms.getParmPlus(id,"sen");
109                        final int y=id.indexOf(""+val,x);
110                        if((val!=0)&&(y>x))
111                        {
112                                final StringBuffer middle=new StringBuffer("");
113                                for(int num=0;num<PhyStats.CAN_SEE_VERBS.length;num++)
114                                {
115                                        if(CMath.bset(val,CMath.pow(2,num)))
116                                                middle.append(PhyStats.CAN_SEE_VERBS[num]+" ");
117                                }
118                                id=id.substring(0,x)+middle.toString().trim()+id.substring(y+((""+val).length()));
119                        }
120                }
121                return tx+"\n("+id+")\n";
122        }

124        public void EATME(MOB mob)
125        {
126                boolean redress=false;
127                if(getSpells().size()>0)
128                        eatIfAble(mob);
129                if((CMParms.getParmPlus(readableText,"beacon")>0)
130                &&(mob.location()!=null))
131                        mob.setStartRoom(mob.location());
132                mob.basePhyStats().setAbility(mob.basePhyStats().ability()+CMParms.getParmPlus(readableText,"abi"));
133                mob.basePhyStats().setArmor(mob.basePhyStats().armor()+CMParms.getParmPlus(readableText,"arm"));
134                mob.basePhyStats().setAttackAdjustment(mob.basePhyStats().attackAdjustment()+CMParms.getParmPlus(readableText,"att"));
135                mob.basePhyStats().setDamage(mob.basePhyStats().damage()+CMParms.getParmPlus(readableText,"dam"));
136                mob.basePhyStats().setDisposition(mob.basePhyStats().disposition()|CMParms.getParmPlus(readableText,"dis"));
137                mob.basePhyStats().setLevel(mob.basePhyStats().level());
138                mob.basePhyStats().setRejuv(mob.basePhyStats().rejuv()+CMParms.getParmPlus(readableText,"rej"));
139                mob.basePhyStats().setSensesMask(mob.basePhyStats().sensesMask()|CMParms.getParmPlus(readableText,"sen"));
140                mob.basePhyStats().setSpeed(mob.basePhyStats().speed()+CMParms.getParmPlus(readableText,"spe"));
141                mob.basePhyStats().setWeight(mob.basePhyStats().weight()+CMParms.getParmPlus(readableText,"wei"));
142                if(CMParms.getParmPlus(readableText,"wei")!=0)
143                        redress=true;
144                mob.basePhyStats().setHeight(mob.basePhyStats().height()+CMParms.getParmPlus(readableText,"hei"));
145                if(CMParms.getParmPlus(readableText,"hei")!=0)
146                        redress=true;
147
148                String val=CMParms.getParmStr(readableText,"gen","").toUpperCase();
149                if((val.length()>0)&&((val.charAt(0)=='M')||(val.charAt(0)=='F')||(val.charAt(0)=='N')))
150                        mob.baseCharStats().setStat(CharStats.STAT_GENDER,val.charAt(0));
151                val=CMParms.getParmStr(readableText,"cla","").toUpperCase();
152                if(val.length()>0)
153                {
154                        final CharClass C=CMClass.findCharClass(val);
155                        if((C!=null)&&(C.availabilityCode()!=0))
156                        {
157                                mob.baseCharStats().setCurrentClass(C);
158                                if((!mob.isMonster())&&(mob.soulMate()==null))
159                                        CMLib.coffeeTables().bump(mob,CoffeeTableRow.STAT_CLASSCHANGE);
160                        }
161                }
162                if(CMParms.getParmPlus(readableText,"lev")!=0)
163                {
164                        int num=CMParms.getParmPlus(readableText,"lev");
165                        if((mob.charStats().getCurrentClass().leveless())
166                        ||(mob.charStats().isLevelCapped(mob.charStats().getCurrentClass()))
167                        ||(mob.charStats().getMyRace().leveless())
168                        ||(CMSecurity.isDisabled(CMSecurity.DisFlag.LEVELS)))
169                        {
170                                
171                        }
172                        else
173                        if(num > 0)
174                        {
175                                for(int i=0;i<num;i++)
176                                {
177                                        if((mob.getExpNeededLevel()==Integer.MAX_VALUE)
178                                        ||(mob.charStats().getCurrentClass().expless())
179                                        ||(mob.charStats().getMyRace().expless()))
180                                                CMLib.leveler().level(mob);
181                                        else
182                                                CMLib.leveler().postExperience(mob,null,null,mob.getExpNeededLevel()+1,false);
183                                }
184                        }
185                        else
186                        if(num < 0)
187                        {
188                                num=num*-1;
189                                for(int i=0;i<num;i++)
190                                {
191                                        if(mob.basePhyStats().level() > 1)
192                                        {
193                                                CMLib.leveler().unLevel(mob);
194                                        }
195                                }
196                        }
197                }
198                val=CMParms.getParmStr(readableText,"rac","").toUpperCase();
199                if((val.length()>0)&&(CMClass.getRace(val)!=null))
200                {
201                        redress=true;
202                        mob.baseCharStats().setMyRace(CMClass.getRace(val));
203                        mob.baseCharStats().getMyRace().startRacing(mob,false);
204                }
205                for(final int i : CharStats.CODES.BASECODES())
206                {
207                        mob.baseCharStats().setStat(i,mob.baseCharStats().getStat(i)+CMParms.getParmPlus(readableText,CMStrings.limit(CharStats.CODES.NAME(i).toLowerCase(),3)));
208                        final int max = CharStats.CODES.toMAXBASE(i);
209                        mob.baseCharStats().setStat(max,mob.baseCharStats().getStat(max)+CMParms.getParmPlus(readableText,"max"+CMStrings.limit(CharStats.CODES.NAME(i).toLowerCase(),3)));
210                }
211
212                mob.baseState().setHitPoints(mob.baseState().getHitPoints()+CMParms.getParmPlus(readableText,"hit"));
213                mob.curState().setHunger(mob.curState().getHunger()+CMParms.getParmPlus(readableText,"hun"));
214                mob.baseState().setMana(mob.baseState().getMana()+CMParms.getParmPlus(readableText,"man"));
215                mob.baseState().setMovement(mob.baseState().getMovement()+CMParms.getParmPlus(readableText,"mov"));
216                mob.curState().setThirst(mob.curState().getThirst()+CMParms.getParmPlus(readableText,"thi"));
217
218                mob.setPractices(mob.getPractices()+CMParms.getParmPlus(readableText,"prac"));
219                mob.setTrains(mob.getTrains()+CMParms.getParmPlus(readableText,"trai"));
220                final int qp=CMParms.getParmPlus(readableText,"ques");
221                if(qp!=0)
222                        mob.setQuestPoint(mob.getQuestPoint()+qp);
223                if(qp>0)
224                        CMLib.players().bumpPrideStat(mob,PrideStat.QUESTPOINTS_EARNED, qp);
225                final int newMoney=CMParms.getParmPlus(readableText,"coin");
226                if(newMoney!=0)
227                        CMLib.beanCounter().setMoney(mob,CMLib.beanCounter().getMoney(mob)+newMoney);
228                final int exp=CMParms.getParmPlus(readableText,"expe");
229                if(exp!=0)
230                        CMLib.leveler().postExperience(mob,null,null,exp,false);
231                mob.recoverCharStats();
232                mob.recoverPhyStats();
233                mob.recoverMaxState();
234                if(redress)
235                        CMLib.utensils().confirmWearability(mob);
236        }

238        @Override
239        public void executeMsg(final Environmental myHost, final CMMsg msg)
240        {
241                if(msg.amITarget(this))
242                {
243                        final MOB mob=msg.source();
244                        switch(msg.targetMinor())
245                        {
246                        case CMMsg.TYP_EAT:
247                                if((msg.sourceMessage()==null)&&(msg.othersMessage()==null))
248                                {
249                                        EATME(mob);
250                                        super.executeMsg(myHost,msg);
251                                }
252                                else
253                                        msg.addTrailerMsg(CMClass.getMsg(msg.source(),msg.target(),msg.tool(),CMMsg.NO_EFFECT,null,msg.targetCode(),msg.targetMessage(),CMMsg.NO_EFFECT,null));
254                                break;
255                        default:
256                                super.executeMsg(myHost,msg);
257                                break;
258                        }
259                }
260                else
261                        super.executeMsg(myHost,msg);
262        }
