036public class StdPotion extends StdDrink implements Potion
037{
038        @Override
039        public String ID()
040        {
041                return "StdPotion";
042        }
043
044        public StdPotion()
045        {
046                super();
047
048                setName("a potion");
049                basePhyStats.setWeight(1);
050                setDisplayText("An empty potion sits here.");
051                setDescription("An empty potion with strange residue.");
052                secretIdentity="What was once a powerful potion.";
053                capacity=1;
054                containType=Container.CONTAIN_LIQUID;
055                baseGoldValue=200;
056                material=RawMaterial.RESOURCE_GLASS;
057                liquidType = RawMaterial.RESOURCE_DRINKABLE;
058                recoverPhyStats();
059        }
060
061        @Override
062        public boolean isDrunk()
063        {
064                return (getSpellList().toUpperCase().indexOf(";DRUNK") >= 0);
065        }
066
067        @Override
068        public int value()
069        {
070                if(isDrunk())
071                        return 0;
072                return super.value();
073        }
074
075        @Override
076        public void setDrunk(boolean isTrue)
077        {
078                if(isTrue&&isDrunk())
079                        return;
080                if((!isTrue)&&(!isDrunk()))
081                        return;
082                if(isTrue)
083                        setSpellList(getSpellList()+";DRUNK");
084                else
085                {
086                        String list="";
087                        final List<Ability> theSpells=getSpells();
088                        for(int v=0;v<theSpells.size();v++)
089                                list+=theSpells.get(v).ID()+";";
090                        setSpellList(list);
091                }
092        }
093
094        @Override
095        public void drinkIfAble(MOB owner, Physical drinkerTarget)
096        {
097                final List<Ability> spells=getSpells();
098                if(owner.isMine(this))
099                {
100                        if((!isDrunk())&&(spells.size()>0))
101                        {
102                                final MOB caster=CMLib.map().getFactoryMOB(owner.location());
103                                final MOB finalCaster=(owner!=drinkerTarget)?owner:caster;
104                                boolean destroyCaster = true;
105                                for(int i=0;i<spells.size();i++)
106                                {
107                                        final Ability thisOneA=(Ability)spells.get(i).copyOf();
108                                        if((drinkerTarget instanceof Item)
109                                        &&((!thisOneA.canTarget(drinkerTarget))&&(!thisOneA.canAffect(drinkerTarget))))
110                                                continue;
111                                        int level=phyStats().level();
112                                        final int lowest=CMLib.ableMapper().lowestQualifyingLevel(thisOneA.ID());
113                                        if(level<lowest)
114                                                level=lowest;
115                                        caster.basePhyStats().setLevel(level);
116                                        caster.phyStats().setLevel(level);
117                                        thisOneA.invoke(finalCaster,drinkerTarget,true,level);
118                                        Ability effectA=drinkerTarget.fetchEffect(thisOneA.ID());
119                                        if((effectA!=null) && (effectA.invoker() == caster))
120                                                destroyCaster =  false; 
121                                        setDrunk(true);
122                                        setLiquidRemaining(0);
123                                }
124                                if(destroyCaster) // don't always, because the invoker, and their level, are often needed.
125                                        caster.destroy();
126                        }
127                }
128        }
129
130        @Override
131        public String getSpellList()
132        {
133                return miscText;
134        }
135
136        @Override
137        public void setSpellList(String list)
138        {
139                miscText = list;
140        }
141
142        public static List<Ability> getSpells(SpellHolder me)
143        {
144                int baseValue=200;
145                final Vector<Ability> theSpells=new Vector<Ability>();
146                final String names=me.getSpellList();
147                final List<String> parsedSpells=CMParms.parseSemicolons(names, true);
148                for(String thisOne : parsedSpells)
149                {
150                        thisOne=thisOne.trim();
151                        String parms="";
152                        final int x=thisOne.indexOf('(');
153                        if((x>0)&&(thisOne.endsWith(")")))
154                        {
155                                parms=thisOne.substring(x+1,thisOne.length()-1);
156                                thisOne=thisOne.substring(0,x).trim();
157                        }
158                        Ability A=CMClass.getAbility(thisOne);
159                        if((A!=null)&&((A.classificationCode()&Ability.ALL_DOMAINS)!=Ability.DOMAIN_ARCHON))
160                        {
161                                A=(Ability)A.copyOf();
162                                A.setMiscText(parms);
163                                baseValue+=(100*CMLib.ableMapper().lowestQualifyingLevel(A.ID()));
164                                theSpells.addElement(A);
165                        }
166                }
167                me.setBaseValue(baseValue);
168                me.recoverPhyStats();
169                return theSpells;
170        }
171
172        @Override
173        public List<Ability> getSpells()
174        {
175                return getSpells(this);
176        }
177
178        @Override
179        public String secretIdentity()
180        {
181                return StdScroll.makeSecretIdentity("potion",super.secretIdentity(),"",getSpells(this));
182        }
183
184        @Override
185        public boolean okMessage(final Environmental myHost, final CMMsg msg)
186        {
187                if((msg.amITarget(this))
188                &&(msg.targetMinor()==CMMsg.TYP_DRINK)
189                &&(msg.othersMessage()==null)
190                &&(msg.sourceMessage()==null))
191                        return true;
192                else
193                if((msg.tool()==this)
194                &&(msg.targetMinor()==CMMsg.TYP_POUR))
195                        return true;
196                return super.okMessage(myHost,msg);
197        }
198
199        @Override
200        public void executeMsg(final Environmental myHost, final CMMsg msg)
201        {
202                if(msg.amITarget(this))
203                {
204                        final MOB mob=msg.source();
205                        switch(msg.targetMinor())
206                        {
207                        case CMMsg.TYP_DRINK:
208                                if((msg.sourceMessage()==null)&&(msg.othersMessage()==null))
209                                {
210                                        drinkIfAble(mob,mob);
211                                        mob.tell(L("@x1 vanishes!",name()));
212                                        destroy();
213                                        mob.recoverPhyStats();
214                                }
215                                else
216                                {
217                                        msg.addTrailerMsg(CMClass.getMsg(msg.source(),msg.target(),msg.tool(),CMMsg.NO_EFFECT,null,msg.targetCode(),msg.targetMessage(),CMMsg.NO_EFFECT,null));
218                                        super.executeMsg(myHost,msg);
219                                }
220                                break;
221                        default:
222                                super.executeMsg(myHost,msg);
223                                break;
224                        }
225                }
226                else
227                if((msg.tool()==this)&&(msg.targetMinor()==CMMsg.TYP_POUR)&&(msg.target() instanceof Physical))
228                {
229                        if((msg.sourceMessage()==null)&&(msg.othersMessage()==null))
230                        {
231                                drinkIfAble(msg.source(),(Physical)msg.target());
232                                msg.source().tell(L("@x1 vanishes!",name()));
233                                destroy();
234                                msg.source().recoverPhyStats();
235                                ((Physical)msg.target()).recoverPhyStats();
236                        }
237                        else
238                        {
239                                msg.addTrailerMsg(CMClass.getMsg(msg.source(),msg.target(),msg.tool(),CMMsg.NO_EFFECT,null,msg.targetCode(),msg.targetMessage(),CMMsg.NO_EFFECT,null));
240                                super.executeMsg(myHost,msg);
241                        }
242                }
243                else
244                        super.executeMsg(myHost,msg);
245        }
