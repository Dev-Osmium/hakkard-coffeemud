074        public static boolean useTheWand(final Ability A, final MOB mob, final int level)
075        {
076                int manaRequired=5;
077                final int q=CMLib.ableMapper().qualifyingLevel(mob,A);
078                if(q>0)
079                {
080                        if(q<CMLib.ableMapper().qualifyingClassLevel(mob,A))
081                                manaRequired=0;
082                        else
083                                manaRequired=5;
084                }
085                else
086                        manaRequired=25;
087                manaRequired-=(2*level);
088                if(manaRequired<5)
089                        manaRequired=5;
090                if(manaRequired>mob.curState().getMana())
091                {
092                        mob.tell(CMLib.lang().L("You don't have enough mana."));
093                        return false;
094                }
095                mob.curState().adjMana(-manaRequired,mob.maxState());
096                return true;
097        }

140        @Override

142        {
143                String id=super.secretIdentity();
144                final Ability A=getSpell();
145                String uses;
146                if(this.maxUses() < 999999)

167        public static boolean checkWave(final MOB mob, final String message, final Wand me)
168        {
169                return (mob.isMine(me))
170                                && (message!=null)
171                                && (me.amBeingWornProperly())
172                                && (message.toUpperCase().indexOf(me.magicWord().toUpperCase()) >= 0);
173        }
174
175        public static void waveIfAble(final MOB mob, final Physical afftarget, String message, final Wand me)
176        {
177                if((mob.isMine(me)) &&(message!=null) &&(!me.amWearingAt(Wearable.IN_INVENTORY)))
178                {
179                        Physical target=null;
180                        if(mob.location()!=null)
181                                target=afftarget;
182                        final int x=message.toUpperCase().indexOf(me.magicWord().toUpperCase());
183                        if(x>=0)
184                        {
185                                message=message.substring(x+me.magicWord().length());
186                                final int y=message.indexOf('\'');
187                                if(y>=0)
188                                        message=message.substring(0,y);
189                                message=message.trim();
190                                final Ability wandUse=mob.fetchAbility("Skill_WandUse");
191                                if((wandUse==null)||(!wandUse.proficiencyCheck(null,0,false)))
192                                        mob.tell(CMLib.lang().L("@x1 glows faintly for a moment, then fades.",me.name()));
193                                else
194                                {
195                                        Ability A=me.getSpell();
196                                        if(A==null)
197                                                mob.tell(CMLib.lang().L("Something seems wrong with @x1.",me.name()));
198                                        else
199                                        if(me.usesRemaining()<=0)
200                                                mob.tell(CMLib.lang().L("@x1 seems spent.",me.name()));
201                                        else
202                                        {
203                                                wandUse.setInvoker(mob);
204                                                A=(Ability)A.newInstance();
205                                                if(useTheWand(A,mob,wandUse.abilityCode()))
206                                                {
207                                                        final Vector<String> V=new Vector<String>();
208                                                        if(target!=null)
209                                                                V.addElement(target.name());
210                                                        V.addAll(CMParms.parse(message));
211                                                        mob.location().show(mob,null,CMMsg.MSG_OK_VISUAL,CMLib.lang().L("@x1 glows brightly.",me.name()));
212                                                        me.setUsesRemaining(me.usesRemaining()-1);
213                                                        int level=me.phyStats().level()
214                                                                        + CMLib.expertises().getExpertiseLevel(mob, "Skill_WandUse", ExpertiseLibrary.Flag.LEVEL);
215                                                        final int lowest=CMLib.ableMapper().lowestQualifyingLevel(A.ID());
216                                                        if(level<lowest)
217                                                                level=lowest;
218                                                        A.invoke(mob, V, target, true, level);
219                                                        wandUse.helpProficiency(mob, 0);
220                                                        return;
221                                                }
222                                        }
223                                }
224                        }
225                }
226        }

228        @Override
229        public void executeMsg(final Environmental myHost, final CMMsg msg)
230        {
231                final MOB mob=msg.source();
232
233                switch(msg.targetMinor())
234                {
235                case CMMsg.TYP_WAND_USE:
236                        if(msg.amITarget(this)&&((msg.tool()==null)||(msg.tool() instanceof Physical)))
237                                waveIfAble(mob,(Physical)msg.tool(),msg.targetMessage());
238                        break;
239                case CMMsg.TYP_SPEAK:
240                        if((msg.sourceMinor()==CMMsg.TYP_SPEAK)
241                        &&(!amWearingAt(Wearable.IN_INVENTORY)))
242                        {
243                                boolean alreadyWanding=false;
244                                final List<CMMsg> trailers =msg.trailerMsgs();
245                                if(trailers!=null)
246                                {
247                                        for(final CMMsg msg2 : trailers)
248                                        {
249                                                if((msg2.targetMinor()==CMMsg.TYP_WAND_USE)
250                                                &&(msg2.target() == this))
251                                                        alreadyWanding=true;
252                                        }
253                                }
254                                final String said=CMStrings.getSayFromMessage(msg.sourceMessage());
255                                if((!alreadyWanding)&&(said!=null)&&(checkWave(mob,said)))
256                                        msg.addTrailerMsg(CMClass.getMsg(msg.source(),this,msg.target(),CMMsg.NO_EFFECT,null,CMMsg.MASK_ALWAYS|CMMsg.TYP_WAND_USE,said,CMMsg.NO_EFFECT,null));
257                        }
258                        break;
259                default:
260                        break;
261                }
262                super.executeMsg(myHost,msg);
263        }

271        protected static String[] CODES={"CLASS","LEVEL","ABILITY","TEXT"};
272
273        @Override
274        public String getStat(final String code)
275        {
276                switch(getCodeNum(code))
277                {
278                case 0:
279                        return ID();
280                case 1:
281                        return "" + basePhyStats().ability();
282                case 2:
283                        return "" + basePhyStats().level();
284                case 3:
285                        return text();
286                }
287                return "";
288        }
289
290        @Override
291        public void setStat(final String code, final String val)
292        {
293                switch(getCodeNum(code))
294                {
295                case 0:
296                        return;
297                case 1:
298                        basePhyStats().setLevel(CMath.s_parseIntExpression(val));
299                        break;
300                case 2:
301                        basePhyStats().setAbility(CMath.s_parseIntExpression(val));
302                        break;
303                case 3:
304                        setMiscText(val);
305                        break;
306                }
307        }
308
309        @Override
310        public String[] getStatCodes()
311        {
312                return CODES;
313        }
314
315        @Override
316        protected int getCodeNum(final String code)
317        {
318                for(int i=0;i<CODES.length;i++)
319                {
320                        if(code.equalsIgnoreCase(CODES[i]))
321                                return i;
322                }
323                return -1;
324        }
325
326        @Override
327        public boolean sameAs(final Environmental E)
328        {
329                if(!(E instanceof StdWand))
330                        return false;
331                final String[] codes=getStatCodes();
332                for(int i=0;i<codes.length;i++)
333                {
334                        if(!E.getStat(codes[i]).equals(getStat(codes[i])))
335                                return false;
336                }
337                return true;
338        }
