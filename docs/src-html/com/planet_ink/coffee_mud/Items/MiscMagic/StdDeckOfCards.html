041        Deck Of Cards
042        This class represents a deck of 52 cards without jokers
043        It inherets from the HandOfCards class for most of its
044        important functionality such as retreiving and adding cards and
045        shuffling and sorting.
046
047        The Deck has added functionality to make the "reigning in" of cards
048        once a game is over easier.  It also has methods to keep track of
049        one or more Hands, each of which must be attributed to a player.
050*/
051
052public class StdDeckOfCards extends StdHandOfCards implements DeckOfCards
053{
054        @Override
055        public String ID()
056        {
057                return "StdDeckOfCards";
058        }
059
060        // a flag to tell us whether the deck instance
061        // has already been filled with cards.
062        boolean alreadyFilled=false;
063
064        // a vector of all the cards in the deck.
065        // this is our private cache, since, as soon
066        // as cards start getting dealt, we would lose
067        // track of them otherwise.
068        protected List<Item> cardsCache=null;
069
070        // this object can manage one or more hands
071        // keyed by the mob/player object.
072        // This functionality is optional.
073        DVector hands=new DVector(2);
074
075        // the constructor for this class doesn't do much except set
076        // some of the display properties of the deck container
077        public StdDeckOfCards()
078        {
079                super();
080                name="A deck of cards";
081                displayText=L("A deck of cards has been left here.");
082                secretIdentity="A magical deck of cards.  Say \"Shuffle\" to me.";
083                recoverPhyStats();
084        }
085
086        @Override
087        protected boolean abilityImbuesMagic()
088        {
089                return false;
090        }
091
092        // makePlayingCard(int cardBitCode)
093        // this method creates a playing card object for
094        // population in the deck.  The card created is
095        // determined by the cardCode, which is a bit masked
096        // value where bits 4-5 determine suit, and lower bits
097        // the value.
098        protected PlayingCard makePlayingCard(int cardBitCode)
099        {
100                final Item I=CMClass.getItem("StdPlayingCard");
101                I.basePhyStats().setAbility(cardBitCode);
102                I.recoverPhyStats();
103                I.setContainer(this);
104                return (PlayingCard)I;
105        }
106
107        // makeAllCards()
108        // this method creates all 52 cards in the deck
109        // and adds them to a vector which is returned.
110        private Vector<PlayingCard> makeAllCards()
111        {
112                final Vector<PlayingCard> allCards=new Vector<PlayingCard>();
113                for (final int suit : PlayingCard.suits)
114                {
115                        for (final int card : PlayingCard.cards)
116                                allCards.addElement(makePlayingCard(suit+card));
117                }
118                return allCards;
119        }
120
121        // fillInTheDeck()
122        // this method creates all 52 cards in the deck
123        // and adds them to a deck owner
124        private List<Item> fillInTheDeck()
125        {
126                alreadyFilled=true;
127                if(!hasContent())
128                {
129                        final Vector<PlayingCard> allCards=makeAllCards();
130                        for(int i=0;i<allCards.size();i++)
131                                addCard(allCards.elementAt(i));
132                }
133                cardsCache=getContents();
134                hands.clear();
135                return cardsCache;
136        }
137
138        // createDeck(Environmental owner)
139        // This static method creates a new deck of cards container,
140        // gives it to the given owner object (mob or room), and
141        // then populates the deck container with all appropriate cards.
142        @Override
143        public DeckOfCards createDeck(Environmental owner)
144        {
145                final StdDeckOfCards deck=new StdDeckOfCards();
146                if(owner instanceof MOB)
147                {
148                        if(deck.owner==null)
149                                ((MOB)owner).addItem(deck);
150                        else
151                                ((MOB)owner).moveItemTo(deck);
152                }
153                else
154                if(owner instanceof Room)
155                {
156                        if(deck.owner==null)
157                                ((Room)owner).addItem(deck);
158                        else
159                                ((Room)owner).moveItemTo(deck,ItemPossessor.Expire.Player_Drop);
160                }
161                deck.fillInTheDeck();
162                return deck;
163        }
164
165        // resetDeckBackTo52Cards()
166        // resets the deck back to 52 cards.  It will
167        // grab cards from all external sources first
168        // and return them to the deck container owner.
169        // If this fails to produce 52 cards, it will
170        // create a set of cards for the deck.
171        // this method also destroys any hands being
172        // managed.
173        @Override
174        public boolean resetDeckBackTo52Cards()
175        {
176                if((cardsCache==null)||(cardsCache.size()==0))
177                        return false;
178
179                // first retrieve all our cards by looping
180                // through our cached list.  If we already
181                // have a card, make sure its faced-down
182                for(int i=0;i<cardsCache.size();i++)
183                {
184                        final PlayingCard card=(PlayingCard)cardsCache.get(i);
185                        if(card.owner()!=owner())
186                                addCard(card);
187                        else
188                                card.turnFaceDown();
189                }
190                // next destroy and clear any hands we may
191                // be managing.
192                for(int h=hands.size()-1;h>=0;h--)
193                        ((Item)hands.elementAt(h,2)).destroy();
194                hands.clear();
195
196                // if something went wrong (because a player cast
197                // disintegrate on their cards or something) we
198                // just ditch the entire deck and rebuild it.
199                if(numberOfCards()<52)
200                {
201                        for(int i=0;i<cardsCache.size();i++)
202                                cardsCache.get(i).destroy();
203                        cardsCache.clear();
204                        final Vector<PlayingCard> allCards=makeAllCards();
205                        for(int i=0;i<allCards.size();i++)
206                                addCard(allCards.elementAt(i));
207                        cardsCache=getContents();
208                }
209                return numberOfCards()==52;
210        }
211
212        // getPlayerHand(MOB player)
213        // If a hand of cards has previously been added to this
214        // deck for internal management, this method will return
215        // that hand given the player object.
216        @Override
217        public HandOfCards getPlayerHand(MOB player)
218        {
219                if(player!=null)
220                for(int i=0;i<hands.size();i++)
221                {
222                        if(hands.elementAt(i,1)==player)
223                                return (HandOfCards)hands.elementAt(i,2);
224                }
225                return null;
226        }
227
228        // addPlayerHand(MOB player, HandOfCards cards)
229        // adds and possibly creates a hand for the given player
230        // if no hand is passed in, a new empty one is created
231        // the hand is then added to our table, keyed by the player
232        // object
233        @Override
234        public HandOfCards addPlayerHand(MOB player, HandOfCards cards)
235        {
236                if(player==null)
237                        return null;
238                if(hands.contains(player))
239                        return (HandOfCards)hands.elementAt(hands.indexOf(player),2);
240                if(cards==null)
241                {
242                        cards=((HandOfCards)CMClass.getMiscMagic("StdHandOfCards")).createEmptyHand(player);
243                }
244                hands.addElement(player,cards);
245                return cards;
246        }
247
248        // removePlayerHand(MOB player)
249        // if the given player object has a hand of cards currently
250        // being managed by this deck, this method will remove all
251        // of the cards from the hand, return them to the deck,
252        // then remove the hand from management, and destroy the hand.
253        @Override
254        public void removePlayerHand(MOB player)
255        {
256                final HandOfCards cards=getPlayerHand(player);
257                if(cards==null)
258                        return;
259                final List<Item> cardSet=cards.getContents();
260                for(int c=0;c<cardSet.size();c++)
261                        addCard((PlayingCard)cardSet.get(c));
262                hands.removeElement(player);
263                cards.destroy();
264        }
265
266        // addCard(PlayingCard card)
267        // this method adds to the base functionality found
268        // in HandOfCards.java by ensuring that all cards
269        // added to the deck are added face down.
270        @Override
271        public boolean addCard(PlayingCard card)
272        {
273                if(card!=null)
274                        card.turnFaceDown();
275                return super.addCard(card);
276        }
277
278        // this is a system event previewer method
279        // its purpose is normally to preview any events occurring
280        // in the same room, see if they are relevant to this object,
281        // and if so, whether this object should modify or cancel
282        // the event before it takes place.
283        // There are two things our deck needs to handle here.
284        @Override
285        public boolean okMessage(final Environmental myHost, final CMMsg msg)
286        {
287                // In this case, we use the instance of an event to trigger
288                // the initial filling of the deck with cards, since, once
289                // this object is capable of handling events, it must already
290                // be "in the world" and ready to receive cards.
291                if((!alreadyFilled)&&(owner()!=null))
292                        fillInTheDeck();
293
294                // This handler also checks to see if anyone is saying "shuffle"
295                // directly to the deck item.  If so, we cancel the message by
296                // returning false (since it would make them look silly to be
297                // talking to a deck of cards), and instead
298                // display a message showing the owner of this deck shuffling it.
299                if((msg.amITarget(this))
300                &&(msg.targetMinor()==CMMsg.TYP_SPEAK)
301                &&(msg.targetMessage()!=null)
302                &&(msg.targetMessage().toUpperCase().indexOf("SHUFFLE")>0))
303                {
304                        if(!shuffleDeck())
305                                msg.source().tell(L("There are no cards left in the deck"));
306                        else
307                        {
308                                final Room R=CMLib.map().roomLocation(this);
309                                if(R!=null)
310                                        R.show(msg.source(),null,this,CMMsg.MASK_ALWAYS|CMMsg.MSG_QUIETMOVEMENT,
311                                                        L("<S-NAME> <S-HAS-HAVE> thoroughly shuffled <O-NAMESELF>."));
312                        }
313                        return false;
314                }
315                return super.okMessage(myHost,msg);
316        }
