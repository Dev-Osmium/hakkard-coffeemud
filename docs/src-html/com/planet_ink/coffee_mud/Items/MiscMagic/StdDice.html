038public class StdDice extends StdItem implements MiscMagic
039{
040        @Override
041        public String ID()
042        {
043                return "StdDice";
044        }
045
046        public StdDice()
047        {
048                super();
049
050                setName("a six sided die");
051                basePhyStats.setWeight(1);
052                basePhyStats().setAbility(6);
053                setDisplayText("A six sided die lies here.");
054                setDescription("It looks like a cube.");
055                secretIdentity="Try THROWing it, or ROLLing it!";
056                baseGoldValue=5;
057                recoverPhyStats();
058                material=RawMaterial.RESOURCE_BONE;
059        }
060
061        protected boolean rollTheBones(final CMMsg msg, final List<String> commands)
062        {
063                final MOB mob=msg.source();
064                String word = commands.remove(0);
065                final Room R=mob.location();
066                if(R==null)
067                        return false;
068
069                final Vector<Item> V=new Vector<Item>();
070                final int maxToDrop=CMLib.english().calculateMaxToGive(mob,commands,true,mob,false);
071                if(maxToDrop<0)
072                        return false;
073
074                String whatToDrop=CMParms.combine(commands,0);
075                boolean allFlag=(commands.size()>0)?commands.get(0).equalsIgnoreCase("all"):false;
076                if(whatToDrop.toUpperCase().startsWith("ALL."))
077                {
078                        allFlag=true;
079                        whatToDrop="ALL "+whatToDrop.substring(4);
080                }
081                if(whatToDrop.toUpperCase().endsWith(".ALL"))
082                {
083                        allFlag=true;
084                        whatToDrop="ALL "+whatToDrop.substring(0,whatToDrop.length()-4);
085                }
086                int addendum=1;
087                String addendumStr="";
088                Item dropThis=null;
089                boolean doBugFix = true;
090                if(V.size()==0)
091                {
092                        while(doBugFix || ((allFlag)&&(addendum<=maxToDrop)))
093                        {
094                                doBugFix=false;
095                                dropThis=mob.fetchItem(null,Wearable.FILTER_UNWORNONLY,whatToDrop+addendumStr);
096                                if((dropThis==null)
097                                &&(V.size()==0)
098                                &&(addendumStr.length()==0)
099                                &&(!allFlag))
100                                {
101                                        dropThis=mob.fetchItem(null,Wearable.FILTER_WORNONLY,whatToDrop);
102                                        if(dropThis!=null)
103                                        {
104                                                if((!dropThis.amWearingAt(Wearable.WORN_HELD))&&(!dropThis.amWearingAt(Wearable.WORN_WIELD)))
105                                                {
106                                                        mob.tell(L("You must remove that first."));
107                                                        return false;
108                                                }
109                                                final CMMsg newMsg=CMClass.getMsg(mob,dropThis,null,CMMsg.MSG_REMOVE,null);
110                                                if(R.okMessage(mob,newMsg))
111                                                        R.send(mob,newMsg);
112                                                else
113                                                        return false;
114                                        }
115                                }
116                                if(dropThis==null)
117                                        break;
118                                if((CMLib.flags().canBeSeenBy(dropThis,mob)||(dropThis instanceof Light))
119                                &&(!V.contains(dropThis))
120                                &&(dropThis instanceof StdDice))
121                                        V.add(dropThis);
122                                addendumStr="."+(++addendum);
123                        }
124                }
125
126                if(V.size()==0)
127                {
128                        if(word.toUpperCase().startsWith("T"))
129                                return true;
130                        else
131                                mob.tell(L("Roll what now?"));
132                        return false;
133                }
134                else
135                for(int i=0;i<V.size();i++)
136                {
137                        final Item I=V.get(i);
138                        if(!I.amDestroyed())
139                        {
140                                final CMMsg msg2=CMClass.getMsg(mob,I,null,CMMsg.MSG_DROP,L("<S-NAME> roll(s) <T-NAME>."));
141                                if(R.okMessage(mob, msg2))
142                                        R.send(mob, msg2);
143                        }
144                }
145                return false;
146        }
147
148        @Override
149        public void executeMsg(final Environmental myHost, final CMMsg msg)
150        {
151                if(msg.target()==this)
152                {
153                        if(msg.sourceMinor()==CMMsg.TYP_GET)
154                                this.setDisplayText(L("@x1 is lying here.",name()));
155                        else
156                        if(msg.sourceMinor()==CMMsg.TYP_DROP)
157                        {
158                                if(phyStats.ability()<2)
159                                {
160                                        basePhyStats.setAbility(2);
161                                        phyStats.setAbility(2);
162                                }
163                                int roll=CMLib.dice().roll(1, phyStats().ability(), 0);
164                                this.setDisplayText(L("@x1 is lying here, showing @x2.",name(),""+roll));
165                        }
166                }
167                super.executeMsg(myHost, msg);
168        }
169        
170        @Override
171        public boolean okMessage(final Environmental myHost, final CMMsg msg)
172        {
173                if(!super.okMessage(myHost, msg))
174                        return false;
175                if(((msg.sourceMinor()==CMMsg.TYP_HUH)
176                        &&(msg.targetMessage()!=null)
177                        &&(msg.targetMessage().length()>0))
178                ||((msg.tool() instanceof Social)
179                        &&(((Social)msg.tool()).baseName().toUpperCase().equals("ROLL"))))
180                {
181                        if(msg.tool() instanceof Social)
182                        {
183                                final Session sess=msg.source().session();
184                                if(msg.source().session()!=null)
185                                {
186                                        List<String> prev = sess.getPreviousCMD();
187                                        if((prev!=null)&&(prev.size()>0)&&("ROLL".startsWith(prev.get(0).toUpperCase())))
188                                                return rollTheBones(msg, new XVector<String>(prev));
189                                }
190                                return rollTheBones(msg, new XVector<String>(new String[]{"ROLL","$"+Name()+"$"}));
191                        }
192                        if(Character.toUpperCase(msg.targetMessage().charAt(0)) == 'R')
193                        {
194                                List<String> parsedFail = CMParms.parse(msg.targetMessage());
195                                if(parsedFail.size()<1)
196                                        return true;
197                                String cmd=parsedFail.get(0).toUpperCase();
198                                if(!("ROLL".startsWith(cmd)))
199                                        return true;
200                                if(parsedFail.size()<2)
201                                {
202                                        msg.source().tell(L("Roll what?"));
203                                        return false;
204                                }
205                                return rollTheBones(msg, parsedFail);
206                        }
207                }
208                else
209                if((msg.sourceMinor()==CMMsg.TYP_COMMANDFAIL)
210                &&(msg.targetMessage()!=null)
211                &&(msg.targetMessage().length()>0))
212                {
213                        if(Character.toUpperCase(msg.targetMessage().charAt(0)) == 'T')
214                        {
215                                List<String> parsedFail = CMParms.parse(msg.targetMessage());
216                                if(parsedFail.size()<2)
217                                        return true;
218                                String cmd=parsedFail.get(0).toUpperCase();
219                                if(!("THROW".startsWith(cmd)))
220                                        return true;
221                                return rollTheBones(msg, parsedFail);
222                        }
223                }
224                return true;
225        }
