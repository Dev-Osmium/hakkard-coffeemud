042        Hand Of Cards
043        This object represents an arbitarily sized collection of
044        PlayingCard objects.  It includes numerous methods for
045        retreiving and adding cards, shuffling and sorting, and
046        most expecially for returning a message of turned-up cards
047        to anyone who LOOKS at a person holding a hand with such
048        cars.
049*/
050public class StdHandOfCards extends StdContainer implements MiscMagic, HandOfCards
051{

055                return "StdHandOfCards";

099        // shuffleDeck()
100        // Shuffles the deck by removing a random card from the
101        // middle of the deck and adding it to the bottom
102        // 52*5 times.
103        @Override
104        public boolean shuffleDeck()
105        {
106                final List<Item> V=getContents();
107                final Environmental own=owner();
108                if(V.size()==0)
109                        return false;
110                for(int i=0;i<V.size()*5;i++)
111                {
112                        final Item I=V.get(CMLib.dice().roll(1,V.size(),-1));
113                        I.setContainer(this);
114                        if(own instanceof MOB)
115                        {
116                                I.removeFromOwnerContainer();
117                                ((MOB)own).addItem(I);
118                        }
119                        else
120                        if(own instanceof Room)
121                        {
122                                I.removeFromOwnerContainer();
123                                ((Room)own).addItem(I,ItemPossessor.Expire.Player_Drop);
124                        }
125                        else
126                        {
127                                backupContents.remove(I);
128                                backupContents.add(I);
129                        }
130                }
131                return true;
132        }

134        // getTopCardFromDeck()
135        // returns the top card item object from the deck
136        @Override
137        public PlayingCard getTopCardFromDeck()
138        {
139                final List<Item> deckContents=getContents();
140                if(deckContents.size()==0)
141                        return null;
142                final PlayingCard card=(PlayingCard)deckContents.get(0);
143                if(card.container() instanceof HandOfCards)
144                        ((HandOfCards)card.container()).removeCard(card);
145                return card;
146        }

148        // addCard(PlayingCard card)
149        // returns the given card item object to
150        // the deck by removing it from its current
151        // owner and adding it back to the decks owner
152        // and container.  If doing this causes a players
153        // hand to be devoid of cards, the hand container
154        // is destroyed.
155        @Override
156        public boolean addCard(PlayingCard card)
157        {
158                if(card==null)
159                        return false;
160                if(card.container() instanceof HandOfCards)
161                        ((HandOfCards)card.container()).removeCard(card);
162
163                if(owner() instanceof MOB)
164                {
165                        if(card.owner()==null)
166                                ((MOB)owner()).addItem(card);
167                        else
168                        if(!((MOB)owner()).isMine(card))
169                                ((MOB)owner()).moveItemTo(card);
170                }
171                else
172                if(owner() instanceof Room)
173                {
174                        if(card.owner()==null)
175                                ((Room)owner()).addItem(card);
176                        else
177                        if(!((Room)owner()).isContent(card))
178                                ((Room)owner()).moveItemTo(card,ItemPossessor.Expire.Player_Drop);
179                }
180                else
181                {
182                        card.setOwner(owner());
183                        if(!backupContents.contains(card))
184                                backupContents.add(card);
185                }
186                card.setContainer(this);
187                if(owner() instanceof MOB)
188                        return ((MOB)owner()).isMine(card);
189                else
190                if(owner() instanceof Room)
191                        return ((Room)owner()).isContent(card);
192                return true;
193        }

204        // removeCard(PlayingCard card)
205        // removes the given card from the
206        // deck and places it in limbo.  calls
207        // to this method should be followed
208        // by an addCard method on another deck.
209        @Override
210        public boolean removeCard(PlayingCard card)
211        {
212                final List<Item> handContents=getContents();
213                if(handContents.contains(card))
214                {
215                        if((card.owner() instanceof MOB)
216                        ||(card.owner() instanceof Room))
217                                card.removeFromOwnerContainer();
218                        else
219                                backupContents.remove(card);
220                        return true;
221                }
222                return false;
223        }

225        // removeAllCards()
226        // removes all cards from the deck and
227        // places them in limbo.  Calls to this
228        // method should be followed by either
229        // a destroy method on the cards themselves
230        // or an addCard method on another deck.
231        @Override
232        public boolean removeAllCards()
233        {
234                final List<Item> handContents=getContents();
235                if(handContents.size()==0)
236                        return false;
237                for(int i=0;i<handContents.size();i++)
238                        removeCard((PlayingCard)handContents.get(i));
239                return true;
240        }

242        // getContentsEncoded()
243        // This method builds a string array equal in size to the deck.
244        // It then returns the contents of the deck encoded in
245        // cardStringCode format.  See convertCardBitCodeToCardStringCode
246        @Override
247        public String[] getContentsEncoded()
248        {
249                final List<Item> contents=getContents();
250                final String[] encodedDeck=new String[contents.size()];
251                for(int i=0;i<contents.size();i++)
252                {
253                        final PlayingCard card=(PlayingCard)contents.get(i);
254                        encodedDeck[i]=card.getStringEncodedSuit()+card.getStringEncodedValue();
255                }
256                return encodedDeck;
257        }

259        // sortByValueAceHigh()
260        // This method is a sort of anti-shuffle.  It puts the cards in
261        // order, first by value, then by suit, with ace considered high.
262        @Override
263        public void sortByValueAceHigh()
264        {
265                final List<Item> unsorted=new Vector<Item>();
266                unsorted.addAll(getContents());
267                if(unsorted.size()==0)
268                        return;
269                // first step is to get them out of the deck and
270                // then re-add them in order.
271                for(int u=0;u<unsorted.size();u++)
272                        removeCard((PlayingCard)unsorted.get(u));
273
274                // now we pick them out in order
275                while(unsorted.size()>0)
276                {
277                        PlayingCard card=(PlayingCard)unsorted.get(0);
278                        int cardBitEncodedValue=card.getBitEncodedValue();
279                        for(int u=1;u<unsorted.size();u++)
280                        {
281                                final PlayingCard card2=(PlayingCard)unsorted.get(u);
282                                final int card2BitEncodedValue=card2.getBitEncodedValue();
283                                if((card2BitEncodedValue<cardBitEncodedValue)
284                                ||((card2BitEncodedValue==cardBitEncodedValue)
285                                        &&(card2.getBitEncodedSuit()<card.getBitEncodedSuit())))
286                                {
287                                        card=card2;
288                                        cardBitEncodedValue=card.getBitEncodedValue();
289                                }
290                        }
291                        unsorted.remove(card);
292                        addCard(card);
293                }
294        }

296        // sortByValueAceLow()
297        // This method is a sort of anti-shuffle.  It puts the cards in
298        // order, first by value, then by suit, with ace low.
299        @Override
300        public void sortByValueAceLow()
301        {
302                final List<Item> unsorted=new Vector<Item>();
303                unsorted.addAll(getContents());
304                if(unsorted.size()==0)
305                        return;
306                // first step is to get them out of the deck and
307                // then re-add them in order.
308                for(int u=0;u<unsorted.size();u++)
309                        removeCard((PlayingCard)unsorted.get(u));
310
311                // now we pick them out in order
312                while(unsorted.size()>0)
313                {
314                        PlayingCard card=(PlayingCard)unsorted.get(0);
315                        int cardBitEncodedValue=card.getBitEncodedValue();
316                        if(cardBitEncodedValue==14)
317                                cardBitEncodedValue=1;
318                        for(int u=1;u<unsorted.size();u++)
319                        {
320                                final PlayingCard card2=(PlayingCard)unsorted.get(u);
321                                int card2BitEncodedValue=card2.getBitEncodedValue();
322                                if(card2BitEncodedValue==14)
323                                        card2BitEncodedValue=1;
324                                if((card2BitEncodedValue<cardBitEncodedValue)
325                                ||((card2BitEncodedValue==cardBitEncodedValue)
326                                        &&(card2.getBitEncodedSuit()<card.getBitEncodedSuit())))
327                                {
328                                        card=card2;
329                                        cardBitEncodedValue=card.getBitEncodedValue();
330                                        if(cardBitEncodedValue==14)
331                                                cardBitEncodedValue=1;
332                                }
333                        }
334                        unsorted.remove(card);
335                        addCard(card);
336                }
337        }

339        // createEmptyHand(Environmental player)
340        // creates an empty HandOfCards object
341        // if the player passed in is not null, it will
342        // add the new hand to the inventory of the given
343        // hand-holder.  Either way, it will return the
344        // empty hand object.
345        @Override
346        public HandOfCards createEmptyHand(Environmental player)
347        {
348                // calling this method without the intention
349                // of putting a card inside is counter-productive.
350                // the other methods should automatically create and
351                // destroy the hands as cards are dealt and returned
352                // to the deck respectively!
353                final HandOfCards hand=(HandOfCards)CMClass.getMiscMagic("StdHandOfCards");
354                if(player instanceof MOB)
355                {
356                        if(hand.owner()==null)
357                                ((MOB)player).addItem(hand);
358                        else
359                                ((MOB)player).moveItemTo(hand);
360                        if(((MOB)player).isMine(hand))
361                                return hand;
362                        return null;
363                }
364                else
365                if(player instanceof Room)
366                {
367                        if(hand.owner()==null)
368                                ((Room)player).addItem(hand);
369                        else
370                                ((Room)player).moveItemTo(hand);
371                        if(((Room)player).isContent(hand))
372                                return hand;
373                        return null;
374                }
375                return hand;
376        }

379        // returns whether this hand contains a card of

381        // a string code is a single letter suit followed

390        // getCard(String cardStringCode)
391        // returns the PlayingCard from this deck or hand if
392        // it is to be found herein.  DOES NOT REMOVE!
393        // removeCard should be called next to do that.
394        // a string code is a single letter suit followed
395        // by a single letter for face cards and the ace,
396        // or a number for other cards.
397        @Override
398        public PlayingCard getCard(String cardStringCode)
399        {
400                final List<Item> handContents=getContents();
401                if(handContents.size()==0)
402                        return null;
403                if(cardStringCode.length()<2)
404                        return null;
405                for(int i=0;i<handContents.size();i++)
406                {
407                        final PlayingCard card=(PlayingCard)handContents.get(i);
408                        if((card.getStringEncodedSuit().charAt(0)==cardStringCode.charAt(0))
409                        &&card.getStringEncodedValue().equalsIgnoreCase(cardStringCode.substring(1)))
410                                return card;
411                }
412                return null;
413        }

493        // getCard(int cardBitCode)
494        // returns the PlayingCard from this deck or hand if
495        // it is to be found herein.  DOES NOT REMOVE!
496        // removeCard should be called next to do that.
497        // a bit code is as described in PlayingCard.java
498        @Override
499        public PlayingCard getCard(int cardBitCode)
500        {
501                if((cardBitCode&(1+2+4+8))==1)
502                        cardBitCode=(cardBitCode&(16+32))+14;
503                final List<Item> handContents=getContents();
504                if(handContents.size()==0)
505                        return null;
506                for(int i=0;i<handContents.size();i++)
507                {
508                        final PlayingCard card=(PlayingCard)handContents.get(i);
509                        if(card.phyStats().ability()==cardBitCode)
510                                return card;
511                }
512                return null;
513        }

591        // this method is a general event handler
592        // to make playing stud-games easier on the players,
593        // we capture the LOOK event in this method and
594        // then tell the looker what the other player
595        // has  "turned up"
596        @Override
597        public void executeMsg(final Environmental host, final CMMsg msg)
598        {
599                if(owner() instanceof MOB)
600                {
601                        // check to see if this event is a LOOK or EXAMINE event
602                        // and see of the target of the action is the owner of
603                        // this hand of cards, and also check quickly whether
604                        // the player HAS any cards.
605                        if(((msg.targetMinor()==CMMsg.TYP_LOOK)||(msg.targetMinor()==CMMsg.TYP_EXAMINE))
606                        &&(msg.target()==owner())
607                        &&(numberOfCards()>0))
608                        {
609                                // now we build a string list of the cards which
610                                // are marked as FACE-UP
611                                final StringBuffer str=new StringBuffer("");
612                                final List<Item> cards=getContents();
613                                for(int c=0;c<cards.size();c++)
614                                {
615                                        final PlayingCard card=(PlayingCard)cards.get(c);
616                                        if(card.isFaceUp())
617                                                str.append(card.name()+", ");
618                                }
619
620                                // now we have that list, so we generate a new OK (always occurs)
621                                // message to notify the looker of what our player has.  We add our
622                                // new message as a "trailer" to ensure it happens after the
623                                // primary event has taken place.  Remember, at this point, we
624                                // are capturing an event in mid-stride.  This event we have
625                                // captured has not completed executing yet.
626                                if(str.length()>0)
627                                        msg.addTrailerMsg(CMClass.getMsg(msg.source(),owner(),null,CMMsg.MSG_OK_VISUAL,L("<T-NAME> <T-HAS-HAVE> the following cards shown: @x1.",str.toString().substring(0,str.length()-2)),CMMsg.NO_EFFECT,null,CMMsg.NO_EFFECT,null));
628                        }
629                }
630                super.executeMsg(host,msg);
631        }
