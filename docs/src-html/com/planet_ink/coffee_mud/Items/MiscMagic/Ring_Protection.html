036public class Ring_Protection extends Ring_Ornamental implements MiscMagic
037{
038        @Override
039        public String ID()
040        {
041                return "Ring_Protection";
042        }
043
044        public Ring_Protection()
045        {
046                super();
047
048                lastLevel = -1;
049        }
050
051        @Override
052        public void affectPhyStats(final Physical affected, final PhyStats affectableStats)
053        {
054                super.affectPhyStats(affected,affectableStats);
055                if((amBeingWornProperly())
056                &&(!this.amWearingAt(Wearable.WORN_HELD)))
057                        affectableStats.setArmor(affectableStats.armor()-phyStats().armor()-phyStats().ability());
058        }
059
060        @Override
061        public void recoverPhyStats()
062        {
063                basePhyStats().setDisposition(basePhyStats().disposition()|PhyStats.IS_BONUS);
064                super.recoverPhyStats();
065                if(lastLevel!=basePhyStats().level())
066                {
067                        lastLevel = basePhyStats().level();
068                        setIdentity();
069                }
070        }
071
072        protected int correctTargetMinor()
073        {
074                switch(this.phyStats().level())
075                {
076                        case SILVER_RING:
077                                return CMMsg.TYP_COLD;
078                        case COPPER_RING:
079                                return CMMsg.TYP_ELECTRIC;
080                        case PLATINUM_RING:
081                                return CMMsg.TYP_GAS;
082                        case GOLD_RING_DIAMOND:
083                                return CMMsg.TYP_FIRE;
084                        case GOLD_RING:
085                                return CMMsg.TYP_ACID;
086                        case GOLD_RING_RUBY:
087                                return CMMsg.TYP_MIND;
088                        case BRONZE_RING:
089                                return CMMsg.TYP_PARALYZE;
090                        case GOLD_RING_OPAL:
091                                return CMMsg.TYP_CAST_SPELL;
092                        case GOLD_RING_TOPAZ:
093                                return -99;
094                        case GOLD_RING_SAPPHIRE:
095                                return CMMsg.TYP_JUSTICE;
096                        case MITHRIL_RING:
097                                return CMMsg.TYP_WEAPONATTACK;
098                        case GOLD_RING_PEARL:
099                                return CMMsg.TYP_WATER;
100                        case GOLD_RING_EMERALD:
101                                return -99;
102                        default:
103                                return -99;
104
105                }
106        }
107
108        protected boolean rollChance()
109        {
110                switch(this.phyStats().level())
111                {
112                        case GOLD_RING_OPAL:
113                                if(Math.random()>.15)
114                                        return false;
115                                return true;
116                        case GOLD_RING_SAPPHIRE:
117                                if(Math.random()>.15)
118                                        return false;
119                                return true;
120                        case MITHRIL_RING:
121                                if(Math.random()>.05)
122                                        return false;
123                                return true;
124                        default:
125                                return true;
126
127                }
128        }
129
130        private void setIdentity()
131        {
132                switch(this.phyStats().level())
133                {
134                        case SILVER_RING:
135                                secretIdentity="The ring of Seven Winters. (Protection from Cold)";
136                                baseGoldValue+=1000;
137                                material=RawMaterial.RESOURCE_SILVER;
138                                break;
139                        case COPPER_RING:
140                                secretIdentity="The ring of Storms. (Protection from Electricity)";
141                                baseGoldValue+=1000;
142                                material=RawMaterial.RESOURCE_COPPER;
143                                break;
144                        case PLATINUM_RING:
145                                secretIdentity="The ring of Sweet Air. (Protection from Gas)";
146                                baseGoldValue+=1000;
147                                material=RawMaterial.RESOURCE_PLATINUM;
148                                break;
149                        case GOLD_RING_DIAMOND:
150                                secretIdentity="The ring of the Eternal Blaze. (Protection from Fire)";
151                                baseGoldValue+=1000;
152                                material=RawMaterial.RESOURCE_GOLD;
153                                break;
154                        case BRONZE_RING:
155                                secretIdentity="The ring of the Bronze Shield. (Protection from Paralysis)";
156                                baseGoldValue+=1000;
157                                material=RawMaterial.RESOURCE_BRONZE;
158                                break;
159                        case GOLD_RING:
160                                secretIdentity="The ring of Sweet Water. (Protection from Acid)";
161                                baseGoldValue+=1000;
162                                material=RawMaterial.RESOURCE_GOLD;
163                                break;
164                        case GOLD_RING_RUBY:
165                                secretIdentity="The ring of Focus. (Protection from Mind Attacks)";
166                                baseGoldValue+=5000;
167                                material=RawMaterial.RESOURCE_GEM;
168                                break;
169                        case GOLD_RING_OPAL:
170                                secretIdentity="Mages Bane. (15% Resistance to Magic)";
171                                baseGoldValue+=1000;
172                                material=RawMaterial.RESOURCE_GEM;
173                                break;
174                        case GOLD_RING_TOPAZ:
175                                basePhyStats().setAbility(60);
176                                secretIdentity="Zimmers Guard. (Ring of Protection +60)";
177                                baseGoldValue+=6000;
178                                material=RawMaterial.RESOURCE_GEM;
179                                break;
180                        case GOLD_RING_SAPPHIRE:
181                                secretIdentity="The ring of Justice. (15% Resistance to Criminal Behavior)";
182                                baseGoldValue+=1000;
183                                material=RawMaterial.RESOURCE_GEM;
184                                break;
185                        case MITHRIL_RING:
186                                secretIdentity="The ring of Fortitude. (5% Resistance to physical attacks)";
187                                baseGoldValue+=2000;
188                                material=RawMaterial.RESOURCE_MITHRIL;
189                                break;
190                        case GOLD_RING_PEARL:
191                                secretIdentity="The ring of the Wave. (Protection from Water Attacks)";
192                                baseGoldValue+=1000;
193                                material=RawMaterial.RESOURCE_PEARL;
194                                break;
195                        case GOLD_RING_EMERALD:
196                                if(basePhyStats().ability()==0)
197                                        basePhyStats().setAbility(50);
198                                secretIdentity="Fox Guard. (Ring of Protection +50)";
199                                baseGoldValue+=5000;
200                                material=RawMaterial.RESOURCE_GEM;
201                                break;
202                        default:
203                                final double pct=Math.random();
204                                if(basePhyStats().ability()==0)
205                                        basePhyStats().setAbility((int)Math.round(pct*49));
206                                baseGoldValue+=basePhyStats().ability()*100;
207                                secretIdentity="A ring of protection + "+basePhyStats().ability()+".";
208                                material=RawMaterial.RESOURCE_STEEL;
209                                break;
210                }
211        }
212
213        @Override
214        public void executeMsg(final Environmental myHost, final CMMsg msg)
215        {
216                super.executeMsg(myHost,msg);
217                if((msg.target()==null)||(!(msg.target() instanceof MOB)))
218                        return ;
219
220                final MOB mob=(MOB)msg.target();
221                if(mob!=this.owner())
222                        return;
223
224                if((msg.targetMinor()==correctTargetMinor())
225                &&(CMath.bset(msg.targetMajor(),CMMsg.MASK_MALICIOUS))
226                &&(!this.amWearingAt(Wearable.IN_INVENTORY))
227                &&(mob.isMine(this))
228                &&(rollChance()))
229                        CMLib.combat().resistanceMsgs(msg.source(),mob,msg);
230        }
231}
