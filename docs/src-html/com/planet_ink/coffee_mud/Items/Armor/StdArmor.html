101        {
102                if(where==0)
103                        return (whereCantWear(mob)==0);
104                if((rawProperLocationBitmap()&where)!=where)
105                        return false;
106                return mob.freeWearPositions(where,getClothingLayer(),getLayerAttributes())>0;

109        protected String armorHealth()
110        {
111                final int[] condSet = new int[]{95, 75, 50, 25, 10, 5, 0};
112                int ordinal=0;
113                for(int i=0;i<condSet.length;i++)
114                {
115                        if(usesRemaining()>=condSet[i])
116                        {
117                                ordinal=i;
118                                break;
119                        }
120                }
121                final String message;
122                switch(material()&RawMaterial.MATERIAL_MASK)
123                {
124                case RawMaterial.MATERIAL_PAPER:
125                case RawMaterial.MATERIAL_CLOTH:
126                        message=CMProps.getListFileChoiceFromIndexedList(CMProps.ListFile.ARMOR_CONDITION_CLOTH, ordinal);
127                        break;
128                case RawMaterial.MATERIAL_ROCK:
129                case RawMaterial.MATERIAL_PRECIOUS:
130                case RawMaterial.MATERIAL_SYNTHETIC:
131                case RawMaterial.MATERIAL_GLASS:
132                        message=CMProps.getListFileChoiceFromIndexedList(CMProps.ListFile.ARMOR_CONDITION_GLASS, ordinal);
133                        break;
134                case RawMaterial.MATERIAL_FLESH:
135                case RawMaterial.MATERIAL_LEATHER:
136                        message=CMProps.getListFileChoiceFromIndexedList(CMProps.ListFile.ARMOR_CONDITION_LEATHER, ordinal);
137                        break;
138                case RawMaterial.MATERIAL_METAL:
139                case RawMaterial.MATERIAL_MITHRIL:
140                        message=CMProps.getListFileChoiceFromIndexedList(CMProps.ListFile.ARMOR_CONDITION_METAL, ordinal);
141                        break;
142                case RawMaterial.MATERIAL_WOODEN:
143                        message=CMProps.getListFileChoiceFromIndexedList(CMProps.ListFile.ARMOR_CONDITION_WOODEN, ordinal);
144                        break;
145                default:
146                        message=CMProps.getListFileChoiceFromIndexedList(CMProps.ListFile.ARMOR_CONDITION_OTHER, ordinal);
147                        break;
148                }
149                return L(message,name(),""+usesRemaining());
150        }

167        @Override
168        public SizeDeviation getSizingDeviation(final MOB mob)
169        {
170                if(mob == null)
171                        return SizeDeviation.FITS;
172
173                if((phyStats().height()>0)
174                &&(mob.phyStats().height()>0))
175                {
176                        int devianceAllowed=200;
177                        if((rawProperLocationBitmap()&SMALL_DEVIATION_WEAR_LOCS)>0)
178                                devianceAllowed=20;
179                        else
180                        if((rawProperLocationBitmap()&STRANGE_DEVIATION_WEAR_LOCS)>0)
181                        {
182                                long wcode=rawProperLocationBitmap();
183
184                                if(CMath.bset(wcode,Wearable.WORN_HELD))
185                                        wcode=wcode-Wearable.WORN_HELD;
186                                if(wcode==Wearable.WORN_FLOATING_NEARBY)
187                                        devianceAllowed=-1;
188                                else
189                                if(wcode==Wearable.WORN_MOUTH)
190                                        devianceAllowed=-1;
191                                else
192                                if(wcode==Wearable.WORN_EYES)
193                                        devianceAllowed=1000;
194                                else
195                                if(wcode==Wearable.WORN_EARS)
196                                        devianceAllowed=1000;
197                                else
198                                if(wcode==Wearable.WORN_NECK)
199                                        devianceAllowed=5000;
200                        }
201                        if((devianceAllowed>0)&&(!CMSecurity.isDisabled(CMSecurity.DisFlag.EQUIPSIZE)))
202                        {
203                                if(mob.phyStats().height()<(phyStats().height()-devianceAllowed))
204                                        return SizeDeviation.TOO_LARGE;
205                                if(mob.phyStats().height()>(phyStats().height()+devianceAllowed))
206                                        return SizeDeviation.TOO_SMALL;
207                        }
208                }
209                return SizeDeviation.FITS;
210        }

212        @Override
213        public boolean okMessage(final Environmental myHost, final CMMsg msg)
214        {
215                if(!super.okMessage(myHost,msg))
216                        return false;
217                if((msg.amITarget(this))
218                &&(msg.targetMinor()==CMMsg.TYP_WEAR))
219                {
220                        final Armor.SizeDeviation deviation = getSizingDeviation(msg.source());
221                        if(deviation != Armor.SizeDeviation.FITS)
222                        {
223                                if(deviation == Armor.SizeDeviation.TOO_LARGE)
224                                {
225                                        msg.source().tell(L("@x1 doesn't fit you -- it's too big.",name()));
226                                        return false;
227                                }
228                                else
229                                if(deviation == Armor.SizeDeviation.TOO_SMALL)
230                                {
231                                        msg.source().tell(L("@x1 doesn't fit you -- it's too small.",name()));
232                                        return false;
233                                }
234                        }
235                }
236                return true;
237        }

581                        &&(owner() instanceof MOB))
582                        {
583                                final MOB owner=(MOB)owner();
584                                setUsesRemaining(100);
585                                msg.addTrailerMsg(CMClass.getMsg(((MOB)owner()),null,null,CMMsg.MSG_OK_VISUAL,L("^I@x1 is destroyed!!^?",name()),CMMsg.NO_EFFECT,null,CMMsg.MSG_OK_VISUAL,L("^I@x1 being worn by <S-NAME> is destroyed!^?",name())));
586                                unWear();
587                                destroy();
588                                owner.recoverPhyStats();
589                                owner.recoverCharStats();
590                                owner.recoverMaxState();
591                                if(owner.location()!=null)
592                                        owner.location().recoverRoomStats();
593                        }
594                }
595        }

607        @Override
608        public void affectPhyStats(final Physical affected, final PhyStats affectableStats)
609        {
610                super.affectPhyStats(affected,affectableStats);
611
612                if((!amWearingAt(Wearable.IN_INVENTORY))
613                &&((!amWearingAt(Wearable.WORN_FLOATING_NEARBY))||(fitsOn(Wearable.WORN_FLOATING_NEARBY)))
614                &&((!amWearingAt(Wearable.WORN_HELD))||(this instanceof Shield)))
615                {
616                        affectableStats.setArmor(affectableStats.armor()-phyStats().armor());
617                        if(phyStats().armor()!=0)
618                        {
619                                final int ability=super.wornLogicalAnd ? (phyStats().ability()*CMath.numberOfSetBits(super.myWornCode)) : phyStats().ability();
620                                if(amWearingAt(Wearable.WORN_TORSO))
621                                        affectableStats.setArmor(affectableStats.armor()-(ability*5));
622                                else
623                                if((amWearingAt(Wearable.WORN_HEAD))||(this.amWearingAt(Wearable.WORN_HELD)))
624                                        affectableStats.setArmor(affectableStats.armor()-(ability*2));
625                                else
626                                if(!amWearingAt(Wearable.WORN_FLOATING_NEARBY))
627                                        affectableStats.setArmor(affectableStats.armor()-ability);
628                        }
629                }
630        }

632        @Override
633        public void affectCharStats(final MOB affected, final CharStats affectableStats)
634        {
635                super.affectCharStats(affected,affectableStats);
636                if(!amWearingAt(Wearable.IN_INVENTORY))
637                switch(material()&RawMaterial.MATERIAL_MASK)
638                {
639                case RawMaterial.MATERIAL_METAL:
640                        if(affectableStats.getStat(CharStats.STAT_SAVE_ELECTRIC)>(Short.MIN_VALUE/2))
641                                affectableStats.setStat(CharStats.STAT_SAVE_ELECTRIC,affectableStats.getStat(CharStats.STAT_SAVE_ELECTRIC)-10);
642                        break;
643                case RawMaterial.MATERIAL_LEATHER:
644                        if(affectableStats.getStat(CharStats.STAT_SAVE_ACID)<(Short.MAX_VALUE/2))
645                                affectableStats.setStat(CharStats.STAT_SAVE_ACID,affectableStats.getStat(CharStats.STAT_SAVE_ACID)+10);
646                        break;
647                case RawMaterial.MATERIAL_MITHRIL:
648                        if(affectableStats.getStat(CharStats.STAT_SAVE_MAGIC)<(Short.MAX_VALUE/2))
649                                affectableStats.setStat(CharStats.STAT_SAVE_MAGIC,affectableStats.getStat(CharStats.STAT_SAVE_MAGIC)+10);
650                        break;
651                case RawMaterial.MATERIAL_CLOTH:
652                case RawMaterial.MATERIAL_PAPER:
653                        if(affectableStats.getStat(CharStats.STAT_SAVE_FIRE)>(Short.MIN_VALUE/2))
654                                affectableStats.setStat(CharStats.STAT_SAVE_FIRE,affectableStats.getStat(CharStats.STAT_SAVE_FIRE)-10);
655                        break;
656                case RawMaterial.MATERIAL_GLASS:
657                case RawMaterial.MATERIAL_ROCK:
658                case RawMaterial.MATERIAL_PRECIOUS:
659                case RawMaterial.MATERIAL_VEGETATION:
660                case RawMaterial.MATERIAL_FLESH:
661                case RawMaterial.MATERIAL_SYNTHETIC:
662                        if(affectableStats.getStat(CharStats.STAT_SAVE_FIRE)<(Short.MAX_VALUE/2))
663                                affectableStats.setStat(CharStats.STAT_SAVE_FIRE,affectableStats.getStat(CharStats.STAT_SAVE_FIRE)+10);
664                        break;
665                case RawMaterial.MATERIAL_ENERGY:
666                        if(affectableStats.getStat(CharStats.STAT_SAVE_PARALYSIS)<(Short.MAX_VALUE/2))
667                                affectableStats.setStat(CharStats.STAT_SAVE_PARALYSIS,affectableStats.getStat(CharStats.STAT_SAVE_PARALYSIS)+10);
668                        break;
669                }
670        }

688        @Override
689        public String secretIdentity()
690        {
691                String id=super.secretIdentity();
692                if(phyStats().ability()>0)
693                        id=name()+" +"+phyStats().ability()+((id.length()>0)?"\n\r":"")+id;
694                else
695                if(phyStats().ability()<0)
696                        id=name()+" "+phyStats().ability()+((id.length()>0)?"\n\r":"")+id;
697                final int timsLevel=CMLib.itemBuilder().timsLevelCalculator(this);
698                if(timsLevel != phyStats.level())
699                        id += " (Power level: "+timsLevel+")";
700                return id+"\n\r"+L("Base Protection: @x1",""+phyStats().armor());
701        }
