202        public static boolean reportError(final ShipEngine me, final Software controlI, final MOB mob, final String literalMessage, final String controlMessage)
203        {
204                if((mob!=null) && (mob.location()==CMLib.map().roomLocation(me)) && (literalMessage!=null))
205                        mob.tell(literalMessage);
206                if(controlMessage!=null)
207                {
208                        if(controlI!=null)
209                                controlI.addScreenMessage(controlMessage);
210                        else
211                        if((mob!=null)&&(me!=null))
212                                mob.tell(CMLib.lang().L("A panel on @x1 reports '@x2'.",me.name(mob),controlMessage));
213                }
214                return false;
215        }

217        public static boolean tellWholeShip(final ShipEngine me, final MOB mob, final int msgCode, final String message)
218        {
219                Room R=CMLib.map().roomLocation(me);
220                if(R==null)
221                        R=mob.location();
222                if(R!=null)
223                {
224                        if(R.getArea() instanceof SpaceShip)
225                        {
226                                for(final Enumeration<Room> r=R.getArea().getProperMap();r.hasMoreElements();)
227                                        r.nextElement().show(mob, null, msgCode, message);
228                        }
229                        else
230                                R.show(mob, null, msgCode, message);
231                }
232                return false;
233        }

235        protected static void sendComputerMessage(final ShipEngine me, final String circuitKey, final MOB mob, final Item controlI, final String code)
236        {
237                for(final Iterator<Computer> c=CMLib.tech().getComputers(circuitKey);c.hasNext();)
238                {
239                        final Computer C=c.next();
240                        if((controlI==null)||(C!=controlI.owner()))
241                        {
242                                final CMMsg msg2=CMClass.getMsg(mob, C, me, CMMsg.NO_EFFECT, null, CMMsg.MSG_ACTIVATE|CMMsg.MASK_CNTRLMSG, code, CMMsg.NO_EFFECT,null);
243                                if(C.okMessage(mob, msg2))
244                                        C.executeMsg(mob, msg2);
245                        }
246                }
247        }

249        private static int getFuelToConsume(final ShipEngine me, final Manufacturer manufacturer, final ShipDir portDir, double thrust)
250        {
251                if((portDir != ShipDir.AFT)&&(portDir != ShipDir.FORWARD))
252                        thrust = 1.0;
253                final int fuel=(int)Math.round(
254                        CMath.ceiling(
255                                thrust
256                                *me.getFuelEfficiency()
257                                *Math.max(.33, Math.abs(2.0-manufacturer.getEfficiencyPct()))
258                                / getFuelDivisor()
259                        )
260                );
261                if((thrust>0)&&(fuel<=0))
262                        return 1;
263                return fuel;
264        }

266        public static boolean executeThrust(final ShipEngine me, final String circuitKey, final MOB mob, final Software controlI, final TechComponent.ShipDir portDir, final double amount)
267        {
268                final LanguageLibrary lang=CMLib.lang();
269                final SpaceObject obj=CMLib.map().getSpaceObject(me, true);
270                final Manufacturer manufacturer=me.getFinalManufacturer();
271                final String rumbleWord = (me instanceof FuelConsumer) ? "rumble" : "hum";
272                if(!(obj instanceof SpaceShip))
273                        return reportError(me, controlI, mob, lang.L("@x1 "+rumbleWord+"s and fires, but nothing happens.",me.name(mob)), lang.L("Failure: @x1: exhaust ports.",me.name(mob)));
274                final SpaceShip ship=(SpaceShip)obj;
275                if((portDir==null)||(amount<0))
276                        return reportError(me, controlI, mob, lang.L("@x1 "+rumbleWord+"s loudly, but accomplishes nothing.",me.name(mob)), lang.L("Failure: @x1: exhaust control.",me.name(mob)));
277                if(!CMParms.contains(me.getAvailPorts(), portDir))
278                        return reportError(me, controlI, mob, lang.L("@x1 "+rumbleWord+"s a little, but accomplishes nothing.",me.name(mob)), lang.L("Failure: @x1: port control.",me.name(mob)));
279                double thrust=me.getInstalledFactor() * amount;
280                if(thrust > me.getMaxThrust())
281                        thrust=me.getMaxThrust();
282                if(me.subjectToWearAndTear())
283                {
284                        if(me.usesRemaining()<75)
285                        {
286                                final double pct = CMath.mul(me.usesRemaining(), 100.0) + (0.35 * manufacturer.getReliabilityPct());
287                                if(pct < 1.0)
288                                        thrust = thrust * pct;
289                        }
290                }
291                else
292                        thrust=manufacturer.getReliabilityPct() * thrust;
293                if(portDir==TechComponent.ShipDir.AFT) // when thrusting aft, the thrust is continual, so save it
294                {
295                        if(amount == 0.0)
296                        {
297                                if(me.getThrust()>0.0)
298                                {
299                                        me.setThrust(0.0);
300                                        //return reportError(me, controlI, mob, lang.L("@x1 goes quiet.",me.name(mob)), lang.L("Info: @x1: Engine shut down.",me.name(mob)));
301                                }
302                                me.setThrust(0.0);
303                                return false;
304                        }
305                        if(me.getThrust()==0.0)
306                        {
307                                me.setThrust(amount); // also, its always the intended amount, not the adjusted amount
308                                //return reportError(me, controlI, mob, lang.L("@x1 roars to life.",me.name(mob)), lang.L("Info: @x1: Engine activated.",me.name(mob)));
309                        }
310                        me.setThrust(amount); // also, its always the intended amount, not the adjusted amount
311                }
312
313                final int fuelToConsume=getFuelToConsume(me, manufacturer, portDir, thrust);
314
315                final double accelleration;
316                if(portDir==TechComponent.ShipDir.AFT) // when thrusting aft, there's a smidgeon more power
317                {
318                        accelleration = (thrust * me.getSpecificImpulse() / ship.getMass());
319                        if(accelleration < me.getMinThrust())
320                                return reportError(me, controlI, mob, lang.L("@x1 "+rumbleWord+"s loudly, but nothing happens.",me.name(mob)), lang.L("Failure: @x1: insufficient thrust.",me.name(mob)));
321                }
322                else
323                        accelleration = thrust;
324
325                //if((amount > 1)&&((portDir!=TechComponent.ShipDir.AFT) || (me.getThrust() > (oldThrust * 10))))
326                //      tellWholeShip(me,mob,CMMsg.MSG_NOISE,CMLib.lang().L("You feel a "+rumbleWord+" and hear the blast of @x1.",me.name(mob)));
327                if(accelleration == 0.0)
328                {
329                        final String code=Technical.TechCommand.COMPONENTFAILURE.makeCommand(TechType.SHIP_ENGINE, "Failure: "+me.name()+": insufficient_thrust_capacity.");
330                        sendComputerMessage(me,circuitKey,mob,controlI,code);
331                        return reportError(me, controlI, mob, lang.L("@x1 "+rumbleWord+"s very loudly, but nothing is happening.",me.name(mob)), lang.L("Failure: @x1: insufficient engine thrust capacity.",me.name(mob)));
332                }
333                else
334                if(me.consumeFuel(fuelToConsume))
335                {
336                        final SpaceObject spaceObject=ship.getShipSpaceObject();
337                        final String code=Technical.TechCommand.ACCELLLERATION.makeCommand(portDir.opposite(),Double.valueOf(accelleration),Boolean.valueOf(me.isConstantThruster()));
338                        final CMMsg msg=CMClass.getMsg(mob, spaceObject, me, CMMsg.NO_EFFECT, null, CMMsg.MSG_ACTIVATE|CMMsg.MASK_CNTRLMSG, code, CMMsg.NO_EFFECT,null);
339                        if(spaceObject.okMessage(mob, msg))
340                        {
341                                spaceObject.executeMsg(mob, msg);
342                                return true;
343                        }
344                }
345                else
346                {
347                        final String code=Technical.TechCommand.COMPONENTFAILURE.makeCommand(TechType.SHIP_ENGINE, "Failure:_"+me.name().replace(' ','_')+":_insufficient_fuel.");
348                        sendComputerMessage(me,circuitKey,mob,controlI,code);
349                        return reportError(me, controlI, mob, lang.L("@x1 "+rumbleWord+"s loudly, then sputters down.",me.name(mob)), lang.L("Failure: @x1: insufficient fuel.",me.name(mob)));
350                }
351                return false;
352        }

354        public static boolean executeCommand(final ShipEngine me, final String circuitKey, final CMMsg msg)
355        {
356                final LanguageLibrary lang=CMLib.lang();
357                final Software controlI=(msg.tool() instanceof Software)?((Software)msg.tool()):null;
358                final MOB mob=msg.source();
359                if(msg.targetMessage()==null)
360                {
361                        me.setThrust(0);
362                        return true;
363                }
364                else
365                {
366                        final String[] parts=msg.targetMessage().split(" ");
367                        final TechCommand command=TechCommand.findCommand(parts);
368                        if(command==null)
369                                return reportError(me, controlI, mob, lang.L("@x1 does not respond.",me.name(mob)), lang.L("Failure: @x1: control failure.",me.name(mob)));
370                        final Object[] parms=command.confirmAndTranslate(parts);
371                        if(parms==null)
372                                return reportError(me, controlI, mob, lang.L("@x1 did not respond.",me.name(mob)), lang.L("Failure: @x1: control syntax failure.",me.name(mob)));
373                        if(command == TechCommand.THRUST)
374                                return executeThrust(me, circuitKey, mob, controlI, (TechComponent.ShipDir)parms[0],((Double)parms[1]).doubleValue());
375                        return reportError(me, controlI, mob, lang.L("@x1 refused to respond.",me.name(mob)), lang.L("Failure: @x1: control command failure.",me.name(mob)));
376                }
377        }

379        public static void executeThrusterMsg(final ShipEngine me, final Environmental myHost, final String circuitKey, final CMMsg msg)
380        {
381                if(msg.amITarget(me))
382                {
383                        switch(msg.targetMinor())
384                        {
385                        case CMMsg.TYP_ACTIVATE:
386                                if(executeCommand(me, circuitKey, msg))
387                                        me.activate(true);
388                                break;
389                        case CMMsg.TYP_DEACTIVATE:
390                                if(me.activated())
391                                {
392                                        // when a constant thruster deactivates, all speed stops
393                                        final SpaceObject obj=CMLib.map().getSpaceObject(me, true);
394                                        if(obj instanceof SpaceShip)
395                                        {
396                                                final MOB mob=msg.source();
397                                                final SpaceShip ship=(SpaceShip)obj;
398                                                final SpaceObject spaceObject=ship.getShipSpaceObject();
399                                                final String code=Technical.TechCommand.ACCELLLERATION.makeCommand(TechComponent.ShipDir.AFT,Double.valueOf(0),Boolean.valueOf(true));
400                                                final CMMsg msg2=CMClass.getMsg(mob, spaceObject, me, CMMsg.NO_EFFECT, null, CMMsg.MSG_ACTIVATE|CMMsg.MASK_CNTRLMSG, code, CMMsg.NO_EFFECT,null);
401                                                if(spaceObject.okMessage(mob, msg2))
402                                                        spaceObject.executeMsg(mob, msg2);
403                                        }
404                                }
405                                me.setThrust(0);
406                                me.activate(false);
407                                break;
408                        case CMMsg.TYP_POWERCURRENT:
409                        {
410                                if(me.activated())
411                                {
412                                        if((me.getThrust()>0.0)
413                                        && (CMParms.contains(me.getAvailPorts(),TechComponent.ShipDir.AFT)))
414                                        {
415                                                final Manufacturer manufacturer=me.getFinalManufacturer();
416                                                //TODO: isn't there a method for this fuel thing?
417                                                final int fuelToConsume=(int)Math.round(CMath.ceiling(me.getThrust()*me.getFuelEfficiency()*Math.max(.33, Math.abs(2.0-manufacturer.getEfficiencyPct()))/getFuelDivisor()));
418                                                if(me.consumeFuel(fuelToConsume))
419                                                {
420                                                        final SpaceObject obj=CMLib.map().getSpaceObject(me, true);
421                                                        if(obj instanceof SpaceShip)
422                                                        {
423                                                                final SpaceObject ship=((SpaceShip)obj).getShipSpaceObject();
424                                                                final String code=Technical.TechCommand.THRUST.makeCommand(TechComponent.ShipDir.AFT,Double.valueOf(me.getThrust()));
425                                                                final CMMsg msg2=CMClass.getMsg(msg.source(), me, null, CMMsg.NO_EFFECT, null, CMMsg.MSG_ACTIVATE|CMMsg.MASK_CNTRLMSG, code, CMMsg.NO_EFFECT,null);
426                                                                if(me.owner() instanceof Room)
427                                                                {
428                                                                        if(me.owner().okMessage(msg.source(), msg2))
429                                                                                ((Room)me.owner()).send(msg.source(), msg2);
430                                                                }
431                                                                else
432                                                                if(ship.okMessage(msg.source(), msg2))
433                                                                        ship.executeMsg(msg.source(), msg2);
434                                                        }
435                                                }
436                                        }
437                                        else
438                                        {
439                                                final CMMsg msg2=CMClass.getMsg(msg.source(), me, me, CMMsg.NO_EFFECT, null, CMMsg.MSG_DEACTIVATE|CMMsg.MASK_CNTRLMSG, "", CMMsg.NO_EFFECT,null);
440                                                if(me.owner() instanceof Room)
441                                                {
442                                                        if(((Room)me.owner()).okMessage(msg.source(), msg2))
443                                                                ((Room)me.owner()).send(msg.source(), msg2);
444                                                }
445                                                else
446                                                if(me.okMessage(msg.source(), msg2))
447                                                        me.executeMsg(msg.source(), msg2);
448                                                final String code=Technical.TechCommand.COMPONENTFAILURE.makeCommand(TechType.SHIP_ENGINE, "Failure: "+me.name()+": insufficient_fuel.");
449                                                sendComputerMessage(me,circuitKey,msg.source(),null,code);
450                                        }
451                                }
452                                break;
453                        }
454                        }
455                }
456        }
