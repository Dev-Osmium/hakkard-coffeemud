036public class StdCompPanel extends StdElecCompContainer implements ElecPanel, TechComponent
037{
038        @Override
039        public String ID()
040        {
041                return "StdCompPanel";
042        }
043
044        protected volatile int          powerNeeds      = 0;
045        protected volatile String       circuitKey      = null;
046
047        public StdCompPanel()
048        {
049                super();
050                setName("an engineering panel");
051                setDisplayText("an engineering panel is on the wall");
052                setDescription("Usually seemless with the wall, these panels can be opened to install new equipment.");
053                super.setDoorsNLocks(true, true, true,false, false,false);
054                basePhyStats().setSensesMask(basePhyStats().sensesMask()|PhyStats.SENSE_ALWAYSCOMPRESSED|PhyStats.SENSE_INSIDEACCESSIBLE|PhyStats.SENSE_ITEMNOTGET);
055                this.activated=true;
056                this.openDelayTicks=0;
057                this.recoverPhyStats();
058        }
059
060        @Override
061        public TechType getTechType()
062        {
063                return TechType.SHIP_PANEL;
064        }
065
066        protected TechType      panelType       = TechType.ANY;
067
068        @Override
069        public TechType panelType()
070        {
071                return panelType;
072        }
073
074        @Override
075        public void setPanelType(TechType type)
076        {
077                panelType = type;
078        }
079
080        @Override
081        public int powerNeeds()
082        {
083                return powerNeeds;
084        }
085
086        @Override
087        public boolean canContain(Item I)
088        {
089                if(!super.canContain(I))
090                        return false;
091                if((I instanceof Technical)
092                &&((panelType()==((Technical)I).getTechType()))||(panelType()==Technical.TechType.ANY))
093                        return true;
094                return false;
095        }
096
097        @Override
098        public void destroy()
099        {
100                if((!destroyed)&&(circuitKey!=null))
101                {
102                        CMLib.tech().unregisterElectronics(this,circuitKey);
103                        circuitKey=null;
104                }
105                super.destroy();
106        }
107
108        @Override
109        public void setOwner(ItemPossessor newOwner)
110        {
111                final ItemPossessor prevOwner=super.owner;
112                super.setOwner(newOwner);
113                if(prevOwner != newOwner)
114                {
115                        if(newOwner instanceof Room)
116                                circuitKey=CMLib.tech().registerElectrics(this,circuitKey);
117                        else
118                        {
119                                CMLib.tech().unregisterElectronics(this,circuitKey);
120                                circuitKey=null;
121                        }
122                }
123        }
124
125        @Override 
126        public boolean okMessage(final Environmental myHost, final CMMsg msg)
127        {
128                if(!super.okMessage(myHost,msg))
129                        return false;
130                if(msg.amITarget(this))
131                {
132                        switch(msg.targetMinor())
133                        {
134                        case CMMsg.TYP_PUT:
135                                if(CMSecurity.isAllowed(msg.source(), msg.source().location(), CMSecurity.SecFlag.CMDITEMS))
136                                        break; // let admins go ahead and put stuff
137                                else
138                                if(msg.tool() instanceof TechComponent)
139                                {
140                                        if(msg.value()<=0)
141                                        {
142                                                final Ability installA=msg.source().fetchAbility("AstroEngineering");
143                                                if(installA==null)
144                                                {
145                                                        msg.source().tell(L("You don't know how to install @x1 into @x2.",((TechComponent)msg.tool()).name(msg.source()),name(msg.source())));
146                                                        return false;
147                                                }
148                                                else
149                                                {
150                                                        installA.invoke(msg.source(),new XVector<String>("INSTALL",msg.tool().Name()), (Physical)msg.target(), false, 0);
151                                                        return false;
152                                                }
153                                        }
154                                }
155                                break;
156                        case CMMsg.TYP_INSTALL:
157                                if(msg.value()<=0)
158                                {
159                                        msg.source().tell(L("You failed to install @x1 into @x2.",((TechComponent)msg.tool()).name(msg.source()),name(msg.source())));
160                                        return false;
161                                }
162                                break;
163                        }
164                }
165                return true;
166        }
167
168        @Override
169        public void executeMsg(final Environmental myHost, final CMMsg msg)
170        {
171                if(msg.amITarget(this))
172                {
173                        switch(msg.targetMinor())
174                        {
175                        case CMMsg.TYP_PUT:
176                                if(msg.tool() instanceof TechComponent)
177                                        ((TechComponent)msg.tool()).setInstalledFactor(CMSecurity.isAllowed(msg.source(), msg.source().location(), CMSecurity.SecFlag.CMDITEMS)?1.0f:0.0f);
178                                break;
179                        case CMMsg.TYP_INSTALL:
180                                if((msg.tool() instanceof TechComponent)&&(msg.value()>=0))
181                                {
182                                        if(msg.value()<=0)
183                                                ((TechComponent)msg.tool()).setInstalledFactor((float)1.0);
184                                        else
185                                                ((TechComponent)msg.tool()).setInstalledFactor((float)CMath.div(msg.value(), 100.0));
186                                        CMMsg msg2=(CMMsg)msg.copyOf();
187                                        msg2.setTargetCode(CMMsg.MSG_PUT);
188                                        msg2.setSourceCode(CMMsg.MSG_PUT);
189                                        msg2.setOthersCode(CMMsg.MSG_PUT);
190                                        super.executeMsg(myHost, msg2);
191                                }
192                                break;
193                        case CMMsg.TYP_ACTIVATE:
194                                if((msg.source().location()!=null)&&(!CMath.bset(msg.targetMajor(), CMMsg.MASK_CNTRLMSG)))
195                                        msg.source().location().show(msg.source(), this, CMMsg.MSG_OK_VISUAL, L("<S-NAME> connect(s) <T-NAME>."));
196                                this.activate(true);
197                                return; // don't let comp container do its thing
198                        case CMMsg.TYP_DEACTIVATE:
199                        {
200                                final Room locR=CMLib.map().roomLocation(this);
201                                final MOB M=CMLib.map().getFactoryMOB(locR);
202                                final CMMsg deactivateMsg = CMClass.getMsg(M, null, null, CMMsg.MASK_ALWAYS|CMMsg.MASK_CNTRLMSG|CMMsg.MSG_DEACTIVATE,null);
203                                for(final Item I : this.getDeepContents())
204                                {
205                                        if(I instanceof Electronics)
206                                        {
207                                                deactivateMsg.setTarget(I);
208                                                if(locR.okMessage(M, deactivateMsg))
209                                                        locR.send(M, deactivateMsg);
210                                        }
211                                }
212                                return; // don't let comp container do its thing
213                        }
214                        case CMMsg.TYP_POWERCURRENT:
215                        {
216                                final Room R=CMLib.map().roomLocation(this);
217                                int powerRemaining=msg.value();
218                                final List<Item> contents=getDeepContents();
219                                final CMMsg powerMsg=CMClass.getMsg(msg.source(), CMMsg.MSG_POWERCURRENT, null);
220                                double totalPowerReq=0.0;
221                                for(int i=contents.size()-1;i>=0;i--)
222                                {
223                                        final Item I=contents.get(i);
224                                        if((I instanceof Electronics)&&(!(I instanceof PowerSource))&&(!(I instanceof PowerGenerator)))
225                                                totalPowerReq+=((((Electronics)I).powerNeeds()<=0)?1.0:((Electronics)I).powerNeeds());
226                                }
227                                for(int i=contents.size()-1;i>=0;i--)
228                                {
229                                        final Item I=contents.get(i);
230                                        if((I instanceof Electronics)&&(!(I instanceof PowerSource))&&(!(I instanceof PowerGenerator)))
231                                        {
232                                                int powerToTake=0;
233                                                if(powerRemaining>0)
234                                                {
235                                                        final double pctToTake=CMath.div(((((Electronics)I).powerNeeds()<=0)?1:((Electronics)I).powerNeeds()),totalPowerReq);
236                                                        powerToTake=(int)Math.round(pctToTake * powerRemaining);
237                                                        if(powerToTake<1)
238                                                                powerToTake=1;
239                                                }
240                                                powerMsg.setValue(powerToTake);
241                                                powerMsg.setTarget(I);
242                                                if((R!=null)&&(R.okMessage(powerMsg.source(), powerMsg)))
243                                                        R.send(powerMsg.source(), powerMsg);
244                                                powerRemaining-=(powerMsg.value()<0)?powerToTake:(powerToTake-powerMsg.value());
245                                        }
246                                }
247                                powerNeeds=(int)Math.round(totalPowerReq);
248                                CMClass.returnMsg(powerMsg);
249                                msg.setValue(powerRemaining);
250                                return; // don't let comp container do its thing
251                        }
252                        }
253                        super.executeMsg(myHost, msg);
254                }
255        }
256        
257        @Override
258        public boolean sameAs(final Environmental E)
259        {
260                if(!(E instanceof StdCompPanel))
261                        return false;
262                return super.sameAs(E);
263        }
