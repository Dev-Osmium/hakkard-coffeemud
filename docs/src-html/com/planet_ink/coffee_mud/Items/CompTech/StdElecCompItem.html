036public class StdElecCompItem extends StdElecItem implements TechComponent
037{
038        @Override
039        public String ID()
040        {
041                return "StdElecCompItem";
042        }
043
044        protected float                         maxRechargePer  = 1.0f;
045
046        protected float                         installedFactor = 1.0f;
047        protected volatile String       circuitKey              = null;
048
049        public StdElecCompItem()
050        {
051                super();
052                setName("an electric component");
053                setDisplayText("an electric component sits here.");
054                setDescription("");
055                baseGoldValue=50000;
056                basePhyStats.setWeight(500);
057                setUsesRemaining(100);
058                basePhyStats().setLevel(1);
059                recoverPhyStats();
060                setMaterial(RawMaterial.RESOURCE_STEEL);
061                setRechargeRate(1.0f);
062        }
063
064        @Override
065        public float getInstalledFactor()
066        {
067                return installedFactor;
068        }
069
070        @Override
071        public void setInstalledFactor(final float pct)
072        {
073                installedFactor = pct;
074        }
075
076        @Override
077        public void setRechargeRate(final float pctCapPer)
078        {
079                this.maxRechargePer = pctCapPer;
080        }
081
082        @Override
083        public float getRechargeRate()
084        {
085                return maxRechargePer;
086        }
087
088        @Override
089        public int powerNeeds()
090        {
091                return (int)Math.min((powerCapacity - power), (int)Math.round((double)powerCapacity * getRechargeRate()));
092        }
093
094        @Override
095        protected double getComputedEfficiency()
096        {
097                return super.getComputedEfficiency() * this.getInstalledFactor();
098        }
099
100        @Override
101        public boolean sameAs(final Environmental E)
102        {
103                if(!(E instanceof StdElecCompItem))
104                        return false;
105                return super.sameAs(E);
106        }
107
108        @Override
109        public boolean isInstalled()
110        {
111                if(!CMLib.flags().isGettable(this))
112                        return true;
113                if(this.container() instanceof ElecPanel)
114                        return (!CMLib.flags().isGettable(this.container()));
115                if(this.container() instanceof TechComponent)
116                        return ((TechComponent)this.container()).isInstalled();
117                return false;
118        }
119
120        protected static final boolean isThisPanelActivated(final ElecPanel E)
121        {
122                if (!E.activated())
123                        return false;
124                if (E.container() instanceof ElecPanel)
125                        return isThisPanelActivated((ElecPanel) E.container());
126                return true;
127        }
128
129        public static final boolean isAllWiringHot(final Electronics E)
130        {
131                if (E instanceof ElecPanel)
132                        return isThisPanelActivated((ElecPanel) E);
133                if (E.container() instanceof ElecPanel)
134                        return isThisPanelActivated((ElecPanel) E.container());
135                return true;
136        }
137
138        @Override
139        public void destroy()
140        {
141                if((!destroyed)&&(circuitKey!=null))
142                {
143                        CMLib.tech().unregisterElectronics(this,circuitKey);
144                        circuitKey=null;
145                }
146                super.destroy();
147        }
148
149        @Override
150        public void setOwner(final ItemPossessor newOwner)
151        {
152                final ItemPossessor prevOwner=super.owner;
153                super.setOwner(newOwner);
154                if(prevOwner != newOwner)
155                {
156                        if(newOwner instanceof Room)
157                                circuitKey=CMLib.tech().registerElectrics(this,circuitKey);
158                        else
159                        {
160                                CMLib.tech().unregisterElectronics(this,circuitKey);
161                                circuitKey=null;
162                        }
163                }
164        }
165
166        protected void sendLocalMessage(final CMMsg msg)
167        {
168                if(owner() instanceof Room)
169                {
170                        if(((Room)owner()).okMessage(msg.source(), msg))
171                                ((Room)owner()).send(msg.source(), msg);
172                }
173                else
174                if(okMessage(msg.source(), msg))
175                        executeMsg(msg.source(), msg);
176        }
177
178        @Override
179        public boolean okMessage(final Environmental host, final CMMsg msg)
180        {
181                if(msg.amITarget(this))
182                {
183                        switch(msg.targetMinor())
184                        {
185                        case CMMsg.TYP_ACTIVATE:
186                                if(!isInstalled())
187                                {
188                                        if(!CMath.bset(msg.targetMajor(), CMMsg.MASK_CNTRLMSG))
189                                                msg.source().tell(L("@x1 is not installed or connected.",name()));
190                                        return false;
191                                }
192                                else
193                                if((!isAllWiringHot(this))&&(!(this instanceof ElecPanel)))
194                                {
195                                        if(!CMath.bset(msg.targetMajor(), CMMsg.MASK_CNTRLMSG))
196                                                msg.source().tell(L("The panel containing @x1 is not activated or connected.",name()));
197                                        return false;
198                                }
199                                break;
200                        case CMMsg.TYP_DEACTIVATE:
201                                break;
202                        case CMMsg.TYP_LOOK:
203                                break;
204                        case CMMsg.TYP_POWERCURRENT:
205                                if((!(this instanceof FuelConsumer))
206                                &&(!(this instanceof PowerGenerator))
207                                && activated()
208                                && (powerNeeds()>0)
209                                && (msg.value()>0))
210                                {
211                                        double amtToTake=Math.min((double)powerNeeds(), (double)msg.value());
212                                        msg.setValue(msg.value()-(int)Math.round(amtToTake));
213                                        amtToTake *= getFinalManufacturer().getEfficiencyPct();
214                                        if(subjectToWearAndTear() && (usesRemaining()<=200))
215                                                amtToTake *= CMath.div(usesRemaining(), 100.0);
216                                        setPowerRemaining(Math.min(powerCapacity(), Math.round(amtToTake) + powerRemaining()));
217                                }
218                                break;
219                        }
220                }
221                return super.okMessage(host, msg);
222        }
223
224        @Override
225        public void executeMsg(final Environmental host, final CMMsg msg)
226        {
227                if(msg.amITarget(this))
228                {
229                        switch(msg.targetMinor())
230                        {
231                        case CMMsg.TYP_DROP:
232                                setInstalledFactor((float)CMath.div(msg.value(),100.0));
233                                break;
234                        case CMMsg.TYP_ACTIVATE: // sometimes one triggers many acts in other places, but you don't want many msgs.
235                                if((msg.source().location() != null)&&(!CMath.bset(msg.targetMajor(), CMMsg.MASK_CNTRLMSG)))
236                                        msg.source().location().show(msg.source(), this, CMMsg.MSG_OK_VISUAL, L("<S-NAME> activate(s) <T-NAME>."));
237                                this.activate(true);
238                                break;
239                        case CMMsg.TYP_DEACTIVATE:
240                                if((msg.source().location() != null)&&(!CMath.bset(msg.targetMajor(), CMMsg.MASK_CNTRLMSG)))
241                                        msg.source().location().show(msg.source(), this, CMMsg.MSG_OK_VISUAL, L("<S-NAME> deactivate(s) <T-NAME>."));
242                                this.activate(false);
243                                break;
244                        case CMMsg.TYP_DAMAGE:
245                                if(subjectToWearAndTear() && (usesRemaining()>0))
246                                {
247                                        if(msg.value()>usesRemaining())
248                                        {
249                                                final Room R=CMLib.map().roomLocation(this);
250                                                final CMMsg msg2=CMClass.getMsg(msg.source(), CMMsg.MSG_DEACTIVATE, L("@x1 sparks and fizzes out.",name()));
251                                                if((R!=null)&&(R.okMessage(msg.source(), msg2)))
252                                                        R.send(msg.source(), msg2);
253                                        }
254                                        else
255                                        {
256                                                this.setUsesRemaining(this.usesRemaining()-msg.value());
257                                        }
258                                }
259                                break;
260                        case CMMsg.TYP_LOOK:
261                                super.executeMsg(host, msg);
262                                if(CMLib.flags().canBeSeenBy(this, msg.source()))
263                                        msg.source().tell(L("@x1 is currently @x2",name(),(activated()?"connected.\n\r":"deactivated/disconnected.\n\r")));
264                                return;
265                        case CMMsg.TYP_REPAIR:
266                                if(CMLib.dice().rollPercentage()<msg.value())
267                                {
268                                        setUsesRemaining(usesRemaining()<100?100:usesRemaining());
269                                        msg.source().tell(L("@x1 is now repaired.\n\r",name()));
270                                }
271                                else
272                                {
273                                        final int repairRequired=100-usesRemaining();
274                                        if(repairRequired>0)
275                                        {
276                                                int repairApplied=(int)Math.round(CMath.mul(repairRequired, CMath.div(msg.value(), 100)));
277                                                if(repairApplied < 0)
278                                                        repairApplied=1;
279                                                setUsesRemaining(usesRemaining()+repairApplied);
280                                                msg.source().tell(L("@x1 is now @x2% repaired.\n\r",name(),""+usesRemaining()));
281                                        }
282                                }
283                                break;
284                        case CMMsg.TYP_ENHANCE:
285                                if((CMLib.dice().rollPercentage()<msg.value())&&(CMLib.dice().rollPercentage()<50))
286                                {
287                                        float addAmt=0.01f;
288                                        if(getInstalledFactor() < 1.0)
289                                        {
290                                                addAmt=(float)(CMath.div(100.0, msg.value()) * 0.1);
291                                                if(addAmt < 0.1f)
292                                                        addAmt=0.1f;
293                                        }
294                                        setInstalledFactor(this.getInstalledFactor()+addAmt);
295                                        msg.source().tell(msg.source(),this,null,L("<T-NAME> is now enhanced.\n\r"));
296                                }
297                                else
298                                {
299                                        msg.source().tell(msg.source(),this,null,L("Your attempt to enhance <T-NAME> has failed.\n\r"));
300                                }
301                                break;
302                        }
303                }
304                super.executeMsg(host, msg);
305        }

307        protected static boolean reportError(final Electronics me, final Software controlI, final MOB mob, final String literalMessage, final String controlMessage)
308        {
309                if((mob!=null) && (mob.location()==CMLib.map().roomLocation(me)) && (literalMessage!=null))
310                        mob.tell(literalMessage);
311                if(controlMessage!=null)
312                {
313                        if(controlI!=null)
314                                controlI.addScreenMessage(controlMessage);
315                        else
316                        if((mob!=null)&&(me!=null))
317                                mob.tell(CMLib.lang().L("A panel on @x1 reports '@x2'.",me.name(mob),controlMessage));
318                }
319                return false;
320        }

323        public boolean subjectToWearAndTear()
324        {
325                return true;
326        }
