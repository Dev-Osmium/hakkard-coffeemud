248        @Override

250        {
251                if((software==null)||(System.currentTimeMillis()>nextSoftwareCheck))
252                {
253                        final List<Item> list=getContents();
254                        final LinkedList<Software> softwareList=new LinkedList<Software>();
255                        for(final Item I : list)
256                        {
257                                if(I instanceof Software)
258                                {
259                                        ((Software)I).setCircuitKey(circuitKey);
260                                        softwareList.add((Software)I);
261                                }
262                        }
263                        nextSoftwareCheck=System.currentTimeMillis()+(10*1000);
264                        software=softwareList;

287        @Override
288        public String readableText()
289        {
290                final StringBuilder str=new StringBuilder("");
291                str.append("\n\r");
292                if(!activated())
293                        str.append(L("The screen is blank.  Try activating/booting it first."));
294                else
295                {
296                        final List<Software> software=getSoftware();
297                        synchronized(software)
298                        {
299                                boolean isInternal=false;
300                                for(final Software S : software)
301                                {
302                                        if(S.getInternalName().equals(currentMenu))
303                                        {
304                                                str.append(S.getCurrentScreenDisplay());
305                                                isInternal=true;
306                                        }
307                                        else
308                                        if(S.getParentMenu().equals(currentMenu))
309                                        {
310                                                str.append(S.getActivationMenu()).append("\n\r");
311                                        }
312                                }
313                                if(isInternal)
314                                {
315                                        str.append(L("\n\rEnter \"<\" to return to the previous menu."));
316                                }
317                                else
318                                if(software.size()>0)
319                                {
320                                        str.append(L("\n\rType in a command:"));
321                                }
322                                else
323                                {
324                                        str.append(L("\n\rThis system is ready to receive software."));
325                                }
326                        }
327                }
328
329                return str.toString();
330        }

332        @Override
333        public List<MOB> getCurrentReaders()
334        {
335                final List<MOB> readers=new LinkedList<MOB>();
336                if(amDestroyed())
337                        return readers;
338                final MOB lastReader=this.lastReader;
339                if(lastReader!=null)
340                {
341                        if(((lastReader==owner())||(lastReader.location()==owner()))
342                        &&(CMLib.flags().isInTheGame(lastReader, true)))
343                                readers.add(lastReader);
344                        else
345                                this.lastReader=null;
346                }
347                for(final Rider R : riders)
348                {
349                        if((R instanceof MOB)
350                        &&(!readers.contains(R)))
351                                readers.add((MOB)R);
352                }
353                return readers;
354        }

356        @Override
357        public boolean okMessage(final Environmental host, final CMMsg msg)
358        {
359                if(msg.amITarget(this))
360                {
361                        switch(msg.targetMinor())
362                        {
363                        case CMMsg.TYP_POWERCURRENT:
364                                break;
365                        case CMMsg.TYP_READ:
366                        case CMMsg.TYP_WRITE:
367                                if(!activated())
368                                {
369                                        msg.source().tell(L("@x1 is not activated/booted up.",name()));
370                                        return false;
371                                }
372                                return true;
373                        case CMMsg.TYP_ACTIVATE:
374                                if((msg.targetMessage()==null)&&(activated()))
375                                {
376                                        msg.source().tell(L("@x1 is already booted up.",name()));
377                                        return false;
378                                }
379                                else
380                                if(powerRemaining()<=0)
381                                {
382                                        if((!CMLib.tech().seekBatteryPower(this, this.circuitKey))
383                                        ||(powerRemaining()<=0))
384                                        {
385                                                msg.source().tell(L("@x1 won't seem to power up. Perhaps it needs power?",name()));
386                                                return false;
387                                        }
388                                }
389                                break;
390                        case CMMsg.TYP_DEACTIVATE:
391                                if((msg.targetMessage()==null)&&(!activated()))
392                                {
393                                        msg.source().tell(L("@x1 is already shut down.",name()));
394                                        return false;
395                                }
396                                break;
397                        }
398                }
399                return super.okMessage(host,msg);
400        }

402        @Override
403        public void executeMsg(final Environmental host, final CMMsg msg)
404        {
405                if(msg.amITarget(this))
406                {
407                        switch(msg.targetMinor())
408                        {
409                        case CMMsg.TYP_DROP:
410                                setInstalledFactor((float)CMath.div(msg.value(),100.0));
411                                break;
412                        case CMMsg.TYP_READ:
413                                if(msg.source().riding()!=this)
414                                        lastReader=msg.source();
415                                break;
416                        case CMMsg.TYP_WRITE:
417                        {
418                                if(msg.targetMessage()!=null)
419                                {
420                                        final List<Software> software=getSoftware();
421                                        final List<CMMsg> msgs=new LinkedList<CMMsg>();
422                                        synchronized(software)
423                                        {
424                                                for(final Software S : software)
425                                                {
426                                                        if(S.getInternalName().equals(currentMenu) && (currentMenu.length()>0))
427                                                        {
428                                                                if(msg.targetMessage().trim().equals("<"))
429                                                                {
430                                                                        msgs.add(CMClass.getMsg(msg.source(),S,null,CMMsg.NO_EFFECT,CMMsg.MASK_ALWAYS|CMMsg.TYP_DEACTIVATE,CMMsg.NO_EFFECT,null));
431                                                                }
432                                                                else
433                                                                if(S.isCommandString(msg.targetMessage(), true))
434                                                                {
435                                                                        msgs.add(CMClass.getMsg(msg.source(),S,null,CMMsg.NO_EFFECT,null,CMMsg.MASK_ALWAYS|CMMsg.TYP_WRITE,msg.targetMessage(),CMMsg.NO_EFFECT,null));
436                                                                }
437                                                        }
438                                                        else
439                                                        if((S.getParentMenu().equals(currentMenu))
440                                                        &&(S.isCommandString(msg.targetMessage(), false)))
441                                                        {
442                                                                msgs.add(CMClass.getMsg(msg.source(),S,null,CMMsg.NO_EFFECT,null,CMMsg.MASK_ALWAYS|CMMsg.TYP_ACTIVATE,msg.targetMessage(),CMMsg.NO_EFFECT,null));
443                                                        }
444                                                }
445                                        }
446                                        boolean readFlag=false;
447                                        boolean menuRead=false;
448                                        final MOB M=msg.source();
449                                        if(msgs.size()==0)
450                                                M.location().show(M, this, null, CMMsg.MASK_ALWAYS|CMMsg.TYP_OK_VISUAL, CMMsg.NO_EFFECT, CMMsg.NO_EFFECT, L("<T-NAME> says '^N\n\rUnknown command. Please read the screen for a menu.\n\r^.^N'"));
451                                        else
452                                        {
453                                                if((!subjectToWearAndTear()) || (Math.random() < CMath.div(usesRemaining(), 100)))
454                                                {
455                                                        for(final CMMsg msg2 : msgs)
456                                                        {
457                                                                if(msg2.target().okMessage(M, msg2))
458                                                                {
459                                                                        msg2.target().executeMsg(M, msg2);
460                                                                        if(msg2.target() instanceof Software)
461                                                                        {
462                                                                                final Software sw=(Software)msg2.target();
463                                                                                if(msg2.targetMinor()==CMMsg.TYP_ACTIVATE)
464                                                                                {
465                                                                                        setActiveMenu(sw.getInternalName());
466                                                                                        readFlag=true;
467                                                                                }
468                                                                                else
469                                                                                if(msg2.targetMinor()==CMMsg.TYP_DEACTIVATE)
470                                                                                {
471                                                                                        setActiveMenu(sw.getParentMenu());
472                                                                                        menuRead=true;
473                                                                                }
474                                                                                else
475                                                                                {
476                                                                                        readFlag=true;
477                                                                                }
478                                                                        }
479                                                                }
480                                                        }
481                                                }
482                                                else
483                                                {
484                                                        final List<MOB> readers=getCurrentReaders();
485                                                        for(final MOB M2 : readers)
486                                                        {
487                                                                if(CMLib.flags().canBeSeenBy(this, M2))
488                                                                        M2.location().show(M2, this, null, CMMsg.MASK_ALWAYS|CMMsg.TYP_OK_VISUAL, CMMsg.NO_EFFECT, CMMsg.NO_EFFECT, L("<T-NAME> blue screens!!\n\r^.^N'"));
489                                                        }
490                                                        deactivateSystem();
491                                                }
492                                        }
493                                        if(readFlag)
494                                                forceReadersSeeNew();
495                                        if(menuRead)
496                                                forceReadersMenu();
497                                }
498                                break;
499                        }
500                        case CMMsg.TYP_GET:
501                        case CMMsg.TYP_PUT:
502                        case CMMsg.TYP_INSTALL:
503                                nextSoftwareCheck=0;
504                                break;
505                        case CMMsg.TYP_LOOK:
506                                super.executeMsg(host, msg);
507                                if(CMLib.flags().canBeSeenBy(this, msg.source()))
508                                        msg.source().tell(L("@x1 is currently @x2",name(),(activated()?"booted up and the screen ready to be read.\n\r":"deactivated.\n\r")));
509                                return;
510                        case CMMsg.TYP_ACTIVATE:
511                                if(!activated())
512                                {
513                                        activate(true);
514                                        setActiveMenu("");
515                                        if((msg.source().location()!=null)&&(!CMath.bset(msg.targetMajor(), CMMsg.MASK_CNTRLMSG)))
516                                        {
517                                                msg.source().location().show(msg.source(),this,null,CMMsg.MSG_OK_VISUAL,L("<S-NAME> boot(s) up <T-NAME>."));
518                                                forceReadersMenu();
519                                        }
520                                }
521                                if((msg.targetMessage()!=null)&&(activated()))
522                                {
523                                        final List<Software> software=getSoftware();
524                                        final List<CMMsg> msgs=new LinkedList<CMMsg>();
525                                        synchronized(software)
526                                        {
527                                                for(final Software S : software)
528                                                {
529                                                        if(S.isActivationString(msg.targetMessage()))
530                                                        {
531                                                                msgs.add(CMClass.getMsg(msg.source(),S,null,CMMsg.NO_EFFECT,null,CMMsg.MASK_ALWAYS|CMMsg.TYP_ACTIVATE,msg.targetMessage(),CMMsg.NO_EFFECT,null));
532                                                        }
533                                                }
534                                        }
535                                        final boolean readFlag=false;
536                                        final MOB M=msg.source();
537                                        if(msgs.size()==0)
538                                        {
539                                                final Room R=M.location();
540                                                if(R!=null)
541                                                        R.show(M, this, null, CMMsg.MASK_ALWAYS|CMMsg.TYP_OK_VISUAL, CMMsg.NO_EFFECT, CMMsg.NO_EFFECT, L("<T-NAME> says '^N\n\rUnknown activation command. Please read the screen for a menu of TYPEable commands.\n\r^.^N'"));
542                                        }
543                                        else
544                                        {
545                                                double damageFailChance=1.0;
546                                                if(subjectToWearAndTear() && (usesRemaining()<75))
547                                                {
548                                                        damageFailChance = CMath.div(usesRemaining(), 100);
549                                                        damageFailChance += (0.35 * getFinalManufacturer().getReliabilityPct());
550                                                }
551                                                if((Math.random()<getInstalledFactor()) && (Math.random()<damageFailChance))
552                                                {
553                                                        for(final CMMsg msg2 : msgs)
554                                                        {
555                                                                if(msg2.target().okMessage(M, msg2))
556                                                                        msg2.target().executeMsg(M, msg2);
557                                                        }
558                                                }
559                                                else
560                                                {
561                                                        final List<MOB> readers=getCurrentReaders();
562                                                        for(final MOB M2 : readers)
563                                                        {
564                                                                if(CMLib.flags().canBeSeenBy(this, M2))
565                                                                        M2.location().show(M2, this, null, CMMsg.MASK_ALWAYS|CMMsg.TYP_OK_VISUAL, CMMsg.NO_EFFECT, CMMsg.NO_EFFECT, L("<T-NAME> blue screens!!\n\r^.^N'"));
566                                                        }
567                                                        deactivateSystem();
568                                                }
569                                        }
570                                        if(readFlag)
571                                                forceReadersSeeNew();
572                                }
573                                break;
574                        case CMMsg.TYP_DEACTIVATE:
575                                if((msg.targetMessage()!=null)&&(activated()))
576                                {
577                                        final List<Software> software=getSoftware();
578                                        final List<CMMsg> msgs=new LinkedList<CMMsg>();
579                                        synchronized(software)
580                                        {
581                                                for(final Software S : software)
582                                                {
583                                                        if(S.isActivationString(msg.targetMessage()))
584                                                        {
585                                                                msgs.add(CMClass.getMsg(msg.source(),S,null,CMMsg.NO_EFFECT,null,CMMsg.MASK_ALWAYS|CMMsg.TYP_DEACTIVATE,msg.targetMessage(),CMMsg.NO_EFFECT,null));
586                                                        }
587                                                }
588                                        }
589                                        final boolean readFlag=false;
590                                        final MOB M=msg.source();
591                                        if(msgs.size()==0)
592                                                M.location().show(M, this, null, CMMsg.MASK_ALWAYS|CMMsg.TYP_OK_VISUAL, CMMsg.NO_EFFECT, CMMsg.NO_EFFECT, L("<T-NAME> says '^N\n\rUnknown deactivation command. Please read the screen for a menu of TYPEable commands.\n\r^.^N'"));
593                                        else
594                                        for(final CMMsg msg2 : msgs)
595                                        {
596                                                if(msg2.target().okMessage(M, msg2))
597                                                        msg2.target().executeMsg(M, msg2);
598                                        }
599                                        if(readFlag)
600                                                forceReadersSeeNew();
601                                }
602                                else
603                                if(activated())
604                                {
605                                        activate(false);
606                                        if((msg.source().location()!=null)&&(!CMath.bset(msg.targetMajor(), CMMsg.MASK_CNTRLMSG)))
607                                                msg.source().location().show(msg.source(),this,null,CMMsg.MSG_OK_VISUAL,L("<S-NAME> shut(s) down <T-NAME>."));
608                                        deactivateSystem();
609                                }
610                                break;
611                        case CMMsg.TYP_POWERCURRENT:
612                                {
613                                        if(!CMProps.getBoolVar(CMProps.Bool.MUDSTARTED))
614                                                nextPowerCycleTmr=System.currentTimeMillis()+(8*1000);
615                                        final int powerToGive=msg.value();
616                                        if(powerToGive>0)
617                                        {
618                                                if(powerRemaining()==0)
619                                                        setPowerRemaining(1);
620                                                nextPowerCycleTmr=System.currentTimeMillis()+(8*1000);
621                                        }
622                                        if(activated())
623                                        {
624                                                final List<Software> software=getSoftware();
625                                                final CMMsg msg2=CMClass.getMsg(msg.source(), null, null, CMMsg.NO_EFFECT,null,CMMsg.MSG_POWERCURRENT,null,CMMsg.NO_EFFECT,null);
626                                                synchronized(software)
627                                                {
628                                                        for(final Software sw : software)
629                                                        {
630                                                                msg2.setTarget(sw);
631                                                                msg2.setValue(((powerToGive>0)?1:0)+(this.getActiveMenu().equals(sw.getInternalName())?1:0));
632                                                                if(sw.okMessage(host, msg2))
633                                                                        sw.executeMsg(host, msg2);
634                                                        }
635                                                }
636                                        }
637                                        forceReadersSeeNew();
638                                        if(System.currentTimeMillis()>nextPowerCycleTmr)
639                                        {
640                                                deactivateSystem();
641                                        }
642                                }
643                                break;
644                        }
645                }
646                else
647                if((msg.source()==this.lastReader)
648                &&(msg.target() instanceof ElecPanel)
649                &&(msg.targetMinor()==CMMsg.TYP_READ))
650                        this.lastReader=null;
651                super.executeMsg(host,msg);
652        }
