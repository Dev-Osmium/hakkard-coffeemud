043public class StdShipGravityGen extends StdElecCompItem
044{
045        @Override
046        public String ID()
047        {
048                return "StdShipGravityGen";
049        }
050        
051        public StdShipGravityGen()
052        {
053                super();
054                setName("a gravity generator");
055                setDisplayText("a gravity generator sits here.");
056                setDescription("");
057        }
058
059        @Override
060        public TechType getTechType()
061        {
062                return Technical.TechType.SHIP_ENVIRO_CONTROL;
063        }
064        
065        private volatile Reference<SpaceShip> myShip       = null;
066
067        @Override
068        public void setOwner(ItemPossessor container)
069        {
070                super.setOwner(container);
071                myShip = null;
072        }
073
074        protected synchronized SpaceShip getMyShip()
075        {
076                if(myShip == null)
077                {
078                        final Area area = CMLib.map().areaLocation(this);
079                        if(area instanceof SpaceShip)
080                                myShip = new WeakReference<SpaceShip>((SpaceShip)area);
081                        else
082                                myShip = new WeakReference<SpaceShip>(null);
083                }
084                return myShip.get();
085        }
086        
087        
088        @Override
089        public boolean okMessage(final Environmental host, final CMMsg msg)
090        {
091                if(!super.okMessage(host, msg))
092                        return false;
093                if(msg.amITarget(this))
094                {
095                        switch(msg.targetMinor())
096                        {
097                        case CMMsg.TYP_POWERCURRENT:
098                        {
099                                if(activated())
100                                {
101                                        final SpaceShip ship=getMyShip();
102                                        if((!ship.getShipFlag(ShipFlag.ARTI_GRAV))
103                                        &&(!ship.getShipFlag(ShipFlag.IN_THE_AIR)))
104                                        {
105                                                final double efficiency = this.getFinalManufacturer().getEfficiencyPct();
106                                                final double reliability = this.getFinalManufacturer().getReliabilityPct();
107                                                double wearAndTear = 1.0;
108                                                if(this.subjectToWearAndTear() && this.usesRemaining()<100)
109                                                        wearAndTear = Math.min(1.0, reliability * CMath.div(this.usesRemaining(), 100.0));
110                                                final double amountNeeded = ship.getShipArea().numberOfProperIDedRooms();
111                                                final double powerUsed = amountNeeded * efficiency; // eff is from 0.5 (great) to 2.0 (terrible)
112                                                double powerFactor = 1.0;
113                                                if(powerUsed > this.powerRemaining())
114                                                {
115                                                        //TODO: auto-draw from auxillary power? batteries?
116                                                        powerFactor = CMath.div(this.powerRemaining(), powerUsed);
117                                                        this.setPowerRemaining(0);
118                                                }
119                                                else
120                                                        this.setPowerRemaining(Math.round(this.powerRemaining() - powerUsed));
121                                                if(CMLib.dice().rollPercentage()<CMath.mul(150.0, wearAndTear * powerFactor))
122                                                        ship.setShipFlag(ShipFlag.ARTI_GRAV, true);
123                                        }
124                                }
125                                break;
126                        }
127                        }
128                }
129                return true;
130        }
131
132        protected static void sendComputerMessage(final ShipWarComponent me, final String circuitKey, final MOB mob, final Item controlI, final String code)
133        {
134                for(final Iterator<Computer> c=CMLib.tech().getComputers(circuitKey);c.hasNext();)
135                {
136                        final Computer C=c.next();
137                        if((controlI==null)||(C!=controlI.owner()))
138                        {
139                                final CMMsg msg2=CMClass.getMsg(mob, C, me, CMMsg.NO_EFFECT, null, CMMsg.MSG_ACTIVATE|CMMsg.MASK_CNTRLMSG, code, CMMsg.NO_EFFECT,null);
140                                if(C.okMessage(mob, msg2))
141                                        C.executeMsg(mob, msg2);
142                        }
143                }
144        }
145
146        @Override
147        public boolean sameAs(final Environmental E)
148        {
149                if(!(E instanceof StdShipDampener))
150                        return false;
151                return super.sameAs(E);
152        }
153}
