090        public void getDirDesc(String dirBuilder, StringBuilder str, boolean useShipDirs)
091        {
092                int numDone=0;
093                int numTotal=0;
094                for(int d=0;d<dirBuilder.length();d++)
095                        numTotal+=(Character.isLowerCase(dirBuilder.charAt(d))?1:0);
096                if(dirBuilder.length()==0)
097                        str.append(" here");
098                else
099                for(int d=0;d<dirBuilder.length();d++)
100                {
101                        if(dirBuilder.charAt(d)=='S')
102                        {
103                                if(numDone==0)
104                                        str.append(L(" inside a ship"));
105                                else
106                                if(numDone<numTotal-1)
107                                        str.append(L(", inside a ship"));
108                                else
109                                        str.append(L(", and then inside a ship"));
110                                numDone++;
111                                continue;
112                        }
113                        String locDesc="";
114                        if(dirBuilder.charAt(d)=='D')
115                        {
116                                locDesc="behind a door ";
117                                d++;
118                        }
119                        if(dirBuilder.charAt(d)=='I')
120                        {
121                                locDesc="inside a room ";
122                                d++;
123                        }
124                        if(dirBuilder.charAt(d)=='o')
125                        {
126                                locDesc="outdoors ";
127                                d++;
128                        }
129                        final int dir=dirBuilder.charAt(d)-'a';
130                        if(numDone==0)
131                                str.append(" ").append(locDesc).append(useShipDirs?CMLib.directions().getShipDirectionName(dir):CMLib.directions().getDirectionName(dir));
132                        else
133                        if(numDone<numTotal-1)
134                                str.append(", ").append(locDesc).append(useShipDirs?CMLib.directions().getShipDirectionName(dir):CMLib.directions().getDirectionName(dir));
135                        else
136                                str.append(", and then ").append(locDesc).append(useShipDirs?CMLib.directions().getShipInDirectionName(dir):CMLib.directions().getInDirectionName(dir));
137                        numDone++;
138                }
139        }

141        public int getScanMsg(Room R, Set<Room> roomsDone, String dirBuilder, int depthLeft, CMMsg scanMsg, StringBuilder str)
142        {
143                if((R==null)||(roomsDone.contains(R)))
144                        return 0;
145                roomsDone.add(R);
146                int numFound=0;
147                final boolean useShipDirs=(R instanceof BoardableShip)||(R.getArea() instanceof BoardableShip);
148                for(int m=0;m<R.numInhabitants();m++)
149                {
150                        final MOB M=R.fetchInhabitant(m);
151                        if(isAlive(M))
152                        {
153                                scanMsg.setTarget(M);
154                                if(R.okMessage(scanMsg.source(), scanMsg))
155                                {
156                                        numFound++;
157                                        str.append("A "+M.charStats().getMyRace().name());
158                                        getDirDesc(dirBuilder, str, useShipDirs);
159                                        str.append(".\n\r");
160                                }
161                        }
162                        for(int i=0;i<M.numItems();i++)
163                        {
164                                final Item I=M.getItem(i);
165                                if(I instanceof CagedAnimal)
166                                {
167                                        final MOB M2=((CagedAnimal)I).unCageMe();
168                                        if(isAlive(M2))
169                                        {
170                                                numFound++;
171                                                str.append("A "+M2.charStats().getMyRace().name());
172                                                getDirDesc(dirBuilder, str, useShipDirs);
173                                                str.append(".\n\r");
174                                        }
175                                        M2.destroy();
176                                }
177                        }
178                }
179                for(int i=0;i<R.numItems();i++)
180                {
181                        final Item I=R.getItem(i);
182                        if(I instanceof CagedAnimal)
183                        {
184                                final MOB M=((CagedAnimal)I).unCageMe();
185                                if(isAlive(M))
186                                {
187                                        numFound++;
188                                        str.append("A "+M.charStats().getMyRace().name());
189                                        getDirDesc(dirBuilder, str, useShipDirs);
190                                        str.append(".\n\r");
191                                }
192                                M.destroy();
193                        }
194                        if((I instanceof SpaceShip)&&(depthLeft>0))
195                        {
196                                Room shipR=null;
197                                for(final Enumeration<Room> r=((SpaceShip)I).getShipArea().getProperMap(); r.hasMoreElements(); )
198                                {
199                                        final Room R2=r.nextElement();
200                                        for(int d=0;d<Directions.NUM_DIRECTIONS();d++)
201                                        {
202                                                if(R2.getRoomInDir(d)==R)
203                                                {
204                                                        final Exit E2=R2.getExitInDir(d);
205                                                        if(E2==null)
206                                                                continue;
207                                                        shipR=R2;
208                                                        break;
209                                                }
210                                        }
211                                        if(shipR!=null)
212                                                break;
213                                }
214                                if(shipR!=null)
215                                        numFound+=getScanMsg(shipR,roomsDone, dirBuilder+'S', depthLeft-1, scanMsg, str);
216                        }
217                }
218                if(depthLeft>0)
219                {
220                        final boolean isIndoors=(R.domainType()&Room.INDOORS)==Room.INDOORS;
221                        for(int d=0;d<Directions.NUM_DIRECTIONS();d++)
222                        {
223                                final Room R2=R.getRoomInDir(d);
224                                final Exit E2=R.getExitInDir(d);
225                                if((R2==null)||(E2==null))
226                                        continue;
227                                final boolean willIndoors=(R.domainType()&Room.INDOORS)==Room.INDOORS;
228                                final boolean willADoor=E2.hasADoor() && !E2.isOpen();
229                                final String dirBCode=willADoor?"D":
230                                                                (isIndoors && (!willIndoors))?"O":
231                                                                (!isIndoors && (willIndoors))?"I":
232                                                                "";
233                                numFound+=getScanMsg(R2, roomsDone, dirBuilder+dirBCode+((char)('a'+d)), depthLeft-1, scanMsg, str);
234                        }
235                }
236                return numFound;
237        }

239        public String getScanMsg()
240        {
241                final Room R=CMLib.map().roomLocation(this);
242                if(R==null)
243                        return "";
244                final StringBuilder str=new StringBuilder("");
245                final int numFound=getScanMsg(R,new HashSet<Room>(), "",phyStats().weight()+1,getScanMsg(R),str);
246                if(activated)
247                        super.setCurrentScreenDisplay("LIFESCAN: Activated: "+numFound+" found last scan.\n\r");
248                else
249                        super.setCurrentScreenDisplay("LIFESCAN: Activate for continual scanning, type for on-demand.\n\r");
250                if(str.length()==0)
251                        return "No life signs detected.";
252                return str.toString().toLowerCase();
253        }
