216        protected TechComponent findComponentByName(final List<? extends TechComponent> list, final String prefix, String name)
217        {
218                if(list.size()==0)
219                        return null;
220                name=name.toUpperCase();
221                if(name.startsWith(prefix))
222                {
223                        final String numStr=name.substring(6);
224                        if(!CMath.isInteger(numStr))
225                                return null;
226                        final int num=CMath.s_int(numStr);
227                        if((num>0)&&(num<=list.size()))
228                                return list.get(num-1);
229                        return null;
230                }
231                TechComponent E=(TechComponent)CMLib.english().fetchEnvironmental(list, name, true);
232                if(E==null)
233                        E=(TechComponent)CMLib.english().fetchEnvironmental(list, name, false);
234                return E;
235        }

257        @Override
258        public boolean isCommandString(final String word, final boolean isActive)
259        {
260                final Vector<String> parsed=CMParms.parse(word);
261                if(parsed.size()==0)
262                        return false;
263                final String uword=parsed.get(0).toUpperCase();
264                if(uword.equals("HELP")
265                ||uword.equals("STOP")
266                ||uword.equals("LAND")
267                ||uword.equals("LAUNCH")
268                ||uword.equals("ACTIVATE")
269                ||uword.equals("DEACTIVATE")
270                ||uword.equals("TARGET")
271                ||uword.equals("FIRE")
272                ||(uword.startsWith("WEAPON")&&(CMath.isInteger(uword.substring(6))))
273                ||(uword.startsWith("ENGINE")&&(CMath.isInteger(uword.substring(6))))
274                ||(uword.startsWith("SENSOR")&&(CMath.isInteger(uword.substring(6))))
275                ||(uword.startsWith("SYSTEM")&&(CMath.isInteger(uword.substring(6))))
276                )
277                        return true;
278                return findEngineByName(uword)!=null;
279        }

281        public List<SpaceObject> takeSensorReport(final TechComponent sensor)
282        {
283                final List<SpaceObject> localSensorReport;
284                synchronized(sensorReport)
285                {
286                        sensorReport.clear();
287                }
288                final String code=Technical.TechCommand.SENSE.makeCommand();
289                final MOB mob=CMClass.getFactoryMOB();
290                try
291                {
292                        final CMMsg msg=CMClass.getMsg(mob, sensor, this, CMMsg.NO_EFFECT, null, CMMsg.MSG_ACTIVATE|CMMsg.MASK_CNTRLMSG, code, CMMsg.NO_EFFECT,null);
293                        if(sensor.owner() instanceof Room)
294                        {
295                                if(((Room)sensor.owner()).okMessage(mob, msg))
296                                        ((Room)sensor.owner()).send(mob, msg);
297                        }
298                        else
299                        if(sensor.okMessage(mob, msg))
300                                sensor.executeMsg(mob, msg);
301                }
302                finally
303                {
304                        mob.destroy();
305                }
306                localSensorReport = new XVector<SpaceObject>(sensorReport.iterator());
307                synchronized(sensorReport)
308                {
309                        sensorReport.clear();
310                        while(this.sensorReports.size()>10)
311                                this.sensorReports.removeLast();
312                        this.sensorReports.addFirst(new Pair<Long,List<SpaceObject>>(new Long(System.currentTimeMillis()),localSensorReport));
313                }
314                return localSensorReport;
315        }

527        protected Double fixInjection(final Double lastInject, final Double lastAccelleration, final double targetAccelleration)
528        {
529                final Double newInject;
530                if(lastAccelleration.doubleValue() < targetAccelleration)
531                {
532                        if(lastAccelleration.doubleValue() < (targetAccelleration * .00001))
533                                newInject = new Double(lastInject.doubleValue()*200.0);
534                        else
535                        if(lastAccelleration.doubleValue() < (targetAccelleration * .001))
536                                newInject = new Double(lastInject.doubleValue()*20.0);
537                        else
538                        if(lastAccelleration.doubleValue() < (targetAccelleration * .1))
539                                newInject = new Double(lastInject.doubleValue()*2.0);
540                        else
541                        if(lastAccelleration.doubleValue() < (targetAccelleration * .5))
542                                newInject = new Double(lastInject.doubleValue()*1.25);
543                        else
544                        if(lastAccelleration.doubleValue() < (targetAccelleration * 0.9))
545                                newInject = new Double(1.07 * lastInject.doubleValue());
546                        else
547                        if(lastAccelleration.doubleValue() < (targetAccelleration * 0.95))
548                                newInject = new Double(1.02 * lastInject.doubleValue());
549                        else
550                        if(lastAccelleration.doubleValue() < (targetAccelleration * 0.99))
551                                newInject = new Double(1.01 * lastInject.doubleValue());
552                        else
553                                newInject = new Double(1.001 * lastInject.doubleValue());
554                }
555                else
556                if(lastAccelleration.doubleValue() > targetAccelleration)
557                {
558                        if(lastAccelleration.doubleValue() > (targetAccelleration * 1000000))
559                                newInject = new Double(lastInject.doubleValue()/200.0);
560                        else
561                        if(lastAccelleration.doubleValue() > (targetAccelleration * 10000))
562                                newInject = new Double(lastInject.doubleValue()/20.0);
563                        else
564                        if(lastAccelleration.doubleValue() > (targetAccelleration * 100))
565                                newInject = new Double(lastInject.doubleValue()/2.0);
566                        else
567                        if(lastAccelleration.doubleValue() > (targetAccelleration * 2))
568                                newInject = new Double(lastInject.doubleValue()/1.25);
569                        else
570                        if(lastAccelleration.doubleValue() > (targetAccelleration * 1.1))
571                                newInject = new Double(0.93 * lastInject.doubleValue());
572                        else
573                        if(lastAccelleration.doubleValue() > (targetAccelleration * 1.05))
574                                newInject = new Double(0.98 * lastInject.doubleValue());
575                        else
576                        if(lastAccelleration.doubleValue() > (targetAccelleration * 1.01))
577                                newInject = new Double(0.99 * lastInject.doubleValue());
578                        else
579                                newInject = new Double(0.999 * lastInject.doubleValue());
580                }

608                }
