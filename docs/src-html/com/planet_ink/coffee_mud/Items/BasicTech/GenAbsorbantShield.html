035public class GenAbsorbantShield extends GenPersonalShield
036{
037        @Override
038        public String ID()
039        {
040                return "GenAbsorbantShield";
041        }
042
043        public GenAbsorbantShield()
044        {
045                super();
046                setName("an absorption shield generator");
047                setDisplayText("an absorption shield generator sits here.");
048                setDescription("The absorption shield generator is worn about the body and activated to use. It absorbs all manner of weapon types. ");
049                setDescription("The integrity shield generator is worn about the body and activated to use. It protects against disruption and disintegration beams. ");
050        }
051
052        @Override
053        protected String fieldOnStr(MOB viewerM)
054        {
055                return L((owner() instanceof MOB)?
056                        "A sparkling field of energy surrounds <O-NAME>.":
057                        "A sparkling field of energy surrounds <T-NAME>.");
058        }
059
060        @Override
061        protected String fieldDeadStr(MOB viewerM)
062        {
063                return L((owner() instanceof MOB)?
064                        "The sparkling field around <O-NAME> flickers and dies out.":
065                        "The sparkling field around <T-NAME> flickers and dies out.");
066        }
067
068        @Override
069        protected boolean doShield(MOB mob, CMMsg msg, double successFactor)
070        {
071                if(msg.value()<=0)
072                        return true;
073                if((successFactor>=1.0)||((successFactor>0.0)&&(msg.value()==1)))
074                {
075                        mob.location().show(mob,msg.source(),null,CMMsg.MSG_OK_VISUAL,L("The sparkling field around <S-NAME> completely absorbs the @x1 attack from <T-NAME>.",msg.tool().name()));
076                        msg.setValue(0);
077                }
078                else
079                if(successFactor>=0.0)
080                {
081                        msg.setValue((int)Math.round(successFactor*msg.value()));
082                        final String showDamage = CMProps.getVar(CMProps.Str.SHOWDAMAGE).equalsIgnoreCase("YES")?" ("+Math.round(successFactor*100.0)+")":"";
083                        if(successFactor>=0.75)
084                                msg.addTrailerMsg(CMClass.getMsg(mob,msg.source(),msg.tool(),CMMsg.MSG_OK_VISUAL,L("The sparkling field around <S-NAME> absorbs most@x1 of the <O-NAMENOART> damage.",showDamage)));
085                        else
086                        if(successFactor>=0.50)
087                                msg.addTrailerMsg(CMClass.getMsg(mob,msg.source(),msg.tool(),CMMsg.MSG_OK_VISUAL,L("The sparkling field around <S-NAME> absorbs much@x1 of the <O-NAMENOART> damage.",showDamage)));
088                        else
089                        if(successFactor>=0.25)
090                                msg.addTrailerMsg(CMClass.getMsg(mob,msg.source(),msg.tool(),CMMsg.MSG_OK_VISUAL,L("The sparkling field around <S-NAME> absorbs some@x1 of the <O-NAMENOART> damage.",showDamage)));
091                        else
092                                msg.addTrailerMsg(CMClass.getMsg(mob,msg.source(),msg.tool(),CMMsg.MSG_OK_VISUAL,L("The sparkling field around <S-NAME> absorbs a little@x1 of the <O-NAMENOART> damage.",showDamage)));
093                }
094                return true;
095        }
096
097        @Override
098        protected boolean doesShield(MOB mob, CMMsg msg, double successFactor)
099        {
100                return activated();
101        }
102}
