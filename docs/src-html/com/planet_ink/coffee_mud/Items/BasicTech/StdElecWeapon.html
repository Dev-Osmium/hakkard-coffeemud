065        public StdElecWeapon()
066        {
067                super();

069                setName("a tech gun");
070                setDisplayText("a tech gun sits here.");
071                setDescription("You can't tell what it is by looking at it.");

073                setRawLogicalAnd(false);
074                setRawProperLocationBitmap(Wearable.WORN_HELD|Wearable.WORN_WIELD);

087        // this method is the reason not to make the types an editable field; the
088        // murder-motel-like interaction between shields and weapons is finely
089        // balanced based on what this does.
090        @Override 
091        public int weaponDamageType()
092        {
093                switch(mode)
094                {
095                case LASER:
096                        return Weapon.TYPE_LASERING;
097                case SONIC:
098                        return Weapon.TYPE_SONICING;
099                case DISRUPT:
100                case DISINTEGRATE:
101                        return Weapon.TYPE_MELTING;
102                case STUN:
103                        return Weapon.TYPE_SONICING;
104                case MAIM:
105                        return Weapon.TYPE_SLASHING;
106                default:
107                        return weaponDamageType;
108                }
109        }

111        protected ModeType getState(String s)
112        {
113                s=s.toUpperCase().trim();
114                for(final ModeType type : modeTypes)
115                {
116                        if(type.toString().startsWith(s))
117                                return type;
118                }
119                return null;
120        }
121
122        protected String getStateName()
123        {
124                return mode.name().toLowerCase();
125        }

127        @Override
128        public void setMiscText(String text)
129        {
130                miscText=text;
131                if(CMath.isInteger(text))
132                {
133                        final int x=CMath.s_int(text);
134                        if((x>=0)&&(x<modeTypes.length))
135                                mode=modeTypes[x];
136                }
137                else
138                {
139                        final ModeType t=(ModeType)CMath.s_valueOf(ModeType.class,text.toUpperCase().trim());
140                        if((t != null)&&(CMParms.indexOf(modeTypes, t)>=0))
141                                mode=t;
142                }
143        }

151        @Override
152        public void recoverPhyStats()
153        {
154                super.recoverPhyStats();
155                if(mode!=null)
156                switch(mode)
157                {
158                case STUN:
159                        phyStats().setDamage(1);
160                        break;
161                case NORMAL:
162                case MAIM:
163                case KILL:
164                case LASER:
165                case SONIC:
166                        break;
167                case DISINTEGRATE:
168                case DISRUPT:
169                        phyStats().setDamage(1+(phyStats().damage()*2));
170                        break;
171                }
172        }

217        @Override
218        public void executeMsg(final Environmental myHost, final CMMsg msg)
219        {
220                if(msg.amITarget(this))
221                {
222                        switch(msg.targetMinor())
223                        {
224                        case CMMsg.TYP_LOOK:
225                        case CMMsg.TYP_EXAMINE:
226                                if(CMLib.flags().canBeSeenBy(this, msg.source()))
227                                {
228                                        super.executeMsg(myHost,msg);
229                                        msg.source().tell(L("@x1 is currently @x2 and is at @x3% power.",name(),(activated()?"activated":"deactivated"),""+Math.round(CMath.div(powerRemaining(),powerCapacity())*100.0)));
230                                        if(CMLib.flags().canBeSeenBy(this, msg.source())&&(activated()))
231                                                msg.source().tell(L("@x1 is currently set to @x2.",name(),getStateName()));
232                                }
233                                break;
234                        case CMMsg.TYP_ACTIVATE:
235                        {
236                                super.executeMsg(myHost,msg);
237                                if((msg.source().location()!=null)&&(!CMath.bset(msg.targetMajor(), CMMsg.MASK_CNTRLMSG)))
238                                        msg.source().location().show(msg.source(), this, CMMsg.MSG_OK_VISUAL, L("<S-NAME> activate(s) <T-NAME>."));
239                                this.activate(true);
240                                if((msg.source().location()!=null)&&(!CMath.bset(msg.targetMajor(), CMMsg.MASK_CNTRLMSG)))
241                                {
242                                        ModeType newState=mode;
243                                        if((msg.targetMessage()!=null)&&(msg.targetMessage().length()>0))
244                                        {
245                                                final List<String> V=CMParms.parse(msg.targetMessage());
246                                                if(V.size()>0)
247                                                {
248                                                        final String s=V.get(0).toUpperCase().trim();
249                                                        newState=this.getState(s);
250                                                        if(newState==null)
251                                                                newState=mode;
252                                                }
253                                        }
254                                        mode=newState;
255                                        msg.source().location().show(msg.source(), this, CMMsg.MSG_OK_VISUAL, L("<S-NAME> set(s) <T-NAME> on @x1.",this.getStateName()));
256                                        recoverPhyStats();
257                                        msg.source().recoverPhyStats();
258                                }
259                                this.activate(true);
260                                break;
261                        }
262                        case CMMsg.TYP_DEACTIVATE:
263                        {
264                                super.executeMsg(myHost,msg);
265                                if((msg.source().location()!=null)&&(!CMath.bset(msg.targetMajor(), CMMsg.MASK_CNTRLMSG)))
266                                        msg.source().location().show(msg.source(), this, CMMsg.MSG_OK_VISUAL, L("<S-NAME> deactivate(s) <T-NAME>."));
267                                this.activate(false);
268                                break;
269                        }
270                        default:
271                                super.executeMsg(myHost,msg);
272                                break;
273                        }
274                }
275                else
276                if((owner() instanceof MOB) && msg.amISource((MOB)owner()) && (!amWearingAt(Wearable.IN_INVENTORY)))
277                {
278                        super.executeMsg(myHost,msg);
279                        final MOB mob=(MOB)owner();
280                        switch(msg.targetMinor())
281                        {
282                        case CMMsg.TYP_WEAPONATTACK:
283                                if((msg.tool()==this) && (weaponClassification()==Weapon.CLASS_THROWN))
284                                        msg.addTrailerMsg(CMClass.getMsg(msg.source(),this,CMMsg.MSG_EXPIRE|CMMsg.MASK_ALWAYS,null));
285                                break;
286                        case CMMsg.TYP_DAMAGE:
287                                if(msg.tool() ==this)
288                                {
289                                        switch(mode)
290                                        {
291                                        case STUN: {
292                                                if(msg.value()>0)
293                                                {
294                                                        final Ability A=CMClass.getAbility("CombatSleep");
295                                                        if((A!=null)&&(msg.target() instanceof Physical))
296                                                                A.invoke(mob, (Physical)msg.target(), true, 0);
297                                                        else
298                                                        if((A==null)&&(msg.target() instanceof MOB))
299                                                        {
300                                                                msg.source().location().show((MOB)msg.target(), null, CMMsg.MSG_OK_VISUAL, L("<S-NAME> go(es) unconscious!"));
301                                                                ((MOB)msg.target()).basePhyStats().setDisposition(((MOB)msg.target()).basePhyStats().disposition()|PhyStats.IS_SLEEPING);
302                                                                ((MOB)msg.target()).phyStats().setDisposition(((MOB)msg.target()).phyStats().disposition()|PhyStats.IS_SLEEPING);
303                                                        }
304                                                        if(mob.getVictim()==msg.source())
305                                                                mob.makePeace(true);
306                                                        if(msg.source().getVictim()==mob)
307                                                                msg.source().makePeace(true);
308                                                        msg.setValue(0);
309                                                }
310                                                break;
311                                        }
312                                        case NORMAL:
313                                        case KILL:
314                                        case LASER:
315                                        case SONIC:
316                                        {
317                                                // this is normal...
318                                                break;
319                                        }
320                                        case DISINTEGRATE:
321                                        {
322                                                final Environmental targ=msg.target();
323                                                if((msg.value()>(basePhyStats().damage()*1.5))
324                                                &&((!(targ instanceof MOB))||(msg.value()>=((MOB)targ).curState().getHitPoints())))
325                                                {
326                                                        final Room R=CMLib.map().roomLocation(targ);
327                                                        if(R!=null)
328                                                        {
329                                                                if(targ instanceof MOB)
330                                                                {
331                                                                        if((!((MOB)targ).amDead())||(((MOB)targ).curState().getHitPoints()>0))
332                                                                                CMLib.combat().postDeath(msg.source(), (MOB)targ, null);
333                                                                        ((MOB)targ).location().show(mob,targ,CMMsg.MSG_OK_ACTION,L("<T-NAME> disintegrate(s)!"));
334                                                                        if(((MOB)targ).amDead())
335                                                                        {
336                                                                                DeadBody corpseI=null;
337                                                                                for(int i=0;i<R.numItems();i++)
338                                                                                {
339                                                                                        final Item I=R.getItem(i);
340                                                                                        if((I instanceof DeadBody)
341                                                                                        &&(I.container()==null)
342                                                                                        &&(((DeadBody)I).getMobName().equals(targ.Name()))
343                                                                                        &&(!((DeadBody)I).isPlayerCorpse()))
344                                                                                        {
345                                                                                                corpseI=(DeadBody)I;
346                                                                                                break;
347                                                                                        }
348                                                                                }
349                                                                                if(corpseI!=null)
350                                                                                {
351                                                                                        corpseI.destroy();
352                                                                                }
353                                                                        }
354                                                                }
355                                                                else
356                                                                if((targ instanceof Item)
357                                                                &&((!(targ instanceof DeadBody)||(!((DeadBody)targ).isPlayerCorpse()))))
358                                                                {
359                                                                        ((MOB)targ).location().show(mob,targ,CMMsg.MSG_OK_ACTION,L("^S<S-NAME> fire(s) <O-NAME> at <T-NAME> and it disintegrates!^?"));
360                                                                        ((Item)targ).destroy();
361                                                                }
362                                                                R.recoverRoomStats();
363                                                        }
364                                                }
365                                                break;
366                                        }
367                                        case MAIM:
368                                        {
369                                                if(msg.value()>0)
370                                                {
371                                                        final Ability A=CMClass.getAbility("Amputation");
372                                                        if((A!=null)&&(msg.target() instanceof Physical))
373                                                                A.invoke(mob, (Physical)msg.target(), true, 0);
374                                                }
375                                                break;
376                                        }
377                                        case DISRUPT:
378                                        {
379                                                final Environmental targ=msg.target();
380                                                if((msg.value()>(basePhyStats().damage()*1.5))
381                                                &&((!(targ instanceof MOB))||(msg.value()>=((MOB)targ).curState().getHitPoints())))
382                                                {
383                                                        final Room R=CMLib.map().roomLocation(targ);
384                                                        if(R!=null)
385                                                        {
386                                                                if(targ instanceof MOB)
387                                                                {
388                                                                        if((!((MOB)targ).amDead())||(((MOB)targ).curState().getHitPoints()>0))
389                                                                                CMLib.combat().postDeath(msg.source(), (MOB)targ, msg);
390                                                                }
391                                                                else
392                                                                if((targ instanceof Item)
393                                                                &&((!(targ instanceof DeadBody)||(!((DeadBody)targ).isPlayerCorpse()))))
394                                                                {
395                                                                        ((MOB)targ).location().show(mob,targ,CMMsg.MSG_OK_ACTION,L("^S<S-NAME> fire(s) <O-NAME> at <T-NAME> and it explodes!^?"));
396                                                                        ((Item)targ).destroy();
397                                                                }
398                                                                R.recoverRoomStats();
399                                                        }
400                                                }
401                                                break;
402                                        }
403                                        }
404                                }
405                                break;
406                        default:
407                                break;
408                        }
409                }
410        }

412        @Override
413        public boolean okMessage(final Environmental myHost, final CMMsg msg)
414        {
415                if(!super.okMessage(myHost, msg))
416                        return false;
417                if((owner() instanceof MOB) && msg.amISource((MOB)owner()) && (!amWearingAt(Wearable.IN_INVENTORY)))
418                {
419                        final MOB mob=(MOB)owner();
420                        switch(msg.targetMinor())
421                        {
422                        case CMMsg.TYP_ACTIVATE:
423                                if((msg.source().location()!=null)&&(!CMath.bset(msg.targetMajor(), CMMsg.MASK_CNTRLMSG)))
424                                {
425                                        if((msg.targetMessage()!=null)&&(msg.targetMessage().length()>0))
426                                        {
427                                                final List<String> V=CMParms.parse(msg.targetMessage());
428                                                if(V.size()>0)
429                                                {
430                                                        final String s=V.get(0).toUpperCase().trim();
431                                                        final ModeType newState=this.getState(s);
432                                                        if(newState==null)
433                                                        {
434                                                                msg.source().tell(L("'@x1' is an unknown setting on @x2",s,name(msg.source())));
435                                                                return false;
436                                                        }
437                                                }
438                                        }
439                                }
440                                break;
441                        case CMMsg.TYP_WEAPONATTACK:
442                                if(msg.tool() ==this)
443                                {
444                                        if((powerRemaining()<=0)||(!activated))
445                                        {
446                                                String msgStr;
447                                                if(weaponClassification()==Weapon.CLASS_RANGED)
448                                                        msgStr="<S-YOUPOSS> <T-NAMENOART> goes *click* and does not fire.";
449                                                else
450                                                if(activated())
451                                                        msgStr="<S-YOUPOSS> <T-NAMENOART> seems to be out of power.";
452                                                else
453                                                        msgStr="<S-YOUPOSS> <T-NAMENOART> seems to be turned off.";
454                                                final CMMsg msg2=CMClass.getMsg(mob, this, null,CMMsg.MSG_OK_VISUAL,msgStr);
455                                                if((mob.location()!=null)&&(mob.location().okMessage(myHost, msg2)))
456                                                        mob.location().send(mob, msg2);
457                                                return false;
458                                        }
459                                }
460                                break;
461                        case CMMsg.TYP_DAMAGE: // remember 50% miss rate
462                                if(msg.tool() ==this)
463                                {
464                                        double successFactor=0.5;
465                                        final Manufacturer m=getFinalManufacturer();
466                                        successFactor=m.getReliabilityPct()*successFactor;
467                                        final long powerConsumed=Math.round(phyStats().damage()*Math.max(.33, Math.abs(2.0-m.getEfficiencyPct())));
468                                        if(powerRemaining()>=powerConsumed)
469                                        {
470                                                setPowerRemaining(powerRemaining()-powerConsumed);
471                                                if(msg.value()>0)
472                                                        msg.setValue((int)Math.round(successFactor*msg.value()));
473                                        }
474                                        else
475                                        {
476                                                setPowerRemaining(0);
477                                        }
478                                        if(mode == ModeType.STUN)
479                                        {
480                                                if(msg.value()>0)
481                                                        msg.setValue(1);
482                                        }
483                                }
484                                break;
485                        }
486                }
487                return true;
488        }
