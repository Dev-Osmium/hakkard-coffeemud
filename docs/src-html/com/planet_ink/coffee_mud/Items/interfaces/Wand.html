034/**
035 * A wand is an item containing a magic spell that can be invoked
036 * on a target, by holding the wand, and then saying the magic word
037 * to the target.
038 * @author Bo Zimmerman
039 *
040 */
041public interface Wand extends MiscMagic
042{
043        /**
044         * Sets the spell ability object that this wand can cast.
045         * @see Wand#getSpell()
046         * @param theSpell the spell ability object that this wand can cast.
047         */
048        public void setSpell(Ability theSpell);
049        
050        /**
051         * Sets the spell ability object that this wand can cast.
052         * @see Wand#setSpell(Ability)
053         * @return the spell ability object that this wand can cast.
054         */
055        public Ability getSpell();
056
057        /**
058         * Returns whether the given mob can use this wand.  They
059         * must be holding it, and must say the magic word, etc.
060         * @param mob the mob to check for waving ability
061         * @param message what the mob said prior to the check
062         * @return true if the wand can be invoked, false otherwise
063         */
064        public boolean checkWave(MOB mob, String message);
065        
066        /**
067         * Checks to see if the mob can invoke this wand against the
068         * given target, given the message they just said out loud.
069         * @param mob the invoker of the wand
070         * @param afftarget the target of the say message
071         * @param message the thing said by the wand invoker
072         */
073        public void waveIfAble(MOB mob, Physical afftarget, String message);
074        
075        /**
076         * Returns the magic word that invokes this wand
077         * @return the magic word that invokes this wand
078         */
079        public String magicWord();
080        
081        /**
082         * Gets the number of times this wand can be invoked before
083         * being drained and empty useless stick.
084         * @see Wand#setMaxUses(int)
085         * @return the number of times this wand can be invoked
086         */
087        public int maxUses();
088        
089        /**
090         * Sets the number of times this wand can be invoked before
091         * being drained and empty useless stick.
092         * @see Wand#maxUses()
093         * @param maxUses the number of times this wand can be invoked
094         */
095        public void setMaxUses(int maxUses);
096}
