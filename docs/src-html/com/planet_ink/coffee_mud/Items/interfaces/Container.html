033/**
034 * A container is an item that contains other items, including other containers.
035 * Next to the Item interface, the Container interface is probably the most basic.
036 * @author Bo Zimmerman
037 *
038 */
039public interface Container extends Item, CloseableLockable
040{
041        /**
042         * Returns all the items in this container, including other
043         * containers and THEIR contents, recursively.
044         * @see Container#getContents()
045         * @return all the items in this container, recursively
046         */
047        public ReadOnlyList<Item> getDeepContents();
048        
049        /**
050         * Returns all the immediate items in this container, including
051         * other containers, but not the contents of those innner containers.
052         * Just the first level contents of this container.
053         * @see Container#getDeepContents()
054         * @return all the immediate items in this container
055         */
056        public ReadOnlyList<Item> getContents();
057        
058        /**
059         * Returns the maximum weight that can fit inside this container.  Weight
060         * is used as a proxy for volume in CoffeeMud.
061         * @see Container#setCapacity(int)
062         * @return the maximum weight that can fit inside this container
063         */
064        public int capacity();
065        
066        /**
067         * Sets the maximum weight that can fit inside this container.  Weight
068         * is used as a proxy for volume in CoffeeMud.
069         * @see Container#capacity()
070         * @param newValue the maximum weight that can fit inside this container
071         */
072        public void setCapacity(int newValue);
073        
074        /**
075         * Returns whether there is anything at all in this container.
076         * @return true if there is anything in this container, false if it is empty
077         */
078        public boolean hasContent();
079        
080        /**
081         * Returns whether this container is allowed to contain the given object.
082         * @see Container#setContainTypes(long)
083         * @see Container#containTypes()
084         * @param I the item to check against the allowed content
085         * @return true if the given item can go in this container, false otherwise
086         */
087        public boolean canContain(Item I);
088        
089        /**
090         * Returns whether the given item is, in fact, inside this container, even
091         * recursively.  So if the given item is in a container, and that container
092         * is in THIS container, it would still be inside.
093         * @param I the item to check against the content
094         * @return true if the item is somewhere inside, false otherwise
095         */
096        public boolean isInside(Item I);
097        
098        /**
099         * Returns a bitmap of the types of things that this container can hold.
100         * @see Container#setContainTypes(long)
101         * @see Container#CONTAIN_ANYTHING
102         * @see Container#CONTAIN_DESCS
103         * @return a bitmap of the types of things that this container can hold
104         */
105        public long containTypes();
106        
107        /**
108         * Sets a bitmap of the types of things that this container can hold.
109         * @see Container#containTypes()
110         * @see Container#CONTAIN_ANYTHING
111         * @see Container#CONTAIN_DESCS
112         * @param containTypes a bitmap of the types of things that this container can hold.
113         */
114        public void setContainTypes(long containTypes);
115        
116        /**
117         * Empties this container into its owner.
118         * @param flatten if true, will also remove all recursive items from their containers
119         */
120        public void emptyPlease(boolean flatten);
121        
122        /**
123         * Container type that overrides all others -- the container can hold anything!
124         * @see Container#setContainTypes(long)
125         * @see Container#containTypes()
126         */
127        public static final int CONTAIN_ANYTHING=0;
128        /** Container Type flag that means the container can hold liquids */
129        public static final int CONTAIN_LIQUID=1;
130        /** Container Type flag that means the container can hold coins */
131        public static final int CONTAIN_COINS=2;
132        /** Container Type flag that means the container can hold swords */
133        public static final int CONTAIN_SWORDS=4;
134        /** Container Type flag that means the container can hold daggers */
135        public static final int CONTAIN_DAGGERS=8;
136        /** Container Type flag that means the container can hold other weapons */
137        public static final int CONTAIN_OTHERWEAPONS=16;
138        /** Container Type flag that means the container can hold one handed weapons */
139        public static final int CONTAIN_ONEHANDWEAPONS=32;
140        /** Container Type flag that means the container can hold corpses */
141        public static final int CONTAIN_BODIES=64;
142        /** Container Type flag that means the container can hold books and such */
143        public static final int CONTAIN_READABLES=128;
144        /** Container Type flag that means the container can hold scrolls */
145        public static final int CONTAIN_SCROLLS=256;
146        /** Container Type flag that means the container can hold caged animals */
147        public static final int CONTAIN_CAGED=512;
148        /** Container Type flag that means the container can hold keys */
149        public static final int CONTAIN_KEYS=1024;
150        /** Container Type flag that means the container can hold drinkables */
151        public static final int CONTAIN_DRINKABLES=2048;
152        /** Container Type flag that means the container can hold clothing */
153        public static final int CONTAIN_CLOTHES=4096;
154        /** Container Type flag that means the container can hold smokeables */
155        public static final int CONTAIN_SMOKEABLES=8192;
156        /** Container Type flag that means the container can hold space ship components */
157        public static final int CONTAIN_SSCOMPONENTS=16384;
158        /** Container Type flag that means the container can hold shoes */
159        public static final int CONTAIN_FOOTWEAR=32768;
160        /** Container Type flag that means the container can hold raw resources */
161        public static final int CONTAIN_RAWMATERIALS=65536;
162        /** Container Type flag that means the container can hold foods */
163        public static final int CONTAIN_EATABLES=131072;
164        
165        /**
166         * Ordinal list of the names of all the container bitmask types.
167         * @see Container#setContainTypes(long)
168         * @see Container#containTypes()
169         */
170        public static final String[] CONTAIN_DESCS={"ANYTHING",
171                                                                                                "LIQUID",
172                                                                                                "COINS",
173                                                                                                "SWORDS",
174                                                                                                "DAGGERS",
175                                                                                                "OTHER WEAPONS",
176                                                                                                "ONE-HANDED WEAPONS",
177                                                                                                "BODIES",
178                                                                                                "READABLES",
179                                                                                                "SCROLLS",
180                                                                                                "CAGED ANIMALS",
181                                                                                                "KEYS",
182                                                                                                "DRINKABLES",
183                                                                                                "CLOTHES",
184                                                                                                "SMOKEABLES",
185                                                                                                "SS COMPONENTS",
186                                                                                                "FOOTWEAR",
187                                                                                                "RAWMATERIALS",
188                                                                                                "EATABLES"
189                                                                                                };
