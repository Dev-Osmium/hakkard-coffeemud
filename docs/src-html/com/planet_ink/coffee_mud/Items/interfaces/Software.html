033/**
034 * Software is a plain technical item that monitors and manipulates
035 * other space ship components and systems, while also managing
036 * a user interface for players/mobs monitoring the software.
037 * These user interfaces are divided into menus.  Each software
038 * can specify a "parent menu", referring to another software
039 * or system that gives access to this software.
040 * @author Bo Zimmerman
041 *
042 */
043public interface Software extends Item, Technical
044{
045        /**
046         * The parent menu that this software gets access from.
047         * When Software is available from root, "" is returned.
048         * @return parent menu that this software gets access from
049         */
050        public String getParentMenu();
051        
052        /**
053         * Returns the internal name of this software.
054         * @return the internal name of this software.
055         */
056        public String getInternalName();
057        
058        /**
059         * Returns whether the given computer-entry command
060         * should be responded to by THIS software object
061         * on an activation command.
062         * @param word the computer-entry command entered
063         * @return true if this software should respond.
064         */
065        public boolean isActivationString(String word);
066        
067        /**
068         * Returns whether the given computer-entry command
069         * should be responded to by THIS software object
070         * on a deactivation command.
071         * @param word the computer-entry command entered
072         * @return true if this software should respond.
073         */
074        public boolean isDeActivationString(String word);
075        
076        /**
077         * Returns whether the given computer-entry command
078         * should be responded to by THIS software object
079         * on a WRITE/ENTER command.
080         * @param word the computer-entry command
081         * @param isActive true if the software is already activated
082         * @return true if this software can respond
083         */
084        public boolean isCommandString(String word, boolean isActive);
085        
086        /**
087         * Returns the menu name of this software, so that it can
088         * be identified on its parent screen.
089         * @return the menu name of this software
090         */
091        public String getActivationMenu();
092        
093        /**
094         * Adds a new message to the screen from this program, which
095         * will be received by those monitoring the computer
096         * @see Software#getScreenMessage()
097         * @see Software#getCurrentScreenDisplay()
098         * @param msg the new message for the screen
099         */
100        public void addScreenMessage(String msg);
101        
102        /**
103         * Returns any new messages from this program when
104         * it is activated and on the screen.  Seen by those
105         * monitoring the computer.
106         * @see Software#addScreenMessage(String)
107         * @see Software#getCurrentScreenDisplay()
108         * @return the new screen messages
109         */
110        public String getScreenMessage();
111        
112        /**
113         * Returns the full screen appearance of this program when
114         * it is activated and on the screen.  Only those intentially
115         * looking at the screen again, or forced by the program, will
116         * see this larger message.
117         * @see Software#addScreenMessage(String)
118         * @see Software#getScreenMessage()
119         * @return the entire screen message
120         */
121        public String getCurrentScreenDisplay();
122        
123        /**
124         * Software runs on computers, and computers run on power systems.
125         * This method tells the software what the power system "circuit" key
126         * is that the computer host is running on, allowing the software to
127         * find other equipment on the same circuit and control it.
128         * @param key the circuit key
129         */
130        public void setCircuitKey(String key);
131}
