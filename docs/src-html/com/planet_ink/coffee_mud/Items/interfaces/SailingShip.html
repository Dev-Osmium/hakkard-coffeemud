036/**
037 * A Sailing Ship, which is an object that's boardable, rooms are attached
038 * to it so you can get in, and its an item that can appear in rooms, have
039 * speed and direction, be in combat, etc.
040 * @author Bo Zimmerman
041 *
042 */
043public interface SailingShip extends BoardableShip, Item, Combatant, Rideable
044{
045        public static final int COURSE_STEER_MASK = 256;
046
047        /**
048         * Returns which direction the ship is currently facing.
049         * @return the direction the ship is facing.
050         */
051        public int getDirectionFacing();
052
053        /**
054         * Sets which direction the ship is currently facing.
055         * @param dir the direction the ship is facing.
056         */
057        public void setDirectionFacing(int dir);
058
059        /**
060         * Returns whether the anchor is down, thus holding the ship in place.
061         * @return true if the anchor is down, holding the ship in place.
062         */
063        public boolean isAnchorDown();
064
065        /**
066         * Returns this ships max speed, typically &gt;= 1
067         * @return this ships max speed, typically &gt;= 1
068         */
069        public int getShipSpeed();
070
071        /**
072         * Sets whether the anchor is down, thus holding the ship in place.
073         * @param truefalse true if the anchor is down, false if the anchor is up
074         */
075        public void setAnchorDown(boolean truefalse);
076
077        /**
078         * Returns the mapping of this ships siege weapons to the coordinates
079         * they are presently aimed at.  Each coordinate is int[x,y]
080         * @return the pairings of weapons to coordinates aimed at
081         */
082        public PairList<Weapon,int[]> getSiegeWeaponAimings();
083
084        /**
085         * Returns the future course of this ship.  A stop-course direction is
086         * always -1, so it is typically the last entry.  Otherwise, each entry
087         * is a compass direction, possibly masked by COURSE_STEER_MASK in order
088         * to specify that it is a TURN ONLY.  Directions not marked as turns
089         * are automatic movements in that direction.
090         * @see SailingShip#setCurrentCourse(List)
091         * @return the future course of this ship.
092         */
093        public List<Integer> getCurrentCourse();
094
095        /**
096         * Sets the future course of this ship.  A stop-course direction is
097         * always -1, so it is typically the last entry.  Otherwise, each entry
098         * is a compass direction, possibly masked by COURSE_STEER_MASK in order
099         * to specify that it is a TURN ONLY.  Directions not marked as turns
100         * are automatic movements in that direction.
101         * @see SailingShip#getCurrentCourse()
102         * @param course the new course to set.
103         */
104        public void setCurrentCourse(List<Integer> course);
105}
