033/**
034 * Musical instruments are typically used by Bard types
035 * to make their magic.  Having hard coded types allows
036 * skills and such to do thing based on these types.
037 * @author Bo Zimmerman
038 */
039public interface MusicalInstrument extends Item
040{
041        /**
042         * The enum of instrument types.  These are general
043         * categories of instruments, so hopefully nothing is
044         * missing.  They are used by skills to identify an
045         * instrument category.
046         * @author Bo Zimmerman
047         *
048         */
049        public enum InstrumentType
050        {
051                CLARINETS,
052                CYMBALS,
053                DRUMS,
054                FLUTES,
055                GUITARS,
056                HARMONICAS,
057                HARPS,
058                HORNS,
059                OBOES,
060                ORGANS,
061                PIANOS,
062                TROMBONES,
063                TRUMPETS,
064                TUBAS,
065                VIOLINS,
066                WOODS,
067                XYLOPHONES,
068                OTHER_INSTRUMENT_TYPE
069                ;
070                private static String[] valueNames = null;
071                public static String[] valueNames()
072                {
073                        if(valueNames != null)
074                        {
075                                return valueNames;
076                        }
077                        valueNames = new String[InstrumentType.values().length];
078                        int i=0;
079                        for(InstrumentType type : InstrumentType.values())
080                                valueNames[i++] = type.name();
081                        return valueNames;
082                }
083        }
084        
085        /**
086         * Returns the instrument type of this instrument
087         * @see InstrumentType
088         * @see MusicalInstrument#setInstrumentType(InstrumentType)
089         * @see MusicalInstrument#setInstrumentType(String)
090         * @see MusicalInstrument#setInstrumentType(int)
091         * @return the instrument type enum object of this instrument
092         */
093        public InstrumentType getInstrumentType();
094        
095        /**
096         * Returns the instrument type name of this instrument
097         * @see InstrumentType
098         * @see MusicalInstrument#setInstrumentType(InstrumentType)
099         * @see MusicalInstrument#setInstrumentType(String)
100         * @see MusicalInstrument#setInstrumentType(int)
101         * @return the instrument type name of this instrument
102         */
103        public String getInstrumentTypeName();
104        
105        /**
106         * Sets the instrument type of this instrument
107         * @see InstrumentType
108         * @see MusicalInstrument#setInstrumentType(InstrumentType)
109         * @see MusicalInstrument#setInstrumentType(String)
110         * @see MusicalInstrument#setInstrumentType(int)
111         * @param type the instrument type enum object of this instrument
112         */
113        public void setInstrumentType(InstrumentType type);
114        
115        /**
116         * Sets the instrument type of this instrument
117         * @see InstrumentType
118         * @see MusicalInstrument#setInstrumentType(InstrumentType)
119         * @see MusicalInstrument#setInstrumentType(String)
120         * @see MusicalInstrument#setInstrumentType(int)
121         * @param typeName the instrument type name for this instrument
122         */
123        public void setInstrumentType(String typeName);
124        
125        /**
126         * Sets the instrument type of this instrument
127         * @see InstrumentType
128         * @see MusicalInstrument#setInstrumentType(InstrumentType)
129         * @see MusicalInstrument#setInstrumentType(String)
130         * @see MusicalInstrument#setInstrumentType(int)
131         * @param typeOrdinal the instrument ordinal of this instrument
132         */
133        public void setInstrumentType(int typeOrdinal);
134}
