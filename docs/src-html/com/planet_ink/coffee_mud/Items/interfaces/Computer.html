001package com.planet_ink.coffee_mud.Items.interfaces;
002
003import java.util.List;
004
005import com.planet_ink.coffee_mud.MOBS.interfaces.MOB;
006
007/**
008 * A computer is a particular type of electronics panel that holds
009 * software, and has readers who monitor the messages this panel
010 * generates, as well as use the Type command to enter data into
011 * this panel's software.
012 * @see Software
013 * @see ElecPanel
014 * @author Bo Zimmerman
015 *
016 */
017public interface Computer extends ElecPanel
018{
019        /**
020         * Returns the list of Software objects installed in this computer.
021         * @see Software
022         * @return the list of Software objects installed in this computer
023         */
024        public List<Software> getSoftware();
025        
026        /**
027         * Returns the list of mobs currently monitoring the output of this
028         * computers software.
029         * @return the list of mobs currently monitoring the output
030         */
031        public List<MOB> getCurrentReaders();
032        
033        /**
034         * Forces all the current readers to "read" the computer, typically
035         * seeing the menu.
036         * @see Computer#getCurrentReaders()
037         * @see Computer#forceReadersSeeNew()
038         */
039        public void forceReadersMenu();
040        
041        /**
042         * Forces all the current readers to see any new messages that 
043         * should be seen by anyone monitoring the computer.
044         * @see Computer#getCurrentReaders()
045         * @see Computer#forceReadersMenu()
046         */
047        public void forceReadersSeeNew();
048        
049        /**
050         * Most software supports different levels of menu, and some software
051         * is even a sub-menu unto itself.  This method forces the system to
052         * recognize one of those menus as current.  The software takes it
053         * from there.
054         * @see Computer#getActiveMenu()
055         * @param internalName the menu to set as current and active
056         */
057        public void setActiveMenu(String internalName);
058        
059        /**
060         * Most software supports different levels of menu, and some software
061         * is even a sub-menu unto itself.  This method returns the current
062         * active menu.
063         * @see Computer#setActiveMenu(String)
064         * @return internalName the menu to set as current and active
065         */
066        public String getActiveMenu();
067}
