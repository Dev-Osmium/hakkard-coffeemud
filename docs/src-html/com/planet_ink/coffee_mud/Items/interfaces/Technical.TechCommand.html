140        /**
141         * A TechCommand is an internal message that is only understood between electrical
142         * objects, typically ship components, but potentially between computer components
143         * of all sorts.  
144         * @author Bo Zimmerman
145         *
146         */
147        public static enum TechCommand
148        {
149                THRUST(TechComponent.ShipDir.class, Double.class),
150                ACCELLERATED(TechComponent.ShipDir.class, Double.class),
151                ACCELLLERATION(TechComponent.ShipDir.class, Double.class, Boolean.class),
152                COMPONENTFAILURE(Technical.TechType.class, String[].class),
153                SENSE(),
154                AIRREFRESH(Double.class, Integer.class), 
155                POWERSET(Long.class),
156                WEAPONTARGETSET(Double.class,Double.class),
157                WEAPONFIRE(),
158                SHIELDSET(ShipDir.class,Integer.class),
159                GRAVITYCHANGE(Boolean.class);
160                private final Class<?>[]        parms;
161
162                private TechCommand(Class<?>... parms)
163                {
164                        this.parms = parms;
165                }
166
167                /**
168                 * Returns the form of the parameters of this tech command
169                 * @return  the form of the parameters of this tech command
170                 */
171                public Class<?>[] getParms()
172                {
173                        return parms;
174                }
175
176                /**
177                 * Creates a new tech command of this enums type using the given parameters,
178                 * and returns the message as a string, or "" if an error occurred due to
179                 * bad parameters.
180                 * @param parts the parameters, which must be perfectly valid for this enum
181                 * @return the encoded tech command
182                 */
183                public String makeCommand(Object... parts)
184                {
185                        if ((parts == null) || (parts.length != parms.length))
186                                return "";
187                        final StringBuilder str = new StringBuilder(toString());
188                        for (int i = 0; i < parms.length; i++)
189                        {
190                                if (parts[i] == null)
191                                        return "";
192                                else 
193                                if (parms[i] == String[].class)
194                                {
195                                        for (; i < parms.length; i++)
196                                                str.append(" ").append(parts[i].toString());
197                                        break;
198                                }
199                                else 
200                                if (!parms[i].isAssignableFrom(parts[i].getClass()))
201                                        return "";
202                                else
203                                if(parts[i] instanceof Double)
204                                {
205                                        final DecimalFormat df = new DecimalFormat("#");
206                                        df.setMaximumFractionDigits(8);
207                                        str.append(" ").append(df.format(((Double)parts[i]).doubleValue()));
208                                }
209                                else
210                                        str.append(" ").append(parts[i].toString());
211                        }
212                        return str.toString();
213                }
214
215                /**
216                 * When a tech command of this enum type is received with its parameters,
217                 * the parameters are parsed into a list and passed to this method to
218                 * confirm their types, translate them to the appropriate types, and
219                 * return the parameters as their original objects in an Object array.
220                 * null is returned if anything goes wrong
221                 * @param parts the command parameters as a string list
222                 * @return the command parameters as their original objects, or null
223                 */
224                @SuppressWarnings({ "unchecked", "rawtypes" })
225                public Object[] confirmAndTranslate(String[] parts)
226                {
227                        if (parts.length != parms.length + 1)
228                                return null;
229                        final Object[] resp = new Object[parts.length - 1];
230                        for (int i = 0; i < parms.length; i++)
231                        {
232                                if (parms[i].isEnum())
233                                {
234                                        resp[i] = CMath.s_valueOf((Class<? extends Enum>) parms[i], parts[i + 1]);
235                                        if (resp[i] == null)
236                                                return null;
237                                }
238                                else 
239                                if (Integer.class.isAssignableFrom(parms[i]) || Long.class.isAssignableFrom(parms[i]))
240                                {
241                                        if (!CMath.isLong(parts[i + 1]))
242                                                return null;
243                                        if (Integer.class.isAssignableFrom(parms[i]))
244                                                resp[i] = Integer.valueOf(parts[i + 1]);
245                                        else
246                                                resp[i] = Long.valueOf(parts[i + 1]);
247                                }
248                                else 
249                                if (Double.class.isAssignableFrom(parms[i]) || Float.class.isAssignableFrom(parms[i]))
250                                {
251                                        if (!CMath.isNumber(parts[i + 1]))
252                                                return null;
253                                        if (Float.class.isAssignableFrom(parms[i]))
254                                                resp[i] = Float.valueOf(parts[i + 1]);
255                                        else
256                                                resp[i] = Double.valueOf(parts[i + 1]);
257                                }
258                                else 
259                                if (Boolean.class.isAssignableFrom(parms[i]))
260                                {
261                                        if (!CMath.isBool(parts[i + 1]))
262                                                return null;
263                                        resp[i] = Boolean.valueOf(parts[i + 1]);
264                                }
265                                else 
266                                if (String.class.isAssignableFrom(parms[i]))
267                                {
268                                        resp[i] = parts[i + 1];
269                                }
270                                else 
271                                if (String[].class.isAssignableFrom(parms[i]))
272                                {
273                                        final StringBuilder rebuilt = new StringBuilder(parts[i + 1]);
274                                        for (i = i + 2; i < parts.length; i++)
275                                                rebuilt.append(" ").append(parts[i]);
276                                        resp[i] = rebuilt.toString();
277                                        return resp;
278                                }
279                        }
280                        return resp;
281                }
282
283                /**
284                 * Returns the techcommand object that matches the first word in this
285                 * parsed command string list.  Only the first entry matters.
286                 * @param parts the entire command string list
287                 * @return the techcommand that matches the first string, or null
288                 */
289                public static TechCommand findCommand(String[] parts)
290                {
291                        if (parts.length == 0)
292                                return null;
293                        return (TechCommand) CMath.s_valueOf(TechCommand.class, parts[0].toUpperCase().trim());
294                }
295        }
