033/**
034 * A Boardable Ship, which is an object that's boardable, rooms are attached
035 * to it so you can get in.
036 * @author Bo Zimmerman
037 *
038 */
039public interface BoardableShip extends Environmental
040{
041        /**
042         * Designates that this ship is landed and docked in the given 
043         * planetary room.
044         * @param R the coordinate toom in which the ship is docked.
045         */
046        public void dockHere(Room R);
047
048        /**
049         * When a boardable ship area is created, it is bound to an item
050         * that resides in a room when docked, or possibly, always. 
051         * @param dockableItem the item that acts as dockable item 
052         */
053        public void setDockableItem(Item dockableItem);
054        
055        /**
056         * Designates that this ship is no longer docked, and whether it
057         * should also be moved into its na
058         * @param moveToOutside true to put in space/ocean,  or false to leave in limbo
059         * @return one of the internal rooms that was connected to the dock
060         */
061        public Room unDock(boolean moveToOutside);
062
063        /**
064         * Returns the Room where this ship is docked, or NULL if in space.
065         * @return the Room where this ship is docked, or NULL if in space.
066         */
067        public Room getIsDocked();
068
069        /**
070         * Space ships are unique in having an Item stand-in for dirt-side access,
071         * as well as an Area object.  This method returns the area object that 
072         * represents the contents of the ship.
073         * @return the official area version of this ship
074         */
075        public Area getShipArea();
076
077        /**
078         * Space ships are unique in having an Item stand-in for dirt-side access,
079         * as well as an Area object.  This method sets the area object that 
080         * represents the contents of the ship.
081         * @param xml area xml for the ship
082         */
083        public void setShipArea(String xml);
084
085        /**
086         * Renames the ship to something else
087         * @param newName the new ship name
088         */
089        public void renameShip(String newName);
090
091        /**
092         * Returns the room ID of the ships home port.
093         * @return the ships home port
094         */
095        public String getHomePortID();
096
097        /**
098         * Sets the room ID of the ships home port
099         * @param portID the ships home port
100         */
101        public void setHomePortID(String portID);
102
103        /**
104         * Ships are unique in having an Item stand-in for port-side access,
105         * as well as an Area object.  This method returns that Item.
106         * @return the official space version of this ship
107         */
108        public Item getShipItem();
109}
