033/**
034 * A Light is an item that can emit light for some period of time.
035 * Lights can be put out, but when they go out of their own accord
036 * they can be destroyed automatically, or not.
037 * @author Bo Zimmerman
038 */
039public interface Light extends Item
040{
041        /**
042         * Sets the duration, in ticks, that this light object will emit
043         * light before stopping for whatever reason.
044         * @see Light#getDuration()
045         * @param duration the duration of the light source, in ticks.
046         */
047        public void setDuration(int duration);
048
049        /**
050         * Gets the duration, in ticks, that this light object will emit
051         * light before stopping for whatever reason.
052         * @see Light#setDuration(int)
053         * @return duration the duration of the light source, in ticks.
054         */
055        public int getDuration();
056        
057        /**
058         * Gets whether this Light item automatically destroys itself after
059         * it is lit, and then its duration runs out.
060         * @see Light#setDestroyedWhenBurntOut(boolean)
061         * @return true if it is destroyed after use, false otherwise 
062         */
063        public boolean destroyedWhenBurnedOut();
064        
065        /**
066         * Sets whether this Light item automatically destroys itself after
067         * it is lit, and then its duration runs out.
068         * @see Light#destroyedWhenBurnedOut()
069         * @param truefalse true if it is destroyed after use, false otherwise 
070         */
071        public void setDestroyedWhenBurntOut(boolean truefalse);
072        
073        /**
074         * Returns true if this particular light item is sensitive to rain
075         * and nasty weather such that rain or water will cause it to stop
076         * being lit.
077         * @return true if it goes out in the rain
078         */
079        public boolean goesOutInTheRain();
080        
081        /**
082         * Gets whether this light item is presently emitting any light. 
083         * @see Light#light(boolean)
084         * @return true if it is lit and emitting light
085         */
086        public boolean isLit();
087        
088        /**
089         * Sets whether this light item is presently emitting any light. 
090         * @see Light#isLit()
091         * @param isLit true if it is lit and emitting light
092         */
093        public void light(boolean isLit);
094
095}
