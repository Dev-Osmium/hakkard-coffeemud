033/**
034 * An item that can be used to do damage to another mob.
035 * That's what a weapon is.
036 * They are classified by "damage type", which speaks generally
037 * to the way the weapon hurts a mob, and a "weapon classification",
038 * which speaks to the way in which the weapon is used, or the skills
039 * required.
040 * @author Bo Zimmerman
041 */
042public interface Weapon extends Item
043{
044        /** One of the weapon type, denotes natural weapon damage */
045        public final static int TYPE_NATURAL=0;
046        /** One of the weapon type, denotes slashing weapon damage */
047        public final static int TYPE_SLASHING=1;
048        /** One of the weapon type, denotes poking weapon damage */
049        public final static int TYPE_PIERCING=2;
050        /** One of the weapon type, denotes blunt weapon damage */
051        public final static int TYPE_BASHING=3;
052        /** One of the weapon type, denotes fire weapon damage */
053        public final static int TYPE_BURNING=4;
054        /** One of the weapon type, denotes explosive weapon damage */
055        public final static int TYPE_BURSTING=5;
056        /** One of the weapon type, denotes shooting weapon damage */
057        public final static int TYPE_SHOOT=6;
058        /** One of the weapon type, denotes cold weapon damage */
059        public final static int TYPE_FROSTING=7;
060        /** One of the weapon type, denotes poisony weapon damage */
061        public final static int TYPE_GASSING=8;
062        /** One of the weapon type, denotes acid weapon damage */
063        public final static int TYPE_MELTING=9;
064        /** One of the weapon type, denotes electric weapon damage */
065        public final static int TYPE_STRIKING=10;
066        /** One of the weapon type, denotes light weapon damage */
067        public final static int TYPE_LASERING=11;
068        /** One of the weapon type, denotes sonic weapon damage */
069        public final static int TYPE_SONICING=12;
070
071        /**
072         * Description code words for the weapon/damage types, indexed
073         * by the weapon/damage type values.
074         */
075        public final static String[] TYPE_DESCS=
076        {
077                "NATURAL",
078                "SLASHING",
079                "PIERCING",
080                "BASHING",
081                "BURNING",
082                "BURSTING",
083                "SHOOTING",
084                "FROSTING",
085                "GASSING",
086                "MELTING",
087                "STRIKING",
088                "LASERING",
089                "SONICING"
090        };
091
092        /** One of the weapon classification, denotes an axe swinging type weapon */
093        public final static int CLASS_AXE=0;
094        /** One of the weapon classification, denotes a blunt type weapon */
095        public final static int CLASS_BLUNT=1;
096        /** One of the weapon classification, denotes an edged type weapon */
097        public final static int CLASS_EDGED=2;
098        /** One of the weapon classification, denotes a flailed type weapon */
099        public final static int CLASS_FLAILED=3;
100        /** One of the weapon classification, denotes a hammer swinging type weapon */
101        public final static int CLASS_HAMMER=4;
102        /** One of the weapon classification, denotes a natural type weapon */
103        public final static int CLASS_NATURAL=5;
104        /** One of the weapon classification, denotes a polearm type weapon */
105        public final static int CLASS_POLEARM=6;
106        /** One of the weapon classification, denotes a ranged type weapon */
107        public final static int CLASS_RANGED=7;
108        /** One of the weapon classification, denotes a sword type weapon */
109        public final static int CLASS_SWORD=8;
110        /** One of the weapon classification, denotes a dagger type weapon */
111        public final static int CLASS_DAGGER=9;
112        /** One of the weapon classification, denotes a staff type weapon */
113        public final static int CLASS_STAFF=10;
114        /** One of the weapon classification, denotes a thrown type weapon */
115        public final static int CLASS_THROWN=11;
116        
117        /**
118         * Description code words for the weapon classifications, indexed
119         * by the weapon classification code values.
120         */
121        public final static String[] CLASS_DESCS=
122        {
123                "AXE",
124                "BLUNT",
125                "EDGED",
126                "FLAILED",
127                "HAMMER",
128                "KARATE",
129                "POLEARM",
130                "RANGED",
131                "SWORD",
132                "DAGGER",
133                "STAFF",
134                "THROWN"
135        };
136
137        /**
138         * Gets the type of damage this weapon does, from the weapon damage
139         * type list of codes.
140         * @see Weapon#TYPE_DESCS
141         * @return get weapon damage type code
142         */
143        public int weaponDamageType();
144
145        /**
146         * Sets the type of damage this weapon does, from the weapon damage
147         * type list of codes.
148         * @see Weapon#TYPE_DESCS
149         * @param newType get weapon damage type code
150         */
151        public void setWeaponDamageType(int newType);
152        
153        /**
154         * Gets the general classification of this weapon, denoting how the
155         * weapon is used and the skills required.  The code comes from the
156         * weapon class list of codes.
157         * @see Weapon#CLASS_DESCS
158         * @return the weapon classification code
159         */
160        public int weaponClassification();
161        
162        /**
163         * Sets the general classification of this weapon, denoting how the
164         * weapon is used and the skills required.  The code comes from the
165         * weapon class list of codes.
166         * @see Weapon#CLASS_DESCS
167         * @param newClassification the weapon classification code
168         */
169        public void setWeaponClassification(int newClassification);
170
171        /**
172         * Sets the minimum and maximum range of this weapon. 
173         * Combined with the size of the room, and who attacks first,
174         * this can set the distance between two combatants.
175         * @see Environmental#maxRange()
176         * @see Environmental#minRange()
177         * @param min the minimum range 0=melee
178         * @param max the maximum range 0=melee
179         */
180        public void setRanges(int min, int max);
181
182        /**
183         * Gets the string that would be shown if someone gets a hit with
184         * this weapon. 
185         * @see Weapon#missString()
186         * @param damageAmount the amount of damage done with this weapon
187         * @return the message string to show someone hit with this weapon
188         */
189        public String hitString(int damageAmount);
190
191        /**
192         * Gets the string that would be shown if someone misses with
193         * this weapon. 
194         * @see Weapon#hitString(int)
195         * @return the message string to show someone misses with this weapon
196         */
197        public String missString();
198}
