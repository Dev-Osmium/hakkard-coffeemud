081        @Override
082        public void setParms(String newParms)
083        {
084                if(newParms.startsWith("+"))
085                {
086                        final List<String> V=CMParms.parseSemicolons(newParms.substring(1),false);
087                        final StringBuffer rsc=new StringBuffer("");
088                        for(int v=0;v<V.size();v++)
089                                rsc.append(V.get(v)+"\n\r");
090                        final ChattyGroup[] addGroups=parseChatData(rsc);
091                        final ArrayList<ChattyEntry> newList=new ArrayList<ChattyEntry>(addedChatEntries.length);
092                        for(final ChattyEntry CE : addedChatEntries)
093                                newList.add(CE);
094                        for(final ChattyGroup CG : addGroups)
095                        {
096                                for(final ChattyEntry CE : CG.entries)
097                                        newList.add(CE);
098                        }
099                        addedChatEntries = newList.toArray(new ChattyEntry[0]);
100                }
101                else
102                {
103                        super.setParms(newParms);
104                        addedChatEntries=new ChattyEntry[0];
105                }
106                responseQue=new SLinkedList<ChattyResponse>();
107                myChatGroup=null;
108        }

122        protected static ChattyGroup newChattyGroup(String name)
123        {
124                final char[] n = name.toCharArray();
125                int last=0;
126                char lookFor=' ';
127                final ArrayList<String> names=new ArrayList<String>();
128                final ArrayList<MaskingLibrary.CompiledZMask> masks=new ArrayList<MaskingLibrary.CompiledZMask>();
129                for(int i=0;i<n.length;i++)
130                {
131                        if(n[i]==lookFor)
132                        {
133                                final String s=name.substring(last,i).trim();
134                                last=i;
135                                if(s.length()>0)
136                                {
137                                        if(lookFor=='/')
138                                                masks.add(CMLib.masking().maskCompile(s));
139                                        else
140                                                names.add(s.toUpperCase());
141                                }
142                                if(lookFor=='/')
143                                        lookFor=' ';
144                        }
145                        else
146                        if(n[i]=='/')
147                        {
148                                lookFor='/';
149                                last=i;
150                        }
151                }
152                final String s=name.substring(last,name.length()).trim();
153                if(s.length()>0)
154                {
155                        if(lookFor=='/')
156                                masks.add(CMLib.masking().maskCompile(s));
157                        else
158                                names.add(s.toUpperCase());
159                }
160                if((names.size()==0)&&(masks.size()==0))
161                        names.add("");
162                return new ChattyGroup(names.toArray(new String[0]),masks.toArray(new MaskingLibrary.CompiledZMask[0]));
163        }

171        protected static ChattyGroup[] unprotectedChatGroupLoad(String parms)
172        {
173                ChattyGroup[] rsc=null;
174                String filename="chat.dat";
175                final int x=parms.indexOf('=');
176                if(x>0)
177                        filename=parms.substring(0,x);
178                rsc=(ChattyGroup[])Resources.getResource("MUDCHAT GROUPS-"+filename.toLowerCase());
179                if(rsc!=null)
180                        return rsc;
181                synchronized(("MUDCHAT GROUPS-"+filename.toLowerCase()).intern())
182                {
183                        rsc=(ChattyGroup[])Resources.getResource("MUDCHAT GROUPS-"+filename.toLowerCase());
184                        if(rsc!=null)
185                                return rsc;
186                        rsc=loadChatData(filename);
187                        Resources.submitResource("MUDCHAT GROUPS-"+filename.toLowerCase(),rsc);
188                        return rsc;
189                }

207        protected static ChattyGroup[] parseChatData(StringBuffer rsc)
208        {
209                final ArrayList<ChattyGroup> chatGroups = new ArrayList<ChattyGroup>();
210                ChattyGroup currentChatGroup=newChattyGroup("");
211                final ArrayList<ChattyEntry> currentChatEntries = new ArrayList<ChattyEntry>();
212                final ArrayList<ChattyEntry> tickyChatEntries = new ArrayList<ChattyEntry>();
213                ChattyEntry currentChatEntry=null;
214                final ArrayList<ChattyTestResponse> currentChatEntryResponses = new ArrayList<ChattyTestResponse>();
215
216                ChattyGroup otherChatGroup;
217                chatGroups.add(currentChatGroup);
218                String str=nextLine(rsc);
219                while(str!=null)
220                {
221                        if(str.length()>0)
222                        {
223                                char c=str.charAt(0);
224                                switch(c)
225                                {
226                                case '"':
227                                        Log.sysOut("MudChat",str.substring(1));
228                                        break;
229                                case '*':
230                                        if((str.length()==1)||("([{<".indexOf(str.charAt(1))<0))
231                                                break;
232                                        c=str.charAt(1);
233                                //$FALL-THROUGH$
234                                case '(':
235                                case '[':
236                                case '{':
237                                case '<':
238                                        if(currentChatEntry!=null)
239                                                currentChatEntry.responses = currentChatEntryResponses.toArray(new ChattyTestResponse[0]);
240                                        currentChatEntryResponses.clear();
241                                        currentChatEntry=new ChattyEntry(str);
242                                        if(currentChatEntry.expression.length()>0)
243                                        {
244                                                if(c=='<')
245                                                        tickyChatEntries.add(currentChatEntry);
246                                                else
247                                                        currentChatEntries.add(currentChatEntry);
248                                        }
249                                        else
250                                                currentChatEntry=null;
251                                        break;
252                                case '>':
253                                        if(currentChatEntry!=null)
254                                                currentChatEntry.responses = currentChatEntryResponses.toArray(new ChattyTestResponse[0]);
255                                        currentChatGroup.entries = currentChatEntries.toArray(new ChattyEntry[0]);
256                                        currentChatGroup.tickies = tickyChatEntries.toArray(new ChattyEntry[0]);
257                                        currentChatEntries.clear();
258                                        tickyChatEntries.clear();
259                                        currentChatGroup=newChattyGroup(str.substring(1).trim());
260                                        if(currentChatGroup == null)
261                                                return null;
262                                        chatGroups.add(currentChatGroup);
263                                        currentChatEntry=null;
264                                        break;
265                                case '@':
266                                        {
267                                                otherChatGroup=matchChatGroup(null,str.substring(1).trim(),chatGroups.toArray(new ChattyGroup[0]));
268                                                if(otherChatGroup==null)
269                                                        otherChatGroup=chatGroups.get(0);
270                                                if(otherChatGroup != currentChatGroup)
271                                                {
272                                                        for(final ChattyEntry CE : otherChatGroup.entries)
273                                                                currentChatEntries.add(CE);
274                                                        for(final ChattyEntry CE : otherChatGroup.tickies)
275                                                                tickyChatEntries.add(CE);
276                                                }
277                                                break;
278                                        }
279                                case '%':
280                                        {
281                                                final StringBuffer rsc2=new StringBuffer(Resources.getFileResource(str.substring(1).trim(),true).toString());
282                                                if (rsc2.length() < 1)
283                                                {
284                                                        Log.sysOut("MudChat", "Error reading resource " + str.substring(1).trim());
285                                                }
286                                                rsc.insert(0,rsc2.toString());
287                                                break;
288                                        }
289                                case '0':
290                                case '1':
291                                case '2':
292                                case '3':
293                                case '4':
294                                case '5':
295                                case '6':
296                                case '7':
297                                case '8':
298                                case '9':
299                                        if(currentChatEntry!=null)
300                                                currentChatEntryResponses.add(new ChattyTestResponse(str));
301                                        break;
302                                }
303                        }
304                        str=nextLine(rsc);
305                }
306                if(currentChatEntry!=null)
307                        currentChatEntry.responses = currentChatEntryResponses.toArray(new ChattyTestResponse[0]);
308                currentChatGroup.entries = currentChatEntries.toArray(new ChattyEntry[0]);
309                currentChatGroup.tickies = tickyChatEntries.toArray(new ChattyEntry[0]);
310                currentChatEntries.clear();
311                tickyChatEntries.clear();
312                return chatGroups.toArray(new ChattyGroup[0]);
313        }

332        public static String nextLine(StringBuffer tsc)
333        {
334                String ret=null;
335                if((tsc!=null)&&(tsc.length()>0))
336                {
337                        int y=tsc.toString().indexOf("\n\r");
338                        if(y<0)
339                                y=tsc.toString().indexOf("\r\n");
340                        if(y<0)
341                        {
342                                y=tsc.toString().indexOf("\n");
343                                if(y<0)
344                                        y=tsc.toString().indexOf("\r");
345                                if(y<0)
346                                {
347                                        tsc.setLength(0);
348                                        ret="";
349                                }
350                                else
351                                {
352                                        ret=tsc.substring(0,y).trim();
353                                        tsc.delete(0,y+1);
354                                }
355                        }
356                        else
357                        {
358                                ret=tsc.substring(0,y).trim();
359                                tsc.delete(0,y+2);
360                        }
361                }
362                return ret;
363
364        }

366        protected static ChattyGroup matchChatGroup(MOB meM, String myName, ChattyGroup[] chatGroups)
367        {
368                myName=myName.toUpperCase();
369                if(myName.equals("DEFAULT"))
370                        return chatGroups[0];
371                for(final ChattyGroup CG : chatGroups)
372                {
373                        if(CG.entries!=null)
374                        {
375                                for(final String name : CG.groupNames)
376                                {
377                                        if(name.equals(myName))
378                                                return CG;
379                                }
380                                if(meM != null)
381                                {
382                                        for(final MaskingLibrary.CompiledZMask mask : CG.groupMasks)
383                                        {
384                                                if(CMLib.masking().maskCheck(mask, meM, true))
385                                                        return CG;
386                                        }
387                                }
388                        }
389                }
390                return null;
391        }

393        protected ChattyGroup getMyBaseChatGroup(MOB forMe, ChattyGroup[] chatGroups)
394        {
395                if((myChatGroup!=null)&&(myOldName.equals(forMe.Name())))
396                        return myChatGroup;
397                myOldName=forMe.Name();
398                ChattyGroup matchedCG=null;
399                if(getParms().length()>0)
400                {
401                        final int x=getParms().indexOf('=');
402                        if(x<0)
403                                matchedCG=matchChatGroup(forMe,getParms(),chatGroups);
404                        else
405                                matchedCG=matchChatGroup(forMe,getParms().substring(x+1).trim(),chatGroups);
406                }
407                if(matchedCG!=null)
408                        return matchedCG;
409                matchedCG=matchChatGroup(forMe,CMLib.english().removeArticleLead(CMStrings.removeColors(myOldName.toUpperCase())),chatGroups);
410                if(matchedCG!=null)
411                        return matchedCG;
412                matchedCG=matchChatGroup(forMe,forMe.charStats().raceName(),chatGroups);
413                if(matchedCG!=null)
414                        return matchedCG;
415                matchedCG=matchChatGroup(forMe,forMe.charStats().getCurrentClass().name(),chatGroups);
416                if(matchedCG!=null)
417                        return matchedCG;
418                return chatGroups[0];
419        }

421        protected ChattyGroup getMyChatGroup(MOB forMe, ChattyGroup[] chatGroups)
422        {
423                if((myChatGroup!=null)&&(myOldName.equals(forMe.Name())))
424                        return myChatGroup;
425                ChattyGroup chatGrp=getMyBaseChatGroup(forMe,chatGroups);
426                if((addedChatEntries==null)||(addedChatEntries.length==0))
427                        return chatGrp;
428                final List<ChattyEntry> newEntries = new ArrayList<ChattyEntry>();
429                newEntries.addAll(Arrays.asList(addedChatEntries));
430                newEntries.addAll(Arrays.asList(chatGrp.entries));
431                chatGrp=chatGrp.clone();
432                chatGrp.entries = newEntries.toArray(new ChattyEntry[0]);
433                return chatGrp;
434        }

436        protected void queResponse(ArrayList<ChattyTestResponse> responses, MOB source, MOB target, String rest)
437        {
438                int total=0;
439                for(final ChattyTestResponse CR : responses)
440                        total+=CR.weight;
441                if(total == 0)
442                        return;
443                ChattyTestResponse selection=null;
444                int select=CMLib.dice().roll(1,total,0);
445                for(final ChattyTestResponse CR : responses)
446                {
447                        select-=CR.weight;
448                        if(select<=0)
449                        {
450                                selection=CR;
451                                break;
452                        }
453                }
454
455                if(selection!=null)
456                {
457                        for(String finalCommand : selection.responses)
458                        {
459                                if(finalCommand.trim().length()==0)
460                                        return;
461                                else
462                                if(finalCommand.startsWith(":"))
463                                {
464                                        finalCommand="emote "+finalCommand.substring(1).trim();
465                                        if(source!=null)
466                                                finalCommand=CMStrings.replaceAll(finalCommand," her "," "+source.charStats().hisher()+" ");
467                                }
468                                else
469                                if(finalCommand.startsWith("!"))
470                                        finalCommand=finalCommand.substring(1).trim();
471                                else
472                                if(finalCommand.startsWith("\""))
473                                        finalCommand="say \""+finalCommand.substring(1).trim()+"\"";
474                                else
475                                if(target!=null)
476                                        finalCommand="sayto \""+target.name()+"\" "+finalCommand.trim();
477
478                                finalCommand=CMStrings.replaceAll(finalCommand,"$r",rest);
479                                if(target!=null)
480                                        finalCommand=CMStrings.replaceAll(finalCommand,"$t",target.name());
481                                if(source!=null)
482                                        finalCommand=CMStrings.replaceAll(finalCommand,"$n",source.name());
483                                if(finalCommand.indexOf("$%")>=0)
484                                {
485                                        if(scriptEngine == null)
486                                        {
487                                                scriptEngine=(ScriptingEngine)CMClass.getCommon("DefaultScriptingEngine");
488                                                scriptEngine.setSavable(false);
489                                                scriptEngine.setVarScope("*");
490                                        }
491                                        final Object[] tmp = new Object[ScriptingEngine.SPECIAL_NUM_OBJECTS];
492                                        finalCommand = scriptEngine.varify(source, target, source, source, null, null, "", tmp, finalCommand);
493                                }
494                                finalCommand=CMStrings.replaceAll(finalCommand,"$$","$");
495                                Vector<String> V=CMParms.parse(finalCommand);
496                                for(final ChattyResponse R : responseQue)
497                                {
498                                        if(CMParms.combine(R.parsedCommand,1).equalsIgnoreCase(finalCommand))
499                                        {
500                                                V=null;
501                                                break;
502                                        }
503                                }
504                                if(V!=null)
505                                        responseQue.add(new ChattyResponse(V,RESPONSE_DELAY));
506                        }
507                }
508        }

510        protected boolean match(MOB speaker, String expression, String message, String[] rest)
511        {
512                final int l=expression.length();
513                if(l==0)
514                        return true;
515                if((expression.charAt(0)=='(')
516                &&(expression.charAt(l-1)==')'))
517                        expression=expression.substring(1,expression.length()-1).trim();
518
519                int end=0;
520                for (; ((end < expression.length()) && (("(&|~").indexOf(expression.charAt(end)) < 0)); end++)
521                {/* loop */
522                }
523                String check=null;
524                if(end<expression.length())
525                {
526                        check=expression.substring(0,end).trim();
527                        expression=expression.substring(end).trim();
528                }
529                else
530                {
531                        check=expression.trim();
532                        expression="";
533                }
534                boolean response=true;
535                if(check.startsWith("="))
536                {
537                        response=check.substring(1).trim().equalsIgnoreCase(message.trim());
538                        if(response)
539                                rest[0]="";
540                }
541                else
542                if(check.startsWith("^"))
543                {
544                        response=message.trim().startsWith(check.substring(1).trim());
545                        if(response)
546                                rest[0]=message.substring(check.substring(1).trim().length());
547                }
548                else
549                if(check.startsWith("/"))
550                {
551                        int expEnd=0;
552                        while((++expEnd)<check.length())
553                        {
554                                if(check.charAt(expEnd)=='/')
555                                        break;
556                        }
557                        response=CMLib.masking().maskCheck(check.substring(1,expEnd).trim(),speaker,false);
558                }
559                else
560                if(check.length()>0)
561                {
562                        final int x=message.toUpperCase().indexOf(check.toUpperCase().trim());
563                        response=(x>=0);
564                        if(response)
565                                rest[0]=message.substring(x+check.length());
566                }
567                else
568                {
569                        response=true;
570                        rest[0]=message;
571                }
572
573                if(expression.length()>0)
574                {
575                        if(expression.startsWith("("))
576                        {
577                                int expEnd=0;
578                                int parenCount=1;
579                                while(((++expEnd)<expression.length())&&(parenCount>0))
580                                {
581                                        if(expression.charAt(expEnd)=='(')
582                                                parenCount++;
583                                        else
584                                        if(expression.charAt(expEnd)==')')
585                                        {
586                                                parenCount--;
587                                                if(parenCount<=0)
588                                                        break;
589                                        }
590                                }
591                                if(expEnd<expression.length()&&(parenCount<=0))
592                                {
593                                        return response && match(speaker,expression.substring(1,expEnd).trim(),message,rest);
594                                }
595                                return response;
596                        }
597                        else
598                        if(expression.startsWith("&"))
599                                return response&&match(speaker,expression.substring(1).trim(),message,rest);
600                        else
601                        if(expression.startsWith("|"))
602                                return response||match(speaker,expression.substring(1).trim(),message,rest);
603                        else
604                        if(expression.startsWith("~"))
605                                return response&&(!match(speaker,expression.substring(1).trim(),message,rest));
606                }
607                return response;
608        }

610        @Override
611        public void executeMsg(final Environmental affecting, final CMMsg msg)
612        {
613                super.executeMsg(affecting,msg);
614
615                if((!canActAtAll(affecting))
616                ||(CMSecurity.isDisabled(CMSecurity.DisFlag.MUDCHAT)))
617                        return;
618                final MOB mob=msg.source();
619                final MOB monster=(MOB)affecting;
620                if((msg.source()==monster)
621                &&(msg.sourceMinor()==CMMsg.TYP_SPEAK)
622                &&(msg.othersMessage()!=null))
623                        lastThingSaid=CMStrings.getSayFromMessage(msg.othersMessage());
624                else
625                if((!mob.isMonster())
626                &&(CMLib.flags().canBeHeardSpeakingBy(mob,monster))
627                &&(CMLib.flags().canBeSeenBy(mob,monster))
628                &&(CMLib.flags().canBeSeenBy(monster,mob)))
629                {
630                        ArrayList<ChattyTestResponse> myResponses=null;
631                        myChatGroup=getMyChatGroup(monster,getChatGroups(getParms()));
632                        final String rest[]=new String[1];
633                        final boolean combat=((monster.isInCombat()))||(mob.isInCombat());
634
635                        String str;
636                        if((msg.targetMinor()==CMMsg.TYP_SPEAK)
637                        &&(msg.amITarget(monster)
638                           ||((msg.target()==null)
639                                  &&(mob.location()==monster.location())
640                                  &&(talkDown<=0)
641                                  &&(mob.location().numPCInhabitants()<3)))
642                        &&(CMLib.flags().canBeHeardSpeakingBy(mob,monster))
643                        &&(myChatGroup!=null)
644                        &&(lastReactedTo!=msg.source())
645                        &&(msg.sourceMessage()!=null)
646                        &&(msg.targetMessage()!=null)
647                        &&((str=CMStrings.getSayFromMessage(msg.sourceMessage()))!=null))
648                        {
649                                str=" "+CMLib.english().stripPunctuation(str)+" ";
650                                for(final ChattyEntry entry : myChatGroup.entries)
651                                {
652                                        final String expression=entry.expression;
653                                        if(entry.combatEntry)
654                                        {
655                                                if(!combat)
656                                                        continue;
657                                        }
658                                        else
659                                        if(combat)
660                                                continue;
661
662                                        if((expression.charAt(0)=='(')
663                                        &&(expression.charAt(expression.length()-1)==')'))
664                                        {
665                                                if(match(mob,expression.substring(1,expression.length()-1),str,rest))
666                                                {
667                                                        myResponses=new ArrayList<ChattyTestResponse>();
668                                                        myResponses.addAll(Arrays.asList(entry.responses));
669                                                        break;
670                                                }
671                                        }
672                                }
673                        }
674                        else // dont interrupt another mob
675                        if((msg.sourceMinor()==CMMsg.TYP_SPEAK)
676                        &&(mob.isMonster())  // this is another mob (not me) talking
677                        &&(CMLib.flags().canBeHeardSpeakingBy(mob,monster))
678                        &&(CMLib.flags().canBeSeenBy(mob,monster)))
679                           talkDown=TALK_WAIT_DELAY;
680                        else // dont parse unless we are done waiting
681                        if((CMLib.flags().canBeHeardMovingBy(mob,monster))
682                        &&(CMLib.flags().canBeSeenBy(mob,monster))
683                        &&(CMLib.flags().canBeSeenBy(monster,mob))
684                        &&(talkDown<=0)
685                        &&(lastReactedTo!=msg.source())
686                        &&(myChatGroup!=null))
687                        {
688                                str=null;
689                                char c1='[';
690                                char c2=']';
691                                if((msg.amITarget(monster)&&(msg.targetMessage()!=null)))
692                                        str=" "+msg.targetMessage()+" ";
693                                else
694                                if(msg.othersMessage()!=null)
695                                {
696                                        c1='{';
697                                        c2='}';
698                                        str=" "+msg.othersMessage()+" ";
699                                }
700                                if(str!=null)
701                                {
702                                        for(final ChattyEntry entry : myChatGroup.entries)
703                                        {
704                                                final String expression=entry.expression;
705                                                if(entry.combatEntry)
706                                                {
707                                                        if(!combat)
708                                                                continue;
709                                                }
710                                                else
711                                                if(combat)
712                                                        continue;
713                                                if((expression.charAt(0)==c1)
714                                                &&(expression.charAt(expression.length()-1)==c2))
715                                                {
716                                                        if(match(mob,expression.substring(1,expression.length()-1),str,rest))
717                                                        {
718                                                                myResponses=new ArrayList<ChattyTestResponse>();
719                                                                myResponses.addAll(Arrays.asList(entry.responses));
720                                                                break;
721                                                        }
722                                                }
723                                        }
724                                }
725                        }
726
727                        if(myResponses!=null)
728                        {
729                                lastReactedTo=msg.source();
730                                lastRespondedTo=msg.source();
731                                queResponse(myResponses,monster,mob,rest[0]);
732                        }
733                }
734        }

736        @Override
737        public boolean tick(final Tickable ticking, final int tickID)
738        {
739                super.tick(ticking,tickID);
740                if((tickID==Tickable.TICKID_MOB)
741                &&(ticking instanceof MOB)
742                &&(!CMSecurity.isDisabled(CMSecurity.DisFlag.MUDCHAT)))
743                {
744                        if(talkDown>0)
745                                talkDown--;
746
747                        if(tickDown>=0)
748                        {
749                                --tickDown;
750                                if(tickDown<0)
751                                {
752                                        myChatGroup=getMyChatGroup((MOB)ticking,getChatGroups(getParms()));
753                                }
754                        }
755                        if((myChatGroup!=null)&&(myChatGroup.tickies.length>0) && canActAtAll(ticking))
756                        {
757                                final boolean combat = ((MOB)ticking).isInCombat();
758                                ArrayList<ChattyTestResponse> myResponses=null;
759                                for(ChattyEntry entry : myChatGroup.tickies)
760                                {
761                                        if(entry.combatEntry)
762                                        {
763                                                if(!combat)
764                                                        continue;
765                                        }
766                                        else
767                                        if(combat)
768                                                continue;
769                                        if(entry.expression.length()>2)
770                                        {
771                                                int val=CMath.s_int(entry.expression.substring(1,entry.expression.length()-1).trim());
772                                                if((val==0)||(CMLib.dice().rollPercentage()>val))
773                                                        continue;
774                                        }
775                                        if(myResponses==null)
776                                                myResponses=new ArrayList<ChattyTestResponse>();
777                                        myResponses.addAll(Arrays.asList(entry.responses));
778                                }
779                                if(myResponses!=null)
780                                {
781                                        queResponse(myResponses,(MOB)ticking,(MOB)ticking,"");
782                                }
783                        }
784                        if(responseQue.size()==0)
785                                lastReactedTo=null;
786                        else
787                        if(!canActAtAll(ticking))
788                        {
789                                responseQue.clear();
790                                return true;
791                        }
792                        else
793                        {
794                                for(final Iterator<ChattyResponse> riter= responseQue.descendingIterator();riter.hasNext();)
795                                {
796                                        final ChattyResponse R = riter.next();
797                                        R.delay--;
798                                        if(R.delay<=0)
799                                        {
800                                                responseQue.remove(R);
801                                                ((MOB)ticking).doCommand(R.parsedCommand,MUDCmdProcessor.METAFLAG_FORCED);
802                                                lastReactedTo=null;
803                                                // you've done one, so get out before doing another!
804                                                break;
805                                        }
806                                }
807                        }
808                }
809                return true;
810        }
