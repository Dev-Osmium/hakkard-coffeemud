027public class FieryRoom extends ActiveTicker 
028{
029        @Override
030        public String ID()
031        {
032                return "FieryRoom"; 
033        }
034
035        @Override
036        protected int canImproveCode()
037        {
038                return Behavior.CAN_ROOMS; 
039        }
040
041        private String          newDisplay              = "";
042        private String          newDesc                 = "";
043        private int                     directDamage    = 10;
044        private int                     eqChance                = 0;
045        private int                     burnTicks               = 12;
046        private boolean         noStop                  = false;
047        private boolean         noNpc                   = false;
048        private boolean         noFireText              = false;
049
050        private String[] FireTexts = {"The fire here crackles and burns."};
051
052        public FieryRoom()
053        {
054                super();
055                minTicks = 5; maxTicks = 10; chance = 100;
056                tickReset();
057        }
058
059        @Override
060        public String accountForYourself()
061        {
062                return "on fire";
063        }
064
065        @Override
066        public void setParms(String newParms)
067        {
068                super.setParms(newParms);
069                newDisplay = CMParms.getParmStr(newParms, "Title", "A Charred Ruin");
070                newDesc = CMParms.getParmStr(newParms, "Description", "Whatever was once here is now nothing more than ash.");
071                directDamage = CMParms.getParmInt(newParms, "damage", 10);
072                eqChance = CMParms.getParmInt(newParms, "eqchance", 0);
073                burnTicks = CMParms.getParmInt(newParms, "burnticks", 12);
074                final Vector<String> V=CMParms.parse(newParms.toUpperCase());
075                noStop=(V.contains("NOSTOP"));
076                noNpc=(V.contains("NONPC"));
077                noFireText=(V.contains("NOFIRETEXT"));
078                setFireTexts();
079        }
080
081        private void setFireTexts()
082        {
083                final String[] newFireTexts = {"The fire here crackles and burns.",
084                                                                  "The intense heat of the fire here is "+(directDamage>0?"very painful":"very unpleasant")+".",
085                                                                  "The flames dance around you"+(eqChance>0?", licking at your clothes.":"."),
086                                                                  "The fire is burning out of control. You fear for your safety"+(noStop?".":" as it looks like this place is being completely consumed."),
087                                                                  "You hear popping and sizzling as something burns.",
088                                                                  "The smoke here is very thick and you worry about whether you will be able to breathe."};
089                FireTexts = newFireTexts;
090        }
091
092        @Override
093        public boolean tick(final Tickable ticking, final int tickID)
094        {
095                super.tick(ticking, tickID);
096                // on every tick, we may do damage OR eq handling.
097                if (!(ticking instanceof Room))
098                        return super.tick(ticking, tickID);
099
100                final Room room = (Room) ticking;
101                if (canAct(ticking, tickID))
102                {
103                        if ( (directDamage > 0) || (eqChance > 0))
104                        {
105                                // for each inhab, do directDamage to them.
106                                for (int i = 0; i < room.numInhabitants(); i++)
107                                {
108                                        final MOB inhab = room.fetchInhabitant(i);
109                                        if(inhab==null)
110                                                continue;
111                                        if (inhab.isMonster())
112                                        {
113                                                boolean reallyAffect = true;
114                                                if (noNpc)
115                                                {
116                                                        reallyAffect = false;
117                                                        if(inhab.getStartRoom()!=room)
118                                                        {
119                                                                final Set<MOB> group = inhab.getGroupMembers(new HashSet<MOB>());
120                                                                for (final Object element : group)
121                                                                {
122                                                                        final MOB follower = (MOB) element;
123                                                                        if (! (follower.isMonster()))
124                                                                        {
125                                                                                reallyAffect = true;
126                                                                                break;
127                                                                        }
128                                                                }
129                                                        }
130                                                }
131                                                if (reallyAffect)
132                                                {
133                                                        dealDamage(inhab);
134                                                        if (CMLib.dice().rollPercentage() > eqChance)
135                                                                eqRoast(inhab);
136                                                }
137                                        }
138                                        else
139                                        {
140                                                if((!CMSecurity.isAllowed(inhab,inhab.location(),CMSecurity.SecFlag.ORDER))
141                                                &&(!CMSecurity.isAllowed(inhab,inhab.location(),CMSecurity.SecFlag.CMDROOMS)))
142                                                {
143                                                        dealDamage(inhab);
144                                                        if (CMLib.dice().rollPercentage() > eqChance)
145                                                                eqRoast(inhab);
146                                                }
147                                        }
148                                }
149                        }
150                        // % chance of burning each item in the room.
151                        roastRoom(room);
152                        // The tick happened.  If NOT NoFireText, Do flame emotes
153                        if(!noFireText)
154                        {
155                                final String pickedText=FireTexts[CMLib.dice().roll(1,FireTexts.length,0)-1];
156                                room.showHappens(CMMsg.MSG_OK_ACTION,pickedText);
157                        }
158                }
159                if (!noStop)
160                {
161                        if(burnTicks==0)
162                        {
163                                // NOSTOP is false.  This means the room gets set
164                                // to the torched text and the behavior goes away.
165                                room.setDisplayText(newDisplay);
166                                room.setDescription(newDesc);
167                                room.delBehavior(this);
168                        }
169                        else
170                                --burnTicks;
171                }
172                return super.tick(ticking, tickID);
173        }
174
175        private void dealDamage(MOB mob)
176        {
177                final MOB M=CMLib.map().getFactoryMOB(mob.location());
178                M.setName(L("fire"));
179                CMLib.combat().postDamage(M, mob, null, directDamage, CMMsg.MASK_ALWAYS | CMMsg.MASK_MALICIOUS|CMMsg.TYP_FIRE, Weapon.TYPE_BURNING,
180                                                        L("The fire here <DAMAGE> <T-NAME>!"));
181                M.destroy();
182        }
183
184        private void eqRoast(MOB mob)
185        {
186                final Item target = getSomething(mob);
187                if (target != null)
188                {
189                        final MOB M=CMLib.map().getFactoryMOB(mob.location());
190                        M.setName(L("fire"));
191                        switch (target.material() & RawMaterial.MATERIAL_MASK)
192                        {
193                                case RawMaterial.MATERIAL_GLASS:
194                                case RawMaterial.MATERIAL_METAL:
195                                case RawMaterial.MATERIAL_MITHRIL:
196                                case RawMaterial.MATERIAL_SYNTHETIC:
197                                case RawMaterial.MATERIAL_PRECIOUS:
198                                case RawMaterial.MATERIAL_ROCK:
199                                case RawMaterial.MATERIAL_UNKNOWN: {
200                                        // all these we'll make get hot and be dropped.
201                                        final int damage = CMLib.dice().roll(1, 6, 1);
202                                        CMLib.combat().postDamage(M, mob, null, damage, CMMsg.MASK_ALWAYS | CMMsg.MASK_MALICIOUS|CMMsg.TYP_FIRE, Weapon.TYPE_BURNING, target.name() + " <DAMAGE> <T-NAME>!");
203                                        if (CMLib.dice().rollPercentage() < mob.charStats().getStat(CharStats.STAT_STRENGTH))
204                                        {
205                                                CMLib.commands().postDrop(mob, target, false, false, false);
206                                        }
207                                        break;
208                                }
209                                default: {
210                                        final Ability burn = CMClass.getAbility("Burning");
211                                        if (burn != null)
212                                        {
213                                                mob.location().showHappens(CMMsg.MSG_OK_ACTION, L("@x1 begins to burn!",target.Name()));
214                                                burn.invoke(M, target, true, 0);
215                                                target.recoverPhyStats();
216                                        }
217                                        break;
218                                }
219                        }
220                        M.destroy();
221                }
222        }
223
224        private static void roastRoom(Room which)
225        {
226                final MOB mob=CMLib.map().getFactoryMOB(which);
227                mob.setName(CMLib.lang().L("fire"));
228                for(int i=0;i<which.numItems();i++)
229                {
230                        final Item target=which.getItem(i);
231                        if(target != null)
232                        {
233                                final Ability burn = CMClass.getAbility("Burning");
234                                if((burn != null)&&(CMLib.dice().rollPercentage()>60))
235                                {
236                                        which.showHappens(CMMsg.MSG_OK_ACTION,CMLib.lang().L("@x1 begins to burn!",target.Name()));
237                                        burn.invoke(mob,target,true,0);
238                                        target.recoverPhyStats();
239                                }
240                        }
241                }
242                mob.destroy();
243        }
244
245        private static Item getSomething(MOB mob)
246        {
247                final Vector<Item> good = new Vector<Item>();
248                final Vector<Item> great = new Vector<Item>();
249                Item target = null;
250                for (int i = 0; i < mob.numItems(); i++)
251                {
252                        final Item I = mob.getItem(i);
253                        if (I.amWearingAt(Wearable.IN_INVENTORY))
254                                good.addElement(I);
255                        else
256                                great.addElement(I);
257                }
258                if (great.size() > 0)
259                        target = great.elementAt(CMLib.dice().roll(1, great.size(), -1));
260                else
261                if (good.size() > 0)
262                        target = good.elementAt(CMLib.dice().roll(1, good.size(), -1));
263                return target;
264        }
