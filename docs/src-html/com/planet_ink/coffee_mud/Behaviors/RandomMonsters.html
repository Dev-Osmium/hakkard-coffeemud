064        @Override
065        public List<String> externalFiles()
066        {
067                final Vector<String> xmlfiles=new Vector<String>();
068                final String theseparms=getParms();
069                final int x=theseparms.indexOf(';');
070                String filename=(x>=0)?theseparms.substring(x+1):theseparms;
071                if(filename.trim().length()==0)
072                        return null;
073                final int start=filename.indexOf("<MOBS>");
074                if((start<0)||(start>20))
075                {
076                        final int extraSemicolon=filename.indexOf(';');
077                        if(extraSemicolon>=0)
078                                filename=filename.substring(0,extraSemicolon);
079                        if(filename.trim().length()>0)
080                                xmlfiles.addElement(filename.trim());
081                        return xmlfiles;
082                }
083                return null;
084        }

086        @Override
087        public void setParms(String newParms)
088        {
089                maintained=new Vector<MOB>();
090                final int x=newParms.indexOf(';');
091                String oldParms=newParms;
092                restrictedLocales=null;
093                if(x>=0)
094                {
095                        oldParms=newParms.substring(0,x).trim();
096                        String extraParms=oldParms;
097                        int extraX=newParms.indexOf("<MOBS>");
098                        if(extraX<0)
099                        {
100                                final String xtra=newParms.substring(x+1);
101                                extraX=xtra.indexOf(';');
102                                if(extraX>=0)
103                                        extraParms=xtra.substring(extraX+1);
104                        }
105                        final Vector<String> V=CMParms.parse(extraParms);
106                        for(int v=0;v<V.size();v++)
107                        {
108                                String s=V.elementAt(v);
109                                if((s.startsWith("+")||s.startsWith("-"))&&(s.length()>1))
110                                {
111                                        if(restrictedLocales==null)
112                                                restrictedLocales=new Vector<Integer>();
113                                        if(s.equalsIgnoreCase("+ALL"))
114                                                restrictedLocales.clear();
115                                        else
116                                        if(s.equalsIgnoreCase("-ALL"))
117                                        {
118                                                restrictedLocales.clear();
119                                                for(int i=0;i<Room.DOMAIN_INDOORS_DESCS.length;i++)
120                                                        restrictedLocales.addElement(Integer.valueOf(Room.INDOORS+i));
121                                                for(int i=0;i<Room.DOMAIN_OUTDOOR_DESCS.length;i++)
122                                                        restrictedLocales.addElement(Integer.valueOf(i));
123                                        }
124                                        else
125                                        {
126                                                final char c=s.charAt(0);
127                                                s=s.substring(1).toUpperCase().trim();
128                                                int code=-1;
129                                                for(int i=0;i<Room.DOMAIN_INDOORS_DESCS.length;i++)
130                                                {
131                                                        if(Room.DOMAIN_INDOORS_DESCS[i].startsWith(s))
132                                                                code=Room.INDOORS+i;
133                                                }
134                                                if(code>=0)
135                                                {
136                                                        if((c=='+')&&(restrictedLocales.contains(Integer.valueOf(code))))
137                                                                restrictedLocales.removeElement(Integer.valueOf(code));
138                                                        else
139                                                        if((c=='-')&&(!restrictedLocales.contains(Integer.valueOf(code))))
140                                                                restrictedLocales.addElement(Integer.valueOf(code));
141                                                }
142                                                code=-1;
143                                                for(int i=0;i<Room.DOMAIN_OUTDOOR_DESCS.length;i++)
144                                                {
145                                                        if(Room.DOMAIN_OUTDOOR_DESCS[i].startsWith(s))
146                                                                code=i;
147                                                }
148                                                if(code>=0)
149                                                {
150                                                        if((c=='+')&&(restrictedLocales.contains(Integer.valueOf(code))))
151                                                                restrictedLocales.removeElement(Integer.valueOf(code));
152                                                        else
153                                                        if((c=='-')&&(!restrictedLocales.contains(Integer.valueOf(code))))
154                                                                restrictedLocales.addElement(Integer.valueOf(code));
155                                                }
156
157                                        }
158                                }
159                        }
160                }
161                super.setParms(oldParms);
162                minMonsters=CMParms.getParmInt(oldParms,"minmonsters",1);
163                maxMonsters=CMParms.getParmInt(oldParms,"maxmonsters",1);
164                if(maxMonsters<minMonsters)
165                        maxMonsters=minMonsters;
166                avgMonsters=CMLib.dice().roll(1,(maxMonsters-minMonsters),minMonsters);
167                parms=newParms;
168                if((restrictedLocales!=null)&&(restrictedLocales.size()==0))
169                        restrictedLocales=null;
170                alreadyTriedLoad=false;
171        }

179        public boolean okRoomForMe(MOB M, Room newRoom)
180        {
181                if(newRoom==null)
182                        return false;
183                if(M==null)
184                        return false;
185                if(restrictedLocales==null)
186                        return true;
187                return !restrictedLocales.contains(Integer.valueOf(newRoom.domainType()));
188        }

190        @SuppressWarnings("unchecked")
191        public List<MOB> getMonsters(Tickable thang, String theseparms)
192        {
193                List<MOB> monsters=null;
194                final int x=theseparms.indexOf(';');
195                String thangName="null";
196                if(thang instanceof Room)
197                        thangName=CMLib.map().getExtendedRoomID((Room)thang);
198                else
199                if((thang instanceof MOB)&&(((MOB)thang).getStartRoom())!=null)
200                        thangName=CMLib.map().getExtendedRoomID(((MOB)thang).getStartRoom());
201                else
202                if(thang!=null)
203                        thangName=thang.name();
204                final String thangID=CMClass.classID(thang);
205                String filename=(x>=0)?theseparms.substring(x+1):theseparms;
206                if(filename.trim().length()==0)
207                {
208                        if(alreadyTriedLoad)
209                                return null;
210                        alreadyTriedLoad=true;
211                        Log.errOut("Blank XML/filename: '"+filename+"' on Behavior RandomMonsters on object "+thangName+" ("+thangID+").");
212                        return null;
213                }
214                final int start=filename.indexOf("<MOBS>");
215                if((start>=0)&&(start<=20))
216                {
217                        int end=start+20;
218                        if(end>filename.length())
219                                end=filename.length();
220                        monsters=(List<MOB>)Resources.getResource("RANDOMMONSTERS-XML/"+filename.length()+"/"+filename.hashCode());
221                        if(monsters!=null)
222                                return monsters;
223                        monsters=new Vector<MOB>();
224                        final String error=CMLib.coffeeMaker().addMOBsFromXML(filename,monsters,null);
225                        if(error.length()>0)
226                        {
227                                if(alreadyTriedLoad)
228                                        return null;
229                                alreadyTriedLoad=true;
230                                Log.errOut("RandomMonsters: Error on import of xml for '"+thangName+"' ("+thangID+"): "+error+".");
231                                return null;
232                        }
233                        if(monsters.size()<=0)
234                        {
235                                if(alreadyTriedLoad)
236                                        return null;
237                                alreadyTriedLoad=true;
238                                Log.errOut("RandomMonsters: No mobs loaded for '"+thangName+"' ("+thangID+").");
239                                return null;
240                        }
241                        for(final MOB M : monsters)
242                                CMLib.threads().deleteAllTicks(M);
243                        Resources.submitResource("RANDOMMONSTERS-XML/"+filename.length()+"/"+filename.hashCode(),monsters);
244                }
245                else
246                {
247                        final int extraSemicolon=filename.indexOf(';');
248                        if(extraSemicolon>=0)
249                                filename=filename.substring(0,extraSemicolon);
250                        filename=filename.trim();
251                        monsters=(List<MOB>)Resources.getResource("RANDOMMONSTERS-"+filename);
252                        if((monsters==null)&&(!alreadyTriedLoad))
253                        {
254                                alreadyTriedLoad=true;
255                                final StringBuffer buf=Resources.getFileResource(filename,true);
256
257                                if((buf==null)||(buf.length()<20))
258                                {
259                                        Log.errOut("RandomMonsters: Unknown XML file: '"+filename+"' for '"+thangName+"' ("+thangID+").");
260                                        return null;
261                                }
262                                if(buf.substring(0,20).indexOf("<MOBS>")<0)
263                                {
264                                        Log.errOut("RandomMonsters: Invalid XML file: '"+filename+"' for '"+thangName+"' ("+thangID+").");
265                                        return null;
266                                }
267                                monsters=new Vector<MOB>();
268                                final String error=CMLib.coffeeMaker().addMOBsFromXML(buf.toString(),monsters,null);
269                                if(error.length()>0)
270                                {
271                                        Log.errOut("RandomMonsters: Error on import of: '"+filename+"' for '"+thangName+"' ("+thangID+"): "+error+".");
272                                        return null;
273                                }
274                                if(monsters.size()<=0)
275                                {
276                                        Log.errOut("RandomMonsters: No mobs loaded: '"+filename+"' for '"+thangName+"' ("+thangID+").");
277                                        return null;
278                                }
279                                for(final MOB M : monsters)
280                                        CMLib.threads().deleteAllTicks(M);
281                                Resources.submitResource("RANDOMMONSTERS-"+filename,monsters);
282                        }
283                }
284                return monsters;
285        }

287        public boolean canFlyHere(MOB M, Room R)
288        {
289                if(R==null)
290                        return true;
291                if(((R.domainType()&Room.DOMAIN_INDOORS_AIR)==Room.DOMAIN_INDOORS_AIR)
292                ||((R.domainType()&Room.DOMAIN_OUTDOORS_AIR)==Room.DOMAIN_OUTDOORS_AIR))
293                {
294                        if(!CMLib.flags().isInFlight(M))
295                                return false;
296                }
297                return true;
298        }

306        @Override
307        public boolean tick(final Tickable ticking, final int tickID)
308        {
309                tickStatus=Tickable.STATUS_START;
310                super.tick(ticking,tickID);
311                if((!CMProps.getBoolVar(CMProps.Bool.MUDSTARTED))
312                ||(CMSecurity.isDisabled(CMSecurity.DisFlag.RANDOMMONSTERS)))
313                {
314                        tickStatus=Tickable.STATUS_NOT;
315                        return true;
316                }
317                for(int i=maintained.size()-1;i>=0;i--)
318                {
319                        try
320                        {
321                                final MOB M=maintained.elementAt(i);
322                                if((M.amDead())||(M.amDestroyed())||(M.location()==null)||(!M.location().isInhabitant(M)))
323                                        maintained.removeElement(M);
324                        }
325                        catch (final Exception e)
326                        {
327                        }
328                }
329                if(maintained.size()>=maxMonsters)
330                {
331                        tickStatus=Tickable.STATUS_NOT;
332                        return true;
333                }
334                tickStatus=Tickable.STATUS_ALIVE;
335                if((canAct(ticking,tickID))||(maintained.size()<minMonsters))
336                {
337                        tickStatus=Tickable.STATUS_MISC;
338                        final List<MOB> monsters=getMonsters(ticking,getParms());
339                        tickStatus=Tickable.STATUS_MISC+1;
340                        if(monsters==null)
341                        {
342                                tickStatus=Tickable.STATUS_NOT;
343                                return true;
344                        }
345                        tickStatus=Tickable.STATUS_MISC+2;
346                        if(maintained.size()<avgMonsters)
347                        {
348                                MOB M=monsters.get(CMLib.dice().roll(1,monsters.size(),-1));
349                                if(M!=null)
350                                {
351                                        tickStatus=Tickable.STATUS_MISC+3;
352                                        M=(MOB)M.copyOf();
353                                        tickStatus=Tickable.STATUS_MISC+4;
354                                        M.setStartRoom(null);
355                                        M.basePhyStats().setRejuv(PhyStats.NO_REJUV);
356                                        M.recoverPhyStats();
357                                        tickStatus=Tickable.STATUS_MISC+5;
358                                        M.text();
359                                        maintained.addElement(M);
360                                        tickStatus=Tickable.STATUS_MISC+6;
361                                        if(ticking instanceof Room)
362                                        {
363                                                tickStatus=Tickable.STATUS_MISC+7;
364                                                if(ticking instanceof GridLocale)
365                                                {
366                                                        tickStatus=Tickable.STATUS_MISC+8;
367                                                        final Room room=((GridLocale)ticking).getRandomGridChild();
368                                                        tickStatus=Tickable.STATUS_MISC+9;
369                                                        M.bringToLife(room,true);
370                                                }
371                                                else
372                                                        M.bringToLife(((Room)ticking),true);
373                                                tickStatus=Tickable.STATUS_MISC+10;
374                                                Resources.removeResource("HELP_"+((Room)ticking).getArea().name().toUpperCase());
375                                        }
376                                        else
377                                        if((ticking instanceof Area)&&(((Area)ticking).metroSize()>0))
378                                        {
379                                                tickStatus=Tickable.STATUS_MISC+11;
380                                                Resources.removeResource("HELP_"+ticking.name().toUpperCase());
381                                                Room room=null;
382                                                tickStatus=Tickable.STATUS_MISC+12;
383                                                if(restrictedLocales==null)
384                                                {
385                                                        tickStatus=Tickable.STATUS_MISC+13;
386                                                        int tries=0;
387                                                        while(((room==null)||(!canFlyHere(M,room)))
388                                                        &&((++tries)<100))
389                                                                room=((Area)ticking).getRandomMetroRoom();
390                                                        tickStatus=Tickable.STATUS_MISC+14;
391                                                }
392                                                else
393                                                {
394                                                        tickStatus=Tickable.STATUS_MISC+15;
395                                                        int tries=0;
396                                                        while(((room==null)||(!okRoomForMe(M,room)))
397                                                        &&((++tries)<100))
398                                                                room=((Area)ticking).getRandomMetroRoom();
399                                                        tickStatus=Tickable.STATUS_MISC+16;
400                                                }
401                                                if((room!=null)&&(room instanceof GridLocale))
402                                                        room=((GridLocale)room).getRandomGridChild();
403                                                if(room!=null)
404                                                        M.bringToLife(room,true);
405                                                else
406                                                        maintained.removeElement(M);
407                                                tickStatus=Tickable.STATUS_MISC+17;
408                                        }
409                                }
410                        }
411                }
412                tickStatus=Tickable.STATUS_NOT;
413                return true;
414        }
