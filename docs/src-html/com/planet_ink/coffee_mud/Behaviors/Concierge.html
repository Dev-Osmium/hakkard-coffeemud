101        protected MOB getTalker(final Environmental o, final Room room)
102        {
103                if(o==null)
104                        return null;
105                if(startRoom==null)
106                        startRoom=room;
107                if((talkerName==null)||(talkerName.length()==0))
108                {
109                        if(o instanceof MOB)
110                                return (MOB)o;
111                        if(fakeTalker==null)
112                        {
113                                fakeTalker=CMClass.getFactoryMOB();
114                                fakeTalker.setName(o.name());
115                        }
116                }
117                else
118                if(o instanceof Rideable)
119                {
120                        final Environmental E=CMLib.english().fetchEnvironmental(((Rideable)o).riders(),talkerName,true);
121                        if(E instanceof MOB)
122                                return (MOB)E;
123                }
124                if(fakeTalker==null)
125                {
126                        fakeTalker=CMClass.getFactoryMOB();
127                        fakeTalker.setName(talkerName);
128                }
129                fakeTalker.setStartRoom(startRoom);
130                if(room != null)
131                        fakeTalker.setLocation(room);
132                else
133                        fakeTalker.setLocation(CMLib.map().roomLocation(o));
134                return fakeTalker;
135        }

165        @Override
166        public void setParms(final String newParm)
167        {
168                super.setParms(newParm);
169                resetDefaults();
170                if((CMath.isInteger(newParm))
171                ||(CMath.isDouble(newParm)))
172                {
173                        basePrice=CMath.s_double(newParm);
174                        return;
175                }
176                final List<String> V=CMParms.parseSemicolons(newParm,true);
177                String s=null;
178                int x=0;
179                double price=0;
180                Room R=null;
181                Area A=null;
182                resetFlags();
183                for(int v=0;v<V.size();v++)
184                {
185                        s=V.get(v);
186                        x=s.indexOf('=');
187                        if(x>=0)
188                        {
189                                String numStr=s.substring(x+1).trim();
190                                if(numStr.startsWith("\"")&&(numStr.endsWith("\"")))
191                                        numStr=numStr.substring(1,numStr.length()-1).trim();
192                                s=s.substring(0,x).trim().toUpperCase();
193                                final boolean isTrue=numStr.toLowerCase().startsWith("t");
194                                if(s.equals("PORTAL"))
195                                {
196                                        portal=isTrue;
197                                        continue;
198                                }
199                                else
200                                if(s.equals("AREAONLY"))
201                                {
202                                        if(isTrue)
203                                                roomRadiusFlags.add(TrackingFlag.AREAONLY);
204                                        continue;
205                                }
206                                else
207                                if(s.equals("NOCLIMB"))
208                                {
209                                        if(isTrue)
210                                                trackingFlags.add(TrackingFlag.NOCLIMB);
211                                        continue;
212                                }
213                                else
214                                if(s.equals("NOWATER"))
215                                {
216                                        if(isTrue)
217                                                trackingFlags.add(TrackingFlag.NOWATER);
218                                        continue;
219                                }
220                                else
221                                if(s.equals("INDOOROK"))
222                                {
223                                        if(isTrue)
224                                        {
225                                                trackingFlags.remove(TrackingFlag.OUTDOORONLY);
226                                                roomRadiusFlags.remove(TrackingFlag.OUTDOORONLY);
227                                        }
228                                        continue;
229                                }
230                                else
231                                if(s.equals("NOAIR"))
232                                {
233                                        if(isTrue)
234                                                trackingFlags.add(TrackingFlag.NOAIR);
235                                        continue;
236                                }
237                                else
238                                if(s.equals("NOLOCKS"))
239                                {
240                                        if(isTrue)
241                                                trackingFlags.add(TrackingFlag.UNLOCKEDONLY);
242                                        continue;
243                                }
244                                else
245                                if(s.equals("NOHOMES"))
246                                {
247                                        if(isTrue)
248                                                trackingFlags.add(TrackingFlag.NOHOMES);
249                                        continue;
250                                }
251                                else
252                                if(s.equals("NOINDOOR"))
253                                {
254                                        if(isTrue)
255                                        {
256                                                trackingFlags.add(TrackingFlag.OUTDOORONLY);
257                                                roomRadiusFlags.add(TrackingFlag.OUTDOORONLY);
258                                        }
259                                        continue;
260                                }
261                                else
262                                if(s.equals("MAXRANGE"))
263                                {
264                                        maxRange=CMath.s_int(numStr);
265                                        continue;
266                                }
267                                else
268                                if(s.equals("GREETING"))
269                                {
270                                        greeting=numStr;
271                                        continue;
272                                }
273                                else
274                                if(s.equals("ENTERMSG"))
275                                {
276                                        mountStr=numStr;
277                                        continue;
278                                }
279                                else
280                                if(s.equals("TALKERNAME"))
281                                {
282                                        talkerName=numStr;
283                                        continue;
284                                }
285                                else
286                                if(s.equals("PERROOM"))
287                                {
288                                        perRoomPrice=CMath.s_double(numStr);
289                                        continue;
290                                }
291                                else
292                                        price=CMath.s_double(numStr);
293                        }
294                        A=null;
295                        R=CMLib.map().getRoom(s);
296                        if(R==null)
297                                A=CMLib.map().findArea(s);
298                        if(A!=null)
299                                rates.add(A,Double.valueOf(price));
300                        else
301                        if((R!=null)&&(!rates.containsFirst(R)))
302                                rates.add(R,Double.valueOf(price));
303                        else
304                                rates.add(s,Double.valueOf(price));
305                }
306                basePrice=price;
307        }

309        protected double getPrice(final Room centerRoom, final Room  destR)
310        {
311                if(destR==null)
312                        return basePrice;
313                final int rateIndex=rates.indexOfFirst(destR);
314                if(rateIndex<0)
315                {
316                        double price=basePrice;
317                        if(this.perRoomPrice > 0.0)
318                        {
319                                final List<Room> trail=CMLib.tracking().findTrailToRoom(centerRoom, destR, trackingFlags, maxRange);
320                                if(trail != null)
321                                        price = price + (perRoomPrice * trail.size());
322                        }
323                        return price;
324                }
325                return rates.get(rateIndex).second.doubleValue();
326        }

345        protected Room findDestination(final Environmental observer, final MOB mob, final Room centerRoom, final String where, final TrackingFlags roomRadiusFlags)
346        {
347                PairVector<String,Double> stringsToDo=null;
348                Room roomR=null;
349                List<Room> roomsInRange=null;
350                if(rates.size()==0)
351                {
352                        final Area A=CMLib.map().findArea(where);
353                        if(A!=null)
354                        {
355                                roomsInRange=getRoomsInRange(centerRoom,roomsInRange,roomRadiusFlags);
356                                roomR=findNearestAreaRoom(A,roomsInRange);
357                        }
358                }
359                else
360                if(rates.size()>0)
361                {
362                        for(int r=rates.size()-1;r>=0;r--)
363                        {
364                                if(rates.get(r).first instanceof String)
365                                {
366                                        final String place=(String)rates.get(r).first;
367                                        if((observer!=null)&&(centerRoom!=null))
368                                        {
369                                                if(stringsToDo==null)
370                                                        stringsToDo=new PairVector<String,Double>();
371                                                stringsToDo.addElement(place,rates.get(r).second);
372                                        }
373                                        rates.removeElementAt(r);
374                                }
375                        }
376                        if((stringsToDo!=null)&&(observer!=null))
377                        {
378                                Room R=null;
379                                String place=null;
380                                for(int r=0;r<stringsToDo.size();r++)
381                                {
382                                        roomsInRange=getRoomsInRange(centerRoom,roomsInRange,roomRadiusFlags);
383                                        place=stringsToDo.get(r).first;
384                                        R=(Room)CMLib.english().fetchEnvironmental(roomsInRange,place,false);
385                                        if(R!=null)
386                                                rates.add(R,stringsToDo.get(r).second);
387                                }
388                                stringsToDo.clear();
389                                stringsToDo=null;
390                        }
391                        if(ratesVec==null)
392                        {
393                                ratesVec=new ArrayList<Room>();
394                                for(final Pair<Object,Double> p : rates)
395                                {
396                                        if(p.first instanceof Area)
397                                                ratesVec.add(((Area)p.first).getRandomMetroRoom());
398                                        else
399                                        if(p.first instanceof Room)
400                                                ratesVec.add((Room)p.first);
401                                }
402                        }
403                        roomR=(Room)CMLib.english().fetchEnvironmental(ratesVec,where,true);
404                        if(roomR==null)
405                                roomR=(Room)CMLib.english().fetchEnvironmental(ratesVec,where,false);
406                        if(roomR==null)
407                        {
408                                final Area A=CMLib.map().findArea(where);
409                                if(A!=null)
410                                {
411                                        roomsInRange=getRoomsInRange(centerRoom,roomsInRange,roomRadiusFlags);
412                                        roomR=findNearestAreaRoom(A,roomsInRange);
413                                }
414                        }
415                }
416                if(roomR==null)
417                {
418                        roomsInRange=getRoomsInRange(centerRoom,roomsInRange,roomRadiusFlags);
419                        roomR=CMLib.map().findFirstRoom(new IteratorEnumeration<Room>(roomsInRange.iterator()), mob, where, false, 5);
420                }
421                return roomR;
422        }

424        protected boolean mayGiveThisMoney(final MOB source, final MOB conceirgeM, final Room room, final Environmental possibleCoins)
425        {
426                if(possibleCoins!=null)
427                {
428                        if(!(possibleCoins instanceof Coins))
429                        {
430                                if(room.isInhabitant(conceirgeM))
431                                {
432                                        CMLib.commands().postSay(conceirgeM,source,L("I'm sorry, I can only accept money."),true,false);
433                                        return false;
434                                }
435                                return true;
436                        }
437
438                        final int destIndex=destinations.indexOfFirst(source);
439                        if(destIndex<0)
440                        {
441                                CMLib.commands().postSay(conceirgeM,source,L("What's this for?  Please tell me where you'd like to go first."),true,false);
442                                return false;
443                        }
444                        else
445                        if(!((Coins)possibleCoins).getCurrency().equalsIgnoreCase(CMLib.beanCounter().getCurrency(conceirgeM)))
446                        {
447                                CMLib.commands().postSay(conceirgeM,source,L("I'm sorry, I don't accept that kind of currency."),true,false);
448                                return false;
449                        }
450                        final Double owed=destinations.get(destinations.indexOfFirst(source)).third;
451                        if(owed.doubleValue()<=0.0)
452                        {
453                                CMLib.commands().postSay(conceirgeM,source,L("Hey, you've already paid me!"),true,false);
454                                return false;
455                        }
456                }
457                return true;
458        }

460        @Override
461        public boolean okMessage(final Environmental host, final CMMsg msg)
462        {
463                if(!super.okMessage(host,msg))
464                        return false;
465                final MOB source=msg.source();
466                if(startRoom==null)
467                        startRoom=source.location();
468                if((!canFreelyBehaveNormal(host))&&(host instanceof MOB))
469                        return true;
470                final Environmental observer=host;
471                final Room room=source.location();
472                if(source != observer)
473                {
474                        if((msg.amITarget(getReceiver(observer,room)))
475                        &&(msg.targetMinor()==CMMsg.TYP_GIVE)
476                        &&(!(observer instanceof Container))
477                        &&(!(observer instanceof MOB)))
478                                return this.mayGiveThisMoney(source, getTalker(observer,room), room, msg.tool());
479                        else
480                        if((msg.amITarget(observer))
481                        &&(msg.targetMinor()==CMMsg.TYP_PUT)
482                        &&(observer instanceof Container))
483                                return this.mayGiveThisMoney(source, getTalker(observer,room), room, msg.tool());
484                        else
485                        if((msg.targetMinor()==CMMsg.TYP_DROP)
486                        &&(!(observer instanceof MOB)))
487                                return this.mayGiveThisMoney(source, getTalker(observer,room), room, msg.target());
488                }
489                return true;
490        }

492        @Override
493        public boolean tick(final Tickable ticking, final int tickID)
494        {
495                if((ticking instanceof Environmental)
496                &&(tickID==Tickable.TICKID_MOB)
497                &&(thingsToSay.size()>0)
498                &&(canFreelyBehaveNormal(ticking))||(!(ticking instanceof MOB)))
499                {
500                        final Room R=CMLib.map().roomLocation((Environmental)ticking);
501                        synchronized(thingsToSay)
502                        {
503                                while(thingsToSay.size()>0)
504                                {
505                                        final MOB source=thingsToSay.get(0).first;
506                                        final MOB observer=getTalker((Environmental)ticking,source.location());
507                                        final String msg=thingsToSay.get(0).second;
508                                        thingsToSay.removeElementAt(0);
509                                        if((R!=null)&&(R.isHere(source)))
510                                                CMLib.commands().postSay(observer,source,msg,true,false);
511                                }
512                        }
513                }
514                return super.tick(ticking,tickID);
515        }

517        protected void executeMoneyDrop(final MOB source, final MOB conciergeM, final Environmental possibleCoins, final CMMsg addToMsg)
518        {
519                if(possibleCoins instanceof Coins)
520                {
521                        final int destIndex=destinations.indexOfFirst(source);
522                        if(destIndex>=0)
523                        {
524                                final Quad<MOB,Room,Double,TrackingFlags> destT=destinations.get(destIndex);
525                                final Room destR=destT.second;
526                                final Double owed=Double.valueOf(destT.third.doubleValue() - ((Coins)possibleCoins).getTotalValue());
527                                if(owed.doubleValue()>0.0)
528                                {
529                                        destT.third=owed;
530                                        CMLib.commands().postSay(conciergeM,source,L("Ok, you still owe @x1.",CMLib.beanCounter().nameCurrencyLong(conciergeM,owed.doubleValue())),true,false);
531                                        return;
532                                }
533                                else
534                                if(owed.doubleValue()<0.0)
535                                {
536                                        final double change=-owed.doubleValue();
537                                        final Coins C=CMLib.beanCounter().makeBestCurrency(conciergeM,change);
538                                        if((change>0.0)&&(C!=null))
539                                        {
540                                                // this message will actually end up triggering the hand-over.
541                                                final CMMsg newMsg=CMClass.getMsg(conciergeM,source,C,CMMsg.MSG_SPEAK,L("^T<S-NAME> say(s) 'Heres your change.' to <T-NAMESELF>.^?"));
542                                                C.setOwner(conciergeM);
543                                                final long num=C.getNumberOfCoins();
544                                                final String curr=C.getCurrency();
545                                                final double denom=C.getDenomination();
546                                                C.destroy();
547                                                C.setNumberOfCoins(num);
548                                                C.setCurrency(curr);
549                                                C.setDenomination(denom);
550                                                destT.third=Double.valueOf(0.0);
551                                                addToMsg.addTrailerMsg(newMsg);
552                                        }
553                                        else
554                                                CMLib.commands().postSay(conciergeM,source,L("Gee, thanks. :)"),true,false);
555                                }
556                                ((Coins)possibleCoins).destroy();
557                                giveMerchandise(source, destR, conciergeM, source.location(), destT.fourth);
558                                destinations.removeElementFirst(source);
559                        }
560                        else
561                        if(!CMLib.flags().canBeSeenBy(source,conciergeM))
562                                CMLib.commands().postSay(conciergeM,null,L("Wha?  Where did this come from?  Cool!"),false,false);
563                }
564        }

571        protected void giveMerchandise(final MOB whoM, Room destination, final Environmental observer, final Room room, final TrackingFlags trackingFlags)
572        {
573
574                if(this.portal)
575                {
576                        final Room R=whoM.location();
577                        final Ability spell=CMClass.getAbility("Spell_Portal");
578                        final CMMsg msg=CMClass.getMsg(whoM,R,spell,CMMsg.MSG_CAST_VERBAL_SPELL,L("A blinding, swirling portal appears here."));
579                        final CMMsg msg2=CMClass.getMsg(whoM,destination,spell,CMMsg.MSG_CAST_VERBAL_SPELL,L("A blinding, swirling portal appears here."));
580                        if((R.okMessage(whoM,msg))&&(destination.okMessage(whoM,msg2)))
581                        {
582                                R.send(whoM,msg);
583                                destination=(Room)msg2.target();
584                                destination.sendOthers(whoM,msg2);
585                                final Item I=CMClass.getItem("GenPortal");
586                                I.setName(L("A portal to @x1",getDestinationName(whoM,destination)));
587                                I.setDisplayText(L("A portal to @x1 swirls here",getDestinationName(whoM,destination)));
588                                I.setReadableText(CMLib.map().getExtendedRoomID(destination));
589                                R.addItem(I, Expire.Monster_EQ);
590                                final Behavior B=CMClass.getBehavior("Decay");
591                                B.setParms("minticks=8 maxticks=12");
592                                I.addBehavior(B);
593                                thingsToSay.addElement(whoM,L("Enter this portal to @x1.",getDestinationName(whoM,destination)));
594                        }
595                }
596                else
597                {
598                        final MOB fromM=getTalker(observer,room);
599                        String name=CMLib.map().getExtendedRoomID(destination);
600                        if(name.length()==0)
601                                name=destination.displayText();
602                        final List<Room> set=new ArrayList<Room>();
603                        CMLib.tracking().getRadiantRooms(fromM.location(),set,trackingFlags,null,maxRange,null);
604                        String trailStr;
605                        if(CMLib.tracking().canValidTrail(fromM.location(), set, name, maxRange, null, 1))
606                                trailStr=CMLib.tracking().getTrailToDescription(fromM.location(),set,name,false,false,maxRange,null,1);
607                        else
608                        {
609                                //set.clear();
610                                final TrackingFlags noAirFlags = trackingFlags.copyOf();
611                                noAirFlags.add(TrackingFlag.NOAIR);
612                                CMLib.tracking().getRadiantRooms(fromM.location(),set,noAirFlags,null,maxRange,null);
613                                trailStr=CMLib.tracking().getTrailToDescription(fromM.location(),set,name,false,false,maxRange,null,1);
614                        }
615                        thingsToSay.addElement(whoM,L("The way to @x1 from here is: @x2",getDestinationName(whoM,destination),trailStr));
616                }
617        }

619        @Override
620        public void executeMsg(final Environmental affecting, final CMMsg msg)
621        {
622                super.executeMsg(affecting,msg);
623
624                if((!canFreelyBehaveNormal(affecting))&&(affecting instanceof MOB))
625                        return;
626
627                final MOB source=msg.source();
628                final Environmental observer=affecting;
629                final Room room=source.location();
630                if(source!=observer)
631                {
632                        switch(msg.targetMinor())
633                        {
634                        case CMMsg.TYP_GIVE:
635                                if(msg.amITarget(getReceiver(observer,room)))
636                                        executeMoneyDrop(source,getTalker(observer,room),msg.tool(),msg);
637                                break;
638                        case CMMsg.TYP_DROP:
639                                if((!(observer instanceof Container))
640                                &&((!(observer instanceof Rideable))||(msg.source().riding()==observer))
641                                &&(!(observer instanceof MOB)))
642                                        executeMoneyDrop(source,getTalker(observer,room),msg.target(),msg);
643                                break;
644                        case CMMsg.TYP_PUT:
645                                if((msg.amITarget(observer))
646                                &&((!(observer instanceof Rideable))||(msg.source().riding()==observer))
647                                &&(observer instanceof Container))
648                                        executeMoneyDrop(source,getTalker(observer,room),msg.tool(),msg);
649                                break;
650                        case CMMsg.TYP_SPEAK:
651                                if((msg.source()==getTalker(observer,room))
652                                &&(msg.target() instanceof MOB)
653                                &&(msg.tool() instanceof Coins)
654                                &&(((Coins)msg.tool()).amDestroyed())
655                                &&(!msg.source().isMine(msg.tool()))
656                                &&(!((MOB)msg.target()).isMine(msg.tool())))
657                                        CMLib.beanCounter().giveSomeoneMoney(msg.source(),(MOB)msg.target(),((Coins)msg.tool()).getTotalValue());
658                                else
659                                if((!msg.source().isMonster())
660                                &&((msg.target()==observer)||(source.location().numPCInhabitants()==1)||(!(observer instanceof MOB)))
661                                &&((!(observer instanceof Rideable))||(msg.source().riding()==observer))
662                                &&(msg.sourceMessage()!=null))
663                                {
664                                        String say=CMStrings.getSayFromMessage(msg.sourceMessage());
665                                        if((say!=null)&&(say.length()>0))
666                                        {
667                                                final List<String> parsedSay = CMParms.parse(say);
668                                                boolean didLoop=true;
669                                                final TrackingFlags myRoomRadiusFlags=CMLib.tracking().newFlags();
670                                                myRoomRadiusFlags.addAll(roomRadiusFlags);
671                                                boolean didAnything=false;
672                                                while(didLoop && (parsedSay.size()>0))
673                                                {
674                                                        didLoop=false;
675                                                        final String s=parsedSay.get(parsedSay.size()-1);
676                                                        if(s.equalsIgnoreCase("noclimb"))
677                                                                didLoop=myRoomRadiusFlags.add(TrackingFlag.NOCLIMB);
678                                                        else
679                                                        if(s.equalsIgnoreCase("nodoors"))
680                                                                didLoop=myRoomRadiusFlags.add(TrackingFlag.OPENONLY);
681                                                        else
682                                                        if(s.equalsIgnoreCase("noswim"))
683                                                                didLoop=myRoomRadiusFlags.add(TrackingFlag.NOWATER);
684                                                        else
685                                                        if(s.equalsIgnoreCase("noprivate"))
686                                                                didLoop=myRoomRadiusFlags.add(TrackingFlag.NOPRIVATEPROPERTY);
687                                                        else
688                                                        if(s.equalsIgnoreCase("nofly"))
689                                                                didLoop=myRoomRadiusFlags.add(TrackingFlag.NOAIR);
690                                                        else
691                                                        if(s.equalsIgnoreCase("nolocks"))
692                                                                didLoop=myRoomRadiusFlags.add(TrackingFlag.UNLOCKEDONLY);
693                                                        else
694                                                        if(s.equalsIgnoreCase("nocrawl"))
695                                                                didLoop=myRoomRadiusFlags.add(TrackingFlag.NOCRAWL);
696                                                        if(didLoop)
697                                                                parsedSay.remove(parsedSay.size()-1);
698                                                        didAnything = didAnything || didLoop;
699                                                }
700                                                Room roomR;
701                                                if(didAnything)
702                                                {
703                                                        say=CMParms.combine(parsedSay);
704                                                        roomR=findDestination(observer,msg.source(),room,say,myRoomRadiusFlags);
705                                                        if((roomR==null)||(roomR.roomID().length()==0))
706                                                        {
707                                                                final Room roomR2=findDestination(observer,msg.source(),room,say,trackingFlags);
708                                                                if(roomR2!=null)
709                                                                        roomR=roomR2;
710                                                        }
711                                                }
712                                                else
713                                                {
714                                                        roomR=findDestination(observer,msg.source(),room,say,trackingFlags);
715                                                        if((roomR==null)||(roomR.roomID().length()==0))
716                                                        {
717                                                                final Room roomR2=findDestination(observer,msg.source(),room,say,myRoomRadiusFlags);
718                                                                if(roomR2!=null)
719                                                                        roomR=roomR2;
720                                                        }
721                                                }
722                                                if(roomR==null)
723                                                {
724                                                        synchronized(thingsToSay)
725                                                        {
726                                                                thingsToSay.addElement(msg.source(),L("I'm sorry, I don't know where '@x1' is.",say));
727                                                                return;
728                                                        }
729                                                }
730                                                final double rate=getPrice(room,roomR);
731                                                final Double owed=Double.valueOf(rate);
732                                                trackingFlags.addAll(myRoomRadiusFlags);
733                                                destinations.removeElementFirst(msg.source());
734                                                if(owed.doubleValue()<=0.0)
735                                                        giveMerchandise(msg.source(), roomR, observer, room, trackingFlags);
736                                                else
737                                                {
738                                                        destinations.addElement(msg.source(),roomR,owed, trackingFlags);
739                                                        final String moneyName=CMLib.beanCounter().nameCurrencyLong(getTalker(observer,room),rate);
740                                                        thingsToSay.addElement(msg.source(),getGiveMoneyMessage(msg.source(),observer,roomR,moneyName));
741                                                }
742                                        }
743                                }
744                                break;
745                        case CMMsg.TYP_LEAVE:
746                                if((msg.target()==room)
747                                &&(!disableComingsAndGoings())
748                                &&(destinations.containsFirst(msg.source())))
749                                        destinations.removeElementFirst(msg.source());
750                                break;
751                        case CMMsg.TYP_ENTER:
752                                if((!msg.source().isMonster())
753                                &&((greeting!=null)&&(greeting.length()>0))
754                                &&(!disableComingsAndGoings())
755                                &&(!destinations.containsFirst(msg.source()))
756                                &&(msg.target()==CMLib.map().roomLocation(observer))
757                                &&(CMLib.flags().canBeSeenBy(msg.source(), getTalker(observer,room))))
758                                        thingsToSay.addElement(msg.source(),greeting);
759                                //$FALL-THROUGH$
760                        case CMMsg.TYP_MOUNT:
761                                if((!msg.source().isMonster())
762                                &&(msg.target()==observer)
763                                &&((mountStr!=null)&&(mountStr.length()>0))
764                                &&(!destinations.containsFirst(msg.source()))
765                                &&(CMLib.flags().canBeSeenBy(msg.source(), getTalker(observer,room))))
766                                        thingsToSay.addElement(msg.source(),mountStr);
767                        }
768                }
769        }
770}
