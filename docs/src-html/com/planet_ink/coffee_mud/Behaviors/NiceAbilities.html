036public class NiceAbilities extends ActiveTicker
037{
038        @Override
039        public String ID()
040        {
041                return "NiceAbilities";
042        }
043
044        @Override
045        protected int canImproveCode()
046        {
047                return Behavior.CAN_MOBS;
048        }
049
050        private List<Ability> mySkills=null;
051        private int numAllSkills=-1;
052
053        public NiceAbilities()
054        {
055                super();
056                minTicks=10; maxTicks=20; chance=100;
057                tickReset();
058        }
059
060        @Override
061        public String accountForYourself()
062        {
063                return "random benevolent skill using";
064        }
065
066        @Override
067        public boolean tick(final Tickable ticking, final int tickID)
068        {
069                super.tick(ticking,tickID);
070                if((canAct(ticking,tickID))&&(ticking instanceof MOB))
071                {
072                        final MOB mob=(MOB)ticking;
073                        final Room thisRoom=mob.location();
074                        if(thisRoom==null)
075                                return true;
076
077                        final double aChance=CMath.div(mob.curState().getMana(),mob.maxState().getMana());
078                        if((Math.random()>aChance)||(mob.curState().getMana()<50))
079                                return true;
080
081                        if(thisRoom.numPCInhabitants()>0)
082                        {
083                                final MOB target=thisRoom.fetchRandomInhabitant();
084                                MOB followMOB = target;
085                                if((target!=null)&&(target.amFollowing()!=null))
086                                        followMOB=target.amUltimatelyFollowing();
087                                if((target!=null)
088                                &&(target!=mob)
089                                &&(followMOB.getVictim()!=mob)
090                                &&(!followMOB.isMonster()))
091                                {
092                                        if((numAllSkills!=mob.numAllAbilities())||(mySkills==null))
093                                        {
094                                                numAllSkills=mob.numAbilities();
095                                                mySkills=new ArrayList<Ability>();
096                                                for(final Enumeration<Ability> e=mob.allAbilities(); e.hasMoreElements();)
097                                                {
098                                                        final Ability tryThisOne=e.nextElement();
099                                                        if((tryThisOne!=null)
100                                                        &&(tryThisOne.abstractQuality()==Ability.QUALITY_BENEFICIAL_OTHERS)
101                                                        &&(((tryThisOne.classificationCode()&Ability.ALL_ACODES)!=Ability.ACODE_PRAYER)
102                                                                ||tryThisOne.appropriateToMyFactions(mob)))
103                                                        {
104                                                                mySkills.add(tryThisOne);
105                                                        }
106                                                }
107                                        }
108                                        if(mySkills.size()>0)
109                                        {
110                                                final Ability tryThisOne=mySkills.get(CMLib.dice().roll(1, mySkills.size(), -1));
111                                                if((mob.fetchEffect(tryThisOne.ID())==null)
112                                                &&(tryThisOne.castingQuality(mob,target)==Ability.QUALITY_BENEFICIAL_OTHERS))
113                                                {
114                                                        tryThisOne.setProficiency(CMLib.ableMapper().getMaxProficiency(mob,true,tryThisOne.ID()));
115                                                        final Vector<String> V=new XVector<String>("$"+target.Name()+"$");
116                                                        V.addElement(target.name());
117                                                        tryThisOne.invoke(mob,V,target,false,0);
118                                                }
119                                        }
120                                }
121                        }
122                }
123                return true;
124        }
125}
