165        protected MOB getBehaversMOB(Tickable ticking)
166        {
167                if(ticking==null)
168                        return null;
169
170                if(ticking instanceof MOB)
171                        return (MOB)ticking;
172                else
173                if(ticking instanceof Item)
174                {
175                        if(((Item)ticking).owner() != null)
176                        {
177                                if(((Item)ticking).owner() instanceof MOB)
178                                        return (MOB)((Item)ticking).owner();
179                        }
180                }
181
182                return null;
183        }
184
185        protected Room getBehaversRoom(Tickable ticking)
186        {
187                if(ticking==null)
188                        return null;
189
190                if(ticking instanceof Room)
191                        return (Room)ticking;
192
193                final MOB mob=getBehaversMOB(ticking);
194                if(mob!=null)
195                        return mob.location();
196
197                if(ticking instanceof Item)
198                {
199                        if(((Item)ticking).owner() != null)
200                        {
201                                if(((Item)ticking).owner() instanceof Room)
202                                        return (Room)((Item)ticking).owner();
203                        }
204                }
205                return null;
206        }

256        @Override
257        public boolean canImprove(PhysicalAgent E)
258        {
259                if((E==null)&&(canImproveCode()==0))
260                        return true;
261                if(E==null)
262                        return false;
263                if((E instanceof MOB)&&((canImproveCode()&Ability.CAN_MOBS)>0))
264                        return true;
265                if((E instanceof Item)&&((canImproveCode()&Ability.CAN_ITEMS)>0))
266                        return true;
267                if((E instanceof Exit)&&((canImproveCode()&Ability.CAN_EXITS)>0))
268                        return true;
269                if((E instanceof Room)&&((canImproveCode()&Ability.CAN_ROOMS)>0))
270                        return true;
271                if((E instanceof Area)&&((canImproveCode()&Ability.CAN_AREAS)>0))
272                        return true;
273                return false;
274        }

308        protected static final String[] CODES   = { "CLASS", "TEXT" };
309
310        @Override
311        public String[] getStatCodes()
312        {
313                return CODES;
314        }
315
316        protected int getCodeNum(final String code)
317        {
318                for(int i=0;i<CODES.length;i++)
319                {
320                        if(code.equalsIgnoreCase(CODES[i]))
321                                return i;
322                }
323                return -1;
324        }
325
326        @Override
327        public String getStat(String code)
328        {
329                switch(getCodeNum(code))
330                {
331                case 0:
332                        return ID();
333                case 1:
334                        return getParms();
335                }
336                return "";
337        }
338
339        @Override
340        public void setStat(String code, String val)
341        {
342                switch(getCodeNum(code))
343                {
344                case 0:
345                        return;
346                case 1:
347                        setParms(val);
348                        break;
349                }
350        }
351
352        @Override
353        public int getSaveStatIndex()
354        {
355                return getStatCodes().length;
356        }
357
358        @Override
359        public boolean isStat(final String code)
360        {
361                return CMParms.indexOf(getStatCodes(), code.toUpperCase().trim()) >= 0;
362        }
363
364        public boolean sameAs(Behavior E)
365        {
366                if(!(E instanceof StdBehavior))
367                        return false;
368                for(int i=0;i<CODES.length;i++)
369                {
370                        if(!E.getStat(CODES[i]).equals(getStat(CODES[i])))
371                                return false;
372                }
373                return true;
374        }
