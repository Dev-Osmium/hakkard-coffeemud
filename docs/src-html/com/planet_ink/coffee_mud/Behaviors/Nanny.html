102        public double getPaidBy(MOB mob)
103        {
104                if(mob==null)
105                        return 0.0;

139        public void addPayment(MOB mob,double amt)
140        {
141                if(mob==null)
142                        return;
143                for(final Payment P : payments)
144                {
145                        if(P.mommyM==mob)

154        public void clearTheSlate(MOB mob)
155        {
156                if(mob==null)
157                        return;
158                for(final Payment P : payments)
159                {
160                        if(P.mommyM==mob)
161                                payments.remove(P);
162                }
163                if(dropOffs != null)
164                {
165                        for(final DropOff D : dropOffs)
166                        {
167                                if(D.mommyM==mob)
168                                {
169                                        boolean found=false;
170                                        for(final DropOff A : associations)
171                                                found = found || (A.mommyM==D.mommyM);
172                                        if(!found)
173                                                associations.add(D);
174                                        dropOffs.remove(D);
175                                        changedSinceLastSave=true;
176                                }
177                        }
178                }
179        }

215        public String getPronoun(List<PhysicalAgent> V)
216        {
217                if(V.size()==0)
218                        return "your stuff";
219                int babies=0;
220                int friends=0;
221                int objects=0;
222                int mounts=0;
223                for(int v=0;v<V.size();v++)
224                {
225                        final PhysicalAgent E=V.get(v);
226                        if(CMLib.flags().isBaby(E)||CMLib.flags().isChild(E))
227                                babies++;
228                        else
229                        if(isMount(E))
230                                mounts++;
231                        else
232                        if(E instanceof Item)
233                                objects++;
234                        else
235                                friends++;
236                }
237                final Vector<String> pros=new Vector<String>();
238                if(babies>0)
239                        pros.addElement("little one"+((babies>1)?"s":""));
240                if(mounts>0)
241                        pros.addElement("mount"+((babies>1)?"s":""));
242                if(friends>0)
243                        pros.addElement("friend"+((babies>1)?"s":""));
244                if(objects>0)
245                        pros.addElement("thing"+((babies>1)?"s":""));
246                final StringBuffer list=new StringBuffer("");
247                for(int p=0;p<pros.size();p++)
248                {
249                        list.append(pros.elementAt(p));
250                        if((pros.size()>1)&&(p==pros.size()-2))
251                                list.append(", and ");
252                        else
253                        if(pros.size()>1)
254                                list.append(", ");
255                }
256                return list.toString().trim();
257        }

301        public PhysicalAgent getDroppedOffObjIfAny(PhysicalAgent P)
302        {
303                if(P==null) 
304                        return null;
305                if(isDroppedOff(P)) 
306                        return P;
307                
308                if(P instanceof Container)
309                {
310                        final List<Item> V=((Container)P).getDeepContents();
311                        Item I=null;
312                        for(int v=0;v<V.size();v++)

316                                if(P!=null) 
317                                        return P;
318                        }
319                }

321        }

324        public boolean okMessage(final Environmental host, final CMMsg msg)
325        {
326                if(!super.okMessage(host,msg))
327                        return false;
328                if(dropOffs==null)
329                        return true;
330                final int targMinor=msg.targetMinor();
331                if((msg.target()==host)
332                &&(targMinor==CMMsg.TYP_GIVE))
333                {
334                        if(isDropOffable(msg.tool()))
335                        {
336                                final String pronoun=this.getPronoun(new XVector<PhysicalAgent>((PhysicalAgent)msg.tool()));
337                                msg.source().tell(msg.source(),host,msg.tool(),L("<T-NAME> won't accept <O-NAME>.  You should probably leave your @x1 here.",pronoun));
338                                return false;
339                        }
340                        if(msg.tool() instanceof Coins)
341                        {
342                                final Coins C=(Coins)msg.tool();
343                                final String myCurrency=CMLib.beanCounter().getCurrency(host);
344                                if(!C.getCurrency().equalsIgnoreCase(myCurrency))
345                                {
346                                        if(host instanceof MOB)
347                                                CMLib.commands().postSay((MOB)host,msg.source(),L("I'm don't accept @x1.  I can only accept @x2.",CMLib.beanCounter().getDenominationName(C.getCurrency(),C.getDenomination()),CMLib.beanCounter().getDenominationName(myCurrency)));
348                                        else
349                                                msg.source().tell(L("The @x1 doesn't accept @x2.  It only accepts @x3.",place,CMLib.beanCounter().getDenominationName(C.getCurrency(),C.getDenomination()),CMLib.beanCounter().getDenominationName(myCurrency)));
350                                        return false;
351                                }
352                        }
353                }
354                else
355                if(((targMinor==CMMsg.TYP_GET)
356                        ||(targMinor==CMMsg.TYP_PULL)
357                        ||(targMinor==CMMsg.TYP_PUSH)
358                        ||(targMinor==CMMsg.TYP_CAST_SPELL))
359                &&(msg.target() instanceof Item)
360                &&(!msg.targetMajor(CMMsg.MASK_INTERMSG))
361                &&(getDroppedOffObjIfAny((Item)msg.target()))!=null)
362                {
363                        final PhysicalAgent obj=getDroppedOffObjIfAny((Item)msg.target());
364                        final String amt=getOwedFor(CMLib.beanCounter().getCurrency(host),obj);
365                        if((msg.source().location()==CMLib.map().roomLocation(host))
366                        &&(host instanceof MOB))
367                        {
368                                if(amt.length()>0)
369                                        CMLib.commands().postSay((MOB)host,msg.source(),L("I'm afraid that @x1 fees of @x2 are still owed.",place,amt));
370                                else
371                                        CMLib.commands().postSay((MOB)host,msg.source(),L("I'm afraid that @x1 fees are still owed on @x2.",place,obj.name()));
372                        }
373                        else
374                        if(amt.length()>0)
375                                msg.source().tell(L("You'll need to pay @x1 fees of @x2 first.",place,amt));
376                        else
377                                msg.source().tell(L("You'll need to pay your @x1 fees  for @x2 first.",place,obj.name()));
378                        return false;
379                }
380                else
381                if((CMath.bset(msg.sourceMajor(),CMMsg.MASK_MALICIOUS)
382                        ||CMath.bset(msg.targetMajor(),CMMsg.MASK_MALICIOUS))
383                &&(msg.target() instanceof PhysicalAgent)
384                &&((getDroppedOffObjIfAny((PhysicalAgent)msg.target())!=null)
385                                ||(msg.target()==host)
386                                ||(msg.target()==CMLib.map().roomLocation(host))))
387                {
388                        if(msg.source()!=host)
389                        {
390                                if((host instanceof MOB)&&(msg.source().location()==CMLib.map().roomLocation(host)))
391                                {
392                                        CMLib.commands().postSay((MOB)host,msg.source(),L("Not in my @x1 you dont!",place));
393                                        final MOB victim=msg.source().getVictim();
394                                        if(victim!=null)
395                                                victim.makePeace(true);
396                                        msg.source().makePeace(true);
397                                }
398                                else
399                                        msg.source().tell(L("You can't do that here."));
400                        }
401                        return false;
402                }
403                else
404                if((msg.sourceMinor()==CMMsg.TYP_LEAVE)
405                &&(msg.target()==CMLib.map().roomLocation(host)))
406                {
407                        final PhysicalAgent obj=getDroppedOffObjIfAny(msg.source());
408                        if(obj!=null)
409                        {
410                                if((msg.tool() instanceof Ability)
411                                &&(msg.source().location()!=null))
412                                {
413                                        final Room R=CMLib.map().roomLocation(host);
414                                        final boolean summon=CMath.bset(((Ability)msg.tool()).flags(),Ability.FLAG_SUMMONING);
415                                        final boolean teleport=CMath.bset(((Ability)msg.tool()).flags(),Ability.FLAG_TRANSPORTING);
416                                        final boolean shere=(msg.source().location()==R);
417                                        if((!shere)&&((summon)||(teleport)))
418                                        {
419                                                if((msg.source().location()!=null)&&(msg.source().location()!=R))
420                                                        msg.source().location().showHappens(CMMsg.MSG_OK_VISUAL,L("Magical energy fizzles and is absorbed into the air!"));
421                                                R.showHappens(CMMsg.MSG_OK_VISUAL,L("Magic energy fizzles and is absorbed into the air."));
422                                                return false;
423                                        }
424                                }
425                                else
426                                {
427                                        final String amt=getOwedFor(CMLib.beanCounter().getCurrency(host),obj);
428                                        if((msg.source().location()==CMLib.map().roomLocation(host))
429                                        &&(host instanceof MOB))
430                                        {
431                                                if(amt.length()>0)
432                                                        CMLib.commands().postSay((MOB)host,msg.source(),L("I'm afraid that @x1 fees of @x2 are still owed on @x3.",place,amt,obj.name()));
433                                                else
434                                                        CMLib.commands().postSay((MOB)host,msg.source(),L("I'm afraid that @x1 fees are still owed on @x2.",place,obj.name()));
435                                        }
436                                        else
437                                        if(amt.length()>0)
438                                                msg.source().tell(L("You'll need to pay @x1 fees of @x2 for @x3 first.",place,amt,obj.name()));
439                                        else
440                                                msg.source().tell(L("You'll need to pay your @x1 fees  for @x2 first.",place,obj.name()));
441                                }
442                                return false;
443                        }
444                }
445                return true;
446        }

459        public boolean isDropOffable(Environmental E)
460        {
461                if(E==null)
462                        return false;
463                if((E instanceof MOB)&&(!((MOB)E).isMonster()))
464                        return false;
465                if((watchesBabies)&&(CMLib.flags().isBaby(E)))
466                        return true;
467                if((watchesChildren)&&(CMLib.flags().isChild(E))&&(!CMLib.flags().isBaby(E)))
468                        return true;
469                if((watchesMounts)&&(isMount(E)))
470                        return true;
471                if((watchesMOBFollowers)&&(E instanceof MOB)&&(!isMount(E))&&(!CMLib.flags().isChild(E))&&(!CMLib.flags().isBaby(E)))
472                        return true;
473                if((this.watchesWagons)
474                &&(E instanceof Rideable)
475                &&(((Rideable)E).rideBasis()==Rideable.RIDEABLE_WAGON))
476                        return true;
477                if((this.watchesCars)
478                &&(E instanceof Item)
479                &&(E instanceof Rideable)
480                &&(((Rideable)E).rideBasis()==Rideable.RIDEABLE_LAND))
481                        return true;
482                if((this.watchesBoats)
483                &&(E instanceof Item)
484                &&(E instanceof Rideable)
485                &&(((Rideable)E).rideBasis()==Rideable.RIDEABLE_WATER))
486                        return true;
487                if((this.watchesAirCars)
488                &&(E instanceof Item)
489                &&(E instanceof Rideable)
490                &&(((Rideable)E).rideBasis()==Rideable.RIDEABLE_AIR))
491                        return true;
492                return false;
493        }

503        public MOB getMommyOf(Physical P)
504        {
505                if((P instanceof Item)
506                &&(((Item)P).owner() instanceof MOB)
507                &&(!((MOB)((Item)P).owner()).isMonster()))
508                        return (MOB)((Item)P).owner();
509                if((P instanceof MOB)
510                &&(((MOB)P).amFollowing()!=null)
511                &&(!((MOB)P).amFollowing().isMonster()))
512                        return ((MOB)P).amFollowing();
513                if((P instanceof MOB)
514                &&(ultimateFollowing(P)!=null)
515                &&(!ultimateFollowing(P).isMonster()))
516                        return ultimateFollowing(P);
517                if(P instanceof Rideable)
518                {
519                        final Rideable R=(Rideable)P;
520                        Environmental E2=null;
521                        for(int r=0;r<R.numRiders();r++)
522                        {
523                                E2=R.fetchRider(r);
524                                if((E2 instanceof MOB)
525                                &&(!((MOB)E2).isMonster()))
526                                        return (MOB)E2;
527                        }
528                }
529                if((P instanceof Rider)
530                &&(((Rider)P).riding()!=null))
531                        return getMommyOf(((Rider)P).riding());
532                return null;
533        }

535        public void addAssociationsIfNecessary(Set<PhysicalAgent> H)
536        {
537                PhysicalAgent P=null;
538                for(final Object o : H)
539                {
540                        if(o instanceof PhysicalAgent)
541                        {
542                                P=(PhysicalAgent)o;
543                                if((P instanceof Rider)&&(((Rider)P).riding()!=null)&&(!H.contains(((Rider)P).riding())))
544                                        H.add(P);
545                        }
546                }
547                for(final Object o : H)
548                {
549                        if(o instanceof PhysicalAgent)
550                        {
551                                P=(PhysicalAgent)o;
552                                if((isDropOffable(P))&&(!isAssociated(P)))
553                                {
554                                        final MOB source=getMommyOf(P);
555                                        if(source!=null)
556                                                associations.add(new DropOff(source,P,System.currentTimeMillis()));
557                                }
558                                if(P instanceof MOB)
559                                {
560                                        final MOB mob=(MOB)P;
561                                        for(int t=0;t<mob.numItems();t++)
562                                        {
563                                                final Item I=mob.getItem(t);
564                                                if(isDropOffable(I)&&(!isAssociated(I)))
565                                                {
566                                                        final MOB source=getMommyOf(I);
567                                                        if(source!=null)
568                                                                associations.add(new DropOff(source,I,System.currentTimeMillis()));
569                                                }
570                                        }
571                                }
572                        }
573                }
574        }

586        @Override
587        public void executeMsg(final Environmental host, final CMMsg msg)
588        {
589                super.executeMsg(host,msg);
590                if(dropOffs==null)
591                        return;
592
593                if((msg.targetMinor()==CMMsg.TYP_ENTER)
594                &&(msg.target()==CMLib.map().roomLocation(host)))
595                {
596                        final String currency=CMLib.beanCounter().getCurrency(host);
597                        final Set H=msg.source().getGroupMembers(new HashSet<MOB>());
598                        msg.source().getRideBuddies(H);
599                        if(!H.contains(msg.source()))
600                                H.add(msg.source());
601                        HashSet<Environmental> H2 = null;
602                        do
603                        {
604                                H2 = new HashSet<Environmental>();
605                                H2.addAll(H);
606                                for (final Object element : H2)
607                                {
608                                        final Environmental E = (Environmental)element;
609                                        if(E instanceof Rideable)
610                                        {
611                                                final Rideable R = (Rideable)E;
612                                                for(int r = 0; r<R.numRiders(); r++)
613                                                        if(!H.contains(R.fetchRider(r)))
614                                                                H.add(R.fetchRider(r));
615                                        }
616                                }
617                        }
618                        while(H.size() > H2.size())
619                                ;
620
621                        addAssociationsIfNecessary(H);
622                        final List<PhysicalAgent> myAssocs=myCurrentAssocs(msg.source());
623                        final StringBuffer list=new StringBuffer("");
624                        for(int m=0;m<myAssocs.size();m++)
625                        {
626                                list.append(myAssocs.get(m).name());
627                                if((myAssocs.size()>1)&&(m==myAssocs.size()-2))
628                                        list.append(", and ");
629                                else
630                                if(myAssocs.size()>1)
631                                        list.append(", ");
632                        }
633                        if(list.length()>0)
634                                sayLaters.addElement(msg.source(),"Welcome to my "+place+", "+msg.source().name()+"! You are welcome to leave " +
635                                                        list.toString()+" here under my care and protection.  Be aware that I charge "
636                                                        +CMLib.beanCounter().abbreviatedPrice(currency,hourlyRate)+" per hour, each.  " +
637                                                        "No payment is due until you return to fetch your "+getPronoun(myAssocs)+".");
638
639                        final double owed=getAllOwedBy(msg.source());
640                        final double paid=getPaidBy(msg.source());
641                        if(owed>0)
642                        {
643                                final List<PhysicalAgent> myStuff=getAllOwedFor(msg.source());

717                &&(!((MOB)msg.target()).isMine(msg.tool())))

746                        parms.append("Wagons,");

750                        parms.append("Boats,");
751                if(watchesAirCars)
752                        parms.append("AirCars,");
753                if(watchesMOBFollowers)
754                        parms.append("Followers,");
755                parms.append("\"");

757                {
758                        parms.append(" |~| ");
759                        final Vector<String> oldNames=new Vector<String>();
760                        String eName=null;

762                        for(final DropOff D : dropOffs)
763                        {

765
766                                eName=D.baby.Name();
767                                oName=D.mommyM.Name();

769                                        eName=getNameCount(oldNames,eName)+"."+eName;
770                                parms.append(CMLib.xml().convertXMLtoTag("ENAM",CMLib.xml().parseOutAngleBrackets(eName)));
771                                parms.append(CMLib.xml().convertXMLtoTag("ONAM",CMLib.xml().parseOutAngleBrackets(oName)));
772                                parms.append(CMLib.xml().convertXMLtoTag("TIME",D.dropOffTime));
773                                parms.append("</DROP>");
774                        }
775                }
776                return parms.toString().trim();
777        }
