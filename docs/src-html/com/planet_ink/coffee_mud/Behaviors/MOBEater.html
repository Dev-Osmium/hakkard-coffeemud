084        @Override
085        public void startBehavior(final PhysicalAgent forMe)
086        {
087                if((forMe!=null)&&(forMe instanceof MOB))
088                {
089                        if(stomachR==null)
090                                stomachR = CMClass.getLocale("StoneRoom");
091                        lastKnownEaterM=(MOB)forMe;
092                        lastKnownLocationR=((MOB)forMe).location();
093                        if(lastKnownLocationR!=null)
094                                stomachR.setArea(lastKnownLocationR.getArea());
095                        stomachR.setDisplayText(L("The Stomach of @x1",forMe.name()));
096                        stomachR.setName(L("the stomach of @x1",forMe.name()));
097                        stomachR.setDescription(L("You are in the stomach of @x1.  It is wet with digestive acids, and the walls are grinding you to a pulp.  You have been Swallowed whole and are being digested.",forMe.name()));
098                        stomachR.addNonUninvokableEffect(CMClass.getAbility("Prop_NoRecall"));
099                        stomachR.addNonUninvokableEffect(CMClass.getAbility("Prop_NoTeleportOut"));
100                        final ExtendableAbility A=(ExtendableAbility)CMClass.getAbility("ExtAbility");
101                        final WeakReference<MOB> eater=new WeakReference<MOB>(lastKnownEaterM);
102                        stomachR.addNonUninvokableEffect(A.setAbilityID("MOBEaterStomachWatcher").setMsgListener(new MsgListener()
103                        {
104                                @Override
105                                public void executeMsg(final Environmental myHost, final CMMsg msg) 
106                                {
107                                        if(A.affecting() instanceof Room)
108                                        {
109                                                if((eater.get()==null)||(eater.get().amDestroyed())||(eater.get().amDead()))
110                                                {
111                                                        CMLib.map().emptyRoom((Room)A.affecting(), null, true);
112                                                }
113                                        }
114                                }
115
116                                @Override
117                                public boolean okMessage(final Environmental myHost, final CMMsg msg)
118                                {
119                                        if((msg.sourceMinor()==CMMsg.TYP_COMMANDFAIL)
120                                        &&(msg.targetMessage()!=null)
121                                        &&(msg.targetMessage().length()>0))
122                                        {
123                                                final char c=Character.toUpperCase(msg.targetMessage().charAt(0));
124                                                if((c=='K')||(c=='A'))
125                                                {
126                                                        List<String> parsedFail = CMParms.parse(msg.targetMessage());
127                                                        String cmd=parsedFail.get(0).toUpperCase();
128                                                        if("ATTACK".startsWith(cmd)||"KILL".startsWith(cmd))
129                                                        {
130                                                                String rest = CMParms.combine(parsedFail,1).toUpperCase().trim();
131                                                                if("HERE".equals(rest)||"STOMACH".startsWith(rest)||"WALLS".startsWith(rest))
132                                                                {
133                                                                        Item I=msg.source().fetchWieldedItem();
134                                                                        if((!(I instanceof Weapon))
135                                                                        ||(I.minRange()>0))
136                                                                        {
137                                                                                msg.source().tell(L("You aren't wielding an appropriate weapon."));
138                                                                                return false;
139                                                                        }
140                                                                        if(msg.source().getPeaceTime()<CMProps.getTickMillis())
141                                                                        {
142                                                                                msg.source().tell(L("You are too busy trying to survive right now!"));
143                                                                                return false;
144                                                                        }
145                                                                        final Weapon weapon=(Weapon)I;
146                                                                        final int dmg = CMLib.combat().adjustedDamage(msg.source(), (Weapon)I, (MOB)forMe, 0, true, false)/10;
147                                                                        MOB M=CMClass.getFactoryMOB(L("Someone inside @x1",forMe.name()), msg.source().phyStats().level(), ((MOB)forMe).location());
148                                                                        try
149                                                                        {
150                                                                                if(stomachR.show(msg.source(), forMe, CMMsg.MSG_NOISYMOVEMENT, L("<S-NAME> @x1 <T-YOUPOSS> stomach with @x2!",CMLib.combat().standardHitWord(weapon.weaponDamageType(),dmg),I.name(msg.source()))))
151                                                                                        CMLib.combat().postDamage(M, (MOB)forMe, I, dmg, CMMsg.MSG_WEAPONATTACK, weapon.weaponDamageType(), L("<S-NAME> <DAMAGE> <T-HIM-HER>!"));
152                                                                        }
153                                                                        finally
154                                                                        {
155                                                                                M.destroy();
156                                                                        }
157                                                                        return false;
158                                                                }
159                                                        }
160                                                }
161                                        }
162                                        return true;
163                                }
164                        }));
165                }
166        }

168        public void kill()
169        {
170                if((lastKnownLocationR==null)
171                ||(stomachR==null)) 
172                        return;
173
174                // ===== move all inhabitants to the dragons location
175                // ===== loop through all inhabitants of the stomach
176                final Vector<Physical> these=new Vector<Physical>();
177                for (int x=0;x<stomachR.numInhabitants();x++)
178                {
179                        // ===== get the tasty morsels
180                        final MOB tastyMorselM = stomachR.fetchInhabitant(x);
181                        if(tastyMorselM!=null)
182                                these.addElement(tastyMorselM);
183                }
184                
185                for(final Enumeration<MOB> p=CMLib.players().players();p.hasMoreElements();)
186                {
187                        final MOB M=p.nextElement();
188                        if((M!=null)&&(M.location()==stomachR)&&(!CMLib.flags().isInTheGame(M,true))&&(!these.contains(M)))
189                        {
190                                M.setLocation(lastKnownLocationR);
191                                for(int f=0;f<M.numFollowers();f++)
192                                {
193                                        MOB F=M.fetchFollower(f);
194                                        if((F!=null)&&(F.location()==stomachR)&&(!CMLib.flags().isInTheGame(F,true))&&(!these.contains(F)))
195                                                F.setLocation(lastKnownLocationR);
196                                }
197                                CMLib.database().DBUpdatePlayerMOBOnly(M);
198                        }
199                }
200
201                // =====move the inventory of the stomach to the room
202                for (int y=0;y<stomachR.numItems();y++)
203                {
204                        final Item PartiallyDigestedItem = stomachR.getItem(y);
205                        if((PartiallyDigestedItem!=null)&&(PartiallyDigestedItem.container()==null))
206                                these.addElement(PartiallyDigestedItem);
207                }
208
209                for(int i=0;i<these.size();i++)
210                {
211                        if(these.elementAt(i) instanceof Item)
212                                lastKnownLocationR.moveItemTo((Item)these.elementAt(i),ItemPossessor.Expire.Player_Drop);
213                        else
214                        if(these.elementAt(i) instanceof MOB)
215                                lastKnownLocationR.bringMobHere((MOB)these.elementAt(i),false);
216                }
217                stomachR.recoverPhyStats();
218                lastKnownLocationR.recoverRoomStats();
219                lastKnownLocationR=null;
220        }

222        @Override
223        public boolean tick(final Tickable ticking, final int tickID)
224        {
225                super.tick(ticking,tickID);
226                if(ticking==null) 
227                        return true;
228                if(!(ticking instanceof MOB))
229                        return true;
230
231                final MOB mob=(MOB)ticking;
232                if(this.lastKnownEaterM!=mob)
233                        this.lastKnownEaterM=mob;
234                if(mob.location()!=null)
235                        lastKnownLocationR=mob.location();
236
237                if((--digestDown)<=0)
238                {
239                        digestDown=4;
240                        digestTastyMorsels(mob);
241                }
242
243                if((canAct(ticking,tickID))
244                &&(((MOB)ticking).isInCombat())
245                &&(!mob.amDead()))
246                        trySwallowWhole(mob);
247                return true;
248        }

250        @Override
251        public void executeMsg(final Environmental myHost, final CMMsg msg)
252        {
253                if(this.lastKnownEaterM != null)
254                {
255                        if(((msg.amISource(this.lastKnownEaterM))&&(msg.sourceMinor()==CMMsg.TYP_DEATH))
256                        ||(!CMLib.flags().isInTheGame(this.lastKnownEaterM, true)))
257                                kill();
258                }
259                super.executeMsg(myHost,msg);
260        }

262        protected boolean trySwallowWhole(MOB mob)
263        {
264                if(stomachR==null) 
265                        return true;
266                if (CMLib.flags().isAliveAwakeMobile(mob,true)
267                        &&(mob.rangeToTarget()==0)
268                        &&(CMLib.flags().canHear(mob)||CMLib.flags().canSee(mob)||CMLib.flags().canSmell(mob)))
269                {
270                        final MOB tastyMorselM = mob.getVictim();
271                        if(tastyMorselM==null) 
272                                return true;
273                        if (tastyMorselM.baseWeight()<(mob.phyStats().weight()/2))
274                        {
275                                // ===== The player has been eaten.
276                                // ===== move the tasty morsel to the stomach
277                                final CMMsg eatMsg=CMClass.getMsg(mob,
278                                                                                                  tastyMorselM,
279                                                                                                  null,
280                                                                                                  CMMsg.MSG_EAT,
281                                                                                                  CMMsg.MASK_ALWAYS|CMMsg.TYP_JUSTICE,
282                                                                                                  CMMsg.MSG_NOISYMOVEMENT,
283                                                                                                  L("<S-NAME> swallow(es) <T-NAMESELF> WHOLE!"));
284                                if(mob.location().okMessage(tastyMorselM,eatMsg))
285                                {
286                                        mob.location().send(tastyMorselM,eatMsg);
287                                        if(eatMsg.value()==0)
288                                        {
289                                                stomachR.bringMobHere(tastyMorselM,false);
290                                                final CMMsg enterMsg=CMClass.getMsg(tastyMorselM,stomachR,null,CMMsg.MSG_ENTER,stomachR.description(),CMMsg.MSG_ENTER,null,CMMsg.MSG_ENTER,L("<S-NAME> slide(s) down the gullet into the stomach!"));
291                                                stomachR.send(tastyMorselM,enterMsg);
292                                        }
293                                }
294                        }
295                }
296                return true;
297        }
298
299        protected boolean digestTastyMorsels(MOB mob)
300        {
301                if(stomachR==null) 
302                        return true;
303                // ===== loop through all inhabitants of the stomach
304                final int morselCount = stomachR.numInhabitants();
305                for (int x=0;x<morselCount;x++)
306                {
307                        // ===== get a tasty morsel
308                        final MOB tastyMorselM = stomachR.fetchInhabitant(x);
309                        if (tastyMorselM != null)
310                        {
311                                final CMMsg DigestMsg=CMClass.getMsg(mob,
312                                                                                                         tastyMorselM,
313                                                                                                         null,
314                                                                                                         CMMsg.MASK_ALWAYS|CMMsg.TYP_ACID,
315                                                                                                         L("<S-NAME> digest(s) <T-NAMESELF>!!"));
316                                // no OKaffectS, since the dragon is not in his own stomach.
317                                stomachR.send(mob,DigestMsg);
318                                int damage=(int)Math.round(tastyMorselM.maxState().getHitPoints() * CMath.div(pctAcidHp, 100));
319                                if(damage<2) 
320                                        damage=2;
321                                if(DigestMsg.value()!=0) 
322                                        damage=damage/2;
323                                CMLib.combat().postDamage(mob,tastyMorselM,null,damage,CMMsg.MASK_ALWAYS|CMMsg.TYP_ACID,Weapon.TYPE_MELTING,L("The stomach acid <DAMAGE> <T-NAME>!"));
324                        }
325                }
326                return true;
327        }
