036public class ScriptableEverymob extends StdBehavior implements ScriptingEngine
037{
038        @Override
039        public String ID()
040        {
041                return "ScriptableEverymob";
042        }
043
044        @Override
045        protected int canImproveCode()
046        {
047                return Behavior.CAN_ROOMS|Behavior.CAN_AREAS;
048        }
049
050        private boolean started=false;
051        private Scriptable sampleB=null;
052
053        @Override
054        public String accountForYourself()
055        {
056                return "complex triggered behaving";
057        }
058
059        private void giveUpTheScript(Area metroA, MOB M)
060        {
061                if((M==null)
062                ||(!M.isMonster())
063                ||(M.getStartRoom()==null)
064                ||(metroA==null)
065                ||(!metroA.inMyMetroArea(M.getStartRoom().getArea()))
066                ||(M.fetchBehavior("Scriptable")!=null))
067                        return;
068                final Scriptable S=new Scriptable();
069                S.setParms(getParms());
070                S.setSavable(false);
071                M.addBehavior(S);
072                S.setSavable(false);
073                sampleB=S;
074        }
075
076        private Area determineArea(Environmental forMe)
077        {
078                if(forMe instanceof Room)
079                        return ((Room)forMe).getArea();
080                else
081                if(forMe instanceof Area)
082                        return (Area)forMe;
083                return null;
084        }
085
086        private Enumeration<Room> determineRooms(Environmental forMe)
087        {
088                if(forMe instanceof Room)
089                        return new XVector<Room>((Room)forMe).elements();
090                else
091                if(forMe instanceof Area)
092                        return ((Area)forMe).getMetroMap();
093                return null;
094        }
095
096        private void giveEveryoneTheScript(Environmental forMe)
097        {
098                if((CMProps.getBoolVar(CMProps.Bool.MUDSTARTED))
099                &&(!started))
100                {
101                        started = true;
102                        final Enumeration<Room> rooms=determineRooms(forMe);
103                        final Area A=determineArea(forMe);
104                        if((A!=null)&&(rooms!=null))
105                        {
106                                Room R=null;
107                                for(;rooms.hasMoreElements();)
108                                {
109                                        R=rooms.nextElement();
110                                        for(int m=0;m<R.numInhabitants();m++)
111                                                giveUpTheScript(A,R.fetchInhabitant(m));
112                                }
113                        }
114                }
115        }
116
117        @Override
118        public boolean tick(final Tickable ticking, final int tickID)
119        {
120                if((!started)&&(ticking instanceof Environmental))
121                        giveEveryoneTheScript((Environmental)ticking);
122                return super.tick(ticking, tickID);
123        }
124
125        @Override
126        public void startBehavior(PhysicalAgent forMe)
127        {
128                giveEveryoneTheScript(forMe);
129        }
130
131        @Override
132        public void executeMsg(final Environmental host, final CMMsg msg)
133        {
134                if((msg.target() instanceof Room)
135                &&(msg.targetMinor()==CMMsg.TYP_LOOK))
136                        giveUpTheScript(determineArea(host),msg.source());
137                super.executeMsg(host,msg);
138        }
