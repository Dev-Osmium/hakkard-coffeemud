035public class ROMGangMember extends StdBehavior
036{
037        @Override
038        public String ID()
039        {
040                return "ROMGangMember";
041        }
042
043        int tickTock=5;
044        String gangName="";
045        String[] messages=new String[0];
046
047        @Override
048        public String accountForYourself()
049        {
050                return "gang membership";
051        }
052
053        @Override
054        public void setParms(String newParms)
055        {
056                super.setParms(newParms);
057                List<String> list=CMParms.parseSemicolons(newParms, true);
058                messages=new String[0];
059                if(list.size()==0)
060                        gangName=CMLib.dice().rollPercentage()>50?"Red":"Blue";
061                else
062                {
063                        gangName=list.remove(0);
064                        messages=list.toArray(messages);
065                }
066        }
067        
068        public void pickAFight(MOB observer)
069        {
070                if(!canFreelyBehaveNormal(observer))
071                        return;
072                final Room R=observer.location();
073                if(R==null)
074                        return;
075                if(R.numPCInhabitants()==0)
076                        return;
077
078                MOB victim=null;
079                String vicParms="";
080                for(int i=0;i<R.numInhabitants();i++)
081                {
082                        final MOB inhab=R.fetchInhabitant(i);
083                        if((inhab!=null)
084                        &&((inhab.isMonster())||(CMLib.clans().findCommonRivalrousClans(inhab,observer).size()==0)))
085                        {
086                                for(final Enumeration<Behavior> e=inhab.behaviors();e.hasMoreElements();)
087                                {
088                                        final Behavior B=e.nextElement();
089                                        if(B.ID().equals(ID())&&(!((ROMGangMember)B).gangName.equals(gangName)))
090                                        {
091                                                victim=inhab;
092                                                vicParms=((ROMGangMember)B).gangName;
093                                        }
094                                        else
095                                        if((B.ID().indexOf("GoodGuardian")>=0)||(B.ID().indexOf("Patrolman")>=0))
096                                                return;
097                                }
098                        }
099                }
100
101                if(victim==null)
102                        return;
103                Item weapon=observer.fetchWieldedItem();
104                if(weapon==null)
105                        weapon=observer.getNaturalWeapon();
106
107                /* say something, then raise hell */
108                if(messages.length>0)
109                {
110                        R.show(observer,null,CMMsg.MSG_SPEAK,"^T<S-NAME> say(s) '"+messages[CMLib.dice().roll(1,messages.length,-1)]+"'.^?");
111                }
112                else
113                {
114                        switch (CMLib.dice().roll(1,7,-1))
115                        {
116                        case 0:
117                                R.show(observer,null,CMMsg.MSG_SPEAK,L("^T<S-NAME> yell(s) 'I've been looking for you, punk!'^?"));
118                                break;
119                        case 1:
120                                R.show(observer,victim,CMMsg.MSG_NOISYMOVEMENT,L("With a scream of rage, <S-NAME> attack(s) <T-NAME>."));
121                                break;
122                        case 2:
123                                R.show(observer,victim,CMMsg.MSG_SPEAK,L("^T<S-NAME> say(s) 'What's slimy @x1 trash like you doing around here?'^?",vicParms));
124                                break;
125                        case 3:
126                                R.show(observer,victim,CMMsg.MSG_SPEAK,L("^T<S-NAME> crack(s) <S-HIS-HER> knuckles and say(s) 'Do ya feel lucky?'^?"));
127                                break;
128                        case 4:
129                                R.show(observer,victim,CMMsg.MSG_SPEAK,L("^T<S-NAME> say(s) 'There's no cops to save you this time!'^?"));
130                                break;
131                        case 5:
132                                R.show(observer,victim,CMMsg.MSG_SPEAK,L("^T<S-NAME> say(s) 'Time to join your brother, spud.'^?"));
133                                break;
134                        case 6:
135                                R.show(observer,victim,CMMsg.MSG_SPEAK,L("^T<S-NAME> say(s) 'Let's rock.'^?"));
136                                break;
137                        }
138                }
139
140                CMLib.combat().postAttack(observer,victim,weapon);
141        }
142
143        @Override
144        public boolean tick(final Tickable ticking, final int tickID)
145        {
146                super.tick(ticking,tickID);
147
148                if(tickID!=Tickable.TICKID_MOB)
149                        return true;
150                final MOB mob=(MOB)ticking;
151                tickTock--;
152                if(tickTock<=0)
153                {
154                        tickTock=CMLib.dice().roll(1,10,0);
155                        pickAFight(mob);
156                }
157                return true;
158        }
159}
