166        public int weather(Environmental host, Room room)
167        {
168                if(room==null)
169                        return 0;
170                final Area A=(host instanceof Area)?(Area)host:CMLib.map().roomLocation(host).getArea();

176        private boolean isInclement(final int weather)
177        {
178                switch(weather)
179                {
180                        case Climate.WEATHER_BLIZZARD:
181                        case Climate.WEATHER_SNOW:
182                        case Climate.WEATHER_RAIN:
183                        case Climate.WEATHER_THUNDERSTORM:
184                        case Climate.WEATHER_SLEET:
185                        case Climate.WEATHER_DUSTSTORM:
186                                return true;
187                        default:
188                                return false;
189                }
190        }

192        @Override
193        public boolean okMessage(final Environmental host, final CMMsg msg)
194        {
195                if(!super.okMessage(host,msg))
196                        return false;
197
198                final Room R=msg.source().location();
199                if((host instanceof Area)
200                &&(R!=null)&&(R.getArea()!=host))
201                        return true;
202                final int weather=weather(host,R);
203                // first handle the effect of storms on ranged
204                // weapons
205
206                if((msg.targetMinor()==CMMsg.TYP_WEAPONATTACK)
207                &&(msg.source().rangeToTarget()!=0)
208                &&(msg.tool() instanceof Item)
209                &&(!(msg.tool() instanceof Electronics))
210                &&((msg.sourceMinor()==CMMsg.TYP_THROW)
211                        ||((msg.tool() instanceof Weapon)
212                                &&((((Weapon)msg.tool()).weaponClassification()==Weapon.CLASS_RANGED)
213                                   ||(((Weapon)msg.tool()).weaponClassification()==Weapon.CLASS_THROWN)))))
214                {
215                        switch(weather)
216                        {
217                        case Climate.WEATHER_WINDY:
218                        case Climate.WEATHER_THUNDERSTORM:
219                        case Climate.WEATHER_BLIZZARD:
220                        case Climate.WEATHER_DUSTSTORM:
221                        {
222                                if((CMLib.dice().rollPercentage()<windsheer)
223                                &&(R!=null))
224                                {
225                                        if(R.show(msg.source(),msg.target(),msg.tool(),CMMsg.MSG_WEATHER,L("^WThe strong wind blows <S-YOUPOSS> attack against <T-NAMESELF> with <O-NAME> off target.^?")))
226                                                return false;
227                                }
228                                break;
229                        }
230                        }
231                }
232                // then try to handle slippage of boats and feet in bad weather
233                if(((msg.sourceMajor(CMMsg.MASK_MOVE)))&&(R!=null)&&(isInclement(weather)))
234                {
235                        switch(R.domainType())
236                        {
237                        case Room.DOMAIN_INDOORS_AIR:
238                        case Room.DOMAIN_OUTDOORS_AIR:
239                                break;
240                        case Room.DOMAIN_INDOORS_WATERSURFACE:
241                        case Room.DOMAIN_OUTDOORS_WATERSURFACE:
242                        {
243                                final Rideable riding=msg.source().riding();
244                                if((riding!=null)
245                                &&((riding.rideBasis()==Rideable.RIDEABLE_WATER)||(riding instanceof BoardableShip))
246                                &&(!CMLib.flags().isABonusItems(riding))
247                                &&(!riding.phyStats().isAmbiance("-ANTIWEATHER")))
248                                {
249                                        String what=null;
250                                        switch(weather)
251                                        {
252                                        case Climate.WEATHER_SNOW:
253                                                if(((R.getClimateType()&Places.CLIMASK_WINDY)!=0)
254                                                &&(CMLib.dice().rollPercentage()<boatSlipChance))
255                                                        what="cold snowy winds"; // never L(
256                                                break;
257                                        case Climate.WEATHER_RAIN:
258                                                if(((R.getClimateType()&Places.CLIMASK_WINDY)!=0)
259                                                &&(CMLib.dice().rollPercentage()<boatSlipChance))
260                                                        what="strong rainy winds"; // never L(
261                                                break;
262                                        case Climate.WEATHER_BLIZZARD:
263                                                if(CMLib.dice().rollPercentage()<(boatSlipChance * 1.5))
264                                                        what="blizzard"; // never L(
265                                                break;
266                                        case Climate.WEATHER_THUNDERSTORM:
267                                                if(CMLib.dice().rollPercentage()<(boatSlipChance * 1.5))
268                                                        what="thunderstorm"; // never L(
269                                                break;
270                                        case Climate.WEATHER_DUSTSTORM:
271                                                if(CMLib.dice().rollPercentage()<boatSlipChance)
272                                                        what="strong winds"; // never L(
273                                                break;
274                                        }
275                                        if(what!=null)
276                                        {
277                                                if(R.show(msg.source(),null,CMMsg.MSG_WEATHER,L("^W<S-NAME> make(s) no progress in the "+what+".^?")))
278                                                {
279                                                        if(riding instanceof BoardableShip)
280                                                        {
281                                                                final Area shipArea=((BoardableShip)riding).getShipArea();
282                                                                if(shipArea != null)
283                                                                {
284                                                                        for(final Enumeration<Room> sr=shipArea.getProperMap();sr.hasMoreElements();)
285                                                                        {
286                                                                                final Room sR=sr.nextElement();
287                                                                                if((sR!=null)&&((sR.domainType()&Room.INDOORS)==0))
288                                                                                        sR.show(msg.source(),null,CMMsg.MSG_OK_ACTION,L("^W<S-NAME> make(s) no progress in the "+what+".^?"));
289                                                                        }
290                                                                }
291                                                                
292                                                        }
293                                                        return false;
294                                                }
295                                        }
296                                }
297                                break;
298                        }
299                        case Room.DOMAIN_INDOORS_UNDERWATER:
300                        case Room.DOMAIN_OUTDOORS_UNDERWATER:
301                                break;
302                        default:
303                        {
304                                String what=null;
305                                switch(weather)
306                                {
307                                case Climate.WEATHER_BLIZZARD:
308                                case Climate.WEATHER_SNOW:
309                                        if(CMLib.dice().rollPercentage()<snowSlipChance)
310                                                what="cold wet"; // never L(
311                                        break;
312                                case Climate.WEATHER_RAIN:
313                                case Climate.WEATHER_THUNDERSTORM:
314                                        if(CMLib.dice().rollPercentage()<rainSlipChance)
315                                                what="slippery wet"; // never L(
316                                        break;
317                                case Climate.WEATHER_SLEET:
318                                        if(CMLib.dice().rollPercentage()<sleetSlipChance)
319                                                what="icy"; // never L(
320                                        break;
321                                }
322                                if((what!=null)
323                                &&(!CMLib.flags().isInFlight(msg.source()))
324                                &&(CMLib.flags().isStanding(msg.source()))
325                                &&(msg.source().riding()==null)
326                                &&(CMLib.dice().rollPercentage()>((msg.source().charStats().getStat(CharStats.STAT_DEXTERITY)*3)+25)))
327                                {
328                                        if(R.show(msg.source(),null,CMMsg.MSG_WEATHER,L("^W<S-NAME> slip(s) on the "+what+" ground.^?")))
329                                        {
330                                                Ability A=CMClass.getAbility("Skill_Trip");
331                                                A.startTickDown(msg.source(), msg.source(), 3);
332                                                msg.source().recoverPhyStats();
333                                                return false;
334                                        }
335                                }
336                        }
337                        }
338                }
339                if((R!=null)
340                &&(weather==Climate.WEATHER_BLIZZARD))
341                {
342                        final Ability A=CMClass.getAbility("Spell_ObscureSelf");
343                        if(A!=null)
344                        {
345                                A.setAffectedOne(msg.source());
346                                if(!A.okMessage(msg.source(),msg))
347                                        return false;
348                        }
349                }
350                return true;
351        }

353        protected boolean isInArea(final Area A, MOB M)
354        {
355                if((M!=null)
356                &&(M.location()!=null)
357                &&(M.location().getArea()!=null))
358                {
359                        final Area mA=M.location().getArea();
360                        if(mA==A)
361                                return true;
362                        if(mA instanceof BoardableShip)
363                        {
364                                Area inA=CMLib.map().areaLocation(((BoardableShip)mA).getShipItem());
365                                return inA==A;
366                        }
367                }
368                return false;
369        }

371        @Override
372        public boolean tick(final Tickable ticking, final int tickID)
373        {
374                final int realLastWeather=super.lastWeather;
375                if(!super.tick(ticking,tickID))
376                        return false;
377                final Area A=CMLib.map().areaLocation(ticking);
378                if(A==null)
379                        return false;
380                final Climate C=A.getClimateObj();
381                if(C==null)
382                        return false;
383                lastWeather=realLastWeather;
384
385                // handle freeze overs
386                if((coldWeather(lastWeather))
387                &&(coldWeather(C.weatherType(null)))
388                &&(lastWeather!=C.weatherType(null))
389                &&(A.getTimeObj().getSeasonCode()==TimeClock.Season.WINTER)
390                &&(CMLib.dice().rollPercentage()<freezeOverChance))
391                {
392                        if(ticking instanceof Room)
393                        {
394                                final Room R=(Room)ticking;
395                                if((R.domainType()==Room.DOMAIN_OUTDOORS_WATERSURFACE)
396                                &&(CMLib.dice().rollPercentage()<freezeOverChance)
397                                &&(R instanceof Drink)
398                                &&(((Drink)R).liquidType()==RawMaterial.RESOURCE_FRESHWATER))
399                                {
400                                        final Ability A2=CMClass.getAbility("Spell_IceSheet");
401                                        if(A2!=null)
402                                        {
403                                                final MOB mob=CMLib.map().getFactoryMOB(R);
404                                                A2.invoke(mob,R,true,0);
405                                                mob.destroy();
406                                        }
407                                }
408                        }
409                        else
410                        for(final Enumeration<Room> e=A.getProperMap();e.hasMoreElements();)
411                        {
412                                final Room R=e.nextElement();
413                                if((R.domainType()==Room.DOMAIN_OUTDOORS_WATERSURFACE)
414                                &&(CMLib.dice().rollPercentage()<freezeOverChance))
415                                {
416                                        final Ability A2=CMClass.getAbility("Spell_IceSheet");
417                                        if(A2!=null)
418                                        {
419                                                final MOB mob=CMLib.map().getFactoryMOB(R);
420                                                A2.invoke(mob,R,true,0);
421                                                mob.destroy();
422                                        }
423                                }
424                        }
425                }
426                if((botherDown--)==1)
427                {
428                        resetBotherTicks();
429                        switch(C.weatherType(null))
430                        {
431                        case Climate.WEATHER_BLIZZARD:
432                        case Climate.WEATHER_SLEET:
433                        case Climate.WEATHER_SNOW:
434                        case Climate.WEATHER_HAIL:
435                        case Climate.WEATHER_THUNDERSTORM:
436                        case Climate.WEATHER_RAIN:
437                        {
438                                final Enumeration<Room> r=(ticking instanceof Room)?new SingleEnumeration<Room>((Room)ticking):A.getProperMap();
439                                for(;r.hasMoreElements();)
440                                {
441                                        final Room R=r.nextElement();
442                                        if(CMLib.map().hasASky(R) && (R.numInhabitants() > 0))
443                                        {
444                                                for(int i=0;i<R.numInhabitants();i++)
445                                                {
446                                                        final MOB mob=R.fetchInhabitant(i);
447                                                        if((mob!=null)
448                                                        &&(!mob.isMonster())
449                                                        &&(CMLib.flags().isAliveAwakeMobile(mob,true))
450                                                        &&(mob.isAttributeSet(MOB.Attrib.AUTOWEATHER)))
451                                                                mob.tell(C.getWeatherDescription(A));
452                                                }
453                                        }
454                                }
455                                break;
456                        }
457                        default:
458                                break;
459                        }
460                }
461                if((diseaseDown--)==1)
462                {
463                        resetDiseaseTicks();
464                        int coldChance=0;
465                        int fluChance=0;
466                        int frostBiteChance=0;
467                        int heatExhaustionChance=0;
468                        switch(C.weatherType(null))
469                        {
470                        case Climate.WEATHER_BLIZZARD:
471                        case Climate.WEATHER_SLEET:
472                        case Climate.WEATHER_SNOW:
473                                coldChance=99;
474                                fluChance=25;
475                                frostBiteChance=15;
476                                break;
477                        case Climate.WEATHER_HAIL:
478                                coldChance=50;
479                                frostBiteChance=10;
480                                break;
481                        case Climate.WEATHER_THUNDERSTORM:
482                        case Climate.WEATHER_RAIN:
483                                coldChance=25;
484                                break;
485                        case Climate.WEATHER_WINTER_COLD:
486                                coldChance=75;
487                                fluChance=10;
488                                frostBiteChance=5;
489                                break;
490                        case Climate.WEATHER_HEAT_WAVE:
491                                heatExhaustionChance=15;
492                                break;
493                        case Climate.WEATHER_DROUGHT:
494                                heatExhaustionChance=20;
495                                break;
496                        }
497
498                        for(final Session S : CMLib.sessions().localOnlineIterable())
499                        {
500                                if((S.mob()==null)
501                                ||(!isInArea(A,S.mob()))
502                                ||(S.mob().isMonster()))
503                                        continue;
504
505                                final MOB M=S.mob();
506                                final Room R=M.location();
507
508                                if((R.getClimateType()&Places.CLIMASK_COLD)>0)
509                                {
510                                        if(coldChance>0)
511                                                coldChance+=10;
512                                        if(coldChance>0) fluChance+=5; // yes, cold is related this way to flu
513                                        if(frostBiteChance>0)
514                                                frostBiteChance=frostBiteChance+(int)Math.round(CMath.mul(frostBiteChance,0.5));
515                                }
516                                if((R.getClimateType()&Places.CLIMASK_HOT)>0)
517                                {
518                                        if(heatExhaustionChance>0)
519                                                heatExhaustionChance+=10;
520                                }
521                                if((R.getClimateType()&Places.CLIMASK_WET)>0)
522                                {
523                                        if(coldChance>0)
524                                                coldChance+=5;
525                                        if(heatExhaustionChance>5)
526                                                heatExhaustionChance-=5;
527                                        if(frostBiteChance>0)
528                                                frostBiteChance=frostBiteChance+(int)Math.round(CMath.mul(frostBiteChance,0.25));
529                                }
530                                final int save=(M.charStats().getSave(CharStats.STAT_SAVE_COLD)+M.charStats().getSave(CharStats.STAT_SAVE_WATER))/2;
531                                if((CMLib.dice().rollPercentage()<(coldChance-save))
532                                &&((C.weatherType(M.location())!=Climate.WEATHER_CLEAR)))
533                                {
534                                        long coveredPlaces=0;
535                                        for (final long element : ALL_COVERED_SPOTS)
536                                        {
537                                                if(M.getWearPositions(element)==0)
538                                                        coveredPlaces=coveredPlaces|element;
539                                        }
540                                        Item I=null;
541                                        for(int i=0;i<M.numItems();i++)
542                                        {
543                                                I=M.getItem(i);
544                                                if((I==null)||(!I.amBeingWornProperly()))
545                                                        continue;
546                                                if(I.amWearingAt(Wearable.WORN_ABOUT_BODY))
547                                                        coveredPlaces=coveredPlaces|Wearable.WORN_TORSO|Wearable.WORN_LEGS;
548                                                for (final long element : ALL_COVERED_SPOTS)
549                                                {
550                                                        if(I.amWearingAt(element))
551                                                                coveredPlaces=coveredPlaces|element;
552                                                }
553                                        }
554                                        if((coveredPlaces!=ALL_COVERED_CODE)&&(!CMSecurity.isDisabled(CMSecurity.DisFlag.AUTODISEASE)))
555                                        {
556                                                Ability COLD=CMClass.getAbility("Disease_Cold");
557                                                if(CMLib.dice().rollPercentage()<(fluChance+(((M.location().getClimateType()&Places.CLIMASK_WET)>0)?10:0)))
558                                                        COLD=CMClass.getAbility("Disease_Flu");
559                                                if((COLD!=null)&&(M.fetchEffect(COLD.ID())==null)&&(!CMSecurity.isAbilityDisabled(A.ID())))
560                                                        COLD.invoke(M,M,true,0);
561                                        }
562                                }
563                                if((CMLib.dice().rollPercentage()<(frostBiteChance-save))
564                                &&((C.weatherType(M.location())!=Climate.WEATHER_CLEAR)))
565                                {
566                                        long unfrostedPlaces=0;
567                                        for (final long element : ALL_FROST_SPOTS)
568                                        {
569                                                if(M.getWearPositions(element)==0)
570                                                        unfrostedPlaces=unfrostedPlaces|element;
571                                        }
572                                        Item I=null;
573                                        for(int i=0;i<M.numItems();i++)
574                                        {
575                                                I=M.getItem(i);
576                                                if((I==null)||(I.amWearingAt(Wearable.IN_INVENTORY)))
577                                                        continue;
578                                                for (final long element : ALL_FROST_SPOTS)
579                                                {
580                                                        if(I.amWearingAt(element))
581                                                                unfrostedPlaces=unfrostedPlaces|element;
582                                                }
583                                        }
584                                        if((unfrostedPlaces!=ALL_FROST_CODE)
585                                        &&(!CMSecurity.isDisabled(CMSecurity.DisFlag.AUTODISEASE)))
586                                        {
587                                                final Ability COLD=CMClass.getAbility("Disease_FrostBite");
588                                                if((COLD!=null)&&(M.fetchEffect(COLD.ID())==null)&&(!CMSecurity.isAbilityDisabled(A.ID())))
589                                                        COLD.invoke(M,M,true,0);
590                                        }
