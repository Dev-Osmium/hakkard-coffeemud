074        @Override
075        public void setParms(String newParms)
076        {
077                favorMobs=false;
078                maintained=new Vector<Item>();
079                restrictedLocales=null;
080                String parms=newParms;
081                if(parms.indexOf(';')>=0)
082                        parms=parms.substring(0,parms.indexOf(';'));
083                final Vector<String> V=CMParms.parse(parms);
084                for(int v=0;v<V.size();v++)
085                {
086                        String s=V.elementAt(v);
087                        if(s.equalsIgnoreCase("MOBS"))
088                                favorMobs=true;
089                        else
090                        if((s.startsWith("+")||s.startsWith("-"))&&(s.length()>1))
091                        {
092                                if(restrictedLocales==null)
093                                        restrictedLocales=new Vector<Integer>();
094                                if(s.equalsIgnoreCase("+ALL"))
095                                        restrictedLocales.clear();
096                                else
097                                if(s.equalsIgnoreCase("-ALL"))
098                                {
099                                        restrictedLocales.clear();
100                                        for(int i=0;i<Room.DOMAIN_INDOORS_DESCS.length;i++)
101                                                restrictedLocales.addElement(Integer.valueOf(Room.INDOORS+i));
102                                        for(int i=0;i<Room.DOMAIN_OUTDOOR_DESCS.length;i++)
103                                                restrictedLocales.addElement(Integer.valueOf(i));
104                                }
105                                else
106                                {
107                                        final char c=s.charAt(0);
108                                        s=s.substring(1).toUpperCase().trim();
109                                        int code=-1;
110                                        for(int i=0;i<Room.DOMAIN_INDOORS_DESCS.length;i++)
111                                        {
112                                                if(Room.DOMAIN_INDOORS_DESCS[i].startsWith(s))
113                                                        code=Room.INDOORS+i;
114                                        }
115                                        if(code>=0)
116                                        {
117                                                if((c=='+')&&(restrictedLocales.contains(Integer.valueOf(code))))
118                                                        restrictedLocales.removeElement(Integer.valueOf(code));
119                                                else
120                                                if((c=='-')&&(!restrictedLocales.contains(Integer.valueOf(code))))
121                                                        restrictedLocales.addElement(Integer.valueOf(code));
122                                        }
123                                        code=-1;
124                                        for(int i=0;i<Room.DOMAIN_OUTDOOR_DESCS.length;i++)
125                                        {
126                                                if(Room.DOMAIN_OUTDOOR_DESCS[i].startsWith(s))
127                                                        code=i;
128                                        }
129                                        if(code>=0)
130                                        {
131                                                if((c=='+')&&(restrictedLocales.contains(Integer.valueOf(code))))
132                                                        restrictedLocales.removeElement(Integer.valueOf(code));
133                                                else
134                                                if((c=='-')&&(!restrictedLocales.contains(Integer.valueOf(code))))
135                                                        restrictedLocales.addElement(Integer.valueOf(code));
136                                        }
137                                }
138                        }
139                }
140                super.setParms(newParms);
141                minItems=CMParms.getParmInt(parms,"minitems",1);
142                maxItems=CMParms.getParmInt(parms,"maxitems",1);
143                maxDups=CMParms.getParmInt(parms,"maxdups",Integer.MAX_VALUE);
144                if(minItems>maxItems)
145                        maxItems=minItems;
146                avgItems=CMLib.dice().roll(1,maxItems-minItems,minItems);
147                enchantPct=CMParms.getParmInt(parms,"enchanted",10);
148                if((restrictedLocales!=null)&&(restrictedLocales.size()==0))
149                        restrictedLocales=null;
150        }

158        public boolean okRoomForMe(Room newRoom)
159        {
160                if(newRoom==null)
161                        return false;
162                if(restrictedLocales==null)
163                        return true;
164                return !restrictedLocales.contains(Integer.valueOf(newRoom.domainType()));
165        }

167        public boolean isStillMaintained(Environmental thang, ShopKeeper SK, Item I)
168        {
169                if((I==null)||(I.amDestroyed()))
170                        return false;
171                if(SK!=null)
172                {
173                        final CoffeeShop shop=(SK instanceof Librarian)?((Librarian)SK).getBaseLibrary():SK.getShop();
174                        return shop.doIHaveThisInStock(I.Name(),null);
175                }
176                if(thang instanceof Area)
177                {
178                        final Room R=CMLib.map().roomLocation(I);
179                        if(R==null)
180                                return false;
181                        return ((Area)thang).inMyMetroArea(R.getArea());
182                }
183                else
184                if(thang instanceof Room)
185                        return CMLib.map().roomLocation(I)==thang;
186                else
187                if(thang instanceof MOB)
188                        return (I.owner()==thang);
189                else
190                if(thang instanceof Container)
191                        return (I.owner()==((Container)thang).owner())&&(I.container()==thang);
192                return I.owner()==CMLib.map().roomLocation(thang);
193        }

240                @Override
241                public boolean tick(final Tickable ticking, final int tickID)
242                {
243                        @SuppressWarnings("unchecked")
244                        List<Item> allItems=(List<Item>)Resources.getResource("ITEMGENERATOR-ALLITEMS");
245                        if(allItems!=null)
246                                return false;
247                        allItems=new Vector<Item>();
248
249                        final List<ItemCraftor> skills=new Vector<ItemCraftor>();
250                        for(final Enumeration<Ability> e=CMClass.abilities();e.hasMoreElements();)
251                        {
252                                final Ability A=e.nextElement();
253                                if(A instanceof ItemCraftor)
254                                        skills.add((ItemCraftor)A.copyOf());
255                        }
256                        List<ItemCraftor.ItemKeyPair> skillSet=null;
257                        for(final ItemCraftor skill : skills)
258                        {
259                                skillSet=skill.craftAllItemSets(false);
260                                if(skillSet!=null)
261                                for(final ItemCraftor.ItemKeyPair materialSet: skillSet)
262                                        allItems.add(materialSet.item);
263                        }
264                        Resources.submitResource("ITEMGENERATOR-ALLITEMS",allItems);
265                        return false;
266                }
267        }

269        @SuppressWarnings("unchecked")
270        public synchronized GeneratedItemSet getItems(Tickable thang, String theseparms)
271        {
272                String mask=parms;
273                if(mask.indexOf(';')>=0)
274                        mask=mask.substring(parms.indexOf(';')+1);
275                GeneratedItemSet items=(GeneratedItemSet)Resources.getResource("ITEMGENERATOR-"+mask.toUpperCase().trim());
276                if(items==null)
277                {
278                        List<Item> allItems=(List<Item>)Resources.getResource("ITEMGENERATOR-ALLITEMS");
279                        if(allItems==null)
280                        {
281                                synchronized(builerTick)
282                                {
283                                        allItems=(List<Item>)Resources.getResource("ITEMGENERATOR-ALLITEMS");
284                                        if(allItems==null)
285                                        {
286                                                if(builerTick[0]==null)
287                                                {
288                                                        builerTick[0]=new ItemGenerationTicker();
289                                                        CMLib.threads().startTickDown(builerTick[0],Tickable.TICKID_ITEM_BEHAVIOR|Tickable.TICKID_LONGERMASK,1234,1);
290                                                }
291                                                return null;
292                                        }
293                                }
294                        }
295                        items=new GeneratedItemSet();
296                        Item I=null;
297                        final MaskingLibrary.CompiledZMask compiled=CMLib.masking().maskCompile(mask);
298                        double totalValue=0;
299                        int maxValue=-1;
300                        for(int a=0;a<allItems.size();a++)
301                        {
302                                I=allItems.get(a);
303                                if((CMLib.masking().maskCheck(compiled,I,true))
304                                &&(!(I instanceof ClanItem)))
305                                {
306                                        if(I.value()>maxValue)
307                                                maxValue=I.value();
308                                        items.add(I);
309                                }
310                        }
311                        for(int a=0;a<items.size();a++)
312                        {
313                                I=items.get(a);
314                                totalValue+=CMath.div(maxValue,I.value()+1);
315                        }
316                        if(items.size()>0)
317                        {
318                                items.maxValue=maxValue;
319                                items.totalValue=totalValue;
320                        }
321                        Resources.submitResource("ITEMGENERATOR-"+mask.toUpperCase().trim(),items);
322                }
323                return items;
324        }

326        @Override
327        public boolean tick(final Tickable ticking, final int tickID)
328        {
329                super.tick(ticking,tickID);
330                if((!CMProps.getBoolVar(CMProps.Bool.MUDSTARTED))
331                ||(!(ticking instanceof Environmental))
332                ||(CMSecurity.isDisabled(CMSecurity.DisFlag.RANDOMITEMS)))
333                        return true;
334                Item I=null;
335                final Environmental E=(Environmental)ticking;
336                final ShopKeeper SK=CMLib.coffeeShops().getShopKeeper(E);
337                for(int i=maintained.size()-1;i>=0;i--)
338                {
339                        try
340                        {
341                                I=maintained.elementAt(i);
342                                if(!isStillMaintained(E,SK,I))
343                                        maintained.removeElement(I);
344                        }
345                        catch (final Exception e)
346                        {
347                        }
348                }
349                if(maintained.size()>=maxItems)
350                        return true;
351                if((canAct(ticking,tickID))||(maintained.size()<minItems))
352                {
353                        final GeneratedItemSet items=getItems(ticking,getParms());
354                        if(items==null)
355                                return true;
356                        if((ticking instanceof Environmental)&&(((Environmental)ticking).amDestroyed()))
357                                return false;
358
359                        if((maintained.size()<avgItems)
360                        &&(items.size()>1))
361                        {
362                                final double totalValue=items.totalValue;
363                                final int maxValue=items.maxValue;
364                                double pickedTotal=Math.random()*totalValue;
365                                double value=-1;
366                                for(int i=2;i<items.size();i++)
367                                {
368                                        I=items.elementAt(i);
369                                        value=CMath.div(maxValue,I.value()+1.0);
370                                        if(pickedTotal<=value)
371                                        {
372                                                break;
373                                        }
374                                        pickedTotal-=value;
375                                }
376                                if(I!=null)
377                                {
378
379                                        if((maxDups<Integer.MAX_VALUE)&&(maxDups>0))
380                                        {
381                                                int numDups=0;
382                                                for(int m=0;m<maintained.size();m++)
383                                                {
384                                                        if(I.sameAs(maintained.elementAt(m)))
385                                                                numDups++;
386                                                }
387                                                if((maxDups>0)&&(numDups>=maxDups))
388                                                        return true;
389                                        }
390
391                                        I=(Item)I.copyOf();
392                                        I.basePhyStats().setRejuv(PhyStats.NO_REJUV);
393                                        I.recoverPhyStats();
394                                        I.text();
395                                        if(SK!=null)
396                                        {
397                                                if(SK.doISellThis(I))
398                                                {
399                                                        maintained.addElement(I);
400                                                        final CoffeeShop shop=(SK instanceof Librarian)?((Librarian)SK).getBaseLibrary():SK.getShop();
401                                                        shop.addStoreInventory(CMLib.itemBuilder().enchant(I,enchantPct),1,-1);
402                                                }
403                                        }
404                                        else
405                                        if(ticking instanceof Container)
406                                        {
407                                                if(((Container)ticking).owner() instanceof Room)
408                                                        ((Room)((Container)ticking).owner()).addItem(CMLib.itemBuilder().enchant(I,enchantPct));
409                                                else
410                                                if(((Container)ticking).owner() instanceof MOB)
411                                                        ((MOB)((Container)ticking).owner()).addItem(CMLib.itemBuilder().enchant(I,enchantPct));
412                                                else
413                                                        return true;
414                                                maintained.addElement(I);
415                                                I.setContainer((Container)ticking);
416                                        }
417                                        else
418                                        if(ticking instanceof MOB)
419                                        {
420                                                ((MOB)ticking).addItem(CMLib.itemBuilder().enchant(I,enchantPct));
421                                                I.wearIfPossible((MOB)ticking);
422                                                maintained.addElement(I);
423                                        }
424                                        else
425                                        {
426                                                Room room=null;
427                                                if(ticking instanceof Room)
428                                                        room=(Room)ticking;
429                                                else
430                                                if(ticking instanceof Area)
431                                                {
432                                                        if(((Area)ticking).metroSize()>0)
433                                                        {
434                                                                Resources.removeResource("HELP_"+ticking.name().toUpperCase());
435                                                                if(restrictedLocales==null)
436                                                                {
437                                                                        int tries=0;
438                                                                        while((room==null)&&((++tries)<100))
439                                                                                room=((Area)ticking).getRandomMetroRoom();
440                                                                }
441                                                                else
442                                                                {
443                                                                        int tries=0;
444                                                                        while(((room==null)||(!okRoomForMe(room)))
445                                                                        &&((++tries)<100))
446                                                                                room=((Area)ticking).getRandomMetroRoom();
447                                                                }
448                                                        }
449                                                        else
450                                                                return true;
451                                                }
452                                                else
453                                                if(ticking instanceof Environmental)
454                                                        room=CMLib.map().roomLocation((Environmental)ticking);
455                                                else
456                                                        return true;
457
458                                                if(room instanceof GridLocale)
459                                                        room=((GridLocale)room).getRandomGridChild();
460                                                if(room!=null)
461                                                {
462                                                        if(CMLib.flags().isGettable(I)&&(!(I instanceof Rideable)))
463                                                        {
464                                                                final Vector<MOB> inhabs=new Vector<MOB>();
465                                                                for(int m=0;m<room.numInhabitants();m++)
466                                                                {
467                                                                        final MOB M=room.fetchInhabitant(m);
468                                                                        if((M.isSavable())&&(M.getStartRoom().getArea().inMyMetroArea(room.getArea())))
469                                                                                inhabs.addElement(M);
470                                                                }
471                                                                if(inhabs.size()>0)
472                                                                {
473                                                                        final MOB M=inhabs.elementAt(CMLib.dice().roll(1,inhabs.size(),-1));
474                                                                        M.addItem(CMLib.itemBuilder().enchant(I,enchantPct));
475                                                                        I.wearIfPossible(M);
476                                                                        maintained.addElement(I);
477                                                                }
478                                                        }
479                                                        if(!favorMobs)
480                                                        {
481                                                                maintained.addElement(I);
482                                                                room.addItem(CMLib.itemBuilder().enchant(I,enchantPct));
483                                                        }
484                                                }
485                                        }
486                                }
487                        }
488                }
489                return true;
490        }
