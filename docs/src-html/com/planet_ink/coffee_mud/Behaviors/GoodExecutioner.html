035public class GoodExecutioner  extends StdBehavior
036{
037        @Override
038        public String ID()
039        {
040                return "GoodExecutioner";
041        }
042
043        @Override
044        public long flags()
045        {
046                return Behavior.FLAG_POTENTIALLYAGGRESSIVE;
047        }
048
049        private boolean                 doPlayers               = false;
050        private boolean                 norecurse               = false;
051        protected long                  deepBreath              = System.currentTimeMillis();
052        private final DVector   protectedOnes   = new DVector(2);
053
054        @Override
055        public void setParms(String newParms)
056        {
057                super.setParms(newParms);
058                newParms=newParms.toUpperCase();
059                final Vector<String> V=CMParms.parse(newParms);
060                doPlayers=V.contains("PLAYERS")||V.contains("PLAYER");
061        }
062
063        @Override
064        public String accountForYourself()
065        {
066                return "aggression to evilness and thieves";
067        }
068
069        @Override
070        public boolean grantsAggressivenessTo(MOB M)
071        {
072                if(norecurse)
073                        return false;
074                norecurse=true;
075                try
076                {
077                        if(M==null)
078                                return false;
079                        if(CMLib.flags().isBoundOrHeld(M))
080                                return false;
081                        if(((!M.isMonster())&&(!doPlayers)))
082                                return false;
083                        final List<Behavior> V=CMLib.flags().flaggedBehaviors(M,Behavior.FLAG_POTENTIALLYAGGRESSIVE);
084                        if((V!=null)
085                        &&((V.size()>1)
086                                ||((V.size()==1)&&(!V.get(0).ID().equals(ID())))))
087                                        return true;
088                        
089                        return ((CMLib.flags().isEvil(M))||(M.baseCharStats().getCurrentClass().baseClass().equalsIgnoreCase("Thief")));
090                }
091                finally
092                {
093                        norecurse = false;
094                }
095        }
096
097        @Override
098        public void executeMsg(final Environmental affecting, final CMMsg msg)
099        {
100                super.executeMsg(affecting,msg);
101                final MOB source=msg.source();
102                if(!canFreelyBehaveNormal(affecting))
103                {
104                        deepBreath=System.currentTimeMillis();
105                        return;
106                }
107                if(msg.sourceMinor()==CMMsg.TYP_LIFE)
108                {
109                        final MOB observer=(MOB)affecting;
110                        if((observer.getVictim() == msg.source())
111                        ||(msg.source().getVictim() == observer))
112                                observer.makePeace(true);
113                        synchronized(protectedOnes)
114                        {
115                                final int x = protectedOnes.indexOf(msg.source().Name());
116                                if(x>=0)
117                                        protectedOnes.setElementAt(x, 2, Long.valueOf(System.currentTimeMillis()));
118                                else
119                                        protectedOnes.addElement(msg.source().Name(),Long.valueOf(System.currentTimeMillis()));
120                        }
121                }
122                if((deepBreath==0)||(System.currentTimeMillis()-deepBreath)>6000)
123                {
124                        synchronized(protectedOnes)
125                        {
126                                for(int p=protectedOnes.size()-1;p>=0;p--)
127                                {
128                                        if((System.currentTimeMillis()-((Long)protectedOnes.elementAt(p, 2)).longValue())>(30 * 1000))
129                                                protectedOnes.removeElementAt(p);
130                                }
131                                if(protectedOnes.contains(msg.source().Name()))
132                                        return;
133                        }
134                        deepBreath=0;
135                        final MOB observer=(MOB)affecting;
136                        // base 90% chance not to be executed
137                        if((source.isMonster()||doPlayers)
138                        &&(source!=observer)
139                        &&(grantsAggressivenessTo(source)))
140                        {
141                                String reason="EVIL";
142                                if(source.baseCharStats().getCurrentClass().baseClass().equalsIgnoreCase("Thief"))
143                                        reason="A THIEF";
144                                final MOB oldFollowing=source.amFollowing();
145                                source.setFollowing(null);
146                                final boolean yep=Aggressive.startFight(observer,source,true,false,source.name().toUpperCase()+" IS "+reason+", AND MUST BE DESTROYED!");
147                                if(!yep)
148                                if(oldFollowing!=null)
149                                        source.setFollowing(oldFollowing);
150                        }
151                }
152        }
153}
