037public class TaxiBehavior extends Concierge
038{
039        @Override
040        public String ID()
041        {
042                return "TaxiBehavior";
043        }
044
045        @Override
046        protected int canImproveCode()
047        {
048                return Behavior.CAN_ITEMS | Behavior.CAN_MOBS;
049        }
050
051        protected volatile Ability      isEnRouter              = null;
052        protected Room                          returnToRoom    = null;
053        protected Room                          destRoom                = null;
054        protected MOB                           riderM                  = null;
055        protected List<Room>            trailTo                 = null;
056        protected List<Rider>           defaultRiders   = null;
057
058        @Override
059        public String accountForYourself()
060        {
061                return "taking you from here to there";
062        }
063
064        @Override
065        protected String getGiveMoneyMessage(final MOB mob, Environmental observer, Environmental destination, String moneyName)
066        {
067                if(observer instanceof MOB)
068                        return L("I can take you to @x1, but you'll need to give me @x2 first.",getDestinationName(mob,destination),moneyName);
069                else
070                if(observer instanceof Container)
071                        return L("I can take you to @x1, but you'll need to put @x2 into @x3 first.",getDestinationName(mob,destination),moneyName,observer.name());
072                else
073                        return L("I can take you to @x1, but you'll need to drop @x2 first.",getDestinationName(mob,destination),moneyName);
074        }
075        
076        @Override
077        protected void giveMerchandise(MOB whoM, Room destR, Environmental observer, Room room, TrackingFlags trackingFlags)
078        {
079                MOB fromM=getTalker(observer,room);
080                final ArrayList<Room> set=new ArrayList<Room>();
081                CMLib.tracking().getRadiantRooms(fromM.location(),set,roomRadiusFlags,null,maxRange,null);
082                trailTo=CMLib.tracking().findTrailToRoom(fromM.location(), destR, trackingFlags, maxRange);
083                thingsToSay.addElement(whoM,L("OK, we're now on our way to @x1.",getDestinationName(whoM,destR)));
084                this.returnToRoom=fromM.location();
085                this.isEnRouter=CMClass.getAbility("Prop_Adjuster");
086                this.isEnRouter.setMiscText("sen+"+PhyStats.CAN_NOT_TRACK);
087                this.destRoom = destR;
088                this.riderM = whoM;
089                if(observer instanceof Affectable)
090                        ((Affectable)observer).addNonUninvokableEffect(this.isEnRouter);
091        }
092        
093        @Override
094        protected boolean disableComingsAndGoings()
095        {
096                return (isEnRouter!=null);
097        }
098
099        @Override
100        protected final MOB getTalker(Environmental o, Room room)
101        {
102                if(o instanceof Rideable)
103                {
104                        if(defaultRiders == null)
105                                defaultRiders = new XVector<Rider>(((Rideable)o).riders());
106                }
107                return super.getTalker(o, room);
108        }
109        
110        private void endTheRide(Environmental observer)
111        {
112                if(this.isEnRouter != null)
113                {
114                        final Room room=CMLib.map().roomLocation(observer);
115                        MOB conciergeM=this.getTalker(observer,room);
116                        if(room==this.destRoom)
117                                CMLib.commands().postSay(conciergeM,null,L("We're here.  Best of luck!."),false,false);
118                        else
119                                CMLib.commands().postSay(conciergeM,null,L("This is as far as I can go.  Best of luck!."),false,false);
120                        Rideable rideable = null;
121                        if(observer instanceof Rideable)
122                                rideable = (Rideable)observer;
123                        else
124                        if((observer instanceof Rider)&&(((Rider)observer).riding()!=null))
125                                rideable=((Rider)observer).riding();
126                        if((rideable!=null)&&(room!=null))
127                        {
128                                MOB mob=this.getTalker(observer, room);
129                                for(final Enumeration<Rider> r = rideable.riders(); r.hasMoreElements(); )
130                                {
131                                        final Rider rider=r.nextElement();
132                                        if(!defaultRiders.contains(rider))
133                                        {
134                                                if(rider instanceof MOB)
135                                                        room.show((MOB)rider, rideable, mob, CMMsg.MASK_ALWAYS|CMMsg.MSG_DISMOUNT, L("<S-NAME> @x1 from <T-NAME>.",rideable.dismountString(rider)));
136                                                else
137                                                        room.show(mob, rideable, rider, CMMsg.MASK_ALWAYS|CMMsg.MSG_DISMOUNT, L("<S-NAME> help(s) <O-NAME> off of <T-NAME>."));
138                                                rider.setRiding(null);
139                                        }
140                                }
141                        }
142                        if(returnToRoom != null)
143                        {
144                                if(observer instanceof MOB)
145                                        CMLib.tracking().wanderFromTo((MOB)observer, returnToRoom, false );
146                                else
147                                if((observer instanceof Item)&&(room != null))
148                                {
149                                        room.showHappens(CMMsg.MSG_OK_ACTION, L("@x1 heads off.",observer.name()));
150                                        returnToRoom.moveItemTo((Item)observer);
151                                }
152                        }
153                        if(isEnRouter != null)
154                        {
155                                if(isEnRouter.affecting() != null)
156                                        isEnRouter.affecting().delEffect(isEnRouter);
157                        }
158                        isEnRouter = null;
159                        returnToRoom = null;
160                        destRoom = null;
161                        trailTo= null;
162                        riderM = null;
163                }
164        }
165        
166        @Override
167        public boolean tick(final Tickable ticking, final int tickID)
168        {
169                if(!super.tick(ticking, tickID))
170                        return false;
171                if((ticking instanceof Environmental) && (isEnRouter != null))
172                {
173                        final Environmental observer=(Environmental)ticking;
174                        if((ticking instanceof MOB) && (!super.canFreelyBehaveNormal(ticking)))
175                                endTheRide(observer);
176                        else
177                        {
178                                final Room locR=CMLib.map().roomLocation(observer);
179                                if(locR==destRoom)
180                                        endTheRide(observer);
181                                else
182                                if(locR!=null)
183                                {
184                                        final int nextDirection=CMLib.tracking().trackNextDirectionFromHere(trailTo, locR, true);
185                                        final Room nextR=locR.getRoomInDir(nextDirection);
186                                        final Exit nextE=locR.getExitInDir(nextDirection);
187                                        if((nextR != null) && (nextE != null) && (nextE.isOpen()))
188                                        {
189                                                if((trackingFlags.contains(TrackingFlag.OUTDOORONLY))&&((nextR.domainType()&Room.INDOORS)!=0))
190                                                        endTheRide(observer);
191                                                else
192                                                if(observer instanceof MOB)
193                                                {
194                                                        if(!CMLib.tracking().walk((MOB)observer, nextDirection,false,false))
195                                                                endTheRide(observer);
196                                                }
197                                                else
198                                                if(observer instanceof Item)
199                                                {
200                                                        if(!CMLib.tracking().walk((Item)observer, nextDirection))
201                                                                endTheRide(observer);
202                                                }
203                                        }
204                                        else
205                                                endTheRide(observer);
206                                }
207                                else
208                                        endTheRide(observer);
209                        }
210                }
211                return true;
212        }
213        
214        @Override
215        protected void resetDefaults()
216        {
217                super.resetDefaults();
218                greeting="Need a lift? If so, come aboard.";
219                mountStr=L("Where are you headed?");
220                isEnRouter = null;
221                returnToRoom = null;
222                destRoom = null;
223                trailTo= null;
224                riderM = null;
225                basePrice=10.0;
226                perRoomPrice=1.0;
227        }
228        
229        @Override
230        public void startBehavior(PhysicalAgent behaving)
231        {
232                super.startBehavior(behaving);
233                if((talkerName.length()==0) && (behaving instanceof Item))
234                        talkerName="the driver";
235        }
236        
237        @Override
238        protected void resetFlags()
239        {
240                trackingFlags = CMLib.tracking().newFlags().plus(TrackingLibrary.TrackingFlag.NOEMPTYGRIDS);
241                trackingFlags.plus(TrackingFlag.OPENONLY);
242                trackingFlags.plus(TrackingFlag.OUTDOORONLY);
243                roomRadiusFlags = CMLib.tracking().newFlags();
244                roomRadiusFlags.plus(TrackingFlag.OPENONLY);
245                trackingFlags.plus(TrackingFlag.OUTDOORONLY);
246        }
247        
248        @Override
249        public void setParms(String newParm)
250        {
251                super.setParms(newParm);
252                if(roomRadiusFlags.contains(TrackingFlag.AREAONLY))
253                        trackingFlags.add(TrackingFlag.AREAONLY);
254        }
