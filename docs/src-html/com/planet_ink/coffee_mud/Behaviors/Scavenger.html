035public class Scavenger extends ActiveTicker
036{
037        @Override
038        public String ID()
039        {
040                return "Scavenger";
041        }
042
043        @Override
044        protected int canImproveCode()
045        {
046                return Behavior.CAN_MOBS;
047        }
048        
049        protected int origItems=-1;
050        protected MaskingLibrary.CompiledZMask mask = null;
051        protected String trashRoomID="";
052
053        public Scavenger()
054        {
055                super();
056                minTicks=1; maxTicks=10; chance=99;
057                origItems=-1;
058                tickReset();
059        }
060
061        @Override
062        public String accountForYourself()
063        {
064                return "scavenging";
065        }
066
067        @Override
068        public void setParms(String newParms)
069        {
070                super.setParms(newParms);
071                String argParms="";
072                String maskStr="";
073                int x=newParms.indexOf(';');
074                if(x>=0)
075                {
076                        argParms=newParms.substring(0,x);
077                        maskStr=newParms.substring(x+1);
078                }
079                trashRoomID=CMParms.getParmStr(argParms,"TRASH","");
080                mask=(maskStr.length()==0)?null:CMLib.masking().getPreCompiledMask(maskStr);
081        }
082        
083        @Override
084        public boolean tick(final Tickable ticking, final int tickID)
085        {
086                if(!super.tick(ticking,tickID))
087                        return false;
088
089                if((canAct(ticking,tickID))&&(ticking instanceof MOB))
090                {
091                        final MOB mob=(MOB)ticking;
092                        final Room thisRoom=mob.location();
093                        if(thisRoom == null)
094                                return true;
095                                
096                        if(origItems<0)
097                                origItems=mob.numItems();
098                        if((mob.phyStats().weight()>=(int)Math.round(CMath.mul(mob.maxCarry(),0.9)))
099                        ||(mob.numItems()>=mob.maxItems()))
100                        {
101                                if(CMLib.flags().isATrackingMonster(mob))
102                                        return true;
103                                if(trashRoomID.equalsIgnoreCase("NO"))
104                                        return true;
105                                final Room R=(trashRoomID.length()>0)?CMLib.map().getRoom(trashRoomID):null;
106                                if((mob.location()==R)&&(R!=null))
107                                {
108                                        Container C=null;
109                                        int maxCapacity=0;
110                                        for(int i=0;i<R.numItems();i++)
111                                        {
112                                                final Item I=R.getItem(i);
113                                                if((I instanceof Container)
114                                                &&(I.container()==null)
115                                                &&(!CMLib.flags().isGettable(I)))
116                                                {
117                                                        if(((Container)I).capacity()>maxCapacity)
118                                                        {
119                                                                C=(Container)I;
120                                                                maxCapacity=((Container)I).capacity();
121                                                        }
122                                                }
123                                        }
124                                        if(C!=null)
125                                                mob.doCommand(new XVector<String>("PUT","ALL",C.Name()),MUDCmdProcessor.METAFLAG_FORCED);
126                                        else
127                                                mob.doCommand(new XVector<String>("DROP","ALL"),MUDCmdProcessor.METAFLAG_FORCED);
128                                        CMLib.tracking().wanderAway(mob,false,true);
129                                }
130                                else
131                                if(R!=null)
132                                {
133                                        final Ability A=CMLib.flags().isTracking(mob) ? null : CMClass.getAbility("Skill_Track");
134                                        if(A!=null)
135                                                A.invoke(mob,CMParms.parse("\""+CMLib.map().getExtendedRoomID(R)+"\""),R,true,0);
136                                }
137                                else
138                                if((origItems>=0)&&(mob.numItems()>origItems))
139                                {
140                                        while((origItems>=0)&&(mob.numItems()>origItems))
141                                        {
142                                                final Item I=mob.getItem(origItems);
143                                                if(I==null)
144                                                {
145                                                        if(origItems>0)
146                                                                origItems--;
147                                                        break;
148                                                }
149                                                if(I.owner()==null)
150                                                        I.setOwner(mob);
151                                                I.destroy();
152                                        }
153                                        mob.recoverPhyStats();
154                                        mob.recoverCharStats();
155                                        mob.recoverMaxState();
156                                }
157                        }
158                        if((thisRoom.numItems()==0)||(thisRoom.numPCInhabitants()>0))
159                                return true;
160                        List<Item> choices=new ArrayList<Item>(thisRoom.numItems()<1000?thisRoom.numItems():1000);
161                        for(int i=0;(i<thisRoom.numItems())&&(choices.size()<1000);i++)
162                        {
163                                final Item thisItem=thisRoom.getItem(i);
164                                if((thisItem!=null)
165                                &&(thisItem.container()==null)
166                                &&(CMLib.flags().isGettable(thisItem))
167                                &&(CMLib.flags().canBeSeenBy(thisItem, mob))
168                                &&(!(thisItem instanceof DeadBody))
169                                &&(mask==null)||(CMLib.masking().maskCheck(mask, thisItem, false)))
170                                        choices.add(thisItem);
171                        }
172                        if(choices.size()==0)
173                                return true;
174                        final Item I=choices.get(CMLib.dice().roll(1,choices.size(),-1));
175                        if(I!=null)
176                                mob.doCommand(new XVector<String>("GET","$"+I.Name()+"$"),MUDCmdProcessor.METAFLAG_FORCED);
177                        choices.clear();
178                        choices=null;
179                }
180                return true;
181        }
182}
