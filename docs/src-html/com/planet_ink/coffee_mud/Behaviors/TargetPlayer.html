039public class TargetPlayer extends ActiveTicker
040{
041        @Override
042        public String ID()
043        {
044                return "TargetPlayer";
045        }
046
047        @Override
048        protected int canImproveCode()
049        {
050                return Behavior.CAN_MOBS;
051        }
052
053        public TargetPlayer()
054        {
055                super();
056                minTicks=3; maxTicks=12; chance=100;
057                tickReset();
058        }
059
060        @Override
061        public String accountForYourself()
062        {
063                return "hero targeting";
064        }
065
066        @Override
067        public boolean tick(final Tickable ticking, final int tickID)
068        {
069                if(canAct(ticking,tickID))
070                {
071                        final MOB mob = (MOB) ticking;
072                        if (mob.getVictim() != null)
073                        {
074                                final Set<MOB> theBadGuys = mob.getVictim().getGroupMembers(new HashSet<MOB>());
075                                MOB shouldFight = null;
076                                for (final Object element : theBadGuys)
077                                {
078                                        final MOB consider = (MOB) element;
079                                        if (consider.isMonster())
080                                                continue;
081                                        if (shouldFight == null)
082                                        {
083                                                shouldFight = consider;
084                                        }
085                                        else
086                                        {
087                                                if (((shouldFight.phyStats()!=null)&&(consider.phyStats()!=null))
088                                                &&(shouldFight.phyStats().level() > consider.phyStats().level()))
089                                                        shouldFight = consider;
090                                        }
091                                }
092                                if(shouldFight!=null)
093                                {
094                                        if(shouldFight.equals(mob.getVictim()))
095                                                return true;
096                                        else
097                                        if(CMLib.flags().canBeSeenBy(shouldFight,mob))
098                                        {
099                                                mob.setVictim(shouldFight);
100                                        }
101                                }
102                        }
103                        return true;
104                }
105                return true;
106        }
107}
