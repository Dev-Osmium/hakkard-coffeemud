035public class Follower extends ActiveTicker
036{
037        @Override
038        public String ID()
039        {
040                return "Follower";
041        }

043        @Override
044        protected int canImproveCode()
045        {
046                return Behavior.CAN_ITEMS | Behavior.CAN_MOBS;
047        }
048
049        protected boolean       realFollow              = false;
050        protected boolean       noFollowers             = false;
051        protected boolean       inventory               = false;
052        protected boolean       wander                  = false;
053        protected int           lastNumPeople   = -1;
054        protected Room          lastRoom                = null;
055        protected MOB           lastOwner               = null;
056        protected String        name                    = null;
057
058        int direction=-1;
059
060        public Follower()
061        {
062                super();
063                minTicks=0;
064                maxTicks=0;
065                direction=-1;
066        }

068        @Override
069        public void setParms(String newParms)
070        {
071                minTicks=0;
072                maxTicks=0;
073                chance=100;
074                super.setParms(newParms);
075                final Vector<String> V=CMParms.parse(newParms.toUpperCase());
076                realFollow=V.contains("GROUP");
077                wander=V.contains("WANDER");
078                noFollowers=V.contains("NOFOLLOWERS");
079                inventory=V.contains("INVENTORY")||V.contains("INV");
080                name=CMParms.getParmStr(newParms, "NAME", "");
081        }

086                return "natural friendly following";

089        private class QuickFollowClass implements Runnable
090        {
091                private boolean                         scheduled       = false;
092                private final MOB                       mob;
093                private final Environmental     affecting;
094
095                public QuickFollowClass(final MOB mob, final Environmental affecting)
096                {
097                        this.mob=mob;
098                        this.affecting=affecting;
099                }
100                
101                @Override
102                public void run()
103                {
104                        if(!scheduled)
105                        {
106                                CMLib.threads().scheduleRunnable(this, 100);
107                                scheduled=true;
108                        }
109                        else
110                        {
111                                final Room R=CMLib.map().roomLocation(affecting);
112                                final Room mobR = mob.location();
113                                if(mobR != R)
114                                {
115                                        direction = CMLib.map().getRoomDir(R, mobR);
116                                        if((direction<0)
117                                        &&(mobR!=null))
118                                        {
119                                                if(affecting instanceof MOB)
120                                                        mobR.bringMobHere((MOB)affecting, true);
121                                                else
122                                                if(affecting instanceof Item)
123                                                        mobR.moveItemTo((Item)affecting);
124                                                
125                                        }
126                                }
127                                if(affecting instanceof MOB)
128                                        followingMOB((MOB)affecting);
129                                else
130                                if(affecting instanceof Item)
131                                        followingItem((Item)affecting);
132                        }
133                }
134                
135        }
136        
137        @Override
138        public void executeMsg(final Environmental affecting, CMMsg msg)
139        {
140                super.executeMsg(affecting,msg);
141
142                final MOB mob=msg.source();
143                if(mob.amDead())
144                        return;
145                if(mob.location()==null)
146                        return;
147
148                if(affecting instanceof MOB)
149                {
150                        if((!canFreelyBehaveNormal(affecting))||(realFollow))
151                                return;
152
153                        if(maxTicks >= 0)
154                        {
155                                if((direction<0)
156                                &&(msg.amITarget(((MOB)affecting).location()))
157                                &&(CMLib.flags().canBeSeenBy(mob,(MOB)affecting))
158                                &&(msg.othersMessage()!=null)
159                                &&((msg.targetMinor()==CMMsg.TYP_LEAVE)
160                                 ||(msg.targetMinor()==CMMsg.TYP_FLEE))
161                                &&((CMLib.masking().maskCheck(getParms(),mob,false))
162                                        ||((name!=null)&&(name.length()>0)&&(mob.Name().equalsIgnoreCase(name))))
163                                &&(CMLib.dice().rollPercentage()<=chance))
164                                {
165                                        String directionWent=msg.othersMessage();
166                                        final int x=directionWent.lastIndexOf(' ');
167                                        if(x>=0)
168                                        {
169                                                directionWent=directionWent.substring(x+1);
170                                                direction=CMLib.directions().getDirectionCode(directionWent);
171                                        }
172                                        else
173                                                direction=-1;
174                                }
175                        }
176                }
177                
178                // handle instant moves
179                if((msg.target() instanceof Room)
180                &&(maxTicks<0)
181                &&((msg.targetMinor()==CMMsg.TYP_LEAVE)
182                        ||(msg.targetMinor()==CMMsg.TYP_FLEE)
183                        ||(msg.targetMinor()==CMMsg.TYP_RECALL)
184                        ||(msg.targetMinor()==CMMsg.TYP_ENTER))
185                &&((CMLib.masking().maskCheck(getParms(),mob,false))
186                        ||((name!=null)&&(name.length()>0)&&(mob.Name().equalsIgnoreCase(name))))
187                &&((!(affecting instanceof MOB))||CMLib.flags().canBeSeenBy(mob,(MOB)affecting))
188                &&(CMLib.dice().rollPercentage()<=chance))
189                {
190                        msg.addTrailerRunnable(new QuickFollowClass(mob,affecting));
191                }
192        }

194        public MOB pickRandomMOBHere(Environmental ticking, Room room)
195        {
196                if(room==null)
197                        return null;
198                if((room.numInhabitants()!=lastNumPeople)
199                ||(room!=lastRoom))
200                {
201                        lastNumPeople=room.numInhabitants();
202                        lastRoom=room;
203                        for(int i=0;i<room.numInhabitants();i++)
204                        {
205                                final MOB M=room.fetchInhabitant(i);
206                                if((M!=null)
207                                &&(M!=ticking)
208                                &&((name == null)||(name.length()==0)||(name.equalsIgnoreCase(M.Name())))
209                                &&(!CMSecurity.isAllowed(M,room,CMSecurity.SecFlag.CMDMOBS))
210                                &&(!CMSecurity.isAllowed(M,room,CMSecurity.SecFlag.CMDROOMS))
211                                &&(CMLib.masking().maskCheck(getParms(),M,false)))
212                                        return M;
213                        }
214                }
215                return null;
216        }

218        @Override
219        public boolean okMessage(final Environmental host, final CMMsg msg)
220        {
221                if(!super.okMessage(host,msg))
222                        return false;
223                if((host instanceof Item)
224                &&(msg.tool()==host)
225                &&(msg.sourceMinor()==CMMsg.TYP_SELL))
226                {
227                        msg.source().tell(L("You can not sell @x1.",host.name()));
228                        return false;
229                }
230                return true;
231        }

233        public void followingMOB(MOB mob)
234        {
235                if(!canFreelyBehaveNormal(mob))
236                        return;
237                final Room room=mob.location();
238                if((noFollowers)&&(mob.numFollowers()>0))
239                        return;
240                if(realFollow)
241                {
242                        if(mob.amFollowing()==null)
243                        {
244                                final MOB M=pickRandomMOBHere(mob,room);
245                                if(M!=null)
246                                        CMLib.commands().postFollow(mob,M,false);
247                        }
248                }
249                else
250                if(direction>=0)
251                {
252                        final Room otherRoom=room.getRoomInDir(direction);
253
254                        if(otherRoom!=null)
255                        {
256                                if((!wander)&&(!otherRoom.getArea().Name().equals(room.getArea().Name())))
257                                        direction=-1;
258                        }
259                        else
260                                direction=-1;
261
262                        if(direction<0)
263                                return;
264
265                        boolean move=true;
266                        for(int m=0;m<room.numInhabitants();m++)
267                        {
268                                final MOB inhab=room.fetchInhabitant(m);
269                                if((inhab!=null)
270                                &&(CMSecurity.isAllowed(inhab,room,CMSecurity.SecFlag.CMDMOBS)
271                                   ||CMSecurity.isAllowed(inhab,room,CMSecurity.SecFlag.CMDROOMS)))
272                                        move=false;
273                        }
274                        if(move)
275                                CMLib.tracking().walk(mob,direction,false,false);
276                        direction=-1;
277                }
278        }
279        
280        public void followingItem(Item I)
281        {
282                if(I.container()!=null)
283                        I.setContainer(null);
284
285                final Room R=CMLib.map().roomLocation(I);
286                if(R==null)
287                        return;
288
289                if(R!=lastOwner.location())
290                        lastOwner.location().moveItemTo(I,ItemPossessor.Expire.Never,ItemPossessor.Move.Followers);
291                if((inventory)&&(R.isInhabitant(lastOwner)))
292                {
293                        CMLib.commands().postGet(lastOwner,null,I,true);
294                        if(!lastOwner.isMine(I))
295                        {
296                                lastOwner.moveItemTo(I);
297                                if(lastOwner.location()!=null)
298                                        lastOwner.location().recoverRoomStats();
299                        }
300                }
301        }

303        @Override
304        public boolean tick(final Tickable ticking, final int tickID)
305        {
306                super.tick(ticking,tickID);
307
308                if((ticking instanceof Item)
309                &&((lastOwner==null)
310                   ||((!inventory)&&(!CMLib.flags().isInTheGame(lastOwner,false)))))
311                {
312                        final Item I=(Item)ticking;
313                        if((I.owner()!=null)
314                        &&(I.owner() instanceof MOB)
315                        &&(CMLib.masking().maskCheck(getParms(),I.owner(),false))
316                        &&(!CMSecurity.isAllowed((MOB)I.owner(),((MOB)I.owner()).location(),CMSecurity.SecFlag.CMDMOBS))
317                        &&(!CMSecurity.isAllowed((MOB)I.owner(),((MOB)I.owner()).location(),CMSecurity.SecFlag.CMDROOMS)))
318                                lastOwner=(MOB)I.owner();
319                        else
320                        if(!inventory)
321                        {
322                                final MOB M=pickRandomMOBHere(I,CMLib.map().roomLocation(I));
323                                if(M!=null)
324                                        lastOwner=M;
325                        }
326                }
327
328                if((!canAct(ticking,tickID))||(maxTicks<0))
329                        return true;
330
331                if(ticking instanceof MOB)
332                {
333                        if(tickID!=Tickable.TICKID_MOB)
334                                return true;
335                        followingMOB((MOB)ticking);
336                }
337                else
338                if((ticking instanceof Item)
339                &&(lastOwner!=null)
340                &&(lastOwner.location()!=null))
341                {
342                        followingItem((Item)ticking);
343                }
344                return true;
345        }
