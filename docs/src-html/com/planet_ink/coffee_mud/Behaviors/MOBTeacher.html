070        protected void setTheCharClass(MOB mob, CharClass C)
071        {
072                if((mob.baseCharStats().numClasses()==1)
073                &&(mob.baseCharStats().getMyClass(0).ID().equals("StdCharClass"))
074                &&(!C.ID().equals("StdCharClass")))
075                {
076                        mob.baseCharStats().setMyClasses(C.ID());
077                        mob.baseCharStats().setMyLevels(""+mob.phyStats().level());
078                        mob.recoverCharStats();
079                        return;
080                }
081                for(int i=0;i<mob.baseCharStats().numClasses();i++)
082                {
083                        final CharClass C1=mob.baseCharStats().getMyClass(i);
084                        if((C1!=null)
085                        &&(mob.baseCharStats().getClassLevel(C1)>0))
086                                mob.baseCharStats().setClassLevel(C1,1);
087                }
088                mob.baseCharStats().setCurrentClass(C);
089                mob.baseCharStats().setClassLevel(C,mob.phyStats().level());
090                mob.recoverCharStats();
091        }

093        protected void classAbles(MOB mob, Map<String,Ability> myAbles, int pct)
094        {
095                final boolean stdCharClass=mob.charStats().getCurrentClass().ID().equals("StdCharClass");
096                final String className=mob.charStats().getCurrentClass().ID();
097                Ability A=null;
098                for(final Enumeration<Ability> a=CMClass.abilities();a.hasMoreElements();)
099                {
100                        A=a.nextElement();
101                        if((A!=null)
102                        &&((stdCharClass&&(CMLib.ableMapper().lowestQualifyingLevel(A.ID())>0)&&(!(A instanceof ArchonOnly)))
103                                ||(CMLib.ableMapper().qualifiesByLevel(mob,A)&&(!CMLib.ableMapper().getSecretSkill(className,true,A.ID()))))
104                        &&((!noCommon)||((A.classificationCode()&Ability.ALL_ACODES)!=Ability.ACODE_COMMON_SKILL))
105                        &&((!stdCharClass)||(CMLib.ableMapper().availableToTheme(A.ID(),Area.THEME_FANTASY,true))))
106                                addAbility(mob,A,pct,myAbles);
107                }
108                for(final ClanGovernment G : CMLib.clans().getStockGovernments())
109                {
110                        G.getClanLevelAbilities(null,null,Integer.valueOf(Integer.MAX_VALUE));
111                        for(final Enumeration<AbilityMapping> m= CMLib.ableMapper().getClassAbles(G.getName(), false);m.hasMoreElements();)
112                        {
113                                final AbilityMapping M=m.nextElement();
114                                final Ability A2=CMClass.getAbility(M.abilityID());
115                                addAbility(mob,A2,pct,myAbles);
116                        }
117                }
118        }

120        @Override
121        public boolean tick(final Tickable ticking, final int tickID)
122        {
123                if((tickID==Tickable.TICKID_MOB)
124                &&(!CMSecurity.isDisabled(CMSecurity.DisFlag.MOBTEACHER))
125                &&((--tickDownToKnowledge)==0)
126                &&(ticking instanceof MOB))
127                {
128                        if(!noExpertises)
129                        {
130                                noExpertises=true;
131                                final MOB mob=(MOB)ticking;
132                                if(teachEverything)
133                                {
134                                        for(final Enumeration<ExpertiseLibrary.ExpertiseDefinition> e=CMLib.expertises().definitions();e.hasMoreElements();)
135                                                mob.addExpertise(e.nextElement().ID());
136                                        trainableExpertises=null;
137                                }
138                                else
139                                {
140                                        boolean someNew=true;
141                                        final CharStats oldBase=(CharStats)mob.baseCharStats().copyOf();
142                                        for(final int i: CharStats.CODES.BASECODES())
143                                                mob.baseCharStats().setStat(i,100);
144                                        for(int i=0;i<mob.baseCharStats().numClasses();i++)
145                                                mob.baseCharStats().setClassLevel(mob.baseCharStats().getMyClass(i),100);
146                                        mob.recoverCharStats();
147                                        while(someNew)
148                                        {
149                                                someNew=false;
150                                                final List<ExpertiseDefinition> V=CMLib.expertises().myQualifiedExpertises(mob);
151                                                ExpertiseLibrary.ExpertiseDefinition def=null;
152                                                for(int v=0;v<V.size();v++)
153                                                {
154                                                        def=V.get(v);
155                                                        if(mob.fetchExpertise(def.ID())==null)
156                                                        {
157                                                                mob.addExpertise(def.ID());
158                                                                someNew=true;
159                                                        }
160                                                }
161                                                if(someNew)
162                                                        trainableExpertises=null;
163                                        }
164                                        mob.setBaseCharStats(oldBase);
165                                        mob.recoverCharStats();
166                                }
167                        }
168
169                }
170                return super.tick(ticking,tickID);
171        }

173        public void addAbility(MOB mob, Ability A, int pct, Map<String,Ability> myAbles)
174        {
175                if(CMLib.dice().rollPercentage()<=pct)
176                {
177                        final Ability A2=myAbles.get(A.ID());
178                        if(A2==null)
179                        {
180                                A=(Ability)A.copyOf();
181                                A.setSavable(false);
182                                A.setProficiency(CMLib.ableMapper().getMaxProficiency(mob,true,A.ID()));
183                                myAbles.put(A.ID(),A);
184                                mob.addAbility(A);
185                        }
186                        else
187                                A2.setProficiency(CMLib.ableMapper().getMaxProficiency(mob,true,A2.ID()));
188                }
189        }

191        protected void ensureCharClass()
192        {
193                myMOB.baseCharStats().setMyClasses("StdCharClass");
194                myMOB.baseCharStats().setMyLevels(""+myMOB.phyStats().level());
195                myMOB.recoverCharStats();
196
197                final Map<String,Ability> myAbles=new HashMap<String,Ability>();
198                Ability A=null;
199                for(final Enumeration<Ability> a=myMOB.allAbilities();a.hasMoreElements();)
200                {
201                        A=a.nextElement();
202                        if(A!=null)
203                                myAbles.put(A.ID(),A);
204                }
205                myMOB.baseCharStats().setStat(CharStats.STAT_INTELLIGENCE,19);
206                myMOB.baseCharStats().setStat(CharStats.STAT_WISDOM,19);
207
208                int pct=100;
209                List<String> V=null;
210                A=CMClass.getAbility(getParms());
211                if(A!=null)
212                {
213                        addAbility(myMOB,A,pct,myAbles);
214                        teachEverything=false;
215                }
216                else
217                        V=CMParms.parse(getParms());
218
219                if(V!=null)
220                {
221                        for(int v=V.size()-1;v>=0;v--)
222                        {
223                                final String s=V.get(v);
224                                if(s.equalsIgnoreCase("NOCOMMON"))
225                                {
226                                        noCommon=true;
227                                        V.remove(v);
228                                }
229                                if(s.equalsIgnoreCase("NOEXPS")||s.equalsIgnoreCase("NOEXP"))
230                                {
231                                        noExpertises=true;
232                                        V.remove(v);
233                                }
234                                if(s.equalsIgnoreCase("NOHLEXPS")||s.equalsIgnoreCase("NOHLEXP"))
235                                {
236                                        noHLExpertises=true;
237                                        V.remove(v);
238                                }
239                        }
240                }
241
242                if(V!=null)
243                {
244                        for(int v=0;v<V.size();v++)
245                        {
246                                final String s=V.get(v);
247                                if(s.endsWith("%"))
248                                {
249                                        pct=CMath.s_int(s.substring(0,s.length()-1));
250                                        continue;
251                                }
252        
253                                A=CMClass.getAbility(s);
254                                final CharClass C=CMClass.findCharClass(s);
255                                if((C!=null)&&(C.availabilityCode()!=0))
256                                {
257                                        teachEverything=false;
258                                        setTheCharClass(myMOB,C);
259                                        classAbles(myMOB,myAbles,pct);
260                                        myMOB.recoverCharStats();
261                                }
262                                else
263                                if(A!=null)
264                                {
265                                        addAbility(myMOB,A,pct,myAbles);
266                                        teachEverything=false;
267                                }
268                                else
269                                {
270                                        final ExpertiseLibrary.ExpertiseDefinition def=CMLib.expertises().getDefinition(s);
271                                        if(def!=null)
272                                        {
273                                                myMOB.addExpertise(def.ID());
274                                                teachEverything=false;
275                                        }
276                                }
277                        }
278                }
279                myMOB.recoverCharStats();
280                if((myMOB.charStats().getCurrentClass().ID().equals("StdCharClass"))
281                &&(teachEverything))
282                        classAbles(myMOB,myAbles,pct);
283                int lvl=myMOB.phyStats().level()/myMOB.baseCharStats().numClasses();
284                if(lvl<1)
285                        lvl=1;
286                for(int i=0;i<myMOB.baseCharStats().numClasses();i++)
287                {
288                        final CharClass C=myMOB.baseCharStats().getMyClass(i);
289                        if((C!=null)&&(myMOB.baseCharStats().getClassLevel(C)>=0))
290                                myMOB.baseCharStats().setClassLevel(C,lvl);
291                }
292                myMOB.recoverCharStats();
293        }

295        @Override
296        public void setParms(String newParms)
297        {
298                super.setParms(newParms);
299                if(myMOB==null)
300                        return;
301                teachEverything=true;
302                noCommon=false;
303                noExpertises=false;
304                tickDownToKnowledge=4;
305                trainableExpertises=null;
306                ensureCharClass();
307        }

309        @Override
310        public boolean okMessage(final Environmental host, final CMMsg msg)
311        {
312                if(host instanceof MOB)
313                {
314                        if(((MOB)host).isAttributeSet(MOB.Attrib.NOTEACH))
315                                ((MOB)host).setAttribute(MOB.Attrib.NOTEACH,false);
316                }
317                return super.okMessage(host,msg);
318        }

320        @Override
321        public void executeMsg(final Environmental affecting, final CMMsg msg)
322        {
323                if(myMOB==null)
324                        return;
325                super.executeMsg(affecting,msg);
326                if(!canFreelyBehaveNormal(affecting))
327                        return;
328                final MOB monster=myMOB;
329                final MOB student=msg.source();
330
331                if((!msg.amISource(monster))
332                &&(!student.isMonster())
333                &&(msg.sourceMessage()!=null)
334                &&((msg.target()==null)||msg.amITarget(monster))
335                &&(msg.targetMinor()==CMMsg.TYP_SPEAK)
336                &&(!CMSecurity.isDisabled(CMSecurity.DisFlag.MOBTEACHER)))
337                {
338                        String sayMsg=CMStrings.getSayFromMessage(msg.sourceMessage());
339                        if(sayMsg==null)
340                        {
341                                final int start=msg.sourceMessage().indexOf('\'');
342                                if(start>0)
343                                {
344                                        sayMsg=msg.sourceMessage().substring(start+1);
345                                        int x=sayMsg.lastIndexOf("\'");
346                                        if(x>0)
347                                                sayMsg=sayMsg.substring(0,x);
348                                }
349                                else
350                                        sayMsg=msg.sourceMessage();
351                        }
352                        int x=sayMsg.toUpperCase().indexOf("TEACH");
353                        if(x<0)
354                                x=sayMsg.toUpperCase().indexOf("GAIN ");
355                        if(x>=0)
356                        {
357                                boolean giveABonus=false;
358                                String s=sayMsg.substring(x+5).trim();
359                                if(s.startsWith("\""))
360                                        s=s.substring(1).trim();
361                                if(s.endsWith("\""))
362                                        s=s.substring(0,s.length()-1);
363                                if(s.toUpperCase().endsWith("PLEASE"))
364                                        s=s.substring(0,s.length()-6).trim();
365                                if(s.startsWith("\""))
366                                        s=s.substring(1).trim();
367                                if(s.endsWith("\""))
368                                        s=s.substring(0,s.length()-1);
369                                if(s.toUpperCase().startsWith("PLEASE "))
370                                {
371                                        giveABonus=true;
372                                        s=s.substring(6).trim();
373                                }
374                                if(s.startsWith("\""))
375                                        s=s.substring(1).trim();
376                                if(s.endsWith("\""))
377                                        s=s.substring(0,s.length()-1);
378                                if(s.toUpperCase().startsWith("ME "))
379                                        s=s.substring(3).trim();
380                                if(s.startsWith("\""))
381                                        s=s.substring(1).trim();
382                                if(s.endsWith("\""))
383                                        s=s.substring(0,s.length()-1);
384                                if(s.toUpperCase().startsWith("PLEASE "))
385                                {
386                                        giveABonus=true;
387                                        s=s.substring(6).trim();
388                                }
389                                if(s.toUpperCase().startsWith("ME "))
390                                        s=s.substring(3).trim();
391                                if(s.startsWith("\""))
392                                        s=s.substring(1).trim();
393                                if(s.endsWith("\""))
394                                        s=s.substring(0,s.length()-1);
395                                if(s.trim().equalsIgnoreCase("LIST"))
396                                {
397                                        CMLib.commands().postSay(monster,student,L("Try the QUALIFY command."),true,false);
398                                        return;
399                                }
400                                if(s.trim().toUpperCase().equals("ALL"))
401                                {
402                                        CMLib.commands().postSay(monster,student,L("I can't teach you everything at once. Try the QUALIFY command."),true,false);
403                                        return;
404                                }
405                                if(!CMLib.flags().canBeSeenBy(student,monster))
406                                {
407                                        CMLib.commands().postSay(monster,student,L("I can't see you, so I can't teach you."),true,false);
408                                        return;
409                                }
410                                if(!CMLib.flags().canBeSeenBy(monster,student))
411                                {
412                                        CMLib.commands().postSay(monster,student,L("You can't see me, so I can't teach you."),true,false);
413                                        return;
414                                }
415                                final Ability myAbility=CMClass.findAbility(s.trim().toUpperCase(),monster);
416                                if(myAbility==null)
417                                {
418                                        ExpertiseLibrary.ExpertiseDefinition theExpertise=null;
419                                        if(trainableExpertises==null)
420                                        {
421                                                trainableExpertises=new LinkedList<ExpertiseLibrary.ExpertiseDefinition>();
422                                                trainableExpertises.addAll(CMLib.expertises().myListableExpertises(monster));
423                                                for(final Enumeration<String> exi=monster.expertises();exi.hasMoreElements();)
424                                                {
425                                                        final Pair<String,Integer> EXI=monster.fetchExpertise(exi.nextElement());
426                                                        if(EXI.getValue()==null)
427                                                        {
428                                                                final ExpertiseLibrary.ExpertiseDefinition def=CMLib.expertises().getDefinition(EXI.getKey());
429                                                                if((def != null) && (!trainableExpertises.contains(def)))
430                                                                        trainableExpertises.add(def);
431                                                        }
432                                                        else
433                                                        {
434                                                                final List<String> childrenIDs=CMLib.expertises().getStageCodes(EXI.getKey());
435                                                                for(final String experID : childrenIDs)
436                                                                {
437                                                                        final ExpertiseLibrary.ExpertiseDefinition def=CMLib.expertises().getDefinition(experID);
438                                                                        if((def != null) && (!trainableExpertises.contains(def)))
439                                                                                trainableExpertises.add(def);
440                                                                }
441                                                        }
442                                                }
443                                        }
444                                        for(final ExpertiseLibrary.ExpertiseDefinition def : trainableExpertises)
445                                        {
446                                                if((def.name().equalsIgnoreCase(s))
447                                                &&(theExpertise==null))
448                                                        theExpertise=def;
449                                        }
450                                        if(theExpertise==null)
451                                        {
452                                                for(final ExpertiseLibrary.ExpertiseDefinition def : trainableExpertises)
453                                                {
454                                                        if((CMLib.english().containsString(def.name(),s)
455                                                        &&(theExpertise==null)))
456                                                                theExpertise=def;
457                                                }
458                                        }
459                                        if(theExpertise!=null)
460                                        {
461                                                if(!CMLib.expertises().postTeach(monster,student,theExpertise))
462                                                        return;
463                                        }
464                                        else
465                                        if((CMClass.findCharClass(s.trim())!=null))
466                                                CMLib.commands().postSay(monster,student,L("I've heard of @x1, but that's an class-- try TRAINing  for it.",s),true,false);
467                                        else
468                                        {
469                                                for(final Enumeration<ExpertiseLibrary.ExpertiseDefinition> e=CMLib.expertises().definitions(); e.hasMoreElements();)
470                                                {
471                                                        final ExpertiseLibrary.ExpertiseDefinition def=e.nextElement();
472                                                        if(def.name().equalsIgnoreCase(s))
473                                                        {
474                                                                theExpertise=def;
475                                                                break;
476                                                        }
477                                                }
478                                                if(theExpertise==null)
479                                                        CMLib.commands().postSay(monster,student,L("I'm sorry, but I've never heard of @x1",s),true,false);
480                                                else
481                                                        CMLib.commands().postSay(monster,student,L("I'm sorry, but I do not know @x1.",theExpertise.name()));
482                                        }
483                                        return;
484                                }
485                                if(giveABonus)
486                                {
487                                        monster.baseCharStats().setStat(CharStats.STAT_INTELLIGENCE,25);
488                                        monster.baseCharStats().setStat(CharStats.STAT_WISDOM,25);
489                                        monster.recoverCharStats();
490                                }
491
492                                final int prof75=(int)Math.round(CMath.mul(CMLib.ableMapper().getMaxProficiency(student,true,myAbility.ID()),0.75));
493                                myAbility.setProficiency(prof75/2);
494                                CMLib.expertises().postTeach(monster,student,myAbility);
495                                monster.baseCharStats().setStat(CharStats.STAT_INTELLIGENCE,19);
496                                monster.baseCharStats().setStat(CharStats.STAT_WISDOM,19);
497                                monster.recoverCharStats();
498                        }
499                }
500
501        }
