035public class MoneyChanger extends StdBehavior
036{
037        @Override
038        public String ID()
039        {
040                return "MoneyChanger";
041        }
042
043        protected Map<String,Double> rates=new Hashtable<String,Double>();
044        protected double spaceMaxCut=0.0;
045        protected long spaceMaxDistance=SpaceObject.Distance.GalaxyRadius.dm;
046        protected double cut=0.05;
047        private boolean complainedAboutSpaceError=false;
048        
049        @Override
050        public String accountForYourself()
051        {
052                return "money changing for a fee";
053        }
054
055        @Override
056        public void startBehavior(PhysicalAgent forMe)
057        {
058                if(forMe==null)
059                        return;
060                if(!(forMe instanceof MOB))
061                        return;
062                // i suppose this is for accepting heavy loads of coin
063                ((MOB)forMe).baseCharStats().setStat(CharStats.STAT_STRENGTH,100);
064                super.startBehavior(forMe);
065        }
066
067        protected final Map<String,Double> getRatesFor(final Environmental affecting, String currency)
068        {
069                if(spaceMaxCut<=0.0)
070                        return rates;
071                currency=currency.toUpperCase();
072                if(rates.containsKey(currency))
073                        return rates;
074                String myCurrency=CMLib.beanCounter().getCurrency(affecting);
075                if(myCurrency.equalsIgnoreCase(currency))
076                {
077                        rates.put(currency, Double.valueOf(cut));
078                        return rates;
079                }
080                SpaceObject homeO=CMLib.map().getSpaceObject(affecting, false);
081                if(homeO!=null)
082                {
083                        myCurrency=CMLib.beanCounter().getCurrency(homeO);
084                        if(myCurrency.equalsIgnoreCase(currency))
085                        {
086                                rates.put(currency, Double.valueOf(cut));
087                                return rates;
088                        }
089                }
090                else
091                {
092                        // no space object, wtf? this SHOULD fail
093                        if(!complainedAboutSpaceError)
094                        {
095                                complainedAboutSpaceError=true;
096                                Log.errOut("MoneyChanger",affecting.Name()+" is not on a planet, so space rates cannot apply!");
097                        }
098                        return rates;
099                }
100                for(final Enumeration<Area> a=CMLib.map().spaceAreas();a.hasMoreElements();)
101                {
102                        Area A=a.nextElement();
103                        if((A!=null)&&(A!=homeO))
104                        {
105                                myCurrency=CMLib.beanCounter().getCurrency(A);
106                                if(myCurrency.equalsIgnoreCase(currency))
107                                {
108                                        SpaceObject oA=(SpaceObject)A;
109                                        long distance=CMLib.map().getDistanceFrom(homeO, oA);
110                                        if((distance<0)||(distance>spaceMaxDistance))
111                                        {
112                                                rates.put(currency, Double.valueOf(spaceMaxCut));
113                                        }
114                                        else
115                                        {
116                                                double pct=CMath.div(distance, spaceMaxDistance);
117                                                double amt=spaceMaxCut*pct;
118                                                if(amt < cut)
119                                                        amt=cut;
120                                                rates.put(currency, Double.valueOf(cut));
121                                        }
122                                        return rates;
123                                }
124                        }
125                }
126                
127                return rates;
128        }
129        
130        protected boolean doIExchangeThisCurrency(final Environmental affecting, final String currency)
131        {
132                final Map<String,Double> rates=getRatesFor(affecting, currency);
133                return ((rates.size()==0)||(rates.containsKey(currency.toUpperCase())));
134        }
135        
136        protected double getMyCut(final Environmental affecting, final String currency)
137        {
138                final Map<String,Double> rates=getRatesFor(affecting,currency);
139                if((rates.size()>0)&&(rates.containsKey(currency.toUpperCase())))
140                        return rates.get(currency.toUpperCase()).doubleValue();
141                return cut;
142        }

144        @Override
145        public void setParms(String newParm)
146        {
147                super.setParms(newParm);
148                rates.clear();
149                cut=0.05;
150                spaceMaxCut=0.0;
151                spaceMaxDistance=SpaceObject.Distance.GalaxyRadius.dm;
152                newParm=newParm.toUpperCase();
153                int x=newParm.indexOf('=');
154                while(x>0)
155                {
156                        int lastSp=newParm.lastIndexOf(' ',x);
157                        if(lastSp<0)
158                                lastSp=0;
159                        if((lastSp>=0)&&(lastSp<x-1)&&(Character.isLetter(newParm.charAt(x-1))))
160                        {
161                                String parm=newParm.substring(lastSp,x).trim().toUpperCase();
162                                while((x<newParm.length())&&(newParm.charAt(x)!='='))
163                                        x++;
164                                if(x<newParm.length())
165                                {
166                                        while((x<newParm.length())
167                                        &&(!Character.isDigit(newParm.charAt(x)))
168                                        &&(newParm.charAt(x)!='.'))
169                                                x++;
170                                        if(x<newParm.length())
171                                        {
172                                                newParm=newParm.substring(x);
173                                                x=0;
174                                                while((x<newParm.length())
175                                                &&((Character.isDigit(newParm.charAt(x)))||(newParm.charAt(x)=='.')))
176                                                        x++;
177                                                double val=CMath.s_double(newParm.substring(0,x));
178                                                if(newParm.substring(0,x).indexOf('.')<0)
179                                                        val= CMath.s_long(newParm.substring(0,x));
180                                                if(x<newParm.length())
181                                                        newParm=newParm.substring(x+1);
182                                                else
183                                                        newParm="";
184                                                if(parm.equalsIgnoreCase("default"))
185                                                        parm="";
186                                                if(parm.equalsIgnoreCase("spacemaxcut"))
187                                                        spaceMaxCut=val/100.0;
188                                                else
189                                                if(parm.equalsIgnoreCase("spacemaxdistance"))
190                                                        spaceMaxDistance=Math.round(CMath.mul(SpaceObject.Distance.GalaxyRadius.dm, val/100.0));
191                                                else
192                                                if(parm.equalsIgnoreCase("cut"))
193                                                        cut=val/100.0;
194                                                else
195                                                        rates.put(parm,Double.valueOf(val/100.0));
196                                        }
197                                }
198                        }
199                        x=newParm.indexOf('=');
200                }
201        }

203        @Override
204        public boolean okMessage(final Environmental host, final CMMsg msg)
205        {
206                if(!super.okMessage(host,msg))
207                        return false;
208                final MOB source=msg.source();
209                if(!canFreelyBehaveNormal(host))
210                        return true;
211                final MOB observer=(MOB)host;
212                if((source!=observer)
213                &&(msg.amITarget(observer))
214                &&(msg.targetMinor()==CMMsg.TYP_GIVE)
215                &&(!CMSecurity.isAllowed(source,source.location(),CMSecurity.SecFlag.CMDROOMS))
216                &&(msg.tool()!=null))
217                {
218                        if(!(msg.tool() instanceof Coins))
219                        {
220                                CMLib.commands().postSay(observer,source,L("I'm sorry, I can only accept money."),true,false);
221                                return false;
222                        }
223                        else
224                        if(!doIExchangeThisCurrency(host,((Coins)msg.tool()).getCurrency()))
225                        {
226                                CMLib.commands().postSay(observer,source,L("I'm sorry, I don't accept that kind of currency."),true,false);
227                                return false;
228                        }
229                        double value=((Coins)msg.tool()).getTotalValue();
230                        final String currency=((Coins)msg.tool()).getCurrency().toUpperCase();
231                        double takeCut=getMyCut(host,currency);
232                        double amountToTake=CMLib.beanCounter().abbreviatedRePrice(observer,value*takeCut);
233                        if((amountToTake>0.0)&&(amountToTake<CMLib.beanCounter().getLowestDenomination(CMLib.beanCounter().getCurrency(observer))))
234                                amountToTake=CMLib.beanCounter().getLowestDenomination(CMLib.beanCounter().getCurrency(observer));
235                        value-=amountToTake;
236                        observer.recoverPhyStats();
237                        final Coins C=CMLib.beanCounter().makeBestCurrency(observer,value);
238                        if((value<=0)||(C==null))
239                        {
240                                CMLib.commands().postSay(observer,source,L("I'm sorry, I can not change such a small amount."),true,false);
241                                return false;
242                        }
243                }
244                return true;
245        }

247        @Override
248        public void executeMsg(final Environmental affecting, final CMMsg msg)
249        {
250                super.executeMsg(affecting,msg);
251                final MOB source=msg.source();
252                if(!canFreelyBehaveNormal(affecting))
253                        return;
254                final MOB observer=(MOB)affecting;
255
256                if((source!=observer)
257                &&(msg.amITarget(observer))
258                &&(msg.targetMinor()==CMMsg.TYP_GIVE)
259                &&(msg.tool() instanceof Coins))
260                {
261                        if((CMLib.flags().canBeSeenBy(source,observer))
262                        &&(CMLib.flags().canBeSeenBy(observer,source)))
263                        {
264                                double value=((Coins)msg.tool()).getTotalValue();
265                                final String currency=((Coins)msg.tool()).getCurrency().toUpperCase();
266                                double takeCut=getMyCut(affecting,currency);
267                                double amountToTake=CMLib.beanCounter().abbreviatedRePrice(observer,value*takeCut);
268                                if((amountToTake>0.0)&&(amountToTake<CMLib.beanCounter().getLowestDenomination(CMLib.beanCounter().getCurrency(observer))))
269                                        amountToTake=CMLib.beanCounter().getLowestDenomination(CMLib.beanCounter().getCurrency(observer));
270                                value-=amountToTake;
271                                observer.recoverPhyStats();
272                                final Coins C=CMLib.beanCounter().makeBestCurrency(observer,value);
273                                if((value>0.0)&&(C!=null))
274                                {
275                                        // this message will actually end up triggering the hand-over.
276                                        final CMMsg newMsg=CMClass.getMsg(observer,source,C,CMMsg.MSG_SPEAK,L("^T<S-NAME> say(s) 'Thank you for your business' to <T-NAMESELF>.^?"));
277                                        C.setOwner(observer);
278                                        final long num=C.getNumberOfCoins();
279                                        final String curr=C.getCurrency();
280                                        final double denom=C.getDenomination();
281                                        C.destroy();
282                                        C.setNumberOfCoins(num);
283                                        C.setCurrency(curr);
284                                        C.setDenomination(denom);
285                                        msg.addTrailerMsg(newMsg);
286                                }
287                                else
288                                        CMLib.commands().postSay(observer,source,L("Gee, thanks. :)"),true,false);
289                                ((Coins)msg.tool()).destroy();
290                        }
291                        else
292                        if(!CMLib.flags().canBeSeenBy(source,observer))
293                                CMLib.commands().postSay(observer,null,L("Wha?  Where did this come from?  Cool!"),true,false);
294                }
295                else
296                if((msg.source()==observer)
297                &&(msg.target() instanceof MOB)
298                &&(msg.targetMinor()==CMMsg.TYP_SPEAK)
299                &&(msg.tool() instanceof Coins)
300                &&(((Coins)msg.tool()).amDestroyed())
301                &&(!msg.source().isMine(msg.tool()))
302                &&(!((MOB)msg.target()).isMine(msg.tool())))
303                        CMLib.beanCounter().giveSomeoneMoney(msg.source(),(MOB)msg.target(),((Coins)msg.tool()).getTotalValue());
304        }
