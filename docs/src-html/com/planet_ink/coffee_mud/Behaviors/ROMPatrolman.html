035public class ROMPatrolman extends StdBehavior
036{
037        @Override
038        public String ID()
039        {
040                return "ROMPatrolman";
041        }
042        int tickTock=0;
043
044        @Override
045        public String accountForYourself()
046        {
047                return "gang member passifying";
048        }
049
050        public void keepPeace(MOB observer)
051        {
052                if(!canFreelyBehaveNormal(observer))
053                        return;
054                MOB victim=null;
055                for(int i=0;i<observer.location().numInhabitants();i++)
056                {
057                        final MOB inhab=observer.location().fetchInhabitant(i);
058                        if((inhab!=null)&&(inhab.isInCombat()))
059                        {
060                                if(inhab.phyStats().level()>inhab.getVictim().phyStats().level())
061                                        victim=inhab;
062                                else
063                                        victim=inhab.getVictim();
064                        }
065                }
066
067                if(victim==null)
068                        return;
069                if(BrotherHelper.isBrother(victim,observer,false))
070                        return;
071                observer.location().show(observer,null,CMMsg.MSG_NOISYMOVEMENT,L("<S-NAME> blow(s) down hard on <S-HIS-HER> whistle. ***WHEEEEEEEEEEEET***"));
072                for(final Enumeration<Room> r=observer.location().getArea().getMetroMap();r.hasMoreElements();)
073                {
074                        final Room R=r.nextElement();
075                        if((R!=observer.location())&&(R.numPCInhabitants()>0))
076                                R.showHappens(CMMsg.MSG_NOISE,L("You hear a shrill whistling sound in the distance."));
077                }
078
079                Item weapon=observer.fetchWieldedItem();
080                if(weapon==null)
081                        weapon=observer.getNaturalWeapon();
082                boolean makePeace=false;
083                boolean fight=false;
084                switch(CMLib.dice().roll(1,7,-1))
085                {
086                case 0:
087                        observer.location().show(observer,null,CMMsg.MSG_SPEAK,L("^T<S-NAME> yell(s) 'All roit! All roit! break it up!'^?"));
088                        makePeace=true;
089                        break;
090                case 1:
091                        observer.location().show(observer,null,CMMsg.MSG_SPEAK,L("^T<S-NAME> sigh(s) 'Society's to blame, but what's a bloke to do?'^?"));
092                        fight=true;
093                        break;
094                case 2:
095                        observer.location().show(observer,null,CMMsg.MSG_SPEAK,L("^T<S-NAME> mumble(s) 'bloody kids will be the death of us all.'^?"));
096                        break;
097                case 3:
098                        observer.location().show(observer,null,CMMsg.MSG_SPEAK,L("^T<S-NAME> yell(s) 'Stop that! Stop that!' and attack(s).^?"));
099                        fight=true;
100                        break;
101                case 4:
102                        observer.location().show(observer,null,CMMsg.MSG_NOISYMOVEMENT,L("<S-NAME> pull(s) out his billy and go(es) to work."));
103                        fight=true;
104                        break;
105                case 5:
106                        observer.location().show(observer,null,CMMsg.MSG_NOISYMOVEMENT,L("<S-NAME> sigh(s) in resignation and proceed(s) to break up the fight."));
107                        makePeace=true;
108                        break;
109                case 6:
110                        observer.location().show(observer,null,CMMsg.MSG_SPEAK,L("^T<S-NAME> say(s) 'Settle down, you hooligans!'^?"));
111                        break;
112                 }
113
114                if(makePeace)
115                {
116                        final Room room=observer.location();
117                        for(int i=0;i<room.numInhabitants();i++)
118                        {
119                                final MOB inhab=room.fetchInhabitant(i);
120                                if((inhab!=null)
121                                &&(inhab.isInCombat())
122                                &&(inhab.getVictim().isInCombat())
123                                &&((observer.phyStats().level()>(inhab.phyStats().level()+5))
124                                &&(!CMLib.flags().isEvil(observer))))
125                                {
126                                        final String msg="<S-NAME> stop(s) <T-NAME> from fighting with "+inhab.getVictim().name();
127                                        final CMMsg msgs=CMClass.getMsg(observer,inhab,CMMsg.MSG_NOISYMOVEMENT,msg);
128                                        if(observer.location().okMessage(observer,msgs))
129                                        {
130                                                final MOB ivictim=inhab.getVictim();
131                                                if(ivictim!=null)
132                                                        ivictim.makePeace(true);
133                                        }
134                                }
135                        }
136                }
137                else
138                if(fight)
139                        CMLib.combat().postAttack(observer,victim,weapon);
140        }
141
142        @Override
143        public boolean tick(final Tickable ticking, final int tickID)
144        {
145                super.tick(ticking,tickID);
146
147                if(tickID!=Tickable.TICKID_MOB)
148                        return true;
149                final MOB mob=(MOB)ticking;
150                tickTock--;
151                if(tickTock<=0)
152                {
153                        tickTock=CMLib.dice().roll(1,3,0);
154                        keepPeace(mob);
155                }
156                return true;
157        }
158}
