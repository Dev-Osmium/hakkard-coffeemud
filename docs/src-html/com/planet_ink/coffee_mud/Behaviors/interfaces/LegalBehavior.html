035/**
036 * A LegalBehavior is a Behavior that provides functionality related to law and
037 * order within a given geographic sphere, which is usually an Area with a
038 * LegalBehavior behavior.  A LegalBehavior keeps track of Warrants against
039 * players and mobs which persist only in memory.  It also controls the behavior
040 * of arresting officers and judges, and dispenses justice by taking mobs and
041 * players to jail, putting them on parole, or issuing warnings or other
042 * punishments.
043 * @see com.planet_ink.coffee_mud.Behaviors.interfaces.Behavior
044 * @see com.planet_ink.coffee_mud.Common.interfaces.LegalWarrant
045 * @see com.planet_ink.coffee_mud.Common.interfaces.Law
046 */
047public interface LegalBehavior extends Behavior
048{
049        /** constant for the number of miliseconds in a real-life day */
050        public static final long ONE_REAL_DAY=(long)1000*60*60*24;
051        /** constant for the number of miliseconds before a warrant expires */
052        public static final long EXPIRATION_MILLIS=ONE_REAL_DAY*7; // 7 real days
053        /** constant for the number of miliseconds before an area is under legal control */
054        public static final long CONTROLTIME=ONE_REAL_DAY*3;
055
056        /**
057         * Returns whether or not the given legal warrant is still a valid, timely
058         * warrant that can be acted upon by law enforcement.
059         * @see com.planet_ink.coffee_mud.Common.interfaces.LegalWarrant
060         * @param W the legal warrant to inspect
061         * @param debugging whether debug information should be sent to the log
062         * @return whether the given warrant is still a valid, timely crime.
063         */
064        public boolean isStillACrime(LegalWarrant W, boolean debugging);
065
066        /**
067         * Inspects the circumstances of, and if necessary, assigns a warrant to
068         * be handled by local law enforcement and judges.  Call this method
069         * if you want to add a warrant for a custom crime with your own qualifying
070         * flags.  For recognized crimes with crime keys, however, use the accuse
071         * command.
072         * @see com.planet_ink.coffee_mud.Behaviors.interfaces.LegalBehavior#accuse(Area, MOB, MOB, String[])
073         * @see com.planet_ink.coffee_mud.Common.interfaces.Law
074         * @see com.planet_ink.coffee_mud.Common.interfaces.Law#PUNISHMENTMASK_DESCS
075         * @param mob the accused character
076         * @param laws the system of laws to use as a basis
077         * @param myArea the geographical domain that the laws apply to
078         * @param target the victim of a crime, if any
079         * @param crimeLocs string of location flags, e.g. !indoors !home keyword !keyword
080         * @param crimeFlags string of crime situation flags, e.g. !recently !combat
081         * @param crime string description of the crime, e.g. robbing T-NAME
082         * @param sentence string sentence action, e.g. warn, parole1, jail1, death
083         * @param warnMsg string the officer will say to explain the seriousness of the crime
084         * @return whether or not the warrant was successfully issued
085         */
086        public boolean fillOutWarrant(MOB mob,
087                                                                  Law laws,
088                                                                  Area myArea,
089                                                                  Environmental target,
090                                                                  String crimeLocs,
091                                                                  String crimeFlags,
092                                                                  String crime,
093                                                                  String sentence,
094                                                                  String warnMsg);
095
096        /**
097         * A method that transfers a warrant out on the accused to a different
098         * framed individual
099         * @param myArea the geographical legal area
100         * @param accused the mob with actual warrants out on him/her
101         * @param framed the person to transfer the first warrant to.
102         * @return whether warrants were actually transferred from the accused to the framed
103         */
104        public boolean frame(Area myArea, MOB accused, MOB framed);
105
106        /**
107         * Assigns an officer and begins the automated arresting procedure. The
108         * target criminal must have a warrant out for this to end well for the
109         * state.
110         * @param myArea the geographic legal area
111         * @param officer the mob to assign as the arresting officer
112         * @param mob the mob to arrest
113         * @return whether the arrest began successfully.
114         */
115        public boolean arrest(Area myArea, MOB officer, MOB mob);
116
117        /**
118         * Returns the set of laws governing the given geographic legal area,
119         * assuming that this LegalBehavior is the behavior governing the same.
120         * @see com.planet_ink.coffee_mud.Common.interfaces.Law
121         * @param myArea  the geographic legal area
122         * @return the Law object that governs the area and behavior
123         */
124        public Law legalInfo(Area myArea);
125
126        /**
127         * Returns whether the given mob is both an officer of the law, and not
128         * otherwise engaged in an arrest, and so is available to make one
129         * @param myArea the geographic legal area
130         * @param mob the pc/npc to test
131         * @return whether the mob is an officer of the law
132         */
133        public boolean isElligibleOfficer(Area myArea, MOB mob);
134
135        /**
136         * Returns whether the given mob has a valid warrant out for his/her arrest.
137         * @param myArea the geographic legal area
138         * @param accused the mob to test
139         * @return whether a warrant is available for the accused
140         */
141        public boolean hasWarrant(Area myArea, MOB accused);
142
143        /**
144         * Returns whether the given mob qualifies as an arresting officer of any
145         * sort in the given legal area.
146         * @param myArea the geographic legal area
147         * @param mob the mob to test for officerhood
148         * @return true if the mob is an officer, false otherwise
149         */
150        public boolean isAnyOfficer(Area myArea, MOB mob);
151
152        /**
153         * Returns whether the given mob qualifies as the judge in the given legal
154         * area.
155         * @param myArea the geographic legal area
156         * @param mob the mob to test for judgehood
157         * @return true if the mob is a judge, false otherwise
158         */
159        public boolean isJudge(Area myArea, MOB mob);
160
161        /**
162         * A method to change the amount of base currency currently
163         * listed as fines owed by the given mob.  A value of 0 erases.
164         * @param d the amount of base currency the mob owes
165         * @param mob the mob who owes money to the state
166         */
167        public void modifyAssessedFines(double d, MOB mob);
168
169        /**
170         * Returns the amount of base currency owed by the given mob, if any.
171         * @param mob the mob who might owe money
172         * @return the amount owed, or 0 if none.
173         */
174        public double finesOwed(MOB mob);
175
176        /**
177         * This method notifies the legal behavior that its laws have changed
178         * and need to be updated.  Call this method whenever the behaviors/
179         * areas laws have changed.
180         * @param myArea the geographic legal area
181         * @return Whether the update was necessary due to the legal parameters
182         */
183        public boolean updateLaw(Area myArea);
184
185        /**
186         * Get the name of the clan that currently rules this area, if applicable.
187         * @see com.planet_ink.coffee_mud.Common.interfaces.Clan
188         * @return Empty string if the area is unruled, or ruled by the system. ClanID otherwise.
189         */
190        public String rulingOrganization();
191
192        /**
193         * If the legal behavior and area are conquerable by clans or foreign organizations,
194         * this method will return the name of the current controlling clan, and some information
195         * about the state of the conquest, such as control points achieved.
196         * @param myArea the geographic legal area
197         * @return information about the conquest of this area, in readable form.
198         */
199        public String conquestInfo(Area myArea);
200
201        /**
202         * Returns whether this legalbehavior governs an area that is presently
203         * legally stable.  Unconquerable areas are always stable, and areas conquered
204         * and controlled for a sufficient amount of time are also stable.
205         * @return Whether order has been restored.
206         */
207        public boolean isFullyControlled();
208
209        /**
210         * Returns the number of control points necessary to conquer the area governed
211         * by this legal behavior.  Not applicable if the legal behavior doesn't permit
212         * government changes or conquest.
213         * @see com.planet_ink.coffee_mud.Behaviors.interfaces.LegalBehavior#setControlPoints(String, int)
214         * @see com.planet_ink.coffee_mud.Behaviors.interfaces.LegalBehavior#getControlPoints(String)
215         * @return the number of control points needed to control this legal behavior
216         */
217        public int controlPoints();
218
219        /**
220         * The present chance (percent) that the area may collapse into revolt and
221         * remove itself from control.  Not applicable if the legal behavior doesn't permit
222         * government changes or conquest.
223         * @return the percent chance of revolt
224         */
225        public int revoltChance();
226
227        /**
228         * Modify the number of control points earned by the given clanID. Not
229         * applicable if the legal behavior doesn't permit government changes or conquest.
230         * @see com.planet_ink.coffee_mud.Behaviors.interfaces.LegalBehavior#controlPoints()
231         * @see com.planet_ink.coffee_mud.Behaviors.interfaces.LegalBehavior#getControlPoints(String)
232         * @param clanID the clan to assign the control points to
233         * @param newControlPoints the number of points to assign
234         */
235        public void setControlPoints(String clanID, int newControlPoints);
236
237        /**
238         * Returns the number of control points earned by the given clanID. Not
239         * applicable if the legal behavior doesn't permit government changes or conquest.
240         * @see com.planet_ink.coffee_mud.Behaviors.interfaces.LegalBehavior#controlPoints()
241         * @see com.planet_ink.coffee_mud.Behaviors.interfaces.LegalBehavior#setControlPoints(String, int)
242         * @param clanID the clan to assign the control points to
243         * @return The number of control points earned by this clan/organization
244         */
245        public int getControlPoints(String clanID);
246
247        /**
248         * Searches the list of warrants, returning those criminal mobs whose names
249         * match the search string, and still have legal warrants available for them.
250         * Use a search name of NULL to return all criminals.
251         * @see com.planet_ink.coffee_mud.MOBS.interfaces.MOB
252         * @param myArea the geographic legal area
253         * @param searchStr the name/search string to use
254         * @return a list of MOB objects
255         */
256        public List<MOB> getCriminals(Area myArea, String searchStr);
257
258        /**
259         * Returns a list of all active legal warrants available on the given
260         * mob.
261         * @see com.planet_ink.coffee_mud.Common.interfaces.LegalWarrant
262         * @param myArea the geographic legal area
263         * @param accused the mob to look for warrants for
264         * @return a list of LegalWarrant objects
265         */
266        public List<LegalWarrant> getWarrantsOf(Area myArea, MOB accused);
267
268        /**
269         * Puts a warrant on the official docket so that officers can act
270         * on them.  This method is called by other methods to finish off
271         * their work.  Any WARRANTS channels are also notified.
272         * @see com.planet_ink.coffee_mud.Behaviors.interfaces.LegalBehavior#fillOutWarrant(MOB, Law, Area, Environmental, String, String, String, String, String)
273         * @see com.planet_ink.coffee_mud.Behaviors.interfaces.LegalBehavior#addWarrant(Area, MOB, MOB, String, String, String, String, String)
274         * @see com.planet_ink.coffee_mud.Common.interfaces.LegalWarrant
275         * @param myArea the geographic legal area
276         * @param W the LegalWarrant to put on the docket
277         * @return whether the warrant was successfully added
278         */
279        public boolean addWarrant(Area myArea, LegalWarrant W);
280
281        /**
282         * Fills out and, if possible, issues a warrant for arrest using the given
283         * crime data.  Calls fillOutWarrant to do its work.
284         * @see com.planet_ink.coffee_mud.Behaviors.interfaces.LegalBehavior#fillOutWarrant(MOB, Law, Area, Environmental, String, String, String, String, String)
285         * @param myArea the geographic legal area
286         * @param accused the accused character
287         * @param victim the victim of a crime, if any
288         * @param crimeLocs string of location flags, e.g. !indoors !home keyword !keyword
289         * @param crimeFlags string of crime situation flags, e.g. !recently !combat
290         * @param crime string description of the crime, e.g. robbing T-NAME
291         * @param sentence string sentence action, e.g. warn, parole1, jail1, death
292         * @param warnMsg string the officer will say to explain the seriousness of the crime
293         * @return whether or not the warrant was successfully issued
294         */
295        public boolean addWarrant(Area myArea, MOB accused, MOB victim, String crimeLocs, String crimeFlags, String crime, String sentence, String warnMsg);
296
297        /**
298         * Removes the given warrants from the list of issued warrants.  Does not update
299         * the old-warrants (prior convictions) record, but erases the warrant completely.
300         * @see com.planet_ink.coffee_mud.Common.interfaces.LegalWarrant
301         * @param myArea the geographic legal area
302         * @param W the legal warrant to remove
303         * @return true if the warrant was found to remove, false otherwise
304         */
305        public boolean deleteWarrant(Area myArea, LegalWarrant W);
306
307        /**
308         * Removes the first warrant for the given accused criminal, for any one of
309         * the given list of official crime KEYS.  Crime KEYS are the key names of
310         * crimed, such as TAXEVASION.
311         * @param myArea the geographic legal area
312         * @param accused the mob possible accused of one of the crimes
313         * @param acquittableLaws the list of crime keys.
314         * @return whether an acquittable crime was found, and removed
315         */
316        public boolean aquit(Area myArea, MOB accused, String[] acquittableLaws);
317
318        /**
319         * Returns whether any of the given Room objects in the jails Vector
320         * is indeed an official Jail room as defined by this legal behavior.
321         * @see com.planet_ink.coffee_mud.Locales.interfaces.Room
322         * @param myArea the geographic legal area
323         * @param jails a list of Room objects to inspect
324         * @return whether any one of the room objects is, in fact, a jail
325         */
326        public boolean isJailRoom(Area myArea, List<Room> jails);
327
328        /**
329         * Issues a LegalWarrant against the accused on behalf of the given
330         * victim, for a crime listed in the list of crime keys.
331         * Calls fillOutWarrant after retreiving the remaining information about
332         * the crime key described by the parameter accusableLaws.
333         * @see com.planet_ink.coffee_mud.Behaviors.interfaces.LegalBehavior#fillOutWarrant(MOB, Law, Area, Environmental, String, String, String, String, String)
334         * @param myArea the geographic legal area
335         * @param accused the accused mob
336         * @param victim the victim of the crime
337         * @param accusableLaws a crime key, such as TAXEVASION
338         * @return whether one of the laws was found, and a warrant successfully filled out
339         */
340        public boolean accuse(Area myArea, MOB accused, MOB victim, String[] accusableLaws);
341        
342        /**
343         * If the warrant reflects someone in prison, it releases them.
344         * If the warrant reflects someone being arrested, it releases
345         * the arrest and temporarily ignores the warrant.
346         * 
347         * @param myArea the legal area
348         * @param warrant the warrant to excuse
349         */
350        public void release(Area myArea, LegalWarrant warrant);
351}
