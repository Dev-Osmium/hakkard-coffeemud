094        @Override
095        public void setParms(String newParms)
096        {
097                String myParms=newParms;
098                listOfRooms=new Vector<String>();
099                roomInfos=new Vector<String>();
100                messageInfo=new Vector<Vector<String>>();
101                mapInfo=new Vector<String>();
102                stubs=new Vector<String>();
103
104                char c=';';
105                int x=myParms.indexOf(c);
106                if(x<0)
107                {
108                        c='/';
109                        x=myParms.indexOf(c);
110                }
111                if(x>0)
112                {
113                        final String parmText=myParms.substring(0,x);
114                        myParms=myParms.substring(x+1);
115                        super.setParms(parmText);
116                }
117                roomInfos=CMParms.parseSemicolons(myParms,false);
118                parms=newParms;
119                xmlInfo=loadInfo();
120                parseMovingXML(xmlInfo);
121        }

123        protected static synchronized String loadInfo()
124        {
125                final StringBuffer str=new StringBuffer("");

127                CMFile //F=new CMFile(Resources.makeFileResourceName("behavior/"+resourceName),null,0);
128                //if((!F.exists()) || (!F.canRead()))
129                        F=new CMFile(Resources.makeFileResourceName(resourceName),null,0);
130                if((F.exists()) && (F.canRead()))
131                {
132                        final List<String> V=Resources.getFileLineVector(F.text());
133                        for(int v=0;v<V.size();v++)
134                                str.append(V.get(v));
135                        final String theString = str.toString();
136                        return theString;
137                }
138                else
139                {
140                        Log.errOut("MovingRoom","Unable to load "+Resources.makeFileResourceName("behavior/"+resourceName)+" or "+Resources.makeFileResourceName(resourceName));
141                        return "";
142                }

145        protected void parseMovingXML(String roomToParse)
146        {
147                final Vector<String> V = new Vector<String>();
148                final String theFullBlock=CMLib.xml().returnXMLBlock(roomToParse, roomInfos.get(0).toString().toUpperCase());
149                final String theStopsBlock=CMLib.xml().returnXMLBlock(theFullBlock, "STOPS");
150                final String theNormalDirBlock=CMLib.xml().returnXMLBlock(theFullBlock, "NORMALDIRECTION");
151                final String theReverseDirBlock=CMLib.xml().returnXMLBlock(theFullBlock, "REVERSEDIRECTION");
152                final String theDescriptionsBlock=CMLib.xml().returnXMLBlock(theFullBlock, "ROOMDESCRIPTIONS");
153                int x=1;
154                String thisone=CMLib.xml().returnXMLValue(theStopsBlock, "STOP1");
155                while (!"".equals(thisone))
156                {
157                        ++x;
158                        listOfRooms.addElement(thisone);
159                        thisone=CMLib.xml().returnXMLValue(theStopsBlock, "STOP"+x);
160                }
161                V.addElement(CMLib.xml().returnXMLValue(theNormalDirBlock, "TRAVELDIRECTION"));
162                V.addElement(CMLib.xml().returnXMLValue(theNormalDirBlock, "DOORSDIRECTION"));
163                V.addElement(CMLib.xml().returnXMLValue(CMLib.xml().returnXMLBlock(theNormalDirBlock, "INSIDE"), "ARRIVALINFO"));
164                V.addElement(CMLib.xml().returnXMLValue(CMLib.xml().returnXMLBlock(theNormalDirBlock, "INSIDE"), "DEPARTINFO"));
165                V.addElement(CMLib.xml().returnXMLValue(CMLib.xml().returnXMLBlock(theNormalDirBlock, "OUTSIDE"), "ARRIVALINFO"));
166                V.addElement(CMLib.xml().returnXMLValue(CMLib.xml().returnXMLBlock(theNormalDirBlock, "OUTSIDE"), "DEPARTINFO"));
167                messageInfo.addElement(new Vector<String>(V));
168                V.removeAllElements();
169                V.addElement(CMLib.xml().returnXMLValue(theReverseDirBlock, "TRAVELDIRECTION"));
170                V.addElement(CMLib.xml().returnXMLValue(theReverseDirBlock, "DOORSDIRECTION"));
171                V.addElement(CMLib.xml().returnXMLValue(CMLib.xml().returnXMLBlock(theReverseDirBlock, "INSIDE"), "ARRIVALINFO"));
172                V.addElement(CMLib.xml().returnXMLValue(CMLib.xml().returnXMLBlock(theReverseDirBlock, "INSIDE"), "DEPARTINFO"));
173                V.addElement(CMLib.xml().returnXMLValue(CMLib.xml().returnXMLBlock(theReverseDirBlock, "OUTSIDE"), "ARRIVALINFO"));
174                V.addElement(CMLib.xml().returnXMLValue(CMLib.xml().returnXMLBlock(theReverseDirBlock, "OUTSIDE"), "DEPARTINFO"));
175                messageInfo.addElement(new Vector<String>(V));
176                V.removeAllElements();
177                final String theNormalDescBlock = (CMLib.xml().returnXMLBlock(theDescriptionsBlock, "NORMALDIRECTION"));
178                final String theReverseDescBlock = (CMLib.xml().returnXMLBlock(theDescriptionsBlock, "REVERSEDIRECTION"));
179                V.addElement(CMLib.xml().returnXMLValue(CMLib.xml().returnXMLBlock(theNormalDescBlock, "INSIDE"), "DOOROPENED"));
180                V.addElement(CMLib.xml().returnXMLValue(CMLib.xml().returnXMLBlock(theNormalDescBlock, "INSIDE"), "DOORCLOSED"));
181                V.addElement(CMLib.xml().returnXMLValue(CMLib.xml().returnXMLBlock(theNormalDescBlock, "OUTSIDE"), "DOOROPENED"));
182                V.addElement(CMLib.xml().returnXMLValue(CMLib.xml().returnXMLBlock(theNormalDescBlock, "OUTSIDE"), "DOORCLOSED"));
183                V.addElement(CMLib.xml().returnXMLValue(CMLib.xml().returnXMLBlock(theReverseDescBlock, "INSIDE"), "DOOROPENED"));
184                V.addElement(CMLib.xml().returnXMLValue(CMLib.xml().returnXMLBlock(theReverseDescBlock, "INSIDE"), "DOORCLOSED"));
185                V.addElement(CMLib.xml().returnXMLValue(CMLib.xml().returnXMLBlock(theReverseDescBlock, "OUTSIDE"), "DOOROPENED"));
186                V.addElement(CMLib.xml().returnXMLValue(CMLib.xml().returnXMLBlock(theReverseDescBlock, "OUTSIDE"), "DOORCLOSED"));
187                messageInfo.addElement(new Vector<String>(V));
188                V.removeAllElements();
189                mapInfo.addElement(CMLib.xml().returnXMLValue(theFullBlock, "ROOMPRINTNAME"));
190                mapInfo.addElement(CMLib.xml().returnXMLValue(theFullBlock, "LINEPRINTNAME"));
191                mapInfo.addElement(CMLib.xml().returnXMLValue(theFullBlock, "DISPLOC"));
192        }

194        protected String fixOutputString(String incoming, Room busstopRoom)
195        {
196                String repWord="";
197                incoming = " " + incoming;
198                int i=0;
199                if (incoming.indexOf("$disproom")>0)
200                {
201                        i = incoming.indexOf("$disproom");
202                        repWord=incoming.substring(1,i)+busstopRoom.displayText()+incoming.substring(i+9);
203                }
204                else
205                if (incoming.indexOf("$traveldir")>0)
206                {
207                        i = incoming.indexOf("$traveldir");
208                        final int pos=listOfRooms.indexOf(CMLib.map().getExtendedRoomID(busstopRoom));
209                        boolean revDirName=false;
210                        if (((pos==0)||(pos==listOfRooms.size()-1))&&(currentStatus==1))
211                                revDirName=true;
212                        List<String> V=new Vector<String>();
213                        if (!revDirName)
214                        {
215                                if (isReversed)
216                                        V=messageInfo.elementAt(CODE_REVERSEBLOCK);
217                                else
218                                        V=messageInfo.elementAt(CODE_NORMALBLOCK);
219                        }
220                        else
221                        {
222                                if (isReversed)
223                                        V=messageInfo.elementAt(CODE_NORMALBLOCK);
224                                else
225                                        V=messageInfo.elementAt(CODE_REVERSEBLOCK);
226                        }
227                        repWord=incoming.substring(1,i)+V.get(0).toString()+incoming.substring(i+10);
228                }
229                else
230                if (incoming.indexOf("$outopendir")>0)
231                {
232                        i = incoming.indexOf("$outopendir");
233                        List<String> V=new Vector<String>();
234                        if (isReversed)
235                                V=messageInfo.elementAt(CODE_REVERSEBLOCK);
236                        else
237                                V=messageInfo.elementAt(CODE_NORMALBLOCK);
238                        repWord=incoming.substring(1,i)+CMLib.directions().getDirectionName(CMLib.directions().getOpDirectionCode(V.get(1).toString()))+incoming.substring(i+11);
239                }
240                else
241                if (incoming.indexOf("$inopendir")>0)
242                {
243                        i = incoming.indexOf("$inopendir");
244                        List<String> V=new Vector<String>();
245                        if (isReversed)
246                                V=messageInfo.elementAt(CODE_REVERSEBLOCK);
247                        else
248                                V=messageInfo.elementAt(CODE_NORMALBLOCK);
249                        repWord=incoming.substring(1,i)+V.get(1).toString()+incoming.substring(i+10);
250                }
251                else
252                {
253                        repWord=incoming.substring(1);
254                        return repWord;
255                }
256                repWord = fixOutputString(repWord,busstopRoom);
257                return repWord;
258        }

260        protected void removeStubs(Room busstopRoom1,Room busstopRoom2)
261        {
262                if (!stubs.isEmpty())
263                for(int s=0;s<stubs.size();s++)
264                {
265                        int i=busstopRoom1.description().indexOf(stubs.elementAt(s).toString());
266                        if (i>0)
267                                busstopRoom1.setDescription(busstopRoom1.description().substring(0,i).trim());
268                        i=busstopRoom2.description().indexOf(stubs.elementAt(s).toString());
269                        if (i>0)
270                                busstopRoom2.setDescription(busstopRoom2.description().substring(0,i).trim());
271                }
272        }

274        @Override
275        public boolean tick(final Tickable ticking, final int tickID)
276        {
277                final Vector<String> normalVec = messageInfo.elementAt(CODE_NORMALBLOCK);
278                final Vector<String> reverseVec = messageInfo.elementAt(CODE_REVERSEBLOCK);
279                final Vector<String> theDescriptions = messageInfo.elementAt(CODE_DESCRIPTIONBLOCK);
280                super.tick(ticking,tickID);
281                if(canAct(ticking,tickID))
282                {
283                        final Room subwayRoom=getBehaversRoom(ticking);
284                        if(subwayRoom==null)
285                        {
286                                Log.errOut("MovingRoom",ticking.ID()+"/"+ticking.name()+" is NOT A DADGUM ROOM!!!!");
287                                return false;
288                        }
289                        if (currentStop==0)
290                        {
291                                isReversed=false;
292                                nextStop=currentStop+1;
293                        }
294                        else
295                        if (currentStop>=(listOfRooms.size()-1))
296                        {
297                                isReversed=true;
298                                nextStop=currentStop-1;
299                        }
300                        else
301                        if (isReversed)
302                        {
303                                nextStop=currentStop-1;
304                        }
305                        else
306                        {
307                                nextStop=currentStop+1;
308                        }
309                        if((currentStop<0)
310                        ||(currentStop>=listOfRooms.size())
311                        ||(nextStop<0)
312                        ||(nextStop>=listOfRooms.size()))
313                        {
314                                Log.errOut("MovingRoom","Moving Room behavior on "+subwayRoom.roomID()+" HAD malformed rooms list in xml file.");
315                                subwayRoom.delBehavior(this);
316                                CMLib.threads().deleteTick(this,-1);
317                                if((ticking instanceof PhysicalAgent)
318                                &&(ticking!=subwayRoom))
319                                        ((PhysicalAgent)ticking).delBehavior(this);
320                                return false;
321                        }
322
323                        final String currentStopS=listOfRooms.elementAt(currentStop);
324                        final String nextStopS=listOfRooms.elementAt(nextStop);
325                        if(ticking instanceof Room)
326                        {
327                                Room currentStopRoom = CMLib.map().getRoom(currentStopS);
328                                if (currentStopRoom == null)
329                                        currentStopRoom=getBehaversRoom(ticking);
330                                Room nextStopRoom =CMLib.map().getRoom(nextStopS);
331                                if (nextStopRoom == null)
332                                        nextStopRoom=getBehaversRoom(ticking);
333                                if (currentStatus==0)
334                                {
335                                        //DEPARTING
336                                        if (isReversed)
337                                        {
338                                                currentStopRoom.showHappens(CMMsg.MSG_OK_ACTION,fixOutputString(reverseVec.elementAt(CODE0_OUTSIDEDEPARTMSG).toString(),nextStopRoom));
339                                                subwayRoom.showHappens(CMMsg.MSG_OK_ACTION,fixOutputString(reverseVec.elementAt(CODE0_INSIDEDEPARTMSG).toString(),nextStopRoom));
340                                                if (((subwayRoom.rawDoors()[CMLib.directions().getGoodDirectionCode(normalVec.elementAt(1).toString())]!=null)
341                                                                && ((currentStop==0)
342                                                                                || (currentStop==(listOfRooms.size()-1))))
343                                                ||(subwayRoom.rawDoors()[CMLib.directions().getGoodDirectionCode(reverseVec.elementAt(1).toString())]!=null))
344                                                {
345                                                        if ((currentStop==0)||(currentStop==(listOfRooms.size()-1)))
346                                                        {
347                                                                currentStopRoom.rawDoors()[CMLib.directions().getOpDirectionCode(normalVec.elementAt(1).toString())]=null;
348                                                                currentStopRoom.setRawExit(CMLib.directions().getOpDirectionCode(normalVec.elementAt(1).toString()),null);
349                                                                subwayRoom.rawDoors()[CMLib.directions().getGoodDirectionCode(normalVec.elementAt(1).toString())]=null;
350                                                                subwayRoom.setRawExit(CMLib.directions().getGoodDirectionCode(normalVec.elementAt(1).toString()),null);
351                                                        }
352                                                        else
353                                                        {
354                                                                currentStopRoom.rawDoors()[CMLib.directions().getOpDirectionCode(reverseVec.elementAt(1).toString())]=null;
355                                                                currentStopRoom.setRawExit(CMLib.directions().getOpDirectionCode(reverseVec.elementAt(1).toString()),null);
356                                                                subwayRoom.rawDoors()[CMLib.directions().getGoodDirectionCode(reverseVec.elementAt(1).toString())]=null;
357                                                                subwayRoom.setRawExit(CMLib.directions().getGoodDirectionCode(reverseVec.elementAt(1).toString()),null);
358                                                        }
359                                                        CMLib.database().DBUpdateExits(subwayRoom);
360                                                        CMLib.database().DBUpdateExits(currentStopRoom);
361                                                        subwayRoom.getArea().fillInAreaRoom(subwayRoom);
362                                                        nextStopRoom.getArea().fillInAreaRoom(currentStopRoom);
363                                                        removeStubs(subwayRoom,currentStopRoom);
364                                                        final String s1=(fixOutputString(theDescriptions.elementAt(CODE1_REVERSEINSIDECLOSED).toString(),nextStopRoom));
365                                                        final String s2=(fixOutputString(theDescriptions.elementAt(CODE1_REVERSEOUTSIDECLOSED).toString(),nextStopRoom));
366                                                        if(!stubs.contains(s1))
367                                                                stubs.addElement(s1);
368                                                        if(!stubs.contains(s2))
369                                                                stubs.addElement(s2);
370                                                        subwayRoom.setDescription(subwayRoom.description()+"  "+s1);
371                                                        currentStopRoom.setDescription(currentStopRoom.description()+"  "+s2);
372                                                }
373                                                else
374                                                {
375                                                        Log.errOut("MovingRoom","Previous room links exists, "+subwayRoom.roomID()+" "+currentStopRoom.roomID()+" not unlinking exit.");
376                                                }
377                                                super.setParms("min="+roomInfos.get(2)+" max="+roomInfos.get(2)+" chance=100;"+roomInfos.get(0)+";"+roomInfos.get(1)+";"+roomInfos.get(2));
378                                                currentStatus=1;
379                                        }
380                                        else
381                                        {
382                                                // departing, not reversed
383                                                currentStopRoom.showHappens(CMMsg.MSG_OK_ACTION,fixOutputString(normalVec.elementAt(CODE0_OUTSIDEDEPARTMSG).toString(),nextStopRoom));
384                                                subwayRoom.showHappens(CMMsg.MSG_OK_ACTION,fixOutputString(normalVec.elementAt(CODE0_INSIDEDEPARTMSG).toString(),nextStopRoom));
385                                                if (((subwayRoom.rawDoors()[CMLib.directions().getGoodDirectionCode(reverseVec.elementAt(1).toString())]!=null)
386                                                                && ((currentStop==0)
387                                                                                || (currentStop==(listOfRooms.size()-1))))
388                                                ||(subwayRoom.rawDoors()[CMLib.directions().getGoodDirectionCode(normalVec.elementAt(1).toString())]!=null))
389                                                {
390                                                        if ((currentStop==0)||(currentStop==(listOfRooms.size()-1)))
391                                                        {
392                                                                currentStopRoom.rawDoors()[CMLib.directions().getOpDirectionCode(reverseVec.elementAt(1).toString())]=null;
393                                                                currentStopRoom.setRawExit(CMLib.directions().getOpDirectionCode(reverseVec.elementAt(1).toString()),null);
394                                                                subwayRoom.rawDoors()[CMLib.directions().getGoodDirectionCode(reverseVec.elementAt(1).toString())]=null;
395                                                                subwayRoom.setRawExit(CMLib.directions().getGoodDirectionCode(reverseVec.elementAt(1).toString()),null);
396                                                        }
397                                                        else
398                                                        {
399                                                                currentStopRoom.rawDoors()[CMLib.directions().getOpDirectionCode(normalVec.elementAt(1).toString())]=null;
400                                                                currentStopRoom.setRawExit(CMLib.directions().getOpDirectionCode(normalVec.elementAt(1).toString()),null);
401                                                                subwayRoom.rawDoors()[CMLib.directions().getGoodDirectionCode(normalVec.elementAt(1).toString())]=null;
402                                                                subwayRoom.setRawExit(CMLib.directions().getGoodDirectionCode(normalVec.elementAt(1).toString()),null);
403                                                        }
404                                                        CMLib.database().DBUpdateExits(subwayRoom);
405                                                        CMLib.database().DBUpdateExits(currentStopRoom);
406                                                        subwayRoom.getArea().fillInAreaRoom(subwayRoom);
407                                                        nextStopRoom.getArea().fillInAreaRoom(currentStopRoom);
408                                                        removeStubs(subwayRoom,currentStopRoom);
409                                                        final String s1=(fixOutputString(theDescriptions.elementAt(CODE1_NORMALINSIDECLOSED).toString(),nextStopRoom));
410                                                        final String s2=(fixOutputString(theDescriptions.elementAt(CODE1_NORMALOUTSIDECLOSED).toString(),nextStopRoom));
411                                                        if(!stubs.contains(s1))
412                                                                stubs.addElement(s1);
413                                                        if(!stubs.contains(s2))
414                                                                stubs.addElement(s2);
415                                                        subwayRoom.setDescription(subwayRoom.description()+"  "+s1);
416                                                        currentStopRoom.setDescription(currentStopRoom.description()+"  "+s2);
417                                                }
418                                                else
419                                                {
420                                                        Log.errOut("MovingRoom","Previous room links exists, "+subwayRoom.roomID()+" "+currentStopRoom.roomID()+" not unlinking exit.");
421                                                }
422                                                super.setParms("min="+roomInfos.get(2)+" max="+roomInfos.get(2)+" chance=100;"+roomInfos.get(0)+";"+roomInfos.get(1)+";"+roomInfos.get(2));
423                                                currentStatus=1;
424                                        }
425                                }
426                                else
427                                {
428                                        //ARRIVING
429                                        if (isReversed)
430                                        {
431                                                subwayRoom.showHappens(CMMsg.MSG_OK_ACTION,fixOutputString(reverseVec.elementAt(2).toString(),nextStopRoom));
432                                                nextStopRoom.showHappens(CMMsg.MSG_OK_ACTION,fixOutputString(reverseVec.elementAt(4).toString(),nextStopRoom));
433                                                currentStatus=0;
434                                                if ((nextStopRoom.rawDoors()[CMLib.directions().getOpDirectionCode(reverseVec.elementAt(1).toString())]==null)
435                                                ||(nextStopRoom.rawDoors()[CMLib.directions().getOpDirectionCode(reverseVec.elementAt(1).toString())]==subwayRoom))
436                                                {
437                                                        final Exit thisNewExit=CMClass.getExit("StdOpenDoorway");
438                                                        subwayRoom.rawDoors()[CMLib.directions().getGoodDirectionCode(reverseVec.elementAt(1).toString())]=nextStopRoom;
439                                                        subwayRoom.setRawExit(CMLib.directions().getGoodDirectionCode(reverseVec.elementAt(1).toString()),thisNewExit);
440                                                        nextStopRoom.rawDoors()[CMLib.directions().getOpDirectionCode(reverseVec.elementAt(1).toString())]=subwayRoom;
441                                                        nextStopRoom.setRawExit(CMLib.directions().getOpDirectionCode(reverseVec.elementAt(1).toString()),thisNewExit);
442                                                        CMLib.database().DBUpdateExits(subwayRoom);
443                                                        CMLib.database().DBUpdateExits(nextStopRoom);
444                                                        subwayRoom.getArea().fillInAreaRoom(subwayRoom);
445                                                        nextStopRoom.getArea().fillInAreaRoom(nextStopRoom);
446                                                        removeStubs(subwayRoom,nextStopRoom);
447                                                        final String s1=(fixOutputString(theDescriptions.elementAt(CODE1_REVERSEINSIDEOPEN).toString(),nextStopRoom));
448                                                        final String s2=(fixOutputString(theDescriptions.elementAt(CODE1_REVERSEOUTSIDEOPEN).toString(),nextStopRoom));
449                                                        if(!stubs.contains(s1))
450                                                                stubs.addElement(s1);
451                                                        if(!stubs.contains(s2))
452                                                                stubs.addElement(s2);
453                                                        subwayRoom.setDescription(subwayRoom.description()+"  "+s1);
454                                                        nextStopRoom.setDescription(nextStopRoom.description()+"  "+s2);
455                                                }
456                                                else
457                                                {
458                                                        Log.errOut("MovingRoom","Previous room links exists, "+subwayRoom.roomID()+" "+nextStopRoom.roomID()+" not linking exits.");
459                                                }
460                                                super.setParms("min="+roomInfos.get(1)+" max="+roomInfos.get(1)+" chance=100;"+roomInfos.get(0)+";"+roomInfos.get(1)+";"+roomInfos.get(2));
461                                                currentStop = nextStop;
462                                        }
463                                        else
464                                        {
465                                                // arriving, not reversed
466                                                subwayRoom.showHappens(CMMsg.MSG_OK_ACTION,fixOutputString(normalVec.elementAt(2).toString(),nextStopRoom));
