035public class RandomTraps extends ActiveTicker
036{
037        @Override
038        public String ID()
039        {
040                return "RandomTraps";
041        }

043        @Override
044        protected int canImproveCode()
045        {
046                return Behavior.CAN_ROOMS|Behavior.CAN_AREAS;
047        }

049        protected Vector<Physical> maintained=new Vector<Physical>();
050        protected int minTraps=1;
051        protected int maxTraps=1;
052        protected int avgTraps=1;
053        protected boolean doAnyItems=false;
054        protected boolean doAnyContainers=false;
055        protected boolean doDooredContainers=false;
056        protected boolean doLockedContainers=false;
057        protected boolean doAnyDoors=false;
058        protected boolean doAnyLockedDoors=false;
059        protected boolean doRooms=false;
060
061        protected Set<Integer> restrictedLocales=null;
062        private int tickStatus=Tickable.STATUS_NOT;

064        @Override

082        @Override
083        public void setParms(String newParms)
084        {
085                maintained=new Vector<Physical>();
086                doAnyItems=false;
087                doAnyContainers=false;
088                doDooredContainers=false;
089                doLockedContainers=false;
090                doAnyDoors=false;
091                doAnyLockedDoors=true;
092                doRooms=false;
093                final int x=newParms.indexOf(';');
094                String oldParms=newParms;
095                restrictedLocales=null;
096                if(x>=0)
097                {
098                        oldParms=newParms.substring(0,x).trim();
099                        String p=CMParms.getParmStr(oldParms,"ROOMS","NO").toUpperCase().trim();
100                        if(p.startsWith("Y"))
101                                doRooms=true;
102                        p=CMParms.getParmStr(oldParms,"ITEMS","NO").toUpperCase().trim();
103                        if(p.startsWith("Y"))
104                        {
105                                doAnyItems=true;
106                                doAnyContainers=true;
107                                doDooredContainers=true;
108                                doLockedContainers=true;
109                        }
110                        else
111                        if(p.startsWith("CONT"))
112                        {
113                                doAnyItems=false;
114                                doAnyContainers=true;
115                                doDooredContainers=true;
116                                doLockedContainers=true;
117                        }
118                        else
119                        if(p.startsWith("LID"))
120                        {
121                                doAnyItems=false;
122                                doAnyContainers=false;
123                                doDooredContainers=true;
124                                doLockedContainers=true;
125                        }
126                        else
127                        if(p.startsWith("LOCK"))
128                        {
129                                doAnyItems=false;
130                                doAnyContainers=false;
131                                doDooredContainers=false;
132                                doLockedContainers=true;
133                        }
134                        else
135                        if(p.startsWith("NO"))
136                        {
137                                doAnyItems=false;
138                                doAnyContainers=false;
139                                doDooredContainers=false;
140                                doLockedContainers=false;
141                        }
142
143                        p=CMParms.getParmStr(oldParms,"EXITS","LOCKED").toUpperCase().trim();
144                        if((p.startsWith("Y"))
145                        ||(p.startsWith("DOOR")))
146                        {
147                                doAnyDoors=true;
148                                doAnyLockedDoors=true;
149                        }
150                        else
151                        if(p.startsWith("LOCK"))
152                        {
153                                doAnyDoors=false;
154                                doAnyLockedDoors=true;
155                        }
156                        else
157                        if(p.startsWith("NO"))
158                        {
159                                doAnyDoors=false;
160                                doAnyLockedDoors=false;
161                        }
162
163                        final Vector<String> V=CMParms.parse(oldParms);
164                        for(int v=0;v<V.size();v++)
165                        {
166                                String s=V.elementAt(v);
167                                if((s.startsWith("+")||(s.startsWith("-")))&&(s.length()>1))
168                                {
169                                        if(restrictedLocales==null)
170                                                restrictedLocales=new TreeSet<Integer>();
171                                        if(s.equalsIgnoreCase("+ALL"))
172                                                restrictedLocales.clear();
173                                        else
174                                        if(s.equalsIgnoreCase("-ALL"))
175                                        {
176                                                restrictedLocales.clear();
177                                                for(int i=0;i<Room.DOMAIN_INDOORS_DESCS.length;i++)
178                                                        restrictedLocales.add(Integer.valueOf(Room.INDOORS+i));
179                                                for(int i=0;i<Room.DOMAIN_OUTDOOR_DESCS.length;i++)
180                                                        restrictedLocales.add(Integer.valueOf(i));
181                                        }
182                                        else
183                                        {
184                                                final char c=s.charAt(0);
185                                                s=s.substring(1).toUpperCase().trim();
186                                                int code=-1;
187                                                for(int i=0;i<Room.DOMAIN_INDOORS_DESCS.length;i++)
188                                                {
189                                                        if(Room.DOMAIN_INDOORS_DESCS[i].startsWith(s))
190                                                                code=Room.INDOORS+i;
191                                                }
192                                                if(code>=0)
193                                                {
194                                                        if((c=='+')&&(restrictedLocales.contains(Integer.valueOf(code))))
195                                                                restrictedLocales.remove(Integer.valueOf(code));
196                                                        else
197                                                        if((c=='-')&&(!restrictedLocales.contains(Integer.valueOf(code))))
198                                                                restrictedLocales.add(Integer.valueOf(code));
199                                                }
200                                                code=-1;
201                                                for(int i=0;i<Room.DOMAIN_OUTDOOR_DESCS.length;i++)
202                                                {
203                                                        if(Room.DOMAIN_OUTDOOR_DESCS[i].startsWith(s))
204                                                                code=i;
205                                                }
206                                                if(code>=0)
207                                                {
208                                                        if((c=='+')&&(restrictedLocales.contains(Integer.valueOf(code))))
209                                                                restrictedLocales.remove(Integer.valueOf(code));
210                                                        else
211                                                        if((c=='-')&&(!restrictedLocales.contains(Integer.valueOf(code))))
212                                                                restrictedLocales.add(Integer.valueOf(code));
213                                                }
214                                        }
215                                }
216                        }
217                }
218                super.setParms(oldParms);
219                minTraps=CMParms.getParmInt(oldParms,"mintraps",1);
220                maxTraps=CMParms.getParmInt(oldParms,"maxtraps",1);
221                if(maxTraps<minTraps)
222                        maxTraps=minTraps;
223                avgTraps=CMLib.dice().roll(1,maxTraps-minTraps,minTraps);
224                parms=newParms;
225                if((restrictedLocales!=null)&&(restrictedLocales.size()==0))
226                        restrictedLocales=null;
227        }

229        protected void makeRoomElligible(Room R, List<Physical> elligible)
230        {
231                if(R==null)
232                        return;
233                if((restrictedLocales!=null)
234                &&(restrictedLocales.contains(Integer.valueOf(R.domainType()))))
235                        return;
236
237                if(R instanceof GridLocale)
238                {
239                        final List<Room> map=((GridLocale)R).getAllRooms();
240                        if(map.size()==0)
241                                elligible.add(R);
242                        else
243                        for (final Room room : ((GridLocale)R).getAllRooms())
244                                elligible.add(room);
245                }
246                else
247                        elligible.add(R);
248        }

250        @Override
251        public boolean tick(final Tickable ticking, final int tickID)
252        {
253                tickStatus=Tickable.STATUS_START;
254                super.tick(ticking,tickID);
255                tickStatus=Tickable.STATUS_MISC+0;
256                for(int i=maintained.size()-1;i>=0;i--)
257                {
258                        if(CMLib.utensils().fetchMyTrap(maintained.elementAt(i))==null)
259                                maintained.removeElementAt(i);
260                }
261                if(maintained.size()>=maxTraps)
262                {
263                        tickStatus=Tickable.STATUS_NOT;
264                        return true;
265                }
266                tickStatus=Tickable.STATUS_MISC+1;
267                if((canAct(ticking,tickID))||(maintained.size()<minTraps))
268                {
269                        tickStatus=Tickable.STATUS_MISC+2;
270                        final List<Trap> allTraps=new ArrayList<Trap>();
271                        if(maintained.size()<avgTraps)
272                        {
273                                for(final Enumeration<Ability> e=CMClass.abilities();e.hasMoreElements();)
274                                {
275                                        final Ability A=e.nextElement();
276                                        if(A instanceof Trap)
277                                                allTraps.add((Trap)A);
278                                }
279                        }
280
281                        tickStatus=Tickable.STATUS_MISC+3;
282
283                        if(maintained.size()<avgTraps)
284                        {
285                                final List<Physical> elligible=new Vector<Physical>();
286                                if(ticking instanceof Room)
287                                {
288                                        tickStatus=Tickable.STATUS_MISC+4;
289                                        makeRoomElligible((Room)ticking,elligible);
290                                }
291                                else
292                                if((ticking instanceof Area)&&(((Area)ticking).metroSize()>0))
293                                {
294                                        tickStatus=Tickable.STATUS_MISC+5;
295                                        for(final Enumeration<Room> m=((Area)ticking).getMetroMap();m.hasMoreElements();)
296                                                makeRoomElligible(m.nextElement(),elligible);
297                                }
298                                else
299                                        return true;
300
301                                tickStatus=Tickable.STATUS_MISC+6;
302                                if(elligible.size()==0)
303                                        return true;
304
305                                final int oldSize=elligible.size();
306                                for(int r=0;r<oldSize;r++)
307                                {
308                                        tickStatus=Tickable.STATUS_MISC+7;
309                                        Room R=null;
310                                        try
311                                        {
312                                                if(elligible.get(r) instanceof Room)
313                                                        R = (Room)elligible.get(r);
314                                        }
315                                        catch (final IndexOutOfBoundsException e)
316                                        {
317                                        }
318                                        if(R==null)
319                                                continue;
320
321                                        if((doAnyDoors)||(doAnyLockedDoors))
322                                        for(int d=Directions.NUM_DIRECTIONS()-1;d>=0;d--)
323                                        {
324                                                tickStatus=Tickable.STATUS_MISC+8;
325                                                if(((R.rawDoors()[d]!=null) // important optimization
326                                                &&(R.rawDoors()[d].roomID().length()>0)))
327                                                {
328                                                        final Exit E=R.getExitInDir(d);
329                                                        if((R.getRoomInDir(d)!=null)
330                                                        &&(E!=null)
331                                                        &&((E.hasADoor())
332                                                        &&(!E.isOpen()))
333                                                        &&(!elligible.contains(E))
334                                                        &&((R.getReverseExit(d)==null)
335                                                           ||(!elligible.contains(R.getReverseExit(d)))))
336                                                        {
337                                                                tickStatus=Tickable.STATUS_MISC+9;
338                                                                if(E.hasALock()&&(E.isLocked())&&(doAnyLockedDoors))
339                                                                        elligible.add(E);
340                                                                else
341                                                                if(doAnyDoors)
342                                                                        elligible.add(E);
343                                                        }
344                                                        tickStatus=Tickable.STATUS_MISC+10;
345                                                }
346                                        }
347
348                                        tickStatus=Tickable.STATUS_MISC+21;
349                                        if((doAnyItems)||(doAnyContainers)||(doDooredContainers)||(doLockedContainers))
350                                        for(int i=0;i<R.numItems();i++)
351                                        {
352                                                tickStatus=Tickable.STATUS_MISC+22;
353                                                final Item I=R.getItem(i);
354                                                if((CMLib.flags().isGettable(I))
355                                                &&(!elligible.contains(I))
356                                                &&(!I.ID().endsWith("Wallpaper")))
357                                                {
358                                                        tickStatus=Tickable.STATUS_MISC+23;
359                                                        if(I instanceof Container)
360                                                        {
361                                                                tickStatus=Tickable.STATUS_MISC+24;
362                                                                final Container C=(Container)I;
363                                                                if(C.hasADoor()&&(!C.isOpen()))
364                                                                {
365                                                                        if(C.hasALock()&&C.isLocked()&&(doLockedContainers))
366                                                                                elligible.add(I);
367                                                                        else
368                                                                        if(doDooredContainers)
369                                                                                elligible.add(I);
370                                                                }
371                                                                else
372                                                                if(doAnyContainers)
373                                                                        elligible.add(I);
374                                                        }
375                                                        else
376                                                        if(doAnyItems)
377                                                                elligible.add(I);
378                                                }
379                                                tickStatus=Tickable.STATUS_MISC+25;
380                                        }
381                                        tickStatus=Tickable.STATUS_MISC+26;
382                                }
383
384                                tickStatus=Tickable.STATUS_MISC+27;
385                                if(!doRooms)
386                                while((elligible.size()>0)&&(elligible.get(0) instanceof Room))
387                                        elligible.remove(0);
388
389                                tickStatus=Tickable.STATUS_MISC+28;
390                                for(int e=elligible.size()-1;e>=0;e--)
391                                {
392                                        tickStatus=Tickable.STATUS_MISC+29;
393                                        if((maintained.contains(elligible.get(e)))
394                                        ||(CMLib.utensils().fetchMyTrap(elligible.get(e))!=null))
395                                                elligible.remove(e);
396                                        tickStatus=Tickable.STATUS_MISC+30;
397                                }
398
399                                if(elligible.size()==0)
400                                        return true;
401
402                                tickStatus=Tickable.STATUS_MISC+31;
403                                final Physical P=elligible.get(CMLib.dice().roll(1,elligible.size(),-1));
404
405                                final List<Trap> elligibleTraps=new Vector<Trap>();
406                                for(int t=0;t<allTraps.size();t++)
407                                {
408                                        if(allTraps.get(t).canSetTrapOn(null,P)
409                                        &&(P.fetchEffect(allTraps.get(t).ID())==null))
410                                                elligibleTraps.add(allTraps.get(t));
411                                }
412
413                                if(elligibleTraps.size()==0)
414                                        return true;
415
416                                tickStatus=Tickable.STATUS_MISC+32;
417                                Trap T=elligibleTraps.get(CMLib.dice().roll(1,elligibleTraps.size(),-1));
418                                T=(Trap)T.copyOf();
419                                T.setProficiency(100);
420                                T.makeLongLasting();
421                                T.setSavable(false);
422                                P.addEffect(T);
423                                maintained.addElement(P);
424                                tickStatus=Tickable.STATUS_MISC+33;
425                        }
426                }
427                tickStatus=Tickable.STATUS_NOT;
428                return true;
429        }
