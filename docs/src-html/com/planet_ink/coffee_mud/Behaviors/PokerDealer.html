246        // checks the pot for the highest bid and returns it
247        private double amountToCall()
248        {
249                double amountToCall=-1;
250                for(int p=0;p<pot.size();p++)
251                {
252                        final Double potAmount=(Double)pot.elementAt(p,2);
253                        if(potAmount.doubleValue()>amountToCall)
254                                amountToCall=potAmount.doubleValue();
255                }
256                return amountToCall;
257        }

259        // checks whether the pot is currently in balance.
260        private boolean isThePotRight()
261        {
262                if(pot.size()<2)
263                        return true;
264                double amountToCall=-1;
265                for(int p=0;p<pot.size();p++)
266                {
267                        final Double potAmount=(Double)pot.elementAt(p,2);
268                        if(amountToCall<0.0)
269                                amountToCall=potAmount.doubleValue();
270                        if(potAmount.doubleValue()!=amountToCall)
271                                return false;
272                }
273                return true;
274        }

276        // returns 0 if the player called a bet
277        // return 1 if the player raised the bet
278        // returns -1 if the player is below the bet
279        public int getCalled0Raised1OrFolded(MOB player)
280        {
281                if(!pot.contains(player))
282                        return -1;
283                final Double inPot=(Double)pot.elementAt(pot.indexOf(player),2);
284                int numHigherThanPlayer=0;
285                int numEqualToPlayer=0;
286                for(int p=0;p<pot.size();p++)
287                {
288                        if(pot.elementAt(p,1)!=player)
289                        {
290                                final Double potAmount=(Double)pot.elementAt(p,2);
291                                if(potAmount.doubleValue()==inPot.doubleValue())
292                                        numEqualToPlayer++;
293                                else
294                                if(potAmount.doubleValue()>inPot.doubleValue())
295                                        numHigherThanPlayer++;
296                        }
297                }
298                if(numHigherThanPlayer>0)
299                        return -1;
300                if(numEqualToPlayer>0)
301                        return 0;
302                return 1;
303        }

305        // general system event previewer.
306        // when events in the same room as this behavior occur,
307        // the messages are sent here before they happen.  the
308        // "host" will be the object that is hosting this behavior.
309        // The "msg" will be the details of the event which has occurred.
310        // This method will return true if its ok to proceed, and false otherwise.
311        @Override
312        public boolean  okMessage(Environmental host, CMMsg msg)
313        {
314                if(!super.okMessage(host,msg))
315                        return false;
316
317                // if someone tries to pick money up off the ground here
318                // they need to be stopped COLD.
319                if(((msg.targetMinor()==CMMsg.TYP_GET)||(msg.targetMinor()==CMMsg.TYP_PUSH)||(msg.targetMinor()==CMMsg.TYP_PULL))
320                &&(msg.source().location()==CMLib.map().roomLocation(host))
321                &&(msg.target() instanceof Coins)
322                &&(!msg.targetMajor(CMMsg.MASK_INTERMSG))
323                &&(msg.source().location().isContent((Coins)msg.target())))
324                {
325                        msg.source().tell(L("No touching the pot!"));
326                        return false;
327                }
328
329                // if someone drops some money on the ground, it
330                // may be a bet, or an anti, or an invalid action
331                // we should check for that first.
332                if((msg.targetMinor()==CMMsg.TYP_DROP)
333                &&(msg.source().location()==CMLib.map().roomLocation(host))
334                &&(msg.target() instanceof Coins)
335                &&(!msg.targetMajor(CMMsg.MASK_INTERMSG)))
336                {
337                        final Double inPot=pot.contains(msg.source())?(Double)pot.elementAt(pot.indexOf(msg.source()),2):null;
338                        final String currency=CMLib.beanCounter().getCurrency(host);
339                        final MOB playerDroppingMoney=msg.source();
340                        final Coins theMoneyDropped=(Coins)msg.target();
341
342                        // if they put down foreign currency, tell them its
343                        // wrong, then abort their attempt to drop it into
344                        // the pot.
345                        if(!currency.equals(theMoneyDropped.getCurrency()))
346                        {
347                                playerDroppingMoney.tell(L("That is not the proper currency.  This table is only dealing in @x1.",CMLib.beanCounter().getDenominationName(currency)));
348                                return false;
349                        }
350
351                        // game hasn't started yet, so only antis allowed
352                        // in this case.
353                        if((gameState&STATE_MASK)==STATE_WAITING_FOR_ANTIS)
354                        {
355                                if(timer<=0)
356                                        timer=System.currentTimeMillis()+(TIME_SECONDSTOSTART*1000);
357                                // if they havn't antied yet
358                                if(inPot==null)
359                                {
360                                        // if they havn't antied enough, abort their drop attempt.
361                                        if(theMoneyDropped.getTotalValue()<anti)
362                                        {
363                                                msg.source().tell(L("Thats not enough.  The anti is @x1.",antiAmount(host)));
364                                                return false;
365                                        }
366
367                                        // if they give too much, queue up a change making message.
368                                        if(theMoneyDropped.getTotalValue()>anti)
369                                        {
370                                                final double change=theMoneyDropped.getTotalValue()-anti;
371                                                Coins C=CMLib.beanCounter().makeBestCurrency(currency,change);
372                                                final CMMsg changeMsg=CMClass.getMsg(playerDroppingMoney,C,null,
373                                                                                                          CMMsg.MASK_ALWAYS|CMMsg.MSG_GET,L("You anti up, picking up @x1 in change.",CMLib.beanCounter().abbreviatedPrice(currency,change)),
374                                                                                                          CMMsg.MASK_ALWAYS|CMMsg.MSG_GET,null,
375                                                                                                          CMMsg.MASK_ALWAYS|CMMsg.MSG_GET,L("<S-NAME> antis up."));
376                                                msg.addTrailerMsg(changeMsg);
377                                                // now we want to clear the drop message,
378                                                // and modify the amount actually being dropped.
379                                                C=CMLib.beanCounter().makeBestCurrency(currency,anti);
380                                                theMoneyDropped.setNumberOfCoins(C.getNumberOfCoins());
381                                                theMoneyDropped.setDenomination(C.getDenomination());
382                                                msg.modify(msg.source(),msg.target(),msg.tool(),msg.sourceCode(),null,msg.targetCode(),null,msg.othersCode(),null);
383                                        }
384                                }
385                                else
386                                {
387                                        // if they've already antied, we abort this.
388                                        playerDroppingMoney.tell(L("You have already antied, and can not anti again."));
389                                        return false;
390                                }
391                        }
392                        else
393                        if(((gameState&STATE_MASK)!=STATE_FIRST_BETTER)
394                        &&((gameState&STATE_MASK)!=STATE_NEXT_BETTER))
395                        {
396                                // now, if the game is already started, but its not a betting round,
397                                // we reject their attempt to bet.
398                                playerDroppingMoney.tell(L("You can not bet at this time."));
399                                // abort the message
400                                return false;
401                        }
402                        else
403                        if(whoseTurn!=playerDroppingMoney)
404                        {
405                                // now, if the game is already started, we allow them only to bet on their turn.
406                                // if it is not their turn to bet, we abort.
407                                if(whoseTurn==null)
408                                        playerDroppingMoney.tell(L("You can not bet at this time.  Please wait your turn."));
409                                else
410                                        playerDroppingMoney.tell(L("It is @x1's turn to bet right now.  Please wait for the dealer to call on you.",whoseTurn.name()));
411                                // abort the message
412                                return false;
413                        }
414                        else
415                        {
416                                // first we determine the amount to call.
417                                final double amountToCall=amountToCall();
418                                // now, we check if they have called yet
419                                if((inPot==null)||(inPot.doubleValue()<amountToCall))
420                                {
421                                        final double amountInPot=(inPot==null)?0.0:inPot.doubleValue();
422                                        final double amountNeededToCall=amountToCall-amountInPot;
423                                        final String instructions="  You may now raise by dropping more money, or end your betting round by saying 'call'.";
424
425                                        // if they havn't given enough to call, allow it, but let them know.
426                                        if(theMoneyDropped.getTotalValue()<amountNeededToCall)
427                                                msg.source().tell(L("That won't be enough.  You'll still need to drop @x1 more to call.",CMLib.beanCounter().abbreviatedPrice(currency,amountNeededToCall-theMoneyDropped.getTotalValue())));
428                                        else
429                                        // if they give too much, make change
430                                        if(theMoneyDropped.getTotalValue()>amountNeededToCall)
431                                        {
432                                                final double change=theMoneyDropped.getTotalValue()-amountNeededToCall;
433                                                Coins C=CMLib.beanCounter().makeBestCurrency(currency,change);
434                                                final CMMsg changeMsg=CMClass.getMsg(playerDroppingMoney,C,null,
435                                                                                                          CMMsg.MASK_ALWAYS|CMMsg.MSG_GET,L("You see the bet, picking up @x1 in change.@x2",CMLib.beanCounter().abbreviatedPrice(currency,change),instructions),
436                                                                                                          CMMsg.MASK_ALWAYS|CMMsg.MSG_GET,null,
437                                                                                                          CMMsg.MASK_ALWAYS|CMMsg.MSG_GET,L("<S-NAME> see(s) the bet."));
438                                                msg.addTrailerMsg(changeMsg);
439                                                // now we want to clear the drop message,
440                                                // and modify the amount actually being dropped.
441                                                C=CMLib.beanCounter().makeBestCurrency(currency,anti);
442                                                theMoneyDropped.setNumberOfCoins(C.getNumberOfCoins());
443                                                theMoneyDropped.setDenomination(C.getDenomination());
444                                                msg.modify(msg.source(),msg.target(),msg.tool(),msg.sourceCode(),null,msg.targetCode(),null,msg.othersCode(),null);
445                                        }
446                                        else
447                                        // if they gave the perfect amount
448                                        {
449                                                //  we just want to change the drop message to our gambler prose.
450                                                final String newMessage="<S-NAME> see(s) the bet.";
451                                                msg.modify(msg.source(),msg.target(),msg.tool(),msg.sourceCode(),newMessage+instructions,msg.targetCode(),newMessage,msg.othersCode(),newMessage);
452                                        }
453                                }
454                                else
455                                {
456                                        // we enforce table stakes, so make the this player is not
457                                        // betting the other players out of the game.
458                                        final double totalDown=amountToCall+theMoneyDropped.getTotalValue();
459                                        for(int p=0;p<pot.size();p++)
460                                        {
461                                                if(pot.elementAt(p,1)!=playerDroppingMoney)
462                                                {
463                                                        final MOB mob =(MOB)pot.elementAt(p,1);
464                                                        if(CMLib.beanCounter().getTotalAbsoluteValue(mob,currency)<totalDown)
465                                                        {
466                                                                msg.source().tell(L("You may only bet up to @x1 due to the table-stakes rule.",CMLib.beanCounter().abbreviatedPrice(currency,CMLib.beanCounter().getTotalAbsoluteValue(mob,currency))));
467                                                                return false;
468                                                        }
469                                                }
470
471                                        }
472                                        // if they are raising, we just change the drop message
473                                        // to our fancy gamblers prose.
474                                        final String newMessage="<S-NAME> raise(s) <T-NAMESELF>.";
475                                        final String instructions="  You may continue raising by dropping more money.  Or say 'raise' to end your betting.";
476                                        msg.modify(msg.source(),msg.target(),msg.tool(),msg.sourceCode(),newMessage+instructions,msg.targetCode(),newMessage,msg.othersCode(),newMessage);
477                                }
478                        }
479                }
480                return true;
481        }

483        // this method will handle the betting and drawing rounds
484        // by changing whose turn it is and, if necessary,
485        // changing the game state.
486        private void nextPlayerNextState(Environmental host)
487        {
488                if(pot.size()<=0)
489                {
490                        whoseTurn=null;
491                        endTheGame(host);
492                        gameState=STATE_WAITING_FOR_ANTIS|STATEMASK_NEED_ANOUNCEMENT;
493                        return;
494                }
495                MOB nextPlayer=(MOB)pot.elementAt(0,1);
496                if(whoseTurn!=null)
497                for(int p=0;p<pot.size()-1;p++)
498                {
499                        if(pot.elementAt(p,1)==whoseTurn)
500                        {
501                                nextPlayer=(MOB)pot.elementAt(p+1,1);
502                                break;
503                        }
504                }
505                whoseTurn=nextPlayer;
506
507                // if we still have more players to consider...
508                switch(gameState)
509                {
510                case STATE_DEALING:
511                case STATE_DONE_DRAWING:
512                        gameState=STATE_FIRST_BETTER|STATEMASK_NEED_ANOUNCEMENT;
513                        break;
514                case STATE_FIRST_BETTER:
515                        if((isThePotRight())&&(whoseTurn==pot.elementAt(0,1)))
516                                gameState=STATE_DONE_BETTING|STATEMASK_NEED_ANOUNCEMENT;
517                        else
518                        if(isThePotRight())
519                                gameState=STATE_FIRST_BETTER|STATEMASK_NEED_ANOUNCEMENT;
520                        else
521                                gameState=STATE_NEXT_BETTER|STATEMASK_NEED_ANOUNCEMENT;
522                        break;
523                case STATE_NEXT_BETTER:
524                        if(isThePotRight())
525                                gameState=STATE_DONE_BETTING|STATEMASK_NEED_ANOUNCEMENT;
526                        else
527                                gameState=STATE_NEXT_BETTER|STATEMASK_NEED_ANOUNCEMENT;
528                        break;
529                case STATE_FIRST_DRAW:
530                        gameState=STATE_NEXT_DRAW|STATEMASK_NEED_ANOUNCEMENT;
531                        break;
532                case STATE_NEXT_DRAW:
533                        if(whoseTurn==pot.elementAt(0,1))
534                                gameState=STATE_DONE_DRAWING|STATEMASK_NEED_ANOUNCEMENT;
535                        else
536                                gameState=STATE_NEXT_DRAW|STATEMASK_NEED_ANOUNCEMENT;
537                        break;
538                }
539        }

541        // general system event handler.
542        // when events in the same room as this behavior occur,
543        // the messages are sent here.  the "host" will be the
544        // object that is hosting this behavior.  The "msg" will
545        // be the details of the event which has occurred.
546        @Override
547        public void executeMsg(final Environmental host, final CMMsg msg)
548        {
549                // if someone enters the same room where this
550                // behavior is operating, and a game is NOT
551                // in progress, they should get a greeting!
552                if((msg.targetMinor()==CMMsg.TYP_LOOK)
553                &&(msg.othersMessage()==null)
554                &&(msg.target() instanceof Room)

816        // The heart of poker.
817        // it determines the best hand held.
818        // it returns a numeric score for the hand
819        // the better the hand, the higher the score
820        // It can handle poker of 1 or more cards.
821        private int determineHand(HandOfCards hand)
822        {
823                List<Item> cards=hand.getContents();
824                if(cards.size()==0)
825                        return -1;
826
827                // first check for flushes
828                int flushSuit=-1;
829                for (final int suit : PlayingCard.suits)
830                {
831                        int tempSuitCount=0;
832                        int tempFlushSuit=0;
833                        for(int c=0;c<cards.size();c++)
834                        {
835                                if(((PlayingCard)cards.get(c)).getBitEncodedSuit()==suit)
836                                {
837                                        tempSuitCount++;
838                                        tempFlushSuit=((PlayingCard)cards.get(c)).getBitEncodedSuit();
839                                }
840                        }
841                        if(tempSuitCount>=5)
842                        {
843                                flushSuit=tempFlushSuit;
844                                break;
845                        }
846                }
847
848                // then for straights with ace high
849                int highStraightCard=-1;
850                if(cards.size()>=5)
851                for(int window=cards.size()-5;window>=0;window--)
852                {
853                        highStraightCard=((PlayingCard)cards.get(window+4)).getBitEncodedValue();
854                        if(highStraightCard==1)
855                                highStraightCard=14;
856                        for(int c=window+1;c<window+5;c++)
857                        {
858                                int cardBitValue=((PlayingCard)cards.get(c)).getBitEncodedValue();
859                                if(cardBitValue==1)
860                                        cardBitValue=14;
861                                if(cardBitValue!=(((PlayingCard)cards.get(c-1)).getBitEncodedValue()+1))
862                                        highStraightCard=-1;
863                        }
864                        if(highStraightCard>=0)
865                                break;
866                }
867                // then for straights with ace low
868                if((highStraightCard<0)&&(cards.size()>=5))
869                {
870                        hand.sortByValueAceLow();
871                        cards=hand.getContents();
872                        for(int window=cards.size()-5;window>=0;window--)
873                        {
874                                highStraightCard=((PlayingCard)cards.get(window+4)).getBitEncodedValue();
875                                for(int c=window+1;c<window+5;c++)
876                                {
877                                        if(((PlayingCard)cards.get(c)).getBitEncodedValue()!=(((PlayingCard)cards.get(c-1)).getBitEncodedValue()+1))
878                                                highStraightCard=-1;
879                                }
880                                if(highStraightCard>=0)
881                                        break;
882                        }
883                        if(highStraightCard<0)
884                        {
885                                hand.sortByValueAceHigh();
886                                cards=hand.getContents();
887                        }
888                }
889
890                // now build a list of matches (pairs, etc)
891                // put them in a set keyed by the face value of the
892                // card
893                DVector matches=new DVector(2);
894                for(int c=1;c<cards.size();c++)
895                {
896                        final Integer value=Integer.valueOf(((PlayingCard)cards.get(c)).getBitEncodedValue());
897                        if(value.intValue()==(((PlayingCard)cards.get(c-1)).getBitEncodedValue()))
898                        {
899                                final int index=matches.indexOf(value);
900                                if(index>=0)
901                                        matches.setElementAt(index,2,Integer.valueOf(1+((Integer)matches.elementAt(index,2)).intValue()));
902                                else
903                                        matches.addElement(value,Integer.valueOf(2));
904                        }
905                }
906
907                // then sort our matches so the MOST matches are highest
908                final DVector sortedMatches=new DVector(2);
909                while(matches.size()>0)
910                {
911                        int num=0;
912                        for(int i=1;i<matches.size();i++)
913                        {
914                                if(((Integer)matches.elementAt(i,2)).intValue()<=((Integer)matches.elementAt(num,2)).intValue())
915                                        num=i;
916                        }
917                        sortedMatches.addElement(matches.elementAt(num,1),matches.elementAt(num,2));
918                        matches.removeElementAt(num);
919                }
920                matches=sortedMatches;
921
922                // a match of 5 cards means 5 of a kind!
923                if((matches.size()>0)
924                &&(((Integer)matches.elementAt(matches.size()-1,2)).intValue()==5))
925                        return HAND_5OFAKIND+((Integer)matches.elementAt(matches.size()-1,1)).intValue();
926
927                // check for straight flush
928                if((highStraightCard==14)
929                &&(flushSuit>0)
930                &&hand.containsCard(flushSuit|10)
931                &&hand.containsCard(flushSuit|11)
932                &&hand.containsCard(flushSuit|12)
933                &&hand.containsCard(flushSuit|13)
934                &&hand.containsCard(flushSuit|1))
935                        return HAND_ROYALFLUSH+flushSuit;
936
937                // check for straight flush
938                if((highStraightCard>=0)
939                &&(flushSuit>0)
940                &&(hand.containsCard(flushSuit|highStraightCard))
941                &&(hand.containsCard(flushSuit|highStraightCard-1))
942                &&(hand.containsCard(flushSuit|highStraightCard-2))
943                &&(hand.containsCard(flushSuit|highStraightCard-3))
944                &&(hand.containsCard(flushSuit|highStraightCard-4)))
945                        return HAND_STRAIGHTFLUSH+(highStraightCard|flushSuit);
946
947                // check for 4 of a kind
948                if((matches.size()>0)
949                &&(((Integer)matches.elementAt(matches.size()-1,2)).intValue()==4))
950                {
951                        for(int c=cards.size()-1;c>=0;c--)
952                        {
953                                if(((PlayingCard)cards.get(c)).getBitEncodedValue()!=((Integer)matches.elementAt(matches.size()-1,1)).intValue())
954                                        return HAND_4OFAKIND+(((Integer)matches.elementAt(matches.size()-1,1)).intValue()<<15)+((PlayingCard)cards.get(c)).getBitEncodedValue();
955                        }
956                        return HAND_4OFAKIND+(((Integer)matches.elementAt(matches.size()-1,1)).intValue()<<15);
957                }
958
959                // check for a full house
960                if((matches.size()>1)
961                &&(((Integer)matches.elementAt(matches.size()-1,2)).intValue()>2)
962                &&(((Integer)matches.elementAt(matches.size()-2,2)).intValue()>1))
963                        return HAND_FULLHOUSE+(((Integer)matches.elementAt(matches.size()-1,1)).intValue()<<15)+((Integer)matches.elementAt(matches.size()-2,1)).intValue();
964
965                // check for a flush
966                if(flushSuit>0)
967                {
968                        int addValue=0;
969                        int addTimes=0;
970                        for(int c=cards.size()-1;c>=0;c--)
971                        {
972                                if(((PlayingCard)cards.get(c)).getBitEncodedSuit()==flushSuit)
973                                {
974                                        if((++addTimes)<4)
975                                                addValue=(addValue<<4)+((PlayingCard)cards.get(c)).getBitEncodedValue();
976                                }
977                        }
978                        return HAND_FLUSH+(flushSuit<<13)+addValue;
979                }
980
981                // check for a straight
982                if(highStraightCard>=0)
983                        return HAND_STRAIGHT+highStraightCard;
984
985                // check for 3 of a kind
986                if((matches.size()>0)
987                &&(((Integer)matches.elementAt(matches.size()-1,2)).intValue()==3))
988                {
989                        final int value=((Integer)matches.elementAt(matches.size()-1,1)).intValue();
990                        int addValue=0;
991                        int addTimes=0;
992                        for(int c=cards.size()-1;c>=0;c--)
993                        {
994                                if(((PlayingCard)cards.get(c)).getBitEncodedValue()!=value)
995                                {
996                                        if((++addTimes)<4)
997                                                addValue=(addValue<<4)+((PlayingCard)cards.get(c)).getBitEncodedValue();
998                                }
999                        }
1000                        return HAND_3OFAKIND+(value<<15)+addValue;
1001                }
1002
1003                // check for 2 pair
1004                if((matches.size()>1)
1005                &&(((Integer)matches.elementAt(matches.size()-1,2)).intValue()>1)
1006                &&(((Integer)matches.elementAt(matches.size()-2,2)).intValue()>1))
1007                {
1008                        final int value=((Integer)matches.elementAt(matches.size()-1,1)).intValue();
1009                        final int value1=((Integer)matches.elementAt(matches.size()-2,1)).intValue();
1010                        int addValue=0;
1011                        int addTimes=0;
1012                        for(int c=cards.size()-1;c>=0;c--)
1013                                if((((PlayingCard)cards.get(c)).getBitEncodedValue()!=value)
1014                                &&(((PlayingCard)cards.get(c)).getBitEncodedValue()!=value1))
1015                                {
1016                                        if((++addTimes)<2)
1017                                                addValue=(addValue<<4)+((PlayingCard)cards.get(c)).getBitEncodedValue();
1018                                }
1019                        return HAND_2PAIR+(value<<15)+(value1<<11)+addValue;
1020                }
1021
1022                // check for a pair
1023                if((matches.size()>0)
1024                &&(((Integer)matches.elementAt(matches.size()-1,2)).intValue()>1))
1025                {
1026                        final int value=((Integer)matches.elementAt(matches.size()-1,1)).intValue();
1027                        int addValue=0;
1028                        int addTimes=0;
1029                        for(int c=cards.size()-1;c>=0;c--)
1030                        {
1031                                if(((PlayingCard)cards.get(c)).getBitEncodedValue()!=value)
1032                                {
1033                                        if((++addTimes)<4)
1034                                                addValue=(addValue<<4)+((PlayingCard)cards.get(c)).getBitEncodedValue();
1035                                }
1036                        }
1037                        return HAND_1PAIR+(value<<15)+addValue;
1038                }
1039
1040                // no good hands, so build a high-card score
1041                int addValue=0;
1042                int addTimes=0;
1043                for(int c=cards.size()-1;c>=0;c--)
1044                {
1045                        if((++addTimes)<5)
1046                                addValue=(addValue<<4)+((PlayingCard)cards.get(c)).getBitEncodedValue();
1047                }
1048                return HAND_HIGHCARD+addValue;
1049        }

1051        // this method calls the determineHand
1052        // method repeatedly for all players,
1053        // determining the score for their hands.
1054        // It then passes back a 2 dimensional
1055        // DVector with the player mob object and
1056        // the score.  It is sorted by score.
1057        // since stud starts betting with high SHOWN cards,
1058        // we need a way to specify that only the face up
1059        // cards count for scoring purposes.
1060        private DVector determineAllSortedScores(boolean faceUpOnly)
1061        {
1062                final DVector unsortedScores=new DVector(2);
1063                for(int p=0;p<pot.size();p++)
1064                {
1065                        final MOB mob=(MOB)pot.elementAt(p,1);
1066                        HandOfCards hand=theDeck().getPlayerHand(mob);
1067                        if(hand==null)
1068                                continue;
1069
1070                        // if only the face cards count, we will pull them out,
1071                        // put them in their own hand, generate a score for them
1072                        // and use their score
1073                        if(faceUpOnly)
1074                        {
1075                                final HandOfCards faceUpHand=((HandOfCards)CMClass.getMiscMagic("StdHandOfCards")).createEmptyHand(null);
1076                                final List<Item> handContents=hand.getContents();
1077                                for(int h=0;h<handContents.size();h++)
1078                                {
1079                                        PlayingCard card=(PlayingCard)handContents.get(h);
1080                                        if(card.isFaceUp())
1081                                        {
1082                                                card=(PlayingCard)card.copyOf();
1083                                                faceUpHand.addCard(card);
1084                                        }
1085                                }
1086                                hand=faceUpHand;
1087                        }
1088                        final int score=determineHand(hand);
1089                        if(score<0)
1090                                continue;
1091                        unsortedScores.addElement(mob,Integer.valueOf(score));
1092                }
1093
1094                // now we have the scores, so sort them
1095                final DVector sortedScores=new DVector(2);
1096                while(unsortedScores.size()>0)
1097                {

1100                        {
1101                                if(((Integer)unsortedScores.elementAt(i,2)).intValue()<=((Integer)unsortedScores.elementAt(topNum,2)).intValue())
1102                                        topNum=i;

1115                final PlayingCard PC=(PlayingCard)CMClass.getItem("StdPlayingCard");

1117                {
1118                        score-=HAND_5OFAKIND;

1120                }
1121                if((score&HAND_MASK)==HAND_ROYALFLUSH)
1122                {
1123                        score-=HAND_ROYALFLUSH;
1124                        return "a Royal Flush of "+PC.getSuitDescription(score);
1125                }
1126                if((score&HAND_MASK)==HAND_STRAIGHTFLUSH)
1127                {
1128                        score-=HAND_STRAIGHTFLUSH;
1129                        return "a Straight Flush of "+PC.getSuitDescription(score)+", "+PC.getCardValueLongDescription(score)+" high";
1130                }
1131                if((score&HAND_MASK)==HAND_4OFAKIND)
1132                {
1133                        score-=HAND_4OFAKIND;
1134                        int base=(score&((1+2+4+8)<<15));
1135                        score=score-base;
1136                        base=base>>15;

1138                }

1140                {

1144                        base=base>>15;

1146                }
1147                if((score&HAND_MASK)==HAND_FLUSH)
1148                {
1149                        score-=HAND_FLUSH;

1165                {

1170                {
1171                        score-=HAND_3OFAKIND;

1176                        for(int i=0;i<4;i++)
1177                        {
1178                                final int value=(score&(1+2+4+8));

1188                }
1189                if((score&HAND_MASK)==HAND_2PAIR)

1197                        combo2=combo2>>11;

1199                        for(int i=0;i<2;i++)
1200                        {
1201                                final int value=(score&(1+2+4+8));
1202                                score-=value;
1203                                score=score>>4;
1204                                if(value>0)
1205                                        str.append(PC.getCardValueLongDescription(value)+" ");
1206                        }
1207                        final String hand="Two Pair, "+PC.getCardValueLongDescription(combo1)+"s and "+PC.getCardValueLongDescription(combo2)+"s";
1208                        if(str.length()>0)
1209                                return  hand+", with "+str.toString().trim()+" high";
1210                        return hand;
1211                }
1212                if((score&HAND_MASK)==HAND_1PAIR)
1213                {
1214                        score-=HAND_1PAIR;
1215                        int combo=(score&((1+2+4+8)<<15));
1216                        score=score-combo;
1217                        combo=combo>>15;
1218                        final StringBuffer str=new StringBuffer("");
1219                        for(int i=0;i<4;i++)
1220                        {
1221                                final int value=(score&(1+2+4+8));
1222                                score-=value;
1223                                score=score>>4;

1228                        if(str.length()>0)

1238                                final int value=(score&(1+2+4+8));

1410        // this method is called by the service engine
1411        // every 3-4 seconds.  It is used to initiate
1412        // timing events, and to initiate non-reactionary
1413        // events such as the dealers play.  What the
1414        // method returns doesn't really matter.  The
1415        // "ticking" variable refers to the object hosting
1416        // this behavior, and tickID is seldom important
1417        // but refers to the reason for the timed event.
1418        @Override
1419        public boolean tick(final Tickable ticking, final int tickID)
1420        {
1421                if(!super.tick(ticking,tickID))
1422                        return false;
1423
1424                final Environmental host=(Environmental)ticking;
1425                final Room R=CMLib.map().roomLocation(host);
1426
1427                // first see if we lost any players
1428                // due to leaving the room or the game.
1429                for(int p=pot.size()-1;p>=0;p--)
1430                {
1431                        final MOB mob=(MOB)pot.elementAt(p,1);
1432                        if(!R.isInhabitant(mob))
1433                        {
1434                                communicate(host,null,"Oops.. we lost "+mob.name()+".",null);
1435                                theDeck().removePlayerHand(mob);
1436                                pot.removeElement(mob);
1437                        }
1438                }
1439
1440                // now, if we are in a waiting state,
1441                // but the number of players has dropped below
1442                // minimum, we end the game
1443                switch((gameState)&STATE_MASK)
1444                {
1445                case STATE_FIRST_DRAW:
1446                case STATE_NEXT_DRAW:
1447                case STATE_FIRST_BETTER:
1448                case STATE_NEXT_BETTER:
1449                case STATE_DEALING:
1450                case STATE_DONE_DRAWING:
1451                case STATE_DONE_BETTING:
1452                {
1453                        if(pot.size()==0)
1454                        {
1455                                communicate(host,null,"Looks like everyone's out.  Time to start over.",null);
1456                                endTheGame(host);
1457                                return true;
1458                        }
1459                        if(pot.size()==1)
1460                        {
1461                                announceWinners(host);
1462                                endTheGame(host);
1463                                return true;
1464                        }
1465                        break;
1466                }
1467                }
1468
1469                // next lets check for announcements
1470                // that are flagged to occur.  This often
1471                // causes a timer to be set giving the player
1472                // a short time to respond.
1473                if((gameState&STATEMASK_NEED_ANOUNCEMENT)>0)
1474                {
1475                        gameState=gameState-STATEMASK_NEED_ANOUNCEMENT;
1476                        switch((gameState)&STATE_MASK)
1477                        {
1478                        case STATE_DEALING:
1479                        case STATE_DONE_DRAWING:
1480                                communicate(host,null,"Ok, everyone has their cards.  Take a few seconds to look them over, and then we'll start betting.",null);
1481                                timer=System.currentTimeMillis()+(TIME_SECONDSTOFIRSTBET*1000);
1482                                break;
1483                        case STATE_FIRST_BETTER:
1484                                if(whoseTurn!=null)
1485                                        communicate(host,whoseTurn,"Ok, "+whoseTurn.name()+", you may bet by dropping money, or you may say 'pass'.",null);
1486                                communicate(host,whoseTurn,"You have "+TIME_SECONDSTOBET+" seconds.",null);
1487                                timer=System.currentTimeMillis()+(TIME_SECONDSTOBET*1000);
1488                                break;
1489                        case STATE_FIRST_DRAW:
1490                        case STATE_NEXT_DRAW:
1491                                communicate(host,whoseTurn,"Ok, "+whoseTurn.name()+", you may request a draw by saying which cards you want dropped.  For example, say '1 3 5' to drop your first, third, and fifth card for a re-draw.  Say 'stand' if you do not want to draw.",null);
1492                                communicate(host,whoseTurn,"You have "+TIME_SECONDSTODRAW+" seconds.",null);
1493                                timer=System.currentTimeMillis()+(TIME_SECONDSTODRAW*1000);
1494                                break;
1495                        case STATE_NEXT_BETTER:
1496                        {
1497                                final Double inPot=pot.contains(whoseTurn)?(Double)pot.elementAt(pot.indexOf(whoseTurn),2):null;
1498                                final double amountToCall=amountToCall();
1499                                final double amountInPot=(inPot==null)?0.0:inPot.doubleValue();
1500                                final double amountNeededToCall=amountToCall-amountInPot;
1501                                final String currency=CMLib.beanCounter().getCurrency(host);
1502                                if(amountNeededToCall==0.0)
1503                                        communicate(host,whoseTurn,"Ok, "+whoseTurn.name()+", you may open betting by dropping money, or you may say 'pass'.",null);
1504                                else
1505                                        communicate(host,whoseTurn,"Ok, "+whoseTurn.name()+", you may call the bet of "+CMLib.beanCounter().abbreviatedPrice(currency,amountNeededToCall)+" by dropping money, or you may say 'fold'.",null);
1506                                communicate(host,whoseTurn,"You have "+TIME_SECONDSTOBET+" seconds.",null);
1507                                timer=System.currentTimeMillis()+(TIME_SECONDSTOBET*1000);
1508                                break;
1509                        }
1510                        case STATE_DONE_BETTING:
1511                        {
1512                                // after betting, the next step
1513                                // depends on the game being played and
1514                                // what round of play it is.  Sometimes
1515                                // the game is over after betting, sometimes
1516                                // more cards are dealt, and sometimes cards
1517                                // are discarded and drawn.
1518                                timer=0;
1519                                whoseTurn=null;
1520                                switch(gameRules)
1521                                {
1522                                case PokerDealer.GAME_STRAIGHTPOKER:
1523                                        announceWinners((Environmental)ticking);
1524                                        endTheGame((Environmental)ticking);
1525                                        break;
1526                                case PokerDealer.GAME_DRAWPOKER:
1527                                        switch(roundOfPlay)
1528                                        {
1529                                        case 0:
1530                                                roundOfPlay++;
1531                                                gameState=STATE_FIRST_DRAW|STATEMASK_NEED_ANOUNCEMENT;
1532                                                timer=0;
1533                                                nextPlayerNextState(host);
1534                                                break;
1535                                        case 1:
1536                                                announceWinners((Environmental)ticking);
1537                                                endTheGame((Environmental)ticking);
1538                                                break;
1539                                        }
1540                                        break;
1541                                case PokerDealer.GAME_5CARDSTUD:
1542                                        switch(roundOfPlay)
1543                                        {
1544                                        case 0:
1545                                        case 1:
1546                                                communicate(host,null,"The next card is up.",null);
1547                                                dealToAll(host,1,1);
1548                                                roundOfPlay++;
1549                                                gameState=STATE_DEALING|STATEMASK_NEED_ANOUNCEMENT;
1550                                                timer=0;
1551                                                break;
1552                                        case 2:
1553                                                communicate(host,null,"The last card is down and dirty.",null);
1554                                                dealToAll(host,1,0);
1555                                                roundOfPlay++;
1556                                                gameState=STATE_DEALING|STATEMASK_NEED_ANOUNCEMENT;
1557                                                timer=0;
1558                                                break;
1559                                        default:
1560                                                announceWinners((Environmental)ticking);
1561                                                endTheGame((Environmental)ticking);
1562                                                break;
1563                                        }
1564                                        break;
1565                                case PokerDealer.GAME_7CARDSTUD:
1566                                        switch(roundOfPlay)
1567                                        {
1568                                        case 0:
1569                                        case 1:
1570                                        case 2:
1571                                                communicate(host,null,"The next card is up.",null);
1572                                                dealToAll(host,1,1);
1573                                                roundOfPlay++;
1574                                                gameState=STATE_DEALING|STATEMASK_NEED_ANOUNCEMENT;
1575                                                timer=0;
1576                                                break;
1577                                        case 3:
1578                                                communicate(host,null,"The last card is down and dirty.",null);
1579                                                dealToAll(host,1,0);
1580                                                roundOfPlay++;
1581                                                gameState=STATE_DEALING|STATEMASK_NEED_ANOUNCEMENT;
1582                                                timer=0;
1583                                                break;
1584                                        default:
1585                                                announceWinners((Environmental)ticking);
1586                                                endTheGame((Environmental)ticking);
1587                                                break;
1588                                        }
1589                                        break;
1590                                }
1591                                break;
1592                        }
1593                        }
1594                }
1595                else
1596                // now, if the player was asked to do something, such as draw or bet,
1597                // and they fail to do so in the amount of time given, we must perform
1598                // a default action FOR them.
1599                if((timer>0)&&(System.currentTimeMillis()>timer))
1600                {
1601                        switch((gameState)&STATE_MASK)
1602                        {
1603                        case STATE_DEALING:
1604                        case STATE_DONE_DRAWING:
1605                                // these events are always initiated by the dealer, so they are
1606                                // always very short.  Therefore, unlike a typical "time-out",
1607                                // they always occur.
1608                                timer=0;
1609                                // in stud games, after a deal, the highest shown
1610                                // hand always starts betting.  Therefore, we
1611                                // will reshuffle the pot to put the highest
1612                                // shown hand first.
1613                                if((gameRules==GAME_5CARDSTUD)||(gameRules==GAME_7CARDSTUD))
1614                                {
1615                                        final DVector newPot=new DVector(2);
1616                                        boolean startAdding=false;
1617                                        final DVector scores=determineAllSortedScores(true);
1618                                        if(scores.size()>0)
1619                                        {
1620                                                final MOB winner=(MOB)scores.elementAt(scores.size()-1,1);
1621                                                final DVector theRest=new DVector(2);
1622                                                for(int p=0;p<pot.size();p++)
1623                                                {
1624                                                        if(startAdding||pot.elementAt(p,1)==winner)
1625                                                        {
1626                                                                startAdding=true;
1627                                                                newPot.addElement(pot.elementAt(p,1),pot.elementAt(p,2));
1628                                                        }
1629                                                        else
1630                                                                theRest.addElement(pot.elementAt(p,1),pot.elementAt(p,2));
1631                                                }
1632                                                for(int p=0;p<theRest.size();p++)
1633                                                        newPot.addElement(theRest.elementAt(p,1),theRest.elementAt(p,2));
1634                                                pot=newPot;
1635                                                communicate(host,null,winner.name()+" is showing the best hand with "+describeHand(((Integer)scores.elementAt(scores.size()-1,2)).intValue())+".",null);
1636                                        }
1637                                }
1638                                // begin the first betting state and announce it.
1639                                whoseTurn=null;
1640                                nextPlayerNextState(host);
1641                                if(whoseTurn!=null)
1642                                        communicate(host,null,"Betting will start with "+whoseTurn.name()+".",null);
