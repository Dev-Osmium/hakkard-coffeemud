035public class AshMephit extends Mephit
036{
037        @Override
038        public String ID()
039        {
040                return "AshMephit";
041        }
042

044        {
045                super();
046        }
047        
048        private final static String localizedStaticName = CMLib.lang().L("Ash Mephit");
049
050        @Override
051        public String name()
052        {
053                return localizedStaticName;
054        }
055
056        @Override
057        public void affectCharStats(final MOB affectedMOB, final CharStats affectableStats)
058        {
059                super.affectCharStats(affectedMOB, affectableStats);
060                affectableStats.setStat(CharStats.STAT_SAVE_PIERCE, affectableStats.getStat(CharStats.STAT_SAVE_PIERCE)+50);
061                affectableStats.setStat(CharStats.STAT_SAVE_FIRE, affectableStats.getStat(CharStats.STAT_SAVE_FIRE)+50);
062                
063                affectableStats.setStat(CharStats.STAT_SAVE_BLUNT, affectableStats.getStat(CharStats.STAT_SAVE_BLUNT)-100);
064        }
065
066        @Override
067        public int lightestWeight()
068        {
069                return 10;
070        }
071
072        @Override
073        public void affectPhyStats(final Physical affected, final PhyStats affectableStats)
074        {
075                super.affectPhyStats(affected, affectableStats);
076        }

078        private final String[]  racialAbilityNames                      = { "Dragonbreath"};
079        private final int[]             racialAbilityLevels                     = { 1, };

114        @Override
115        public String healthText(final MOB viewer, final MOB mob)
116        {
117                final double pct=(CMath.div(mob.curState().getHitPoints(),mob.maxState().getHitPoints()));
118
119                if(pct<.10)
120                        return L("^r@x1^r is unstable and almost disintegrated!^N",mob.name(viewer));
121                else
122                if(pct<.20)
123                        return L("^r@x1^r is nearing disintegration.^N",mob.name(viewer));
124                else
125                if(pct<.30)
126                        return L("^r@x1^r is noticeably disintegrating.^N",mob.name(viewer));
127                else
128                if(pct<.40)
129                        return L("^y@x1^y is very damaged and slightly disintegrated.^N",mob.name(viewer));
130                else
131                if(pct<.50)
132                        return L("^y@x1^y is very damaged.^N",mob.name(viewer));
133                else
134                if(pct<.60)
135                        return L("^p@x1^p is starting to show major damage.^N",mob.name(viewer));
136                else
137                if(pct<.70)
138                        return L("^p@x1^p is definitely damaged.^N",mob.name(viewer));
139                else
140                if(pct<.80)
141                        return L("^g@x1^g is disheveled and mildly damaged.^N",mob.name(viewer));
142                else
143                if(pct<.90)
144                        return L("^g@x1^g is noticeably disheveled.^N",mob.name(viewer));
145                else
146                if(pct<.99)
147                        return L("^g@x1^g is slightly disheveled.^N",mob.name(viewer));
148                else
149                        return L("^c@x1^c is in perfect condition.^N",mob.name(viewer));
150        }
151
152        @Override 
153        public DeadBody getCorpseContainer(MOB mob, Room room)
154        {
155                final DeadBody body = super.getCorpseContainer(mob, room);
156                if(body != null)
157                {
158                        body.setMaterial(RawMaterial.RESOURCE_ASH);
159                }
160                return body;
161        }
162        
163        private static Vector<RawMaterial>      resources       = new Vector<RawMaterial>();
164
165        @Override
166        public List<RawMaterial> myResources()
167        {
168                synchronized(resources)
169                {
170                        if(resources.size()==0)
171                        {
172                                resources.addElement(makeResource
173                                        (L("a pound of ash"),RawMaterial.RESOURCE_ASH));
174                        }
175                }
176                return resources;
177        }
