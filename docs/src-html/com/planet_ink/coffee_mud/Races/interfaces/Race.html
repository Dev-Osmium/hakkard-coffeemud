542        public final static Map<String,Integer> BODYPARTHASH_RL_LOWER=new SHashtable<String,Integer>(new Enumeration<Pair<String,Integer>>()
543        {
544                private int                                                                     index   = 0;
545                private final Stack<Pair<String, Integer>>      others  = new Stack<Pair<String, Integer>>();
546
547                @Override
548                public boolean hasMoreElements()
549                {
550                        return (others.size() > 0) || (index < BODYPARTSTR.length);
551                }
552
553                @Override
554                public Pair<String, Integer> nextElement()
555                {
556                        if (!hasMoreElements())
557                                throw new NoSuchElementException();
558                        if (others.size() > 0)
559                                return others.pop();
560                        others.push(new Pair<String, Integer>(BODYPARTSTR[index].toLowerCase(), Integer.valueOf(index)));
561                        others.push(new Pair<String, Integer>("left " + BODYPARTSTR[index].toLowerCase(), Integer.valueOf(index)));
562                        others.push(new Pair<String, Integer>("right " + BODYPARTSTR[index].toLowerCase(), Integer.valueOf(index)));
563                        index++;
564                        return others.pop();
565                }
566        });
