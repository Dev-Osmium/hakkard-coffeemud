525        @Override
526        public void executeMsg(final Environmental myHost, final CMMsg msg)
527        {
528                // the sex rules
529                if(!(myHost instanceof MOB))
530                        return;
531
532                final MOB myChar=(MOB)myHost;
533                if((msg.source()==myChar)
534                &&(this.getXPAdjustment()!=0)
535                &&((msg.sourceMinor()==CMMsg.TYP_EXPCHANGE) // race gets execmsg before expchange application, so this works.
536                        ||(msg.sourceMinor()==CMMsg.TYP_RPXPCHANGE))
537                &&(msg.value()!=0))
538                        msg.setValue(msg.value() + (int)Math.round(msg.value() * CMath.div(getXPAdjustment(), 100.0)));
539
540                if((msg.tool() instanceof Social)
541                &&(msg.amITarget(myChar)||(msg.source()==myChar))
542                &&(myChar.location()==msg.source().location())
543                &&(msg.sourceMinor()!=CMMsg.TYP_CHANNEL)
544                &&(msg.tool().Name().startsWith("MATE ")
545                        ||msg.tool().Name().startsWith("SEX ")))
546                {
547                        if(msg.tool().Name().endsWith("SELF"))
548                        {
549                                if((msg.source()==myChar)
550                                &&(fertile())
551                                &&(msg.source().fetchWornItems(Wearable.WORN_LEGS|Wearable.WORN_WAIST,(short)-2048,(short)0).size()==0))
552                                {
553                                        if(msg.source().maxState().getFatigue()>Long.MIN_VALUE/2)
554                                                msg.source().curState().adjFatigue(CharState.FATIGUED_MILLIS,msg.source().maxState());
555                                        if(myChar.maxState().getFatigue()>Long.MIN_VALUE/2)
556                                                myChar.curState().adjFatigue(CharState.FATIGUED_MILLIS,myChar.maxState());
557                                        final Ability A=CMClass.getAbility("Spell_Blindness");
558                                        if(A!=null)
559                                                A.invoke(myChar,myChar,true,myChar.phyStats().level());
560                                }
561                        }
562                        else
563                        if((msg.target()==myChar)
564                        &&(msg.tool().Name().endsWith("<T-NAME>")))
565                        {
566                                final boolean srcExhausted=((msg.source().curState().getMovement()<(msg.source().maxState().getMovement()/2))
567                                                ||(msg.source().curState().getFatigue()>=CharState.FATIGUED_MILLIS));
568                                final boolean meExhausted=((myChar.curState().getMovement()<(myChar.maxState().getMovement()/2))
569                                                ||(myChar.curState().getFatigue()>=CharState.FATIGUED_MILLIS));
570                                if((myChar.charStats().getStat(CharStats.STAT_GENDER)==('F'))
571                                &&(msg.source().charStats().getStat(CharStats.STAT_GENDER)==('M'))
572                                &&(myChar.fetchWornItems(Wearable.WORN_LEGS|Wearable.WORN_WAIST,(short)-2048,(short)0).size()==0)
573                                &&(msg.source().fetchWornItems(Wearable.WORN_LEGS|Wearable.WORN_WAIST,(short)-2048,(short)0).size()==0)
574                                &&(msg.source().charStats().getMyRace().canBreedWith(this))
575                                &&((msg.source().charStats().getStat(CharStats.STAT_AGE)==0)
576                                                ||((msg.source().charStats().ageCategory()>Race.AGE_CHILD)
577                                                                &&(msg.source().charStats().ageCategory()<Race.AGE_OLD)))
578                                &&((myChar.charStats().getStat(CharStats.STAT_AGE)==0)
579                                                ||((myChar.charStats().ageCategory()>Race.AGE_CHILD)
580                                                                &&(myChar.charStats().ageCategory()<Race.AGE_OLD))))
581                                {
582                                        if(srcExhausted)
583                                                msg.source().tell(L("You are exhausted!"));
584                                        else
585                                        {
586                                                if(msg.source().maxState().getFatigue()>Long.MIN_VALUE/2)
587                                                        msg.source().curState().adjFatigue(CharState.FATIGUED_MILLIS,msg.source().maxState());
588                                                msg.source().curState().adjMovement(-msg.source().maxState().getMovement()/2, msg.source().maxState());
589                                        }
590                                        if(meExhausted)
591                                                myChar.tell(L("You are exhausted!"));
592                                        else
593                                        {
594                                                if(myChar.maxState().getFatigue()>Long.MIN_VALUE/2)
595                                                        myChar.curState().adjFatigue(CharState.FATIGUED_MILLIS,myChar.maxState());
596                                                myChar.curState().adjMovement(-myChar.maxState().getMovement()/2, myChar.maxState());
597                                        }
598                                        if(!srcExhausted && !meExhausted && (CMLib.dice().rollPercentage()<10))
599                                        {
600                                                final Ability A=CMClass.getAbility("Pregnancy");
601                                                if((A!=null)
602                                                &&(myChar.fetchAbility(A.ID())==null)
603                                                &&(myChar.fetchEffect(A.ID())==null))
604                                                        A.invoke(msg.source(),myChar,true,0);
605                                        }
606                                }
607                        }
608                }
