171        @Override
172        public Weapon myNaturalWeapon()
173        {
174                if(naturalWeapon==null)
175                {
176                        naturalWeapon=CMClass.getWeapon("StdWeapon");
177                        naturalWeapon.setName(L("a pair of huge claws"));
178                        naturalWeapon.setMaterial(RawMaterial.RESOURCE_BONE);
179                        naturalWeapon.setUsesRemaining(1000);
180                        naturalWeapon.setWeaponDamageType(Weapon.TYPE_SLASHING);
181                }
182                return naturalWeapon;
183        }

185        @Override
186        public String makeMobName(char gender, int age)
187        {
188                switch(age)
189                {
190                        case Race.AGE_INFANT:
191                        case Race.AGE_TODDLER:
192                        case Race.AGE_CHILD:
193                                return name().toLowerCase()+" cub";
194                        default :
195                                return super.makeMobName(gender, age);
196                }
197        }

199        @Override
200        public String healthText(final MOB viewer, final MOB mob)
201        {
202                final double pct=(CMath.div(mob.curState().getHitPoints(),mob.maxState().getHitPoints()));
203
204                if(pct<.10)
205                        return L("^r@x1^r is hovering on deaths door!^N",mob.name(viewer));
206                else
207                if(pct<.20)
208                        return L("^r@x1^r is covered in blood and matted hair.^N",mob.name(viewer));
209                else
210                if(pct<.30)
211                        return L("^r@x1^r is bleeding badly from lots of wounds.^N",mob.name(viewer));
212                else
213                if(pct<.40)
214                        return L("^y@x1^y has large patches of bloody matted fur.^N",mob.name(viewer));
215                else
216                if(pct<.50)
217                        return L("^y@x1^y has some bloody matted fur.^N",mob.name(viewer));
218                else
219                if(pct<.60)
220                        return L("^p@x1^p has a lot of cuts and gashes.^N",mob.name(viewer));
221                else
222                if(pct<.70)
223                        return L("^p@x1^p has a few cut patches.^N",mob.name(viewer));
224                else
225                if(pct<.80)
226                        return L("^g@x1^g has a cut patch of fur.^N",mob.name(viewer));
227                else
228                if(pct<.90)
229                        return L("^g@x1^g has some disheveled fur.^N",mob.name(viewer));
230                else
231                if(pct<.99)
232                        return L("^g@x1^g has some misplaced hairs.^N",mob.name(viewer));
233                else
234                        return L("^c@x1^c is in perfect health.^N",mob.name(viewer));
235        }

237        @Override
238        public List<RawMaterial> myResources()
239        {
240                synchronized(resources)
241                {
242                        if(resources.size()==0)
243                        {
244                                resources.addElement(makeResource
245                                (L("some @x1 claws",name().toLowerCase()),RawMaterial.RESOURCE_BONE));
246                                for(int i=0;i<5;i++)
247                                {
248                                        resources.addElement(makeResource
249                                        (L("a strip of @x1 fur",name().toLowerCase()),RawMaterial.RESOURCE_FUR,L("@x1 fur",name().toLowerCase())));
250                                        resources.addElement(makeResource
251                                        (L("a strip of @x1 hide",name().toLowerCase()),RawMaterial.RESOURCE_HIDE));
252                                }
253                                for(int i=0;i<3;i++)
254                                {
255                                        resources.addElement(makeResource
256                                        (L("a pound of @x1 meat",name().toLowerCase()),RawMaterial.RESOURCE_MEAT));
257                                }
258                                resources.addElement(makeResource
259                                (L("some @x1 blood",name().toLowerCase()),RawMaterial.RESOURCE_BLOOD));
260                                resources.addElement(makeResource
261                                (L("a pile of @x1 bones",name().toLowerCase()),RawMaterial.RESOURCE_BONE));
262                        }
263                }
264                return resources;
265        }
