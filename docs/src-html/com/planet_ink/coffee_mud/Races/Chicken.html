191        @Override
192        public String makeMobName(char gender, int age)
193        {
194                switch(age)
195                {
196                        case Race.AGE_INFANT:
197                        case Race.AGE_TODDLER:
198                        case Race.AGE_CHILD:
199                        {
200                                switch(gender)
201                                {
202                                case 'M':
203                                case 'm':
204                                        return "cockrell";
205                                case 'F':
206                                case 'f':
207                                        return "chick";
208                                default:
209                                        return super.makeMobName(gender, age);
210                                }
211                        }
212                        case Race.AGE_YOUNGADULT:
213                        case Race.AGE_MATURE:
214                        case Race.AGE_MIDDLEAGED:
215                        {
216                                switch(gender)
217                                {
218                                case 'M':
219                                case 'm':
220                                        return "rooster";
221                                case 'F':
222                                case 'f':
223                                        return "hen";
224                                default:
225                                        return super.makeMobName(gender, age);
226                                }
227                        }
228                        case Race.AGE_OLD:
229                        case Race.AGE_VENERABLE:
230                        case Race.AGE_ANCIENT:
231                        {
232                                switch(gender)
233                                {
234                                case 'M':

249        @Override
250        public String healthText(final MOB viewer, final MOB mob)
251        {
252                final double pct=(CMath.div(mob.curState().getHitPoints(),mob.maxState().getHitPoints()));
253
254                if(pct<.10)
255                        return L("^r@x1^r is hovering on deaths door!^N",mob.name(viewer));
256                else
257                if(pct<.20)
258                        return L("^r@x1^r is covered in blood and matted feathers.^N",mob.name(viewer));
259                else
260                if(pct<.30)
261                        return L("^r@x1^r is bleeding badly from lots of wounds.^N",mob.name(viewer));
262                else
263                if(pct<.40)
264                        return L("^y@x1^y has numerous bloody matted feathers.^N",mob.name(viewer));
265                else
266                if(pct<.50)
267                        return L("^y@x1^y has some bloody matted feathers.^N",mob.name(viewer));
268                else
269                if(pct<.60)
270                        return L("^p@x1^p has a lot of missing feathers.^N",mob.name(viewer));
271                else
272                if(pct<.70)
273                        return L("^p@x1^p has a few missing feathers.^N",mob.name(viewer));
274                else
275                if(pct<.80)
276                        return L("^g@x1^g has a missing feather.^N",mob.name(viewer));
277                else
278                if(pct<.90)
279                        return L("^g@x1^g has a few feathers out of place.^N",mob.name(viewer));
280                else
281                if(pct<.99)
282                        return L("^g@x1^g has a some ruffled feathers.^N",mob.name(viewer));
283                else
284                        return L("^c@x1^c is in perfect health.^N",mob.name(viewer));
285        }

287        @Override
288        public boolean tick(final Tickable ticking, final int tickID)
289        {
290                if(!super.tick(ticking,tickID))
291                        return false;
292                if((tickID==Tickable.TICKID_MOB)
293                &&(ticking instanceof MOB)
294                &&(((MOB)ticking).charStats().getStat(CharStats.STAT_GENDER)=='F'))
295                {
296                        if((CMLib.dice().rollPercentage()>99)&&(((MOB)ticking).numItems()<9))
297                        {
298                                final Item I=CMClass.getItem("GenFoodResource");
299                                I.setName(L("an egg"));
300                                I.setDisplayText(L("an egg has been left here."));
301                                I.setMaterial(RawMaterial.RESOURCE_EGGS);
302                                I.setDescription(L("It looks like a chicken egg!"));
303                                I.basePhyStats().setWeight(1);
304                                CMLib.materials().addEffectsToResource(I);
305                                ((MOB)ticking).addItem((Item)I.copyOf());
306                        }
307                        if((((MOB)ticking).numItems()>5)
308                        &&(((MOB)ticking).location()!=null)
309                        &&(((MOB)ticking).location().findItem(null,"an egg")==null))
310                        {
311                                final Item I=((MOB)ticking).findItem("an egg");
312                                if(I!=null)
313                                {
314                                        ((MOB)ticking).location().show(((MOB)ticking),null,CMMsg.MSG_NOISYMOVEMENT,L("<S-NAME> lay(s) an egg."));
315                                        I.removeFromOwnerContainer();
316                                        I.executeMsg((MOB)ticking,CMClass.getMsg((MOB)ticking,I,null,CMMsg.TYP_ROOMRESET,null));
317                                        ((MOB)ticking).location().addItem(I,ItemPossessor.Expire.Resource);
318                                        ((MOB)ticking).location().recoverRoomStats();
319                                }
320                        }
321                }
322                return true;
323        }

334                                resources.addElement(makeResource
335                                (L("some @x1 feathers",name().toLowerCase()),RawMaterial.RESOURCE_FEATHERS));
336                                resources.addElement(makeResource
337                                (L("some @x1 meat",name().toLowerCase()),RawMaterial.RESOURCE_POULTRY));
338                                resources.addElement(makeResource
339                                (L("some @x1 blood",name().toLowerCase()),RawMaterial.RESOURCE_BLOOD));
340                                resources.addElement(makeResource
341                                (L("a pile of @x1 bones",name().toLowerCase()),RawMaterial.RESOURCE_BONE));
342                        }
