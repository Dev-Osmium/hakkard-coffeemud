156        @Override
157        public void affectPhyStats(final Physical affected, final PhyStats affectableStats)
158        {
159                super.affectPhyStats(affected,affectableStats);
160                affectableStats.setDisposition(affectableStats.disposition()|PhyStats.IS_SNEAKING);
161        }
162
163        @Override
164        public void affectCharStats(final MOB affectedMOB, final CharStats affectableStats)
165        {
166                super.affectCharStats(affectedMOB, affectableStats);
167                affectableStats.setRacialStat(CharStats.STAT_STRENGTH,5);
168                affectableStats.setRacialStat(CharStats.STAT_DEXTERITY,5);
169                affectableStats.setRacialStat(CharStats.STAT_INTELLIGENCE,1);
170                affectableStats.setStat(CharStats.STAT_SAVE_POISON,affectableStats.getStat(CharStats.STAT_SAVE_POISON)+100);
171        }

185        @Override
186        public Weapon myNaturalWeapon()
187        {
188                if(naturalWeapon==null)
189                {
190                        naturalWeapon=CMClass.getWeapon("StdWeapon");
191                        naturalWeapon.setName(L("a nasty stinger"));
192                        naturalWeapon.setMaterial(RawMaterial.RESOURCE_BONE);
193                        naturalWeapon.setUsesRemaining(1000);
194                        naturalWeapon.setWeaponDamageType(Weapon.TYPE_PIERCING);
195                }
196                return naturalWeapon;
197        }
198
199        @Override
200        public List<RawMaterial> myResources()
201        {
202                synchronized(resources)
203                {
204                        if(resources.size()==0)
205                        {
206                                resources.addElement(makeResource
207                                (L("some @x1 pincers",name().toLowerCase()),RawMaterial.RESOURCE_BONE));
208                        }
209                }
210                return resources;
211        }
212
213        @Override
214        public String makeMobName(char gender, int age)
215        {
216                switch(age)
217                {
218                        case Race.AGE_INFANT:
219                        case Race.AGE_TODDLER:
220                        case Race.AGE_CHILD:
221                                return "baby "+name().toLowerCase();
222                        default :
223                                return super.makeMobName('N', age);
224                }
225        }
