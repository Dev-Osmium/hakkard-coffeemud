181        @Override
182        public String makeMobName(char gender, int age)
183        {
184                switch(age)
185                {
186                        case Race.AGE_INFANT:
187                        case Race.AGE_TODDLER:
188                                return name().toLowerCase()+" chick";
189                        case Race.AGE_CHILD:
190                                return "young "+name().toLowerCase();
191                        default :
192                                return super.makeMobName(gender, age);
193                }
194        }

196        @Override
197        public List<Item> outfit(MOB myChar)
198        {
199                if(outfitChoices==null)
200                {
201                        // Have to, since it requires use of special constructor
202                        final Armor s1=CMClass.getArmor("GenShirt");
203                        if(s1 == null)
204                                return new Vector<Item>();
205                        outfitChoices=new Vector<Item>();
206                        s1.setName(L("a delicate green shirt"));
207                        s1.setDisplayText(L("a delicate green shirt sits gracefully here."));
208                        s1.setDescription(L("Obviously fine craftmenship, with sharp folds and intricate designs."));
209                        s1.text();
210                        outfitChoices.add(s1);
211
212                        final Armor s2=CMClass.getArmor("GenShoes");
213                        s2.setName(L("a pair of sandals"));
214                        s2.setDisplayText(L("a pair of sandals lie here."));
215                        s2.setDescription(L("Obviously fine craftmenship, these light leather sandals have tiny woodland drawings in them."));
216                        s2.text();
217                        outfitChoices.add(s2);
218
219                        final Armor p1=CMClass.getArmor("GenPants");
220                        p1.setName(L("some delicate leggings"));
221                        p1.setDisplayText(L("a pair delicate brown leggings sit here."));
222                        p1.setDescription(L("Obviously fine craftmenship, with sharp folds and intricate designs.  They look perfect for dancing in!"));
223                        p1.text();
224                        outfitChoices.add(p1);
225
226                        final Armor s3=CMClass.getArmor("GenBelt");
227                        outfitChoices.add(s3);
228                }
229                return outfitChoices;
230        }

238        @Override
239        public String healthText(final MOB viewer, final MOB mob)
240        {
241                final double pct=(CMath.div(mob.curState().getHitPoints(),mob.maxState().getHitPoints()));
242
243                if(pct<.10)
244                        return L("^r@x1^r is facing mortality!^N",mob.name(viewer));
245                else
246                if(pct<.20)
247                        return L("^r@x1^r is covered in blood.^N",mob.name(viewer));
248                else
249                if(pct<.30)
250                        return L("^r@x1^r is bleeding badly from lots of wounds.^N",mob.name(viewer));
251                else
252                if(pct<.40)
253                        return L("^y@x1^y has numerous bloody wounds and gashes.^N",mob.name(viewer));
254                else
255                if(pct<.50)
256                        return L("^y@x1^y has some bloody wounds and gashes.^N",mob.name(viewer));
257                else
258                if(pct<.60)
259                        return L("^p@x1^p has a few bloody wounds.^N",mob.name(viewer));
260                else
261                if(pct<.70)
262                        return L("^p@x1^p is cut and bruised.^N",mob.name(viewer));
263                else
264                if(pct<.80)
265                        return L("^g@x1^g has some minor cuts and bruises.^N",mob.name(viewer));
266                else
267                if(pct<.90)
268                        return L("^g@x1^g has a few bruises and scratches.^N",mob.name(viewer));
269                else
270                if(pct<.99)
271                        return L("^g@x1^g has a few small bruises.^N",mob.name(viewer));
272                else
273                        return L("^c@x1^c is in perfect health.^N",mob.name(viewer));
274        }

276        @Override
277        public List<RawMaterial> myResources()
278        {
279                synchronized(resources)
280                {
281                        if(resources.size()==0)
282                        {
283                                resources.addElement(makeResource
284                                (L("a pair of @x1 ears",name().toLowerCase()),RawMaterial.RESOURCE_MEAT));
285                                resources.addElement(makeResource
286                                (L("some @x1 feathers",name().toLowerCase()),RawMaterial.RESOURCE_FEATHERS,L("@x1 FEATHER",name().toUpperCase())));
287                                resources.addElement(makeResource
288                                (L("some @x1 blood",name().toLowerCase()),RawMaterial.RESOURCE_BLOOD));
289                                resources.addElement(makeResource
290                                (L("a pile of @x1 bones",name().toLowerCase()),RawMaterial.RESOURCE_BONE));
291                        }
292                }
293                return resources;
294        }
