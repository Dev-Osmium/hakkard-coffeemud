112        private static Vector<RawMaterial>      resources       = new Vector<RawMaterial>();
113
114        @Override
115        public void affectCharStats(final MOB affectedMOB, final CharStats affectableStats)
116        {
117                affectableStats.setStat(CharStats.STAT_GENDER, 'F');
118        }
119
120        @Override
121        public List<RawMaterial> myResources()
122        {
123                synchronized(resources)
124                {
125                        if(resources.size()==0)
126                        {
127                                resources.addElement(makeResource
128                                (L("some greasy @x1 claws",name().toLowerCase()),RawMaterial.RESOURCE_BONE));
129                                for(int i=0;i<2;i++)
130                                {
131                                        resources.addElement(makeResource
132                                        (L("some dirty @x1 feathers",name().toLowerCase()),RawMaterial.RESOURCE_FEATHERS,L("@x1 FEATHER",name().toUpperCase())));
133                                }
134                                resources.addElement(makeResource
135                                (L("some @x1 meat",name().toLowerCase()),RawMaterial.RESOURCE_POULTRY));
136                                resources.addElement(makeResource
137                                (L("some @x1 blood",name().toLowerCase()),RawMaterial.RESOURCE_BLOOD));
138                                resources.addElement(makeResource
139                                (L("a pile of @x1 bones",name().toLowerCase()),RawMaterial.RESOURCE_BONE));
140                        }
141                }
142                return resources;
143        }
