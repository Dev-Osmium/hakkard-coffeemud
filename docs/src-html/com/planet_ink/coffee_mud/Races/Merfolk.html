217        @Override
218        public List<Item> outfit(MOB myChar)
219        {
220                if(outfitChoices==null)
221                {
222                        // Have to, since it requires use of special constructor
223                        final Armor s1=CMClass.getArmor("GenShirt");
224                        if(s1 == null)
225                                return new Vector<Item>();
226                        outfitChoices=new Vector<Item>();
227                        s1.setName(L("a delicate scaley green shirt"));
228                        s1.setDisplayText(L("a delicate scaley green shirt sits here."));
229                        s1.setDescription(L("Obviously fine craftmenship, with sparking scales and intricate designs."));
230                        s1.setMaterial(RawMaterial.RESOURCE_SCALES);
231                        s1.text();
232                        outfitChoices.add(s1);
233
234                        final Armor s3=CMClass.getArmor("GenBelt");
235                        outfitChoices.add(s3);
236                }
237                return outfitChoices;
238        }

246        @Override
247        public String makeMobName(char gender, int age)
248        {
249                switch(age)
250                {
251                        case Race.AGE_INFANT:
252                                return "merbaby";
253                        case Race.AGE_TODDLER:
254                                return "mertoddler";
255                        case Race.AGE_CHILD:
256                                switch(gender)
257                                {
258                                case 'M':
259                                case 'm':
260                                        return "merboy";
261                                case 'F':
262                                case 'f':
263                                        return "mergirl";
264                                default:
265                                        return "young "+name().toLowerCase();
266                                }
267                        case Race.AGE_YOUNGADULT:
268                        case Race.AGE_MATURE:
269                        case Race.AGE_MIDDLEAGED:
270                        default:
271                        {
272                                switch(gender)
273                                {
274                                case 'M':
275                                case 'm':
276                                        return "merman";
277                                case 'F':
278                                case 'f':
279                                        return "mermaid";
280                                default:
281                                        return name().toLowerCase();
282                                }
283                        }
284                        case Race.AGE_OLD:
285                        case Race.AGE_VENERABLE:
286                        case Race.AGE_ANCIENT:
287                        {
288                                switch(gender)
289                                {
290                                case 'M':
291                                case 'm':
292                                        return "old merman";
293                                case 'F':
294                                case 'f':
295                                        return "old mermaid";
296                                default:
297                                        return "old " + name().toLowerCase();
298                                }
299                        }
300                }
301        }

303        @Override
304        public String healthText(final MOB viewer, final MOB mob)
305        {
306                final double pct=(CMath.div(mob.curState().getHitPoints(),mob.maxState().getHitPoints()));
307
308                if(pct<.10)
309                        return L("^r@x1^r is facing a cold death!^N",mob.name(viewer));
310                else
311                if(pct<.20)
312                        return L("^r@x1^r is covered in blood.^N",mob.name(viewer));
313                else
314                if(pct<.30)
315                        return L("^r@x1^r is bleeding badly from lots of wounds.^N",mob.name(viewer));
316                else
317                if(pct<.40)
318                        return L("^y@x1^y has numerous bloody wounds and gashes.^N",mob.name(viewer));
319                else
320                if(pct<.50)
321                        return L("^y@x1^y has some bloody wounds and gashes.^N",mob.name(viewer));
322                else
323                if(pct<.60)
324                        return L("^p@x1^p has a few bloody wounds.^N",mob.name(viewer));
325                else
326                if(pct<.70)
327                        return L("^p@x1^p is cut and bruised heavily.^N",mob.name(viewer));
328                else
329                if(pct<.80)
330                        return L("^g@x1^g has some minor cuts and bruises.^N",mob.name(viewer));
331                else
332                if(pct<.90)
333                        return L("^g@x1^g has a few bruises and scratched scales.^N",mob.name(viewer));
334                else
335                if(pct<.99)
336                        return L("^g@x1^g has a few small bruises.^N",mob.name(viewer));
337                else
338                        return L("^c@x1^c is in perfect health.^N",mob.name(viewer));
339        }

341        @Override
342        public List<RawMaterial> myResources()
343        {
344                synchronized(resources)
345                {
346                        if(resources.size()==0)
347                        {
348                                resources.addElement(makeResource
349                                (L("a pair of @x1 fins",name().toLowerCase()),RawMaterial.RESOURCE_MEAT));
350                                resources.addElement(makeResource
351                                (L("some @x1 blood",name().toLowerCase()),RawMaterial.RESOURCE_BLOOD));
352                                resources.addElement(makeResource
353                                (L("a pile of @x1 bones",name().toLowerCase()),RawMaterial.RESOURCE_BONE));
354                        }
355                }
356                return resources;
357        }
