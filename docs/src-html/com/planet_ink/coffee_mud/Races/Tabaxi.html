148        @Override
149        public void affectPhyStats(final Physical affected, final PhyStats affectableStats)
150        {
151                super.affectPhyStats(affected,affectableStats);
152                        affectableStats.setSensesMask(affectableStats.sensesMask()|PhyStats.CAN_SEE_INFRARED); //would like to turn this darkvison if it is okay with Zac.
153                affectableStats.setSpeed(affectableStats.speed() * 2.0);
154        }
155
156        @Override
157        public void affectCharStats(final MOB affectedMOB, final CharStats affectableStats)
158        {
159                affectableStats.setStat(CharStats.STAT_DEXTERITY,affectableStats.getStat(CharStats.STAT_DEXTERITY)+3);
160                affectableStats.setStat(CharStats.STAT_MAX_DEXTERITY_ADJ,affectableStats.getStat(CharStats.STAT_MAX_DEXTERITY_ADJ)+3);
161                affectableStats.setStat(CharStats.STAT_STRENGTH,affectableStats.getStat(CharStats.STAT_STRENGTH)+3);
162                affectableStats.setStat(CharStats.STAT_MAX_STRENGTH_ADJ,affectableStats.getStat(CharStats.STAT_MAX_STRENGTH_ADJ)+3);
163                affectableStats.setStat(CharStats.STAT_INTELLIGENCE,affectableStats.getStat(CharStats.STAT_INTELLIGENCE)-4);
164                affectableStats.setStat(CharStats.STAT_MAX_INTELLIGENCE_ADJ,affectableStats.getStat(CharStats.STAT_MAX_INTELLIGENCE_ADJ)-4);
165                affectableStats.setStat(CharStats.STAT_WISDOM,affectableStats.getStat(CharStats.STAT_WISDOM)-2);
166                affectableStats.setStat(CharStats.STAT_MAX_WISDOM_ADJ,affectableStats.getStat(CharStats.STAT_MAX_WISDOM_ADJ)-2);
167        }

169        @Override
170        public List<RawMaterial> myResources()
171        {
172                synchronized(resources)
173                {
174                        if(resources.size()==0)
175                        {
176                                resources.addElement(makeResource
177                                (L("a set of @x1 fangs",name().toLowerCase()),RawMaterial.RESOURCE_BONE));
178                                for(int i=0;i<5;i++)
179                                {
180                                        resources.addElement(makeResource
181                                        (L("a strip of @x1 leather",name().toLowerCase()),RawMaterial.RESOURCE_LEATHER));
182                                }
183                                for(int i=0;i<2;i++)
184                                {
185                                        resources.addElement(makeResource
186                                        (L("a pound of @x1 meat",name().toLowerCase()),RawMaterial.RESOURCE_BEEF));
187                                }
188                                resources.addElement(makeResource
189                                (L("some @x1 blood",name().toLowerCase()),RawMaterial.RESOURCE_BLOOD));
190                                resources.addElement(makeResource
191                                (L("a pile of @x1 bones",name().toLowerCase()),RawMaterial.RESOURCE_BONE));
192                        }
193                }
194                return resources;
195        }
