044
045        @Override

050
051        @Override

057        @Override

059        {

061        }
062
063        @Override
064        public int heightVariance()
065        {
066                return 12;
067        }
068
069        @Override
070        public int lightestWeight()
071        {
072                return 200;
073        }
074
075        @Override
076        public int weightVariance()
077        {
078                return 200;
079        }
080
081        @Override
082        public long forbiddenWornBits()
083        {
084                return 0;
085        }
086
087        private final static String localizedStaticRacialCat = CMLib.lang().L("Slime");
088
089        @Override
090        public String racialCategory()
091        {
092                return localizedStaticRacialCat;
093        }
094
095        @Override
096        public boolean fertile()
097        {
098                return true;
099        }
100
101        @Override
102        public String arriveStr()
103        {
104                return "sloshes in";
105        }
106
107        @Override
108        public String leaveStr()
109        {
110                return "sloshes";
111        }
112
113        @Override
114        public Weapon myNaturalWeapon()
115        {
116                if(naturalWeapon==null)
117                {
118                        naturalWeapon=CMClass.getWeapon("StdWeapon");
119                        naturalWeapon.setName(L("an appendange"));
120                        naturalWeapon.setRanges(0,2);
121                        naturalWeapon.setMaterial(RawMaterial.RESOURCE_BERRIES);
122                        naturalWeapon.setUsesRemaining(1000);
123                        naturalWeapon.setWeaponDamageType(Weapon.TYPE_BASHING);
124                }
125                return naturalWeapon;
126        }
127
128        @Override 
129        public DeadBody getCorpseContainer(MOB mob, Room room)
130        {
131                final DeadBody body = super.getCorpseContainer(mob, room);
132                if(body != null)
133                {
134                        body.setMaterial(RawMaterial.RESOURCE_BLOOD);
135                }
136                return body;
137        }
138        
139        private static Vector<RawMaterial>      resources = new Vector<RawMaterial>();
140
141        @Override
142        public List<RawMaterial> myResources()
143        {
144                synchronized(resources)
145                {
146                        if(resources.size()==0)
147                        {
148                                resources.addElement(makeResource
149                                (L("a palm-full of @x1",name().toLowerCase()),RawMaterial.RESOURCE_BLOOD));
150                        }
151                }
152                return resources;
153        }
154
155        @Override
156        public String makeMobName(char gender, int age)
157        {
158                switch(age)
159                {
160                        case Race.AGE_INFANT:
161                        case Race.AGE_TODDLER:
162                        case Race.AGE_CHILD:
163                                return name().toLowerCase()+" puddle";
164                        default :
165                                return super.makeMobName('N', age);
166                }
167        }
168        
169}
