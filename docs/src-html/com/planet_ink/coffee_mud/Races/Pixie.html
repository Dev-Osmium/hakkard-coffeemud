196        @Override
197        public void executeMsg(final Environmental myHost, final CMMsg msg)
198        {
199                if((msg.source()==myHost)
200                &&(msg.target() instanceof Food)
201                &&((((Food)msg.target()).material()&RawMaterial.MATERIAL_MASK)==RawMaterial.MATERIAL_FLESH)
202                &&(((Food)msg.target()).material()!=RawMaterial.RESOURCE_WAX))
203                {
204                        if(msg.targetMinor()==CMMsg.TYP_EAT)
205                        {
206                                final Ability A=CMClass.getAbility("Poison_Heartstopper");
207                                if(A!=null)
208                                        A.invoke(msg.source(),msg.source(),true,0);
209                        }
210                        else
211                        if((msg.targetMinor()==CMMsg.TYP_GET)||(msg.targetMinor()==CMMsg.TYP_PUSH)||(msg.targetMinor()==CMMsg.TYP_PULL))
212                        {
213                                final Ability A=CMClass.getAbility("Poison_Hives");
214                                if(A!=null)
215                                        A.invoke(msg.source(),msg.source(),true,0);
216                        }
217                }
218                super.executeMsg(myHost, msg);
219        }

223        {
224                if(outfitChoices==null)

226                        // Have to, since it requires use of special constructor
227                        final Armor s1=CMClass.getArmor("GenShirt");

263        @Override
264        public String healthText(final MOB viewer, final MOB mob)
265        {
266                final double pct=(CMath.div(mob.curState().getHitPoints(),mob.maxState().getHitPoints()));
267
268                if(pct<.10)
269                        return L("^r@x1^r is facing mortality!^N",mob.name(viewer));
270                else
271                if(pct<.20)
272                        return L("^r@x1^r is covered in blood.^N",mob.name(viewer));
273                else
274                if(pct<.30)
275                        return L("^r@x1^r is bleeding badly from lots of wounds.^N",mob.name(viewer));
276                else
277                if(pct<.40)
278                        return L("^y@x1^y has numerous bloody wounds and gashes.^N",mob.name(viewer));
279                else
280                if(pct<.50)
281                        return L("^y@x1^y has some bloody wounds and gashes.^N",mob.name(viewer));
282                else
283                if(pct<.60)
284                        return L("^p@x1^p has a few bloody wounds.^N",mob.name(viewer));
285                else
286                if(pct<.70)
287                        return L("^p@x1^p is cut and bruised.^N",mob.name(viewer));
288                else
289                if(pct<.80)
290                        return L("^g@x1^g has some minor cuts and bruises.^N",mob.name(viewer));
291                else
292                if(pct<.90)
293                        return L("^g@x1^g has a few bruises and scratches.^N",mob.name(viewer));
294                else
295                if(pct<.99)
296                        return L("^g@x1^g has a few small bruises.^N",mob.name(viewer));
297                else
298                        return L("^c@x1^c is in perfect health.^N",mob.name(viewer));
299        }

301        private static Vector<RawMaterial>      resources       = new Vector<RawMaterial>();
302
303        @Override
304        public List<RawMaterial> myResources()
305        {
306                synchronized(resources)
307                {
308                        if(resources.size()==0)
309                        {
310                                resources.addElement(makeResource
311                                (L("a pair of @x1 ears",name().toLowerCase()),RawMaterial.RESOURCE_MEAT));
312                                resources.addElement(makeResource
313                                (L("some @x1 blood",name().toLowerCase()),RawMaterial.RESOURCE_BLOOD));
314                                resources.addElement(makeResource
315                                (L("a pile of @x1 bones",name().toLowerCase()),RawMaterial.RESOURCE_BONE));
316                        }
317                }
318                return resources;
319        }
