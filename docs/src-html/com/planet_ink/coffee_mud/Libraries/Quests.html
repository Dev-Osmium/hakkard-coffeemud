169        @Override
170        public Object getHolidayFile()
171        {
172                Quest Q=fetchQuest("holidays");
173                if((Q==null)
174                ||(!Q.script().toUpperCase().trim().equalsIgnoreCase(holidayDefinition)))
175                {
176                        Q=null;
177                        final CMFile lF=new CMFile("//"+Resources.makeFileResourceName(holidayFilename),null);
178                        final CMFile vF=new CMFile("::"+Resources.makeFileResourceName(holidayFilename),null);
179                        if((lF.exists())&&(!vF.exists())&&(lF.canRead())&&(vF.canWrite()))
180                        {
181                                final byte[] O=lF.raw();
182                                vF.saveRaw(O);
183                        }
184                        Q=(Quest)CMClass.getCommon("DefaultQuest");
185                        Q.setScript(holidayDefinition,true);
186                        addQuest(Q);
187                        CMLib.database().DBUpdateQuest(Q);
188                        Q=fetchQuest("holidays");
189                        if(Q==null)
190                                return "A quest named 'holidays', with the script definition '"+holidayDefinition+"' has not been created.  Enter the following to create this quest:\n\r"
191                                          +"CREATE QUEST "+holidayDefinition+"\n\r"
192                                          +"SAVE QUESTS";
193                }
194                final CMFile F=new CMFile(Resources.makeFileResourceName(holidayFilename),null);
195                if((!F.exists())||(!F.canRead())||(!F.canWrite()))
196                {
197                        return "The file '"+Resources.makeFileResourceName(holidayFilename)+"' does not exist, and is required for this feature.";
198                }
199                final List<String> V=Resources.getFileLineVector(F.text());
200                final List<String> steps=parseQuestSteps(V,0,true);
201                return steps;
202        }

206        public String listHolidays(Area A, String otherParms)
207        {
208                final Object resp=getHolidayFile();
209                if(resp instanceof String)
210                        return (String)resp;
211                List<String> steps=null;
212                if(resp instanceof List)
213                        steps=(List<String>)resp;
214                else
215                        return "Unknown error.";
216                String areaName=A.Name().toUpperCase().trim();
217                if(otherParms.equalsIgnoreCase("ALL"))
218                        areaName=null;
219                final StringBuffer str=new StringBuffer(L("^xDefined Quest Holidays^?\n\r"));
220                List<String> line=null;
221                String var=null;
222                List<String> V=null;
223                str.append("^H#  "+CMStrings.padRight(L("Holiday Name"),20)+CMStrings.padRight(L("Area Name(s)"),50)+"^?\n\r");
224                for(int s=1;s<steps.size();s++)
225                {
226                        final String step=steps.get(s);
227                        V=Resources.getFileLineVector(new StringBuffer(step));
228                        final List<List<String>> cmds=CMLib.quests().parseQuestCommandLines(V,"SET",0);
229                        List<String> areaLine=null;
230                        List<String> nameLine=null;
231                        for(int v=0;v<cmds.size();v++)
232                        {
233                                line=cmds.get(v);
234                                if(line.size()>1)
235                                {
236                                        var=line.get(1).toUpperCase();
237                                        if (var.equals("AREAGROUP"))
238                                        {
239                                                areaLine = line;
240                                        }
241                                        if (var.equals("NAME"))
242                                        {
243                                                nameLine = line;
244                                        }
245                                }
246                        }
247                        if(nameLine!=null)
248                        {
249                                boolean contains=true;//(areaName==null);
250                                if(areaLine!=null)
251                                {
252                                        if((!contains) && (areaName != null))
253                                        for(int l=2;l<areaLine.size();l++)
254                                        {
255                                                if(areaName.equalsIgnoreCase(areaLine.get(l)))
256                                                        {
257                                                                contains = true;
258                                                                break;
259                                                        }
260                                        }
261                                }
262                                else
263                                {
264                                        areaLine=new XVector<String>("","","*special*");
265                                        contains=true;
266                                }
267                                if(contains)
268                                {
269                                        final String name=CMParms.combine(nameLine,2);
270                                        str.append(CMStrings.padRight(""+s,3)+CMStrings.padRight(name,20)+CMStrings.padRight(CMParms.combineQuoted(areaLine,2),30)+"\n\r");
271                                }
272                        }
273                }
274                return str.toString();
275        }

277        protected void promptText(MOB mob, TriadList<String,String,Integer> sets, String var, int showNumber, int showFlag, String prompt, String help, boolean emptyOK)
278        throws java.io.IOException
279        {
280                final int index=sets.indexOfFirst(var);
281                final String oldVal=index>=0?(String)sets.get(index).second:"";
282                while((mob.session()!=null)&&(!mob.session().isStopped()))
283                {
284                        final String newVAL=CMLib.genEd().prompt(mob,oldVal,showNumber,showFlag,prompt,emptyOK);
285                        if(newVAL.equals("?"))
286                        {
287                                mob.tell(help);
288                                continue;
289                        }
290                        else
291                        if(index>=0)
292                                sets.get(index).second = newVAL;
293                        else
294                        if(!newVAL.equals(oldVal))
295                                sets.add(var,newVAL,Integer.valueOf(-1));
296                        break;
297                }
298        }

497        @Override
498        public HolidayData getEncodedHolidayData(String dataFromStepsFile)
499        {
500                final List<String> stepV=Resources.getFileLineVector(new StringBuffer(dataFromStepsFile));
501                for(int v=0;v<stepV.size();v++)
502                        stepV.set(v,CMStrings.replaceAll(stepV.get(v),"\\;",";"));
503                final TriadList<String,String,Integer> settings=new TriadVector<String,String,Integer>();
504                final TriadList<String,String,Integer> behaviors=new TriadVector<String,String,Integer>();
505                final TriadList<String,String,Integer> properties=new TriadVector<String,String,Integer>();
506                final TriadList<String,String,Integer> stats=new TriadVector<String,String,Integer>();
507                Vector<String> lineV=null;
508                String line=null;
509                String var=null;
510                String cmd=null;
511                int pricingMobIndex=-1;
512                final String[] SETTINGS={"NAME","WAIT","DATE","DURATION","MUDDAY","AREAGROUP","MOBGROUP"};
513                for(int v=0;v<stepV.size();v++)
514                {
515                        line=stepV.get(v);
516                        lineV=CMParms.parse(line);
517                        if(lineV.size()>1)
518                        {
519                                cmd=lineV.elementAt(0).toUpperCase();
520                                var=lineV.elementAt(1).toUpperCase();
521                                if(cmd.equals("SET")&&(CMParms.indexOf(SETTINGS,var)>=0))
522                                {
523                                        if(!settings.containsFirst(var))
524                                        {
525                                                String str=CMParms.combineQuoted(lineV,2);
526                                                if(str.toUpperCase().startsWith("ANY "))
527                                                        str=str.substring(4);
528                                                if(str.toUpperCase().startsWith("RESELECT MASK="))
529                                                        str=str.substring(14);
530                                                if(str.toUpperCase().startsWith("MASK="))
531                                                        str=str.substring(5);
532                                                settings.add(var,str,Integer.valueOf(v));
533                                        }
534                                        else
535                                        if((var.equalsIgnoreCase("MOBGROUP"))
536                                        &&(pricingMobIndex<0)
537                                        &&(CMParms.combine(lineV,2).toUpperCase().indexOf("SHOPKEEPER")>0))
538                                                pricingMobIndex=v;
539                                }
540                                if(cmd.equals("GIVE")&&("BEHAVIOR".equalsIgnoreCase(var))&&(lineV.size()>2)&&(pricingMobIndex<0))
541                                {
542                                        var=lineV.elementAt(2).toUpperCase();
543                                        behaviors.add(var,CMParms.combineQuoted(lineV,3),Integer.valueOf(v));
544                                }
545                                if(cmd.equals("GIVE")&&("AFFECT".equalsIgnoreCase(var))&&(lineV.size()>2)&&(pricingMobIndex<0))
546                                {
547                                        var=lineV.elementAt(2).toUpperCase();
548                                        properties.add(var,CMParms.combineQuoted(lineV,3),Integer.valueOf(v));
549                                }
550                                if(cmd.equals("GIVE")&&("STAT".equalsIgnoreCase(var))&&(lineV.size()>2))
551                                {
552                                        var=lineV.elementAt(2).toUpperCase();
553                                        if((pricingMobIndex<0)||(var.equals("PRICEMASKS")))
554                                                stats.add(var,CMParms.combineQuoted(lineV,3),Integer.valueOf(v));
555                                }
556                        }
557                }
558                final Integer pricingMobIndexI = Integer.valueOf(pricingMobIndex);
559                return new HolidayData()
560                {

563                        {

565                        }
566

571                        }

576                                return properties;
577                        }
578

581                        {
582                                return stats;
583                        }

601        public void modifyHoliday(MOB mob, int holidayNumber)
602        {
603                final Object resp=getHolidayFile();
604                if(resp instanceof String)
605                {
606                        mob.tell((String)resp);
607                        return;
608                }
609                List<String> steps=null;
610                if(resp instanceof List)
611                        steps=(List<String>)resp;
612                else
613                {
614                        mob.tell(L("Unknown error."));
615                        return;
616                }
617                if((holidayNumber<=0)||(holidayNumber>=steps.size()))
618                {
619                        mob.tell(L("@x1 does not exist as a holiday -- enter LIST HOLIDAYS.",""+holidayNumber));
620                        return;
621                }
622
623                final String step=steps.get(holidayNumber);
624                final HolidayData encodedData=getEncodedHolidayData(step);
625                final TriadList<String,String,Integer> settings=encodedData.settings();
626                final TriadList<String,String,Integer> behaviors=encodedData.behaviors();
627                final TriadList<String,String,Integer> properties=encodedData.properties();
628                final TriadList<String,String,Integer> stats=encodedData.stats();
629
630                final int oldNameIndex=settings.indexOfFirst("NAME");
631                if((mob.isMonster())||(oldNameIndex<0))
632                        return;
633                final String oldName=settings.get(oldNameIndex).second;
634                boolean ok=false;
635                int showFlag=-1;
636                if(CMProps.getIntVar(CMProps.Int.EDITORTYPE)>0)
637                        showFlag=-999;
638                try
639                {
640                        while((mob.session()!=null)&&(!mob.session().isStopped())&&(!ok))
641                        {
642                                int showNumber=0;
643                                promptText(mob,settings,"NAME",++showNumber,showFlag,"Holiday Name","It's, well, a name.",false);
644                                showNumber=promptDuration(mob,settings,showNumber,showFlag);
645                                if(settings.indexOfFirst("AREAGROUP")>=0)
646                                        promptText(mob,settings,"AREAGROUP",++showNumber,showFlag,"Areas List (?)","Area names are space separated, and words grouped using double-quotes",false);
647                                if(settings.indexOfFirst("MOBGROUP")>=0)
648                                        promptText(mob,settings,"MOBGROUP",++showNumber,showFlag,"Mask for mobs that apply (?)",CMLib.masking().maskHelp("\n\r","disallow"),false);
649                                promptText(mob,properties,"MOOD",++showNumber,showFlag,"Mood setting (?)","NULL/Empty (to not use a Mood), or one of: FORMAL, POLITE, HAPPY, SAD, ANGRY, RUDE, MEAN, PROUD, GRUMPY, EXCITED, SCARED, LONELY",true);
650                                promptText(mob,behaviors,"AGGRESSIVE",++showNumber,showFlag,"Aggressive setting (?)",CMLib.help().getHelpText("Aggressive",mob,true)+"\n\r\n\r** NULL/Empty (to not use Aggressive **",true);
651                                showNumber=genPricing(mob,stats,++showNumber,showFlag);
652                                showNumber=genMudChat(mob,"MUDCHAT",behaviors,++showNumber,showFlag);
653                                showNumber=genBehaviors(mob,behaviors,++showNumber,showFlag);
654                                showNumber=genProperties(mob,properties,++showNumber,showFlag);
655                                if(showFlag<-900)
