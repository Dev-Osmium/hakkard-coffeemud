111        protected boolean parseMcpStart(final String s, final boolean[] exec, final Map<String,String> keyValuePairs)
112        {
113                McpParseStartState state = McpParseStartState.START;
114                String lastKey = "";
115                int startIndex = 0;
116                for(int i=0;i<s.length();i++)
117                {
118                        final char c=s.charAt(i);
119                        switch(state)
120                        {
121                        case FINISH_COMMAND:
122                                if(!Character.isWhitespace(c))
123                                {
124                                        startIndex=i;
125                                        state=McpParseStartState.IN_MCPKEY;
126                                }
127                                else
128                                {
129                                        Log.errOut("Invalid MCP "+state.toString()+": "+c+": "+s);
130                                        return false;
131                                }
132                                break;
133                        case FINISH_KEY:
134                                if(Character.isWhitespace(c))
135                                {
136                                        keyValuePairs.put(lastKey, "");
137                                        state = McpParseStartState.FINISH_VAL;
138                                }
139                                else
140                                {
141                                        if(c=='\"')
142                                        {
143                                                startIndex = i+1;
144                                                state = McpParseStartState.IN_QUOTEVAL;
145                                        }
146                                        else
147                                        {
148                                                startIndex = i;
149                                                state = McpParseStartState.IN_VAL;
150                                        }
151                                }
152                                break;
153                        case FINISH_MCPKEY:
154                                if(!Character.isWhitespace(c))
155                                {
156                                        startIndex = i;
157                                        state = McpParseStartState.IN_KEY;
158                                }
159                                exec[0]=true;
160                                break;
161                        case FINISH_VAL:
162                                if(!Character.isWhitespace(c))
163                                {
164                                        startIndex = i;
165                                        state = McpParseStartState.IN_KEY;
166                                }
167                                break;
168                        case IN_COMMAND:
169                                if(Character.isWhitespace(c))
170                                {
171                                        keyValuePairs.put(MCP_COMMAND_KEY(), s.substring(startIndex,i));
172                                        if(s.substring(startIndex,i).equalsIgnoreCase("mcp"))
173                                                state = McpParseStartState.FINISH_MCPKEY;
174                                        else
175                                                state = McpParseStartState.FINISH_COMMAND;
176                                }
177                                break;
178                        case FINISH_IN_KEY:
179                                if(!Character.isWhitespace(c))
180                                {
181                                        Log.errOut("Invalid MCP "+state.toString()+": "+c+": "+s);
182                                        return false;
183                                }
184                                else
185                                {
186                                        state = McpParseStartState.FINISH_KEY;
187                                }
188                                break;
189                        case IN_KEY:
190                                if(Character.isWhitespace(c))
191                                {
192                                        Log.errOut("Invalid MCP "+state.toString()+": "+c+": "+s);
193                                        return false;
194                                }
195                                else
196                                if(c==':')
197                                {
198                                        lastKey = s.substring(startIndex,i);
199                                        if(lastKey.trim().endsWith("*"))
200                                        {
201                                                exec[0]=false;
202                                        }
203                                        state = McpParseStartState.FINISH_IN_KEY;
204                                }
205                                break;
206                        case IN_MCPKEY:
207                                if(Character.isWhitespace(c))
208                                {
209                                        keyValuePairs.put(MCP_KEYSENT_KEY(), s.substring(startIndex,i));
210                                        state = McpParseStartState.FINISH_MCPKEY;
211                                }
212                                break;
213                        case IN_QUOTEVAL:
214                                if((c=='\"')&&(s.charAt(i-1)!='\\'))
215                                {
216                                        keyValuePairs.put(lastKey, s.substring(startIndex,i));
217                                        state = McpParseStartState.FINISH_VAL;
218                                }
219                                break;
220                        case IN_VAL:
221                                if(Character.isWhitespace(c))
222                                {
223                                        keyValuePairs.put(lastKey, s.substring(startIndex,i));
224                                        state = McpParseStartState.FINISH_VAL;
225                                }
226                                break;
227                        case START:
228                                if(!Character.isWhitespace(c))
229                                {
230                                        startIndex=i;
231                                        state=McpParseStartState.IN_COMMAND;
232                                }
233                                else
234                                {
235                                        Log.errOut("Invalid MCP "+state.toString()+": "+c+": "+s);
236                                        return false;
237                                }
238                                break;
239                        }
240                }
241                switch(state)
242                {
243                case IN_VAL:
244                        keyValuePairs.put(lastKey, s.substring(startIndex));
245                        break;
246                case FINISH_KEY:
247                        keyValuePairs.put(lastKey, "");
248                        break;
249                case IN_MCPKEY:
250                        keyValuePairs.put(MCP_KEYSENT_KEY(), s.substring(startIndex));
251                        exec[0]=true;
252                        break;
253                default:
254                        Log.errOut("Invalid MCP END "+state.toString()+": " + s);
255                        return false;
256                }
257                if((!keyValuePairs.containsKey(MCP_COMMAND_KEY()))
258                ||((!keyValuePairs.get(MCP_COMMAND_KEY()).equalsIgnoreCase("mcp"))
259                        &&(!keyValuePairs.containsKey(MCP_KEYSENT_KEY()))))
260                {
261                        Log.errOut("Invalid MCP -- missing Command or Key: " + s);
262                        return false;
263                }
264                if((!exec[0]) && (!keyValuePairs.containsKey(MCP_DATA_TAG())))
265                {
266                        Log.errOut("Missing MCP Data Tag: " + s);
267                        return false;
268                }
269                if((!exec[0]) && (!containsMcpStarTag(keyValuePairs)))
270                {
271                        Log.errOut("Missing MCP Star Tag: " + s);
272                        return false;
273                }
274                return true;
275        }

278        {
279                START,
280                IN_CONTMCPKEY,
281                FINISH_CONTMCPKEY,
282                IN_CONTKEY,
283                FINISH_IN_CONTKEY,
284                FINISH_CONTKEY,
285                IN_LINE

288        protected boolean parseMcpCont(final String s, final boolean[] exec, final Map<String,String> keyValuePairs)
289        {
290                McpParseContState state = McpParseContState.START;
291                String lastKey = "";
292                int startIndex = 0;
293                exec[0] = false; // never end on a continue tag
294                for(int i=1;i<s.length();i++)
295                {
296                        final char c=s.charAt(i);
297                        switch(state)
298                        {
299                        case FINISH_CONTKEY:
300                                startIndex = i;
301                                state = McpParseContState.IN_LINE;
302                                break;
303                        case FINISH_CONTMCPKEY:
304                                if(!Character.isWhitespace(c))
305                                {
306                                        startIndex = i;
307                                        state = McpParseContState.IN_CONTKEY;
308                                }
309                                break;
310                        case IN_CONTKEY:
311                                if(Character.isWhitespace(c))
312                                {
313                                        Log.errOut("Invalid MCPC "+state.toString()+": "+c+": "+s);
314                                        return false;
315                                }
316                                else
317                                if(c==':')
318                                {
319                                        lastKey = s.substring(startIndex,i);
320                                        if(!keyValuePairs.containsKey(lastKey+"*"))
321                                        {
322                                                Log.errOut("Unknown CONT KEY: "+lastKey+": "+state.toString()+": "+c+": "+s);
323                                                return false;
324                                        }
325                                        state = McpParseContState.FINISH_IN_CONTKEY;
326                                }
327                                break;
328                        case FINISH_IN_CONTKEY:
329                                if(!Character.isWhitespace(c))
330                                {
331                                        Log.errOut("Unknown/Invalid/Bad MCP: "+state.toString()+": "+c+": "+s);
332                                        return false;
333                                }
334                                state = McpParseContState.FINISH_CONTKEY;
335                                break;
336                        case IN_CONTMCPKEY:
337                                if(Character.isWhitespace(c))
338                                {
339                                        if((!keyValuePairs.containsKey(MCP_DATA_TAG()))
340                                        ||(!s.substring(startIndex,i).equals(keyValuePairs.get(MCP_DATA_TAG()))))
341                                        {
342                                                Log.errOut("Unknown/Invalid CONT MCP KEY: "+s.substring(startIndex,i)+": "+state.toString()+": "+c+": "+s);
343                                                return false;
344                                        }
345                                        state = McpParseContState.FINISH_CONTMCPKEY;
346                                }
347                                break;
348                        case START:
349                                if(!Character.isWhitespace(c))
350                                {
351                                        startIndex=i;
352                                        state=McpParseContState.IN_CONTMCPKEY;
353                                }
354                                break;
355                        case IN_LINE:
356                                break;
357                        default:
358                                break;
359                        }
360                }
361                switch(state)
362                {
363                case IN_LINE:
364                        if(keyValuePairs.containsKey(lastKey))
365                                keyValuePairs.put(lastKey, keyValuePairs.get(lastKey)+"\n\r"+s.substring(startIndex));
366                        else
367                                keyValuePairs.put(lastKey, s.substring(startIndex));
368                        break;
369                case FINISH_IN_CONTKEY:
370                case FINISH_CONTKEY:
371                        if(keyValuePairs.containsKey(lastKey))
372                                keyValuePairs.put(lastKey, keyValuePairs.get(lastKey)+"\n\r");
373                        else
374                                keyValuePairs.put(lastKey, "");
375                        break;
376                default:
377                        Log.errOut("Invalid MCP CONT "+state.toString()+": " + s);
378                        return false;
379                }
380                return true;
381        }

383        protected boolean parseMcpEnd(final String s, final boolean[] exec, final Map<String,String> keyValuePairs)
384        {
385                final String endKey = s.substring(1).trim();
386                if((!keyValuePairs.containsKey(MCP_COMMAND_KEY()))
387                ||((!keyValuePairs.get(MCP_COMMAND_KEY()).equalsIgnoreCase("mcp"))
388                        &&(!endKey.equalsIgnoreCase(keyValuePairs.get(MCP_DATA_TAG())))))
389                {
390                        Log.errOut("Unknown/Invalid CONT MCP KEY: "+endKey+": "+s);
391                        return false;
392                }
393                exec[0] = true;
394                return true;
395        }

397        @Override
398        @SuppressWarnings("unchecked")
399        public boolean mcp(final Session session, final StringBuilder str, final String[] mcpKey,
400                                           final Map<String,float[]> clientSupported, final Map<String,String> keyValuePairs)
401        {
402                Map<String,MCPPackage> mcpPackages = (Map<String,MCPPackage>)Resources.getResource("MCP_COMPILED_PACKAGES");
403                if(mcpPackages == null)
404                {
405                        synchronized(this)
406                        {
407                                mcpPackages = (Map<String,MCPPackage>)Resources.getResource("MCP_COMPILED_PACKAGES");
408                                if(mcpPackages == null)
409                                {
410                                        mcpPackages = new Hashtable<String,MCPPackage>();
411                                        final List<MCPPackage> pkgs = new ArrayList<MCPPackage>();
412                                        if(CMClass.loadObjectListToObj(pkgs, "com/planet_ink/coffee_mud/Libraries/mcppkgs/", CMProps.instance().getStr("MCPPACKAGES"), "com.planet_ink.coffee_mud.Libraries.interfaces.ProtocolLibrary$MCPPackage"))
413                                        {
414                                                for(final MCPPackage pkg : pkgs )
415                                                {
416                                                        mcpPackages.put(pkg.packageName(), pkg);
417                                                }
418                                        }
419                                        Resources.submitResource("MCP_COMPILED_PACKAGES", mcpPackages);
420                                }
421                        }
422                }
423
424                final String s = str.substring(3).trim();
425                final boolean[] execute = new boolean[1];
426                if(s.length()==0)
427                        return false;
428                if(s.charAt(0)=='*')
429                {
430                        if(!parseMcpCont(s,execute,keyValuePairs))
431                        {
432                                return false;
433                        }
434                }
435                else
436                if(s.charAt(0)==':')
437                {
438                        if(!parseMcpEnd(s,execute,keyValuePairs))
439                        {
440                                return false;
441                        }
442                }
443                else
444                {
445                        if(!parseMcpStart(s,execute,keyValuePairs))
446                        {
447                                return false;
448                        }
449                }
450                if(execute[0])
451                {
452                        if((!keyValuePairs.containsKey(MCP_COMMAND_KEY()))
453                        ||((!keyValuePairs.get(MCP_COMMAND_KEY()).equals("mcp"))
454                                &&(!keyValuePairs.containsKey(MCP_KEYSENT_KEY()))))
455                        {
456                                Log.errOut("Invalid MCP PROCESS -- no command or key sent: " + s);
457                                return false;
458                        }
459                        final String pkgCmd = keyValuePairs.get(MCP_COMMAND_KEY());
460                        final String keySent = keyValuePairs.get(MCP_KEYSENT_KEY());
461                        if((keySent != null)&&(mcpKey[0]!=null)&&(!keySent.equals(mcpKey[0])))
462                        {
463                                Log.errOut("Invalid MCP PROCESS -- invalid key sent: " + keySent+": "+mcpKey[0]+": "+s);
464                                return false;
465                        }
466                        else
467                        if(pkgCmd.equals("mcp"))
468                        {
469                                if(keyValuePairs.containsKey("authentication-key"))
470                                {
471                                        mcpKey[0] = keyValuePairs.get("authentication-key");
472                                }
473                                else
474                                {
475                                        Log.errOut("Invalid MCP PROCESS -- NO key Sent: "+s);
476                                        return false;
477                                }
478                                for(final String commandKey : mcpPackages.keySet())
479                                {
480                                        final MCPPackage pkg = mcpPackages.get(commandKey);
481                                        session.rawPrintln("#$#mcp-negotiate-can "+mcpKey[0]+" package: "+pkg.packageName()+" min-version: "+pkg.minVersion()+" max-version: "+pkg.maxVersion());
482                                }
483                                session.rawPrintln("#$#mcp-negotiate-end "+mcpKey[0]);
484                        }
485                        else
486                        {
487                                MCPPackage pkg = mcpPackages.get(pkgCmd);
488                                if(pkg == null)
489                                {
490                                        for(final String commandKey : mcpPackages.keySet())
491                                        {
492                                                if(pkgCmd.startsWith(commandKey+"-"))
493                                                {
494                                                        pkg = mcpPackages.get(commandKey);
495                                                        break;
496                                                }
497                                        }
498                                }
499                                if(pkg != null)
500                                {
501                                        pkg.executePackage(session, pkgCmd, clientSupported, keyValuePairs);
502                                }
503                        }
504                        keyValuePairs.clear();
505                }
506                return true;
507        }

509        @Override
510        public String[] mxpImagePath(String fileName)
511        {
512                if((fileName==null)||(fileName.trim().length()==0))
513                        return new String[]{"",""};
514                final String mxpImagePath=CMProps.getVar(Str.MXPIMAGEPATH);
515                if((mxpImagePath.length()==0)
516                ||(CMSecurity.isDisabled(CMSecurity.DisFlag.MXP)))
517                        return new String[]{"",""};
518                int x=fileName.lastIndexOf('=');
519                String preFilename="";
520                if(x>=0)
521                {
522                        preFilename=fileName.substring(0,x+1);
523                        fileName=fileName.substring(x+1);
524                }
525                x=fileName.lastIndexOf('/');
526                if(x>=0)
527                {
528                        preFilename+=fileName.substring(0,x+1);
529                        fileName=fileName.substring(x+1);
530                }
531                if(mxpImagePath.endsWith("/"))
532                        return new String[]{mxpImagePath+preFilename,fileName};
533                return new String[]{mxpImagePath+"/"+preFilename,fileName};
534        }

599        @Override @SuppressWarnings({"unchecked","rawtypes"})
600        public String getDefaultMXPImage(final Object O)
601        {
602                if((CMProps.getVar(Str.MXPIMAGEPATH).length()==0)
603                ||(CMSecurity.isDisabled(CMSecurity.DisFlag.MXP)))
604                        return "";
605                Map<String,String> H=(Map)Resources.getResource("PARSED: mxp_images.ini");
606                if(H==null)
607                {
608                        H=new Hashtable<String,String>();
609                        final List<String> V=Resources.getFileLineVector(new CMFile("resources/mxp_images.ini",null).text());
610                        if((V!=null)&&(V.size()>0))
611                        {
612                                String s=null;
613                                int x=0;
614                                for(int v=0;v<V.size();v++)
615                                {
616                                        s=V.get(v).trim();
617                                        if(s.startsWith("//")||s.startsWith(";"))
618                                                continue;
619                                        x=s.indexOf('=');
620                                        if(x<0)
621                                                continue;
622                                        if(s.substring(x+1).trim().length()>0)
623                                                H.put(s.substring(0,x),s.substring(x+1));
624                                }
625                        }
626                        Resources.submitResource("PARSED: mxp_images.ini",H);
627                }
628                String image=null;
629                if(O instanceof Race)
630                {
631                        image=getHashedMXPImage(H,"RACE_"+((Race)O).ID().toUpperCase());
632                        if(image==null)
633                                image=getHashedMXPImage(H,"RACECAT_"+((Race)O).racialCategory().toUpperCase().replace(' ','_'));
634                        if(image==null)
635                                image=getHashedMXPImage(H,"RACE_*");
636                        if(image==null)
637                                image=getHashedMXPImage(H,"RACECAT_*");
638                }
639                else
640                if(O instanceof MOB)
641                {
642                        final String raceName=((MOB)O).charStats().raceName();
643                        Race R=null;
644                        for(final Enumeration<Race> e=CMClass.races();e.hasMoreElements();)
645                        {
646                                R=e.nextElement();
647                                if(raceName.equalsIgnoreCase(R.name()))
648                                        image=getDefaultMXPImage(R);
649                        }
650                        if(image==null)
651                                image=getDefaultMXPImage(((MOB)O).charStats().getMyRace());
652                }
653                else
654                if(O instanceof Room)
655                {
656                        image=getHashedMXPImage(H,"ROOM_"+((Room)O).ID().toUpperCase());

995                if(o instanceof StringBuilder)

1016                else
1017                        return o;

1020        @SuppressWarnings({ "unchecked", "rawtypes" })
1021        protected Map<String,Object> buildMsdpMap(final char[] data, final int dataSize)
1022        {
1023                final Stack<Object> stack=new Stack<Object>();
1024                stack.push(new HashMap<StringBuilder,Object>());
1025                StringBuilder str=null;
1026                StringBuilder var=null;
1027                StringBuilder valVar=null;
1028                int x=-1;
1029                while(++x<dataSize)
1030                {
1031                        switch(data[x])
1032                        {
1033                        case Session.MSDP_VAR: // start a string
1034                                str=new StringBuilder("");
1035                                var=str;
1036                                if(stack.peek() instanceof Map)
1037                                        ((Map)stack.peek()).put(str, "");
1038                                else
1039                                if(stack.peek() instanceof List)
1040                                        ((List)stack.peek()).add(str);
1041                                break;
1042                        case Session.MSDP_VAL:
1043                        {
1044                                valVar=var;
1045                                var=null;
1046                                str=new StringBuilder("");
1047                                break;
1048                        }
1049                        case Session.MSDP_TABLE_OPEN: // open a table
1050                        {
1051                                final Map<StringBuilder,Object> M=new HashMap<StringBuilder,Object>();
1052                                if((stack.peek() instanceof Map)&&(valVar!=null))
1053                                        ((Map)stack.peek()).put(valVar, M);
1054                                else
1055                                if(stack.peek() instanceof List)
1056                                        ((List)stack.peek()).add(M);
1057                                valVar=null;
1058                                stack.push(M);
1059                                break;
1060                        }
1061                        case Session.MSDP_TABLE_CLOSE: // done with table
1062                                if((stack.size()>1)&&(stack.peek() instanceof Map))
1063                                        stack.pop();
1064                                break;
1065                        case Session.MSDP_ARRAY_OPEN: // open an array
1066                        {
1067                                final List<Object> M=new LinkedList<Object>();
1068                                if((stack.peek() instanceof Map)&&(valVar!=null))
1069                                        ((Map)stack.peek()).put(valVar, M);
1070                                else
1071                                if(stack.peek() instanceof List)
1072                                        ((List)stack.peek()).add(M);
1073                                valVar=null;
1074                                stack.push(M);
1075                                break;
1076                        }
1077                        case Session.MSDP_ARRAY_CLOSE: // close an array
1078                                if((stack.size()>1)&&(stack.peek() instanceof List))
1079                                        stack.pop();
1080                                break;
1081                        default:
1082                                if((stack.peek() instanceof Map)&&(valVar!=null))
1083                                        ((Map)stack.peek()).put(valVar, str);
1084                                else
1085                                if((stack.peek() instanceof List)&&(!((List)stack.peek()).contains(str)))
1086                                        ((List)stack.peek()).add(str);
1087                                valVar=null;
1088                                if(str!=null)
1089                                        str.append(data[x]);
1090                                break;
1091                        }
1092                }
1093                return (Map<String,Object>)msdpStringify(stack.firstElement());
1094        }
