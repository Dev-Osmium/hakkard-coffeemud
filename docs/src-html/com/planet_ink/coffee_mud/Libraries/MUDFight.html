136        public Set<MOB> allPossibleCombatants(final MOB mob, final boolean includePlayers)
137        {
138                final SHashSet<MOB> h=new SHashSet<MOB>();
139                final Room thisRoom=mob.location();
140                if(thisRoom==null)
141                        return null;
142                final Set<MOB> h1=mob.getGroupMembers(new HashSet<MOB>());
143                for(int m=0;m<thisRoom.numInhabitants();m++)
144                {
145                        final MOB inhab=thisRoom.fetchInhabitant(m);
146                        if((inhab!=null)
147                        &&(inhab!=mob)
148                        &&(!h1.contains(inhab))
149                        &&(CMLib.flags().isSeeable(inhab)||CMLib.flags().canMove(inhab))
150                        &&(CMLib.flags().isSeeable(inhab)||(!CMLib.flags().isCloaked(inhab)))
151                        &&((includePlayers)||(!mob.isMonster())||(!inhab.isMonster())))
152                                h.addUnsafe(inhab);
153                }
154                return h;
155        }

274        @Override
275        public boolean doTurnBasedCombat(final MOB mob, final Room R, final Area A)
276        {
277                int index = R.getCombatTurnMobIndex();
278                MOB M=null;
279                if(mob.actions() < 1.0)
280                {
281                        if((index >= R.numInhabitants())||((M=R.fetchInhabitant(index))==mob)||(M==null)||(!M.isInCombat()))
282                        {
283                                if((index<0)||(index>=R.numInhabitants()-1))
284                                        index=-1;
285                                for(index++;index<R.numInhabitants();index++)
286                                {
287                                        M=R.fetchInhabitant(index);
288                                        if((M!=null)&&(M.isInCombat()))
289                                        {
290                                                M.setActions((M.actions() - Math.floor(M.actions())) + (CMLib.flags().isSitting(M) ? M.phyStats().speed() / 2.0 : M.phyStats().speed()));
291                                                R.setCombatTurnMobIndex(index);
292                                                break;
293                                        }
294                                }
295                                return false;
296                        }
297                        else
298                        {
299                                return true;
300                        }
301                }
302                else
303                if((index < R.numInhabitants())&&((M=R.fetchInhabitant(index))!=null)&&(M!=mob)&&(M.isInCombat()))
304                        mob.setActions(mob.actions()-Math.floor(mob.actions()));
305                return false;
306        }

366                return rollToHit(adjustedAttackBonus(attacker,defender),adjustedArmor(defender),attackerFudgeBonusAmt);

369        @Override
370        public boolean rollToHit(final int attack, final int defence, final int adjustment)
371        {
372                double myArmor= -((double)defence);
373                if(myArmor==0)
374                        myArmor=1.0;
375                else
376                if(myArmor<0.0)
377                        myArmor=-CMath.div(1.0,myArmor);
378                double hisAttack=attack;
379                if(hisAttack==0.0)
380                        hisAttack=1.0;
381                else
382                if(hisAttack<0.0)
383                        hisAttack=-CMath.div(1.0,myArmor);
384                return CMLib.dice().normalizeAndRollLess((int)Math.round(50.0*(hisAttack/myArmor)) + adjustment);
385        }
