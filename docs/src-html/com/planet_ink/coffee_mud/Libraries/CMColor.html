074                @Override

076                {
077                        if(!(cs instanceof ColorState))

080                                && (((ColorState)cs).backgroundCode() == backgroundCode);

085                {

099        @Override
100        public final ColorState valueOf(final char fg, final char bg)
101        {
102                final Integer keyI=Integer.valueOf((bg * 65536) + fg);
103                if(cache.containsKey(keyI))
104                        return cache.get(keyI);
105                final ColorState newColorState = new ColorStateImpl(fg,bg);
106                cache.put(keyI,newColorState);
107                return newColorState;
108        }

110        protected int translateSingleCMCodeToANSIOffSet(final String code)
111        {
112                if(code.length()==0)
113                        return -1;
114                if(!code.startsWith("^"))
115                        return -1;
116                int i=code.length()-1;
117                while(i>=0)
118                {
119                        if(Character.isLetter(code.charAt(i)))
120                                return "krgybpcw".indexOf(Character.toLowerCase(code.charAt(i)));
121                        else
122                                i++;
123                }
124                return 3;
125        }

127        public String translateCMCodeToFGNumber(String code)
128        {
129                if(code.length()==0)
130                        return code;
131                if(!code.startsWith("^"))
132                        return code;
133                final int background=code.indexOf('|');
134                if(background>0)
135                        code=code.substring(0,background);
136                int bold=0;
137                for(int i=0;i<code.length();i++)
138                {
139                        if(Character.isLowerCase(code.charAt(i)))
140                                bold=1;
141                }
142                return bold+";"+(30+translateSingleCMCodeToANSIOffSet(code))+"m";
143        }

145        @Override
146        public String translateCMCodeToANSI(final String code)
147        {
148                if(code.length()==0)
149                        return code;
150                if(!code.startsWith("^"))
151                        return code;
152                final int background=code.indexOf('|');
153                int bold=0;
154                for(int i=0;i<code.length();i++)
155                {
156                        if(Character.isLowerCase(code.charAt(i)))
157                                bold=1;
158                }
159                final String finalColor;
160                if(background>0)
161                        finalColor= "\033["+(40+translateSingleCMCodeToANSIOffSet(code.substring(0,background)))+";"+bold+";"+(30+translateSingleCMCodeToANSIOffSet(code.substring(background+1)))+"m";
162                else
163                        finalColor = "\033["+bold+";"+(30+translateSingleCMCodeToANSIOffSet(code))+"m";
164                return finalColor;
165        }

167        @Override
168        public String translateANSItoCMCode(final String code)
169        {
170                if(code == null)
171                        return "";
172                if(code.length()==0)
173                        return code;
174                if(code.indexOf('^')==0)
175                        return code;
176                if(code.indexOf('|')>0)
177                        return code;
178                String code1=null;
179                String code2=null;
180                final boolean bold=(code.indexOf(";1;")>0)||(code.indexOf("[1;")>0);
181                for(int i=0;i<COLORS_INCARDINALORDER.length;i++)
182                {
183                        if((code1==null)&&(code.indexOf(""+(40+i))>0))
184                                code1="^"+Character.toUpperCase(COLORS_INCARDINALORDER[i].getCodeChar());
185                        if((code2==null)&&(code.indexOf(""+(30+i))>0))
186                                code2="^"+(bold?COLORS_INCARDINALORDER[i].getCodeChar():Character.toUpperCase(COLORS_INCARDINALORDER[i].getCodeChar()));
187                }
188                if((code1!=null)&&(code2!=null))
189                        return code1+"|"+code2;
190                else
191                if((code1==null)&&(code2!=null))
192                        return code2;
193                else
194                if((code1!=null)&&(code2==null))
195                        return code1;
196                else
197                        return "^W";
198        }

200        @Override
201        public String mixHTMLCodes(final String code1, final String code2)
202        {
203                String html=null;
204                if((code1==null)||(code1.length()==0))
205                        html=code2;
206                else
207                if((code2==null)||(code2.length()==0))
208                        html=code1;
209                else
210                if(code1.startsWith(" ")&&(code2.startsWith("<FONT")))
211                        html=code2+code1;
212                else
213                if(code2.startsWith(" ")&&(code1.startsWith("<FONT")))
214                        html=code1+code2;
215                else
216                if(code1.startsWith("<")&&(code2.startsWith("<")))
217                        html=code1+">"+code2;
218                else
219                if(!code1.startsWith("<"))
220                        html=code2;
221                else
222                        html=code1;
223                if(html.startsWith(" "))
224                        return "<FONT"+html;
225                return html;
226        }

228        @Override
229        public String mixColorCodes(final String code1, String code2)
230        {
231                if((code1==null)||(code1.length()==0))
232                        return code2;
233                if((code2==null)||(code2.length()==0))
234                        return code1;
235                if(code1.charAt(code1.length()-1)!=code2.charAt(code2.length()-1))
236                        return code1+code2;
237                if(code2.startsWith("\033["))
238                        code2=code2.substring("\033[".length());
239                return code1.substring(0,code1.length()-1)+";"+code2;
240        }

242        @Override
243        public CMMsg fixSourceFightColor(final CMMsg msg)
244        {
245                if(msg.sourceMessage()!=null)
246                        msg.setSourceMessage(CMStrings.replaceAll(msg.sourceMessage(),"^F","^f"));
247                if(msg.targetMessage()!=null)
248                        msg.setTargetMessage(CMStrings.replaceAll(msg.targetMessage(),"^F","^e"));
249                return msg;
250        }

252        protected final SpecialColor findCodeColor(final String name)
253        {
254                if(this.nameMap.size()==0)
255                {
256                        for(final SpecialColor code : SpecialColor.values())
257                        {
258                                nameMap.put(code.getCodeString(), code);
259                        }
260                }
261                if(this.nameMap.containsKey(name))
262                        return this.nameMap.get(name);
263                return (SpecialColor)CMath.s_valueOf(SpecialColor.class, name);
264        }
265
266        protected final char findCodeChar(final String name)
267        {
268                final SpecialColor code = findCodeColor(name);
269                if(code != null)
270                        return code.getCodeChar();
271                return ' ';
272        }
