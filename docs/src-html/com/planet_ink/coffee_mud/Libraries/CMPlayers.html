458        {

469        @Override
470        public int bumpPrideStat(final MOB mob, final AccountStats.PrideStat stat, final int amt)
471        {
472                if((amt != 0)&&(mob!=null))
473                {
474                        final PlayerStats pstats=mob.playerStats();
475                        if(pstats != null)
476                        {
477                                pstats.bumpPrideStat(stat, amt);
478                                adjustTopPrideStats(topPlayers,mob.Name(),stat,pstats);
479                                if(pstats.getAccount() != null)
480                                {
481                                        pstats.getAccount().bumpPrideStat(stat,amt);
482                                        adjustTopPrideStats(topAccounts,pstats.getAccount().getAccountName(),stat,pstats.getAccount());
483                                }
484                                return amt;
485                        }
486                }
487                return 0;
488        }

490        protected void adjustTopPrideStats(final List<Pair<String,Integer>>[][] topWhat, final String name, final AccountStats.PrideStat stat, final AccountStats astats)
491        {
492                for(final TimeClock.TimePeriod period : TimeClock.TimePeriod.values())
493                {
494                        final List<Pair<String,Integer>> top=topWhat[period.ordinal()][stat.ordinal()];
495                        if(top == null)
496                                continue;
497                        synchronized(top)
498                        {
499                                final int pVal=astats.getPrideStat(period, stat);
500                                if(pVal <= 0)
501                                        removePrideStat(top,name,0);
502                                else
503                                {
504                                        boolean found=false;
505                                        for(int i=0;i<top.size();i++)
506                                        {
507                                                if(top.get(i).first.equals(name))
508                                                {
509                                                        found=true;
510                                                        top.get(i).second=Integer.valueOf(pVal);
511                                                        break;
512                                                }
513                                                else
514                                                if(pVal > top.get(i).second.intValue())
515                                                {
516                                                        top.add(i,new Pair<String,Integer>(name,Integer.valueOf(pVal)));
517                                                        removePrideStat(top,name,i+1);
518                                                        found=true;
519                                                        break;
520                                                }
521                                        }
522                                        if((!found)&&(top.size()<PRIDE_TOP_SIZE))
523                                                top.add(new Pair<String,Integer>(name,Integer.valueOf(pVal)));
524                                }
525                        }
526                }
527        }

529        @Override
530        public void renamePlayer(final MOB mob, final String oldName)
531        {
532                final String newName = mob.Name();
533                CMLib.database().DBPlayerNameChange(oldName, newName);
534                for(final Enumeration<MOB> p=CMLib.players().players();p.hasMoreElements();)
535                {
536                        final MOB playerM=p.nextElement();
537                        if(playerM.getWorshipCharID().equalsIgnoreCase(oldName))
538                                playerM.setWorshipCharID(newName);
539                        if(playerM.getLiegeID().equalsIgnoreCase(oldName))
540                                playerM.setLiegeID(newName);
541                }
542                for(int q=0;q<CMLib.quests().numQuests();q++)
543                {
544                        final Quest Q=CMLib.quests().fetchQuest(q);
545                        if(Q.wasWinner(oldName))
546                        {
547                                Q.declareWinner("-"+oldName);
548                                Q.declareWinner(newName);
549                        }
550                }
551                final PlayerStats pstats = mob.playerStats();
552                if(pstats!=null)
553                {
554                        final PlayerAccount account = pstats.getAccount();
555                        if(account != null)
556                        {
557                                account.delPlayer(oldName);
558                                account.addNewPlayer(mob);
559                                CMLib.database().DBUpdateAccount(account);
560                                account.setLastUpdated(System.currentTimeMillis());
561                        }
562                        for(final Enumeration<Item> e=pstats.getExtItems().items(); e.hasMoreElements();)
563                        {
564                                final Item I=e.nextElement();
565                                if(I instanceof PrivateProperty)
566                                {
567                                        if(((PrivateProperty)I).getOwnerName().equalsIgnoreCase(oldName))
568                                                ((PrivateProperty)I).setOwnerName(newName);
569                                }
570                        }
571                }
572
573                for(final Enumeration<Room> r=CMLib.map().roomsFilled();r.hasMoreElements();)
574                {
575                        Room R=r.nextElement();
576                        if((R!=null)&&(R.roomID().length()>0))
577                        {
578                                synchronized(("SYNC"+R.roomID()).intern())
579                                {
580                                        R=CMLib.map().getRoom(R);
581                                        boolean changed=false;
582                                        for(final Enumeration<Ability> a=R.effects();a.hasMoreElements();)
583                                        {
584                                                final Ability A=a.nextElement();
585                                                if(A==null)
586                                                        continue;
587                                                if(A instanceof LandTitle)
588                                                {
589                                                        if(((LandTitle)A).getOwnerName().equals(oldName))
590                                                        {
591                                                                ((LandTitle)A).setOwnerName(newName);
592                                                                changed=true;
593                                                        }
594                                                }
595                                                else
596                                                if(A.text().equals(oldName))
597                                                {
598                                                        changed=true;
599                                                        A.setMiscText(newName);
600                                                }
601                                        }
602                                        if(changed)
603                                        {
604                                                CMLib.database().DBUpdateRoom(R);
605                                        }
606                                }
607                        }
608                }
609        }

611        @Override
612        public void obliteratePlayer(MOB deadMOB, final boolean deleteAssets, final boolean quiet)
613        {
614                if(deadMOB==null)
615                        return;
616                if(getPlayer(deadMOB.Name())!=null)
617                {
618                        deadMOB=getPlayer(deadMOB.Name());
619                        delPlayer(deadMOB);
620                }
621                for(final Session S : CMLib.sessions().allIterable())
622                {
623                        if((!S.isStopped())&&(S.mob()!=null)&&(S.mob().Name().equals(deadMOB.Name())))
624                                deadMOB=S.mob();
625                }
626                if(deadMOB.playerStats()!=null)
627                {
628                        final PlayerAccount A=deadMOB.playerStats().getAccount();
629                        if(A!=null)
630                                A.delPlayer(deadMOB);
631                }
632                final Room deadLoc=deadMOB.location();
633                if(deleteAssets)
634                {
635                        final CMMsg msg=CMClass.getMsg(deadMOB,null,CMMsg.MSG_RETIRE,(quiet)?null:L("A horrible death cry is heard throughout the land."));
636                        if(deadLoc!=null)
637                                deadLoc.send(deadMOB,msg);
638                        try
639                        {
640                                for(final Enumeration<Room> r=CMLib.map().rooms();r.hasMoreElements();)
641                                {
642                                        final Room R=r.nextElement();
643                                        if((R!=null)&&(R!=deadLoc))
644                                        {
645                                                if(R.okMessage(deadMOB,msg))
646                                                        R.sendOthers(deadMOB,msg);
647                                                else
648                                                {
649                                                        addPlayer(deadMOB);
650                                                        return;
651                                                }
652                                        }
653                                }
654                        }
655                        catch(final NoSuchElementException e)
656                        {
657                        }
658                }
659                final StringBuffer newNoPurge=new StringBuffer("");
660                final List<String> protectedOnes=Resources.getFileLineVector(Resources.getFileResource("protectedplayers.ini",false));
661                boolean somethingDone=false;
662                if((protectedOnes!=null)&&(protectedOnes.size()>0))
663                {
664                        for(int b=0;b<protectedOnes.size();b++)
665                        {
666                                final String B=protectedOnes.get(b);
667                                if(!B.equalsIgnoreCase(deadMOB.name()))
668                                        newNoPurge.append(B+"\n");
669                                else
670                                        somethingDone=true;
671                        }
672                        if(somethingDone)
673                                Resources.updateFileResource("::protectedplayers.ini",newNoPurge);
674                }
675
676                final PlayerStats pStats = deadMOB.playerStats();
677                if(pStats != null)
678                        pStats.getExtItems().delAllItems(true);
679                final List<String> channels=CMLib.channels().getFlaggedChannelNames(ChannelsLibrary.ChannelFlag.PLAYERPURGES);
680                for(int i=0;i<channels.size();i++)
681                {
682                        String name=deadMOB.Name();
683                        if((pStats != null)
684                        &&(pStats.getAccount()!=null))
685                                name+=" ("+pStats.getAccount().getAccountName()+")";
686                        CMLib.commands().postChannel(channels.get(i),deadMOB.clans(),CMLib.lang().fullSessionTranslation("@x1 has just been deleted.",name),true);
687                }
688                CMLib.coffeeTables().bump(deadMOB,CoffeeTableRow.STAT_PURGES);
689
690                CMLib.database().DBDeletePlayerOnly(deadMOB.Name());
691                deadMOB.delAllItems(false);
692                for(int i=0;i<deadMOB.numItems();i++)
693                {
694                        final Item I=deadMOB.getItem(i);
695                        if(I!=null)
696                                I.setContainer(null);
697                }
698                deadMOB.delAllItems(false);
699                CMLib.database().DBUpdatePlayerItems(deadMOB);
700                while(deadMOB.numFollowers()>0)
701                {
702                        final MOB follower=deadMOB.fetchFollower(0);
703                        if(follower!=null)
704                                follower.setFollowing(null);
705                }
706                if(deleteAssets)
707                {

1010                                                        p.second[stat.ordinal()]=CMath.s_int(statVal);
1011                                        }
1012                                }

1015                }

1139        private boolean autoPurge()
1140        {
1141                if(CMSecurity.isDisabled(CMSecurity.DisFlag.AUTOPURGE))
1142                        return true;
1143                final String mask=CMProps.getVar(CMProps.Str.AUTOPURGE);
1144                if(mask.hashCode() != this.autoPurgeHash)
1145                {
1146                        final int lastLevel=CMProps.getIntVar(CMProps.Int.LASTPLAYERLEVEL)+100;
1147                        final long[][] presorted=CMLib.utensils().compileConditionalRange(CMParms.parseCommas(mask.trim(),true), 2, 0, lastLevel);
1148                        autoPurgeDaysLevels=new long[lastLevel+1];
1149                        prePurgeLevels=new long[lastLevel+1];
1150                        for (int i = 0; i < autoPurgeDaysLevels.length; i++)
1151                                autoPurgeDaysLevels[i] = 0;
1152                        for (int i = 0; i < prePurgeLevels.length; i++)
1153                                prePurgeLevels[i] = 0;
1154                        for(int i=0;i<presorted.length;i++)
1155                        {
1156                                final long[] set=presorted[i];
1157                                if((set==null)||(set.length<2))
1158                                {
1159                                        Log.errOut("CMPlayers","Error in AUTOPURGE definition #"+(i+1)+" in coffeemud.ini file! Fix immediately!!");
1160                                        continue;
1161                                }
1162                                final long val=set[0];
1163                                if(set[0]<=0)
1164                                        continue;
1165                                final long prepurge=set[1];
1166                                long realVal=(val*TimeManager.MILI_DAY);
1167                                long purgePoint=realVal-(prepurge*TimeManager.MILI_DAY);
1168                                if(val <= 0)
1169                                {
1170                                        realVal = 0;
1171                                        purgePoint = 0;
1172                                }
1173                                if(autoPurgeDaysLevels[i]==0)
1174                                        autoPurgeDaysLevels[i]=realVal;
1175                                if(prePurgeLevels[i]==0)
1176                                        prePurgeLevels[i]=purgePoint;
1177                        }
1178                        this.autoPurgeHash=mask.hashCode();
1179                }
1180                setThreadStatus(serviceClient,"autopurge process");
1181                final List<PlayerLibrary.ThinPlayer> allUsers=CMLib.database().getExtendedUserList();
1182                List<String> protectedOnes=Resources.getFileLineVector(Resources.getFileResource("protectedplayers.ini",false));
1183                if(protectedOnes==null)
1184                        protectedOnes=new Vector<String>();
1185
1186                final List<String> warnedOnes=Resources.getFileLineVector(Resources.getFileResource("warnedplayers.ini",false));
1187                if(CMSecurity.isDebugging(CMSecurity.DbgFlag.AUTOPURGE))
1188                        Log.debugOut(serviceClient.getName(),"Autopurge process start. "+allUsers.size()+" users loaded, "+protectedOnes.size()+" protected, and "+warnedOnes.size()+" previously warned.");
1189
1190                final StringBuilder warnStr=new StringBuilder("");
1191                final Map<String,Long> warnMap=new TreeMap<String,Long>();

1198                                if(codedWarnStr.trim().length()>0)
1199                                {

1202                                        final int lastSpace=codedWarnStr.lastIndexOf(' ');
1203                                        final long warningDateTime=CMath.s_long(codedWarnStr.substring(lastSpace+1).trim());
1204                                        if((warningDateTime > 0)
1205                                        && (System.currentTimeMillis() < (warningDateTime + (10 * TimeManager.MILI_DAY))))

1209                                                        Log.debugOut(serviceClient.getName(),"Warn loaded: "+warnedName+" last warned on "+CMLib.time().date2String(warningDateTime));
1210                                                warnMap.put(warnedName, new Long(warningDateTime));

1290                                &&(foundWarningDateTime > 0)
1291                                &&((System.currentTimeMillis()-foundWarningDateTime)>TimeManager.MILI_DAY))
1292                                {
1293                                        final MOB M=getLoadPlayer(name);
1294                                        if((M!=null)&&(!CMSecurity.isASysOp(M))&&(!CMSecurity.isAllowedAnywhere(M, CMSecurity.SecFlag.NOPURGE)))
1295                                        {
1296                                                obliteratePlayer(M,true, true);
1297                                                M.destroy();
1298                                                Log.sysOut(serviceClient.getName(),"AutoPurged user "+name+". Last logged in "+(CMLib.time().date2String(userLastLoginDateTime))+".");
1299                                        }
1300                                }
1301                        }
1302                        else
1303                        if(CMSecurity.isDebugging(CMSecurity.DbgFlag.AUTOPURGE))
1304                                Log.debugOut(serviceClient.getName(),name+" last on "+CMLib.time().date2String(userLastLoginDateTime)+" will be warned on "+CMLib.time().date2String(warnDateTime)+" and purged on "+CMLib.time().date2String(purgeDateTime));
1305                        if(warnChanged)

1307                }

1309                // accounts!
1310                if((!CMSecurity.isDisabled(CMSecurity.DisFlag.PURGEACCOUNTS))&&(CMProps.getIntVar(CMProps.Int.ACCOUNTPURGEDAYS)>0))
1311                {
1312                        for(final Enumeration<PlayerAccount> pe=CMLib.players().accounts("",null); pe.hasMoreElements();)
1313                        {
1314                                final PlayerAccount PA=pe.nextElement();
1315                                if((PA.numPlayers() > 0)
1316                                ||(isProtected(protectedOnes, PA.getAccountName())))
1317                                        continue;
1318                                final long lastDateTimePurge = PA.getLastDateTime() + (TimeManager.MILI_DAY * CMProps.getIntVar(CMProps.Int.ACCOUNTPURGEDAYS));
1319                                final long lastUpdatedPurge = PA.getLastUpdated() + (TimeManager.MILI_DAY * CMProps.getIntVar(CMProps.Int.ACCOUNTPURGEDAYS));
1320                                final long accountExpPurge = PA.getAccountExpiration() + (TimeManager.MILI_DAY * CMProps.getIntVar(CMProps.Int.ACCOUNTPURGEDAYS));
1321                                long lastTime = lastDateTimePurge;
1322                                if(lastUpdatedPurge > lastTime)
1323                                        lastTime=lastUpdatedPurge;
1324                                if(accountExpPurge > lastTime)
1325                                        lastTime=accountExpPurge;
1326                                if(System.currentTimeMillis()>lastTime)
1327                                {
1328                                        Log.sysOut(serviceClient.getName(),"AutoPurged account "+PA.getAccountName()+".");
1329                                        CMLib.players().obliterateAccountOnly(PA);
1330                                }
1331                        }
1332                }
1333                return true;

1336        private void warnPrePurge(final MOB mob, long timeLeft)
1337        {
1338                // check for valid recipient
1339                if(mob==null)
1340                        return;
1341
1342                if((mob.playerStats()==null)
1343                ||(mob.playerStats().getEmail().length()==0)) // no email addy to forward TO
1344                {
1345                        if(CMSecurity.isDebugging(CMSecurity.DbgFlag.AUTOPURGE))
1346                                Log.debugOut(serviceClient.getName(),"Unable to warn "+mob.Name()+" due to lacking an email address.");
1347                        return;
1348                }
1349
1350                final boolean canReceiveRealEmail =
1351                                (mob.playerStats()!=null)
1352                                &&(mob.playerStats().getEmail().length()>0)
1353                                &&(mob.isAttributeSet(MOB.Attrib.AUTOFORWARD))
1354                                &&((mob.playerStats().getAccount()==null)
1355                                        ||(!mob.playerStats().getAccount().isSet(AccountFlag.NOAUTOFORWARD)));
1356                if(!canReceiveRealEmail)
1357                {
1358                        if(CMSecurity.isDebugging(CMSecurity.DbgFlag.AUTOPURGE))
1359                                Log.debugOut(serviceClient.getName(),mob.Name()+" Opting out of auto-forward.");
1360                }
1361
1362                if(CMSecurity.isDisabled(DisFlag.SMTPCLIENT))
1363                {
1364                        if(CMSecurity.isDebugging(CMSecurity.DbgFlag.AUTOPURGE))
1365                                Log.debugOut(serviceClient.getName(),"Unable to warn "+mob.Name()+" due to smtp disable.");
1366                        return;
1367                }
1368
1369                //  timeLeft is in millis
1370                final String from="AutoPurgeWarning";
1371                final String to=mob.Name();
1372                final String subj=CMProps.getVar(CMProps.Str.MUDNAME)+" Autopurge Warning: "+to;
1373                String textTimeLeft="";
1374                if(timeLeft<0)
1375                        timeLeft = 1000*60*60*24;
1376                if(timeLeft>(1000*60*60*24*2))
1377                {
1378                        final int days=(int)CMath.div((double)timeLeft,1000*60*60*24);
1379                        textTimeLeft = days + " days";
1380                }
1381                else
1382                {
1383                        final int hours=(int)CMath.div((double)timeLeft,1000*60*60);
1384                        textTimeLeft = hours + " hours";
1385                }
1386
1387                final String msg=L("Your character, @x1, is going to be autopurged by the system in @x2.  ",to,textTimeLeft)+
1388                                                 L("If you would like to keep this character active, please re-login.  This is an automated message, please do not reply.");
1389                CMLib.smtp().emailOrJournal(from, from, to, subj, CMLib.coffeeFilter().simpleOutFilter(msg));
1390        }
