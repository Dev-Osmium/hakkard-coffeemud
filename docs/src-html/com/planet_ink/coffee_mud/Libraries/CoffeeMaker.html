054        public String getGenMOBTextUnpacked(final MOB mob, final String newText)
055        {
056                if((newText!=null)&&((newText.length()>10)||newText.startsWith("%DBID>")))
057                {
058                        if(newText.startsWith("%DBID>"))
059                        {
060                                final int x=newText.indexOf('@');
061                                final String dbstr=CMLib.database().DBReadRoomMOBMiscText(newText.substring(6,x),
062                                                                                                                                  ((Object)mob).getClass().getName()+newText.substring(x).trim());

065                                Log.errOut("Unable to re-read mob data: "+newText);
066                                return null;
067                        }
068                        return newText;
069                }
070                return null;
071        }

092        @Override
093        public int envFlags(final Environmental E)
094        {
095                int f=0;
096                if(E instanceof Item)
097                {
098                        final Item item=(Item)E;
099                        if(!CMath.bset(item.basePhyStats().sensesMask(),PhyStats.SENSE_ITEMNODROP))
100                                f=f|1;
101                        if(!CMath.bset(item.basePhyStats().sensesMask(),PhyStats.SENSE_ITEMNOTGET))
102                                f=f|2;
103                        if(CMath.bset(item.basePhyStats().sensesMask(),PhyStats.SENSE_ITEMREADABLE))
104                                f=f|4;
105                        if(!CMath.bset(item.basePhyStats().sensesMask(),PhyStats.SENSE_ITEMNOREMOVE))
106                                f=f|8;
107                }
108                else
109                if(E instanceof Exit)
110                {
111                        final Exit exit=(Exit)E;
112                        if(exit.isReadable())
113                                f=f|4;
114                }
115                if(E instanceof CloseableLockable)
116                {
117                        f = f | 16; // new style flag
118                        final CloseableLockable container=(CloseableLockable)E;
119                        if(container.hasADoor())
120                                f=f|32;
121                        if(container.hasALock())
122                                f=f|64;
123                        if(container.defaultsClosed())
124                                f=f|128;
125                        if(container.defaultsLocked())
126                                f=f|256;
127                }
128                return f;
129        }

133        {
134                if(E instanceof Item)
135                {
136                        final Item item=(Item)E;
137                        // deprecated, but unfortunately, its here to stay.
138                        CMLib.flags().setDroppable(item,CMath.bset(f,1));
139                        CMLib.flags().setGettable(item,CMath.bset(f,2));
140                        CMLib.flags().setReadable(item,CMath.bset(f,4));
141                        CMLib.flags().setRemovable(item,CMath.bset(f,8));
142                }
143                else
144                if(E instanceof Exit)
145                {
146                        final Exit exit=(Exit)E;
147                        exit.setReadable(CMath.bset(f,4));
148                }
149
150                if(E instanceof CloseableLockable)
151                {
152                        final CloseableLockable container=(CloseableLockable)E;
153                        if((CMath.bset(f, 16))||(E instanceof Exit)) // this will be a 'new method' flag
154                        {
155                                final boolean hasDoor=CMath.bset(f,32);
156                                final boolean hasLock=CMath.bset(f,64);
157                                final boolean defaultsClosed=CMath.bset(f,128);
158                                final boolean defaultsLocked=CMath.bset(f,256);
159                                container.setDoorsNLocks(hasDoor,(!hasDoor)||(!defaultsClosed),defaultsClosed,hasLock,hasLock&&defaultsLocked,defaultsLocked);
160                        }
161                        else
162                        {
163                                final boolean hasDoor=CMath.bset(f,32);
164                                final boolean hasLock=CMath.bset(f,64);
165                                container.setDoorsNLocks(hasDoor,!hasDoor,hasDoor,hasLock,hasLock,hasLock);
166                        }
167                }
168        }

195                else
196                        return (E.isGeneric()?getGenPropertiesStr(E):"") + (fromTop?getOrdPropertiesStr(E):"");

210        protected String getOrdPropertiesStr(final Environmental E)
211        {
212                final XMLLibrary xml=CMLib.xml();
213                final StringBuilder str=new StringBuilder("");
214                if(E instanceof Room)
215                {
216                        str.append(xml.convertXMLtoTag("RCLIM", ((Room)E).getClimateTypeCode()));
217                        str.append(xml.convertXMLtoTag("RATMO", ((Room)E).getAtmosphereCode()));
218                        if(E instanceof GridLocale)
219                        {
220                                str.append(xml.convertXMLtoTag("XGRID",((GridLocale)E).xGridSize()));
221                                str.append(xml.convertXMLtoTag("YGRID",((GridLocale)E).yGridSize()));
222                        }
223                        if(E instanceof LocationRoom)
224                                str.append(xml.convertXMLtoTag("COREDIR",CMParms.toListString(((LocationRoom)E).getDirectionFromCore())));
225                        str.append(getExtraEnvPropertiesStr(E));
226                        str.append(getGenScripts((Room)E,false));
227                }
228                else
229                if(E instanceof Area)
230                {
231                        final Area myArea=(Area)E;
232                        final StringBuilder parentstr = new StringBuilder();
233                        final StringBuilder childrenstr = new StringBuilder();
234                        str.append(xml.convertXMLtoTag("ARCHP",myArea.getArchivePath()));
235                        final Area defaultParentArea=CMLib.map().getDefaultParentArea();
236                        for(final Enumeration<Area> e=myArea.getParents(); e.hasMoreElements();)
237                        {
238                                final Area A=e.nextElement();
239                                if((A!=defaultParentArea)||(A.getTimeObj()!=CMLib.time().globalClock()))
240                                {
241                                        parentstr.append("<PARENT>");
242                                        parentstr.append(xml.convertXMLtoTag("PARENTNAMED", A.name()));
243                                        parentstr.append("</PARENT>");
244                                }
245                        }
246                        str.append(xml.convertXMLtoTag("PARENTS",parentstr.toString()));
247                        for(final Enumeration<Area> e=myArea.getChildren(); e.hasMoreElements();)
248                        {
249                                final Area A=e.nextElement();
250                                childrenstr.append("<CHILD>");
251                                childrenstr.append(xml.convertXMLtoTag("CHILDNAMED", A.name()));
252                                childrenstr.append("</CHILD>");
253                        }
254                        str.append(xml.convertXMLtoTag("CHILDREN",childrenstr.toString()));
255                        str.append(getExtraEnvPropertiesStr(E));
256                        str.append(getGenScripts((Area)E,false));
257                        str.append(xml.convertXMLtoTag("AUTHOR",myArea.getAuthorID()));
258                        str.append(xml.convertXMLtoTag("CURRENCY",myArea.getCurrency()));
259                        if(myArea instanceof BoardableShip)
260                                str.append(xml.convertXMLtoTag("DISP",xml.parseOutAngleBrackets(myArea.displayText())));
261                        final Vector<String> V=new Vector<String>();
262                        String flag=null;
263                        for(final Enumeration<String> f=myArea.areaBlurbFlags();f.hasMoreElements();)
264                        {
265                                flag=f.nextElement();
266                                V.addElement((flag+" "+myArea.getBlurbFlag(flag)).trim());
267                        }
268                        str.append(xml.convertXMLtoTag("BLURBS",xml.getXMLList(V)));
269                        str.append(xml.convertXMLtoTag("AATMO",((Area)E).getAtmosphereCode()));
270                        if(E instanceof GridZones)
271                                str.append(xml.convertXMLtoTag("XGRID",((GridZones)E).xGridSize())
272                                                  +xml.convertXMLtoTag("YGRID",((GridZones)E).yGridSize()));
273                        if(E instanceof AutoGenArea)
274                        {
275                                str.append(xml.convertXMLtoTag("AGXMLPATH",xml.parseOutAngleBrackets(((AutoGenArea)E).getGeneratorXmlPath())));
276                                str.append(xml.convertXMLtoTag("AGAUTOVAR",xml.parseOutAngleBrackets(CMParms.toEqListString(((AutoGenArea)E).getAutoGenVariables()))));
277                        }
278                }
279                else
280                if(E instanceof Ability)
281                        str.append(xml.convertXMLtoTag("AWRAP",E.text()));
282                else
283                if(E instanceof Item)
284                {
285                        final Item I=(Item)E;
286                        str.append((((I instanceof Container)&&(((Container)I).capacity()>0))
287                                ?xml.convertXMLtoTag("IID",""+I):""));
288                        str.append(xml.convertXMLtoTag("IWORN",""+I.rawWornCode()));
289                        str.append(xml.convertXMLtoTag("ILOC",""+((I.container()!=null)?(""+I.container()):"")));
290                        str.append(xml.convertXMLtoTag("IUSES",""+I.usesRemaining()));
291                        str.append(xml.convertXMLtoTag("ILEVL",""+I.basePhyStats().level()));
292                        str.append(xml.convertXMLtoTag("IABLE",""+I.basePhyStats().ability()));
293                        str.append((E.isGeneric()?"":xml.convertXMLtoTag("ITEXT",""+I.text())));
294                }
295                else
296                if(E instanceof MOB)
297                {
298                        final MOB M=(MOB)E;
299                        str.append(xml.convertXMLtoTag("MLEVL",""+M.basePhyStats().level()));
300                        str.append(xml.convertXMLtoTag("MABLE",""+M.basePhyStats().ability()));
301                        str.append(xml.convertXMLtoTag("MREJV",""+M.basePhyStats().rejuv()));
302                        str.append(((E.isGeneric()?"":xml.convertXMLtoTag("MTEXT",""+M.text()))));
303                }
304                return str.toString();
305        }

381        protected String getPlayerExtraInventory(final MOB M)
382        {
383                final StringBuilder itemstr=new StringBuilder("");
384                final ItemCollection coll=M.playerStats().getExtItems();
385                final List<Item> finalCollection=new LinkedList<Item>();
386                final List<Item> extraItems=new LinkedList<Item>();
387                for(int i=coll.numItems()-1;i>=0;i--)
388                {
389                        final Item thisItem=coll.getItem(i);
390                        if((thisItem!=null)&&(!thisItem.amDestroyed()))
391                        {
392                                final Item cont=thisItem.ultimateContainer(null);
393                                if(cont.owner() instanceof Room)
394                                        finalCollection.add(thisItem);
395                        }
396                }
397                for(final Item thisItem : finalCollection)
398                {
399                        if(thisItem instanceof Container)
400                        {
401                                final List<Item> contents=((Container)thisItem).getDeepContents();
402                                for(final Item I : contents)
403                                        if(!finalCollection.contains(I))
404                                                extraItems.add(I);
405                        }
406                }
407                finalCollection.addAll(extraItems);
408                final HashSet<String> done=new HashSet<String>();
409                for(final Item thisItem : finalCollection)
410                {
411                        if(!done.contains(""+thisItem))
412                        {
413                                CMLib.catalog().updateCatalogIntegrity(thisItem);
414                                final Item cont=thisItem.ultimateContainer(null);
415                                final String roomID=((cont.owner()==null)&&(thisItem instanceof SpaceObject)&&(CMLib.map().isObjectInSpace((SpaceObject)thisItem)))?
416                                                ("SPACE."+CMParms.toListString(((SpaceObject)thisItem).coordinates())):CMLib.map().getExtendedRoomID((Room)cont.owner());
417                                itemstr.append("<ITEM>");
418                                itemstr.append(CMLib.xml().convertXMLtoTag("ICLASS",CMClass.classID(thisItem)));
419                                itemstr.append((thisItem instanceof Container)
420                                                ?CMLib.xml().convertXMLtoTag("IID",""+thisItem):"");
421                                itemstr.append(CMLib.xml().convertXMLtoTag("IDATA",getPropertiesStr(thisItem,true)));
422                                itemstr.append("<IROOM ID=\""+roomID+"\" EXPIRE="+thisItem.expirationDate()+" />");
423                                itemstr.append("</ITEM>");
424                                done.add(""+thisItem);
425                        }
426                }
427                return (CMLib.xml().convertXMLtoTag("EXTRAINV",itemstr.toString()));
428        }

430        protected void setPlayerExtraInventory(final MOB M, final List<XMLTag> buf)
431        {
432                final ItemCollection coll=M.playerStats().getExtItems();
433                final List<XMLLibrary.XMLTag> V=CMLib.xml().getContentsFromPieces(buf,"EXTRAINV");
434                if(V==null)
435                {
436                        Log.errOut("CoffeeMaker","Error parsing 'EXTRAINV' of "+identifier(M,null)+".  Load aborted");
437                        return;
438                }
439                final Hashtable<String,Container> IIDmap=new Hashtable<String,Container>();
440                final Hashtable<Item,String> LOCmap=new Hashtable<Item,String>();
441                for(int i=0;i<V.size();i++)
442                {
443                        final XMLTag iblk=V.get(i);
444                        if((!iblk.tag().equalsIgnoreCase("ITEM"))||(iblk.contents()==null))
445                        {
446                                Log.errOut("CoffeeMaker","Error parsing 'ITEM' of "+identifier(M,null)+".  Load aborted");
447                                return;
448                        }
449                        final Item newOne=CMClass.getItem(iblk.getValFromPieces("ICLASS"));
450                        if(newOne instanceof ArchonOnly)
451                                continue;
452                        if(newOne==null)
453                        {
454                                Log.errOut("CoffeeMaker","Unknown item "+iblk.getValFromPieces("ICLASS")+" on "+identifier(M,null)+", skipping.");
455                                continue;
456                        }
457                        final List<XMLLibrary.XMLTag> idat=iblk.getContentsFromPieces("IDATA");
458                        if(idat==null)
459                        {
460                                Log.errOut("CoffeeMaker","Error parsing 'ITEM DATA' of "+identifier(M,null)+".  Load aborted");
461                                return;
462                        }
463                        final String ILOC=CMLib.xml().getValFromPieces(idat,"ILOC");
464                        coll.addItem(newOne);
465
466                        final XMLLibrary.XMLTag irm=iblk.getPieceFromPieces("IROOM");
467                        if(irm ==null)
468                        {
469                                Log.errOut("CoffeeMaker","Error parsing 'ITEM IROOM' of "+identifier(M,null)+".  Load aborted");
470                                return;
471                        }
472                        final String roomID=irm.parms().get("ID");
473                        final long expirationDate=CMath.s_long(irm.parms().get("EXPIRE"));
474                        if(roomID.startsWith("SPACE.") && (newOne instanceof SpaceObject))
475                                CMLib.map().addObjectToSpace((SpaceObject)newOne,CMParms.toLongArray(CMParms.parseCommas(roomID.substring(6), true)));
476                        else
477                        {
478                                final Room itemR=CMLib.map().getRoom(roomID);
479                                if(itemR!=null)
480                                {
481                                        if(newOne instanceof BoardableShip)
482                                                ((BoardableShip)newOne).dockHere(itemR);
483                                        else
484                                                itemR.addItem(newOne);
485                                        newOne.setExpirationDate(expirationDate);
486                                }
487                        }
488                        if(ILOC.length()>0)
489                                LOCmap.put(newOne,ILOC);
490                        setPropertiesStr(newOne,idat,true);
491                        if(newOne instanceof Container)
492                                IIDmap.put(CMLib.xml().getValFromPieces(iblk.contents(),"IID"),(Container)newOne);
493                }
494                for(final Enumeration<Item> i=coll.items();i.hasMoreElements();)
495                {
496                        final Item item=i.nextElement();
497                        if(item!=null)
498                        {
499                                final String ILOC=LOCmap.get(item);
500                                if(ILOC!=null)
501                                        item.setContainer(IIDmap.get(ILOC));
502                        }
503                }
504
505        }

531                }
532
533                if(E instanceof Item)
534                {
535                        final Item item=(Item)E;
536                        text.append(
537                         xml.convertXMLtoTag("IDENT",item.rawSecretIdentity())
538                        +xml.convertXMLtoTag("VALUE",item.baseGoldValue())
539                        //+xml.convertXMLtoTag("USES",item.usesRemaining()) // handled 'from top' & in db
540                        +xml.convertXMLtoTag("MTRAL",item.material())
541                        +xml.convertXMLtoTag("READ",item.readableText())

926        protected String unpackRoomFromXML(final Area forceArea, final List<XMLTag> xml, final boolean andContent, final boolean andSave)
927        {
928                Area myArea;
929                String areaName;
930                if(forceArea!=null)
931                {
932                        myArea=forceArea;
933                        areaName = forceArea.Name();
934                }
935                else
936                {
937                        areaName = CMLib.xml().getValFromPieces(xml,"RAREA");
938                        myArea=CMLib.map().getArea(areaName);
939                }
940                if(myArea==null)
941                        return unpackErr("Room","null RAREA '"+areaName+"'",xml);
942                final String roomClass=CMLib.xml().getValFromPieces(xml,"RCLAS");
943                final Room newRoom=CMClass.getLocale(roomClass);
944                if(newRoom==null)
945                        return unpackErr("Room","null RCLAS '"+roomClass+"'",xml);
946                newRoom.setRoomID(CMLib.xml().getValFromPieces(xml,"ROOMID"));
947                if(newRoom.roomID().equals("NEW"))
948                        newRoom.setRoomID(myArea.getNewRoomID(newRoom,-1));
949                if((forceArea==null) && CMLib.map().getRoom(newRoom.roomID())!=null)
950                        return "Room Exists: "+newRoom.roomID();
951                newRoom.setArea(myArea);
952                if(andSave)
953                        CMLib.database().DBCreateRoom(newRoom);
954                newRoom.setDisplayText(CMLib.xml().getValFromPieces(xml,"RDISP"));
955                newRoom.setDescription(CMLib.xml().getValFromPieces(xml,"RDESC"));
956                newRoom.setMiscText(CMLib.xml().restoreAngleBrackets(CMLib.xml().getValFromPieces(xml,"RTEXT")));
957
958                // now EXITS!
959                final List<XMLLibrary.XMLTag> xV=CMLib.xml().getContentsFromPieces(xml,"ROOMEXITS");
960                if(xV==null)
961                        return unpackErr("Room","null 'ROOMEXITS' in room "+newRoom.roomID(),xml);
962                for(int x=0;x<xV.size();x++)
963                {
964                        final XMLTag xblk=xV.get(x);
965                        if((!xblk.tag().equalsIgnoreCase("REXIT"))||(xblk.contents()==null))
966                                return unpackErr("Room","??"+xblk.tag()+" in room "+newRoom.roomID(),xblk);
967                        final int dir=xblk.getIntFromPieces("XDIRE");
968                        final String doorID=xblk.getValFromPieces("XDOOR");
969                        if((dir<0)||(dir>=Directions.NUM_DIRECTIONS()))
970                        {
971                                if((dir>255)&&(!(newRoom instanceof GridLocale)))
972                                        return unpackErr("Room","Not GridLocale, tried "+dir+" exit for room '"+newRoom.roomID()+"'",xblk);
973                                else
974                                if(dir>255)
975                                {
976                                        final String xdata=xblk.getValFromPieces("XDATA");
977                                        final List<String> CEs=CMParms.parseSemicolons(xdata.trim(),true);
978                                        for(int ces=0;ces<CEs.size();ces++)
979                                        {
980                                                final Vector<String> SCE=CMParms.parse(CEs.get(ces).trim());
981                                                final GridLocale.CrossExit CE=new GridLocale.CrossExit();
982                                                if(SCE.size()<3)
983                                                        continue;
984                                                CE.x=CMath.s_int(SCE.elementAt(0));
985                                                CE.y=CMath.s_int(SCE.elementAt(1));

1618                        {

1620                                ||(end2<=start)

1625                                        end--;

1627                                }
1628                        }
1629                }
1630                final StringBuffer str=new StringBuffer("*1>");
