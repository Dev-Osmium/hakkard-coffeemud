208        public int getTotalBonusStatPoints(final PlayerStats playerStats, final PlayerAccount account)
209        {
210                final int basemax = CMProps.getIntVar(CMProps.Int.BASEMAXSTAT);
211                final int basemin = CMProps.getIntVar(CMProps.Int.BASEMINSTAT);
212
213                int points = CMProps.getIntVar(CMProps.Int.MAXSTAT);
214                // Make sure there are enough points
215                if (points < ((basemin + 1) * CharStats.CODES.BASECODES().length))
216                        points = (basemin + 1) * CharStats.CODES.BASECODES().length;
217
218                // Make sure there aren't too many points
219                if (points > (basemax - 1) * CharStats.CODES.BASECODES().length)
220                        points = (basemax - 1) * CharStats.CODES.BASECODES().length;
221
222                if(playerStats != null)
223                        points += playerStats.getBonusCharStatPoints();
224                if(account != null)
225                        points += account.getBonusCharStatPoints();
226
227                // Subtract stat minimums from point total to get distributable points
228                return points - (basemin * CharStats.CODES.BASECODES().length);
229        }

233        {
234                final int basemax = CMProps.getIntVar(CMProps.Int.BASEMAXSTAT);
235
236                final int[] stats=new int[CharStats.CODES.BASECODES().length];
237                for(int i=0;i<stats.length;i++)
238                        stats[i]=baseCharStats.getStat(CharStats.CODES.BASECODES()[i]);
239
240                while (pointsLeft > 0)
241                {
242                        final int whichNum = CMLib.dice().roll(1,CharStats.CODES.BASECODES().length,-1);
243                        if(stats[whichNum]<basemax)
244                        {
245                                stats[whichNum]++;
246                                --pointsLeft;
247                        }
248                }
249
250                for(int i=0;i<stats.length;i++)
251                        baseCharStats.setStat(CharStats.CODES.BASECODES()[i],stats[i]);

350        public boolean isOkName(String login, final boolean spacesOk)
351        {
352                if(login.length()>20)
353                        return false;
354                if(login.length()<3)
355                        return false;
356
357                if((!spacesOk)&&(login.trim().indexOf(' ')>=0))
358                        return false;
359
360                login=login.toUpperCase().trim();
361                final Vector<String> V=CMParms.parse(login);
362                for(int v=V.size()-1;v>=0;v--)
363                {
364                        final String str=V.elementAt(v);
365                        if((" THE A AN ").indexOf(" "+str+" ")>=0)
366                                V.removeElementAt(v);
367                }
368
369                if(isBadName(login))
370                        return false;
371
372                for(int c=0;c<login.length();c++)
373                {
374                        if((login.charAt(c)!=' ')&&(!Character.isLetter(login.charAt(c))))
375                                return false;
376                }
377
378                for(final Enumeration<Deity> d=CMLib.map().deities();d.hasMoreElements();)
379                {
380                        final MOB D=d.nextElement();
381                        if((CMLib.english().containsString(D.ID(),login))
382                        ||(CMLib.english().containsString(D.Name(),login)))
383                                return false;
384                }
385                for(final Enumeration<MOB> m=CMClass.mobTypes();m.hasMoreElements();)
386                {
387                        final MOB M=m.nextElement();
388                        if((CMLib.english().containsString(M.Name(),login))
389                        ||(CMLib.english().containsString(M.name(),login)))
390                                return false;
391                }
392
393                for(final Enumeration<Clan> e=CMLib.clans().clans();e.hasMoreElements();)
394                {
395                        final Clan C=e.nextElement();
396                        if((CMLib.english().containsString(C.clanID(),login))
397                        ||(CMLib.english().containsString(C.name(),login)))
398                                return false;
399                }
400
401                return !CMSecurity.isBanned(login);
402        }

575        private Map<String,List<String>> getLoginScripts()
576        {
577                final Hashtable<String,List<String>> extraScripts=new Hashtable<String,List<String>>();
578                final String[] VALID_SCRIPT_CODES={"PASSWORD","EMAIL","ANSI","THEME","RACE","GENDER","STATS","CLASS","FACTIONS","END"};
579                final List<String> extras=CMParms.parseCommas(CMProps.getVar(CMProps.Str.CHARCREATIONSCRIPTS),true);
580                for(int e=0;e<extras.size();e++)
581                {
582                        String s=extras.get(e);
583                        final int x=s.indexOf(':');
584                        String code="END";
585                        if(x>0)
586                        {
587                                code=s.substring(0,x).toUpperCase().trim();
588                                boolean found=false;
589                                for (final String element : VALID_SCRIPT_CODES)
590                                {
591                                        if(element.equals(code))
592                                        {
593                                                code=element;
594                                                found=true;
595                                                break;
596                                        }
597                                        else
598                                        if(element.startsWith(code))
599                                        {
600                                                code=element;
601                                                found=true;
602                                                break;
603                                        }
604                                }
605                                if(!found)
606                                {
607                                        Log.errOut("CharCreation","Error in CHARCREATIONSCRIPTS, invalid code: "+code);
608                                        continue;
609                                }
610                                s=s.substring(x+1);
611                        }
612                        List<String> V=extraScripts.get(code);
613                        if(V==null)
614                        {
615                                V=new Vector<String>();
616                                extraScripts.put(code,V);
617                        }
618                        V.add(s.trim());
619                }
620                return extraScripts;
621        }
