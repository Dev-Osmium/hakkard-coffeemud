061        public boolean isCommonClanRelations(final Clan C1, final Clan C2, final int relation)
062        {
063                if((C1==null)||(C2==null))
064                        return relation==Clan.REL_NEUTRAL;
065                int i1=C1.getClanRelations(C2.clanID());
066                int i2=C2.getClanRelations(C1.clanID());
067                if((i1==i2)
068                &&((i1==Clan.REL_WAR)
069                   ||(i1==Clan.REL_ALLY)))
070                   return i1==relation;
071                for(final Enumeration<Clan> e=clans();e.hasMoreElements();)
072                {
073                        final Clan C=e.nextElement();
074                        if((C!=C1)&&(C!=C2))
075                        {
076                                if((i1!=Clan.REL_WAR)
077                                &&(C1.getClanRelations(C.clanID())==Clan.REL_ALLY)
078                                &&(C.getClanRelations(C2.clanID())==Clan.REL_WAR))
079                                        i1=Clan.REL_WAR;
080                                if((i2!=Clan.REL_WAR)
081                                &&(C2.getClanRelations(C.clanID())==Clan.REL_ALLY)
082                                &&(C.getClanRelations(C1.clanID())==Clan.REL_WAR))
083                                        i2=Clan.REL_WAR;
084                        }
085                }
086                if(i1==i2)
087                        return relation==i1;
088
089                if(Clan.REL_NEUTRALITYGAUGE[i1]<Clan.REL_NEUTRALITYGAUGE[i2])
090                        return relation==i1;
091                return relation==i2;
092        }

273        public int getClanRelations(final Clan C1, final Clan C2)
274        {
275                if((C1==null)||(C2==null))
276                        return Clan.REL_NEUTRAL;
277                final int i1=C1.getClanRelations(C2.clanID());
278                final int i2=C2.getClanRelations(C1.clanID());
279                final int rel=Clan.RELATIONSHIP_VECTOR[i1][i2];
280                if(rel==Clan.REL_WAR)
281                        return Clan.REL_WAR;
282                if(rel==Clan.REL_ALLY)
283                        return Clan.REL_ALLY;
284                for(final Enumeration<Clan> e=clans();e.hasMoreElements();)
285                {
286                        final Clan C=e.nextElement();
287                        if((C!=C1)
288                        &&(C!=C2)
289                        &&(((C1.getClanRelations(C.clanID())==Clan.REL_ALLY)&&(C.getClanRelations(C2.clanID())==Clan.REL_WAR)))
290                                ||((C2.getClanRelations(C.clanID())==Clan.REL_ALLY)&&(C.getClanRelations(C1.clanID())==Clan.REL_WAR)))
291                                        return Clan.REL_WAR;
292                }
293                return rel;
294        }

415                for(final Enumeration<Tattoo> e=M.tattoos();e.hasMoreElements();)

417                        final Tattoo T=e.nextElement();
418                        for(final String familyTattoo : familyTattoos)
419                        {
420                                if(T.getTattooName().startsWith(familyTattoo))
421                                {
422                                        final String name=T.getTattooName().substring(familyTattoo.length());
423                                        final MOB M2=CMLib.players().getLoadPlayer(name.toLowerCase());
424                                        if((M2 != null)&&isFamilyOfMembership(M2,members))
425                                                return true;
426                                }
427                        }

429                return false;

587        @Override
588        public boolean goForward(final MOB mob, final Clan C, final List<String> commands, final Clan.Function function, final boolean voteIfNecessary)
589        {
590                if((mob==null)||(C==null))
591                        return false;
592                final Pair<Clan,Integer> clanRole=mob.getClanRole(C.clanID());
593                if(clanRole==null)
594                        return false;
595                final int role=clanRole.second.intValue();
596                final Clan.Authority allowed=C.getAuthority(role,function);
597                if(allowed==Clan.Authority.CAN_DO)
598                        return true;
599                if(allowed==Clan.Authority.CAN_NOT_DO)
600                        return false;
601                if(function==Clan.Function.ASSIGN)
602                {
603                        if(C.getAuthority(role,Clan.Function.VOTE_ASSIGN)!=Clan.Authority.CAN_DO)
604                                return false;
605                }
606                else
607                if(C.getAuthority(role,Clan.Function.VOTE_OTHER)!=Clan.Authority.CAN_DO)
608                        return false;
609                if(!voteIfNecessary)
610                        return true;
611                final String matter=CMParms.combine(commands,0);
612                int numFound=0;
613                for(final Enumeration<Clan.ClanVote> e=C.votes();e.hasMoreElements();)
614                {
615                        final Clan.ClanVote CV=e.nextElement();
616                        if((CV.voteStarter.equalsIgnoreCase(mob.Name()))
617                        &&(CV.voteStatus==Clan.VSTAT_STARTED)
618                        &&(++numFound>5))
619                        {
620                                mob.tell(L("This matter must be voted upon, but you already have started enough votes for one person."));
621                                return false;
622                        }
623                        if((CV.matter.equalsIgnoreCase(matter)
624                        &&(CV.voteStatus==Clan.VSTAT_STARTED)))
625                        {
626                                mob.tell(L("This matter must be voted upon, and is already BEING voted upon.  Use CLANVOTE to see."));
627                                return false;
628                        }
629                }
630                if(mob.session()==null)
631                        return false;
632                try
633                {
634                        final int numVotes=C.getNumVoters(function);
635                        if(numVotes==1)
636                                return true;
637
638                        if(mob.session().confirm(L("This matter must be voted upon.  Would you like to start the vote now (y/N)?"),"N"))
639                        {
640                                final Clan.ClanVote CV=new Clan.ClanVote();
641                                CV.matter=matter;
642                                CV.voteStarter=mob.Name();
643                                CV.function=function.ordinal();
644                                CV.voteStarted=System.currentTimeMillis();
645                                CV.votes=new PairVector<String,Boolean>();
646                                CV.voteStatus=Clan.VSTAT_STARTED;
647                                C.addVote(CV);
648                                C.updateVotes();
649                                final Clan.Function voteFunctionType = (function == Clan.Function.ASSIGN) ? Clan.Function.VOTE_ASSIGN : Clan.Function.VOTE_OTHER;
650                                final List<Integer> votingRoles = new Vector<Integer>();
651                                for(int i=0;i<C.getRolesList().length;i++)
652                                {
653                                        if(C.getAuthority(i, voteFunctionType)==Clan.Authority.CAN_DO)
654                                                votingRoles.add(Integer.valueOf(i));
655                                }
656                                if(votingRoles.size()>0)
657                                {
658                                        final String firstRoleName = C.getRoleName(votingRoles.iterator().next().intValue(), true, true);
659                                        final String rest = " "+firstRoleName+" should use CLANVOTE to participate.";
660                                        if(votingRoles.size() >= (C.getRolesList().length-2))
661                                        {
662                                                if(function == Clan.Function.ASSIGN)
663                                                        clanAnnounce(mob,"The "+C.getGovernmentName()+" "+C.clanID()+" has a new election to vote upon. "+rest);
664                                                else
665                                                        clanAnnounce(mob,"The "+C.getGovernmentName()+" "+C.clanID()+" has a new matter to vote upon. "+rest);
666                                        }
667                                        else
668                                        if(votingRoles.size()==1)
669                                                clanAnnounce(mob,"The "+C.getGovernmentName()+" "+C.clanID()+" has a new matter to vote upon. "+rest);
670                                        else
671                                        {
672                                                final String[] roleNames = new String[votingRoles.size()];
673                                                for(int i=0;i<votingRoles.size();i++)
674                                                {
675                                                        final Integer roleID=votingRoles.get(i);
676                                                        roleNames[i]=C.getRoleName(roleID.intValue(), true, true);
677                                                }
678                                                final String list = CMLib.english().toEnglishStringList(roleNames);
679                                                clanAnnounce(mob,"The "+C.getGovernmentName()+" "+C.clanID()+" has a new matter to vote upon. "
680                                                                +list+" should use CLANVOTE to participate.");
681                                        }
682                                }
683                                mob.tell(L("Your vote has started.  Use CLANVOTE to cast your vote."));
684                                return false;
685                        }
686                }
687                catch(final java.io.IOException e)
688                {
689                }
690                mob.tell(L("Without a vote, this command can not be executed."));
