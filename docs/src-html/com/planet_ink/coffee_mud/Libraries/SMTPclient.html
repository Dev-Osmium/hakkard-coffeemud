051        Attribute doMXLookup( String hostName )
052        {
053                try
054                {
055                        final Hashtable<String,String> env = new Hashtable<String,String>();
056                        env.put("java.naming.factory.initial",
057                                        "com.sun.jndi.dns.DnsContextFactory");
058                        final DirContext ictx = new InitialDirContext( env );
059                        final Attributes attrs = ictx.getAttributes( hostName, new String[] { "MX" });
060                        final Attribute attr = attrs.get( "MX" );
061                        if( attr == null )
062                                return( null );
063                        return( attr );
064                }
065                catch(final javax.naming.NamingException x)
066                {
067                }
068                return null;
069        }

090        /**
091         *   Create a SMTP object pointing to the specified host
092         *   @param SMTPServerInfo the host to connect to.
093         *   @param port the port to connect to.
094         *   @throws UnknownHostException the host was unknown
095         *   @throws IOException a socket error
096         */
097        public SMTPclient( String SMTPServerInfo, int port) throws UnknownHostException,IOException
098        {
099                auth = new SMTPHostAuth(SMTPServerInfo);
100                final int portIndex=auth.host.lastIndexOf(':');
101                if(portIndex>0)
102                {
103                        port=CMath.s_int(auth.host.substring(portIndex+1));
104                        auth.host=auth.host.substring(0,portIndex);
105                }
106                sock = new Socket( auth.getHost(), port );
107                reply = new BufferedReader(new InputStreamReader(sock.getInputStream()));
108                sock.setSoTimeout(DEFAULT_TIMEOUT);
109                send = new PrintWriter( sock.getOutputStream() );
110                final boolean debug = CMSecurity.isDebugging(CMSecurity.DbgFlag.SMTPCLIENT);
111                String rstr = reply.readLine();
112                if(debug)
113                        Log.debugOut("SMTPclient",rstr);
114                if ((rstr==null)||(!rstr.startsWith("220"))) throw new ProtocolException(rstr);
115                while (rstr.indexOf('-') == 3)
116                {
117                        rstr = reply.readLine();
118                        if(debug)
119                                Log.debugOut("SMTPclient",rstr);
120                        if (!rstr.startsWith("220")) throw new ProtocolException(rstr);
121                }
122        }

124        public SMTPclient (String emailAddress) throws IOException,
125                                                                                                   BadEmailAddressException
126        {
127                int x=this.getEmailAddressError(emailAddress);
128                if(x>=0)
129                        throw new BadEmailAddressException("Malformed email address");
130                x=emailAddress.indexOf('@');
131                final String domain=emailAddress.substring(x+1).trim();
132                final Vector<String> addys=new Vector<String>();
133                final Attribute mx=doMXLookup(domain);
134                boolean connected=false;
135                try
136                {
137                        if((mx!=null)&&(mx.size()>0))
138                        for(final NamingEnumeration<?> e=mx.getAll();e.hasMore();)
139                                addys.addElement((String)e.next());
140                }
141                catch(final javax.naming.NamingException ne)
142                {
143                }
144                if(addys.size()==0)
145                        addys.addElement(domain);
146                for (String hostid : addys)
147                {
148                        final int y=hostid.lastIndexOf(' ');
149                        if(y>=0)
150                                hostid=hostid.substring(y+1).trim();
151                        try
152                        {
153                                sock = new Socket( hostid, DEFAULT_PORT );
154                                reply = new BufferedReader(new InputStreamReader(sock.getInputStream()));
155                                sock.setSoTimeout(DEFAULT_TIMEOUT);
156                                send = new PrintWriter( sock.getOutputStream() );
157                                final boolean debug = CMSecurity.isDebugging(CMSecurity.DbgFlag.SMTPCLIENT);
158                                String rstr = reply.readLine();
159                                if(debug)
160                                        Log.debugOut("SMTPclient",rstr);
161                                if ((rstr==null)||(!rstr.startsWith("220"))) throw new ProtocolException(rstr);
162                                while (rstr.indexOf('-') == 3)
163                                {
164                                        rstr = reply.readLine();
165                                        if(debug)
166                                                Log.debugOut("SMTPclient",rstr);
167                                        if (!rstr.startsWith("220")) throw new ProtocolException(rstr);
168                                }
169                                connected=true;
170                                break;
171                        }
172                        catch(final Exception ex)
173                        {
174                                // just try the next one.
175                        }
176                }
177                if(!connected)
178                        throw new IOException("Unable to connect to '"+domain+"'.");
179        }

181        @Override
182        public boolean emailIfPossible(String fromName, String toName, String subj, String msg)
183        {
184                try
185                {
186                        SMTPLibrary.SMTPClient SC=null;
187                        final String toEmail=makeValidEmailAddress(toName);
188                        if(!isValidEmailAddress(toEmail))
189                        {
190                                Log.errOut("SMTPClient","User/Account not found: "+toName);
191                                return false;
192                        }
193                        if(CMProps.getVar(CMProps.Str.SMTPSERVERNAME).length()>0)
194                                SC=CMLib.smtp().getClient(CMProps.getVar(CMProps.Str.SMTPSERVERNAME),SMTPLibrary.DEFAULT_PORT);
195                        else
196                                SC=CMLib.smtp().getClient(toEmail);
197
198                        final String domain=CMProps.getVar(CMProps.Str.MUDDOMAIN).toLowerCase();
199                        SC.sendMessage(fromName+"@"+domain,
200                                                   fromName+"@"+domain,
201                                                   toEmail,
202                                                   toEmail,
203                                                   subj,
204                                                   CMLib.coffeeFilter().simpleOutFilter(msg));
205                        return true;
206                }
207                catch(final Exception e)
208                {
209                        Log.errOut("SMTPClient",e.getMessage());
210                        return false;
211                }
212        }

250        public void emailOrJournal(String from, String replyTo, String to, String subject, String message)
251        {
252                final String smtpServerInfo = CMProps.getVar(CMProps.Str.SMTPSERVERNAME);
253                final String fromEmail=(from.indexOf('@')>0)?makeValidEmailAddress(from):(from+'@'+CMProps.getVar(Str.MUDDOMAIN).toLowerCase());
254                final String toEmail=makeValidEmailAddress(to);
255                final String replyToEmail=(replyTo.indexOf('@')>0)?makeValidEmailAddress(replyTo):(replyTo+'@'+CMProps.getVar(Str.MUDDOMAIN).toLowerCase());
256                final String unSubUrl;
257                if(!isValidEmailAddress(to))
258                        unSubUrl = CMLib.utensils().getUnsubscribeURL(to);
259                else
260                        unSubUrl = null;
261                message+=L("\n\r\n\rThis message was sent through the @x1 mail server at @x2, port @x3.  ",
262                                CMProps.getVar(CMProps.Str.MUDNAME), CMProps.getVar(CMProps.Str.MUDDOMAIN), CMProps.getVar(CMProps.Str.MUDPORTS))+
263                                L("Please contact the administrators regarding any abuse of this system.\n\r")+
264                                ((unSubUrl == null) ? "" : L("To unsubscribe, visit: @x1  \n\r",unSubUrl));
265                if(!emailIfPossible(smtpServerInfo, fromEmail, replyToEmail, toEmail, subject, message))
266                {
267                        if(CMProps.getIntVar(CMProps.Int.MAXMAILBOX)>0)
268                        {
269                                final int count=CMLib.database().DBCountJournal(CMProps.getVar(CMProps.Str.MAILBOX),null,to);
270                                if(count>=CMProps.getIntVar(CMProps.Int.MAXMAILBOX))
271                                {
272                                        Log.warnOut("Unable to email "+to+" because max mailbox has been reached.");
273                                        return;
274                                }
275                        }
276                        CMLib.database().DBWriteJournal(CMProps.getVar(CMProps.Str.MAILBOX), from, to, subject, message);
277                }

296        /*
297         * Checks to see if the given string is a valid email address.
298         * If it is not, it returns the location from 0-length+1 where
299         * the problem occurs.
300         * @param addy the email address
301         * @return -1, or the location of the error from 0-length+1
302         */
303        private int getEmailAddressError(String addy)
304        {
305                if(addy==null)
306                        return 0;
307                final int x=addy.indexOf('@');
308                if(x<0)
309                        return addy.length();
310                final String localPart=addy.substring(0,x).trim();
311                final String network=addy.substring(x+1);
312                if(localPart.length()==0)
313                        return x;
314                if(network.length()==0)
315                        return addy.length();
316                if((localPart.startsWith("\"")&&localPart.endsWith("\"")))
317                {
318                        final int z=localPart.substring(1,localPart.length()-1).indexOf("\"");
319                        if(z>=0)
320                                return 1+z;
321                }
322                else
323                for(int l=0;l<localPart.length();l++)
324                {
325                        if(EMAIL_VALID_LOCAL_CHARS.indexOf(localPart.charAt(l))<0)
326                                return l;
327                }
328                if(localPart.startsWith("."))
329                        return 0;
330                if(localPart.endsWith("."))
331                        return x-1;
332                if(localPart.length()>64)
333                        return x;
334                for(int l=0;l<network.length();l++)
335                {
336                        if(EMAIL_VALID_DOMAIN_CHARS.indexOf(network.charAt(l))<0)
337                                return x+1+l;
338                }
339                if(network.startsWith("-"))
340                        return x+1;
341                if(network.endsWith("-"))
342                        return addy.length();
343                if(network.startsWith("."))
344                        return x+1;
345                if(network.length()>255)
346                        return addy.length();
347                return -1;
348        }

350        public String makeValidEmailAddress(String name)
351        {
352                if(!isValidEmailAddress(name))
353                {
354                        if(CMLib.players().playerExists(name))
355                                return CMLib.players().getLoadPlayer(name).playerStats().getEmail();
356                        else
357                        if(CMLib.players().accountExists(name))
358                                return CMLib.players().getLoadAccount(name).getEmail();
359                        else
360                        if(CMLib.players().playerExistsAllHosts(name))
361                                return CMLib.players().getPlayerAllHosts(name).playerStats().getEmail();
362                        else
363                        if(CMLib.players().accountExistsAllHosts(name))
364                                return CMLib.players().getAccountAllHosts(name).getEmail();
365                }
366                return name;
367        }

378        /**
379        * Send a message
380        *
381        * Usage:  Mailer.sendmsg(S, From, To, Subject, Message);
382        * @param froaddress  Address sending from
383        * @param reply_address Address reply to
384        * @param to_address Address sending to
385        * @param mockto_address Address sending to
386        * @param subject Subject line
387        * @param message Message content
388        */
389        @Override
390        public synchronized void sendMessage(String froaddress,
391                                                                                 String reply_address,
392                                                                                 String to_address,
393                                                                                 String mockto_address,
394                                                                                 String subject,
395                                                                                 String message)
396                throws IOException
397        {
398                froaddress=froaddress.replace(' ','_');
399                reply_address=reply_address.replace(' ','_');
400                to_address=to_address.replace(' ','_');
401
402                if(CMSecurity.isDisabled(CMSecurity.DisFlag.SMTPCLIENT))
403                {
404                        Log.debugOut("SMTPclient", "Message not sent: "+froaddress+"/"+reply_address+"/"+to_address+"/"+mockto_address+"/"+subject+"/"+message);
405                        return;
406                }
407
408                String rstr;
409                final boolean debug = CMSecurity.isDebugging(CMSecurity.DbgFlag.SMTPCLIENT);
410                final StringBuffer fixMsg=new StringBuffer(message);
411                for(int f=0;f<fixMsg.length();f++)
412                {
413                        if((fixMsg.charAt(f)=='\n')
414                        &&(f>0)
415                        &&(fixMsg.charAt(f-1)!='\r'))
416                        {
417                                if((f<fixMsg.length()-1)&&(fixMsg.charAt(f+1)=='\r'))
418                                {
419                                        fixMsg.setCharAt(f,'\r');
420                                        fixMsg.setCharAt(f+1,'\n');
421                                }
422                                else
423                                {
424                                        fixMsg.insert(f,'\r');
425                                        f++;
426                                }
427                        }
428                        else
429                        if((fixMsg.charAt(f)=='\r')
430                        &&(f<fixMsg.length()-1)
431                        &&(fixMsg.charAt(f+1)!='\n'))
432                        {
433                                if((f>0)&&(fixMsg.charAt(f-1)=='\n'))
434                                {
435                                        fixMsg.setCharAt(f-1,'\r');
436                                        fixMsg.setCharAt(f,'\n');
437                                }
438                                else
439                                {
440                                        fixMsg.insert(f+1,'\n');
441                                        f++;
442                                }
443                        }
444                }
445                message=fixMsg.toString();
446
447                final String host = CMProps.getVar(CMProps.Str.MUDDOMAIN);
448                if((auth != null) && (auth.getAuthType().length()>0))
449                        sendLine(debug,"EHLO " + host);
450                else
451                        sendLine(debug,"HELO " + host);
452                rstr = reply.readLine();
453                if(debug)
454                        Log.debugOut("SMTPclient",rstr);
455                if ((rstr==null)||(!rstr.startsWith("250"))) 
456                        throw new ProtocolException(""+rstr);
457
458                if((auth != null) && (auth.getAuthType().length()>0))
459                {
460                        if(auth.getAuthType().equalsIgnoreCase("login"))
461                                sendLine(debug,"AUTH " + auth.getAuthType()+" "+auth.getLogin());
462                        else
463                                sendLine(debug,"AUTH " + auth.getAuthType());
464                        while(rstr.startsWith("250"))
465                                rstr = reply.readLine();
466                        if(debug)
467                                Log.debugOut("SMTPclient",rstr);
468                        if (!rstr.startsWith("334")) throw new ProtocolException(""+rstr);
469                        if(auth.getAuthType().equalsIgnoreCase("plain"))
470                                sendLine(debug,auth.getPlainLogin());
471                        else
472                        if(auth.getAuthType().equalsIgnoreCase("login"))
473                                sendLine(debug,auth.getPassword());
474                        rstr = reply.readLine();
475                        if(debug)
476                                Log.debugOut("SMTPclient",rstr);
477                        if ((rstr==null)||(!rstr.startsWith("235"))) 
478                                throw new ProtocolException(""+rstr);
479                }
480                sendLine(debug,"MAIL FROM:<" + froaddress+">");
481                rstr = reply.readLine();
482                if(debug)
483                        Log.debugOut("SMTPclient",rstr);
484                if ((rstr==null)||(!rstr.startsWith("250"))) 
485                        throw new ProtocolException(""+rstr);
486                sendLine(debug,"RCPT TO:<" + to_address+">");
487                rstr = reply.readLine();
488                if(debug)
489                        Log.debugOut("SMTPclient",rstr);
490                if ((rstr==null)||(!rstr.startsWith("250"))) 
491                        throw new ProtocolException(""+rstr);
492                sendLine(debug,"DATA");
493                rstr = reply.readLine();
494                if(debug)
495                        Log.debugOut("SMTPclient",rstr);
496                if ((rstr==null)||(!rstr.startsWith("354"))) 
497                        throw new ProtocolException(""+rstr);
498                sendLine(debug,"MIME-Version: 1.0");
499                if((message.indexOf("<HTML>")>=0)&&(message.indexOf("</HTML>")>=0))
500                        sendLine(debug,"Content-Type: text/html");
501                else
502                        sendLine(debug,"Content-Type: text/plain; charset=iso-8859-1");
503                sendLine(debug,"Content-Transfer-Encoding: 7bit");
504                sendLine(debug,"Date: " + CMLib.time().smtpDateFormat(System.currentTimeMillis()));
505                sendLine(debug,"From: " + froaddress);
506                sendLine(debug,"Subject: " + subject);
507                sendLine(debug,"Sender: " + froaddress);
508                sendLine(debug,"Reply-To: " + reply_address);
509                sendLine(debug,"To: " + mockto_address);
510
511                // Sending a blank line ends the header part.
512                send.print(EOL);
513
514                // Now send the message proper
515                sendLine(debug,message);
516                sendLine(debug,".");
517
518                rstr = reply.readLine();
519                if(debug)
520                        Log.debugOut("SMTPclient",rstr);
521                if (!rstr.startsWith("250")) 
522                        throw new ProtocolException(rstr);
523        }

525        /**
526        * return members of a list on an email server.
527        * 250-First Last [emailaddress]\r
528        *
529        * Usage:  List=Mailer.getListMembers(List);
530        * @param list member list
531        * @return String List of members
532        * @throws IOException a socket error
533        * @throws ProtocolException an internal error
534        */
535        public synchronized String getListMembers( String list) throws IOException, ProtocolException 
536        {
537
538                String sendString;
539
540                InetAddress local;
541                try
542                {
543                  local = InetAddress.getLocalHost();
544                }
545                catch (final UnknownHostException ioe)
546                {
547                  System.err.println("No local IP address found - is your network up?");
548                  throw ioe;
549                }
550                final String host = local.getHostName();
551                send.print("HELO " + host);
552                send.print(EOL);
553                send.flush();
554                String rstr = reply.readLine();
555                if (!rstr.startsWith("250")) throw new ProtocolException(rstr);
556                sendString = "EXPN " + list ;
557                send.print(sendString);
558                send.print(EOL);
559                send.flush();
560                rstr="";
561                try
562                {
563                        while(true)
564                        {
565                                rstr+=reply.readLine();
566                                sock.setSoTimeout(1000);
567                        }
568                }
569                catch(final java.io.InterruptedIOException x)
570                { // not really an error, just a control break
571                }
572                sock.setSoTimeout(DEFAULT_TIMEOUT);
573                if (!rstr.startsWith("250")) throw new ProtocolException(rstr);
574                return rstr;
575        }

609        private class SMTPHostAuth
610        {
611                public SMTPHostAuth(String unparsedServerInfo)
612                {
613                        final List<String> info=CMParms.parseCommas(unparsedServerInfo,false);
614                        if(info.size()==0)
615                                return;
616                        host = info.remove(0);
617                        if((info.size()==0)||(host.length()==0))
618                                return;
619                        final String s=info.get(0);
620                        if(s.equalsIgnoreCase("plain")||s.equalsIgnoreCase("login"))
621                                authType=info.remove(0).toUpperCase().trim();
622                        else
623                                authType="PLAIN";
624                        if(info.size()==0)
625                        {
626                                authType="";
627                                return;
628                        }
629                        login=info.remove(0);
630                        if(info.size()==0)
631                                return;
632                        password=info.remove(0);
633                }

635                private String  host            = "";
636                private String  authType        = "";
637                private String  login           = "";
638                private String  password        = "";

647                        return authType;

650                public String getPlainLogin()
651                {
652                        final byte[] buffer = new byte[2 + login.length() + password.length()];
653                        int bufDex=0;
654                        buffer[bufDex++]=0;
655                        for(int i=0;i<login.length();i++)
656                                buffer[bufDex++]=(byte)login.charAt(i);
657                        buffer[bufDex++]=0;
658                        for(int i=0;i<password.length();i++)
659                                buffer[bufDex++]=(byte)password.charAt(i);
660                        return B64Encoder.B64encodeBytes(buffer);
661                }
662
663                public String getLogin()
664                {
665                        final byte[] buffer = new byte[login.length()];
666                        int bufDex=0;
667                        for(int i=0;i<login.length();i++)
668                                buffer[bufDex++]=(byte)login.charAt(i);
669                        return B64Encoder.B64encodeBytes(buffer);
670                }
671
672                public String getPassword()
673                {
674                        final byte[] buffer = new byte[password.length()];
675                        int bufDex=0;
676                        for(int i=0;i<password.length();i++)
677                                buffer[bufDex++]=(byte)password.charAt(i);
