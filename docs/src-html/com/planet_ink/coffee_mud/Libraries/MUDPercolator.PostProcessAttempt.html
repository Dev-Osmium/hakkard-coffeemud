081        @Override
082        public void buildDefinedIDSet(final List<XMLTag> xmlRoot, final Map<String,Object> defined)
083        {
084                if(xmlRoot==null)
085                        return;
086                for(int v=0;v<xmlRoot.size();v++)
087                {
088                        final XMLLibrary.XMLTag piece = xmlRoot.get(v);
089                        final String id = piece.getParmValue("ID");
090                        if((id!=null)&&(id.length()>0))
091                        {
092                                final Object o=defined.get(id.toUpperCase());
093                                if(o != null)
094                                {
095                                        if(!(o instanceof XMLTag))
096                                                Log.errOut("Duplicate ID: "+id+" (first tag did not resolve to a complex piece -- it wins.)");
097                                        else
098                                        {
099                                                final Boolean pMergeVal;
100                                                if((piece.parms()==null)||(!piece.parms().containsKey("MERGE")))
101                                                        pMergeVal = null;
102                                                else
103                                                        pMergeVal = Boolean.valueOf(CMath.s_bool(piece.parms().get("MERGE")));
104                                                final Boolean oMergeVal;
105                                                if((((XMLTag)o).parms()==null)||(!((XMLTag)o).parms().containsKey("MERGE")))
106                                                        oMergeVal = null;
107                                                else
108                                                        oMergeVal = Boolean.valueOf(CMath.s_bool(((XMLTag)o).parms().get("MERGE")));
109                                                if((pMergeVal == null)||(oMergeVal == null))
110                                                        Log.errOut("Duplicate ID: "+id+" (no MERGE tag found to permit this operation -- first tag wins.)");
111                                                else
112                                                if(pMergeVal.booleanValue() && oMergeVal.booleanValue())
113                                                {
114                                                        final XMLTag src=piece;
115                                                        final XMLTag tgt=(XMLTag)o;
116                                                        if(!src.tag().equalsIgnoreCase(tgt.tag()))
117                                                                Log.errOut("Unable to merge tags with ID: "+id+", they are of different types.");
118                                                        else
119                                                        for(final String parm : src.parms().keySet())
120                                                        {
121                                                                final String srcParmVal=src.parms().get(parm);
122                                                                final String tgtParmVal=tgt.parms().get(parm);
123                                                                if(tgtParmVal==null)
124                                                                        tgt.parms().put(parm,srcParmVal);
125                                                                else
126                                                                if(tgtParmVal.equalsIgnoreCase(srcParmVal)||parm.equalsIgnoreCase("ID"))
127                                                                {
128                                                                        /* do nothing -- nothing to do */
129                                                                }
130                                                                else
131                                                                if(parm.equalsIgnoreCase("REQUIRES"))
132                                                                {
133                                                                        final Map<String,String> srcParms=CMParms.parseEQParms(srcParmVal, REQUIRES_DELIMITERS, true);
134                                                                        final Map<String,String> tgtParms=CMParms.parseEQParms(tgtParmVal, REQUIRES_DELIMITERS, true);
135                                                                        for(final String srcKey : srcParms.keySet())
136                                                                        {
137                                                                                final String srcVal=srcParms.get(srcKey);
138                                                                                final String tgtVal=tgtParms.get(srcKey);
139                                                                                if(tgtVal == null)
140                                                                                        tgtParms.put(srcKey, srcVal);
141                                                                                else
142                                                                                if(UPPER_REQUIRES_KEYWORDS.contains(srcVal.toUpperCase()))
143                                                                                {
144                                                                                        if(!srcVal.equalsIgnoreCase(tgtVal))
145                                                                                                Log.errOut("Unable to merge REQUIRES parm on tags with ID: "+id+", mismatch in requirements for '"+srcKey+"'.");
146                                                                                }
147                                                                                else
148                                                                                if(UPPER_REQUIRES_KEYWORDS.contains(tgtVal.toUpperCase()))
149                                                                                        Log.errOut("Unable to merge REQUIRES parm on tags with ID: "+id+", mismatch in requirements for '"+srcKey+"'.");
150                                                                                else
151                                                                                        tgtParms.put(srcKey,srcVal+";"+tgtVal);
152                                                                        }
153                                                                        tgt.parms().put(parm, CMParms.combineEQParms(tgtParms, ','));
154                                                                }
155                                                                else
156                                                                if(parm.equalsIgnoreCase("CONDITION")||parm.equalsIgnoreCase("VALIDATE"))
157                                                                        tgt.parms().put(parm, tgtParmVal+" and "+srcParmVal);
158                                                                else
159                                                                if(parm.equalsIgnoreCase("INSERT")||parm.equalsIgnoreCase("DEFINE")||parm.equalsIgnoreCase("PREDEFINE"))
160                                                                        tgt.parms().put(parm, tgtParmVal+","+srcParmVal);
161                                                                else
162                                                                        Log.errOut("Unable to merge SELECT parm on tags with ID: "+id+".");
163                                                        }
164                                                        for(final XMLLibrary.XMLTag subPiece : src.contents())
165                                                        {
166                                                                final Boolean subMergeVal;
167                                                                if((subPiece.parms()==null)||(!subPiece.parms().containsKey("MERGE")))
168                                                                        subMergeVal = null;
169                                                                else
170                                                                        subMergeVal = Boolean.valueOf(CMath.s_bool(subPiece.parms().get("MERGE")));
171                                                                if((subMergeVal == null)||(subMergeVal.booleanValue()))
172                                                                        tgt.contents().add(subPiece);
173                                                        }
174                                                }
175                                        }
176                                }
177                                else
178                                        defined.put(id.toUpperCase().trim(),piece);
179                        }
180                        final String load = piece.getParmValue("LOAD");
181                        if((load!=null)&&(load.length()>0))
182                        {
183                                piece.parms().remove("LOAD");
184                                XMLTag loadedPiece=(XMLTag)defined.get("SYSTEM_LOADED_XML_FILES");
185                                if(loadedPiece==null)
186                                {
187                                        loadedPiece=CMLib.xml().createNewTag("SYSTEM_LOADED_XML_FILES","");
188                                        defined.put("SYSTEM_LOADED_XML_FILES", loadedPiece);
189                                }
190                                final CMFile file = new CMFile(load,null,CMFile.FLAG_LOGERRORS|CMFile.FLAG_FORCEALLOW);
191                                if(loadedPiece.getPieceFromPieces( file.getAbsolutePath().toUpperCase())==null)
192                                {
193                                        loadedPiece.contents().add(CMLib.xml().createNewTag(file.getAbsolutePath().toUpperCase(),"true"));
194                                        if(file.exists() && file.canRead())
195                                        {
196                                                final List<XMLTag> addPieces=CMLib.xml().parseAllXML(file.text());
197                                                piece.contents().addAll(addPieces);
198                                        }
199                                }
200                        }
201                        buildDefinedIDSet(piece.contents(),defined);
202                }
203        }

244        @SuppressWarnings("unchecked")
245        protected final String PostProcessAttempter(final Map<String,Object> defined, final PostProcessAttempt attempter) throws CMException
246        {
247                try
248                {
249                        attempter.defined=defined;
250                        final String result = attempter.attempt();
251                        return result;
252                }
253                catch(final PostProcessException pe)
254                {
255                        List<PostProcessAttempt> posties=(List<PostProcessAttempt>)defined.get(MUDPercolator.POST_PROCESSING_STAT_SETS);
256                        if(posties==null)
257                        {
258                                posties=new Vector<PostProcessAttempt>();
259                                defined.put(MUDPercolator.POST_PROCESSING_STAT_SETS, posties);
260                        }
261                        attempter.firstRun=false;
262                        final Map<String,Object> definedCopy;
263                        if(defined instanceof Hashtable)
264                                definedCopy=(Map<String,Object>)((Hashtable<String,Object>)defined).clone();
265                        else
266                                definedCopy=new Hashtable<String,Object>(defined);
267                        attempter.defined=definedCopy;
268                        posties.add(attempter);
269                        return null;
270                }
271        }

292        public Room buildRoom(final XMLTag piece, final Map<String,Object> defined, final Exit[] exits, final int direction) throws CMException
293        {
294                addDefinition("DIRECTION",CMLib.directions().getDirectionName(direction).toLowerCase(),defined);
295
296                final String classID = findStringNow("class",piece,defined);
297                final Room R = CMClass.getLocale(classID);
298                if(R == null)
299                        throw new CMException("Unable to build room on classID '"+classID+"', Data: "+CMParms.toKeyValueSlashListString(piece.parms())+":"+CMStrings.limit(piece.value(),100));
300                addDefinition("ROOM_CLASS",classID,defined);
301                final List<String> ignoreStats=new XVector<String>(new String[]{"CLASS","DISPLAY","DESCRIPTION"});
302                fillOutRequiredStatCodeSafe(R, ignoreStats, "ROOM_", "TITLE", "DISPLAY", piece, defined);
303                fillOutRequiredStatCodeSafe(R, ignoreStats, "ROOM_", "DESCRIPTION", "DESCRIPTION", piece, defined);
304                fillOutCopyCodes(R, ignoreStats, "ROOM_", piece, defined);
305                fillOutStatCodes(R, ignoreStats, "ROOM_", piece, defined);
306                final List<MOB> mV = findMobs(piece,defined);
307                for(int i=0;i<mV.size();i++)
308                {
309                        final MOB M=mV.get(i);
310                        M.setSavable(true);
311                        M.bringToLife(R,true);
312                }
313                final List<Item> iV = findItems(piece,defined);
314                for(int i=0;i<iV.size();i++)
315                {
316                        final Item I=iV.get(i);
317                        R.addItem(I);
318                        I.setSavable(true);
319                        I.setExpirationDate(0);
320                }
321                final List<Ability> aV = findAffects(R,piece,defined,null);
322                for(int i=0;i<aV.size();i++)
323                {
324                        final Ability A=aV.get(i);
325                        A.setSavable(true);
326                        R.addNonUninvokableEffect(A);
327                }
328                final List<Behavior> bV = findBehaviors(R,piece,defined);
329                for(int i=0;i<bV.size();i++)
330                {
331                        final Behavior B=bV.get(i);
332                        B.setSavable(true);
333                        R.addBehavior(B);
334                }
335                for(int dir=0;dir<Directions.NUM_DIRECTIONS();dir++)
336                {
337                        Exit E=exits[dir];
338                        if((E==null)&&(defined.containsKey("ROOMLINK_"+CMLib.directions().getDirectionChar(dir).toUpperCase())))
339                        {
340                                defined.put("ROOMLINK_DIR",CMLib.directions().getDirectionChar(dir).toUpperCase());
341                                final Exit E2=findExit(R,piece, defined);
342                                if(E2!=null)
343                                        E=E2;
344                                defined.remove("ROOMLINK_DIR");
345                                if(CMSecurity.isDebugging(CMSecurity.DbgFlag.MUDPERCOLATOR))
346                                        Log.debugOut("MUDPercolator","EXIT:NEW:"+((E==null)?"null":E.ID())+":DIR="+CMLib.directions().getDirectionChar(dir).toUpperCase()+":ROOM="+R.getStat("DISPLAY"));
347                        }
348                        else
349                        if((CMSecurity.isDebugging(CMSecurity.DbgFlag.MUDPERCOLATOR))&&defined.containsKey("ROOMLINK_"+CMLib.directions().getDirectionChar(dir).toUpperCase()))
350                                Log.debugOut("MUDPercolator","EXIT:OLD:"+((E==null)?"null":E.ID())+":DIR="+CMLib.directions().getDirectionChar(dir).toUpperCase()+":ROOM="+R.getStat("DISPLAY"));
351                        R.setRawExit(dir, E);
352                        R.startItemRejuv();
353                }
354                return R;
355        }

416        @Override
417        public Area findArea(final XMLTag piece, final Map<String,Object> defined, final int directions) throws CMException
418        {
419                try
420                {
421                        final String tagName="AREA";
422                        final List<XMLLibrary.XMLTag> choices = getAllChoices(null,null,null,tagName, piece, defined,true);
423                        if((choices==null)||(choices.size()==0))
424                        {
425                                return null;
426                        }
427                        while(choices.size()>0)
428                        {
429                                final XMLLibrary.XMLTag valPiece = choices.get(CMLib.dice().roll(1,choices.size(),-1));
430                                choices.remove(valPiece);
431                                if(valPiece.parms().containsKey("VALIDATE") && !testCondition(null,null,null,CMLib.xml().restoreAngleBrackets(valPiece.getParmValue("VALIDATE")),valPiece, defined))
432                                        continue;
433                                final Map<String,Object> rDefined=new Hashtable<String,Object>();
434                                rDefined.putAll(defined);
435                                defineReward(null,null,null,valPiece.getParmValue("DEFINE"),valPiece,null,rDefined,true);
436                                final Area A=buildArea(valPiece,rDefined,directions);
437                                for (final String key : rDefined.keySet())
438                                {
439                                        if(key.startsWith("_"))
440                                                defined.put(key,rDefined.get(key));
441                                }
442                                return A;
443                        }
444                }
445                catch(final PostProcessException pe)
446                {
447                        throw new CMException("Unable to post process this object type: "+pe.getMessage(),pe);
448                }
449                return null;
450        }

453        {
454                defined.put("DIRECTION",CMLib.directions().getDirectionName(direction).toLowerCase());
455
456                final String classID = findStringNow("class",piece,defined);
457                final Area A = CMClass.getAreaType(classID);
458                if(A == null)
459                        throw new CMException("Unable to build area on classID '"+classID+"', Data: "+CMParms.toKeyValueSlashListString(piece.parms())+":"+CMStrings.limit(piece.value(),100));
460                defined.put("AREA_CLASS",classID);
461                final String name = findStringNow(A,null,"AREA_","NAME",piece,defined);
462                if(CMLib.map().getArea(name)!=null)
463                {
464                        A.destroy();
465                        throw new CMException("Unable to create area '"+name+"', you must destroy the old one first.");
466                }
467                A.setName(name);
468                defined.put("AREA_NAME",name);
469
470                PostProcessAttempter(defined,new PostProcessAttempt()
471                {
472                        @Override
473                        public String attempt() throws CMException, PostProcessException
474                        {
475                                final String author = findOptionalString(A,null,"AREA_","author",piece,this.defined, false);
476                                if(author != null)
477                                {
478                                        A.setAuthorID(author);
479                                        defined.put("AREA_AUTHOR",author);
480                                }
481                                return author;
482                        }
483                });
484                PostProcessAttempter(defined,new PostProcessAttempt()
485                {
486                        @Override
487                        public String attempt() throws CMException, PostProcessException
488                        {
489                                final String description = findOptionalString(A,null,"AREA_","description",piece,this.defined, false);
490                                if(description != null)
491                                {
492                                        A.setDescription(description);
493                                        defined.put("AREA_DESCRIPTION",description);
494                                }
495                                return description;
496                        }
497                });
498
499                if(fillInArea(piece, defined, A, direction))
500                        return A;
501                throw new CMException("Unable to build area for some reason.");
502        }

525        protected Room layOutRooms(final Area A, final LayoutManager layoutManager, final int size, final int direction, final XMLTag piece, final Map<String,Object> defined) throws CMException
526        {
527                if(CMSecurity.isDebugging(CMSecurity.DbgFlag.MUDPERCOLATOR))
528                        Log.debugOut("MudPercolator","Using LayoutManager:"+layoutManager.name());
529                final Random random=new Random(System.currentTimeMillis());
530                final List<LayoutNode> roomsToLayOut = layoutManager.generate(size,direction);
531                if((roomsToLayOut==null)||(roomsToLayOut.size()==0))
532                        throw new CMException("Unable to fill area of size "+size+" off layout "+layoutManager.name());
533                int numLeafs=0;
534                for(int i=0;i<roomsToLayOut.size();i++)
535                {
536                        final LayoutNode node=roomsToLayOut.get(i);
537                        if(node.type()==LayoutTypes.leaf)
538                                numLeafs++;
539                }
540                defined.put("AREA_NUMLEAFS", ""+numLeafs);
541
542                // now break our rooms into logical groups, generate those rooms.
543                final List<List<LayoutNode>> roomGroups = new Vector<List<LayoutNode>>();
544                final LayoutNode magicRoomNode = roomsToLayOut.get(0);
545                final HashSet<LayoutNode> nodesAlreadyGrouped=new HashSet<LayoutNode>();
546                boolean keepLooking=true;
547                while(keepLooking)
548                {
549                        keepLooking=false;
550                        for(int i=0;i<roomsToLayOut.size();i++)
551                        {
552                                final LayoutNode node=roomsToLayOut.get(i);
553                                if(node.type()==LayoutTypes.leaf)
554                                {
555                                        final Vector<LayoutNode> group=new Vector<LayoutNode>();
556                                        group.add(node);
557                                        nodesAlreadyGrouped.add(node);
558                                        for(final Integer linkDir : node.links().keySet())
559                                        {
560                                                final LayoutNode dirNode=node.links().get(linkDir);
561                                                if(!nodesAlreadyGrouped.contains(dirNode))
562                                                {
563                                                        if((dirNode.type()==LayoutTypes.leaf)
564                                                        ||(dirNode.type()==LayoutTypes.interior)&&(node.isFlagged(LayoutFlags.offleaf)))
565                                                        {
566                                                                group.addElement(dirNode);
567                                                                nodesAlreadyGrouped.add(dirNode);
568                                                        }
569                                                }
570                                        }
571                                        for(final LayoutNode n : group)
572                                                roomsToLayOut.remove(n);
573                                        if(group.size()>0)
574                                        {
575                                                // randomize the leafs a bit
576                                                if(roomGroups.size()==0)
577                                                        roomGroups.add(group);
578                                                else
579                                                        roomGroups.add(random.nextInt(roomGroups.size()),group);
580                                        }
581                                        keepLooking=true;
582                                        break;
583                                }
584                        }
585                }
586                keepLooking=true;
587                while(keepLooking)
588                {
589                        keepLooking=false;
590                        for(int i=0;i<roomsToLayOut.size();i++)
591                        {
592                                final LayoutNode node=roomsToLayOut.get(i);
593                                if(node.type()==LayoutTypes.street)
594                                {
595                                        final List<LayoutNode> group=new Vector<LayoutNode>();
596                                        group.add(node);
597                                        nodesAlreadyGrouped.add(node);
598                                        final LayoutRuns run=node.getFlagRuns();
599                                        if(run==LayoutRuns.ns)
600                                        {
601                                                layoutFollow(node,LayoutTypes.street,Directions.NORTH,nodesAlreadyGrouped,group);
602                                                layoutFollow(node,LayoutTypes.street,Directions.SOUTH,nodesAlreadyGrouped,group);
603                                        }
604                                        else
605                                        if(run==LayoutRuns.ew)
606                                        {
607                                                layoutFollow(node,LayoutTypes.street,Directions.EAST,nodesAlreadyGrouped,group);
608                                                layoutFollow(node,LayoutTypes.street,Directions.WEST,nodesAlreadyGrouped,group);
609                                        }
610                                        else
611                                        if(run==LayoutRuns.nesw)
612                                        {
613                                                layoutFollow(node,LayoutTypes.street,Directions.NORTHEAST,nodesAlreadyGrouped,group);
614                                                layoutFollow(node,LayoutTypes.street,Directions.SOUTHWEST,nodesAlreadyGrouped,group);
615                                        }
616                                        else
617                                        if(run==LayoutRuns.nwse)
618                                        {
619                                                layoutFollow(node,LayoutTypes.street,Directions.NORTHWEST,nodesAlreadyGrouped,group);
620                                                layoutFollow(node,LayoutTypes.street,Directions.SOUTHEAST,nodesAlreadyGrouped,group);
621                                        }
622                                        else
623                                        {
624                                                int topDir=-1;
625                                                int topSize=-1;
626                                                for(int d=0;d<Directions.NUM_DIRECTIONS();d++)
627                                                {
628                                                        if(node.links().get(Integer.valueOf(d))!=null)
629                                                        {
630                                                                final List<LayoutNode> grpCopy=new XVector<LayoutNode>(group);
631                                                                final HashSet<LayoutNode> nodesAlreadyGroupedCopy=(HashSet<LayoutNode>)nodesAlreadyGrouped.clone();
632                                                                layoutFollow(node,LayoutTypes.street,d,nodesAlreadyGroupedCopy,grpCopy);
633                                                                if(node.links().get(Integer.valueOf(Directions.getOpDirectionCode(d)))!=null)
634                                                                        layoutFollow(node,LayoutTypes.street,Directions.getOpDirectionCode(d),nodesAlreadyGroupedCopy,grpCopy);
635                                                                if(grpCopy.size()>topSize)
636                                                                {
637                                                                        topSize=grpCopy.size();
638                                                                        topDir=d;
639                                                                }
640                                                        }
641                                                }
642                                                if(topDir>=0)
643                                                {
644                                                        layoutFollow(node,LayoutTypes.street,topDir,nodesAlreadyGrouped,group);
645                                                        if(node.links().get(Integer.valueOf(Directions.getOpDirectionCode(topDir)))!=null)
646                                                                layoutFollow(node,LayoutTypes.street,Directions.getOpDirectionCode(topDir),nodesAlreadyGrouped,group);
647                                                }
648                                        }
649                                        for(final LayoutNode n : group)
650                                                roomsToLayOut.remove(n);
651                                        if(group.size()>0)
652                                                roomGroups.add(group);
653                                        keepLooking=true;
654                                }
655                        }
656                }
657
658                while(roomsToLayOut.size() >0)
659                {
660                        final LayoutNode node=roomsToLayOut.get(0);
661                        final Vector<LayoutNode> group=new Vector<LayoutNode>();
662                        group.add(node);
663                        nodesAlreadyGrouped.add(node);
664                        layoutRecursiveFill(node,nodesAlreadyGrouped,group,node.type());
665                        for(final LayoutNode n : group)
666                                roomsToLayOut.remove(n);
667                        roomGroups.add(group);
668                }
669
670                // make CERTAIN that the magic first room in the layout always
671                // gets ID#0.
672                List<LayoutNode> magicGroup=null;
673                for(int g=0;g<roomGroups.size();g++)
674                {
675                        final List<LayoutNode> group=roomGroups.get(g);
676                        if(group.contains(magicRoomNode))
677                        {
678                                magicGroup=group;
679                                break;
680                        }
681                }
682                if(CMSecurity.isDebugging(CMSecurity.DbgFlag.MUDPERCOLATOR))
683                for(int g=0;g<roomGroups.size();g++)
684                {
685                        final Vector<LayoutNode> group=(Vector<LayoutNode>)roomGroups.get(g);
686                        if(CMSecurity.isDebugging(CMSecurity.DbgFlag.MUDPERCOLATOR))
687                                Log.debugOut("MudPercolator","GROUP:"+A.Name()+": "+group.firstElement().type().toString()+": "+group.size());
688                }
689                final Map<List<LayoutNode>,Map<String,Object>> groupDefinitions=new Hashtable<List<LayoutNode>,Map<String,Object>>();
690                for(final List<LayoutNode> group : roomGroups)
691                {
692                        final Map<String,Object> newDefined=new Hashtable<String,Object>();
693                        newDefined.putAll(defined);
694                        groupDefinitions.put(group, newDefined);
695                }
696
697                Map<String,Object> groupDefined = groupDefinitions.get(magicGroup);
698                final Room magicRoom = processRoom(A,direction,piece,magicRoomNode,groupDefined);
699                for(final Map<String,Object> otherDefineds : groupDefinitions.values())
700                {
701                        otherDefineds.remove("ROOMTAG_NODEGATEEXIT");
702                        otherDefineds.remove("ROOMTAG_GATEEXITROOM");
703                }
704                updateLayoutDefinitions(defined,groupDefined,groupDefinitions,roomGroups);
705
706                //now generate the rooms and add them to the area
707                for(final List<LayoutNode> group : roomGroups)
708                {
709                        groupDefined = groupDefinitions.get(group);
710
711                        for(final LayoutNode node : group)
712                        {
713                                if(node!=magicRoomNode)
714                                        processRoom(A,direction,piece,node,groupDefined);
715                        }
716                        updateLayoutDefinitions(defined,groupDefined,groupDefinitions,roomGroups);
717                }
718
719                for(final Integer linkDir : magicRoomNode.links().keySet())
720                {
721                        final LayoutNode linkNode=magicRoomNode.getLink(linkDir.intValue());
722                        if((magicRoom.getRawExit(linkDir.intValue())==null) || (linkNode.room() == null))
723                                Log.errOut("MUDPercolator","Generated an unpaired node for "+magicRoom.roomID());
724                        else
725                                magicRoom.rawDoors()[linkDir.intValue()]=linkNode.room();
726                }
727
728                //now do a final-link on the rooms
729                for(final List<LayoutNode> group : roomGroups)
730                {
731                        for(final LayoutNode node : group)
732                        {
733                                final Room R=node.room();
734                                if(node != magicRoomNode)
735                                {
736                                        if((R==null)||(node.links().keySet().size()==0))
737                                                Log.errOut("MUDPercolator",layoutManager.name()+" generated a linkless node: "+node.toString());
738                                        else
739                                        for(final Integer linkDir : node.links().keySet())
740                                        {
741                                                final LayoutNode linkNode=node.getLink(linkDir.intValue());
742                                                if((R.getRawExit(linkDir.intValue())==null) || (linkNode.room() == null))
743                                                        Log.errOut("MUDPercolator","Generated an unpaired node for "+R.roomID());
744                                                else
745                                                        R.rawDoors()[linkDir.intValue()]=linkNode.room();
746                                        }
747                                }
748                        }
749                }
750                return magicRoom;
751        }

805        protected Room processRoom(final Area A, final int direction, final XMLTag piece, final LayoutNode node, final Map<String,Object> groupDefined)
806                throws CMException
807        {
808                for(final LayoutTags key : node.tags().keySet())
809                        groupDefined.put("ROOMTAG_"+key.toString().toUpperCase(),node.tags().get(key));
810                final Exit[] exits=new Exit[Directions.NUM_DIRECTIONS()];
811                for(final Integer linkDir : node.links().keySet())
812                {
813                        final LayoutNode linkNode = node.links().get(linkDir);
814                        if(linkNode.room() != null)
815                        {
816                                final int opDir=Directions.getOpDirectionCode(linkDir.intValue());
817                                exits[linkDir.intValue()]=linkNode.room().getExitInDir(opDir);
818                                groupDefined.put("ROOMTITLE_"+CMLib.directions().getDirectionChar(linkDir.intValue()).toUpperCase(),linkNode.room().displayText(null));
819                        }
820                        groupDefined.put("NODETYPE_"+CMLib.directions().getDirectionChar(linkDir.intValue()).toUpperCase(),linkNode.type().name());
821                        //else groupDefined.put("ROOMTITLE_"+CMLib.directions().getDirectionChar(linkDir.intValue()).toUpperCase(),"");
822                        groupDefined.put("ROOMLINK_"+CMLib.directions().getDirectionChar(linkDir.intValue()).toUpperCase(),"true");
823                }
824                if(CMSecurity.isDebugging(CMSecurity.DbgFlag.MUDPERCOLATOR))
825                {
826                        if(CMSecurity.isDebugging(CMSecurity.DbgFlag.MUDPERCOLATOR))
827                                Log.debugOut("MUDPercolator",A.Name()+": type: "+node.type().toString());
828                        final StringBuffer defs=new StringBuffer("");
829                        for (final String key : groupDefined.keySet())
830                        {
831                                defs.append(key+"="+CMStrings.limit(groupDefined.get(key).toString(),10)+",");
832                        }
833                        if(CMSecurity.isDebugging(CMSecurity.DbgFlag.MUDPERCOLATOR))
834                                Log.debugOut("MUDPercolator","DEFS: "+defs.toString());
835                }
836                final Room R=findRoom(A,piece, groupDefined, exits, direction);
837                if(R==null)
838                        throw new CMException("Failure to generate room from "+piece.value());
839                R.setRoomID(A.getNewRoomID(null,-1));
840                if(CMSecurity.isDebugging(CMSecurity.DbgFlag.MUDPERCOLATOR))
841                        Log.debugOut("MUDPercolator","ROOMID: "+R.roomID());
842                R.setArea(A);
843                A.addProperRoom(R);
844                node.setRoom(R);
845                groupDefined.remove("ROOMTAG_NODEGATEEXIT");
846                groupDefined.remove("ROOMTAG_GATEEXITROOM");
847                for(final LayoutTags key : node.tags().keySet())
848                        groupDefined.remove("ROOMTAG_"+key.toString().toUpperCase());
849                for(final Integer linkDir : node.links().keySet())
850                {
851                        groupDefined.remove("NODETYPE_"+CMLib.directions().getDirectionChar(linkDir.intValue()).toUpperCase());
852                        groupDefined.remove("ROOMLINK_"+CMLib.directions().getDirectionChar(linkDir.intValue()).toUpperCase());
853                        groupDefined.remove("ROOMTITLE_"+CMLib.directions().getDirectionChar(linkDir.intValue()).toUpperCase());
854                }
855                return R;
856        }

1029        protected List<Varidentifier> parseVariables(final String str)
1030        {
1031                int x=str.indexOf('$');
1032                final List<Varidentifier> list=new XVector<Varidentifier>();
1033                while((x>=0)&&(x<str.length()-1))
1034                {
1035                        final Varidentifier var = new Varidentifier();
1036                        var.outerStart=x;
1037                        x++;
1038                        if((x<str.length())&&(str.charAt(x)=='{'))
1039                        {
1040                                int varstart=var.outerStart;
1041                                x++;
1042                                while((x<str.length()-2)&&(str.charAt(x+1)==':'))
1043                                {
1044                                        if((str.charAt(x)=='l')||(str.charAt(x)=='L'))
1045                                                var.toLowerCase=true;
1046                                        else
1047                                        if((str.charAt(x)=='u')||(str.charAt(x)=='U'))
1048                                                var.toUpperCase=true;
1049                                        else
1050                                        if((str.charAt(x)=='p')||(str.charAt(x)=='P'))
1051                                                var.toPlural=true;
1052                                        else
1053                                        if((str.charAt(x)=='c')||(str.charAt(x)=='C'))
1054                                                var.toCapitalized=true;
1055                                        x+=2;
1056                                        varstart+=2;
1057                                }
1058                                while((x<str.length())&&(str.charAt(x)!='}'))
1059                                        x++;
1060                                var.var = str.substring(varstart+2,x);
1061                                if(x<str.length())
1062                                        x++;
1063                                var.outerEnd=x;
1064                        }
1065                        else
1066                        if((x<str.length())&&(str.charAt(x)=='['))
1067                        {
1068                                final int varstart=var.outerStart;
1069                                x++;
1070                                while((x<str.length())&&(str.charAt(x)!=']'))
1071                                        x++;
1072                                var.var = str.substring(varstart+2,x);
1073                                var.isMathExpression=true;
1074                                if(x<str.length())
1075                                        x++;
1076                                var.outerEnd=x;
1077                        }

1634        {
1635                int levelLimit=-1;
1636                int levelFloor=-1;
1637                Class<?>[] deriveClasses=new Class[0];
1638                final Map.Entry<Character, String>[] otherParms=CMStrings.splitMulti(recipe, splitters);

1641                        return recipe;
1642                for(int i=1;i<otherParms.length;i++)
1643                {

1645                        {

1647                                if(CMath.isMathExpression(lvlStr))
1648                                {
1649                                        levelLimit=CMath.parseIntExpression(lvlStr);

1651                                                levelLimit=-1;
1652                                }
1653                        }
1654                        else
1655                        if(otherParms[i].getKey().charValue()=='>')
1656                        {
1657                                final String lvlStr=strFilterNow(null,null,null,otherParms[i].getValue().trim(),piece, defined);
1658                                if(CMath.isMathExpression(lvlStr))
1659                                {
1660                                        levelFloor=CMath.parseIntExpression(lvlStr);
1661                                        if((levelFloor==0)||((levelFloor>levelLimit)&&(levelLimit>0)))
1662                                                levelFloor=-1;
1663                                }
1664                        }
1665                        else
1666                        if(otherParms[i].getKey().charValue()=='=')
1667                        {
1668                                final String classStr=strFilterNow(null,null,null,otherParms[i].getValue().trim(),piece, defined);
1669                                final Object O=CMClass.getItemPrototype(classStr);
1670                                if(O!=null)
1671                                {

1673                                        deriveClasses[deriveClasses.length-1]=O.getClass();
1674                                }
1675                                else
1676                                        throw new CMException("Unknown metacraft class= "+classStr);
1677                        }
1678                }
1679                if(levelLimit>0)
1680                        filter.first=Integer.valueOf(levelLimit);
1681                if(levelFloor>0)
1682                        filter.second=Integer.valueOf(levelFloor);
1683                if(deriveClasses.length>0)
1684                        filter.third=deriveClasses;
1685                return recipe;
1686        }
