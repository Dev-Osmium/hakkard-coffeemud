060        public String niceCommaList(final List<?> V, final boolean andTOrF)
061        {
062                String id="";
063                for(int v=0;v<V.size();v++)
064                {
065                        String s=null;
066                        if(V.get(v) instanceof Environmental)
067                                s=((Environmental)V.get(v)).name();
068                        else
069                        if(V.get(v) instanceof String)
070                                s=(String)V.get(v);
071                        else
072                                continue;
073                        if(V.size()==1)
074                                id+=s;
075                        else
076                        if(v==(V.size()-1))
077                                id+=((andTOrF)?"and ":"or ")+s;
078                        else
079                                id+=s+", ";
080                }
081                return id;
082        }

242        {
243                double s=-1.0;
244                try
245                {
246                        final int n = number;
247                        final Object[] objs = new Object[n] ;
248                        final Environmental cl = E;
249                        final Runtime rt = Runtime.getRuntime() ;
250                        final long m0 =rt.totalMemory() - rt.freeMemory() ;
251                        System.gc() ;
252                        Thread.sleep( 500 ) ;
253                        for (int i = 0 ; i < n ; ++i)
254                                objs[i] = E = (Environmental)cl.copyOf();
255                        System.gc() ;
256                        Thread.sleep( 1000 ) ;
257                        final long m1 =rt.totalMemory() - rt.freeMemory() ;
258                        final long dm = m1 - m0 ;
259                        s = (double)dm / (double)n ;
260                        if(s<0.0)
261                                return memoryUse(E,number);
262                }
263                catch (final Exception e)
264                {
265                        return -1;
266                }
267                return s;

298        {
299                if(target instanceof Room)
300                {
301                        final Room R=(Room)target;
302                        for(int m=0;m<R.numInhabitants();m++)
303                        {
304                                final MOB M=R.fetchInhabitant(m);
305                                if(M!=null)
306                                        extinguish(source,M,mundane);
307                        }
308                        for(int i=0;i<R.numItems();i++)
309                        {
310                                final Item I=R.getItem(i);
311                                if(I!=null)
312                                        extinguish(source,I,mundane);
313                        }
314                        return;
315                }
316                for(int a=target.numEffects()-1;a>=0;a--) // personal effects
317                {
318                        final Ability A=target.fetchEffect(a);
319                        if((A!=null)&&((!mundane)||((A.classificationCode()&Ability.ALL_ACODES)==Ability.ACODE_PROPERTY)))
320                        {
321                                if((CMath.bset(A.flags(),Ability.FLAG_HEATING)&&(!mundane))
322                                ||(CMath.bset(A.flags(),Ability.FLAG_FIREBASED))
323                                ||((A.ID().equalsIgnoreCase("Spell_SummonElemental")&&A.text().toUpperCase().indexOf("FIRE")>=0)))
324                                        A.unInvoke();
325                        }
326                }
327                if((target instanceof MOB)&&(!mundane))
328                {
329                        final MOB tmob=(MOB)target;
330                        if(tmob.charStats().getMyRace().ID().equals("FireElemental"))
331                                CMLib.combat().postDeath(source,(MOB)target,null);
332                        for(int i=0;i<tmob.numItems();i++)
333                        {
334                                final Item I=tmob.getItem(i);
335                                if(I!=null)
336                                        extinguish(tmob,I,mundane);
337                        }
338                }
339                if((target instanceof Light)&&(((Light)target).isLit()))
340                {
341                        ((Light)target).tick(target,Tickable.TICKID_LIGHT_FLICKERS);
342                        ((Light)target).light(false);
343                }
344        }

379        @Override
380        public int disenchantItem(final Item target)
381        {
382                int level=target.basePhyStats().level();
383                boolean doneSomething=false;
384                if(target instanceof Wand)
385                {
386                        final Ability A=((Wand)target).getSpell();
387                        if(A!=null)
388                                level=level-CMLib.ableMapper().lowestQualifyingLevel(A.ID())+2;
389                        ((Wand)target).setSpell(null);
390                        ((Wand)target).setUsesRemaining(0);
391                        doneSomething=true;
392                }
393                else
394                if(target instanceof SpellHolder)
395                {
396                        ((SpellHolder)target).setSpellList("");
397                        doneSomething=true;
398                }
399                else
400                if((target.phyStats().ability()>0)
401                &&(!(target instanceof Coins)))
402                {
403                        level=level-(target.basePhyStats().ability()*3);
404                        target.basePhyStats().setAbility(0);
405                        doneSomething=true;
406                }
407
408                final LinkedList<Ability> affects=new LinkedList<Ability>();
409                for(final Enumeration<Ability> a=target.effects();a.hasMoreElements();)
410                {
411                        final Ability A=a.nextElement();
412                        if(A!=null)
413                                affects.add(A);
414                }
415                for(final Ability A : affects)
416                {
417                        A.unInvoke();
418                        level=level-1;
419                        target.delEffect(A);
420                        doneSomething=true;
421                }
422                if(target.amDestroyed())
423                        return 0;
424                if(doneSomething)
425                        return level;
426                return -999;
427        }

568                                                        for(int r=0;r<V.size();r++)
569                                                        {
570                                                                final Item I=V.get(r);

572                                                                {
573                                                                        mob.delItem(I);

575                                                                }

579                                                                        I.phyStats().setRejuv(PhyStats.NO_REJUV);
580                                                                }
581                                                        }
582                                                }
583                                        }
584                                        if(mob instanceof ShopKeeper)
585                                        {
586                                                final List<Item> rivalItems=new Vector<Item>();
587                                                final CoffeeShop shop=(mob instanceof Librarian)?((Librarian)mob).getBaseLibrary():((ShopKeeper)mob).getShop();

596                                                                        rivalItems.add((Item)E);
597                                                        }
598                                                }
599                                                for(final Item I : rivalItems)
600                                                {
601                                                        if(CMLib.dice().rollPercentage()>I.basePhyStats().rejuv())
602                                                        {
603                                                                shop.delAllStoreInventory(I);

605                                                        }

608                                                                I.basePhyStats().setRejuv(PhyStats.NO_REJUV);
609                                                                I.phyStats().setRejuv(PhyStats.NO_REJUV);
610                                                        }
611                                                }
612                                        }

615                                                CMLib.beanCounter().addMoney(mob, Math.random()*mob.getMoneyVariation());

619                                        mob.recoverPhyStats();
620                                        mob.recoverCharStats();
621                                        mob.recoverMaxState();
622                                }
623                        }
624                }
625                return newLastTickedDateTime;
626        }

628        @Override
629        public void recursiveDropMOB(final MOB mob, final Room room, final Item thisContainer, boolean bodyFlag)
630        {
631                // caller is responsible for recovering any env
632                // stat changes!
633
634                if(CMLib.flags().isHidden(thisContainer))
635                        thisContainer.basePhyStats().setDisposition(thisContainer.basePhyStats().disposition()&((int)PhyStats.ALLMASK-PhyStats.IS_HIDDEN));
636                mob.delItem(thisContainer);
637                thisContainer.unWear();
638                if(!bodyFlag)
639                        bodyFlag=(thisContainer instanceof DeadBody);
640                if(bodyFlag)
641                {
642                        room.addItem(thisContainer);
643                        thisContainer.setExpirationDate(0);
644                }
645                else
646                        room.addItem(thisContainer,ItemPossessor.Expire.Player_Drop);
647                thisContainer.recoverPhyStats();
648                boolean nothingDone=true;
649                do
650                {
651                        nothingDone=true;
652                        for(int i=0;i<mob.numItems();i++)
653                        {
654                                final Item thisItem=mob.getItem(i);
655                                if((thisItem!=null)&&(thisItem.container()==thisContainer))
656                                {
657                                        recursiveDropMOB(mob,room,thisItem,bodyFlag);
658                                        nothingDone=false;
659                                        break;
660                                }
661                        }
662                }
663                while(!nothingDone);
664        }

832        protected TriadVector<Integer,Integer,MaskingLibrary.CompiledZMask> parseLootPolicyFor(final MOB mob)
833        {
834                if((mob==null)||(!mob.isMonster()))
835                        return noLootPolicy;
836                final String lootPolicyStr=CMProps.getVar(CMProps.Str.ITEMLOOTPOLICY);
837                if((lootPolicy==null)||(lastLootPolicyHash!=lootPolicyStr.hashCode()))
838                {
839                        final List<String> lootPolicies=(!mob.isMonster())?new Vector<String>():CMParms.parseCommas(CMProps.getVar(CMProps.Str.ITEMLOOTPOLICY),true);
840                        final TriadVector<Integer,Integer,MaskingLibrary.CompiledZMask> policies=new TriadVector<Integer,Integer,MaskingLibrary.CompiledZMask>();
841                        for(int p=0;p<lootPolicies.size();p++)
842                        {
843                                String s=lootPolicies.get(p).toUpperCase().trim();
844                                if(s.length()==0)
845                                        continue;
846                                MaskingLibrary.CompiledZMask compiledMask=null;
847                                final int maskDex=s.indexOf("MASK=");
848                                if(maskDex>=0)
849                                {
850                                        s=s.substring(0,maskDex).trim();
851                                        compiledMask=CMLib.masking().maskCompile(lootPolicies.get(p).substring(maskDex+5).trim());
852                                }
853                                else
854                                        compiledMask=CMLib.masking().createEmptyMask();
855                                final Vector<String> parsed=CMParms.parse(s);
856                                int pct=100;
857                                for(int x=0;x<parsed.size();x++)
858                                {
859                                        if(CMath.isInteger(parsed.elementAt(x)))
860                                                pct=CMath.s_int(parsed.elementAt(x));
861                                        else
862                                        if(CMath.isPct(parsed.elementAt(x)))
863                                                pct=(int)Math.round(CMath.s_pct(parsed.elementAt(x))*100.0);
864                                }
865                                int flags=0;
866                                if(parsed.contains("RUIN"))
867                                        flags|=LOOTFLAG_RUIN;
868                                else
869                                if(parsed.contains("LOSS"))
870                                        flags|=LOOTFLAG_LOSS;
871                                if(flags==0)
872                                        flags|=LOOTFLAG_LOSS;
873                                if(parsed.contains("WORN"))
874                                        flags|=LOOTFLAG_WORN;
875                                else
876                                if(parsed.contains("UNWORN"))
877                                        flags|=LOOTFLAG_UNWORN;
878                                policies.addElement(Integer.valueOf(pct),Integer.valueOf(flags),compiledMask);
879                        }
880                        lootPolicy=policies;
881                        lastLootPolicyHash=lootPolicyStr.hashCode();
882                }
883                return lootPolicy;
884        }
