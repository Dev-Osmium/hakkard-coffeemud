027public class GridCityLayout extends AbstractLayout
028{
029        @Override
030        public String name()
031        {
032                return "GRIDCITY";
033        }
034
035        @Override
036        public List<LayoutNode> generate(final int num, final int dir)
037        {
038                final Vector<LayoutNode> set = new Vector<LayoutNode>();
039                final int diameter = (int)Math.round(Math.sqrt(num));
040                final int plusX = (diff(diameter,diameter,num) > diff(diameter+1,diameter,num)) ? 1 : 0;
041
042                final LayoutSet lSet = new LayoutSet(set,num);
043                lSet.drawABox(diameter+plusX,diameter);
044                int middle=(diameter+plusX)/2;
045                LayoutNode firstNode=null;
046                for(int x=0;x<diameter+plusX;x+=2)
047                {
048                        LayoutNode n = lSet.getNode(new long[]{x,0});
049                        if(n!=null)
050                        {
051                                if((dir==Directions.NORTH)&&(x>=middle-1)&&(x<=middle+1))
052                                        firstNode=n;
053                                for(int y=0;y<diameter-1;y++)
054                                {
055                                        lSet.use(n,LayoutTypes.street);
056                                        n.flagRun(LayoutRuns.ns);
057                                        LayoutNode nn = lSet.getNextNode(n, Directions.NORTH);
058                                        if(nn==null)
059                                                nn=lSet.makeNextNode(n, Directions.NORTH);
060                                        n.crossLink(nn);
061                                        n=nn;
062                                }
063                                lSet.use(n,LayoutTypes.street);
064                                n.flagRun(LayoutRuns.ns);
065                                if((dir==Directions.SOUTH)&&(x>=middle-1)&&(x<=middle+1))
066                                        firstNode=n;
067                        }
068                }
069                final int endX=diameter+plusX-1;
070                middle=(-diameter+1)/2;
071                for(int y=-2;y>=-diameter+1;y-=2)
072                {
073                        LayoutNode n = lSet.getNode(new long[]{0,y});
074                        if(n!=null)
075                        {
076                                if((dir==Directions.EAST)&&(y>=middle-1)&&(y<=middle+1))
077                                        firstNode=n;
078                                for(int x=0;x<endX;x++)
079                                {
080                                        lSet.use(n,LayoutTypes.street);
081                                        n.flagRun(LayoutRuns.ew);
082                                        LayoutNode nn = lSet.getNextNode(n, Directions.EAST);
083                                        if(nn==null)
084                                                nn=lSet.makeNextNode(n, Directions.EAST);
085                                        n.crossLink(nn);
086                                        n=nn;
087                                }
088                                lSet.use(n,LayoutTypes.street);
089                                n.flagRun(LayoutRuns.ew);
090                                if((dir==Directions.WEST)&&(y>=middle-1)&&(y<=middle+1))
091                                        firstNode=n;
092                        }
093                }
094                boolean north=true;
095                for(int y=0;y>=(-diameter)+1;y-=2)
096                {
097                        for(int x=1;x<endX;x++)
098                        {
099                                final LayoutNode n = lSet.getNode(new long[]{x,y});
100                                LayoutNode nn = lSet.getNextNode(n, Directions.NORTH);
101                                if(nn==null)
102                                {
103                                        if(north)
104                                        {
105                                                if(y>(-diameter)+2)
106                                                {
107                                                        nn = lSet.makeNextNode( n, Directions.NORTH);
108                                                }
109                                        }
110                                        if(nn != null)
111                                        {
112                                                n.crossLink(nn);
113                                                lSet.use(nn,LayoutTypes.interior);
114                                        }
115                                        north = !north;
116                                }
117                        }
118                }
119                for(int y=-2;y>=(-diameter)+1;y-=2)
120                {
121                        for(int x=1;x<endX;x++)
122                        {
123                                final LayoutNode n = lSet.getNode(new long[]{x,y});
124                                LayoutNode nn = lSet.getNextNode(n, Directions.SOUTH);
125                                if(nn==null)
126                                {
127                                        nn = lSet.makeNextNode( n, Directions.SOUTH);
128                                        if(nn != null)
129                                        {
130                                                n.crossLink(nn);
131                                                lSet.use(nn,LayoutTypes.interior);
132                                        }
133                                }
134                        }
135                }
136                lSet.fillInFlags();
137                if(firstNode == null)
138                {
139                        if(set.size()>0)
140                                firstNode=set.get(0);
141                }
142                if(firstNode != null)
143                {
144                        set.remove(firstNode);
145                        set.insertElementAt(firstNode, 0);
146                }
147                else
148                        Log.errOut("GridCityLayout","Not able to find start room for direction: "+dir);
149                return set;
150        }
151
152}
