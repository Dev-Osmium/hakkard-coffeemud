034public class LayoutSet
035{
036        Random r = new Random();
037
038        private long total  = 0;
039        private final Map<Long, LayoutNode> used = new Hashtable<Long, LayoutNode>();
040        private List<LayoutNode> set = null;
041
042        public LayoutSet(final List<LayoutNode> V, final long total)
043        {
044                this.total = total;
045                this.set = V;
046        }
047
048        public List<LayoutNode> set()
049        {
050                return set;
051        }
052
053        public Long getHashCode(final long x, final long y)
054        {
055                return (Long.valueOf((x * total) + y));
056        }
057
058        public boolean isUsed(final long[] xy)
059        {
060                return isUsed(xy[0], xy[1]);
061        }
062
063        public boolean isUsed(final long x, final long y)
064        {
065                return used.containsKey(getHashCode(x, y));
066        }
067
068        public boolean isUsed(final LayoutNode n)
069        {
070                return isUsed(n.coord()) && set.contains(n);
071        }
072
073        public void unUse(final LayoutNode n)
074        {
075                used.remove(getHashCode(n.coord()[0],n.coord()[1]));
076                set.remove(n);
077        }
078
079        public boolean use(final LayoutNode n, final LayoutTypes nodeType)
080        {
081                if(isUsed(n.coord()))
082                        return false;
083                used.put(getHashCode(n.coord()[0],n.coord()[1]),n);
084                set.add(n);
085                if(nodeType != null)
086                        n.reType(nodeType);
087                return true;
088        }
089
090        public LayoutNode getNode(final long[] xy)
091        {
092                return getNode(xy[0], xy[1]);
093        }
094
095        public LayoutNode getNode(final long x, final long y)
096        {
097                return used.get(getHashCode(x, y));
098        }
099
100        public boolean spaceAvailable()
101        {
102                return set.size() < total;
103        }
104
105        public long[] makeNextCoord(final long[] n, final int dir)
106        {
107                switch(dir)
108                {
109                case Directions.NORTH:
110                        return new long[]{n[0],n[1]-1};
111                case Directions.SOUTH:
112                        return new long[]{n[0],n[1]+1};
113                case Directions.EAST:
114                        return new long[]{n[0]+1,n[1]};
115                case Directions.WEST:
116                        return new long[]{n[0]-1,n[1]};
117                default:
118                        return makeNextCoord(n,CMLib.dice().pick(Directions.CODES())); // picks one of the above, and only one of the above
119                }
120        }
121
122        public LayoutNode makeNextNode(final LayoutNode n, final int dir)
123        {
124                final long[] l = makeNextCoord(n.coord(),dir);
125                if(l!=null)
126                        return new DefaultLayoutNode(l);
127                return null;
128        }
129
130        public LayoutNode getNextNode(final LayoutNode n, final int dir)
131        {
132                final LayoutNode next = makeNextNode(n,dir);
133                return getNode(next.coord());
134        }

136        public void drawABox(final int width, final int height)
137        {
138                LayoutNode n = new DefaultLayoutNode(new long[]{0,0});
139                n.flag(LayoutFlags.corner);
140                for(int y=0;y<height-1;y++)
141                {
142                        use(n,LayoutTypes.surround);
143                        LayoutNode nn = getNextNode(n, Directions.NORTH);
144                        if(nn==null)
145                                nn=makeNextNode(n, Directions.NORTH);
146                        n.crossLink(nn);
147                        nn.flagRun(LayoutRuns.ns);
148                        n=nn;
149                }
150                n.flag(LayoutFlags.corner);
151                use(n,LayoutTypes.surround);
152                n = new DefaultLayoutNode(new long[]{width-1,0});
153                n.flag(LayoutFlags.corner);
154                for(int y=0;y<height-1;y++)
155                {
156                        use(n,LayoutTypes.surround);
157                        LayoutNode nn = getNextNode(n, Directions.NORTH);
158                        if(nn==null)
159                                nn=makeNextNode(n, Directions.NORTH);
160                        n.crossLink(nn);
161                        nn.flagRun(LayoutRuns.ns);
162                        n=nn;
163                }
164                n.flag(LayoutFlags.corner);
165                use(n,LayoutTypes.surround);
166                n = getNode(new long[]{0,0});
167                n.flag(LayoutFlags.corner);
168                for(int x=0;x<width-1;x++)
169                {
170                        use(n,LayoutTypes.surround);
171                        LayoutNode nn = getNextNode(n, Directions.EAST);
172                        if(nn==null)
173                                nn=makeNextNode(n, Directions.EAST);
174                        n.crossLink(nn);
175                        nn.flagRun(LayoutRuns.ew);
176                        n=nn;
177                }
178                n.flag(LayoutFlags.corner);
179                use(n,LayoutTypes.surround);
180                n = getNode(new long[]{0,-height+1});
181                n.flag(LayoutFlags.corner);
182                for(int x=0;x<width-1;x++)
183                {
184                        use(n,LayoutTypes.surround);
185                        LayoutNode nn = getNextNode(n, Directions.EAST);
186                        if(nn==null)
187                                nn=makeNextNode(n, Directions.EAST);
188                        n.crossLink(nn);
189                        nn.flagRun(LayoutRuns.ew);
190                        n=nn;
191                }
192                n.flag(LayoutFlags.corner);
193                use(n,LayoutTypes.surround);
194        }

196        public boolean fillMaze(final LayoutNode p)
197        {
198                final Vector<Integer> dirs = new Vector<Integer>();
199                for(int i=0;i<4;i++)
200                        dirs.add(Integer.valueOf(i));
201                final Vector<Integer> rdirs = new Vector<Integer>();
202                while(dirs.size()>0)
203                {
204                        final int x = r.nextInt(dirs.size());
205                        final Integer dir = dirs.elementAt(x);
206                        dirs.removeElementAt(x);
207                        rdirs.addElement(dir);
208                }
209                for(int r=0;r<rdirs.size();r++)
210                {
211                        final Integer dir = rdirs.elementAt(r);
212                        LayoutNode p2 = getNextNode(p, dir.intValue());
213                        if(p2 == null)
214                        {
215                                p2 = makeNextNode(p, dir.intValue());
216                                p.crossLink(p2);
217                                use(p2,LayoutTypes.interior);
218                                fillMaze(p2);
219                        }
220                }
221                return true;
222        }

224        public void clipLongStreets()
225        {
226                final Vector<LayoutNode> set2= new Vector<LayoutNode>(set());
227                for (final LayoutNode p : set2)
228                {
229                        if(isUsed(p) && p.isStreetLike())
230                        {
231                                for(int d=0;d<4;d++)
232                                {
233                                        if(p.getLink(d)==null)
234                                        {
235                                                final LayoutNode p2 =getNextNode(p, d);
236                                                if((p2!=null)
237                                                &&(!p.links().containsValue(p2)))
238                                                {
239                                                        final Iterator<LayoutNode> nodes=p.links().values().iterator();
240                                                        final LayoutNode p_1=nodes.next();
241                                                        final LayoutNode p_2=nodes.next();
242                                                        p.deLink();
243                                                        p_1.crossLink(p_2);
244                                                        unUse(p);
245                                                        final LayoutNode p3 = makeNextNode(p2, Directions.getOpDirectionCode(d));
246                                                        p2.crossLink(p3);
247                                                        use(p3, LayoutTypes.leaf);
248                                                        break;
249                                                }
250                                        }
251                                }
252                        }
253                }
254        }

256        public void fillInFlags()
257        {
258                for (final LayoutNode n : set())
259                {
260                        final int[] dirs=new int[n.links().size()];
261                        int x=0;
262                        for(final Integer dirLink : n.links().keySet())
263                                dirs[x++]=dirLink.intValue();
264                        n.setExits(dirs);
265                        if((dirs.length==1)&&(!n.isFlagged(LayoutFlags.gate)))
266                                n.reType(LayoutTypes.leaf);
267                }
268                for (final LayoutNode n : set())
269                {
270                        if(n.links().size()==2)
271                        {
272                                LayoutFlags flag = null;
273                                if(n.type()==LayoutTypes.interior)
274                                {
275                                        for(final Integer dirLink : n.links().keySet())
276                                        {
277                                                final LayoutNode n2=n.links().get(dirLink);
278                                                if((n2!=null)&&(n2.type()==LayoutTypes.leaf))
279                                                        flag=LayoutFlags.offleaf;
280                                        }
281                                }
282                                if(flag!=null)
283                                        n.flag(flag);
284                                else
285                                {
286                                        final Iterator<Integer> dirs=n.links().keySet().iterator();
287                                        final Integer lN1=dirs.next();
288                                        final Integer lN2=dirs.next();
289                                        if(lN1.intValue() != Directions.getOpDirectionCode(lN2.intValue()))
290                                                n.flag(LayoutFlags.corner);
291                                }
292                        }
293                        else
294                        if((n.links().size()==3)
295                        &&(((n.type()==LayoutTypes.street)
296                                ||(n.type()!=LayoutTypes.surround))))
297                        {
298                                boolean allStreet = true;
299                                for(final Integer dirLink : n.links().keySet())
300                                {
301                                        final LayoutNode n2=n.links().get(dirLink);
302                                        if((n2==null)
303                                        ||((n2.type()!=LayoutTypes.street)
304                                                &&(n2.type()!=LayoutTypes.surround)))
305                                                        allStreet = false;
306                                }
307                                if(allStreet)
308                                        n.flag(LayoutFlags.tee);
309                        }
310                        else
311                        if((n.links().size()==4)
312                        &&(((n.type()==LayoutTypes.street)
313                                ||(n.type()!=LayoutTypes.surround))))
314                        {
315                                boolean allStreet = true;
316                                for(final Integer dirLink : n.links().keySet())
317                                {
318                                        final LayoutNode n2=n.links().get(dirLink);
319                                        if((n2==null)
320                                        ||((n2.type()!=LayoutTypes.street)
321                                                &&(n2.type()!=LayoutTypes.surround)))
322                                                        allStreet = false;
323                                }
324                                if(allStreet)
325                                        n.flag(LayoutFlags.intersection);
326                        }
327                }
328        }
