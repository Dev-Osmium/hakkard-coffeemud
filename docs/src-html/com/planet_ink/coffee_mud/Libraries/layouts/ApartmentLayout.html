030public class ApartmentLayout extends AbstractLayout
031{
032        @Override
033        public String name()
034        {
035                return "APARTMENT";
036        }
037
038        public void setRunFromDirection(final LayoutNode node, final int dir)
039        {
040                switch(dir)
041                {
042                case Directions.NORTH:
043                case Directions.SOUTH:
044                        node.flagRun(LayoutRuns.ns);
045                        break;
046                case Directions.EAST:
047                case Directions.WEST:
048                        node.flagRun(LayoutRuns.ew);
049                        break;
050                default:
051                        setRunFromDirection(node,CMLib.dice().pick(Directions.CODES())); // picks one of the above, and only one of the above
052                        break;
053                }
054        }
055
056        @Override
057        public java.util.List<LayoutNode> generate(final int num, final int dir)
058        {
059                final Vector<LayoutNode> set = new Vector<LayoutNode>();
060                int hallwayLength=num/3;
061                int numHallways=1;
062                while(hallwayLength > 7)
063                {
064                        hallwayLength = hallwayLength / 2;
065                        numHallways *= 2;
066                }
067
068                final LayoutSet lSet = new LayoutSet(set,num);
069                LayoutNode n = null;
070                int hallwayDirection=dir;
071                int sidewayDirection=dir;
072                final int fullNumHallway = numHallways;
073                switch(dir)
074                {
075                case Directions.NORTH:
076                        n = new DefaultLayoutNode(new long[] { 0, hallwayLength });
077                        sidewayDirection = Directions.EAST;
078                        break;
079                case Directions.SOUTH:
080                        n = new DefaultLayoutNode(new long[] { fullNumHallway, 0 });
081                        sidewayDirection = Directions.WEST;
082                        break;
083                case Directions.EAST:
084                        n = new DefaultLayoutNode(new long[] { 0, 0 });
085                        sidewayDirection = Directions.SOUTH;
086                        break;
087                case Directions.WEST:
088                        n = new DefaultLayoutNode(new long[] { fullNumHallway, hallwayLength });
089                        sidewayDirection = Directions.NORTH;
090                        break;
091                case Directions.NORTHEAST:
092                        n = new DefaultLayoutNode(new long[] { 0, hallwayLength });
093                        sidewayDirection = Directions.EAST;
094                        hallwayDirection = Directions.NORTH;
095                        break;
096                case Directions.NORTHWEST:
097                        n = new DefaultLayoutNode(new long[] { fullNumHallway, hallwayLength });
098                        sidewayDirection = Directions.WEST;
099                        hallwayDirection = Directions.NORTH;
100                        break;
101                case Directions.SOUTHEAST:
102                        n = new DefaultLayoutNode(new long[] { 0, 0 });
103                        sidewayDirection = Directions.SOUTH;
104                        hallwayDirection = Directions.EAST;
105                        break;
106                case Directions.SOUTHWEST:
107                        n = new DefaultLayoutNode(new long[] { fullNumHallway, 0 });
108                        sidewayDirection = Directions.WEST;
109                        hallwayDirection = Directions.SOUTH;
110                        break;
111                }
112                if(n!=null)
113                {
114                        final java.util.List<LayoutNode> hallways=new ArrayList<LayoutNode>();
115                        hallways.add(n);
116                        //n.flagGateExit(dir);
117                        lSet.use(n,LayoutTypes.street);
118                        n.flag(LayoutFlags.gate);
119                        n.flag(LayoutFlags.tee);
120                        setRunFromDirection(n,sidewayDirection);
121                        for(int h=1;h<numHallways;h++)
122                        {
123                                LayoutNode prevNode=n;
124                                LayoutNode nextNode=lSet.makeNextNode(prevNode, sidewayDirection);
125                                lSet.use(nextNode,LayoutTypes.street);
126                                nextNode.crossLink(prevNode);
127                                setRunFromDirection(nextNode,sidewayDirection);
128
129                                prevNode=nextNode;
130                                nextNode=lSet.makeNextNode(prevNode, sidewayDirection);
131                                lSet.use(nextNode,LayoutTypes.street);
132                                nextNode.crossLink(prevNode);
133                                setRunFromDirection(nextNode,sidewayDirection);
134
135                                prevNode=nextNode;
136                                nextNode=lSet.makeNextNode(prevNode, sidewayDirection);
137                                lSet.use(nextNode,LayoutTypes.street);
138                                nextNode.crossLink(prevNode);
139                                setRunFromDirection(nextNode,sidewayDirection);
140                                nextNode.flag(LayoutFlags.tee);
141
142                                hallways.add(nextNode);
143                                n=nextNode;
144                        }
145                        for(final LayoutNode hallwayNode : hallways)
146                        {
147                                LayoutNode prevNode=hallwayNode;
148                                for(int h=0;h<hallwayLength;h++)
149                                {
150                                        LayoutNode nextNode=lSet.makeNextNode(prevNode, hallwayDirection);
151                                        lSet.use(nextNode,LayoutTypes.street);
152                                        nextNode.crossLink(prevNode);
153                                        setRunFromDirection(nextNode,hallwayDirection);
154
155                                        prevNode=nextNode; // this should stick
156
157                                        nextNode=lSet.makeNextNode(prevNode, sidewayDirection);
158                                        lSet.use(nextNode,LayoutTypes.leaf);
159                                        nextNode.crossLink(prevNode);
160
161                                        nextNode=lSet.makeNextNode(prevNode, Directions.getOpDirectionCode(sidewayDirection));
162                                        lSet.use(nextNode,LayoutTypes.leaf);
163                                        nextNode.crossLink(prevNode);
164                                }
165                        }
166                        lSet.fillInFlags();
167                }
168                return set;
169        }
