024public class LayoutTester
025{
026        
027        private final static String[] DEFAULT_DIRECTIONS_LIST_COMPASS   = "North,South,East,West,Up,Down,There,Northeast,Northwest,Southeast,Southwest".split(",");
028        
029        public static void draw(LayoutManager layout, int size, int dir)
030        {
031                final List<LayoutNode> V=layout.generate(size, dir);
032                final LayoutNode firstNode = (V.size()==0) ? null : V.get(0);
033
034                System.out.println("Layout "+layout.name()+", size="+V.size()+", dir="+DEFAULT_DIRECTIONS_LIST_COMPASS[dir]+": "+continuityCheck(V));
035                long lowestX=Long.MAX_VALUE;
036                long lowestY=Long.MAX_VALUE;
037                long highestX=Long.MIN_VALUE;
038                long highestY=Long.MIN_VALUE;
039                final Hashtable<Long,Vector<LayoutNode>> HY = new Hashtable<Long,Vector<LayoutNode>>();
040                for(final LayoutNode ls : V)
041                {
042                        if(ls.coord()[0]<lowestX)
043                                lowestX = ls.coord()[0];
044                        if(ls.coord()[1]<lowestY)
045                                lowestY = ls.coord()[1];
046                        if(ls.coord()[0]>highestX)
047                                highestX = ls.coord()[0];
048                        if(ls.coord()[1]>highestY)
049                                highestY = ls.coord()[1];
050                        if(!HY.containsKey(Long.valueOf(ls.coord()[1])))
051                                HY.put(Long.valueOf(ls.coord()[1]), new Vector<LayoutNode>());
052                        HY.get(Long.valueOf(ls.coord()[1])).add(ls);
053
054                }
055                for(long y=lowestY;y<=highestY;y++)
056                {
057                        final Vector<LayoutNode> ys = HY.get(Long.valueOf(y));
058                        if(ys != null)
059                        {
060                                final Hashtable<Long,LayoutNode> H = new Hashtable<Long,LayoutNode>();
061                                for(final LayoutNode xs : ys) 
062                                        H.put(Long.valueOf(xs.coord()[0]),xs);
063                                for(int i=0;i<3;i++)
064                                {
065                                        for(long x=lowestX;x<=highestX;x++)
066                                        {
067                                                if(H.containsKey(Long.valueOf(x)))
068                                                {
069                                                        LayoutNode n = H.get(Long.valueOf(x));
070                                                        System.out.print(n.getColorRepresentation((firstNode==n ? 'O':'*'),i));
071                                                }
072                                                else
073                                                        System.out.print("   ");
074                                        }
075                                        System.out.println("");
076                                }
077                        }
078                }
079        }
080
081        public static boolean continuityCheck(List<LayoutNode> set)
082        {
083                for(int s=0;s<set.size();s++)
084                {
085                        final LayoutNode node = set.get(s);
086                        for(final Iterator<LayoutNode> e=node.links().values().iterator();e.hasNext();)
087                        {
088                                if(!set.contains(e.next()))
089                                        return false;
090                        }
091                }
092                return true;
093        }
