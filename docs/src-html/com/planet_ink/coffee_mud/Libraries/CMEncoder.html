038        private byte[]                  encodeBuffer    = new byte[65536];
039        private final Deflater  compresser              = new Deflater(Deflater.BEST_COMPRESSION);
040        private final Inflater  decompresser    = new Inflater();

048        @Override
049        public synchronized String decompressString(byte[] b)
050        {
051                try
052                {
053                        if ((b == null)||(b.length==0)) return "";
054
055                        decompresser.reset();
056                        decompresser.setInput(b);
057
058                        synchronized (encodeBuffer)
059                        {
060                                final int len = decompresser.inflate(encodeBuffer);
061                                return new String(encodeBuffer, 0, len, CMProps.getVar(CMProps.Str.CHARSETINPUT));
062                        }
063                }
064                catch (final Exception ex)
065                {
066                        Log.errOut(Thread.currentThread().getName(), "Error occurred during decompression: "+ex.getMessage());
067                        encodeBuffer=new byte[65536];
068                        return "";
069                }
070        }
071
072        @Override
073        public synchronized byte[] compressString(String s)
074        {
075                byte[] result = null;
076
077                try
078                {
079                        compresser.reset();
080                        compresser.setInput(s.getBytes(CMProps.getVar(CMProps.Str.CHARSETINPUT)));
081                        compresser.finish();
082
083                        synchronized (encodeBuffer)
084                        {
085                                if(s.length()>encodeBuffer.length)
086                                        encodeBuffer=new byte[s.length()];
087                                encodeBuffer[0]=0;
088
089                                final int len = compresser.deflate(encodeBuffer);
090                                result = new byte[len];
091                                System.arraycopy(encodeBuffer, 0, result, 0, len);
092                        }
093                }
094                catch (final Exception ex)
095                {
096                        Log.errOut("MUD", "Error occurred during compression: "+ex.getMessage());
097                        encodeBuffer=new byte[65536];
098                }
099
100                return result;
101        }

103        @Override
104        public String makeRandomHashString(final String password)
105        {
106                final int salt=(int)Math.round(CMath.random() * Integer.MAX_VALUE);
107                final int passHash=(password+salt).toLowerCase().hashCode();
108                return "|"+B64Encoder.B64encodeBytes(ByteBuffer.allocate(4).putInt(salt).array())
109                          +"|"+B64Encoder.B64encodeBytes(ByteBuffer.allocate(4).putInt(passHash).array());
110        }
111
112        @Override
113        public String makeRepeatableHashString(final String str)
114        {
115                final int passHash=str.toLowerCase().hashCode();
116                return "|"+B64Encoder.B64encodeBytes(enDeCrypt(ByteBuffer.allocate(4).putInt(passHash).array()));
117        }

119        @Override
120        public boolean isARandomHashString(final String password)
121        {
122                return ((password.length()>2) && (password.startsWith("|")) && (password.indexOf('|',1)>1));
123        }
124
125        @Override
126        public boolean checkPasswordAgainstRandomHashString(final String passwordString, final String hashString)
127        {
128                final int hashDex=hashString.indexOf('|',1);
129                final int salt=ByteBuffer.wrap(B64Encoder.B64decode(hashString.substring(1,hashDex))).getInt();
130                final int hash=ByteBuffer.wrap(B64Encoder.B64decode(hashString.substring(hashDex+1))).getInt();
131                return hash==(passwordString+salt).toLowerCase().hashCode();
132        }
133
134        @Override
135        public boolean checkHashStringPairs(final String hashString1, final String hashString2)
136        {
137                final int hashDex1=hashString1.indexOf('|',1);
138                final int salt1=ByteBuffer.wrap(B64Encoder.B64decode(hashString1.substring(1,hashDex1))).getInt();
139                final int hash1=ByteBuffer.wrap(B64Encoder.B64decode(hashString1.substring(hashDex1+1))).getInt();
140                
141                final int hashDex2=hashString2.indexOf('|',1);
142                final int salt2=ByteBuffer.wrap(B64Encoder.B64decode(hashString2.substring(1,hashDex2))).getInt();
143                final int hash2=ByteBuffer.wrap(B64Encoder.B64decode(hashString2.substring(hashDex2+1))).getInt();
144                
145                return (hash1==(hashString2+salt1).toLowerCase().hashCode()) || (hash2==(hashString1+salt2).toLowerCase().hashCode());
146        }

148        @Override
149        public String generateRandomPassword()
150        {
151                final StringBuilder str=new StringBuilder("");
152                for(int i=0;i<10;i++)
153                {
154                        if((i%2)==0)
155                                str.append(CMLib.dice().roll(1, 10, -1));
156                        else
157                                str.append((char)('a'+CMLib.dice().roll(1, 26, -1)));
158                }
159                return str.toString();
160        }

162        @Override
163        public boolean passwordCheck(String pass1, String pass2)
164        {
165                if(pass1 == null)
166                {
167                        return (pass2==null);
168                }
169                else
170                if(pass2 == null)
171                        return false;
172                
173                if(pass1.equalsIgnoreCase(pass2))
174                        return true;
175                if(isARandomHashString(pass2))
176                {
177                        if(isARandomHashString(pass1))
178                                return checkHashStringPairs(pass1,pass2);
179                        else
180                                return checkPasswordAgainstRandomHashString(pass1, pass2);
181                }
182                else
183                {
184                        if(isARandomHashString(pass1))
185                                return checkPasswordAgainstRandomHashString(pass2, pass1);
186                        return pass1.equalsIgnoreCase(pass2);
187                }
188        }

190        public static byte[] getFilter()
191        {
192                if(encrFilter==null)
193                {
194                        // this is coffeemud's unsophisticated xor(mac address) encryption system.
195                        final byte[] filterc = new String("wrinkletellmetrueisthereanythingasnastyasyouwellmaybesothenumber7470issprettybad").getBytes();
196                        encrFilter=new byte[256];
197                        try
198                        {
199                                for(int i=0;i<256;i++)
200                                        encrFilter[i]=filterc[i % filterc.length];
201                                final String domain=CMProps.getVar(CMProps.Str.MUDDOMAIN);
202                                if(domain.length()>0)
203                                {
204                                        for(int i=0;i<256;i++)
205                                                encrFilter[i]^=domain.charAt(i % domain.length());
206                                }
207                                final String name=CMProps.getVar(CMProps.Str.MUDNAME);
208                                if(name.length()>0)
209                                {
210                                        for(int i=0;i<256;i++)
211                                                encrFilter[i]^=name.charAt(i % name.length());
212                                }
213                                final String email=CMProps.getVar(CMProps.Str.ADMINEMAIL);
214                                if(email.length()>0)
215                                {
216                                        for(int i=0;i<256;i++)
217                                                encrFilter[i]^=email.charAt(i % email.length());
218                                }
219                                for(final Enumeration<NetworkInterface> nie = NetworkInterface.getNetworkInterfaces(); nie.hasMoreElements();)
220                                {
221                                        final NetworkInterface ni = nie.nextElement();
222                                        if(ni != null)
223                                        {
224                                                final byte[] mac = ni.getHardwareAddress();
225                                                if((mac != null) && (mac.length > 0))
226                                                {
227                                                        for(int i=0;i<256;i++)
228                                                                encrFilter[i]^=Math.abs(mac[i % mac.length]);
229                                                }
230                                        }
231                                }
232                        }
233                        catch(final Exception e)
234                        {
235                                e.printStackTrace();
236                        }
237                }
238                return encrFilter;
239        }

241        protected byte[] enDeCrypt(byte[] bytes)
242        {
243                final byte[] encrFilter=getFilter();
244                for ( int i = 0, j = 0; i < bytes.length; i++, j++ )
245                {
246                        if ( j >= encrFilter.length ) 
247                                j = 0;
248                        bytes[i]=(byte)((bytes[i] ^ encrFilter[j]) & 0xff);
249                }
250                return bytes;
251        }

253        @Override
254        public String filterEncrypt(String str)
255        {
256                try
257                {
258                        final byte[] buf=B64Encoder.B64encodeBytes(enDeCrypt(str.getBytes()),B64Encoder.DONT_BREAK_LINES).getBytes();
259                        final StringBuilder s=new StringBuilder("");
260                        for(final byte b : buf)
261                        {
262                                String s2=Integer.toHexString(b);
263                                while(s2.length()<2)
264                                        s2="0"+s2;
265                                s.append(s2);
266                        }
267                        return s.toString();
268                }
269                catch(final Exception e)
270                {
271                        return "";
272                }
273        }
274
275        @Override
276        public String filterDecrypt(String str)
277        {
278                try
279                {
280                        final byte[] buf=new byte[str.length()/2];
281                        for(int i=0;i<str.length();i+=2)
282                                buf[i/2]=(byte)(Integer.parseInt(str.substring(i,i+2),16) & 0xff);
283                        return new String(enDeCrypt(B64Encoder.B64decode(new String(buf))));
284                }
285                catch(final Exception e)
286                {
287                        return "";
288                }
289        }
