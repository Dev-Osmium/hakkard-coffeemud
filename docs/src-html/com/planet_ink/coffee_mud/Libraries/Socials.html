048        @Override
049        public void putSocialsInHash(final Map<String,List<Social>> soc, final List<String> lines)
050        {
051                for(int v=0;v<lines.size();v++)
052                {
053                        String getline=lines.get(v);
054                        int x=getline.indexOf("\t");
055                        if(x>=0)
056                        {
057                                final Social socobj=(Social)CMClass.getCommon("DefaultSocial");
058                                final String s=getline.substring(0,x).toUpperCase();
059                                if(s.length()>0)
060                                switch(s.charAt(0))
061                                {
062                                case 'W':
063                                        socobj.setSourceCode(CMMsg.MSG_SPEAK);
064                                        break;
065                                case 'M':
066                                        socobj.setSourceCode(CMMsg.MSG_HANDS);
067                                        break;
068                                case 'S':
069                                        socobj.setSourceCode(CMMsg.MSG_NOISE);
070                                        break;
071                                case 'O':
072                                        socobj.setSourceCode(CMMsg.MSG_NOISYMOVEMENT);
073                                        break;
074                                default:
075                                        socobj.setSourceCode(CMMsg.MSG_HANDS);
076                                        break;
077                                }
078                                if(s.length()>1)
079                                switch(s.charAt(1))
080                                {
081                                case 'T':
082                                        socobj.setOthersCode(CMMsg.MSG_HANDS);
083                                        socobj.setTargetCode(CMMsg.MSG_HANDS);
084                                        break;
085                                case 'S':
086                                        socobj.setOthersCode(CMMsg.MSG_NOISE);
087                                        socobj.setTargetCode(CMMsg.MSG_NOISE);
088                                        break;
089                                case 'W':
090                                        socobj.setOthersCode(CMMsg.MSG_SPEAK);
091                                        socobj.setTargetCode(CMMsg.MSG_SPEAK);
092                                        break;
093                                case 'V':
094                                        socobj.setOthersCode(CMMsg.MSG_NOISYMOVEMENT);
095                                        socobj.setTargetCode(CMMsg.MSG_NOISYMOVEMENT);
096                                        break;
097                                case 'O':
098                                        socobj.setOthersCode(CMMsg.MSG_OK_VISUAL);
099                                        socobj.setTargetCode(CMMsg.MSG_OK_VISUAL);
100                                        break;
101                                default:
102                                        socobj.setOthersCode(CMMsg.MSG_NOISYMOVEMENT);
103                                        socobj.setTargetCode(CMMsg.MSG_NOISYMOVEMENT);
104                                        break;
105                                }
106                                getline=getline.substring(x+1);
107                                x=getline.indexOf("\t");
108                                if(x>=0)
109                                {
110                                        socobj.setName(getline.substring(0,x).toUpperCase());
111                                        getline=getline.substring(x+1);
112                                        x=getline.indexOf("\t");
113                                        if(x>=0)
114                                        {
115                                                socobj.setSourceMessage(getline.substring(0,x));
116                                                getline=getline.substring(x+1);
117                                                x=getline.indexOf("\t");
118                                                if(x>=0)
119                                                {
120                                                        socobj.setOthersMessage(getline.substring(0,x));
121                                                        getline=getline.substring(x+1);
122                                                        x=getline.indexOf("\t");
123                                                        if(x>=0)
124                                                        {
125                                                                socobj.setTargetMessage(getline.substring(0,x));
126                                                                getline=getline.substring(x+1);
127                                                                x=getline.indexOf("\t");
128                                                                if(x>=0)
129                                                                {
130                                                                        socobj.setFailedMessage(getline.substring(0,x));
131                                                                        getline=getline.substring(x+1);
132                                                                        x=getline.indexOf("\t");
133                                                                        if(x>=0)
134                                                                                socobj.setSoundFile(getline.substring(0,x));
135                                                                        else
136                                                                                socobj.setSoundFile(getline);
137                                                                }
138                                                                else
139                                                                        socobj.setFailedMessage(getline);
140
141                                                        }
142                                                }
143                                                put(soc,socobj.Name(),socobj);
144                                        }
145                                }
146                        }
147                }
148        }

152        {
153                Map<String,List<Social>> soc=(Map<String,List<Social>>)Resources.getResource("PARSED_SOCIALS: "+filename);

171        private String realName(String name)
172        {
173                String shortName=name.toUpperCase().trim();
174                final int spdex=shortName.indexOf(' ');
175                if(spdex>0)
176                        shortName=shortName.substring(0,spdex);
177                return shortName;
178        }

180        private void put(Map<String,List<Social>> H, String name, Social S)
181        {
182                name=realName(name);
183                List<Social> V2=H.get(name);
184                if(V2==null)
185                {
186                        V2=new Vector<Social>(4);
187                        H.put(name,V2);
188                }
189                for(int v=0;v<V2.size();v++)
190                {
191                        if(V2.get(v).Name().equalsIgnoreCase(S.Name()))
192                        {
193                                V2.remove(v);
194                                break;
195                        }
196                }
197                V2.add(S);
198        }

206        @Override
207        public void remove(String name)
208        {
209                final String realName=realName(name);
210                final List<Social> V2=getSocialHash().get(realName);
211                if(V2==null)
212                        return;
213                for(int v=0;v<V2.size();v++)
214                {
215                        if(V2.get(v).Name().equalsIgnoreCase(name))
216                        {
217                                V2.remove(v);
218                                if(V2.size()==0)
219                                {
220                                        getSocialHash().remove(realName);
221                                        unloadDerivedResources();
222                                }
223                                break;
224                        }
225                }
226        }

373        /**
374         * Finds the first and only the first instance of the second parameter string in the first string,
375         * and replaces it with the third string.  Returns the first string with or without changes. 
376         * This method is case sensitive.
377         * @param str the string to look inside of
378         * @param thisStr the string to look for inside the first string
379         * @param withThisStr the string to replace the second string with, if found.
380         * @return the string modified, or not modified if no replacement was made.
381         */
382        protected final static String replaceFirstWordStartsWith(final String str, final String thisStr, final String withThisStr)
383        {
384                if((str==null)
385                ||(thisStr==null)
386                ||(withThisStr==null)
387                ||(str.length()==0)
388                ||(thisStr.length()==0))
389                        return str;
390                if(Character.toUpperCase(str.charAt(0))==Character.toUpperCase(thisStr.charAt(0)))
391                {
392                        if(str.toUpperCase().startsWith(thisStr.toUpperCase())
393                        &&(str.toUpperCase().equals(thisStr.toUpperCase())
394                                ||((str.length()>thisStr.length())&&(Character.isWhitespace(str.charAt(thisStr.length()))))))
395                                return withThisStr+str.substring(thisStr.length());
396                        if(thisStr.toUpperCase().startsWith(str.toUpperCase()))
397                                return withThisStr;
398                        int x=str.indexOf(' ');
399                        if(x>1)
400                        {
401                                if(thisStr.toUpperCase().startsWith(str.substring(0,x).toUpperCase()))
402                                        return withThisStr+str.substring(x+1);
403                        }
404                }
405                return str;
406        }

408        @Override
409        public boolean modifySocialInterface(MOB mob, String socialString)
410                throws IOException
411        {
412                final Vector<String> socialsParse=CMParms.parse(socialString);
413                if(socialsParse.size()==0)
414                {
415                        mob.tell(L("Which social?"));
416                        return false;
417                }
418                final String name=socialsParse.firstElement().toUpperCase().trim();
419                String rest=socialsParse.size()>1?CMParms.combine(socialsParse,1):"";
420                List<Social> socials=getSocialsSet(socialsParse.firstElement());
421                if(((socials==null)||(socials.size()==0))
422                &&((mob.session()==null)
423                        ||(!mob.session().confirm(L("The social '@x1' does not exist.  Create it (y/N)? ",name),"N"))))
424                        return false;
425                if(socials==null)
426                        socials=new Vector<Social>();
427                boolean resaveSocials=true;
428                while((resaveSocials)&&(mob.session()!=null)&&(!mob.session().isStopped()))
429                {
430                        resaveSocials=false;
431                        Social soc=null;
432                        boolean pickNewSocial=true;
433                        while((pickNewSocial)&&(mob.session()!=null)&&(!mob.session().isStopped()))
434                        {
435                                pickNewSocial=false;
436                                final StringBuffer str=new StringBuffer(L("\n\rSelect a target:\n\r"));
437                                int selection=-1;
438                                for(int v=0;v<socials.size();v++)
439                                {
440                                        final Social S=socials.get(v);
441                                        final int x=S.Name().indexOf(' ');
442                                        final int y=(x<0)?-1:S.Name().indexOf(' ',x+1);
443                                        if(x<0)
444                                        {
445                                                str.append((v+1)+") No Target (NONE)\n\r");
446                                                continue;
447                                        }
448                                        if((rest.length()>0)
449                                        &&(S.Name().substring(x+1).toUpperCase().trim().equalsIgnoreCase(rest.toUpperCase().trim())))
450                                                selection=(v+1);
451                                        if(S.Name().substring(x+1).toUpperCase().trim().startsWith("<T-NAME>"))
452                                        {
453                                                str.append((v+1)+") MOB Targeted (MOBTARGET)");
454                                                if(y>x)
455                                                        str.append(" with argument ("+S.Name().substring(y+1)+")");
456                                                str.append("\n\r");
457                                                continue;
458                                        }
459                                        if(S.Name().substring(x+1).toUpperCase().trim().startsWith("<I-NAME>"))
460                                        {
461                                                str.append((v+1)+") Room Item Targeted (ITEMTARGET)");
462                                                if(y>x)
463                                                        str.append(" with argument ("+S.Name().substring(y+1)+")");
464                                                str.append("\n\r");
465                                                continue;
466                                        }
467                                        if(S.Name().substring(x+1).toUpperCase().trim().startsWith("<V-NAME>"))
468                                        {
469                                                str.append((v+1)+") Inventory Targeted (INVTARGET)");
470                                                if(y>x)
471                                                        str.append(" with argument ("+S.Name().substring(y+1)+")");
472                                                str.append("\n\r");
473                                                continue;
474                                        }
475                                        if(S.Name().substring(x+1).toUpperCase().trim().startsWith("<E-NAME>"))
476                                        {
477                                                str.append((v+1)+") Equipment Targeted (EQUIPTARGET)");
478                                                if(y>x)
479                                                        str.append(" with argument ("+S.Name().substring(y+1)+")");
480                                                str.append("\n\r");
481                                                continue;
482                                        }
483                                        str.append((v+1)+") "+S.Name().substring(x+1).toUpperCase().trim());
484                                        if(y>x)
485                                                str.append(" with argument ("+S.Name().substring(y+1)+")");
486                                        str.append("\n\r");
487                                }
488                                str.append(L("@x1) Add a new target\n\r",""+(socials.size()+1)));
489                                String s=null;
490                                if((rest.length()>0)&&(selection<0))
491                                        selection=(socials.size()+1);
492                                else
493                                if(selection<0)
494                                {
495                                        mob.session().safeRawPrintln(str.toString());
496                                        s=mob.session().prompt(L("\n\rSelect an option or RETURN: "),"");
497                                        if(!CMath.isInteger(s))
498                                        {
499                                                soc=null;
500                                                break;
501                                        }
502                                        selection=CMath.s_int(s);
503                                }
504                                if((selection>0)&&(selection<=socials.size()))
505                                {
506                                        soc=socials.get(selection-1);
507                                        break;
508                                }
509                                String newOne=rest;
510                                if(newOne.length()==0)
511                                {
512                                        newOne="?";
513                                        while((newOne.equals("?"))&&(!mob.session().isStopped()))
514                                        {
515                                                newOne=mob.session().prompt(L("\n\rNew target (?): "),"").toUpperCase().trim();
516                                                if(newOne.equals("?"))
517                                                        mob.session().println(L("Choices: MOBTARGET, ITEMTARGET, INVTARGET, EQUIPTARGET, NONE, ALL, SELF"));
518                                        }
519                                        if(newOne.trim().length()==0)
520                                        {
521                                                pickNewSocial=true;
522                                                continue;
523                                        }
524                                }
525                                newOne=newOne.toUpperCase().trim();
526                                final String[] validTargets={"NONE","ALL","SELF","<T-NAME>","<I-NAME>","<V-NAME>","<E-NAME>"};
527                                final String[] friendlyTargets={"NONE","ALL","SELF","MOBS","ITEMS","INVENTORY","EQUIPMENT"};
528                                newOne=replaceFirstWordStartsWith(newOne, "TNAME", "<T-NAME>");
529                                newOne=replaceFirstWordStartsWith(newOne, "INAME", "<I-NAME>");
530                                newOne=replaceFirstWordStartsWith(newOne, "VNAME", "<V-NAME>");
531                                newOne=replaceFirstWordStartsWith(newOne, "ENAME", "<E-NAME>");
532                                newOne=replaceFirstWordStartsWith(newOne, "T-NAME", "<T-NAME>");
533                                newOne=replaceFirstWordStartsWith(newOne, "I-NAME", "<I-NAME>");
534                                newOne=replaceFirstWordStartsWith(newOne, "V-NAME", "<V-NAME>");
535                                newOne=replaceFirstWordStartsWith(newOne, "E-NAME", "<E-NAME>");
536                                newOne=replaceFirstWordStartsWith(newOne, "MOBTARGET", "<T-NAME>");
537                                newOne=replaceFirstWordStartsWith(newOne, "ITEMTARGET", "<I-NAME>");
538                                newOne=replaceFirstWordStartsWith(newOne, "INVTARGET", "<V-NAME>");
539                                newOne=replaceFirstWordStartsWith(newOne, "EQUIPTARGET", "<E-NAME>");
540                                final int spaceDex=newOne.indexOf(' ');
541                                final String tag=(spaceDex<0) ? newOne : newOne.substring(0,spaceDex).trim();
542                                final int foundDex=CMParms.indexOf(validTargets, tag);
543                                final String friendlyTag = (foundDex < 0) ? "" : friendlyTargets[foundDex];
544                                if(newOne.startsWith("<"))
545                                {
546                                        if(foundDex<0)
547                                        {
548                                                mob.tell(L("@x1 is an invalid target.  Enter ? for some choices."));
549                                                pickNewSocial=true;
550                                                continue;
551                                        }
552                                }
553                                if(newOne.equalsIgnoreCase("NONE"))
554                                        newOne="";
555                                else
556                                if((foundDex<0)
557                                &&(!mob.session().confirm(L("'@x1' is a non-standard target.  Are you sure (y/N)? ",tag),"N")))
558                                {
559                                        rest="";
560                                        pickNewSocial=true;
561                                }
562                                else
563                                if((spaceDex > 0)
564                                &&(!mob.session().confirm(L("Target: '@x1' is a valid target, with an argument of @x2.  Is this OK (Y/n)? ",friendlyTag,newOne.substring(spaceDex+1)),"Y")))
565                                {
566                                        rest="";
567                                        pickNewSocial=true;
568                                }
569                                else
570                                        newOne=" "+newOne;
571                                if(!pickNewSocial)
572                                {
573                                        for(int i=0;i<socials.size();i++)
574                                        {
575                                                if(socials.get(i).Name().equals(name+newOne))
576                                                {
577                                                        mob.tell(L("This social already exists.  Pick it off the list above."));
578                                                        pickNewSocial=true;
579                                                        break;
580                                                }
581                                        }
582                                        if(!pickNewSocial)
583                                        {
584                                                soc=makeDefaultSocial(name,newOne);
585                                                addSocial(soc);
586                                                if(!socials.contains(soc))
587                                                        socials.add(soc);
588                                                resaveSocials=true;
589                                        }
590                                }
591                        }
592                        if(soc!=null)
593                        {
594                                boolean ok=false;
595                                int showFlag=-1;
596                                if(CMProps.getIntVar(CMProps.Int.EDITORTYPE)>0)
597                                        showFlag=-999;
598                                while(!ok)
599                                {
600                                        int showNumber=0;

628                                                break;
629                                        }
630                                        if(showFlag>0)

649                                                ok=true;

652                        }

654                        {
655                                save(mob);

657                                soc=null;
