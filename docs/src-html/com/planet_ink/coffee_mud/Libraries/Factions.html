120        @Override
121        public String rangeDescription(Faction.FRange FR, String andOr)
122        {
123                if(FR==null)
124                        return "";
125                final Faction F=FR.getFaction();
126                final Vector<FRange> relevantFactions=new Vector<FRange>();
127                for(final Enumeration<FRange> e=F.ranges();e.hasMoreElements();)
128                {
129                        final Faction.FRange FR2=e.nextElement();
130                        if(FR2.codeName().equalsIgnoreCase(FR.codeName()))
131                                relevantFactions.addElement(FR2);
132                }
133                if(relevantFactions.size()==0)
134                        return "";
135                if(relevantFactions.size()==1)
136                        return F.name()+" of "+relevantFactions.firstElement().name();
137                final StringBuffer buf=new StringBuffer(F.name()+" of ");
138                for(int i=0;i<relevantFactions.size()-1;i++)
139                        buf.append(relevantFactions.elementAt(i).name()+", ");
140                buf.append(andOr+relevantFactions.lastElement().name());
141                return buf.toString();
142        }

144        private Faction buildFactionFromXML(StringBuffer buf, String factionID)
145        {
146                final Faction F=(Faction)CMClass.getCommon("DefaultFaction");
147                F.initializeFaction(buf,factionID);
148                for(final Enumeration<FRange> e=F.ranges();e.hasMoreElements();)
149                {
150                        final Faction.FRange FR=e.nextElement();
151                        final String CodeName=(FR.codeName().length()>0)?FR.codeName().toUpperCase():FR.name().toUpperCase();
152                        if(!hashedFactionRanges.containsKey(CodeName))
153                                hashedFactionRanges.put(CodeName,FR);
154                        final String SimpleUniqueCodeName = F.name().toUpperCase()+"."+FR.name().toUpperCase();
155                        if(!hashedFactionRanges.containsKey(SimpleUniqueCodeName))
156                                hashedFactionRanges.put(SimpleUniqueCodeName,FR);
157                        final String UniqueCodeName = SimpleUniqueCodeName.replace(' ','_');
158                        if(!hashedFactionRanges.containsKey(UniqueCodeName))
159                                hashedFactionRanges.put(UniqueCodeName,FR);
160                        final String SimpleUniqueIDName = F.factionID().toUpperCase()+"."+FR.name().toUpperCase();
161                        if(!hashedFactionRanges.containsKey(SimpleUniqueIDName))
162                                hashedFactionRanges.put(SimpleUniqueIDName,FR);
163                }
164                return addFaction(F) ? F : null;
165        }

202        @Override
203        public Faction getFaction(String factionID)
204        {
205                if(factionID==null)
206                        return null;
207                Faction F=factionSet.get(factionID.toUpperCase());
208                if((F==null)&&(!factionID.toLowerCase().endsWith(".ini")))
209                {
210                        F=getFaction(factionID+".ini");
211                }
212                if(F!=null)
213                {
214                        if(F.isDisabled())
215                                return null;
216                        if(!F.amDestroyed())
217                                return F;
218                        factionSet.remove(F.factionID().toUpperCase());
219                        Resources.removeResource(F.factionID());
220                        return null;
221                }
222                final CMFile f=new CMFile(Resources.makeFileResourceName(makeFactionFilename(factionID)),null,CMFile.FLAG_LOGERRORS);
223                if(!f.exists())
224                        return null;
225                final StringBuffer buf=f.text();
226                if((buf!=null)&&(buf.length()>0))
227                {
228                        return buildFactionFromXML(buf, factionID);
229                }
230                return null;
231        }

493        protected Faction makeReactionFaction(String prefix, String classID, String Name, String code, String baseTemplateFilename)
494        {
495                final String codedName=Name.toUpperCase().trim().replace(' ','_');
496                final String factionID=prefix+codedName;
497                Faction templateF=getFaction(this.makeFactionFilename(codedName.toLowerCase()));
498                if(templateF==null)
499                        templateF=getFaction(baseTemplateFilename);
500                if(templateF==null)
501                {
502                        Log.errOut("Factions","Could not find base template '"+baseTemplateFilename+"'");
503                        return null;
504                }
505                StringBuffer buf = rebuildFactionProperties(templateF);
506                factionSet.remove(templateF.factionID().toUpperCase().trim());
507                String bufStr = buf.toString();
508                bufStr = CMStrings.replaceAll(bufStr,"<CODE>",code);
509                bufStr = CMStrings.replaceAll(bufStr,"<NAME>",Name);
510                bufStr = CMStrings.replaceAll(bufStr,"<FACTIONID>",factionID);
511                bufStr = CMStrings.replaceAll(bufStr,"<CLASSID>",classID);
512                buf=new StringBuffer(bufStr);
513                final Faction F=buildFactionFromXML(buf, factionID);
514                F.setInternalFlags(F.getInternalFlags() | Faction.IFLAG_IGNOREAUTO);
515                F.setInternalFlags(F.getInternalFlags() | Faction.IFLAG_NEVERSAVE);
516                F.setInternalFlags(F.getInternalFlags() | Faction.IFLAG_CUSTOMTICK);
517                return addFaction(F) ? F : null;
518        }

520        public Faction[] getSpecialFactions(final MOB mob, final Room R)
521        {
522                if((mob==null)||(R==null))
523                        return null;
524                Faction F=null;
525                final String autoReactionTypeStr=CMProps.getVar(CMProps.Str.AUTOREACTION).toUpperCase().trim();
526                if((autoReactionTypeStr==null)||(autoReactionTypeStr.length()==0))
527                        return null;
528                if(autoReactionTypeStr.equals("AREA"))
529                {
530                        final Area A=R.getArea();
531                        if((A!=null)&&(!CMath.bset(A.flags(), Area.FLAG_INSTANCE_CHILD)))
532                        {
533                                final String areaCode = A.Name().toUpperCase().trim().replace(' ','_');
534                                F=getFaction("AREA_"+areaCode);
535                                if(F==null)
536                                        F=makeReactionFaction("AREA_",A.ID(),A.Name(),areaCode,"examples/areareaction.ini");
537                                if(F==null)
538                                        return null;
539                                return new Faction[]{F};
540                        }
541                }
542                else
543                if(autoReactionTypeStr.equals("NAME"))
544                {
545                        final Vector<Faction> Fs=new Vector<Faction>();
546                        for(int i=0;i<R.numInhabitants();i++)
547                        {
548                                final MOB M=R.fetchInhabitant(i);
549                                if((M!=null)&&(M!=mob)&&(M.isMonster()))
550                                {
551                                        final String nameCode = M.Name().toUpperCase().trim().replace(' ','_');
552                                        F=getFaction("NAME_"+nameCode);
553                                        if(F==null)
554                                                F=makeReactionFaction("NAME_",M.ID(),M.Name(),nameCode,"examples/namereaction.ini");
555                                        if(F!=null)
556                                                Fs.add(F);
557                                }
558                        }
559                        if(Fs.size()==0)
560                                return null;
561                        return Fs.toArray(new Faction[0]);
562                }
563                else
564                if(autoReactionTypeStr.equals("RACE"))
565                {
566                        final Vector<Faction> Fs=new Vector<Faction>();
567                        final HashSet<Race> done=new HashSet<Race>(2);
568                        for(int i=0;i<R.numInhabitants();i++)
569                        {
570                                final MOB M=R.fetchInhabitant(i);
571                                if((M!=null)&&(M!=mob)&&(M.isMonster())&&(!done.contains(M.charStats().getMyRace())))
572                                {
573                                        final Race rR=M.charStats().getMyRace();
574                                        done.add(rR);
575                                        final String nameCode = rR.name().toUpperCase().trim().replace(' ','_');
576                                        F=getFaction("RACE_"+nameCode);
577                                        if(F==null)
578                                                F=makeReactionFaction("RACE_",rR.ID(),rR.name(),nameCode,"examples/racereaction.ini");
579                                        if(F!=null)
580                                                Fs.add(F);
581                                }
582                        }
583                        if(Fs.size()==0)
584                                return null;
585                        return Fs.toArray(new Faction[0]);
586                }
587                return null;
588        }

591        public void updatePlayerFactions(MOB mob, Room R, boolean forceAutoCheck)
592        {
593                if((mob==null)||(R==null))
594                        return;
595                else
596                {
597                        Faction F=null;
598                        for(final Enumeration<Faction> e=factions();e.hasMoreElements();)
599                        {
600                                F=e.nextElement();
601                                if(F!=null)
602                                {
603                                        if(((forceAutoCheck || (F.getInternalFlags()&Faction.IFLAG_IGNOREAUTO)==0))
604                                        &&(!F.hasFaction(mob))
605                                        &&(F.findAutoDefault(mob)!=Integer.MAX_VALUE))
606                                                mob.addFaction(F.factionID(),F.findAutoDefault(mob));
607                                }
608                        }
609                }
610                final Faction[] Fs=getSpecialFactions(mob,R);
611                if(Fs!=null)
612                {
613                        for(final Faction F : Fs)
614                        {
615                                if((F!=null)&&(!F.hasFaction(mob))&&(F.findAutoDefault(mob)!=Integer.MAX_VALUE))
616                                        mob.addFaction(F.factionID(),F.findAutoDefault(mob));
617                        }
618                }
619        }

621        protected void addOutsidersAndTimers(Faction F, Vector<Faction.FactionChangeEvent> outSiders, Vector<Faction.FactionChangeEvent> timers)
622        {
623                Faction.FactionChangeEvent[] CEs=null;
624                CEs=F.getChangeEvents("ADDOUTSIDER");
625                if(CEs!=null)
626                {
627                        for (final FactionChangeEvent ce : CEs)
628                                outSiders.addElement(ce);
629                }
630                CEs=F.getChangeEvents("TIME");
631                if(CEs!=null)
632                {
633                        for (final FactionChangeEvent ce : CEs)
634                        {
635                                if(ce.triggerParameters().length()==0)
636                                        timers.addElement(ce);
637                                else
638                                {
639                                        int[] ctr=(int[])ce.stateVariable(0);
640                                        if(ctr==null)
641                                        {
642                                                ctr=new int[]{CMath.s_int(ce.getTriggerParm("ROUNDS"))};
643                                                ce.setStateVariable(0,ctr);
644                                        }
645                                        if((--ctr[0])<=0)
646                                        {
647                                                ctr[0]=CMath.s_int(ce.getTriggerParm("ROUNDS"));
648                                                timers.addElement(ce);
649                                        }
650                                }
651                        }
652                }
653        }

660                try
