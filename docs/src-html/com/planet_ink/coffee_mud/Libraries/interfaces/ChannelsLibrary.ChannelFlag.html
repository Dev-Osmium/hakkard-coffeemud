034/**
035 * The main chat channels management library.  This is one of the
036 * older libraries in the system, and as such, still reflects the
037 * use of indexed vectors that was common.  It still makes sense
038 * given the way those numbers are used in CMMsg message codes to
039 * identify the channels.
040 *
041 * Over the years, the ability to filter the channels by user
042 * was added, along with flags to automatically insert information
043 * or enable other parts of the system to generate channel
044 * messages.
045 *
046 * @author Bo Zimmerman
047 */
048public interface ChannelsLibrary extends CMLibrary
049{
050        /**
051         * Returns the number of registered channels
052         * @see ChannelsLibrary#getChannelIndex(String)
053         * @return the number of registered channels
054         */
055        public int getNumChannels();
056
057        /**
058         * Returns the CMChannel object for the given
059         * registered channel from 0 - getNumChannels();
060         * Basically this is almost all you need, but
061         * there's more....
062         * @see ChannelsLibrary#getNumChannels()
063         * @param channelNumber the index of the channel
064         * @return the CMChannel object
065         */
066        public CMChannel getChannel(int channelNumber);
067
068        /**
069         * Generates a list of previous channel messages, in ChannelMsg
070         * object format.  This may potentially hit the database.
071         * @see ChannelsLibrary#getChannelIndex(String)
072         * @see ChannelsLibrary.ChannelMsg
073         * @param channelNumber the channel id number/index
074         * @param numNewToSkip starting message number (0 based)
075         * @param numToReturn total number of messages to return
076         * @return the list of messages
077         */
078        public List<ChannelMsg> getChannelQue(int channelNumber, int numNewToSkip, int numToReturn);
079
080        /**
081         * Returns whether the given Mob can read a channel message from the given sender, on
082         * a particular channel.
083         * @see ChannelsLibrary#getNumChannels()
084         * @see ChannelsLibrary#mayReadThisChannel(MOB, int, boolean)
085         * @see ChannelsLibrary#mayReadThisChannel(MOB, boolean, Session, int)
086         * @see ChannelsLibrary#mayReadThisChannel(MOB, boolean, MOB, int, boolean)
087         * @param sender the sender of the channel message
088         * @param areaReq true if the message can only be read by someone in the senders Area
089         * @param M the potential receiver of the message to confirm
090         * @param channelNumber the channel index number
091         * @return true if the mob can read the senders message, false otherwise
092         */
093        public boolean mayReadThisChannel(MOB sender, boolean areaReq, MOB M, int channelNumber);
094
095        /**
096         * Returns whether the given Mob can read a channel message from the given sender, on
097         * a particular channel.
098         * @see ChannelsLibrary#getChannelIndex(String)
099         * @see ChannelsLibrary#mayReadThisChannel(MOB, int, boolean)
100         * @see ChannelsLibrary#mayReadThisChannel(MOB, boolean, Session, int)
101         * @see ChannelsLibrary#mayReadThisChannel(MOB, boolean, MOB, int)
102         * @param sender the sender of the channel message
103         * @param areaReq true if the message can only be read by someone in the senders Area
104         * @param M the potential receiver of the message to confirm
105         * @param channelNumber the channel index number
106         * @param offlineOK true if the channel reader can be read, or offline, false if they must be online
107         * @return true if the mob can read the senders message, false otherwise
108         */
109        public boolean mayReadThisChannel(MOB sender, boolean areaReq, MOB M, int channelNumber, boolean offlineOK);
110
111        /**
112         * Returns whether the given Mob can read a channel message from the given sender, on
113         * a particular channel.
114         * @see ChannelsLibrary#getNumChannels()
115         * @see ChannelsLibrary#mayReadThisChannel(MOB, int, boolean)
116         * @see ChannelsLibrary#mayReadThisChannel(MOB, boolean, MOB, int)
117         * @see ChannelsLibrary#mayReadThisChannel(MOB, boolean, MOB, int, boolean)
118         * @param sender the sender of the channel message
119         * @param areaReq true if the message can only be read by someone in the senders Area
120         * @param ses the potential receiver session of the message to confirm
121         * @param channelNumber the channel index number
122         * @return true if the mob can read the senders message, false otherwise
123         */
124        public boolean mayReadThisChannel(MOB sender, boolean areaReq, Session ses, int channelNumber);
125
126        /**
127         * Returns whether the given Mob can read a channel message from the given sender, on
128         * a particular channel.
129         * @see ChannelsLibrary#getChannelIndex(String)
130         * @see ChannelsLibrary#mayReadThisChannel(MOB, boolean, MOB, int)
131         * @see ChannelsLibrary#mayReadThisChannel(MOB, boolean, Session, int)
132         * @see ChannelsLibrary#mayReadThisChannel(MOB, boolean, MOB, int, boolean)
133         * @param M the potential receiver of the message to confirm
134         * @param channelNumber the channel index number
135         * @param zapCheckOnly the channel index number
136         * @return true if the mob can read the senders message, false otherwise
137         */
138        public boolean mayReadThisChannel(MOB M, int channelNumber, boolean zapCheckOnly);
139
140        /**
141         * Adds the given channel message CMMsg event message to the que for the given
142         * channel number.  Will cause a database write.
143         * @see ChannelsLibrary#getNumChannels()
144         * @see ChannelsLibrary#getChannelIndex(String)
145         * @param channelNumber the channel index number
146         * @param msg the channel message msg that was sent around
147         */
148        public void channelQueUp(int channelNumber, CMMsg msg);
149
150        /**
151         * Returns the official index number of the channel with the
152         * given name or which starts with the given name.  It is
153         * case insensitive search.
154         * @see ChannelsLibrary#getNumChannels()
155         * @param channelName the channel string to search for
156         * @return the official index number of the channel
157         */
158        public int getChannelIndex(String channelName);
159
160        /**
161         * Returns the bitmask appropriate to the channel of the given
162         * name, or which starts with the given string.  The bitmask
163         * is used for the individual player toggle switches.
164         * @param channelName the name of the channel
165         * @return the bitmask appropriate to the channel
166         */
167        public int getChannelCodeNumber(String channelName);
168
169        /**
170         * Returns the list of channel names that have the given
171         * ChannelFlag set.
172         * @see ChannelsLibrary.ChannelFlag
173         * @param flag the flag to find channels for
174         * @return the list of channel names with the flag set.
175         */
176        public List<String> getFlaggedChannelNames(ChannelFlag flag);
177
178        /**
179         * Returns the friendly readable description of the channel
180         * with the given name, or which starts with the given string.
181         * It's mostly requirements to read or write to the channel.
182         * @param channelName the name of the channel
183         * @return the friendly readable description of the channel
184         */
185        public String getExtraChannelDesc(String channelName);
186
187        /**
188         * Returns all the CMChannel objects for any channels flagged
189         * as being mapped to IMC2.
190         * @return all the CMChannel objects for IMC2
191         */
192        public List<CMChannel> getIMC2ChannelsList();
193
194        /**
195         * Returns all the CMChannel objects for any channels flagged
196         * as being mapped to I3.
197         * @return all the CMChannel objects for I3
198         */
199        public List<CMChannel> getI3ChannelsList();
200
201        /**
202         * Returns an array of all the names of all the channels.
203         * @see ChannelsLibrary#getNumChannels()
204         * @return an array of all the names of all the channels.
205         */
206        public String[] getChannelNames();
207
208        /**
209         * Returns the official channel name of the channel with the
210         * given name or which starts with the given name.  It is
211         * case insensitive search.
212         * @see ChannelsLibrary#getNumChannels()
213         * @param channelName the channel string to search for
214         * @return the official channel name of the channel
215         */
216        public String findChannelName(String channelName);
217
218        /**
219         * Searches all users online for any sessions that are snooping on the
220         * given session, and also are not permitted to read the given channel
221         * number.  It then returns all the offending sessions, after forcing
222         * them to stop snooping.
223         * @see ChannelsLibrary#getNumChannels()
224         * @see ChannelsLibrary#restoreInvalidSnoopers(Session, List)
225         * @param mySession the session to search for snoopers of
226         * @param channelNumber the channel number to cross reference snoopers by
227         * @return all the offending sessions
228         */
229        public List<Session> clearInvalidSnoopers(Session mySession, int channelNumber);
230
231        /**
232         * Iterates through the given list of sessions and forces them all to snoop on
233         * the given session.  This makes sense if you check the clearInvalidSnoopers
234         * method.
235         * @see ChannelsLibrary#clearInvalidSnoopers(Session, int)
236         * @param mySession the session to search restoring snooping on
237         * @param invalid the list of sessions to force resnooping
238         */
239        public void restoreInvalidSnoopers(Session mySession, List<Session> invalid);
240
241        /**
242         * Clears the channels list and then reloads it from the three given comma-delimited
243         * list of coded strings.
244         * @param list the main game channel list
245         * @param ilist the list of i3 channels
246         * @param imc2list the list of imc2 channels
247         * @return the total number of channels loaded
248         */
249        public int loadChannels(String list, String ilist, String imc2list);
250
251        /**
252         * Sends the given channel message from the given sender to the given session on the
253         * given channelNumbered channel.
254         * @see ChannelsLibrary#getNumChannels()
255         * @see ChannelsLibrary#createAndSendChannelMessage(MOB, String, String, boolean)
256         * @param ses the session to send the channel message to
257         * @param areareq true if the sender and session must be in the same area, false otherwise
258         * @param channelNumber the channel index number of the message
259         * @param msg the constructed CMMsg channel message
260         * @param sender the sender of the channel message
261         * @return true if the message was sent, and false otherwise
262         */
263        public boolean sendChannelCMMsgTo(Session ses, boolean areareq, int channelNumber, CMMsg msg, MOB sender);
264
265        /**
266         * Creates and sends the given string message on the channel with the given name to all sessions who are
267         * allowed to receive the message from the given sender.
268         * @see ChannelsLibrary#sendChannelCMMsgTo(Session, boolean, int, CMMsg, MOB)
269         * @param mob the sender of the message
270         * @param channelName the name of the channel to send the message on
271         * @param message the string message to send on the channel
272         * @param systemMsg true to format as a system message, false for a normal chat message
273         */
274        public void createAndSendChannelMessage(MOB mob, String channelName, String message, boolean systemMsg);
275
276        /**
277         * Creates a new channel object.
278         * @see ChannelsLibrary.CMChannel
279         * @see ChannelsLibrary.ChannelFlag
280         * @param name the channel name
281         * @param i3Name empty string, or the mapped name of the i3 channel
282         * @param imc2Name empty string, or the mapped name of the imc2 channel
283         * @param mask the zapper mask for who may read the channel
284         * @param flags the channel flags to set for the given channel
285         * @param colorOverrideANSI empty string for default, or the color code for this channel
286         * @param colorOverrideWords empty string for default, or the color code for this channel
287         * @return the newly created channel object
288         */
289        public CMChannel createNewChannel(final String name, final String i3Name, final String imc2Name,
290                                                                          final String mask, final Set<ChannelFlag> flags,
291                                                                          final String colorOverrideANSI, final String colorOverrideWords);

293        /**
294         * Basic Channel definition
295         * @author Bo Zimmerman
296         */
297        public static interface CMChannel
298        {
299                /**
300                 * The name of the channel, always uppercased
301                 * @return name of the channel
302                 */
303                public String name();
304
305                /**
306                 * An empty string, or the name of the I3 channel that
307                 * this channel is mapped to.
308                 * @return the name of the I3 channel or ""
309                 */
310                public String i3name();
311
312                /**
313                 * An empty string, or the name of the IMC2 channel that
314                 * this channel is mapped to.
315                 * @return the name of the IMC2 channel or ""
316                 */
317                public String imc2Name();
318
319                /**
320                 * The zapper mask to filter in those who may read this
321                 * channel.
322                 * @see MaskingLibrary
323                 * @return the zapper mask for who may read
324                 */
325                public String mask();
326
327                /**
328                 * Empty string, or a color ansi codes to override the
329                 * default channel color
330                 * @see ColorLibrary
331                 * @return Empty string, or a color ansi codes to override
332                 */
333                public String colorOverrideANSICodes();
334
335                /**
336                 * Empty string, or a color code words to override the
337                 * default channel color
338                 * @see ColorLibrary
339                 * @return Empty string, or a color code words to override
340                 */
341                public String colorOverrideWords();
342
343                /**
344                 * The channel flags for this channel.
345                 * @see ChannelsLibrary.ChannelFlag
346                 * @return channel flags for this channel.
347                 */
348                public Set<ChannelFlag> flags();
349
350                /**
351                 * The internal cached queue of previous
352                 * channel messages.
353                 * @see ChannelsLibrary.ChannelMsg
354                 * Always trimmed when it gets too large/
355                 * @return internal cached queue of messages
356                 */
357                public SLinkedList<ChannelMsg> queue();
358        }
359
360        /**
361         * The entry for the channel que, containing the CMMsg
362         * object that was sent as the original message as well
363         * as the timestamp when it was sent.
364         * @author Bo Zimmerman
365         *
366         */
367        public interface ChannelMsg
368        {
369                /**
370                 * The CMMsg object that was sent as the original message
371                 * @return CMMsg object that was sent as the original message
372                 */
373                public CMMsg msg();
374
375                /**
376                 * The timestamp of when the message was sent
377                 * @return timestamp of when the message was sent
378                 */
379                public long sentTimeMillis();
380        }
