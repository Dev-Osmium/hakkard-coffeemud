094        /**
095         * Interface allowing java or javascript plugins to the MCP protocol
096         * @author Bo Zimmerman
097         *
098         */
099        public interface MCPPackage
100        {
101                /**
102                 * Return the name of the overall package
103                 * @return the name of the overall package
104                 */
105                public String packageName();
106
107                /**
108                 * Returns the minimum version supported here.
109                 * @return the minimum version supported here.
110                 */
111                public float minVersion();
112
113                /**
114                 * Returns the maximum version supported here.
115                 * @return the maximum version supported here.
116                 */
117                public float maxVersion();
118
119                /**
120                 * Execute the package
121                 * @param session the session for which the package is being executed
122                 * @param command the actual command being executed by this package
123                 * @param clientSupported map of client packages supported followed by min-maxes
124                 * @param variables key/value pairs for this package
125                 */
126                public void executePackage(Session session, String command, Map<String,float[]> clientSupported, Map<String,String> variables);
127        }
