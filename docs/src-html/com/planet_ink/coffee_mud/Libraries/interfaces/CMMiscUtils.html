037/**
038 * The MiscUtils (Miscellaneous Utilities) library should not exist.
039 * Its purpose is to expose universal functionality that doesn't
040 * seem to fit anywhere else.  Usually, however, it does fit
041 * elsewhere just fine, but the elsewhere felt too crowded already
042 * to receive it.  I'm sure this library will always exist, but the
043 * interface is definitely subject to change.
044 * 
045 * @author Bo Zimmerman
046 */
047public interface CMMiscUtils extends CMLibrary
048{
049        /**
050         * Checks the given player mob for the format of their prompt in
051         * their playerstats and generates a fully formed prompt, complete
052         * with all variables filled in.
053         * @param mob the mob to build a prompt for
054         * @param prompt TODO
055         * @return the fully filled in customized prompt string
056         */
057        public String builtPrompt(MOB mob, String prompt);
058
059        /**
060         * Returns the current mud-month-day-year for the object, depending
061         * on the timeclock where the environmental is located.
062         * @param E the object curious about the date
063         * @return the date string
064         */
065        public String getFormattedDate(Environmental E);
066
067        /**
068         * Returns a rediculous best guess on the amount of memory used
069         * by the given environmental.  
070         * @param E the object to check for a footprint of
071         * @param number the accuracy -- higher is better
072         * @return the amount of memory used, very approximately
073         */
074        public double memoryUse ( Environmental E, int number );
075
076        /**
077         * Nice english comma-delimited list, with oxford commas
078         * and trailing "and" or "or" at the end.  If the list is
079         * of environmental, it will use the name, otherwise, it 
080         * casts everything as a string and lists it.
081         * @param V the objects to list
082         * @param andTOrF true for trailing and, false for trailing or
083         * @return the readable comma list.
084         */
085        public String niceCommaList(List<?> V, boolean andTOrF);
086
087        /**
088         * This strange method takes a list of space-delimited expressions of the
089         * form [CONDITION]number number number number, etc. E.G.: &gt;1 3 2 5 3 2.
090         * Each list must contain the given num of digits. If the condition falls
091         * within the given start of range and end of range, then the condition
092         * range of entries in the returned array is populated with the values on
093         * that row.
094         * @param condV the list of fully expressions
095         * @param numDigits the min number of digits in each expression
096         * @param startOfRange the starting range to return
097         * @param endOfRange the ending range to return &gt; startOfRange
098         * @return the list of number lists
099         */
100        public long[][] compileConditionalRange(List<String> condV, int numDigits, final int startOfRange, final int endOfRange);
101
102        /**
103         * Outfits the given mob with the list of given items.  If the mob
104         * does not have an item with the same name as one on the list, it 
105         * is given to them magically.  If they can, they will then wear it.
106         * @param mob the mob to outfit
107         * @param items the items to outfit the mob with
108         */
109        public void outfit(MOB mob, List<Item> items);
110
111        /**
112         * Returns the language being spoke by the given object (mob, usually).
113         * A null return usually means Common. 
114         * @param P the mob to check
115         * @return the language being spoken.
116         */
117        public Language getLanguageSpoken(Physical P);
118
119        /**
120         * Returns whether the given Item is reachable by the given mob.
121         * It may not be reachable if the mob is riding something, and the
122         * item is on the ground, etc.  If the object given is not an
123         * item, or null, true is always returned.
124         * @param mob the mob who wants to reach
125         * @param E the Item to reach for
126         * @return true, or false if it's unreachable
127         */
128        public boolean reachableItem(MOB mob, Environmental E);
129
130        /**
131         * Recursively extinguishes everything from the given target on down.
132         * If a room is given, everything in the room, including the room,
133         * is extinguished.   Ordinary fires and torches also go out.  
134         * The mundane flag is to prevent extinguishing elementals and magic
135         * flame spells.
136         * @param source the mob doing the extinguishing
137         * @param target the thing to extinguish
138         * @param mundane true to skip magic and elemental targets
139         */
140        public void extinguish(MOB source, Physical target, boolean mundane);
141
142        /**
143         * Given the allowedArmorLevel code and the mob, this method returns
144         * whether the given mob is only wearing permitted items on the
145         * applicable armor slots.  
146         * @see CharClass#ARMOR_DESCS
147         * @param mob the mob to check
148         * @param allowedArmorLevel the allowed armor level
149         * @return true if the mob is good to go, false otherwise
150         */
151        public boolean armorCheck(MOB mob, int allowedArmorLevel);
152
153        /**
154         * Given the allowedArmorLevel code and the mob, this method returns
155         * whether the given mob is permitted to wear the given item is
156         * applicable armor slots given the armor level code.  
157         * @see CharClass#ARMOR_DESCS
158         * @param mob the mob to check
159         * @param I the item to check
160         * @param allowedArmorLevel the allowed armor level
161         * @return true if the mob is good to go for that item, false otherwise
162         */
163        public boolean armorCheck(MOB mob, Item I, int allowedArmorLevel);
164
165        /**
166         * Drops all items from the given mob into the given room which are inside the given
167         * container (or are the given item).  The bodyFlag ensures that the contents are
168         * not marked for cleanup.  This method does the deed, but does not generate any
169         * new messages.  It also does not recover the state of the mob.
170         * @param mob the mob who is dropping
171         * @param room the room where it's being dropped
172         * @param thisContainer the item or container where the items must be
173         * @param bodyFlag true if the container is a body, false otherwise
174         */
175        public void recursiveDropMOB(MOB mob, Room room, Item thisContainer, boolean bodyFlag);
176
177        /**
178         * Returns a copy of the given item/container and a copy of every item in that
179         * container, recursively.
180         * @param theContainer the container or item to copy
181         * @return a list of copies of all the contents
182         */
183        public List<Item> deepCopyOf(Item theContainer);
184
185        /**
186         * This method removes all equipment from the mob and quickly attempt to re-wear/hold/wield
187         * it all where it originally was by issueing wear messages which are previewed and executed.
188         * In the end, the mob will be wearing everything that the system will let them wear.
189         * Moreover, this is all done silently.
190         * @param mob the mob to confirm the equipment of
191         */
192        public void confirmWearability(MOB mob);
193
194        /**
195         * Assumes that every inventory item, equipped item, and store inventory
196         * are for an NPC mob, where "rejuv" doesn't really matter.  In those cases,
197         * the "rejuv" stat is, for non-rivalrous items, a pct chance of it being
198         * retained on this mob.  For rivalrous items (two wielded swords, for example),
199         * it is a weighted chance of being selected.  Electronics items also have their
200         * random stats determined at this time. 
201         * @param mob the npc mob to process variable equipment on
202         * @return 0 for success, -1 if an admin is in the room, so nothing could be done.
203         */
204        public int processVariableEquipment(MOB mob);
205
206        /**
207         * Creates one of the deprecated traps depending on what sort of object is passed
208         * in, whether it has a lid or a lock, etc.
209         * @param unlockThis the exit, container, room, whatever
210         * @return the trap to add to the physical thing, or null
211         */
212        public Trap makeADeprecatedTrap(Physical unlockThis);
213        
214        /**
215         * Creates and sets a deprecated trap on the given exit, room,
216         * container, or whatever.
217         * @see CMMiscUtils#makeADeprecatedTrap(Physical)
218         * @see CMMiscUtils#setTrapped(Physical, Trap)
219         * @see CMMiscUtils#fetchMyTrap(Physical)
220         * @param myThang the thing to set the trap on.
221         */
222        public void setTrapped(Physical myThang);
223        
224        /**
225         * Sets the given deprecated trap on the given exit, room,
226         * container, or whatever.
227         * @see CMMiscUtils#makeADeprecatedTrap(Physical)
228         * @see CMMiscUtils#setTrapped(Physical)
229         * @param myThang the thing to set the trap on.
230         * @param theTrap the deprecated trap to set on it
231         */
232        public void setTrapped(Physical myThang, Trap theTrap);
233        
234        /**
235         * Returns any trap found on the given thing, or null.
236         * @see CMMiscUtils#makeADeprecatedTrap(Physical)
237         * @param myThang the thing to check for a trap
238         * @return the trap found, or null
239         */
240        public Trap fetchMyTrap(Physical myThang);
241
242        /**
243         * If any mob (probably a player) is possessing the
244         * given mob, this will return that mob, or null
245         * @param mob the mob to check for possession
246         * @return the mob possessing the given mob
247         */
248        public MOB getMobPossessingAnother(MOB mob);
249
250        /**
251         * Normally just sends the message to the room by calling
252         * Room.send.  However, if the target of the message is 
253         * an exit, then the several exits involved would also
254         * informed by having executeMsg called on them.
255         * @param msg the message to send
256         * @param room the room to send the message to
257         * @param dirCode if known, the direction of the target exit
258         */
259        public void roomAffectFully(CMMsg msg, Room room, int dirCode);
260
261        /**
262         * Returns any corpses found in the given container, recursively.
263         * @param container the container that possibly has corpses
264         * @return a list of any corpses found, or an empty list
265         */
266        public List<DeadBody> getDeadBodies(Environmental container);
267
268        /**
269         * Resurrects the given body according to all system rules.
270         * 
271         * @param tellMob if the corpse could not be resurrected, tell this mob.
272         * @param corpseRoom room to bring the mob to after resurrection, probably same as body's location
273         * @param body the corpse to resurrect
274         * @param XPLevel if &gt; 0, and rules allow, bonus xp restored
275         * @return true if the resurrection happened, false otherwise
276         */
277        public boolean resurrect(MOB tellMob, Room corpseRoom, DeadBody body, int XPLevel);
278
279        /**
280         * This method parses the item ruinning rules and possibly ruins the given item
281         * by returning the ruined version.  Or it might do nothing
282         * and just return the item.
283         * @param mob the mob to get ruin policies for
284         * @param I the item to potentially ruin
285         * @return the ruined item, or the original item, depending
286         */
287        public Item isRuinedLoot(MOB mob, Item I);
288
289        /**
290         * Always converts the given item into the Ruined version
291         * @see CMMiscUtils#isRuinedLoot(MOB, Item)
292         * @param I the item to ruin
293         * @return the new, ruined version
294         */
295        public Item ruinItem(Item I);
296
297        /**
298         * Iterates through every mob and player in the game, replacing the old race
299         * object with the new one.
300         * @see CMMiscUtils#reloadCharClasses(CharClass)
301         * @param newR the new race object
302         * @param oldR the old race object
303         */
304        public void swapRaces(Race newR, Race oldR);
305
306        /**
307         * Iterates through every mob and player in the game, replacing the old char class
308         * object given with the new one of the same ID from CMClass.
309         * @see CMMiscUtils#swapRaces(Race, Race)
310         * @param oldC the old charclass object
311         */
312        public void reloadCharClasses(CharClass oldC);
313
314        /**
315         * Returns whether the given item can be destroyed by the given mob, probably magically.
316         * This is a recursive check if the item is a container.
317         * @param mob the mob who wants to destroy the item
318         * @param I the item to destroy
319         * @param ignoreBodies true to ignore corpse checks, false otherwise
320         * @param ignoreWeight TODO
321         * @return true if the item can be destroyed, false otherwise
322         */
323        public boolean canBePlayerDestroyed(final MOB mob, final Item I, final boolean ignoreBodies, boolean ignoreWeight);
324
325        /**
326         * Calls unInvoke on all effects on the given environmental.  
327         * This may not cause the effects to disappear, depending on the
328         * behavior of each effect.
329         * @param E the object to diseffect
330         * @return true if the item still exists, false if it was destroyed
331         */
332        public boolean disInvokeEffects(Environmental E);
333
334        /**
335         * Removes magical effects from wands, and other spell holders,
336         * deletes any effects after attempting to uninvoke.  This method
337         * probably needs more thought, since not all effects are
338         * magic (though they usually are)
339         * 
340         * The return value is a very strange number: 0 if the item is 
341         * destroyed, -999 if nothing done, or the item level minus the 
342         * magic value of what was done to it.
343         * @param target the object to disenchant.
344         * @return a bizarre number 
345         */
346        public int disenchantItem(Item target);
347        
348        /**
349         * Absolutely returns the correct race when mixing races of the
350         * two given IDs.  Applies system rules to the generation.
351         * 
352         * @param race1 the mother race
353         * @param race2 the father race
354         * @param ignoreRules TODO
355         * @return the mixed race
356         */
357        public Race getMixedRace(String race1, String race2, boolean ignoreRules);
358        
359        /**
360         * Breaks apart a given generic mixed race ID to figure
361         * out which races were combined to make it up.
362         * @param raceID the raceID to break apart
363         * @return a list of constituant races, or empty.
364         */
365        public List<Race> getConstituantRaces(final String raceID);
366        
367        /**
368         * Returns a list of worn equipment at the top layer, where it can be
369         * seen.  An optional wornMask can narrow the locations allowed.
370         * The require list is the item, plus the worn location.
371         * 
372         * @param mob the mob to get equipment from
373         * @param wornMask 0, or a bitmap of allowed wear locations
374         * @return the list of items and wear locations
375         */
376        public PairList<Item, Long> getSeenEquipment(final MOB mob, final long wornMask);
377        
378        /**
379         * Does its best to come up with a specially encoded unsubscribe url for 
380         * a particular user.  The url will be unique to the user, but is man-in-the-
381         * middle attackable, as it is a simple hash.
382         * @param name the account or player name
383         * @return the full web url with http and everything
384         */
385        public String getUnsubscribeURL(final String name);
386}
