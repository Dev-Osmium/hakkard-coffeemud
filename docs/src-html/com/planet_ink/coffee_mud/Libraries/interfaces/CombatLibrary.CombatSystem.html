034/**
035 * Fighting, healing, determining combatants, death, attacking,

044        /**
045         * An enumeration of the several combat systems.
046         * These are specified in the coffeemud.ini file.
047         *
048         * @author Bo Zimmerman
049         *
050         */
051        public enum CombatSystem
052        {
053                /**
054                 * The default system allows one action point to be used
055                 * by each combatant on a skill, per round.  All other action
056                 * points are automatically used on basic attacks, every round.
057                 */
058                DEFAULT,
059
060                /**
061                 * The queue system allows all action points to be used
062                 * by each combatant on skills.  All unused action points are
063                 * spent on basic attacks every round.
064                 */
065                QUEUE,
066
067                /**
068                 * The manual system allows all action points to be used
069                 * by each combatant on skills or on basic attacks, but
070                 * all attack must be manually entered.  Any unused points
071                 * are lost every round.
072                 */
073                MANUAL,
074
075                /**
076                 * The turn based system gives each combatant an amount of time
077                 * to use all their action points on skills or basic attacks,
078                 * not permitting any other combatant to act until the previous
079                 * one completes.
080                 */
081                TURNBASED
082        }
