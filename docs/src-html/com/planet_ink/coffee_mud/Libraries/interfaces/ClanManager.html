035/**
036 * This library manages the list of Clans in the game, and helps
037 * administer their most basic functions.
038 * @see com.planet_ink.coffee_mud.Common.interfaces.Clan
039 */
040public interface ClanManager extends CMLibrary
041{
042        /**
043         * Returns a list of all available clans names in the game,
044         * as Clan objects.
045         * @see com.planet_ink.coffee_mud.Common.interfaces.Clan
046         * @return a list of all available clans names in the game,
047         */
048        public Enumeration<String> clansNames();
049
050        /**
051         * Returns a list of all available clan names for all clan
052         * managers that share a single map.  This is for multi-host
053         * muds.
054         * @return all clan names across all shared map hosts.
055         */
056        public Enumeration<String> clansNamesAllHosts();
057
058        /**
059         * Returns the number of clans in the game.
060         * @return the number of clans in the game.
061         */
062        public int numClans();
063
064        /**
065         * This method is used to determine the basic relationship between two clans.  The
066         * two clans are evaluated, based on their declared relationship to each other, and
067         * the relations they inherit from allys.  It is then compared with the passed in
068         * relationship constant.  If they match, true is returned, and false otherwise.
069         * @see com.planet_ink.coffee_mud.Common.interfaces.Clan#REL_DESCS
070         * @param clanName1 the first clan to evaluate
071         * @param clanName2 the second clan to evaluate
072         * @param relation the clan relation to compare to
073         * @return true if the common relationship matches the given relation, and false otherwise
074         */
075        public boolean isCommonClanRelations(String clanName1, String clanName2, int relation);
076
077        /**
078         * This method is used to determine the basic relationship between two clan members.  The
079         * two mobs first rivalrous clans are evaluated, based on their declared relationship to
080         * each other, and the relations they inherit from allys.  It is then compared with the
081         * passed the War status.  If they match, true is returned, and false otherwise.
082         * @see com.planet_ink.coffee_mud.Common.interfaces.Clan#REL_DESCS
083         * @param M1 the first mobs first rivalrous clan to evaluate
084         * @param M2 the second mobs first rivalrous clan to evaluate
085         * @return true if the common relationship matches war, and false otherwise
086         */
087        public boolean isAtClanWar(MOB M1, MOB M2);
088
089        /**
090         * Returns whether the two mobs share ANY common clans, even
091         * non-rivalrous ones.
092         * @param M1 first mob
093         * @param M2 second mob
094         * @return true if they share a clan, false otherwise
095         */
096        public boolean isAnyCommonClan(MOB M1, MOB M2);
097
098        /**
099         * This method is used to determine the basic relationship between two mobs clans. The
100         * two sets of clans are evaluated, based on their declared relationship to each other, and
101         * the relations they inherit from allys.  This relationship is then checked against the given
102         * relation.  If any pairing matches the relation, true is returned.
103         * @see com.planet_ink.coffee_mud.Common.interfaces.Clan#REL_DESCS
104         * @param M1 the first mob whose clans to evaluate
105         * @param M2 the second mob whose clans to evaluate
106         * @param relation the relation to look for
107         * @return true if any clans relate in the given way, false otherwise
108         */
109        public boolean findAnyClanRelations(MOB M1, MOB M2, int relation);
110
111        /**
112         * This method is used to determine the basic relationship between two clans.  The
113         * two clans are evaluated, based on their declared relationship to each other, and
114         * the relations they inherit from allys.  This relationship is then returned as
115         * a relation constant number.
116         * @see com.planet_ink.coffee_mud.Common.interfaces.Clan#REL_DESCS
117         * @param clanID1 the first clan to evaluate
118         * @param clanID2 the second clan to evaluate
119         * @return the relation code integer
120         */
121        public int getClanRelations(String clanID1, String clanID2);
122
123        /**
124         * Get last time governments were loaded/updated
125         * @return time in ms
126         */
127        public long getLastGovernmentLoad();
128
129        /**
130         * Returns the clan which may or may not be assigned to the absolute file path specified.
131         * @param sitePath the absolute path of the web dir assigned to this clan
132         * @return a clan at that path, or NULL for none
133         */
134        public Clan getWebPathClan(String sitePath);
135
136        /**
137         * Returns whether the given clan set contains a clan that
138         * allows the given mob to do the given function in the clan.
139         * @param mob the mob to check the clans of
140         * @param func the function to check for
141         * @return true if allowed, false otherwise
142         */
143        public boolean checkClanPrivilege(MOB mob, Clan.Function func);
144
145        /**
146         * Returns whether the given mob belongs to the given clan, and if
147         * so, whether they can do the given function in the clan.
148         * @param mob the mob to look for privileges for
149         * @param clanID the clanID to check for to check for
150         * @param func the function to check for
151         * @return true if allowed, false otherwise
152         */
153        public boolean checkClanPrivilege(MOB mob, String clanID, Clan.Function func);
154
155        /**
156         * If the given mob belongs to a clan, and if they can do the
157         * given function in a clan, this will return that clan object
158         * and the integer.
159         * @param mob the mob to check the clans of
160         * @param func the function to check for
161         * @return the clan and role integer
162         */
163        public Pair<Clan,Integer> findPrivilegedClan(MOB mob, Clan.Function func);
164
165        /**
166         * If the given mob belongs to a clan, and if they can do the
167         * given function in a clan, this will return those clan objects
168         * and their role integer.
169         * @param mob the mob to check the clans of
170         * @param func the function to check for
171         * @return the clan and role integers
172         */
173        public List<Pair<Clan,Integer>> findPrivilegedClans(MOB mob, Clan.Function func);
174
175        /**
176         * Returns a list of clans that the source mob belongs to which the filter
177         * mob does NOT also belong.  The clans will be rivalrous only.
178         * @param clanSourceMob the mob to source the clan list from
179         * @param filterMob the mob to use to filter out source clans
180         * @return the clan and role integers
181         */
182        public List<Pair<Clan,Integer>> findRivalrousClans(MOB clanSourceMob, MOB filterMob);
183
184        /**
185         * Searches for all clans that can be rivalrous with other clans that are
186         * commonly shared between two mobs, along with their two roles
187         * @param mob1 the first mob to check the clans of
188         * @param mob2 the second mob to check the clans of
189         * @return potentially rivalrous clans and the mobs roles in them
190         */
191        public List<Triad<Clan,Integer,Integer>> findCommonRivalrousClans(MOB mob1, MOB mob2);
192
193        /**
194         * Searches for a clan in the list that can be rivalrous with other clans.
195         * First one found is returned.
196         * @param mob the mob to check the clans of
197         * @return First clan of the mobs that is rivalrous
198         */
199        public Clan findRivalrousClan(MOB mob);
200
201        /**
202         * Searches for a clan in the list that can be conquerable with other clans.
203         * First one found is returned.
204         * @param mob the mob to check the clans of
205         * @return First clan of the mobs that is conquerable of others
206         */
207        public Clan findConquerableClan(MOB mob);
208
209        /**
210         * Searches for all clans that can be rivalrous with other clans.
211         * @param mob the mob to check the clans of
212         * @return List of the mobs clans that are rivalrous
213         */
214        public List<Pair<Clan,Integer>> findRivalrousClans(MOB mob);
215
216        /**
217         * Returns the Clan object associated with the given clan name
218         * @see com.planet_ink.coffee_mud.Common.interfaces.Clan
219         * @param id the clan name
220         * @return the Clan object associated with the given clan name
221         */
222        public Clan getClan(String id);
223
224        /**
225         * Returns the Clan object associated with the given clan name
226         * from any host sharing the same map as the caller.
227         *
228         * @see com.planet_ink.coffee_mud.Common.interfaces.Clan
229         * @param id the clan name
230         * @return the Clan object associated with the given clan name
231         */
232        public Clan getClanAnyHost(String id);
233
234        /**
235         * Returns the Clan object associated with the given clan name, or
236         * if the name is not found, the name that most closely matches it.
237         * @see com.planet_ink.coffee_mud.Common.interfaces.Clan
238         * @param id the clan name
239         * @return the Clan object associated with the given clan name
240         */
241        public Clan findClan(String id);
242
243        /**
244         * Returns an enumeration of all the Clans in the game
245         * @return an enumeration of all the Clans in the game
246         */
247        public Enumeration<Clan> clans();
248
249        /**
250         * Returns all clans and their last-cached accept-positions
251         * The auto-positions may be inaccurate.
252         * @return all clans and their last-cached accept-positions
253         */
254        public Iterable<Pair<Clan,Integer>> clanRoles();
255
256        /**
257         * Adds the given clan to the games list
258         * @param C the clan to add
259         */
260        public void addClan(Clan C);
261
262        /**
263         * Removes the given clan from the games list
264         * @param C the clan to remove
265         */
266        public void removeClan(Clan C);
267
268        /**
269         * Returns the clan associated with a given specific web path.
270         * It is derived from the clanwebsites entry in coffeemud.ini file.
271         * @param webPath a full vfs resource path
272         * @return the clan associated with the path, or null
273         */
274        public Clan getWebPathClanMapping(String webPath);
275
276        /**
277         * Returns the template vfs path associated with a given specific
278         * web path. It is derived from the clanwebsites entry in
279         * coffeemud.ini file.
280         * @param webPath a full vfs resource path
281         * @return the template path associated with the resource path, or null
282         */
283        public String getClanWebTemplateDir(String webPath);
284
285        /**
286         * Forces all clans to go through their maintenance process, which
287         * normally only occurs infrequently.  This does things like handle
288         * automatic promotions, manage votes, and clean out inactive clans.
289         */
290        public void tickAllClans();
291
292        /**
293         * Sends a message to the games official CLAN chat channel.  This
294         * is normally for messages that may interest all clans.
295         * @param msg the message to send
296         */
297        public void clanAnnounceAll(String msg);
298
299        /**
300         * Returns a descriptive name for the given trophy code number.
301         * @see com.planet_ink.coffee_mud.Common.interfaces.Clan.Trophy
302         * @param trophy the trophy code number
303         * @return the descriptive name
304         */
305        public String translatePrize(Trophy trophy);
306
307        /**
308         * Returns whether this mud has activated its trophy system for clans.
309         * @return whether this mud has activated its trophy system for clans.
310         */
311        public boolean trophySystemActive();
312        /**
313         * Returns whether the given MOB is a member of any of the families
314         * represented by the given list of clan members.
315         * @param M the mob to evaluate
316         * @param members the members of a clan
317         * @return true if the mob is a family member, and false otherwise
318         */
319        public boolean isFamilyOfMembership(MOB M, List<MemberRecord> members);
320
321        /**
322         * Returns the list of clans this mob belongs to in the given category.
323         * @param M the mob to evaluate
324         * @param category the clan goverment category
325         * @return the list of clans this mob belongs to
326         */
327        public List<Pair<Clan,Integer>> getClansByCategory(MOB M, String category);
328
329        /**
330         * Returns help on the government type named, if it is available
331         * @param mob the viewer of the government type
332         * @param named the possible name of the government
333         * @param exact true to only match exact, or false otherwise
334         * @return null, or the help for the government named
335         */
336        public String getGovernmentHelp(MOB mob, String named, boolean exact);
337
338        /**
339         * Returns a government definition object of the given internal
340         * stock clangovernments.xml id.  See /resources/clangovernments.xml
341         * Also:
342         * @see com.planet_ink.coffee_mud.Common.interfaces.ClanGovernment
343         * @param typeid the internal typeid
344         * @return the clan government object
345         */
346        public ClanGovernment getStockGovernment(int typeid);
347
348        /**
349         * Returns the default government definition object.
350         * See /resources/clangovernments.xml
351         * Also:
352         * @see com.planet_ink.coffee_mud.Common.interfaces.ClanGovernment
353         * @return the clan government object
354         */
355        public ClanGovernment getDefaultGovernment();
356
357        /**
358         * Creates, but does not add, a sample government object
359         * @return a sample government object
360         */
361        public ClanGovernment createSampleGovernment();
362
363        /**
364         * Returns all government definition objects from internal
365         * stock clangovernments.xml file.  See /resources/clangovernments.xml
366         * Also:
367         * @see com.planet_ink.coffee_mud.Common.interfaces.ClanGovernment
368         * @return the clan government object
369         */
370        public ClanGovernment[] getStockGovernments();
371
372        /**
373         * Converts a given clan government object into xml.
374         * @see com.planet_ink.coffee_mud.Common.interfaces.ClanGovernment
375         * @see com.planet_ink.coffee_mud.Libraries.interfaces.ClanManager#makeGovernmentXML(com.planet_ink.coffee_mud.Common.interfaces.ClanGovernment[])
376         * @see com.planet_ink.coffee_mud.Libraries.interfaces.ClanManager#parseGovernmentXML(StringBuffer)
377         * @param gvt the clan government object
378         * @return the xml
379         */
380        public String makeGovernmentXML(ClanGovernment gvt);
381
382        /**
383         * Converts a given clan government objects into xml.
384         * @see com.planet_ink.coffee_mud.Common.interfaces.ClanGovernment
385         * @see com.planet_ink.coffee_mud.Libraries.interfaces.ClanManager#makeGovernmentXML(com.planet_ink.coffee_mud.Common.interfaces.ClanGovernment)
386         * @see com.planet_ink.coffee_mud.Libraries.interfaces.ClanManager#parseGovernmentXML(StringBuffer)
387         * @param gvts the clan government objects
388         * @return the xml
389         */
390        public String makeGovernmentXML(ClanGovernment gvts[]);
391
392        /**
393         * Converts xml into clan government objects.
394         * @see com.planet_ink.coffee_mud.Common.interfaces.ClanGovernment
395         * @see com.planet_ink.coffee_mud.Libraries.interfaces.ClanManager#makeGovernmentXML(com.planet_ink.coffee_mud.Common.interfaces.ClanGovernment)
396         * @see com.planet_ink.coffee_mud.Libraries.interfaces.ClanManager#makeGovernmentXML(com.planet_ink.coffee_mud.Common.interfaces.ClanGovernment[])
397         * @param xml the xml
398         * @return the clan government objects
399         */
400        public ClanGovernment[] parseGovernmentXML(StringBuffer xml);
401
402        /**
403         * Forces the stock governments to be re-saved to clangovernments.xml.
404         */
405        public void reSaveGovernmentsXML();
406
407        /**
408         * Creates the new stock government
409         * @param name governmentname
410         * @return new stock government
411         */
412        public ClanGovernment createGovernment(String name);
413
414        /**
415         * Deletes the stock government.. confusing all the clans that
416         * currently use it.
417         * @param government the government to delete
418         * @return true if it was there to remove, false otherwise
419         */
420        public boolean removeGovernment(ClanGovernment government);
421
422        /**
423         * Makes an announcement to the clan announcement channel from
424         * the given mob.  These are channels marked in the coffeemud.ini file
425         * as receiving clan info messages.  These messages are only seen
426         * by the authors clan members.
427         * @param mob the mob who is announcing the message
428         * @param msg string message to send to the clan info channels
429         */
430        public void clanAnnounce(MOB mob, String msg);
431
432        /**
433         * Examines the given command string, which is based on the given clan function code.
434         * If the given Clan requires a vote to accomplish it, the vote will be created. If
435         * the given clan forbids the given mob from performing the given function, or from
436         * even starting a vote on it, the method will return false. If the mob is allowed
437         * to perform
438         * @see com.planet_ink.coffee_mud.Common.interfaces.Clan.Function#ACCEPT
439         * @param mob the player who wants to perform the function
440         * @param C the clan that the player belongs to
441         * @param commands the command list describing the function that wants to be executed
442         * @param function the function code described by the commands list
443         * @param voteIfNecessary true to start a vote if one is needed, false to just return true.
444         * @return true to execute the given command, and false not to.
445         */
446        public boolean goForward(MOB mob, Clan C, List<String> commands, Clan.Function function, boolean voteIfNecessary);
447
448        /**
449         * Force the clans maintenance thread
450         */
451        public void forceTick();
452}
