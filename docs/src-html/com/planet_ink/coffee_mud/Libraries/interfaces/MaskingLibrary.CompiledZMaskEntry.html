205                private final String[] keys;
206                private ZapperKey(final String... exts)
207                {
208                        final List<String> k = new ArrayList<String>();
209                        k.add(name().startsWith("_") ? ("-"+name().substring(1)) : ("+"+name()));
210                        for(final String x : exts)
211                        {
212                                if(!k.contains(x))
213                                        k.add(x);
214                        }
215                        keys = k.toArray(new String[0]);
216                }
217
218                public String[] keys()
219                {
220                        return keys;
221                }
