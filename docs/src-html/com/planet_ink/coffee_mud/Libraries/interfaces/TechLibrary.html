034public interface TechLibrary extends CMLibrary
035{
036        
037        /**
038         * Returns the appropriate electronics key for this electronic item,
039         * area, or room.  It represents the electronical currents that tie
040         * technology together.
041         * @param o the object to inspect
042         * @return the key to use, or null if none can be found
043         */
044        public String getElectronicsKey(final CMObject o);
045
046        /**
047         * Unregisters an electronic component that belonged
048         * in a complex circuitry, like a panel or a
049         * generator.
050         * @param E the electronic component to unregister
051         * @param oldKey the last key registered to this device
052         */
053        public void unregisterElectronics(Electronics E, String oldKey);
054
055        /**
056         * Unregisters all electronic components that belonged
057         * in a complex circuitry, like a panel or a
058         * generator, of the given key.
059         * @param oldKey the last key registered to this device
060         */
061        public void unregisterAllElectronics(final String oldKey);
062
063        /**
064         * Registers an electronic component that belongs
065         * in a complex circuitry, like a panel or a
066         * generator
067         * @param E the electronic component to register
068         * @param oldKey the last key registered to this device
069         * @return the new key assigned to this item (or old key)
070         */
071        public String registerElectrics(Electronics E, String oldKey);
072
073        /**
074         * Returns whether the currents at the given key are still
075         * active.  Some currents go inactive when players leave
076         * the game, or their areas may be suspended by the system.
077         * @param key the current key
078         * @return true if the area is active
079         */
080        public boolean isCurrentActive(final String key);
081
082        /**
083         * Returns a new exclusive list of all the registered electronics
084         * keys being processed.
085         * @return the list of keys
086         */
087        public List<String> getMakeRegisteredKeys();
088
089        /**
090         * For the given key, return an eclusive list of all the electronics
091         * that belong to that key.
092         * @param key the key to return electronics for
093         * @return the list of electronic object of that key
094         */
095        public List<Electronics> getMakeRegisteredElectronics(String key);
096
097        /**
098         * Certain Key Systems may automatically force batteries in their circuit
099         * to activate in order to provide that system with power. This will make
100         * that attempt.
101         * @param E the key device to seek power
102         * @param key this devices key
103         * @return true if an attempt to give power was made, false otherwise.
104         */
105        public boolean seekBatteryPower(final ElecPanel E, final String key);
106
107        /**
108         * Retrieves the default manufacturer for new products
109         * @return the default manufacturer for new products
110         */
111        public Manufacturer getDefaultManufacturer();
112
113        /**
114         * Adds and saves a new manufacturer to the list.
115         * @param manufacturer the one to add
116         */
117        public void addManufacturer(Manufacturer manufacturer);
118
119        /**
120         * Removes a new manufacturer from the list.
121         * @param manufacturer the one to remove
122         */
123        public void delManufacturer(Manufacturer manufacturer);
124
125        /**
126         * Updates a new manufacturer in the list.
127         * @param manufacturer the one to update
128         */
129        public void updateManufacturer(Manufacturer manufacturer);
130
131        /**
132         * Retrieves the manufacturer of the given name, or null
133         * if it is not found.  Can not handle RANDOM!!
134         * @param name the manufacturer to fetch
135         * @return the manufacturer found, or null
136         */
137        public Manufacturer getManufacturer(String name);
138
139        /**
140         * Retrieves the manufacturer of the given name, or null
141         * if it is not found.  If random is sent, it will find
142         * a random manufacturer for the given item.
143         * @param E for random manufacturers, the item to check
144         * @param name the manufacturer to fetch
145         * @return the manufacturer found, or null
146         */
147        public Manufacturer getManufacturerOf(Electronics E, String name);
148
149        /**
150         * Returns an iterator of manufacturers
151         * @return the set to return
152         */
153        public Iterator<Manufacturer> manufacterers();
154
155        /**
156         * Return all the rooms containing computers in the
157         * given circuit key.
158         * @param key the circuit key
159         * @return an iterator of rooms
160         */
161        public Iterator<Room> getComputerRooms(String key);
162        /**
163         * Return all the computers in the given circuit key.
164         * @param key the circuit key
165         * @return an iterator of computers
166         */
167        public Iterator<Computer> getComputers(String key);
168
169        /**
170         * Returns a random gaussian-distributed tech level from the
171         * current low tech level bound to bound+10.
172         * @return a random valid tech level;
173         */
174        public int getRandomGlobalTechLevel();
175
176        /**
177         * Returns the global tech level
178         * @return the global tech level
179         */
180        public int getGlobalTechLevel();
181
182        /**
183         * "Fixes" the tech level of the given item by making sure
184         * it's manufacturer is non-random, and by assigning a random
185         * tech level within the valid range, and modifying its name
186         * and displaytext to reflect the new tech level
187         * @param I An electronics item that needs fixing
188         * @param newTechLevel the new tech level
189         */
190        public void fixItemTechLevel(Electronics I, int newTechLevel);
191        
192        /**
193         * If the two given objects are within an appropriate distance
194         * from each other, this will return the correct amount of
195         * accelleration g-force applied by the second object to the 
196         * first. typically <= 1G
197         * 
198         * @param S the object being pulled
199         * @param cO the object pulling
200         * @return the amount of gravity force, or 0
201         */
202        public double getGravityForce(SpaceObject S, SpaceObject cO);
203}
