032/**
033 * The color library does just what it says, provides access to methods
034 * to translate colors from words, to ansi codes, to html tags, and
035 * combine them all in ways appropriate for their use.
036 * 
037 * @author Bo Zimmerman
038 *
039 */
040public interface ColorLibrary extends CMLibrary
041{
042        /**
043         * These are the straight color codes and character codes
044         * such as blink adn so forth.  Includes back and foreground,
045         * and lots of codes for each one, including ansi codes,
046         * html tags, ^ codes, and so forth.
047         * 
048         * @author Bo Zimmerman
049         */
050        public enum Color
051        {
052                WHITE("\033[1;37m","<FONT COLOR=WHITE",'w'),
053                LIGHTGREEN("\033[1;32m","<FONT COLOR=LIGHTGREEN",'g'),
054                LIGHTBLUE("\033[1;34m","<FONT COLOR=BLUE",'b'),
055                LIGHTRED("\033[1;31m","<FONT COLOR=RED",'r'),
056                YELLOW("\033[1;33m","<FONT COLOR=YELLOW",'y'),
057                LIGHTCYAN("\033[1;36m","<FONT COLOR=CYAN",'c'),
058                LIGHTPURPLE("\033[1;35m","<FONT COLOR=VIOLET",'p'),
059                GREY("\033[0;37m","<FONT COLOR=LIGHTGREY",'W'),
060                GREEN("\033[0;32m","<FONT COLOR=GREEN",'G'),
061                BLUE("\033[0;34m","<FONT COLOR=#000099",'B'),
062                RED("\033[0;31m","<FONT COLOR=#993300",'R'),
063                BROWN("\033[0;33m","<FONT COLOR=#999966",'Y'),
064                CYAN("\033[0;36m","<FONT COLOR=DARKCYAN",'C'),
065                PURPLE("\033[0;35m","<FONT COLOR=PURPLE",'P'),
066                DARKGREY("\033[1;30m","<FONT COLOR=GRAY",'k'),
067                BLACK("\033[0;30m","<FONT COLOR=BLACK",'K'),
068                NONE("\033[0;0m","</I></U></BLINK></B></FONT"),
069                BOLD("\033[1m","<B"),
070                UNDERLINE("\033[4m","<U"),
071                BLINK("\033[5m","<BLINK"),
072                ITALICS("\033[6m","<I"),
073                BGWHITE("\033[47m"," style=\"background-color: white\""),
074                BGGREEN("\033[42m"," style=\"background-color: green\""),
075                BGBLUE("\033[44m"," style=\"background-color: #000099\""),
076                BGRED("\033[41m"," style=\"background-color: #993300\""),
077                BGYELLOW("\033[43m"," style=\"background-color: #999966\""),
078                BGCYAN("\033[46m"," style=\"background-color: darkcyan\""),
079                BGPURPLE("\033[45m"," style=\"background-color: purple\""),
080                BGBLACK("\033[40m"," style=\"background-color: black\""),
081                BGDEFAULT("\033[49m"," style=\"background-color: white\""),
082                ;
083                
084                private final String    ansiCode;
085                private final String    htmlTag;
086                private final char              codeLetter;
087                private final boolean   isBasicColor;
088                private final boolean   isExtendedColor;
089                
090                private Color(String ansiCode, String htmlTag, char codeLetter)
091                {
092                        this.ansiCode = ansiCode;
093                        this.codeLetter = codeLetter;
094                        this.htmlTag = htmlTag;
095                        isBasicColor = ((this.codeLetter != 'K') && (this.codeLetter != '\0'));
096                        isExtendedColor = (this.codeLetter != '\0');
097                }
098                
099                private Color(String ansiCode, String htmlTag)
100                {
101                        this(ansiCode, htmlTag, '\0');
102                }
103                
104                /**
105                 * True if its a basic 16 color, but also non-black.
106                 * @return its a basic 16 color, but also non-black.
107                 */
108                public final boolean isBasicColor()
109                {
110                        return isBasicColor;
111                }
112                
113                /**
114                 * True if its a basic 16 color, including black.
115                 * @return its a basic 16 color, including black.
116                 */
117                public final boolean isExtendedColor()
118                {
119                        return isExtendedColor;
120                }
121
122                /**
123                 * Returns the HTML tag for this color code
124                 * @return the HTML tag for this color code
125                 */
126                public final String getHtmlTag()
127                {
128                        return htmlTag;
129                }
130
131                /**
132                 * Returns the ANSI escape codes for this color.
133                 * @return the ANSI escape codes for this color.
134                 */
135                public final String getANSICode()
136                {
137                        return ansiCode;
138                }
139
140                /**
141                 * Returns the ^ char code, or 0 if its not a basic color
142                 * @return the ^ char code, or 0 if its not a basic color
143                 */
144                public final char getCodeChar()
145                {
146                        return codeLetter;
147                }
148                
149                /**
150                 * Returns the name, but with - instead of _
151                 * @return the name, but with - instead of _
152                 */
153                public final String getCodeString()
154                {
155                        return name();
156                }
157                
158        }
159

203        /**
204         * The special color codes are named color codes which are
205         * usually renameable by the user to other basic colors.
206         * 
207         * @author Bo Zimmerman
208         *
209         */
210        public enum SpecialColor
211        {
212                YOU_FIGHT('f'),
213                FIGHT_YOU('e'),
214                FIGHT('F'),
215                SPELL('S'),
216                EMOTE('E'),
217                TALK('T'),
218                TELL('t'),
219                CHANNEL('Q'),
220                CHANNELFORE('q'),
221                IMPORTANT1('x'),
222                IMPORTANT2('X'),
223                IMPORTANT3('Z'),
224                ROOMTITLE('O'),
225                ROOMDESC('L'),
226                DIRECTION('D'),
227                DOORDESC('d'),
228                ITEM('I'),
229                MOB('M'),
230                HITPOINTS('h'),
231                MANA('m'),
232                MOVES('v'),
233                NORMAL('N'),
234                HIGHLIGHT('H'),
235                UNEXPDIRECTION('U'),
236                UNEXPDOORDESC('u'),
237                WEATHER('J')
238                ;
239                
240                private final char              cchar;
241                private final String    underStr;
242                private final String    escapeCode;
243                
244                private SpecialColor(char escapeChar)
245                {
246                        this.cchar = escapeChar;
247                        this.underStr = name().replace('_', '-');
248                        this.escapeCode = "^"+cchar;
249                }
250                
251                /**
252                 * Returns the ^ color code
253                 * @return the ^ color code
254                 */
255                public final char getCodeChar()
256                {
257                        return cchar;
258                }
259                
260                /**
261                 * Returns the FULL ^ color code
262                 * @return the FULL ^ color code
263                 */
264                public final String getEscapeCode()
265                {
266                        return escapeCode;
267                }
268                
269                /**
270                 * Returns the name with _ replaced by -
271                 * @return the name with _ replaced by -
272                 */
273                public final String getCodeString()
274                {
275                        return underStr;
276                }
277        }

279        /**
280         * Color code prefix to designate background color (^ for foreground)
281         */
282        public static final char COLORCODE_BACKGROUND='~';
283        
284        /**
285         * Color code prefix to designate foreground ansi color
286         */
287        public static final char COLORCODE_FANSI256='#';
288        
289        /**
290         * Color code prefix to designate background ansi color
291         */
292        public static final char COLORCODE_BANSI256='|';
293
294        /**
295         * A color state is a class saved for users so the system knows
296         * what the current color situation is, which allows it to go
297         * back when it changes.
298         * @author Bo Zimmerman
299         *
300         */
301        public interface ColorState
302        {
303                /**
304                 * Returns the foreground basic 16 color code
305                 * @return the foreground basic 16 color code
306                 */
307                public char foregroundCode();
308                
309                /**
310                 * Returns the background basic 16 color code
311                 * @return the background basic 16 color code
312                 */
313                public char backgroundCode();
314        }
315
316        /**
317         * Clears the color code lookup tables so that the next
318         * translations will come from the properties.
319         */
320        public void clearLookups();
321
322        /**
323         * Translates a basic 16 or special color code, anything
324         * that starts with ^ to the ansi escape sequence.
325         * @see ColorLibrary#translateANSItoCMCode(String)
326         * @param code the ^ code
327         * @return the ansi escape sequence
328         */
329        public String translateCMCodeToANSI(String code);
330
331        /**
332         * Translates an ansi escape sequence to a 
333         * basic 16 or special color code, anything
334         * that starts with ^ to the .
335         * @see ColorLibrary#translateCMCodeToANSI(String)
336         * @param code ansi escape sequence
337         * @return the ^ code
338         */
339        public String translateANSItoCMCode(String code);
340
341        /**
342         * Given a foreground and background set of html tags,
343         * this method generates a single useable html tag
344         * from the two.
345         * @see ColorLibrary#mixColorCodes(String, String)
346         * @param code1 the first html tag
347         * @param code2 the other html tag, or null
348         * @return the combined html tag
349         */
350        public String mixHTMLCodes(String code1, String code2);
351
352        /**
353         * Given a foreground and background set of ansi escape codes,
354         * this method generates a single useable ansi escape code
355         * from the two.
356         * @see ColorLibrary#mixHTMLCodes(String, String)
357         * @param code1 the first ansi escape code
358         * @param code2 the other ansi escape code, or null
359         * @return the combined ansi escape code
360         */
361        public String mixColorCodes(String code1, String code2);
362
363        /**
364         * Does nothing more impressive than adding the color codes
365         * for combat to the source and target messages if possible.
366         * @param msg the message to colorize
367         * @return the same CMMsg sent, returned.
368         */
369        public CMMsg fixSourceFightColor(CMMsg msg);
370
371        /**
372         * Returns the standard html tag lookup table, which
373         * maps an array indexed by the ^ color codes to an html
374         * tag.
375         * @see ColorLibrary#standardColorLookups()
376         * @return the html tag lookup table
377         */
378        public String[] standardHTMLlookups();
379        
380        /**
381         * Returns the standard ansi escape color lookup table, which
382         * maps an array indexed by the ^ color codes to an ansi escape 
383         * color
384         * @see ColorLibrary#standardHTMLlookups()
385         * @return the ansi escape color lookup table
386         */
387        public String[] standardColorLookups();
388        
389        /**
390         * Generates a new ColorState object from the given 
391         * foreground and background basic 16 ^ color codes.
392         * @see ColorLibrary.ColorState
393         * @param fg the foreground color code char
394         * @param bg the background color code char
395         * @return the new ColorState object
396         */
397        public ColorState valueOf(final char fg, final char bg);
398        
399        /**
400         * A ColorState object that represents the basic normal
401         * color, typically grey with no background.
402         * @see ColorLibrary.ColorState
403         * @return the normal color ColorState
404         */
405        public ColorState getNormalColor();
