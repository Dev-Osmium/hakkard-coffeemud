034/**
035 * Library for configureing and managing the rules for what resource
036 * components (magic dust, tools, etc) are required every time a 
037 * particular skill is used.  Normally this would be part of the 
038 * skill itself, but since this feature was added so late, it's 
039 * separate.  
040 * 
041 *  Also here are common skill limit utilities, for determining
042 *  how many common skills a player can learn.
043 *  
044 * @author Bo Zimmerman
045 */
046public interface AbilityComponents extends CMLibrary
047{
048        /**
049         * Checks whether the given mob has the given components
050         * required to use a skill available to him/her, and if found,
051         * returns them as a FoundComponents list.
052         * @see AbilityComponents#getAbilityComponents(String)
053         * @param mob the mob whose inventory or room or both to check
054         * @param req the ability components rules definition
055         * @param mithrilOK true to allow mithril as a metal substitute
056         * @return null if missing components, or the list of found components
057         */
058        public List<Object> componentCheck(MOB mob, List<AbilityComponent> req, boolean mithrilOK);
059        
060        /**
061         * Returns a very dirty approximate of a sample of what components appear
062         * to be required.  Named items are not required, but raw material components
063         * are created, and most rules are ignored in order to generate as many as
064         * possible.
065         * @see AbilityComponents#getAbilityComponents(String)
066         * @param req the ability components rules definition
067         * @param mithrilOK true to allow mithril as a metal substitute
068         * @return a list of sample items
069         */
070        public List<Item> componentsSample(List<AbilityComponent> req, boolean mithrilOK);
071        
072        /**
073         * If the ability component recipe used to build the list of found
074         * components needed to use a skill requires that any of the componenets
075         * are destroyed.
076         * @see AbilityComponents#componentCheck(MOB, List, boolean)
077         * @see MaterialLibrary.DeadResourceRecord
078         * @param found the components found with componentCheck
079         * @return the record of the components destroyed
080         */
081        public MaterialLibrary.DeadResourceRecord destroyAbilityComponents(List<Object> found);
082        
083        /**
084         * Returns a friendly readable form of the component requirements
085         * of the given Ability/Skill ID(), or null if that ability has
086         * no requirements.  Since requirements may differ by player
087         * mask, the player mob is also required.
088         * @see AbilityComponents#getAbilityComponentDesc(MOB, List)
089         * @see AbilityComponents#getAbilityComponentDesc(MOB, AbilityComponent, boolean)
090         * @param mob the player mob who wants to know
091         * @param AID the Ability ID() of the skill whose components to check
092         * @return a friendly readable form of the component requirements
093         */
094        public String getAbilityComponentDesc(MOB mob, String AID);
095        
096        /**
097         * Returns a friendly readable form of the component requirements
098         * of the given Ability/Skill Component List, or null if it has
099         * no requirements.  Since requirements may differ by player
100         * mask, the player mob is also required.
101         * @see AbilityComponents#getAbilityComponentDesc(MOB, AbilityComponent, boolean)
102         * @see AbilityComponents#getAbilityComponentDesc(MOB, String)
103         * @param mob the player mob who wants to know
104         * @param req the coded requirements list
105         * @return a friendly readable form of the component requirements
106         */
107        public String getAbilityComponentDesc(MOB mob, List<AbilityComponent> req);
108        
109        /**
110         * Returns a friendly readable description of a specific component in 
111         * the given decoded ability components definition list.  If the 
112         * component does not refer to the given mob, "" is returned.  
113         * @see AbilityComponents#getAbilityComponentDesc(MOB, List)
114         * @see AbilityComponents#getAbilityComponentDesc(MOB, String)
115         * @param mob the mob to check this components applicability to.
116         * @param comp the complete ability component decoded
117         * @param useConnector true to use a connector AND/OR, false otherwise
118         * @return a friendly readable description of a specific component
119         */
120        public String getAbilityComponentDesc(MOB mob, AbilityComponent comp, boolean useConnector);
121        
122        /**
123         * Returns the master ability component map, keyed by the Ability ID.
124         * @return the master ability component map, keyed by the Ability ID.
125         */
126        public Map<String, List<AbilityComponent>> getAbilityComponentMap();
127        
128        /**
129         * Adds a new coded ability component to the given component map.
130         * The component is coded as found in components.txt, with ID=parms.
131         * @param s the new coded ability component string 
132         * @param H the map to add the new component to
133         * @return an error string, or null if everything went well.
134         */
135        public String addAbilityComponent(String s, Map<String, List<AbilityComponent>> H);
136        
137        /**
138         * Gets the decoded ability component definition for a given Ability ID().
139         * This is then used by other methods to determine whether a user has the
140         * necessary components, or for manipulation of the definition.
141         * @param AID the Ability ID()
142         * @return the decoded ability component definition for a given Ability ID()
143         */
144        public List<AbilityComponent> getAbilityComponents(String AID);
145        
146        /**
147         * Breaks an ability component decoded objects into a series of key/value pairs,
148         * where the first is always the connector, and the keys are as follows:
149         * ANDOR, DISPOSITION, FATE, AMOUNT, COMPONENTID, SUBTYPE, MASK. In that order.
150         * This is primarily for simplifying editors.
151         * @see AbilityComponents#setAbilityComponentCodedFromCodedPairs(PairList, AbilityComponent)
152         * @param comp the decoded ability component to produce fields from
153         * @return the key/value pairs of the ability component values.
154         */
155        public PairList<String, String> getAbilityComponentCoded(AbilityComponent comp);
156        
157        /**
158         * Copies the key/value pairs from a PairList of specific abilitycomponent fields
159         * into the given AbilityComponent object.       The first pairlist entry is always the 
160         * connector, and the keys are as follows: ANDOR, DISPOSITION, FATE, AMOUNT, 
161         * COMPONENTID, SUBTYPE, MASK. In that order.
162         * This is primarily for simplifying editors.
163         * @see AbilityComponents#getAbilityComponentCoded(AbilityComponent)
164         * @param decodedDV  the key/value pairs of the ability component values.
165         * @param comp the decoded ability component to copy field data into 
166         */
167        public void setAbilityComponentCodedFromCodedPairs(PairList<String, String> decodedDV, AbilityComponent comp);
168        
169        /**
170         * Reconstructs the coded ability component definition string (ID=parms)
171         * from the internal cached structures, given a particular Ability ID.
172         * @see AbilityComponents#getAbilityComponentCodedString(List)
173         * @param AID the Ability ID()
174         * @return the coded ability component definition string (ID=parms)
175         */
176        public String getAbilityComponentCodedString(String AID);
177        
178        /**
179         * Reconstructs the coded ability component definition string (ID=parms)
180         * from the given cached decoded structures list.
181         * @see AbilityComponents#getAbilityComponentCodedString(String)
182         * @param comps the decoded ability components definition list
183         * @return the coded ability component definition string (ID=parms)
184         */
185        public String getAbilityComponentCodedString(List<AbilityComponent> comps);
186        
187        /**
188         * Creates a new blank ability component object
189         * @return a new blank ability component object
190         */
191        public AbilityComponent createBlankAbilityComponent();
192        
193        /**
194         * Alters and saved the ability components definition to on the
195         * filesystem (components.txt).  
196         * @param compID the ID of the component being altered
197         * @param delete true to delete, false to add or modify
198         */
199        public void alterAbilityComponentFile(String compID, boolean delete);
200        
201        /**
202         * Returns the character-class based common skill ability limits
203         * object applicable to the given mob, or zeroes if there's
204         * a problem.
205         * @see AbilityComponents#getSpecialSkillLimit(MOB, Ability)
206         * @see AbilityComponents#getSpecialSkillRemainder(MOB, Ability)
207         * @see AbilityComponents#getSpecialSkillRemainders(MOB)
208         * @see AbilityLimits
209         * @param studentM the mob to find limits for
210         * @return the character-class based common skill ability limits
211         */
212        public AbilityLimits getSpecialSkillLimit(MOB studentM);
213        
214        /**
215         * Returns the character-class based common skill ability limits
216         * object applicable to the given mob and the given ability.
217         * @see AbilityComponents#getSpecialSkillLimit(MOB)
218         * @see AbilityComponents#getSpecialSkillRemainder(MOB, Ability)
219         * @see AbilityComponents#getSpecialSkillRemainders(MOB)
220         * @see AbilityLimits
221         * @see AbilityLimits#specificSkillLimit()
222         * @param studentM the mob to find limits for
223         * @param A the ability object to find limits for
224         * @return the character-class based common skill ability limits
225         */
226        public AbilityLimits getSpecialSkillLimit(MOB studentM, Ability A);
227        
228        /**
229         * Returns the character-class based common skill ability limits
230         * object applicable to the given mob and the given ability, and
231         * then subtracts the number of each common skill already learned
232         * to derive a remaining number of each type.
233         * @see AbilityComponents#getSpecialSkillLimit(MOB, Ability)
234         * @see AbilityComponents#getSpecialSkillLimit(MOB)
235         * @see AbilityComponents#getSpecialSkillRemainders(MOB)
236         * @see AbilityLimits
237         * @see AbilityLimits#specificSkillLimit()
238         * @param studentM the mob to find limits for
239         * @param A the ability object to find limits for
240         * @return the character-class based common skill ability remainders
241         */
242        public AbilityLimits getSpecialSkillRemainder(MOB studentM, Ability A);
243        
244        /**
245         * Returns the character-class based common skill ability limits
246         * object applicable to the given mob, and
247         * then subtracts the number of each common skill already learned
248         * to derive a remaining number of each type.
249         * @see AbilityComponents#getSpecialSkillLimit(MOB, Ability)
250         * @see AbilityComponents#getSpecialSkillLimit(MOB)
251         * @see AbilityComponents#getSpecialSkillRemainder(MOB, Ability)
252         * @see AbilityLimits
253         * @param studentM the mob to find limits for
254         * @return the character-class based common skill ability limits
255         */
256        public AbilityLimits getSpecialSkillRemainders(MOB studentM);
257        
258        /**
259         * Ability Limits object, denoting how many of different types
260         * of common skills and langs that a player can learn, including an
261         * entry for a specific skill.
262         * @author Bo Zimmerman
263         */
264        public static interface AbilityLimits
265        {
266                /**
267                 * Returns number of common skills
268                 * @return number of common skills
269                 */
270                public int commonSkills();
271
272                /**
273                 * Returns max number of common skills
274                 * @return max number of common skills
275                 */
276                public int maxCommonSkills();
277                
278                /**
279                 * Sets number of common skills
280                 * @param newVal number of common skills
281                 * @return this
282                 */
283                public AbilityLimits commonSkills(int newVal);
284
285                /**
286                 * Returns number of crafting skills
287                 * @return number of crafting skills
288                 */
289                public int craftingSkills();
290
291                /**
292                 * Returns max number of crafting skills
293                 * @return max number of crafting skills
294                 */
295                public int maxCraftingSkills();
296
297                /**
298                 * Sets number of crafting skills
299                 * @param newVal number of crafting skills
300                 * @return this
301                 */
302                public AbilityLimits craftingSkills(int newVal);
303
304                /**
305                 * Returns number of non-crafting skills
306                 * @return number of non-crafting skills
307                 */
308                public int nonCraftingSkills();
309
310                /**
311                 * Returns max number of non-crafting skills
312                 * @return max number of non-crafting skills
313                 */
314                public int maxNonCraftingSkills();
315
316                /**
317                 * Sets number of non-crafting skills
318                 * @param newVal number of non-crafting skills
319                 * @return this
320                 */
321                public AbilityLimits nonCraftingSkills(int newVal);
322
323                /**
324                 * Returns max number of language skills
325                 * @return max number of language skills
326                 */
327                public int maxLanguageSkills();
328
329                /**
330                 * Sets number of language skills
331                 * @param newVal number of language skills
332                 * @return this
333                 */
334                public AbilityLimits languageSkills(int newVal);
335
336                /**
337                 * Returns number of language skills
338                 * @return number of language skills
339                 */
340                public int languageSkills();
341
342                /**
343                 * Returns number of given specific ability type 
344                 * limit.
345                 * @return i don't know how to say it
346                 */
347                public int specificSkillLimit();
348
349                /**
350                 * Sets number of given specific ability type 
351                 * limit.
352                 * @param newVal a new number
353                 * @return this
354                 */
355                public AbilityLimits specificSkillLimit(int newVal);
356        }
