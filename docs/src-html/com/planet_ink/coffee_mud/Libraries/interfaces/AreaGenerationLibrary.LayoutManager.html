036/**
037 * The AreaGenerationLibrary is the random generator, which takes a special
038 * flavor of xml as input to randomly generate anything from strings, to
039 * mobs and items, whole populated rooms, or entire areas.  Most of the methods
040 * in this library provide a finer control to the generation process. 
041 * @author Bo Zimmerman
042 */
043public interface AreaGenerationLibrary extends CMLibrary
044{
045        /**
046         * Given an area-generation xml file, this method will return all the defined tag ids. This
047         * allows the caller to specifically select one of them for generation.
048         * @param xmlRoot the root of the area-generation xml file
049         * @param defined a map of ids to objects, whether strings or tags.
050         */
051        public void buildDefinedIDSet(List<XMLTag> xmlRoot, Map<String,Object> defined);
052
053        /**
054         * Given a specific ITEM generation tag, this method will return the items selected
055         * by that tag piece.
056         * @see AreaGenerationLibrary#buildDefinedIDSet(List, Map)
057         * @param piece the identified tag that can return items
058         * @param defined the defined id set from the entire xml document
059         * @return the list of items generated from the tag.
060         * @throws CMException any parsing or generation errors
061         */
062        public List<Item> findItems(XMLTag piece, Map<String,Object> defined) throws CMException;
063
064        /**
065         * Given a specific MOB generation tag, this method will return the MOBs selected
066         * by that tag piece.
067         * @see AreaGenerationLibrary#buildDefinedIDSet(List, Map)
068         * @param piece the identified tag that can return MOBs
069         * @param defined the defined id set from the entire xml document
070         * @return the list of MOBs generated from the tag.
071         * @throws CMException any parsing or generation errors
072         */
073        public List<MOB> findMobs(XMLTag piece, Map<String,Object> defined) throws CMException;
074
075        /**
076         * Returns a string of the given tag name type, from the given top-level xml tag piece that
077         * resolves to a string, and with the given pre-defined id set.
078         * @see AreaGenerationLibrary#buildDefinedIDSet(List, Map)
079         * @param tagName the name of the string tag
080         * @param piece the top level piece, probably of type tagname
081         * @param defined the pre-defined id set from the entire xml document
082         * @return the string this resolves to
083         * @throws CMException any parsing or generation errors
084         */
085        public String findString(String tagName, XMLTag piece, Map<String,Object> defined) throws CMException;
086
087        /**
088         * Given a specific ROOM generation tag, this method will return the room selected
089         * by that tag piece, with the entrace to it being in the given direction
090         * @see AreaGenerationLibrary#buildDefinedIDSet(List, Map)
091         * @param piece the identified tag that can return a room
092         * @param defined the defined id set from the entire xml document
093         * @param exits pre-defined exits from this room, if any
094         * @param direction the direction of entrance to this room
095         * @return the room generated from the tag.
096         * @throws CMException any parsing or generation errors
097         */
098        public Room buildRoom(XMLTag piece, Map<String,Object> defined, Exit[] exits, int direction) throws CMException;
099
100        /**
101         * Does nothing but check the requirements to build the given xml tag piece, and compares it with
102         * the variables in the given id definition map, to see if all requirements are met.  If not,
103         * it throws an exception.
104         * @param piece the xml tag piece you want to build
105         * @param defined the defined id set from the entire xml document, and the user
106         * @throws CMException any parsing or generation errors
107         */
108        public void checkRequirements(XMLTag piece, Map<String,Object> defined) throws CMException;
109
110        /**
111         * Check the requirements to build the given xml tag piece, and compares it with
112         * the variables in the given id definition map, to see if all requirements are met.  If not,
113         * it returns a map of the undefined or mis-defined ids as keys, mapped to the data type name.
114         * @param defined the defined id set from the entire xml document, and the user
115         * @param piece the xml tag piece you want to build
116         * @return a map of the undefined or mis-defined ids as keys, mapped to the data type name
117         */
118        public Map<String,String> getUnfilledRequirements(Map<String,Object> defined, XMLTag piece);
119        
120        /**
121         * Given a specific AREA generation tag, this method will return the area selected
122         * by that tag piece, with the entrance to it being in the given direction
123         * @see AreaGenerationLibrary#buildDefinedIDSet(List, Map)
124         * @param piece the identified tag that can return a room
125         * @param defined the defined id set from the entire xml document
126         * @param directions the direction of entrance to this room
127         * @return the area generated from the tag.
128         * @throws CMException any parsing or generation errors
129         */
130        public Area findArea(XMLTag piece, Map<String,Object> defined, int directions) throws CMException;
131        
132        /**
133         * Given a specific AREA generation tag, and an empty area, this method will 
134         * populate the area with rooms from the given tag piece, with the entrance 
135         * to it being in the given direction.
136         * @see AreaGenerationLibrary#buildDefinedIDSet(List, Map)
137         * @param piece the identified tag that can return a room
138         * @param defined the defined id set from the entire xml document
139         * @param A the area to put the rooms into
140         * @param direction the direction of entrance to this room
141         * @return true if everything went well, an exception otherwise
142         * @throws CMException any parsing or generation errors
143         */
144        public boolean fillInArea(XMLTag piece, Map<String,Object> defined, Area A, int direction) throws CMException;
145        
146        /**
147         * Returns the layout manager of the given name.
148         * @param named the name of the layout manager
149         * @return the layout manager of the given name.
150         */
151        public LayoutManager getLayoutManager(String named);
152        
153        /**
154         * Sometimes an object cannot be generated at a given time because some tag which
155         * will be defined later has not yet been defined, and cannot be resolved at
156         * generation time.  In those cases, this method is called with the defined id
157         * list as it was returned during primary area, room, item, mob, or string generation.
158         * The post-processes will be extracted from the defined ids, and then processed
159         * on those objects.
160         * @param defined the defined ids after they've gone through generating an object
161         * @throws CMException any parsing or generation errors
162         */
163        public void postProcess(final Map<String,Object> defined) throws CMException;
164        
165        /**
166         * If an xml tag is selected manually, by something outside the library, then 
167         * certain post-selection processes are not properly done, making generation
168         * based on the tag potentially impossible.  In those cases, before generation
169         * of the object from the selected xml tag piece, this method is called to handle
170         * post-selection processing (called rewarding, as in a reward for being picked).
171         * @param piece the selected xml tag piece
172         * @param defined the defined id set from the entire xml document
173         * @throws CMException any parsing or generation errors
174         */
175        public void defineReward(XMLTag piece, Map<String,Object> defined) throws CMException;
176        
177        /**
178         * If an xml tag is selected manually, by something outside the library, then 
179         * certain pre-selection processes are not properly done, making generation
180         * based on the tag potentially impossible.  In those cases, before generation
181         * of the object from the selected xml tag piece, this method is called to handle
182         * pre-selection processing (called rewarding, as in a reward for being picked).
183         * @param piece the selected xml tag piece
184         * @param defined the defined id set from the entire xml document
185         * @throws CMException any parsing or generation errors
186         */
187        public void preDefineReward(XMLTag piece, Map<String,Object> defined) throws CMException;
188        
189        /**
190         * Given a root xml tag and a tag name, this method will return all matching xml tag pieces.
191         * @param tagName the name of the tag to search for
192         * @param piece the root xml tag piece
193         * @param defined the defined id set from the entire xml document
194         * @return the list of found xml tags that match the tag name
195         * @throws CMException any parsing or generation errors
196         */
197        public List<XMLTag> getAllChoices(String tagName, XMLTag piece, Map<String,Object> defined) throws CMException;
198
199        /**
200         * Adjusts the levels of all mobs, items, and mob-items in the room by adjusting 
201         * them from their place in an existing range to a new range.Does not save -- that's up to you.
202         * @param room the room to adjust
203         * @param oldMin current minimum level range for the rooms area
204         * @param oldMax current maximum level range for the rooms area
205         * @param newMin new minimum level range for the rooms area
206         * @param newMax new maximum level range for the rooms area 
207         * @return true if the room was modified, false otherwise.
208         */
209        public boolean relevelRoom(Room room, int oldMin, int oldMax, int newMin, int newMax);
210        
211        /**
212         * Area generators work by first laying out a set of rooms into a
213         * configuration called a Layout. Layouts are scalable configurations
214         * that can handle any size, or originating direction of entry. 
215         * A Layout Manager then, is an algorithm for generating the room set
216         * given a size and initial direction.  Each Layout Manager then 
217         * generates a layout of rooms according to its kind.
218         * @author Bo Zimmerman
219         *
220         */
221        public static interface LayoutManager
222        {
223                /**
224                 * The name of the layout manager
225                 * @return name of the layout manager
226                 */
227                public String name();
228                
229                /**
230                 * Generates a list of layout nodes, each
231                 * of which reprents a single room in the area.
232                 * The first node is always the entry room, which
233                 * is entered from the direction given to this method.
234                 * @see AreaGenerationLibrary.LayoutNode
235                 * @param num the number of nodes to generate
236                 * @param dir the direction of entry into this group of nodes.
237                 * @return a list of layout nodes
238                 */
239                public List<LayoutNode> generate(int num, int dir);
240        }
241
242        /**
243         * A layout node represents a single room in a layout
244         * manager.  It contains numerous methods for linking
245         * nodes together the way rooms are, and for flagging
246         * nodes to give some idea of where the node fits in
247         * various layouts.
248         * @see AreaGenerationLibrary.LayoutTags
249         * @see AreaGenerationLibrary.LayoutTypes
250         * @see AreaGenerationLibrary.LayoutFlags
251         * @see AreaGenerationLibrary.LayoutRuns
252         * @author Bo Zimmerman
253         *
254         */
255        public static interface LayoutNode
256        {
257                /**
258                 * Uses this nodes coordinates, as well as the coordinates
259                 * of the given connected node, this method will create a 
260                 * link between them of the appropriate direction
261                 * see delLink(LayoutNode)
262                 * @see AreaGenerationLibrary.LayoutNode#getLink(int)
263                 * @see AreaGenerationLibrary.LayoutNode#links()
264                 * @see AreaGenerationLibrary.LayoutNode#deLink()
265                 * @param to the node to connect this one to.
266                 */
267                public void crossLink(LayoutNode to);
268                
269                /**
270                 * Removes all directional links between this node and the
271                 * given node.
272                 * see crossLink(LayoutNode)
273                 * @see AreaGenerationLibrary.LayoutNode#getLink(int)
274                 * @see AreaGenerationLibrary.LayoutNode#links()
275                 * @see AreaGenerationLibrary.LayoutNode#deLink()
276                 * @param linkNode the node to de-link from.
277                 */
278                public void delLink(LayoutNode linkNode);
279                
280                /**
281                 * Returns a link from this node in the given direction.  
282                 * crossLink must have already been called to establish the 
283                 * link.
284                 * see crossLink(LayoutNode)
285                 * see delLink(LayoutNode)
286                 * @see AreaGenerationLibrary.LayoutNode#links()
287                 * @see AreaGenerationLibrary.LayoutNode#deLink()
288                 * @param d the direction code
289                 * @return the node in the given direction
290                 */
291                public LayoutNode getLink(int d);
292                
293                /**
294                 * Removes all links from other nodes to this one, and removes
295                 * all links from this node to others.  crossLink must have been
296                 * called for there to be any links.
297                 * see crossLink(LayoutNode)
298                 * see delLink(LayoutNode)
299                 * @see AreaGenerationLibrary.LayoutNode#links()
300                 * @see AreaGenerationLibrary.LayoutNode#getLink(int)
301                 */
302                public void deLink();
303                
304                /**
305                 * Returns a map of directions to other layoutnodes.  These
306                 * being the established links between nodes. crossLink must have been
307                 * called for there to be any links.
308                 * see crossLink(LayoutNode)
309                 * see delLink(LayoutNode)
310                 * @see AreaGenerationLibrary.LayoutNode#deLink()
311                 * @see AreaGenerationLibrary.LayoutNode#getLink(int)
312                 * @return the map of links
313                 */
314                public Map<Integer, LayoutNode> links();
315                
316                /**
317                 * Returns comma-delimited values of each LayoutTag type,
318                 * through a may of layouttag to string mapping.
319                 * @see AreaGenerationLibrary.LayoutTags
320                 * @return the map of layouttags to strings
321                 */
322                public Map<LayoutTags, String> tags();
323                
324                /**
325                 * Returns x,y coordinates of this node.
326                 * These are generated by the layout manager.
327                 * @return x,y coordinates of this node
328                 */
329                public long[] coord();
330                
331                /**
332                 * Returns true if the first two linked nodes to this
333                 * node are in the same general direction, this making
334                 * this node look like a street node.
335                 * @return true if this node is part of a street.
336                 */
337                public boolean isStreetLike();
338                
339                /**
340                 * Flags this node with one of the visual layout flags,
341                 * telling whether it is at an intersection, or on a
342                 * corner, or something like that.
343                 * @param flag the flag to set.
344                 */
345                public void flag(LayoutFlags flag);
346                
347                /**
348                 * Returns whether this node is flagged with one of the
349                 * visual layout flags, telling whether it is an intersectino,
350                 * or on a corner, or something like that.
351                 * see flag(LayoutFlags)
352                 * @param flag the flag to check for
353                 * @return true if this node is flagged, false otherwise
354                 */
355                public boolean isFlagged(LayoutFlags flag);
356                
357                /**
358                 * Flags this node as being part of a street-like run.
359                 * @see AreaGenerationLibrary.LayoutRuns
360                 * @param dirs the direction of the run
361                 */
362                public void flagRun(LayoutRuns dirs);
363                
364                /**
365                 * Return whether this node is flagged as being part of a 
366                 * street-like run.
367                 * @see AreaGenerationLibrary.LayoutRuns
368                 * @return the direction of the run, or null
369                 */
370                public LayoutRuns getFlagRuns();
371                
372                /**
373                 * Returns the layout type flag that this node represents,
374                 * denoting whether it is a leaf, a street, or some other
375                 * role.
376                 * @see AreaGenerationLibrary.LayoutTypes
377                 * @return the type of this node
378                 */
379                public LayoutTypes type();
380                
381                /**
382                 * Sets the layout type flag that this node represents,
383                 * denoting whether it is a leaf, a street, or some other
384                 * role.
385                 * @see LayoutTypes
386                 * @param type the type of this node
387                 */
388                public void reType(LayoutTypes type);
389                
390                /**
391                 * Sets the exit available link directions, without setting
392                 * the nodes they are linked to -- what good is this? 
393                 * Well, it sets the LayoutTags.NODEEXITS string, which
394                 * makes some sense I guess.
395                 * @param dirs the array of Direction codes to set.
396                 */
397                public void setExits(int[] dirs);
398                
399                /**
400                 * Returns one line of a 3x3 character representation. Each call
401                 * returns 3 characters, with the middle being the roomchar, and
402                 * the rest depending on the links.  Call this three times with
403                 * 0, 1, and 2, to get all three lines.
404                 * @param roomChar the char to use for this room
405                 * @param line which line, 0, 1, or 2
406                 * @return the 3 character string for this line.
407                 */
408                public String getColorRepresentation(char roomChar, int line);
409                
410                /**
411                 * Returns the room object assigned to this node.
412                 * @see AreaGenerationLibrary.LayoutNode#setRoom(Room)
413                 * @return the room object assigned to this node.
414                 */
415                public Room room();
416                
417                /**
418                 * Sets the room object assigned to this node.
419                 * @see AreaGenerationLibrary.LayoutNode#room()
420                 * @param room the room object assigned to this node.
421                 */
422                public void setRoom(Room room);
423        }
424

437                NODEEXITS

446        public enum LayoutTypes

448                surround,
449                leaf,
450                street,
451                square,
452                interior
453        }
454
455        /**
456         * Enum of the type of room layoutnode
457         * @see AreaGenerationLibrary.LayoutNode
458         * @author Bo Zimmerman
459         *
460         */
461        public enum LayoutFlags
462        {
463                corner,
464                gate,
465                intersection,
466                tee,
467                offleaf
468        }
469
470        /**
471         * If this room layoutnode is a street type, what
472         * direction does it run?
473         * @see AreaGenerationLibrary.LayoutNode
474         * @author Bo Zimmerman

476         */
477        public enum LayoutRuns
478        {
479                ew,
480                ns,
481                ud,
482                nesw,
483                nwse
484        }
