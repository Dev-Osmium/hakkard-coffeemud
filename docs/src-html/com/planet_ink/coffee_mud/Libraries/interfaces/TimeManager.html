036public interface TimeManager extends CMLibrary
037{
038        /**
039         * Returns the numeric representation of the month
040         *
041         * Usage: month2MM("January");
042         * @param monthName The month name
043         * @return String The number of the month as a string
044         */
045        public String month2MM(String monthName);
046        /**
047         * Return the name of the month, given a number
048         *
049         * Usage: String Mnth=getMonthName(m,GiveShort).charStats();
050         * @param monthNum Month number to convert
051         * @param giveShort Give abbreviation if true
052         * @return String Month name
053         */
054        public String getMonthName(int monthNum, boolean giveShort);
055
056        /**
057         * Returns whether the given string would parse to a valid
058         * date.  If true is returned, the change of getting a valid
059         * date from string2Date is much higher.
060         * @param dateTimeStr the strong to parse
061         * @return true if its possibly valid, false definitely not
062         */
063        public boolean isValidDateString(String dateTimeStr);
064
065        /**
066         * Converts a string of some form into a Calendar object.
067         *
068         * Usage: string2Millis(GetRes(Results,"StartDateTime"));
069         * @param dateTimeStr The string to retrieve from
070         * @return Calendar Calendar object
071         */
072        public long string2Millis(String dateTimeStr);
073
074        /**
075         * Converts a string of some form into a Calendar object.
076         *
077         * Usage: Calendar.string2Date(GetRes(Results,"StartDateTime"));
078         * @param dateTimeStr The string to retrieve from
079         * @return Calendar Calendar object
080         */
081        public Calendar string2Date(String dateTimeStr);
082
083        /**
084         * Converts a number to two digits.
085         * @param num the number
086         * @return the number as two digits
087         */
088        public String twoDigits(long num);
089
090        /**
091         * Returns the regular Hours given the hours in the
092         * international format (military time)
093         *
094         * Usage: ConvertHour(GetIn(req, "ENDHR"))
095         * @param hours24 Hours in military format
096         * @return String Hours in regular format
097         **/
098        public String convertHour(String hours24);
099
100        /**
101         * Returns the AMPM stamp given the international Hours portion the Time
102         *
103         * Usage: getAMPM(GetIn(req, "ENDHR"))
104         * @param TheHour Hours in military format
105         * @return String AM or PM stamp
106         **/
107        public String getAMPM(String TheHour);
108
109        /**
110         * Get the zone id given the timezone string
111         *
112         * Usage: GetTheZoneID(MeetTZ.getRawOffset())+"\n";
113         * @param theRawOffset The time zone's raw offset to convert
114         * @return String The time zone ID
115         */
116        public String getTheIntZoneID(int theRawOffset);
117
118        /**
119         *  Returns the time zone of the given ID
120         *
121         * Usage: MEETZN = GetTheTimeZone(ID);
122         * @param theID The ID of the abbreviated time zone.
123         * @return String The time zone name
124         */
125        public String getTheTimeZone(String theID);
126
127        /**
128         * Returns the month for a given date
129         *
130         * Usage: String ENDMM=date2MonthString(time,true);
131         * @param time The time in miliseconds
132         * @param shortName true to use shortened months
133         * @return String The month name
134         **/
135        public String date2MonthString(long time, boolean shortName);
136
137        /**
138         * Returns time in HH:MMap format, unless the day is other
139         * than today, in which case it prepends yy/m/d.
140         *
141         * Usage: String ENDMM=date2BriefString(time,true);
142         * @param time The time in miliseconds
143         * @return String The date string
144         **/
145        public String date2BriefString(long time);
146
147        /**
148         * Returns the month/day for a given date
149         *
150         * Usage: String ENDMM=date2MonthDateString();
151         * @param time The time in miliseconds
152         * @param shortName true to use shortened months
153         * @return String the month/day name
154         **/
155        public String date2MonthDateString(long time, boolean shortName);
156
157        /**
158         * Return the time in HH:MM AP format.
159         * @param time the time in millis
160         * @return the time in format
161         */
162        public String date2APTimeString(long time);
163
164        /**
165         * Returns the DD portion of a given date
166         *
167         * Usage: String ENDDD=date2DayOfMonthString();
168         * @param time The time in miliseconds
169         * @return String The day
170         **/
171        public String date2DayOfMonthString(long time);
172
173        /**
174        * Returns the YYYY portion of a given date
175        * Returns the DD portion of a given date
176        *
177        * Usage: String ENDYYYY=date2YYYYString();
178         * @param time The time in miliseconds
179        * @return String The year
180        **/
181        public String date2YYYYString(long time);
182
183        /**
184        * Returns the Hours portion of a given Time
185        *
186        * Usage: String ENDHR=date2HRString();
187        * @param time time used
188        * @return String The hour
189        **/
190        public String date2HRString(long time);
191
192        /**
193        * Returns the Minutes portion of a given Time
194        *
195        * Usage: String ENDMIN=date2MINString();
196        * @param time The time in miliseconds
197        * @return String The minutes
198        **/
199        public String date2MINString(long time);
200
201        /**
202         * format the date for an smtp message
203         *
204         * Usage:  smtpDateFormat(98374987234)
205        * @param time The time in miliseconds
206        * @return String The minutes
207         */
208         public String smtpDateFormat(long time);
209
210        /**
211         *  Returns the time zone of the server
212         *
213         * Usage: MEETZN = T2ZoneString();
214         * @param time The time in miliseconds
215         * @return String The time zone
216         */
217        public String date2ZoneString(long time);
218        /**
219         * Returns the Minutes portion of a given Time
220         *
221         * Usage: String ST_AMPM=date2AMPMString(time);
222         * @param time The time in miliseconds
223         * @return String AM or PM stamp
224         **/
225        public String date2AMPMString(long time);
226        /**
227         * Converts a given date into a string of form:
228         * MM/DD/YYYY HH:MM AP
229         *
230         * Usage: date2String()
231         * @param C The time in calendar
232         * @return String Formatted date/time
233         */
234        public String date2String(Calendar C);
235        /**
236         * Converts a given date into a string of form:
237         * MM/DD/YYYY HH:MM AP
238         *
239         * Usage: date2String(time)
240         * @param time The time in miliseconds
241         * @return String Formatted date/time
242         */
243        public String date2String(long time);
244        /**
245         * Converts a given number of milliseconds,
246         * into a number of rl years, months, days,
247         * hours, minutes, and seconds.  If in short form,
248         * returns y, m, d, h, m, and s.
249         *
250         * Usage: date2EllapsedTime(time)
251         * @param time The time in miliseconds
252         * @param minUnit The smallest unit to round down to
253         * @param shortest true for short form, false otherwise
254         * @return String Formatted ellapsed time
255         */
256        public String date2EllapsedTime(long time, TimeUnit minUnit, boolean shortest);
257
258        /**
259         * Picks the single best representation of the given ellapsed time, 
260         * from seconds to minutes to hours to days, and returns the correct
261         * amount along with the single letter suffix
262         * @param t time in milliseconds
263         * @return ellapsed time approximation.
264         */
265        public String date2BestShortEllapsedTime(long t);
266        /**
267         * Converts a given number of milliseconds,
268         * into a number of rl years, months, days,
269         * hours, minutes, and seconds.  If in short form,
270         * returns y, m, d, h, m, and s.  This method
271         * will automatically determine the smallest reasonable
272         * unit of time to show.
273         *
274         * Usage: date2SmartEllapsedTime(time)
275         * @param time The time in miliseconds
276         * @param shortest true for short form, false otherwise
277         * @return String Formatted ellapsed time
278         */
279        public String date2SmartEllapsedTime(long time, boolean shortest);
280
281        /**
282         * Converts a given date into a string of form:
283         * MM/DD/YYYY HH:MM AP
284         *
285         * Usage: date2SecondsString(time)
286         * @param time The time in miliseconds
287         * @return String Formatted date/time
288         */
289        public String date2SecondsString(long time);
290        /**
291         * Converts a given date into a string of form:
292         * MM/DD/YYYY
293         *
294         * Usage: date2DateString(time)
295         * @param time The time in miliseconds
296         * @return String Formatted date
297         */
298        public String date2DateString(long time);
299        /**
300         * Converts a given date into a string of form:
301         * MM/DD/YY
302         *
303         * Usage: date2Date2String(time)
304         * @param time The time in miliseconds
305         * @return String Formatted date
306         */
307        public String date2Date2String(long time);
308
309        /**
310         * Parses either a tick expression, or an
311         * expression ending with the word minutes,
312         * hours, seconds, days, mudhours, muddays,
313         * mudweeks, mudmonths, or mudyears
314         * @param val the expression
315         * @return the number of ticks represented by the string
316         */
317        public int parseTickExpression(String val);
318
319        /**
320         * Parses whether a tick expression, or an
321         * expression ending with the word minutes,
322         * hours, seconds, days, mudhours, muddays,
323         * mudweeks, mudmonths, or mudyears
324         * @param val the expression
325         * @return whether a number of ticks is represented by the string
326         */
327        public boolean isTickExpression(String val);
328
329        /**
330         * Returns the most local clock for the given object.
331         * @param P an item, room, mob, whatever.
332         * @see com.planet_ink.coffee_mud.Common.interfaces.TimeClock
333         * @return the local clock
334         */
335        public TimeClock localClock(Physical P);
336
337        /**
338         * Returns the mud wide global time clock
339         * object as defined by the coffeemud.ini
340         * file.
341         * @see com.planet_ink.coffee_mud.Common.interfaces.TimeClock
342         * @return the global clock
343         */
344        public TimeClock globalClock();
345
346        /** constant for the names of the months in a rl year */
347        public final static String[] MONTHS={
348                "January","February","March","April","May","June","July","August","September","October","November","December"
349        };
350
351        /** constant for the short names of the months in a rl year */
352        public final static String[] SHORTMONTHS={
353                "Jan","Feb","Mar","Apr","May","Jun","Jul","Aug","Sep","Oct","Nov","Dec"
354        };
355
356        /** constant for the number of milliseconds in a second */
357        public final static long MILI_SECOND=1000;
358        /** constant for the number of milliseconds in a minute */
359        public final static long MILI_MINUTE=MILI_SECOND*60;
360        /** constant for the number of milliseconds in a rl hour */
361        public final static long MILI_HOUR=MILI_MINUTE*60;
362        /** constant for the number of milliseconds in a rl day */
363        public final static long MILI_DAY=MILI_HOUR*24;
364        /** constant for the number of milliseconds in a rl week */
365        public final static long MILI_WEEK=MILI_DAY*7;
366        /** constant for the number of milliseconds in a rl month */
367        public final static long MILI_MONTH=MILI_DAY*30;
368        /** constant for the number of milliseconds in a rl year */
369        public final static long MILI_YEAR=MILI_DAY*365;
