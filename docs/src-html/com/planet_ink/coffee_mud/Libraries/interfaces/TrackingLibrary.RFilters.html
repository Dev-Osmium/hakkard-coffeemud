109        public static enum TrackingFlag
110        {
111                NOHOMES(new RFilter()
112                {
113                        @Override
114                        public boolean isFilteredOut(final Room hostR, final Room R, final Exit E, final int dir)
115                        {
116                                return CMLib.law().getLandTitle(R) != null;
117                        }
118                }),
119                OPENONLY(new RFilter()
120                {
121                        @Override
122                        public boolean isFilteredOut(final Room hostR, final Room R, final Exit E, final int dir)
123                        {
124                                return !E.isOpen();
125                        }
126                }),
127                UNLOCKEDONLY(new RFilter()
128                {
129                        @Override
130                        public boolean isFilteredOut(final Room hostR, final Room R, final Exit E, final int dir)
131                        {
132                                return E.hasALock();
133                        }
134                }),
135                AREAONLY(new RFilter()
136                {
137                        @Override
138                        public boolean isFilteredOut(final Room hostR, final Room R, final Exit E, final int dir)
139                        {
140                                return (R!=null)&&(hostR!=null)&&(hostR.getArea()!=R.getArea());
141                        }
142                }),
143                NOHIDDENAREAS(new RFilter()
144                {
145                        @Override
146                        public boolean isFilteredOut(final Room hostR, final Room R, final Exit E, final int dir)
147                        {
148                                return ((R!=null)&&(CMLib.flags().isHidden(R)))
149                                        || ((R!=null)&&(R.getArea()!=null)&&(CMLib.flags().isHidden(R.getArea())));
150                        }
151                }),
152                NOEMPTYGRIDS(new RFilter()
153                {
154                        @Override
155                        public boolean isFilteredOut(final Room hostR, final Room R, final Exit E, final int dir)
156                        {
157                                return (R.getGridParent() != null) && (R.getGridParent().roomID().length() == 0);
158                        }
159                }),
160                NOAIR(new RFilter()
161                {
162                        @Override
163                        public boolean isFilteredOut(final Room hostR, final Room R, final Exit E, final int dir)
164                        {
165                                return (R.domainType() == Room.DOMAIN_INDOORS_AIR) || (R.domainType() == Room.DOMAIN_OUTDOORS_AIR);
166                        }
167                }),
168                NOPRIVATEPROPERTY(new RFilter()
169                {
170                        @Override
171                        public boolean isFilteredOut(final Room hostR, final Room R, final Exit E, final int dir)
172                        {
173                                final LegalLibrary law=CMLib.law();
174                                final LandTitle title = law.getLandTitle(R);
175                                return  (title == null) || (title.getOwnerName() == null) || (title.getOwnerName().length()==0);
176                        }
177                }),
178                NOWATER(new RFilter()
179                {
180                        @Override
181                        public boolean isFilteredOut(final Room hostR, final Room R, final Exit E, final int dir)
182                        {
183                                return CMLib.flags().isWateryRoom(R);
184                        }
185                }),
186                WATERSURFACEONLY(new RFilter()
187                {
188                        @Override
189                        public boolean isFilteredOut(final Room hostR, final Room R, final Exit E, final int dir)
190                        {
191                                return !CMLib.flags().isWaterySurfaceRoom(R);
192                        }
193                }),
194                WATERSURFACEORSHOREONLY(new RFilter()
195                {
196                        @Override
197                        public boolean isFilteredOut(final Room hostR, final Room R, final Exit E, final int dir)
198                        {
199                                if(R==null)
200                                        return true;
201                                if((R.domainType()==Room.DOMAIN_OUTDOORS_AIR)
202                                ||(R.domainType()==Room.DOMAIN_INDOORS_AIR))
203                                        return true;
204                                if(CMLib.flags().isWaterySurfaceRoom(R)
205                                || (R.ID().equals("Shore"))
206                                || (R.domainType() == Room.DOMAIN_OUTDOORS_SEAPORT))
207                                        return false;
208                                boolean foundWater=false;
209                                for(final int dir2 : Directions.CODES())
210                                {
211                                        final Room R2=R.getRoomInDir(dir2);
212                                        if((R2!=null)&&(CMLib.flags().isWaterySurfaceRoom(R2)))
213                                                foundWater=true;
214                                }
215                                return (!foundWater);
216                        }
217                }),
218                SHOREONLY(new RFilter()
219                {
220                        @Override
221                        public boolean isFilteredOut(final Room hostR, final Room R, final Exit E, final int dir)
222                        {
223                                if(R==null)
224                                        return true;
225                                if((R.domainType()==Room.DOMAIN_OUTDOORS_AIR)
226                                || (R.domainType()==Room.DOMAIN_INDOORS_AIR)
227                                || (CMLib.flags().isWaterySurfaceRoom(R) ))
228                                        return true;
229                                if((R.ID().equals("Shore"))
230                                || (R.domainType() == Room.DOMAIN_OUTDOORS_SEAPORT))
231                                        return false;
232                                boolean foundWater=false;
233                                for(final int dir2 : Directions.CODES())
234                                {
235                                        final Room R2=R.getRoomInDir(dir2);
236                                        if((R2!=null)&&(CMLib.flags().isWaterySurfaceRoom(R2)))
237                                                foundWater=true;
238                                }
239                                return (!foundWater);
240                        }
241                }),
242                UNDERWATERONLY(new RFilter()
243                {
244                        @Override
245                        public boolean isFilteredOut(final Room hostR, final Room R, final Exit E, final int dir)
246                        {
247                                return !CMLib.flags().isUnderWateryRoom(R);
248                        }
249                }),
250                FLOORSONLY(new RFilter()
251                {
252                        @Override
253                        public boolean isFilteredOut(final Room hostR, final Room R, final Exit E, final int dir)
254                        {
255                                return (R.getRoomInDir(Directions.DOWN)!=null);
256                        }
257                }),
258                CEILINGSSONLY(new RFilter()
259                {
260                        @Override
261                        public boolean isFilteredOut(final Room hostR, final Room R, final Exit E, final int dir)
262                        {
263                                return (R.getRoomInDir(Directions.UP)!=null);
264                        }
265                }),
266                NOCLIMB(new RFilter()
267                {
268                        @Override
269                        public boolean isFilteredOut(final Room hostR, final Room R, final Exit E, final int dir)
270                        {
271                                return (CMLib.flags().isClimbing(R) || CMLib.flags().isClimbing(E));
272                        }
273                }),
274                NOCRAWL(new RFilter()
275                {
276                        @Override
277                        public boolean isFilteredOut(final Room hostR, final Room R, final Exit E, final int dir)
278                        {
279                                return (CMLib.flags().isCrawlable(R) || CMLib.flags().isCrawlable(E));
280                        }
281                }),
282                OUTDOORONLY(new RFilter()
283                {
284                        @Override
285                        public boolean isFilteredOut(final Room hostR, final Room R, final Exit E, final int dir)
286                        {
287                                return (R.domainType() & Room.INDOORS) != 0;
288                        }
289                }),
290                INDOORONLY(new RFilter()
291                {
292                        @Override
293                        public boolean isFilteredOut(final Room hostR, final Room R, final Exit E, final int dir)
294                        {
295                                return (R.domainType() & Room.INDOORS) == 0;
296                        }
297                });
298                public RFilter myFilter;
299                private TrackingFlag(final RFilter filter)
300                {
301                        this.myFilter=filter;
302                }
303        }
