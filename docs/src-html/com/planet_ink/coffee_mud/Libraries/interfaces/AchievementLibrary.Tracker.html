038/**
039 * The achievement library forms the basis of the achievement system,
040 * which accepts player events as input, as well as the file based achievement
041 * definition files, and tracks player and account completion of achievements
042 * through trackers.
043 * @see AchievementLibrary.Event
044 * @see AchievementLibrary.Tracker
045 * @see AchievementLibrary.Achievement
046 * @author Bo Zimmerman
047 *
048 */
049public interface AchievementLibrary extends CMLibrary
050{
051        /**
052         * The list of arguments/parameters common to all achievement event types
053         */
054        public final String[] BASE_ACHIEVEMENT_PARAMETERS = new String[] { "EVENT", "DISPLAY", "TITLE", "REWARDS" };
055        
056        /**
057         * Events define the type of achievement, describing specific arguments that 
058         * the achievement of each event type needs. It is also used to specify the
059         * type of player event that has occurred, allowing the appropriate achievement
060         * to be determined and tracked.
061         * @author Bo Zimmerman
062         *
063         */
064        public enum Event
065        {
066                KILLS("Number of Kills",new String[]{"NUM","ZAPPERMASK","PLAYERMASK"}),
067                STATVALUE("A Stat value",new String[]{"VALUE","ABOVEBELOW","STAT"}),
068                FACTION("A Faction level",new String[]{"VALUE","ABOVEBELOW","ID"}),
069                EXPLORE("Exploration",new String[]{"PERCENT","AREA"}),
070                CRAFTING("Crafting",new String[]{"NUM","ABILITYID"}),
071                MENDER("Mending",new String[]{"NUM","ABILITYID"}),
072                SKILLUSE("Using Skills",new String[]{"NUM","ABILITYID"}),
073                SOCIALUSE("Using Socials",new String[]{"NUM","SOCIALID"}),
074                QUESTOR("Completing Quests",new String[]{"NUM","PLAYERMASK","QUESTMASK"}),
075                ACHIEVER("Completing Achievements",new String[]{"ACHIEVEMENTLIST"}), 
076                ROOMENTER("Entering a Room",new String[]{"ROOMID"}),
077                LEVELSGAINED("Gaining Levels",new String[]{"NUM","PLAYERMASK"}),
078                CLASSLEVELSGAINED("Gaining Class Levels",new String[]{"NUM","CLASS","PLAYERMASK"}),
079                TIMEPLAYED("Time Played",new String[]{"SECONDS","PLAYERMASK"}),
080                JUSTBE("Character State",new String[]{"PLAYERMASK"}),
081                DEATHS("Dieing",new String[]{"NUM","ZAPPERMASK","PLAYERMASK"}),
082                RETIRE("Retiring",new String[]{"NUM","PLAYERMASK"}),
083                REMORT("Remorting",new String[]{"NUM","PLAYERMASK"}),
084                GOTITEM("Got an item",new String[]{"NUM","ITEMMASK","PLAYERMASK"}),
085                FACTIONS("A group of factions",new String[]{"VALUE","ABOVEBELOW","IDMASK","NUM"}),
086                BIRTHS("Births",new String[]{"NUM","ZAPPERMASK","PLAYERMASK"}),
087                RACEBIRTH("Race Creation",new String[]{"NUM","ZAPPERMASK","PLAYERMASK"}),
088                PLAYERBORN("Being a Player Born",new String[]{"PLAYERMASK"}),
089                PLAYERBORNPARENT("Being a Player Parent",new String[]{"NUM","ZAPPERMASK","PLAYERMASK"}),
090                ;
091                private final String[] parameters;
092                private final String displayName;
093
094                private Event(final String displayName, final String[] extraParameters)
095                {
096                        this.displayName = displayName;
097                        parameters = CMParms.combine(BASE_ACHIEVEMENT_PARAMETERS, extraParameters);
098                }
099
100                /**
101                 * Returns the friendly display name of this event.
102                 * @return the friendly display name of this event.
103                 */
104                public String displayName()
105                {
106                        return this.displayName;
107                }
108                
109                /**
110                 * Returns all arguments, required and optional, to this achievement event type
111                 * @return all arguments to this achievement event type
112                 */
113                public String[] getParameters()
114                {
115                        return parameters;
116                }
117
118                /**
119                 * Returns a string list of all event types.
120                 * @return a string list of all event types.
121                 */
122                public static String[] getEventChoices()
123                {
124                        final List<String> choices=new ArrayList<String>(Event.values().length);
125                        for(Event E : Event.values())
126                                choices.add(E.name());
127                        return choices.toArray(new String[0]);
128                }
129        }

131        /**
132         * The achievement interface provides basic information about the specific achievement,
133         * as defined in the achievements.ini definition file.  It also allows a tracker to
134         * be created for situation where progress in the event requires tracking each
135         * player event.
136         * @author Bo Zimmerman
137         *
138         */
139        public interface Achievement
140        {
141                /**
142                 * Returns whether this is a player or account achievement.
143                 * @see com.planet_ink.coffee_mud.Common.interfaces.AccountStats.Agent
144                 * @return the agent type
145                 */
146                public AccountStats.Agent getAgent();
147
148                /**
149                 * Returns the event that defines the general type of this achievement.
150                 * @see Event
151                 * @return the event that defines the general type of this achievement.
152                 */
153                public Event getEvent();
154
155                /**
156                 * Gets the tattoo that is added to players or accounts to designate
157                 * that this achievement has been completed.  It is also the unique
158                 * key for all achievements.
159                 * @return the tattoo that is added to players or accounts
160                 */
161                public String getTattoo();
162
163                /**
164                 * Creates a new tracker object with the given progress count as
165                 * the default/starting value.
166                 * @see Achievement#getTargetCount()
167                 * @param oldCount the initial value for progress, if applicable
168                 * @return a new tracker object with the given progress count
169                 */
170                public Tracker getTracker(int oldCount);
171
172                /**
173                 * Parses the parameters defined by the event type of this achievement 
174                 * to produce the unique achievement that this is. The parameters are
175                 * in key=value pairs, space delimited.
176                 * @see Event#getParameters()
177                 * @see Achievement#getRawParmVal(String)
178                 * @param parms the parameter values to parse
179                 * @return "" if all went well, or an error message
180                 */
181                public String parseParms(String parms);
182
183                /**
184                 * Returns the friendly display name of this achievement.
185                 * @return the friendly display name of this achievement.
186                 */
187                public String getDisplayStr();
188
189                /**
190                 * Returns the list of coded awards given to players who
191                 * complete this achievement.  The format is a string with
192                 * a number followed by a string of the type of thing
193                 * to award, such as QP, XP, or a currency.
194                 * @see AchievementLibrary.Award
195                 * @return the list of decoded awards given to players
196                 */
197                public Award[] getRewards();
198
199                /**
200                 * For achievements that require tracking progress, 
201                 * this returns the target count that the tracker
202                 * must report.
203                 * @return the target count that the tracker
204                 */
205                public int getTargetCount();
206
207                /**
208                 * Returns true if this achievement is completed
209                 * when the tracker count is &gt; than the target count,
210                 * and false if the opposite is true.
211                 * @return true for a &gt; completion
212                 */
213                public boolean isTargetFloor();
214
215                /**
216                 * Returns true if this achievement requires progress
217                 * tracking, which means it must be saved with the player
218                 * and/or account record to keep track of.
219                 * @return true if this achievement requires progress
220                 */
221                public boolean isSavableTracker();
222
223                /**
224                 * Allows access to the parameters passed into this
225                 * achievement to create it.  
226                 * @see Event#getParameters()
227                 * @see Achievement#parseParms(String)
228                 * @param str the name of the parameter to query
229                 * @return the value of the parameter 
230                 */
231                public String getRawParmVal(String str);
232        }

403        /**
404         * A tracker object assigned to a particular player or account
405         * for a particular achievement, allowing the achievement to 
406         * track progress if it needs to, or just providing a way
407         * to quickly query completion otherwise.
408         * @author Bo Zimmerman
409         *
410         */
411        public interface Tracker extends Cloneable
412        {
413                /**
414                 * The achievement to which this tracker belongs.
415                 * @return achievement to which this tracker belongs.
416                 */
417                public Achievement getAchievement();
418
419                /**
420                 * Returns true if the given mob has completed this
421                 * achievement, even if the tattoo has not yet been
422                 * assigned.
423                 * @param mob the player being checked
424                 * @return true if completion has happened, false otherwise
425                 */
426                public boolean isAchieved(MOB mob);
427
428                /**
429                 * For events which require tracked progress, this method is
430                 * called to give this tracker a potential bump, after testing
431                 * the given mob and the given arguments to see if the 
432                 * achievement deserves a bump in progress.
433                 * @param mob the player who did something trackable
434                 * @param bumpNum the amount to bump the progress by
435                 * @param parms optional arguments unique to the Event
436                 * @return true if a bump occurred, false otherwise
437                 */
438                public boolean testBump(MOB mob, int bumpNum, Object... parms);
439
440                /**
441                 * Returns the count/score to show for the given mob.  If the
442                 * achievement of this tracker is Savable, then the mob may be
443                 * null, since the count would then be internally stored.
444                 * @param mob the mob to get a count for -- required ONLY for unsavable
445                 * @return the score for this achievement and this mob
446                 */
447                public int getCount(MOB mob);
448                
449                /**
450                 * Returns a copy of this tracker, unattached to the
451                 * tracker it is a copy of.
452                 * @return a copy of this tracker
453                 */
454                public Tracker copyOf();
455        }

464         * @param addIfPossible true if, on success, the new achievment is added, false otherwise
465         * @return the error message, or "" for success
466         */
