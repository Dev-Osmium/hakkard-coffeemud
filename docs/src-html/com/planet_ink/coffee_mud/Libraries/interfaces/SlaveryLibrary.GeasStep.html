034public interface SlaveryLibrary extends CMLibrary
035{
036        public enum Step
037        {
038                EVAL,
039                INT1,
040                INT2,
041                INT3,
042                INT4,
043                INT5,
044                ALLDONE;//==-99
045                
046                public Step nextStep()
047                {
048                        switch(this)
049                        {
050                        case EVAL:
051                                return INT1;
052                        case INT1:
053                                return INT2;
054                        case INT2:
055                                return INT3;
056                        case INT3:
057                                return INT4;
058                        case INT4:
059                                return INT5;
060                        case INT5:
061                                return ALLDONE;
062                        default:
063                                return this;
064                        }
065                }
066        }
067
068        public List<Map<String,String>> findMatch(MOB mob, List<String> prereq);
069        public String cleanWord(String s);
070        public GeasSteps processRequest(MOB you, MOB me, String req);
071
072        public static interface GeasStep
073        {
074                public boolean botherIfAble(String msgOrQ);
075                
076                public boolean sayResponse(MOB speaker, MOB target, String response);
077                
078                public String step();
079                
080                public void setStep(Step step);
081                
082                public MOB getBotheredMob();
083        }
084        
085        public static interface GeasSteps extends List<GeasStep>
086        {
087                public void step();
088                
089                public void move(int moveCode);
090                
091                public boolean sayResponse(MOB speaker, MOB target, String response);
092                
093                public MOB stepperM();
094                
095                public Set<MOB> getBotheredMobs();
096                
097                public boolean isDone();
098        }
099}
