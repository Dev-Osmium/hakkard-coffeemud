035public class AutoTitles extends StdLibrary implements AutoTitlesLibrary
036{
037        @Override
038        public String ID()
039        {
040                return "AutoTitles";
041        }
042
043        private TriadSVector<String,String,MaskingLibrary.CompiledZMask> autoTitles=null;
044
045        private static final String titleFilename = "titles.ini";
046        
047        private String getTitleFilename()
048        {
049                CMFile F = new CMFile(Resources.makeFileResourceName(titleFilename),null);
050                if(F.exists() && (F.canRead()))
051                        return titleFilename;
052                final String oldFilename = titleFilename.substring(0,titleFilename.length()-4)+".txt"; 
053                F = new CMFile(Resources.makeFileResourceName(oldFilename),null);
054                if(F.exists() && (F.canRead()))
055                        return oldFilename;
056                return titleFilename;
057        }
058
059        @Override
060        public String evaluateAutoTitle(String row, boolean addIfPossible)
061        {
062                if(row.trim().startsWith("#")||row.trim().startsWith(";")||(row.trim().length()==0))
063                        return null;
064                int x=row.indexOf('=');
065                while((x>=1)&&(row.charAt(x-1)=='\\'))
066                        x=row.indexOf('=',x+1);
067                if(x<0)
068                        return "Error: Invalid line! Not comment, whitespace, and does not contain an = sign!";
069                final String title=row.substring(0,x).trim();
070                final String mask=row.substring(x+1).trim();
071
072                if(title.length()==0)
073                        return "Error: Blank title: "+title+"="+mask+"!";
074                if(mask.length()==0)
075                        return "Error: Blank mask: "+title+"="+mask+"!";
076                if(addIfPossible)
077                {
078                        if(autoTitles==null)
079                                reloadAutoTitles();
080                        for(final Triad<String,String,MaskingLibrary.CompiledZMask> triad : autoTitles)
081                        {
082                                if(triad.first.equalsIgnoreCase(title))
083                                        return "Error: Duplicate title: "+title+"="+mask+"!";
084                        }
085                        autoTitles.add(new Triad<String,String,MaskingLibrary.CompiledZMask>(title,mask,CMLib.masking().maskCompile(mask)));
086                }
087                return null;
088        }
089
090        @Override
091        public boolean isExistingAutoTitle(String title)
092        {
093                if(autoTitles==null)
094                        reloadAutoTitles();
095                title=title.trim();
096                for(final Triad<String,String,MaskingLibrary.CompiledZMask> triad : autoTitles)
097                {
098                        if(triad.first.equalsIgnoreCase(title))
099                                return true;
100                }
101                return false;
102        }
103
104        @Override
105        public Enumeration<String> autoTitles()
106        {
107                if(autoTitles==null)
108                        reloadAutoTitles();
109                return autoTitles.firstElements();
110        }
111
112        @Override
113        public String getAutoTitleMask(String title)
114        {
115                if(autoTitles==null)
116                        reloadAutoTitles();
117                for(final Triad<String,String,MaskingLibrary.CompiledZMask> triad : autoTitles)
118                {
119                        if(triad.first.equalsIgnoreCase(title))
120                                return triad.second;
121                }
122                return "";
123        }
124
125        @Override
126        public boolean evaluateAutoTitles(MOB mob)
127        {
128                if(mob==null)
129                        return false;
130                final PlayerStats P=mob.playerStats();
131                if(P==null)
132                        return false;
133                if(autoTitles==null)
134                        reloadAutoTitles();
135                String title=null;
136                MaskingLibrary.CompiledZMask mask=null;
137                int pdex=0;
138                final List<String> ptV=P.getTitles();
139                boolean somethingDone=false;
140                synchronized(ptV)
141                {
142                        for(final Triad<String,String,MaskingLibrary.CompiledZMask> triad : autoTitles)
143                        {
144                                mask=triad.third;
145                                title=triad.first;
146                                pdex=ptV.indexOf(title);
147                                if(pdex<0)
148                                {
149                                        final String fixedTitle = CMStrings.removeColors(title).replace('\'', '`');
150                                        for(int p=ptV.size()-1;p>=0;p--)
151                                        {
152                                                try
153                                                {
154                                                        final String tit=CMStrings.removeColors(ptV.get(p)).replace('\'', '`');
155                                                        if(tit.equalsIgnoreCase(fixedTitle))
156                                                        {
157                                                                pdex=p;
158                                                                break;
159                                                        }
160                                                }
161                                                catch(final java.lang.IndexOutOfBoundsException ioe)
162                                                {
163                                                }
164                                        }
165                                }
166
167                                if(CMLib.masking().maskCheck(mask,mob,true))
168                                {
169                                        if(pdex<0)
170                                        {
171                                                if(ptV.size()>0)
172                                                        ptV.add(0,title);
173                                                else
174                                                        ptV.add(title);
175                                                somethingDone=true;
176                                        }
177                                }
178                                else
179                                if(pdex>=0)
180                                {
181                                        somethingDone=true;
182                                        ptV.remove(pdex);
183                                }
184                        }
185                }
186                return somethingDone;
187        }

189        protected void dispossesTitle(String title)
190        {
191                final List<String> list=CMLib.database().getUserList();
192                final String fixedTitle = CMStrings.removeColors(title).replace('\'', '`');
193                for(final String playerName : list)
194                {
195                        final MOB M=CMLib.players().getLoadPlayer(playerName);
196                        if(M.playerStats()!=null)
197                        {
198                                final List<String> ptV=M.playerStats().getTitles();
199                                synchronized(ptV)
200                                {
201                                        int pdex=ptV.indexOf(title);
202                                        if(pdex<0)
203                                        {
204                                                for(int p=ptV.size()-1;p>=0;p--)
205                                                {
206                                                        try
207                                                        {
208                                                                final String tit=CMStrings.removeColors(ptV.get(p)).replace('\'', '`');
209                                                                if(tit.equalsIgnoreCase(fixedTitle))
210                                                                {
211                                                                        pdex=p;
212                                                                        break;
213                                                                }
214                                                        }
215                                                        catch(final java.lang.IndexOutOfBoundsException ioe)
216                                                        {
217                                                        }
218                                                }
219                                        }
220                                        if(pdex>=0)
221                                        {
222                                                ptV.remove(pdex);
223                                                if(!CMLib.flags().isInTheGame(M,true))
224                                                        CMLib.database().DBUpdatePlayerPlayerStats(M);
225                                        }
226                                }
227                        }
228                }
229        }

231        @Override
232        public void appendAutoTitle(String text)
233        {
234                Resources.removeResource(this.getTitleFilename());
235                final CMFile F=new CMFile(Resources.makeFileResourceName(titleFilename),null,CMFile.FLAG_LOGERRORS);
236                F.saveText(text,true);
237                reloadAutoTitles();
238        }
239
240        @Override
241        public String deleteTitleAndResave(String title)
242        {
243                dispossesTitle(title);
244                final CMFile F=new CMFile(Resources.makeFileResourceName(titleFilename),null,CMFile.FLAG_LOGERRORS);
245                if(F.exists())
246                {
247                        final boolean removed=Resources.findRemoveProperty(F, title);
248                        if(removed)
249                        {
250                                Resources.removeResource(titleFilename);
251                                reloadAutoTitles();
252                                return null;
253                        }
254                        return "Unable to delete title!";
255                }
256                return "Unable to open "+titleFilename+"!";
257        }
258
259        @Override
260        public void reloadAutoTitles()
261        {
262                autoTitles=new TriadSVector<String,String,MaskingLibrary.CompiledZMask>();
263                final List<String> V=Resources.getFileLineVector(Resources.getFileResource(this.getTitleFilename(),true));
264                String WKID=null;
265                for(int v=0;v<V.size();v++)
266                {
267                        final String row=V.get(v);
268                        WKID=evaluateAutoTitle(row,true);
269                        if(WKID==null)
270                                continue;
271                        if(WKID.startsWith("Error: "))
272                                Log.errOut("CharCreation",WKID);
273                }
274                for(final Enumeration<MOB> e=CMLib.players().players();e.hasMoreElements();)
275                {
276                        final MOB M=e.nextElement();
277                        if(M.playerStats()!=null)
278                        {
279                                if((evaluateAutoTitles(M))&&(!CMLib.flags().isInTheGame(M,true)))
280                                        CMLib.database().DBUpdatePlayerPlayerStats(M);
281                        }
282                }
283        }

285        @Override
286        public String getAutoTitleInstructions()
287        {
288                final StringBuffer buf=new CMFile(Resources.makeFileResourceName(this.getTitleFilename()),null,CMFile.FLAG_LOGERRORS).text();
289                final StringBuffer inst=new StringBuffer("");
290                List<String> V=new Vector<String>();
291                if(buf!=null)
292                        V=Resources.getFileLineVector(buf);
293                for(int v=0;v<V.size();v++)
294                {
295                        if(V.get(v).startsWith("#"))
296                                inst.append(V.get(v).substring(1)+"\n\r");
297                        else
298                        if(V.get(v).length()>0)
299                                break;
300                }
301                return inst.toString();
302        }
