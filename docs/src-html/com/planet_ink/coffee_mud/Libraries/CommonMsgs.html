057        @Override
058        public boolean handleUnknownCommand(final MOB mob, final List<String> command)
059        {
060                if(mob==null)
061                        return false;
062                final Room R=mob.location();
063                String msgStr;
064                if(R==null)
065                {
066                        mob.tell(L("Huh?"));
067                        return false;
068                }
069                if(command.size()>0)
070                {
071                        final String word=CMLib.english().getAnEvokeWord(mob,command.get(0));
072                        if(word!=null)
073                                msgStr=L("You don't know how to @x1 that.",word.toLowerCase());
074                        else
075                                msgStr=L("Huh?");
076                }
077                else
078                        msgStr=L("Huh?");
079                final CMMsg msg=CMClass.getMsg(mob,null,null,CMMsg.MSG_HUH,msgStr,CMParms.combineQuoted(command,0),null);
080                if(!R.okMessage(mob,msg))
081                        return false;
082                R.send(mob,msg);
083                return true;
084        }

197        {
198                MsgMonitor M;
199                for (final WeakReference<MsgMonitor> W : globalMonitors)
200                {
201                        M=W.get();

209        @Override
210        public void addGlobalMonitor(final MsgMonitor M)
211        {
212                if(M==null)
213                        return;
214                for (final WeakReference<MsgMonitor> W : globalMonitors)
215                {
216                        if(W.get()==M)
217                                return;
218                }
219                globalMonitors.add(new WeakReference<MsgMonitor>(M));
220        }

227                for (final WeakReference<MsgMonitor> W : globalMonitors)

232        }

586        public void handleBeingMobSniffed(final CMMsg msg)
587        {
588                if(!(msg.target() instanceof MOB))
589                        return;
590                final MOB sniffingmob=msg.source();
591                final MOB sniffedmob=(MOB)msg.target();
592                if((sniffedmob.playerStats()!=null)
593                &&(sniffedmob.soulMate()==null)
594                &&(sniffedmob.playerStats().getHygiene()>=PlayerStats.HYGIENE_DELIMIT))
595                {

772                }
