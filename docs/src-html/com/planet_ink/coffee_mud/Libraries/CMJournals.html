239        @Override
240        public int loadCommandJournals(String list)
241        {
242                clearCommandJournals();
243                while(list.length()>0)
244                {
245                        int x=list.indexOf(',');
246
247                        String item=null;
248                        if(x<0)
249                        {
250                                item=list.trim();
251                                list="";
252                        }
253                        else
254                        {
255                                item=list.substring(0,x).trim();
256                                list=list.substring(x+1);
257                        }
258                        x=item.indexOf(' ');
259                        final Hashtable<CommandJournalFlags,String> flags=new Hashtable<CommandJournalFlags,String>();
260                        String mask="";
261                        if(x>0)
262                        {
263                                mask=item.substring(x+1).trim();
264                                for(int pf=0;pf<CommandJournalFlags.values().length;pf++)
265                                {
266                                        final String flag = CommandJournalFlags.values()[pf].toString();
267                                        final int keyx=mask.toUpperCase().indexOf(flag);
268                                        if(keyx>=0)
269                                        {
270                                                int keyy=mask.indexOf(' ',keyx+1);
271                                                if(keyy<0)
272                                                        keyy=mask.length();
273                                                if((keyx==0)||(Character.isWhitespace(mask.charAt(keyx-1))))
274                                                {
275                                                        String parm=mask.substring(keyx+flag.length(),keyy).trim();
276                                                        if((parm.length()==0)||(parm.startsWith("=")))
277                                                        {
278                                                                if(parm.startsWith("="))
279                                                                        parm=parm.substring(1);
280                                                                flags.put(CommandJournalFlags.values()[pf],parm);
281                                                                mask=mask.substring(0,keyx).trim()+" "+mask.substring(keyy).trim();
282                                                        }
283                                                }
284                                        }
285                                }
286                                item=item.substring(0,x);
287                        }
288                        final String name=item.toUpperCase().trim();
289                        CMSecurity.registerJournal(name);
290                        String flagVal = flags.get(CommandJournalFlags.ASSIGN);
291                        if(flagVal == null)
292                                flags.put(CommandJournalFlags.ASSIGN, "ALL");
293                        else
294                        {
295                                List<String> flagValL=CMParms.parseAny(flagVal.toUpperCase().trim(), ':', true);
296                                if(!flagValL.contains("ALL"))
297                                        flagValL.add("ALL");
298                                StringBuilder newFlags=new StringBuilder("");
299                                for(String flag : flagValL)
300                                        newFlags.append(flag).append(':');
301                                flags.put(CommandJournalFlags.ASSIGN, newFlags.toString());
302                        }
303                        final String journalAdminMask = mask;
304                        commandJournals.put(name,new CommandJournal()
305                        {
306                                @Override
307                                public String NAME()
308                                {
309                                        return name;
310                                }
311
312                                @Override
313                                public String mask()
314                                {
315                                        return journalAdminMask;
316                                }
317
318                                @Override
319                                public String JOURNAL_NAME()
320                                {
321                                        return "SYSTEM_" + NAME().toUpperCase().trim() + "S";
322                                }
323

325                                public String getFlag(CommandJournalFlags flag)
326                                {
327                                        return flags.get(flag);
328                                }
329
330                                @Override
331                                public String getScriptFilename()
332                                {
333                                        return flags.get(CommandJournalFlags.SCRIPT);
334                                }
335                        });
336                }
337                return commandJournals.size();
338        }

340        @Override
341        public boolean canReadMessage(JournalEntry entry, String srchMatch, MOB readerM, boolean ignorePrivileges)
342        {
343                if(entry==null)
344                        return false;
345                final String to=entry.to();
346                if((srchMatch!=null)
347                &&(srchMatch.length()>0)
348                &&((to.toLowerCase().indexOf(srchMatch)<0)
349                &&(entry.from().toLowerCase().indexOf(srchMatch)<0)
350                &&(entry.subj().toLowerCase().indexOf(srchMatch)<0)
351                &&(entry.msg().toLowerCase().indexOf(srchMatch)<0)))
352                        return false;
353                boolean priviledged=false;
354                if(readerM!=null)
355                        priviledged=CMSecurity.isAllowedAnywhere(readerM,CMSecurity.SecFlag.JOURNALS)&&(!ignorePrivileges);
356                if(to.equalsIgnoreCase("all")
357                ||((readerM!=null)
358                        &&(priviledged
359                                ||to.equalsIgnoreCase(readerM.Name())
360                                ||(to.toUpperCase().trim().startsWith("MASK=")&&(CMLib.masking().maskCheck(to.trim().substring(5),readerM,true))))))
361                        return true;
362                return false;
363        }

386        @Override
387        public void registerClanForum(Clan clan, String allClanForumDefs)
388        {
389                if(clan==null)
390                        return;
391                this.clanForums.remove(clan.clanID());
392                if(allClanForumDefs==null)
393                        return;
394                final List<String> set=CMParms.parseCommas(allClanForumDefs,true);
395                final StringBuilder myForumList=new StringBuilder("");
396                for(String s : set)
397                {
398                        s=s.trim();
399                        if(s.startsWith("["))
400                        {
401                                final int x=s.indexOf(']');
402                                final String cat=s.substring(1,x).trim();
403                                if(clan.getGovernment().getCategory().equalsIgnoreCase(cat))
404                                {
405                                        s=s.substring(x+1).trim();
406                                        s=CMStrings.replaceAll(s, "<CLANTYPE>", clan.getGovernmentName());
407                                        s=CMStrings.replaceAll(s, "<CLANNAME>", clan.getName());
408                                        s=CMStrings.replaceAll(s, ",", ".");
409                                        if(myForumList.length()>0)
410                                                myForumList.append(',');
411                                        myForumList.append(s);
412                                }
413                        }
414                }
415                final List<ForumJournal> journals = parseForumJournals(myForumList.toString());
416                if((journals!=null)&&(journals.size()>0))
417                        this.clanForums.put(clan.clanID(), journals);
418        }

420        public List<ForumJournal> parseForumJournals(String list)
421        {
422                final List<ForumJournal> journals = new Vector<ForumJournal>(1);
423                while(list.length()>0)
424                {
425                        int x=list.indexOf(',');
426                        String item;
427                        if(x<0)
428                        {
429                                item=list.trim();
430                                list="";
431                        }
432                        else
433                        {
434                                item=list.substring(0,x).trim();
435                                list=list.substring(x+1);
436                        }
437                        final Hashtable<ForumJournalFlags,String> flags=new Hashtable<ForumJournalFlags,String>();
438                        x=item.indexOf('=');
439                        if(x > 0)
440                        {
441                                int y=x;
442                                while((y>0)&&(!Character.isWhitespace(item.charAt(y))))
443                                        y--;
444                                final String rest = item.toUpperCase().substring(y+1).trim();
445                                item=item.substring(0,y);
446                                final Vector<Integer> flagDexes = new Vector<Integer>();
447                                x=rest.indexOf('=');
448                                while(x > 0)
449                                {
450                                        y=x;
451                                        while((y>0)&&(!Character.isWhitespace(rest.charAt(y))))
452                                                y--;
453                                        if(y>0)
454                                        {
455                                                try
456                                                {
457                                                        ForumJournalFlags.valueOf(rest.substring(y,x).toUpperCase().trim());
458                                                        flagDexes.addElement(Integer.valueOf(y));
459                                                }
460                                                catch(final Exception e)
461                                                {
462                                                }
463                                        }
464                                        x=rest.indexOf('=',x+1);
465                                }
466                                flagDexes.addElement(Integer.valueOf(rest.length()));
467                                int lastStart=0;
468                                for(final Integer flagDex : flagDexes)
469                                {
470                                        final String piece = rest.substring(lastStart,flagDex.intValue());
471                                        lastStart=flagDex.intValue();
472                                        x=piece.indexOf('=');
473                                        try
474                                        {
475                                                final ForumJournalFlags flagVar = ForumJournalFlags.valueOf(piece.substring(0,x).toUpperCase().trim());
476                                                final String flagVal = piece.substring(x+1);
477                                                flags.put(flagVar, flagVal);
478                                        }
479                                        catch(final Exception e)
480                                        {
481                                        }
482                                }
483                        }
484                        String mask;
485                        mask=flags.remove(ForumJournalFlags.READ);
486                        final String readMask=(mask != null)?mask.trim():"";
487                        mask=flags.remove(ForumJournalFlags.POST);
488                        final String postMask=(mask != null)?mask.trim():"";
489                        mask=flags.remove(ForumJournalFlags.REPLY);
490                        final String replyMask=(mask != null)?mask.trim():"";
491                        mask=flags.remove(ForumJournalFlags.ADMIN);
492                        final String adminMask=(mask != null)?mask.trim():"";
493                        final String name = item.trim();
494                        journals.add(new ForumJournal()
495                        {
496                                @Override
497                                public String NAME()
498                                {
499                                        return name;
500                                }
501
502                                @Override
503                                public String readMask()
504                                {
505                                        return readMask;
506                                }
507
508                                @Override
509                                public String postMask()
510                                {
511                                        return postMask;
512                                }
513
514                                @Override
515                                public String replyMask()
516                                {
517                                        return replyMask;
518                                }
519
520                                @Override
521                                public String adminMask()
522                                {
523                                        return adminMask;
524                                }

528                                {
529                                        return flags.get(flag);

534                                {
535                                        if(mask.length()>0)
536                                        {
537                                                if(M==null)
538                                                        return false;
539                                                return CMLib.masking().maskCheck(mask, M, true);
540                                        }
541                                        return true;

544                                @Override
545                                public boolean authorizationCheck(MOB M, ForumJournalFlags fl)
546                                {
547                                        if(!maskCheck(M,readMask))
548                                                return false;
549                                        if(fl==ForumJournalFlags.READ)
550                                                return true;
551                                        if(fl==ForumJournalFlags.POST)
552                                                return maskCheck(M,postMask);
553                                        else
554                                        if(fl==ForumJournalFlags.REPLY)
555                                                return maskCheck(M,replyMask);
556                                        else
557                                        if(fl==ForumJournalFlags.ADMIN)
558                                                return maskCheck(M,adminMask);
559                                        return false;
560                                }
561                        });
562                }
563                return journals;
564        }

595        @Override
596        public String getScriptValue(MOB mob, String journal, String oldValue)
597        {
598                final CommandJournal CMJ=getCommandJournal(journal);
599                if(CMJ==null)
600                        return oldValue;
601                final String scriptFilename=CMJ.getScriptFilename();
602                if((scriptFilename==null)||(scriptFilename.trim().length()==0))
603                        return oldValue;
604                final ScriptingEngine S=(ScriptingEngine)CMClass.getCommon("DefaultScriptingEngine");
605                S.setSavable(false);
606                S.setVarScope("*");
607                S.setScript("LOAD="+scriptFilename);
608                S.setVar(mob.Name(),"VALUE", oldValue);
609                final CMMsg msg2=CMClass.getMsg(mob,mob,null,CMMsg.MSG_OK_VISUAL,null,null,L("COMMANDJOURNAL_@x1",CMJ.NAME()));
610                S.executeMsg(mob, msg2);
611                S.dequeResponses();
612                S.tick(mob,Tickable.TICKID_MOB);
613                final String response=S.getVar("*","VALUE");
614                if(response!=null)
615                        return response;
616                return oldValue;
617        }

637        public void expirationJournalSweep()
638        {
639                setThreadStatus(serviceClient,"expiration journal sweeping");
640                try
641                {
642                        for(final Enumeration<CommandJournal> e=commandJournals();e.hasMoreElements();)
643                        {
644                                final CommandJournal CMJ=e.nextElement();
645                                final String num=CMJ.getFlag(CommandJournalFlags.EXPIRE);
646                                if((num!=null)&&(CMath.isNumber(num))&&(CMath.s_double(num)>0.0))
647                                {
648                                        setThreadStatus(serviceClient,"updating journal "+CMJ.NAME());
649                                        final long expirationDate = System.currentTimeMillis() - Math.round(CMath.mul(TimeManager.MILI_DAY,CMath.s_double(num)));
650                                        final List<JournalEntry> items=CMLib.database().DBReadJournalMsgsOlderThan(CMJ.JOURNAL_NAME(),null,expirationDate);
651                                        for(int i=items.size()-1;i>=0;i--)
652                                        {
653                                                final JournalEntry entry=items.get(i);
654                                                final String from=entry.from();
655                                                final String message=entry.msg();
656                                                Log.sysOut(Thread.currentThread().getName(),"Expired "+CMJ.NAME()+" from "+from+": "+message);
657                                                CMLib.database().DBDeleteJournal(CMJ.JOURNAL_NAME(),entry.key());
658                                        }
659                                        setThreadStatus(serviceClient,"command journal sweeping");
660                                }
661                        }
662                }
663                catch(final NoSuchElementException nse)
664                {
665                }
666                try
667                {
668                        final List<Pair<String,String>> deleteThese = new LinkedList<Pair<String,String>>();
669                        for(final Enumeration<ForumJournal> e=forumJournals();e.hasMoreElements();)
670                        {
671                                final ForumJournal FMJ=e.nextElement();
672                                final String num=FMJ.getFlag(ForumJournalFlags.EXPIRE);
673                                if((num!=null)&&(CMath.isNumber(num))&&(CMath.s_double(num)>0.0))
674                                {
675                                        setThreadStatus(serviceClient,"updating journal "+FMJ.NAME());
676                                        final long expirationDate = System.currentTimeMillis() - Math.round(CMath.mul(TimeManager.MILI_DAY,CMath.s_double(num)));
677                                        final List<JournalEntry> items=CMLib.database().DBReadJournalMsgsOlderThan(FMJ.NAME(),null,expirationDate);
678                                        for(int i=items.size()-1;i>=0;i--)
679                                        {
680                                                final JournalEntry entry=items.get(i);
681                                                if(!CMath.bset(entry.attributes(), JournalEntry.ATTRIBUTE_PROTECTED))
682                                                {
683                                                        final String from=entry.from();
684                                                        final String message=entry.msg();
685                                                        Log.debugOut(Thread.currentThread().getName(),"Expired "+FMJ.NAME()+" from "+from+": "+message);
686                                                        deleteThese.add(new Pair<String,String>(FMJ.NAME(),entry.key()));
687                                                }
688                                        }
689                                        setThreadStatus(serviceClient,"forum journal sweeping");
690                                }
691                        }
692                        for(final Pair<String,String> p : deleteThese)
693                                CMLib.database().DBDeleteJournal(p.first,p.second);
694                }
695                catch(final NoSuchElementException nse)
696                {
697                }
698        }
