116        @Override
117        public boolean isACity(final Area A)
118        {
119                int other=0;
120                int streets=0;
121                int buildings=0;
122                Room R=null;
123                for(final Enumeration<Room> e=A.getCompleteMap();e.hasMoreElements();)
124                {
125                        R=e.nextElement();
126                        if((R==null)||(R.roomID()==null)||(R.roomID().length()==0))
127                                continue;
128                        if(R.domainType()==Room.DOMAIN_OUTDOORS_CITY)
129                                streets++;
130                        else
131                        if((R.domainType()==Room.DOMAIN_INDOORS_METAL)
132                        ||(R.domainType()==Room.DOMAIN_INDOORS_STONE)
133                        ||(R.domainType()==Room.DOMAIN_INDOORS_WOOD))
134                                buildings++;
135                        else
136                                other++;
137                }
138                if((streets<(other/2))||((streets+buildings)<other))
139                        return false;
140                return true;
141        }

143        @Override
144        public List<LandTitle> getAllUniqueLandTitles(final Enumeration<Room> e, final String owner, final boolean includeRentals)
145        {
146                final Vector<LandTitle> V=new Vector<LandTitle>();
147                final HashSet<Room> roomsDone=new HashSet<Room>();
148                final Set<String> titlesDone = new HashSet<String>();
149                Room R=null;
150                for(;e.hasMoreElements();)
151                {
152                        R=e.nextElement();
153                        LandTitle T=getLandTitle(R);
154                        if((T!=null)
155                        &&(!V.contains(T))
156                        &&(!titlesDone.contains(T.getTitleID()))
157                        &&(includeRentals||(!T.rentalProperty()))
158                        &&((owner==null)
159                                ||(owner.length()==0)
160                                ||(owner.equals("*")&&(T.getOwnerName().length()>0))
161                                ||(T.getOwnerName().equals(owner))))
162                        {
163                                titlesDone.add(T.getTitleID());
164                                int price = T.getPrice();
165                                final List<Room> V2=T.getAllTitledRooms();
166                                for(int v=0;v<V2.size();v++)
167                                {
168                                        final Room R2=V2.get(v);
169                                        if(R2 != null)
170                                        {
171                                                if(!roomsDone.contains(R2))
172                                                {
173                                                        roomsDone.add(R2);
174                                                        final LandTitle T2=getLandTitle(R2);
175                                                        if((T2 != null)
176                                                        &&(T2.getPrice() > price))
177                                                        {
178                                                                T=T2;
179                                                                price=T2.getPrice();
180                                                        }
181                                                }
182                                        }
183                                }
184                                V.addElement(T); // should be the best taxable title
185                        }
186                }
187                return V;
188        }

260        {
261                final Room dR=room.getRoomInDir(Directions.DOWN);
262                if((dR!=null)
263                &&(dR.domainType() != Room.DOMAIN_INDOORS_CAVE)
264                &&((dR.getAtmosphere()&RawMaterial.MATERIAL_MASK)!=RawMaterial.MATERIAL_ROCK))
265                {
266                        if(isHomePeerRoom(dR))
267                                return true;
268                        final Set<Room> peerRooms=getHomePeersOnThisFloor(room,new HashSet<Room>());
269                        for(final Room R : peerRooms)

273                                &&(pdR.domainType() != Room.DOMAIN_INDOORS_CAVE)
274                                &&((pdR.getAtmosphere()&RawMaterial.MATERIAL_ROCK)==0)
275                                &&(isHomePeerRoom(pdR)))
276                                        return true;
277                        }
278                }
279                return false;

305        public boolean isRoomSimilarlyTitled(final LandTitle title, final Room R)
306        {
307                final LandTitle ptitle = ifLandTitle(R);
308                if(ptitle ==null)
309                        return false;
310                if(ptitle.getOwnerName().length()==0)
311                {
312                        for(int d=0;d<Directions.NUM_DIRECTIONS();d++)
313                        {
314                                final Room sideRoom=R.getRoomInDir(d);
315                                final LandTitle psTitle=ifLandTitle(sideRoom);
316                                if(psTitle.getOwnerName().equals(title.getOwnerName()))
317                                        return true;
318                        }
319                        return false;
320                }
321                else
322                        return ptitle.getOwnerName().equals(title.getOwnerName());

325        @Override
326        public Set<Room> getHomePeersOnThisFloor(final Room room, final Set<Room> doneRooms)
327        {
328                for(int d=0;d<Directions.NUM_DIRECTIONS();d++)
329                {
330                        if((d!=Directions.UP)&&(d!=Directions.DOWN)&&(d!=Directions.GATE))
331                        {
332                                final Room sideRoom=room.getRoomInDir(d);
333                                if(isHomePeerRoom(sideRoom)  &&(!doneRooms.contains(sideRoom)))
334                                {
335                                        doneRooms.add(sideRoom);
336                                        doneRooms.addAll(getHomePeersOnThisFloor(sideRoom, doneRooms));
337                                }
338                        }
339                }
340                return doneRooms;
341        }

429        @Override
430        public boolean doesAnyoneHavePrivilegesHere(final MOB mob, final String overrideID, final Room R)
431        {
432                if((mob==null)||(R==null))
433                        return false;
434                if((doesHavePriviledgesHere(mob,R))||((overrideID.length()>0)&&(mob.Name().equals(overrideID))))
435                        return true;
436                if(overrideID.length()>0)
437                {
438                        final Pair<Clan,Integer> clanPair=mob.getClanRole(overrideID);
439                        if((clanPair!=null)&&(clanPair.first.getAuthority(clanPair.second.intValue(), Clan.Function.HOME_PRIVS)!=Clan.Authority.CAN_NOT_DO))
440                                return true;
441                }
442                for(int i=0;i<R.numInhabitants();i++)
443                {
444                        final MOB M=R.fetchInhabitant(i);
445                        if(doesHavePriviledgesHere(M,R))
446                                return true;
447                        if(overrideID.length()>0)
448                        {
449                                if(M.Name().equals(overrideID))
450                                        return true;
451                                final Pair<Clan,Integer> clanPair=M.getClanRole(overrideID);
452                                if((clanPair!=null)&&(clanPair.first.getAuthority(clanPair.second.intValue(), Clan.Function.HOME_PRIVS)!=Clan.Authority.CAN_NOT_DO))
453                                        return true;
454                        }
455                }
456                return false;
457        }

679        @Override
680        public void colorRoomForSale(Room R, final LandTitle title, final boolean reset)
681        {
682                synchronized(("SYNC"+R.roomID()).intern())
683                {
684                        R=CMLib.map().getRoom(R);
685                        final String theStr=CMLib.lang().L(title.rentalProperty()?RENTSTR:SALESTR);
686                        final String otherStr=CMLib.lang().L(title.rentalProperty()?SALESTR:RENTSTR);
687                        int x=R.description().indexOf(otherStr);
688                        while(x>=0)
689                        {
690                                R.setDescription(R.description().substring(0,x));
691                                CMLib.database().DBUpdateRoom(R);
692                                x=R.description().indexOf(otherStr);

756        @Override
757        public boolean robberyCheck(final PrivateProperty record, final CMMsg msg, final boolean quiet)
758        {
759                if(((msg.targetMinor()==CMMsg.TYP_GET)&&(!msg.isTarget(CMMsg.MASK_INTERMSG)))
760                ||(msg.targetMinor()==CMMsg.TYP_PUSH)
761                ||(msg.targetMinor()==CMMsg.TYP_PULL))
762                {
763                        final Room R=msg.source().location();
764                        if((msg.target() instanceof Item)
765                        &&(((Item)msg.target()).owner() ==R)
766                        &&((!(msg.tool() instanceof Item))||(msg.source().isMine(msg.tool())))
767                        &&(!msg.sourceMajor(CMMsg.MASK_ALWAYS))
768                        &&(record.getOwnerName().length()>0)
769                        &&(R!=null)
770                        &&(msg.othersMessage()!=null)
771                        &&(msg.othersMessage().length()>0)
772                        &&(!shopkeeperMobPresent(R))
773                        &&(!doesHavePriviledgesHere(msg.source(),R)))
774                        {
775                                if((!canAttackThisProperty(msg.source(), record))
776                                &&(!msg.source().isAttributeSet(Attrib.PLAYERKILL)))
777                                {
778                                        if(!quiet)
779                                                msg.source().tell(L("You either need to turn on your PK flag, or be in a clan war to rob this property."));
780                                        return false;
781                                }
782                                final LegalBehavior B=getLegalBehavior(R);
783                                if(B!=null)
784                                {
785                                        final MOB D=getPropertyOwner(record);
786                                        if((D!=null)&&(D!=msg.source()))
787                                        {
788                                                // now check for witnesses
789                                                boolean witnessFound=false;
790                                                final List<Room> rooms=new LinkedList<Room>();
791                                                rooms.add(R);
792                                                for(int d=0;d<Directions.NUM_DIRECTIONS();d++)
793                                                {
794                                                        final Room R2=R.getRoomInDir(d);
795                                                        if(R2!=null)
796                                                                rooms.add(R2);
797                                                }
798                                                for(final Room R2 : rooms)
799                                                {
800                                                        for(int m=0;m<R2.numInhabitants();m++)
801                                                        {
802                                                                final MOB M=R2.fetchInhabitant(m);
803                                                                if((M!=null)
804                                                                && doesHavePriviledgesHere(M,R) // not a typo
805                                                                &&(CMLib.flags().canBeSeenBy(msg.source(), M)))
806                                                                        witnessFound=true;
807                                                        }
808                                                }
809                                                if(witnessFound)
810                                                        B.accuse(getLegalObject(R),msg.source(),D,new String[]{"PROPERTYROB","THIEF_ROBBERY"});
811                                        }
812                                }
813                                Ability propertyProp=((Item)msg.target()).fetchEffect("Prop_PrivateProperty");
814                                if(propertyProp==null)
815                                {
816                                        propertyProp=CMClass.getAbility("Prop_PrivateProperty");
817                                        propertyProp.setMiscText("owner=\""+record.getOwnerName()+"\" expiresec=60");
818                                        ((Item)msg.target()).addNonUninvokableEffect(propertyProp);
819                                }
820                        }
821                }
822                return true;
823        }

825        @Override
826        public PrivateProperty getPropertyRecord(final Item item)
827        {
828                if(item==null)
829                        return null;
830                if(item instanceof PrivateProperty)
831                        return (PrivateProperty)item;
832                if(item.numEffects()==0)
833                        return null;
834                for(final Enumeration<Ability> a=item.effects();a.hasMoreElements();)
835                {
836                        final Ability A=a.nextElement();
837                        if(A instanceof PrivateProperty)
838                                return (PrivateProperty)A;
839                }
840                return null;
841        }

843        @Override
844        public boolean mayOwnThisItem(final MOB mob, final Item item)
845        {
846                final PrivateProperty record = getPropertyRecord(item);
847                if(record != null)
848                {
849                        if(doesHaveWeakPrivilegesWith(mob,record))
850                                return true;
851                        final MOB following=mob.amFollowing();
852                        while((following!=null)&&(following!=mob))
853                        {
854                                if(doesHaveWeakPrivilegesWith(following,record))
855                                        return true;
856                        }
857                        if(item.owner() instanceof Room)
858                        {
859                                final Room R=(Room)item.owner();
860                                if(doesHavePriviledgesHere(mob,R))
861                                        return true;
862                        }
863                        return false;
864                }
865                if(item.owner() instanceof Room)
866                {
867                        final Room R=(Room)item.owner();
868                        final PrivateProperty roomRecord = getPropertyRecord(R);
869                        if(roomRecord != null)
870                        {
871                                if(doesHaveWeakPrivilegesWith(mob,roomRecord))
872                                        return true;
873                                if((roomRecord.getOwnerName()!=null)
874                                &&(roomRecord.getOwnerName().length()>0))
875                                        return false;
876                        }
877                }
878                return true;
879        }
