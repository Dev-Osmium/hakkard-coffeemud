144        @Override
145        public List<ChannelMsg> getChannelQue(final int channelNumber, final int numNewToSkip, final int numToReturn)
146        {
147                if((channelNumber>=0)&&(channelNumber<channelList.size()))
148                {
149                        final CMChannel channel = channelList.get(channelNumber);
150                        final LinkedList<ChannelMsg> msgs=new LinkedList<ChannelMsg>();
151                        if(numNewToSkip < channel.queue().size())
152                        {
153                                int skipNum=numNewToSkip;
154                                for(final ChannelMsg msg : channel.queue())
155                                {
156                                        if((--skipNum < 0)&&(msgs.size() < numToReturn))
157                                                msgs.addFirst(msg);
158                                }
159                        }
160
161                        if(msgs.size()>=numToReturn)
162                                return msgs;
163                        if(channel.flags().contains(ChannelsLibrary.ChannelFlag.NOBACKLOG))
164                                return msgs;
165                        final List<Pair<String,Long>> backLog=CMLib.database().getBackLogEntries(channel.name(), numNewToSkip, numToReturn);
166                        if(backLog.size()<=msgs.size())
167                                return msgs;
168                        final List<ChannelMsg> allMsgs = new XVector<ChannelMsg>();
169                        for(int x=0;x<backLog.size()-msgs.size();x++)
170                        {
171                                final CMMsg msg=CMClass.getMsg();
172                                msg.parseFlatString(backLog.get(x).first);
173                                final long time = backLog.get(x).second.longValue();
174                                allMsgs.add(new ChannelMsg()
175                                {
176                                        @Override
177                                        public CMMsg msg()
178                                        {
179                                                return msg;
180                                        }
181
182                                        @Override
183                                        public long sentTimeMillis()
184                                        {
185                                                return time;
186                                        }
187                                });
188                        }
189                        allMsgs.addAll(msgs);
190                        return allMsgs;
191                }
192                return emptyQueue;
193        }

201        @Override
202        public boolean mayReadThisChannel(final MOB sender,
203                                                                          final boolean areaReq,
204                                                                          final MOB M,
205                                                                          final int channelNumber,
206                                                                          final boolean offlineOK)
207        {
208                if((sender==null)||(M==null))
209                        return false;
210                final PlayerStats pstats=M.playerStats();
211                if(pstats==null)
212                        return false;
213                final Room R=M.location();
214                if(((!offlineOK))
215                &&((M.amDead())||(R==null)))
216                        return false;
217                final CMChannel chan=getChannel(channelNumber);
218                if(chan==null)
219                        return false;
220                if(chan.flags().contains(ChannelFlag.CLANONLY)||chan.flags().contains(ChannelFlag.CLANALLYONLY))
221                {
222                        // only way to fail an all-clan send is to have NO clan.
223                        if(!CMLib.clans().checkClanPrivilege(M, Clan.Function.CHANNEL))
224                                return false;
225
226                        if((!CMLib.clans().isAnyCommonClan(sender,M))
227                        &&((!chan.flags().contains(ChannelFlag.CLANALLYONLY))
228                                ||(!CMLib.clans().findAnyClanRelations(M,sender,Clan.REL_ALLY))))
229                                return false;
230                }
231
232                if((!pstats.isIgnored(sender))
233                &&(CMLib.masking().maskCheck(chan.mask(),M,true))
234                &&((!areaReq)
235                   ||(sender.location()==null)
236                   ||(R==null)
237                   ||(R.getArea()==sender.location().getArea()))
238                &&(!CMath.isSet(pstats.getChannelMask(),channelNumber)))
239                        return true;
240                return false;
241        }

243        @Override
244        public boolean mayReadThisChannel(final MOB sender, final boolean areaReq, final Session ses, final int channelNumber)
245        {
246                if(ses==null)
247                        return false;
248                final MOB M=ses.mob();
249
250                if((sender==null)
251                ||(M==null)
252                ||(M.amDead())
253                ||(M.location()==null))
254                        return false;
255                final PlayerStats pstats=M.playerStats();
256                if(pstats==null)
257                        return false;
258                String senderName=sender.Name();
259                final int x=senderName.indexOf('@');
260                if(x>0)
261                        senderName=senderName.substring(0,x);
262                final CMChannel chan=getChannel(channelNumber);
263                if(chan==null)
264                        return false;
265                if(chan.flags().contains(ChannelFlag.CLANONLY)||chan.flags().contains(ChannelFlag.CLANALLYONLY))
266                {
267                        // only way to fail an all-clan send is to have NO clan.
268                        if(!CMLib.clans().checkClanPrivilege(M, Clan.Function.CHANNEL))
269                                return false;
270                        if((!CMLib.clans().isAnyCommonClan(sender,M))
271                        &&((!chan.flags().contains(ChannelFlag.CLANALLYONLY))
272                                ||(!CMLib.clans().findAnyClanRelations(M,sender,Clan.REL_ALLY))))
273                                return false;
274                }
275
276                final Room R=M.location();
277                if((!ses.isStopped())
278                &&(R!=null)
279                &&(!pstats.isIgnored(sender))
280                &&(!pstats.isIgnored(senderName))
281                &&(CMLib.masking().maskCheck(chan.mask(),M,true))
282                &&((!areaReq)
283                   ||(sender.location()==null)
284                   ||(R.getArea()==sender.location().getArea()))
285                &&(!CMath.isSet(pstats.getChannelMask(),channelNumber)))
286                        return true;
287                return false;
288        }

291        public boolean mayReadThisChannel(final MOB M, final int channelNumber, final boolean zapCheckOnly)
292        {
293                if(M==null)
294                        return false;
295
296                if(channelNumber>=getNumChannels())
297                        return false;
298
299                final CMChannel chan=getChannel(channelNumber);
300                if(chan==null)
301                        return false;
302                if((chan.flags().contains(ChannelFlag.CLANONLY)||chan.flags().contains(ChannelFlag.CLANALLYONLY))
303                &&(!CMLib.clans().checkClanPrivilege(M, Clan.Function.CHANNEL)))
304                        return false;
305
306                if(((zapCheckOnly)||((!M.amDead())&&(M.location()!=null)))
307                &&(CMLib.masking().maskCheck(chan.mask(),M,true))
308                &&(!CMath.isSet(M.playerStats().getChannelMask(),channelNumber)))
309                        return true;
310                return false;
311        }

313        @Override
314        public void channelQueUp(final int channelNumber, final CMMsg msg)
315        {
316                CMLib.map().sendGlobalMessage(msg.source(),CMMsg.TYP_CHANNEL,msg);
317                final CMChannel channel=getChannel(channelNumber);
318                final SLinkedList<ChannelMsg> q=channel.queue();
319                synchronized(q)
320                {
321                        if(q.size()>=QUEUE_SIZE)
322                                q.removeLast();
323                        final long now = System.currentTimeMillis();
324                        q.addFirst(new ChannelMsg()
325                        {
326                                @Override
327                                public CMMsg msg()
328                                {
329                                        return msg;
330                                }
331
332                                @Override
333                                public long sentTimeMillis()
334                                {
335                                        return now;
336                                }
337                        });
338                }
339                if((!channel.flags().contains(ChannelsLibrary.ChannelFlag.NOBACKLOG))
340                &&(!CMSecurity.isDisabled(CMSecurity.DisFlag.CHANNELBACKLOGS))
341                &&(!CMProps.getVar(CMProps.Str.CHANNELBACKLOG).equals("0")))
342                {
343                        try
344                        {
345                                CMLib.database().addBackLogEntry(getChannel(channelNumber).name(), msg.toFlatString());
346                        }
347                        catch(final Exception e)
348                        {
349                                Log.errOut(e);
350                        }
351                }
352        }

355        public int getChannelIndex(String channelName)
356        {
357                channelName=channelName.toUpperCase();
358                for(int c=0;c<channelList.size();c++)
359                {
360                        if((channelList.get(c)).name().equals(channelName))
361                                return c;
362                }
363                for(int c=0;c<channelList.size();c++)
364                {
365                        if((channelList.get(c)).name().startsWith(channelName))
366                                return c;
367                }
368                return -1;

417        @Override
418        public String getExtraChannelDesc(final String channelName)
419        {
420                final StringBuilder str=new StringBuilder("");
421                final int dex = getChannelIndex(channelName);
422                if(dex >= 0)
423                {
424                        final CMChannel chan=getChannel(dex);
425                        final Set<ChannelFlag> flags = chan.flags();
426                        final String mask = chan.mask();
427                        if(flags.contains(ChannelFlag.CLANALLYONLY))
428                                str.append(L(" This is a channel for clans and their allies."));
429                        if(flags.contains(ChannelFlag.CLANONLY))
430                                str.append(L(" Only members of the same clan can see messages on this channel."));
431                        if(flags.contains(ChannelFlag.PLAYERREADONLY)
432                        ||flags.contains(ChannelFlag.READONLY)
433                        ||flags.contains(ChannelFlag.ARCHONREADONLY))
434                                str.append(L(" This channel is read-only."));
435                        if(flags.contains(ChannelFlag.SAMEAREA))
436                                str.append(L(" Only people in the same area can see messages on this channel."));
437                        if((mask!=null)&&(mask.trim().length()>0))
438                                str.append(L(" The following may read this channel : @x1",CMLib.masking().maskDesc(mask)));
439                }
440                return str.toString();
441        }

512        public String parseOutFlags(final String mask, final Set<ChannelFlag> flags, final String[] colorOverride)
513        {
514                final List<String> V=CMParms.parseSpaces(mask,true);
515                for(int v=V.size()-1;v>=0;v--)
516                {
517                        final String s=V.get(v).toUpperCase();
518                        if(CMParms.contains(CMParms.toStringArray(ChannelFlag.values()), s))
519                        {
520                                V.remove(v);
521                                flags.add(ChannelFlag.valueOf(s));
522                        }
523                        else
524                        {
525                                final Color C=(Color)CMath.s_valueOf(Color.class, s);
526                                if(C!=null)
527                                {
528                                        V.remove(v);
529                                        if(s.startsWith("BG"))
530                                                colorOverride[0]=colorOverride[0]+C.getANSICode();
531                                        else
532                                                colorOverride[0]=C.getANSICode()+colorOverride[0];
533                                        colorOverride[1]+=" "+s;
534                                }
535                        }
536                }
537                final StringBuilder str=new StringBuilder();
538                for(final String s : V)
539                        str.append(s).append(" ");
540                return str.toString().trim();
541        }

543        @Override
544        public int loadChannels(String list, String ilist, String imc2list)
545        {
546                clearChannels();
547                while(list.length()>0)
548                {
549                        int x=list.indexOf(',');
550
551                        String item=null;
552                        if(x<0)
553                        {
554                                item=list.trim();
555                                list="";
556                        }
557                        else
558                        {
559                                item=list.substring(0,x).trim();
560                                list=list.substring(x+1);
561                        }
562                        x=item.indexOf(' ');
563                        final CMChannel chan;
564                        if(x>0)
565                        {
566                                final String[] colorOverride=new String[]{"",""};
567                                final Set<ChannelFlag> flags = new HashSet<ChannelFlag>();
568                                final String mask=parseOutFlags(item.substring(x+1).trim(),flags,colorOverride);
569                                item=item.substring(0,x);
570                                chan = this.createNewChannel(item.toUpperCase().trim(), mask, flags, colorOverride[0], colorOverride[1]);
571                        }
572                        else
573                                chan = this.createNewChannel(item.toUpperCase().trim());
574                        channelList.add(chan);
575                }
576                while(ilist.length()>0)
577                {
578                        final int x=ilist.indexOf(',');
579
580                        String item=null;
581                        if(x<0)
582                        {
583                                item=ilist.trim();
584                                ilist="";
585                        }
586                        else
587                        {
588                                item=ilist.substring(0,x).trim();
589                                ilist=ilist.substring(x+1);
590                        }
591                        final int y1=item.indexOf(' ');
592                        final int y2=item.lastIndexOf(' ');
593                        if((y1<0)||(y2<=y1))
594                                continue;
595                        final String lvl=item.substring(y1+1,y2).trim();
596                        final String ichan=item.substring(y2+1).trim();
597                        item=item.substring(0,y1);
598                        final Set<ChannelFlag> flags = new HashSet<ChannelFlag>();
599                        final String nameStr=item.toUpperCase().trim();
600                        final String[] colorOverride=new String[]{"",""};
601                        final String maskStr=parseOutFlags(lvl,flags,colorOverride);
602                        final String i3nameStr=ichan;
603                        final CMChannel chan = this.createNewChannel(nameStr, i3nameStr, "", maskStr, flags, colorOverride[0], colorOverride[1]);
604                        channelList.add(chan);
605                }
606                while(imc2list.length()>0)
607                {
608                        final int x=imc2list.indexOf(',');
609
610                        String item=null;
611                        if(x<0)
612                        {
613                                item=imc2list.trim();
614                                imc2list="";
615                        }
616                        else
617                        {
618                                item=imc2list.substring(0,x).trim();
619                                imc2list=imc2list.substring(x+1);
620                        }
621                        final int y1=item.indexOf(' ');
622                        final int y2=item.lastIndexOf(' ');
623                        if((y1<0)||(y2<=y1))
624                                continue;
625                        final Set<ChannelFlag> flags = new HashSet<ChannelFlag>();
626                        final String lvl=item.substring(y1+1,y2).trim();
627                        final String ichan=item.substring(y2+1).trim();
628                        item=item.substring(0,y1);
629                        final String nameStr=item.toUpperCase().trim();
630                        final String[] colorOverride=new String[]{"",""};
631                        final String maskStr=parseOutFlags(lvl,flags,colorOverride);
632                        final String imc2Name=ichan;
633                        final CMChannel chan = this.createNewChannel(nameStr, "", imc2Name, maskStr, flags, colorOverride[0], colorOverride[1]);
634                        channelList.add(chan);
635                }
636                baseChannelNames=new String[channelList.size()];
637                for(int i=0;i<channelList.size();i++)
638                        baseChannelNames[i]=channelList.get(i).name();
639                if(channelList.size()>31)
640                {
641                        Log.errOut("CMChannels", "Too many channels defined: "+channelList.size()+".");
642                }
643                if(!CMSecurity.isDisabled(CMSecurity.DisFlag.CHANNELAUCTION))
644                {
645                        channelList.add(this.createNewChannel("AUCTION"));
646                }
647                return channelList.size();
648        }

650        @Override
651        public boolean sendChannelCMMsgTo(final Session ses, final boolean areareq, final int channelInt, final CMMsg msg, final MOB sender)
652        {
653                final MOB M=ses.mob();
654                if(M==null)
655                        return false;
656                final ChannelsLibrary libChk=(ChannelsLibrary)CMLib.library((char)ses.getGroupID(),Library.CHANNELS);
657                final CMChannel channel=this.getChannel(channelInt);
658                if((libChk != null)&&(libChk != this)&&(channel!=null))
659                {
660                        final int newDex=libChk.getChannelIndex(channel.name());
661                        if(newDex < 0)
662                                return false;
663                }
664                final Room R=M.location();
665                boolean didIt=false;
666                if(mayReadThisChannel(sender,areareq,ses,channelInt)
667                &&(R!=null)
668                &&((sender.location()==R)||(R.okMessage(ses.mob(),msg))))
669                {
670                        if(ses.getClientTelnetMode(Session.TELNET_GMCP)&&(channel!=null))
671                        {
672                                ses.sendGMCPEvent("comm.channel", "{\"chan\":\""+channel.name()+"\",\"msg\":\""+
673                                                MiniJSON.toJSONString(CMLib.coffeeFilter().fullOutFilter(null, M, msg.source(), msg.target(), msg.tool(), CMStrings.removeColors((M==msg.source())?msg.sourceMessage():msg.othersMessage()), false))
674                                                +"\",\"player\":\""+msg.source().name()+"\"}");
675                        }
676                        M.executeMsg(M,msg);
677                        didIt=true;
678                        if(msg.trailerMsgs()!=null)
679                        {
680                                for(final CMMsg msg2 : msg.trailerMsgs())
681                                {
682                                        if((msg!=msg2)&&(R.okMessage(M,msg2)))
683                                                M.executeMsg(M,msg2);
684                                }
685                                msg.trailerMsgs().clear();
686                        }
687                        if(msg.trailerRunnables()!=null)
688                        {
689                                for(final Runnable r : msg.trailerRunnables())
690                                        CMLib.threads().executeRunnable(r);
691                                msg.trailerRunnables().clear();
692                        }
693                }
694                return didIt;
695        }

697        @Override
698        public void createAndSendChannelMessage(final MOB mob, String channelName, String message, final boolean systemMsg)
699        {
700                if(mob == null)
701                        return;
702                final int channelInt=getChannelIndex(channelName);
703                if(channelInt<0)
704                        return;
705
706                final PlayerStats pStats=mob.playerStats();
707                final Room R=mob.location();
708
709                message=CMProps.applyINIFilter(message,CMProps.Str.CHANNELFILTER);
710                final CMChannel chan=getChannel(channelInt);
711                final Set<ChannelFlag> flags=chan.flags();
712                channelName=chan.name();
713                final String channelColor="^Q";
714
715                String nameAppendage="";
716                if(chan.flags().contains(ChannelsLibrary.ChannelFlag.ADDACCOUNT)
717                &&(pStats.getAccount()!=null))
718                        nameAppendage+=" ("+pStats.getAccount().name()+")";
719                if(chan.flags().contains(ChannelsLibrary.ChannelFlag.ADDROOM)
720                &&(R!=null))
721                        nameAppendage+=" in "+CMStrings.replaceAll(R.displayText(mob),"\'","");
722                if(chan.flags().contains(ChannelsLibrary.ChannelFlag.ADDAREA)
723                &&(R!=null)&&(R.getArea()!=null))
724                        nameAppendage+=" at "+CMStrings.replaceAll(R.getArea().name(mob),"\'","");
725
726                CMMsg msg=null;
727                if(systemMsg)
728                {
729                        String str="["+channelName+"]"+nameAppendage+" '"+message+"'^</CHANNEL^>^?^.";
730                        if((!mob.name().startsWith("^"))||(mob.name().length()>2))
731                                str="<S-NAME>"+nameAppendage+" "+str;
732                        msg=CMClass.getMsg(mob,null,null,
733                                        CMMsg.MASK_CHANNEL|CMMsg.MASK_ALWAYS|CMMsg.MSG_SPEAK,channelColor+"^<CHANNEL \""+channelName+"\"^>"+str,
734                                        CMMsg.NO_EFFECT,null,
735                                        CMMsg.MASK_CHANNEL|(CMMsg.TYP_CHANNEL+channelInt),channelColor+"^<CHANNEL \""+channelName+"\"^>"+str);
736                }
737                else
738                if(message.startsWith(",")
739                ||(message.startsWith(":")
740                        &&(message.length()>1)
741                        &&(Character.isLetter(message.charAt(1))||message.charAt(1)==' ')))
742                {
743                        String msgstr=message.substring(1);
744                        final Vector<String> V=CMParms.parse(msgstr);
745                        Social S=CMLib.socials().fetchSocial(V,true,false);
746                        if(S==null)
747                                S=CMLib.socials().fetchSocial(V,false,false);
748                        if(S!=null)
749                                msg=S.makeChannelMsg(mob,channelInt,channelName,V,false);
750                        else
751                        {
752                                msgstr=CMProps.applyINIFilter(msgstr,CMProps.Str.EMOTEFILTER);
753                                if(msgstr.trim().startsWith("'")||msgstr.trim().startsWith("`"))
754                                        msgstr=msgstr.trim();
755                                else
756                                        msgstr=" "+msgstr.trim();
757                                final String srcstr="^<CHANNEL \""+channelName+"\"^>["+channelName+"] "+mob.name()+nameAppendage+msgstr+"^</CHANNEL^>^N^.";
758                                final String reststr="^<CHANNEL \""+channelName+"\"^>["+channelName+"] <S-NAME>"+nameAppendage+msgstr+"^</CHANNEL^>^N^.";
759                                msg=CMClass.getMsg(mob,null,null,
760                                                CMMsg.MASK_CHANNEL|CMMsg.MASK_ALWAYS|CMMsg.MSG_SPEAK,channelColor+""+srcstr,
761                                                CMMsg.NO_EFFECT,null,
762                                                CMMsg.MASK_CHANNEL|(CMMsg.TYP_CHANNEL+channelInt),channelColor+reststr);
763                        }
764                }
765                else
766                {
767                        msg=CMClass.getMsg(mob,null,null,
768                                        CMMsg.MASK_CHANNEL|CMMsg.MASK_ALWAYS|CMMsg.MSG_SPEAK,channelColor+"^<CHANNEL \""+channelName+"\"^>"+L("You")+nameAppendage+" "+channelName+" '"+message+"'^</CHANNEL^>^N^.",
769                                        CMMsg.NO_EFFECT,null,
770                                        CMMsg.MASK_CHANNEL|(CMMsg.TYP_CHANNEL+channelInt),channelColor+"^<CHANNEL \""+channelName+"\"^><S-NAME>"+nameAppendage+" "+CMLib.english().makePlural(channelName)+" '"+message+"'^</CHANNEL^>^N^.");
771                }
772
773                if((chan.flags().contains(ChannelsLibrary.ChannelFlag.ACCOUNTOOC)
774                        ||(chan.flags().contains(ChannelsLibrary.ChannelFlag.ACCOUNTOOCNOADMIN) && (!CMSecurity.isStaff(mob))))
775                &&(pStats!=null)
776                &&(pStats.getAccount()!=null)
777                &&(msg.source()==mob))
778                {
779                        final String accountName=pStats.getAccount().getAccountName();
780                        if(msg.sourceMessage()!=null)
781                                msg.setSourceMessage(CMStrings.replaceAll(msg.sourceMessage(), "<S-NAME>", accountName));
782                        if(msg.targetMessage()!=null)
783                                msg.setTargetMessage(CMStrings.replaceAll(msg.targetMessage(), "<S-NAME>", accountName));
784                        if(msg.othersMessage()!=null)
785                                msg.setOthersMessage(CMStrings.replaceAll(msg.othersMessage(), "<S-NAME>", accountName));
786                }
787
788                if((chan.flags().contains(ChannelsLibrary.ChannelFlag.REALNAMEOOC)
789                        ||(chan.flags().contains(ChannelsLibrary.ChannelFlag.REALNAMEOOCNOADMIN) && (!CMSecurity.isStaff(mob))))
790                &&(pStats!=null)
791                &&(pStats.getAccount()!=null)
792                &&(msg.source()==mob))
793                {
794                        final String realName=mob.Name();
795                        if(msg.sourceMessage()!=null)
796                                msg.setSourceMessage(CMStrings.replaceAll(msg.sourceMessage(), "<S-NAME>", realName));
797                        if(msg.targetMessage()!=null)
798                                msg.setTargetMessage(CMStrings.replaceAll(msg.targetMessage(), "<S-NAME>", realName));
799                        if(msg.othersMessage()!=null)
800                                msg.setOthersMessage(CMStrings.replaceAll(msg.othersMessage(), "<S-NAME>", realName));
801                }
802
803                if(chan.flags().contains(ChannelsLibrary.ChannelFlag.NOLANGUAGE))
804                        msg.setTool(getCommonLanguage());
805
806                CMLib.commands().monitorGlobalMessage(R, msg);
807                if((R!=null)
808                &&((!R.isInhabitant(mob))||(R.okMessage(mob,msg))))
809                {
810                        if(chan.flags().contains(ChannelsLibrary.ChannelFlag.TWITTER))
811                                tweet(message);
812                        final boolean areareq=flags.contains(ChannelsLibrary.ChannelFlag.SAMEAREA);
813                        channelQueUp(channelInt,msg);
814                        for(final Session S : CMLib.sessions().localOnlineIterable())
815                                sendChannelCMMsgTo(S,areareq,channelInt,msg,mob);
816                }
817                if((CMLib.intermud().i3online()&&(CMLib.intermud().isI3channel(channelName)))
818                ||(CMLib.intermud().imc2online()&&(CMLib.intermud().isIMC2channel(channelName))))
819                        CMLib.intermud().i3channel(mob,channelName,message);
820        }

843        /**
844         * Requires including special library and special configuration.
845         * @param msg the message to tweet
846         */
847        private void tweet(String msg)
848        {
849                msg = CMStrings.scrunchWord(CMStrings.removeColors(msg), 280);
850                try
851                {
852                        final Class<?> cbClass = Class.forName("twitter4j.conf.ConfigurationBuilder");
853                        final Object cbObj = cbClass.newInstance();
854                        final Method cbM1=cbClass.getMethod("setOAuthConsumerKey", String.class);
855                        cbM1.invoke(cbObj,CMProps.getProp("TWITTER-OAUTHCONSUMERKEY"));
856                        final Method cbM2=cbClass.getMethod("setOAuthConsumerSecret", String.class);
857                        cbM2.invoke(cbObj,CMProps.getProp("TWITTER-OAUTHCONSUMERSECRET"));
858                        final Method cbM3=cbClass.getMethod("setOAuthAccessToken", String.class);
859                        cbM3.invoke(cbObj,CMProps.getProp("TWITTER-OAUTHACCESSTOKEN"));
860                        final Method cbM4=cbClass.getMethod("setOAuthAccessTokenSecret", String.class);
861                        cbM4.invoke(cbObj,CMProps.getProp("TWITTER-OAUTHACCESSTOKENSECRET"));
862                        final Method cbM5=cbClass.getMethod("build");
863                        final Object cbBuildObj = cbM5.invoke(cbObj);
864
865                        final Class<?> cfClass = Class.forName("twitter4j.conf.Configuration");
866                        final Class<?> afClass = Class.forName("twitter4j.auth.AuthorizationFactory");
867                        final Method adM1 = afClass.getMethod("getInstance",cfClass);
868                        final Object auObj = adM1.invoke(null, cbBuildObj);
869
870                        final Class<?> auClass = Class.forName("twitter4j.auth.Authorization");
871                        final Class<?> tfClass = Class.forName("twitter4j.TwitterFactory");
872                        final Object tfObj = tfClass.newInstance();
873                        final Method tfM1 = tfClass.getMethod("getInstance", auClass);
874                        final Object twObj = tfM1.invoke(tfObj, auObj);
875
876                        final Class<?> twClass = Class.forName("twitter4j.Twitter");
877                        final Method twM1 = twClass.getMethod("updateStatus", String.class);
878                        twM1.invoke(twObj, msg);
879                }
880                catch (final Exception e)
881                {
882                        Log.errOut(e);
883                }
884        }

886        @Override
887        public boolean tick(final Tickable ticking, final int tickID)
888        {
889                try
890                {
891                        if(!CMSecurity.isDisabled(CMSecurity.DisFlag.UTILITHREAD))
892                        {
893                                tickStatus=Tickable.STATUS_ALIVE;
894                                try
895                                {
896                                        final String propStr=CMProps.getVar(CMProps.Str.CHANNELBACKLOG).toUpperCase().trim();
897                                        if((!CMSecurity.isDisabled(CMSecurity.DisFlag.CHANNELBACKLOGS))
898                                        &&(!propStr.equalsIgnoreCase("INFINITY"))
899                                        &&(!propStr.equalsIgnoreCase("FOREVER")))
900                                        {
901                                                if(CMath.isInteger(propStr))
902                                                        CMLib.database().trimBackLogEntries(getChannelNames(), CMath.s_int(propStr), 0);
903                                                else
904                                                {
905                                                        final String[] ss=propStr.split(" ");
906                                                        if((ss.length!=2)&&(!CMath.isInteger(ss[0])))
907                                                                Log.errOut("CMChannels","Malformed CHANNELBACKLOG entry in coffeemud.ini file: "+propStr);
908                                                        else
909                                                        if(ss[1].equals("DAYS")||ss[1].equals("DAY"))
910                                                                CMLib.database().trimBackLogEntries(getChannelNames(), Integer.MAX_VALUE, System.currentTimeMillis() - (CMath.s_int(ss[0]) * TimeManager.MILI_DAY));
911                                                        else
912                                                        if(ss[1].equals("WEEKS")||ss[1].equals("WEEK"))
913                                                                CMLib.database().trimBackLogEntries(getChannelNames(), Integer.MAX_VALUE, System.currentTimeMillis() - (CMath.s_int(ss[0]) * TimeManager.MILI_WEEK));
914                                                        else
915                                                        if(ss[1].equals("MONTHS")||ss[1].equals("MONTHS"))
916                                                                CMLib.database().trimBackLogEntries(getChannelNames(), Integer.MAX_VALUE, System.currentTimeMillis() - (CMath.s_int(ss[0]) * TimeManager.MILI_MONTH));
917                                                        else
918                                                        if(ss[1].equals("YEARSS")||ss[1].equals("YEAR"))
919                                                                CMLib.database().trimBackLogEntries(getChannelNames(), Integer.MAX_VALUE, System.currentTimeMillis() - (CMath.s_int(ss[0]) * TimeManager.MILI_YEAR));
920                                                        else
921                                                                Log.errOut("CMChannels","Malformed CHANNELBACKLOG entry in coffeemud.ini file: "+propStr);
922                                                }
923                                        }
924                                }
925                                finally
926                                {
927                                }
928                        }
929                }
930                finally
931                {
932                        tickStatus=Tickable.STATUS_NOT;
933                        setThreadStatus(serviceClient,"sleeping");
934                }
935                return true;
936        }
