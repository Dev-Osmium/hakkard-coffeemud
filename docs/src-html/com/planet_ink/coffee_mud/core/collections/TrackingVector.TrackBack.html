021/**
022 * A tracking vector keeps track of which other tracking vectors that items of
023 * the same type have been added to so that, if desired, items can be removed
024 * from all vectors at once. The constructor requires a map to keep track of the
025 * objects.
026 * 
027 * This class is required for RTree
028 * 
029 * @author Bo Zimmerman
030 * 
031 * @param <T> the type, or something?
032 */
033public class TrackingVector<T> extends Vector<T>
034{
035        private static final long                                                                        serialVersionUID = 3331770309040710349L;
036
037        private final Map<T, List<WeakReference<TrackingVector<T>>>> tracker;
038        private final WeakReference<TrackingVector<T>>                           myRef                    = new WeakReference<TrackingVector<T>>(this);
039        private final WeakReference<TrackBack<T>>                                        trackBackRef;
040
041        public interface TrackBack<T>
042        {
043                public void removed(T o);
044        }
045
046        public TrackingVector(Map<T, List<WeakReference<TrackingVector<T>>>> tracker)
047        {
048                super();
049                this.tracker = tracker;
050                trackBackRef = null;
051        }
052
053        public TrackingVector(Map<T, List<WeakReference<TrackingVector<T>>>> tracker, int sz)
054        {
055                super(sz);
056                this.tracker = tracker;
057                trackBackRef = null;
058        }
059
060        public TrackingVector(Map<T, List<WeakReference<TrackingVector<T>>>> tracker, TrackBack<T> obj)
061        {
062                super();
063                this.tracker = tracker;
064                this.trackBackRef = new WeakReference<TrackBack<T>>(obj);
065        }
066
067        public TrackingVector(Map<T, List<WeakReference<TrackingVector<T>>>> tracker, int sz, TrackBack<T> obj)
068        {
069                super(sz);
070                this.tracker = tracker;
071                this.trackBackRef = new WeakReference<TrackBack<T>>(obj);
072        }
073
074        protected void addTrackedEntry(T e)
075        {
076                synchronized (tracker)
077                {
078                        if (tracker.containsKey(e))
079                        {
080                                final List<WeakReference<TrackingVector<T>>> l = tracker.get(e);
081                                if (!l.contains(myRef))
082                                        l.add(myRef);
083                        }
084                        else
085                        {
086                                final List<WeakReference<TrackingVector<T>>> l = new Vector<WeakReference<TrackingVector<T>>>();
087                                l.add(myRef);
088                                tracker.put(e, l);
089                        }
090                }
091        }
092
093        protected void removeTrackedEntry(Object e)
094        {
095                synchronized (tracker)
096                {
097                        if (tracker.containsKey(e))
098                        {
099                                final List<WeakReference<TrackingVector<T>>> l = tracker.get(e);
100                                l.remove(myRef);
101                        }
102                }
103        }
104
105        public void removeAllTrackedEntries(T e)
106        {
107                synchronized (tracker)
108                {
109                        if (tracker.containsKey(e))
110                        {
111                                final List<WeakReference<TrackingVector<T>>> l = tracker.get(e);
112                                for (final WeakReference<TrackingVector<T>> ref : l)
113                                {
114                                        if (ref.get() != null)
115                                                ref.get().removeOnlyFromMe(e);
116                                }
117                                tracker.remove(e);
118                        }
119                }
120        }
121
122        @Override
123        public synchronized boolean add(T e)
124        {
125                if (super.add(e))
126                {
127                        addTrackedEntry(e);
128                        return true;
129                }
130                return false;
131        }
132
133        @Override
134        public synchronized void addElement(T e)
135        {
136                super.addElement(e);
137                addTrackedEntry(e);
138        }
139
140        @Override
141        public void add(int arg0, T arg1)
142        {
143                super.add(arg0, arg1);
144                addTrackedEntry(arg1);
145        }
146
147        @Override
148        public synchronized boolean addAll(Collection<? extends T> arg0)
149        {
150                for (final T o : arg0)
151                        addTrackedEntry(o);
152                return super.addAll(arg0);
153        }
154
155        @Override
156        public synchronized boolean addAll(int arg0, Collection<? extends T> arg1)
157        {
158                for (final T o : arg1)
159                        addTrackedEntry(o);
160                return super.addAll(arg0, arg1);
161        }
162
163        @Override
164        public void clear()
165        {
166                for (final T e : this)
167                        removeTrackedEntry(e);
168                super.clear();
169        }
170
171        @Override
172        public boolean remove(Object arg0)
173        {
174                if (removeOnlyFromMe(arg0))
175                {
176                        removeTrackedEntry(arg0);
177                        return true;
178                }
179                return false;
180        }
181
182        @SuppressWarnings("unchecked")
183        public boolean removeOnlyFromMe(Object arg0)
184        {
185                final boolean success = super.remove(arg0);
186                if ((trackBackRef != null) && (trackBackRef.get() != null))
187                        trackBackRef.get().removed((T) arg0);
188                return success;
189        }
190
191        @Override
192        public synchronized T remove(int arg0)
193        {
194                final T x = super.remove(arg0);
195                if (x != null)
196                {
197                        removeTrackedEntry(x);
198                        if ((trackBackRef != null) && (trackBackRef.get() != null))
199                                trackBackRef.get().removed(x);
200                }
201                return x;
202        }
203
204        @Override
205        public synchronized boolean removeAll(Collection<?> arg0)
206        {
207                for (final Object e : arg0)
208                        removeTrackedEntry(e);
209                return super.removeAll(arg0);
210        }
211}
