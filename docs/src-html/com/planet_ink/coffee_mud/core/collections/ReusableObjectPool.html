030public class ReusableObjectPool<T extends CMObject>
031{
032        private final NotifyingCMObjectVector<T>                masterList;
033        private final Stack<NotifyingCMObjectVector<T>> masterPool      = new Stack<NotifyingCMObjectVector<T>>();
034        private volatile int                                                    created         = 0;
035        private volatile int                                                    requests        = 0;
036        private volatile int                                                    distributed     = 0;
037        private volatile int                                                    returned        = 0;
038        private final int                                                               minEntries;
039        private final Object                                                    sync            = new Object();
040        private final PoolFixer                                                 fixer           = new PoolFixer();
041
042        private class PoolFixer implements Runnable
043        {
044                private boolean isRunning       = false;
045
046                public void startFixing()
047                {
048                        if (isRunning)
049                        {
050                                fix();
051                                return;
052                        }
053                        synchronized (this)
054                        {
055                                if (!isRunning)
056                                {
057                                        isRunning = true;
058                                        new Thread(Thread.currentThread().getThreadGroup(), this, "PoolFixer" + Thread.currentThread().getThreadGroup().getName().charAt(0)).start();
059                                        return;
060                                }
061                        }
062                        fix();
063                }
064
065                public void fix()
066                {
067                        Runtime.getRuntime().runFinalization();
068                        System.gc();
069                        try
070                        {
071                                Thread.sleep(100);
072                        }
073                        catch (final Exception e)
074                        {
075                        }
076                        Runtime.getRuntime().runFinalization();
077                        try
078                        {
079                                Thread.sleep(100);
080                        }
081                        catch (final Exception e)
082                        {
083                        }
084                        System.gc();
085                }
086
087                @Override
088                public void run()
089                {
090                        try
091                        {
092                                int iters = 0;
093                                while (masterPool.size() < minEntries)
094                                {
095                                        fix();
096                                        if ((masterPool.size() < minEntries) && (iters > 5))
097                                        {
098                                                iters = 0;
099                                                synchronized (sync)
100                                                {
101                                                        masterPool.add(makeNewEntry());
102                                                }
103                                        }
104                                        iters++;
105                                }
106                        }
107                        finally
108                        {
109                                synchronized (this)
110                                {
111                                        isRunning = false;
112                                }
113                        }
114                }
115        }
116
117        private class NotifyingCMObjectVector<K extends CMObject> extends Vector<T>
118        {
119                private static final long       serialVersionUID        = 1L;
120
121                public NotifyingCMObjectVector(final List<T> V)
122                {
123                        super(V);
124                }
125
126                public NotifyingCMObjectVector(int size)
127                {
128                        super(size);
129                }
130
131                @Override
132                protected void finalize() throws Throwable
133                {
134                        final NotifyingCMObjectVector<T> V = new NotifyingCMObjectVector<T>(this);
135                        synchronized (sync)
136                        {
137                                returned++;
138                                masterPool.push(V);
139                        }
140                        super.finalize();
141                }
142        }
143
144        public ReusableObjectPool(final List<T> initialEntry, final int minEntries)
145        {
146                if (initialEntry.size() == 0)
147                        this.masterList = new NotifyingCMObjectVector<T>(new ReadOnlyList<T>(initialEntry));
148                else
149                        this.masterList = new NotifyingCMObjectVector<T>(initialEntry);
150                this.minEntries = (minEntries < 2) ? 2 : minEntries;
151                for (int i = 0; i < minEntries; i++)
152                        masterPool.add(makeNewEntry());
153        }
154
155        public int getMasterPoolSize()
156        {
157                return masterPool.size();
158        }
159
160        public int getListSize()
161        {
162                return masterList.size();
163        }
164
165        @SuppressWarnings("unchecked")
166        private final NotifyingCMObjectVector<T> makeNewEntry()
167        {
168                final NotifyingCMObjectVector<T> myList = new NotifyingCMObjectVector<T>(masterList.size());
169                for (final T o : masterList)
170                        myList.add((T) o.copyOf());
171                created++;
172                if (created == minEntries * 1000)
173                        Log.errOut("ReuseOP", "Reusable Object Pool pass all reason: " + CMParms.toListString(masterList));
174                return myList;
175        }
176
177        public final List<T> get()
178        {
179                if (masterList.size() == 0)
180                        return masterList;
181                requests++;
182                if (masterPool.size() < minEntries / 2)
183                        fixer.startFixing();
184                synchronized (sync)
185                {
186                        if (!masterPool.isEmpty())
187                        {
188                                final NotifyingCMObjectVector<T> myList = masterPool.pop();
189                                if (myList != null)
190                                {
191                                        distributed++;
192                                        return myList;
193                                }
194                        }
195                }
196                fixer.fix();
197                return makeNewEntry();

199                // for(final T o : masterList)
200                // myList.add((T)o.copyOf());
201                // return myList;
202        }
