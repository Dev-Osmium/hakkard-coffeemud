033public class PrioritizingLimitedMap<T extends Comparable<T>, K> implements Map<T, K>
034{
035        protected int        itemLimit;
036        protected final int      origItemLimit;
037        protected final long touchAgeLimitMillis;
038        protected final long maxAgeLimitMillis;
039        protected final int      threshHoldToExpand;
040        
041        protected final AtomicBoolean trimming  = new AtomicBoolean(false);
042
043        private class LinkedEntry<V, W> extends Pair<V, W>
044        {
045                public volatile LinkedEntry<V, W>       next            = null;
046                public volatile LinkedEntry<V, W>       prev            = null;
047                public volatile int                                     priority        = 0;
048                public volatile boolean                         causedExpand= false;
049                public volatile int                                     index           = 0;
050                public volatile long                            lastTouch       = System.currentTimeMillis();
051                public final long                                       birthDate       = System.currentTimeMillis();
052
053                public LinkedEntry(V frst, W scnd)
054                {
055                        super(frst, scnd);
056                }
057        }
058
059        protected volatile LinkedEntry<T, K>                    head    = null;
060        protected volatile LinkedEntry<T, K>                    tail    = null;
061        protected final SortedMap<T, LinkedEntry<T, K>> map             = Collections.synchronizedSortedMap(new TreeMap<T, LinkedEntry<T, K>>());
062
063        /**
064         * Constructed a "limit" tree-based map. All the parameters here are
065         * somewhat fuzzy. The itemLimit will be ignored if lots of youngsters come
066         * flooding in. If the itemLimit is exceeded by more than 2* the touch/max
067         * age, then those ages will be similarly divided to clean out newer and
068         * newer entries in order to at least approach the itemLimit standards.
069         * itemLimit, is therefore, a long-run ideal. The threshold to expand is the
070         * only hard limit.
071         * 
072         * @param itemLimit
073         *            the number of items to try to limit this map to
074         * @param touchAgeLimitMillis
075         *            the age of last-touching that makes an item too old to keep
076         * @param maxAgeLimitMillis
077         *            the longest amount of time any entry is allowed to live,
078         *            regardless of touching
079         * @param threshHoldToExpand
080         *            the number of touches on any given item before the limit
081         *            expands to accommodate
082         */
083        public PrioritizingLimitedMap(int itemLimit, long touchAgeLimitMillis, long maxAgeLimitMillis, int threshHoldToExpand)
084        {
085                if (itemLimit <= 0)
086                        itemLimit = 1;
087                this.itemLimit = itemLimit;
088                this.origItemLimit = itemLimit;
089                this.touchAgeLimitMillis = (touchAgeLimitMillis > Integer.MAX_VALUE) ? Integer.MAX_VALUE : touchAgeLimitMillis;
090                this.maxAgeLimitMillis = (maxAgeLimitMillis > Integer.MAX_VALUE) ? Integer.MAX_VALUE : maxAgeLimitMillis;
091                this.threshHoldToExpand = threshHoldToExpand;
092        }
093
094        /**
095         * Constructed a "limit" tree-based map. All the parameters here are
096         * somewhat fuzzy. The itemLimit will be ignored if lots of youngsters come
097         * flooding in. If the itemLimit is exceeded by more than 2* the touch/max
098         * age, then those ages will be similarly divided to clean out newer and
099         * newer entries in order to at least approach the itemLimit standards.
100         * itemLimit, is therefore, a long-run ideal.
101         * 
102         * @param itemLimit
103         *            the number of items to try to limit this map to
104         * @param touchAgeLimitMillis
105         *            the age of last-touching that makes an item too old to keep
106         * @param maxAgeLimitMillis
107         *            the longest amount of time any entry is allowed to live,
108         *            regardless of touching
109         */
110        public PrioritizingLimitedMap(int itemLimit, long touchAgeLimitMillis, long maxAgeLimitMillis)
111        {
112                this(itemLimit, touchAgeLimitMillis, maxAgeLimitMillis, Integer.MAX_VALUE);
113        }
114
115        @Override
116        public K get(Object key)
117        {
118                final LinkedEntry<T, K> p = map.get(key);
119                if (p != null)
120                {
121                        markFoundAgain(p);
122                        if(!trimming.get())
123                        {
124                                trimming.set(true);
125                                try
126                                {
127                                        trimDeadwood(1);
128                                }
129                                finally
130                                {
131                                        trimming.set(false);
132                                }
133                        }
134                        return p.second;
135                }
136                return null;
137        }
138
139        @Override
140        public synchronized void clear()
141        {
142                head = null;
143                tail = null;
144                map.clear();
145        }
146
147        @Override
148        public boolean containsKey(Object arg0)
149        {
150                return map.containsKey(arg0);
151        }
152
153        public Enumeration<T> prioritizedKeys()
154        {
155                return new Enumeration<T>() 
156                {
157                        private LinkedEntry<T, K>       ptr     = head;
158
159                        @Override
160                        public boolean hasMoreElements()
161                        {
162                                return ptr != null;
163                        }
164
165                        @Override
166                        public T nextElement()
167                        {
168                                if (ptr != null)
169                                {
170                                        final T elem = ptr.first;
171                                        ptr = ptr.next;
172                                        return elem;
173                                }
174                                return null;
175                        }
176                };
177        }
178
179        @Override
180        public synchronized boolean containsValue(Object arg0)
181        {
182                for (final LinkedEntry<T, K> p : map.values())
183                {
184                        if (p.first == arg0)
185                                return true;
186                }
187                return false;
188        }
189
190        @Override
191        public synchronized Set<java.util.Map.Entry<T, K>> entrySet()
192        {
193                final Set<java.util.Map.Entry<T, K>> c = new TreeSet<java.util.Map.Entry<T, K>>();
194                for (final T t : map.keySet())
195                        c.add(new Pair<T, K>(t, map.get(t).second));
196                return c;
197        }
198
199        @Override
200        public boolean isEmpty()
201        {
202                return map.isEmpty();
203        }
204
205        @Override
206        public Set<T> keySet()
207        {
208                return map.keySet();
209        }

211        private void markFoundAgain(LinkedEntry<T, K> p)
212        {
213                p.priority++;
214                p.lastTouch = System.currentTimeMillis();
215                LinkedEntry<T, K> pp = p.prev;
216                while ((pp != null) && (p.priority > pp.priority))
217                {
218                        final LinkedEntry<T, K> pn = p.next;
219                        final int ppIndex = pp.index;
220                        pp.index = p.index;
221                        p.index = ppIndex;
222                        p.prev = pp.prev;
223                        p.next = pp;
224                        if (pp.prev == null)
225                                head = p;
226                        else
227                                pp.prev.next = p;
228                        pp.prev = p;
229                        if (pn != null)
230                                pn.prev = pp;
231                        else
232                                tail = pp;
233                        pp.next = pn;
234                        pp = p.prev;
235                }
236        }
237
238        private void trimDeadwood(int multiplier)
239        {
240                if (map.size() > itemLimit)
241                {
242                        LinkedEntry<T, K> prev = tail;
243                        final long touchTimeout = System.currentTimeMillis() - (touchAgeLimitMillis / multiplier);
244                        final long maxAgeTimeout = System.currentTimeMillis() - (maxAgeLimitMillis / multiplier);
245                        int expands = 0;
246                        int counter = 0;
247                        while ((prev != null) && (prev != head) && (prev.index <= 0) && (map.size() > itemLimit))
248                        {
249                                if (counter++ > map.size())
250                                        break; // failsafe to broken links
251                                final LinkedEntry<T, K> pprev = prev.prev;
252                                if ((prev.lastTouch < touchTimeout) || (prev.birthDate < maxAgeTimeout))
253                                        remove(prev.first);
254                                else 
255                                if ((prev.priority > threshHoldToExpand) && (!prev.causedExpand))
256                                {
257                                        prev.causedExpand = true;
258                                        expands++;
259                                }
260                                prev = pprev;
261                        }
262                        itemLimit += expands;
263                        if ((map.size() > itemLimit) && (((multiplier + 1) * itemLimit) < map.size()))
264                                trimDeadwood(multiplier + 1); // addition to better enforce the
265                                                                                          // item limit w/o being crazy
266                }
267        }
268
269        @Override
270        public synchronized K put(T arg0, K arg1)
271        {
272                LinkedEntry<T, K> p = map.get(arg0);
273                if (p == null)
274                {
275                        p = new LinkedEntry<T, K>(arg0, arg1);
276                        map.put(arg0, p);
277                        if (tail == null)
278                        {
279                                head = p;
280                                tail = p;
281                                p.index = itemLimit;
282                        }
283                        else
284                        {
285                                p.index = tail.index - 1;
286                                tail.next = p;
287                                p.prev = tail;
288                                tail = p;
289                        }
290                }
291                else
292                {
293                        if (p.second != arg1)
294                                p.second = arg1;
295                        markFoundAgain(p);
296                }
297                if(!trimming.get())
298                {
299                        trimming.set(true);
300                        try
301                        {
302                                trimDeadwood(1);
303                        }
304                        finally
305                        {
306                                trimming.set(false);
307                        }
308                }
309                return arg1;
310        }
311
312        @Override
313        public synchronized void putAll(Map<? extends T, ? extends K> arg0)
314        {
315                for (final T t : arg0.keySet())
316                        put(t, arg0.get(t));
317        }
318
319        @Override
320        public synchronized K remove(Object arg0)
321        {
322                if (!map.containsKey(arg0))
323                        return null;
324                final LinkedEntry<T, K> p = map.remove(arg0);
325                if (map.size() == 0)
326                {
327                        head = null;
328                        tail = null;
329                        return p.second;
330                }
331                LinkedEntry<T, K> pn = p.next;
332                while ((pn != null) && (tail != pn))
333                {
334                        pn.index++;
335                        pn = pn.next;
336                }
337                if (head == p)
338                        head = p.next;
339                if (tail == p)
340                        tail = p.prev;
341                if (p.prev != null)
342                        p.prev.next = p.next;
343                if (p.next != null)
344                        p.next.prev = p.prev;
345                return p.second;
346        }
347
348        @Override
349        public int size()
350        {
351                return map.size();
352        }
353
354        @Override
355        public synchronized Collection<K> values()
356        {
357                final Collection<K> c = new Vector<K>(map.size());
358                for (final T t : map.keySet())
359                        c.add(map.get(t).second);
360                return c;
361        }
