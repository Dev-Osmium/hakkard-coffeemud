020public class SHashtable<K, F> implements CMap<K, F>, java.io.Serializable
021{
022        private static final long            serialVersionUID   = 6687178785122561993L;
023        private volatile Hashtable<K, F>        H;
024
025        public SHashtable()
026        {
027                super();
028                H = new Hashtable<K, F>();
029        }
030
031        public SHashtable(int size)
032        {
033                super();
034                H = new Hashtable<K, F>(size);
035        }
036
037        public SHashtable(Enumeration<Pair<K, F>> e)
038        {
039                super();
040                this.H = new Hashtable<K, F>();
041                if (H != null)
042                {
043                        for (; e.hasMoreElements();)
044                        {
045                                final Pair<K, F> p = e.nextElement();
046                                put(p.first, p.second);
047                        }
048                }
049        }
050
051        public SHashtable(Map<K, F> H)
052        {
053                super();
054                this.H = new Hashtable<K, F>();
055                if (H != null)
056                {
057                        for (final K o : H.keySet())
058                                put(o, H.get(o));
059                }
060        }
061
062        @SuppressWarnings("unchecked")
063        public SHashtable(Object[][] H)
064        {
065                super();
066                this.H = new Hashtable<K, F>();
067                if (H != null)
068                {
069                        for (final Object[] o : H)
070                                this.H.put((K) o[0], (F) o[1]);
071                }
072        }
073
074        @SuppressWarnings("unchecked")
075        public synchronized Hashtable<K, F> toHashtable()
076        {
077                return (Hashtable<K, F>) H.clone();
078        }
079
080        public synchronized Vector<String> toStringVector(String divider)
081        {
082                final Vector<String> V = new Vector<String>(size());
083                for (final Object S : keySet())
084                {
085                        if (S != null)
086                        {
087                                final Object O = get(S);
088                                if (O == null)
089                                        V.add(S.toString() + divider);
090                                else
091                                        V.add(S.toString() + divider + O.toString());
092                        }
093                }
094                return V;
095        }
096
097        @SuppressWarnings("unchecked")
098
099        @Override
100        public synchronized void clear()
101        {
102                H = (Hashtable<K, F>) H.clone();
103                H.clear();
104        }
105
106        @SuppressWarnings("unchecked")
107        public synchronized SHashtable<K, F> copyOf()
108        {
109                final SHashtable<K, F> SH = new SHashtable<K, F>();
110                SH.H = (Hashtable<K, F>) H.clone();
111                return SH;
112        }
113
114        public synchronized boolean contains(Object arg0)
115        {
116                return H.contains(arg0);
117        }
118
119        @Override
120        public synchronized boolean containsKey(Object arg0)
121        {
122                return H.containsKey(arg0);
123        }
124
125        @Override
126        public synchronized boolean containsValue(Object arg0)
127        {
128                return H.containsValue(arg0);
129        }
130
131        public synchronized Enumeration<F> elements()
132        {
133                return H.elements();
134        }
135
136        @Override
137        public synchronized Set<java.util.Map.Entry<K, F>> entrySet()
138        {
139                return H.entrySet();
140        }
141
142        @Override
143        public boolean equals(Object arg0)
144        {
145                return this == arg0;
146        }
147
148        @Override
149        public synchronized F get(Object arg0)
150        {
151                return H.get(arg0);
152        }
153
154        @Override
155        public int hashCode()
156        {
157                return super.hashCode();
158        }
159
160        @Override
161        public synchronized boolean isEmpty()
162        {
163                return H.isEmpty();
164        }
165
166        public synchronized Enumeration<K> keys()
167        {
168                return H.keys();
169        }
170
171        @Override
172        public synchronized Set<K> keySet()
173        {
174                return H.keySet();
175        }
176
177        @SuppressWarnings("unchecked")
178
179        @Override
180        public synchronized F put(K arg0, F arg1)
181        {
182                H = (Hashtable<K, F>) H.clone();
183                return H.put(arg0, arg1);
184        }
185
186        @SuppressWarnings("unchecked")
187
188        @Override
189        public synchronized F remove(Object arg0)
190        {
191                H = (Hashtable<K, F>) H.clone();
192                return H.remove(arg0);
193        }
194
195        @Override
196        public synchronized int size()
197        {
198                return H.size();
199        }
200
201        @Override
202        public String toString()
203        {
204                return super.toString();
205        }
206
207        @Override
208        public synchronized Collection<F> values()
209        {
210                return new ReadOnlyCollection<F>(H.values());
211        }
212
213        @SuppressWarnings("unchecked")
214
215        @Override
216        public synchronized void putAll(Map<? extends K, ? extends F> arg0)
217        {
218                H = (Hashtable<K, F>) H.clone();
219                H.putAll(arg0);
220        }
221
222}
