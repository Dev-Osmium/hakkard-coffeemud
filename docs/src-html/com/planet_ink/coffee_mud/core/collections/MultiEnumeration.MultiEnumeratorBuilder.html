024public class MultiEnumeration<K> implements Enumeration<K>
025{
026        private final LinkedList<Enumeration<K>> enums=new LinkedList<Enumeration<K>>();
027        private volatile Enumeration<K> enumer=null;
028
029        public static interface MultiEnumeratorBuilder<K>
030        {
031                public MultiEnumeration<K> getList();
032        }
033        
034        public MultiEnumeration(Enumeration<K>[] esets)
035        {
036                if((esets!=null)&&(esets.length>0))
037                        for(final Enumeration<K> E : esets)
038                                if(E!=null)
039                                        enums.add(E);
040        }
041
042        public MultiEnumeration(Collection<Enumeration<K>> esets)
043        {
044                if(esets!=null)
045                        enums.addAll(esets);
046        }
047
048        public MultiEnumeration(Enumeration<K> eset)
049        {
050                enums.add(eset);
051        }
052
053        public MultiEnumeration<K> addEnumeration(Enumeration<K> set)
054        {
055                if(set != null)
056                        enums.add(set);
057                return this;
058        }
059
060        @Override
061        public boolean hasMoreElements()
062        {
063                boolean hasMore = (enumer != null) && enumer.hasMoreElements();
064                while(!hasMore)
065                {
066                        if(enums.size()==0)
067                        {
068                                enumer=null;
069                                return false;
070                        }
071                        enumer=enums.removeFirst();
072                        hasMore = (enumer != null) && enumer.hasMoreElements();
073                }
074                return hasMore;
075        }
076
077        @Override
078        public K nextElement()
079        {
080                if(!hasMoreElements())
081                        throw new NoSuchElementException();
082                return enumer.nextElement();
083        }
084}
