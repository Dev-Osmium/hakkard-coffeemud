020public class FullConvertingList<L,K> implements List<K>
021{
022        private final List<L> list;
023        private final FullConverter<L, K> converter;
024        private Class<?> lClass = null;
025        private Class<?> kClass = null;
026
027        public FullConvertingList(List<L> l, FullConverter<L, K> conv)
028        {
029                list=l;
030                converter=conv;
031                for(Method M : conv.getClass().getMethods())
032                {
033                        if(M.getName().equals("reverseConvert") && (M.getParameterTypes().length>0))
034                        {
035                                lClass = M.getReturnType();
036                                kClass = M.getParameterTypes()[0];
037                                break;
038                        }
039                }
040        }
041
042        @Override
043        public boolean add(K arg0)
044        {
045                return list.add(converter.reverseConvert(arg0));
046        }
047
048        @Override
049        public void add(int arg0, K arg1)
050        {
051                list.add(arg0,converter.reverseConvert(arg1));
052        }
053        
054        @SuppressWarnings("unchecked")
055        protected Object convertToListType(final Object arg0)
056        {
057                if(arg0 == null)
058                        return null;
059                if(lClass.isAssignableFrom(arg0.getClass()))
060                        return arg0;
061                if(kClass.isAssignableFrom(arg0.getClass()))
062                        return converter.reverseConvert((K)arg0);
063                return null;
064        }
065
066        @Override
067        public boolean addAll(Collection<? extends K> arg0)
068        {
069                if(arg0==null)
070                        return false;
071                boolean didAll = true;
072                for(K arg01 : arg0)
073                        didAll = add(arg01) && didAll;
074                return didAll;
075        }
076
077        @Override
078        public boolean addAll(int arg0, Collection<? extends K> arg1)
079        {
080                if(arg1==null)
081                        return false;
082                for(K arg01 : arg1)
083                        add(arg0, arg01);
084                return true;
085        }
086
087        @Override
088        public void clear()
089        {
090                list.clear();
091        }
092
093        @Override
094        public boolean contains(Object arg0)
095        {
096                final Object o = convertToListType(arg0);
097                if(o == null)
098                        return false;
099                return list.contains(o);
100        }
101
102        @Override
103        public boolean containsAll(Collection<?> arg0)
104        {
105                if(arg0==null)
106                        return false;
107                for(Object o : arg0)
108                {
109                        o = convertToListType(o);
110                        if(o == null)
111                                return false;
112                        if(!list.contains(o))
113                                return false;
114                }
115                return true;
116        }
117
118        @Override
119        public K get(int arg0)
120        {
121                return converter.convert(arg0,list.get(arg0));
122        }
123
124        @Override
125        public int indexOf(Object arg0)
126        {
127                if(list.size()==0)
128                        return -1;
129                final Object o = convertToListType(arg0);
130                if(o == null)
131                        return -1;
132                return list.indexOf(o);
133        }
134
135        @Override
136        public boolean isEmpty()
137        {
138                return list.isEmpty();
139        }
140
141        @Override
142        public Iterator<K> iterator()
143        {
144                return new FullConvertingIterator<L, K>(list.iterator(),converter);
145        }
146
147        @Override
148        public int lastIndexOf(Object arg0)
149        {
150                if(list.size()==0)
151                        return -1;
152                final Object o = convertToListType(arg0);
153                if(o == null)
154                        return -1;
155                return list.lastIndexOf(o);
156        }
157
158        @Override
159        public ListIterator<K> listIterator()
160        {
161                return new FullConvertingListIterator<L,K>(list.listIterator(), converter);
162        }
163
164        @Override
165        public ListIterator<K> listIterator(int arg0)
166        {
167                return new FullConvertingListIterator<L,K>(list.listIterator(arg0), converter);
168        }
169
170        @Override
171        public boolean remove(Object arg0)
172        {
173                final Object o = convertToListType(arg0);
174                if(o == null)
175                        return false;
176                return list.remove(o);
177        }
178
179        @Override
180        public K remove(int arg0)
181        {
182                return converter.convert(arg0, list.remove(arg0));
183        }
184
185        @Override
186        public boolean removeAll(Collection<?> arg0)
187        {
188                if(arg0==null)
189                        return false;
190                boolean didAll = true;
191                for(Object o : arg0)
192                        didAll = remove(o) && didAll;
193                return didAll;
194        }
195
196        @Override
197        public boolean retainAll(Collection<?> arg0)
198        {
199                throw new java.lang.IllegalArgumentException();
200        }
201
202        @Override
203        public K set(int arg0, K arg1)
204        {
205                throw new java.lang.IllegalArgumentException();
206        }
207
208        @Override
209        public int size()
210        {
211                return list.size();
212        }
213
214        @Override
215        public List<K> subList(int arg0, int arg1)
216        {
217                return new FullConvertingList<L,K>(list.subList(arg0,arg1),converter);
218        }
219
220        @Override
221        public Object[] toArray()
222        {
223                final Object[] obj=new Object[list.size()];
224                for(int x=obj.length-1;x>=0;x--)
225                        obj[x]=converter.convert(x,list.get(x));
226                return obj;
227        }
228
229        @SuppressWarnings("unchecked")
230
231        @Override
232        public <T> T[] toArray(T[] a) 
233        {
234                if (a.length < list.size())
235                        return (T[]) Arrays.copyOf(toArray(), list.size(), a.getClass());
236                System.arraycopy(toArray(), 0, a, 0, list.size());
237                if (a.length > list.size())
238                        a[list.size()] = null;
239                return a;
240        }
241}
