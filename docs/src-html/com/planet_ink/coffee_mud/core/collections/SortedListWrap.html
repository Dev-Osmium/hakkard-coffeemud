001package com.planet_ink.coffee_mud.core.collections;
002
003import java.util.Collection;
004import java.util.Iterator;
005import java.util.List;
006import java.util.ListIterator;
007
008/**
009 * List wrapper for an existing List that turns it into a sorted one.
010 * @author Bo Zimmerman
011 *
012 * @param <T> the type?
013 */
014public class SortedListWrap<T extends Comparable<T>> implements List<T>
015{
016        private final List<T> list;
017        public SortedListWrap(List<T> list)
018        {
019                this.list=list;
020        }
021
022        @SuppressWarnings({"unchecked", "rawtypes"})
023        protected int compareTo(T arg0, Object arg1)
024        {
025                if(arg0 != null)
026                        return ((Comparable)arg0).compareTo(arg1);
027                return -1;
028        }
029
030        @Override
031        public synchronized boolean add(T arg0)
032        {
033                if(arg0==null)
034                        return false;
035                if(list.size()==0)
036                        return list.add(arg0);
037                int start=0;
038                int end=list.size()-1;
039                int comp=-1;
040                int mid=-1;
041                while(start<=end)
042                {
043                        mid=(end+start)/2;
044                        comp=compareTo(list.get(mid),arg0);
045                        if(comp==0)
046                                break;
047                        else
048                        if(comp>0)
049                                end=mid-1;
050                        else
051                                start=mid+1;
052                }
053                if(comp==0)
054                        list.add(mid,arg0);
055                else
056                if(comp>0)
057                {
058                        while((mid>=0)&&(compareTo(list.get(mid),arg0)>0))
059                                mid--;
060                        if(mid>=list.size()-1)
061                                list.add(arg0);
062                        else
063                        if(mid<0)
064                                list.add(0,arg0);
065                        else
066                                list.add(mid+1,arg0);
067                }
068                else
069                {
070                        while((mid<list.size())&&(compareTo(list.get(mid),arg0)<0))
071                                mid++;
072                        if(mid>=list.size())
073                                list.add(arg0);
074                        else
075                                list.add(mid,arg0);
076                }
077                return true;
078        }
079
080        @Override
081        public void add(int arg0, T arg1)
082        {
083                throw new java.lang.UnsupportedOperationException();
084        }
085
086        @Override
087        public synchronized boolean addAll(Collection<? extends T> arg0)
088        {
089                boolean tf=true;
090                for(final T t : arg0)
091                        tf=tf&&add(t);
092                return tf;
093        }
094
095        @Override
096        public boolean addAll(int arg0, Collection<? extends T> arg1)
097        {
098                throw new java.lang.UnsupportedOperationException();
099        }
100
101        @Override
102        public synchronized void clear()
103        {
104                list.clear();
105        }
106
107        @Override
108        public boolean contains(Object arg0)
109        {
110                return indexOf(arg0)>=0;
111        }
112
113        @Override
114        public boolean containsAll(Collection<?> arg0)
115        {
116                for(final Object o : arg0)
117                {
118                        if(!contains(o))
119                                return false;
120                }
121                return true;
122        }
123
124        @Override
125        public T get(int arg0)
126        {
127                return list.get(arg0);
128        }
129
130        @Override
131        public synchronized int indexOf(Object arg0)
132        {
133                if(list.size()==0)
134                        return -1;
135                int start=0;
136                int end=list.size()-1;
137                while(start<=end)
138                {
139                        final int mid=(end+start)/2;
140                        final int comp=compareTo(list.get(mid),arg0);
141                        if(comp==0)
142                                return mid;
143                        else
144                        if(comp>0)
145                                end=mid-1;
146                        else
147                                start=mid+1;
148                }
149                return -1;
150        }
151
152        @Override
153        public boolean isEmpty()
154        {
155                return list.isEmpty();
156        }
157
158        @Override
159        public Iterator<T> iterator()
160        {
161                return new ReadOnlyIterator<T>(list.iterator());
162        }
163
164        @Override
165        public synchronized int lastIndexOf(Object arg0)
166        {
167                int firstIndex=indexOf(arg0);
168                if(firstIndex<0)
169                        return -1;
170                while((firstIndex<list.size())&&(compareTo(list.get(firstIndex),arg0)==0))
171                        firstIndex++;
172                return firstIndex;
173        }
174
175        @Override
176        public ListIterator<T> listIterator()
177        {
178                return new ReadOnlyListIterator<T>(list.listIterator());
179        }
180
181        @Override
182        public ListIterator<T> listIterator(int arg0)
183        {
184                return new ReadOnlyListIterator<T>(list.listIterator(arg0));
185        }
186
187        @Override
188        public synchronized boolean remove(Object arg0)
189        {
190                final int index=indexOf(arg0);
191                if(index >= 0)
192                        return remove(index)==arg0;
193                return false;
194        }
195
196        @Override
197        public synchronized T remove(int arg0)
198        {
199                return list.remove(arg0);
200        }
201
202        @Override
203        public synchronized boolean removeAll(Collection<?> arg0)
204        {
205                return list.removeAll(arg0);
206        }
207
208        @Override
209        public synchronized boolean retainAll(Collection<?> arg0)
210        {
211                return list.retainAll(arg0);
212        }
213
214        @Override
215        public T set(int arg0, T arg1)
216        {
217                throw new java.lang.UnsupportedOperationException();
218        }
219
220        @Override
221        public int size()
222        {
223                return list.size();
224        }
225
226        @Override
227        public List<T> subList(int arg0, int arg1)
228        {
229                return list.subList(arg0, arg1);
230        }
231
232        @Override
233        public Object[] toArray()
234        {
235                return list.toArray();
236        }
237
238        @SuppressWarnings("hiding")
239
240        @Override
241        public <T> T[] toArray(T[] arg0)
242        {
243                return list.toArray(arg0);
244        }
245}
