044public class GetSys extends CM1Command
045{
046        @Override 
047        public String getCommandWord()
048        { 
049                return "GETSYS";
050        }
051        
052        public GetSys(RequestHandler req, String parameters)
053        {
054                super(req, parameters);
055        }
056
057        @Override
058        public void run()
059        {
060                try
061                {
062                        String type = parameters.toUpperCase().trim();
063                        if(type.startsWith("PROP "))
064                        {
065                                type = type.substring(5).trim();
066                                if(CMProps.isPropName(type))
067                                        req.sendMsg("[OK "+CMProps.getProp(type)+"]");
068                                else
069                                        req.sendMsg("[FAIL NO PROP "+type+"]");
070                        }
071                        else
072                        if(type.startsWith("DEBUG "))
073                        {
074                                type = type.substring(6).trim();
075                                if(CMParms.indexOfIgnoreCase(CMSecurity.DbgFlag.values(), type)>=0)
076                                        req.sendMsg("[OK "+CMSecurity.isDebuggingSearch(type)+"]");
077                                else
078                                        req.sendMsg("[FAIL NO DEBUG "+type+"]");
079                        }
080                        else
081                        if(type.startsWith("DISABLE "))
082                        {
083                                type = type.substring(8).trim();
084                                if(CMParms.indexOfIgnoreCase(CMSecurity.DisFlag.values(), type)>=0)
085                                        req.sendMsg("[OK "+CMSecurity.isAnyFlagDisabled(type)+"]");
086                                else
087                                        req.sendMsg("[FAIL NO DISABLE "+type+"]");
088                        }
089                        else
090                        if(type.startsWith("ENABLE "))
091                        {
092                                req.sendMsg("[FAIL NO ENABLE "+type+"]");
093                        }
094                        else
095                        {
096                                if(CMProps.isPropName(type))
097                                        req.sendMsg("[OK "+CMProps.getProp(type)+"]");
098                                else
099                                if(CMParms.indexOfIgnoreCase(CMSecurity.DbgFlag.values(), type)>=0)
100                                        req.sendMsg("[OK "+CMSecurity.isDebuggingSearch(type)+"]");
101                                else
102                                if(CMParms.indexOfIgnoreCase(CMSecurity.DisFlag.values(), type)>=0)
103                                        req.sendMsg("[OK "+CMSecurity.isAnyFlagDisabled(type)+"]");
104                                else
105                                        req.sendMsg("[FAIL NO SYS "+type+"]");
106                        }
107                }
108                catch(final Exception ioe)
109                {
110                        Log.errOut(className,ioe);
111                        req.close();
112                }
113        }
114
115        @Override
116        public boolean passesSecurityCheck(MOB user, PhysicalAgent target)
117        {
118                return ((user != null) && (CMSecurity.isASysOp(user)));
119        }
120        
121        @SuppressWarnings("rawtypes")
122        protected String getAllProps()
123        {
124                StringBuilder str= new StringBuilder("PROP [?], DEBUG [?], DISABLE [?], ENABLE [?] ");
125                for(Class<? extends Enum> c : CMProps.PROP_CLASSES)
126                {
127                        str.append(CMParms.toListString(c.getEnumConstants()));
128                        str.append(", ");
129                }
130                str.append(CMParms.toListString(CMSecurity.DbgFlag.class.getEnumConstants()));
131                str.append(", ");
132                str.append(CMParms.toListString(CMSecurity.DisFlag.class.getEnumConstants()));
133                str.append(", ");
134                return str.toString().substring(0,str.length()-2);
135        }
136        
137        @Override
138        public String getHelp(MOB user, PhysicalAgent target, String rest)
139        {
140                return "USAGE: "+getCommandWord()+" "+getAllProps();
141        }
142}
