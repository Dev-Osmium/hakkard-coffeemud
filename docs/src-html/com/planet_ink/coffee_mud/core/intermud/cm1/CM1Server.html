041public class CM1Server extends Thread
042{
043        private String           name = "";
044        private int              port = 27755;
045        private boolean          shutdownRequested = false;
046        private boolean          isShutdown = false;
047        private Selector         servSelector = null;
048        private final String iniFile;
049        private CMProps          page;
050        private ServerSocketChannel
051                                                 servChan = null;
052        private final SHashtable<SocketChannel,RequestHandler>
053                                                 handlers = new SHashtable<SocketChannel,RequestHandler>();
054
055        public CM1Server(String serverName, String iniFile)
056        {
057                super(serverName);
058                if(!loadPropPage(iniFile))
059                        throw new IllegalArgumentException();
060                final int serverPort = page.getInt("PORT");
061                this.iniFile=iniFile;
062                name=serverName+"@"+serverPort;
063                setName(name);
064                port=serverPort;
065                shutdownRequested = false;
066        }
067
068        public String getINIFilename() 
069        { 
070                return iniFile;
071        }
072
073        protected boolean loadPropPage(String iniFile)
074        {
075                if (page==null || !page.isLoaded())
076                {
077                        page=new CMProps (iniFile);
078                        if(!page.isLoaded())
079                        {
080                                Log.errOut(getName(),"failed to load " + iniFile);
081                                return false;
082                        }
083                }
084                return true;
085        }
086
087        @Override
088        public void run()
089        {
090                while(!shutdownRequested)
091                {
092                        try
093                        {
094                                servChan = ServerSocketChannel.open();
095                                final ServerSocket serverSocket = servChan.socket();
096                                servSelector = Selector.open();
097                                if((page.getStr("BIND")!=null)&&(page.getStr("BIND").trim().length()>0))
098                                        serverSocket.bind (new InetSocketAddress(InetAddress.getByName(page.getStr("BIND")),port));
099                                else
100                                        serverSocket.bind (new InetSocketAddress (port));
101                                Log.sysOut("Started "+name+" on port "+port);
102                                servChan.configureBlocking (false);
103                                servChan.register (servSelector, SelectionKey.OP_ACCEPT);
104                        }
105                        catch(final IOException e)
106                        {
107                                Log.errOut(e);
108                                Log.errOut("CM1Server failed to start.");
109                                shutdownRequested=true;
110                                break;
111                        }
112                        try
113                        {
114                                shutdownRequested = false;
115                                while (!shutdownRequested)
116                                {
117                                        try
118                                        {
119                                                final int n = servSelector.select();
120                                                if (n == 0) 
121                                                        continue;
122
123                                                final Iterator<SelectionKey> it = servSelector.selectedKeys().iterator();
124                                                while (it.hasNext())
125                                                {
126                                                        final SelectionKey key = it.next();
127                                                        if (key.isAcceptable())
128                                                        {
129                                                                final ServerSocketChannel server = (ServerSocketChannel) key.channel();
130                                                                final SocketChannel channel = server.accept();
131                                                                if (channel != null)
132                                                                {
133                                                                        final RequestHandler handler=new RequestHandler(channel,page.getInt("IDLETIMEOUTMINS"));
134                                                                        channel.configureBlocking (false);
135                                                                        channel.register (servSelector, SelectionKey.OP_READ, handler);
136                                                                        handlers.put(channel,handler);
137                                                                        handler.sendMsg("CONNECTED TO "+name.toUpperCase());
138                                                                }
139                                                                //sayHello (channel);
140                                                        }
141                                                        try
142                                                        {
143                                                                if (key.isReadable())
144                                                                {
145                                                                        final RequestHandler handler = (RequestHandler)key.attachment();
146                                                                        if((!handler.isRunning())&&(!handler.needsClosing()))
147                                                                                CMLib.threads().executeRunnable(handler);
148                                                                }
149                                                        }
150                                                        finally
151                                                        {
152                                                                it.remove();
153                                                        }
154                                                }
155                                                for(final SocketChannel schan : handlers.keySet())
156                                                {
157                                                        try
158                                                        {
159                                                                final RequestHandler handler=handlers.get(schan);
160                                                                if((handler!=null)&&(handler.needsClosing()))
161                                                                        handler.shutdown();
162                                                        }
163                                                        catch(final Exception e)
164                                                        {
165                                                        }
166                                                }
167                                        }
168                                        catch(final CancelledKeyException t)
169                                        {
170                                                // ignore
171                                        }
172                                }
173                        }
174                        catch(final Exception t)
175                        {
176                                Log.errOut("CM1Server",t);
177                        }
178                        finally
179                        {
180                                if(servSelector != null)
181                                {
182                                        try
183                                        {
184                                                servSelector.close();
185                                        }
186                                        catch(final Exception e)
187                                        {
188                                        }
189                                }
190                                if(servChan != null)
191                                {
192                                        try
193                                        {
194                                                servChan.close();
195                                        }
196                                        catch(final Exception e)
197                                        {
198                                        }
199                                }
200                                for(final SocketChannel schan : handlers.keySet())
201                                {
202                                        try
203                                        {
204                                                final RequestHandler handler=handlers.get(schan);
205                                                if(handler!=null)
206                                                        handler.shutdown();
207                                        }
208                                        catch(final Exception e)
209                                        {
210                                        }
211                                }
212                                handlers.clear();
213                                Log.sysOut("CM1Server","Shutdown complete");
214                        }
215                }
216                isShutdown = true;
217        }
218
219        public void shutdown()
220        {
221                shutdownRequested = true;
222                final long time = System.currentTimeMillis();
223                while((System.currentTimeMillis()-time<30000) && (!isShutdown))
224                {
225                        CMLib.s_sleep(1000);
226                        if(servSelector != null)
227                        {
228                                try 
229                                {
230                                        servSelector.close();
231                                }
232                                catch(final Exception e)
233                                {
234                                }
235                        }
236                        CMLib.s_sleep(1000);
237                        if((servChan != null)&&(!isShutdown))
238                        {
239                                try 
240                                {
241                                        servChan.close();
242                                }
243                                catch(final Exception e)
244                                {
245                                }
246                        }
247                        CMLib.s_sleep(1000);
248                        this.interrupt();
249                }
250        }
