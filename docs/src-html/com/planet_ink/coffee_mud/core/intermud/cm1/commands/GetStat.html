047        @Override
048        public String getCommandWord()
049        {

053        public GetStat(final RequestHandler req, final String parameters)
054        {
055                super(req, parameters);
056        }

058        protected static final String[] STATTYPES       = { "SESSION", "MOB", "CHAR", "STATE", "PHYSICAL", "BASECHAR", "MAXSTATE", "BASESTATE", "BASEPHYSICAL", "PLAYERSTATS", "ITEM", "EXIT", "ROOM", "AREA" };
059        protected static final String[] TYPESTYPE       = { "P", "MP", "MP", "MP", "MPIREA", "MP", "MP", "MP", "MPIREA", "P", "I", "E", "R", "A" };
060
061        protected static final String[] PHYSSTATS       = { "EFFECT" };
062        protected static final String[] PHYASTATS       = { "BEHAVIOR" };
063        protected static final String[] MOBASTATS       = { "ABILITY", "FACTION", "EXPERTISE", "FOLLOWER" };
064        protected static final String[] ITEMSTATS       = { "ITEM" };
065        protected static final String[] ROOMSTATS       = { "MOB" };
066
067        public char getTypeCode(final Physical P)
068        {
069                if (P instanceof MOB)
070                        return ((MOB) P).isMonster() ? 'M' : 'P';
071                if (P instanceof Item)
072                        return 'I';
073                if (P instanceof Room)
074                        return 'R';
075                if (P instanceof Exit)
076                        return 'E';
077                if (P instanceof Area)
078                        return 'A';
079                return ' ';
080        }
081
082        public boolean isApplicableTypeCode(final String type, final Physical P)
083        {
084                final char c = getTypeCode(P);
085                for (int i = 0; i < STATTYPES.length; i++)
086                {
087                        if (STATTYPES[i].equalsIgnoreCase(type))
088                                return TYPESTYPE[i].indexOf(c) >= 0;
089                }
090                return false;
091        }
092
093        public String[] getApplicableStatCodes(final Physical P)
094        {
095                final char c = getTypeCode(P);
096                final List<String> majorCodes = new LinkedList<String>();
097                for (int i = 0; i < STATTYPES.length; i++)
098                {
099                        if (TYPESTYPE[i].indexOf(c) >= 0)
100                                majorCodes.add(STATTYPES[i]);
101                }
102                return majorCodes.toArray(new String[0]);
103        }
104
105        public Modifiable getModifiable(final String type, final Physical P)
106        {
107                final int x = CMParms.indexOf(STATTYPES, type.toUpperCase().trim());
108                if (x < 0)
109                        return null;
110                if (!isApplicableTypeCode(type, P))
111                        return null;
112
113                switch (x)
114                {
115                case 0:
116                        return ((MOB) P).session();
117                case 1:
118                        return P;
119                case 2:
120                        return ((MOB) P).charStats();
121                case 3:
122                        return ((MOB) P).curState();
123                case 4:
124                        return P.phyStats();
125                case 5:
126                        return ((MOB) P).baseCharStats();
127                case 6:
128                        return ((MOB) P).maxState();
129                case 7:
130                        return ((MOB) P).baseState();
131                case 8:
132                        return P.basePhyStats();
133                case 9:
134                        return ((MOB) P).playerStats();
135                case 10:
136                        return P;
137                case 11:
138                        return P;
139                case 12:
140                        return P;
141                case 13:
142                        return P;
143                }
144                return null;
145        }
146
147        public boolean UseGenBuilder(final Physical P, final Modifiable m)
148        {
149                return (P != null) && (!P.isGeneric()) && ((m instanceof MOB) || (m instanceof Item));
150        }
151
152        public String[] getStatCodes(final Physical P, final Modifiable m)
153        {
154                SLinkedList<String> codes;
155                if (!UseGenBuilder(P, m))
156                        codes = new SLinkedList<String>(m.getStatCodes());
157                else
158                if (m instanceof MOB)
159                        codes = new SLinkedList<String>(CMParms.toStringArray(GenericBuilder.GenMOBCode.values()));
160                else
161                if (m instanceof Item)
162                        codes = new SLinkedList<String>(CMParms.toStringArray(GenericBuilder.GenItemCode.values()));
163                else
164                        return new String[0];
165                if (m instanceof Physical)
166                        codes.addAll(PHYSSTATS);
167                if (m instanceof PhysicalAgent)
168                        codes.addAll(PHYASTATS);
169                if (m instanceof MOB)
170                        codes.addAll(MOBASTATS);
171                if (m instanceof ItemPossessor)
172                        codes.addAll(ITEMSTATS);
173                if (m instanceof Room)
174                        codes.addAll(ROOMSTATS);
175                return codes.toArray(new String[0]);
176        }
177
178        public boolean isAStat(final Physical P, final Modifiable m, final String stat)
179        {
180                if (!UseGenBuilder(P, m))
181                        return m.isStat(stat);
182                final String[] codes = getStatCodes(P, m);
183                if (codes != null)
184                        for (final String code : codes)
185                                if (code.equalsIgnoreCase(stat))
186                                        return true;
187                return false;
188        }

190        public boolean isAuthorized(final MOB user, final PhysicalAgent target)
191        {
192                if (target instanceof MOB)
193                {
194                        if (CMLib.players().playerExists(target.Name()))
195                                return CMSecurity.isAllowed(user, user.location(), CMSecurity.SecFlag.CMDPLAYERS);
196                        return CMSecurity.isAllowed(user, user.location(), CMSecurity.SecFlag.CMDMOBS);
197                }
198                else
199                if (target instanceof Item)
200                        return CMSecurity.isAllowed(user, user.location(), CMSecurity.SecFlag.CMDITEMS);
201                else
202                if (target instanceof Room)
203                        return CMSecurity.isAllowed(user, user.location(), CMSecurity.SecFlag.CMDROOMS);
204                else
205                if (target instanceof Exit)
206                        return CMSecurity.isAllowed(user, user.location(), CMSecurity.SecFlag.CMDEXITS);
207                else
208                if (target instanceof Area)
209                        return CMSecurity.isAllowed(user, user.location(), CMSecurity.SecFlag.CMDAREAS);
210                else
211                        return false;
212        }

214        @Override
215        public void run()
216        {
217                try
218                {
219                        final PhysicalAgent P = req.getTarget();
220                        if (P == null)
221                        {
222                                req.sendMsg("[FAIL NO TARGET]");
223                                return;
224                        }
225                        if (!isAuthorized(req.getUser(), P))
226                        {
227                                req.sendMsg("[FAIL UNAUTHORIZED]");
228                                return;
229                        }
230                        String stat = "";
231                        String type = parameters.toUpperCase().trim();
232                        String rest = "";
233                        // char adjuster=' ';
234                        int x = parameters.indexOf(' ');
235                        if (x > 0)
236                        {
237                                type = parameters.substring(0, x).toUpperCase().trim();
238                                stat = parameters.substring(x + 1).toUpperCase().trim();
239                                x = stat.lastIndexOf(' ');
240                                if (x > 0)
241                                {
242                                        rest = stat.substring(x + 1).trim();
243                                        stat = stat.substring(0, x);
244                                }
245                                if ((stat.length() > 0) && (!Character.isLetter(stat.charAt(0))))
246                                {
247                                        // adjuster=stat.charAt(0);
248                                        stat = stat.substring(1);
249                                }
250                        }
251                        final Modifiable mod = getModifiable(type, P);
252                        if (mod == null)
253                        {
254                                req.sendMsg("[FAIL " + getHelp(req.getUser(), P, "") + "]");
255                                return;
256                        }
257                        if ((stat.length() == 0) || (!isAStat(P, mod, stat)))
258                        {
259                                req.sendMsg("[FAIL USAGE: GETSTAT " + type + " " + CMParms.toListString(getStatCodes(P, mod)) + "]");
260                                return;
261                        }
262                        if (mod instanceof Physical)
263                        {
264                                switch (CMParms.indexOf(PHYSSTATS, stat))
265                                {
266                                case -1:
267                                        break;
268                                case 0:
269                                {
270                                        if (rest.trim().length() == 0)
271                                                req.sendMsg("[OK " + ((Physical) mod).numEffects() + "]");
272                                        else
273                                        {
274                                                final Ability A = ((Physical) mod).fetchEffect(CMath.s_int(rest));
275                                                if (A == null)
276                                                        req.sendMsg("[FAIL NO EFFECT " + rest + "]");
277                                                else
278                                                        req.sendMsg("[OK " + A.ID() + " " + A.text() + "]");
279                                        }
280                                        return;
281                                }
282                                }
283                        }
284                        if (mod instanceof PhysicalAgent)
285                        {
286                                switch (CMParms.indexOf(PHYASTATS, stat))
287                                {
288                                case -1:
289                                        break;
290                                case 0:
291                                {
292                                        if (rest.trim().length() == 0)
293                                                req.sendMsg("[OK " + ((PhysicalAgent) mod).numBehaviors() + "]");
294                                        else
295                                        {
296                                                final Behavior A = ((PhysicalAgent) mod).fetchBehavior(CMath.s_int(rest));
297                                                if (A == null)
298                                                        req.sendMsg("[FAIL NO BEHAVIOR " + rest + "]");
299                                                else
300                                                        req.sendMsg("[OK " + A.ID() + " " + A.getParms() + "]");
301                                        }
302                                        return;
303                                }
304                                }
305                        }
306                        if (mod instanceof MOB)
307                        {
308                                switch (CMParms.indexOf(MOBASTATS, stat))
309                                {
310                                case -1:
311                                        break;
312                                case 0:
313                                {
314                                        if (rest.trim().length() == 0)
315                                                req.sendMsg("[OK " + ((MOB) mod).numAllAbilities() + "]");
316                                        else
317                                        {
318                                                final Ability A = ((MOB) mod).fetchAbility(CMath.s_int(rest));
319                                                if (A == null)
320                                                        req.sendMsg("[FAIL NO ABILITY " + rest + "]");
321                                                else
322                                                        req.sendMsg("[OK " + A.ID() + " " + A.proficiency() + " " + A.text() + "]");
323                                        }
324                                        return;
325                                }
326                                case 1:
327                                {
328                                        if (rest.trim().length() == 0)
329                                        {
330                                                final StringBuilder factions = new StringBuilder("");
331                                                for (final Enumeration<String> f = ((MOB) mod).factions(); f.hasMoreElements();)
332                                                        factions.append(' ').append(f);
333                                                req.sendMsg("[OK" + factions.toString() + "]");
334                                        }
335                                        else
336                                        {
337                                                final Faction F = CMLib.factions().getFaction(rest);
338                                                if (F == null)
339                                                        req.sendMsg("[FAIL " + rest + " NOT EXIST]");
340                                                else
341                                                {
342                                                        final int f = ((MOB) mod).fetchFaction(F.factionID());
343                                                        if (f == Integer.MAX_VALUE)
344                                                                req.sendMsg("[FAIL NO FACTION " + F.factionID() + "]");
345                                                        else
346                                                                req.sendMsg("[OK " + f + "]");
347                                                }
348                                        }
349                                        return;
350                                }
351                                case 2:
352                                {
353                                        if (rest.trim().length() == 0)
354                                        {
355                                                int numExpertises = 0;
356                                                for (final Enumeration<String> i = ((MOB) mod).expertises(); i.hasMoreElements(); numExpertises++)
357                                                        i.nextElement();
358                                                req.sendMsg("[OK " + numExpertises + "]");
359                                        }
360                                        else
361                                        {
362                                                int whichExpertise = CMath.s_int(rest);
363                                                final Enumeration<String> i = ((MOB) mod).expertises();
364                                                String EX = null;
365                                                while ((whichExpertise >= 0) && (i.hasMoreElements()))
366                                                {
367                                                        EX = i.nextElement();
368                                                        whichExpertise--;
369                                                }
370                                                if ((whichExpertise >= 0) || (EX == null))
371                                                        req.sendMsg("[FAIL NO EXPERTISE " + rest + "]");
372                                                else
373                                                        req.sendMsg("[OK " + EX + "]");
374                                        }
375                                        return;
376                                }
377                                case 3:
378                                {
379                                        if (rest.trim().length() == 0)
380                                                req.sendMsg("[OK " + ((MOB) mod).numFollowers() + "]");
381                                        else
382                                        {
383                                                final MOB M = ((MOB) mod).fetchFollower(CMath.s_int(rest));
384                                                if (M == null)
385                                                        req.sendMsg("[FAIL NO FOLLOWER " + rest + "]");
386                                                else
387                                                        req.sendMsg("[OK " + M.Name() + "]");
388                                        }
389                                        return;
390                                }
391                                }
392                        }
393                        if (mod instanceof ItemPossessor)
394                        {
395                                switch (CMParms.indexOf(ITEMSTATS, stat))
396                                {
397                                case -1:
398                                        break;
399                                case 0:
400                                {
401                                        if (rest.trim().length() == 0)
402                                                req.sendMsg("[OK " + ((ItemPossessor) mod).numItems() + "]");
403                                        else
404                                        {
405                                                final Item I = ((ItemPossessor) mod).getItem(CMath.s_int(rest));
406                                                if (I == null)
407                                                        req.sendMsg("[FAIL NO ITEM " + rest + "]");
408                                                else
409                                                        req.sendMsg("[OK " + I.Name() + "]");
410                                        }
411                                        return;
412                                }
413                                }
414                        }
415                        if (mod instanceof Room)
416                        {
417                                switch (CMParms.indexOf(ROOMSTATS, stat))
418                                {
419                                case -1:
420                                        break;
421                                case 0:
422                                {
423                                        if (rest.trim().length() == 0)
424                                                req.sendMsg("[OK " + ((Room) mod).numInhabitants() + "]");
425                                        else
426                                        {
427                                                final MOB M = ((Room) mod).fetchInhabitant(CMath.s_int(rest));
428                                                if (M == null)
429                                                        req.sendMsg("[FAIL NO MOB " + rest + "]");
430                                                else
431                                                        req.sendMsg("[OK " + M.Name() + "]");
432                                        }
433                                        return;
434                                }
435                                }
436                        }
437                        if (!UseGenBuilder(P, mod))
438                        {
439                                req.sendMsg("[OK " + mod.getStat(stat) + "]");
440                        }
441                        else
442                        {
443                                final String[] codes = this.getStatCodes(P, mod);
444                                for (final String code : codes)
445                                {
446                                        if (code.equalsIgnoreCase(stat))
447                                        {
448                                                if (P instanceof MOB)
449                                                        req.sendMsg("[OK " + CMLib.coffeeMaker().getGenMobStat((MOB) P, stat) + "]");
450                                                else
451                                                if (P instanceof Item)
452                                                        req.sendMsg("[OK " + CMLib.coffeeMaker().getGenItemStat((Item) P, stat) + "]");
453                                        }
454                                }
455                        }
456                }
457                catch (final Exception ioe)
458                {
459                        Log.errOut(className, ioe);
460                        req.close();
461                }
462        }

470        @Override
471        public String getHelp(final MOB user, final PhysicalAgent target, String rest)
472        {
473                Modifiable mod = null;
474                if ((rest != null) && (rest.trim().length() > 0))
475                {
476                        final int x = rest.indexOf(' ');
477                        if (x > 0)
478                                rest = rest.substring(0, x).toUpperCase().trim();
479                        if (isApplicableTypeCode(rest, target))
480                                mod = getModifiable(rest.toUpperCase().trim(), target);
481                }
482                if (mod == null)
483                        return "USAGE: " + getCommandWord() + " " + CMParms.toListString(getApplicableStatCodes(target));
484                else
485                if (rest != null)
486                        return "USAGE: " + getCommandWord() + " " + rest.toUpperCase().trim() + " " + CMParms.toListString(getStatCodes(target, mod));
487                else
488                        return "USAGE: " + getCommandWord() + " " + CMParms.toListString(getStatCodes(target, mod));
489        }
