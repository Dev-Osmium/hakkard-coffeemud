024 * A PersistenceException occurs whenever some error
025 * condition interrupts a persistence operation.
026 * This class maintains a chain of exceptions so
027 * that an easy trace can be done to find out what
028 * caused the problem.
029 */
030
031import java.util.Enumeration;
032import java.util.NoSuchElementException;
033
034/**
035 * An exception that gets thrown on persistence operations.
036 * If it was triggered by a specific exception, that
037 * exception will be added to the exception chain so
038 * any class catching this exception can get detailed
039 * information on what went wrong.
040 * @author George Reese (borg@imaginary.com)
041 * @version 1.0
042 */
043public class PersistenceException extends Exception
044{
045        public static final long        serialVersionUID        = 0333444555666L;
046        private final Exception         prior;
047
048        public PersistenceException()
049        {
050                this("No reason given.");
051        }
052
053        /**
054         * Constructs a new PersistenceException with the
055         * specified exception explanation.
056         * @param reason the reason for the exception
057         */
058        public PersistenceException(String reason)
059        {
060                this(reason, null);
061        }
062
063        /**
064         * Constructs a new PersistenceException that occured
065         * because another exception was encountered during a
066         * persistence operation such as a save or restore.
067         * @param e the exception causing this exception to be created
068         */
069        public PersistenceException(Exception e)
070        {
071                this("A persistence exception occurred: " + e.getMessage(), e);
072        }
073
074        /**
075         * Constructs a new PersistenceException based on a previous
076         * exception during a persistence operation with the specified
077         * exception explanation.
078         * @param reason the explanation for the exception
079         * @param e the exception causing this exception to be created
080         */
081        public PersistenceException(String reason, Exception e)
082        {
083                super(reason);
084                prior = e;
085        }
086
087        /**
088         * @return the exception which caused this one to be created
089         */
090        public Exception getPriorException()
091        {
092                return prior;
093        }
094
095        /**
096         * @return the full chain of exceptions leading to this one
097         */
098        public Enumeration<Exception> getExceptionChain()
099        {
100                return new PersistenceExceptionEnumeration(this);
101        }
102}
103
104final class PersistenceExceptionEnumeration implements Enumeration<Exception> 
105{
106        private Exception exception;
107
108        public PersistenceExceptionEnumeration()
109        {
110                this(null);
111        }
112
113        public PersistenceExceptionEnumeration(Exception e)
114        {
115                exception = e;
116        }
117
118        @Override
119        public boolean hasMoreElements()
120        {
121                if( exception == null )
122                        return false;
123                return true;
124        }
125
126        @Override
127        public Exception nextElement()
128        {
129                Exception e;
130
131                if( exception == null )
132                        throw new NoSuchElementException();
133                e = exception;
134                if( e instanceof PersistenceException )
135                        exception = ((PersistenceException)e).getPriorException();
136                else exception = null;
137                return e;
138        }
139}
