036public class FingerReply extends Packet 
037{
038        public String visible_name="";
039        public String title = "";
040        public String real_name = "";
041        public String e_mail="";
042        public String loginout_time = "";
043        public String idle_time = "";
044        public String ip_time = "";
045        public String extra = "";
046
047        public FingerReply()
048        {
049                super();
050                type = Packet.FINGER_REPLY;
051        }
052
053        public FingerReply(String to_whom, String mud)
054        {
055                super();
056                type = Packet.FINGER_REPLY;
057                target_mud = mud;
058                target_name = to_whom;
059        }
060
061        public FingerReply(Vector<?> v) throws InvalidPacketException
062        {
063                super(v);
064                try
065                {
066                        type = Packet.FINGER_REPLY;
067                        try
068                        {
069                                visible_name = v.elementAt(6).toString();
070                                title = v.elementAt(7).toString();
071                                real_name = v.elementAt(8).toString();
072                                e_mail = v.elementAt(9).toString();
073                                loginout_time = v.elementAt(10).toString();
074                                idle_time = v.elementAt(11).toString();
075                                ip_time = v.elementAt(12).toString();
076                                extra = v.elementAt(13).toString();
077                        }
078                        catch (final Exception e)
079                        {
080                        }
081                }
082                catch( final ClassCastException e )
083                {
084                        throw new InvalidPacketException();
085                }
086        }
087
088        @Override
089        public void send() throws InvalidPacketException 
090        {
091                super.send();
092        }
093
094        @Override
095        public String toString()
096        {
097                String cmd = "({\"finger-reply\",5,\"" + I3Server.getMudName() +
098                                 "\",0,\"" + target_mud + "\",\"" + target_name + "\",";
099                final String[] responses={ visible_name, title, real_name, e_mail,
100                                                        loginout_time, idle_time, ip_time, extra };
101                for(final String nom : responses)
102                {
103                        if(nom.length()==0)
104                                cmd += "0,";
105                        else
106                        if(CMath.isNumber(nom))
107                                cmd += "" + nom + ",";
108                        else
109                                cmd += "\"" + nom + "\",";
110                }
111                cmd += "})";
112                return cmd;
113
114        }
115}
