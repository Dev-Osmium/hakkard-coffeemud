025 * to some data store.
026 */
027
028/**
029 * Specified methods which must be implemented by
030 * objects wishing to be saved.
031 * @author George Reese (borg@imaginary.com)
032 * @version 1.0
033 */
034public interface Persistent 
035{
036        /**
037         * The Persistent has not yet been modified since last
038         * save or restore.
039         */
040        static public final int UNMODIFIED = 0;
041        /**
042         * The Peersistence has been modified since last save
043         * or restore.
044         */
045        static public final int MODIFIED = 1;
046        /**
047         * The Persistent is a brand new object.
048         */
049        static public final int NEW = 2;
050        /**
051         * The Persistent needs to be deleted from the data store.
052         */
053        static public final int DELETED = 3;
054
055        /**
056         * Prescribes a method for restoration from a data
057         * store.  An implementation will usually call the
058         * object's PersistentPeer method to perform the
059         * actual save.
060         * @throws PersistenceException thrown when an error occurs in restoring
061         * @see com.planet_ink.coffee_mud.core.intermud.i3.persist.PersistentPeer
062         */
063        public abstract void restore() throws PersistenceException;
064
065        /**
066         * Prescribes a method for saving this object's data
067         * to a data store.  An implementation will usually
068         * check to see if the object has been modified and
069         * trigger the actual saving mechanism in its
070         * PersistentPeer implementation.
071         * @throws PersistenceException thrown when an error occurs in saving
072         * @see com.planet_ink.coffee_mud.core.intermud.i3.persist.PersistentPeer
073         */
074        public abstract void save() throws PersistenceException;
075}
