047@SuppressWarnings({"unchecked","rawtypes"})
048public class IMudPeer implements PersistentPeer
049{
050        Object myobj=null;
051        boolean isRestoring=false;
052        String myID="";
053
054        /**
055         * Gets data about this peer from storage and gives it
056         * back to the object for which this peer exists.
057         * @exception com.planet_ink.coffee_mud.core.intermud.i3.persist.PersistenceException if an error occurs during restore
058         */
059        @Override
060        public void restore() throws PersistenceException
061        {
062                isRestoring=true;
063                if(myobj instanceof Intermud)
064                {
065                        try
066                        {
067                                final CMFile F=new CMFile("resources/ppeer."+myID,null);
068                                if(!F.exists())
069                                        return;
070
071                                final ObjectInputStream in=new ObjectInputStream(new ByteArrayInputStream(F.raw()));
072                                Object newobj;
073                                newobj=in.readObject();
074                                if(newobj instanceof Integer)
075                                        ((Intermud)myobj).password=((Integer)newobj).intValue();
076                                newobj=in.readObject();
077                                if(newobj instanceof Hashtable)
078                                        ((Intermud)myobj).banned=(Hashtable)newobj;
079                                newobj=in.readObject();
080                                if(newobj instanceof ChannelList)
081                                        ((Intermud)myobj).channels=(ChannelList)newobj;
082                                newobj=in.readObject();
083                                if(newobj instanceof MudList)
084                                        ((Intermud)myobj).muds=(MudList)newobj;
085                                newobj=in.readObject();
086                                if(newobj instanceof List)
087                                ((Intermud)myobj).name_servers=(List)newobj;
088                        }
089                        catch(final Exception e)
090                        {
091                                Log.errOut("IMudPeer","Unable to read /resources/ppeer."+myID);
092                        }
093                }
094                isRestoring=false;
095        }
096
097        /**
098         * Triggers a save of its peer.  Implementing classes
099         * should do whatever it takes to save the object in
100         * this method.
101         * @exception com.planet_ink.coffee_mud.core.intermud.i3.persist.PersistenceException if a problem occurs in saving
102         */
103        @Override
104        public void save() throws PersistenceException
105        {
106                if(myobj instanceof Intermud)
107                {
108                        try
109                        {
110                                final ByteArrayOutputStream bout=new ByteArrayOutputStream();
111                                final ObjectOutputStream out=new ObjectOutputStream(bout);
112                                out.writeObject(Integer.valueOf(((Intermud)myobj).password));
113                                out.writeObject(((Intermud)myobj).banned);
114                                out.writeObject(((Intermud)myobj).channels);
115                                out.writeObject(((Intermud)myobj).muds);
116                                out.writeObject(((Intermud)myobj).name_servers);
117                                out.flush();
118                                bout.flush();
119                                new CMFile("::resources/ppeer."+myID,null).saveRaw(bout.toByteArray());
120                                out.close();
121                                bout.close();
122                        }
123                        catch(final Exception e)
124                        {
125                                Log.errOut("IMudPeer",e.getMessage());
126                        }
127                }
128        }
129
130        /**
131         * Assigns a persistent object to this peer for
132         * persistence operations.
133         * @param ob the implementation of com.planet_ink.coffee_mud.core.intermud.i3.persist.Persistent that this is a peer for
134         * @see com.planet_ink.coffee_mud.core.intermud.i3.persist.Persistent
135         */
136        @Override
137        public void setPersistent(Persistent ob)
138        {
139                myobj=ob;
140                myID=ob.getClass().getName().substring(ob.getClass().getName().lastIndexOf('.')+1);
141        }
142
143        /**
144         * An implementation uses this to tell its Persistent
145         * that it is in the middle of restoring.
146         * @return true if a restore operation is in progress
147         */
148        @Override
149        public boolean isRestoring()
150        {
151                return isRestoring;
152        }
153}
