168        public String fixColors(String str)
169        {
170                final StringBuffer buf=new StringBuffer(str);
171                int startedAt=-1;
172                for(int i=0;i<buf.length();i++)
173                {
174                        if(buf.charAt(i)=='%')
175                        {
176                                if(startedAt<0)
177                                        startedAt=i;
178                                else
179                                if(((i+1)<buf.length())&&(buf.charAt(i+1)=='^'))
180                                {
181                                        String found=null;
182                                        final String code=buf.substring(startedAt,i+2);
183                                        for (final String[] element : i3ansi_conversion)
184                                        {
185                                                if(code.equals(element[1]))
186                                                {
187                                                        found=element[0];
188                                                        break;
189                                                }
190                                        }
191                                        if(found!=null)
192                                        {
193                                                buf.replace(startedAt,i+2,found);
194                                                i=startedAt+1;
195                                        }
196                                        startedAt=-1;
197                                }
198                        }
199                }
200                return buf.toString();
201        }

203        public String socialFixIn(String str)
204        {
205
206                str=CMStrings.replaceAll(str,"$N","<S-NAME>");
207                str=CMStrings.replaceAll(str,"$n","<S-NAME>");
208                str=CMStrings.replaceAll(str,"$T","<T-NAMESELF>");
209                str=CMStrings.replaceAll(str,"$t","<T-NAMESELF>");
210                str=CMStrings.replaceAll(str,"$O","<T-NAMESELF>");
211                str=CMStrings.replaceAll(str,"$o","<T-NAMESELF>");
212                str=CMStrings.replaceAll(str,"$m","<S-HIM-HER>");
213                str=CMStrings.replaceAll(str,"$M","<T-HIM-HER>");
214                str=CMStrings.replaceAll(str,"$s","<S-HIS-HER>");
215                str=CMStrings.replaceAll(str,"$S","<T-HIS-HER>");
216                str=CMStrings.replaceAll(str,"$e","<S-HE-SHE>");
217                str=CMStrings.replaceAll(str,"$E","<T-HE-SHE>");
218                str=CMStrings.replaceAll(str,"`","\'");
219                if(str.equals("$"))
220                        return "";
221                return str.trim();
222        }

236        /**
237         * Handles an incoming I3 packet asynchronously.
238         * An implementation should make sure that asynchronously
239         * processing the incoming packet will not have any
240         * impact, otherwise you could end up with bizarre
241         * behaviour like an intermud chat line appearing
242         * in the middle of a room description.  If your
243         * mudlib is not prepared to handle multiple threads,
244         * just stack up incoming packets and pull them off
245         * the stack during your main thread of execution.
246         * @param packet the incoming packet
247         */
248        @Override
249        public void receive(Packet packet)
250        {
251                switch(packet.type)
252                {
253                case Packet.CHAN_EMOTE:
254                case Packet.CHAN_MESSAGE:
255                case Packet.CHAN_TARGET:
256                        {
257                                lastPacketReceivedTime=System.currentTimeMillis();
258                                final ChannelPacket ck=(ChannelPacket)packet;
259                                final String channelName=ck.channel;
260                                CMMsg msg=null;
261
262                                if((ck.sender_mud!=null)&&(ck.sender_mud.equalsIgnoreCase(getMudName())))
263                                        return;
264                                if((ck.channel==null)||(ck.channel.length()==0))
265                                        return;
266                                final String channelColor="^Q";
267                                final int channelInt=CMLib.channels().getChannelIndex(channelName);
268                                int channelCode=channelInt;
269                                if(channelInt < 0)
270                                {
271                                        channelCode=47;
272                                }
273                                ck.message=fixColors(CMProps.applyINIFilter(ck.message,CMProps.Str.CHANNELFILTER));
274                                if(ck.message_target!=null)
275                                        ck.message_target=fixColors(CMProps.applyINIFilter(ck.message_target,CMProps.Str.CHANNELFILTER));
276                                final MOB mob=CMClass.getFactoryMOB();
277                                mob.setName(ck.sender_name+"@"+ck.sender_mud);
278                                mob.setLocation(getUniversalRoom());
279                                MOB targetMOB=null;
280                                boolean killtargetmob=false;
281                                if(ck.type==Packet.CHAN_TARGET)
282                                {
283                                        if((ck.target_mud!=null)&&(ck.target_mud.equalsIgnoreCase(getMudName())))
284                                                targetMOB=CMLib.players().getLoadPlayer(ck.target_name);
285                                        if((ck.target_visible_name!=null)&&(ck.target_mud!=null)&&(targetMOB==null))
286                                        {
287                                                killtargetmob=true;
288                                                targetMOB=CMClass.getFactoryMOB();
289                                                targetMOB.setName(ck.target_visible_name+"@"+ck.target_mud);
290                                                targetMOB.setLocation(getUniversalRoom());
291                                        }
292                                        String msgs=socialFixIn(ck.message);
293                                        msgs=CMProps.applyINIFilter(msgs,CMProps.Str.EMOTEFILTER);
294                                        String targmsgs=socialFixIn(ck.message_target);
295                                        targmsgs=CMProps.applyINIFilter(targmsgs,CMProps.Str.EMOTEFILTER);
296                                        final String str=channelColor+"^<CHANNEL \""+channelName+"\"^>["+channelName+"] "+msgs+"^</CHANNEL^>^N^.";
297                                        final String str2=channelColor+"^<CHANNEL \""+channelName+"\"^>["+channelName+"] "+targmsgs+"^</CHANNEL^>^N^.";
298                                        msg=CMClass.getMsg(mob,targetMOB,null,CMMsg.NO_EFFECT,null,CMMsg.MASK_CHANNEL|(CMMsg.TYP_CHANNEL+channelCode),str2,CMMsg.MASK_CHANNEL|(CMMsg.TYP_CHANNEL+channelCode),str);
299                                }
300                                else
301                                if(ck.type==Packet.CHAN_EMOTE)
302                                {
303                                        String msgs=socialFixIn(ck.message);
304                                        msgs=CMProps.applyINIFilter(msgs,CMProps.Str.EMOTEFILTER);
305                                        final String str=channelColor+"^<CHANNEL \""+channelName+"\"^>["+channelName+"] "+msgs+"^</CHANNEL^>^N^.";
306                                        msg=CMClass.getMsg(mob,null,null,CMMsg.NO_EFFECT,null,CMMsg.NO_EFFECT,null,CMMsg.MASK_CHANNEL|(CMMsg.TYP_CHANNEL+channelCode),str);
307                                }
308                                else
309                                {
310                                        final String str=channelColor+"^<CHANNEL \""+channelName+"\"^>"+mob.name()+" "+channelName+"(S) '"+ck.message+"'^</CHANNEL^>^N^.";
311                                        msg=CMClass.getMsg(mob,null,null,CMMsg.NO_EFFECT,null,CMMsg.NO_EFFECT,null,CMMsg.MASK_CHANNEL|(CMMsg.TYP_CHANNEL+channelCode),str);
312                                }
313                                CMLib.commands().monitorGlobalMessage(mob.location(), msg);
314                                if(channelInt>=0)
315                                        CMLib.channels().channelQueUp(channelInt,msg);
316                                for(final Session S : CMLib.sessions().localOnlineIterable())
317                                {
318                                        final MOB M=S.mob();
319                                        if(((channelInt<0)||CMLib.channels().mayReadThisChannel(mob,false,S,channelInt))
320                                        &&(M.location()!=null)
321                                        &&(M.location().okMessage(M,msg)))
322                                                M.executeMsg(M,msg);
323                                }
324                                mob.destroy();
325                                if((targetMOB!=null)&&(killtargetmob))
326                                        targetMOB.destroy();
327                        }
328                        break;
329                case Packet.LOCATE_QUERY:
330                        {
331                                lastPacketReceivedTime=System.currentTimeMillis();
332                                final LocateQueryPacket lk=(LocateQueryPacket)packet;
333                                String stat="online";
334                                String name=CMStrings.capitalizeAndLower(lk.user_name);
335                                final MOB smob=findSessMob(lk.user_name);
336                                if(smob!=null)
337                                {
338                                        if(CMLib.flags().isCloaked(smob))
339                                                stat="exists, but not logged in";
340                                }
341                                else
342                                if(CMLib.players().getPlayer(lk.user_name)!=null)
343                                        stat="exists, but not logged in";
344                                else
345                                if(CMLib.players().playerExists(lk.user_name))
346                                        stat="exists, but is not online";
347                                else
348                                        name=null;
349                                if(name!=null)
350                                {
351                                        final LocateReplyPacket lpk=new LocateReplyPacket(lk.sender_name,lk.sender_mud,name,0,stat);
352                                        try
353                                        {
354                                                lpk.send();
355                                        }
356                                        catch (final Exception e)
357                                        {
358                                                Log.errOut("IMudClient", e);
359                                        }
360                                }
361                        }
362                        break;
363                case Packet.LOCATE_REPLY:
364                        {
365                                lastPacketReceivedTime=System.currentTimeMillis();
366                                final LocateReplyPacket lk=(LocateReplyPacket)packet;
367                                final MOB smob=findSessMob(lk.target_name);
368                                if(smob!=null)
369                                        smob.tell(fixColors(lk.located_visible_name)+"@"+fixColors(lk.located_mud_name)+" ("+lk.idle_time+"): "+fixColors(lk.status));
370                        }
371                        break;
372                case Packet.FINGER_REQUEST:
373                        {
374                                lastPacketReceivedTime=System.currentTimeMillis();
375                                final FingerRequest lk=(FingerRequest)packet;
376                                Packet pkt;
377                                final MOB M=CMLib.players().getLoadPlayer(lk.target_name);
378                                if(M==null)
379                                        pkt=new ErrorPacket(lk.sender_name,lk.sender_mud,"unk-user","User "+lk.target_name+" is not known here.","0");
380                                else
381                                {
382                                        final FingerReply fpkt = new FingerReply(lk.sender_name,lk.sender_mud);
383                                        pkt=fpkt;
384                                        fpkt.e_mail="0";
385                                        final Session sess=M.session();
386                                        if((sess==null)||(!sess.isAfk()))
387                                                fpkt.idle_time="-1";
388                                        else
389                                                fpkt.idle_time=Long.toString(sess.getIdleMillis()/1000);
390                                        fpkt.ip_time="0"; // what IS this?
391                                        fpkt.loginout_time=CMLib.time().date2String(M.playerStats().getLastDateTime());
392                                        fpkt.real_name="0"; // don't even know this
393                                        if(M.titledName().equals(M.name()))
394                                                fpkt.title="An ordinary "+M.charStats().displayClassName();
395                                        else
396                                                fpkt.title=M.titledName();
397                                        fpkt.visible_name=M.name();
398                                        fpkt.extra=M.name()+" is a "+M.charStats().raceName()+" "+M.charStats().displayClassName();
399                                }
400                                try
401                                {
402                                        pkt.send();
403                                }
404                                catch (final Exception e)
405                                {
406                                        Log.errOut("IMudClient", e);
407                                }
408                        }
409                        break;
410                case Packet.FINGER_REPLY:
411                        {
412                                lastPacketReceivedTime=System.currentTimeMillis();
413                                final FingerReply lk=(FingerReply)packet;
414                                final MOB smob=findSessMob(lk.target_name);
415                                if(smob!=null)
416                                {
417                                        final StringBuilder response=new StringBuilder("");
418                                        if((lk.visible_name.length()>0)&&(!lk.visible_name.equals("0")))
419                                                response.append("^H").append(CMStrings.padRight(L("Name"),10)).append(": ^N").append(lk.visible_name).append("\n\r");
420                                        if((lk.title.length()>0)&&(!lk.title.equals("0")))
421                                                response.append("^H").append(CMStrings.padRight(L("Title"),10)).append(": ^N").append(lk.title).append("\n\r");
422                                        if((lk.real_name.length()>0)&&(!lk.real_name.equals("0")))
423                                                response.append("^H").append(CMStrings.padRight(L("Real Name"),10)).append(": ^N").append(lk.real_name).append("\n\r");
424                                        if((lk.e_mail.length()>0)&&(!lk.e_mail.equals("0")))
425                                                response.append("^H").append(CMStrings.padRight(L("Email"),10)).append(": ^N").append(lk.e_mail).append("\n\r");
426                                        if((lk.loginout_time.length()>0)&&(!lk.loginout_time.equals("0")))
427                                                response.append("^H").append(CMStrings.padRight(L("Logged"),10)).append(": ^N").append(lk.loginout_time).append("\n\r");
428                                        if((lk.ip_time.length()>0)&&(!lk.ip_time.equals("0")))
429                                                response.append("^H").append(CMStrings.padRight(L("IP Time"),10)).append(": ^N").append(lk.ip_time).append("\n\r");
430                                        if((lk.extra.length()>0)&&(!lk.extra.equals("0")))
431                                                response.append("^H").append(CMStrings.padRight(L("Extra"),10)).append(": ^N").append(lk.extra).append("\n\r");
432                                        smob.tell(response.toString());
433                                }
434                        }
435                        break;
436                case Packet.MAUTH_REQUEST:
437                        {
438                                lastPacketReceivedTime=System.currentTimeMillis();
439                                final MudAuthRequest lk=(MudAuthRequest)packet;
440                                if(lk.sender_mud.equalsIgnoreCase(I3Server.getMudName()))
441                                {
442                                        if(CMSecurity.isDebugging(DbgFlag.I3))
443                                                Log.debugOut("I3","Received my own mud-auth.");
444                                }
445                                else
446                                        Log.sysOut("I3","MUD "+lk.sender_mud+" wants to mud-auth.");
447                                final MudAuthReply pkt = new MudAuthReply(lk.sender_mud, System.currentTimeMillis());
448                                try
449                                {
450                                        pkt.send();
451                                }
452                                catch (final Exception e)
453                                {
454                                        Log.errOut("IMudClient", e);
455                                }
456                        }
457                        break;
458                case Packet.MAUTH_REPLY:
459                        {
460                                lastPacketReceivedTime=System.currentTimeMillis();
461                                final MudAuthReply lk=(MudAuthReply)packet;
462                                if(lk.sender_mud.equalsIgnoreCase(I3Server.getMudName()))
463                                {
464                                        if(CMSecurity.isDebugging(DbgFlag.I3))
465                                                Log.debugOut("I3","I replied to my mud-auth.");
466                                        lastPacketReceivedTime=System.currentTimeMillis();
467                                }
468                                else
469                                        Log.sysOut("I3","MUD "+lk.sender_mud+" replied to my mud-auth with key "+lk.key+".");
470                        }
471                        break;
472                case Packet.WHO_REPLY:
473                        {
474                                lastPacketReceivedTime=System.currentTimeMillis();
475                                final WhoPacket wk=(WhoPacket)packet;
476                                final MOB smob=findSessMob(wk.target_name);
477                                if(smob!=null)
478                                {
479                                        final StringBuffer buf=new StringBuffer("\n\rwhois@"+fixColors(wk.sender_mud)+":\n\r");
480                                        final Vector V=wk.who;
481                                        if(V.size()==0)
482                                                buf.append("Nobody!");
483                                        else
484                                        for(int v=0;v<V.size();v++)
485                                        {
486                                                final Vector V2=(Vector)V.elementAt(v);
487                                                final String nom = fixColors((String)V2.elementAt(0));
488                                                int idle=0;
489                                                if(V2.elementAt(1) instanceof Integer)
490                                                        idle = ((Integer)V2.elementAt(1)).intValue();
491                                                final String xtra = fixColors(V2.elementAt(2).toString());
492                                                buf.append("["+CMStrings.padRight(nom,20)+"] "+xtra+" ("+idle+")\n\r");
493                                        }
494                                        smob.session().wraplessPrintln(buf.toString());
495                                        break;
496                                }
497                        }
498                        break;
499                case Packet.CHAN_WHO_REP:
500                        {
501                                lastPacketReceivedTime=System.currentTimeMillis();
502                                final ChannelWhoReply wk=(ChannelWhoReply)packet;
503                                final MOB smob=findSessMob(wk.target_name);
504                                if(smob!=null)
505                                {
506                                        final StringBuffer buf=new StringBuffer("\n\rListening on "+wk.channel+"@"+fixColors(wk.sender_mud)+":\n\r");
507                                        final Vector V=wk.who;
508                                        if(V.size()==0)
509                                                buf.append("Nobody!");
510                                        else
511                                        for(int v=0;v<V.size();v++)
512                                        {
513                                                final String nom = fixColors((String)V.elementAt(v));
514                                                buf.append("["+CMStrings.padRight(nom,20)+"]\n\r");
515                                        }
516                                        smob.session().wraplessPrintln(buf.toString());
517                                        smob.session().setPromptFlag(true);
518                                        break;
519                                }
520                        }
521                        break;
522                case Packet.CHAN_WHO_REQ:
523                        {
524                                lastPacketReceivedTime=System.currentTimeMillis();
525                                final ChannelWhoRequest wk=(ChannelWhoRequest)packet;
526                                final ChannelWhoReply wkr=new ChannelWhoReply();
527                                wkr.target_name=wk.sender_name;
528                                wkr.target_mud=wk.sender_mud;
529                                wkr.channel=wk.channel;
530                                final int channelInt=CMLib.channels().getChannelIndex(wk.channel);
531                                final Vector<String> whoV=new Vector<String>();
532                                for(final Session S : CMLib.sessions().localOnlineIterable())
533                                {
534                                        final MOB M=S.mob();
535                                        if((CMLib.channels().mayReadThisChannel(M,false,S,channelInt))
536                                        &&(M!=null)
537                                        &&(!CMLib.flags().isCloaked(M)))
538                                                whoV.addElement(M.name());
539                                }
540                                wkr.who=whoV;
541                                try
542                                {
543                                        wkr.send();
544                                }
545                                catch (final Exception e)
546                                {
547                                        Log.errOut("IMudClient", e);
548                                }
549                        }
550                        break;
551                case Packet.CHAN_USER_REQ:
552                        {
553                                lastPacketReceivedTime=System.currentTimeMillis();
554                                final ChannelUserRequest wk=(ChannelUserRequest)packet;
555                                final ChannelUserReply wkr=new ChannelUserReply();
556                                wkr.target_name=wk.sender_name;
557                                wkr.target_mud=wk.sender_mud;
558                                wkr.userRequested=wk.userToRequest;
559                                final MOB M=CMLib.players().getLoadPlayer(wk.userToRequest);
560                                if(M!=null)
561                                {
562                                        wkr.userVisibleName = M.name();
563                                        wkr.gender=(char)M.charStats().getStat(CharStats.STAT_GENDER);
564                                        try
565                                        {
566                                                wkr.send();
567                                        }
568                                        catch (final Exception e)
569                                        {
570                                                Log.errOut("IMudClient", e);
571                                        }
572                                }
573                        }
574                        break;
575                case Packet.WHO_REQUEST:
576                        {
577                                lastPacketReceivedTime=System.currentTimeMillis();
578                                final WhoPacket wk=(WhoPacket)packet;
579                                final WhoPacket wkr=new WhoPacket();
580                                wkr.type=Packet.WHO_REPLY;
581                                wkr.target_name=wk.sender_name;
582                                wkr.target_mud=wk.sender_mud;
583                                final Vector whoV=new Vector();
584                                for(final Session S : CMLib.sessions().localOnlineIterable())
585                                {
586                                        MOB smob=S.mob();
587                                        if((smob!=null)&&(smob.soulMate()!=null))
588                                                smob=smob.soulMate();
589                                        if((!S.isStopped())&&(smob!=null)
590                                        &&(!smob.amDead())
591                                        &&(CMLib.flags().isInTheGame(smob,true))
592                                        &&(!CMLib.flags().isCloaked(smob)))
593                                        {
594                                                final Vector whoV2=new Vector();
595                                                whoV2.addElement(smob.name());
596                                                whoV2.addElement(Integer.valueOf((int)(S.getIdleMillis()/1000)));
597                                                whoV2.addElement(smob.charStats().displayClassLevel(smob,true));
598                                                whoV.addElement(whoV2);
599                                        }
600                                }
601                                wkr.who=whoV;
602                                try
603                                {
604                                        wkr.send();
605                                }
606                                catch (final Exception e)
607                                {
608                                        Log.errOut("IMudClient", e);
609                                }
610                        }
611                        break;
612                case Packet.TELL:
613                        {
614                                lastPacketReceivedTime=System.currentTimeMillis();
615                                final TellPacket tk=(TellPacket)packet;
616                                final MOB smob=findSessMob(tk.target_name);
617                                if(smob!=null)
618                                {
619                                        MOB mob=null;
620                                        final PlayerStats pstats=smob.playerStats();
621                                        if(pstats!=null)
622                                        {
623                                                if((pstats.getReplyToMOB()!=null)&&(pstats.getReplyToMOB().Name().indexOf('@')>=0))
624                                                        mob=pstats.getReplyToMOB();
625                                                else
626                                                        mob=CMClass.getFactoryMOB();
627                                                pstats.setReplyTo(mob, PlayerStats.REPLY_TELL);
628                                        }
629                                        else
630                                                mob=CMClass.getFactoryMOB();
631                                        mob.setName(tk.sender_name+"@"+tk.sender_mud);
632                                        mob.setLocation(getUniversalRoom());
633                                        tk.message=fixColors(CMProps.applyINIFilter(tk.message,CMProps.Str.SAYFILTER));
634                                        CMLib.commands().postSay(mob,smob,tk.message,true,true);
635                                }
636                        }
637                        break;
638                default:
639                        Log.errOut("IMudInterface","Unknown type: "+packet.type);
640                        break;
641                }
642        }

657         * Register a fake channel
658         * @param chan the remote channel name
659         * @return the local channel name for the specified new local channel name
660         * @see com.planet_ink.coffee_mud.core.intermud.i3.packets.ImudServices#getLocalChannel
661         */
662        @Override
663        public boolean addChannel(CMChannel chan)
664        {
665                if((getLocalChannel(chan.i3name()).length()==0)
666                &&(getRemoteChannel(chan.name()).length()==0))
667                {
668                        channels.add(chan);
669                        return true;
670                }
671                return false;
672        }

681        {

683                {

685                        {

688                        }
689                }
690                return false;
691        }
