045        /**
046         * Given a user name, this method will build a unique
047         * key.  This unique key has nothing to do with the
048         * unique object id.  The idea behind a key name is to
049         * ensure that you do not end up with a user named
050         * Descartes and a user named Des Cartes.  It removes
051         * all non-alphabetic characters and makes the string
052         * lower case.
053         * @exception InvalidNameException thrown if the name produces an unuseable key
054         * @param nom the visual name to create a key from
055         * @return the new key
056         * @throws InvalidNameException an error telling you to pick a new name
057         */
058        static public String createKeyName(final String nom) throws InvalidNameException
059        {
060                final StringBuffer buff = new StringBuffer(nom.toLowerCase());
061                String key = "";
062                int i;
063
064                for(i=0; i<buff.length(); i++)
065                {
066                        final char c = buff.charAt(i);
067
068                        if( c >= 'a' && c <= 'z' )
069                        {
070                                key = key + c;
071                        }
072                        else
073                        if( c != '\'' && c != '-' && c != ' ' )
074                        {
075                                throw new InvalidNameException(c + " is an invalid character for names.");
076                        }
077                }
078                if( key.length() < 3 )
079                {
080                        throw new InvalidNameException("Your name must have at least three alphabetic characters.");
081                }
082                else
083                if( key.length() > 29 )
084                {
085                        throw new InvalidNameException("Your name is too long.");
086                }
087                return key;
088        }

090        /**
091         * Given a user name, this method will find the Interactive
092         * instance associated with that user name.  If no such user is
093         * currently logged in, this method will return null.
094         * @param nom the name of the desired user
095         * @return the Interactive object for the specified name or null if no such user exists
096         */
097        static public Interactive findUser(String nom)
098        {
099                final ServerUser[] users = I3Server.getInteractives();
100                int i;
101
102                try
103                {
104                        nom = createKeyName(nom);
105                }
106                catch( final InvalidNameException e )
107                {
108                        return null;
109                }
110                for(i=0; i<users.length; i++)
111                {
112                        final Interactive user = (Interactive)users[i];
113
114                        if( user.getKeyName().equals(nom) )
115                        {
116                                return user;
117                        }
118                }
119                return null;
120        }

166        /**
167         * Stops any running I/O threads for this interactive, closes the
168         * user socket, and marks the object for destruction according to
169         * the requirements of the ServerObject interface.
170         * @see com.planet_ink.coffee_mud.core.intermud.i3.server.ServerObject#getDestructed
171         */
172        @Override
173        public synchronized void destruct()
174        {
175                output_stream.flush();
176                if( input_thread != null )
177                {
178                        input_thread.stop();
179                        input_thread = null;
180                }
181                try
182                {
183                        if(socket!=null)
184                                socket.close();
185                }
186                catch( final java.io.IOException e )
187                {
188                        Log.errOut("IMInteractive",e);
189                }
190                destructed = true;
191        }

193        /**
194         * Called whenever a command is pulled off the incoming
195         * command stack.  If there is an instance of the Input
196         * class to which input is supposed to be redirected,
197         * then the command is sent there.  Otherwise it is sent
198         * to the parser.  Muds wishing to implement their own
199         * parser system should
200         * @param cmd the command to be executed
201         * @see #processInput
202         */
203        protected synchronized void input(final String cmd)
204        {
205                Input ob = null;
206
207                if( redirect.size() > 0 )
208                {
209                        ob = (Input)redirect.elementAt(0);
210                        redirect.removeElementAt(0);
211                }
212                if( ob != null )
213                {
214                        ob.input(this, cmd);
215                }
216                else
217                if( body != null )
218                {
219                        body.executeCommand(cmd);
220                }
221                if( redirect.size() < 1 )
222                {
223                        sendMessage(getPrompt(), true);
224                }
225        }

234                socket = null;

238                }
239                destruct();

255        /**
256         * The server triggers this method once each server cycle to see
257         * if the user has any input waiting to be processed.  This method
258         * checks the input queue.  If there is input waiting, it updates
259         * the last command time and calls the input() method with the
260         * waiting command.  Otherwise it simply returns.
261         * @see #input
262         * @see com.planet_ink.coffee_mud.core.intermud.i3.server.ServerUser#processInput
263         * @see com.planet_ink.coffee_mud.core.intermud.i3.server.ServerThread#tick(com.planet_ink.coffee_mud.core.interfaces.Tickable, int)
264         */
265        @Override
266        public synchronized final void processInput()
267        {
268                if( input_thread != null )
269                {
270                        final String msg = input_thread.nextMessage();
271
272                        if( msg != null )
273                        {
274                                last_command_time = new java.util.Date();
275                                input(msg);
276                        }
277                }
278        }

619/**
620 * The InputThread class handles asynchronous user input and queues
621 * it up to be picked up by the user synchronously.  In English,
622 * the user can be entering information at any point in time
623 * while the server is running.  You want, however, that a command
624 * be executed in a specific order.  This class therefore stuffs commands
625 * into a queue when they arrive.  When the user is ready, it pulls a
626 * single command off to be executed.
627 * Created: 27 September 1996
628 * Last modified 27 September 1996
629 * @author George Reese (borg@imaginary.com)
630 * @version 1.0
631 * @see com.planet_ink.coffee_mud.core.intermud.i3.net.Interactive
632 */
633@SuppressWarnings({"unchecked","rawtypes"})
634class InputThread implements Runnable
635{
636        private final List<String>                      input_buffer;
637        private final BufferedReader                    stream;
638        private boolean                                 destructed;
639        private final Thread                            thread;
640        private final Interactive                       user;
641        private volatile long                   internalSize=0;
642
643        /**
644         * Constructs and starts the thread which accepts user
645         * input.  As a user enters a command, the command is
646         * added to a input_buffer.  During each server cycle, the
647         * Interactive object for this thread pulls off one
648         * command and executes it.
649         * @exception java.io.IOException thrown if no input stream can be created
650         * @param s the socket connected to the user's machine
651         * @param u the Interactive attached to this thread
652         */
653        public InputThread(final Socket s, final Interactive u) throws java.io.IOException {
654                destructed = false;
655                user = u;
656                input_buffer = new Vector(10);
657                stream = new java.io.BufferedReader(new java.io.InputStreamReader(s.getInputStream()));
658                thread = new Thread(Thread.currentThread().getThreadGroup(),this,"I3_"+Thread.currentThread().getName());
659                thread.setDaemon(true);
660                thread.start();
661        }
662
663        /**
664         * As long as the user is connected, this thread accepts
665         * input from the user machine.  If the user drops link,
666         * this will call loseLink() in the interactive object.
667         * @see com.planet_ink.coffee_mud.core.intermud.i3.net.Interactive#loseLink
668         */
669        @Override
670        public void run()
671        {
672                while( !destructed )
673                {
674                        String msg;
675
676                        try
677                        {
678                                msg = stream.readLine();
679                        }
680                        catch( final java.io.IOException e )
681                        {
682                                synchronized( user )
683                                {
684                                        user.loseLink();
685                                }
686                                return;
687                        }
688                        synchronized( this )
689                        {
690                                if(msg != null)
691                                {
692                                        input_buffer.add(msg);
693                                        internalSize+=(msg.length()*2);
694                                }
695                        }
696                        if(internalSize > (10 * 1024 * 1024))
697                        {
698                                Log.errOut("Excessive buffer size: "+internalSize);
699                        }
700                        try
701                        {
702                                Thread.sleep(10);
703                        }
704                        catch (final InterruptedException e)
705                        {
706                        }
707                }
708        }
709
710        /**
711         * The interactive object for this input thread will
712         * call stop if the interactive is destructed for
713         * any reason.
714         */
715        public void stop()
716        {
717                destructed = true;
718                CMLib.killThread(thread,500,1);
719                input_buffer.clear();
720        }
721
722        protected synchronized String nextMessage()
723        {
724                String msg;
725
726                synchronized( input_buffer )
727                {
728                        if( input_buffer.size() > 0 )
729                        {
730                                msg = input_buffer.remove(0);
731                                internalSize-=(msg.length()*2);
732                        }
733                        else
734                        {
735                                msg = null;
736                        }
737                }
738                return msg;
739        }
740}
