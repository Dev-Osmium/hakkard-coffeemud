029/**
030 * Any object which should persist over time should
031 * have a PersistentPeer which handles saving it.
032 * This allows a separate object to worry about
033 * persistence issues, specifically whether the object
034 * should save to a flatfile or database, and how that
035 * saving occurs.
036 * @author George Reese (borg@imaginary.com)
037 * @version 1.0
038 */
039public interface PersistentPeer {
040        /**
041         * Gets data about this peer from storage and gives it
042         * back to the object for which this peer exists.
043         * @exception com.planet_ink.coffee_mud.core.intermud.i3.persist.PersistenceException if an error occurs during restore
044         */
045        public abstract void restore() throws PersistenceException;
046
047        /**
048         * Triggers a save of its peer.  Implementing classes
049         * should do whatever it takes to save the object in
050         * this method.
051         * @exception com.planet_ink.coffee_mud.core.intermud.i3.persist.PersistenceException if a problem occurs in saving
052         */
053        public abstract void save() throws PersistenceException;
054
055        /**
056         * Assigns a persistent object to this peer for
057         * persistence operations.
058         * @param ob the implementation of com.planet_ink.coffee_mud.core.intermud.i3.persist.Persistent that this is a peer for
059         * @see com.planet_ink.coffee_mud.core.intermud.i3.persist.Persistent
060         */
061        public abstract void setPersistent(Persistent ob);
062
063        /**
064         * An implementation uses this to tell its Persistent
065         * that it is in the middle of restoring.
066         * @return true if a restore operation is in progress
067         */
068        public abstract boolean isRestoring();
069}
