025/**
026 * This class represents an I3 channel.  The ChannelList
027 * class keeps a list of this channels for tracking.
028 * @author George Reese (borg@imaginary.com)
029 * @version 1.0
030 */
031
032/**
033 * Copyright (c) 1996 George Reese
034 * This source code may not be modified, copied,
035 * redistributed, or used in any fashion without the
036 * express written consent of George Reese.
037 */
038public class Channel implements Serializable
039{
040        public static final long serialVersionUID=0;
041        /**
042         * The name of the channel
043         */
044        public String channel;
045        /**
046         * The modification status of this channel
047         * @see com.planet_ink.coffee_mud.core.intermud.i3.persist.Persistent
048         */
049        public int modified;
050        /**
051         * The mud which controls the channel
052         */
053        public String owner;
054        /**
055         * The type of the mud channel
056         */
057        public int    type;
058
059        /**
060         * Constructs a new mud channel object
061         */
062        public Channel()
063        {
064                super();
065                modified = Persistent.UNMODIFIED;
066        }
067
068        /**
069         * Constructs a copy of an existing channel
070         * @param other the other channel
071         */
072        public Channel(Channel other)
073        {
074                super();
075                channel = other.channel;
076                modified = other.modified;
077                owner = other.owner;
078                type = other.type;
079        }
080}
