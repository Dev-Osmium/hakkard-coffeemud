044        public int DBCount(String journal, String from, String to)
045        {
046                if(journal==null) 
047                        return 0;
048
049                journal = DB.injectionClean(journal);
050                from    = DB.injectionClean(from);
051                to              = DB.injectionClean(to);
052
053                synchronized(journal.toUpperCase().intern())
054                {
055                        int ct=0;
056                        DBConnection D=null;
057                        try
058                        {
059                                D=DB.DBFetch();
060                                final ResultSet R=D.query("SELECT CMFROM,CMTONM FROM CMJRNL WHERE CMJRNL='"+journal+"'");
061                                if((from==null)&&(to==null))
062                                        ct=D.getRecordCount(R);
063                                else
064                                {
065                                        while(R.next())
066                                        {
067                                                if(((from==null)||(from.equalsIgnoreCase(DBConnections.getRes(R,"CMFROM"))))
068                                                &&((to==null)||(to.equalsIgnoreCase(DBConnections.getRes(R,"CMTONM")))))
069                                                        ct++;
070                                        }
071                                }
072                        }
073                        catch(final Exception sqle)
074                        {
075                                Log.errOut("Journal",sqle);
076                        }
077                        finally
078                        {
079                                DB.DBDone(D);
080                        }
081                        return ct;
082                }
083        }

179        protected JournalEntry DBReadJournalEntry(ResultSet R)
180        {
181                final JournalEntry entry=(JournalEntry)CMClass.getCommon("DefaultJournalEntry");
182                entry.key               (DBConnections.getRes(R,"CMJKEY"));
183                entry.from              (DBConnections.getRes(R,"CMFROM"));
184
185                final String dateStr = DBConnections.getRes(R,"CMDATE");
186                entry.to                (DBConnections.getRes(R,"CMTONM"));
187                entry.subj              (DBConnections.getRes(R,"CMSUBJ"));
188                entry.parent    (DBConnections.getRes(R,"CMPART"));
189                entry.attributes(CMath.s_long(DBConnections.getRes(R,"CMATTR")));
190                entry.data              (DBConnections.getRes(R,"CMDATA"));
191                entry.update    (CMath.s_long(DBConnections.getRes(R,"CMUPTM")));
192                entry.msgIcon   (DBConnections.getRes(R, "CMIMGP"));
193                entry.views             (CMath.s_int(DBConnections.getRes(R, "CMVIEW")));
194                entry.replies   (CMath.s_int(DBConnections.getRes(R, "CMREPL")));
195                entry.msg               (DBConnections.getRes(R,"CMMSGT"));
196
197                final int datestrdex=dateStr.indexOf('/');
198                if(datestrdex>=0)
199                {
200                        entry.update(CMath.s_long(dateStr.substring(datestrdex+1)));
201                        entry.date(CMath.s_long(dateStr.substring(0,datestrdex)));
202                }
203                else
204                {
205                        entry.date(CMath.s_long(dateStr));
206                        if(entry.update()<entry.date())
207                                entry.update(entry.date());
208                }
209
210                final String subject=entry.subj().toUpperCase();
211                if((subject.startsWith("MOTD"))
212                ||(subject.startsWith("MOTM"))
213                ||(subject.startsWith("MOTY")))
214                {
215                        final char c=subject.charAt(3);
216                        entry.subj(entry.subj().substring(4));
217                        long last=entry.date();
218                        if(c=='D') 
219                                last=last+TimeManager.MILI_DAY;
220                        else
221                        if(c=='M') 
222                                last=last+TimeManager.MILI_MONTH;
223                        else
224                        if(c=='Y') 
225                                last=last+TimeManager.MILI_YEAR;
226                        entry.update(last);
227                }
228                return entry;
229        }

231        public List<JournalEntry> DBReadJournalPageMsgs(String journal, String parent, String searchStr, long newerDate, int limit)
232        {
233                journal         = DB.injectionClean(journal);
234                parent          = DB.injectionClean(parent);
235                searchStr       = DB.injectionClean(searchStr);
236
237                final boolean searching=((searchStr!=null)&&(searchStr.length()>0));
238
239                final Vector<JournalEntry> journalV=new Vector<JournalEntry>();
240                DBConnection D=null;
241                try
242                {
243                        D=DB.DBFetch();
244                        String sql="SELECT * FROM CMJRNL WHERE";
245                        if(newerDate==0)
246                                sql+=" CMUPTM > 0"; // <0 are the meta msgs
247                        else
248                        if((parent==null)||(parent.length()>0))
249                                sql+=" CMUPTM > " + newerDate;
250                        else
251                                sql+=" CMUPTM < " + newerDate;
252
253                        if((journal!=null)&&(journal.length()>0))
254                                sql += " AND CMJRNL='"+journal+"'";
255                        if(parent != null)
256                                sql += " AND CMPART='"+parent+"'";
257                        if(searching)
258                                sql += " AND (CMSUBJ LIKE '%"+searchStr+"%' OR CMMSGT LIKE '%"+searchStr+"%')";
259                        sql += " ORDER BY CMUPTM";
260                        if((parent==null)||(parent.length()>0))
261                                sql += " ASC";
262                        else
263                                sql += " DESC";
264
265                        final ResultSet R=D.query(sql);
266                        int cardinal=0;
267                        JournalEntry entry;
268                        final Map<String,JournalEntry> parentKeysDone=new HashMap<String,JournalEntry>();
269                        while(((cardinal < limit)||(limit==0)) && R.next())
270                        {
271                                entry = DBReadJournalEntry(R);
272                                if((parent!=null)&&(CMath.bset(entry.attributes(),JournalEntry.ATTRIBUTE_STUCKY)))
273                                        continue;
274                                if(searching)
275                                {
276                                        final long updated=entry.update();
277                                        final String parentKey=((entry.parent()!=null)&&(entry.parent().length()>0)) ? entry.parent() : entry.key();
278                                        if(parentKeysDone.containsKey(parentKey))
279                                        {
280                                                parentKeysDone.get(parentKey).update(updated);
281                                                continue;
282                                        }
283                                        if((entry.parent()!=null)&&(entry.parent().length()>0))
284                                        {
285                                                final JournalEntry oldEntry=entry;
286                                                entry=DBReadJournalEntry(journal, entry.parent());
287                                                if(entry==null)
288                                                        entry=oldEntry;
289                                        }
290                                        parentKeysDone.put(entry.key(),entry);
291                                        entry.update(updated);
292                                }
293                                entry.cardinal(++cardinal);
294                                journalV.addElement(entry);
295                        }
296                        if(CMSecurity.isDebugging(CMSecurity.DbgFlag.CMJRNL))
297                                Log.debugOut("JournalLoader","Query ("+journalV.size()+"): "+sql);
298                        if((journalV.size()>0)&&(parent!=null)) // set last entry -- make sure its not stucky
299                        {
300                                journalV.lastElement().lastEntry(true);
301                                while(R.next())
302                                {
303                                        final long attributes=R.getLong("CMATTR");
304                                        if(CMath.bset(attributes,JournalEntry.ATTRIBUTE_STUCKY))
305                                                continue;
306                                        journalV.lastElement().lastEntry(false);
307                                        break;
308                                }
309                        }
310                        else
311                        if((journalV.size()>0)&&(!R.next()))
312                                journalV.lastElement().lastEntry(true);
313                }
314                catch(final Exception sqle)
315                {
316                        Log.errOut("Journal",sqle);
317                        return null;
318                }
319                finally
320                {
321                        DB.DBDone(D);
322                }
323                return journalV;
324        }

385        public List<JournalEntry> makeJournalEntryList(final String journalID, final ResultSet R, long limit) throws SQLException
386        {
387                List<String> ids = new ArrayList<String>();
388                while(R.next() && (--limit>=0))
389                        ids.add(R.getString("CMJKEY"));
390                R.close();
391                final String lastEntry = (ids.size()==0) ? "" : ids.get(ids.size()-1);
392                return new FullConvertingList<String,JournalEntry>(ids, new FullConverter<String,JournalEntry>()
393                {
394                        @Override
395                        public JournalEntry convert(int cardinal, String obj)
396                        {
397                                final JournalEntry j=DBReadJournalEntry(journalID, obj);
398                                if(j != null)
399                                {
400                                        j.cardinal(cardinal);
401                                        if(j.key().equals(lastEntry))
402                                                j.lastEntry(true);
403                                        return j;
404                                }
405                                return null;
406                        }
407
408                        @Override
409                        public String reverseConvert(JournalEntry obj)
410                        {
411                                return obj==null ? "" : obj.key();
412                        }
413                });
414        }

551        public int getFirstMsgIndex(List<JournalEntry> journal, String from, String to, String subj)
552        {
553                from    = DB.injectionClean(from);
554                to              = DB.injectionClean(to);
555                subj    = DB.injectionClean(subj);
556
557                if(journal==null) 
558                        return -1;
559                for(int i=0;i<journal.size();i++)
560                {
561                        final JournalEntry E=journal.get(i);
562                        if((from!=null)&&(!(E.from()).equalsIgnoreCase(from)))
563                                continue;
564                        if((to!=null)&&(!(E.to()).equalsIgnoreCase(to)))
565                                continue;
566                        if((subj!=null)&&(!(E.subj()).equalsIgnoreCase(subj)))
567                                continue;
568                        return i;
569                }
570                return -1;
571        }

654        public void DBDeletePlayerPrivateJournalEntries(String name)
655        {
656                name = DB.injectionClean(name);
657                DBConnection D=null;
658                try
659                {
660                        D=DB.DBFetch();
661                        final Vector<String> deletableEntriesV=new Vector<String>();
662                        final Vector<String[]> notifiableParentsV=new Vector<String[]>();
663                        //Resources.submitResource("JOURNAL_"+journal);
664                        final String sql="SELECT * FROM CMJRNL WHERE CMTONM='"+name+"'";
665                        final ResultSet R=D.query(sql);
666                        while(R.next())
667                        {
668                                final String journalName=R.getString("CMJRNL");
669                                if((CMLib.journals().getForumJournal(journalName)!=null)
670                                ||(CMLib.journals().getCommandJournal(journalName)!=null))
671                                        continue;
672                                final String key=R.getString("CMJKEY");
673                                final String parent=R.getString("CMPART");
674                                if((parent!=null)&&(parent.length()>0))
675                                        notifiableParentsV.add(new String[]{journalName,parent});
676                                deletableEntriesV.add(key);
677                        }
678                        for(final String[] parentKey : notifiableParentsV)
679                        {
680                                if(!deletableEntriesV.contains(parentKey[0]))
681                                {
682                                        final JournalEntry parentEntry=DBReadJournalEntry(parentKey[0], parentKey[1]);
683                                        if(parentEntry!=null)
684                                                DBUpdateMessageReplies(parentEntry.key(),parentEntry.replies()-1);
685                                }
686                        }
687                        for(final String s : deletableEntriesV)
688                                D.update("DELETE FROM CMJRNL WHERE CMJKEY='"+s+"' OR CMPART='"+s+"'",0);
689                }
690                catch(final Exception sqle)
691                {
692                        Log.errOut("Journal",sqle);
693                }
694                finally
695                {
696                        DB.DBDone(D);
697                }
698        }

862                if(journal==null)

868                                return null;

871                        final String oldmsg=entry.msg();
872                        final int replies = entry.replies()+1;
873                        message=oldmsg+JournalsLibrary.JOURNAL_BOUNDARY
874                         +"^yReply from^N: "+from+"    ^yDate/Time ^N: "+CMLib.time().date2String(now)+"%0D"
875                         +message;
876                        entry.msg(message);
877                        final String sql="UPDATE CMJRNL SET CMUPTM="+now+", CMMSGT=?, CMREPL="+replies+" WHERE CMJKEY='"+oldkey+"'";
878                        if(CMSecurity.isDebugging(CMSecurity.DbgFlag.CMJRNL))
879                                Log.debugOut("JournalLoader",sql);
880                        DB.updateWithClobs(sql,message);

918        public void DBWrite(String journal, JournalEntry entry)
919        {
920                journal                 =DB.injectionClean(journal);
921                entry.data              (DB.injectionClean(entry.data()));
922                entry.from              (DB.injectionClean(entry.from()));
923                entry.key               (DB.injectionClean(entry.key()));

928                entry.to                (DB.injectionClean(entry.to()));
929
930                if(journal==null) 
931                        return;
932                synchronized(journal.toUpperCase().intern())
933                {
934                        final long now=System.currentTimeMillis();
935                        if(entry.subj().length()>255)
936                                entry.subj(entry.subj().substring(0,255));
937                        if(entry.key()==null)
938                                entry.key(journal+now+Math.random());
939                        if(entry.date()==0)
940                                entry.date(now);
941                        if(entry.update()==0)
942                                entry.update(now);
943                        final String sql = "INSERT INTO CMJRNL ("
944                                +"CMJKEY, "
945                                +"CMJRNL, "
946                                +"CMFROM, "
947                                +"CMDATE, "
948                                +"CMTONM, "
949                                +"CMSUBJ, "
950                                +"CMPART, "
951                                +"CMATTR, "
952                                +"CMDATA, "
953                                +"CMUPTM, "
954                                +"CMIMGP, "
955                                +"CMVIEW, "
956                                +"CMREPL, "
957                                +"CMMSGT "
958                                +") VALUES ('"
959                                +entry.key()
960                                +"','"+journal
961                                +"','"+entry.from()
962                                +"','"+entry.date()
963                                +"','"+entry.to()
964                                +"',?"
965                                +",'"+entry.parent()
966                                +"',"+entry.attributes()
967                                +",'"+entry.data()
968                                +"',"+entry.update()
969                                +",'"+entry.msgIcon()
970                                +"',"+entry.views()
971                                +","+entry.replies()
972                                +",?)";
973                        if(CMSecurity.isDebugging(CMSecurity.DbgFlag.CMJRNL))
974                                Log.debugOut("JournalLoader",sql);
975                        DB.updateWithClobs(sql , entry.subj(), entry.msg());
976                        if((entry.parent()!=null)&&(entry.parent().length()>0))
977                        {
978                                // this constitutes a threaded reply -- update the counter

980                                if(parentEntry!=null)
981                                        DBUpdateMessageReplies(parentEntry.key(),parentEntry.replies()+1);
982                        }
983                        if(System.currentTimeMillis()==now) // ensures unique keys.
984                                CMLib.s_sleep(1);
985                }
986        }
