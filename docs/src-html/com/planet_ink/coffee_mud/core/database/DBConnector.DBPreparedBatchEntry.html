054        public static final class DBPreparedBatchEntry

090                                                final String dbService,

093                                                final Map<String,String> dbParms,
094                                                final int numConnections,

099                                                final boolean retryErrorQueue)
100        {
101                super();
102                this.dbClass=dbClass;
103                this.dbService=dbService;
104                this.dbUser=dbUser;
105                this.dbPass=dbPass;
106                this.dbParms=dbParms;
107                this.numConnections=numConnections;
108                this.doErrorQueueing=doErrorQueueing;
109                this.newErrorQueueing=retryErrorQueue;
110                this.dbReuse=reuse;
111                this.dbTransact=transact;

113                if(this.dbPingIntMins<=0)
114                        this.dbPingIntMins=Integer.MAX_VALUE;

117        public void reconnect()
118        {
119                if (dbConnections != null)
120                {
121                        dbConnections.deregisterDriver();
122                        dbConnections.killConnections();
123                }
124                dbConnections=new DBConnections(dbClass,dbService,dbUser,dbPass,dbParms,numConnections,dbReuse,dbTransact,doErrorQueueing);
125                if(dbConnections.amIOk()&&newErrorQueueing)
126                        dbConnections.retryQueuedErrors();
127        }

139        public int getRecordCount(final DBConnection D, final ResultSet R)
140        {
141                if(D!=null)
142                        return D.getRecordCount(R);
143                int recordCount=0;
144                try
145                {
146                        R.last();
147                        recordCount=R.getRow();
148                        R.beforeFirst();
149                }
150                catch(final Exception e)
151                {
152                        e.printStackTrace();
153                }
154                return recordCount;
155        }

421        /** return a status string, or "" if everything is ok.
422         *
423         * Usage: errorStatus();
424         * @return StringBuffer    complete error status
425         */
426        public StringBuffer errorStatus()
427        {
428                if(dbConnections==null)
429                        return new StringBuffer("Not connected.");
430                final StringBuffer status=dbConnections.errorStatus();
431                if(status.length()==0)
432                        return new StringBuffer("OK! Connections in use="+dbConnections.numInUse()+"/"+dbConnections.numConnectionsMade());
433                return new StringBuffer("<BR>"+status.toString().replaceAll("\n","<BR>")+"Connections in use="+dbConnections.numInUse()+"/"+dbConnections.numConnectionsMade());
434        }
