053        public void DBReadClanItems(Map<String,Clan> clans)
054        {
055                DBConnection D=null;
056                try
057                {
058                        D=DB.DBFetch();
059                        ResultSet R=D.query("SELECT * FROM CMCLIT");
060                        while(R.next())
061                        {
062                                final String clanID=DBConnections.getRes(R,"CMCLID");
063                                final Clan C=clans.get(clanID);
064                                if(C==null)
065                                {
066                                        //Log.errOut("Clan","Couldn't find clan '"+clanID+"'");
067                                }
068                                else
069                                {
070                                        final String itemID=DBConnections.getRes(R,"CMITID");
071                                        final Item newItem=CMClass.getItem(itemID);
072                                        if(newItem==null)
073                                                Log.errOut("Clan","Couldn't find item '"+itemID+"'");
074                                        else
075                                        {
076                                                String text=DBConnections.getResQuietly(R,"CMITTX");
077                                                int roomX;
078                                                if(text.startsWith("<ROOM") && ((roomX=text.indexOf("/>"))>=0))
079                                                {
080                                                        final String roomXML=text.substring(0,roomX+2);
081                                                        text=text.substring(roomX+2);
082                                                        newItem.setMiscText(text);
083                                                        final List<XMLLibrary.XMLTag> xml=CMLib.xml().parseAllXML(roomXML);
084                                                        if((xml!=null)&&(xml.size()>0))
085                                                        {
086                                                                final String roomID=xml.get(0).parms().get("ID");
087                                                                final long expirationDate=CMath.s_long(xml.get(0).parms().get("EXPIRE"));
088                                                                if(roomID.startsWith("SPACE.") && (newItem instanceof SpaceObject))
089                                                                        CMLib.map().addObjectToSpace((SpaceObject)newItem,CMParms.toLongArray(CMParms.parseCommas(roomID.substring(6), true)));
090                                                                else
091                                                                {
092                                                                        final Room itemR=CMLib.map().getRoom(roomID);
093                                                                        if(itemR!=null)
094                                                                        {
095                                                                                if(newItem instanceof BoardableShip)
096                                                                                        ((BoardableShip)newItem).dockHere(itemR);
097                                                                                else
098                                                                                        itemR.addItem(newItem);
099                                                                                newItem.setExpirationDate(expirationDate);
100                                                                        }
101                                                                }
102                                                        }
103                                                }
104                                                else
105                                                {
106                                                        newItem.setMiscText(text);
107                                                }
108                                                newItem.setUsesRemaining((int)DBConnections.getLongRes(R,"CMITUR"));
109                                                newItem.basePhyStats().setLevel((int)DBConnections.getLongRes(R,"CMITLV"));
110                                                newItem.basePhyStats().setAbility((int)DBConnections.getLongRes(R,"CMITAB"));
111                                                newItem.basePhyStats().setHeight((int)DBConnections.getLongRes(R,"CMHEIT"));
112                                                newItem.recoverPhyStats();
113                                                CMLib.map().registerWorldObjectLoaded(null, null, newItem);
114                                                C.getExtItems().addItem(newItem);
115                                        }
116                                }
117                                updateBootStatus("Clan Items");
118                        }
119                }
120                catch(final Exception sqle)
121                {
122                        Log.errOut("Clan",sqle);
123                }
124                finally
125                {
126                        DB.DBDone(D);
127                }
128        }

130        public List<Clan> DBRead()
131        {
132                List<Clan> clanList=new ArrayList<Clan>();
133                DBConnection D=null;
134                try
135                {
136                        D=DB.DBFetch();
137                        ResultSet R=D.query("SELECT * FROM CMCLAN");
138                        recordCount=DB.getRecordCount(D,R);
139                        final Map<String,Clan> clans=new Hashtable<String,Clan>();
140                        while(R.next())
141                        {
142                                currentRecordPos=R.getRow();
143                                final String name=DBConnections.getRes(R,"CMCLID");
144                                final Clan C=(Clan)CMClass.getCommon("DefaultClan");
145                                C.setName(name);
146                                C.setPremise(DBConnections.getRes(R,"CMDESC"));
147                                C.setAcceptanceSettings(DBConnections.getRes(R,"CMACPT"));
148                                C.setStatus(CMath.s_int(DBConnections.getRes(R, "CMSTAT"))); // this will mangle laststatuschangetime, so must be before politics
149                                C.setPolitics(DBConnections.getRes(R,"CMPOLI"));
150                                C.setRecall(DBConnections.getRes(R,"CMRCLL"));
151                                C.setDonation(DBConnections.getRes(R,"CMDNAT"));
152                                C.setMorgue(DBConnections.getRes(R,"CMMORG"));
153                                C.setTrophies(CMath.s_int(DBConnections.getRes(R, "CMTROP")));
154                                //CMLib.clans().addClan(C);
155                                clanList.add(C);
156                                clans.put(C.clanID(), C);
157                                updateBootStatus("Clans");
158                        }
159                        R.close();
160                }
161                catch(final Exception sqle)
162                {
163                        Log.errOut("Clan",sqle);
164                }
165                finally
166                {
167                        DB.DBDone(D);
168                }
169                // log comment
170                return clanList;
171        }

173        public void DBUpdate(Clan C)
174        {
175                final String sql="UPDATE CMCLAN SET "
176                                +"CMDESC='"+C.getPremise()+"',"
177                                +"CMACPT='"+C.getAcceptanceSettings()+"',"
178                                +"CMPOLI=?,"
179                                +"CMRCLL='"+C.getRecall()+"',"
180                                +"CMDNAT='"+C.getDonation()+"',"
181                                +"CMSTAT="+C.getStatus()+","
182                                +"CMMORG='"+C.getMorgue()+"',"
183                                +"CMTROP="+C.getTrophies()+""
184                                +" WHERE CMCLID='"+C.clanID()+"'";
185                DB.updateWithClobs(sql, C.getPolitics());
186        }

188        protected String getDBItemUpdateString(final Clan C, final Item thisItem)
189        {
190                CMLib.catalog().updateCatalogIntegrity(thisItem);
191                final String container=((thisItem.container()!=null)?(""+thisItem.container()):"");
192                return "INSERT INTO CMCLIT (CMCLID, CMITNM, CMITID, CMITTX, CMITLO, CMITWO, "
193                +"CMITUR, CMITLV, CMITAB, CMHEIT"
194                +") values ('"+C.clanID()+"','"+(thisItem)+"','"+thisItem.ID()+"',?,'"+container+"',"+thisItem.rawWornCode()+","
195                +thisItem.usesRemaining()+","+thisItem.basePhyStats().level()+","+thisItem.basePhyStats().ability()+","
196                +thisItem.basePhyStats().height()+")";
197        }

199        private List<DBPreparedBatchEntry> getDBItemUpdateStrings(final Clan C)
200        {
201                final HashSet<String> done=new HashSet<String>();
202                final List<DBPreparedBatchEntry> strings=new LinkedList<DBPreparedBatchEntry>();
203                final ItemCollection coll=C.getExtItems();
204                final List<Item> finalCollection=new LinkedList<Item>();
205                final List<Item> extraItems=new LinkedList<Item>();
206                for(int i=coll.numItems()-1;i>=0;i--)
207                {
208                        final Item thisItem=coll.getItem(i);
209                        if((thisItem!=null)&&(!thisItem.amDestroyed()))
210                        {
211                                final Item cont=thisItem.ultimateContainer(null);
212                                if(cont.owner() instanceof Room)
213                                        finalCollection.add(thisItem);
214                        }
215                }
216                for(final Item thisItem : finalCollection)
217                {
218                        if(thisItem instanceof Container)
219                        {
220                                final List<Item> contents=((Container)thisItem).getDeepContents();
221                                for(final Item I : contents)
222                                        if(!finalCollection.contains(I))
223                                                extraItems.add(I);
224                        }
225                }
226                finalCollection.addAll(extraItems);
227                for(final Item thisItem : finalCollection)
228                {
229                        if(!done.contains(""+thisItem))
230                        {
231                                CMLib.catalog().updateCatalogIntegrity(thisItem);
232                                final Item cont=thisItem.ultimateContainer(null);
233                                final String sql=getDBItemUpdateString(C,thisItem);
234                                final String roomID=((cont.owner()==null)&&(thisItem instanceof SpaceObject)&&(CMLib.map().isObjectInSpace((SpaceObject)thisItem)))?
235                                                ("SPACE."+CMParms.toListString(((SpaceObject)thisItem).coordinates())):CMLib.map().getExtendedRoomID((Room)cont.owner());
236                                final String text="<ROOM ID=\""+roomID+"\" EXPIRE="+thisItem.expirationDate()+" />"+thisItem.text();
237                                strings.add(new DBPreparedBatchEntry(sql,text));
238                                done.add(""+thisItem);
239                        }
240                }
241                return strings;
242        }

244        public void DBUpdateItems(final Clan C)
245        {
246                if((C==null)||(C.clanID()==null)||(C.clanID().length()==0))
247                        return;
248                final List<DBPreparedBatchEntry> statements=new LinkedList<DBPreparedBatchEntry>();
249                statements.add(new DBPreparedBatchEntry("DELETE FROM CMCLIT WHERE CMCLID='"+C.clanID()+"'"));
250                statements.addAll(getDBItemUpdateStrings(C));
251                DB.updateWithClobs(statements);
252        }

254        public void DBCreate(Clan C)
255        {
256                if(C.clanID().length()==0)
257                        return;
258                final String sql="INSERT INTO CMCLAN ("
259                        +"CMCLID,"
260                        +"CMTYPE,"
261                        +"CMDESC,"
262                        +"CMACPT,"
263                        +"CMPOLI,"
264                        +"CMRCLL,"
265                        +"CMDNAT,"
266                        +"CMSTAT,"
267                        +"CMMORG,"
268                        +"CMTROP"
269                        +") values ("
270                        +"'"+C.clanID()+"',"
271                        +"0,"
272                        +"'"+C.getPremise()+"',"
273                        +"'"+C.getAcceptanceSettings()+"',"
274                        +"?,"
275                        +"'"+C.getRecall()+"',"
276                        +"'"+C.getDonation()+"',"
277                        +""+C.getStatus()+","
278                        +"'"+C.getMorgue()+"',"
279                        +""+C.getTrophies()
280                        +")";
281                        DB.updateWithClobs(sql, C.getPolitics());
282        }

284        public void DBDelete(Clan C)
285        {
286                DBConnection D=null;
287                try
288                {
289                        D=DB.DBFetch();
290                        D.update("DELETE FROM CMCLAN WHERE CMCLID='"+C.clanID()+"'",0);
291                        D.update("DELETE FROM CMCLIT WHERE CMCLID='"+C.clanID()+"'",0);
292                        D.update("DELETE FROM CMCHCL WHERE CMCLAN='"+C.clanID()+"'",0);
293                }
294                catch(final Exception sqle)
295                {
296                        Log.errOut("Clan",sqle);
297                }
298                finally
299                {
300                        DB.DBDone(D);
301                }
302        }
