041/**
042 * CMSecurity is the singleton that handles all security checks for all manner
043 * of resource access by users.  It also handles certain flag systems, such
044 * as debug flags, resource re-save flags, and system disabling flags.
045 * 
046 * Supported: (see SecFlag)
047 * ORDER includes TAKE, GIVE, DRESS, mob passivity, all follow
048 * ABOVELAW (also law books),
049 * WIZINV (includes see WIZINV),
050 * CMDMOBS (also prevents walkaways)
051 * SUPERSKILL (never fails skills),
052 * IMMORT (never dies),
053 * KILL* for deleting journal entries
054 * FS:relative path from /coffeemud/ -- read/write access to regular file sys
055 * VFS:relative path from /coffeemud/ -- read/write access to virtual file sys
056 * LIST: (affected by killx, cmdplayers, loadunload, cmdclans, ban, nopurge,
057 *   cmditems, cmdmobs, cmdrooms, sessions, cmdareas, listadmin, stat)
058 * 
059 * Like many similar systems, this class is thread-group-sensitive on some
060 * or all resources, allowing different security views to be presented to 
061 * different "muds" according to the thread group the calling thread belongs to.
062 * @author Bo Zimmerman
063 */

191        /**
192         * Iterates through all the properties on the given property page, finding
193         * any security group definitions and, when found, registering them with 
194         * the security system.
195         * @param page the properties page to go through.
196         */
197        public static final void parseGroups(final Properties page)
198        {
199                clearGroups();
200                if(page==null)
201                        return;
202                final List<Pair<String,String>> allGroups=new LinkedList<Pair<String,String>>();
203                for(final Enumeration<Object> e=page.keys();e.hasMoreElements();)
204                {
205                        final String key=(String)e.nextElement();
206                        if(key.startsWith("GROUP_"))
207                        {
208                                final String groupName=key.substring(6).trim().toUpperCase();
209                                allGroups.add(new Pair<String,String>(groupName,(String)page.get(key)));
210                                addGroup(groupName,"");
211                        }
212                }
213                for (final Pair<String, String> p : allGroups)
214                {
215                        addGroup(p.first,p.second);
216                }
217        }

219        /**
220         * Accepts a string representing a security flag, or another group name, or a filesystem path,
221         * or some other security designation, and returns the appropriate object, such as
222         * a SecGroup, SecPath, SecFlag, or String, that it represents.
223         * @param s the security thing to parse and identify
224         * @return the internal security object that the string represents
225         */
226        public Object parseSecurityFlag(String s)
227        {
228                final SecGroup group=groups.get(s);
229                if(group != null)
230                        return group;
231                else
232                if(s.startsWith("FS:"))
233                {
234                        s=s.substring(3).trim();
235                        boolean isAreaOnly=false;
236                        if(s.startsWith("AREA "))
237                        {
238                                s=s.substring(5).trim();
239                                isAreaOnly=true;
240                        }
241                        while(s.startsWith("/"))
242                                s=s.substring(1).trim();
243                        return new SecPath(s,false,isAreaOnly);
244
245                }
246                else
247                if(s.startsWith("VFS:"))
248                {
249                        s=s.substring(4);
250                        boolean isAreaOnly=false;
251                        if(s.startsWith("AREA "))
252                        {
253                                s=s.substring(5).trim();
254                                isAreaOnly=true;
255                        }
256                        while(s.startsWith("/"))
257                                s=s.substring(1).trim();
258                        return new SecPath(s,true,isAreaOnly);
259                }
260                else
261                if(journalFlags.contains(s))
262                        return s;
263                else
264                if(s.startsWith("KILL") && journalFlags.contains(s.substring(4)))
265                        return s.substring(4);
266                else
267                if(s.startsWith("KILL") && s.endsWith("S") && journalFlags.contains(s.substring(4,s.length()-1)))
268                        return s.substring(4,s.length()-1);
269                else
270                if(s.endsWith("S") && journalFlags.contains(s.substring(0,s.length()-1)))
271                        return s.substring(0,s.length()-1);
272                else
273                {
274                        s=s.replace(' ','_');
275                        SecFlag flag=(SecFlag)CMath.s_valueOf(SecFlag.class, s);
276                        if(flag==null)
277                                flag=(SecFlag)CMath.s_valueOf(SecFlag.class, s+"S");
278                        if((flag==null)&&(s.equals("POOF")))
279                                flag=(SecFlag)CMath.s_valueOf(SecFlag.class, "GOTO");
280                        if((flag==null)&&(s.equals("AREA_POOF")))
281                                flag=(SecFlag)CMath.s_valueOf(SecFlag.class, "AREA_GOTO");
282                        if(flag==null)
283                                return null;
284                        else
285                                return flag;
286                }
287        }

289        /**
290         * Create a new security group object with the given name, and the given
291         * set of security flags/group names as a string list.  
292         * @param name the new security group name
293         * @param set the string list of flags/group names
294         * @return the new security group object
295         */
296        public final SecGroup createGroup(String name, final List<String> set)
297        {
298                final Set<SecFlag>       newFlags=new HashSet<SecFlag>();
299                final List<SecGroup> newGroups=new LinkedList<SecGroup>();
300                final List<SecPath>  newPaths=new LinkedList<SecPath>();
301                final Set<String>        newJFlags=new HashSet<String>();
302                for(int v=0;v<set.size();v++)
303                {
304                        final String s=set.get(v).trim().toUpperCase();
305                        final Object o=this.parseSecurityFlag(s);
306                        if(o==null)
307                                Log.errOut("CMSecurity","Unknown security flag: "+s+" in group "+name);
308                        else
309                        if(o instanceof SecGroup)
310                                newGroups.add((SecGroup)o);
311                        else
312                        if(o instanceof SecPath)
313                                newPaths.add((SecPath)o);
314                        else
315                        if(o instanceof SecFlag)
316                                newFlags.add((SecFlag)o);
317                        else
318                        if(o instanceof String)
319                                newJFlags.add((String)o);
320                        else
321                                Log.errOut("CMSecurity","Unparsed security flag: "+s+" in group "+name);
322                }
323                return new SecGroup(name,newFlags,newGroups,newPaths,newJFlags);
324        }

365         * @param mob the user/player to check
366         * @return true if the MOB represents a SysOp (Archon), false otherwise.

368        public static final boolean isASysOp(final MOB mob)
369        {
370                final MaskingLibrary.CompiledZMask mask=i().compiledSysop;
371                if(mask == null)
372                {
373                        final String maskStr = CMProps.instance().getStr("SYSOPMASK","");

375                        {
376                                Log.errOut("Compiled Archon Mask was not set! Using props entry.");
377                                instance().compiledSysop=CMLib.masking().maskCompile(maskStr);
378                        }
379                        else
380                        if((secs['0'] != null)
381                        &&(secs['0'] != i())
382                        &&(secs['0'].compiledSysop != null))
383                        {
384                                Log.errOut("Compiled Archon Mask was not set! Using base instance.");
385                                instance().compiledSysop=secs['0'].compiledSysop;
386                        }
387                        else
388                        if((secs[0] != null)
389                        &&(secs[0] != i())
390                        &&(secs[0].compiledSysop != null))
391                        {
392                                Log.errOut("Compiled Archon Mask was not set! Using 0th instance.");
393                                instance().compiledSysop=secs[0].compiledSysop;
394                        }
395                        else
396                        {
397                                instance().compiledSysop=CMLib.masking().maskCompile("-ANYCLASS +Archon");
398                                Log.errOut("Compiled Archon Mask was not set! Using core default.");
399                        }
400                        return isASysOp(mob);
401                }
402                return CMLib.masking().maskCheck(mask,mob,true)
403                                ||((mob!=null)
404                                        &&(mob.soulMate()!=null)
405                                        &&(mob.soulMate().isAttributeSet(MOB.Attrib.SYSOPMSGS))
406                                        &&(isASysOp(mob.soulMate())));

473        /**
474         * Returns a comprehensive list of all filesystem directories which the given user/player, in the given room
475         * location, may access
476         * @param mob the user/player to check
477         * @param room the room location of the above user/player
478         * @return the list of all directories this user/player can access
479         */
480        public static final List<String> getAccessibleDirs(final MOB mob, final Room room)
481        {
482                final List<String> DIRSV=new Vector<String>();
483                if(isASysOp(mob))
484                {
485                        DIRSV.add("/");
486                        return DIRSV;
487                }
488                if(mob==null)
489                        return DIRSV;
490                if((mob.playerStats()==null)
491                ||((mob.soulMate()!=null)&&(!mob.soulMate().isAttributeSet(MOB.Attrib.SYSOPMSGS))))
492                        return DIRSV;
493                final boolean subop=((room!=null)&&(room.getArea()!=null)&&(room.getArea().amISubOp(mob.Name())));
494                final Iterator[] allGroups={mob.playerStats().getSecurityFlags().paths(),
495                                 mob.baseCharStats().getCurrentClass().getSecurityFlags(mob.baseCharStats().getCurrentClassLevel()).paths()};
496                for(final Iterator<SecPath> g=new MultiIterator<SecPath>(allGroups);g.hasNext();)
497                {
498                        final SecPath p=g.next();
499                        if((!p.isAreaOnly)||(subop))
500                                DIRSV.add((p.isVfs?"::":"//")+p.path);
501                }
502                String dir=null;
503                for(int d=0;d<DIRSV.size();d++)
504                {
505                        dir=DIRSV.get(d);
506                        if(dir.startsWith("//"))
507                        {
508                                dir=dir.substring(2);
509                                String path="";
510                                String subPath=null;
511                                while(dir.startsWith("/"))
512                                        dir=dir.substring(1);
513                                while(dir.length()>0)
514                                {
515                                        while(dir.startsWith("/"))
516                                                dir=dir.substring(1);
517                                        final int x=dir.indexOf('/');
518                                        subPath=dir;
519                                        if(x>0)
520                                        {
521                                                subPath=dir.substring(0,x).trim();
522                                                dir=dir.substring(x+1).trim();
523                                        }
524                                        else
525                                        {
526                                                subPath=dir.trim();
527                                                dir="";
528                                        }
529                                        final CMFile F=new CMFile(path,null,CMFile.FLAG_LOGERRORS);
530                                        if((F.exists())&&(F.canRead())&&(F.isDirectory()))
531                                        {
532                                                final String[] files=F.list();
533                                                for (final String file : files)
534                                                {
535                                                        if(file.equalsIgnoreCase(subPath))
536                                                        {
537                                                                if(path.length()>0)
538                                                                        path+="/";
539                                                                path+=file;
540                                                                break;
541                                                        }
542                                                }
543                                        }
544                                }
545                                DIRSV.set(d,"//"+path);
546                        }
547                }
548                return DIRSV;
549        }
