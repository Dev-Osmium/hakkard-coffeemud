026/**
027 * A JSON parser.
028 * Not much to say.  It can take a valid json string and generate a standing

030 * from such an object.

084        /**
085         * Given a normal string, this method will return a JSON-Safe
086         * string, which means escaped crlf, escaped tabs and backslashes, etc.
087         * @param value the unsafe string
088         * @return the JSON safe string
089         */
090        public static String toJSONString(final String value)
091        {
092                final StringBuilder strBldr=new StringBuilder("");
093                for(final char c : value.toCharArray())
094                {
095                        switch(c)
096                        {
097                        case '\"':
098                        case '\\':
099                                strBldr.append('\\').append(c);
100                                break;
101                        case '\b': 
102                                strBldr.append('\\').append('b'); 
103                                break;
104                        case '\f': 
105                                strBldr.append('\\').append('f'); 
106                                break;
107                        case '\n': 
108                                strBldr.append('\\').append('n'); 
109                                break;
110                        case '\r': 
111                                strBldr.append('\\').append('r'); 
112                                break;
113                        case '\t': 
114                                strBldr.append('\\').append('t'); 
115                                break;
116                        default: 
117                                strBldr.append(c); 
118                                break;
119                        }
120                }
121                return strBldr.toString();
122        }

124        /**
125         * An official JSON object.  Implemented as a Map, this class
126         * has numerous methods for accessing the internal keys and

134                /**
135                 * Internal method that returns a raw value object, or throws
136                 * an exception if the key is not found
137                 * @param key the key to look for
138                 * @return the raw Object the key is mapped to
139                 * @throws MJSONException the key was not found
140                 */
141                private Object getCheckedObject(String key) throws MJSONException
142                {
143                        if(!containsKey(key))
144                                throw new MJSONException("Key '"+key+"' not found");
145                        return get(key);
146                }
147
148                /**
149                 * Returns a JSONObject mapped in THIS object by the given key.
150                 * Throws an exception if anything goes wrong.
151                 * @param key the key of the object
152                 * @return the JSON Object mapped to by that key
153                 * @throws MJSONException a missing key, or not a JSON Object
154                 */
155                public JSONObject getCheckedJSONObject(String key) throws MJSONException
156                {
157                        final Object o = getCheckedObject(key);
158                        if(!(o instanceof JSONObject))
159                                throw new MJSONException("Key '"+key+"' is not a JSON object");
160                        return (JSONObject)o;
161                }

269                /**
270                 * Correctly appends the given thing to the given stringbuffer which
271                 * is assumed to be in the middle of a JSON obect definition, right
272                 * after the key and the colon:
273                 * @param value the StringBuffer to append a value to
274                 * @param obj the value to append -- a string, null, array, or number
275                 */
276                public static void appendJSONValue(final StringBuilder value, Object obj)
277                {
278                        if(obj instanceof String)
279                        {
280                                value.append("\"").append(toJSONString((String)obj)).append("\"");
281                        }
282                        else
283                        if(obj == NULL)
284                        {
285                                value.append("null");
286                        }
287                        else
288                        if(obj instanceof Object[])
289                        {
290                                value.append("[");
291                                final Object[] array=(Object[])obj;
292                                for(int i=0; i<array.length; i++)
293                                {
294                                        if(i>0)
295                                                value.append(",");
296                                        appendJSONValue(value, array[i]);
297                                }
298                                value.append("]");
299                        }
300                        else
301                        if(obj != null)
302                        {
303                                value.append(obj.toString());
304                        }
305                }

307                /**
308                 * Returns a full JSON document representation of this JSON object
309                 */
310                @Override
311                public String toString()
312                {
313                        final StringBuilder value = new StringBuilder("");
314                        value.append("{");
315                        for(final Iterator<String> k = keySet().iterator(); k.hasNext();)
316                        {
317                                final String keyVar = k.next();
318                                value.append("\"").append(toJSONString(keyVar)).append("\":");
319                                final Object obj = get(keyVar);
320                                appendJSONValue(value, obj);
321                                if(k.hasNext())
322                                {
323                                        value.append(",");
324                                }
325                        }
326                        value.append("}");
327                        return value.toString();
328                }

330                public Object jsonDeepCopy(Object obj)
331                {
332                        if(obj instanceof JSONObject)
333                                return ((JSONObject)obj).copyOf();
334                        else
335                        if(obj.getClass().isArray())
336                        {
337                                final Object[] newArray = Arrays.copyOf((Object[])obj, ((Object[])obj).length);
338                                for(int i=0;i<newArray.length;i++)
339                                        newArray[i] = jsonDeepCopy(newArray[i]);
340                                return newArray;
341                        }
342                        else
343                                return obj;
344                }
345
346                public JSONObject copyOf()
347                {
348                        final JSONObject newObj = new JSONObject();
349                        for(String key : this.keySet())
350                                newObj.put(key, jsonDeepCopy(this.get(key)));
351                        return newObj;
352                }
353        }

355        /**
356         * Parse either an Long, or Double object from the doc buffer
357         * @param doc the full JSON document
358         * @param index one dimensional array containing current index into the doc
359         * @return either an Long or a Double
360         * @throws MJSONException any parsing errors
361         */
362        private Object parseNumber(char[] doc, int[] index) throws MJSONException
363        {
364                final int numStart = index[0];
365                NumberParseState state = NumberParseState.INITIAL;
366                while(index[0] < doc.length)
367                {
368                        final char c=doc[index[0]];
369                        switch(state)
370                        {
371                        case INITIAL:
372                                if(c=='0')
373                                        state=NumberParseState.NEEDDOT;
374                                else
375                                if(c=='-')
376                                        state=NumberParseState.NEEDN0DIGIT;
377                                else
378                                if(Character.isDigit(c))
379                                        state=NumberParseState.HAVEDIGIT;
380                                else
381                                        throw new MJSONException("Expected digit at "+index[0]);
382                                break;
383                        case NEEDN0DIGIT:
384                                if(c=='0')
385                                        throw new MJSONException("Expected digit at "+index[0]);
386                                else
387                                if(Character.isDigit(c))
388                                        state=NumberParseState.HAVEDIGIT;
389                                else
390                                        throw new MJSONException("Expected digit at "+index[0]);
391                                break;
392                        case HAVEDIGIT:
393                                if(c=='.')
394                                        state=NumberParseState.NEEDDOTDIGIT;
395                                else
396                                if((c=='E')||(c=='e'))
397                                        state=NumberParseState.HAVEE;
398                                else
399                                if(Character.isDigit(c))
400                                        state=NumberParseState.HAVEDIGIT;
401                                else
402                                {
403                                        index[0]--;
404                                        return Long.valueOf(new String(doc,numStart,index[0]-numStart+1));
405                                }
406                                break;
407                        case NEEDDOT:
408                                if(c=='.')
409                                        state=NumberParseState.NEEDDOTDIGIT;
410                                else
411                                {
412                                        index[0]--;
413                                        return Long.valueOf(new String(doc,numStart,index[0]-numStart+1));
414                                }
415                                break;
416                        case NEEDDOTDIGIT:
417                                if(Character.isDigit(c))
418                                        state=NumberParseState.HAVEDOTDIGIT;
419                                else
420                                        throw new MJSONException("Expected digit at "+index[0]);
421                                break;
422                        case HAVEDOTDIGIT:
423                                if(Character.isDigit(c))
424                                        state=NumberParseState.HAVEDOTDIGIT;
425                                else
426                                if((c=='e')||(c=='E'))
427                                        state=NumberParseState.HAVEE;
428                                else
429                                {
430                                        index[0]--;
431                                        return Double.valueOf(new String(doc,numStart,index[0]-numStart+1));
432                                }
433                                break;
434                        case HAVEE:
435                                if(c=='0')
436                                        throw new MJSONException("Expected non-zero digit at "+index[0]);
437                                else
438                                if(Character.isDigit(c)||(c=='+')||(c=='-'))
439                                        state=NumberParseState.HAVEEDIGIT;
440                                else
441                                        throw new MJSONException("Expected +- or non-zero digit at "+index[0]);
442                                break;
443                        case HAVEEDIGIT:
444                                if(!Character.isDigit(c))
445                                {
446                                        index[0]--;
447                                        return Double.valueOf(new BigDecimal(new String(doc,numStart,index[0]-numStart+1)).doubleValue());
448                                }
449                                break;
450                        }
451                        index[0]++;
452                }
453                throw new MJSONException("Unexpected end of number at"+index[0]);
454        }

456        /**
457         * Given a char array, and index into it, returns the byte value of the 1 hex
458         * digits at the indexed point of the char array.
459         * @param doc the json doc containing a hex number
460         * @param index the index into that json doc where the hex number begins
461         * @return the byte value of the 1 digit hex number
462         * @throws MJSONException a parse error meaning it wasn't a hex number at all
463         */
464        private byte getByteFromHex(char[] doc, int index) throws MJSONException
465        {
466                final char c = doc[index];
467                if((c >= '0') && (c <= '9'))
468                        return (byte)(c-'0');
469                if((c >= 'a') && (c <= 'f'))
470                        return (byte)(10 + (c-'a'));
471                if((c >= 'A') && (c <= 'F'))
472                        return (byte)(10 + (c-'A'));
473                throw new MJSONException("Illegal hex digit at "+index);
474        }

476        /**
477         * Given a json document char array, and an index into it, parses a string at 
478         * the indexed point of the char array and returns its value.
479         * @param doc the json doc containing the string
480         * @param index the index into that json doc where the string begins
481         * @return the value of the found string
482         * @throws MJSONException a parse exception, meaning no string was there
483         */
484        private String parseString(char[] doc, int[] index) throws MJSONException
485        {
486                final StringBuilder value=new StringBuilder("");
487                if(doc[index[0]]!='\"')
488                {
489                        throw new MJSONException("Expected quote at: "+doc[index[0]]);
490                }
491                index[0]++;
492                while(index[0] < doc.length)
493                {
494                        final char c=doc[index[0]];
495                        if(c=='\"')
496                                return value.toString();
497                        else
498                        if(c=='\\')
499                        {
500                                if(index[0] == doc.length-1)
501                                        throw new MJSONException("Unfinished escape");
502                                index[0]++;
503                                switch(doc[index[0]])
504                                {
505                                case '\"':
506                                case '\\':
507                                case '/':
508                                        value.append(doc[index[0]]);
509                                        break;
510                                case 'b': 
511                                        value.append('\b'); 
512                                        break;
513                                case 'f': 
514                                        value.append('\f');
515                                        break;
516                                case 'n': 
517                                        value.append('\n'); 
518                                        break;
519                                case 'r': 
520                                        value.append('\r'); 
521                                        break;
522                                case 't': 
523                                        value.append('\t'); 
524                                        break;
525                                case 'u':
526                                {
527                                        if(index[0] >= doc.length-5)
528                                                throw new MJSONException("Unfinished unicode escape at "+index[0]);
529                                        final byte[] hexBuf=new byte[4];
530                                        hexBuf[0] = getByteFromHex(doc,++index[0]);
531                                        hexBuf[1] = getByteFromHex(doc,++index[0]);
532                                        hexBuf[2] = getByteFromHex(doc,++index[0]);
533                                        hexBuf[3] = getByteFromHex(doc,++index[0]);
534                                        try
535                                        {
536                                                value.append(new String(hexBuf, "Cp1251"));
537                                        }
538                                        catch (final UnsupportedEncodingException e)
539                                        {
540                                                throw new MJSONException("Illegal character at"+index[0],e);
541                                        }
542                                        break;
543                                }
544                                default:
545                                        throw new MJSONException("Illegal escape character: "+doc[index[0]]);
546                                }
547                        }
548                        else
549                                value.append(c);
550                        index[0]++;
551                }
552                throw new MJSONException("Unfinished string at "+index[0]);
553        }

555        /**
556         * Given a json document char array, and an index into it, parses an array at 
557         * the indexed point of the char array and returns its value object.
558         * @param doc the json doc containing the array
559         * @param index the index into that json doc where the array begins
560         * @return the value object of the found array
561         * @throws MJSONException a parse exception, meaning no array was there
562         */
563        private Object[] parseArray(char[] doc, int[] index) throws MJSONException
564        {
565                ArrayParseState state=ArrayParseState.INITIAL;
566                final List<Object> finalSet=new ArrayList<Object>();
567                while(index[0] < doc.length)
568                {
569                        final char c=doc[index[0]];
570                        if(!Character.isWhitespace(c))
571                        {
572                                switch(state)
573                                {
574                                case INITIAL:
575                                        if(c=='[')
576                                                state = ArrayParseState.NEEDOBJECT;
577                                        else
578                                                throw new MJSONException("Expected String at "+index[0]);
579                                        break;
580                                case EXPECTOBJECT:
581                                        finalSet.add(parseElement(doc,index));
582                                        state = ArrayParseState.GOTOBJECT;
583                                        break;
584                                case NEEDOBJECT:
585                                        if(c==']')
586                                                return finalSet.toArray(new Object[0]);
587                                        else
588                                        {
589                                                finalSet.add(parseElement(doc,index));
590                                                state = ArrayParseState.GOTOBJECT;
591                                        }
592                                        break;
593                                case GOTOBJECT:
594                                        if(c==']')
595                                                return finalSet.toArray(new Object[0]);
596                                        else
597                                        if(c==',')
598                                                state = ArrayParseState.EXPECTOBJECT;
599                                        else
600                                                throw new MJSONException("Expected ] or , at "+index[0]);
601                                        break;
602                                }
603                        }
604                        index[0]++;
605                }
606                throw new MJSONException("Expected ] at "+index[0]);
607        }

609        /**
610         * Given a json document char array, and an index into it, parses a value object at 
611         * the indexed point of the char array and returns its value object.  A value object
612         * may be anything from a string, array, a JSON object, boolean, null, or a number.
613         * @param doc the json doc containing the value 
614         * @param index the index into that json doc where the value begins
615         * @return the value object of the found value
616         * @throws MJSONException a parse exception, meaning no recognized value was there
617         */
618        private Object parseElement(char[] doc, int[] index) throws MJSONException
619        {
620                switch(doc[index[0]])
621                {
622                case '\"':
623                        return parseString(doc,index);
624                case '[':
625                        return parseArray(doc,index);
626                case '{':
627                        return parseObject(doc,index);
628                case '-':
629                case '0':
630                case '1':
631                case '2':
632                case '3':
633                case '4':
634                case '5':
635                case '6':
636                case '7':
637                case '8':
638                case '9':
639                        return parseNumber(doc,index);
640                case 't':
641                        if((index[0] < doc.length-5) && (new String(doc,index[0],4).equals("true")))
642                        {
643                                index[0]+=3;
644                                return Boolean.TRUE;
645                        }
646                        throw new MJSONException("Invalid true at "+index[0]);
647                case 'f':
648                        if((index[0] < doc.length-6) && (new String(doc,index[0],5).equals("false")))
649                        {
650                                index[0]+=4;
651                                return Boolean.FALSE;
652                        }
653                        throw new MJSONException("Invalid false at "+index[0]);
654                case 'n':
655                        if((index[0] < doc.length-5) && (new String(doc,index[0],4).equals("null")))
656                        {
657                                index[0]+=3;
658                                return NULL;
659                        }
660                        throw new MJSONException("Invalid null at "+index[0]);
661                default:
662                        throw new MJSONException("Unknown character at "+index[0]);
663                }
664        }

666        /**
667         * Given a json document char array, and an index into it, parses a JSON object at 
668         * the indexed point of the char array and returns it as a mapped JSON object.
669         * @param doc the json doc containing the JSON object
670         * @param index the index into that json doc where the JSON object begins
671         * @return the value object of the found JSON object
672         * @throws MJSONException a parse exception, meaning no JSON object was there
673         */
674        private JSONObject parseObject(char[] doc, int[] index) throws MJSONException
675        {
676                final JSONObject map = new JSONObject();
677                String key = null;
678                ObjectParseState state = ObjectParseState.INITIAL;
679                while(index[0] < doc.length)
680                {
681                        final char c=doc[index[0]];
682                        if(!Character.isWhitespace(c))
683                        {
684                                switch(state)
685                                {
686                                case INITIAL:
687                                        if(c=='{')
688                                                state = ObjectParseState.NEEDKEY;
689                                        else
690                                                throw new MJSONException("Expected Key/String at "+index[0]);
691                                        break;
692                                case NEEDKEY:
693                                        if(c=='\"')
694                                        {
695                                                key = parseString(doc,index);
696                                                state = ObjectParseState.GOTKEY;
697                                        }
698                                        else
699                                        if(c=='}')
700                                                return map;
701                                        else
702                                                throw new MJSONException("Expected Key/String at "+index[0]);
703                                        break;
704                                case GOTKEY:
705                                        if(c==':')
706                                                state = ObjectParseState.NEEDOBJECT;
707                                        else
708                                                throw new MJSONException("Expected Colon at "+index[0]);
709                                        break;
710                                case NEEDOBJECT:
711                                        map.put(key, parseElement(doc,index));
712                                        state = ObjectParseState.GOTOBJECT;
713                                        break;
714                                case GOTOBJECT:
715                                        if(c==',')
716                                                state = ObjectParseState.NEEDKEY;
717                                        else
718                                        if(c=='}')
719                                                return map;
720                                        else
721                                                throw new MJSONException("Expected } or , at "+index[0]);
722                                        break;
723                                }
724                        }
725                        index[0]++;
726                }
727                throw new MJSONException("Expected } at "+index[0]);
728        }

730        /**
731         * Given a string containing a JSON object, this method will parse it
732         * into a mapped JSONObject object recursively.
733         * @param doc the JSON document that contains a top-level JSON object
734         * @return the JSON object at the top level
735         * @throws MJSONException the parse error
736         */
737        public JSONObject parseObject(String doc) throws MJSONException
738        {
739                try
740                {
741                        return parseObject(doc.toCharArray(), new int[]{0});
742                }
743                catch (final MJSONException e)
744                {
745                        throw e;
746                }
747                catch (final Exception e)
748                {
749                        throw new MJSONException("Internal error",e);
750                }
751        }

753        /**
754         * Converts a pojo field to a JSON value.
755         * @param type the class type
756         * @param val the value
757         * @return the json value
758         */
759        public String fromPOJOFieldtoJSON(Class<?> type, Object val)
760        {
761                final StringBuilder str=new StringBuilder("");
762                if(val==null)
763                        str.append("null");
764                else
765                if(type.isArray())
766                {
767                        str.append("[");
768                        final int length = Array.getLength(val);
769                        for (int i=0; i<length; i++) 
770                        {
771                                Object e = Array.get(val, i);
772                                if(i>0)
773                                        str.append(",");
774                                str.append(fromPOJOFieldtoJSON(type.getComponentType(),e));
775                        }
776                        str.append("]");
777                }
778                else
779                if(type == String.class)
780                        str.append("\"").append(toJSONString(val.toString())).append("\"");
781                else
782                if(type.isPrimitive())
783                        str.append(val.toString());
784                else
785                if((type == Float.class)||(type==Integer.class)||(type==Double.class)||(type==Boolean.class)
786                 ||(type==Long.class)||(type==Short.class)||(type==Byte.class))
787                        str.append(val.toString());
788                else
789                        str.append(fromPOJOtoJSON(val));
790                return str.toString();
791        }

793        /**
794         * Converts a pojo object to a JSON document.
795         * @param o the object to convert
796         * @return the json document
797         */
798        public String fromPOJOtoJSON(Object o)
799        {
800                StringBuilder str=new StringBuilder("");
801                str.append("{");
802                final Field[] fields = o.getClass().getDeclaredFields();
803                boolean firstField=true;
804                for(final Field field : fields)
805                {
806                        try
807                        {
808                                field.setAccessible(true);
809                                if(field.isAccessible())
810                                {
811                                        if(!firstField)
812                                                str.append(",");
813                                        else
814                                                firstField=false;
815                                        str.append("\"").append(field.getName()).append("\":");
816                                        str.append(fromPOJOFieldtoJSON(field.getType(),field.get(o)));
817                                }
818                        }
819                        catch (IllegalArgumentException e)
820                        {
821                        }
822                        catch (IllegalAccessException e)
823                        {
824                        }
825                }
826                str.append("}");
827                return str.toString();
828        }

839        /**
840         * Converts a json object to a pojo object.
841         * @param jsonObj the json object
842         * @param o the object to convert
843         */
844        public void fromJSONtoPOJO(MiniJSON.JSONObject jsonObj, Object o) throws MJSONException
845        {
846                final Field[] fields = o.getClass().getDeclaredFields();
847                for(final Field field : fields)
848                {
849                        try
850                        {
851                                field.setAccessible(true);
852                                if(field.isAccessible() && jsonObj.containsKey(field.getName()))
853                                {
854                                        
855                                        Object jo = jsonObj.get(field.getName());
856                                        if((jo == null) || (jo == MiniJSON.NULL))
857                                                field.set(o, null);
858                                        else
859                                        if(field.getType().isArray() && (jo instanceof Object[]))
860                                        {
861                                                final Object[] objs = (Object[])jo;
862                                                final Object tgt;
863                                                final Class<?> cType = field.getType().getComponentType();
864                                                tgt = Array.newInstance(cType, objs.length);
865                                                for(int i=0;i<objs.length;i++)
866                                                {
867                                                        if(objs[i].getClass() == cType)
868                                                                Array.set(tgt, i, objs[i]);
869                                                        else
870                                                        if((cType == Float.class)&&(objs[i] instanceof Double))
871                                                                Array.set(tgt, i, Float.valueOf(((Double)objs[i]).floatValue()));
872                                                        else
873                                                        if((cType == Integer.class)&&(objs[i] instanceof Long))
874                                                                Array.set(tgt, i, Integer.valueOf(((Long)objs[i]).intValue()));
875                                                        else
876                                                        if((cType == Byte.class)&&(objs[i] instanceof Long))
877                                                                Array.set(tgt, i, Byte.valueOf(((Long)objs[i]).byteValue()));
878                                                        else
879                                                        if((cType == Short.class)&&(objs[i] instanceof Long))
880                                                                Array.set(tgt, i, Short.valueOf(((Long)objs[i]).shortValue()));
881                                                        else
882                                                        if(cType.isPrimitive())
883                                                        {
884                                                                if(cType == boolean.class)
885                                                                        Array.setBoolean(tgt, i, Boolean.valueOf(objs[i].toString()).booleanValue());
886                                                                else
887                                                                if(cType == int.class)
888                                                                        Array.setInt(tgt, i, Long.valueOf(objs[i].toString()).intValue());
889                                                                else
890                                                                if(cType == short.class)
891                                                                        Array.setShort(tgt, i, Long.valueOf(objs[i].toString()).shortValue());
892                                                                else
893                                                                if(cType == byte.class)
894                                                                        Array.setByte(tgt, i, Long.valueOf(objs[i].toString()).byteValue());
895                                                                else
896                                                                if(cType == long.class)
897                                                                        Array.setLong(tgt, i, Long.valueOf(objs[i].toString()).longValue());
898                                                                else
899                                                                if(cType == float.class)
900                                                                        Array.setFloat(tgt, i, Double.valueOf(objs[i].toString()).floatValue());
901                                                                else
902                                                                if(cType == double.class)
903                                                                        Array.setDouble(tgt, i, Double.valueOf(objs[i].toString()).doubleValue());
904                                                        }
905                                                        else
906                                                        if(objs[i] instanceof JSONObject)
907                                                        {
908                                                                Object newObj = cType.newInstance();
909                                                                fromJSONtoPOJO((JSONObject)objs[i], newObj);
910                                                                Array.set(tgt, i, newObj);
911                                                        }
912                                                }
913                                                field.set(o, tgt);
914                                        }
915                                        else
916                                        if((field.getType() == String.class)&&(jo instanceof String))
917                                                field.set(o, jo);
918                                        else
919                                        if(field.getType().isPrimitive())
920                                        {
921                                                final Class<?> cType=field.getType();
922                                                if(cType == boolean.class)
923                                                        field.setBoolean(o, Boolean.valueOf(jo.toString()).booleanValue());
924                                                else
925                                                if(cType == int.class)
926                                                        field.setInt(o, Long.valueOf(jo.toString()).intValue());
927                                                else
928                                                if(cType == short.class)
929                                                        field.setShort(o, Long.valueOf(jo.toString()).shortValue());
930                                                else
931                                                if(cType == byte.class)
932                                                        field.setByte(o, Long.valueOf(jo.toString()).byteValue());
933                                                else
934                                                if(cType == long.class)
935                                                        field.setLong(o, Long.valueOf(jo.toString()).longValue());
936                                                else
937                                                if(cType == float.class)
938                                                        field.setFloat(o, Double.valueOf(jo.toString()).floatValue());
939                                                else
940                                                if(cType == double.class)
941                                                        field.setDouble(o, Double.valueOf(jo.toString()).doubleValue());
942                                                else
943                                                        field.set(o, jo);
944                                        }
945                                        else
946                                        if(jo instanceof JSONObject)
947                                        {
948                                                Object newObj = field.getType().newInstance();
949                                                fromJSONtoPOJO((JSONObject)jo, newObj);
950                                                field.set(o, newObj);
951                                        }
952                                        else
953                                        if((field.getType() == Integer.class)&&(jo instanceof Long))
954                                                field.set(o, Integer.valueOf(((Long)jo).intValue()));
955                                        else
956                                        if((field.getType() == Short.class)&&(jo instanceof Long))
957                                                field.set(o, Short.valueOf(((Long)jo).shortValue()));
958                                        else
959                                        if((field.getType() == Byte.class)&&(jo instanceof Long))
960                                                field.set(o, Byte.valueOf(((Long)jo).byteValue()));
961                                        else
962                                        if((field.getType() == Long.class)&&(jo instanceof Long))
963                                                field.set(o, Long.valueOf(((Long)jo).longValue()));
964                                        else
965                                        if((field.getType() == Double.class)&&(jo instanceof Double))
966                                                field.set(o, Double.valueOf(((Double)jo).doubleValue()));
967                                        else
968                                        if((field.getType() == Float.class)&&(jo instanceof Double))
969                                                field.set(o, Float.valueOf(((Double)jo).floatValue()));
970                                        else
971                                        if((field.getType() == Boolean.class)&&(jo instanceof Boolean))
972                                                field.set(o, Boolean.valueOf(((Boolean)jo).booleanValue()));
973                                        else
974                                                field.set(o, jo);
975                                }
976                        }
977                        catch (IllegalArgumentException e)
978                        {
979                                throw new MJSONException(e.getMessage(),e);
980                        }
981                        catch (IllegalAccessException e)
982                        {
983                                throw new MJSONException(e.getMessage(),e);
984                        }
985                        catch (InstantiationException e)
986                        {
987                                throw new MJSONException(e.getMessage(),e);
988                        }
989                }
990        }
