037public class CMThreadPoolExecutor extends ThreadPoolExecutor
038{
039        protected Map<Runnable,Thread>  active = new HashMap<Runnable,Thread>();
040        protected long                                  timeoutMillis;
041        protected CMThreadFactory               threadFactory;
042        protected int                                   queueSize = 0;
043        protected String                                poolName = "Pool";
044        protected volatile long                 lastRejectTime = 0;
045        protected volatile int                  rejectCount = 0;
046
047        protected static class CMArrayBlockingQueue<E> extends ArrayBlockingQueue<E>{
048                private static final long serialVersionUID = -4557809818979881831L;
049                public CMThreadPoolExecutor executor = null;
050
051                public CMArrayBlockingQueue(int capacity)
052                {
053                        super(capacity);
054                }
055
056                @Override 
057                public boolean offer(E o)
058                {
059                        final int allWorkingThreads = executor.getActiveCount() + super.size();
060                        return (allWorkingThreads < executor.getPoolSize()) && super.offer(o);
061                }
062        }
063
064        public CMThreadPoolExecutor(String poolName,
065                                                                int corePoolSize, int maximumPoolSize,
066                                                                long keepAliveTime, TimeUnit unit,
067                                                                long timeoutMins, int queueSize)
068        {
069                super(corePoolSize, maximumPoolSize, keepAliveTime, unit, new CMArrayBlockingQueue<Runnable>(queueSize));
070                ((CMArrayBlockingQueue<Runnable>)this.getQueue()).executor=this;
071                timeoutMillis=timeoutMins * 60 * 1000;
072                this.poolName=poolName;
073                threadFactory=new CMThreadFactory(poolName);
074                setThreadFactory(threadFactory);
075                this.queueSize=queueSize;
076                setRejectedExecutionHandler(new RejectedExecutionHandler()
077                {
078                        @Override
079                        public void rejectedExecution(Runnable r, ThreadPoolExecutor executor)
080                        {
081                                try
082                                {
083                                        executor.getQueue().put(r);
084                                }
085                                catch (final InterruptedException e)
086                                {
087                                        throw new RejectedExecutionException(e);
088                                }
089                        }
090                });
091        }
092
093        @Override
094        protected void beforeExecute(Thread t, Runnable r)
095        {
096                synchronized(active)
097                {
098                        if(t instanceof CMFactoryThread)
099                                ((CMFactoryThread)t).setRunnable(r);
100                        active.put(r,t);
101                }
102        }
103
104        @Override
105        protected void afterExecute(Runnable r, Throwable t)
106        {
107                synchronized(active)
108                {
109                        final Thread th=active.get(r);
110                        if(th instanceof CMFactoryThread)
111                                ((CMFactoryThread)th).setRunnable(null);
112                        active.remove(r);
113                }
114        }
115
116        @Override
117        public int getActiveCount()
118        {
119                return active.size();
120        }
121
122        public boolean isActive(Runnable r)
123        {
124                return active.containsKey(r);
125        }
126
127        public boolean isActiveOrQueued(Runnable r)
128        {
129                return active.containsKey(r) || getQueue().contains(r);
130        }
131
132        @Override
133        public void execute(Runnable r)
134        {
135                try
136                {
137                        if(this.getQueue().contains(r))
138                                return;
139                        super.execute(r);
140                        if((rejectCount>0)&&(System.currentTimeMillis()-lastRejectTime)>5000)
141                        {
142                                Log.warnOut(rejectCount+" Pool_"+poolName,"Threads rejected.");
143                                rejectCount=0;
144                        }
145                }
146                catch(final RejectedExecutionException e)
147                {
148                        if(r instanceof CMRunnable)
149                        {
150                                final Collection<CMRunnable> runsKilled = getTimeoutOutRuns(1);
151                                for(final CMRunnable runnable : runsKilled)
152                                {
153                                        if(runnable instanceof Session)
154                                        {
155                                                final Session S=(Session)runnable;
156                                                final StringBuilder sessionInfo=new StringBuilder("");
157                                                sessionInfo.append("status="+S.getStatus()+" ");
158                                                sessionInfo.append("active="+S.activeTimeMillis()+" ");
159                                                sessionInfo.append("online="+S.getMillisOnline()+" ");
160                                                sessionInfo.append("lastloop="+(System.currentTimeMillis()-S.getInputLoopTime())+" ");
161                                                sessionInfo.append("addr="+S.getAddress()+" ");
162                                                sessionInfo.append("mob="+((S.mob()==null)?"null":S.mob().Name()));
163                                                Log.errOut("Pool_"+poolName,"Timed-Out Runnable: "+sessionInfo.toString());
164                                        }
165                                        else
166                                        if(runnable instanceof TickableGroup)
167                                        {
168                                                final TickableGroup G=(TickableGroup)runnable;
169                                                Log.errOut("Pool_"+poolName,"Timed-Out Runnable: "+G.getName()+"-"+G.getStatus()+"\n\r");
170                                        }
171                                        else
172                                                Log.errOut("Pool_"+poolName,"Timed-Out Runnable: "+runnable.toString());
173                                }
174                        }
175                        lastRejectTime=System.currentTimeMillis();
176                        rejectCount++;
177                }
178        }
179
180        public Collection<CMRunnable> getTimeoutOutRuns(int maxToKill)
181        {
182                final LinkedList<CMRunnable> timedOut=new LinkedList<CMRunnable>();
183                if(timeoutMillis<=0)
184                        return timedOut;
185                final LinkedList<Thread> killedOut=new LinkedList<Thread>();
186                synchronized(active)
187                {
188                        try
189                        {
190                                for (final Runnable runnable : active.keySet())
191                                {
192                                        if(runnable instanceof CMRunnable)
193                                        {
194                                                final CMRunnable cmRunnable=(CMRunnable)runnable;
195                                                final Thread thread=active.get(runnable);
196                                                if(cmRunnable.activeTimeMillis() > timeoutMillis)
197                                                {
198                                                        if(timedOut.size() >= maxToKill)
199                                                        {
200                                                                CMRunnable leastWorstOffender=null;
201                                                                for(final CMRunnable r : timedOut)
202                                                                {
203                                                                        if((leastWorstOffender != null)
204                                                                        &&(r.activeTimeMillis() < leastWorstOffender.activeTimeMillis()))
205                                                                                leastWorstOffender=r;
206                                                                }
207                                                                if(leastWorstOffender!=null)
208                                                                {
209                                                                        if(cmRunnable.activeTimeMillis() < leastWorstOffender.activeTimeMillis())
210                                                                                continue;
211                                                                        else
212                                                                                timedOut.remove(leastWorstOffender);
213                                                                }
214                                                        }
215                                                        timedOut.add(cmRunnable);
216                                                        killedOut.add(thread);
217                                                }
218                                        }
219                                }
220                        }
221                        catch(final Exception e)
222                        {
223                        }
224                }
225                try
226                {
227                        while(killedOut.size()>0)
228                        {
229                                final Thread t = killedOut.remove();
230                                active.remove(t);
231                                CMLib.killThread(t,100,3);
232                        }
233                }
234                catch(final Exception e)
235                {
236                }
237                return timedOut;
238        }
