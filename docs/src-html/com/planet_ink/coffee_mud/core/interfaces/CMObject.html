032/**
033 * The general base interface which is implemented by every class
034 * which the CoffeeMud ClassLoader (CMClass) handles.
035 * @see com.planet_ink.coffee_mud.core.CMClass
036 * @author Bo Zimmerman
037 *
038 */
039public interface CMObject extends Cloneable, Comparable<CMObject>
040{
041        /**
042         * The CoffeeMud Java Class ID shared by all instances of
043         * this object.  Unlike the Java Class name, this method
044         * does not include package information.  However, it must
045         * return a String value unique to its class category in
046         * the ClassLoader.  Class categories include Libraries, Common,
047         * Areas, Abilities, Behaviors, CharClasses, Commands, Exits
048         * Locales, MOBS, Races, WebMacros, Basic Items, Armor,
049         * Weapons, ClanItems, Tech.  The name is typically identical
050         * to the class name.
051         * @return the name of this class
052         */
053        public String ID();
054        /**
055         * The displayable name of this object.  May be modified by phyStats() object. Is
056         * derived from the Name().
057         * @see  Environmental#Name()
058         * @return the modified final name of this object on the map.
059         */
060        public String name();
061        /**
062         * Returns a new instance of this class.
063         * @return a new instance of this class
064         */
065        public CMObject newInstance();
066        /**
067         * Similar to Cloneable.clone(), but does its best to make sure that
068         * any internal objects to this class are also copyOfed.
069         * @return a clone of this object
070         */
071        public CMObject copyOf();
072
073        /**
074         * Called ONCE after all objects are loaded, but before the map is read in
075         * during initialization.
076         */
077        public void initializeClass();
078        
079}
