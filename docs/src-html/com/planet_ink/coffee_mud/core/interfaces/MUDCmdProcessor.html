027/**
028*
029* Something that is capable of taking user-entered commands, parsing
030* them, and then executing them.
031* @author Bo Zimmerman
032*
033*/
034public interface MUDCmdProcessor
035{
036
037        /** constant mask for the metaflags parameter for execute and preexecute, means being mpforced*/
038        public static final int METAFLAG_MPFORCED=1;
039        /** constant mask for the metaflags parameter for execute and preexecute, means being ordered*/
040        public static final int METAFLAG_ORDER=2;
041        /** constant mask for the metaflags parameter for execute and preexecute, means being possessed*/
042        public static final int METAFLAG_POSSESSED=4;
043        /** constant mask for the metaflags parameter for execute and preexecute, means being snooped*/
044        public static final int METAFLAG_SNOOPED=8;
045        /** constant mask for the metaflags parameter for execute and preexecute, means being forced with AS*/
046        public static final int METAFLAG_AS=16;
047        /** constant mask for the metaflags parameter for execute and preexecute, means being forced with spells*/
048        public static final int METAFLAG_FORCED=32;
049        /** constant mask for the metaflags parameter for execute and preexecute, means being a meta-command*/
050        public static final int METAFLAG_ASMESSAGE=64;
051        /** constant mask for the metaflags parameter for execute and preexecute, means always after prev command*/
052        public static final int METAFLAG_INORDER=128;
053        
054        /**
055         * Parses the given command string tokens to determine what kind of command is
056         * to be executed, and depending on how many actions the player has remaining
057         * this tick, either executes the command or puts it on the que for automatic
058         * execution once sufficient actions are available.
059         * @see MUDCmdProcessor#dequeCommand()
060         * @see MUDCmdProcessor#clearCommandQueue()
061         * @see MUDCmdProcessor#doCommand(List, int)
062         * @see MUDCmdProcessor#actions()
063         * @param commands the parsed command string tokens
064         * @param metaFlags meta-command flags to send to the command, if any
065         * @param actionCost either 0 to let the action object decide, or an override cost
066         */
067        public void enqueCommand(List<String> commands, int metaFlags, double actionCost);
068        
069        /**
070         * Parses the given commands string tokens to determine what kind of commands are
071         * to be executed, and depending on how many actions the player has remaining
072         * this tick, either executes the commands or puts them on the que for automatic
073         * execution once sufficient actions are available.
074         * @see MUDCmdProcessor#dequeCommand()
075         * @see MUDCmdProcessor#clearCommandQueue()
076         * @see MUDCmdProcessor#doCommand(List, int)
077         * @see MUDCmdProcessor#actions()
078         * @param commands the parsed commands string tokens
079         * @param metaFlags meta-command flags to send to the command, if any
080         */
081        public void enqueCommands(List<List<String>> commands, int metaFlags);
082        
083        /**
084         * Parses the given command string tokens to determine what kind of command is
085         * to be executed, and depending on how many actions the player has remaining
086         * this tick, either executes the command or puts it on the que for automatic
087         * execution once sufficient actions are available.
088         * @see MUDCmdProcessor#dequeCommand()
089         * @see MUDCmdProcessor#clearCommandQueue()
090         * @see MUDCmdProcessor#doCommand(List, int)
091         * @see MUDCmdProcessor#actions()
092         * @param commands the parsed command string tokens
093         * @param metaFlags meta-command flags to send to the command, if any
094         * @param actionCost either 0 to let the action object decide, or an override cost
095         */
096        public void prequeCommand(List<String> commands, int metaFlags, double actionCost);
097        
098        /**
099         * Parses the given command string tokens to determine what kind of command is
100         * to be executed, and depending on how many actions the player has remaining
101         * this tick, either executes the command or puts it on the que for automatic
102         * execution once sufficient actions are available.
103         * @see MUDCmdProcessor#dequeCommand()
104         * @see MUDCmdProcessor#clearCommandQueue()
105         * @see MUDCmdProcessor#doCommand(List, int)
106         * @param commands the parsed command string tokens
107         * @param metaFlags meta-command flags to send to the command, if any
108         */
109        public void prequeCommands(List<List<String>> commands, int metaFlags);
110
111        /**
112         * If this processor has enough action points to perform the top command on the que,
113         * then this method will execute that action.
114         * It returns whether there are still more actions on the que.
115         * @see MUDCmdProcessor#enqueCommand(List, int, double)
116         * @see MUDCmdProcessor#clearCommandQueue()
117         * @see MUDCmdProcessor#doCommand(List, int)
118         * @see MUDCmdProcessor#commandQueSize()
119         * @see MUDCmdProcessor#actions()
120         * @return true if there are more actions on the que, false otherwise
121         */
122        public boolean dequeCommand();
123        
124        /**
125         * Cancels and empties the command que of this processor.
126         * @see MUDCmdProcessor#dequeCommand()
127         * @see MUDCmdProcessor#enqueCommand(List, int, double)
128         * @see MUDCmdProcessor#doCommand(List, int)
129         * @see MUDCmdProcessor#commandQueSize()
130         * @see MUDCmdProcessor#actions()
131         */
132        public void clearCommandQueue();
133        
134        /**
135         * Returns the number of commands on this processors que
136         * @see MUDCmdProcessor#dequeCommand()
137         * @see MUDCmdProcessor#clearCommandQueue()
138         * @see MUDCmdProcessor#doCommand(List, int)
139         * @see MUDCmdProcessor#enqueCommand(List, int, double)
140         * @see MUDCmdProcessor#actions()
141         * @return the number of commands on this processors que
142         */
143        public int commandQueSize();
144        
145        /**
146         * Forces this processor to parse the given command string tokens,
147         * determine a command to execute, and then execute it, 
148         * regardless of whether there are any action points
149         * remaining.
150         * @see MUDCmdProcessor#dequeCommand()
151         * @see MUDCmdProcessor#clearCommandQueue()
152         * @see MUDCmdProcessor#commandQueSize()
153         * @see MUDCmdProcessor#enqueCommand(List, int, double)
154         * @see MUDCmdProcessor#actions()
155         * @param commands the command string tokens
156         * @param metaFlags any meta-command flags
157         */
158        public void doCommand(List<String> commands, int metaFlags);
159        
160        /**
161         * Returns the number of action points that this processor
162         * has to spend towards performing any commands, skills,
163         * or whatever needs action points to be spent on them.
164         * They are usually replenished once per tick.
165         * @see MUDCmdProcessor#enqueCommand(List, int, double)
166         * @see MUDCmdProcessor#setActions(double)
167         * @return the number of action points
168         */
169        public double actions();
170        
171        /**
172         * Sets the number of action points that this processor
173         * has to spend towards performing any commands, skills,
174         * or whatever needs action points to be spent on them.
175         * They are usually replenished once per tick.
176         * @see MUDCmdProcessor#enqueCommand(List, int, double)
177         * @see MUDCmdProcessor#actions()
178         * @param remain the number of action points
179         */
180        public void setActions(double remain);
181}
