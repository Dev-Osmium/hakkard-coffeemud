032/**
033 * The interface represents a 3d cubed thing.
034 * @author Bo Zimmerman
035 *
036 */
037public interface BoundedObject
038{
039        public BoundedCube getBounds();
040
041        public static class BoundedCube implements Comparable<BoundedCube>
042        {
043                public long     lx, ty, iz = 0;
044                public long     rx, by, oz = 0;
045
046                public BoundedCube()
047                {
048                        super();
049                }
050
051                public BoundedCube(long lx, long rx, long ty, long by, long iz, long oz)
052                {
053                        super();
054                        this.lx = lx; 
055                        this.rx = rx;
056                        this.ty = ty;
057                        this.by = by;
058                        this.iz = iz;
059                        this.oz = oz;
060                }
061
062                public BoundedCube(long[] coords, long radius)
063                {
064                        super();
065                        this.lx = coords[0] - radius; 
066                        this.rx = coords[0] + radius;
067                        this.ty = coords[1] - radius; 
068                        this.by = coords[1] + radius;
069                        this.iz = coords[2] - radius; 
070                        this.oz = coords[2] + radius;
071                }
072
073                public BoundedCube(BoundedCube l)
074                {
075                        super();
076                        set(l);
077                }
078
079                public void set(BoundedCube l)
080                {
081                        this.lx = l.lx; 
082                        this.rx = l.rx;
083                        this.ty = l.ty; 
084                        this.by = l.by;
085                        this.iz = l.iz; 
086                        this.oz = l.oz;
087                }
088                
089                public void union(BoundedCube l)
090                {
091                        if(l.lx < lx)
092                                lx=l.lx;
093                        if(l.rx > rx)
094                                rx=l.rx;
095                        if(l.ty < ty)
096                                ty=l.ty;
097                        if(l.by > by)
098                                by=l.by;
099                        if(l.iz < iz)
100                                iz=l.iz;
101                        if(l.oz > oz)
102                                oz=l.oz;
103                }
104
105                public long radius()
106                {
107                        return Math.round(Math.sqrt(((rx - lx) * (rx - lx))
108                                                                           +((by - ty) * (by - ty))
109                                                                           +((oz - iz) * (oz - iz))));
110                }
111
112                public BoundedCube expand(double[] direction, long distance)
113                {
114                        final BoundedCube cube=new BoundedCube(this);
115                        final double x1=Math.cos(direction[0])*Math.sin(direction[1]);
116                        final double y1=Math.sin(direction[0])*Math.sin(direction[1]);
117                        final double z1=Math.cos(direction[1]);
118                        final long[] oldCenter=new long[]{((lx+rx)/2),((ty+rx)/2),((iz+oz)/2)};
119                        final long[] newCenter=new long[]{
120                                        oldCenter[0]+Math.round(CMath.mul(distance,x1)),
121                                        oldCenter[1]+Math.round(CMath.mul(distance,y1)),
122                                        oldCenter[2]+Math.round(CMath.mul(distance,z1))};
123                        if(newCenter[0]>oldCenter[0])
124                                cube.rx+=newCenter[0]-oldCenter[0];
125                        else
126                                cube.lx+=newCenter[0]-oldCenter[0];
127                        if(newCenter[1]>oldCenter[1])
128                                cube.ty+=newCenter[1]-oldCenter[1];
129                        else
130                                cube.by+=newCenter[1]-oldCenter[1];
131                        if(newCenter[2]>oldCenter[2])
132                                cube.iz+=newCenter[2]-oldCenter[2];
133                        else
134                                cube.oz+=newCenter[2]-oldCenter[2];
135                        return cube;
136                }
137
138                public boolean intersects(BoundedCube two)
139                {
140                        if(two==null)
141                                return false;
142                        return ( 
143                                ((lx <= two.lx && two.lx <= rx) || (two.lx <= lx && lx <= two.rx))
144                        &&      ((ty <= two.ty && two.ty <= by) || (two.ty <= ty && ty <= two.by))
145                        &&      ((iz <= two.iz && two.iz <= oz) || (two.iz <= iz && iz <= two.oz)) 
146                        );
147                }
148
149                public boolean contains(long x, long y, long z)
150                {
151                        return ((x >= lx)
152                                  &&(x <= rx)
153                                  &&(y >= ty)
154                                  &&(y <= by)
155                                  &&(z >= iz)
156                                  &&(z <= oz));
157                }
158
159                public long width()
160                {
161                        return rx - lx;
162                }
163
164                public long height()
165                {
166                        return by - ty;
167                }
168
169                public long depth()
170                {
171                        return oz - iz;
172                }
173
174                @Override
175                public int compareTo(BoundedCube o)
176                {
177                        if(lx<o.lx)
178                                return -1;
179                        if(lx>o.lx)
180                                return 1;
181                        if(rx<o.rx)
182                                return -1;
183                        if(rx>o.rx)
184                                return 1;
185                        if(ty<o.ty)
186                                return -1;
187                        if(ty>o.ty)
188                                return 1;
189                        if(by<o.by)
190                                return -1;
191                        if(by>o.by)
192                                return 1;
193                        if(iz<o.iz)
194                                return -1;
195                        if(iz>o.iz)
196                                return 1;
197                        if(oz<o.oz)
198                                return -1;
199                        if(oz>o.oz)
200                                return 1;
201                        return 0;
202                }
203        }
204}
