022/**
023*
024* A place where people might be, can be either abstract (like an area),
025* or concrete (like a Room)
026* @author Bo Zimmerman
027*
028*/
029public interface Places extends PhysicalAgent
030{
031        /** a constant code for {@link Places#getAtmosphereCode()} that denotes that the atmo is inherited from a parent */
032        public final static int ATMOSPHERE_INHERIT = -1;
033
034        /**
035         * Returns the resource (or -1) that represents the atmosphere of this area.
036         * Since most rooms inherit their atmosphere from the area, this is important.
037         * Return -1 to have this area inherit its atmosphere from parents (which
038         * would ultimately go back to RESOURCE_AIR)
039         * @see com.planet_ink.coffee_mud.Items.interfaces.RawMaterial#MATERIAL_GAS
040         * @return the RawMaterial resource, or -1
041         */
042        public int getAtmosphereCode();
043
044        /**
045         * Returns the resource that represents the atmosphere of this area.
046         * Since most rooms inherit their atmosphere from the area, this is important.
047         * Could have this area inherit its atmosphere from parents (which
048         * would ultimately go back to RESOURCE_AIR)
049         * @see com.planet_ink.coffee_mud.Items.interfaces.RawMaterial#MATERIAL_GAS
050         * @return the RawMaterial resource
051         */
052        public int getAtmosphere();
053
054        /**
055         * Sets the resource (or -1) that represents the atmosphere of this area.
056         * Since most rooms inherit their atmosphere from the area, this is important.
057         * Return -1 to have this area inherit its atmosphere from parents (which
058         * would ultimately go back to RESOURCE_AIR)
059         * @see com.planet_ink.coffee_mud.Items.interfaces.RawMaterial#MATERIAL_GAS
060         * @param resourceCode the RawMaterial resource to use
061         */
062        public void setAtmosphere(int resourceCode);
063
064        /**
065         * Returns a bitmap of climate flags for this area which will be used to influence
066         * the weather for the area in addition to season and other factors.
067         * @see com.planet_ink.coffee_mud.core.interfaces.Places#CLIMASK_COLD
068         * @return a CLIMASK bitmap
069         */
070        public int getClimateTypeCode();
071        /**
072         * Returns a bitmap of climate flags for this area which will be used to influence
073         * the weather for the area in addition to season and other factors.
074         * @see com.planet_ink.coffee_mud.core.interfaces.Places#CLIMASK_COLD
075         * @param newClimateType a CLIMASK bitmap
076         */
077        public void setClimateType(int newClimateType);
078
079        /**
080         * Returns a bitmap of the climate for this place.  If the climate is CLIMASK_INHERIT,
081         * then it will look to parent objects, such as areas, and parent areas, until it
082         * eventually finds a non-inherit, or returns CLIMASK_NORMAL;
083         * @see com.planet_ink.coffee_mud.core.interfaces.Places#CLIMASK_COLD
084         * @return a derived climate
085         */
086        public int getClimateType();
087
088        /**     Bitmap climate flag meaning that the area has inherited weather.  @see com.planet_ink.coffee_mud.core.interfaces.Places#climateType() */
089        public final static int CLIMASK_INHERIT = -1;
090        /**     Bitmap climate flag meaning that the area has normal weather.  @see com.planet_ink.coffee_mud.core.interfaces.Places#climateType() */
091        public final static int CLIMASK_NORMAL=0;
092        /**     Bitmap climate flag meaning that the area has wet weather.  @see com.planet_ink.coffee_mud.core.interfaces.Places#climateType() */
093        public final static int CLIMASK_WET=1;
094        /**     Bitmap climate flag meaning that the area has cold weather.  @see com.planet_ink.coffee_mud.core.interfaces.Places#climateType() */
095        public final static int CLIMASK_COLD=2;
096        /**     Bitmap climate flag meaning that the area has windy weather.  @see com.planet_ink.coffee_mud.core.interfaces.Places#climateType() */
097        public final static int CLIMASK_WINDY=4;
098        /**     Bitmap climate flag meaning that the area has hot weather.  @see com.planet_ink.coffee_mud.core.interfaces.Places#climateType() */
099        public final static int CLIMASK_HOT=8;
100        /**     Bitmap climate flag meaning that the area has dry weather.  @see com.planet_ink.coffee_mud.core.interfaces.Places#climateType() */
101        public final static int CLIMASK_DRY=16;
102        /**     Indexed description of the CLIMASK_ bitmap constants in all possible combinations.
103         * @see com.planet_ink.coffee_mud.core.interfaces.Places#CLIMASK_NORMAL
104         */
105        public final static String[] CLIMATE_DESCS={"NORMAL","WET","COLD","WINDY","HOT","DRY"};
106        /**     Number of CLIMASK_ constants.  @see com.planet_ink.coffee_mud.core.interfaces.Places#climateType() */
107        public final static int NUM_CLIMATES=6;
108        /**     Bitmap climate flag meaning that the area has all weather modifiers.  @see com.planet_ink.coffee_mud.core.interfaces.Places#climateType() */
109        public final static int ALL_CLIMATE_MASK=31;
110}
