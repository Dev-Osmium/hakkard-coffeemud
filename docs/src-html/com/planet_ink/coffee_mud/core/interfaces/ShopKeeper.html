034/**
035 * An interface for objects capable of being a shopkeeper, usually a MOB or an Ability.
036 * @author Bo Zimmerman
037 *
038 */
039public interface ShopKeeper extends Environmental, Economics
040{
041        /** shopkeeper type constant, means they buy anything */
042        public final static int DEAL_ANYTHING=0;
043        /** shopkeeper type constant, means they buy items not covered by other constants */
044        public final static int DEAL_GENERAL=1;
045        /** shopkeeper type constant, means they buy armor */
046        public final static int DEAL_ARMOR=2;
047        /** shopkeeper type constant, means they buy magic items*/
048        public final static int DEAL_MAGIC=3;
049        /** shopkeeper type constant, means they buy weapons*/
050        public final static int DEAL_WEAPONS=4;
051        /** shopkeeper type constant, means they buy pets*/
052        public final static int DEAL_PETS=5;
053        /** shopkeeper type constant, means they buy leather stuff*/
054        public final static int DEAL_LEATHER=6;
055        /** shopkeeper type constant, means they buy only what they are told to sell*/
056        public final static int DEAL_INVENTORYONLY=7;
057        /** shopkeeper type constant, means they train players in skills they sell*/
058        public final static int DEAL_TRAINER=8;
059        /** shopkeeper type constant, means they cast spells they sell on players*/
060        public final static int DEAL_CASTER=9;
061        /** shopkeeper type constant, means they buy jewelry*/
062        public final static int DEAL_JEWELLER=10;
063        /** shopkeeper type constant, means they buy potions*/
064        public final static int DEAL_ALCHEMIST=11;
065        /** shopkeeper type constant, means they are a banker for players, and implement the Banker interface*/
066        public final static int DEAL_BANKER=12;
067        /** shopkeeper type constant, means they buy and sell property in their area to players*/
068        public final static int DEAL_LANDSELLER=13;
069        /** shopkeeper type constant, means they buy electronics items*/
070        public final static int DEAL_ANYTECHNOLOGY=14;
071        /** shopkeeper type constant, means they buy and sell property in their area to clans*/
072        public final static int DEAL_CLANDSELLER=15;
073        /** shopkeeper type constant, means they buy and sell raw foodstuff*/
074        public final static int DEAL_FOODSELLER=16;
075        /** shopkeeper type constant, means they buy and sell raw meats*/
076        public final static int DEAL_BUTCHER=17;
077        /** shopkeeper type constant, means they buy and sell raw non-meat foodstuffs*/
078        public final static int DEAL_GROWER=18;
079        /** shopkeeper type constant, means they buy raw leathers and hides*/
080        public final static int DEAL_HIDESELLER=19;
081        /** shopkeeper type constant, means they buy raw lumber*/
082        public final static int DEAL_LUMBERER=20;
083        /** shopkeeper type constant, means they buy raw metals*/
084        public final static int DEAL_METALSMITH=21;
085        /** shopkeeper type constant, means they buy raw stones*/
086        public final static int DEAL_STONEYARDER=22;
087        /** shopkeeper type constant, means they are a banker for clans, and implement the Banker interface*/
088        public final static int DEAL_CLANBANKER=23;
089        /** shopkeeper type constant, means they sell InnKeys*/
090        public final static int DEAL_INNKEEPER=24;
091        /** shopkeeper type constant, means they buy and sell SpaceShip areas to players*/
092        public final static int DEAL_SHIPSELLER=25;
093        /** shopkeeper type constant, means they buy and sell SpaceShip areas to clans*/
094        public final static int DEAL_CSHIPSELLER=26;
095        /** shopkeeper type constant, means they buy and sell intelligent mobs as slaves*/
096        public final static int DEAL_SLAVES=27;
097        /** shopkeeper type constant, means they handle mail for players, and implement the Postman interface*/
098        public final static int DEAL_POSTMAN=28;
099        /** shopkeeper type constant, means they handle mail for clans, and implement the Postman interface*/
100        public final static int DEAL_CLANPOSTMAN=29;
101        /** shopkeeper type constant, means they handle auctions, and implement the Auctioneer interface*/
102        public final static int DEAL_AUCTIONEER=30;
103        /** shopkeeper type constant, means they buy and sell musical instruments*/
104        public final static int DEAL_INSTRUMENTS=31;
105        /** shopkeeper type constant, means they buy and sell any books*/
106        public final static int DEAL_BOOKS=32;
107        /** shopkeeper type constant, means they buy and sell any readables*/
108        public final static int DEAL_READABLES=33;
109
110        /** shopkeeper integer sets denoting the DEAL_* constants which conflict with each other */
111        public final static int[][] DEAL_CONFLICTS={
112                {DEAL_POSTMAN,DEAL_CLANPOSTMAN},
113                {DEAL_CLANBANKER,DEAL_BANKER},
114                {DEAL_SHIPSELLER,DEAL_CSHIPSELLER,DEAL_LANDSELLER,DEAL_CLANDSELLER},
115                {DEAL_TRAINER,DEAL_CASTER},
116        };
117
118        /** A list of strings describing the DEAL_* constants, in their numeric value order. */
119        public final static String[] DEAL_DESCS=
120        {
121                "ANYTHING","GENERAL","ARMOR","MAGIC","WEAPONS",
122                "PETS","LEATHER","INVENTORY ONLY","TRAINER",
123                "CASTER","JEWELRY","POTIONS","BANKER","LAND",
124                "ANY TECHNOLOGY","CLAN LAND","FOODS","MEATS",
125                "VEGETABLES","HIDES","LUMBER","METALS","ROCKS",
126                "CLAN BANKER", "INN KEEPER", "SHIP SELLER",
127                "CLAN SHIP SELLER", "SLAVES", "POSTMAN", "CLAN POSTMAN",
128                "AUCTIONEER","INSTRUMENTS","BOOKS","READABLES"
129        };
130
131        /**
132         * This class represents a given price for a given item in the shopkeepers inventory. It is usually
133         * calculated for a given buyer.
134         */
135        public static class ShopPrice
136        {
137                /** the price of the item in base currency gold */
138                public double absoluteGoldPrice=0.0;
139                /** the number of experience points required to purchase the item */
140                public int experiencePrice=0;
141                /** the number of quest points required to purchase the item */
142                public int questPointPrice=0;
143        }
144
145        /**
146         * the CoffeeShop method to access the shopkeepers store of goods
147         * @see com.planet_ink.coffee_mud.Common.interfaces.CoffeeShop
148         * @return the CoffeeShop object
149         */
150        public CoffeeShop getShop();
151
152        /**
153         * Returns the ShopKeeper DEAL_* mask describing what is sold or bought by this ShopKeeper
154         * @see com.planet_ink.coffee_mud.core.interfaces.ShopKeeper#DEAL_DESCS
155         * @see com.planet_ink.coffee_mud.core.interfaces.ShopKeeper#setWhatIsSoldMask(long)
156         * @return the dealer type constants to the 2nd power, shifted 8 bits left
157         */
158        public long getWhatIsSoldMask();
159        /**
160         * Returns whether the given type of good is sold by this shopkeeper.
161         * @param deal  the ShopKeeper DEAL_* constant describing what is sold or bought by this ShopKeeper
162         * @see com.planet_ink.coffee_mud.core.interfaces.ShopKeeper#DEAL_DESCS
163         * @see com.planet_ink.coffee_mud.core.interfaces.ShopKeeper#getWhatIsSoldMask()
164         * @return true if the shopkeeper will make such a deal
165         */
166        public boolean isSold(int deal);
167        /**
168         * Sets the encoded ShopKeeper DEAL_* constants describing what is sold or bought by this ShopKeeper
169         * @see com.planet_ink.coffee_mud.core.interfaces.ShopKeeper#DEAL_DESCS
170         * @see com.planet_ink.coffee_mud.core.interfaces.ShopKeeper#addSoldType(int)
171         * @param newSellCode the dealer type constants to the 2nd power, shifted 8 bits left
172         */
173        public void setWhatIsSoldMask(long newSellCode);
174
175        /**
176        * Adds the ShopKeeper DEAL_* constants describing what is sold or bought by this ShopKeeper
177        * to the existing shopkeeper mask.  A value of 0 will clear the whole mask.
178         * @see com.planet_ink.coffee_mud.core.interfaces.ShopKeeper#DEAL_DESCS
179         * @see com.planet_ink.coffee_mud.core.interfaces.ShopKeeper#isSold(int)
180        * @param dealType the ShopKeeper DEAL_* constants describing what is sold or bought by this ShopKeeper
181        */
182        public void addSoldType(int dealType);
183
184        /**
185         * Based on the value of this ShopKeepers whatIsSold() method, this will return a displayable string
186         * describing that type.
187         * @see ShopKeeper#isSold(int)
188         * @return a description of the whatIsSold() code
189         */
190        public String storeKeeperString();
191        /**
192         * Returns whether this ShopKeeper deals in the type of item passed in.  The determination is based
193         * on the whatIsSold() code.
194         * @see ShopKeeper#isSold(int)
195         * @param thisThang the item to determine if the shopkeeper deals  in
196         * @return whether the shopkeeper deals in the type of item passed in
197         */
198        public boolean doISellThis(Environmental thisThang);
199        
200        /**
201         * Sets the zapper mask which applies to items to determine whether they are bought and solid
202         * by this shopkeeper.
203         * @see ShopKeeper#isSold(int)
204         * @see ShopKeeper#getWhatIsSoldZappermask()
205         * @see MaskingLibrary
206         * @param newSellMask the item zappermask
207         */
208        public void setWhatIsSoldZappermask(String newSellMask);
209
210        /**
211         * Returns the zapper mask which applies to items to determine whether they are bought and solid
212         * by this shopkeeper.
213         * @see ShopKeeper#isSold(int)
214         * @see ShopKeeper#setWhatIsSoldZappermask(String)
215         * @see MaskingLibrary
216         * @return the item zappermask
217         */
218        public String getWhatIsSoldZappermask();
219}
