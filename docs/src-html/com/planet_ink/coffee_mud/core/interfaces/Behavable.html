024/**
025*
026* Something that can behave -- means almost everything!
027* @author Bo Zimmerman
028*
029*/
030public interface Behavable
031{
032        /**
033         * Add a new behavior to this object.  After calling this method,
034         * recoverPhyStats() should be called next in case this behavior object modifies the stats.
035         * A Behavior with a given ID() can only be added once per object.
036         * @see com.planet_ink.coffee_mud.Behaviors.interfaces.Behavior
037         * @see com.planet_ink.coffee_mud.core.interfaces.Affectable#recoverPhyStats()
038         * @param to The behavior object to add.
039         */
040        public void addBehavior(Behavior to);
041
042        /**
043         * Delete a behavior from this object.  After calling this method,
044         * recoverPhyStats() should be called next in case this behavior object modified the stats.
045         * @see com.planet_ink.coffee_mud.Behaviors.interfaces.Behavior
046         * @see com.planet_ink.coffee_mud.core.interfaces.Affectable#recoverPhyStats()
047         * @param to The behavior object to remove.
048         */
049        public void delBehavior(Behavior to);
050
051        /**
052         * The number of behaviors this object has.
053         * @see com.planet_ink.coffee_mud.Behaviors.interfaces.Behavior
054         * @return the number of behaviors
055         */
056        public int numBehaviors();
057
058        /**
059         * Returns a behavior object on this object. May return null even if the index
060         * is correct to mark a race condition.
061         * @see com.planet_ink.coffee_mud.Behaviors.interfaces.Behavior
062         * @see com.planet_ink.coffee_mud.core.interfaces.PhysicalAgent#numBehaviors()
063         * @param index which object to return
064         * @return the behavior object
065         */
066        public Behavior fetchBehavior(int index);
067
068        /**
069         * Returns a behavior object listed on this object. The object will
070         * be the one with the same ID() string as passed in.
071         * @see com.planet_ink.coffee_mud.Behaviors.interfaces.Behavior
072         * @see CMObject#ID()
073         * @param ID the Behavior ID to look for
074         * @return the behavior object
075         */
076        public Behavior fetchBehavior(String ID);
077
078        /**
079         * Returns an enumerator of all the behaviors on this object.
080         * @see com.planet_ink.coffee_mud.Behaviors.interfaces.Behavior
081         * @return an enumerator of all the behaviors on this object.
082         */
083        public Enumeration<Behavior> behaviors();
084
085        /**
086         * Applies the given code to each behavior on this object
087         * @param applier code to execute against each object
088         */
089        public void eachBehavior(final EachApplicable<Behavior> applier);
090
091        /**
092         * Add a new runnable script to this object.  Objects which are
093         * not mobs or areas will gain a temporary tick service for
094         * this script.
095         * @see com.planet_ink.coffee_mud.Common.interfaces.ScriptingEngine
096         * @param s the scripting engine, fully populated, to add
097         */
098        public void addScript(ScriptingEngine s);
099
100        /**
101         * Removes all behaviors from this object.
102         */
103        public void delAllBehaviors();
104
105        /**
106         * Remove a running script from this object.
107         * @see com.planet_ink.coffee_mud.Common.interfaces.ScriptingEngine
108         * @param s the specific scripting engine to remove
109         */
110        public void delScript(ScriptingEngine s);
111
112        /**
113         * Removes all executing scripts from this object.
114         */
115        public void delAllScripts();
116
117        /**
118         * Return the number of scripts running on this object
119         * @return number of scripts
120         */
121        public int numScripts();
122
123        /**
124         * Retrieve one of the enumerated scripts running on this
125         * object
126         * @see com.planet_ink.coffee_mud.Common.interfaces.ScriptingEngine
127         * @param x which script to return
128         * @return the scripting engine
129         */
130        public ScriptingEngine fetchScript(int x);
131
132        /**
133         * Returns an enumerator of all the scripts on this object.
134         * @see com.planet_ink.coffee_mud.Common.interfaces.ScriptingEngine
135         * @return an enumerator of all the scripts on this object.
136         */
137        public Enumeration<ScriptingEngine> scripts();
138
139        /**
140         * Applies the given code to each scripting engine on this object
141         * @param applier code to execute against each object
142         */
143        public void eachScript(final EachApplicable<ScriptingEngine> applier);
144}
