036/**
037 * This class represents a thread, consisting of a group of Tickable objects  receiving periodic calls
038 * to their tick(Tickable,int) methods by this thread object
039 * @see Tickable
040 * @see Tickable#tick(Tickable, int)
041 * @author Bo Zimmerman
042 *
043 */
044public interface TickableGroup extends CMRunnable
045{
046        /**
047         * Returns the current or last TickClient object which this thread made a tick(Tickable,int) method
048         * call to.
049         * @see Tickable
050         * @see Tickable#tick(Tickable, int)
051         * @return the TickClient object last accessed
052         */
053        public TickClient getLastTicked();
054
055        /**
056         * Returns the name of this group
057         * @return the name
058         */
059        public String getName();
060
061        /**
062         * Returns the amount of time, in ms, between calls to this group
063         * @return the time between ticks
064         */
065        public long getTickInterval();
066
067        /**
068         * Returns the next time, in ms, when this group should tick.
069         * @return the next time, in ms, when this group should tick.
070         */
071        public long getNextTickTime();
072
073        /**
074         * Orders this group to do any necessary cleanup before going away.
075         */
076        public void shutdown();
077
078        /**
079         * Adds another ticking object to this group
080         * @see TickClient
081         * @param C the client to add
082         */
083        public void addTicker(TickClient C);
084
085        /**
086         * Removes the given ticking object from this group
087         * @return whether anything was deleted.
088         * @see TickClient
089         * @param C the client to remove
090         */
091        public boolean delTicker(TickClient C);
092
093        /**
094         * Returns an iterator of all the ticking objects
095         * @return an iterator of all the ticking objects
096         */
097        public Iterator<TickClient> tickers();
098
099        /**
100         * Returns the number of ticking objects in this group
101         * @return the number of ticking objects in this group
102         */
103        public int numTickers();
104
105        /**
106         * Returns an iterator of all clients matching the given criteria
107         * @param T the tickable object to look for
108         * @param tickID the tickid to match, or -1 for all
109         * @return an iterator of all clients matching the given criteria
110         */
111        public Iterator<TickClient> getTickSet(final Tickable T, final int tickID);
112
113        /**
114         * Returns an iterator of all the items of the given type in the given room.
115         * @param itemTypes 0=mobs or items, 1=items, 2=mobs
116         * @param R the room to look in
117         * @return an iterator of all the items of the given type in the given room.
118         */
119        public Iterator<TickClient> getLocalItems(int itemTypes, Room R);
120
121        /**
122         * Returns true if this group contains the given ticking object with
123         * the given tickID.
124         * @param T the tickable object
125         * @param tickID the tickid to match, or -1 for any
126         * @return true if its in here, false otherwise
127         */
128        public boolean contains(final Tickable T, final int tickID);
129
130        /**
131         * Returns the tickclient at the given index from 0 to numTickers
132         * @see TickableGroup#numTickers()
133         * @param i the index
134         * @return the tickclient, or null
135         */
136        public TickClient fetchTickerByIndex(int i);
137
138        /**
139         * Returns the displayable status of this ticking object
140         * @return the displayable status of this ticking object
141         */
142        public String getStatus();
143
144        /**
145         * Returns the time, in ms, that this object last ticked.
146         * @return the time, in ms, that this object last ticked.
147         */
148        public long getLastStartTime();
149
150        /**
151         * Returns the time, in ms, that this object last stopped ticking.
152         * @return the time, in ms, that this object last stopped ticking
153         */
154        public long getLastStopTime();
155
156        /**
157         * Returns the number of ms that this object has ticked in total
158         * @return the number of ms that this object has ticked in total
159         */
160        public long getMilliTotal();
161
162        /**
163         * Returns the number of times that this object has ticked in total
164         * @return the number of times that this object has ticked in total
165         */
166        public long getTickTotal();
167
168        /**
169         * Returns whether this group is only permitted a single object
170         * @return true if this group is only permitted a single object
171         */
172        public boolean isSolitaryTicker();
173
174        /**
175         * Returns true if this group currently is getting thread time
176         * @return true if this group currently is getting thread time
177         */
178        public boolean isAwake();
179
180        /**
181         * Returns the name of the thread group that this ticking group
182         * belongs to (and thus, under which it should run later)
183         * @return the name of the thread group
184         */
185        public String getThreadGroupName();
186
187        /**
188         * If this group is currently getting thread time, this will
189         * return a reference to that thread object, or null otherwise.
190         * @return the thread currently giving this group time, or null
191         */
192        public Thread getCurrentThread();
193}
