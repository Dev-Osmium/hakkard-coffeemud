036/**
037 * Interface for objects which represents real estate purchasable by players.  May
038 * be found implemented by Abilities which are placed as effects on the room  objects
039 * for sale, or implemented as Items representing the sellable title.
040 * @author Bo Zimmerman
041 */
042public interface LandTitle extends PrivateProperty
043{
044        /**
045         * Get the roomID or the Area name of the property for sale
046         * @return the roomID or the Area  name of the property for sale
047         */
048        public String landPropertyID();
049        
050        /**
051         * Set the roomID or the Area name of the property for sale
052         * @param landID the roomID or the Area  name of the property for sale
053         */
054        public void setLandPropertyID(String landID);
055
056        /**
057         * Checks for changes in the content or condition of the rooms represented
058         * by this title and saves the changes to the database, if necessary.
059         * @param optPlayerList - null, or a vector of player names for quick confirms
060         * @see LandTitle#updateTitle()
061         */
062        public void updateLot(Set<String> optPlayerList);
063        
064        /**
065         * Simply resaves the rooms represented by this title to reflect change
066         * in ownership or price.  The state of the rooms is not inspected or
067         * updated as in updateLot.
068         * @see LandTitle#updateLot(Set)
069         */
070        public void updateTitle();
071
072        /**
073         * The complete set of room objects represented by this title
074         * @see com.planet_ink.coffee_mud.Locales.interfaces.Room
075         * @return a list of the complete set of Room objects represented by this title
076         */
077        public List<Room> getAllTitledRooms();
078        
079        /**
080         * The complete set of room objects that are tied together by one or more titles.
081         * @see com.planet_ink.coffee_mud.Locales.interfaces.Room
082         * @return a list of the complete set of Room objects represented by property
083         */
084        public List<Room> getConnectedPropertyRooms();
085
086        /**
087         * Returns a unique identifier corresponding to getConnectedPropertyRooms.
088         * An identifier that uniquely identifies all the connected lots of this property,
089         * even if they are unowned, or owned by different people.  Think of it as a "subdivision"
090         * when lots are variously owned, or a "mansion id" when lots are owned by one person.
091         * @return unique identifier
092         */
093        public String getUniqueLotID();
094
095        /**
096         * Whether this property is a rental.
097         * @return true if the property is rental, false if ownable outright
098         */
099        public boolean rentalProperty();
100
101        /**
102         * Sets whether this property is a rental.
103         * @param truefalse true if the property is rental, false if ownable outright
104         */
105        public void setRentalProperty(boolean truefalse);
106        
107        /**
108         * If back taxes is owned on this property, this is how the value is set.  The
109         * value should be in base coffeemud currency
110         * @param amount the back  taxes owed
111         */
112        public void setBackTaxes(int amount);
113
114        /**
115         * If back taxes is owned on this property, this is how the value is retrieved.  The
116         * value should be in base coffeemud currency
117         * @return amount the back  taxes owed
118         */
119        public int backTaxes();
120
121        /**
122         * If this title allows expansion construction, this this will return the new
123         * title to assign to the newly constructed rooms.
124         * @see LandTitle#allowsExpansionConstruction()
125         * @return the new title to assigne to a new room.
126         */
127        public LandTitle generateNextRoomTitle();
128        
129        /**
130         * Returns whether this title allows property to be expanded through masonry or construction.
131         * @see LandTitle#generateNextRoomTitle()
132         * @return true if expansion is OK, false otherwise
133         */
134        public boolean allowsExpansionConstruction();
135
136        /**
137         * Returns whether this title creates connected walls between new rooms as if the rooms
138         * were laid out in a grid.
139         * @return true if the rooms can lay out as a grid, false otherwise
140         */
141        public boolean gridLayout();
142
143        /**
144         * Sets whether this title creates connected walls between new rooms as if the rooms
145         * were laid out in a grid.
146         * @param gridLayout true if the rooms can lay out as a grid, false otherwise
147         */
148        public void setGridLayout(boolean gridLayout);
149
150}
