033/**
034 * This interface is implemented by any object which wishes to get periodic thread time from
035 * the threads engine.  Almost all CoffeeMud objects implement this interface
036 * @author Bo Zimmerman
037 *
038 */
039public interface Tickable extends CMObject
040{
041        /**
042         * The nice displayable name of this instance of this object
043         * @return the displayable name of this object instance
044         */
045        @Override
046        public String name();
047        /**
048         * A coded status for this object during the period where
049         * its tick method is being called. The statis is defined,
050         * at least in part, by constants in this interface STATUS_*.
051         * STATUS_NOT should be returned when the objects tick method
052         * is not currently in execution.  It should never return
053         * STATUS_NOT when the objects tick method is in execution.
054         * @see Tickable#tick(Tickable, int)
055         * @return the numeric status of this object
056         */
057        public int getTickStatus();
058
059        /**
060         * this is the method which is called periodically by the threading engine.  How often it
061         * is called depends on the parameters passed to the threadding engine when it is submitted
062         * for thread access.  Typically the period is once per TIME_TICK period, but that is
063         * determined when the object is submitted to the thread engine.
064         * @see Tickable
065         * @see com.planet_ink.coffee_mud.core.threads.ServiceEngine
066         * @see TickableGroup
067         * @param ticking a reference to this Tickable object
068         * @param tickID the TICKID_ constant describing this periodic call, as defined in Tickable
069         * @return true always, unless this object no longer wishes to ever tick again, in which case false
070         */
071        public boolean tick(final Tickable ticking, final int tickID);
072
073        /** modifies a tickID to designate a longer wait before declaring dead.*/
074        public final static int TICKID_LONGERMASK=256;
075        /** modifies a tickID to designate that this object should tick by itself.*/
076        public final static int TICKID_SOLITARYMASK=65536;
077
078        /** the most common tickid, representing the tick of a mob*/
079        public final static int TICKID_MOB=0;
080        /** the tickid representing the tick of a behavior on an item*/
081        public final static int TICKID_ITEM_BEHAVIOR=1;
082        /** the tickid representing the tick that automatically closes open doors*/
083        public final static int TICKID_EXIT_REOPEN=2;
084        /** the tickid representing the decay tick on a corpse*/
085        public final static int TICKID_DEADBODY_DECAY=3;
086        /** the tickid representing the going out of a light*/
087        public final static int TICKID_LIGHT_FLICKERS=4;
088        /** the tickid representing the resetting of a trap*/
089        public final static int TICKID_TRAP_RESET=5;
090        /** the tickid representing the destruction of a trap*/
091        public final static int TICKID_TRAP_DESTRUCTION=6;
092        /** the tickid representing the returning of an innkey to the front desk*/
093        public final static int TICKID_ITEM_BOUNCEBACK=7;
094        /** the tickid representing the tick of a behavior on a room*/
095        public final static int TICKID_ROOM_BEHAVIOR=8;
096        /** the tickid representing the standard tick of an area*/
097        public final static int TICKID_AREA=9;
098        /** the tickid representing the resetting of an item that rejuvenates*/
099        public final static int TICKID_ROOM_ITEM_REJUV=10;
100        /** the tickid representing the ticking of a behavior on an exit*/
101        public final static int TICKID_EXIT_BEHAVIOR=11;
102        /** the tickid representing the ticking of a spell on a non-mob object*/
103        public final static int TICKID_SPELL_AFFECT=12;
104        /** the tickid representing the ticking of a quest script*/
105        public final static int TICKID_QUEST=13;
106        /** the tickid representing the ticking of a clan object*/
107        public final static int TICKID_CLAN=14;
108        /** the tickid representing the ticking of a clan item*/
109        public final static int TICKID_CLANITEM=15;
110        /** the tickid representing the ticking of the smtp server*/
111        public final static int TICKID_EMAIL=TICKID_SOLITARYMASK|16;
112        /** the tickid representing the impending stop of a service*/
113        public final static int TICKID_READYTOSTOP=17;
114        /** the tickid for a live auction tick*/
115        public final static int TICKID_LIVEAUCTION=18;
116        /** the tickid for a slow auction service tick*/
117        public final static int TICKID_TIMEAUCTION=19;
118        /** the tickid for a miscellaneous tickable service*/
119        public final static int TICKID_MISCELLANEOUS=20;
120        /** the tickid for a miscellaneous tickable service*/
121        public final static int TICKID_PROPERTY_SPECIAL=21;
122        /** the tickid for an electrical proactive action service*/
123        public final static int TICKID_ELECTRONICS=22;
124        /** the tickid for a system support task*/
125        public final static int TICKID_SUPPORT=23;
126        /** the tickid for an special combat service*/
127        public final static int TICKID_SPECIALMANEUVER=24;
128        /** the tickid for an special combat service*/
129        public final static int TICKID_SPECIALCOMBAT=25;
130        /** the tickid for a beam weapon in motion*/
131        public final static int TICKID_BEAMWEAPON=26;
132
133        /** a tick status constant representing  the state of waiting for tick access */
134        public static int STATUS_NOT=0;
135        /** a tick status constant representing the state of just starting its tick access */
136        public static int STATUS_START=1;
137        /** a tick status representing CLASS part  of its tick access  */
138        public static int STATUS_CLASS=2;
139        /** a tick status representing RACE part  of its tick access  */
140        public static int STATUS_RACE=3;
141        /** a tick status representing COMBAT part  of its tick access  */
142        public static int STATUS_FIGHT=4;
143        /** a tick status representing WEATHER part  of its tick access  */
144        public static int STATUS_WEATHER=5;
145        /** a tick status representing DEAD part  of its tick access  */
146        public static int STATUS_DEAD=6;
147        /** a tick status representing ALIVE part  of its tick access  */
148        public static int STATUS_ALIVE=7;
149        /** a tick status representing OTHER part  of its tick access  */
150        public static int STATUS_REBIRTH=8;
151        /** a tick status representing ALIVE part  of its tick access  */
152        public static int STATUS_OTHER=98;
153        /** a tick status representing the end of its tick access  */
154        public static int STATUS_END=99;
155        /** a tick status MASK representing the tick access of an associated behavior */
156        public static int STATUS_BEHAVIOR=512;
157        /** a tick status MASK representing the tick access of an associated effect */
158        public static int STATUS_AFFECT=1024;
159        /** a tick status MASK representing the tick access of an associated script */
160        public static int STATUS_SCRIPT=2048;
161        /** a tick status MASK  whose meaning is class dependent */
162        public static int STATUS_MISC=4096;
163        /** a tick status MASK  whose meaning is class dependent */
164        public static int STATUS_MISC2=8192;
165        /** a tick status MASK  whose meaning is class dependent */
166        public static int STATUS_MISC3=16384;
167        /** a tick status MASK  whose meaning is class dependent */
168        public static int STATUS_MISC4=32768;
169        /** a tick status MASK  whose meaning is class dependent */
170        public static int STATUS_MISC5=65536;
171        /** a tick status MASK  whose meaning is class dependent */
172        public static int STATUS_MISC6=131072;
173        /** a tick status MASK  whose meaning is class dependent */
174        public static int STATUS_MISC7=131072*2;
175}
