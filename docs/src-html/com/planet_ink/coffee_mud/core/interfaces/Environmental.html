038/**
039 * The core of most object types in CoffeeMud. Much of the most common fields and
040 * data are derived from this.  It represents something that appears in the
041 * environment, either physically, or as energy or other phenomenon.
042 *
043 * @author Bo Zimmerman
044 *
045 */
046public interface Environmental extends Tickable, StatsAffecting, MsgListener, Contingent, Modifiable
047{
048        /**
049         * The raw unmodified name of this object as stored in the database.
050         * This is the name set by builders and used as a basis for the name() method.
051         * @see  Environmental#name()
052         * @return the raw base name of this object on the map.
053         */
054        public String Name();
055        /**
056         * Sets the raw unmodified name of this object as stored in the database.
057         * This is the name set by builders and used as a basis for the name() method.
058         * @see  Environmental#Name()
059         * @param newName the raw base name of this object on the map.
060         */
061        public void setName(String newName);
062
063        /**
064         * Gets the raw string used to show what this object looks like in the room.
065         * May be used as a basis for other methods, such as the title of rooms, and
066         * what an exit looks like when open.  The value for Items may be null if the item
067         * is not displayed when the room is seen.
068         * @return the string describing how this object looks in the room
069         */
070        public String displayText();
071        /**
072         * Sets the raw string used to show what this object looks like in the room.
073         * May be used as a basis for other methods, such as the title of rooms, and
074         * what an exit looks like when open.  The value for Items may be null if the item
075         * is not displayed when the room is seen.
076         * @param newDisplayText the string describing how this object looks in the room
077         */
078        public void setDisplayText(String newDisplayText);
079
080        /**
081         * The basic description of this object, as shown when the item is directly LOOKed at.
082         * @return the basic detail description of this object
083         */
084        public String description();
085        /**
086         * Sets the basic description of this object, as shown when the item is directly LOOKed at.
087         * @param newDescription the basic detail description of this object
088         */
089        public void setDescription(String newDescription);
090
091        /**
092         * Returns the fully qualified and determined name of the image file displayed for this
093         * object when MXP is used.  If rawImage() is non-empty, it will return rawImage, and
094         * otherwise use the mxp default data file.
095         * @return the name of the mxp image to display for this object.
096         */
097        public String image();
098        /**
099         * Returns the raw name of the image file to display for this object when MXP is used.  This
100         * is the value set by the builder, and may be returned by image() if it is non-empty.
101         * @return the raw name of the mxp image file
102         */
103        public String rawImage();
104        /**
105         * Sets the raw name of the image file to display for this object when MXP is used.  This
106         * is the value set by the builder, and may be returned by image() if it is non-empty.
107         * @param newImage the raw name of the mxp image file
108         */
109        public void setImage(String newImage);
110
111        /**
112         * Whether the fields of this item are set in code, or set by builders.  Generic means that
113         * they are set by builders, in which case XML is returned by the text() method containing
114         * all of the values for all the fields.
115         * @see com.planet_ink.coffee_mud.core.interfaces.Environmental#text()
116         * @return whether this item is modifiable by builders
117         */
118        public boolean isGeneric();
119
120        /**
121         * For objects which have false for their isGeneric method, this is used to set any internally
122         * coded strings to change the nature or behavior of the object.  For objects which have true
123         * for their isGeneric method, this is used to set and parse the XML which will be used to
124         * fill out the fields in this object.  Since Ability objects are never Generic, this will always
125         * be where parameters are read from an Ability instance.
126         * @param newMiscText either an open internal text string, or XML
127         */
128        public void setMiscText(String newMiscText);
129        /**
130         * For objects which have false for their isGeneric method, this is used to set any internally
131         * coded strings to change the nature or behavior of the object.  For objects which have true
132         * for their isGeneric method, this is used to set and parse the XML which will be used to
133         * fill out the fields in this object. Since Ability objects are never Generic, this will always
134         * be where parameters are read from an Ability instance.
135         * @return either an open internal text string, or XML
136         */
137        public String text();
138
139        /**
140         * Unimplemented as of yet, but will hold a string telling the system what
141         * the proper format of any miscText data.  Will use the CMParms.MTFORMAT_*
142         * constants for definition.
143         * @see com.planet_ink.coffee_mud.core.CMParms
144         * @see com.planet_ink.coffee_mud.core.interfaces.Environmental#text()
145         * @see com.planet_ink.coffee_mud.core.interfaces.Environmental#setMiscText(String)
146         * @return the format expected for the miscText field
147         */
148        public String miscTextFormat();
149
150        /**
151         * Whether this object instance is functionally identical to the object passed in.  Works by repeatedly
152         * calling getStat on both objects and comparing the values.
153         * @see com.planet_ink.coffee_mud.core.interfaces.Modifiable#getStatCodes()
154         * @see com.planet_ink.coffee_mud.core.interfaces.Modifiable#getStat(String)
155         * @param E the object to compare this one to
156         * @return whether this object is the same as the one passed in
157         */
158        public boolean sameAs(final Environmental E);
159
160        /**
161         * If this object expires, it should have a timestamp saying when it expires, in real time.
162         * When it expires, a MSG_EXPIRE message will be sent to it.
163         * @see com.planet_ink.coffee_mud.core.interfaces.Environmental#setExpirationDate(long)
164         * @return the time stamp when this thing expires
165         */
166        public long expirationDate();
167        /**
168         * If this object expires, it should have a timestamp saying when it expires, in real time.
169         * When it expires, a MSG_EXPIRE message will be sent to it.
170         * @see com.planet_ink.coffee_mud.core.interfaces.Environmental#expirationDate()
171         * @param dateTime the time stamp when this thing expires
172         */
173        public void setExpirationDate(long dateTime);
174        /**
175         * the maximum range of this object, if applicable.  Can refer to the size of a room,
176         * the range of a weapon, or the calculated range of a mob in combat.
177         * @return the maximum range
178         */
179        public int maxRange();
180        /**
181         * the minimum range of this object, if applicable.  Can refer to the size of a room,
182         * the range of a weapon, or the calculated range of a mob in combat.  Usually 0.
183         * @return the minimum range
184         */
185        public int minRange();
186
187        /**
188         * Localize an internal string -- shortcut. Same as calling:
189         * @see com.planet_ink.coffee_mud.Libraries.interfaces.LanguageLibrary#fullSessionTranslation(String, String...)
190         * Call with the string to translate, which may contain variables of the form @x1, @x2, etc. The array in xs
191         * is then used to replace the variables AFTER the string is translated.
192         * @param str the string to translate
193         * @param xs the array of variables to replace
194         * @return the translated string, with all variables in place
195         */
196        public String L(final String str, final String ... xs);
197}
