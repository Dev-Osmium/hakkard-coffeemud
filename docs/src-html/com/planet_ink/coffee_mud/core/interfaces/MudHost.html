038/**
039 * The interface implemented by the main mud application.  Includes several timing constants.
040 * @author Bo Zimmerman
041 *
042 */
043public interface MudHost
044{
045        /** the number of milliseconds between each savethread execution */
046        public final static long TIME_SAVETHREAD_SLEEP=60*60000; // 60 minutes, right now.
047        /** the number of milliseconds between each utilithread execution */
048        public final static long TIME_UTILTHREAD_SLEEP=10 * 60000;
049        /** for multi-host systems, the thread code denoting the main/first mud host */
050        public final static char MAIN_HOST='0';
051
052        /**
053         * the hostname of the mud server
054         * @return hostname or ip address
055         */
056        public String getHost();
057
058        /**
059         * the port a given MUD server instance is listening on
060         * @return the port numbered listened on by this mud instance
061         */
062        public int getPort();
063
064        /**
065         * The hostname, port of the most public web server for this host
066         * Includes trailing slash
067         * @return http://mydomain:myport/
068         */
069        public String geWebHostUrl();
070
071        /**
072         * An order to permanently shutdown the entire mud system
073         * @param S a player session to send status messages to.  May be null.
074         * @param keepItDown true to shutdown, false to restart
075         * @param externalCommand if keepItDown is false, an external command to execute
076         */
077        public void shutdown(Session S, boolean keepItDown, String externalCommand);
078
079        /**
080         * Retrieve a string telling the status of mud startup or shutdown
081         * @return status of mud startup or shutdown
082         */
083        public String getStatus();
084
085        /**
086         * Retrieve the number of seconds since startup
087         * @return number of seconds since startup
088         */
089        public long getUptimeSecs();
090
091        /**
092         * Return any internal threads that are a source of troubles
093         * @return a list of threads that need service or killing
094         */
095        public List<Runnable> getOverdueThreads();
096
097        /**
098         * Return the viewable name of the language supported by this host.
099         * @return the language supported by this host.
100         */
101        public String getLanguage();
102
103        /**
104         * Flexible interface for tinkering with mud-host settings.
105         * Commands to be defined later, or now, or whatever.
106         * @param cmd space-delimited (parsable) command/parm list
107         * @return any return variables
108         * @throws java.lang.Exception any exceptions
109         */
110        public String executeCommand(String cmd)
111                throws Exception;
112
113        /**
114         * Because thread groups are used to track configurations,
115         * here is a way to get it.
116         * @return the thread group this host was created under.
117         */
118        public ThreadGroup threadGroup();
119
120        /**
121         * Sets whether this mud is accepting connections
122         * @see com.planet_ink.coffee_mud.core.interfaces.MudHost#isAcceptingConnections()
123         * @param truefalse whether it is accepting connections
124         */
125        public void setAcceptConnections(boolean truefalse);
126
127        /**
128         * Handles a connection from a user, and internal states
129         * @param sock the socket the connection was made on
130         * @throws java.net.SocketException socket exceptions
131         * @throws java.io.IOException io exceptions
132         */
133        public void acceptConnection(Socket sock) throws SocketException, IOException;
134
135        /**
136         * Sets whether this mud is accepting connections
137         * @see com.planet_ink.coffee_mud.core.interfaces.MudHost#setAcceptConnections(boolean)
138         * @return true/false whether it is accepting connections
139         */
140        public boolean isAcceptingConnections();
141}
