033/**
034 * A Drinkable object containing its own liquid material type, and liquid capacity management.
035 * @author Bo Zimmerman
036 *
037 */
038public interface Drink extends Environmental, Decayable
039{
040        /**
041         * The amount of thirst points quenched every time this item is drank from.
042         * @see com.planet_ink.coffee_mud.Common.interfaces.CharState
043         * @return amount of thirst quenched
044         */
045        public int thirstQuenched();
046        /**
047         * The total amount of liquid possible to be contained in this liquid container.
048         * @return total liquid contained herein.
049         */
050        public int liquidHeld();
051        /**
052         * The amount of liquid remaining in this liquid container.  Will always be less
053         * less than liquidHeld();
054         * @see Drink#liquidHeld()
055         * @return amount of liquid remaining in this liquid container.
056         */
057        public int liquidRemaining();
058        /**
059         * The material type of the liquid in this container.  Although a class implementing
060         * the Drink interface can sometimes be a liquid itself (like GenLiquidResource), most
061         * often, a Drink interface implementing class is a mob without a material to draw from
062         * or an Item having its own non-liquid material (like a leather waterskin containing milk).
063         * Either way, this is necessary.  The material types are constants in RawMaterial.
064         * @see com.planet_ink.coffee_mud.Items.interfaces.RawMaterial
065         * @return the type of liquid contained herein
066         */
067        public int liquidType();
068        /**
069         * Sets the material type of the liquid in this container.  Although a class implementing
070         * the Drink interface can sometimes be a liquid itself (like GenLiquidResource), most
071         * often, a Drink interface implementing class is a mob without a material to draw from
072         * or an Item having its own non-liquid material (like a leather waterskin containing milk).
073         * Either way, this is necessary.  The material types are constants in RawMaterial.
074         * @see com.planet_ink.coffee_mud.Items.interfaces.RawMaterial
075         * @param newLiquidType the type of liquid contained herein
076         */
077        public void setLiquidType(int newLiquidType);
078        /**
079         * Set the amount of thirst points quenched every time this item is drank from.
080         * @see com.planet_ink.coffee_mud.Common.interfaces.CharState
081         * @param amount of thirst quenched
082         */
083        public void setThirstQuenched(int amount);
084        /**
085         * Sets the total amount of liquid possible to be contained in this liquid container.
086         * @param amount total liquid contained herein.
087         */
088        public void setLiquidHeld(int amount);
089        /**
090         * Sets the amount of liquid remaining in this liquid container.  Will always be less
091         * less than liquidHeld();
092         * @see Drink#setLiquidHeld(int)
093         * @param amount amount of liquid remaining in this liquid container.
094         */
095        public void setLiquidRemaining(int amount);
096        /**
097         * Whether this liquid container still contains any liquid.
098         * @return whether any liquid is left.
099         */
100        public boolean containsDrink();
101        /**
102         * Settable only internally, this method returns whether this entire object  is
103         * destroyed immediately after it is drank from  --  like a potion.
104         * @return Whether the item survives after drinking.
105         */
106        public boolean disappearsAfterDrinking();
107
108        /**
109         * Given the liquid source, the amount of liquid which would need to be taken
110         * from the source liquid source to fill up THIS liquid source.
111         * @param theSource the liquid source to fill up from
112         * @return the amount to take from the liquid source
113         */
114        public int amountTakenToFillMe(Drink theSource);
115}
