025/**
026*
027* Something that can be tattooed with a named marker.
028* @see com.planet_ink.coffee_mud.Common.interfaces.Tattoo
029* @author Bo Zimmerman
030*
031*/
032public interface Tattooable extends CMObject
033{
034        /**
035         * Add a new tattoo to this object
036         * @see Tattooable
037         * @see Tattooable#addTattoo(String, int)
038         * @see Tattooable#addTattoo(String)
039         * @param of the tattoo object to add
040         */
041        public void addTattoo(Tattoo of);
042        
043        /**
044         * Add a new tattoo to this object
045         * @see Tattooable
046         * @see Tattooable#addTattoo(String, int)
047         * @see Tattooable#addTattoo(Tattoo)
048         * @param of the permanent tattoo name to add
049         */
050        public void addTattoo(String of);
051        /**
052         * Add a new tattoo to this object
053         * @see Tattooable
054         * @see Tattooable#addTattoo(String)
055         * @see Tattooable#addTattoo(Tattoo)
056         * @param of the permanent tattoo name to add
057         * @param tickDown the ticks of life for this tattoo
058         */
059        public void addTattoo(String of, int tickDown);
060        
061        /**
062         * Remove a specific tattoo from this object
063         * @see Tattooable
064         * @see Tattooable#delTattoo(String)
065         * @param of the tattoo object to remove
066         */
067        public void delTattoo(Tattoo of);
068        
069        /**
070         * Remove a specific tattoo from this object
071         * @see Tattooable
072         * @see Tattooable#delTattoo(Tattoo)
073         * @param of the tattoo name of the tattoo to remove
074         */
075        public void delTattoo(String of);
076
077        /**
078         * Returns an enumeration of all the tattoos on this object.
079         * @see Tattooable
080         * @return an enumeration of all the tattoos on this object.
081         */
082        public Enumeration<Tattoo> tattoos();
083        
084        /**
085         * Returns the tattoo of the given marker name
086         * @see Tattooable
087         * @see Tattooable#findTattooStartsWith(String)
088         * @param of the marker name to return a tattoo for
089         * @return the tattoo of the given marker name
090         */
091        public Tattoo findTattoo(String of);
092        
093        /**
094         * Returns the tattoo that starts with the given marker name
095         * @see Tattooable
096         * @see Tattooable#findTattoo(String)
097         * @param of the marker name to return a tattoo starter for
098         * @return the tattoo that starts with the given marker name
099         */
100        public Tattoo findTattooStartsWith(String of);
101}
