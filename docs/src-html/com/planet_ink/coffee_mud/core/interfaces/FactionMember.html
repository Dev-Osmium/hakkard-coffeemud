028/**
029*
030* Something that can belong to one or more factions, and have rank
031* (also called "faction") with it.
032* @see com.planet_ink.coffee_mud.Common.interfaces.Faction
033* @author Bo Zimmerman
034*
035*/
036public interface FactionMember
037{
038        /**
039         * Adds a new faction to this member, with the given initial rank
040         * @see FactionMember
041         * @param of the facton ID to add to this
042         * @param start the initial rank/value in the new faction
043         */
044        public void addFaction(String of, int start);
045
046        /**
047         * If this is already a member of the given faction, then the
048         * value or rank in that faction will be adjusted by the given
049         * amount.  If this is NOT yet a member of the given faction, then
050         * the faction is added with the amount as an initial value.
051         * @see FactionMember
052         * @param of the facton ID to add or alter
053         * @param amount the amount to alter the faction by, or initial value
054         */
055        public void adjustFaction(String of, int amount);
056
057        /**
058         * Returns an enumeration of all the faction id this is a member of.
059         * @see FactionMember
060         * @return an enumeration of all the faction id this is a member of.
061         */
062        public Enumeration<String> factions();
063
064        /**
065         * Returns an enumeration of the faction range ids that represent the
066         * rank that this member has in each of their factions.  So there is
067         * one range returned per faction.
068         * @see FactionMember
069         * @return an enumeration of the faction ranges
070         */
071        public List<String> fetchFactionRanges();
072
073        /**
074         * Returns whether this is a member of the given faction and has some
075         * rank.
076         * @see FactionMember
077         * @param which the faction id to search for
078         * @return true if this is a member of the given faction, false otherwise
079         */
080        public boolean hasFaction(String which);
081
082        /**
083         * Returns the faction rank/value that this member has in the given
084         * faction id.
085         * @see FactionMember
086         * @param which the faction id to return rank/value in
087         * @return the rank value, or Integer.MAX_VALUE if no value found
088         */
089        public int fetchFaction(String which);
090
091        /**
092         * Returns the complete contextual faction data for this
093         * member.
094         * @see com.planet_ink.coffee_mud.Common.interfaces.Faction.FData
095         * @param which the faction id to return rank/value in
096         * @return the faction data
097         */
098        public Faction.FData fetchFactionData(String which);
099
100        /**
101         * Returns a friendly viewable list of all the factions that this is
102         * a member of, along with the rank/value this member has in that
103         * faction.  The list is semicolon-delimited.
104         * @see FactionMember
105         * @return friendly viewable list of all the factions
106         */
107        public String getFactionListing();
108
109        /**
110         * Removes this as a member of the given faction, losing all rank.
111         * @see FactionMember
112         * @param which the faction id to remove
113         */
114        public void removeFaction(String which);
115
116        /**
117         * Copies the factions that the given member belongs to into this.
118         * @see FactionMember
119         * @param source the source of the factions to copy
120         */
121        public void copyFactions(FactionMember source);
122}
