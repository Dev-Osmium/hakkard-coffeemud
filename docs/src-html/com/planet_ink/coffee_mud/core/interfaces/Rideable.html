036/**
037 * The interface for an item or mob which may be ridden
038 * @see com.planet_ink.coffee_mud.core.interfaces.Rider
039 * @author Bo Zimmerman
040 *
041 */
042public interface Rideable extends Rider
043{
044        /** constant for the  rideType() method.  Means it is ridden over land*/
045        public final static int RIDEABLE_LAND=0;
046        /** constant for the  rideType() method.  Means it is ridden over water*/
047        public final static int RIDEABLE_WATER=1;
048        /** constant for the  rideType() method.  Means it is ridden through the air*/
049        public final static int RIDEABLE_AIR=2;
050        /** constant for the  rideType() method.  Means it is sat upon*/
051        public final static int RIDEABLE_SIT=3;
052        /** constant for the  rideType() method.  Means it is slept upon*/
053        public final static int RIDEABLE_SLEEP=4;
054        /** constant for the  rideType() method.  Means it is sat at*/
055        public final static int RIDEABLE_TABLE=5;
056        /** constant for the  rideType() method.  Means it is entered into*/
057        public final static int RIDEABLE_ENTERIN=6;
058        /** constant for the  rideType() method.  Means it is climbed*/
059        public final static int RIDEABLE_LADDER=7;
060        /** constant for the  rideType() method.  Means it is pulled by others*/
061        public final static int RIDEABLE_WAGON=8;
062        /** constant for the  rideType() method.  Means it is climbed and travels with player*/
063        public final static String[] RIDEABLE_DESCS=
064        {
065                "LAND-BASED","WATER-BASED","AIR-FLYING","FURNITURE-SIT","FURNITURE-SLEEP","FURNITURE-TABLE",
066                "ENTER-IN","LADDER","WAGON"
067        };
068
069        /**
070         * Whether the type of rideable is mobile.
071         * @see Rideable
072         * @return Whether this type of rideable is mobile
073         */
074        public boolean isMobileRideBasis();
075
076        /**
077         * The type of rideable object this is.
078         * @see Rideable
079         * @return the RIDEABLE_* constant describing how this is ridden
080         */
081        public int rideBasis();
082        /**
083         * Sets type of rideable object this is.
084         * @see Rideable
085         * @param basis the RIDEABLE_* constant describing how this is ridden
086         */
087        public void setRideBasis(int basis);
088        /**
089         * The number of Riders which may ride upon this Rideable
090         * @see Rider
091         * @return the maximum riders
092         */
093        public int riderCapacity();
094        /**
095         * Sets the number of Riders which may ride upon this Rideable
096         * @see Rider
097         * @param newCapacity the maximum riders
098         */
099        public void setRiderCapacity(int newCapacity);
100        /**
101         * Returns the number of riders currently mounted on this Rideable
102         * @see Rider
103         * @return the number of current Riders
104         */
105        public int numRiders();
106        /**
107         * Returns an iterator of the riders on this rideable
108         * @see Rider
109         * @return the riders
110         */
111        public Enumeration<Rider> riders();
112
113        /**
114         * Returns a particular Rider mounted on this Rideable.  May return null
115         * in the case of a race condition
116         * @see Rider
117         * @see Rideable
118         * @see Rideable#numRiders()
119         * @param which which rider to detch
120         * @return the rider riding
121         */
122        public Rider fetchRider(int which);
123
124        /**
125         * Adds a new Rider to this Rideable.  Is called by Rider.setRiding(Rideable)
126         * Should also call Rider.setRiding(Rideable) without recursion
127         * @see Rider
128         * @see Rideable
129         * @param mob Rider object, either an item or a mob
130         */
131        public void addRider(Rider mob);
132
133        /**
134         * Removes a Rider from this Rideable.  Is called by Rider.setRiding(null)
135         * Should also call Rider.setRiding(null) without recursion
136         * @see Rider
137         * @see Rideable
138         * @param mob Rider object, either an item or a mob
139         */
140        public void delRider(Rider mob);
141
142        /**
143         * Returns whether Rider is currently mounted on this Rideable
144         * @see Rider
145         * @see Rideable
146         * @param mob the Rider to check this Rideable for
147         * @return true if the Rider is mounted on this Rideable
148         */
149        public boolean amRiding(Rider mob);
150
151        /**
152         * Returns a string grammatically correct for the given rider when
153         * they are mounted on this Rideable
154         * @see Rider
155         * @see Rideable#setStateString(String)
156         * @see Rideable#getStateString()
157         * @param R The rider object to make grammatically correct.
158         * @return a string describing  the riders state of riding this Rideable
159         */
160        public String stateString(Rider R);
161
162        /**
163         * Returns the custom string grammatically correct for the given rider when
164         * they are mounted on this Rideable
165         * @see Rider
166         * @see Rideable#setStateString(String)
167         * @see Rideable#stateString(Rider)
168         * @return a custom string describing  the riders state of riding this Rideable
169         */
170        public String getStateString();
171
172        /**
173         * Returns a string grammatically correct for the given rider when
174         * they are mounted on this Rideable
175         * @see Rider
176         * @see Rideable#stateString(Rider)
177         * @see Rideable#getStateString()
178         * @param str a string describing  the riders state of riding this Rideable
179         */
180        public void setStateString(String str);
181
182        /**
183         * Returns a verb string describing what one does when one rides
184         * this Rideable from room to room.
185         * @see Rider
186         * @see Rideable#setRideString(String)
187         * @see Rideable#getRideString()
188         * @param R The rider object to make grammatically correct.
189         * @return a string describing  the riders verb of riding this Rideable somewhere
190         */
191        public String rideString(Rider R);
192
193        /**
194         * Returns a custom verb string describing what one does when one rides
195         * this Rideable from room to room.
196         * @see Rider
197         * @see Rideable#setRideString(String)
198         * @see Rideable#rideString(Rider)
199         * @return a custom string describing  the riders verb of riding this Rideable somewhere
200         */
201        public String getRideString();
202
203        /**
204         * Sets a verb string describing what one does when one rides
205         * this Rideable from room to room.
206         * @see Rider
207         * @see Rideable#rideString(Rider)
208         * @see Rideable#getRideString()
209         * @param str a string describing  the riders verb of riding this Rideable somewhere
210         */
211        public void setRideString(String str);
212
213        /**
214         * Returns a string grammatically correct for the given rider when
215         * they are putting something on this Rideable
216         * @see Rider
217         * @see Rideable#setPutString(String)
218         * @see Rideable#getPutString()
219         * @param R The rider object to make grammatically correct.
220         * @return a string describing  the riders state of putting something on this Rideable
221         */
222        public String putString(Rider R);
223
224        /**
225         * Returns a custom string grammatically correct for the given rider when
226         * they are putting something on this Rideable
227         * @see Rider
228         * @see Rideable#setPutString(String)
229         * @see Rideable#putString(Rider)
230         * @return a custom string describing  the riders state of putting something on this Rideable
231         */
232        public String getPutString();
233
234        /**
235         * Set a string grammatically correct for the given rider when
236         * they are putting something on this Rideable
237         * @see Rider
238         * @see Rideable#putString(Rider)
239         * @see Rideable#getPutString()
240         * @param str a string describing  the riders state of putting something on this Rideable
241         */
242        public void setPutString(final String str);
243
244        /**
245         * Returns a string grammatically correct for this Rideable when
246         * Riders are mounted
247         * @see Rider
248         * @see Rideable#setStateStringSubject(String)
249         * @see Rideable#getStateStringSubject()
250         * @param R The rider object to make grammatically correct.
251         * @return a string describing the Riderable state of being ridden
252         */
253        public String stateStringSubject(Rider R);
254
255        /**
256         * Returns a custom string grammatically correct for this Rideable when
257         * Riders are mounted
258         * @see Rider
259         * @see Rideable#setStateStringSubject(String)
260         * @see Rideable#stateStringSubject(Rider)
261         * @return a custom string describing the Riderable state of being ridden
262         */
263        public String getStateStringSubject();
264
265        /**
266         * Sets a string grammatically correct for this Rideable when
267         * Riders are mounted
268         * @see Rider
269         * @see Rideable#stateStringSubject(Rider)
270         * @see Rideable#getStateStringSubject()
271         * @param str a string describing the Riderable state of being ridden
272         */
273        public void setStateStringSubject(String str);
274
275        /**
276         * Whether this Rideable moves when the Rider wants to move it.  Largely derived
277         * from rideBasis().
278         * @see Rideable#rideBasis()
279         * @return whether this item moves with the rider
280         */
281        public boolean mobileRideBasis();
282
283        /**
284         * Returns a string grammatically correct for the given rider when
285         * they are mounting this Rideable
286         * @see Rider
287         * @see Rideable#setMountString(String)
288         * @see Rideable#getMountString()
289         * @see Rideable#RIDEABLE_DESCS
290         * @param commandType one of the RIDEABLE_ constants as a type
291         * @param R The rider object to make grammatically correct.
292         * @return a string describing the riders state of mounting this Rideable
293         */
294        public String mountString(int commandType, Rider R);
295
296        /**
297         * Returns a custom string grammatically correct for the given rider when
298         * they are mounting this Rideable
299         * @see Rider
300         * @see Rideable#setMountString(String)
301         * @see Rideable#mountString(int, Rider)
302         * @see Rideable#RIDEABLE_DESCS
303         * @return a custom string describing the riders state of mounting this Rideable
304         */
305        public String getMountString();
306
307        /**
308         * Setss a string grammatically correct for the given rider when
309         * they are mounting this Rideable
310         * @see Rider
311         * @see Rideable#mountString(int, Rider)
312         * @see Rideable#getDismountString()
313         * @see Rideable#RIDEABLE_DESCS
314         * @param str a string describing the riders state of mounting this Rideable
315         */
316        public void setMountString(String str);
317
318        /**
319         * Returns a string grammatically correct for the given rider when
320         * they are dismounting this Rideable
321         * @see Rider
322         * @see Rideable#setDismountString(String)
323         * @see Rideable#getDismountString()
324         * @param R The rider object to make grammatically correct.
325         * @return a string describing the riders state of dismounting this Rideable
326         */
327        public String dismountString(Rider R);
328
329        /**
330         * Returns a custom string grammatically correct for the given rider when
331         * they are dismounting this Rideable
332         * @see Rider
333         * @see Rideable#setDismountString(String)
334         * @see Rideable#dismountString(Rider)
335         * @return a custom string describing the riders state of dismounting this Rideable
336         */
337        public String getDismountString();
338
339        /**
340         * Sets a string grammatically correct for the given rider when
341         * they are dismounting this Rideable
342         * @see Rider
343         * @see Rideable#dismountString(Rider)
344         * @see Rideable#getDismountString()
345         * @param str a string describing the riders state of dismounting this Rideable
346         */
347        public void setDismountString(String str);
348
349        /**
350         * Adds all of the MOB Riders on this Rideable to the given Set and returns it
351         * @param list the hashset into which to add all the mob riders
352         * @return the same list passed in, filled
353         */
354        public Set<MOB> getRideBuddies(Set<MOB> list);
355}
