033/**
034 * An interface for all mechanisms that has a lockable door or lid, such
035 * as containers and exits
036 * @see com.planet_ink.coffee_mud.Exits.interfaces.Exit
037 * @see com.planet_ink.coffee_mud.Items.interfaces.Container
038 */
039public interface CloseableLockable extends Physical
040{
041        /**
042         * Returns whether this is OPEN and may be accessed or travelled through
043         * @return whether this is OPEN and may be accessed or travelled through
044         */
045        public boolean isOpen();
046
047        /**
048         * Returns whether this is LOCKED, and must be unlocked before
049         * being used.
050         * @see com.planet_ink.coffee_mud.core.interfaces.CloseableLockable#hasADoor()
051         * @see com.planet_ink.coffee_mud.core.interfaces.CloseableLockable#hasALock()
052         * @return true if locked, false otherwise
053         */
054        public boolean isLocked();
055
056        /**
057         * Returns whether this has a door, and must be opened before
058         * being used.
059         * @see com.planet_ink.coffee_mud.core.interfaces.CloseableLockable#isOpen()
060         * @see com.planet_ink.coffee_mud.core.interfaces.CloseableLockable#isLocked()
061         * @return true if a door is present, false otherwise.
062         */
063        public boolean hasADoor();
064
065        /**
066         * Returns whether this has a lock on it, and, if locked,
067         * must be unlocked before being used.
068         * @see com.planet_ink.coffee_mud.core.interfaces.CloseableLockable#isOpen()
069         * @see com.planet_ink.coffee_mud.core.interfaces.CloseableLockable#isLocked()
070         * @see com.planet_ink.coffee_mud.core.interfaces.CloseableLockable#hasADoor()
071         * @return whether a lock is present
072         */
073        public boolean hasALock();
074
075        /**
076         * For exits with a door and lock, this returns whether this
077         * defaults in a closed and locked state.
078         * @see com.planet_ink.coffee_mud.core.interfaces.CloseableLockable#hasADoor()
079         * @see com.planet_ink.coffee_mud.core.interfaces.CloseableLockable#defaultsClosed()
080         * @return true if defaults closed and locked, false otherwise
081         */
082        public boolean defaultsLocked();
083
084        /**
085         * For exits with a door, this returns whether this
086         * defaults in a closed state.
087         * @see com.planet_ink.coffee_mud.core.interfaces.CloseableLockable#hasADoor()
088         * @see com.planet_ink.coffee_mud.core.interfaces.CloseableLockable#defaultsLocked()
089         * @return true if defaults closed, false otherwise
090         */
091        public boolean defaultsClosed();
092
093        /**
094         * Modifies the various door/lock settings for this thing.
095         * @see com.planet_ink.coffee_mud.core.interfaces.CloseableLockable#isOpen()
096         * @see com.planet_ink.coffee_mud.core.interfaces.CloseableLockable#isLocked()
097         * @see com.planet_ink.coffee_mud.core.interfaces.CloseableLockable#hasADoor()
098         * @see com.planet_ink.coffee_mud.core.interfaces.CloseableLockable#hasALock()
099         * @see com.planet_ink.coffee_mud.core.interfaces.CloseableLockable#defaultsClosed()
100         * @see com.planet_ink.coffee_mud.core.interfaces.CloseableLockable#defaultsLocked()
101         * @param hasADoor whether this has a door
102         * @param isOpen whether this is open for use
103         * @param defaultsClosed whether this defaults closed
104         * @param hasALock whether this has a lock
105         * @param isLocked whether this is presently locked
106         * @param defaultsLocked whether this defaults closed and locked.
107         */
108        public void setDoorsNLocks(boolean hasADoor,
109                                                           boolean isOpen,
110                                                           boolean defaultsClosed,
111                                                           boolean hasALock,
112                                                           boolean isLocked,
113                                                           boolean defaultsLocked);
114
115        /**
116         * For things with doors and locks, this returns the unique string
117         * representing the key code required to unlock/lock it.
118         * @see com.planet_ink.coffee_mud.core.interfaces.CloseableLockable#setKeyName(String)
119         * @return the key code
120         */
121        public String keyName();
122
123        /**
124         * For things with doors and locks, this sets the unique string
125         * representing the key code required to unlock/lock it.
126         * @see com.planet_ink.coffee_mud.core.interfaces.CloseableLockable#keyName()
127         * @param keyName the new key code
128         */
129        public void setKeyName(String keyName);
130        
131        /**
132         * Returns the number of ticks that this remains open when a mob
133         * or player changes it from a closed to an open state, and this thing
134         * defaults in a closed state.
135         * @see com.planet_ink.coffee_mud.core.interfaces.CloseableLockable#setOpenDelayTicks(int)
136         * @see com.planet_ink.coffee_mud.core.interfaces.Tickable#TICKID_EXIT_REOPEN
137         * @return the number of ticks this remains open
138         */
139        public int openDelayTicks();
140
141        /**
142         * Sets the number of ticks that this will remain open when a mob
143         * or player changes it from a closed to an open state, and this thing
144         * defaults in a closed state.
145         * @see com.planet_ink.coffee_mud.core.interfaces.CloseableLockable#openDelayTicks()
146         * @see com.planet_ink.coffee_mud.core.interfaces.Tickable#TICKID_EXIT_REOPEN
147         * @param numTicks the number of ticks this will remain open
148         */
149        public void setOpenDelayTicks(int numTicks);
150}
