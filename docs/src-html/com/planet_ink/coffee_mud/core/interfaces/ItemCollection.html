023import com.planet_ink.coffee_mud.core.CMLib;
024
025/**
026 * An abstract interface for something capable of grouping items together
027 * @author Bo Zimmerman
028 *
029 */
030public interface ItemCollection extends CMObject
031{
032        /**
033         * Adds a new item to its possessor. By default, the item is added
034         * in a default resting state -- no containers, timeouts, or other
035         * modifiers are set.  Duplicates will not be permitted.
036         * @see com.planet_ink.coffee_mud.core.interfaces.ItemPossessor#delItem(Item)
037         * @param item the item to add
038         */
039        public void addItem(Item item);
040
041        /**
042         * Removes the item from this possessor.
043         * @param item the item to remove
044         */
045        public void delItem(Item item);
046
047        /**
048         * Returns the total number of items at this possessor, regardless
049         * of container status.
050         * @return the total number of items
051         */
052        public int numItems();
053
054        /**
055         * Returns the item at the given index, regardless of container status,
056         * visibility, or other modifiers.
057         * @param i the index of the item
058         * @return the item at that index, or null if its not found
059         */
060        public Item getItem(int i);
061
062        /**
063         * Returns a random item in this collection, or null
064         * @return a random item in this collection, or null
065         */
066        public Item getRandomItem();
067
068        /**
069         * An enumeration of all the items at this possessor.
070         * @return enumeration of all the items at this possessor.
071         */
072        public Enumeration<Item> items();
073
074        /**
075         * Returns the item in the given container that matches the
076         * given itemID, whether by full name, description, class ID,
077         * or partial name (if no fuller name is found).  Handles indexing
078         * for duplicate-named items.
079         * @param goodLocation the container to look in, or null for none
080         * @param itemID the name or partial name of the item to fetch
081         * @return the item found, or null
082         */
083        public Item findItem(Item goodLocation, String itemID);
084
085        /**
086         * Returns the item in this possessor that matches the
087         * given itemID, whether by full name, description, class ID,
088         * or partial name (if no fuller name is found).  Handles indexing
089         * for duplicate-named items.
090         * @param itemID the name or partial name of the item to fetch
091         * @return the item found, or null
092         */
093        public Item findItem(String itemID);
094
095        /**
096         * Returns all items in the given container that matches the
097         * given itemID, whether by full name, description, class ID,
098         * or partial name (if no fuller names are found).
099         *
100         * @param goodLocation the container to look in, or null for none
101         * @param itemID the name or partial name of the item to fetch
102         * @return the item found, or null
103         */
104        public List<Item> findItems(Item goodLocation, String itemID);
105
106        /**
107         * Returns all items in this possessor that matches the
108         * given itemID, whether by full name, description, class ID,
109         * or partial name (if no fuller names are found).
110         *
111         * @param itemID the name or partial name of the item to fetch
112         * @return the item found, or null
113         */
114        public List<Item> findItems(String itemID);
115
116        /**
117         * Returns whether the given item is in this possessors list.
118         * @param item the item to check
119         * @return true if the item was found, and false otherwise
120         */
121        public boolean isContent(Item item);
122        /**
123         * Removes all items from this collection
124         * @param destroy true to also destroy the items
125         */
126        public void delAllItems(boolean destroy);
127        /**
128         * Applies the given code to each item in this collection
129         * @param applier code to execute against each object
130         */
131        public void eachItem(final EachApplicable<Item> applier);
132}
