024/**
025*
026* Something that can be affected, and has environmental stats that
027* can be affected as well.
028* @author Bo Zimmerman
029*
030*/
031public interface Affectable
032{
033        /**
034         * Object containing a set of base, unmodified, mostly numeric fields.  The values on the fields
035         * in this object will be as they were set by the builder. This object is used as a basis for
036         * the recoverPhyStats() method.  See the PhyStats interface for information on the fields herein.
037         * @see com.planet_ink.coffee_mud.core.interfaces.Affectable#phyStats()
038         * @see com.planet_ink.coffee_mud.core.interfaces.Affectable#recoverPhyStats()
039         * @see com.planet_ink.coffee_mud.Common.interfaces.PhyStats
040         * @return a set of state fields
041         */
042        public PhyStats basePhyStats();
043
044        /**
045         * Re-sets the object containing a set of base, unmodified, mostly numeric fields.  The values on the fields
046         * in this object will be as they were set by the builder. This object is used as a basis for
047         * the recoverPhyStats() method.  See the PhyStats interface for information on the fields herein. This
048         * method is rarely called -- the fields therein are usually set using setter methods from the PhyStats
049         * interface on the object itself.
050         * @see com.planet_ink.coffee_mud.core.interfaces.Affectable#phyStats()
051         * @see com.planet_ink.coffee_mud.core.interfaces.Affectable#recoverPhyStats()
052         * @see com.planet_ink.coffee_mud.Common.interfaces.PhyStats
053         * @param newStats a set of state fields
054         */
055        public void setBasePhyStats(PhyStats newStats);
056
057        /**
058         * Object containing a set of current, modified, usable, mostly numeric fields.  This object is based on
059         * the object from basePhyStats() and then updated and modified by the recoverPhyStats() method.
060         * See the PhyStats interface for information on the fields herein.
061         * @see com.planet_ink.coffee_mud.core.interfaces.Affectable#basePhyStats()
062         * @see com.planet_ink.coffee_mud.core.interfaces.Affectable#recoverPhyStats()
063         * @see com.planet_ink.coffee_mud.Common.interfaces.PhyStats
064         * @return the current set of state fields
065         */
066        public PhyStats phyStats();
067
068        /**
069         * This method copies the basePhyStats() object into the phyStats() object, then makes repeated calls to
070         * all surrounding objects  with affectPhyStats(Environmental,PhyStats) method.   Surrounding  objects
071         * include the room where the object is located, the Ability objects in the Effects list, the Behaviors
072         * in the behaviors list, and race/charclass/area if applicable.  Those methods will then make all necessary
073         * adjustments to the values in the new phyStats() object.  When it returns, phyStats() will have a totally
074         * updated object.  This method must be called in code whenever the object is placed on the map, or when
075         * anything changes in its environment, such as location, effects, or other states.
076         * @see com.planet_ink.coffee_mud.core.interfaces.Affectable#basePhyStats()
077         * @see com.planet_ink.coffee_mud.core.interfaces.Affectable#phyStats()
078         * @see com.planet_ink.coffee_mud.core.interfaces.Affectable#addEffect(Ability)
079         * @see com.planet_ink.coffee_mud.core.interfaces.PhysicalAgent#addBehavior(Behavior)
080         * @see com.planet_ink.coffee_mud.Common.interfaces.PhyStats
081         */
082        public void recoverPhyStats();
083
084        /**
085         * Add a new effect to this object, whether permanent or temporary.  After calling this method,
086         * recoverPhyStats() should be called next in case this ability object modifies the stats.
087         * An Ability with a given ID() can only be added once per object.
088         * @see com.planet_ink.coffee_mud.Abilities.interfaces.Ability
089         * @see com.planet_ink.coffee_mud.core.interfaces.Affectable#recoverPhyStats()
090         * @param to The ability object to add as an effect.
091         */
092        public void addEffect(Ability to);
093
094        /**
095         * Same as addEffect(Ability), but will set the Ability object as never being able to be uninvoked.
096         * recoverPhyStats() method  should be called next.
097         * An Ability with a given ID() can only be added once per object.
098         * @see com.planet_ink.coffee_mud.Abilities.interfaces.Ability
099         * @see com.planet_ink.coffee_mud.core.interfaces.Affectable#recoverPhyStats()
100         * @param to The ability object to add as an effect.
101         */
102        public void addNonUninvokableEffect(Ability to);
103
104        /**
105         * Delete an effect from this object, whether permanent or temporary.  After calling this method,
106         * recoverPhyStats() should be called next in case this ability object modified the stats.
107         * @see com.planet_ink.coffee_mud.Abilities.interfaces.Ability
108         * @see com.planet_ink.coffee_mud.core.interfaces.Affectable#recoverPhyStats()
109         * @param to The ability object to remove as an effect on this object
110         */
111        public void delEffect(Ability to);
112
113        /**
114         * Returns the number of ability objects listed as effects on this object.
115         * @see com.planet_ink.coffee_mud.Abilities.interfaces.Ability
116         * @return the number of effects this object has
117         */
118        public int numEffects();
119
120        /**
121         * Returns an ability object listed as an effect on this object. May return null even if the index
122         * is correct to mark a race condition.
123         * @see com.planet_ink.coffee_mud.Abilities.interfaces.Ability
124         * @see com.planet_ink.coffee_mud.core.interfaces.Affectable#numEffects()
125         * @param index which object to return
126         * @return the ability object effecting this object
127         */
128        public Ability fetchEffect(int index);
129
130        /**
131         * Returns an ability object listed as an effect on this object. The object will
132         * be the one with the same ID() string as passed in.
133         * @see com.planet_ink.coffee_mud.Abilities.interfaces.Ability
134         * @see CMObject#ID()
135         * @param ID the Ability ID to look for in effects
136         * @return the ability object effecting this object
137         */
138        public Ability fetchEffect(String ID);
139
140        /**
141         * Returns an enumerator of abilities listed as effects on this object.
142         * @see com.planet_ink.coffee_mud.Abilities.interfaces.Ability
143         * @return the enumerator of ability objects effecting this object
144         */
145        public Enumeration<Ability> effects();
146
147        /**
148         * Optionally uninvokes and then certainly removes all effects
149         * from this object.
150         * @param unInvoke send true to uninvoke before deleting
151         */
152        public void delAllEffects(boolean unInvoke);
153
154        /**
155         * Applies the given code to each effect on this object
156         * @param applier code to execute against each object
157         */
158        public void eachEffect(final EachApplicable<Ability> applier);
159}
