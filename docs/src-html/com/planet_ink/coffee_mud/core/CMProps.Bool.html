582        /**
583         * Creates a properties object for the callers thread group using the given input stream 
584         * as input for the properties.
585         * @param in a stream from which to draw the properties.
586         */
587        public CMProps(InputStream in)
588        {
589                final char c=Thread.currentThread().getThreadGroup().getName().charAt(0);
590                if(props[c]==null)
591                        props[c]=this;
592                try
593                {
594                        this.load(in);
595                        loaded=true;
596                }
597                catch(final IOException e)
598                {
599                        loaded=false;
600                }
601        }

603        /**
604         * Creates a properties object for the callers thread group using the given file path 
605         * as input for the properties.
606         * @param filename a file from which to draw the properties.
607         */
608        public CMProps(String filename)
609        {
610                final char c=Thread.currentThread().getThreadGroup().getName().charAt(0);
611                if(props[c]==null)
612                        props[c]=this;
613                try
614                {
615                        final CMFile F=new CMFile(filename,null);
616                        if(F.exists())
617                        {
618                                this.load(new ByteArrayInputStream(F.textUnformatted().toString().getBytes()));
619                                loaded=true;
620                        }
621                        else
622                                loaded=false;
623                }
624                catch(final IOException e)
625                {
626                        loaded=false;
627                }
628        }

1644         * @param fieldName the name of the field being parsed, for error messages

1646         * @param fields the pre-separated list of cost definitions to finish parsing

1648        public static final void setUpCosts(final String fieldName, final Map<String,ExpertiseLibrary.SkillCostDefinition> map, final List<String> fields)
1649        {

1655                                continue;

1669                                continue;
1670                        }
1671                        String keyField="";

1673                        {
1674                                final int skillIndex=formula.indexOf(' ');
1675                                if(skillIndex<0)
1676                                {

1678                                        continue;
1679                                }

1681                                formula=formula.substring(skillIndex+1).trim();

1689                        {
1690                                Log.errOut("CMProps","Error parsing coffeemud.ini '"+fieldName+"' has duplicate key:"+((keyField.length()==0)?"<EMPTY>":keyField));
