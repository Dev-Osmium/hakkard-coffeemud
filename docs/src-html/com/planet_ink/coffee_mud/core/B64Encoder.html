134        protected static byte[] encode3to4(final byte[] source, final int srcOffset, final int numSigBytes,
135                                                                           final byte[] destination, final int destOffset )
136        {
137                final int inBuff =   ( numSigBytes > 0 ? ((source[ srcOffset     ] << 24) >>>  8) : 0 )
138                                         | ( numSigBytes > 1 ? ((source[ srcOffset + 1 ] << 24) >>> 16) : 0 )
139                                         | ( numSigBytes > 2 ? ((source[ srcOffset + 2 ] << 24) >>> 24) : 0 );
140
141                switch( numSigBytes )
142                {
143                        case 3:
144                                destination[ destOffset         ] = ALPHABET[ (inBuff >>> 18)            ];
145                                destination[ destOffset + 1 ] = ALPHABET[ (inBuff >>> 12) & 0x3f ];
146                                destination[ destOffset + 2 ] = ALPHABET[ (inBuff >>>  6) & 0x3f ];
147                                destination[ destOffset + 3 ] = ALPHABET[ (inBuff       ) & 0x3f ];
148                                return destination;
149
150                        case 2:
151                                destination[ destOffset         ] = ALPHABET[ (inBuff >>> 18)            ];
152                                destination[ destOffset + 1 ] = ALPHABET[ (inBuff >>> 12) & 0x3f ];
153                                destination[ destOffset + 2 ] = ALPHABET[ (inBuff >>>  6) & 0x3f ];
154                                destination[ destOffset + 3 ] = EQUALS_SIGN;
155                                return destination;
156
157                        case 1:
158                                destination[ destOffset         ] = ALPHABET[ (inBuff >>> 18)            ];
159                                destination[ destOffset + 1 ] = ALPHABET[ (inBuff >>> 12) & 0x3f ];
160                                destination[ destOffset + 2 ] = EQUALS_SIGN;
161                                destination[ destOffset + 3 ] = EQUALS_SIGN;
162                                return destination;
163
164                        default:
165                                return destination;
166                }
167        }

174        public static String B64encodeObject( final java.io.Serializable serializableObject, final int options )
175        {
176                final java.io.ByteArrayOutputStream  baos  = new java.io.ByteArrayOutputStream();
177                java.io.OutputStream               b64os = null;
178                java.io.ObjectOutputStream     oos   = null;
179                java.util.zip.GZIPOutputStream gzos  = null;
180
181                final int gzip             = (options & GZIP);
182                final int dontBreakLines = (options & DONT_BREAK_LINES);
183
184                try
185                {
186                        b64os = new B64OutputStream( baos, ENCODE | dontBreakLines );
187
188                        if( gzip == GZIP )
189                        {
190                                gzos = new java.util.zip.GZIPOutputStream( b64os );
191                                oos  = new java.io.ObjectOutputStream( gzos );
192                        }
193                        else
194                                oos   = new java.io.ObjectOutputStream( b64os );
195
196                        oos.writeObject( serializableObject );
197                }
198                catch( final java.io.IOException e )
199                {
200                        e.printStackTrace();
201                        return null;
202                }
203                finally
204                {
205                        try
206                        {
207                                if (oos != null)
208                                        oos.close();
209                        }
210                        catch (final Exception e)
211                        {
212                        }
213                        try
214                        {
215                                if (gzos != null)
216                                        gzos.close();
217                        }
218                        catch (final Exception e)
219                        {
220                        }
221                        try
222                        {
223                                if (b64os != null)
224                                        b64os.close();
225                        }
226                        catch (final Exception e)
227                        {
228                        }
229                        try
230                        {
231                                baos.close();
232                        }
233                        catch (final Exception e)
234                        {
235                        }
236                }
237
238                try
239                {
240                        return new String( baos.toByteArray(), PREFERRED_ENCODING );
241                }
242                catch (final java.io.UnsupportedEncodingException uue)
243                {
244                        return new String( baos.toByteArray() );
245                }
246        }

263        public static String B64encodeBytes( final byte[] source, final int off, final int len, final int options )
264        {
265                final int dontBreakLines = ( options & DONT_BREAK_LINES );
266                final int gzip             = ( options & GZIP   );
267
268                if( gzip == GZIP )
269                {
270                        final java.io.ByteArrayOutputStream  baos  = new java.io.ByteArrayOutputStream();
271                        java.util.zip.GZIPOutputStream gzos  = null;
272                        B64OutputStream                 b64os = null;
273
274                        try
275                        {
276                                b64os = new B64OutputStream( baos, ENCODE | dontBreakLines );
277                                gzos  = new java.util.zip.GZIPOutputStream( b64os );
278
279                                gzos.write( source, off, len );
280                                gzos.close();
281                        }
282                        catch( final java.io.IOException e )
283                        {
284                                e.printStackTrace();
285                                return null;
286                        }
287                        finally
288                        {
289                                try
290                                {
291                                        if (gzos != null)
292                                                gzos.close();
293                                }
294                                catch (final Exception e)
295                                {
296                                }
297                                try
298                                {
299                                        if (b64os != null)
300                                                b64os.close();
301                                }
302                                catch (final Exception e)
303                                {
304                                }
305                                try
306                                {
307                                        baos.close();
308                                }
309                                catch (final Exception e)
310                                {
311                                }
312                        }
313                        try
314                        {
315                                return new String( baos.toByteArray(), PREFERRED_ENCODING );
316                        }
317                        catch (final java.io.UnsupportedEncodingException uue)
318                        {
319                                return new String( baos.toByteArray() );
320                        }
321                }
322                final boolean breakLines = dontBreakLines == 0;
323
324                final int    len43   = len * 4 / 3;
325                final byte[] outBuff = new byte[   ( len43 )                                    // Main 4:3
326                                                                   + ( (len % 3) > 0 ? 4 : 0 )          // Account for padding
327                                                                   + (breakLines ? ( len43 / MAX_LINE_LENGTH ) : 0) ]; // New lines
328                int d = 0;
329                int e = 0;
330                final int len2 = len - 2;
331                int lineLength = 0;
332                for( ; d < len2; d+=3, e+=4 )
333                {
334                        encode3to4( source, d+off, 3, outBuff, e );
335
336                        lineLength += 4;
337                        if( breakLines && lineLength == MAX_LINE_LENGTH )
338                        {
339                                outBuff[e+4] = NEW_LINE;
340                                e++;
341                                lineLength = 0;
342                        }
343                }
344
345                if( d < len )
346                {
347                        encode3to4( source, d+off, len - d, outBuff, e );
348                        e += 4;
349                }
350                try
351                {
352                        return new String( outBuff, 0, e, PREFERRED_ENCODING );
353                }
354                catch (final java.io.UnsupportedEncodingException uue)
355                {
356                        return new String( outBuff, 0, e );
357                }
358
359        }

361        protected static int decode4to3( final byte[] source, final int srcOffset, final byte[] destination, final int destOffset )
362        {
363                if( source[ srcOffset + 2] == EQUALS_SIGN )
364                {
365                        final int outBuff =   ( ( DECODABET[ source[ srcOffset  ] ] & 0xFF ) << 18 )
366                                                  | ( ( DECODABET[ source[ srcOffset + 1] ] & 0xFF ) << 12 );
367
368                        destination[ destOffset ] = (byte)( outBuff >>> 16 );
369                        return 1;
370                }
371                else
372                if( source[ srcOffset + 3 ] == EQUALS_SIGN )
373                {
374                        final int outBuff =   ( ( DECODABET[ source[ srcOffset   ] ] & 0xFF ) << 18 )
375                                                  | ( ( DECODABET[ source[ srcOffset + 1 ] ] & 0xFF ) << 12 )
376                                                  | ( ( DECODABET[ source[ srcOffset + 2 ] ] & 0xFF ) <<  6 );
377
378                        destination[ destOffset         ] = (byte)( outBuff >>> 16 );
379                        destination[ destOffset + 1 ] = (byte)( outBuff >>>  8 );
380                        return 2;
381                }
382                else
383                {
384                        try
385                        {
386                                final int outBuff =   ( ( DECODABET[ source[ srcOffset   ] ] & 0xFF ) << 18 )
387                                                          | ( ( DECODABET[ source[ srcOffset + 1 ] ] & 0xFF ) << 12 )
388                                                          | ( ( DECODABET[ source[ srcOffset + 2 ] ] & 0xFF ) <<  6)
389                                                          | ( ( DECODABET[ source[ srcOffset + 3 ] ] & 0xFF )      );
390
391                                destination[ destOffset         ] = (byte)( outBuff >> 16 );
392                                destination[ destOffset + 1 ] = (byte)( outBuff >>  8 );
393                                destination[ destOffset + 2 ] = (byte)( outBuff           );
394
395                                return 3;
396                        }
397                        catch( final Exception e)
398                        {
399                                Log.errOut("CMEncoder",e);
400                                return -1;
401                        }
402                }
403        }

405        public static byte[] B64decode( final byte[] source, final int off, final int len )
406        {
407                final int    len34   = len * 3 / 4;
408                final byte[] outBuff = new byte[ len34 ]; // Upper limit on size of output
409                int    outBuffPosn = 0;
410
411                final byte[] b4          = new byte[4];
412                int    b4Posn    = 0;
413                int    i                 = 0;
414                byte   sbiCrop   = 0;
415                byte   sbiDecode = 0;
416                for( i = off; i < off+len; i++ )
417                {
418                        sbiCrop = (byte)(source[i] & 0x7f); // Only the low seven bits
419                        sbiDecode = DECODABET[ sbiCrop ];
420
421                        if( sbiDecode >= WHITE_SPACE_ENC ) // White space, Equals sign or better
422                        {
423                                if( sbiDecode >= EQUALS_SIGN_ENC )
424                                {
425                                        b4[ b4Posn++ ] = sbiCrop;
426                                        if( b4Posn > 3 )
427                                        {
428                                                outBuffPosn += decode4to3( b4, 0, outBuff, outBuffPosn );
429                                                b4Posn = 0;
430
431                                                if( sbiCrop == EQUALS_SIGN )
432                                                        break;
433                                        }
434                                }
435                        }
436                        else
437                        {
438                                System.err.println( "Bad Base64 input character at " + i + ": " + source[i] + "(decimal)" );
439                                return null;
440                        }
441                }
442
443                final byte[] out = new byte[ outBuffPosn ];
444                System.arraycopy( outBuff, 0, out, 0, outBuffPosn );
445                return out;
446        }

448        public static byte[] B64decode( final String s )
449        {
450                byte[] bytes;
451                try
452                {
453                        bytes = s.getBytes( PREFERRED_ENCODING );
454                }
455                catch( final java.io.UnsupportedEncodingException uee )
456                {
457                        bytes = s.getBytes();
458                }
459                bytes = B64decode( bytes, 0, bytes.length );
460                if( bytes != null && bytes.length >= 4 )
461                {
462
463                        final int head = (bytes[0] & 0xff) | ((bytes[1] << 8) & 0xff00);
464                        if( java.util.zip.GZIPInputStream.GZIP_MAGIC == head )
465                        {
466                                java.io.ByteArrayInputStream  bais = null;
467                                java.util.zip.GZIPInputStream gzis = null;
468                                java.io.ByteArrayOutputStream baos = null;
469                                final byte[] buffer = new byte[2048];
470                                int    length = 0;
471
472                                try
473                                {
474                                        baos = new java.io.ByteArrayOutputStream();
475                                        bais = new java.io.ByteArrayInputStream( bytes );
476                                        gzis = new java.util.zip.GZIPInputStream( bais );
477
478                                        while( ( length = gzis.read( buffer ) ) >= 0 )
479                                        {
480                                                baos.write(buffer,0,length);
481                                        }
482                                        bytes = baos.toByteArray();
483
484                                }
485                                catch( final java.io.IOException e )
486                                {
487                                }
488                                finally
489                                {
490                                        try
491                                        {
492                                                if (baos != null)
493                                                        baos.close();
494                                        }
495                                        catch (final Exception e)
496                                        {
497                                        }
498                                        try
499                                        {
500                                                if (gzis != null)
501                                                        gzis.close();
502                                        }
503                                        catch (final Exception e)
504                                        {
505                                        }
506                                        try
507                                        {
508                                                if (bais != null)
509                                                        bais.close();
510                                        }
511                                        catch (final Exception e)
512                                        {
513                                        }
514                                }
515                        }
516                }
517
518                return bytes;
519        }

628        public static byte[] B64decodeFromFile( final String filename )
629        {
630                byte[] decodedData = null;
631                B64InputStream bis = null;
632                try
633                {
634                        final java.io.File file = new java.io.File( filename );
635                        byte[] buffer = null;
636                        int length   = 0;
637                        int numBytes = 0;
638
639                        if( file.length() > Integer.MAX_VALUE )
640                        {
641                                System.err.println( "File is too big for this convenience method (" + file.length() + " bytes)." );
642                                return null;
643                        }
644                        buffer = new byte[ (int)file.length() ];
645
646                        bis = new B64InputStream(
647                                          new java.io.BufferedInputStream(
648                                          new java.io.FileInputStream( file ) ), DECODE );
649
650                        while( ( numBytes = bis.read( buffer, length, 4096 ) ) >= 0 )
651                                length += numBytes;
652
653                        decodedData = new byte[ length ];
654                        System.arraycopy( buffer, 0, decodedData, 0, length );
655
656                }
657                catch( final java.io.IOException e )
658                {
659                        System.err.println( "Error decoding from file " + filename );
660                }
661                finally
662                {
663                        try
664                        {
665                                if (bis != null)
666                                        bis.close();
667                        }
668                        catch (final Exception e)
669                        {
670                        }
671                }
672
673                return decodedData;
674        }

676        public static String B64encodeFromFile( final String filename )
677        {
678                String encodedData = null;
679                B64InputStream bis = null;
680                try
681                {
682                        final java.io.File file = new java.io.File( filename );
683                        final byte[] buffer = new byte[ (int)(file.length() * 1.4) ];
684                        int length   = 0;
685                        int numBytes = 0;
686                        bis = new B64InputStream(
687                                          new java.io.BufferedInputStream(
688                                          new java.io.FileInputStream( file ) ), ENCODE );
689                        while( ( numBytes = bis.read( buffer, length, 4096 ) ) >= 0 )
690                                length += numBytes;
691                        encodedData = new String( buffer, 0, length, PREFERRED_ENCODING );
692
693                }
694                catch( final java.io.IOException e )
695                {
696                        System.err.println( "Error encoding from file " + filename );
697                }
698                finally
699                {
700                        try
701                        {
702                                if (bis != null)
703                                        bis.close();
704                        }
705                        catch (final Exception e)
706                        {
707                        }
708                }
709
710                return encodedData;
711        }

713        private static class B64InputStream extends java.io.FilterInputStream
714        {
715                private final boolean   encode;                 // Encoding or decoding
716                private int                             position;               // Current position in the buffer
717                private final byte[]    buffer;                 // Small buffer holding
718                                                                                                // converted data
719                private final int               bufferLength;   // Length of buffer (3 or 4)
720                private int                             numSigBytes;    // Number of meaningful bytes in the buffer
721                private int                             lineLength;
722                private final boolean   breakLines;             // Break lines at less than 80characters
723
724                public B64InputStream( final java.io.InputStream in, final int options )
725                {
726                        super( in );
727                        this.breakLines   = (options & DONT_BREAK_LINES) != DONT_BREAK_LINES;
728                        this.encode       = (options & ENCODE) == ENCODE;
729                        this.bufferLength = encode ? 4 : 3;
730                        this.buffer   = new byte[ bufferLength ];
731                        this.position = -1;
732                        this.lineLength = 0;
733                }
734
735                @Override
736                public int read() throws java.io.IOException
737                {
738                        if( position < 0 )
739                        {
740                                if( encode )
741                                {
742                                        final byte[] b3 = new byte[3];
743                                        int numBinaryBytes = 0;
744                                        for( int i = 0; i < 3; i++ )
745                                        {
746                                                try
747                                                {
748                                                        final int b = in.read();
749
750                                                        if( b >= 0 )
751                                                        {
752                                                                b3[i] = (byte)b;
753                                                                numBinaryBytes++;
754                                                        }
755
756                                                }
757                                                catch( final java.io.IOException e )
758                                                {
759                                                        if( i == 0 )
760                                                                throw e;
761
762                                                }
763                                        }
764
765                                        if( numBinaryBytes > 0 )
766                                        {
767                                                encode3to4( b3, 0, numBinaryBytes, buffer, 0 );
768                                                position = 0;
769                                                numSigBytes = 4;
770                                        }
771                                        else
772                                        {
773                                                return -1;
774                                        }
775                                }
776                                else
777                                {
778                                        final byte[] b4 = new byte[4];
779                                        int i = 0;
780                                        for( i = 0; i < 4; i++ )
781                                        {
782                                                int b = 0;
783                                                do{ b = in.read(); }
784                                                while( b >= 0 && DECODABET[ b & 0x7f ] <= WHITE_SPACE_ENC )
785                                                        ;
786
787                                                if( b < 0 )
788                                                        break; // Reads a -1 if end of stream
789
790                                                b4[i] = (byte)b;
791                                        }
792
793                                        if( i == 4 )
794                                        {
795                                                numSigBytes = decode4to3( b4, 0, buffer, 0 );
796                                                position = 0;
797                                        }
798                                        else
799                                        if( i == 0 )
800                                        {
801                                                return -1;
802                                        }
803                                        else
804                                        {
805                                                throw new java.io.IOException( "Improperly padded Base64 input." );
806                                        }
807
808                                }
809                        }
810                        if( position >= 0 )
811                        {
812                                if( position >= numSigBytes )
813                                        return -1;
814
815                                if( encode && breakLines && lineLength >= MAX_LINE_LENGTH )
816                                {
817                                        lineLength = 0;
818                                        return '\n';
819                                }
820                                lineLength++;
821                                final int b = buffer[ position++ ];
822                                if( position >= bufferLength )
823                                        position = -1;
824                                return b & 0xFF;
825                        }
826                        throw new java.io.IOException( "Error in Base64 code reading stream." );
827                }
828
829                @Override
830                public int read( final byte[] dest, final int off, final int len ) throws java.io.IOException
831                {
832                        int i;
833                        int b;
834                        for( i = 0; i < len; i++ )
835                        {
836                                b = read();
837
838                                if( b >= 0 )
839                                        dest[off + i] = (byte)b;
840                                else
841                                if( i == 0 )
842                                        return -1;
843                                else
844                                        break;
845                        }
846                        return i;
847                }
848        }

850        private static class B64OutputStream extends java.io.FilterOutputStream
851        {
852                private final boolean   encode;
853                private int                             position;
854                private byte[]                  buffer;
855                private final int               bufferLength;
856                private int                             lineLength;
857                private final boolean   breakLines;
858                private final byte[]    b4;                             // Scratch used in a few places
859                private final boolean   suspendEncoding;
860
861                public B64OutputStream( final java.io.OutputStream out, final int options )
862                {
863                        super( out );
864                        this.breakLines   = (options & DONT_BREAK_LINES) != DONT_BREAK_LINES;
865                        this.encode       = (options & ENCODE) == ENCODE;
866                        this.bufferLength = encode ? 3 : 4;
867                        this.buffer       = new byte[ bufferLength ];
868                        this.position     = 0;
869                        this.lineLength   = 0;
870                        this.suspendEncoding = false;
871                        this.b4                   = new byte[4];
872                }
873
874                @Override
875                public void write(final int theByte) throws java.io.IOException
876                {
877                        if( suspendEncoding )
878                        {
879                                super.out.write( theByte );
880                                return;
881                        }
882                        if( encode )
883                        {
884                                buffer[ position++ ] = (byte)theByte;
885                                if( position >= bufferLength )
886                                {
887                                        out.write( encode3to4( b4, buffer, bufferLength ) );
888
889                                        lineLength += 4;
890                                        if( breakLines && lineLength >= MAX_LINE_LENGTH )
891                                        {
892                                                out.write( NEW_LINE );
893                                                lineLength = 0;
894                                        }
895                                        position = 0;
896                                }
897                        }
898                        else
899                        {
900                                if( DECODABET[ theByte & 0x7f ] > WHITE_SPACE_ENC )
901                                {
902                                        buffer[ position++ ] = (byte)theByte;
903                                        if( position >= bufferLength )
904                                        {
905                                                final int len = decode4to3( buffer, 0, b4, 0 );
906                                                out.write( b4, 0, len );
907                                                position = 0;
908                                        }
909                                }
910                                else
911                                if( DECODABET[ theByte & 0x7f ] != WHITE_SPACE_ENC )
912                                {
913                                        throw new java.io.IOException( "Invalid character in Base64 data." );
914                                }
915                        }
916                }
917
918                @Override
919                public void write( final byte[] theBytes, final int off, final int len ) throws java.io.IOException
920                {
921                        if( suspendEncoding )
922                        {
923                                super.out.write( theBytes, off, len );
924                                return;
925                        }
926                        for( int i = 0; i < len; i++ )
927                        {
928                                write( theBytes[ off + i ] );
929                        }
930                }
931
932                public void flushBase64() throws java.io.IOException
933                {
934                        if( position > 0 )
935                        {
936                                if( encode )
937                                {
938                                        out.write( encode3to4( b4, buffer, position ) );
939                                        position = 0;
940                                }
941                                else
942                                {
943                                        throw new java.io.IOException( "Base64 input not properly padded." );
944                                }
945                        }
946                }
947
948                @Override
949                public void close() throws java.io.IOException
950                {
951                        flushBase64();
952                        super.close();
953
954                        buffer = null;
955                        out    = null;
956                }
957        }
