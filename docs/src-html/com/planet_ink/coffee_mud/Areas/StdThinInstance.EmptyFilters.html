082        protected Area getParentArea()
083        {
084                if((parentArea!=null)&&(parentArea.get()!=null))
085                        return parentArea.get();
086                final int x=Name().indexOf('_');
087                if(x<0)
088                        return null;
089                if(!CMath.isNumber(Name().substring(0,x)))
090                        return null;
091                final Area parentA = CMLib.map().getArea(Name().substring(x+1));
092                if((parentA==null)
093                ||(!qualifiesToBeParentArea(parentA)))
094                        return null;
095                parentArea=new WeakReference<Area>(parentA);
096                return parentA;
097        }

099        @Override
100        public Room getRoom(String roomID)
101        {
102                if(!CMath.bset(flags(),Area.FLAG_INSTANCE_CHILD))
103                        return super.getRoom(roomID);
104
105                if(!isRoom(roomID))
106                        return null;
107                final Room existingR=super.getRoom(roomID);
108                if(((existingR==null)||(existingR.amDestroyed()))&&(roomID!=null))
109                {
110                        final Area parentA=getParentArea();
111                        if(parentA==null)
112                                return null;
113
114                        if(roomID.toUpperCase().startsWith(Name().toUpperCase()+"#"))
115                                roomID=Name()+roomID.substring(Name().length()); // for case sensitive situations
116                        final Room parentR=parentA.getRoom(parentA.Name()+getStrippedRoomID(roomID));
117                        if(parentR==null)
118                                return null;
119
120                        final Room newR=CMLib.database().DBReadRoomObject(parentR.roomID(), false);
121                        if(newR==null)
122                                return null;
123                        final TreeMap<String,Room> V=new TreeMap<String,Room>();
124                        V.put(roomID,newR);
125                        CMLib.database().DBReadRoomExits(newR.roomID(), newR, false);
126                        CMLib.database().DBReadContent(newR.roomID(), newR, true);
127                        newR.clearSky();
128                        if(newR instanceof GridLocale)
129                                ((GridLocale)newR).clearGrid(null);
130                        for(int d=Directions.NUM_DIRECTIONS()-1;d>=0;d--)
131                                newR.rawDoors()[d]=null;
132                        newR.setRoomID(roomID);
133                        newR.setArea(this);
134                        addProperRoom(newR);
135
136                        synchronized(("SYNC"+roomID).intern())
137                        {
138                                final Room[] newDoors = newR.rawDoors();
139                                for(int d=Directions.NUM_DIRECTIONS()-1;d>=0;d--)
140                                {
141                                        final Room dirR=parentR.rawDoors()[d];
142                                        if(dirR!=null)
143                                        {
144                                                final String myRID=dirR.roomID();
145                                                if((myRID!=null)
146                                                &&(myRID.length()>0)
147                                                &&(dirR.getArea()==parentA))
148                                                {
149                                                        final String localDirRID=convertToMyArea(myRID);
150                                                        final Room localDirR=getProperRoom(localDirRID);
151                                                        if(localDirR!=null)
152                                                                newDoors[d]=localDirR;
153                                                        else
154                                                        if(localDirRID==null)
155                                                                Log.errOut("StdThinInstance","Error in linked room ID "+parentR.roomID()+", dir="+d);
156                                                        else
157                                                        {
158                                                                newDoors[d]=CMClass.getLocale("ThinRoom");
159                                                                newDoors[d].setRoomID(localDirRID);
160                                                                newDoors[d].setArea(this);
161                                                        }
162                                                }
163                                                else
164                                                        newDoors[d]=dirR;
165                                        }
166                                }
167                        }
168                        for(final Enumeration<MOB> e=newR.inhabitants();e.hasMoreElements();)
169                                e.nextElement().bringToLife(newR,true);
170                        newR.startItemRejuv();
171                        fillInAreaRoom(newR);
172                        newR.setExpirationDate(System.currentTimeMillis()+WorldMap.ROOM_EXPIRATION_MILLIS);
173                        return newR;
174                }
175                return existingR;
176        }

178        protected static boolean isInThisArea(final Area childA, final MOB mob)
179        {
180                if(mob != null)
181                {
182                        final Room R=mob.location();
183                        if(R!=null)
184                        {
185                                final Area A=R.getArea();
186                                if(R.getArea()==childA)
187                                        return true;
188                                if(A instanceof BoardableShip)
189                                {
190                                        final Room R2=CMLib.map().roomLocation(((BoardableShip)A).getShipItem());
191                                        if((R2!=null)
192                                        &&(R2.getArea()==childA))
193                                                return true;
194                                }
195                        }
196                }
197                return false;
198        }

200        protected Set<Physical> getProtectedSet(final Area childA, final List<WeakReference<MOB>> registeredMobList)
201        {
202                final Set<Physical> protectTheseList=new HashSet<Physical>();
203                for(final WeakReference<MOB> wmob : registeredMobList)
204                {
205                        final MOB M=wmob.get();
206                        if(M!=null)
207                        {
208                                final Room R=M.location();
209                                if(CMLib.flags().isInTheGame(M,true)
210                                &&(R!=null)
211                                &&(R.getArea()==childA))
212                                        protectTheseList.add(M);
213                        }
214                }
215                for(final Enumeration<MOB> m = CMLib.players().players();m.hasMoreElements();)
216                {
217                        final MOB M=m.nextElement();
218                        if(M!=null)
219                        {
220                                final Room R=M.location();
221                                if((R!=null)
222                                &&(R.getArea()==childA))
223                                        protectTheseList.add(M);
224                        }
225                }
226                for(final Enumeration<BoardableShip> s = CMLib.map().ships();s.hasMoreElements();)
227                {
228                        final BoardableShip B=s.nextElement();
229                        if(B!=null)
230                        {
231                                final Item I=B.getShipItem();
232                                if((I!=null)
233                                &&(I.baseGoldValue()>2)
234                                &&(I.owner() instanceof Room))
235                                {
236                                        final Room R=(Room)I.owner();
237                                        if((R!=null)
238                                        &&(R.getArea()==childA)
239                                        &&(B instanceof PrivateProperty)
240                                        &&(((PrivateProperty)B).getOwnerName()!=null)
241                                        &&(((PrivateProperty)B).getOwnerName().length()>0))
242                                        {
243                                                protectTheseList.add(I);
244                                        }
245                                }
246                        }
247                }
248                final List<MOB> addFollowers=new ArrayList<MOB>(0);
249                for(final Physical p : protectTheseList)
250                {
251                        if(p instanceof MOB)
252                        {
253                                final MOB pmob=(MOB)p;
254                                final Set<MOB> grp=pmob.getGroupMembers(new HashSet<MOB>());
255                                for(final MOB M : grp)
256                                {
257                                        if(pmob!=M)
258                                        {
259                                                final Room R=M.location();
260                                                if((R!=null)
261                                                &&(R.getArea()==childA))
262                                                        addFollowers.add(M);
263                                        }
264                                }
265                        }
266                }
267                protectTheseList.addAll(addFollowers);
268                return protectTheseList;
269        }

323        protected void flushInstance(AreaInstanceChild child)
324        {
325                final Area childA=child.A;
326                final Set<Physical> protectedList = getProtectedSet(childA, child.mobs);
327                childA.setAreaState(State.STOPPED); // kill motion!
328                Room returnBoatsToR = null;
329                for(final Physical P : protectedList)
330                {
331                        if(P instanceof MOB)
332                        {
333                                final MOB wmob=(MOB)P;
334                                final Room R=wmob.location();
335                                if((R!=null)
336                                &&(R.getArea()==childA))
337                                {
338                                        final Room startRoom=wmob.getStartRoom();
339                                        if(startRoom != null)
340                                        {
341                                                if(R.isInhabitant(wmob))
342                                                        startRoom.bringMobHere(wmob, true);
343                                                if(wmob.location()!=startRoom)
344                                                        wmob.setLocation(startRoom);
345                                        }
346                                }
347                        }
348                        else
349                        if(P instanceof Item)
350                        {
351                                final Item I = (Item)P;
352                                Room startRoom = returnBoatsToR;
353                                Room R=CMLib.map().roomLocation(I);
354                                if(R!=null)
355                                {
356                                        if((startRoom == null)
357                                        &&(I instanceof SailingShip))
358                                        {
359                                                Room R1=CMLib.tracking().getRadiantRoomTarget(R, new AllWaterFilters(), new OutOfAreaFilter(childA));
360                                                if(R1!=null)
361                                                {
362                                                        returnBoatsToR=R1;
363                                                        startRoom=R1;
364                                                }
365                                        }
366                                        if(startRoom == null)
367                                        {
368                                                Room R1=CMLib.tracking().getRadiantRoomTarget(R, new EmptyFilters(), new OutOfAreaFilter(childA));
369                                                if(R1!=null)
370                                                        startRoom=R1;
371                                        }
372                                        if(startRoom != null)
373                                        {
374                                                if(R.isHere(I))
375                                                        startRoom.moveItemTo(I);
376                                                R=CMLib.map().roomLocation(I);
377                                                if(R!=startRoom)
378                                                        I.setOwner(startRoom);
379                                        }
380                                }
381                        }
382                }
383                final MOB mob=CMClass.getFactoryMOB();
384                try
385                {
386                        final CMMsg msg=CMClass.getMsg(mob,null,null,CMMsg.MSG_EXPIRE,null);
387                        final LinkedList<Room> propRooms = new LinkedList<Room>();
388                        for(final Enumeration<Room> r=childA.getFilledProperMap();r.hasMoreElements();)
389                                propRooms.add(r.nextElement());
390                        // msgs only, handles saves and stuff, but ignores grid rooms!!
391                        for(final Iterator<Room> r=propRooms.iterator();r.hasNext();)
392                        {
393                                final Room R=r.next();
394                                try
395                                {
396                                        try
397                                        {
398                                                R.clearSky();
399                                                msg.setTarget(R);
400                                                R.executeMsg(mob,msg);
401                                                try
402                                                {
403                                                        CMLib.map().emptyRoom(R, null, true);
404                                                }
405                                                catch(Exception e)
406                                                {
407                                                        Log.errOut(e);
408                                                }
409                                        }
410                                        catch(Exception e)
411                                        {
412                                                Log.errOut(e);
413                                        }
414                                        R.destroy(); // destroys the mobs and items.  the Deadly Thing.
415                                }
416                                catch(Exception e)
417                                {
418                                        Log.errOut(e);
419                                }
420                        }
421                        propRooms.clear();
422                        CMLib.map().delArea(childA);
423                        childA.destroy();
424                }
425                finally
426                {
427                        mob.destroy();
428                }
429        }

431        @Override
432        public void destroy()
433        {
434                super.destroy();
435                if(this.doesManageChildAreas())
436                {
437                        List<AreaInstanceChild> cs = new ArrayList<AreaInstanceChild>(instanceChildren.size());
438                        synchronized(instanceChildren)
439                        {
440                                for(int i=instanceChildren.size()-1;i>=0;i--)
441                                        cs.add(instanceChildren.get(i));
442                                instanceChildren.clear();
443                        }
444                        for(AreaInstanceChild I : cs)
445                        {
446                                final Area A=I.A;
447                                try
448                                {
449                                        flushInstance(I);
450                                }
451                                catch(Exception e)
452                                {
453                                        Log.errOut(e);
454                                }
455                                CMLib.map().delArea(A);
456                                A.destroy();
457                        }
458                }
459        }

468        @Override
469        public boolean tick(final Tickable ticking, final int tickID)
470        {
471                if(!super.tick(ticking, tickID))
472                        return false;
473                if(!doesManageChildAreas())
474                        return true;
475                if((--childCheckDown)<=0)
476                {
477                        childCheckDown=CMProps.getMillisPerMudHour()/CMProps.getTickMillis();
478                        final List<AreaInstanceChild> flushThese = new ArrayList<AreaInstanceChild>(1);
479                        synchronized(instanceChildren)
480                        {
481                                for(int i=instanceChildren.size()-1;i>=0;i--)
482                                {
483                                        final Area childA=instanceChildren.get(i).A;
484                                        if(childA.getAreaState() != Area.State.ACTIVE) // this is the one and only criteria -- if its not active, flush it.
485                                        {
486                                                final AreaInstanceChild child = instanceChildren.remove(i);
487                                                flushThese.add(child);
488                                        }
489                                }
490                        }
491                        for(final AreaInstanceChild inst : flushThese)
492                                flushInstance(inst);
493                }
494                return true;
495        }

497        @Override
498        public void executeMsg(final Environmental myHost, final CMMsg msg)
499        {
500                super.executeMsg(myHost, msg);
501                if(CMath.bset(flags(),Area.FLAG_INSTANCE_CHILD))
502                {
503                        if((msg.sourceMinor()==CMMsg.TYP_SPEAK)
504                        &&(msg.sourceMessage()!=null)
505                        &&((msg.sourceMajor()&CMMsg.MASK_MAGIC)==0))
506                        {
507                                final String said=CMStrings.getSayFromMessage(msg.sourceMessage());
508                                if("RESET INSTANCE".equalsIgnoreCase(said))
509                                {
510                                        Room returnToRoom=null;
511                                        final Room thisRoom=msg.source().location();
512                                        if(thisRoom.getArea()==this)
513                                        {
514                                                for(int d=0;d<Directions.NUM_DIRECTIONS();d++)
515                                                {
516                                                        final Room R=thisRoom.getRoomInDir(d);
517                                                        if((R!=null)&&(R.getArea()!=null)&&(R.getArea()!=this))
518                                                                returnToRoom=R;
519                                                }
520                                        }
521                                        if(returnToRoom==null)
522                                        {
523                                                msg.addTrailerMsg(CMClass.getMsg(msg.source(),null,null,CMMsg.MSG_OK_ACTION,CMMsg.NO_EFFECT,CMMsg.NO_EFFECT, L("You must be at an entrance to reset the area.")));
524                                                return;
525                                        }
526                                        final Area A=this.getParentArea();
527                                        if(A instanceof StdThinInstance)
528                                        {
529                                                final StdThinInstance parentA=(StdThinInstance)A;
530                                                final List<AreaInstanceChild> resetThese = new LinkedList<AreaInstanceChild>();
531                                                synchronized(instanceChildren)
532                                                {
533                                                        for(int i=0;i<parentA.instanceChildren.size();i++)
534                                                        {
535                                                                if(parentA.instanceChildren.get(i).A==this)
536                                                                {
537                                                                        final AreaInstanceChild child = parentA.instanceChildren.remove(i);
538                                                                        resetThese.add(child);
539                                                                }
540                                                        }
541                                                }
542                                                for(final AreaInstanceChild child : resetThese)
543                                                {
544                                                        final List<WeakReference<MOB>> V=child.mobs;
545                                                        for(final WeakReference<MOB> wM : V)
546                                                        {
547                                                                final MOB M=wM.get();
548                                                                if((M!=null)
549                                                                &&CMLib.flags().isInTheGame(M,true)
550                                                                &&(M.location()!=null)
551                                                                &&(M.location()!=returnToRoom)
552                                                                &&(M.location().getArea()==this))
553                                                                {
554                                                                        returnToRoom.bringMobHere(M, true);
555                                                                        CMLib.commands().postLook(M, true);
556                                                                }
557                                                        }
558                                                        setAreaState(Area.State.PASSIVE);
559                                                        flushInstance(child);
560                                                        msg.addTrailerMsg(CMClass.getMsg(msg.source(),CMMsg.MSG_OK_ACTION,L("The instance has been reset.")));
561                                                }
562                                        }
563                                        msg.addTrailerMsg(CMClass.getMsg(msg.source(),CMMsg.MSG_OK_ACTION,L("The instance failed to reset.")));
564                                }
565                        }
566                        else
567                        if((msg.sourceMinor()==CMMsg.TYP_QUIT)&&(CMLib.map().isHere(msg.source(), this)))
568                        {
569                                final MOB mob = msg.source();
570                                CMLib.tracking().forceRecall(mob, true);
571                        }
572                }
573        }

575        @Override 
576        public int[] getAreaIStats()
577        {
578                if(!CMProps.getBoolVar(CMProps.Bool.MUDSTARTED))
579                        return emptyStats;
580                final Area parentArea=getParentArea();
581                final String areaName = (parentArea==null)?Name():parentArea.Name();
582                int[] statData=(int[])Resources.getResource("STATS_"+areaName.toUpperCase());
583                if(statData!=null)
584                        return statData;
585                synchronized(("STATS_"+areaName).intern())
586                {
587                        if(parentArea==null)
588                        {
589                                statData=super.getAreaIStats();
590                                if(statData==emptyStats)
591                                {
592                                        final Iterator<AreaInstanceChild> childE=instanceChildren.iterator();
593                                        int ct=0;
594                                        if(childE.hasNext())
595                                        {
596                                                statData=new int[Area.Stats.values().length];
597                                                for(;childE.hasNext();)
598                                                {
599                                                        final int[] theseStats=childE.next().A.getAreaIStats();
600                                                        if(theseStats != emptyStats)
601                                                        {
602                                                                ct++;
603                                                                for(int i=0;i<theseStats.length;i++)
604                                                                        statData[i]+=theseStats[i];
605                                                        }
606                                                }
607                                        }
608                                        if(ct==0)
609                                                return emptyStats;
610                                        for(int i=0;i<statData.length;i++)
611                                                statData[i]=statData[i]/ct;
612                                }
613                                Resources.removeResource("HELP_"+areaName.toUpperCase());
614                                Resources.submitResource("STATS_"+areaName.toUpperCase(),statData);
615                        }
616                        else
617                        {
618                                return super.getAreaIStats();
619                        }
620                }
621                return statData;
622        }

629        protected Area createRedirectArea(List<MOB> mobs)
630        {
631                final StdThinInstance newA=(StdThinInstance)this.copyOf();
632                newA.properRooms=new STreeMap<String, Room>(new Area.RoomIDComparator());
633                newA.parentArea=null;
634                newA.properRoomIDSet = null;
635                newA.metroRoomIDSet = null;
636                newA.blurbFlags=new STreeMap<String,String>();
637                newA.setName((++instanceCounter)+"_"+Name());
638                newA.flags |= Area.FLAG_INSTANCE_CHILD;
639                for(final Enumeration<String> e=getProperRoomnumbers().getRoomIDs();e.hasMoreElements();)
640                        newA.addProperRoomnumber(newA.convertToMyArea(e.nextElement()));
641                CMLib.map().addArea(newA);
642                newA.setAreaState(Area.State.ACTIVE); // starts ticking
643                final List<WeakReference<MOB>> newMobList = new SVector<WeakReference<MOB>>(5);
644                for(final MOB mob : mobs)
645                        newMobList.add(new WeakReference<MOB>(mob));
646                final AreaInstanceChild child = new AreaInstanceChild(newA,newMobList);
647                instanceChildren.add(child);
648                return newA;
649        }

651        @Override
652        public boolean okMessage(final Environmental myHost, final CMMsg msg)
653        {
654                if(!super.okMessage(myHost, msg))
655                        return false;
656                if(!this.doesManageChildAreas())
657                        return true;
658                if(CMath.bset(flags(),Area.FLAG_INSTANCE_PARENT))
659                        setAreaState(Area.State.PASSIVE);
660                if((msg.sourceMinor()==CMMsg.TYP_ENTER)
661                &&(msg.target() instanceof Room)
662                &&(doesManageMobLists())
663                &&(isRoom((Room)msg.target()))
664                &&((!CMSecurity.isAllowed(msg.source(),(Room)msg.target(),CMSecurity.SecFlag.CMDAREAS))
665                        ||(!msg.source().isAttributeSet(MOB.Attrib.SYSOPMSGS)))
666                &&(((msg.source().getStartRoom()==null)||(msg.source().getStartRoom().getArea()!=this))))
667                {
668                        int myDex=-1;
669                        Area redirectA = null;
670                        final Set<MOB> grp = msg.source().getGroupMembers(new HashSet<MOB>());
671                        if(msg.source().isMonster()
672                        &&(msg.source().riding() instanceof BoardableShip))
673                        {
674                                final List<MOB> mobSet=new LinkedList<MOB>();
675                                boolean playerFound=false;
676                                final Area subA=((BoardableShip)msg.source().riding()).getShipArea();
677                                for(final Enumeration<Room> r=subA.getProperMap();r.hasMoreElements();)
678                                {
679                                        final Room R=r.nextElement();
680                                        if(R!=null)
681                                        {
682                                                for(final Enumeration<MOB> m=R.inhabitants();m.hasMoreElements();)
683                                                {
684                                                        final MOB M=m.nextElement();
685                                                        if(M!=null)
686                                                        {
687                                                                mobSet.add(M);
688                                                                playerFound = playerFound || (!M.isMonster());
689                                                        }
690                                                }
691                                        }
692                                }
693                                if(playerFound)
694                                {
695                                        grp.addAll(mobSet);
696                                }
697                        }
698                        synchronized(instanceChildren)
699                        {
700                                for(int i=0;i<instanceChildren.size();i++)
701                                {
702                                        final List<WeakReference<MOB>> V=instanceChildren.get(i).mobs;
703                                        for (final WeakReference<MOB> weakReference : V)
704                                        {
705                                                if(msg.source() == weakReference.get())
706                                                {
707                                                        myDex=i; 
708                                                        break;
709                                                }
710                                        }
711                                }
712                                for(int i=0;i<instanceChildren.size();i++)
713                                {
714                                        if(i!=myDex)
715                                        {
716                                                final List<WeakReference<MOB>> V=instanceChildren.get(i).mobs;
717                                                for(int v=V.size()-1;v>=0;v--)
718                                                {
719                                                        final WeakReference<MOB> wmob=V.get(v);
720                                                        if(wmob==null)
721                                                                continue;
722                                                        final MOB M=wmob.get();
723                                                        if(grp.contains(M))
724                                                        {
725                                                                if(myDex<0)
726                                                                {
727                                                                        myDex=i;
728                                                                        break;
729                                                                }
730                                                                else
731                                                                if((CMLib.flags().isInTheGame(M,true))
732                                                                &&(M.location().getArea()!=instanceChildren.get(i).A))
733                                                                {
734                                                                        V.remove(wmob);
735                                                                        instanceChildren.get(myDex).mobs.add(new WeakReference<MOB>(M));
736                                                                }
737                                                        }
738                                                }
739                                        }
740                                }
741                                if(myDex>=0)
742                                        redirectA=instanceChildren.get(myDex).A;
743                        }
744                        if(myDex<0)
745                                redirectA = createRedirectArea(new XVector<MOB>(grp));
746                        if(redirectA instanceof StdThinInstance)
747                        {
748                                Room R=redirectA.getRoom(((StdThinInstance)redirectA).convertToMyArea(CMLib.map().getExtendedRoomID((Room)msg.target())));
749                                int tries=1000;
750                                while((R==null)&&((--tries)>0))
751                                {
752                                        R=redirectA.getRandomProperRoom();
753                                        if(R!=null)
754                                        {
755                                                msg.setTarget(R);
756                                                if((!CMLib.flags().canAccess(msg.source(),R))
757                                                ||(CMLib.law().getLandTitle(R)!=null)
758                                                ||(!R.okMessage(msg.source(), msg)))
759                                                        R=null;
760                                        }
761                                }
762                                if(R!=null)
763                                        msg.setTarget(R);
764                        }
765                }
766                return true;
767        }
