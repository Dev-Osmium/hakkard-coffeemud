036public class StdThinGridArea extends StdGridArea
037{
038        @Override
039        public String ID()
040        {
041                return "StdThinGridArea";
042        }
043
044        @Override
045        public long flags()
046        {
047                return Area.FLAG_THIN;
048        }
049
050        public RoomnumberSet myRoomSet=null;
051
052        @Override
053        public void addProperRoom(Room R)
054        {
055                if(R!=null)
056                        R.setExpirationDate(System.currentTimeMillis() + WorldMap.ROOM_EXPIRATION_MILLIS);
057                super.addProperRoom(R);
058        }
059
060        @Override
061        public Room getRoom(String roomID)
062        {
063                if(!isRoom(roomID))
064                        return null;
065                Room R=super.getRoom(roomID);
066                if(((R==null)||(R.amDestroyed()))&&(roomID!=null))
067                {
068                        if(roomID.toUpperCase().startsWith(Name().toUpperCase()+"#"))
069                                roomID=Name()+roomID.substring(Name().length()); // for case sensitive situations
070                        R=CMLib.database().DBReadRoomObject(roomID,false);
071                        if(R!=null)
072                        {
073                                R.setArea(this);
074                                addProperRoom(R);
075                                CMLib.database().DBReadRoomExits(roomID,R,false);
076                                CMLib.database().DBReadContent(roomID,R,true);
077                                fillInAreaRoom(R);
078                                R.setExpirationDate(System.currentTimeMillis()+WorldMap.ROOM_EXPIRATION_MILLIS);
079                        }
080                }
081                return R;
082        }
083
084        @Override 
085        public int getPercentRoomsCached()
086        {
087                final double totalRooms=getProperRoomnumbers().roomCountAllAreas();
088                if(totalRooms==0.0)
089                        return 100;
090                final double currentRooms=getCachedRoomnumbers().roomCountAllAreas();
091                return (int)Math.round((currentRooms/totalRooms)*100.0);
092        }
093
094        @Override 
095        protected int[] buildAreaIStats()
096        {
097                if(!CMProps.getBoolVar(CMProps.Bool.MUDSTARTED))
098                        return emptyStats;
099                final int totalRooms=getProperRoomnumbers().roomCountAllAreas();
100                final int percent=getPercentRoomsCached();
101                if((totalRooms>15)&&(percent<90))
102                        return emptyStats;
103                if((totalRooms>5)&&(percent<50))
104                        return emptyStats;
105                if(percent<10)
106                        return emptyStats;
107                return super.buildAreaIStats();
108        }
109
110        public boolean isRoom(String roomID)
111        {
112                return getProperRoomnumbers().contains(roomID);
113        }
114
115        @Override
116        public boolean isRoom(Room R)
117        {
118                if(R==null)
119                        return false;
120                if(super.isRoom(R))
121                        return true;
122                if(R.roomID().length()==0)
123                        return false;
124                return isRoom(R.roomID());
125        }
126
127        @Override
128        public Enumeration<Room> getProperMap()
129        {
130                return new IteratorEnumeration<Room>(properRooms.values().iterator());
131        }
132
133        @Override
134        public Enumeration<Room> getMetroMap()
135        {
136                final int minimum=getProperRoomnumbers().roomCountAllAreas()/10;
137                if(getCachedRoomnumbers().roomCountAllAreas()<minimum)
138                {
139                        for(int r=0;r<minimum;r++)
140                                getRandomProperRoom();
141                }
142                final MultiEnumeration<Room> multiEnumerator = new MultiEnumeration<Room>(new RoomIDEnumerator(this));
143                for(final Iterator<Area> a=getChildrenReverseIterator();a.hasNext();)
144                        multiEnumerator.addEnumeration(a.next().getMetroMap());
145                return new CompleteRoomEnumerator(multiEnumerator);
146        }
147
148        @Override
149        public Enumeration<Room> getCompleteMap()
150        {
151                return new CompleteRoomEnumerator(new RoomIDEnumerator(this));
152        }
153}
