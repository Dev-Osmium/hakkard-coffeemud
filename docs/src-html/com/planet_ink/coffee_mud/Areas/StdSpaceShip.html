275        @Override
276        public long getMass()
277        {
278                final long mass=this.mass;
279                if(mass<0)
280                {
281                        int newMass=phyStats().weight();
282                        for(final Enumeration<Room> r=getProperMap(); r.hasMoreElements();)
283                        {
284                                final Room R=r.nextElement();
285                                if(R!=null)
286                                {
287                                        for(int i=0;i<R.numItems();i++)
288                                        {
289                                                final Item I=R.getItem(i);
290                                                if(I!=null)
291                                                        newMass += I.phyStats().weight();
292                                        }
293                                        for(int i=0;i<R.numInhabitants();i++)
294                                        {
295                                                final MOB M=R.fetchInhabitant(i);
296                                                if(M!=null)
297                                                        newMass += M.phyStats().weight();
298                                        }
299                                }
300                        }
301                        this.mass=newMass;
302                }
303                return this.mass;
304        }

407        @Override
408        public void executeMsg(final Environmental myHost, final CMMsg msg)
409        {
410                super.executeMsg(myHost, msg);
411
412                if((msg.sourceMinor()==CMMsg.TYP_DROP)||(msg.sourceMinor()==CMMsg.TYP_GET))
413                        mass=-1;
414
415                if(msg.amITarget(this))
416                {
417                        switch(msg.targetMinor())
418                        {
419                        case CMMsg.TYP_ACTIVATE:
420                                if(CMath.bset(msg.targetMajor(), CMMsg.MASK_CNTRLMSG))
421                                {
422                                        final String[] parts=msg.targetMessage().split(" ");
423                                        final TechCommand command=TechCommand.findCommand(parts);
424                                        if(command!=null)
425                                        {
426                                                final Object[] parms=command.confirmAndTranslate(parts);
427                                                if(parms!=null)
428                                                {
429                                                        if(command==Technical.TechCommand.AIRREFRESH)
430                                                        {
431                                                                if((staleAirList.size()==0)
432                                                                &&(msg.tool() instanceof Item)
433                                                                &&(((Item)msg.tool()).owner() instanceof Room)
434                                                                &&(((Room)((Item)msg.tool()).owner()).getAtmosphere()<=0))
435                                                                        doStaleCheck();
436                                                                if(staleAirList.size()>0)
437                                                                {
438                                                                        final double pct=((Double)parms[0]).doubleValue();
439                                                                        final int atmoResource=((Integer)parms[1]).intValue();
440                                                                        int numToClear=(int)Math.round(CMath.mul(staleAirList.size(),pct));
441                                                                        while((numToClear>0)&&(staleAirList.size()>0))
442                                                                        {
443                                                                                final String roomID=staleAirList.iterator().next();
444                                                                                staleAirList.remove(roomID);
445                                                                                changeRoomAir(getRoom(roomID),null,atmoResource);
446                                                                                numToClear--;
447                                                                        }
448                                                                        changeRoomAir(getRandomMetroRoom(),null,atmoResource);
449                                                                        for(final Pair<Room,Integer> p  : shipExitCache)
450                                                                                changeRoomAir(p.first,null,atmoResource);
451                                                                }
452                                                                //if(CMSecurity.isDebugging(DbgFlag.SPACESHIP))
453                                                                //      Log.debugOut("Refreshed the air in "+Name()+", stale rooms: "+staleAirList.size());
454                                                        }
455                                                }
456                                        }
457                                }
458                                break;
459                        }
460                }
461                else
462                {
463                        switch(msg.sourceMinor())
464                        {
465                        case CMMsg.TYP_STAND:
466                        case CMMsg.TYP_DISMOUNT:
467                        {
468                                if(this.getShipFlag(ShipFlag.NO_GRAVITY))
469                                {
470                                        msg.addTrailerRunnable(new Runnable()
471                                        {
472                                                @Override
473                                                public void run()
474                                                {
475                                                        final Ability floater = getGravityFloat();
476                                                        if(floater != null)
477                                                                floater.invoke(floater.invoker(), msg.source(), false, 0);
478                                                }
479                                        });
480                                }
481                                break;
482                        }
483                        }
484                        switch(msg.targetMinor())
485                        {
486                        case CMMsg.TYP_DROP:
487                        case CMMsg.TYP_THROW:
488                                if((msg.target() instanceof Item)
489                                ||(msg.tool() instanceof Item))
490                                {
491                                        if(this.getShipFlag(ShipFlag.NO_GRAVITY))
492                                        {
493                                                final Item I=(msg.target() instanceof Item) ? (Item)msg.target(): (Item)msg.tool();
494                                                msg.addTrailerRunnable(new Runnable()
495                                                {
496                                                        @Override
497                                                        public void run()
498                                                        {
499                                                                final Ability floater = getGravityFloat();
500                                                                if(floater != null)
501                                                                        floater.invoke(floater.invoker(), I, false, 0);
502                                                        }
503                                                });
504                                        }
505                                }
506                                break;
507                        }
508                }
509        }

521        protected boolean changeRoomAir(final Room R, final Room notifyRoom, final int atmoResource)
522        {
523                if(R==null)
524                        return false;
525                if(R.getAtmosphere()!=atmoResource)
526                {
527                        if(atmoResource==0)
528                        {
529                                R.showHappens(CMMsg.MSG_OK_ACTION, L("@x1 rushes out of the room.",RawMaterial.CODES.NAME(R.getAtmosphere()).toLowerCase()));
530                                if((notifyRoom!=null)&&(notifyRoom!=R))
531                                        notifyRoom.showHappens(CMMsg.MSG_OK_ACTION, L("@x1 rushes out of the room.",RawMaterial.CODES.NAME(R.getAtmosphere()).toLowerCase()));
532                        }
533                        else
534                        {
535                                R.showHappens(CMMsg.MSG_OK_ACTION, L("@x1 rushes into the room.",RawMaterial.CODES.NAME(atmoResource).toLowerCase()));
536                                if((notifyRoom!=null)&&(notifyRoom!=R))
537                                        notifyRoom.showHappens(CMMsg.MSG_OK_ACTION, L("@x1 rushes into the room.",RawMaterial.CODES.NAME(atmoResource).toLowerCase()));
538                        }
539                        if(atmoResource==getAtmosphere())
540                                R.setAtmosphere(-1);
541                        else
542                                R.setAtmosphere(atmoResource);
543                        return true;
544                }
545                return false;
546        }

548        protected void moveAtmosphereOut(final Set<Room> doneRooms, final Room startRoom, final int atmo)
549        {
550                final LinkedList<Room> toDoRooms=new LinkedList<Room>();
551                toDoRooms.add(startRoom);
552                while(toDoRooms.size()>0)
553                {
554                        final Room R=toDoRooms.removeFirst();
555                        doneRooms.add(R);
556                        if(atmo == RawMaterial.RESOURCE_NOTHING)
557                        {
558                                if((R.roomID().length()>0)
559                                &&(!staleAirList.contains(R.roomID())))
560                                        staleAirList.add(R.roomID());
561                        }
562                        else
563                        if(atmo == RawMaterial.RESOURCE_AIR)
564                                staleAirList.remove(R.roomID());
565                        if(changeRoomAir(R,startRoom,atmo))
566                                break;
567                        for(int d=0;d<Directions.NUM_DIRECTIONS();d++)
568                        {
569                                final Room R2=R.getRoomInDir(d);
570                                final Exit E2=R.getExitInDir(d);
571                                if((R2!=null)
572                                &&(R2.getArea()==R.getArea())
573                                &&(E2!=null)
574                                &&(E2.isOpen())
575                                &&(!doneRooms.contains(R2)))
576                                        toDoRooms.add(R2);
577                        }
578                }
579        }

581        protected void doStaleCheck()
582        {
583                nextStaleCheck=System.currentTimeMillis()+STALE_AIR_INTERVAL;
584                for(final Enumeration<Room> r=getProperMap();r.hasMoreElements();)
585                {
586                        final Room R=r.nextElement();
587                        if(!staleAirList.contains(R.roomID()))
588                        {
589                                if(R.numInhabitants()>0)
590                                        staleAirList.add(R.roomID());
591                        }
592                        else
593                                R.setAtmosphere(RawMaterial.RESOURCE_NOTHING); // WE NOW HAVE A VACUUM HERE!!!
594                }
595        }

613        protected void doGravityChanges()
614        {
615                if((lastEngine != null)
616                &&(!lastEngine.amDestroyed())
617                &&(lastEngine.activated())
618                &&(lastEngine.getThrust()>1.0)
619                &&(CMLib.map().areaLocation(lastEngine)==this))
620                        lastEThrust = lastEngine.getThrust();
621                else
622                {
623                        lastEngine = null;
624                        lastEThrust = 0.0;
625                        final List<Electronics> electronics=CMLib.tech().getMakeRegisteredElectronics(CMLib.tech().getElectronicsKey(this));
626                        for(final Electronics E : electronics)
627                        {
628                                if((E instanceof ShipEngine)
629                                &&(E.activated())
630                                &&((lastEngine == null)
631                                        ||(((ShipEngine)E).getThrust() > lastEngine.getThrust())))
632                                                lastEngine = (ShipEngine)E;
633                        }
634                        lastEThrust =  (lastEngine == null) ? 0.0 : lastEngine.getThrust();
635                }
636                final boolean gravExistsNow =
637                                getShipFlag(ShipFlag.IN_THE_AIR)
638                        || getShipFlag(ShipFlag.ARTI_GRAV)
639                        || (getIsDocked() != null)
640                        || (lastEThrust > 0.2);
641
642                setShipFlag(ShipFlag.ARTI_GRAV, false);
643                if(gravExistsNow == getShipFlag(ShipFlag.NO_GRAVITY)) // opposite, so it needs changing
644                {
645                        final Ability floater = getGravityFloat();
646                        if(floater != null)
647                        {
648                                final SpaceObject spaceObject=getShipSpaceObject();
649                                final String code=Technical.TechCommand.GRAVITYCHANGE.makeCommand(Boolean.valueOf(gravExistsNow));
650                                final String msgStr;
651                                if(gravExistsNow)
652                                        msgStr=L("You feel the pull of gravity returning.");
653                                else
654                                        msgStr=L("You no longer feel the pull of gravity.");
655                                final CMMsg msg=CMClass.getMsg(floater.invoker(), spaceObject, me, CMMsg.NO_EFFECT, null, CMMsg.MSG_ACTIVATE|CMMsg.MASK_CNTRLMSG, code, CMMsg.MSG_QUIETMOVEMENT,msgStr);
656                                CMMsg gmsg;
657                                if(lastEThrust >= (SpaceObject.ACCELLERATION_PASSOUT-0.49))
658                                {
659                                        gmsg=CMClass.getMsg(floater.invoker(), null, me, CMMsg.NO_EFFECT, null, CMMsg.NO_EFFECT, null, CMMsg.MSG_GRAVITY, null);
660                                        gmsg.setValue((int)Math.round(lastEThrust));
661                                }
662                                else
663                                        gmsg=null;
664                                boolean cancelled = false;
665                                for(final Enumeration<Room> r=getProperMap();r.hasMoreElements();)
666                                {
667                                        final Room R=r.nextElement();
668                                        if(R!=null)
669                                        {
670                                                if(!R.okMessage(msg.source(), msg))
671                                                        cancelled=true;
672                                                if((gmsg != null)
673                                                && ((!R.okMessage(gmsg.source(), gmsg))
674                                                        ||((gmsg.value() < (SpaceObject.ACCELLERATION_PASSOUT-0.49)))))
675                                                        gmsg = null;
676                                        }
677                                }
678                                setShipFlag(ShipFlag.NO_GRAVITY, !gravExistsNow);
679                                if(gmsg != null)
680                                {
681                                        for(final Enumeration<Room> r=getProperMap();r.hasMoreElements();)
682                                        {
683                                                final Room R=r.nextElement();
684                                                if(R!=null)
685                                                        R.send(gmsg.source(), gmsg);
686                                        }
687                                }
688                                if(cancelled)
689                                {
690                                        return;
691                                }
692                                for(final Enumeration<Room> r=getProperMap();r.hasMoreElements();)
693                                {
694                                        final Room R=r.nextElement();
695                                        if(R!=null)
696                                                R.send(floater.invoker(), msg);
697                                }
698                                for(final Enumeration<Room> r=getProperMap();r.hasMoreElements();)
699                                {
700                                        final Room R=r.nextElement();
701                                        if(R!=null)
702                                        {
703                                                for(int i=0;i<R.numInhabitants();i++)
704                                                {
705                                                        final MOB M=R.fetchInhabitant(i);
706                                                        if(M!=null)
707                                                                floater.invoke(floater.invoker(), M, gravExistsNow, 0);
708                                                }
709                                                for(int i=0;i<R.numItems();i++)
710                                                {
711                                                        final Item I=R.getItem(i);
712                                                        if(I!=null)
713                                                                floater.invoke(floater.invoker(), I, gravExistsNow, 0);
714                                                }
715                                        }
716                                }
717                        }
718                }
719        }

721        protected void doAtmosphereChanges()
722        {
723                final Set<Room> doneRooms=new HashSet<Room>();
724                for(final Pair<Room,Integer> p : shipExitCache)
725                {
726                        final Room R=p.first;
727                        final Exit E=R.getExitInDir(p.second.intValue());
728                        if((E!=null)&&(E.isOpen()))
729                        {
730                                final Room exitRoom=R;
731                                final Room otherRoom=R.getRoomInDir(p.second.intValue());
732                                final int atmo=otherRoom.getAtmosphere();
733                                moveAtmosphereOut(doneRooms,exitRoom,atmo);
734                        }
735                }
736                if((System.currentTimeMillis() > nextStaleWarn)
737                &&(staleAirList.size()>0))
738                {
739                        nextStaleWarn = System.currentTimeMillis() + STALE_WARN_INTERVAL;
740                        for(final Enumeration<Room> r=getProperMap();r.hasMoreElements();)
741                        {
742                                final Room R=r.nextElement();
743                                if((staleAirList.contains(R.roomID()))
744                                &&(R.numInhabitants()>0)
745                                &&(R.numPCInhabitants()>0))
746                                {
747                                        final int atmo=R.getAtmosphere();
748                                        if(atmo>0)
749                                        for(int i=0;i<R.numInhabitants();i++)
750                                        {
751                                                final MOB M=R.fetchInhabitant(i);
752                                                if((M!=null)
753                                                &&(!M.isMonster())
754                                                &&(!CMLib.flags().canBreatheThis(M,RawMaterial.RESOURCE_NOTHING)))
755                                                        M.tell(L("The @x1 is seeming a bit stale.",RawMaterial.CODES.NAME(atmo).toLowerCase()));
756                                        }
757                                }
758                        }
759                }
760                if(System.currentTimeMillis() >= nextStaleCheck)
761                {
762                        final int numStaleRooms = staleAirList.size();
763                        doStaleCheck();
764                        if(staleAirList.size()>numStaleRooms)
765                                nextStaleWarn = System.currentTimeMillis() + STALE_WARN_INTERVAL;
766                        if(CMSecurity.isDebugging(DbgFlag.SPACESHIP) && (staleAirList.size()>0))
767                                Log.debugOut("Used up the air in "+Name()+", stale rooms: "+staleAirList.size());
768
769                }
770        }

772        @Override
773        public boolean tick(final Tickable ticking, final int tickID)
774        {
775                if(!super.tick(ticking, tickID))
776                        return false;
777                tickStatus=Tickable.STATUS_START;
778                if(tickID==Tickable.TICKID_AREA)
779                {
780                        doAtmosphereChanges();
781                        doGravityChanges();
782                        final BoardableShip item=this.shipItem;
783                        if(item != null)
784                                item.tick(ticking, tickID);
785                }
786                tickStatus=Tickable.STATUS_NOT;
787                return true;
788        }

805        private static final String[] LOCAL_CODES={"RADIUS","AUTHOR"};
806        private static String[] codes=null;

808        @Override
809        public String[] getStatCodes()
810        {
811                if(codes!=null)
812                        return codes;
813                final String[] superCodes=super.getStatCodes();
814                codes=new String[superCodes.length+LOCAL_CODES.length];
815                int i=0;
816                for(;i<superCodes.length;i++)
817                        codes[i]=superCodes[i];
818                for(int x=0;x<LOCAL_CODES.length;i++,x++)
819                        codes[i]=LOCAL_CODES[x];
820                return codes;
821        }

831        {
832                if(getCodeNum(code) < super.getStatCodes().length)

835                {
836                case 0:
837                        return "" + getOMLCoeff();
838                case 1:

841                return "";
842        }
