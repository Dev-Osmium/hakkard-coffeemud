080        protected Area getParentArea()
081        {
082                if((parentArea!=null)&&(parentArea.get()!=null))
083                        return parentArea.get();
084                final int x=Name().indexOf('_');
085                if(x<0)
086                        return null;
087                if(!CMath.isNumber(Name().substring(0,x)))
088                        return null;
089                final Area parentA = CMLib.map().getArea(Name().substring(x+1));
090                if((parentA==null)
091                ||(!CMath.bset(parentA.flags(),Area.FLAG_INSTANCE_PARENT))
092                ||(CMath.bset(parentA.flags(),Area.FLAG_INSTANCE_CHILD)))
093                        return null;
094                parentArea=new WeakReference<Area>(parentA);
095                return parentA;
096        }

104        @Override
105        public int[] getAreaIStats()
106        {
107                if(!CMProps.getBoolVar(CMProps.Bool.MUDSTARTED))
108                        return emptyStats;
109                final Area parentArea=getParentArea();
110                final String areaName = (parentArea==null)?Name():parentArea.Name();
111                int[] statData=(int[])Resources.getResource("STATS_"+areaName.toUpperCase());
112                if(statData!=null)
113                        return statData;
114                final List<Area> workList = new LinkedList<Area>();
115                synchronized(("STATS_"+Name()).intern())
116                {
117                        if(parentArea==null)
118                        {
119                                for(final Enumeration<AreaInstanceChild> childE=instanceChildren.elements();childE.hasMoreElements();)
120                                        workList.add(childE.nextElement().A);
121                        }
122                        else
123                        {
124                                statData=buildAreaIStats();
125                        }
126                }
127                if(parentArea == null)
128                {
129                        int ct=0;
130                        statData=new int[Area.Stats.values().length];
131                        for(final Area childA : workList)
132                        {
133                                final int[] theseStats=childA.getAreaIStats();
134                                if(theseStats != emptyStats)
135                                {
136                                        ct++;
137                                        for(int i=0;i<theseStats.length;i++)
138                                                statData[i]+=theseStats[i];
139                                }
140                        }
141                        if(ct==0)
142                                return emptyStats;
143                        for(int i=0;i<statData.length;i++)
144                                statData[i]=statData[i]/ct;
145                        Resources.removeResource("HELP_"+areaName.toUpperCase());
146                        Resources.submitResource("STATS_"+areaName.toUpperCase(),statData);
147                }
148                return statData;
149        }

151        @Override
152        public boolean tick(final Tickable ticking, final int tickID)
153        {
154                if(!super.tick(ticking, tickID))
155                        return false;
156                if(CMath.bset(flags(),Area.FLAG_INSTANCE_CHILD))
157                        return true;
158                if((--childCheckDown)<=0)
159                {
160                        childCheckDown=CMProps.getMillisPerMudHour()/CMProps.getTickMillis();
161                        final LinkedList<AreaInstanceChild> workList = new LinkedList<AreaInstanceChild>();
162                        final List<AreaInstanceChild> children = this.instanceChildren;
163                        if(children == null)
164                                return true;
165                        synchronized(children)
166                        {
167                                for(int i=children.size()-1;i>=0;i--)
168                                {
169                                        final AreaInstanceChild child = children.get(i);
170                                        final Area childA=child.A;
171                                        if(childA.getAreaState() != Area.State.ACTIVE)
172                                        {
173                                                workList.add(child);
174                                        }
175                                }
176                        }
177                        for(final AreaInstanceChild child : workList)
178                        {
179                                final Area childA=child.A;
180                                final boolean anyInside=getAreaPlayerMOBs(child).size()>0;
181                                if(!anyInside)
182                                {
183                                        synchronized(children)
184                                        {
185                                                children.remove(child);
186                                        }
187                                        final MOB mob=CMClass.sampleMOB();
188                                        final LinkedList<Room> propRooms = new LinkedList<Room>();
189                                        for(final Enumeration<Room> r=childA.getProperMap();r.hasMoreElements();)
190                                                propRooms.add(r.nextElement());
191                                        for(final MOB M : getAreaPlayerMOBs(child))
192                                        {
193                                                if(M!=null)
194                                                        safePlayerMOBMove(M,childA,M.getStartRoom());
195                                        }
196                                        for(final Iterator<Room> r=propRooms.iterator();r.hasNext();)
197                                        {
198                                                final Room R=r.next();
199                                                R.executeMsg(mob,CMClass.getMsg(mob,R,null,CMMsg.MSG_EXPIRE,null));
200                                        }
201                                        propRooms.clear();
202                                        CMLib.map().delArea(childA);
203                                        childA.destroy();
204                                }
205                        }
206                }
207                return true;
208        }

210        protected List<MOB> getAreaPlayerMOBs(final AreaInstanceChild rec)
211        {
212                final List<WeakReference<MOB>> V=rec.mobs;
213                final List<MOB> list=new LinkedList<MOB>();
214                list.addAll(new ConvertingList<WeakReference<MOB>,MOB>(V,new Converter<WeakReference<MOB>,MOB>() {
215                        @Override
216                        public MOB convert(final WeakReference<MOB> obj)
217                        {
218                                return obj.get();
219                        }
220                }));
221                for(final Iterator<Session> s=CMLib.sessions().sessions();s.hasNext();)
222                {
223                        final Session S=s.next();
224                        final MOB M=(S==null) ? null : S.mob();
225                        if(M != null)
226                        {
227                                final Room R=M.location();
228                                if((R!=null)
229                                &&((R.getArea()==this)||(R.getArea()==rec.A))
230                                &&(!list.contains(M)))
231                                        list.add(M);
232                        }
233                }
234                return list;
235        }

237        protected void safePlayerMOBMove(final MOB M, final Area A, Room returnToRoom)
238        {
239                if(M != null)
240                {
241                        final Room R= M.location();
242                        if(returnToRoom == null)
243                                returnToRoom = CMLib.map().getRandomRoom();
244                        if((R!=null)
245                        &&(R!=returnToRoom)
246                        &&((R.getArea()==A)||(R.getArea()==this)))
247                        {
248                                if(CMLib.flags().isInTheGame(M,true)||(R.isInhabitant(M)))
249                                {
250                                        returnToRoom.bringMobHere(M, true);
251                                        CMLib.commands().postLook(M, true);
252                                }
253                                else
254                                        M.setLocation(returnToRoom);
255                        }
256                }
257        }

259        @Override
260        public void executeMsg(final Environmental myHost, final CMMsg msg)
261        {
262                super.executeMsg(myHost, msg);
263                if(CMath.bset(flags(),Area.FLAG_INSTANCE_CHILD))
264                {
265                        if((msg.sourceMinor()==CMMsg.TYP_SPEAK)
266                        &&(msg.sourceMessage()!=null)
267                        &&((msg.sourceMajor()&CMMsg.MASK_MAGIC)==0))
268                        {
269                                final String said=CMStrings.getSayFromMessage(msg.sourceMessage());
270                                if("RESET INSTANCE".equalsIgnoreCase(said))
271                                {
272                                        Room returnToRoom=null;
273                                        final Room thisRoom=msg.source().location();
274                                        if(thisRoom.getArea()==this)
275                                        {
276                                                for(int d=0;d<Directions.NUM_DIRECTIONS();d++)
277                                                {
278                                                        final Room R=thisRoom.getRoomInDir(d);
279                                                        if((R!=null)&&(R.getArea()!=null)&&(R.getArea()!=this))
280                                                                returnToRoom=R;
281                                                }
282                                        }
283                                        if(returnToRoom==null)
284                                        {
285                                                msg.addTrailerMsg(CMClass.getMsg(msg.source(),null,null,CMMsg.MSG_OK_ACTION,CMMsg.NO_EFFECT,CMMsg.NO_EFFECT, L("You must be at an entrance to reset the area.")));
286                                                return;
287                                        }
288                                        final Area A=this.getParentArea();
289                                        if(A instanceof StdAutoGenInstance)
290                                        {
291                                                final StdAutoGenInstance parentA=(StdAutoGenInstance)A;
292                                                AreaInstanceChild rec = null;
293                                                synchronized(parentA.instanceChildren)
294                                                {
295                                                        for(int i=parentA.instanceChildren.size()-1;i>=0;i--)
296                                                        {
297                                                                if(parentA.instanceChildren.get(i).A==this)
298                                                                {
299                                                                        rec = parentA.instanceChildren.get(i);
300                                                                        break;
301                                                                }
302                                                        }
303                                                }
304                                                if(rec != null)
305                                                {
306                                                        final List<MOB> list = getAreaPlayerMOBs(rec);
307                                                        for(final MOB M : list)
308                                                                safePlayerMOBMove(M,rec.A,returnToRoom);
309                                                        synchronized(parentA.instanceChildren)
310                                                        {
311                                                                parentA.instanceChildren.remove(rec);
312                                                        }
313                                                        final MOB mob=CMClass.sampleMOB();
314                                                        final LinkedList<Room> propRooms = new LinkedList<Room>();
315                                                        for(final Enumeration<Room> r=getProperMap();r.hasMoreElements();)
316                                                                propRooms.add(r.nextElement());
317                                                        for(final Iterator<Room> e=propRooms.iterator();e.hasNext();)
318                                                        {
319                                                                final Room R=e.next();
320                                                                R.executeMsg(mob,CMClass.getMsg(mob,R,null,CMMsg.MSG_EXPIRE,null));
321                                                        }
322                                                        msg.addTrailerMsg(CMClass.getMsg(msg.source(),CMMsg.MSG_OK_ACTION,L("The instance has been reset.")));
323                                                        CMLib.map().delArea(this);
324                                                        destroy();
325                                                        return;
326                                                }
327                                        }
328                                        msg.addTrailerMsg(CMClass.getMsg(msg.source(),CMMsg.MSG_OK_ACTION,L("The instance failed to reset.")));
329                                }
330                        }
331                        else
332                        if((msg.sourceMinor()==CMMsg.TYP_QUIT)&&(CMLib.map().isHere(msg.source(), this)))
333                        {
334                                final MOB mob = msg.source();
335                                CMLib.tracking().forceRecall(mob, true);
336                        }
337                }
338        }

340        @Override
341        public boolean okMessage(final Environmental myHost, final CMMsg msg)
342        {
343                if(!super.okMessage(myHost, msg))
344                        return false;
345                if(CMath.bset(flags(),Area.FLAG_INSTANCE_CHILD))
346                        return true;
347                setAreaState(Area.State.PASSIVE);
348                if((msg.sourceMinor()==CMMsg.TYP_ENTER)
349                &&(msg.target() instanceof Room)
350                &&(CMath.bset(flags(),Area.FLAG_INSTANCE_PARENT))
351                &&(isRoom((Room)msg.target()))
352                &&(((msg.source().getStartRoom()==null)||(msg.source().getStartRoom().getArea()!=this))))
353                {
354                        if(msg.source().isMonster())
355                        {
356                                final Set<MOB> friends=msg.source().getGroupMembers(new HashSet<MOB>());
357                                boolean playerInvolved=false;
358                                for(final MOB M : friends)
359                                        playerInvolved = playerInvolved || (!M.isMonster());
360                                if(!playerInvolved)
361                                {
362                                        msg.source().tell(L("You'll need to be accompanied by an adult to enter there."));
363                                        return false;
364                                }
365                        }
366                        final Set<MOB> grp = msg.source().getGroupMembers(new HashSet<MOB>());
367                        final List<AreaInstanceChild> childSearchGroup;
368                        synchronized(instanceChildren)
369                        {
370                                childSearchGroup = new XVector<AreaInstanceChild>(instanceChildren);
371                        }
372                        int myDex=-1;
373                        AreaInstanceChild myRec = null;
374                        for(int i=0;i<childSearchGroup.size();i++)
375                        {
376                                final List<WeakReference<MOB>> V=childSearchGroup.get(i).mobs;
377                                for (final WeakReference<MOB> weakReference : V)
378                                {
379                                        if(msg.source() == weakReference.get())
380                                        {
381                                                myDex = i;
382                                                myRec=childSearchGroup.get(i);
383                                                break;
384                                        }
385                                }
386                        }
387                        for(int i=0;i<childSearchGroup.size();i++)
388                        {
389                                if(i!=myDex)
390                                {
391                                        final AreaInstanceChild instRec = childSearchGroup.get(i);
392                                        final List<WeakReference<MOB>> V=instRec.mobs;
393                                        for(int v=V.size()-1;v>=0;v--)
394                                        {
395                                                final WeakReference<MOB> wmob=V.get(v);
396                                                if(wmob==null)
397                                                        continue;
398                                                final MOB M=wmob.get();
399                                                if(grp.contains(M))
400                                                {
401                                                        if(myDex<0)
402                                                        {
403                                                                myDex=i;
404                                                                myRec = childSearchGroup.get(i);
405                                                                break;
406                                                        }
407                                                        else
408                                                        if((CMLib.flags().isInTheGame(M,true))
409                                                        &&(M.location().getArea()!=instRec.A))
410                                                        {
411                                                                V.remove(wmob);
412                                                                if((myRec!=null)&&(myRec.mobs!=null))
413                                                                        myRec.mobs.add(new WeakReference<MOB>(M));
414                                                        }
415                                                }
416                                        }
417                                }
418                        }
419                        Area redirectA = null;
420                        int direction = CMLib.map().getRoomDir(msg.source().location(), (Room)msg.target());
421                        if((direction<0)&&(msg.tool() instanceof Exit))
422                                direction = CMLib.map().getExitDir(msg.source().location(), (Exit)msg.tool());
423                        if(direction < 0)
424                        {
425                                msg.source().tell(L("Can't figure out where you're coming from?!"));
426                                return false;
427                        }
428                        if(myDex<0)
429                        {
430                                final StdAutoGenInstance newA=(StdAutoGenInstance)this.copyOf();
431                                newA.properRooms=new STreeMap<String, Room>(new Area.RoomIDComparator());
432                                newA.properRoomIDSet = null;
433                                newA.metroRoomIDSet = null;
434                                newA.blurbFlags=new STreeMap<String,String>();
435                                newA.setName((++instanceCounter)+"_"+Name());
436                                newA.flags |= Area.FLAG_INSTANCE_CHILD;
437                                final Set<MOB> myGroup=msg.source().getGroupMembers(new HashSet<MOB>());
438                                final StringBuffer xml = Resources.getFileResource(getGeneratorXmlPath(), true);
439                                if((xml==null)||(xml.length()==0))
440                                {
441                                        msg.source().tell(L("Unable to load this area.  Please try again later."));
442                                        return false;
443                                }
444                                final List<XMLLibrary.XMLTag> xmlRoot = CMLib.xml().parseAllXML(xml);
445                                final Hashtable<String,Object> definedIDs = new Hashtable<String,Object>();
446                                CMLib.percolator().buildDefinedIDSet(xmlRoot,definedIDs);
447                                String idName = "";
448                                final List<String> idChoices = new Vector<String>();
449                                for(final String key : getAutoGenVariables().keySet())
450                                {
451                                        if(key.equalsIgnoreCase("AREA_ID")||key.equalsIgnoreCase("AREA_IDS")||key.equalsIgnoreCase("AREAID")||key.equalsIgnoreCase("AREAIDS"))
452                                                idChoices.addAll(CMParms.parseCommas(getAutoGenVariables().get(key),true));
453                                }
454                                if(idChoices.size()==0)
455                                {
456                                        for(final Object key : definedIDs.keySet())
457                                        {
458                                                final Object val=definedIDs.get(key);
459                                                if((key instanceof String)
460                                                &&(val instanceof XMLTag)
461                                                &&(((XMLTag)val).tag().equalsIgnoreCase("area")))
462                                                {
463                                                        final XMLTag piece=(XMLTag)val;
464                                                        final String inserter = piece.getParmValue("INSERT");
465                                                        if(inserter!=null)
466                                                        {
467                                                                final List<String> V=CMParms.parseCommas(inserter,true);
468                                                                for(int v=0;v<V.size();v++)
469                                                                {
470                                                                        String s = V.get(v);
471                                                                        if(s.startsWith("$"))
472                                                                                s=s.substring(1).trim();
473                                                                        final XMLTag insertPiece =(XMLTag)definedIDs.get(s.toUpperCase().trim());
474                                                                        if(insertPiece == null)
475                                                                                continue;
476                                                                        if(insertPiece.tag().equalsIgnoreCase("area"))
477                                                                        {
478                                                                                if(!idChoices.contains(s.toUpperCase().trim()))
479                                                                                        idChoices.add(s.toUpperCase().trim());
480                                                                        }
481                                                                }
482                                                        }
483                                                        else
484                                                                idChoices.add((String)key);
485                                                }
486                                        }
487                                }
488
489                                if(idChoices.size()>0)
490                                        idName=idChoices.get(CMLib.dice().roll(1, idChoices.size(), -1)).toUpperCase().trim();
491
492                                if((!(definedIDs.get(idName) instanceof XMLTag))
493                                ||(!((XMLTag)definedIDs.get(idName)).tag().equalsIgnoreCase("area")))
494                                {
495                                        msg.source().tell(L("The area id '@x1' has not been defined in the data file.",idName));
496                                        return false;
497                                }
498                                final ScriptingEngine scrptEng=(ScriptingEngine)CMClass.getCommon("DefaultScriptingEngine");
499                                final Object[] scriptObjs = new Object[ScriptingEngine.SPECIAL_NUM_OBJECTS];
500                                final List<Double> levels=new ArrayList<Double>();
501                                final Set<MOB> followers=msg.source().getGroupMembers(new HashSet<MOB>());
502                                if(!followers.contains(msg.source()))
503                                        followers.add(msg.source());
504                                double totalLevels=0.0;
505                                for(final MOB M : followers)
506                                {
507                                        final Double D=Double.valueOf(M.basePhyStats().level());
508                                        levels.add(D);
509                                        totalLevels+=D.doubleValue();
510                                }
511                                final Double[] sortedLevels=levels.toArray(new Double[0]);
512                                final double lowestLevel=sortedLevels[0].doubleValue();
513                                final double medianLevel=sortedLevels[(int)Math.round(Math.floor(sortedLevels.length/2))].doubleValue();
514                                final double averageLevel=Math.round(10.0*totalLevels/(sortedLevels.length))/10.0;
515                                final double highestLevel=sortedLevels[sortedLevels.length-1].doubleValue();
516                                final double groupSize=Double.valueOf(followers.size()).doubleValue();
517                                final double values[]={msg.source().basePhyStats().level(),lowestLevel,medianLevel,averageLevel,highestLevel,totalLevels,groupSize};
518                                for(final String key : getAutoGenVariables().keySet())
519                                {
520                                        if(!(key.equalsIgnoreCase("AREA_ID")||key.equalsIgnoreCase("AREA_IDS")||key.equalsIgnoreCase("AREAID")||key.equalsIgnoreCase("AREAIDS")))
521                                        {
522                                                final String rawValue = CMath.replaceVariables(getAutoGenVariables().get(key),values);
523                                                final String val=scrptEng.varify(msg.source(), newA, msg.source(), msg.source(), null, null, msg.sourceMessage(), scriptObjs, rawValue);
524                                                definedIDs.put(key.toUpperCase(),val);
525                                        }
526                                }
527                                definedIDs.put("AREANAME", Name());
528                                if(!definedIDs.containsKey("AREASIZE"))
529                                        definedIDs.put("AREASIZE", "50");
530                                if(!definedIDs.containsKey("LEVEL_RANGE"))
531                                        definedIDs.put("LEVEL_RANGE", (msg.source().basePhyStats().level()-4)+"?"+(msg.source().basePhyStats().level()));
532                                if(!definedIDs.containsKey("AGGROCHANCE"))
533                                        definedIDs.put("AGGROCHANCE", ""+msg.source().basePhyStats().level());
534                                try
535                                {
536                                        XMLTag piece=(XMLTag)definedIDs.get(idName);
537                                        final List<XMLTag> pieces = CMLib.percolator().getAllChoices(piece.tag(), piece, definedIDs);
538                                        if(pieces.size()>0)
539                                                piece=pieces.get(CMLib.dice().roll(1, pieces.size(), -1));
540//TODO: fix this
541//FIXME: this can maybe pick town.
542                                        if(!definedIDs.containsKey("THEME"))
543                                        {
544                                                final Map<String,String> unfilled = CMLib.percolator().getUnfilledRequirements(definedIDs,piece);
545                                                final List<String> themes = CMParms.parseCommas(unfilled.get("THEME"), true);
546                                                if(themes.size()>0)
547                                                        definedIDs.put("THEME", themes.get(CMLib.dice().roll(1, themes.size(), -1)).toUpperCase().trim());
548                                        }
549                                        try
550                                        {
551                                                CMLib.percolator().checkRequirements(piece, definedIDs);
552                                        }
553                                        catch(final CMException cme)
554                                        {
555                                                msg.source().tell(L("Required ids for @x1 were missing: @x2",idName,cme.getMessage()));
556                                                return false;
557                                        }
558                                        for(final MOB M : myGroup)
559                                                M.tell(L("^x------------------------------------------------------\n\rPreparing to enter @x1, please stand by...\n\r------------------------------------------------------^N^.",Name()));
560                                        definedIDs.put("ROOMTAG_NODEGATEEXIT", CMLib.directions().getDirectionName(Directions.getOpDirectionCode(direction)));
561                                        definedIDs.put("ROOMTAG_GATEEXITROOM", msg.source().location());
562                                        if(!CMLib.percolator().fillInArea(piece, definedIDs, newA, direction))
563                                        {
564                                                msg.source().tell(L("Failed to enter the new area.  Try again later."));
565                                                return false;
566                                        }
567                                        CMLib.percolator().postProcess(definedIDs);
568                                }
569                                catch(final CMException cme)
570                                {
571                                        Log.errOut("StdAutoGenInstance",cme);
572                                        msg.source().tell(L("Failed to finish entering the new area.  Try again later."));
573                                        return false;
574                                }
575                                redirectA=newA;
576                                CMLib.map().addArea(newA);
577                                newA.setAreaState(Area.State.ACTIVE); // starts ticking
578                                final List<WeakReference<MOB>> newMobList = new SVector<WeakReference<MOB>>(5);
579                                newMobList.add(new WeakReference<MOB>(msg.source()));
580                                final AreaInstanceChild child = new AreaInstanceChild(redirectA,newMobList);
581                                synchronized(instanceChildren)
582                                {
583                                        instanceChildren.add(child);
584                                }
585
586                                getAreaIStats(); // if this is the first child ever, this will force stat making
587
588                                final Room R=redirectA.getRoom(redirectA.Name()+"#0");
589                                if(R!=null)
590                                {
591                                        Exit E=R.getExitInDir(Directions.getOpDirectionCode(direction));
592                                        if(E==null)
593                                                E = CMClass.getExit("Open");
594                                        final int opDir=Directions.getOpDirectionCode(direction);
595                                        if(R.getRoomInDir(opDir)!=null)
596                                                msg.source().tell(L("An error has caused the following exit to be one-way."));
597                                        else
598                                        {
599                                                R.setRawExit(opDir, E);
600                                                R.rawDoors()[opDir]=msg.source().location();
601                                        }
602                                }
603                        }
604                        else
605                        if(myRec != null)
606                                redirectA=myRec.A;
607                        if(redirectA instanceof StdAutoGenInstance)
608                        {
609                                final Room R=redirectA.getRoom(redirectA.Name()+"#0");
610                                if(R!=null)
611                                {
612                                        msg.setTarget(R);
613                                }
614                        }
615                }
616                return true;
617        }

621        @Override
622        public String getStat(final String code)
623        {
624                if(CMParms.indexOfIgnoreCase(STDAREACODES, code)>=0)
625                        return super.getStat(code);
626                else
627                switch(getCodeNum(code))
628                {
629                        case 0:
630                                return this.getGeneratorXmlPath();
631                        case 1:
632                                return CMParms.toEqListString(this.getAutoGenVariables());
633                        default:
634                                break;
635                }
636                return "";
637        }

639        @Override
640        public void setStat(final String code, final String val)
641        {
642                if(CMParms.indexOfIgnoreCase(STDAREACODES, code)>=0)
643                        super.setStat(code, val);
644                else
645                switch(getCodeNum(code))
646                {
647                        case 0:
648                                setGeneratorXmlPath(val);
649                                break;
650                        case 1:
651                                setAutoGenVariables(val);
652                                break;
653                        default:
654                                break;
655                }
656        }

658        @Override
659        protected int getCodeNum(final String code)
660        {
661                for(int i=0;i<MYCODES.length;i++)
662                {
663                        if(code.equalsIgnoreCase(MYCODES[i]))
664                                return i;
665                }
666                return -1;
667        }

669        private static String[] codes=null;
670
671        @Override
672        public String[] getStatCodes()
673        {
674                if(codes!=null)
675                        return codes;
676                final String[] MYCODES=CMProps.getStatCodesList(StdAutoGenInstance.MYCODES,this);
677                final String[] superCodes=STDAREACODES;
678                codes=new String[superCodes.length+MYCODES.length];
679                int i=0;
680                for(;i<superCodes.length;i++)
681                        codes[i]=superCodes[i];
682                for(int x=0;x<MYCODES.length;i++,x++)
683                        codes[i]=MYCODES[x];
684                return codes;
685        }

687        @Override
688        public boolean sameAs(final Environmental E)
689        {
690                if(!(E instanceof StdAutoGenInstance))
691                        return false;
692                final String[] codes=getStatCodes();
693                for(int i=0;i<codes.length;i++)
694                {
695                        if(!E.getStat(codes[i]).equals(getStat(codes[i])))
696                                return false;
697                }
698                return true;
699        }
