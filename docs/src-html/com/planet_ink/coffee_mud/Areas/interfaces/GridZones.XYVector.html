020/**
021 * GridZones is a cross-object interface that applies both to Areas, and Locales.
022 * It represents an area (or room) organized like a Grid.
023 * @author Bo Zimmerman
024 */
025public interface GridZones extends Environmental
026{
027        /**
028         * Returns whether the given Room is a child of this
029         * particular Grid instance.
030         * @see com.planet_ink.coffee_mud.Locales.interfaces.Room
031         * @param loc a Room object
032         * @return whether the room is a child
033         */
034        public boolean isMyGridChild(Room loc);
035        /**
036         * Returns the fully-qualified Room ID of this room
037         * @see com.planet_ink.coffee_mud.Locales.interfaces.Room
038         * @see com.planet_ink.coffee_mud.Areas.interfaces.GridZones#getGridChild(String)
039         * @param loc a Room object
040         * @return a fully qualified room ID
041         */
042        public String getGridChildCode(Room loc);
043        /**
044         * Returns the fully-qualified Room ID of this room
045         * @see com.planet_ink.coffee_mud.Locales.interfaces.Room
046         * @see com.planet_ink.coffee_mud.Areas.interfaces.GridZones#getGridChildCode(Room)
047         * @param childCode a fully-qualified Room ID for this Room
048         * @return a Room object
049         */
050        public Room getGridChild(String childCode);
051        /**
052         * Returns a random Room object that is a child of this one.
053         * @see com.planet_ink.coffee_mud.Locales.interfaces.Room
054         * @return a Room object
055         */
056        public Room getRandomGridChild();
057        /**
058         * Returns the X coordinate of the given Room object
059         * @see com.planet_ink.coffee_mud.Locales.interfaces.Room
060         * @param loc the Room object
061         * @return the x coordinate of the room
062         */
063        public int getGridChildX(Room loc);
064        /**
065         * Returns the Y coordinate of the given Room object
066         * @see com.planet_ink.coffee_mud.Locales.interfaces.Room
067         * @param loc the Room object
068         * @return the y coordinate of the room
069         */
070        public int getGridChildY(Room loc);
071
072        /**
073         * Returns the XY coordinates of the Room with the given roomID
074         * in XYVector format.
075         * @see XYVector
076         * @param roomID the roomID of the room to get coordinates for
077         * @return coordinates in XYVector format.
078         */
079        public XYVector getRoomXY(String roomID);
080
081        /**
082         * Returns the XY coordinates of the Room, if a child of
083         * this gridzone, in XYVector format.
084         * @see XYVector
085         * @param room the room to get coordinates for
086         * @return coordinates in XYVector format.
087         */
088        public XYVector getRoomXY(Room room);
089
090        /**
091         * Returns the total width of this grid.
092         * @return the width
093         */
094        public int xGridSize();
095        /**
096         * Returns the total height of this grid
097         * @return the height
098         */
099        public int yGridSize();
100        /**
101         * Sets the total width of this grid.
102         * @param x the width
103         */
104        public void setXGridSize(int x);
105        /**
106         * Sets the total height of this grid.
107         * @param y the height
108         */
109        public void setYGridSize(int y);
110        /**
111         * Returns the Room object at the given coordinates.
112         * @see com.planet_ink.coffee_mud.Locales.interfaces.Room
113         * @param x the x coordinate
114         * @param y the y coordinate
115         * @return the Room object at those coordinates
116         */
117        public Room getGridChild(int x, int y);
118
119        /**
120         * Returns the Room object at the given coordinates.
121         * @see com.planet_ink.coffee_mud.Locales.interfaces.Room
122         * @see XYVector
123         * @param xy the x and y coordinate
124         * @return the Room object at those coordinates
125         */
126        public Room getGridChild(XYVector xy);
127
128        /**
129         * A class for holding x/y coordinates. Used by GridZones
130         * as a way to hold such coordinates in a single place,
131         * and easily compare them to each other.
132         * @author Bo Zimmerman
133         */
134        public static class XYVector
135        {
136                public int x;
137                public int y;
138                public XYVector(int x, int y)
139                {
140                        this.x=x;
141                        this.y=y;
142                }
143        }
144}
