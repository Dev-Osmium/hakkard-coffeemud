600        @SuppressWarnings("rawtypes")
601        protected void cloneFix(final MOB M)
602        {
603                if (M == null)
604                        return;
605                me=this;
606                if (!isGeneric())
607                {
608                        final PhyStats oldBase=basePhyStats;
609                        basePhyStats = (PhyStats) M.basePhyStats().copyOf();
610                        phyStats = (PhyStats) M.phyStats().copyOf();
611                        basePhyStats.setAbility(oldBase.ability());
612                        basePhyStats.setRejuv(oldBase.rejuv());
613                        basePhyStats.setLevel(oldBase.level());
614                        phyStats.setAbility(oldBase.ability());
615                        phyStats.setRejuv(oldBase.rejuv());
616                        phyStats.setLevel(oldBase.level());
617                }
618                else
619                {
620                        basePhyStats = (PhyStats) M.basePhyStats().copyOf();
621                        phyStats = (PhyStats) M.phyStats().copyOf();
622                }
623
624                affectPhyStats = new ApplyAffectPhyStats<Ability>(this);
625                recoverAffectP  = new ApplyRecAffectPhyStats<Item>(this);
626                affectCharStats= new ApplyAffectCharStats(this);
627                affectCharState = new ApplyAffectCharState(this);
628
629                affects = new SVector<Ability>();
630                baseCharStats = (CharStats) M.baseCharStats().copyOf();
631                charStats = (CharStats) M.charStats().copyOf();
632                baseState = (CharState) M.baseState().copyOf();
633                curState = (CharState) M.curState().copyOf();
634                maxState = (CharState) M.maxState().copyOf();
635                removeFromGame = false;
636
637                inventory= new SVector<Item>();
638                abilitys= new CMUniqSortSVec<Ability>();
639                abilityUseTrig = new int[3];
640                abilityUseCache= new STreeMap<String,int[][]>();
641                behaviors= new CMUniqSortSVec<Behavior>();
642                tattoos = new CMUniqNameSortSVec<Tattoo>();
643                expertises = new STreeMap<String,Integer>();
644                followers = null;
645                commandQue = new LinkedList<QMCommand>();
646                scripts = new SVector<ScriptingEngine>();
647                racialAffects = null;
648                clanAffects     = null;
649                factions = new SHashtable<String, FData>(1);
650                possWieldedItem= null;
651                possHeldItem = null;
652                clans = new OrderedMap<String,Pair<Clan,Integer>>();
653
654                for(final Pair<Clan,Integer> p : M.clans())
655                {
656                        setClan(p.first.clanID(), p.second.intValue());
657                }
658                for(final Enumeration<String> e=M.factions();e.hasMoreElements();)
659                {
660                        final String fac=e.nextElement();
661                        addFaction(fac, M.fetchFaction(fac));
662                }
663                for(final Enumeration<Tattoo> e=M.tattoos();e.hasMoreElements();)
664                {
665                        final Tattoo t=e.nextElement();
666                        addTattoo((Tattoo)t.copyOf());
667                }
668                for(final Enumeration<String> s=M.expertises();s.hasMoreElements();)
669                        addExpertise(s.nextElement());
670
671                Item I = null;
672                for (int i = 0; i < M.numItems(); i++)
673                {
674                        I = M.getItem(i);
675                        if (I != null)
676                                addItem((Item) I.copyOf());
677                }
678                Item I2 = null;
679                for (int i = 0; i < numItems(); i++)
680                {
681                        I = getItem(i);
682                        if ((I != null)
683                        && (I.container() != null)
684                        && (!isMine(I.container())))
685                        {
686                                for (final Enumeration<Item> e = M.items(); e.hasMoreElements();)
687                                {
688                                        I2 = e.nextElement();
689                                        if ((I2 == I.container()) && (I2 instanceof Container))
690                                        {
691                                                I.setContainer((Container) I2);
692                                                break;
693                                        }
694                                }
695                        }
696                }
697                Ability A = null;
698                for (int i = 0; i < M.numAbilities(); i++)
699                {
700                        A = M.fetchAbility(i);
701                        if (A != null)
702                                addAbility((Ability) A.copyOf());
703                }
704                try
705                {
706                        amDestroyed=true;
707                        for (final Enumeration<Ability> a = M.personalEffects(); a.hasMoreElements();)
708                        {
709                                A = a.nextElement();
710                                if (A != null)
711                                {
712                                        A = (Ability) A.copyOf();
713                                        addEffect(A);
714                                        if (A.canBeUninvoked())
715                                        {
716                                                A.unInvoke();
717                                                delEffect(A);
718                                        }
719                                }
720                        }
721                }
722                finally
723                {
724                        amDestroyed=false;
725                }
726                for (final Enumeration<Behavior> e = M.behaviors(); e.hasMoreElements();)
727                {
728                        final Behavior B = e.nextElement();
729                        if (B != null) // iteration during a clone would just be messed up.
730                                behaviors.addElement((Behavior) B.copyOf());
731                }
732                ScriptingEngine SE = null;
733                for (final Enumeration<ScriptingEngine> e = M.scripts(); e.hasMoreElements();)
734                {
735                        SE = e.nextElement();
736                        if (SE != null)
737                                addScript((ScriptingEngine) SE.copyOf());
738                }
739        }

746                        final StdMOB E = (StdMOB) this.clone();
