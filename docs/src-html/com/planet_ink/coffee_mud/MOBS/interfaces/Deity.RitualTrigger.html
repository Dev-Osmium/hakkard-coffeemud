033/**
034 * A Deity is a type of mob that provides benefits (and punishments) for
035 * players that worship them, especially Clerics.  They also provide 
036 * for mini-mini quests (called rituals or sins) for obtaining the benefits
037 * or punishments, as well as mini-mini quests for clerics to do services.
038 * 
039 * This interface defines the language for specifying the rituals, 
040 * as well as the methods for accessing them and their benefits and
041 * detriments.
042 * @author Bo Zimmerman
043 *
044 */
045public interface Deity extends MOB
046{
047        /**
048         * The definition of the key words in the ritual definitions.
049         * Most of these require a parameter of one sort or another,
050         * depending on the deity class.  The command phrases
051         * are separated by &amp; (for and) or | for or.
052         * @author Bo Zimmerman
053         *
054         */
055        public enum RitualTrigger
056        {
057                SAY(CMMsg.TYP_SPEAK),
058                TIME(),
059                PUTTHING(CMMsg.TYP_PUT,"PUT"),
060                BURNTHING(CMMsg.TYP_FIRE),
061                EAT(CMMsg.TYP_EAT),
062                DRINK(CMMsg.TYP_DRINK),
063                INROOM(CMMsg.TYP_LOOK),
064                RIDING(),
065                CAST(CMMsg.TYP_CAST_SPELL),
066                EMOTE(CMMsg.TYP_EMOTE),
067                PUTVALUE(CMMsg.TYP_PUT),
068                PUTMATERIAL(CMMsg.TYP_PUT),
069                BURNMATERIAL(CMMsg.TYP_FIRE,"BURN"),
070                BURNVALUE(CMMsg.TYP_FIRE),
071                SITTING("SIT"),
072                STANDING("STAND"),
073                SLEEPING("SLEEP"),
074                READING(CMMsg.TYP_READ,"READ"),
075                RANDOM(),
076                CHECK(),
077                WAIT(),
078                YOUSAY(),
079                OTHERSAY(),
080                ALLSAY(),
081                ;
082                private final int msgCode;
083                private final String shortName;
084                
085                private RitualTrigger(int msgCode, String shortName)
086                {
087                        this.msgCode = msgCode;
088                        this.shortName = shortName;
089                }
090
091                private RitualTrigger(int msgCode)
092                {
093                        this.msgCode = msgCode;
094                        this.shortName = name();
095                }
096
097                private RitualTrigger(String shortName)
098                {
099                        this.msgCode = -999;
100                        this.shortName = name();
101                }
102
103                private RitualTrigger()
104                {
105                        this.msgCode = -999;
106                        this.shortName = name();
107                }
108                
109                public int getCMMsgCode()
110                {
111                        return msgCode;
112                }
113                
114                public String getShortName()
115                {
116                        return shortName;
117                }
118        }
