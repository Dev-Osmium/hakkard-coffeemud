037/**
038 * A PostOffice is a kind of shopkeeper that belongs to a "chain" which
039 * shares access to a common store of post office boxes to hold items
040 * for players.  Post officers simulate delivery by transferring the
041 * items given to them into the appropriate boxes.  Players must then
042 * go to a post office in the same chain to retrieve items delivered
043 * to them.  
044 * Post offices can serve an entire clan, or a single player.
045 * They can handle several different types of charges, including COD.
046 * 
047 * @author Bo Zimmerman
048 */
049public interface PostOffice extends ShopKeeper
050{
051        /**
052         * When the given mob tries to send or withdraw a package, this method is 
053         * called to get the proper FROM address, which is either the mob themselves 
054         * or their clan, if they are (optionally) permitted by their rank.
055         * If checked is true, and the mob does NOT have clan privileges, then an
056         * error message is given to the mob and null is returned.
057         * @see com.planet_ink.coffee_mud.Common.interfaces.Clan.Function
058         * @param mob the mob who is trying to send the package
059         * @param func either Clan.Function.WITHDRAW or Clan.FUNCTION.DEPOSIT
060         * @param checked true if the mob must have clan privileges, false if not.
061         * @return the mobs name, their clan name, or null
062         */
063        public String getSenderName(MOB mob, Clan.Function func, boolean checked);
064
065        /**
066         * Low level method to add an item directly to a postal box.
067         * @see PostOffice#delFromBox(String, Item)
068         * @see PostOffice#emptyBox(String)
069         * @param boxName the box name, such as a player or clan name
070         * @param thisThang the item to add to the postal box
071         * @param from player or clan the item is from
072         * @param to player or clan the item is to
073         * @param holdTime when the package was sent, real time, in milliseconds
074         * @param COD the amount of money which must be paid to withdraw the item
075         */
076        public void addToBox(String boxName, Item thisThang, String from, String to, long holdTime, double COD);
077
078        /**
079         * Low level method to remove an item from a postal box
080         * @see PostOffice#addToBox(String, Item, String, String, long, double)
081         * @see PostOffice#emptyBox(String)
082         * @param boxName the box name, such as a player or clan name
083         * @param thisThang the item to remove from the postal box
084         * @return true if the item was removed, or false otherwise
085         */
086        public boolean delFromBox(String boxName, Item thisThang);
087
088        /**
089         * Removes all items from the given post office box
090         * @see PostOffice#addToBox(String, Item, String, String, long, double)
091         * @see PostOffice#delFromBox(String, Item)
092         * @param boxName the box name, such as a player or clan name
093         */
094        public void emptyBox(String boxName);
095
096        /**
097         * Searches the given postal box for an item with the given
098         * name, or one with the given string as a substring.
099         * @param boxName the player or clan name of the postal box
100         * @param likeThis the search string for the item name
101         * @return the item found, or null
102         */
103        public Item findBoxContents(String boxName, String likeThis);
104        
105        /**
106         * Returns a mapping of postal branches in this chain to forwarding 
107         * addresses, for the given postal box.
108         * @see PostOffice#deleteBoxHere(String)
109         * @see PostOffice#createBoxHere(String, String)
110         * @param boxName the player or clan name of the box
111         * @return a mapping of postal branches to forwarding addresses.
112         */
113        public Map<String, String> getOurOpenBoxes(String boxName);
114
115        /**
116         * Creates a new post office box for the given player or clan
117         * name, with the given forwarding address.
118         * @see PostOffice#deleteBoxHere(String)
119         * @see PostOffice#getOurOpenBoxes(String)
120         * @param boxName the player or clan name of the new box
121         * @param forward the player or clan name of the forwarding
122         */
123        public void createBoxHere(String boxName, String forward);
124
125        /**
126         * Removes the given post office box for the given player or
127         * clan name.
128         * @see PostOffice#createBoxHere(String, String)
129         * @see PostOffice#getOurOpenBoxes(String)
130         * @param boxName the player or clan name of the new box
131         */
132        public void deleteBoxHere(String boxName);
133
134        /**
135         * Parses an postal package entry, which is formatted
136         * as item xml preceded by various semicolon-delimited
137         * numbers and data. 
138         * @see PostOffice.MailPiece
139         * @param data the postal package formatted string
140         * @return the MailPiece object
141         */
142        public MailPiece parsePostalItemData(String data);
143
144        /**
145         * Returns the postal chain, a string shared by all post 
146         * offices that deal with the same postal boxes and
147         * customers, able to send mail to each other.
148         * @see PostOffice#setPostalChain(String)
149         * @return the name of the postal chain
150         */
151        public String postalChain();
152
153        /**
154         * Sets the postal chain, a string shared by all post 
155         * offices that deal with the same postal boxes and
156         * customers, able to send mail to each other.
157         * @see PostOffice#postalChain()
158         * @param name  the name of the postal chain
159         */
160        public void setPostalChain(String name);
161
162        /**
163         * Returns the branch of the postal chain that this specific
164         * postoffice employee belongs to.  Typically the extended
165         * room id, this is a unique identifier for this branch of
166         * the postal chain.  Each PostOffice should be unique!
167         * @see PostOffice#findProperBranch(String)
168         * @return the branch of the postal chain
169         */
170        public String postalBranch(); // based on individual shopkeeper
171
172        /**
173         * Returns the proper branch of this postal chain to which to
174         * deliver packages belonging to the given mob or clan name.
175         * @see PostOffice#postalBranch()
176         * @param name the player or clan name receiving the package
177         * @return the best branch in this postal chain to deliver to
178         */
179        public String findProperBranch(String name);
180
181        /**
182         * Gets the minimum postage to send a package from this branch.
183         * @see PostOffice#setMinimumPostage(double)
184         * @return the minimum postage to send a package from this branch.
185         */
186        public double minimumPostage();
187
188        /**
189         * Sets the minimum postage to send a package from this branch.
190         * @see PostOffice#minimumPostage()
191         * @param d the minimum postage to send a package from this branch.
192         */
193        public void setMinimumPostage(double d);
194
195        /**
196         * Gets the postage charged per pound after the first, to send
197         * packages from this branch.
198         * @see PostOffice#setPostagePerPound(double)
199         * @return the postage charged per pound after the first
200         */
201        public double postagePerPound();
202
203        /**
204         * Sets the postage charged per pound after the first, to send
205         * packages from this branch.
206         * @see PostOffice#postagePerPound()
207         * @param d the postage charged per pound after the first
208         */
209        public void setPostagePerPound(double d);
210
211        /**
212         * Gets the fee charged per pound per mud month to hold a 
213         * package in a postal box at this branch.
214         * @see PostOffice#setHoldFeePerPound(double)
215         * @return the fee charged per pound per mud month
216         */
217        public double holdFeePerPound();
218
219        /**
220         * Sets the fee charged per pound per mud month to hold a 
221         * package in a postal box at this branch.
222         * @see PostOffice#holdFeePerPound()
223         * @param d the fee charged per pound per mud month
224         */
225        public void setHoldFeePerPound(double d);
226
227        /**
228         * Gets the fee to open a new postal box at this branch.
229         * @see PostOffice#setFeeForNewBox(double)
230         * @return the fee to open a new postal box at this branch.
231         */
232        public double feeForNewBox();
233
234        /**
235         * Sets the fee to open a new postal box at this branch.
236         * @see PostOffice#feeForNewBox()
237         * @param d the fee to open a new postal box at this branch.
238         */
239        public void setFeeForNewBox(double d);
240
241        /**
242         * Returns the maximum number of mud-months that a package
243         * will be held before it gets put on sale at this branch.
244         * @see PostOffice#setMaxMudMonthsHeld(int)
245         * @return the maximum number of mud-months to hold
246         */
247        public int maxMudMonthsHeld();
248
249        /**
250         * Sets the maximum number of mud-months that a package
251         * will be held before it gets put on sale at this branch.
252         * @see PostOffice#setMaxMudMonthsHeld(int)
253         * @param months the maximum number of mud-months to hold
254         */
255        public void setMaxMudMonthsHeld(int months);
256
257        /**
258         * Represents a complete mail package/letter, postal
259         * object. Holding all relevant info about a package
260         * from one person to another.
261         * @author Bo Zimmerman
262         *
263         */
264        public static class MailPiece
265        {
266                public String   from    = "";
267                public String   to              = "";
268                public String   time    = "";
269                public String   cod             = "";
270                public String   classID = "";
271                public String   xml             = "";
272        }
