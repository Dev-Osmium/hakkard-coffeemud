037/**
038 * A Banker is a kind of shopkeeper that belongs to a "chain" which
039 * shares access to a common store of player/clan accounts to hold money
040 * and items.  Players must go to a banker in the same chain to retrieve 
041 * money and items deposited with the chain.  Bankers respond to new
042 * commands such as DEPOSIT and WITHDRAW that normal shopkeepers do not.
043 * Bankers also respect marriage by giving both parters access to each
044 * others accounts.
045 * 
046 * Bankers can serve an entire clan, or a single player.
047 * 
048 * @author Bo Zimmerman
049 */
050public interface Banker extends ShopKeeper
051{
052        /**
053         * The default value by which the value of items is divided to determine minimum money
054         * to continue item storage.  Default is needing 10% of item value in money to store.
055         */
056        public final static double MIN_ITEM_BALANCE_DIVISOR=10.0;
057
058        /**
059         * Gets the interest rate paid (or cost) on deposited money.
060         * A positive value is a payment, a negative is a cost.
061         * @see Banker#setCoinInterest(double)
062         * @return the interest rate paid (or cost) on deposited money.
063         */
064        public double getCoinInterest();
065        
066        /**
067         * Sets the interest rate paid (or cost) on deposited money.
068         * A positive value is a payment, a negative is a cost.
069         * @see Banker#getCoinInterest()
070         * @param interest the interest rate paid (or cost) on deposited money.
071         */
072        public void setCoinInterest(double interest);
073        
074        /**
075         * Gets the interest rate paid (or cost) on the value of deposited
076         * items.
077         * A positive value is a payment, a negative is a cost.
078         * @see Banker#setItemInterest(double)
079         * @return the interest rate paid (or cost) on deposited items.
080         */
081        public double getItemInterest();
082        
083        /**
084         * Sets the interest rate paid (or cost) on the value of deposited
085         * items.
086         * A positive value is a payment, a negative is a cost.
087         * @see Banker#getItemInterest()
088         * @param interest the interest rate paid (or cost) on deposited items.
089         */
090        public void setItemInterest(double interest);
091        
092        /**
093         * Gets the interest rate paid (or cost) on loaned out funds as  debt.
094         * A positive value is a cost, a negative is a bonus.
095         * @see Banker#setLoanInterest(double)
096         * @return the interest rate paid (or cost) on loaned money debt.
097         */
098        public double getLoanInterest();
099        
100        /**
101         * Sets the interest rate paid (or cost) on loaned out funds as  debt.
102         * A positive value is a cost, a negative is a bonus.
103         * @see Banker#getLoanInterest()
104         * @param interest the interest rate paid (or cost) on loaned money debt.
105         */
106        public void setLoanInterest(double interest);
107        
108        /**
109         * Gets the name of the bank chain to which this banker belongs.
110         * @see Banker#setBankChain(String)
111         * @return the bank chain name
112         */
113        public String bankChain();
114        
115        /**
116         * Sets the name of the bank chain to which this banker belongs.
117         * @see Banker#bankChain()
118         * @param name the bank chain name
119         */
120        public void setBankChain(String name);
121        
122        /**
123         * Returns all the player and clan names who have open accounts
124         * at this bank.
125         * @return all the player and clan names who have open accounts
126         */
127        public List<String> getAccountNames();
128        
129        /**
130         * When the given mob tries to deposit or withdraw something, this method is 
131         * called to get the proper account name, which is either the mob themselves 
132         * or their clan, if they are (optionally) permitted by their rank.
133         * If checked is true, and the mob does NOT have clan privileges, then an
134         * error message is given to the mob and null is returned.
135         * @see com.planet_ink.coffee_mud.Common.interfaces.Clan.Function
136         * @param mob the mob who is trying to deposit or withdraw or list or something
137         * @param func either Clan.Function.WITHDRAW or Clan.FUNCTION.DEPOSIT or LIST
138         * @param checked true if the mob must have clan privileges, false if not.
139         * @return the mobs name, their clan name, or null
140         */
141        public String getBankClientName(MOB mob, Clan.Function func, boolean checked);
142
143        /**
144         * Returns debt information for the given depositor to this bank chain.
145         * @see com.planet_ink.coffee_mud.Libraries.interfaces.MoneyLibrary.DebtItem
146         * @param depositorName the player or clan name that owes the bank money
147         * @return information about the debt
148         */
149        public MoneyLibrary.DebtItem getDebtInfo(String depositorName);
150
151        /**
152         * Deposits a new item into the given account.  Coin items are, of course,
153         * money.
154         * @see Banker#delAllDeposits(String)
155         * @see Banker#delDepositInventory(String, Item)
156         * @see Banker#getBalance(String)
157         * @param depositorName the account to deposit into, like mob or clan name 
158         * @param item the item to deposit
159         * @param container the container the item is in, which also needs depositing
160         */
161        public void addDepositInventory(String depositorName, Item item, Item container);
162
163        /**
164         * Deletes item into the given account.  Coin items are, of course,
165         * money.  The items are returned in a list to the caller.  If the 
166         * item is a container, all contained items are also returned.
167         * @see Banker#delAllDeposits(String)
168         * @see Banker#addDepositInventory(String, Item, Item)
169         * @see Banker#getBalance(String)
170         * @param depositorName the account to delete from, like mob or clan name 
171         * @param likeItem the likeItem to delete
172         * @return the collection of items deleted from the account
173         */
174        public List<Item> delDepositInventory(String depositorName, Item likeItem);
175        
176        /**
177         * Empties all the items and money from a given depositors box.
178         * @see Banker#delDepositInventory(String, Item)
179         * @see Banker#addDepositInventory(String, Item, Item)
180         * @see Banker#getBalance(String)
181         * @param depositorName the account to empty, like mob or clan name 
182         */
183        public void delAllDeposits(String depositorName);
184        
185        /**
186         * Returns the money balance in the account, in base value
187         * @see Banker#delDepositInventory(String, Item)
188         * @see Banker#addDepositInventory(String, Item, Item)
189         * @see Banker#delAllDeposits(String)
190         * @param depositorName the account to empty, like mob or clan name 
191         * @return the money balance, in base value
192         */
193        public double getBalance(String depositorName);
194        
195        /**
196         * Returns the number of items deposited, including money items
197         * @see Banker#getDepositedItems(String)
198         * @see Banker#findDepositInventory(String, String)
199         * @see Banker#totalItemsWorth(String)
200         * @param depositorName the account to size up, like mob or clan name
201         * @return the number of items in the account
202         */
203        public int numberDeposited(String depositorName);
204        
205        /**
206         * Returns all of the items deposited in the account. Make sure
207         * you destroy these when you are done looking at them!
208         * @see Banker#numberDeposited(String)
209         * @see Banker#findDepositInventory(String, String)
210         * @see Banker#totalItemsWorth(String)
211         * @param depositorName the account to return, like mob or clan name
212         * @return the list of all items in the account
213         */
214        public List<Item> getDepositedItems(String depositorName);
215        
216        /**
217         * Searches the deposit inventory for an item with a substring name
218         * like the one given, returning the first found.  If the search
219         * string resembles a number at all, this will return the Coins
220         * item in the account.
221         * @see Banker#getDepositedItems(String)
222         * @see Banker#numberDeposited(String)
223         * @see Banker#totalItemsWorth(String)
224         * @param mob the player or clan name of the account to search
225         * @param likeThis the search string
226         * @return the item found, or null.
227         */
228        public Item findDepositInventory(String mob, String likeThis);
229        
230        /**
231         * Returns the base money value of all items deposited in the given account.
232         * @see Banker#getDepositedItems(String)
233         * @see Banker#numberDeposited(String)
234         * @see Banker#findDepositInventory(String, String)
235         * @param depositorName the account to account for, like mob or clan name
236         * @return the base money value of all items deposited
237         */
238        public double totalItemsWorth(String depositorName);
239}
