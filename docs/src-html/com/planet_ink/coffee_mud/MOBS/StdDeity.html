208        public String getTriggerDesc(List<DeityTrigger> V)
209        {
210                if((V==null)||(V.size()==0))
211                        return L("Never");
212                final StringBuffer buf=new StringBuffer("");
213                for(int v=0;v<V.size();v++)
214                {
215                        final DeityTrigger DT=V.get(v);
216                        if(v>0)
217                                buf.append(", "+((DT.previousConnect==RitualConnector.AND)?L("and "):L("or ")));
218                        switch(DT.triggerCode)
219                        {
220                        case SAY:
221                                buf.append(L("the player should say '@x1'",DT.parm1.toLowerCase()));
222                                break;
223                        case READING:
224                                if(DT.parm1.equals("0"))
225                                        buf.append(L("the player should read something"));
226                                else
227                                        buf.append(L("the player should read '@x1'",DT.parm1.toLowerCase()));
228                                break;
229                        case TIME:
230                                buf.append(L("the hour of the day is @x1",DT.parm1.toLowerCase()));
231                                break;
232                        case PUTTHING:
233                                buf.append(L("the player should put @x1 in @x2",DT.parm1.toLowerCase(),DT.parm2.toLowerCase()));
234                                break;
235                        case BURNTHING:
236                                buf.append(L("the player should burn @x1",DT.parm1.toLowerCase()));
237                                break;
238                        case DRINK:
239                                buf.append(L("the player should drink @x1",DT.parm1.toLowerCase()));
240                                break;
241                        case EAT:
242                                buf.append(L("the player should eat @x1",DT.parm1.toLowerCase()));
243                                break;
244                        case INROOM:
245                                {
246                                if(DT.parm1.equalsIgnoreCase("holy")
247                                ||DT.parm1.equalsIgnoreCase("unholy")
248                                ||DT.parm1.equalsIgnoreCase("balance"))
249                                        buf.append(L("the player should be in the deities room of infused @x1-ness.",DT.parm1.toLowerCase()));
250                                else
251                                {
252                                        final Room R=CMLib.map().getRoom(DT.parm1);
253                                        if(R==null)
254                                                buf.append(L("the player should be in some unknown place"));
255                                        else
256                                                buf.append(L("the player should be in '@x1'",R.displayText(null)));
257                                }
258                                }
259                                break;
260                        case RIDING:
261                                buf.append(L("the player should be on @x1",DT.parm1.toLowerCase()));
262                                break;
263                        case CAST:
264                                {
265                                final Ability A=CMClass.findAbility(DT.parm1);
266                                if(A==null)
267                                        buf.append(L("the player should cast '@x1'",DT.parm1));
268                                else
269                                        buf.append(L("the player should cast '@x1'",A.name()));
270                                }
271                                break;
272                        case EMOTE:
273                                buf.append(L("the player should emote '@x1'",DT.parm1.toLowerCase()));
274                                break;
275                        case RANDOM:
276                                buf.append(DT.parm1+"% of the time");
277                                break;
278                        case WAIT:
279                                buf.append(L("wait @x1 seconds",""+((CMath.s_int(DT.parm1)*CMProps.getTickMillis())/1000)));
280                                break;
281                        case YOUSAY:
282                                buf.append(L("then you will automatically say '@x1'",DT.parm1.toLowerCase()));
283                                break;
284                        case OTHERSAY:
285                                buf.append(L("then all others will say '@x1'",DT.parm1.toLowerCase()));
286                                break;
287                        case ALLSAY:
288                                buf.append(L("then all will say '@x1'",DT.parm1.toLowerCase()));
289                                break;
290                        case CHECK:
291                                buf.append(CMLib.masking().maskDesc(DT.parm1));
292                                break;
293                        case PUTVALUE:
294                                buf.append(L("the player should put an item worth at least @x1 in @x2",DT.parm1.toLowerCase(),DT.parm2.toLowerCase()));
295                                break;
296                        case PUTMATERIAL:
297                                {
298                                        String material="something";
299                                        final int t=CMath.s_int(DT.parm1);
300                                        RawMaterial.Material m;
301                                        if(((t&RawMaterial.RESOURCE_MASK)==0)
302                                        &&((m=RawMaterial.Material.findByMask(t))!=null))
303                                                material=m.desc().toLowerCase();
304                                        else
305                                        if(RawMaterial.CODES.IS_VALID(t))
306                                                material=RawMaterial.CODES.NAME(t).toLowerCase();
307                                        buf.append(L("the player puts an item made of @x1 in @x2",material,DT.parm2.toLowerCase()));
308                                }
309                                break;
310                        case BURNMATERIAL:
311                                {
312                                        String material="something";
313                                        final int t=CMath.s_int(DT.parm1);
314                                        RawMaterial.Material m;
315                                        if(((t&RawMaterial.RESOURCE_MASK)==0)
316                                        &&((m=RawMaterial.Material.findByMask(t))!=null))
317                                                material=m.desc().toLowerCase();
318                                        else
319                                        if(RawMaterial.CODES.IS_VALID(t))
320                                                material=RawMaterial.CODES.NAME(t).toLowerCase();
321                                        buf.append(L("the player should burn an item made of @x1",material));
322                                }
323                                break;
324                        case BURNVALUE:
325                                buf.append(L("the player should burn an item worth at least @x1",DT.parm1.toLowerCase()));
326                                break;
327                        case SITTING:
328                                buf.append(L("the player should sit down"));
329                                break;
330                        case STANDING:
331                                buf.append(L("the player should stand up"));
332                                break;
333                        case SLEEPING:
334                                buf.append(L("the player should go to sleep"));
335                                break;
336                        }
337                }
338                return buf.toString();
339        }

375        @Override
376        public boolean okMessage(final Environmental myHost, final CMMsg msg)
377        {
378                if(!super.okMessage(myHost,msg))
379                        return false;
380                if((msg.targetMinor()==CMMsg.TYP_EXPIRE)
381                &&(msg.target()==location())
382                &&(CMLib.flags().isInTheGame(this,true)))
383                        return false;
384                else
385                if(msg.amITarget(this))
386                switch(msg.targetMinor())
387                {
388                case CMMsg.TYP_SERVE:
389                        if(msg.source().getMyDeity()==this)
390                        {
391                                msg.source().tell(L("You already worship @x1.",name()));
392                                if(msg.source().isMonster())
393                                        CMLib.commands().postSay(msg.source(),null,L("I already worship @x1.",msg.source().getMyDeity().name()));
394                                return false;
395                        }
396                        if(msg.source().getMyDeity()!=null)
397                        {
398                                msg.source().tell(L("You already worship @x1.",msg.source().getMyDeity().name()));
399                                if(msg.source().isMonster())
400                                        CMLib.commands().postSay(msg.source(),null,L("I already worship @x1.",msg.source().getMyDeity().name()));
401                                return false;
402                        }
403                        if(msg.source().charStats().getCurrentClass().baseClass().equalsIgnoreCase("Cleric"))
404                        {
405                                if(!CMLib.masking().maskCheck(getClericRequirements(),msg.source(),true))
406                                {
407                                        msg.source().tell(L("You are unworthy of serving @x1.",name()));
408                                        if(msg.source().isMonster())
409                                                CMLib.commands().postSay(msg.source(),null,L("I am unworthy of serving @x1.",name()));
410                                        return false;
411                                }
412                        }
413                        else
414                        if(!CMLib.masking().maskCheck(getWorshipRequirements(),msg.source(),true))
415                        {
416                                msg.source().tell(L("You are unworthy of @x1.",name()));
417                                if(msg.source().isMonster())
418                                        CMLib.commands().postSay(msg.source(),null,L("I am unworthy of @x1.",name()));
419                                return false;
420                        }
421                        break;
422                case CMMsg.TYP_REBUKE:
423                        if(!msg.source().getWorshipCharID().equals(Name()))
424                        {
425                                msg.source().tell(L("You do not worship @x1.",name()));
426                                return false;
427                        }
428                        break;
429                }
430                return true;
431        }

433        public synchronized void bestowBlessing(MOB mob, Ability Blessing)
434        {
435                final Room prevRoom=location();
436                mob.location().bringMobHere(this,false);
437                if(Blessing!=null)
438                {
439                        final Vector<String> V=new Vector<String>();
440                        if(Blessing.canTarget(Ability.CAN_MOBS))
441                        {
442                                V.addElement(mob.name()+"$");
443                                Blessing.invoke(this,V,mob,true,mob.phyStats().level());
444                        }
445                        else
446                        if(Blessing.canTarget(Ability.CAN_ITEMS))
447                        {
448                                Item I=mob.fetchWieldedItem();
449                                if(I==null)
450                                        I=mob.fetchHeldItem();
451                                if(I==null)
452                                        I=mob.fetchItem(null,Wearable.FILTER_WORNONLY,"all");
453                                if(I==null)
454                                        I=mob.fetchItem(null,Wearable.FILTER_UNWORNONLY,"all");
455                                if(I==null)
456                                        return;
457                                V.addElement("$"+I.name()+"$");
458                                addItem(I);
459                                Blessing.invoke(this,V,I,true,mob.phyStats().level());
460                                delItem(I);
461                                if(!mob.isMine(I))
462                                        mob.addItem(I);
463                        }
464                        else
465                                Blessing.invoke(this,mob,true,mob.phyStats().level());
466                }
467                prevRoom.bringMobHere(this,false);
468                if(mob.location()!=prevRoom)
469                {
470                        if(mob.getVictim()==this)
471                                mob.makePeace(true);
472                        if(getVictim()==mob)
473                                makePeace(true);
474                }
475        }

534        {
535                norecurse=true;
536                try
537                {
538                        if((!alreadyBlessed(mob))&&(numBlessings()>0))
539                        {
540                                mob.location().show(this,mob,CMMsg.MSG_OK_VISUAL,L("You feel the presence of <S-NAME> in <T-NAME>."));
541                                if((mob.charStats().getCurrentClass().baseClass().equals("Cleric"))
542                                ||(CMSecurity.isASysOp(mob)))
543                                {
544                                        for(int b=0;b<numBlessings();b++)
545                                        {
546                                                final Ability Blessing=fetchBlessing(b);
547                                                if(Blessing!=null)
548                                                        bestowBlessing(mob,Blessing);
549                                        }
550                                }
551                                else
552                                {
553                                        final int randNum=CMLib.dice().roll(1,numBlessings(),-1);
554                                        final Ability Blessing=fetchBlessing(randNum);
555                                        if((Blessing!=null)&&(!fetchBlessingCleric(randNum)))
556                                                bestowBlessing(mob,Blessing);
557                                }
558                        }
559                }
560                catch(final Exception e)
561                {
562                        Log.errOut("StdDeity",e);
563                }
564                norecurse=false;
565        }
