216        @Override
217        public boolean delFromBox(String boxName, Item thisThang)
218        {
219                final List<PlayerData> V=getBoxRowPDData(boxName);
220                boolean found=false;
221                for(int v=V.size()-1;v>=0;v--)
222                {
223                        final DatabaseEngine.PlayerData PD=V.get(v);
224                        if((PD!=null)
225                        &&(PD.key().startsWith(postalBranch()+";")))
226                        {
227                                final Item I=makeItem(parsePostalItemData(PD.xml()));
228                                if(I==null)
229                                        continue;
230                                if(thisThang.sameAs(I))
231                                {
232                                        found=true;
233                                        CMLib.database().DBDeletePlayerData(PD.who(),PD.section(),PD.key());
234                                        break;
235                                }
236                        }
237                }
238                return found;
239        }

357        @Override
358        public MailPiece parsePostalItemData(String data)
359        {
360                final MailPiece piece = new MailPiece();
361                for(int i=0;i<5;i++)
362                {
363                        final int x=data.indexOf(';');
364                        if(x<0)
365                        {
366                                Log.errOut("StdPostman","Man formed postal data: "+data);
367                                return null;
368                        }
369                        switch(i)
370                        {
371                        case 0:
372                                piece.from = data.substring(0, x);
373                                break;
374                        case 1:
375                                piece.to = data.substring(0, x);
376                                break;
377                        case 2:
378                                piece.time = data.substring(0, x);
379                                break;
380                        case 3:
381                                piece.cod = data.substring(0, x);
382                                break;
383                        case 4:
384                                piece.classID = data.substring(0, x);
385                                break;
386                        }
387                        data=data.substring(x+1);
388                }
389                piece.xml = data;
390                return piece;
391        }

425        protected double getHoldingCost(MailPiece data, int chargeableWeight)
426        {
427                if(data== null)
428                        return 0.0;
429                if(getStartRoom()==null)
430                        return 0.0;
431                double amt=0.0;
432                final TimeClock TC=CMLib.time().localClock(getStartRoom());
433                final long time=System.currentTimeMillis()-CMath.s_long(data.time);
434                final long millisPerMudMonth=TC.getDaysInMonth()*CMProps.getMillisPerMudHour()*TC.getHoursInDay();
435                if(time<=0)
436                        return amt;
437                amt+=CMath.mul(CMath.mul(Math.floor(CMath.div(time,millisPerMudMonth)),holdFeePerPound()),chargeableWeight);
438                return amt;
439        }

441        protected double getCODChargeForPiece(MailPiece data)
442        {
443                if(data==null)
444                        return 0.0;
445                final int chargeableWeight=getChargeableWeight(makeItem(data));
446                final double COD=CMath.s_double(data.cod);
447                double amt=0.0;
448                if(COD>0.0)
449                        amt=getSimplePostage(chargeableWeight)+COD;
450                return amt+getHoldingCost(data,chargeableWeight);
451        }

453        protected String getBranchPostableTo(String toWhom, String branch, Map<String, String> allBranchBoxes)
454        {
455                String forward=allBranchBoxes.get(branch);
456                if(forward==null)
457                        return null;
458                if(forward.equalsIgnoreCase(toWhom))
459                        return branch;
460                final PostOffice P=CMLib.map().getPostOffice(postalChain(),forward);
461                if(P!=null)
462                {
463                        forward=allBranchBoxes.get(P.postalBranch());
464                        if((forward!=null)&&forward.equalsIgnoreCase(toWhom))
465                                return P.postalBranch();
466                }
467                return null;
468        }

470        @Override
471        public String findProperBranch(String toWhom)
472        {
473                final MOB M=CMLib.players().getLoadPlayer(toWhom);
474                if(M!=null)
475                {
476                        if(M.getStartRoom()!=null)
477                        {
478                                final Map<String,String> allBranchBoxes=getOurOpenBoxes(toWhom);
479                                final PostOffice P=CMLib.map().getPostOffice(postalChain(),M.getStartRoom().getArea().Name());
480                                String branch=null;
481                                if(P!=null)
482                                {
483                                        branch=getBranchPostableTo(toWhom,P.postalBranch(),allBranchBoxes);
484                                        if(branch!=null)
485                                                return branch;
486                                        if(allBranchBoxes.size()==0)
487                                        {
488                                                P.createBoxHere(toWhom,toWhom);
489                                                return P.postalBranch();
490                                        }
491                                }
492                                branch=getBranchPostableTo(toWhom,postalBranch(),allBranchBoxes);
493                                if(branch!=null)
494                                        return branch;
495                                for(final String tryBranch : allBranchBoxes.keySet())
496                                {
497                                        branch=getBranchPostableTo(toWhom,tryBranch,allBranchBoxes);
498                                        if(branch!=null)
499                                                return branch;
500                                }
501                                if(P!=null)
502                                {
503                                        P.deleteBoxHere(toWhom);
504                                        P.createBoxHere(toWhom,toWhom);
505                                        return P.postalBranch();
506                                }
507                        }
508                }
509                else
510                if(CMLib.clans().getClanAnyHost(toWhom)!=null)
511                {
512                        final Map<String,String> allBranchBoxes=getOurOpenBoxes(toWhom);
513                        String branch=getBranchPostableTo(toWhom,postalBranch(),allBranchBoxes);
514                        if(branch!=null)
515                                return branch;
516                        for(final String tryBranch : allBranchBoxes.keySet())
517                        {
518                                branch=getBranchPostableTo(toWhom,tryBranch,allBranchBoxes);
519                                if(branch!=null)
520                                        return branch;
521                        }
522                }
523                return null;
524        }

526        public long postalWaitTime()
527        {
528                if((postalWaitTime<0)&&(getStartRoom()!=null))
529                        postalWaitTime=10038;
530                else

536        @Override
537        public boolean tick(final Tickable ticking, final int tickID)
538        {
539                if(!super.tick(ticking,tickID))
540                        return false;
541                if(!CMProps.getBoolVar(CMProps.Bool.MUDSTARTED))
542                        return true;
543
544                if((tickID==Tickable.TICKID_MOB)&&(getStartRoom()!=null))
545                {
546                        boolean proceed=false;
547                        // handle interest by watching the days go by...
548                        // each chain is handled by one branch,
549                        // since pending mail all looks the same.
550                        Long L=postalTimes.get(postalChain()+"/"+postalBranch());
551                        if((L==null)||(L.longValue()<System.currentTimeMillis()))
552                        {
553                                proceed=(L!=null);
554                                L=Long.valueOf(System.currentTimeMillis()+postalWaitTime());
555                                postalTimes.remove(postalChain()+"/"+postalBranch());
556                                postalTimes.put(postalChain()+"/"+postalBranch(),L);
557                        }
558                        if(proceed)
559                        {
560                                List<PlayerData> V=getBoxRowPDData(postalChain());
561                                // first parse all the pending mail,
562                                // and remove it from the sorter
563                                final Vector<MailPiece> parsed=new Vector<MailPiece>();
564                                if(V==null)
565                                        V=new Vector<PlayerData>();
566                                for(int v=0;v<V.size();v++)
567                                {
568                                        final DatabaseEngine.PlayerData PD=V.get(v);
569                                        parsed.addElement(parsePostalItemData(PD.xml()));
570                                        CMLib.database().DBDeletePlayerData(PD.who(),PD.section(),PD.key());
571                                }
572                                PostOffice P=null;
573                                for(int v=0;v<parsed.size();v++)
574                                {
575                                        final MailPiece V2=parsed.elementAt(v);
576                                        final String toWhom=V2.to;
577                                        String deliveryBranch=findProperBranch(toWhom);
578                                        if(deliveryBranch!=null)
579                                        {
580                                                P=CMLib.map().getPostOffice(postalChain(),deliveryBranch);
581                                                final Item I=makeItem(V2);
582                                                if((P!=null)&&(I!=null))
583                                                {
584                                                        P.addToBox(toWhom,I,V2.from,V2.to,CMath.s_long(V2.time),CMath.s_double(V2.cod));
585                                                        continue;
586                                                }
587                                        }
588                                        final String fromWhom=V2.from;
589                                        deliveryBranch=findProperBranch(fromWhom);
590                                        if(deliveryBranch!=null)
591                                        {
592                                                P=CMLib.map().getPostOffice(postalChain(),deliveryBranch);
593                                                final Item I=makeItem(V2);
594                                                if((P!=null)&&(I!=null))
595                                                        P.addToBox(fromWhom,I,V2.to,"POSTMASTER",System.currentTimeMillis(),0.0);
596                                        }
597                                }
598                                V=CMLib.database().DBReadPlayerSectionData(postalChain());
599                                TimeClock TC=null;
600                                if(getStartRoom()!=null)
601                                        TC=getStartRoom().getArea().getTimeObj();
602                                if((TC!=null)&&(maxMudMonthsHeld()>0))
603                                {
604                                        for(int v=0;v<V.size();v++)
605                                        {
606                                                final DatabaseEngine.PlayerData V2=V.get(v);
607                                                if(V2.key().startsWith(postalBranch()+";"))
608                                                {
609                                                        final MailPiece data=parsePostalItemData(V2.xml());
610                                                        if((data!=null)&&(getStartRoom()!=null))
611                                                        {
612                                                                final long time=System.currentTimeMillis()-CMath.s_long(data.time);
613                                                                final long millisPerMudMonth=TC.getDaysInMonth()*CMProps.getMillisPerMudHour()*TC.getHoursInDay();
614                                                                if(time>0)
615                                                                {
616                                                                        final int months=(int)Math.round(Math.floor(CMath.div(time,millisPerMudMonth)));
617                                                                        if(months>maxMudMonthsHeld())
618                                                                        {
619                                                                                final Item I=makeItem(data);
620                                                                                CMLib.database().DBDeletePlayerData(V2.who(),V2.section(),V2.key());
621                                                                                if(I!=null)
622                                                                                        getShop().addStoreInventory(I);
623                                                                        }
624                                                                }
625                                                        }
626                                                }
627                                        }
628                                }
629                        }
630                }
631                return true;
632        }

642        @Override
643        public void executeMsg(final Environmental myHost, final CMMsg msg)
644        {
645                final MOB mob=msg.source();
646                if(msg.amITarget(this))
647                {
648                        switch(msg.targetMinor())
649                        {
650                        case CMMsg.TYP_GIVE:
651                        case CMMsg.TYP_DEPOSIT:
652                                if(CMLib.flags().isAliveAwakeMobileUnbound(mob,true))
653                                {
654                                        if(msg.tool() instanceof Container)
655                                                ((Container)msg.tool()).emptyPlease(true);
656                                        final Session S=msg.source().session();
657                                        if((!msg.source().isMonster())&&(S!=null)&&(msg.tool() instanceof Item))
658                                        {
659                                                autoGive(msg.source(),this,(Item)msg.tool());
660                                                if(isMine(msg.tool()))
661                                                {
662                                                        final StdPostman me=this;
663                                                        S.prompt(new InputCallback(InputCallback.Type.PROMPT,"",0)
664                                                        {
665                                                                @Override
666                                                                public void showPrompt()
667                                                                {
668                                                                        S.promptPrint(L("Address this to whom? "));
669                                                                }
670
671                                                                @Override
672                                                                public void timedOut()
673                                                                {
674                                                                        autoGive(me,msg.source(),(Item)msg.tool());
675                                                                }
676
677                                                                @Override 
678                                                                public void callBack()
