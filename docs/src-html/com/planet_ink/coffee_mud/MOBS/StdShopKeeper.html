082        @Override
083        public boolean isSold(int mask)
084        {
085                if (mask == 0)
086                        return whatIsSoldMask == 0;
087                if ((whatIsSoldMask & 255) == mask)
088                        return true;
089                return CMath.bset(whatIsSoldMask >> 8, CMath.pow(2, mask - 1));
090        }
091
092        @Override
093        public void addSoldType(int mask)
094        {
095                if (mask == 0)
096                        whatIsSoldMask = 0;
097                else
098                {
099                        if ((whatIsSoldMask > 0) && (whatIsSoldMask < 256))
100                                whatIsSoldMask = (CMath.pow(2, whatIsSoldMask - 1) << 8);
101
102                        for (int c = 0; c < ShopKeeper.DEAL_CONFLICTS.length; c++)
103                        {
104                                for (int c1 = 0; c1 < ShopKeeper.DEAL_CONFLICTS[c].length; c1++)
105                                {
106                                        if (ShopKeeper.DEAL_CONFLICTS[c][c1] == mask)
107                                        {
108                                                for (c1 = 0; c1 < ShopKeeper.DEAL_CONFLICTS[c].length; c1++)
109                                                {
110                                                        if ((ShopKeeper.DEAL_CONFLICTS[c][c1] != mask) 
111                                                        && (isSold(ShopKeeper.DEAL_CONFLICTS[c][c1])))
112                                                                addSoldType(-ShopKeeper.DEAL_CONFLICTS[c][c1]);
113                                                }
114                                                break;
115                                        }
116                                }
117                        }
118
119                        if (mask < 0)
120                                whatIsSoldMask = CMath.unsetb(whatIsSoldMask, (CMath.pow(2, (-mask) - 1) << 8));
121                        else
122                                whatIsSoldMask |= (CMath.pow(2, mask - 1) << 8);
123                }
124        }

181        protected void doInventoryReset()
182        {
183                invResetTickDown = finalInvResetRate(); // we should now be at a
184                // positive number.
185                if (invResetTickDown <= 0)
186                        invResetTickDown = Ability.TICKS_FOREVER;
187                else
188                {
189                        final CoffeeShop shop=(this instanceof Librarian)?((Librarian)this).getBaseLibrary():this.getShop();
190                        shop.emptyAllShelves();
191                        if (miscText != null)
192                        {
193                                String shoptext;
194                                if (CMProps.getBoolVar(CMProps.Bool.MOBCOMPRESS) && (miscText instanceof byte[]))
195                                        shoptext = CMLib.coffeeMaker().getGenMOBTextUnpacked(this, CMLib.encoder().decompressString((byte[]) miscText));
196                                else
197                                        shoptext = CMLib.coffeeMaker().getGenMOBTextUnpacked(this, CMStrings.bytesToStr(miscText));
198                                final List<XMLLibrary.XMLTag> xml = CMLib.xml().parseAllXML(shoptext);
199                                if (xml != null)
200                                {
201                                        CMLib.coffeeMaker().populateShops(this, xml);
202                                        recoverPhyStats();
203                                        recoverCharStats();
204                                }
205                        }
206                }
207        }

209        @Override
210        public boolean tick(final Tickable ticking, final int tickID)
211        {
212                if (!super.tick(ticking, tickID))
213                        return false;
214                if ((tickID == Tickable.TICKID_MOB) && (isGeneric()))
215                {
216                        if ((--invResetTickDown) <= 0)
217                        {
218                                doInventoryReset();
219                        }
220                        if ((--budgetTickDown) <= 0)
221                        {
222                                budgetTickDown = 100;
223                                budgetRemaining = Long.MAX_VALUE / 2;
224                                Pair<Long,TimeClock.TimePeriod> budget = finalBudget();
225                                if(budget != null)
226                                {
227                                        budgetRemaining = budget.first.longValue();
228                                        budgetTickDown = 100;
229                                        
230                                        final Room R=location();
231                                        final TimeClock C=((R != null) && (R.getArea() != null)) ? R.getArea().getTimeObj() : null;
232                                        final int ticksPerDay = CMProps.getIntVar(CMProps.Int.TICKSPERMUDDAY);
233                                        int hoursInDay = 1;
234                                        int daysInMonth = 1;
235                                        int monthsInYear = 1;
236                                        int daysInWeek = 1;
237                                        if(C!=null)
238                                        {
239                                                if(C.getHoursInDay()>0)
240                                                        hoursInDay=C.getHoursInDay();
241                                                if(C.getDaysInMonth()>0)
242                                                        daysInMonth=C.getDaysInMonth();
243                                                if(C.getMonthsInYear()>0)
244                                                        monthsInYear=C.getMonthsInYear();
245                                                if(C.getDaysInWeek()>0)
246                                                        daysInWeek=C.getDaysInWeek();
247                                        }
248                                        switch(budget.second)
249                                        {
250                                        case ALLTIME:
251                                                budgetTickDown = 100;
252                                                break;
253                                        case DAY:
254                                                budgetTickDown = ticksPerDay;
255                                                break;
256                                        case HOUR:
257                                                budgetTickDown = ticksPerDay / hoursInDay;
258                                                break;
259                                        case MONTH:
260                                                budgetTickDown = ticksPerDay * daysInMonth;
261                                                break;
262                                        case SEASON:
263                                                budgetTickDown = ticksPerDay * (daysInMonth * monthsInYear / 4);
264                                                break;
265                                        case WEEK:
266                                                budgetTickDown = ticksPerDay * daysInWeek;
267                                                break;
268                                        case YEAR:
269                                                budgetTickDown = ticksPerDay * (daysInMonth * monthsInYear);
270                                                break;
271                                        }
272                                }
273                                budgetMax = budgetRemaining;
274                        }
275                }
276                return true;
277        }

279        @Override
280        public boolean okMessage(final Environmental myHost, final CMMsg msg)
281        {
282                if (msg.amITarget(this))
283                {
284                        switch (msg.targetMinor())
285                        {
286                        case CMMsg.TYP_VALUE:
287                        case CMMsg.TYP_SELL:
288                        {
289                                if (!CMLib.coffeeShops().ignoreIfNecessary(msg.source(), finalIgnoreMask(), this))
290                                        return false;
291                                if (CMLib.coffeeShops().standardSellEvaluation(this, msg.source(), msg.tool(), this, budgetRemaining, budgetMax, msg.targetMinor() == CMMsg.TYP_SELL))
292                                        return super.okMessage(myHost, msg);
293                                return false;
294                        }
295                        case CMMsg.TYP_BID:
296                        {
297                                CMLib.commands().postSay(this, msg.source(), L("I'm afraid my prices are firm."), false, false);
298                                return false;
299                        }
300                        case CMMsg.TYP_BUY:
301                        case CMMsg.TYP_VIEW:
302                        {
303                                if (!CMLib.coffeeShops().ignoreIfNecessary(msg.source(), finalIgnoreMask(), this))
304                                        return false;
305                                if ((msg.targetMinor() == CMMsg.TYP_BUY) 
306                                && (msg.tool() != null) 
307                                && (!msg.tool().okMessage(myHost, msg)))
308                                        return false;
309                                if (CMLib.coffeeShops().standardBuyEvaluation(this, msg.source(), msg.tool(), this, msg.targetMinor() == CMMsg.TYP_BUY))
310                                        return super.okMessage(myHost, msg);
311                                return false;
312                        }
313                        case CMMsg.TYP_LIST:
314                        {
315                                if (!CMLib.coffeeShops().ignoreIfNecessary(msg.source(), finalIgnoreMask(), this))
316                                        return false;
317                                return super.okMessage(myHost, msg);
318                        }
319                        default:
320                                break;
321                        }
322                }
323                return super.okMessage(myHost, msg);
324        }

326        @Override
327        public void executeMsg(final Environmental myHost, final CMMsg msg)
328        {
329                if (msg.amITarget(this))
330                {
331                        final MOB mob = msg.source();
332                        switch (msg.targetMinor())
333                        {
334                        case CMMsg.TYP_GIVE:
335                                if (CMLib.flags().isAliveAwakeMobileUnbound(mob, true))
336                                {
337                                        if ((msg.tool() != null)
338                                        && ((CMSecurity.isAllowed(msg.source(), location(), CMSecurity.SecFlag.ORDER) 
339                                                || (CMLib.law().doesHavePriviledgesHere(msg.source(), getStartRoom()))
340                                                || (CMSecurity.isAllowed(msg.source(), location(), CMSecurity.SecFlag.CMDMOBS) && (isMonster())) 
341                                                || (CMSecurity.isAllowed(msg.source(), location(), CMSecurity.SecFlag.CMDROOMS) && (isMonster()))))
342                                        && ((doISellThis(msg.tool())) || (isSold(DEAL_INVENTORYONLY))))
343                                        {
344                                                CMLib.commands().postSay(this, msg.source(), L("Yes, I will now sell @x1.", msg.tool().name()), false, false);
345                                                getShop().addStoreInventory(msg.tool(), 1, -1);
346                                                if (isGeneric())
347                                                        text();
348                                                return;
349                                        }
350                                }
351                                super.executeMsg(myHost, msg);
352                                break;
353                        case CMMsg.TYP_VALUE:
354                        {
355                                super.executeMsg(myHost, msg);
356                                if (CMLib.flags().isAliveAwakeMobileUnbound(mob, true))
357                                {
358                                        final double pawningPrice = CMLib.coffeeShops().pawningPrice(this, mob, msg.tool(), this, getShop()).absoluteGoldPrice;
359                                        final String currencyShort = CMLib.beanCounter().nameCurrencyShort(this, pawningPrice);
360                                        CMLib.commands().postSay(this, mob, L("I'll give you @x1 for @x2.", currencyShort, msg.tool().name()), true, false);
361                                }
362                                break;
363                        }
364                        case CMMsg.TYP_SELL: // sell TO -- this is a shopkeeper purchasing from a player
365                        {
366                                super.executeMsg(myHost, msg);
367                                if (CMLib.flags().isAliveAwakeMobileUnbound(mob, true))
368                                {
369                                        final double paid = CMLib.coffeeShops().transactPawn(this, msg.source(), this, msg.tool());
370                                        if (paid > Double.MIN_VALUE)
371                                        {
372                                                budgetRemaining = budgetRemaining - Math.round(paid);
373                                                if (mySession != null)
374                                                        mySession.stdPrintln(msg.source(), msg.target(), msg.tool(), msg.targetMessage());
375                                                if (!CMath.bset(msg.targetMajor(), CMMsg.MASK_OPTIMIZE))
376                                                        mob.location().recoverRoomStats();
377                                                if (isGeneric())
378                                                        text();
379                                        }
380                                }
381                                break;
382                        }
383                        case CMMsg.TYP_VIEW:
384                        {
385                                super.executeMsg(myHost, msg);
386                                if (CMLib.flags().isAliveAwakeMobileUnbound(mob, true))
387                                {
388                                        if ((msg.tool() != null) 
389                                        && (getShop().doIHaveThisInStock("$" + msg.tool().Name() + "$", mob)))
390                                        {
391                                                final String prefix = L("Interested in @x1? Here is some information for you: ",msg.tool().Name());
392                                                final String viewDesc = prefix + CMLib.coffeeShops().getViewDescription(msg.source(), msg.tool());
393                                                CMLib.commands().postSay(this, msg.source(), viewDesc, true, false);
394                                        }
395                                }
396                                break;
397                        }
398                        case CMMsg.TYP_BUY: // buy-from -- this is a player buying from a
399                                                                // shopkeeper
400                        {
401                                super.executeMsg(myHost, msg);
402                                if (CMLib.flags().isAliveAwakeMobileUnbound(mob, true))
403                                {
404                                        final MOB mobFor = CMLib.coffeeShops().parseBuyingFor(msg.source(), msg.targetMessage());
405                                        if ((msg.tool() != null) && (getShop().doIHaveThisInStock("$" + msg.tool().Name() + "$", mobFor)) 
406                                        && (location() != null))
407                                        {
408                                                final Environmental item = getShop().getStock("$" + msg.tool().Name() + "$", mobFor);
409                                                if (item != null)
410                                                        CMLib.coffeeShops().transactMoneyOnly(this, msg.source(), this, item, !isMonster());
411
412                                                final List<Environmental> products = getShop().removeSellableProduct("$" + msg.tool().Name() + "$", mobFor);
413                                                if (products.size() == 0)
414                                                        break;
415                                                final Environmental product = products.get(0);
416
417                                                if (product instanceof Item)
418                                                {
419                                                        if (!CMLib.coffeeShops().purchaseItems((Item) product, products, this, mobFor))
420                                                                return;
421                                                }
422                                                else 
423                                                if (product instanceof MOB)
424                                                {
425                                                        if (CMLib.coffeeShops().purchaseMOB((MOB) product, this, this, mobFor))
426                                                        {
427                                                                msg.modify(msg.source(), msg.target(), product, msg.sourceCode(), msg.sourceMessage(), msg.targetCode(), msg.targetMessage(), msg.othersCode(), msg.othersMessage());
428                                                                product.executeMsg(myHost, msg);
429                                                        }
430                                                }
431                                                else 
432                                                if (product instanceof Ability)
433                                                        CMLib.coffeeShops().purchaseAbility((Ability) product, this, this, mobFor);
434
435                                                if (mySession != null)
436                                                        mySession.stdPrintln(msg.source(), msg.target(), msg.tool(), msg.targetMessage());
437                                                if (!CMath.bset(msg.targetMajor(), CMMsg.MASK_OPTIMIZE))
438                                                        mob.location().recoverRoomStats();
439                                        }
440                                }
441                                break;
442                        }
443                        case CMMsg.TYP_LIST:
444                        {
445                                super.executeMsg(myHost, msg);
446                                if (CMLib.flags().isAliveAwakeMobileUnbound(mob, true))
447                                {
448                                        final String forMask = CMLib.coffeeShops().getListForMask(msg.targetMessage());
449                                        List<Environmental> inventory = new XVector<Environmental>(getShop().getStoreInventory());
450                                        inventory = CMLib.coffeeShops().addRealEstateTitles(inventory, mob, getShop(), getStartRoom());
451                                        final int limit = CMParms.getParmInt(finalPrejudiceFactors(), "LIMIT", 0);
452                                        final String s = CMLib.coffeeShops().getListInventory(this, mob, inventory, limit, this, forMask);
453                                        if (s.length() > 0)
454                                                mob.tell(s);
455                                }
456                                break;
457                        }
458                        default:
459                                super.executeMsg(myHost, msg);
460                                break;
461                        }
462                }
463                else
464                        super.executeMsg(myHost, msg);
465        }
