180        @Override
181        public boolean okMessage(final Environmental myHost, final CMMsg msg)
182        {
183                final boolean retval = super.okMessage(myHost,msg);
184
185                if((msg.amITarget(this))
186                &&(CMath.bset(msg.targetMajor(),CMMsg.MASK_MALICIOUS))
187                &&(msg.targetMinor()==CMMsg.TYP_CAST_SPELL))
188                {
189                        if(CMLib.dice().rollPercentage() <= magicResistance)
190                        {
191                                msg.source().tell(L("The drow priestess resisted your spell!"));
192                                return false;
193                        }
194                }
195                return retval;
196        }
197
198        @Override
199        public boolean tick(final Tickable ticking, final int tickID)
200        {
201                if((!amDead())&&(tickID==Tickable.TICKID_MOB))
202                {
203                        if (isInCombat())
204                        {
205                                if((--spellDown)<=0)
206                                {
207                                        spellDown=3;
208                                        castSpell();
209                                }
210                                if((--darkDown)<=0)
211                                {
212                                        darkDown=4;
213                                        castDarkness();
214                                }
215                        }
216
217                }
218                return super.tick(ticking,tickID);
219        }
220
221        public boolean castSpell()
222        {
223                Ability prayer = null;
224                int tries = 10;
225                if(CMLib.dice().rollPercentage() < 70)
226                {
227                        prayer = fetchRandomAbility();
228                        while(((--tries)>0)&&((prayer==null)||(this.basePhyStats().level() < CMLib.ableMapper().lowestQualifyingLevel(prayer.ID()))))
229                                prayer = fetchRandomAbility();
230                }
231                else
232                {
233                        prayer = CMClass.getAbility("Prayer_CureSerious");
234                        prayer.setProficiency(CMLib.dice().roll(5, 10, 50));
235                }
236
237                if(prayer!=null)
238                        return prayer.invoke(this,null,false,0);
239                return false;
240
241        }
242
243        @Override
244        protected boolean castDarkness()
245        {
246                if(this.location()==null)
247                        return true;
248                if(CMLib.flags().isInDark(this.location()))
249                        return true;
250
251                Ability dark=CMClass.getAbility("Spell_Darkness");
252                dark.setProficiency(100);
253                if(this.fetchAbility(dark.ID())==null)
254                        this.addAbility(dark);
255                else
256                        dark =this.fetchAbility(dark.ID());
257
258                if(dark!=null)
259                        dark.invoke(this,null,false,0);
260                return true;
261        }
