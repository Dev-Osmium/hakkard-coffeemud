097        @Override
098        public void executeMsg(final Environmental myHost, final CMMsg msg)
099        {
100                super.executeMsg(myHost,msg);
101                if(msg.amITarget(this)&&(msg.targetMinor()==CMMsg.TYP_DRINK))
102                {
103                        final MOB mob=msg.source();
104                        final boolean thirsty=mob.curState().getThirst()<=0;
105                        final boolean full=!mob.curState().adjThirst(thirstQuenched(),mob.maxState().maxThirst(mob.baseWeight()));
106                        if(thirsty)
107                                mob.tell(L("You are no longer thirsty."));
108                        else
109                        if(full)
110                                mob.tell(L("You have drunk all you can."));
111                }
112                else
113                if((msg.tool()==this)
114                &&(msg.targetMinor()==CMMsg.TYP_FILL)
115                &&(msg.target() instanceof Container)
116                &&(((Container)msg.target()).capacity()>0))
117                {
118                        final Container container=(Container)msg.target();
119                        final Item I=CMLib.materials().makeItemResource(liquidType());
120                        I.setMaterial(liquidType());
121                        I.setBaseValue(RawMaterial.CODES.VALUE(liquidType()));
122                        I.basePhyStats().setWeight(1);
123                        CMLib.materials().addEffectsToResource(I);
124                        I.recoverPhyStats();
125                        I.setContainer(container);
126                        if(container.owner()!=null)
127                                if(container.owner() instanceof MOB)
128                                        ((MOB)container.owner()).addItem(I);
129                                else
130                                if(container.owner() instanceof Room)
131                                        ((Room)container.owner()).addItem(I,ItemPossessor.Expire.Resource);
132                }
133        }
134
135        @Override
136        public boolean tick(final Tickable ticking, final int tickID)
137        {
138                if(!super.tick(ticking, tickID))
139                        return false;
140                if((liquidRemaining() < liquidHeld())
141                &&(liquidHeld()<Integer.MAX_VALUE/2))
142                {
143                        final Room R=CMLib.map().roomLocation(this);
144                        if(R!=null)
145                        {
146                                final Area A=R.getArea();
147                                if(A!=null)
148                                {
149                                        final TimeClock clock=A.getTimeObj();
150                                        final int ticksPerMudday=(int)((clock.getHoursInDay() * CMProps.getMillisPerMudHour()) / CMProps.getTickMillis());
151                                        double amt=CMath.div(liquidHeld(),ticksPerMudday);
152                                        refill += amt;
153                                        int amtNow=(int)CMath.round(Math.floor(refill));
154                                        if(amtNow > 0)
155                                        {
156                                                setLiquidRemaining(liquidRemaining() + amtNow);
157                                                if(liquidRemaining() > liquidHeld())
158                                                        setLiquidRemaining(liquidHeld());
159                                                refill -= amtNow;
160                                        }
161                                }
162                        }
163                }
164                return true;
165        }

233        private final static String[] MYCODES={"QUENCHED","LIQUIDHELD","LIQUIDTYPE"};
234
235        @Override
236        public String getStat(String code)
237        {
238                if(super.isStat(code))
239                        return super.getStat(code);
240                else
241                switch(getCodeNum(code))
242                {
243                case 0:
244                        return "" + thirstQuenched();
245                case 1:
246                        return "" + liquidHeld();
247                case 2:
248                        return "" + liquidType();
249                default:
250                        return CMProps.getStatCodeExtensionValue(getStatCodes(), xtraValues, code);
251                }
252        }
253
254        @Override
255        public void setStat(String code, String val)
256        {
257                if(super.isStat(code))
258                        super.setStat(code, val);
259                else
260                switch(getCodeNum(code))
261                {
262                case 0:
263                        setThirstQuenched(CMath.s_parseIntExpression(val));
264                        break;
265                case 1:
266                        setLiquidHeld(CMath.s_parseIntExpression(val));
267                        break;
268                case 2:
269                {
270                        int x = CMath.s_parseListIntExpression(RawMaterial.CODES.NAMES(), val);
271                        x = ((x >= 0) && (x < RawMaterial.RESOURCE_MASK)) ? RawMaterial.CODES.GET(x) : x;
272                        setLiquidType(x);
273                        break;
274                }
275                default:
276                        CMProps.setStatCodeExtensionValue(getStatCodes(), xtraValues, code, val);
277                        break;
278                }
279        }
280
281        @Override
282        protected int getCodeNum(final String code)
283        {
284                for(int i=0;i<MYCODES.length;i++)
285                {
286                        if(code.equalsIgnoreCase(MYCODES[i]))
287                                return i;
288                }
289                return -1;
290        }
291
292        private static String[] codes   = null;
293
294        @Override
295        public String[] getStatCodes()
296        {
297                if(codes!=null)
298                        return codes;
299                final String[] MYCODES=CMProps.getStatCodesList(GenCow.MYCODES,this);
300                final String[] superCodes=CMParms.toStringArray(super.getStatCodes());
301                codes=new String[superCodes.length+MYCODES.length];
302                int i=0;
303                for(;i<superCodes.length;i++)
304                        codes[i]=superCodes[i];
305                for(int x=0;x<MYCODES.length;i++,x++)
306                        codes[i]=MYCODES[x];
307                return codes;
308        }
309
310        @Override
311        public boolean sameAs(final Environmental E)
312        {
313                if(!(E instanceof GenCow))
314                        return false;
315                final String[] codes=getStatCodes();
316                for(int i=0;i<codes.length;i++)
317                {
318                        if(!E.getStat(codes[i]).equals(getStat(codes[i])))
319                                return false;
320                }
321                return true;
322        }
