035public class Cow extends StdMOB implements Drink
036{
037        @Override
038        public String ID()
039        {
040                return "Cow";
041        }
042
043        protected int liquidType = RawMaterial.RESOURCE_MILK;
044        
045        public Cow()
046        {
047                super();
048                username="a cow";
049                setDescription("A large lumbering beast that looks too slow to get out of your way.");
050                setDisplayText("A fat happy cow wanders around here.");
051                CMLib.factions().setAlignment(this,Faction.Align.NEUTRAL);
052                setMoney(0);
053                setWimpHitPoint(0);
054
055                basePhyStats().setDamage(1);
056                basePhyStats().setSpeed(1.0);
057                basePhyStats().setAbility(0);
058                basePhyStats().setLevel(2);
059                basePhyStats().setArmor(90);
060                baseCharStats().setStat(CharStats.STAT_GENDER, 'F');
061                baseCharStats().setMyRace(CMClass.getRace("Cow"));
062                baseCharStats().getMyRace().startRacing(this,false);
063
064                baseState.setHitPoints(CMLib.dice().roll(basePhyStats().level(),20,basePhyStats().level()));
065
066                recoverMaxState();
067                resetToMaxState();
068                recoverPhyStats();
069                recoverCharStats();
070        }
071
072        @Override
073        public long decayTime()
074        {
075                return 0;
076        }
077
078        @Override
079        public void setDecayTime(long time)
080        {
081        }
082
083        @Override
084        public boolean okMessage(final Environmental myHost, final CMMsg msg)
085        {
086                if(msg.amITarget(this)&&(msg.targetMinor()==CMMsg.TYP_DRINK))
087                        return true;
088                return super.okMessage(myHost,msg);
089        }
090
091        @Override
092        public void executeMsg(final Environmental myHost, final CMMsg msg)
093        {
094                super.executeMsg(myHost,msg);
095                if(msg.amITarget(this)&&(msg.targetMinor()==CMMsg.TYP_DRINK))
096                {
097                        final MOB mob=msg.source();
098                        final boolean thirsty=mob.curState().getThirst()<=0;
099                        final boolean full=!mob.curState().adjThirst(thirstQuenched(),mob.maxState().maxThirst(mob.baseWeight()));
100                        if(thirsty)
101                                mob.tell(L("You are no longer thirsty."));
102                        else
103                        if(full)
104                                mob.tell(L("You have drunk all you can."));
105                }
106                else
107                if((msg.tool()==this)
108                &&(msg.targetMinor()==CMMsg.TYP_FILL)
109                &&(msg.target() instanceof Container)
110                &&(((Container)msg.target()).capacity()>0))
111                {
112                        final Container container=(Container)msg.target();
113                        final Item I=CMClass.getItem("GenLiquidResource");
114                        I.setName(L("some milk"));
115                        I.setDisplayText(L("some milk has been left here."));
116                        I.setDescription(L("It looks like milk"));
117                        I.setMaterial(RawMaterial.RESOURCE_MILK);
118                        I.setBaseValue(RawMaterial.CODES.VALUE(RawMaterial.RESOURCE_MILK));
119                        I.basePhyStats().setWeight(1);
120                        CMLib.materials().addEffectsToResource(I);
121                        I.recoverPhyStats();
122                        I.setContainer(container);
123                        if(container.owner()!=null)
124                                if(container.owner() instanceof MOB)
125                                        ((MOB)container.owner()).addItem(I);
126                                else
127                                if(container.owner() instanceof Room)
128                                        ((Room)container.owner()).addItem(I,ItemPossessor.Expire.Resource);
129                }
130        }
