035public class Citizen extends StdMOB
036{
037        @Override
038        public String ID()
039        {
040                return "Citizen";
041        }
042
043        public Citizen()
044        {
045                super();
046                username="a citizen";
047                setDescription("");
048                baseCharStats().setStat(CharStats.STAT_GENDER,(CMLib.dice().rollPercentage()>50)?'M':'F');
049                setDisplayText("A citizen goes about "+baseCharStats().hisher()+" business.");
050                CMLib.factions().setAlignment(this,Faction.Align.GOOD);
051                setMoney(10);
052                basePhyStats.setWeight(150);
053                setWimpHitPoint(0);
054
055                baseCharStats().setStat(CharStats.STAT_INTELLIGENCE,7);
056                baseCharStats().setMyRace(CMClass.getRace("Human"));
057                baseCharStats().getMyRace().startRacing(this,false);
058
059                basePhyStats().setAbility(0);
060                basePhyStats().setLevel(1);
061                basePhyStats().setArmor(90);
062
063                baseState.setHitPoints(CMLib.dice().roll(basePhyStats().level(),20,basePhyStats().level()));
064
065                recoverMaxState();
066                resetToMaxState();
067                recoverPhyStats();
068                recoverCharStats();
069        }
070
071}
