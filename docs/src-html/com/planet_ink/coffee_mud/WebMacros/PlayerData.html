120        public static String getBasic(MOB M, BASICS basic)
121        {
122                final StringBuffer str=new StringBuffer("");
123                switch(basic)
124                {
125                case NAME:
126                        str.append(M.Name() + ", ");
127                        break;
128                case DESCRIPTION:
129                        str.append(M.description() + ", ");
130                        break;
131                case LASTDATETIME:
132                        if (M.playerStats() != null)
133                                str.append(CMLib.time().date2String(M.playerStats().getLastDateTime()) + ", ");
134                        break;
135                case EMAIL:
136                        if (M.playerStats() != null)
137                                str.append(M.playerStats().getEmail() + ", ");
138                        break;
139                case RACENAME:
140                        str.append(M.baseCharStats().getMyRace().name() + ", ");
141                        break;
142                case CHARCLASS:
143                        str.append(M.baseCharStats().getCurrentClass().name(M.baseCharStats().getCurrentClassLevel()) + ", ");
144                        break;
145                case LEVEL:
146                        str.append(M.basePhyStats().level() + ", ");
147                        break;
148                case LEVELSTR:
149                        str.append(M.baseCharStats().displayClassLevel(M, true) + ", ");
150                        break;
151                case CLASSLEVEL:
152                        str.append(M.baseCharStats().getClassLevel(M.baseCharStats().getCurrentClass()) + ", ");
153                        break;
154                case CLASSES:
155                {
156                        for(int c=M.charStats().numClasses()-1;c>=0;c--)
157                        {
158                                final CharClass C=M.charStats().getMyClass(c);
159                                str.append(C.name(M.baseCharStats().getCurrentClassLevel())+" ("+M.charStats().getClassLevel(C)+") ");
160                        }
161                        str.append(", ");
162                        break;
163                }
164                case MAXCARRY:
165                        if (M.maxCarry() > (Integer.MAX_VALUE / 3))
166                                str.append("NA, ");
167                        else
168                                str.append(M.maxCarry() + ", ");
169                        break;
170                case ATTACKNAME:
171                        str.append(CMStrings.capitalizeAndLower(CMStrings.removeColors(CMLib.combat().fightingProwessStr(M))) + ", ");
172                        break;
173                case ARMORNAME:
174                        str.append(CMStrings.capitalizeAndLower(CMStrings.removeColors(CMLib.combat().armorStr(M))) + ", ");
175                        break;
176                case DAMAGENAME:
177                        str.append(CMLib.combat().adjustedDamage(M, null, null, 0, false, true) + ", ");
178                        break;
179                case HOURS:
180                        str.append(Math.round(CMath.div(M.getAgeMinutes(), 60.0)) + ", ");
181                        break;
182                case PRACTICES:
183                        str.append(M.getPractices() + ", ");
184                        break;
185                case EXPERIENCE:
186                        str.append(M.getExperience() + ", ");
187                        break;
188                case EXPERIENCELEVEL:
189                        if(M.getExpNeededLevel()==Integer.MAX_VALUE)
190                                str.append("N/A, ");
191                        else
192                                str.append(M.getExpNextLevel()+", ");
193                        break;
194                case TRAINS:
195                        str.append(M.getTrains() + ", ");
196                        break;
197                case MONEY:
198                        str.append(CMLib.beanCounter().getMoney(M) + ", ");
199                        break;
200                case DEITYNAME:
201                        str.append(M.getWorshipCharID() + ", ");
202                        break;
203                case LIEGE:
204                        str.append(M.getLiegeID() + ", ");
205                        break;
206                case CLANNAMES:
207                case CLANROLE: // deprecated
208                {
209                        final StringBuilder buf = new StringBuilder("");
210                        for (final Pair<Clan, Integer> p : M.clans())
211                                buf.append(p.first.getName()).append(", ");
212                        if (buf.length() > 2)
213                                str.append(buf.substring(0, buf.length() - 2));
214                        break;
215                }
216                case ALIGNMENTNAME:
217                        str.append(M.fetchFaction(CMLib.factions().AlignID()) + ", ");
218                        break;
219                case ALIGNMENTSTRING:
220                {
221                        final Faction.FRange FR=CMLib.factions().getRange(CMLib.factions().AlignID(),M.fetchFaction(CMLib.factions().AlignID()));
222                        if(FR!=null)
223                                str.append(FR.name()+", ");
224                        else
225                                str.append(M.fetchFaction(CMLib.factions().AlignID()));
226                        break;
227                }
228                case WIMP:
229                        str.append(M.getWimpHitPoint() + ", ");
230                        break;
231                case STARTROOM:
232                        if(M.getStartRoom()!=null)
233                                str.append(M.getStartRoom().displayText()+", ");
234                        break;
235                case LOCATION:
236                        if(M.location()!=null)
237                                str.append(M.location().displayText()+", ");
238                        break;
239                case STARTROOMID:
240                        if(M.getStartRoom()!=null)
241                                str.append(M.getStartRoom().roomID()+", ");
242                        break;
243                case LOCATIONID:
244                        if(M.location()!=null)
245                                str.append(M.location().roomID()+", ");
246                        break;
247                case INVENTORY:
248                {
249                        for(int inv=0;inv<M.numItems();inv++)
250                        {
251                                final Item I=M.getItem(inv);
252                                if((I!=null)&&(I.container()==null))
253                                        str.append(I.name()+", ");
254                        }
255                        break;
256                }
257                case WEIGHT:
258                        str.append(M.basePhyStats().weight() + ", ");
259                        break;
260                case ENCUMBRANCE:
261                        str.append(M.phyStats().weight() + ", ");
262                        break;
263                case GENDERNAME:
264                        str.append(CMStrings.capitalizeAndLower(M.baseCharStats().genderName()) + ", ");
265                        break;
266                case LASTDATETIMEMILLIS:
267                        if(M.playerStats()!=null)
268                                str.append(M.playerStats().getLastDateTime()+", ");
269                        break;
270                case HITPOINTS:
271                        str.append(M.curState().getHitPoints() + ", ");
272                        break;
273                case MANA:
274                        str.append(M.curState().getMana() + ", ");
275                        break;
276                case MOVEMENT:
277                        str.append(M.curState().getMovement() + ", ");
278                        break;
279                case RIDING:
280                        if(M.riding()!=null)
281                                str.append(M.riding().name()+", ");
282                        break;
283                case HEIGHT:
284                        str.append(M.basePhyStats().height() + ", ");
285                        break;
286                case LASTIP:
287                {
288                        if(!M.isMonster())
289                                str.append(M.session().getAddress()+", ");
290                        else
291                        if(M.playerStats()!=null)
292                                str.append(M.playerStats().getLastIP()+", ");
293                        break;
294                }
295                case QUESTPOINTS:
296                        str.append(M.getQuestPoint() + ", ");
297                        break;
298                case MAXHITPOINTS:
299                        str.append(M.maxState().getHitPoints() + ", ");
300                        break;
301                case MAXMANA:
302                        str.append(M.maxState().getMana() + ", ");
303                        break;
304                case MAXMOVEMENT:
305                        str.append(M.maxState().getMovement() + ", ");
306                        break;
307                case IMAGE:
308                        str.append(M.rawImage() + ", ");
309                        break;
310                case MAXITEMS:
311                        str.append(M.maxItems() + ", ");
312                        break;
313                case IMGURL:
314                {
315                        final String[] paths=CMLib.protocol().mxpImagePath(M.image());
316                        if(paths[0].length()>0)
317                                str.append(paths[0]+paths[1]+", ");
318                        break;
319                }
320                case HASIMG:
321                {
322                        if(CMLib.protocol().mxpImagePath(M.image())[0].length()>0)
323                                str.append("true, ");
324                        else
325                                str.append("false, ");
326                        break;
327                }
328                case NOTES:
329                        if(M.playerStats()!=null)
330                                str.append(M.playerStats().getNotes()+", ");
331                        break;
332                case LEVELS:
333                        if(M.playerStats()!=null)
334                        {
335                                long lastDateTime=-1;
336                                for(int level=0;level<=M.phyStats().level();level++)
337                                {
338                                        final long dateTime=M.playerStats().leveledDateTime(level);
339                                        final long ageMinutes=M.playerStats().leveledMinutesPlayed(level);
340                                        if((dateTime>1529122205)&&(dateTime!=lastDateTime))
341                                        {
342                                                lastDateTime = dateTime;
343                                                str.append("<TR>");
344                                                if(level==0)
345                                                        str.append("<TD><FONT COLOR=WHITE>Created</FONT></TD>");
346                                                else
347                                                        str.append("<TD><FONT COLOR=WHITE>"+level+"</FONT></TD>");
348                                                str.append("<TD><FONT COLOR=WHITE>"+CMLib.time().date2String(dateTime)+"</FONT></TD>");
349                                                str.append("<TD><FONT COLOR=WHITE>"+CMLib.time().date2EllapsedTime(ageMinutes * 60000L,TimeUnit.MINUTES,true)+"</FONT></TD>");
350                                                str.append("</TR>");
351                                        }
352                                }
353                                str.append(", ");
354                        }
355                        break;
356                case ATTACK:
357                        str.append(M.basePhyStats().attackAdjustment() + ", ");
358                        break;
359                case DAMAGE:
360                        str.append(M.basePhyStats().damage() + ", ");
361                        break;
362                case ARMOR:
363                        str.append(M.basePhyStats().armor() + ", ");
364                        break;
365                case SPEEDNAME:
366                        str.append(M.phyStats().speed() + ", ");
367                        break;
368                case SPEED:
369                        str.append(M.basePhyStats().speed() + ", ");
370                        break;
371                case EXPERTISE:
372                {
373                        for(final Enumeration<String> x=M.expertises();x.hasMoreElements();)
374                        {
375                                final String E=x.nextElement();
376                                final ExpertiseLibrary.ExpertiseDefinition X=CMLib.expertises().getDefinition(E);
377                                if(X==null)
378                                        str.append(E+", ");
379                                else
380                                        str.append(X.name()+", ");
381                        }
382                        break;
383                }
384                case TATTOOS:
385                {
386                        for(final Enumeration<Tattoo> e=M.tattoos();e.hasMoreElements();)
387                         str.append(e.nextElement().toString()+", ");
388                        break;
389                }
390                case SECURITY:
391                {
392                        if(M.playerStats()!=null)
393                        {
394                                final List<String> flags=CMParms.parseSemicolons(M.playerStats().getSetSecurityFlags(null), true);
395                                for(int b=0;b<flags.size();b++)
396                                {
397                                        final String B=flags.get(b);
398                                        if(B!=null)
399                                                str.append(B+", ");
400                                }
401                        }
402                        break;
403                }
404                case TITLES:
405                {
406                        if(M.playerStats()!=null)
407                        {
408                                for(int b=0;b<M.playerStats().getTitles().size();b++)
409                                {
410                                        final String B=M.playerStats().getTitles().get(b);
411                                        if(B!=null)
412                                                str.append(B+", ");
413                                }
414                        }
415                        break;
416                }
417                case FACTIONNAMES:
418                {
419                        for(final Enumeration<String> e=M.factions();e.hasMoreElements();)
420                        {
421                                final String FID=e.nextElement();
422                                final Faction F=CMLib.factions().getFaction(FID);
423                                final int value=M.fetchFaction(FID);
424                                if(F!=null)
425                                        str.append(F.name()+" ("+value+"), ");
426                        }
427                        break;
428                }
429                case ACCTEXPUSED:
430                        str.append(CMProps.getBoolVar(CMProps.Bool.ACCOUNTEXPIRATION) ? "true" : "false");
431                        break;
432                case ACCTEXP:
433                {
434                        if(M.playerStats()!=null)
435                        {
436                                if(CMSecurity.isAllowedEverywhere(M, CMSecurity.SecFlag.NOEXPIRE))
437                                        str.append("Never");
438                                else
439                                        str.append(CMLib.time().date2String(M.playerStats().getAccountExpiration()));
440                        }
441                        break;
442                }
443                case FOLLOWERNAMES:
444                {
445                        for(int f=0;f<M.numFollowers();f++)
446                                str.append(M.fetchFollower(f).name()).append(", ");
447                        //Vector<MOB> V=CMLib.database().DBScanFollowers(M);
448                        //for(int v=0;v<V.size();v++)
449                        //    str.append(((MOB)V.elementAt(v)).name()).append(", ");
450                        break;
451                }
452                case ACCOUNT:
453                        if((M.playerStats()!=null)&&(M.playerStats().getAccount()!=null))
454                                str.append(M.playerStats().getAccount().getAccountName());
455                        break;
456                case BASEHITPOINTS:
457                        str.append(M.baseState().getHitPoints()).append(", ");
458                        break;
459                case BASEMANA:
460                        str.append(M.baseState().getMana()).append(", ");
461                        break;
462                case BASEMOVEMENT:
463                        str.append(M.baseState().getMovement()).append(", ");
464                        break;
465                }
466                return str.toString();
467        }

469        @Override
470        public String runMacro(HTTPRequest httpReq, String parm, HTTPResponse httpResp)
471        {
472                if(!CMProps.getBoolVar(CMProps.Bool.MUDSTARTED))
473                        return CMProps.getVar(CMProps.Str.MUDSTATUS);
474
475                final java.util.Map<String,String> parms=parseParms(parm);
476                final String last=httpReq.getUrlParameter("PLAYER");
477                if(last==null)
478                        return " @break@";
479                if(last.length()>0)
480                {
481                        MOB M=CMLib.players().getLoadPlayer(last);
482                        if(M==null)
483                        {
484                                final MOB authM=Authenticate.getAuthenticatedMob(httpReq);
485                                if((authM!=null)&&(authM.Name().equalsIgnoreCase(last)))
486                                        M=authM;
487                                else
488                                        return " @break@";
489                        }
490
491                        final boolean firstTime=(!httpReq.isUrlParameter("ACTION"))
492                                                        ||(httpReq.getUrlParameter("ACTION")).equals("FIRSTTIME");
493                        final StringBuffer str=new StringBuffer("");
494                        for(MOB.Attrib a : MOB.Attrib.values())
495                        {
496                                if(parms.containsKey(a.getName()))
497                                {
498                                        boolean set=M.isAttributeSet(a);
499                                        if(a.isAutoReversed()) 
500                                                set=!set;
501                                        str.append((set?"ON":"OFF")+",");
502                                }
503                        }
504                        for(final int i : CharStats.CODES.ALLCODES())
505                        {
506                                final String stat=CharStats.CODES.NAME(i);
507                                if(!stat.equalsIgnoreCase("GENDER"))
508                                {
509                                        final CharStats C=M.charStats();
510                                        if(parms.containsKey(stat))
511                                        {
512                                                String old=httpReq.getUrlParameter(stat);
513                                                if((firstTime)||(old.length()==0))
514                                                {
515                                                        if((!CharStats.CODES.isBASE(i))&&(i!=CharStats.STAT_GENDER))
516                                                                old=""+C.getSave(i);
517                                                        else
518                                                                old=""+C.getStat(i);
519                                                }
520                                                str.append(old+", ");
521                                        }
522                                }
523                        }
524                        for(final int i : CharStats.CODES.ALLCODES())
525                        {
526                                final String stat=CharStats.CODES.NAME(i);
527                                if(!stat.equalsIgnoreCase("GENDER"))
528                                {
529                                        final CharStats C=M.baseCharStats();
530                                        if(parms.containsKey("BASE"+stat))
531                                        {
532                                                String old=httpReq.getUrlParameter("BASE"+stat);
533                                                if((firstTime)||(old.length()==0))
534                                                        old=""+C.getStat(i);
535                                                str.append(old+", ");
536                                        }
537                                }
538                        }
539                        for(int i=0;i<BASICS.values().length;i++)
540                        {
541                                if(parms.containsKey(BASICS.values()[i].name()))
542                                {
543                                        if(httpReq.isUrlParameter(BASICS.values()[i].name()))
544                                                str.append(httpReq.getUrlParameter(BASICS.values()[i].name())+", ");
545                                        else
546                                                str.append(getBasic(M,BASICS.values()[i]));
547                                }
548                        }
549                        if(parms.containsKey("RACE"))
550                        {
551                                String old=httpReq.getUrlParameter("RACE");
552                                if((firstTime)||(old.length()==0))
553                                        old=""+M.baseCharStats().getMyRace().ID();
554                                for(final Enumeration<Race> r=MobData.sortedRaces(httpReq);r.hasMoreElements();)
555                                {
556                                        final Race R2=r.nextElement();
557                                        str.append("<OPTION VALUE=\""+R2.ID()+"\"");
558                                        if(R2.ID().equals(old))
559                                                str.append(" SELECTED");
560                                        str.append(">"+R2.name());
561                                }
562                        }
563                        if(parms.containsKey("DEITY"))
564                        {
565                                String old=httpReq.getUrlParameter("DEITY");
566                                if(firstTime)
567                                        old=M.getWorshipCharID();
568                                str.append("<OPTION "+((old.length()==0)?"SELECTED":"")+" VALUE=\"\">Godless");
569                                for(final Enumeration<Deity> e=CMLib.map().deities();e.hasMoreElements();)
570                                {
571                                        final Deity E=e.nextElement();
572                                        str.append("<OPTION VALUE=\""+E.Name()+"\"");
573                                        if(E.Name().equalsIgnoreCase(old))
574                                                str.append(" SELECTED");
575                                        str.append(">"+E.Name());
576                                }
577                        }
578                        if(parms.containsKey("TITLELIST"))
579                        {
580                                if(M.playerStats()!=null)
581                                {
582                                        int b=0;
583                                        final Vector<String> titles=new Vector<String>();
584                                        if(firstTime)
585                                                titles.addAll(M.playerStats().getTitles());
586                                        else
587                                        while(httpReq.isUrlParameter("TITLE"+b))
588                                        {
589                                                final String B=httpReq.getUrlParameter("TITLE"+b);
590                                                if((B!=null)&&(B.trim().length()>0))
591                                                        titles.addElement(B);
592                                                b++;
593                                        }
594                                        for(b=0;b<titles.size();b++)
595                                        {
596                                                final String B=titles.elementAt(b);
597                                                if(B!=null)
598                                                        str.append("<INPUT TYPE=TEXT NAME=TITLE"+b+" SIZE="+B.length()+" VALUE=\""+CMStrings.replaceAll(B,"\"","&quot;")+"\"><BR>");
599                                        }
600                                        str.append("<INPUT TYPE=TEXT NAME=TITLE"+titles.size()+" SIZE=60 VALUE=\"\">");
601                                }
602                        }
603                        if(parms.containsKey("ALIGNMENT"))
604                        {
605                                String old=httpReq.getUrlParameter("ALIGNMENT");
606                                if((firstTime)||(old.length()==0))
607                                        old=""+M.fetchFaction(CMLib.factions().AlignID());
608                                if(CMLib.factions().getFaction(CMLib.factions().AlignID())!=null)
609                                {
610                                        for(final Faction.Align v : Faction.Align.values())
611                                        {
612                                                if(v!=Faction.Align.INDIFF)
613                                                {
614                                                        str.append("<OPTION VALUE="+v.toString());
615                                                        if(old.equalsIgnoreCase(v.toString()))
616                                                                str.append(" SELECTED");
617                                                        str.append(">"+CMStrings.capitalizeAndLower(v.toString().toLowerCase()));
618                                                }
619                                        }
620                                }
621                        }
622                        if(parms.containsKey("BASEGENDER"))
623                        {
624                                String old=httpReq.getUrlParameter("BASEGENDER");
625                                if(firstTime)
626                                        old=""+(char)M.baseCharStats().getStat(CharStats.STAT_GENDER);
627                                str.append("<OPTION VALUE=M "+((old.equalsIgnoreCase("M"))?"SELECTED":"")+">M");
628                                str.append("<OPTION VALUE=F "+((old.equalsIgnoreCase("F"))?"SELECTED":"")+">F");
629                                str.append("<OPTION VALUE=N "+((old.equalsIgnoreCase("N"))?"SELECTED":"")+">N");
630                        }
631                        if(parms.containsKey("PLAYERCLANRESET"))
632                                httpReq.removeUrlParameter("CLAN");
633                        else
634                        if(parms.containsKey("PLAYERCLANNEXT"))
635                        {
636                                final String prevClan=httpReq.getUrlParameter("CLAN");
637                                boolean found=false;
638                                boolean next=false;
639                                for(final Pair<Clan, Integer> p : M.clans())
640                                {
641                                        if((prevClan == null)||(next))
642                                        {
643                                                httpReq.addFakeUrlParameter("CLAN", p.first.clanID());
644                                                found=true;
645                                                break;
646                                        }
647                                        else
648                                        if(p.first.clanID().equals(prevClan))
649                                                next=true;
650                                }
651                                if(!found)
652                                {
653                                        httpReq.addFakeUrlParameter("CLAN","");
654                                        if(parms.containsKey("EMPTYOK"))
655                                                str.append("<!--EMPTY-->, ");
656                                        else
