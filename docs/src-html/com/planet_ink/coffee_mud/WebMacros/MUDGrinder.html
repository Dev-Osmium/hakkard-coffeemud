055        public static Area getAreaObject(String ID)
056        {
057                if(ID==null)
058                        return null;
059                if(ID.length()==0)
060                        return null;
061                if(ID.startsWith("ITEM#<")&&(ID.endsWith(">")))
062                {
063                        String xmlIsh=ID.substring(5);
064                        String playerCode=null;
065                        String roomCode=null;
066                        String mobCode=null;
067                        String itemCode=null;
068                        //String shopItemCode=null;
069                        int x=xmlIsh.indexOf(">");
070                        while(x>=0)
071                        {
072                                xmlIsh=xmlIsh.substring(1);
073                                x--;
074                                while((x<xmlIsh.length()-1)&&(xmlIsh.charAt(x+1)!='<'))
075                                        x=xmlIsh.indexOf(">",x+1);
076                                if(x>=0)
077                                {
078                                        String thing=xmlIsh.substring(0,x);
079                                        if(thing.startsWith("P=")&&(x>2))
080                                                playerCode=thing.substring(2);
081                                        else
082                                        if(thing.startsWith("I=")&&(x>2))
083                                                itemCode=thing.substring(2);
084                                        else
085                                        if(thing.startsWith("SI=")&&(x>3))
086                                                itemCode=thing.substring(3);
087                                        else
088                                        if(thing.startsWith("R=")&&(x>2))
089                                                roomCode=thing.substring(2);
090                                        else
091                                        if(thing.startsWith("M=")&&(x>2))
092                                                mobCode=thing.substring(2);
093                                }
094                                if((x>=0)&&(x<xmlIsh.length()-1))
095                                        xmlIsh=xmlIsh.substring(x+1);
096                                else
097                                        break;
098                                x=xmlIsh.indexOf(">");
099                        }
100                        if(itemCode == null)
101                        {
102                                return null;
103                        }
104                        MOB playerM=null;
105                        Room R=null;
106                        if(playerCode!=null)
107                                playerM=CMLib.players().getLoadPlayer(playerCode);
108                        if((playerCode==null)&&(roomCode!=null))
109                        {
110                                if(!roomCode.equalsIgnoreCase("ANY"))
111                                        R=getRoomObject((HTTPRequest)null,roomCode);
112                        }
113                        Item I=null;
114                        MOB M=null;
115                        final String sync=("SYNC"+((R!=null)?R.roomID():playerCode));
116                        synchronized(sync.intern())
117                        {
118                                if(R!=null)
119                                        R=CMLib.map().getRoom(R);
120                                
121                                if((playerM!=null)&&(R==null))
122                                {
123                                        I=RoomData.getItemFromCode(playerM,itemCode);
124                                        M=playerM;
125                                }
126                                else
127                                if((mobCode!=null)&&(mobCode.length()>0))
128                                {
129                                        if(R!=null)
130                                                M=RoomData.getMOBFromCode(R,mobCode);
131                                        else
132                                                M=RoomData.getMOBFromCode(RoomData.getMOBCache(),mobCode);
133                                        if(M!=null)
134                                        {
135                                                I=RoomData.getItemFromCode(M,itemCode);
136                                                if(I==null)
137                                                        I=RoomData.getItemFromCode((MOB)null,itemCode);
138                                        }
139                                }
140                                else
141                                if(R!=null)
142                                {
143                                        I=RoomData.getItemFromCode(R,itemCode);
144                                        if(I==null)
145                                                I=RoomData.getItemFromCode((Room)null,itemCode);
146                                }
147                                if(I==null)
148                                {
149                                        if(itemCode.startsWith("CATALOG-")||itemCode.startsWith("NEWCATA-"))
150                                        {
151                                                I=CMLib.catalog().getCatalogItem(itemCode.substring(8));
152                                                if(I==null)
153                                                        I=CMClass.getItem("GenItem");
154                                                else
155                                                        I=(Item)I.copyOf();
156                                        }
157                                        else
158                                                I=RoomData.getItemFromAnywhere(RoomData.getItemCache(),itemCode);
159                                }
160                        }
161                        if(I instanceof BoardableShip)
162                        {
163                                return ((BoardableShip)I).getShipArea();
164                        }
165                }
166                else
167                {
168                        return CMLib.map().getArea(ID);
169                }
170                return null;
171        }

173        public static Room getRoomObject(String AREA, String ID)
174        {
175                if(ID==null)
176                        return null;
177                if(ID.length()==0)
178                        return null;
179                if((AREA!=null)&&(AREA.startsWith("ITEM#<"))&&(AREA.endsWith(">")))
180                {
181                        final Area A=getAreaObject(AREA);
182                        if(A!=null)
183                                return A.getRoom(ID);
184                }
185                final Room R=CMLib.map().getRoom(ID);
186                if(R!=null)
187                        return R;
188                final Area A=getAreaObject(AREA);
189                if(A!=null)
190                        return A.getRoom(ID);
191                return null;
192        }

194        public static Room getRoomObject(HTTPRequest req, String ID)
195        {
196                if(ID==null)
197                        return null;
198                if(ID.length()==0)
199                        return null;
200                if(req != null)
201                {
202                        String areaID=req.getUrlParameter("AREA");
203                        if((areaID!=null)&&(areaID.startsWith("ITEM#<"))&&(areaID.endsWith(">")))
204                                return getRoomObject(areaID, ID);
205                }
206                final Room R=CMLib.map().getRoom(ID);
207                if(R!=null)
208                        return R;
209                return (req==null) ? null : getRoomObject(req.getUrlParameter("AREA"), ID);
210        }
