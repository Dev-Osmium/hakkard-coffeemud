036public class ComponentPieceData extends StdWebMacro
037{
038        @Override
039        public String name()
040        {
041                return "ComponentPieceData";
042        }
043
044        @Override
045        public String runMacro(HTTPRequest httpReq, String parm, HTTPResponse httpResp)
046        {
047                final java.util.Map<String,String> parms=parseParms(parm);
048                final String compID=httpReq.getUrlParameter("COMPONENT");
049                if(compID==null)
050                        return " @break@";
051                final String last=httpReq.getUrlParameter("COMPONENTPIECE");
052                if(last==null)
053                        return " @break@";
054                if(last.length()>0)
055                {
056                        final String fixedCompID=compID.replace(' ','_').toUpperCase();
057                        final StringBuilder str=new StringBuilder("");
058                        if(parms.containsKey("MASK")||parms.containsKey("MASKEDIT"))
059                        {
060                                String s=httpReq.getUrlParameter(fixedCompID+"_PIECE_MASK_"+last);
061                                if(s==null)
062                                        s="";
063                                str.append(s);
064                        }
065                        if(parms.containsKey("STRING"))
066                        {
067                                final String type=httpReq.getUrlParameter(fixedCompID+"_PIECE_TYPE_"+last);
068                                final String strType=httpReq.getUrlParameter(fixedCompID+"_PIECE_STRING_"+last);
069                                final AbilityComponent.CompType C=(AbilityComponent.CompType)CMath.s_valueOf(AbilityComponent.CompType.values(), type);
070                                if((C==null)||(C==AbilityComponent.CompType.STRING)||(!CMath.isNumber(strType)))
071                                        str.append(strType);
072                                else
073                                        str.append("TODO");
074                        }
075                        if(parms.containsKey("STRINGEDIT"))
076                        {
077                                final String type=httpReq.getUrlParameter(fixedCompID+"_PIECE_TYPE_"+last);
078                                String strType=httpReq.getUrlParameter(fixedCompID+"_PIECE_STRING_"+last);
079                                if(strType==null)
080                                        strType="item name";
081                                String subType=httpReq.getUrlParameter(fixedCompID+"_PIECE_STYPE_"+last);
082                                if(subType==null)
083                                        subType="";
084                                final AbilityComponent.CompType C=(AbilityComponent.CompType)CMath.s_valueOf(AbilityComponent.CompType.values(), type);
085                                if((C==null)||(C==AbilityComponent.CompType.STRING))
086                                {
087                                        str.append("<INPUT TYPE=TEXT NAME=\""+fixedCompID+"_PIECE_STRING_"+last+"\" VALUE=\"");
088                                        str.append(strType);
089                                        str.append("\">");
090                                }
091                                else
092                                {
093                                        str.append("<SELECT NAME=\""+fixedCompID+"_PIECE_STRING_"+last+"\">");
094                                        if(C==AbilityComponent.CompType.MATERIAL)
095                                        {
096                                                for(final RawMaterial.Material m : RawMaterial.Material.values())
097                                                {
098                                                        str.append("<OPTION VALUE="+m.mask());
099                                                        if((m.mask()==CMath.s_long(strType))||(m.name().equals(strType)))
100                                                                str.append(" SELECTED");
101                                                        str.append(">"+m.noun());
102                                                }
103                                        }
104                                        else
105                                        if(C==AbilityComponent.CompType.RESOURCE)
106                                        {
107                                                for(int i=0;i<RawMaterial.CODES.TOTAL();i++)
108                                                {
109                                                        str.append("<OPTION VALUE="+RawMaterial.CODES.GET(i));
110                                                        if((RawMaterial.CODES.GET(i)==CMath.s_long(strType))||(RawMaterial.CODES.NAME(i).equals(strType)))
111                                                                str.append(" SELECTED");
112                                                        str.append(">"+RawMaterial.CODES.NAME(i));
113                                                }
114                                        }
115                                        str.append("</SELECT>&nbsp;&nbsp;Sub-Type: ");
116                                        str.append("<INPUT TYPE=TEXT SIZE=2 NAME=\""+fixedCompID+"_PIECE_STYPE_"+last+"\" VALUE=\""+subType+"\">"); 
117                                }
118                        }
119                        if(parms.containsKey("AMOUNT")||parms.containsKey("AMOUNTEDIT"))
120                        {
121                                String s=httpReq.getUrlParameter(fixedCompID+"_PIECE_AMOUNT_"+last);
122                                if(s==null)
123                                        s="1";
124                                str.append(s);
125                        }
126                        if(parms.containsKey("CONSUMED"))
127                        {
128                                final String consumed=httpReq.getUrlParameter(fixedCompID+"_PIECE_CONSUMED_"+last);
129                                if((consumed!=null)&&(consumed.equalsIgnoreCase("on")||consumed.equalsIgnoreCase("checked")))
130                                        str.append("consumed");
131                                else
132                                        str.append("kept");
133                        }
134                        if(parms.containsKey("CONSUMEDEDIT"))
135                        {
136                                final String consumed=httpReq.getUrlParameter(fixedCompID+"_PIECE_CONSUMED_"+last);
137                                if((consumed!=null)&&(consumed.equalsIgnoreCase("on")||consumed.equalsIgnoreCase("checked")))
138                                        str.append("checked");
139                                else
140                                        str.append("");
141                        }
142                        if(parms.containsKey("CONNECTOR"))
143                                str.append(httpReq.getUrlParameter(fixedCompID+"_PIECE_CONNECTOR_"+last));
144                        if(parms.containsKey("CONNECTOREDIT"))
145                        {
146                                str.append("<OPTION VALUE=\"DELETE\">Delete Component");
147                                for(final AbilityComponent.CompConnector conn : AbilityComponent.CompConnector.values())
148                                {
149                                        str.append("<OPTION VALUE=\""+conn.toString()+"\" ");
150                                        if(conn.toString().equalsIgnoreCase(httpReq.getUrlParameter(fixedCompID+"_PIECE_CONNECTOR_"+last)))
151                                                str.append("SELECTED ");
152                                        str.append(">"+CMStrings.capitalizeAndLower(conn.toString()));
153                                }
154                        }
155                        if(parms.containsKey("TYPE"))
156                                str.append(httpReq.getUrlParameter(fixedCompID+"_PIECE_TYPE_"+last));
157                        if(parms.containsKey("SUBTYPE"))
158                                str.append(httpReq.getUrlParameter(fixedCompID+"_PIECE_STYPE_"+last));
159                        if(parms.containsKey("TYPEEDIT"))
160                        {
161                                String currType = httpReq.getUrlParameter(fixedCompID+"_PIECE_TYPE_"+last);
162                                if(currType == null)
163                                        currType = "STRING";
164                                for(final AbilityComponent.CompType conn : AbilityComponent.CompType.values())
165                                {
166                                        str.append("<OPTION VALUE=\""+conn.toString()+"\" ");
167                                        if(conn.toString().equalsIgnoreCase(currType))
168                                                str.append("SELECTED ");
169                                        str.append(">"+CMStrings.capitalizeAndLower(conn.toString()));
170                                }
171                        }
172                        if(parms.containsKey("LOCATION"))
173                                str.append(httpReq.getUrlParameter(fixedCompID+"_PIECE_LOCATION_"+last));
174                        if(parms.containsKey("LOCATIONEDIT"))
175                        {
176                                for(final AbilityComponent.CompLocation conn : AbilityComponent.CompLocation.values())
177                                {
178                                        str.append("<OPTION VALUE=\""+conn.toString()+"\" ");
179                                        if(conn.toString().equalsIgnoreCase(httpReq.getUrlParameter(fixedCompID+"_PIECE_LOCATION_"+last)))
180                                                str.append("SELECTED ");
181                                        str.append(">"+CMStrings.capitalizeAndLower(conn.toString()));
182                                }
183                        }
184                        String strstr=str.toString();
185                        if(strstr.endsWith(", "))
186                                strstr=strstr.substring(0,strstr.length()-2);
187                        return clearWebMacros(strstr);
188                }
189                return "";
190        }
191}
