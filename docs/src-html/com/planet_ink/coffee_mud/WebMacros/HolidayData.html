358        public static StringBuffer behaviors(TriadList<String,String,Integer> behaviors, HTTPRequest httpReq, java.util.Map<String,String> parms, int borderSize)
359        {
360                final StringBuffer str=new StringBuffer("");
361                if(parms.containsKey("BEHAVIORS"))
362                {
363                        final Vector<String> theclasses=new Vector<String>();
364                        final Vector<String> theparms=new Vector<String>();
365                        if(httpReq.isUrlParameter("BEHAV1"))
366                        {
367                                int num=1;
368                                String behav=httpReq.getUrlParameter("BEHAV"+num);
369                                String theparm=httpReq.getUrlParameter("BDATA"+num);
370                                while((behav!=null)&&(theparm!=null))
371                                {
372                                        if(behav.length()>0)
373                                        {
374                                                theclasses.addElement(behav);
375                                                String t=theparm;
376                                                t=CMStrings.replaceAll(t,"\"","&quot;");
377                                                theparms.addElement(t);
378                                        }
379                                        num++;
380                                        behav=httpReq.getUrlParameter("BEHAV"+num);
381                                        theparm=httpReq.getUrlParameter("BDATA"+num);
382                                }
383                        }
384                        else
385                        for(int b=0;b<behaviors.size();b++)
386                        {
387                                final Behavior B=CMClass.getBehavior(behaviors.get(b).first);
388                                if((B!=null)
389                                &&(!B.ID().equalsIgnoreCase("MUDCHAT"))
390                                &&(!B.ID().equalsIgnoreCase("AGGRESSIVE")))
391                                {
392                                        theclasses.addElement(CMClass.classID(B));
393                                        String t=behaviors.get(b).second;
394                                        t=CMStrings.replaceAll(t,"\"","&quot;");
395                                        theparms.addElement(t);
396                                }
397                        }
398                        str.append("<TABLE WIDTH=100% BORDER=\""+borderSize+"\" CELLSPACING=0 CELLPADDING=0>");
399                        for(int i=0;i<theclasses.size();i++)
400                        {
401                                final String theclass=theclasses.elementAt(i);
402                                final String theparm=theparms.elementAt(i);
403                                str.append("<TR><TD WIDTH=30%>");
404                                str.append("<SELECT ONCHANGE=\"EditBehavior(this);\" NAME=BEHAV"+(i+1)+">");
405                                str.append("<OPTION VALUE=\"\">Delete!");
406                                str.append("<OPTION VALUE=\""+theclass+"\" SELECTED>"+theclass);
407                                str.append("</SELECT>");
408                                str.append("</TD><TD WIDTH=70%>");
409                                str.append("<INPUT TYPE=TEXT SIZE=40 NAME=BDATA"+(i+1)+" VALUE=\""+theparm+"\">");
410                                str.append("</TD></TR>");
411                        }
412                        str.append("<TR><TD WIDTH=30%>");
413                        str.append("<SELECT ONCHANGE=\"AddBehavior(this);\" NAME=BEHAV"+(theclasses.size()+1)+">");
414                        str.append("<OPTION SELECTED VALUE=\"\">Select a Behavior");
415
416                        Object[] sortedB=null;
417                        final Vector<String> sortMeB=new Vector<String>();
418                        for(final Enumeration b=CMClass.behaviors();b.hasMoreElements();)
419                        {
420                                final Behavior B=(Behavior)b.nextElement();
421                                sortMeB.addElement(CMClass.classID(B));
422                        }
423                        sortedB=(new TreeSet<String>(sortMeB)).toArray();
424                        for (final Object element : sortedB)
425                        {
426                                final String cnam=(String)element;
427                                str.append("<OPTION VALUE=\""+cnam+"\">"+cnam);
428                        }
429                        str.append("</SELECT>");
430                        str.append("</TD><TD WIDTH=70%>");
431                        str.append("<INPUT TYPE=TEXT SIZE=40 NAME=BDATA"+(theclasses.size()+1)+" VALUE=\"\">");
432                        str.append("</TD></TR>");
433                        str.append("</TABLE>");
434                }
435                return str;
436        }

508        public static StringBuffer priceFactors(TriadList<String,String,Integer> stats, HTTPRequest httpReq, java.util.Map<String,String> parms, int borderSize)
509        {
510                final StringBuffer str=new StringBuffer("");
511                if(parms.containsKey("PRICEFACTORS"))
512                {
513                        final Vector<String> theclasses=new Vector<String>();
514                        final Vector<String> theparms=new Vector<String>();
515                        if(httpReq.isUrlParameter("PRCFAC1"))
516                        {
517                                int num=1;
518                                String behav=httpReq.getUrlParameter("PRCFAC"+num);
519                                String theparm=httpReq.getUrlParameter("PMASK"+num);
520                                while((behav!=null)&&(theparm!=null))
521                                {
522                                        if(!behav.trim().endsWith("%"))
523                                                behav=behav.trim()+"%";
524                                        if((behav.length()>0)&&(CMath.isPct(behav)))
525                                        {
526                                                theclasses.addElement(behav);
527                                                String t=theparm;
528                                                t=CMStrings.replaceAll(t,"\"","&quot;");
529                                                theparms.addElement(t);
530                                        }
531                                        num++;
532                                        behav=httpReq.getUrlParameter("PRCFAC"+num);
533                                        theparm=httpReq.getUrlParameter("PMASK"+num);
534                                }
535                        }
536                        else
537                        {
538                                final int pndex=stats.indexOfFirst("PRICEMASKS");
539                                final String priceStr=(pndex<0)?"":(String)stats.get(pndex).second;
540                                final List<String> priceV=CMParms.parseCommas(priceStr,true);
541                                for(int v=0;v<priceV.size();v++)
542                                {
543                                        final String priceLine=priceV.get(v);
544                                        double priceFactor=0.0;
545                                        String mask="";
546                                        final int x=priceLine.indexOf(' ');
547                                        if(x<0)
548                                                priceFactor=CMath.s_double(priceLine);
549                                        else
550                                        {
551                                                priceFactor=CMath.s_double(priceLine.substring(0,x));
552                                                mask=priceLine.substring(x+1).trim();
553                                        }
554                                        theclasses.addElement((priceFactor*100.0)+"%");
555                                        mask=CMStrings.replaceAll(mask,"\"","&quot;");
556                                        theparms.addElement(mask);
557                                }
558                        }
559                        str.append("<TABLE WIDTH=100% BORDER=\""+borderSize+"\" CELLSPACING=0 CELLPADDING=0>");
560                        final String sfont=(parms.containsKey("FONT"))?("<FONT "+(parms.get("FONT"))+">"):"";
561                        final String efont=(parms.containsKey("FONT"))?"</FONT>":"";
562                        if(parms.containsKey("HEADERCOL1")||parms.containsKey("HEADERCOL2"))
563                        {
564                                str.append("<TR><TD WIDTH=25%>");
565                                if(parms.containsKey("HEADERCOL1"))
566                                        str.append(sfont + (parms.get("HEADERCOL1")) + efont);
567                                str.append("</TD><TD WIDTH=75%>");
568                                if(parms.containsKey("HEADERCOL2"))
569                                        str.append(sfont + (parms.get("HEADERCOL2")) + efont);
570                                str.append("</TD></TR>");
571                        }
572                        for(int i=0;i<theclasses.size();i++)
573                        {
574                                final String theclass=theclasses.elementAt(i);

712        protected Vector<String> getMoodList()
713        {
714                final Vector<String> V=new Vector<String>();
715                final Ability A=CMClass.getAbility("Mood");
716                if(A==null)
717                        return V;
718                int x=0;
719                A.setMiscText(""+x);
720                while((A.text().length()>0)&&(!V.contains(A.text())))
721                {
722                        V.addElement(A.text().toUpperCase().trim());
723                        x++;
724                        A.setMiscText(""+x);
725                }
726                return V;
727        }
