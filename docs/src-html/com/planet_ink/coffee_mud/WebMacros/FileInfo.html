037public class FileInfo extends StdWebMacro
038{
039        @Override
040        public String name()
041        {
042                return "FileInfo";
043        }
044
045        @Override
046        public boolean isAdminMacro()
047        {
048                return true;
049        }
050
051        public String trimSlash(String path)
052        {
053                path=path.trim();
054                while(path.startsWith("/"))
055                        path=path.substring(1);
056                while(path.endsWith("/"))
057                        path=path.substring(0,path.length()-1);
058                return path;
059        }
060
061        @Override
062        public String runMacro(HTTPRequest httpReq, String parm, HTTPResponse httpResp)
063        {
064                final java.util.Map<String,String> parms=parseParms(parm);
065                String path=httpReq.getUrlParameter("PATH");
066                if(path==null)
067                        path="";
068                String file=httpReq.getUrlParameter("FILE");
069                if(file==null)
070                        file="";
071                final MOB M = Authenticate.getAuthenticatedMob(httpReq);
072                if(M==null)
073                        return "[authentication error]";
074                try
075                {
076                        final String filepath=path.endsWith("/")?path+file:path+"/"+file;
077                        final String pathKey="CMFSFILE_"+trimSlash(filepath);
078                        CMFile F=(CMFile)httpReq.getRequestObjects().get(pathKey);
079                        if(F==null)
080                        {
081                                F=new CMFile(filepath,M);
082                                httpReq.getRequestObjects().put(pathKey, F);
083                        }
084                        if(parms.containsKey("ISDIRECTORY"))
085                                return ""+F.isDirectory();
086                        if(parms.containsKey("ISFILE"))
087                                return ""+F.isFile();
088                        if(parms.containsKey("ISLOCAL"))
089                                return ""+F.canLocalEquiv();
090                        if(parms.containsKey("ISBOTH"))
091                                return ""+(F.canLocalEquiv()&&(F.canVFSEquiv()));
092                        if(parms.containsKey("ISVFS"))
093                                return ""+F.canVFSEquiv();
094                        if(parms.containsKey("ISTEXT"))
095                        {
096                                final int x=F.getName().lastIndexOf('.');
097                                if(x<0)
098                                        return "false";
099                                final String mime=MIMEType.All.getMIMEType(F.getName().substring(x)).getType();
100                                if(mime.toUpperCase().startsWith("TEXT"))
101                                        return "true";
102                                return "false";
103                        }
104                        if(parms.containsKey("ISBINARY"))
105                        {
106                                final int x=F.getName().lastIndexOf('.');
107                                if(x<0)
108                                        return "true";
109                                final String mime=MIMEType.All.getMIMEType(F.getName().substring(x)).getType();
110                                if(mime.toUpperCase().startsWith("TEXT"))
111                                        return "false";
112                                return "true";
113                        }
114                        if(parms.containsKey("NAME"))
115                                return ""+F.getName();
116                        if(parms.containsKey("DATA"))
117                                return F.textUnformatted().toString();
118                        if(parms.containsKey("TEXTDATA"))
119                        {
120                                String s=F.text().toString();
121                                s=s.replaceAll("\n\r","\n");
122                                s=CMStrings.replaceAll(s,"&","&amp;");
123                                s=CMStrings.replaceAll(s,"@","&#64;");
124                                s=CMStrings.replaceAll(s,"<","&lt;");
125                                return s;
126                        }
127                }
128                catch(final Exception e)
129                {
130                        return "[error]";
131                }
132                return "";
133        }
134}
