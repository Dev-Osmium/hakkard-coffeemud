036public class AbilityPlayerNext extends StdWebMacro
037{
038        @Override
039        public String name()
040        {
041                return "AbilityPlayerNext";
042        }
043
044        @Override
045        public String runMacro(HTTPRequest httpReq, String parm, HTTPResponse httpResp)
046        {
047                if(!CMProps.getBoolVar(CMProps.Bool.MUDSTARTED))
048                        return CMProps.getVar(CMProps.Str.MUDSTATUS);
049
050                final java.util.Map<String,String> parms=parseParms(parm);
051                final String last=httpReq.getUrlParameter("ABILITY");
052                if(parms.containsKey("RESET"))
053                {
054                        if(last!=null)
055                                httpReq.removeUrlParameter("ABILITY");
056                        return "";
057                }
058                final String ableType=httpReq.getUrlParameter("ABILITYTYPE");
059                if((ableType!=null)&&(ableType.length()>0))
060                        parms.put(ableType,ableType);
061                final String domainType=httpReq.getUrlParameter("DOMAIN");
062                if((domainType!=null)&&(domainType.length()>0))
063                        parms.put("DOMAIN",domainType);
064
065                String lastID="";
066                final String playerName=httpReq.getUrlParameter("PLAYER");
067                MOB M=null;
068                if((playerName!=null)&&(playerName.length()>0))
069                        M=CMLib.players().getLoadPlayer(playerName);
070                if(M==null)
071                {
072                        if(parms.containsKey("EMPTYOK"))
073                                return "<!--EMPTY-->";
074                        return " @break@";
075                }
076
077                final Vector<Ability> abilities=new Vector<Ability>();
078                HashSet<String> foundIDs=new HashSet<String>();
079                for(final Enumeration<Ability> a=M.allAbilities();a.hasMoreElements();)
080                {
081                        final Ability A=a.nextElement();
082                        if((A!=null)&&(!foundIDs.contains(A.ID())))
083                        {
084                                foundIDs.add(A.ID());
085                                abilities.addElement(A);
086                        }
087                }
088                foundIDs.clear();
089                foundIDs=null;
090                for(int a=0;a<abilities.size();a++)
091                {
092                        final Ability A=abilities.elementAt(a);
093                        boolean okToShow=true;
094                        final int classType=A.classificationCode()&Ability.ALL_ACODES;
095                        final String className=httpReq.getUrlParameter("CLASS");
096
097                        if((className!=null)&&(className.length()>0))
098                        {
099                                final int level=CMLib.ableMapper().getQualifyingLevel(className,true,A.ID());
100                                if(level<0)
101                                        okToShow=false;
102                                else
103                                {
104                                        final String levelName=httpReq.getUrlParameter("LEVEL");
105                                        if((levelName!=null)&&(levelName.length()>0)&&(CMath.s_int(levelName)!=level))
106                                                okToShow=false;
107                                }
108                        }
109                        else
110                        {
111                                final int level=CMLib.ableMapper().getQualifyingLevel("Archon",true,A.ID());
112                                if(level<0)
113                                        okToShow=false;
114                                else
115                                {
116                                        final String levelName=httpReq.getUrlParameter("LEVEL");
117                                        if((levelName!=null)&&(levelName.length()>0)&&(CMath.s_int(levelName)!=level))
118                                                okToShow=false;
119                                }
120                        }
121                        if(okToShow)
122                        {
123                                if(parms.containsKey("DOMAIN")&&(classType==Ability.ACODE_SPELL))
124                                {
125                                        final String domain=parms.get("DOMAIN");
126                                        if(!domain.equalsIgnoreCase(Ability.DOMAIN_DESCS[(A.classificationCode()&Ability.ALL_DOMAINS)>>5]))
127                                                okToShow=false;
128                                }
129                                else
130                                {
131                                        boolean containsOne=false;
132                                        for (final String element : Ability.ACODE_DESCS)
133                                        {
134                                                if(parms.containsKey(element))
135                                                {
136                                                        containsOne=true;
137                                                        break;
138                                                }
139                                        }
140                                        if(containsOne&&(!parms.containsKey(Ability.ACODE_DESCS[classType])))
141                                                okToShow=false;
142                                }
143                        }
144                        if(parms.containsKey("NOT"))
145                                okToShow=!okToShow;
146                        if(okToShow)
147                        {
148                                if((last==null)||((last.length()>0)&&(last.equals(lastID))&&(!A.ID().equals(lastID))))
149                                {
150                                        httpReq.addFakeUrlParameter("ABILITY",A.ID());
151                                        return "";
152                                }
153                                lastID=A.ID();
154                        }
155                }
156                httpReq.addFakeUrlParameter("ABILITY","");
157                if(parms.containsKey("EMPTYOK"))
158                        return "<!--EMPTY-->";
159                return " @break@";
160        }
161}
