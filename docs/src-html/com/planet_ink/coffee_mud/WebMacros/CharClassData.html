046        private String classDropDown(String old)
047        {
048                final StringBuffer str=new StringBuffer("");
049                str.append("<OPTION VALUE=\"\" "+((old.length()==0)?"SELECTED":"")+">None");

054                        C2=e.nextElement();

061                        else
062                        if(C2.isGeneric())
063                                str.append("<OPTION VALUE=\""+C2.ID()+"\" "+((old.equalsIgnoreCase(C2.ID())||old.equalsIgnoreCase(C2ID))?"SELECTED":"")+">"+C2.ID()+" (Generic)");
