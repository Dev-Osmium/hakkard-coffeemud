036public class ComponentPieceNext extends StdWebMacro
037{
038        @Override
039        public String name()
040        {
041                return "ComponentPieceNext";
042        }
043
044        @Override
045        public boolean isAdminMacro()
046        {
047                return true;
048        }
049
050        @Override
051        public String runMacro(HTTPRequest httpReq, String parm, HTTPResponse httpResp)
052        {
053                final java.util.Map<String,String> parms=parseParms(parm);
054                final String compID=httpReq.getUrlParameter("COMPONENT");
055                if(compID==null)
056                        return " @break@";
057                final String fixedCompID=compID.replace(' ','_').toUpperCase();
058                if(!httpReq.isUrlParameter(fixedCompID+"_PIECE_CONNECTOR_1"))
059                {
060                        final List<AbilityComponent> set=CMLib.ableComponents().getAbilityComponents(compID);
061                        if(set!=null)
062                        {
063                                int index=1;
064                                for(final AbilityComponent A : set)
065                                {
066                                        httpReq.addFakeUrlParameter(fixedCompID+"_PIECE_MASK_"+index, A.getMaskStr());
067                                        if(A.getType()==AbilityComponent.CompType.STRING)
068                                                httpReq.addFakeUrlParameter(fixedCompID+"_PIECE_STRING_"+index, A.getStringType());
069                                        else
070                                                httpReq.addFakeUrlParameter(fixedCompID+"_PIECE_STRING_"+index, Long.toString(A.getLongType()));
071                                        httpReq.addFakeUrlParameter(fixedCompID+"_PIECE_AMOUNT_"+index, Integer.toString(A.getAmount()));
072                                        httpReq.addFakeUrlParameter(fixedCompID+"_PIECE_CONNECTOR_"+index, A.getConnector().toString());
073                                        httpReq.addFakeUrlParameter(fixedCompID+"_PIECE_LOCATION_"+index, A.getLocation().toString());
074                                        httpReq.addFakeUrlParameter(fixedCompID+"_PIECE_TYPE_"+index, A.getType().toString());
075                                        httpReq.addFakeUrlParameter(fixedCompID+"_PIECE_STYPE_"+index, A.getSubType().toString());
076                                        httpReq.addFakeUrlParameter(fixedCompID+"_PIECE_CONSUMED_"+index, A.isConsumed()?"on":"");
077                                        index++;
078                                }
079                        }
080                        else
081                        {
082                                httpReq.addFakeUrlParameter(fixedCompID+"_PIECE_MASK_1", "");
083                                httpReq.addFakeUrlParameter(fixedCompID+"_PIECE_STRING_1", "item name");
084                                httpReq.addFakeUrlParameter(fixedCompID+"_PIECE_AMOUNT_1", "1");
085                                httpReq.addFakeUrlParameter(fixedCompID+"_PIECE_CONNECTOR_1", "AND");
086                                httpReq.addFakeUrlParameter(fixedCompID+"_PIECE_LOCATION_1", "INVENTORY");
087                                httpReq.addFakeUrlParameter(fixedCompID+"_PIECE_TYPE_1", "STRING");
088                                httpReq.addFakeUrlParameter(fixedCompID+"_PIECE_STYPE_1", "");
089                                httpReq.addFakeUrlParameter(fixedCompID+"_PIECE_CONSUMED_1", "on");
090                        }
091                }
092                else
093                {
094                        int oldIndex=1;
095                        int newIndex=1;
096                        while(httpReq.isUrlParameter(fixedCompID+"_PIECE_CONNECTOR_"+oldIndex))
097                        {
098                                final String type=httpReq.getUrlParameter(fixedCompID+"_PIECE_CONNECTOR_"+oldIndex);
099                                if((type.length()>0)&&(!type.equalsIgnoreCase("DELETE")))
100                                {
101                                        if(newIndex != oldIndex)
102                                        {
103                                                httpReq.addFakeUrlParameter(fixedCompID+"_PIECE_MASK_"+newIndex, httpReq.getUrlParameter(fixedCompID+"_PIECE_MASK_"+oldIndex));
104                                                httpReq.addFakeUrlParameter(fixedCompID+"_PIECE_STRING_"+newIndex, httpReq.getUrlParameter(fixedCompID+"_PIECE_STRING_"+oldIndex));
105                                                httpReq.addFakeUrlParameter(fixedCompID+"_PIECE_AMOUNT_"+newIndex, httpReq.getUrlParameter(fixedCompID+"_PIECE_AMOUNT_"+oldIndex));
106                                                httpReq.addFakeUrlParameter(fixedCompID+"_PIECE_CONNECTOR_"+newIndex, httpReq.getUrlParameter(fixedCompID+"_PIECE_CONNECTOR_"+oldIndex));
107                                                httpReq.addFakeUrlParameter(fixedCompID+"_PIECE_LOCATION_"+newIndex, httpReq.getUrlParameter(fixedCompID+"_PIECE_LOCATION_"+oldIndex));
108                                                httpReq.addFakeUrlParameter(fixedCompID+"_PIECE_TYPE_"+newIndex, httpReq.getUrlParameter(fixedCompID+"_PIECE_TYPE_"+oldIndex));
109                                                httpReq.addFakeUrlParameter(fixedCompID+"_PIECE_STYPE_"+newIndex, httpReq.getUrlParameter(fixedCompID+"_PIECE_STYPE_"+oldIndex));
110                                                httpReq.addFakeUrlParameter(fixedCompID+"_PIECE_CONSUMED_"+newIndex, httpReq.getUrlParameter(fixedCompID+"_PIECE_CONSUMED_"+oldIndex));
111                                        }
112                                        newIndex++;
113                                }
114                                oldIndex++;
115                        }
116                        httpReq.removeUrlParameter(fixedCompID+"_PIECE_MASK_"+newIndex);
117                        httpReq.removeUrlParameter(fixedCompID+"_PIECE_STRING_"+newIndex);
118                        httpReq.removeUrlParameter(fixedCompID+"_PIECE_AMOUNT_"+newIndex);
119                        httpReq.removeUrlParameter(fixedCompID+"_PIECE_CONNECTOR_"+newIndex);
120                        httpReq.removeUrlParameter(fixedCompID+"_PIECE_LOCATION_"+newIndex);
121                        httpReq.removeUrlParameter(fixedCompID+"_PIECE_TYPE_"+newIndex);
122                        httpReq.removeUrlParameter(fixedCompID+"_PIECE_STYPE_"+newIndex);
123                        httpReq.removeUrlParameter(fixedCompID+"_PIECE_CONSUMED_"+newIndex);
124                }
125                final String last=httpReq.getUrlParameter("COMPONENTPIECE");
126                if(parms.containsKey("RESET"))
127                {
128                        if(last!=null)
129                                httpReq.removeUrlParameter("COMPONENTPIECE");
130                        return "";
131                }
132                String lastID="";
133                for(int index=1;httpReq.isUrlParameter(fixedCompID+"_PIECE_CONNECTOR_"+index);index++)
134                {
135                        final String id=Integer.toString(index);
136                        if((last==null)||((last.length()>0)&&(last.equals(lastID))&&(!id.equalsIgnoreCase(lastID))))
137                        {
138                                httpReq.addFakeUrlParameter("COMPONENTPIECE",id);
139                                return "";
140                        }
141                        lastID=id;
142                }
143                httpReq.addFakeUrlParameter("COMPONENTPIECE","");
144                if(parms.containsKey("EMPTYOK"))
145                        return "<!--EMPTY-->";
146                return " @break@";
147        }
148}
