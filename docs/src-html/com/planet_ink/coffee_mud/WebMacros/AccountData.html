036public class AccountData extends StdWebMacro
037{
038        @Override
039        public String name()
040        {
041                return "AccountData";
042        }
043
044        @Override
045        public boolean isAdminMacro()
046        {
047                return true;
048        }
049
050        @Override
051        public String runMacro(HTTPRequest httpReq, String parm, HTTPResponse httpResp)
052        {
053                final java.util.Map<String,String> parms=parseParms(parm);
054                final String last=httpReq.getUrlParameter("ACCOUNT");
055                if(last==null)
056                        return "";
057                if(last.length()>0)
058                {
059                        final PlayerAccount A = CMLib.players().getLoadAccount(last);
060                        if(A==null)
061                                return "";
062                        if(parms.containsKey("NAME")||parms.containsKey("ACCOUNT"))
063                                return clearWebMacros(A.getAccountName());
064                        if(parms.containsKey("CLASS"))
065                                return clearWebMacros(A.ID());
066                        if(parms.containsKey("LASTIP"))
067                                return ""+A.getLastIP();
068                        if(parms.containsKey("LASTDATETIME"))
069                                return ""+CMLib.time().date2String(A.getLastDateTime());
070                        if(parms.containsKey("EMAIL"))
071                                return ""+A.getEmail();
072                        if(parms.containsKey("BONUSLANGS"))
073                                return ""+A.getBonusLanguageLimits();
074                        if(parms.containsKey("BONUSCHARLIMIT"))
075                                return ""+A.getBonusCharsLimit();
076                        if(parms.containsKey("BONUSCHARONLINE"))
077                                return ""+A.getBonusCharsOnlineLimit();
078                        if(parms.containsKey("BONUSALLCOMMONSKILLS"))
079                                return ""+A.getBonusCommonSkillLimits();
080                        if(parms.containsKey("BONUSCRAFTINGSKILLS"))
081                                return ""+A.getBonusCraftingSkillLimits();
082                        if(parms.containsKey("BONUSNONCRAFTINGSKILLS"))
083                                return ""+A.getBonusNonCraftingSkillLimits();
084                        if(parms.containsKey("NOTES"))
085                                return ""+A.getNotes();
086                        if(parms.containsKey("TATTOOS"))
087                        {
088                                StringBuilder str = new StringBuilder("");
089                                for(final Enumeration<Tattoo> e=A.tattoos();e.hasMoreElements();)
090                                        str.append(e.nextElement().toString()+", ");
091                                return str.toString();
092                        }
093                        if(parms.containsKey("ACCTEXPIRATION"))
094                        {
095                                if(A.isSet(PlayerAccount.AccountFlag.NOEXPIRE))
096                                        return "Never";
097                                return ""+CMLib.time().date2String(A.getAccountExpiration());
098                        }
099                        for(final PlayerAccount.AccountFlag flag : PlayerAccount.AccountFlag.values())
100                        {
101                                if(parms.containsKey("IS"+flag.name()))
102                                        return ""+A.isSet(flag);
103                        }
104                        if(parms.containsKey("FLAGS"))
105                        {
106                                final String old=httpReq.getUrlParameter("FLAGS");
107                                List<String> set=null;
108                                if(old==null)
109                                {
110                                        final String matList=A.getStat("FLAG");
111                                        set=CMParms.parseCommas(matList,true);
112                                }
113                                else
114                                {
115                                        String id="";
116                                        set=new Vector<String>();
117                                        for(int i=0;httpReq.isUrlParameter("FLAG"+id);id=""+(++i))
118                                                set.add(httpReq.getUrlParameter("FLAG"+id));
119                                }
120                                final StringBuffer str=new StringBuffer("");
121                                for (final PlayerAccount.AccountFlag element : PlayerAccount.AccountFlag.values())
122                                {
123                                        str.append("<OPTION VALUE=\""+element.name()+"\"");
124                                        if(set.contains(element.name()))
125                                                str.append(" SELECTED");
126                                        str.append(">"+CMStrings.capitalizeAndLower(element.name()));
127                                }
128                                str.append(", ");
129                        }
130                        if(parms.containsKey("IGNORE"))
131                                return ""+CMParms.toListString(A.getIgnored());
132                }
133                return "";
134        }
135}
