036@SuppressWarnings("unchecked")
037public class AreaScriptNext extends StdWebMacro
038{
039        @Override
040        public String name()
041        {
042                return "AreaScriptNext";
043        }
044
045        @Override
046        public boolean isAdminMacro()
047        {
048                return true;
049        }

051        protected class AreaScriptInstance
052        {
053                public List<String>     path;
054                public String           instanceKey;
055                public String           fileName;
056                public String           key;
057                public String           customScript    = "";
058
059                public AreaScriptInstance(String instanceKey, List<String> path,
060                                                                  String key, String fileName)
061                {
062                        this.path=path;
063                        this.instanceKey=instanceKey;
064                        this.fileName=fileName;
065                        this.key=key;
066                }
067        }
068
069        public AreaScriptInstance addScript(TreeMap<String,ArrayList<AreaScriptInstance>> list,
070                        ArrayList<String> prefix, String scriptKey, String immediateHost, String key, String file)
071        {
072                final ArrayList<String> next=(ArrayList<String>)prefix.clone();
073                if(immediateHost!=null)
074                        next.add(immediateHost);
075                ArrayList<AreaScriptInstance> subList =list.get(key);
076                if(subList == null)
077                {
078                        subList = new ArrayList<AreaScriptInstance>();
079                        list.put(key,subList);
080                }
081                final AreaScriptInstance inst = new AreaScriptInstance(scriptKey, next, key, file);
082                subList.add(inst);
083                return inst;
084        }
085
086        public void addScripts(TreeMap<String,ArrayList<AreaScriptInstance>> list, ArrayList<String> prefix, PhysicalAgent E)
087        {
088                if(E==null)
089                        return;
090                for(final Enumeration<Behavior> e=E.behaviors();e.hasMoreElements();)
091                {
092                        final Behavior B=e.nextElement();
093                        if(B instanceof ScriptingEngine)
094                        {
095                                if(!B.isSavable())
096                                        continue;
097                                final ScriptingEngine SE=(ScriptingEngine)B;
098                                final List<String> files=B.externalFiles();
099                                if(files!=null)
100                                for(int f=0;f<files.size();f++)
101                                        addScript(list, prefix, SE.getScriptResourceKey(),B.ID(),files.get(f).toLowerCase(), files.get(f));
102                                final String nonFiles=((ScriptingEngine)B).getVar("*","COFFEEMUD_SYSTEM_INTERNAL_NONFILENAME_SCRIPT");
103                                if((nonFiles!=null)&&(nonFiles.trim().length()>0))
104                                {
105                                        final AreaScriptInstance inst =
106                                                addScript(list, prefix, SE.getScriptResourceKey(), B.ID(),"Custom",nonFiles);
107                                        inst.customScript = nonFiles.trim();
108                                }
109                        }
110                }
111                for(final Enumeration<ScriptingEngine> e=E.scripts();e.hasMoreElements();)
112                {
113                        final ScriptingEngine SE=e.nextElement();
114                        if(!SE.isSavable())
115                                continue;
116                        final List<String> files=SE.externalFiles();
117                        for(int f=0;f<files.size();f++)
118                                addScript(list, prefix, SE.getScriptResourceKey(),null,files.get(f).toLowerCase(), files.get(f));
119                        final String nonFiles=SE.getVar("*","COFFEEMUD_SYSTEM_INTERNAL_NONFILENAME_SCRIPT");
120                        if(nonFiles.trim().length()>0)
121                        {
122                                final AreaScriptInstance inst =
123                                        addScript(list, prefix, SE.getScriptResourceKey(), null,"Custom",nonFiles);
124                                inst.customScript = nonFiles.trim();
125                        }
126                }
127        }
128
129        public void addShopScripts(TreeMap<String,ArrayList<AreaScriptInstance>> list, ArrayList<String> prefix, PhysicalAgent E)
130        {
131                if(E==null)
132                        return;
133                final ShopKeeper SK=CMLib.coffeeShops().getShopKeeper(E);
134                if(SK!=null)
135                {
136                        final CoffeeShop shop=(SK instanceof Librarian)?((Librarian)SK).getBaseLibrary():SK.getShop();
137                        for(final Iterator<Environmental> i=shop.getStoreInventory();i.hasNext();)
138                        {
139                                final Environmental E2=i.next();
140                                final ArrayList<String> newPrefix=(ArrayList<String>)prefix.clone();
141                                newPrefix.add(E2.name());
142                                if(E2 instanceof PhysicalAgent)
143                                        addScripts(list,newPrefix,(PhysicalAgent)E2);
144                        }
145                }
146        }

148        public TreeMap<String,ArrayList<AreaScriptInstance>> getAreaScripts(HTTPRequest httpReq, String area)
149        {
150                TreeMap<String,ArrayList<AreaScriptInstance>> list;
151                list = (TreeMap<String,ArrayList<AreaScriptInstance>>)httpReq.getRequestObjects().get("AREA_"+area+" SCRIPTSLIST");
152                if(list == null)
153                {
154                        list=new TreeMap<String,ArrayList<AreaScriptInstance>>();
155                        Area A=CMLib.map().getArea(area);
156                        if(A==null)
157                                A=CMLib.map().findArea(area);
158                        if(A==null)
159                                return list;
160                        Room R=null;
161                        WorldMap.LocatedPair LP=null;
162                        PhysicalAgent AE=null;
163                        ArrayList<String> prefix = new ArrayList<String>();
164                        for(final Enumeration<WorldMap.LocatedPair> ae=CMLib.map().scriptHosts(A);ae.hasMoreElements();)
165                        {
166                                LP=ae.nextElement(); if(LP==null) continue;
167                                AE=LP.obj(); if(AE==null) continue;
168                                R=LP.room(); if(R==null) R=CMLib.map().getStartRoom(AE);
169
170                                prefix = new ArrayList<String>();
171                                prefix.add(A.name());
172
173                                if(AE instanceof Area)
174                                {
175                                        // don't add room to prefix
176                                }
177                                else
178                                if(AE instanceof Room)
179                                        prefix.add(CMLib.map().getExtendedRoomID((Room)AE));
180                                else
181                                {
182                                        if(R!=null)
183                                                prefix.add(CMLib.map().getExtendedRoomID(R));
184                                        if(AE instanceof Item)
185                                        {
186                                                final ItemPossessor IP=((Item)AE).owner();
187                                                if(IP instanceof MOB)
188                                                        prefix.add(IP.Name());
189                                        }
190                                        prefix.add(AE.Name());
191                                }
192
193                                addScripts(list,prefix,AE);
194                                addShopScripts(list,prefix,AE);
195                        }
196                        httpReq.getRequestObjects().put("AREA_"+area+" SCRIPTSLIST",list);
197                }
198                return list;
199        }

201        @Override
202        public String runMacro(HTTPRequest httpReq, String parm, HTTPResponse httpResp)
203        {
204                final java.util.Map<String,String> parms=parseParms(parm);
205                final String area=httpReq.getUrlParameter("AREA");
206                if((area==null)||(area.length()==0))
207                        return "@break@";
208                String last=httpReq.getUrlParameter("AREASCRIPT");
209                if(parms.containsKey("RESET"))
210                {
211                        if(last!=null)
212                                httpReq.removeUrlParameter("AREASCRIPT");
213                        return "";
214                }
215                String lastID="";
216                final TreeMap<String,ArrayList<AreaScriptInstance>> list = getAreaScripts(httpReq,area);
217                for(final String scriptName : list.keySet())
218                {
219                        if((last==null)||((last.length()>0)&&(last.equals(lastID))&&(!scriptName.equals(lastID))))
220                        {
221                                httpReq.addFakeUrlParameter("AREASCRIPT",scriptName);
222                                last=scriptName;
223                                return "";
224                        }
225                        lastID=scriptName;
226                }
227                httpReq.addFakeUrlParameter("AREASCRIPT","");
228                if(parms.containsKey("EMPTYOK"))
229                        return "<!--EMPTY-->";
230                return " @break@";
231        }
