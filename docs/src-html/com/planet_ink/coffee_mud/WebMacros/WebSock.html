172        private class WebSockHandler implements ProtocolHandler
173        {
174                private volatile WSState        state           = WSState.S0;
175                private volatile int            subState        = 0;
176                private volatile long           dataLen         = 0;
177                private volatile byte           opCode          = 0;
178                private final byte[]            mask            = new byte[4];
179                private volatile boolean        finished        = false;
180                private volatile int            maskPos         = 0;
181                private volatile long           lastPing        = System.currentTimeMillis();
182                private HTTPIOHandler           ioHandler       = null;
183
184                private final CoffeePipeSocket lsock;
185                private final CoffeePipeSocket rsock;
186                private final OutputStream out;
187                private final InputStream in;
188                private final ByteArrayOutputStream payload = new ByteArrayOutputStream();
189                private final ByteArrayOutputStream msg         = new ByteArrayOutputStream();
190
191                public WebSockHandler(HTTPRequest httpReq) throws IOException
192                {
193                        synchronized(this)
194                        {
195                                for(final MudHost h : CMLib.hosts())
196                                {
197                                        try
198                                        {
199                                                final CoffeeIOPipes pipes = new CoffeeIOPipes(65536);
200                                                lsock=new CoffeePipeSocket(httpReq.getClientAddress(),pipes.getLeftPipe(),pipes.getRightPipe());
201                                                rsock=new CoffeePipeSocket(httpReq.getClientAddress(),pipes.getRightPipe(),pipes.getLeftPipe());
202                                                h.acceptConnection(rsock);
203                                                out=lsock.getOutputStream();
204                                                in=lsock.getInputStream();
205                                                return;
206                                        }
207                                        catch(final IOException e)
208                                        {
209                                                throw e;
210                                        }
211                                }
212                        }
213                        throw new IOException("No host found.");
214                }
215
216                private byte[] encodeResponse(byte[] resp, int type)
217                {
218                        ByteArrayOutputStream bout=new ByteArrayOutputStream();
219                        bout.write(0x80 + (byte)type); // output
220                        if(resp.length < 126)
221                        {
222                                bout.write(resp.length & 0x7f);
223                        }
224                        else
225                        if(resp.length < 65535)
226                        {
227                                bout.write(126 & 0x7f);
228                                int byte1=resp.length / 256;
229                                int byte2 = resp.length - (byte1 * 256);
230                                bout.write(byte1);
231                                bout.write(byte2);
232                        }
233                        else
234                        {
235                                bout.write(127 & 0x7f);
236                                long len = resp.length;
237                                int byte1=(int)(len / 16777216);
238                                len = len - (byte1 * 16777216);
239                                int byte2=(int)(len / 65536);
240                                len = len - (byte2 * 65536);
241                                int byte3=(int)(len / 256);
242                                len = len - (byte3 * 256);
243                                bout.write(byte1 & 0xff);
244                                bout.write(byte2 & 0xff);
245                                bout.write(byte3 & 0xff);
246                                bout.write((byte)(len & 0xff));
247                        }
248                        try
249                        {
250                                bout.write(resp);
251                        }
252                        catch (IOException e)
253                        {
254                                e.printStackTrace();
255                        }
256                        return bout.toByteArray();
257                }

259                private void reset(ByteBuffer buffer)
260                {
261                        msg.reset();
262                        payload.reset();
263                        state           = WSState.S0;
264                        subState        = 0;
265                        dataLen         = 0;
266                        opCode          = 0;
267                        finished        = false;
268                        maskPos         = 0;
269                }
270
271                private DataBuffers getOutput(DataBuffers outBuffer)
272                {
273                        if(outBuffer == null)
274                                return new CWDataBuffers();
275                        return outBuffer;
276                }
277                
278                private DataBuffers done(HTTPRequest request, ByteBuffer buffer, DataBuffers outBuffer) throws HTTPException
279                {
280                        switch(opCode)
281                        {
282                        case 1: // text data
283                        {
284                                this.lastPing=System.currentTimeMillis();
285                                final byte[] newPayload = msg.toByteArray();
286                                if(newPayload.length>0)
287                                {
288                                        try
289                                        {
290                                                out.write(newPayload);
291                                        }
292                                        catch (IOException e)
293                                        {
294                                                Log.errOut(e);
295                                        }
296                                }
297                                reset(buffer);
298                                outBuffer = getOutput(outBuffer);
299                                try
300                                {
301                                        final int avail=in.available();
302                                        final byte[] buf=new byte[avail];
303                                        final int num=in.read(buf);
304                                        if(num>0)
305                                        {
306                                                final byte[] outBuf = encodeResponse((num==avail)?buf:Arrays.copyOf(buf, num),1);
307                                                outBuffer.add(outBuf, System.currentTimeMillis(), true);
308                                        }
309                                }
310                                catch (IOException e)
311                                {
312                                        Log.errOut(e);
313                                }
314                                break;
315                        }
316                        case 9: // ping
317                        {
318                                this.lastPing=System.currentTimeMillis();
319                                final byte[] newPayload = msg.toByteArray();
320                                newPayload[0] = (byte)((newPayload[0] & 0xf0) + 10); // 10=pong
321                                reset(buffer);
322                                outBuffer = getOutput(outBuffer);
323                                outBuffer.add(newPayload,System.currentTimeMillis(),true);
324                                break;
325                        }
326                        case 10: // pong -- ignore
327                                break;
328                        case 11: // close
329                                break;
330                        default:
331                                break;
332                        }
333                        return outBuffer;
334                }

336                @Override
337                public DataBuffers processBuffer(HTTPIOHandler handler, HTTPRequest request, ByteBuffer buffer) throws HTTPException
338                {
339                        if(ioHandler==null)
340                                ioHandler = handler;
341                        DataBuffers outBuffers = null;
342                        buffer.flip(); // turn the writeable buffer into a "readable" one
343                        while(buffer.position() < buffer.limit())
344                        {
345                                byte b=buffer.get(); // keep this here .. it ensures the while loop ends
346                                msg.write(b);
347                                switch(state)
348                                {
349                                case M1:
350                                        mask[0]=b;
351                                        state=WSState.M2;
352                                        break;
353                                case M2:
354                                        mask[1]=b;
355                                        state=WSState.M3;
356                                        break;
357                                case M3:
358                                        mask[2]=b;
359                                        state=WSState.M4;
360                                        break;
361                                case M4:
362                                        maskPos=0;
363                                        mask[3]=b;
364                                        if(dataLen > 0)
365                                                state=WSState.PAYLOAD;
366                                        else
367                                        {
368                                                if(finished)
369                                                        outBuffers = done(request,buffer,outBuffers);
370                                                state=WSState.S0;
371                                        }
372                                        break;
373                                case P1:
374                                {
375                                        if((b & 0x80) == 0)
376                                        {
377                                                throw HTTPException.standardException(HTTPStatus.S403_FORBIDDEN);
378                                        }
379                                        this.dataLen = (b & 0x7F);
380                                        if(this.dataLen < 126 )
381                                                state = WSState.M1;
382                                        else
383                                        if(this.dataLen == 126)
384                                        {
385                                                this.dataLen = 0;
386                                                this.subState = 2;
387                                                state = WSState.PX;
388                                        }
389                                        else
390                                        {
391                                                this.dataLen = 0;
392                                                this.subState = 4;
393                                                state = WSState.PX;
394                                        }
395                                        break;
396                                }
397                                case PAYLOAD:
398                                        b = (byte)((b & 0xff) ^ mask[maskPos]);
399                                        maskPos = (maskPos+1) % 4;
400                                        payload.write(b);
401                                        if(--dataLen <= 0)
402                                        {
403                                                if(finished)
404                                                        outBuffers = done(request,buffer,outBuffers);
405                                                state=WSState.S0;
406                                        }
407                                        break;
408                                case PX:
409                                        if(subState > 0)
410                                        {
411                                                subState--;
412                                                dataLen = (dataLen << 8) + (b & 0xff);
413                                                if(subState == 0)
414                                                        state = WSState.M1;
415                                        }
416                                        else
417                                                state = WSState.M1;
418                                        break;
419                                case S0:
420                                {
421                                        opCode = (byte)(b & 0x0f);
422                                        finished = (b & 0x80) == 0x80; 
423                                        state = WSState.P1;
424                                        break;
425                                }
426                                }
427                                //System.out.print(Integer.toHexString(buffer.get() & 0xff)+" ");
428                        }
429                        buffer.clear();
430                        return outBuffers;
431                }

433                @Override
434                public boolean isTimedOut()
435                {
436                        final long idle = System.currentTimeMillis() - lastPing;
437                        if ((idle > (30 * 1000)))
438                        {
439                                this.closeAndWait();
440                        }
441                        return false;
442                }

444                @Override
445                public void closeAndWait()
446                {
447                        try
448                        {
449                                lsock.close();
450                                rsock.close();
451                        }
452                        catch (IOException e)
453                        {
454                                Log.errOut(e);
455                        }
456                        synchronized(handlers)
457                        {
458                                handlers.remove(this);
459                        }
460                }

463        @Override
464        public String runMacro(HTTPRequest httpReq, String parm, HTTPResponse httpResp) throws HTTPServerException
465        {
466                if(!CMProps.getBoolVar(CMProps.Bool.MUDSTARTED))
467                        return "false;";
468                if(!initialized)
469                {
470                        initialize();
471                }
472                
473                if((httpReq.getHeader("upgrade")!=null)
474                &&("websocket".equalsIgnoreCase(httpReq.getHeader("upgrade")))
475                &&(httpReq.getHeader("connection")!=null)
476                &&(httpReq.getHeader("connection").toLowerCase().indexOf("upgrade")>=0))
477                {
478                        final HTTPException exception = new HTTPException(HTTPStatus.S101_SWITCHING_PROTOCOLS);
479                        try
480                        {
481                                final String key = httpReq.getHeader("sec-websocket-key");
482                                final String tokenStr = key + "258EAFA5-E914-47DA-95CA-C5AB0DC85B11";
483                                final MessageDigest cript = MessageDigest.getInstance("SHA-1");
484                                cript.reset();
485                                cript.update(tokenStr.getBytes("utf8"));
486                                final String token = B64Encoder.B64encodeBytes(cript.digest());
487                                httpResp.setStatusCode(101);
488                                exception.getErrorHeaders().put(HTTPHeader.Common.CONNECTION, HTTPHeader.Common.UPGRADE.toString());
489                                exception.getErrorHeaders().put(HTTPHeader.Common.UPGRADE, httpReq.getHeader("upgrade"));
490                                exception.getErrorHeaders().put(HTTPHeader.Common.SEC_WEBSOCKET_ACCEPT, token);
491                                if(httpReq.getHeader("origin")!=null)
492                                        exception.getErrorHeaders().put(HTTPHeader.Common.ORIGIN, httpReq.getHeader("origin"));
493                                final StringBuilder locationStr;
494                                if(httpReq.getFullHost().startsWith("https:"))
495                                        locationStr = new StringBuilder("wss://"+httpReq.getHost());
496                                else
497                                        locationStr = new StringBuilder("ws://"+httpReq.getHost());
498                                if(httpReq.getClientPort() != 80)
499                                        locationStr.append(":").append(httpReq.getClientPort());
500                                locationStr.append(httpReq.getUrlPath());
501                                exception.getErrorHeaders().put(HTTPHeader.Common.SEC_WEBSOCKET_LOCATION, locationStr.toString());
502                                final WebSockHandler newHandler = new WebSockHandler(httpReq);
503                                synchronized(handlers)
504                                {
505                                        handlers.add(newHandler);
506                                }
507                                exception.setNewProtocolHandler(newHandler);
508                                httpReq.getRequestObjects().put("___SIPLETHANDLER", newHandler);
509                        }
510                        catch (Exception e)
511                        {
512                                Log.errOut(e);
513                                throw new HTTPServerException(e.getMessage());
514                        }
515                        throw new HTTPServerException(exception);
516                }
517                return "";
518        }
