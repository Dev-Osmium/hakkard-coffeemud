036public class AutoTitleData extends StdWebMacro
037{
038        @Override
039        public String name()
040        {
041                return "AutoTitleData";
042        }
043
044        @Override
045        public String runMacro(HTTPRequest httpReq, String parm, HTTPResponse httpResp)
046        {
047                final java.util.Map<String,String> parms=parseParms(parm);
048                final String last=httpReq.getUrlParameter("AUTOTITLE");
049                if((last==null)&&(!parms.containsKey("EDIT")))
050                        return " @break@";
051
052                if(parms.containsKey("EDIT"))
053                {
054                        final MOB M = Authenticate.getAuthenticatedMob(httpReq);
055                        if(M==null)
056                                return "[authentication error]";
057                        if(!CMSecurity.isAllowed(M,M.location(),CMSecurity.SecFlag.TITLES))
058                                return "[authentication error]";
059                        final String req=httpReq.getUrlParameter("ISREQUIRED");
060                        String newTitle=httpReq.getUrlParameter("TITLE");
061                        if((req!=null)&&(req.equalsIgnoreCase("on")))
062                                newTitle="{"+newTitle+"}";
063                        final String newMask=httpReq.getUrlParameter("MASK");
064                        if((newTitle==null)||(newMask==null)||(newTitle.length()==0))
065                                return "[missing data error]";
066
067                        if((last!=null)&&((last.length()==0)&&(CMLib.titles().isExistingAutoTitle(newTitle))))
068                        {
069                                CMLib.titles().reloadAutoTitles();
070                                return "[new title already exists!]";
071                        }
072
073                        final String error=CMLib.titles().evaluateAutoTitle(newTitle+"="+newMask,false);
074                        if(error!=null)
075                                return "[error: "+error+"]";
076
077                        if((last!=null)&&(CMLib.titles().isExistingAutoTitle(last)))
078                        {
079                                final String err=CMLib.titles().deleteTitleAndResave(last);
080                                if(err!=null)
081                                {
082                                        CMLib.titles().reloadAutoTitles();
083                                        return err;
084                                }
085                        }
086                        CMLib.titles().appendAutoTitle("\n"+newTitle+"="+newMask);
087                }
088                else
089                if(parms.containsKey("DELETE"))
090                {
091                        final MOB M = Authenticate.getAuthenticatedMob(httpReq);
092                        if(M==null)
093                                return "[authentication error]";
094                        if(!CMSecurity.isAllowed(M,M.location(),CMSecurity.SecFlag.TITLES))
095                                return "[authentication error]";
096                        if(last==null)
097                                return " @break@";
098                        if(!CMLib.titles().isExistingAutoTitle(last))
099                                return "Unknown title!";
100                        final String err=CMLib.titles().deleteTitleAndResave(last);
101                        if(err==null)
102                                return "Auto-Title deleted.";
103                        return err;
104                }
105                else
106                if(last==null)
107                        return " @break@";
108                final StringBuffer str=new StringBuffer("");
109
110                if(parms.containsKey("MASK"))
111                {
112                        String mask=httpReq.getUrlParameter("MASK");
113                        if((mask==null)&&(last!=null)&&(last.length()>0))
114                                mask=CMLib.titles().getAutoTitleMask(last);
115                        if(mask!=null)
116                                str.append(CMStrings.replaceAll(mask,"\"","&quot;")+", ");
117                }
118                if(parms.containsKey("TITLE"))
119                {
120                        String title=httpReq.getUrlParameter("TITLE");
121                        if(title==null)
122                                title=last;
123                        if(title!=null)
124                        {
125                                if(title.startsWith("{")&&title.endsWith("}"))
126                                        title=title.substring(1,title.length()-1);
127                                str.append(title+", ");
128                        }
129                }
130                if(parms.containsKey("ISREQUIRED"))
131                {
132                        String req=httpReq.getUrlParameter("ISREQUIRED");
133                        if((req==null)&&(last!=null))
134                                req=(last.startsWith("{")&&last.endsWith("}"))?"on":"";
135                        if(req!=null)
136                                str.append((req.equalsIgnoreCase("on")?"CHECKED":"")+", ");
137                }
138                if(parms.containsKey("MASKDESC"))
139                {
140                        String mask=httpReq.getUrlParameter("MASK");
141                        if((mask==null)&&(last!=null)&&(last.length()>0))
142                                mask=CMLib.titles().getAutoTitleMask(last);
143                        if(mask!=null)
144                                str.append(CMLib.masking().maskDesc(mask)+", ");
145                }
146                String strstr=str.toString();
147                if(strstr.endsWith(", "))
148                        strstr=strstr.substring(0,strstr.length()-2);
149                return clearWebMacros(strstr);
150        }
151}
