050        private String raceDropDown(HTTPRequest httpReq, String old)
051        {
052                final StringBuffer str=new StringBuffer("");
053                str.append("<OPTION VALUE=\"\" "+((old.length()==0)?"SELECTED":"")+">None");
054                Race R2=null;
055                String R2ID=null;
056                for(final Enumeration e=MobData.sortedRaces(httpReq);e.hasMoreElements();)
057                {
058                        R2=(Race)e.nextElement();
059                        R2ID="com.planet_ink.coffee_mud.Races."+R2.ID();
060                        if(R2.isGeneric() && CMClass.checkForCMClass(CMObjectType.RACE,R2ID))
061                        {
062                                str.append("<OPTION VALUE=\""+R2.ID()+"\" "+((old.equalsIgnoreCase(R2.ID()))?"SELECTED":"")+">"+R2.ID()+" (Generic)");
063                                str.append("<OPTION VALUE=\""+R2ID+"\" "+((old.equalsIgnoreCase(R2ID))?"SELECTED":"")+">"+R2ID);
064                        }
065                        else
066                        if(R2.isGeneric())
067                                str.append("<OPTION VALUE=\""+R2.ID()+"\" "+((old.equalsIgnoreCase(R2.ID())||old.equalsIgnoreCase(R2ID))?"SELECTED":"")+">"+R2.ID()+" (Generic)");
068                        else
069                                str.append("<OPTION VALUE=\""+R2ID+"\" "+((old.equalsIgnoreCase(R2.ID())||old.equalsIgnoreCase(R2ID))?"SELECTED":"")+">"+R2ID);
070                }
071                return str.toString();
072        }

074        public static StringBuffer estats(PhyStats E, char c, HTTPRequest httpReq, java.util.Map<String,String> parms, int borderSize)
075        {
076                final StringBuffer str=new StringBuffer("");
077                final PairVector<String,String> theclasses=new PairVector<String,String>();
078                if(httpReq.isUrlParameter(c+"ESTATS1"))
079                {
080                        int num=1;
081                        String behav=httpReq.getUrlParameter(c+"ESTATS"+num);
082                        while(behav!=null)
083                        {
084                                if(behav.length()>0)
085                                {
086                                        if(!behav.equalsIgnoreCase("REJUV"))
087                                        {
088                                                String prof=httpReq.getUrlParameter(c+"ESTATSV"+num);
089                                                if(prof==null)
090                                                        prof="0";
091                                                prof=""+CMath.s_int(prof);
092                                                theclasses.addElement(behav,prof);
093                                        }
094                                }
095                                num++;
096                                behav=httpReq.getUrlParameter(c+"ESTATS"+num);
097                        }
098                }
099                else
100                {
101                        for(int i=0;i<E.getStatCodes().length;i++)
102                        {
103                                if(CMath.s_int(E.getStat(E.getStatCodes()[i]))!=0)
104                                        theclasses.addElement(E.getStatCodes()[i],Integer.toString(CMath.s_int(E.getStat(E.getStatCodes()[i]))));
105                        }
106                }
107                str.append("<TABLE WIDTH=100% BORDER="+borderSize+" CELLSPACING=0 CELLPADDING=0>");
108                for(int i=0;i<theclasses.size();i++)
109                {
110                        final String theclass=theclasses.elementAt(i).first;
111                        str.append("<TR><TD WIDTH=35%>");
112                        str.append("<SELECT ONCHANGE=\"EditAffect(this);\" NAME="+c+"ESTATS"+(i+1)+">");
113                        str.append("<OPTION VALUE=\"\">Delete!");
114                        str.append("<OPTION VALUE=\""+theclass+"\" SELECTED>"+theclass);
115                        str.append("</SELECT>");
116                        str.append("</TD>");
117                        str.append("<TD WIDTH=65%>");
118                        str.append("<INPUT TYPE=TEXT NAME="+c+"ESTATSV"+(i+1)+" VALUE=\""+theclasses.elementAt(i).second+"\" SIZE=4 MAXLENGTH=4>");
119                        str.append("</TD>");
120                        str.append("</TR>");
121                }
122                str.append("<TR><TD WIDTH=35%>");
123                str.append("<SELECT ONCHANGE=\"AddAffect(this);\" NAME="+c+"ESTATS"+(theclasses.size()+1)+">");
124                str.append("<OPTION SELECTED VALUE=\"\">Select a stat");
125                for(int i=0;i<E.getStatCodes().length;i++)
126                {
127                        if((CMath.isNumber(E.getStat(E.getStatCodes()[i])))&&(!theclasses.containsFirst(E.getStatCodes()[i])))
128                                str.append("<OPTION VALUE=\""+E.getStatCodes()[i]+"\">"+E.getStatCodes()[i]);
129                }
130                str.append("</SELECT>");
131                str.append("</TD>");
132                str.append("<TD WIDTH=65%>");
133                str.append("<INPUT TYPE=TEXT NAME="+c+"ESTATSV"+(theclasses.size()+1)+" VALUE=\"\" SIZE=4 MAXLENGTH=4>");
134                str.append("</TD>");
135                str.append("</TR>");
136                str.append("</TABLE>");
137                return str;
138        }

269        public static StringBuffer itemList(List<? extends Item> items, char c, HTTPRequest httpReq, java.util.Map<String,String> parms, int borderSize, boolean one)
270        {
271                if(items==null)
272                        items=new Vector<Item>();
273                final StringBuffer str=new StringBuffer("");
274                final Vector<Item> classes=new Vector<Item>();
275                List<Item> itemlist=null;
276                if(httpReq.isUrlParameter(c+"ITEM1"))
277                {
278                        itemlist=RoomData.getItemCache();
279                        for(int i=1;;i++)
280                        {
281                                final String MATCHING=httpReq.getUrlParameter(c+"ITEM"+i);
282                                if(MATCHING==null)
283                                        break;
284                                Item I2=RoomData.getItemFromAnywhere(itemlist,MATCHING);
285                                if(I2==null)
286                                {
287                                        I2=RoomData.getItemFromAnywhere(items,MATCHING);
288                                        if(I2!=null)
289                                                RoomData.contributeItems(new XVector<Item>(I2));
290                                }
291                                if(I2!=null)
292                                        classes.addElement(I2);
293                                if(one)
294                                        break;
295                        }
296                }
297                else
298                {
299                        classes.addAll(items);
300                        itemlist=RoomData.contributeItems(classes);
301                }
302                str.append("<TABLE WIDTH=100% BORDER=\""+borderSize+"\" CELLSPACING=0 CELLPADDING=0>");
303                int numItems=0;
304                if(!one)
305                for(int i=0;i<classes.size();i++)
306                {
307                        numItems++;
308                        final Item I=classes.elementAt(i);
309                        str.append("<TR>");
310                        str.append("<TD WIDTH=90%>");
311                        str.append("<SELECT ONCHANGE=\"AddItem(this);\" NAME="+c+"ITEM"+(numItems)+">");
312                        if(!one)
313                                str.append("<OPTION VALUE=\"\">Delete!");
314                        if(items.contains(I))
315                                str.append("<OPTION SELECTED VALUE=\""+RoomData.getItemCode(classes,I)+"\">"+I.Name()+" ("+I.ID()+")");
316                        else
317                        if(itemlist.contains(I))
318                                str.append("<OPTION SELECTED VALUE=\""+I+"\">"+I.Name()+" ("+I.ID()+")");
319                        else
320                                str.append("<OPTION SELECTED VALUE=\""+I.ID()+"\">"+I.Name()+" ("+I.ID()+")");
321                        str.append("</SELECT>");
322                        str.append("</TD>");
323                        str.append("<TD WIDTH=10%>");
324                        str.append("<INPUT TYPE=BUTTON NAME="+c+"EDITITEM"+(i+1)+" VALUE=EDIT ONCLICK=\"EditItem('"+RoomData.getItemCode(classes,I)+"','"+c+"ITEM"+(numItems)+"');\">");
325                        str.append("</TD></TR>");
326                }
327                str.append("<TR><TD WIDTH=90%>");
328                str.append("<SELECT ONCHANGE=\"AddItem(this);\" NAME="+c+"ITEM"+(numItems+1)+">");
329                if(!one)
330                        str.append("<OPTION SELECTED VALUE=\"\">Select a new Item");
331                for(final Item I : itemlist)
332                {
333                        if(one&&(classes.contains(I)))
334                        {
335                                if(items.contains(I))
336                                        str.append("<OPTION SELECTED VALUE=\""+RoomData.getItemCode(classes,I)+"\">"+I.Name()+" ("+I.ID()+")");
337                                else
338                                        str.append("<OPTION SELECTED VALUE=\""+I+"\">"+I.Name()+" ("+I.ID()+")");
339                        }
340                        else
341                                str.append("<OPTION VALUE=\""+I+"\">"+I.Name()+RoomData.getObjIDSuffix(I));
342                }
343                if(one)
344                {
345                        final List<String> sortMe=new ArrayList();
346                        CMClass.addAllItemClassNames(sortMe,true,true,false,CMProps.getIntVar(CMProps.Int.MUDTHEME));
347                        Collections.sort(sortMe);
348                        Set<String> found=new TreeSet<String>();
349                        for (final Object element : sortMe)
350                        {
351                                boolean selected=false;
352                                for(int x=0;x<classes.size();x++)
353                                {
354                                        if(classes.elementAt(x).ID().equals(element))
355                                        { 
356                                                selected=true;
357                                                found.add(classes.elementAt(x).ID());
358                                                break;
359                                        }
360                                }
361                                str.append("<OPTION "+(selected?"SELECTED":"")+" VALUE=\""+(String)element+"\">"+(String)element);
362                        }
363                        for(int x=0;x<classes.size();x++)
364                        {
365                                if(!found.contains(classes.elementAt(x).ID()))
366                                        str.append("<OPTION SELECTED VALUE=\""+classes.elementAt(x).ID()+"\">"+classes.elementAt(x).ID());
367                        }
368                }
369                else
370                {
371                        StringBuffer mposs=(StringBuffer)Resources.getResource("MUDGRINDER-OTHERPOSS");
372                        if(mposs==null)
373                        {
374                                mposs=new StringBuffer("");
375                                final List<String> sortMe=new ArrayList<String>();
376                                CMClass.addAllItemClassNames(sortMe,true,true,false,CMProps.getIntVar(CMProps.Int.MUDTHEME));
377                                Collections.sort(sortMe);
378                                for (final Object element : sortMe)
379                                        mposs.append("<OPTION VALUE=\""+(String)element+"\">"+(String)element);
380                                Resources.submitResource("MUDGRINDER-OTHERPOSS",mposs);
381                        }
382                        str.append(mposs);
383                }
384                str.append("</SELECT>");
385                str.append("</TD>");
386                str.append("<TD WIDTH=10%>");
387                str.append("<INPUT TYPE=BUTTON NAME="+c+"ADDITEM VALUE=\"NEW\" ONCLICK=\"AddNewItem('"+c+"ITEM"+(numItems+1)+"');\">");
388                str.append("</TD></TR></TABLE>");
389                return str;
390        }

392        public static StringBuffer dynAbilities(final MOB mob, List<Ability> ables, String ID, Modifiable obj, HTTPRequest httpReq, java.util.Map<String,String> parms, int borderSize, String font)
393        {
394                final StringBuffer str=new StringBuffer("");
395                final DVector theclasses=new DVector(6);
396                final boolean supportsRoles=CMParms.contains(obj.getStatCodes(), "GETRABLEROLE");
397                if(httpReq.isUrlParameter("RABLES1"))
398                {
399                        int num=1;
400                        String behav=httpReq.getUrlParameter("RABLES"+num);
401                        while(behav!=null)
402                        {
403                                if(behav.length()>0)
404                                {
405                                        String prof=httpReq.getUrlParameter("RABPOF"+num);
406                                        if(prof==null)
407                                                prof="0";
408                                        String qual=httpReq.getUrlParameter("RABQUA"+num);
409                                        if(qual==null)
410                                                qual="";
411                                        String parm=httpReq.getUrlParameter("RABPRM"+num);
412                                        if(parm==null)
413                                                parm="";
414                                        String levl=httpReq.getUrlParameter("RABLVL"+num);
415                                        if(levl==null)
416                                                levl="0";
417                                        String roles=null;
418                                        if(supportsRoles)
419                                                roles=httpReq.getUrlParameter("RABROL"+num);
420                                        if(roles==null) 
421                                                roles="";
422                                        theclasses.addElement(behav,prof,qual,levl,parm,roles);
423                                }
424                                num++;
425                                behav=httpReq.getUrlParameter("RABLES"+num);
426                        }
427                }
428                else
429                {
430                        QuadVector<String,Integer,Integer,Boolean> cables;
431                        if(obj instanceof Race)
432                                cables=((Race)obj).culturalAbilities();
433                        else
434                                cables=new QuadVector<String,Integer,Integer,Boolean>();
435                        for (final Ability A : ables)
436                        {
437                                if((A!=null)&&(!cables.containsFirst(A.ID())))
438                                {
439                                        AbilityMapper.AbilityMapping ableMap=CMLib.ableMapper().getAbleMap(ID, A.ID());
440                                        final boolean defaultGain = ableMap.autoGain();
441                                        final int qualifyingLevel = ableMap.qualLevel();
442                                        final String defaultParm = ableMap.defaultParm();
443                                        String roles=null;
444                                        if(supportsRoles && (obj instanceof ClanGovernment))
445                                        {
446                                                roles="";
447                                                for(String key : ableMap.extFields().keySet())
448                                                {
449                                                        ClanPosition P=((ClanGovernment)obj).findPositionRole(key);
450                                                        if(P!=null)
451                                                                roles+=", "+P.getID();
452                                                }
453                                                if(roles.length()>2)
454                                                        roles=roles.substring(2);
455                                        }
456                                        if(roles==null) 
457                                                roles="";
458                                        theclasses.addElement(A.ID(),A.proficiency()+"",defaultGain?"":"on",qualifyingLevel+"",defaultParm,roles);
459                                }
460                        }
461                }
462                if(font==null)
463                        font="<FONT COLOR=WHITE><B>";
464                str.append("<TABLE WIDTH=100% BORDER="+borderSize+" CELLSPACING=0 CELLPADDING=0>");
465                for(int i=0;i<theclasses.size();i++)
466                {
467                        final String theclass=(String)theclasses.get(i,1);
468                        str.append("<TR><TD COLSPAN=4><TABLE BORDER=0 WIDTH=100% CELLSPACING=0 CELLPADDING=0><TR><TD WIDTH=35%>");
469                        str.append("<SELECT ONCHANGE=\"EditAffect(this);\" NAME=RABLES"+(i+1)+">");
470                        str.append("<OPTION VALUE=\"\">Delete!");
471                        str.append("<OPTION VALUE=\""+theclass+"\" SELECTED>"+theclass);
472                        str.append("</SELECT>");
473                        str.append("</TD>");
474                        str.append("<TD WIDTH=25%>");
475                        str.append(font+"Lvl:</B></FONT> <INPUT TYPE=TEXT NAME=RABLVL"+(i+1)+" VALUE=\""+theclasses.get(i,4)+"\" SIZE=3 MAXLENGTH=3>");
476                        str.append("</TD>");
477                        str.append("<TD WIDTH=15%>");
478                        str.append(font+"<INPUT TYPE=TEXT NAME=RABPOF"+(i+1)+" VALUE=\""+theclasses.get(i,2)+"\" SIZE=3 MAXLENGTH=3>%</B></I></FONT>");
479                        str.append("</TD>");
480                        str.append("<TD WIDTH=25%>");
481                        str.append("<INPUT TYPE=CHECKBOX NAME=RABQUA"+(i+1)+" "+(theclasses.get(i,3).toString().equalsIgnoreCase("on")?"CHECKED":"")+">"+font+"Qualify Only</B></FONT></I>");
482                        str.append("</TD>");
483                        str.append("</TR>");
484                        str.append("<TR><TD WIDTH=35%>&nbsp;</TD><TD COLSPAN=3>");
485                        str.append(font+"Parms:</B></FONT> <INPUT TYPE=TEXT NAME=RABPRM"+(i+1)+" VALUE=\""+theclasses.get(i,5)+"\" SIZE=40 MAXLENGTH=100>");
486                        str.append("</TD></TR>");
487                        if(supportsRoles)
488                        {
489                                str.append("<TR><TD WIDTH=35%>&nbsp;</TD><TD COLSPAN=3>");
490                                str.append(font+"Roles:</B></FONT> <INPUT TYPE=TEXT NAME=RABROL"+(i+1)+" VALUE=\""+theclasses.get(i,6)+"\" SIZE=40 MAXLENGTH=100>");
491                                str.append("</TD></TR>");
492                        }
493                        str.append("</TABLE></TD></TR>");
494                }
495                str.append("<TR><TD COLSPAN=4><TABLE BORDER=0 WIDTH=100% CELLSPACING=0 CELLPADDING=0><TR><TD WIDTH=35%>");
496                str.append("<SELECT ONCHANGE=\"AddAffect(this);\" NAME=RABLES"+(theclasses.size()+1)+">");
497                str.append("<OPTION SELECTED VALUE=\"\">Select an Ability");
498                for(final Enumeration<Ability> a=CMClass.abilities();a.hasMoreElements();)
499                {
500                        final Ability A=a.nextElement();
501                        final String cnam=A.ID();
