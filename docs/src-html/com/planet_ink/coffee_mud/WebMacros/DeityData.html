037@SuppressWarnings({"unchecked","rawtypes"})
038public class DeityData extends StdWebMacro
039{
040        @Override
041        public String name()
042        {
043                return "DeityData";
044        }
045
046        // valid parms include description, worshipreq, clericreq,
047        // worshiptrig, clerictrig, worshipsintrig,clericsintrig,powertrig
048
049        private List<PlayerLibrary.ThinPlayer> getDeityData(HTTPRequest httpReq, String deityName)
050        {
051                List<PlayerLibrary.ThinPlayer> folData=(List<PlayerLibrary.ThinPlayer>)httpReq.getRequestObjects().get("DEITYDATAFOR-"+deityName.toUpperCase().trim());
052                if(folData!=null)
053                        return folData;
054                folData = CMLib.database().worshippers(deityName);
055                httpReq.getRequestObjects().put("DEITYDATAFOR-"+deityName.toUpperCase().trim(),folData);
056                return folData;
057        }
058
059        @Override
060        public String runMacro(HTTPRequest httpReq, String parm, HTTPResponse httpResp)
061        {
062                final java.util.Map<String,String> parms=parseParms(parm);
063                final String last=httpReq.getUrlParameter("DEITY");
064                if(last==null)
065                        return " @break@";
066                if(last.length()>0)
067                {
068                        Deity D=CMLib.map().getDeity(last);
069                        if(D!=null)
070                        {
071                                final StringBuffer str=new StringBuffer("");
072                                if(parms.containsKey("DESCRIPTION"))
073                                        str.append(D.description()+", ");
074                                if(parms.containsKey("NAME"))
075                                        str.append(D.Name()+", ");
076                                if(parms.containsKey("LOCATION"))
077                                {
078                                        if(D.getStartRoom()==null)
079                                                str.append("Nowhere, ");
080                                        else
081                                                str.append(CMLib.map().getExtendedRoomID(D.getStartRoom())+": "+D.getStartRoom().displayText()+", ");
082                                }
083                                if(parms.containsKey("AREA")&&(D.getStartRoom()!=null))
084                                {
085                                        if(parms.containsKey("ENCODED"))
086                                        {
087                                                try
088                                                {
089                                                        str.append(URLEncoder.encode(D.getStartRoom().getArea().Name(),"UTF-8")+", ");
090                                                }
091                                                catch(final Exception e)
092                                                {
093                                                }
094                                        }
095                                        else
096                                                str.append(D.getStartRoom().getArea().Name()+", ");
097                                }
098                                if(parms.containsKey("ROOM")&&(D.getStartRoom()!=null))
099                                {
100                                        if(parms.containsKey("ENCODED"))
101                                        {
102                                                try
103                                                {
104                                                        str.append(URLEncoder.encode(D.getStartRoom().roomID(),"UTF-8")+", ");
105                                                }
106                                                catch(final Exception e)
107                                                {
108                                                }
109                                        }
110                                        else
111                                                str.append(D.getStartRoom().roomID()+", ");
112                                }
113                                if(parms.containsKey("MOBCODE"))
114                                {
115                                        final String roomID=D.getStartRoom().roomID();
116                                        List<MOB> classes=(List)httpReq.getRequestObjects().get("DEITYLIST-"+roomID);
117                                        if(classes==null)
118                                        {
119                                                classes=new Vector<MOB>();
120                                                Room R=(Room)httpReq.getRequestObjects().get(roomID);
121                                                if(R==null)
122                                                {
123                                                        R=MUDGrinder.getRoomObject(httpReq, roomID);
124                                                        if(R==null)
125                                                                return "No Room?!";
126                                                        final Vector<Deity> restoreDeities = new Vector<Deity>();
127                                                        for(final Enumeration<Deity> e=CMLib.map().deities();e.hasMoreElements();)
128                                                        {
129                                                                final Deity D2 = e.nextElement();
130                                                                if((D2.getStartRoom()!=null)
131                                                                &&(CMLib.map().getExtendedRoomID(D2.getStartRoom()).equalsIgnoreCase(CMLib.map().getExtendedRoomID(R))))
132                                                                        restoreDeities.addElement(D2);
133                                                        }
134                                                        CMLib.map().resetRoom(R);
135                                                        R=MUDGrinder.getRoomObject(httpReq, roomID);
136                                                        for(int d=restoreDeities.size()-1;d>=0;d--)
137                                                        {
138                                                                final Deity D2=restoreDeities.elementAt(d);
139                                                                if(CMLib.map().getDeity(D2.Name())!=null)
140                                                                        restoreDeities.removeElementAt(d);
141                                                        }
142                                                        for(final Enumeration e=restoreDeities.elements();e.hasMoreElements();)
143                                                        {
144                                                                final Deity D2=(Deity)e.nextElement();
145                                                                for(int i=0;i<R.numInhabitants();i++)
146                                                                {
147                                                                        final MOB M=R.fetchInhabitant(i);
148                                                                        if((M instanceof Deity)
149                                                                        &&(M.Name().equals(D2.Name())))
150                                                                                CMLib.map().registerWorldObjectLoaded(R.getArea(),R,M);
151                                                                }
152                                                        }
153                                                        httpReq.getRequestObjects().put(roomID,R);
154                                                        D=CMLib.map().getDeity(last);
155                                                }
156                                                synchronized(("SYNC"+roomID).intern())
157                                                {
158                                                        R=CMLib.map().getRoom(R);
159                                                        for(int m=0;m<R.numInhabitants();m++)
160                                                        {
161                                                                final MOB M=R.fetchInhabitant(m);
162                                                                if(M.isSavable())
163                                                                        classes.add(M);
164                                                        }
165                                                        RoomData.contributeMOBs(classes);
166                                                }
167                                                httpReq.getRequestObjects().put("DEITYLIST-"+roomID,classes);
168                                        }
169                                        if(parms.containsKey("ENCODED"))
170                                        {
171                                                try
172                                                {
173                                                        str.append(URLEncoder.encode(RoomData.getMOBCode(classes,D),"UTF-8")+", ");
174                                                }
175                                                catch(final Exception e)
176                                                {
177                                                }
178                                        }
179                                        else
180                                                str.append(RoomData.getMOBCode(classes,D)+", ");
181                                }
182                                if(parms.containsKey("WORSHIPREQ"))
183                                        str.append(D.getWorshipRequirementsDesc()+", ");
184                                if(parms.containsKey("CLERICREQ"))
185                                        str.append(D.getClericRequirementsDesc()+", ");
186                                if(parms.containsKey("SERVICETRIG"))
187                                        str.append(D.getServiceTriggerDesc()+", ");
188                                if(D.numCurses()>0)
189                                {
190                                        if(parms.containsKey("WORSHIPSINTRIG"))
191                                                str.append(D.getWorshipSinDesc()+", ");
192                                        if(parms.containsKey("CLERICSINTRIG"))
193                                                str.append(D.getClericSinDesc()+", ");
194                                }
195
196                                if(D.numPowers()>0)
197                                if(parms.containsKey("POWERTRIG"))
198                                        str.append(D.getClericPowerupDesc()+", ");
199                                if(D.numBlessings()>0)
200                                {
201                                        if(parms.containsKey("WORSHIPTRIG"))
202                                                str.append(D.getWorshipTriggerDesc()+", ");
203                                        if(parms.containsKey("CLERICTRIG"))
204                                                str.append(D.getClericTriggerDesc()+", ");
205                                }
206                                if(parms.containsKey("NUMFOLLOWERS"))
207                                {
208                                        final List<PlayerLibrary.ThinPlayer> data=getDeityData(httpReq,D.Name());
209                                        final int num=data.size();
210                                        str.append(num+", ");
211                                }
212                                if(parms.containsKey("NUMPRIESTS"))
213                                {
214                                        final List<PlayerLibrary.ThinPlayer> data=getDeityData(httpReq,D.Name());
215                                        int num=0;
216                                        //DV.addElement(username, cclass, ""+level, race);
217                                        for(int d=0;d<data.size();d++)
218                                        {
219                                                final CharClass C=CMClass.getCharClass(data.get(d).charClass());
220                                                if((C!=null)&&(C.baseClass().equalsIgnoreCase("CLERIC")))
221                                                        num++;
222                                        }
223                                        str.append(num+", ");
224                                }
225                                String strstr=str.toString();
226                                if(strstr.endsWith(", "))
227                                        strstr=strstr.substring(0,strstr.length()-2);
228                                return clearWebMacros(strstr);
229                        }
230                }
231                return "";
232        }
233}
