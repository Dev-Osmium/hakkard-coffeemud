079        public static Enumeration<Race> sortedRaces(final HTTPRequest httpReq)
080        {
081                if(httpReq.getRequestObjects() != null)
082                {
083                        if(httpReq.getRequestObjects().containsKey("SYSTEM_SORTED_RACES"))
084                                return ((Vector<Race>)httpReq.getRequestObjects().get("SYSTEM_SORTED_RACES")).elements();
085                }
086                final TreeMap<String,Race> map=new TreeMap<String,Race>();
087                for(final Enumeration<Race> r = CMClass.races();r.hasMoreElements();)
088                {
089                        final Race R=r.nextElement();
090                        map.put(R.name(), R);
091                }
092                final Vector<Race> V=new Vector<Race>(map.size());
093                for(final String raceName : map.keySet())
094                {
095                        V.add(map.get(raceName));
096                }
097                if(httpReq.getRequestObjects() != null)
098                {
099                        httpReq.getRequestObjects().put("SYSTEM_SORTED_RACES",V);
100                }
101                return V.elements();
102        }

451        public static StringBuffer factions(final MOB E, final HTTPRequest httpReq, final java.util.Map<String,String> parms, final int borderSize)
452        {
453                final StringBuffer str=new StringBuffer("");
454                if(parms.containsKey("FACTIONS"))
455                {
456                        final Vector<String> theclasses=new Vector<String>();
457                        final Vector<String> theparms=new Vector<String>();
458                        if(httpReq.isUrlParameter("FACTION1"))
459                        {
460                                int num=1;
461                                String facti=httpReq.getUrlParameter("FACTION"+num);
462                                String theparm=httpReq.getUrlParameter("FACTDATA"+num);
463                                if(theparm==null)
464                                        theparm="";
465                                while((facti!=null)&&(theparm!=null))
466                                {
467                                        if(facti.length()>0)
468                                        {
469                                                theclasses.addElement(facti);
470                                                String t=theparm;
471                                                t=CMStrings.replaceAll(t,"\"","&quot;");
472                                                theparms.addElement(t);
473                                        }
474                                        num++;
475                                        facti=httpReq.getUrlParameter("FACTION"+num);
476                                        theparm=httpReq.getUrlParameter("FACTDATA"+num);
477                                }
478                        }
479                        else
480                        {
481                                for(final Enumeration<Faction> f=CMLib.factions().factions();f.hasMoreElements();)
482                                {
483                                        final Faction F=f.nextElement();
484                                        if(F.showInEditor() && (!E.hasFaction(F.factionID())))
485                                        {
486                                                final int autoDefault = F.findAutoDefault(E);
487                                                if(autoDefault != Integer.MAX_VALUE)
488                                                        E.addFaction(F.factionID(), autoDefault);
489                                        }
490                                }
491                                // the auto factions!
492                                for(final Enumeration<String> e=E.factions();e.hasMoreElements();)
493                                {
494                                        final Faction f=CMLib.factions().getFaction(e.nextElement());
495                                        if(f!=null)
496                                        {
497                                                theclasses.addElement(f.factionID());
498                                                theparms.addElement(Integer.toString(E.fetchFaction(f.factionID())));
499                                        }
500                                }
501                        }
502                        str.append("<TABLE WIDTH=100% BORDER=\""+borderSize+"\" CELLSPACING=0 CELLPADDING=0>");
503                        for(int i=0;i<theclasses.size();i++)
504                        {
505                                final String theclass=theclasses.elementAt(i);
506                                final Faction F=CMLib.factions().getFaction(theclass);
507                                if(F==null)
508                                        continue;
509                                String theparm=theparms.elementAt(i);
510                                str.append("<TR><TD WIDTH=50%>");
511                                str.append("<SELECT ONCHANGE=\"EditFaction(this);\" NAME=FACTION"+(i+1)+">");
512                                str.append("<OPTION VALUE=\"\">Delete!");
513                                str.append("<OPTION VALUE=\""+theclass+"\" SELECTED>"+F.name());
514                                str.append("</SELECT>");
515                                str.append("</TD><TD WIDTH=50%>");
516                                str.append("<SELECT NAME=FACTDATA"+(i+1)+">");
517                                if(theparm.length()==0)
518                                        theparm=""+F.findDefault(E);
519                                final Faction.FRange FR=CMLib.factions().getRange(F.factionID(),CMath.s_int(theparm));
520                                if(FR==null)
521                                        str.append("<OPTION VALUE=\""+CMath.s_int(theparm)+"\">"+CMath.s_int(theparm));
522                                final List<Faction.FRange> sortedRanges = new XVector<Faction.FRange>(F.ranges());
523                                Collections.sort(sortedRanges, new Comparator<Faction.FRange>()
524                                {
525                                        @Override
526                                        public int compare(final FRange o1, final FRange o2)
527                                        {
528                                                return Integer.valueOf((o1.low()+o1.high())/2).compareTo(Integer.valueOf((o2.low()+o2.high())/2));
529                                        }
530                                });
531
532                                for(final Iterator<Faction.FRange> e=sortedRanges.iterator();e.hasNext();)
533                                {
534                                        final Faction.FRange FR2=e.next();
