039public class ClanGovernmentData extends StdWebMacro
040{
041        @Override
042        public String name()
043        {
044                return "ClanGovernmentData";
045        }
046
047        @Override
048        public String runMacro(HTTPRequest httpReq, String parm, HTTPResponse httpResp)
049        {
050                final java.util.Map<String,String> parms=parseParms(parm);
051                final String last=httpReq.getUrlParameter("GOVERNMENT");
052                if(last==null)
053                        return " @break@";
054                if(last.length()>0)
055                {
056                        ClanGovernment G=null;
057                        if(CMath.isInteger(last))
058                        {
059                                final int lastID=CMath.s_int(last);
060                                G=CMLib.clans().getStockGovernment(lastID);
061                        }
062                        if(G==null)
063                        {
064                                final Set<Integer> usedTypeIDs=new HashSet<Integer>();
065                                for(final ClanGovernment G2 : CMLib.clans().getStockGovernments())
066                                        usedTypeIDs.add(Integer.valueOf(G2.getID()));
067                                G=CMLib.clans().createSampleGovernment();
068                                G.setName(httpReq.getUrlParameter("NAME"));
069                                G.setCategory(httpReq.getUrlParameter("CATEGORY"));
070                                for(int i=0;i<CMLib.clans().getStockGovernments().length;i++)
071                                {
072                                        if(!usedTypeIDs.contains(Integer.valueOf(i)))
073                                        {
074                                                G.setID(i);
075                                                break;
076                                        }
077                                }
078                        }
079                        final StringBuffer str=new StringBuffer("");
080
081                        // ******************************************************************************************
082                        // do govt positions FIRST!
083                        // ******************************************************************************************
084                        final List<ClanPosition> posList=new Vector<ClanPosition>();
085                        String posDexStr="0";
086                        int posDex=0;
087                        final Set<Integer> usedRoleIDs=new HashSet<Integer>();
088                        if(!httpReq.isUrlParameter("GPOSID_"+posDexStr))
089                        {
090                                for(final ClanPosition P : G.getPositions())
091                                {
092                                        posList.add(P);
093                                        usedRoleIDs.add(Integer.valueOf(P.getRoleID()));
094                                }
095                        }
096                        else
097                        while(httpReq.isUrlParameter("GPOSID_"+posDexStr) && httpReq.getUrlParameter("GPOSID_"+posDexStr).trim().length()>0)
098                        {
099                                final String oldID=httpReq.getUrlParameter("GPOSID_"+posDexStr);
100                                final String oldName=httpReq.getUrlParameter("GPOSNAME_"+posDexStr);
101                                final String oldPluralName=httpReq.getUrlParameter("GPOSPLURALNAME_"+posDexStr);
102                                final int oldRoleID=CMath.s_int(httpReq.getUrlParameter("GPOSROLEID_"+posDexStr));
103                                usedRoleIDs.add(Integer.valueOf(oldRoleID));
104                                final int oldRank=CMath.s_int(httpReq.getUrlParameter("GPOSRANK_"+posDexStr));
105                                final int oldMax=CMath.s_int(httpReq.getUrlParameter("GPOSMAX_"+posDexStr));
106                                final String oldMask=httpReq.getUrlParameter("GPOSINNERMASK_"+posDexStr);
107                                final String oldIsPublicStr=httpReq.getUrlParameter("GPOSISPUBLIC_"+posDexStr);
108                                final boolean oldIsPublic=oldIsPublicStr==null?false:oldIsPublicStr.equalsIgnoreCase("on");
109                                final Clan.Authority powerFuncs[]=new Clan.Authority[Clan.Function.values().length];
110                                for(int f=0;f<Clan.Function.values().length;f++)
111                                        powerFuncs[f]=Clan.Authority.CAN_NOT_DO;
112                                String authDexStr="";
113                                int authDex=0;
114                                while(httpReq.getUrlParameter("GPOSPOWER_"+posDexStr+"_"+authDexStr)!=null)
115                                {
116                                        final Clan.Function auth = (Clan.Function)CMath.s_valueOf(Clan.Function.values(),httpReq.getUrlParameter("GPOSPOWER_"+posDexStr+"_"+authDexStr));
117                                        powerFuncs[auth.ordinal()]=Clan.Authority.CAN_DO;
118                                        authDex++;
119                                        authDexStr=Integer.toString(authDex);
120                                }
121                                final ClanPosition P=(ClanPosition)CMClass.getCommon("DefaultClanPosition");
122                                P.setID(oldID);
123                                P.setRoleID(oldRoleID);
124                                P.setRank(oldRank);
125                                P.setName(oldName);
126                                P.setPluralName(oldPluralName);
127                                P.setMax(oldMax);
128                                P.setInnerMaskStr(oldMask);
129                                P.setFunctionChart(powerFuncs);
130                                P.setPublic(oldIsPublic);
131                                posList.add(P);
132                                posDex++;
133                                posDexStr=Integer.toString(posDex);
134                        }
135
136                        final String cmpos=httpReq.getUrlParameter("GOVTPOSITION");
137                        ClanPosition gPos = null;
138                        if((cmpos!=null)&&(cmpos.length()>0)&&(CMath.s_int(cmpos)>=0)&&(CMath.s_int(cmpos)<posList.size()))
139                                gPos=posList.get(CMath.s_int(cmpos));
140
141                        if((gPos!=null)&&parms.containsKey("GPOSID_"+cmpos))
142                                str.append(gPos.getID()+", ");
143                        if((gPos!=null)&&parms.containsKey("GPOSROLEID_"+cmpos))
144                                str.append(gPos.getRoleID()+", ");
145                        if((gPos!=null)&&parms.containsKey("GPOSNAME_"+cmpos))
146                                str.append(gPos.getName()+", ");
147                        if((gPos!=null)&&parms.containsKey("GPOSPLURALNAME_"+cmpos))
148                                str.append(gPos.getPluralName()+", ");
149                        if((gPos!=null)&&parms.containsKey("GPOSRANK_"+cmpos))
150                                str.append(gPos.getRank()+", ");
151                        if((gPos!=null)&&parms.containsKey("GPOSMAX_"+cmpos))
152                                str.append(gPos.getMax()+", ");
153                        if((gPos!=null)&&parms.containsKey("GPOSINNERMASK_"+cmpos))
154                                str.append(gPos.getInnerMaskStr()+", ");
155                        if((gPos!=null)&&parms.containsKey("GPOSISPUBLIC_"+cmpos))
156                                str.append(gPos.isPublic()?"checked, ":"");
157                        if((gPos!=null)&&parms.containsKey("GPOSPOWER_"+cmpos+"_"))
158                        {
159                                for(final Clan.Function func : Clan.Function.values())
160                                {
161                                        str.append("<OPTION VALUE=\""+func.toString()+"\"");
162                                        if(gPos.getFunctionChart()[func.ordinal()]==Authority.CAN_DO)
163                                                str.append(" SELECTED");
164                                        str.append(">"+func.toString());
165                                }
166                        }
167                        if(parms.containsKey("GPOSPOWERLIST"))
168                        {
169                                for(final Clan.Function func : Clan.Function.values())
170                                        str.append("<OPTION VALUE=\""+func.toString()+"\">"+func.toString());
171                        }
172
173                        if(parms.containsKey("NEXTPOSITIONID"))
174                        {
175                                for(int i=0;i<posList.size()+10;i++)
176                                {
177                                        if(!usedRoleIDs.contains(Integer.valueOf(i)))
178                                        {
179                                                str.append(i+", ");
180                                                break;
181                                        }
182                                }
183                        }
184
185                        // iterators
186                        if(parms.containsKey("POSITIONSTART"))
187                        {
188                                if(httpReq.getUrlParameter("GOVTPOSITION")!=null)
189                                        httpReq.removeUrlParameter("GOVTPOSITION");
190                                return "";
191                        }
192                        if(parms.containsKey("POSITIONNEXT"))
193                        {
194                                String lastPos="";
195                                for(int p=0;p<posList.size();p++)
196                                {
197                                        if((cmpos==null)||((cmpos.length()>0)&&(cmpos.equals(lastPos))&&(!(""+p).equals(lastPos))))
198                                        {
199                                                httpReq.addFakeUrlParameter("GOVTPOSITION",(""+p));
200                                                return "";
201                                        }
202                                        lastPos=(""+p);
203                                }
204                                httpReq.addFakeUrlParameter("LASTGOVTPOSITION",""+posList.size());
205                                httpReq.addFakeUrlParameter("GOVTPOSITION","");
206                                if(parms.containsKey("EMPTYOK"))
207                                        return "<!--EMPTY-->";
208                                return " @break@";
209                        }
210
211                        if(parms.containsKey("NAME"))
212                        {
213                                String old=httpReq.getUrlParameter("NAME");
214                                if(old==null)
215                                        old=G.getName();
216                                str.append(old+", ");
217                        }
218                        if(parms.containsKey("AUTOROLE"))
219                        {
220                                String old=httpReq.getUrlParameter("AUTOROLE");
221                                if(old==null)
222                                        old=""+G.getAutoRole();
223                                final int autoPos=CMath.s_int(old);
224                                for(final ClanPosition pos : posList)
225                                        str.append("<OPTION VALUE="+pos.getRoleID()+" "+((autoPos==pos.getRoleID())?"SELECTED":"")+">"+pos.getName());
226                        }
227                        if(parms.containsKey("ACCEPTPOS"))
228                        {
229                                String old=httpReq.getUrlParameter("ACCEPTPOS");
230                                if(old==null)
231                                        old=""+G.getAcceptPos();
232                                final int autoPos=CMath.s_int(old);
233                                for(final ClanPosition pos : posList)
234                                        str.append("<OPTION VALUE="+pos.getRoleID()+" "+((autoPos==pos.getRoleID())?"SELECTED":"")+">"+pos.getName());
235                        }
236                        if(parms.containsKey("SHORTDESC"))
237                        {
238                                String old=httpReq.getUrlParameter("SHORTDESC");
239                                if(old==null)
240                                        old=G.getShortDesc();
241                                str.append(old+", ");
242                        }
243                        if(parms.containsKey("CATEGORY"))
244                        {
245                                String old=httpReq.getUrlParameter("CATEGORY");
246                                if(old==null)
247                                        old=G.getCategory();
248                                str.append(old+", ");
249                        }
250                        if(parms.containsKey("REQUIREDMASK"))
251                        {
252                                String old=httpReq.getUrlParameter("REQUIREDMASK");
253                                if(old==null)
254                                        old=G.getRequiredMaskStr();
255                                str.append(old+", ");
256                        }
257                        if(parms.containsKey("ENTRYSCRIPT"))
258                        {
259                                String old=httpReq.getUrlParameter("ENTRYSCRIPT");
260                                if(old==null)
261                                        old=G.getEntryScript();
262                                str.append(old+", ");
263                        }
264                        if(parms.containsKey("EXITSCRIPT"))
265                        {
266                                String old=httpReq.getUrlParameter("EXITSCRIPT");
267                                if(old==null)
268                                        old=G.getExitScript();
269                                str.append(old+", ");
270                        }
271                        if(parms.containsKey("ISPUBLIC"))
272                        {
273                                String old=httpReq.getUrlParameter("ISPUBLIC");
274                                if(old==null)
275                                        old=G.isPublic()?"on":"";
276                                str.append(old.equalsIgnoreCase("on")?"checked, ":"");
277                        }
278                        if(parms.containsKey("ISDEFAULT"))
279                        {
280                                String old=httpReq.getUrlParameter("ISDEFAULT");
281                                if(old==null)
282                                        old=G.isDefault()?"on":"";
283                                str.append(old.equalsIgnoreCase("on")?"checked, ":"");
284                        }
285                        if(parms.containsKey("ISFAMILYONLY"))
286                        {
287                                String old=httpReq.getUrlParameter("ISFAMILYONLY");
288                                if(old==null)
289                                        old=G.isFamilyOnly()?"on":"";
290                                str.append(old.equalsIgnoreCase("on")?"checked, ":"");
291                        }
292                        if(parms.containsKey("OVERRIDEMINMEMBERS"))
293                        {
294                                String old=httpReq.getUrlParameter("OVERRIDEMINMEMBERS");
295                                if(old==null)
296                                        old=G.getOverrideMinMembers()==null?"":G.getOverrideMinMembers().toString();
297                                str.append(old+", ");
298                        }
299                        if(parms.containsKey("CONQUESTENABLED"))
300                        {
301                                String old=httpReq.getUrlParameter("CONQUESTENABLED");
302                                if(old==null)
303                                        old=G.isConquestEnabled()?"on":"";
304                                str.append(old.equalsIgnoreCase("on")?"checked, ":"");
305                        }
306                        if(parms.containsKey("CONQUESTITEMLOYALTY"))
307                        {
308                                String old=httpReq.getUrlParameter("CONQUESTITEMLOYALTY");
309                                if(old==null)
310                                        old=G.isConquestItemLoyalty()?"on":"";
311                                str.append(old.equalsIgnoreCase("on")?"checked, ":"");
312                        }
313                        if(parms.containsKey("CONQUESTDEITYBASIS"))
314                        {
315                                String old=httpReq.getUrlParameter("CONQUESTDEITYBASIS");
316                                if(old==null)
317                                        old=G.isConquestByWorship()?"on":"";
318                                str.append(old.equalsIgnoreCase("on")?"checked, ":"");
319                        }
320                        if(parms.containsKey("ISRIVALROUS"))
321                        {
322                                String old=httpReq.getUrlParameter("ISRIVALROUS");
323                                if(old==null)
324                                        old=G.isRivalrous()?"on":"";
325                                str.append(old.equalsIgnoreCase("on")?"checked, ":"");
326                        }
327                        if(parms.containsKey("MAXVOTEDAYS"))
328                        {
329                                String old=httpReq.getUrlParameter("MAXVOTEDAYS");
330                                if(old==null)
331                                        old=Integer.toString(G.getMaxVoteDays());
332                                str.append(CMath.s_int(old)+", ");
333                        }
334                        if(parms.containsKey("XPLEVELFORMULA"))
335                        {
336                                String old=httpReq.getUrlParameter("XPLEVELFORMULA");
337                                if(old==null)
338                                        old=G.getXpCalculationFormulaStr();
339                                str.append(CMath.s_int(old)+", ");
340                        }
341                        if(parms.containsKey("VOTEQUORUMPCT"))
342                        {
343                                String old=httpReq.getUrlParameter("VOTEQUORUMPCT");
344                                if(old==null)
345                                        old=Integer.toString(G.getVoteQuorumPct());
346                                str.append(CMath.s_int(old)+", ");
347                        }
348                        if(parms.containsKey("AUTOPROMOTEBY"))
349                        {
350                                String old=httpReq.getUrlParameter("AUTOPROMOTEBY");
351                                if(old==null)
352                                        old=""+G.getAutoPromoteBy().toString();
353                                for(final Clan.AutoPromoteFlag flag : Clan.AutoPromoteFlag.values())
354                                        str.append("<OPTION VALUE="+flag.toString()+" "+((old.equals(flag.toString()))?"SELECTED":"")+">"+flag.toString());
355                        }
356                        if(parms.containsKey("VOTEFUNCS"))
357                        {
358                                final String old=httpReq.getUrlParameter("VOTEFUNCS");
359                                final Set<String> voteFuncs=new HashSet<String>();
360                                if(old==null)
361                                {
362                                        if(posList.size()>0)
363                                        {
364                                                final ClanPosition P=posList.get(0);
365                                                for(final Clan.Function func : Clan.Function.values())
366                                                {
367                                                        if(P.getFunctionChart()[func.ordinal()]==Clan.Authority.MUST_VOTE_ON)
368                                                                voteFuncs.add(func.toString());
369                                                }
370                                        }
371                                }
372                                else
373                                {
374                                        voteFuncs.add(old);
375                                        int x=1;
376                                        while(httpReq.getUrlParameter("VOTEFUNCS"+x)!=null)
377                                        {
378                                                voteFuncs.add(httpReq.getUrlParameter("VOTEFUNCS"+x));
379                                                x++;
380                                        }
381                                }
382                                for(final Clan.Function func : Clan.Function.values())
383                                {
384                                        str.append("<OPTION VALUE=\""+func.toString()+"\"");
385                                        if(voteFuncs.contains(func.toString()))
386                                                str.append(" SELECTED");
387                                        str.append(">"+func.toString());
388                                }
389                        }
390                        if(parms.containsKey("LONGDESC"))
391                        {
392                                String old=httpReq.getUrlParameter("LONGDESC");
393                                if(old==null)
394                                        old=G.getLongDesc();
395                                str.append(old+", ");
396                        }
397                        if(parms.containsKey("RABLE"))
398                        {
399                                final MOB mob=Authenticate.getAuthenticatedMob(httpReq);
400                                str.append(RaceData.dynAbilities(mob,G.getClanLevelAbilities(null,null,Integer.valueOf(Integer.MAX_VALUE)),G.getName(),G,httpReq,parms,1,parms.get("FONT"))+", ");
401                        }
402                        if(parms.containsKey("REFFS"))
403                                str.append(RaceData.dynEffects(G.getName(),G,httpReq,parms,1,parms.get("FONT"))+", ");
404
405                        String strstr=str.toString();
406                        if(strstr.endsWith(", "))
407                                strstr=strstr.substring(0,strstr.length()-2);
408                        return clearWebMacros(strstr);
409                }
410                return "";
411        }
