037public class CrossBaseClassAbilities extends StdWebMacro
038{
039        @Override
040        public String name()
041        {
042                return "CrossBaseClassAbilities";
043        }
044
045        @Override
046        public String runMacro(HTTPRequest httpReq, String parm, HTTPResponse httpResp)
047        {
048                final StringBuffer buf=new StringBuffer("");
049                final java.util.Map<String,String> parms=parseParms(parm);
050                if(parms.containsKey("DOMAINSKILLSBYCLASS"))
051                {
052                        final String className=parms.get("CHARCLASS");
053                        if(className==null)
054                                return " @break@";
055                        final String domainStr=parms.get("DOMAIN");
056                        if(domainStr==null)
057                                return " @break@";
058                        final CharClass C=CMClass.findCharClass(className);
059                        if(C==null)
060                                return " @break@";
061                        int domain=CMParms.indexOf(Ability.DOMAIN_DESCS, domainStr.toUpperCase().trim());
062                        if(domain<0)
063                                return " @break@";
064                        domain = domain << 5;
065                        int levelCap=C.getLevelCap();
066                        if(levelCap < 0)
067                                levelCap = CMProps.getIntVar(CMProps.Int.LASTPLAYERLEVEL);
068                        List<AbilityMapper.AbilityMapping> list = CMLib.ableMapper().getUpToLevelListings(C.ID(), levelCap, true, false);
069                        int ct=0;
070                        for(AbilityMapper.AbilityMapping mapping : list)
071                        {
072                                Ability A=(Ability)CMClass.getPrototypeByID(CMClass.CMObjectType.ABILITY, mapping.abilityID());
073                                if(A==null)
074                                        Log.errOut("CrossBase!","Ability not found?! : "+mapping.abilityID());
075                                else
076                                if(((A.classificationCode()&Ability.ALL_DOMAINS)==domain)
077                                &&(!CMLib.ableMapper().getSecretSkill(C.ID(), false, A.ID())))
078                                        ct++;
079                        }
080                        buf.append(Integer.toString(ct));
081                }
082                else
083                if(parms.containsKey("DOMAINSKILLSBYUNASSIGNED"))
084                {
085                        final String domainStr=parms.get("DOMAIN");
086                        if(domainStr==null)
087                                return " @break@";
088                        int domain=CMParms.indexOf(Ability.DOMAIN_DESCS, domainStr.toUpperCase().trim());
089                        if(domain<0)
090                                return " @break@";
091                        domain = domain << 5;
092                        int ct=0;
093                        for(final Enumeration<Ability> a= CMClass.abilities();a.hasMoreElements();)
094                        {
095                                Ability A=a.nextElement();
096                                if(((A.classificationCode()&Ability.ALL_DOMAINS)==domain)
097                                &&(!CMLib.ableMapper().qualifiesByAnyCharClass(A.ID()))
098                                &&(!CMLib.ableMapper().getSecretSkill(A.ID())))
099                                        ct++;
100                        }
101                        buf.append(Integer.toString(ct));
102                }
103                else
104                {
105                        String baseClass=httpReq.getUrlParameter("BASECLASS");
106                        if(baseClass==null) 
107                                return " @break@";
108                        if(baseClass.length()>0)
109                        {
110                                final Vector<String> charClasses=new Vector<String>();
111                                for(final Enumeration<CharClass> c=CMClass.charClasses();c.hasMoreElements();)
112                                {
113                                        final CharClass C=c.nextElement();
114                                        if((CMProps.isTheme(C.availabilityCode()))
115                                        &&(C.baseClass().equals(baseClass))
116                                        &&(!charClasses.contains(C.ID())))
117                                                charClasses.addElement(C.ID());
118                                }
119
120                                final Vector<String> abilities=new Vector<String>();
121                                final Vector<Integer> levelssum=new Vector<Integer>();
122                                final Vector<Integer> numberare=new Vector<Integer>();
123                                for(int c=0;c<charClasses.size();c++)
124                                {
125                                        final String className=charClasses.elementAt(c);
126                                        for(final Enumeration<Ability> a=CMClass.abilities();a.hasMoreElements();)
127                                        {
128                                                final Ability A=a.nextElement();
129                                                final int level=CMLib.ableMapper().getQualifyingLevel(className,true,A.ID());
130                                                if((level>=0)&&(!CMLib.ableMapper().getSecretSkill(className,false,A.ID())))
131                                                {
132                                                        final int dex=abilities.indexOf(A.ID());
133                                                        if(dex<0)
134                                                        {
135                                                                abilities.addElement(A.ID());
136                                                                levelssum.addElement(Integer.valueOf(level));
137                                                                numberare.addElement(Integer.valueOf(1));
138                                                        }
139                                                        else
140                                                        {
141                                                                final Integer I=levelssum.elementAt(dex);
142                                                                levelssum.setElementAt(Integer.valueOf(I.intValue()+level),dex);
143                                                                final Integer I2=numberare.elementAt(dex);
144                                                                numberare.setElementAt(Integer.valueOf(I2.intValue()+1),dex);
145                                                        }
146                                                }
147                                        }
148                                }
149
150                                final Vector<String> sortedAbilities=new Vector<String>();
151                                while(abilities.size()>0)
152                                {
153                                        double lowAvg=Double.MAX_VALUE;
154                                        int lowDex=-1;
155                                        for(int i=0;i<abilities.size();i++)
156                                        {
157                                                final Integer I=levelssum.elementAt(i);
158                                                final Integer I2=numberare.elementAt(i);
159                                                final double avg=CMath.div(I.intValue(),I2.intValue());
160                                                if(avg<lowAvg)
161                                                {
162                                                        lowAvg=avg;
163                                                        lowDex=i;
164                                                }
165                                        }
166                                        if(lowDex>=0)
167                                        {
168                                                sortedAbilities.addElement(abilities.elementAt(lowDex));
169                                                abilities.removeElementAt(lowDex);
170                                                levelssum.removeElementAt(lowDex);
171                                                numberare.removeElementAt(lowDex);
172                                        }
173                                }
174
175                                buf.append("<BR><BR><BR><B><H3>"+baseClass+"</H3></B>\n\r");
176                                buf.append("<TABLE WIDTH=100% CELLSPACING=0 CELLPADDING=0 BORDER=1>\n\r");
177                                buf.append("<TR>");
178                                buf.append("<TD><B><FONT COLOR=WHITE>Skill</FONT></B></TD>");
179                                for(int c=0;c<charClasses.size();c++)
180                                {
181                                        final String charClass=charClasses.elementAt(c);
182                                        buf.append("<TD><B><FONT COLOR=WHITE>"+charClass+"</FONT></B></TD>");
183                                }
184                                buf.append("</TR>\n\r");
185                                for(int a=0;a<sortedAbilities.size();a++)
186                                {
187                                        final String able=sortedAbilities.elementAt(a);
188                                        buf.append("<TR><TD><B><FONT COLOR=WHITE>"+able+"</FONT></B></TD>");
189                                        for(int c=0;c<charClasses.size();c++)
190                                        {
191                                                final String charClass=charClasses.elementAt(c);
192                                                final int level=CMLib.ableMapper().getQualifyingLevel(charClass,true,able);
193                                                if(level>=0)
194                                                        buf.append("<TD><FONT COLOR=CYAN>"+level+"</FONT></TD>");
195                                                else
196                                                        buf.append("<TD><BR></TD>");
197                                        }
198                                        buf.append("</TR>\n\r");
199                                }
200                                buf.append("</TABLE>");
201                        }
202                }
203                return clearWebMacros(buf);
204        }
