041public class AchievementData extends StdWebMacro
042{
043        @Override
044        public String name()
045        {
046                return "AchievementData";
047        }
048
049        public void rebuildTrackers(String tattoo)
050        {
051                for(final Enumeration<MOB> m = CMLib.players().players();m.hasMoreElements();)
052                {
053                        final MOB M=m.nextElement();
054                        if(M.playerStats()!=null)
055                        {
056                                M.playerStats().rebuildAchievementTracker(M, tattoo);
057                        }
058                }
059        }
060        
061        public String deleteAchievement(String tattoo)
062        {
063                Achievement A=CMLib.achievements().deleteAchievement(tattoo);
064                if(A!=null)
065                {
066                        rebuildTrackers(A.getTattoo());
067                        CMLib.achievements().resaveAchievements(A.getTattoo());
068                }
069                return "";
070        }
071
072        @Override
073        public String runMacro(HTTPRequest httpReq, String parm, HTTPResponse httpResp)
074        {
075                if(!CMProps.getBoolVar(CMProps.Bool.MUDSTARTED))
076                        return CMProps.getVar(CMProps.Str.MUDSTATUS);
077                final java.util.Map<String,String> parms=parseParms(parm);
078                final String last=httpReq.getUrlParameter("ACHIEVEMENT");
079                if((last==null)&&(!parms.containsKey("EDIT")))
080                        return " @break@";
081
082                String agentStr = parms.get("AGENT");
083                if(agentStr == null)
084                        agentStr=httpReq.getUrlParameter("AGENT");
085                AccountStats.Agent agent = ((agentStr == null)||(agentStr.length()==0)||(!CMProps.isUsingAccountSystem())) ? 
086                                AccountStats.Agent.PLAYER : (AccountStats.Agent)CMath.s_valueOf(AccountStats.Agent.class, agentStr.toUpperCase().trim());
087                if(agent == null)
088                {
089                        agent = AccountStats.Agent.PLAYER;
090                }
091                
092                if(parms.containsKey("EDIT"))
093                {
094                        final MOB M = Authenticate.getAuthenticatedMob(httpReq);
095                        if(M==null)
096                                return "[authentication error]";
097                        if(!CMSecurity.isAllowed(M,M.location(),CMSecurity.SecFlag.ACHIEVEMENTS))
098                                return "[authentication error]";
099
100                        String row = "";
101
102                        String newTattoo=httpReq.getUrlParameter("TATTOO");
103                        if(newTattoo==null)
104                                return "[missing TATTOO error]";
105                        row=newTattoo.toUpperCase().trim()+"=";
106                        if((last!=null)&&((last.length()==0)&&(CMLib.achievements().getAchievement(newTattoo)!=null)))
107                        {
108                                return "[new achievement tattoo already exists!]";
109                        }
110
111                        String newEvent=httpReq.getUrlParameter("EVENT");
112                        if((newEvent==null)||(!CMStrings.contains(Event.getEventChoices(), newEvent)))
113                                return "[missing EVENT error]";
114                        final Event E=(Event)CMath.s_valueOf(Event.class, newEvent);
115                        row+="EVENT=\""+newEvent+"\" ";
116
117                        String newDisplay=httpReq.getUrlParameter("DISPLAY");
118                        if(newDisplay==null)
119                                return "[missing DISPLAY error]";
120                        row+="DISPLAY=\""+CMStrings.escape(newDisplay)+"\" ";
121                        
122                        String newTitle=httpReq.getUrlParameter("TITLE");
123                        if((newTitle != null)&&(newTitle.length()>0))
124                                row+="TITLE=\""+CMStrings.escape(newTitle)+"\" ";
125
126                        String newRewards=httpReq.getUrlParameter("REWARDS");
127                        if((newRewards != null)&&(newRewards.length()>0))
128                                row+="REWARDS=\""+CMStrings.escape(newRewards)+"\" ";
129
130                        for(String s : E.getParameters())
131                        {
132                                if(!CMStrings.contains(AchievementLibrary.BASE_ACHIEVEMENT_PARAMETERS, s))
133                                {
134                                        String newValue=httpReq.getUrlParameter(s);
135                                        if((newValue != null)&&(newValue.length()>0))
136                                        {
137                                                row+=s+"=\""+CMStrings.escape(newValue)+"\" ";
138                                        }
139                                }
140                        }
141
142                        String error=CMLib.achievements().evaluateAchievement(agent,row,false);
143                        if(error!=null)
144                                return "[error: "+error+"]";
145
146                        if(!parms.containsKey("CHECKONLY"))
147                        {
148                                if((last!=null)&&(CMLib.achievements().getAchievement(last)!=null))
149                                {
150                                        final String err=deleteAchievement(last);
151                                        if((err!=null)&&(err.length()>0))
152                                        {
153                                                return err;
154                                        }
155                                }
156                        }
157
158                        error=CMLib.achievements().evaluateAchievement(agent,row,true);
159                        if((error!=null)&&(error.length()>0))
160                                return "[error: "+error+"]";
161                        if(!parms.containsKey("CHECKONLY"))
162                        {
163                                CMLib.achievements().resaveAchievements(last);
164                        }
165                }
166                else
167                if(parms.containsKey("DELETE"))
168                {
169                        final MOB M = Authenticate.getAuthenticatedMob(httpReq);
170                        if(M==null)
171                                return "[authentication error]";
172                        if(!CMSecurity.isAllowed(M,M.location(),CMSecurity.SecFlag.ACHIEVEMENTS))
173                                return "[authentication error]";
174                        if(last==null)
175                                return " @break@";
176                        if(CMLib.achievements().getAchievement(last)==null)
177                                return "Unknown title!";
178                        final String err=deleteAchievement(last);
179                        if((err==null)||(err.length()==0))
180                                return "Achievement deleted.";
181                        return err;
182                }
183                else
184                if(last==null)
185                        return " @break@";
186                final StringBuffer str=new StringBuffer("");
187                
188                Achievement A=CMLib.achievements().getAchievement(last);
189                Event E;
190                String eventName=httpReq.getUrlParameter("EVENT");
191                if(eventName==null)
192                        eventName=(A!=null)?A.getEvent().name():"KILLS";
193                E=(Event)CMath.s_valueOf(Event.class, eventName);
194                if(E==null)
195                        E=Event.KILLS;
196
197                if(parms.containsKey("TATTOO"))
198                {
199                        String value=httpReq.getUrlParameter("TATTOO");
200                        if((value==null)&&(A!=null))
201                                value=A.getTattoo();
202                        if(value!=null)
203                                str.append(CMStrings.replaceAll(value,"\"","&quot;")+", ");
204                }
205                if(parms.containsKey("EVENT"))
206                {
207                        if(eventName!=null)
208                                str.append(CMStrings.replaceAll(eventName,"\"","&quot;")+", ");
209                }
210                if(parms.containsKey("HELP") && parms.containsKey("FIELD"))
211                {
212                        String field=parms.get("FIELD");
213                        @SuppressWarnings("unchecked")
214                        Map<String,Map<String,String>> map=(Map<String,Map<String,String>>)httpReq.getRequestObjects().get("SYSTEM_ACHIEVEMENT_HELP_MAP");
215                        if(map == null)
216                        {
217                                map = CMLib.achievements().getAchievementsHelpMap();
218                                httpReq.getRequestObjects().put("SYSTEM_ACHIEVEMENT_HELP_MAP",map);
219                        }
220                        String help = CMLib.achievements().getAchievementsHelpFromMap(map, E, field);
221                        if(help != null)
222                        {
223                                str.append("<PRE>");
224                                str.append(CMStrings.deEscape(help).replaceAll("\n\r","\n"));
225                                str.append("</PRE>");
226                                str.append(", ");
227                        }
228                }
229                if(parms.containsKey("EVENTOPTIONS"))
230                {
231                        StringBuilder s=new StringBuilder("");
232                        for(Event E2 : Event.values())
233                        {
234                                s.append("<OPTION VALUE="+E2.name()+" ");
235                                if(E2==E)
236                                        s.append("SELECTED ");
237                                s.append(">"+E2.name());
238                        }
239                        str.append(s.toString()+", ");
240                }
241                if(parms.containsKey("DISPLAY"))
242                {
243                        String value=httpReq.getUrlParameter("DISPLAY");
244                        if((value==null)&&(A!=null))
245                                value=A.getDisplayStr();
246                        if(value!=null)
247                                str.append(CMStrings.replaceAll(value,"\"","&quot;")+", ");
248                }
249                if(parms.containsKey("TITLE"))
250                {
251                        String value=httpReq.getUrlParameter("TITLE");
252                        if((value==null)&&(A!=null))
253                        {
254                                for(Award award: A.getRewards())
255                                {
256                                        if(award.getType()==AwardType.TITLE)
257                                        {
258                                                value=((TitleAward)award).getTitle();
259                                        }
260                                }
261                        }
262                        if(value!=null)
263                                str.append(CMStrings.replaceAll(value,"\"","&quot;")+", ");
264                }
265                if(parms.containsKey("REWARDS"))
266                {
267                        String value=httpReq.getUrlParameter("REWARDS");
268                        if((value==null)&&(A!=null))
269                                value=CMLib.achievements().getAwardString(A.getRewards());
270                        if(value!=null)
271                                str.append(CMStrings.replaceAll(value,"\"","&quot;")+", ");
272                }
273                if(parms.containsKey("COUNT"))
274                {
275                        String value=httpReq.getUrlParameter("COUNT");
276                        if((value==null)&&(A!=null))
277                                value=""+A.getTargetCount();
278                        if(value!=null)
279                                str.append(CMStrings.replaceAll(value,"\"","&quot;")+", ");
280                }
281                if(parms.containsKey("MISC"))
282                {
283                        String value=httpReq.getUrlParameter("MISC");
284                        if((value==null)&&(A!=null))
285                        {
286                                value="";
287                                for(String otherParmName : E.getParameters())
288                                {
289                                        if(!CMStrings.contains(AchievementLibrary.BASE_ACHIEVEMENT_PARAMETERS, otherParmName))
290                                        {
291                                                value += CMStrings.deEscape(A.getRawParmVal(otherParmName))+" ";
292                                        }
293                                }
294                        }
295                        if(value!=null)
296                                str.append(CMStrings.replaceAll(value,"\"","&quot;")+", ");
297                }
298                if(parms.containsKey("OTHERPARMNEXT"))
299                {
300                        final String lastOtherParm=httpReq.getUrlParameter("OTHERPARM");
301                        if(parms.containsKey("RESET"))
302                        {
303                                if(lastOtherParm!=null)
304                                        httpReq.removeUrlParameter("OTHERPARM");
305                                return "";
306                        }
307                        String lastOtherParmID="";
308                        for(String otherParmName : E.getParameters())
309                        {
310                                if(!CMStrings.contains(AchievementLibrary.BASE_ACHIEVEMENT_PARAMETERS, otherParmName))
311                                {
312                                        if((lastOtherParm==null)||((lastOtherParm.length()>0)&&(lastOtherParm.equals(lastOtherParmID))&&(!otherParmName.equals(lastOtherParmID))))
313                                        {
314                                                httpReq.addFakeUrlParameter("OTHERPARM",otherParmName);
315                                                httpReq.addFakeUrlParameter("OTHERPARMFIELD",CMStrings.capitalizeAndLower(otherParmName));
316                                                return "";
317                                        }
318                                        lastOtherParmID=otherParmName;
319                                }
320                        }
321                        httpReq.addFakeUrlParameter("OTHERPARM","");
322                        if(parms.containsKey("EMPTYOK"))
323                                return "<!--EMPTY-->";
324                        return " @break@";
325                }
326                if(parms.containsKey("PLAYERPROGRESS"))
327                {
328                        if(A == null)
329                                return "[no achievement error]";
330                        final String playerID=httpReq.getUrlParameter("PLAYER");
331                        if((playerID!=null)&&(playerID.length()>0))
332                        {
333                                final MOB M=CMLib.players().getLoadPlayer(playerID);
334                                if(M!=null)
335                                {
336                                        PlayerStats pStats = M.playerStats();
337                                        if(pStats!=null)
338                                        {
339                                                AchievementLibrary.Tracker T = pStats.getAchievementTracker(A, M);
340                                                str.append(""+T.getCount(M)).append(", ");
341                                        }
342                                        else
343                                                return "[bad player error]";
344                                }
345                                else
346                                        return "[unknown player error]";
347                        }
348                        else
349                                return "[player reference error]";
350                        
351                }
352                
353                if(parms.containsKey("ISPLAYERACHIEVED") || parms.containsKey("ISPLAYERPROGRESS"))
354                {
355                        if(A == null)
356                                return "[no achievement error]";
357                        final String playerID=httpReq.getUrlParameter("PLAYER");
358                        if((playerID!=null)&&(playerID.length()>0))
359                        {
360                                final MOB M=CMLib.players().getLoadPlayer(playerID);
361                                if(M!=null)
362                                {
363                                        PlayerStats pStats = M.playerStats();
364                                        if(pStats!=null)
365                                        {
366                                                AchievementLibrary.Tracker T = pStats.getAchievementTracker(A, M);
367                                                if(parms.containsKey("ISPLAYERACHIEVED"))
368                                                {
369                                                        final boolean achieved = M.findTattoo(A.getTattoo()) != null;
370                                                        if(parms.containsKey("ISPLAYERPROGRESS"))
371                                                                str.append((achieved || (T.getCount(M) != 0)) ? "true" : "false").append(", ");
372                                                        else
373                                                                str.append(achieved ? "true" : "false").append(", ");
374                                                }
375                                                else
376                                                        str.append(T.getCount(M) != 0 ? "true" : "false").append(", ");
377                                        }
378                                        else
379                                                return "[bad player error]";
380                                }
381                                else
382                                        return "[unknown player error]";
383                        }
384                        else
385                                return "[player reference error]";
386                }
387                for(String otherParmName : E.getParameters())
388                {
389                        if(!CMStrings.contains(AchievementLibrary.BASE_ACHIEVEMENT_PARAMETERS, otherParmName))
390                        {
391                                if(parms.containsKey(otherParmName))
392                                {
393                                        String newValue=httpReq.getUrlParameter(otherParmName);
394                                        if((newValue==null)&&(A!=null))
395                                                newValue=CMStrings.deEscape(A.getRawParmVal(otherParmName));
396                                        if(newValue!=null)
397                                                str.append(CMStrings.replaceAll(newValue,"\"","&quot;")+", ");
398                                }
399                        }
400                }
401                String strstr=str.toString();
402                if(strstr.endsWith(", "))
403                        strstr=strstr.substring(0,strstr.length()-2);
404                return clearWebMacros(strstr);
405        }
