036public class ChannelBackLogNext extends StdWebMacro
037{
038        @Override
039        public String name()
040        {
041                return "ChannelBackLogNext";
042        }
043
044        @Override
045        public String runMacro(HTTPRequest httpReq, String parm, HTTPResponse httpResp)
046        {
047                final java.util.Map<String,String> parms=parseParms(parm);
048                String last=httpReq.getUrlParameter("CHANNELBACKLOG");
049                String pageNumStr=httpReq.getUrlParameter("CHANNELBACKLOGPAGE");
050                String pageSizeStr=httpReq.getUrlParameter("CHANNELBACKLOGPAGESIZE");
051                if(parms.containsKey("RESET"))
052                {
053                        if(last!=null)
054                                httpReq.removeUrlParameter("CHANNELBACKLOG");
055                        return "";
056                }
057                final String channel=httpReq.getUrlParameter("CHANNEL");
058                if(channel==null)
059                        return " @break@";
060                final int channelInt=CMLib.channels().getChannelIndex(channel);
061                if(channelInt<0)
062                        return " @break@";
063                httpReq.addFakeUrlParameter("CHANNELBACKLOGNEXTPAGE", "false");
064                final MOB mob = Authenticate.getAuthenticatedMob(httpReq);
065                if(mob!=null)
066                {
067                        if(CMLib.channels().mayReadThisChannel(mob,channelInt,true))
068                        {
069                                int pageSize = CMath.s_int(pageSizeStr);
070                                if(pageSize <=0)
071                                        pageSize = 100;
072                                int pageNum = CMath.s_int(pageNumStr); // page 0 is OK
073                                @SuppressWarnings("unchecked")
074                                List<ChannelsLibrary.ChannelMsg> que=(List<ChannelsLibrary.ChannelMsg>)httpReq.getRequestObjects().get("CHANNELMSG_"+channelInt+" QUE "+pageSize+" "+pageNum);
075                                if(que==null)
076                                {
077                                        final List<ChannelsLibrary.ChannelMsg> oldQue=CMLib.channels().getChannelQue(channelInt,pageNum * pageSize, pageSize);
078                                        que=new ArrayList<ChannelsLibrary.ChannelMsg>(oldQue.size());
079                                        que.addAll(oldQue);
080                                        httpReq.getRequestObjects().put("CHANNELMSG_"+channelInt+" QUE "+pageSize+" "+pageNum,que);
081                                }
082                                httpReq.addFakeUrlParameter("CHANNELBACKLOGNEXTPAGE", Boolean.toString(que.size()>=pageSize));
083
084                                final long now=System.currentTimeMillis();
085                                long elapsedTime;
086                                while(true)
087                                {
088                                        final int num=CMath.s_int(last);
089                                        final int doNum=que.size()-num-1;
090                                        last=""+(num+1);
091                                        httpReq.addFakeUrlParameter("CHANNELBACKLOG",last);
092                                        if((doNum<0)||(doNum>=que.size()))
093                                        {
094                                                httpReq.addFakeUrlParameter("CHANNELBACKLOG","");
095                                                if(parms.containsKey("EMPTYOK"))
096                                                        return "<!--EMPTY-->";
097                                                return " @break@";
098                                        }
099                                        final boolean areareq=CMLib.channels().getChannel(channelInt).flags().contains(ChannelsLibrary.ChannelFlag.SAMEAREA);
100
101                                        final ChannelsLibrary.ChannelMsg cmsg=que.get(doNum);
102                                        final CMMsg msg=cmsg.msg();
103                                        String str=null;
104                                        if((mob==msg.source())&&(msg.sourceMessage()!=null))
105                                                str=msg.sourceMessage();
106                                        else
107                                        if((mob==msg.target())&&(msg.targetMessage()!=null))
108                                                str=msg.targetMessage();
109                                        else
110                                        if(msg.othersMessage()!=null)
111                                                str=msg.othersMessage();
112                                        else
113                                                str="";
114                                        str=CMStrings.removeColors(str);
115                                        elapsedTime=now-cmsg.sentTimeMillis();
116                                        elapsedTime=Math.round(elapsedTime/1000L)*1000L;
117                                        if(elapsedTime<0)
118                                        {
119                                                Log.errOut("Channel","Wierd elapsed time: now="+now+", then="+cmsg.sentTimeMillis());
120                                                elapsedTime=0;
121                                        }
122                                        str += " ("+CMLib.time().date2SmartEllapsedTime(elapsedTime,false)+" ago)";
123                                        if(CMLib.channels().mayReadThisChannel(msg.source(),areareq,mob,channelInt,true))
124                                                return clearWebMacros(CMLib.coffeeFilter().fullOutFilter(mob.session(),mob,msg.source(),msg.target(),msg.tool(),CMStrings.removeColors(str),false));
125                                }
126                        }
127                        return "";
128                }
129                return "";
130        }
131}
