036@SuppressWarnings({"unchecked","rawtypes"})
037public class ExitData extends StdWebMacro
038{
039        @Override
040        public String name()
041        {
042                return "ExitData";
043        }
044
045        private static final String[] okparms={
046                "NAME","CLASSES","DISPLAYTEXT","DESCRIPTION",
047                "LEVEL","LEVELRESTRICTED","ISTRAPPED","HASADOOR",
048                "CLOSEDTEXT","DEFAULTSCLOSED","OPENWORD","CLOSEWORD",
049                "HASALOCK","DEFAULTSLOCKED","KEYNAME","ISREADABLE",
050                "READABLETEXT","ISCLASSRESTRICTED","RESTRICTEDCLASSES",
051                "ISALIGNMENTRESTRICTED","RESTRICTEDALIGNMENTS",
052                "MISCTEXT","ISGENERIC","DOORNAME","IMAGE","OPENTICKS"};
053        public static String dispositions(Physical P,
054                                                                          boolean firstTime,
055                                                                          HTTPRequest httpReq,
056                                                                          java.util.Map<String,String> parms)
057        {
058                final StringBuffer str=new StringBuffer("");
059                for(int d=0;d<PhyStats.IS_CODES.length;d++)
060                {
061                        if(parms.containsKey(PhyStats.IS_CODES[d]))
062                        {
063                                String parm=httpReq.getUrlParameter(PhyStats.IS_CODES[d]);
064                                if(firstTime)
065                                        parm=(((P.basePhyStats().disposition()&(1<<d))>0)?"on":"");
066                                if((parm!=null)&&(parm.length()>0))
067                                        str.append("checked");
068                        }
069                }
070                return str.toString();
071        }
072
073        @Override
074        public String runMacro(HTTPRequest httpReq, String parm, HTTPResponse httpResp)
075        {
076                final java.util.Map<String,String> parms=parseParms(parm);
077
078                final String last=httpReq.getUrlParameter("ROOM");
079                if(last==null)
080                        return " @break@";
081                Room R=(Room)httpReq.getRequestObjects().get(last);
082                if(R==null)
083                {
084                        R=MUDGrinder.getRoomObject(httpReq, last);
085                        if(R==null)
086                                return "No Room?!";
087                        httpReq.getRequestObjects().put(last,R);
088                }
089
090                if(!CMProps.getBoolVar(CMProps.Bool.MUDSTARTED))
091                        return CMProps.getVar(CMProps.Str.MUDSTATUS);
092
093                final String linkdir=httpReq.getUrlParameter("LINK");
094                if(linkdir==null)
095                        return "@break@";
096                final int link=CMLib.directions().getGoodDirectionCode(linkdir);
097                if((link<0)||(link>=Directions.NUM_DIRECTIONS()))
098                        return " @break@";
099
100                Exit X=R.getRawExit(link);
101
102                // important generic<->non generic swap!
103                final String newClassID=httpReq.getUrlParameter("CLASSES");
104                if((newClassID!=null)&&(!newClassID.equals(CMClass.classID(X))))
105                                X=CMClass.getExit(newClassID);
106
107                final boolean firstTime=(!httpReq.isUrlParameter("ACTION"))
108                                        ||(!httpReq.getUrlParameter("ACTION").equals("MODIFYEXIT"))
109                                        ||(((httpReq.isUrlParameter("CHANGEDCLASS"))&&(httpReq.getUrlParameter("CHANGEDCLASS")).equals("true")));
110
111                if(X==null)
112                        return "@break@";
113
114                final StringBuffer str=new StringBuffer("");
115                for(int o=0;o<okparms.length;o++)
116                if(parms.containsKey(okparms[o]))
117                {
118                        String old=httpReq.getUrlParameter(okparms[o]);
119                        if(old==null)
120                                old="";
121                        switch(o)
122                        {
123                        case 0: // name
124                                if(firstTime)
125                                        old=X.Name();
126                                str.append(old);
127                                break;
128                        case 1: // classes
129                                {
130                                        if(firstTime)
131                                                old=CMClass.classID(X);
132                                        Object[] sorted=(Object[])Resources.getResource("MUDGRINDER-EXITS");
133                                        if(sorted==null)
134                                        {
135                                                final Vector<String> sortMe=new Vector<String>();
136                                                for(final Enumeration e=CMClass.exits();e.hasMoreElements();)
137                                                        sortMe.addElement(CMClass.classID(e.nextElement()));
138                                                sorted=(new TreeSet(sortMe)).toArray();
139                                                Resources.submitResource("MUDGRINDER-EXITS",sorted);
140                                        }
141                                        for (final Object element : sorted)
142                                        {
143                                                final String cnam=(String)element;
144                                                str.append("<OPTION VALUE=\""+cnam+"\"");
145                                                if(old.equals(cnam))
146                                                        str.append(" SELECTED");
147                                                str.append(">"+cnam);
148                                        }
149                                }
150                                break;
151                        case 2: // displaytext
152                                if(firstTime)
153                                        old=X.displayText();
154                                str.append(old);
155                                break;
156                        case 3: // description
157                                if(firstTime)
158                                        old=X.description();
159                                str.append(old);
160                                break;
161                        case 4: // level
162                                if(firstTime)
163                                        old=""+X.basePhyStats().level();
164                                str.append(old);
165                                break;
166                        case 5: // levelrestricted;
167                                break;
168                        case 6: // istrapped
169                                break;
170                        case 7: // hasadoor
171                                if(firstTime)
172                                        old=X.hasADoor()?"checked":"";
173                                else
174                                if(old.equals("on"))
175                                        old="checked";
176                                str.append(old);
177                                break;
178                        case 8: // closedtext
179                                if(firstTime)
180                                        old=X.closedText();
181                                if(old.length()==0)
182                                        old="a closed door";
183                                str.append(old);
184                                break;
185                        case 9: // defaultsclosed
186                                if(firstTime)
187                                        old=X.defaultsClosed()?"checked":"";
188                                else
189                                if(old.equals("on"))
190                                        old="checked";
191                                str.append(old);
192                                break;
193                        case 10: // openword
194                                if(firstTime)
195                                        old=X.openWord();
196                                if(old.length()==0)
197                                        old="open";
198                                str.append(old);
199                                break;
200                        case 11: // closeword
201                                if(firstTime)
202                                        old=X.closeWord();
203                                if(old.length()==0)
204                                        old="close";
205                                str.append(old);
206                                break;
207                        case 12: // hasalock
208                                if(firstTime)
209                                        old=X.hasALock()?"checked":"";
210                                else
211                                if(old.equals("on"))
212                                        old="checked";
213                                str.append(old);
214                                break;
215                        case 13: // defaultslocked
216                                if(firstTime)
217                                        old=X.defaultsLocked()?"checked":"";
218                                else
219                                if(old.equals("on"))
220                                        old="checked";
221                                str.append(old);
222                                break;
223                        case 14: // keyname
224                                if(firstTime)
225                                        old=X.keyName();
226                                str.append(old);
227                                break;
228                        case 15: // isreadable
229                                if(firstTime)
230                                        old=X.isReadable()?"checked":"";
231                                else
232                                if(old.equals("on"))
233                                        old="checked";
234                                str.append(old);
235                                break;
236                        case 16: // readable text
237                                if(firstTime)
238                                        old=X.readableText();
239                                str.append(old);
240                                break;
241                        case 17: // isclassrestricuted
242                                break;
243                        case 18: // restrictedclasses
244                                break;
245                        case 19: // isalignmentrestricuted
246                                break;
247                        case 20: // restrictedalignments
248                                break;
249                        case 21: // misc text
250                                if(firstTime)
251                                        old=X.text();
252                                str.append(old);
253                                break;
254                        case 22: // is generic
255                                if(X.isGeneric())
256                                        return "true";
257                                return "false";
258                        case 23: // door name
259                                if(firstTime)
260                                        old=X.doorName();
261                                if(old.length()==0)
262                                        old="door";
263                                str.append(old);
264                                break;
265                        case 24: // image
266                                if(firstTime)
267                                        old=X.rawImage();
268                                str.append(old);
269                                break;
270                        case 25: // open ticks
271                                if((firstTime)||(old.length()==0))
272                                        old=Integer.toString(X.openDelayTicks());
273                                str.append(old);
274                                break;
275                        }
276                        if(firstTime)
277                                httpReq.addFakeUrlParameter(okparms[o],old.equals("checked")?"on":old);
278
279                }
280                str.append(ExitData.dispositions(X,firstTime,httpReq,parms));
281                str.append(AreaData.affects(X,httpReq,parms,1));
282                str.append(AreaData.behaves(X,httpReq,parms,1));
283                X.recoverPhyStats();
284                X.text();
285
286                String strstr=str.toString();
287                if(strstr.endsWith(", "))
288                        strstr=strstr.substring(0,strstr.length()-2);
289                return clearWebMacros(strstr);
290        }
291}
