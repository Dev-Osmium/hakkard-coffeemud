037public class PollData extends StdWebMacro
038{
039        @Override
040        public String name()
041        {
042                return "PollData";
043        }
044
045        @Override
046        public String runMacro(HTTPRequest httpReq, String parm, HTTPResponse httpResp)
047        {
048                final java.util.Map<String,String> parms=parseParms(parm);
049                String last=httpReq.getUrlParameter("POLL");
050                if((last==null)&&(!parms.containsKey("EDIT")))
051                        return " @break@";
052
053                final String replaceCommand=httpReq.getUrlParameter("REPLACE");
054                if((replaceCommand != null)
055                && (replaceCommand.length()>0)
056                && (replaceCommand.indexOf('=')>0))
057                {
058                        final int eq=replaceCommand.indexOf('=');
059                        final String field=replaceCommand.substring(0,eq);
060                        final String value=replaceCommand.substring(eq+1);
061                        httpReq.addFakeUrlParameter(field, value);
062                        httpReq.addFakeUrlParameter("REPLACE","");
063                }
064
065                if(parms.containsKey("EDIT"))
066                {
067                        final MOB M = Authenticate.getAuthenticatedMob(httpReq);
068                        if(M==null)
069                                return "[authentication error]";
070                        if(!CMSecurity.isAllowed(M,M.location(),CMSecurity.SecFlag.POLLS))
071                                return "[authentication error]";
072                        final String newTitle=httpReq.getUrlParameter("TITLE");
073                        if((newTitle==null)||(newTitle.length()==0))
074                                return "[missing data error]";
075
076                        if((last!=null)&&(last.length()==0)&&(CMLib.polls().getPoll(newTitle)!=null))
077                                return "[new title already exists!]";
078                        Poll P=CMLib.polls().getPoll(last);
079                        boolean create=false;
080                        if(P==null)
081                        {
082                                P=(Poll)CMClass.getCommon("DefaultPoll");
083                                create=true;
084                        }
085
086                        String old=httpReq.getUrlParameter("TITLE");
087                        if(old!=null)
088                                P.setName(old);
089                        old=httpReq.getUrlParameter("SUBJECT");
090                        if(old!=null)
091                                P.setSubject(old);
092                        old=httpReq.getUrlParameter("DESCRIPTION");
093                        if(old!=null)
094                                P.setDescription(old);
095                        old=httpReq.getUrlParameter("QUALZAPPER");
096                        if(old!=null)
097                                P.setQualZapper(old);
098                        long flag=0;
099                        old=httpReq.getUrlParameter("ISACTIVE");
100                        if((old!=null)&&(old.equalsIgnoreCase("on")))
101                                flag|=Poll.FLAG_ACTIVE;
102                        old=httpReq.getUrlParameter("ISPREVIEWRESULTS");
103                        if((old!=null)&&(old.equalsIgnoreCase("on")))
104                                flag|=Poll.FLAG_PREVIEWRESULTS;
105                        old=httpReq.getUrlParameter("ISABSTAIN");
106                        if((old!=null)&&(old.equalsIgnoreCase("on")))
107                                flag|=Poll.FLAG_ABSTAIN;
108                        old=httpReq.getUrlParameter("ISVOTEBYIP");
109                        if((old!=null)&&(old.equalsIgnoreCase("on")))
110                                flag|=Poll.FLAG_VOTEBYIP;
111                        old=httpReq.getUrlParameter("ISHIDERESULTS");
112                        if((old!=null)&&(old.equalsIgnoreCase("on")))
113                                flag|=Poll.FLAG_HIDERESULTS;
114                        old=httpReq.getUrlParameter("ISNOTATLOGIN");
115                        if((old!=null)&&(old.equalsIgnoreCase("on")))
116                                flag|=Poll.FLAG_NOTATLOGIN;
117                        P.setFlags(flag);
118                        old=httpReq.getUrlParameter("DOESEXPIRE");
119                        if((old==null)||(!old.equalsIgnoreCase("on")))
120                                P.setExpiration(0);
121                        else
122                        {
123                                String AP="AM";
124                                int hr=CMath.s_int(httpReq.getUrlParameter("HOUR"));
125                                if(hr>12)
126                                {
127                                        hr-=12;
128                                        AP="PM";
129                                }
130                                P.setExpiration(CMLib.time().string2Date(
131                                   (CMath.s_int(httpReq.getUrlParameter("MONTH"))+1)
132                                   +"/"+httpReq.getUrlParameter("DAY")
133                                   +"/"+httpReq.getUrlParameter("YEAR")
134                                   +" "+hr
135                                   +":"+httpReq.getUrlParameter("MINUTE")
136                                   +" "+AP).getTimeInMillis());
137                        }
138                        P.getOptions().clear();
139                        int num=0;
140                        while(httpReq.isUrlParameter("OPTION"+(++num)))
141                        {
142                                old=httpReq.getUrlParameter("OPTION"+num);
143                                if((old!=null)&&(old.trim().length()>0))
144                                        P.getOptions().add(new Poll.PollOption(old));
145                        }
146                        if(create)
147                        {
148                                Log.sysOut(M.Name()+" created poll "+P.getName());
149                                CMLib.polls().createPoll(P);
150                                return "Poll "+P.getName()+" created";
151                        }
152                        CMLib.polls().updatePoll(last, P);
153                        Log.sysOut(M.Name()+" updated poll "+P.getName());
154                        return "Poll "+last+" updated";
155                }
156                else
157                if(parms.containsKey("DELETE"))
158                {
159                        final MOB M = Authenticate.getAuthenticatedMob(httpReq);
160                        if(M==null)
161                                return "[authentication error]";
162                        if(!CMSecurity.isAllowed(M,M.location(),CMSecurity.SecFlag.POLLS))
163                                return "[authentication error]";
164                        if(last==null)
165                                return " @break@";
166                        final Poll P=CMLib.polls().getPoll(last);
167                        if(P==null)
168                                return "Unknown poll!";
169                        CMLib.polls().deletePoll(P);
170                        Log.sysOut(M.Name()+" deleted poll "+last);
171                        return "Poll deleted.";
172                }
173                else
174                {
175                        if(last==null)
176                                return " @break@";
177                        if(last.length()>0)
178                        {
179                                final String newPollID=httpReq.getUrlParameter("NEWPOLL");
180                                Poll P=(Poll)httpReq.getRequestObjects().get("POLL-"+last);
181                                if((P==null)
182                                &&(newPollID!=null)
183                                &&(newPollID.length()>0)
184                                &&(CMLib.polls().getPoll(newPollID)==null))
185                                {
186                                        P=(Poll)CMClass.getCommon("DefaultPoll");
187                                        P.setName(newPollID);
188                                        last=newPollID;
189                                        httpReq.addFakeUrlParameter("POLL",newPollID);
190                                }
191                                if(P==null)
192                                {
193                                        P=CMLib.polls().getPoll(last);
194                                        if(P!=null)
195                                                CMLib.polls().loadPollIfNecessary(P);
196                                }
197                                if(parms.containsKey("ISNEWPOLL"))
198                                        return ""+(CMLib.polls().loadPollByName(last)==null);
199                                if(P!=null)
200                                {
201                                        final StringBuffer str=new StringBuffer("");
202                                        final boolean input=parms.containsKey("INPUT");
203                                        if(parms.containsKey("TITLE"))
204                                        {
205                                                String old=httpReq.getUrlParameter("TITLE");
206                                                if(old==null)
207                                                        old=P.getName();
208                                                if(old!=null)
209                                                        str.append(old+", ");
210                                        }
211                                        if(parms.containsKey("SUBJECT"))
212                                        {
213                                                String old=httpReq.getUrlParameter("SUBJECT");
214                                                if(old==null)
215                                                        old=P.getSubject();
216                                                if(old!=null)
217                                                        str.append(old+", ");
218                                        }
219                                        if(parms.containsKey("DESCRIPTION"))
220                                        {
221                                                String old=httpReq.getUrlParameter("DESCRIPTION");
222                                                if(old==null)
223                                                        old=P.getDescription();
224                                                if(old!=null)
225                                                        str.append(old+", ");
226                                        }
227                                        if(parms.containsKey("QUALZAPPER"))
228                                        {
229                                                String old=httpReq.getUrlParameter("QUALZAPPER");
230                                                if(old==null)
231                                                        old=P.getQualZapper();
232                                                if(old!=null)
233                                                        str.append(old+", ");
234                                        }
235                                        if(parms.containsKey("ISACTIVE"))
236                                        {
237                                                String old=httpReq.getUrlParameter("ISACTIVE");
238                                                if(old==null)
239                                                        old=(CMath.bset(P.getFlags(),Poll.FLAG_ACTIVE)?"on":"");
240                                                str.append((old.equalsIgnoreCase("on")?"CHECKED":"")+", ");
241                                        }
242                                        if(parms.containsKey("ISPREVIEWRESULTS"))
243                                        {
244                                                String old=httpReq.getUrlParameter("ISPREVIEWRESULTS");
245                                                if(old==null)
246                                                        old=(CMath.bset(P.getFlags(),Poll.FLAG_PREVIEWRESULTS)?"on":"");
247                                                str.append((old.equalsIgnoreCase("on")?"CHECKED":"")+", ");
248                                        }
249                                        if(parms.containsKey("ISABSTAIN"))
250                                        {
251                                                String old=httpReq.getUrlParameter("ISABSTAIN");
252                                                if(old==null)
253                                                        old=(CMath.bset(P.getFlags(),Poll.FLAG_ABSTAIN)?"on":"");
254                                                str.append((old.equalsIgnoreCase("on")?"CHECKED":"")+", ");
255                                        }
256                                        if(parms.containsKey("ISVOTEBYIP"))
257                                        {
258                                                String old=httpReq.getUrlParameter("ISVOTEBYIP");
259                                                if(old==null)
260                                                        old=(CMath.bset(P.getFlags(),Poll.FLAG_VOTEBYIP)?"on":"");
261                                                str.append((old.equalsIgnoreCase("on")?"CHECKED":"")+", ");
262                                        }
263                                        if(parms.containsKey("ISHIDERESULTS"))
264                                        {
265                                                String old=httpReq.getUrlParameter("ISHIDERESULTS");
266                                                if(old==null)
267                                                        old=(CMath.bset(P.getFlags(),Poll.FLAG_HIDERESULTS)?"on":"");
268                                                str.append((old.equalsIgnoreCase("on")?"CHECKED":"")+", ");
269                                        }
270                                        if(parms.containsKey("ISNOTATLOGIN"))
271                                        {
272                                                String old=httpReq.getUrlParameter("ISNOTATLOGIN");
273                                                if(old==null)
274                                                        old=(CMath.bset(P.getFlags(),Poll.FLAG_NOTATLOGIN)?"on":"");
275                                                str.append((old.equalsIgnoreCase("on")?"CHECKED":"")+", ");
276                                        }
277                                        if(parms.containsKey("DOESEXPIRE"))
278                                        {
279                                                String old=httpReq.getUrlParameter("DOESEXPIRE");
280                                                if(old==null)
281                                                        old=(P.getExpiration()!=0)?"on":"";
282                                                str.append(""+old.equalsIgnoreCase("on"));
283                                        }
284                                        if(parms.containsKey("EXPIRATION")) // req input
285                                        {
286                                                Calendar exp=Calendar.getInstance();
287                                                if(httpReq.isUrlParameter("MONTH"))
288                                                {
289                                                        String AP="AM";
290                                                        int hr=CMath.s_int(httpReq.getUrlParameter("HOUR"));
291                                                        if(hr>12)
292                                                        {
293                                                                hr-=12;
294                                                                AP="PM";
295                                                        }
296
297                                                        final String date=(CMath.s_int(httpReq.getUrlParameter("MONTH"))+1)
298                                                                           +"/"+httpReq.getUrlParameter("DAY")
299                                                                           +"/"+httpReq.getUrlParameter("YEAR")
300                                                                           +" "+hr
301                                                                           +":"+httpReq.getUrlParameter("MINUTE")
302                                                                           +" "+AP;
303
304                                                        exp=CMLib.time().string2Date(date);
305                                                }
306                                                else
307                                                {
308                                                        long time=P.getExpiration();
309                                                        if(time<1000)
310                                                                time=System.currentTimeMillis();
311                                                        exp.setTimeInMillis(time);
312                                                }
313                                                if(!input)
314                                                        str.append(CMLib.time().date2String(exp));
315                                                else
316                                                {
317                                                        str.append("<SELECT NAME=MONTH>");
318                                                        for(int m=0;m<12;m++)
319                                                        {
320                                                                str.append("<OPTION VALUE=\""+m+"\"");
321                                                                if(m==exp.get(Calendar.MONTH))
322                                                                        str.append(" SELECTED");
323                                                                str.append(">"+CMLib.time().getMonthName(m+1,false));
324                                                        }
325                                                        str.append("</SELECT>");
326                                                        str.append("/");
327                                                        str.append("<SELECT NAME=DAY>");
328                                                        final int days = exp.getActualMaximum(Calendar.DAY_OF_MONTH); // 28
329                                                        for(int d=1;d<=days;d++)
330                                                        {
331                                                                str.append("<OPTION VALUE=\""+d+"\"");
332                                                                if(d==exp.get(Calendar.DAY_OF_MONTH))
333                                                                        str.append(" SELECTED");
334                                                                str.append(">"+d);
335                                                        }
336                                                        str.append("</SELECT>");
337                                                        str.append("/");
338                                                        str.append("<SELECT NAME=YEAR>");
339                                                        int year=Calendar.getInstance().get(Calendar.YEAR);
340                                                        if((exp.get(Calendar.YEAR)>1900)&&(exp.get(Calendar.YEAR)<year))
341                                                                year=exp.get(Calendar.YEAR);
342                                                        final int doneYear=Calendar.getInstance().get(Calendar.YEAR)+10;
343                                                        for(int y=year;y<=doneYear;y++)
344                                                        {
345                                                                str.append("<OPTION VALUE=\""+y+"\"");
346                                                                if(y==exp.get(Calendar.YEAR))
347                                                                        str.append(" SELECTED");
348                                                                str.append(">"+y);
349                                                        }
350                                                        str.append("</SELECT>");
351                                                        str.append("&nbsp;&nbsp;");
352                                                        str.append("<SELECT NAME=HOUR>");
353                                                        for(int h=1;h<=24;h++)
354                                                        {
355                                                                str.append("<OPTION VALUE=\""+h+"\"");
356                                                                if(h==exp.get(Calendar.HOUR_OF_DAY))
357                                                                        str.append(" SELECTED");
358                                                                str.append(">"+h);
359                                                        }
360                                                        str.append("</SELECT>");
361                                                        str.append(":");
362                                                        str.append("<SELECT NAME=MINUTE>");
363                                                        for(int m=0;m<=59;m++)
364                                                        {
365                                                                str.append("<OPTION VALUE=\""+m+"\"");
366                                                                if(m==exp.get(Calendar.MINUTE))
367                                                                        str.append(" SELECTED");
368                                                                str.append(">"+CMLib.time().twoDigits(m));
369                                                        }
370                                                        str.append("</SELECT>");
371                                                }
372                                        }
373                                        if(parms.containsKey("NUMOPTIONS"))
374                                                str.append(""+P.getOptions().size()+", ");
375                                        if(parms.containsKey("NUMRESULTS"))
376                                                str.append(""+P.getResults().size()+", ");
377                                        if(parms.containsKey("OPTIONS"))
378                                        {
379                                                int which=-1;
380                                                if(parms.containsKey("NUM"))
381                                                        which=CMath.s_int(parms.get("NUM"));
382                                                if(!httpReq.isUrlParameter("OPTION1"))
383                                                {
384                                                        for(int v=0;v<P.getOptions().size();v++)
385                                                        {
386                                                                final Poll.PollOption O=P.getOptions().get(v);
387                                                                httpReq.addFakeUrlParameter("OPTION"+(v+1),O.text);
388                                                        }
389                                                }
390                                                int num=0;
391                                                int showNum=0;
392                                                String sfont=httpReq.getUrlParameter("FONT");
393                                                String efont="";
394                                                if(sfont==null)
395                                                        sfont="<BR>";
396                                                else
397                                                {
398                                                        final String s=sfont.toUpperCase().trim();
399                                                        if(s.indexOf("<I>")>=0)
400                                                                efont+="</I>";
401                                                        if(s.indexOf("<B>")>=0)
402                                                                efont+="</B>";
403                                                        if(s.indexOf("<FONT")>=0)
404                                                                efont+="</FONT>";
405                                                        if(s.indexOf("<P")>=0)
406                                                                efont+="</P>";
407                                                        if(s.indexOf("<TD")>=0)
408                                                                efont+="</TD>";
409                                                        if(s.indexOf("<TR")>=0)
410                                                                efont+="</TR>";
411                                                }
412
413                                                while(httpReq.isUrlParameter("OPTION"+(++num)))
414                                                {
415                                                        final String option=httpReq.getUrlParameter("OPTION"+num);
416                                                        if((option.length()>0)
417                                                        &&((which<0)||(which==showNum)))
418                                                        {
419                                                                ++showNum;
420                                                                if(!input)
421                                                                        str.append(sfont + option + efont);
422                                                                else
423                                                                        str.append(sfont + "<INPUT TYPE=TEXT NAME=OPTION"+(showNum)+" SIZE=60 MAXLENGTH=255 VALUE=\""+option+"\">" + efont);
424                                                        }
425                                                }
426                                                if(input)
427                                                {
428                                                        ++showNum;
429                                                        str.append(sfont + "<INPUT TYPE=TEXT NAME=OPTION"+(showNum)+" SIZE=60 MAXLENGTH=255 VALUE=\"\">" + efont);
430                                                }
431                                        }
432                                        if(parms.containsKey("RESULTS"))
433                                        {
434                                                final List<Poll.PollOption> options=P.getOptions();
435                                                final List<PollResult> results=P.getResults();
436                                                int which=-1;
437                                                if(parms.containsKey("NUM"))
438                                                        which=CMath.s_int(parms.get("NUM"));
439                                                final int[] votes=new int[options.size()+(CMath.bset(P.getFlags(),Poll.FLAG_ABSTAIN)?1:0)];
440                                                PollResult R=null;
441                                                int choice=0;
442                                                int total=0;
443                                                for(int r=0;r<P.getResults().size();r++)
444                                                {
445                                                        R=results.get(r);
446                                                        choice=CMath.s_int(R.answer);
447                                                        if(((choice<=0)&&CMath.bset(P.getFlags(),Poll.FLAG_ABSTAIN))
448                                                        ||((choice>=0)&&(choice<=options.size())))
449                                                        {
450                                                                total++;
451                                                                if(choice<=0)
452                                                                        votes[votes.length-1]++;
453                                                                else
454                                                                        votes[choice-1]++;
455                                                        }
456                                                }
457                                                String sfont=httpReq.getUrlParameter("FONT");
458                                                String efont="";
459                                                if(sfont==null)
460                                                        sfont="<BR>";
461                                                else
462                                                {
463                                                        final String s=sfont.toUpperCase().trim();
464                                                        if(s.indexOf("<I>")>=0)
465                                                                efont+="</I>";
466                                                        if(s.indexOf("<B>")>=0)
467                                                                efont+="</B>";
468                                                        if(s.indexOf("<FONT")>=0)
469                                                                efont+="</FONT>";
470                                                        if(s.indexOf("<P")>=0)
471                                                                efont+="</P>";
472                                                        if(s.indexOf("<TD")>=0)
473                                                                efont+="</TD>";
474                                                        if(s.indexOf("<TR")>=0)
475                                                                efont+="</TR>";
476                                                }
477                                                final boolean pct=parms.containsKey("PCT")||parms.containsKey("PERCENT");
478                                                final boolean showNum=(!pct)||parms.containsKey("VOTES")||parms.containsKey("VOTE")||parms.containsKey("COUNT");
479                                                for(int o=0;o<options.size();o++)
480                                                {
481                                                        if((which<0)||(which==(o+1)))
482                                                        {
483                                                                str.append(sfont);
484                                                                if(showNum)
485                                                                        str.append(votes[o]);
486                                                                if(pct)
487                                                                {
488                                                                        if(showNum)
489                                                                                str.append(" (" + CMath.toPct(((double)votes[o])/((double)total))+")");
490                                                                        else
491                                                                                str.append(CMath.toPct(((double)votes[o])/((double)total)));
492                                                                }
493                                                                str.append(efont+", ");
494                                                        }
495
496                                                }
497                                        }
498                                        if(parms.containsKey("MASKDESC"))
499                                        {
500                                                String mask=httpReq.getUrlParameter("MASK");
501                                                if((mask==null)&&(last.length()>0))
502                                                        mask=CMLib.titles().getAutoTitleMask(last);
503                                                if(mask!=null)
504                                                        str.append(CMLib.masking().maskDesc(mask)+", ");
505                                        }
506                                        String strstr=str.toString();
507                                        if(strstr.endsWith(", "))
508                                                strstr=strstr.substring(0,strstr.length()-2);
509                                        return clearWebMacros(strstr);
510                                }
511                        }
512                }
513                return " @break@";
514        }
