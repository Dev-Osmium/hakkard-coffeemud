036public class AllQualifyData extends StdWebMacro
037{
038        @Override
039        public String name()
040        {
041                return "AllQualifyData";
042        }
043
044        @Override
045        public boolean isAdminMacro()
046        {
047                return true;
048        }
049
050        @Override
051        public String runMacro(HTTPRequest httpReq, String parm, HTTPResponse httpResp)
052        {
053                final java.util.Map<String,String> parms=parseParms(parm);
054                final String last=httpReq.getUrlParameter("ALLQUALID");
055                String which=httpReq.getUrlParameter("ALLQUALWHICH");
056                if(parms.containsKey("WHICH"))
057                        which=parms.get("WHICH");
058                final String origiWhich=which;
059                if((which==null)||(which.length()==0))
060                        which="ALL";
061                final Map<String,Map<String,AbilityMapper.AbilityMapping>> allQualMap=CMLib.ableMapper().getAllQualifiesMap(httpReq.getRequestObjects());
062                final Map<String,AbilityMapper.AbilityMapping> map=allQualMap.get(which.toUpperCase().trim());
063                if(map==null)
064                        return "";
065
066                AbilityMapper.AbilityMapping mapped=map.get(last);
067                if(mapped==null)
068                {
069                        if((origiWhich==null) && (allQualMap.get("EACH")!=null))
070                                mapped=allQualMap.get("EACH").get(last);
071                        if(mapped==null)
072                        {
073                                mapped = CMLib.ableMapper().newAbilityMapping();
074                                mapped.abilityID(last);
075                        }
076                }
077                final StringBuilder str=new StringBuilder("");
078                if(parms.containsKey("NAME"))
079                {
080                        final Ability A=CMClass.getAbility(last);
081                        if(A!=null)
082                                str.append(A.name()).append(", ");
083                }
084
085                if(parms.containsKey("LEVEL"))
086                {
087                        String lvl=httpReq.getUrlParameter("LEVEL");
088                        if(lvl==null)
089                                lvl=Integer.toString(mapped.qualLevel());
090                        else
091                                lvl=Integer.toString(CMath.s_int(lvl));
092                        str.append(lvl).append(", ");
093                }
094
095                if(parms.containsKey("PROF"))
096                {
097                        String prof=httpReq.getUrlParameter("PROF");
098                        if(prof==null)
099                                prof=Integer.toString(mapped.defaultProficiency());
100                        else
101                                prof=Integer.toString(CMath.s_int(prof));
102                        str.append(prof).append(", ");
103                }
104
105                if(parms.containsKey("MASK"))
106                {
107                        String s=httpReq.getUrlParameter("MASK");
108                        if(s==null)
109                                s=mapped.extraMask();
110                        str.append(s).append(", ");
111                }
112
113                if(parms.containsKey("AUTOGAIN"))
114                {
115                        String s=httpReq.getUrlParameter("AUTOGAIN");
116                        if(s==null)
117                                s=mapped.autoGain()?"on":"";
118                        str.append(s.equalsIgnoreCase("on")?"true":"false").append(", ");
119                }
120
121                if(parms.containsKey("REQUIRES"))
122                {
123                        if(!httpReq.isUrlParameter("REQABLE1"))
124                        {
125                                int pnum=1;
126                                for(final String s : CMParms.parseCommas(mapped.originalSkillPreReqList(),true))
127                                {
128                                        String ableID=s;
129                                        String lvl="";
130                                        final int x=s.indexOf('(');
131                                        if(s.endsWith(")")&&(x>1))
132                                        {
133                                                ableID=s.substring(0,x);
134                                                lvl=s.substring(x+1,s.length()-1).trim();
135                                        }
136                                        final Ability A=CMClass.getAbility(ableID);
137                                        if(A!=null)
138                                        {
139                                                httpReq.addFakeUrlParameter("REQABLE"+pnum, ableID);
140                                                httpReq.addFakeUrlParameter("REQLEVEL"+pnum, lvl);
141                                                pnum++;
142                                        }
143                                }
144                                httpReq.addFakeUrlParameter("REQABLE"+pnum, "");
145                                httpReq.addFakeUrlParameter("REQLEVEL"+pnum, "");
146                        }
147                        else
148                        {
149                                int curChkNum=1;
150                                int curWriteNum=1;
151                                while(httpReq.isUrlParameter("REQABLE"+curChkNum))
152                                {
153                                        final String curVal=httpReq.getUrlParameter("REQABLE"+curChkNum);
154                                        if(curVal.equals("DEL")||curVal.equals("DELETE")||curVal.trim().length()==0)
155                                        {
156                                                curChkNum++;
157                                                continue;
158                                        }
159                                        httpReq.addFakeUrlParameter("REQABLE"+curWriteNum, curVal);
160                                        httpReq.addFakeUrlParameter("REQLEVEL"+curWriteNum, httpReq.getUrlParameter("REQLEVEL"+curChkNum));
161                                        curChkNum++;
162                                        curWriteNum++;
163                                }
164                                httpReq.removeUrlParameter("REQABLE"+curWriteNum);
165                                httpReq.removeUrlParameter("REQLEVEL"+curWriteNum);
166                        }
167                        if(parms.containsKey("RESET"))
168                        {
169                                httpReq.removeUrlParameter("REQUIRESNUM");
170                                httpReq.removeUrlParameter("REQUIRESNAME1");
171                                httpReq.removeUrlParameter("REQUIRESNAME2");
172                                return "";
173                        }
174                        if(parms.containsKey("NEXT"))
175                        {
176                                String lastR=httpReq.getUrlParameter("REQUIRESNUM");
177                                String lastID="";
178                                int curChkNum=1;
179                                int curWriteNum=1;
180                                while(httpReq.isUrlParameter("REQABLE"+curChkNum))
181                                {
182                                        final String thisName=Integer.toString(curChkNum);
183                                        if((lastR==null)||((lastR.length()>0)&&(lastR.equals(lastID))&&(!thisName.equals(lastID))))
184                                        {
185                                                httpReq.addFakeUrlParameter("REQUIRESNUM",thisName);
186                                                lastR=thisName;
187                                                httpReq.addFakeUrlParameter("REQUIRESNAME1","REQABLE"+curWriteNum);
188                                                httpReq.addFakeUrlParameter("REQUIRESNAME2","REQLEVEL"+curWriteNum);
189                                                return "";
190                                        }
191                                        curChkNum++;
192                                        curWriteNum++;
193                                        lastID=thisName;
194                                }
195                                httpReq.addFakeUrlParameter("REQUIRESNUM","");
196                                httpReq.addFakeUrlParameter("REQUIRESNAME1","REQABLE"+curWriteNum);
197                                httpReq.addFakeUrlParameter("REQUIRESNAME2","REQLEVEL"+curWriteNum);
198                                if(parms.containsKey("EMPTYOK"))
199                                        return "<!--EMPTY-->";
200                                return " @break@";
201                        }
202                        if(parms.containsKey("ABLEEDIT"))
203                        {
204                                final String lastR=httpReq.getUrlParameter("REQUIRESNUM");
205                                final String ableID=httpReq.getUrlParameter("REQABLE"+lastR);
206                                if((ableID!=null)&&(ableID.length()>0))
207                                {
208                                        str.append("<OPTION VALUE=\"DEL\">Delete!");
209                                        final Ability A=CMClass.getAbility(ableID);
210                                        if(A!=null)
211                                                str.append("<OPTION VALUE=\""+A.ID()+"\" SELECTED>"+A.ID());
212                                }
213                                else
214                                {
215                                        str.append("<OPTION VALUE=\"\">Add New");
216                                        for(final Enumeration<Ability> a=CMClass.abilities();a.hasMoreElements();)
217                                        {
218                                                final Ability A=a.nextElement();
219                                                if((A.classificationCode()&Ability.ALL_DOMAINS)==Ability.DOMAIN_ARCHON)
220                                                        continue;
221                                                str.append("<OPTION VALUE=\""+A.ID()+"\">"+A.ID());
222                                        }
223                                }
224                                str.append(", ");
225                        }
226
227                }
228
229                String strstr=str.toString();
230                if(strstr.endsWith(", "))
231                        strstr=strstr.substring(0,strstr.length()-2);
232                return clearWebMacros(strstr);
233        }
234}
