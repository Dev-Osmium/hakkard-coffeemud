036public class QuestMgr extends StdWebMacro
037{
038        @Override
039        public String name()
040        {
041                return "QuestMgr";
042        }
043
044        @Override
045        public boolean isAdminMacro()
046        {
047                return true;
048        }
049
050        @Override
051        public String runMacro(HTTPRequest httpReq, String parm, HTTPResponse httpResp)
052        {
053                final java.util.Map<String,String> parms=parseParms(parm);
054                Quest Q=null;
055                final MOB M = Authenticate.getAuthenticatedMob(httpReq);
056                final String name=(M==null)?"Someone":M.Name();
057                if(parms.containsKey("CREATE"))
058                {
059                        Q=(Quest)CMClass.getCommon("DefaultQuest");
060                        final String err=populateQuest(httpReq,Q,false);
061                        if(err.length()>0)
062                                return err;
063                        CMLib.quests().addQuest(Q);
064                        CMLib.quests().save();
065                        httpReq.addFakeUrlParameter("QUEST",Q.name());
066                        Log.sysOut("QuestMgr",name+" created quest '"+Q.name()+"'");
067                        return "Quest '"+Q.name()+"' created.";
068                }
069
070                final String last=httpReq.getUrlParameter("QUEST");
071                if(last==null)
072                        return "";
073                if(last.length()>0)
074                {
075                        Q=CMLib.quests().fetchQuest(last);
076                        if(Q==null)
077                        {
078                                final String newLast=CMStrings.replaceAll(last,"*","@");
079                                for(int q=0;q<CMLib.quests().numQuests();q++)
080                                {
081                                        if((""+CMLib.quests().fetchQuest(q)).equals(newLast))
082                                        {
083                                                Q=CMLib.quests().fetchQuest(q);
084                                                break;
085                                        }
086                                }
087                        }
088                        if(Q==null)
089                                return "";
090                        if(parms.containsKey("MODIFY"))
091                        {
092                                final String err=populateQuest(httpReq,Q,parms.containsKey("REDIRECT"));
093                                if(err.length()>0)
094                                        return err;
095                                httpReq.addFakeUrlParameter("QUEST",Q.name());
096                                CMLib.quests().save();
097                                Log.sysOut("QuestMgr",name+" modified quest '"+Q.name()+"'");
098                        }
099                        if(parms.containsKey("DELETE"))
100                        {
101                                CMLib.quests().delQuest(Q);
102                                CMLib.quests().save();
103                                httpReq.addFakeUrlParameter("QUEST","");
104                                final CMFile F=new CMFile(Resources.makeFileResourceName("quests/"+Q.name()+".quest"),M,CMFile.FLAG_FORCEALLOW);
105                                if(F.exists())
106                                {
107                                        Log.sysOut("QuestMgr",name+" deleted quest '"+Q.name()+"'");
108                                        if(F.delete())
109                                                return "Quest script file '"+Resources.makeFileResourceName("quests/"+Q.name()+".quest")+"' deleted.";
110                                }
111                                return "Quest '"+Q.name()+"' removed from quests list.";
112                        }
113                        if(parms.containsKey("START"))
114                        {
115                                if(Q.running())
116                                        return "Quest '"+Q.name()+"' was already running.";
117                                Q.startQuest();
118                                return "Quest '"+Q.name()+"' started.";
119                        }
120                        if(parms.containsKey("STOP"))
121                        {
122                                if(!Q.running())
123                                        return "Quest '"+Q.name()+"' was not running.";
124                                CMLib.coffeeTables().bump(Q,CoffeeTableRow.STAT_QUESTSTOP);
125                                Q.stopQuest();
126                                return "Quest '"+Q.name()+"' stopped.";
127                        }
128                        if(parms.containsKey("ENABLE"))
129                        {
130                                if(!Q.suspended())
131                                        return "Quest '"+Q.name()+"' was not disabled.";
132                                Q.setSuspended(false);
133                                CMLib.database().DBUpdateQuest(Q);
134                                return "Quest '"+Q.name()+"' enabled.";
135                        }
136                        if(parms.containsKey("DISABLE"))
137                        {
138                                if(Q.suspended())
139                                        return "Quest '"+Q.name()+"' was already disabled.";
140                                Q.setSuspended(true);
141                                if(Q.running())
142                                        Q.stopQuest();
143                                CMLib.database().DBUpdateQuest(Q);
144                                return "Quest '"+Q.name()+"' disabled.";
145                        }
146                        if(parms.containsKey("STEP"))
147                        {
148                                if(!Q.running())
149                                        return "Quest '"+Q.name()+"' was not running.";
150                                Q.stepQuest();
151                                return "Quest '"+Q.name()+"' stepped.";
152                        }
153                }
154                return "";
155        }
156
157        public String populateQuest(HTTPRequest httpReq, Quest Q, boolean redirect)
158        {
159                Q.script();
160                String script=httpReq.getUrlParameter("RAWTEXT");
161                final String unRedirectedScript=script;
162                CMFile redirectF=null;
163                if(redirect
164                &&(Q.script().toUpperCase().trim().startsWith("LOAD=")))
165                {
166                        final String fileName=Q.script().trim().substring(5);
167                        redirectF=new CMFile(Resources.makeFileResourceName(fileName),null,CMFile.FLAG_LOGERRORS);
168                        if((!redirectF.exists())||(!redirectF.canRead()))
169                                return "Unable to load redirected file '"+fileName+"'";
170                }
171                else
172                        redirect=false;
173
174                script=CMStrings.replaceAll(script,"&amp;","&");
175
176                String postFix="";
177                final int x=script.toUpperCase().indexOf("<?XML");
178                if(x>=0)
179                {
180                        postFix=script.substring(x);
181                        script=script.substring(0,x);
182                }
183                script=CMStrings.replaceAll(script,"'","`");
184                if(redirect)
185                        script=CMStrings.replaceAll(script,";","\\;");
186                else
187                {
188                        script=CMStrings.replaceAll(script,"\n",";");
189                        script=CMStrings.replaceAll(script,"\r",";");
190                        script=CMStrings.replaceAll(script,";;",";");
191                        script=CMStrings.replaceAll(script,";;",";");
192                }
193                script=script.trim();
194                while(script.endsWith(";"))
195                        script=script.substring(0,script.length()-1);
196                script=script.trim();
197                if((script==null)||(script.trim().length()==0))
198                        return "No script was specified.";
199                if((redirect)&&(redirectF!=null))
200                {
201                        redirectF.saveText(script+postFix);
202                        script=unRedirectedScript;
203                        Q.setScript(Q.script(),true);
204                }
205                else
206                        Q.setScript(script+postFix,true);
207                if(Q.name().length()==0)
208                        return "You must specify a VALID quest string.  This one contained no name.";
209                else
210                if(Q.duration()<0)
211                        return "You must specify a VALID quest string.  This one contained no duration.";
212                else
213                for(int q=0;q<CMLib.quests().numQuests();q++)
214                {
215                        final Quest Q1=CMLib.quests().fetchQuest(q);
216                        if(Q1.name().equalsIgnoreCase(Q.name())&&(Q1!=Q))
217                                return "A quest with that name already exists.";
218                }
219                return "";
220        }
