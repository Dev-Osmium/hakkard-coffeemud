038public class StatRejuvCharts extends StdWebMacro
039{
040        @Override
041        public String name()
042        {
043                return "StatRejuvCharts";
044        }
045
046        protected String getReq(HTTPRequest httpReq, String tag)
047        {
048                String s=httpReq.getUrlParameter(tag);
049                if(s==null)
050                        s="";
051                return s;
052        }
053
054        public final int avgMath(final int stat, final int level, final int add, final CompiledFormula formula)
055        {
056                final double[] variables={
057                                level,
058                                stat,
059                                (double)stat+7,
060                                stat,
061                                (double)stat+7,
062                                stat,
063                                (double)stat+7,
064                                stat,
065                                stat
066                        };
067                return add+(level*(int)Math.round(CMath.parseMathExpression(formula, variables, 0.0)));
068        }
069        
070        @Override
071        public String runMacro(HTTPRequest httpReq, String parm, HTTPResponse httpResp)
072        {
073                final StringBuffer buf=new StringBuffer("");
074                final String which=httpReq.getUrlParameter("WHICH");
075                final MOB mob=CMClass.getMOB("StdMOB");
076                mob.baseState().setMana(100);
077                mob.baseState().setMovement(100);
078                mob.baseState().setHitPoints(100);
079                mob.recoverMaxState();
080                mob.resetToMaxState();
081                mob.curState().setHunger(1000);
082                mob.curState().setThirst(1000);
083                if((which!=null)&&(which.equals("HP")))
084                        buf.append("<BR>Chart: Hit Points<BR>");
085                else
086                if((which!=null)&&(which.equals("MN")))
087                        buf.append("<BR>Chart: Mana<BR>");
088                else
089                if((which!=null)&&(which.equals("MV")))
090                        buf.append("<BR>Chart: Movement<BR>");
091                else
092                        buf.append("<BR>Chart: Hit Points<BR>");
093                buf.append("Flags: ");
094                int disposition=0;
095
096                if((getReq(httpReq,"SITTING").length()>0))
097                {
098                        disposition=PhyStats.IS_SITTING;
099                        buf.append("Sitting ");
100                }
101                if((getReq(httpReq,"SLEEPING").length()>0))
102                {
103                        disposition=PhyStats.IS_SLEEPING;
104                        buf.append("Sleeping ");
105                }
106                if((getReq(httpReq,"FLYING").length()>0))
107                {
108                        disposition=PhyStats.IS_FLYING;
109                        buf.append("Flying ");
110                }
111                if((getReq(httpReq,"SWIMMING").length()>0))
112                {
113                        disposition=PhyStats.IS_SWIMMING;
114                        buf.append("Swimming ");
115                }
116                if((getReq(httpReq,"RIDING").length()>0))
117                {
118                        mob.setRiding((Rideable)CMClass.getMOB("GenRideable"));
119                        buf.append("Riding ");
120                }
121                final boolean hungry=(httpReq.getUrlParameter("HUNGRY")!=null)&&(httpReq.getUrlParameter("HUNGRY").length()>0);
122                if(hungry)
123                {
124                        buf.append("Hungry ");
125                        mob.curState().setHunger(0);
126                }
127                final boolean thirsty=(httpReq.getUrlParameter("THIRSTY")!=null)&&(httpReq.getUrlParameter("THIRSTY").length()>0);
128                if(thirsty)
129                {
130                        buf.append("Thirsty ");
131                        mob.curState().setThirst(0);
132                }
133                mob.basePhyStats().setDisposition(disposition);
134                mob.recoverPhyStats();
135
136                final int MAX_STAT=25;
137                final int MAX_LEVEL=90;
138                final int SKIP_STAT=2;
139                final int SKIP_LEVEL=5;
140                
141                final int[][] hitpointcharts=new int[MAX_LEVEL+1][MAX_STAT+1];
142                final int[][] manacharts=new int[MAX_LEVEL+1][MAX_STAT+1];
143                final int[][] movementcharts=new int[MAX_LEVEL+1][MAX_STAT+1];
144                final int sh=CMProps.getIntVar(CMProps.Int.STARTHP);
145                final int sm=CMProps.getIntVar(CMProps.Int.STARTMANA);
146                final int sv=CMProps.getIntVar(CMProps.Int.STARTMOVE);
147                final Map<CharClass,CompiledFormula> hpformulas=new Hashtable<CharClass,CompiledFormula>(); 
148                final Map<CharClass,CompiledFormula> mnformulas=new Hashtable<CharClass,CompiledFormula>(); 
149                final Map<CharClass,CompiledFormula> mvformulas=new Hashtable<CharClass,CompiledFormula>(); 
150                for(final Enumeration<CharClass> c=CMClass.charClasses();c.hasMoreElements();)
151                {
152                        final CharClass C1=c.nextElement();
153                        hpformulas.put(C1, CMath.compileMathExpression(C1.getHitPointsFormula()));
154                        mnformulas.put(C1, CMath.compileMathExpression(C1.getManaFormula()));
155                        mvformulas.put(C1, CMath.compileMathExpression(C1.getMovementFormula()));
156                }
157                
158                for(int l=1;l<=MAX_LEVEL;l+=SKIP_LEVEL)
159                {
160                        for(int s=4;s<=MAX_STAT;s+=SKIP_STAT)
161                        {
162                                int num=0;
163                                for(final Enumeration<CharClass> c=CMClass.charClasses();c.hasMoreElements();)
164                                {
165                                        final CharClass C1=c.nextElement();
166                                        num++;
167                                        hitpointcharts[l][s]+=avgMath(s,l,sh,hpformulas.get(C1));
168                                        manacharts[l][s]+=avgMath(s,l,sm,mnformulas.get(C1));
169                                        movementcharts[l][s]+=avgMath(s,l,sv,mvformulas.get(C1));
170                                }
171                                hitpointcharts[l][s]/=num;
172                                manacharts[l][s]/=num;
173                                movementcharts[l][s]/=num;
174                        }
175                }
176                
177                buf.append("<P><TABLE WIDTH=100% BORDER=1>");
178                buf.append("<TR><TD><B><FONT COLOR=WHITE>STATS:</FONT></B></TD>");
179                for(int stats=4;stats<=MAX_STAT;stats+=SKIP_STAT)
180                        buf.append("<TD><B><FONT COLOR=WHITE>"+stats+"</FONT></B></TD>");
181                buf.append("</TR>");
182                
183                CMath.CompiledFormula stateHitPointRecoverFormula = null;
184                CMath.CompiledFormula stateManaRecoverFormula = null;
185                CMath.CompiledFormula stateMovesRecoverFormula  = null;
186                stateHitPointRecoverFormula = CMath.compileMathExpression(CMProps.getVar(CMProps.Str.FORMULA_HITPOINTRECOVER));
187                stateManaRecoverFormula = CMath.compileMathExpression(CMProps.getVar(CMProps.Str.FORMULA_MANARECOVER));
188                stateMovesRecoverFormula = CMath.compileMathExpression(CMProps.getVar(CMProps.Str.FORMULA_MOVESRECOVER));
189
190                for(int level=1;level<=MAX_LEVEL;level+=SKIP_LEVEL)
191                {
192                        buf.append("<TR>");
193                        buf.append("<TD><B><FONT COLOR=WHITE>LVL "+level+"</FONT></B></TD>");
194                        for(int stats=4;stats<=MAX_STAT;stats+=SKIP_STAT)
195                        {
196                                for(final int c: CharStats.CODES.BASECODES())
197                                        mob.baseCharStats().setStat(c,stats);
198                                mob.recoverCharStats();
199                                mob.basePhyStats().setLevel(level);
200                                mob.recoverPhyStats();
201                                mob.curState().setMana(0);
202                                mob.curState().setMovement(0);
203                                mob.curState().setHitPoints(0);
204
205                                final CharStats charStats=mob.charStats();
206                                final CharState curState=mob.curState();
207                                final boolean isSleeping=(CMLib.flags().isSleeping(mob));
208                                final boolean isSittingOrRiding=(!isSleeping) && ((CMLib.flags().isSitting(mob))||(mob.riding()!=null));
209                                final boolean isFlying=(!isSleeping) && (!isSittingOrRiding) && CMLib.flags().isFlying(mob);
210                                final boolean isSwimming=(!isSleeping) && (!isSittingOrRiding) && (!isFlying) && CMLib.flags().isSwimming(mob);
211                                final double[] vals=new double[]{
212                                        charStats.getStat(CharStats.STAT_CONSTITUTION),
213                                        mob.phyStats().level(),
214                                        (curState.getHunger()<1)?1.0:0.0,
215                                        (curState.getThirst()<1)?1.0:0.0,
216                                        (curState.getFatigue()>CharState.FATIGUED_MILLIS)?1.0:0.0,
217                                        isSleeping?1.0:0.0,
218                                        isSittingOrRiding?1.0:0.0,
219                                        isFlying?1.0:0.0,
220                                        isSwimming?1.0:0.0
221                                };
222
223                                if((which!=null)&&(which.equals("HP")))
224                                {
225                                        final long hpGain = Math.round(CMath.parseMathExpression(stateHitPointRecoverFormula, vals, 0.0));
226                                        buf.append("<TD><FONT COLOR=CYAN>"+hitpointcharts[level][stats]+"/"+hpGain+"="+(hitpointcharts[level][stats]/hpGain)+"</FONT></TD>");
227                                }
228                                else
229                                if((which!=null)&&(which.equals("MN")))
230                                {
231                                        vals[0]=((charStats.getStat(CharStats.STAT_INTELLIGENCE)+charStats.getStat(CharStats.STAT_WISDOM)));
232                                        final long manaGain = Math.round(CMath.parseMathExpression(stateManaRecoverFormula, vals, 0.0));
233                                        buf.append("<TD><FONT COLOR=PINK>"+manacharts[level][stats]+"/"+manaGain+"="+(manacharts[level][stats]/manaGain)+"</FONT></TD>");
234                                }
235                                else
236                                if((which!=null)&&(which.equals("MV")))
237                                {
238                                        vals[0]=charStats.getStat(CharStats.STAT_STRENGTH);
239                                        final long moveGain = Math.round(CMath.parseMathExpression(stateMovesRecoverFormula, vals, 0.0));
240                                        buf.append("<TD><FONT COLOR=YELLOW>"+movementcharts[level][stats]+"/"+moveGain+"="+(movementcharts[level][stats]/moveGain)+"</FONT></TD>");
241                                }
242                                else
243                                {
244                                        final long hpGain = Math.round(CMath.parseMathExpression(stateHitPointRecoverFormula, vals, 0.0));
245                                        buf.append("<TD><FONT COLOR=CYAN>"+hitpointcharts[level][stats]+"/"+hpGain+"="+(hitpointcharts[level][stats]/hpGain)+"</FONT></TD>");
246                                }
247                        }
248                        buf.append("</TR>");
249                }
250                mob.destroy();
251                buf.append("</TABLE>");
252                return clearWebMacros(buf);
253        }
