038public class JournalFunction extends StdWebMacro
039{
040        @Override
041        public String name()
042        {
043                return "JournalFunction";
044        }
045
046        @Override
047        public String runMacro(final HTTPRequest httpReq, final String parm, final HTTPResponse httpResp)
048        {
049                if(!CMProps.getBoolVar(CMProps.Bool.MUDSTARTED))
050                        return CMProps.getVar(CMProps.Str.MUDSTATUS);
051
052                final java.util.Map<String,String> parms=parseParms(parm);
053                final String journalName=httpReq.getUrlParameter("JOURNAL");
054                if(journalName==null)
055                        return "Function not performed -- no Journal specified.";
056
057                final Clan setClan=CMLib.clans().getClan(httpReq.getUrlParameter("CLAN"));
058                final JournalsLibrary.ForumJournal forum=CMLib.journals().getForumJournal(journalName,setClan);
059                final MOB M = Authenticate.getAuthenticatedMob(httpReq);
060                if(CMLib.journals().isArchonJournalName(journalName))
061                {
062                        if((M==null)||(!CMSecurity.isASysOp(M)))
063                                return " @break@";
064                }
065                if(parms.containsKey("DESTROYFOREVER"))
066                {
067                        if((forum!=null)&&(!forum.authorizationCheck(M, ForumJournalFlags.ADMIN)))
068                                return "Destruction cancelled -- You are not authorized to delete this forum.";
069                        if(!CMSecurity.isAllowedEverywhere(M,CMSecurity.SecFlag.JOURNALS))
070                                return "Destruction cancelled -- You are not authorized.";
071                        CMLib.database().DBDeleteJournal(journalName, null);
072                        return "Journal "+journalName+" deleted.";
073                }
074                if(parms.containsKey("SUBSCRIBE"))
075                {
076                        if(forum==null)
077                                return "Subscription cancelled -- no forum.";
078                        if(CMLib.journals().subscribeToJournal(journalName, M.Name(), true))
079                        {
080                                return "Now subscribed to "+journalName+".";
081                        }
082                        return "New subscribtion to "+journalName+" failed.";
083                }
084                if(parms.containsKey("UNSUBSCRIBE"))
085                {
086                        if(forum==null)
087                                return "UnSubscription cancelled -- no forum.";
088                        if(CMLib.journals().unsubscribeFromJournal(journalName, M.Name(), true))
089                        {
090                                return "Now unsubscribed from "+journalName+".";
091                        }
092                        return "Unsubscription from "+journalName+" failed -- were you ever subscribed?";
093                }
094                String from="Anonymous";
095                if(M!=null)
096                        from=M.Name();
097                if(parms.containsKey("NEWPOST"))
098                {
099                        if((forum!=null)&&(!forum.authorizationCheck(M, ForumJournalFlags.POST)))
100                                return "Post not submitted -- Unauthorized.";
101                        String to=httpReq.getUrlParameter("TO");
102                        if((to==null)||(M==null)||(to.equalsIgnoreCase("all")))
103                                to="ALL";
104                        if((!to.equals("ALL"))&&(!to.toUpperCase().trim().startsWith("MASK=")))
105                        {
106                                if(!CMLib.players().playerExists(to) && (!CMLib.players().accountExists(to)))
107                                        return "Post not submitted -- TO user does not exist.  Try 'All'.";
108                                to=CMStrings.capitalizeAndLower(to);
109                        }
110                        else
111                        if(journalName.equalsIgnoreCase(CMProps.getVar(CMProps.Str.MAILBOX))
112                        &&(!CMSecurity.isAllowedEverywhere(M,CMSecurity.SecFlag.JOURNALS)))
113                                return "Post not submitted -- You are not authorized to send email to ALL.";
114                        String subject=httpReq.getUrlParameter("SUBJECT");
115                        if(subject==null)
116                                subject="";
117                        final String parent=httpReq.getUrlParameter("PARENT");
118                        if((subject.length()==0)&&(parent==null))
119                                return "Post not submitted -- No subject!";
120                        if((parent!=null)&&(parent.length()>0)&&(subject.length()==0))
121                        {
122                                JournalEntry parentEntry = null;
123                                parentEntry=CMLib.database().DBReadJournalEntry(journalName, parent);
124                                if(parentEntry!=null)
125                                        subject="RE: "+parentEntry.subj();
126                        }
127                        final String date=httpReq.getUrlParameter("DATE");
128                        final String icon=httpReq.getUrlParameter("MSGICON");
129                        final List<String> flags=CMParms.parseCommas(httpReq.getUrlParameter("FLAGS"), true);
130                        if((flags.size()>0)&&(forum!=null)&&(!forum.authorizationCheck(M, ForumJournalFlags.ADMIN)))
131                                return "Post not submitted -- Unauthorized flags.";
132                        final String text=httpReq.getUrlParameter("NEWTEXT");
133                        if((text==null)||(text.length()==0))
134                                return "Post not submitted -- No text!";
135                        if(journalName.equalsIgnoreCase(CMProps.getVar(CMProps.Str.MAILBOX))
136                        &&(CMProps.getIntVar(CMProps.Int.MAXMAILBOX)>0)
137                        &&(!to.equalsIgnoreCase("ALL")))
138                        {
139                                final int count=CMLib.database().DBCountJournal(journalName,null,to);
140                                if(count>=CMProps.getIntVar(CMProps.Int.MAXMAILBOX))
141                                        return "Post not submitted -- Mailbox is full!";
142                        }
143                        final JournalEntry msg = (JournalEntry)CMClass.getCommon("DefaultJournalEntry");
144                        msg.from(from);
145                        msg.subj(clearWebMacros(subject));
146                        msg.msg(clearWebMacros(text));
147                        if((date!=null) && (CMath.isLong(date)))
148                                msg.date(CMath.s_long(date));
149                        else
150                                msg.date(System.currentTimeMillis());
151                        msg.update(System.currentTimeMillis());
152                        msg.parent((parent==null)?"":parent);
153                        msg.msgIcon((icon==null)?"":icon);
154                        if(flags.contains("STUCKY"))
155                                msg.attributes(msg.attributes()|JournalEntry.ATTRIBUTE_STUCKY);
156                        if(flags.contains("PROTECTED"))
157                                msg.attributes(msg.attributes()|JournalEntry.ATTRIBUTE_PROTECTED);
158                        msg.data("");
159                        msg.to(to);
160                        // check for dups
161                        final List<JournalEntry> chckEntries = CMLib.database().DBReadJournalMsgsNewerThan(journalName, to, msg.date()-1);
162                        for(final JournalEntry entry : chckEntries)
163                        {
164                                if((entry.date() == msg.date())
165                                &&(entry.from().equals(msg.from()))
166                                &&(entry.subj().equals(msg.subj()))
167                                &&(entry.parent().equals(msg.parent())))
168                                        return "";
169                        }
170                        CMLib.database().DBWriteJournal(journalName,msg);
171                        JournalInfo.clearJournalCache(httpReq, journalName);
172                        if(parent!=null)
173                                CMLib.database().DBTouchJournalMessage(parent);
174                        CMLib.journals().clearJournalSummaryStats(forum);
175                        return "Post submitted.";
176                }
177                else
178                if(parms.containsKey("ADMINSUBMIT"))
179                {
180                        if(forum==null)
181                                return "Changes not submitted -- Unsupported.";
182                        else
183                        if(!forum.authorizationCheck(M, ForumJournalFlags.ADMIN))
184                                return "Changes not submitted -- Unauthorized.";
185                        final String longDesc=fixForumString(httpReq.getUrlParameter("LONGDESC"));
186                        final String shortDesc=fixForumString(httpReq.getUrlParameter("SHORTDESC"));
187                        final String imgPath=httpReq.getUrlParameter("IMGPATH");
188                        final JournalsLibrary.JournalMetaData metaData = CMLib.journals().getJournalStats(forum);
189                        if(metaData == null)
190                                return "Changes not submitted -- No Stats!";
191                        if(longDesc!=null)
192                                metaData.longIntro(clearWebMacros(longDesc));
193                        if(shortDesc!=null)
194                                metaData.shortIntro(clearWebMacros(shortDesc));
195                        if(imgPath!=null)
196                                metaData.imagePath(clearWebMacros(imgPath));
197                        CMLib.database().DBUpdateJournalMetaData(journalName, metaData);
198                        CMLib.journals().clearJournalSummaryStats(forum);
199                        return "Changed applied.";
200                }
201                String parent=httpReq.getUrlParameter("JOURNALPARENT");
202                if(parent==null)
203                        parent="";
204                String dbsearch=httpReq.getUrlParameter("DBSEARCH");
205                if(dbsearch==null)
206                        dbsearch="";
207                final String page=httpReq.getUrlParameter("JOURNALPAGE");
208                final String mpage=httpReq.getUrlParameter("MESSAGEPAGE");
209                final List<JournalEntry> msgs=JournalInfo.getMessages(journalName,forum,page,mpage,parent,dbsearch,httpReq.getRequestObjects());
210                String msgKey=httpReq.getUrlParameter("JOURNALMESSAGE");
211                int cardinalNumber = CMath.s_int(httpReq.getUrlParameter("JOURNALCARDINAL"));
212                String srch=httpReq.getUrlParameter("JOURNALMESSAGESEARCH");
213                if(srch!=null)
214                        srch=srch.toLowerCase();
215                final boolean doThemAll=parms.containsKey("EVERYTHING");
216                if(doThemAll)
217                {
218                        final JournalEntry entry = JournalInfo.getNextEntry(msgs, null);
219                        if(entry==null)
220                                msgKey="";
221                        else
222                                msgKey=entry.key();
223                        cardinalNumber=1;
224                }
225                final StringBuffer messages=new StringBuffer("");
226                boolean keepProcessing=((msgKey!=null)&&(msgKey.length()>0));
227                String fieldSuffix="";
228                while(keepProcessing)
229                {
230                        if(doThemAll)
231                        {
232                                parms.clear();
233                                parms.put("EVERYTHING","EVERYTHING");
234                                final String fate=httpReq.getUrlParameter("FATE"+msgKey);
235                                final String replyemail=httpReq.getUrlParameter("REPLYEMAIL"+msgKey);
236                                cardinalNumber = CMath.s_int(httpReq.getUrlParameter("CARDINAL"+msgKey));
237                                if((fate!=null)&&(fate.length()>0)&&(CMStrings.isUpperCase(fate)))
238                                        parms.put(fate,fate);
239                                if((replyemail!=null)&&(replyemail.length()>0)&&(CMStrings.isUpperCase(replyemail)))
240                                        parms.put(replyemail,replyemail);
241                                if(parms.size()==1)
242                                {
243                                        JournalEntry entry = JournalInfo.getNextEntry(msgs, msgKey);
244                                        while((entry!=null) && (!CMLib.journals().canReadMessage(entry,srch,M,parms.containsKey("NOPRIV"))))
245                                                entry = JournalInfo.getNextEntry(msgs, entry.key());
246
247                                        if(entry==null)
248                                                keepProcessing=false;
249                                        else
250                                                msgKey=entry.key();
251                                        continue;
252                                }
253                                fieldSuffix=msgKey;
254                        }
255                        else
256                                keepProcessing=false;
257                        JournalEntry entry = JournalInfo.getEntry(msgs, msgKey);
258                        if((entry==null)&&parms.containsKey("DELETEREPLY"))
259                                entry=CMLib.database().DBReadJournalEntry(journalName, msgKey);
260                        if(entry == null)
261                                return "Function not performed -- illegal journal message specified.<BR>";
262                        if(!doThemAll)
263                                entry.cardinal(cardinalNumber);
264                        final String to=entry.to();
265                        final String author=entry.from();
266                        if((M!=null)
267                        &&(CMSecurity.isAllowedAnywhere(M,CMSecurity.SecFlag.JOURNALS)
268                                ||(to.equalsIgnoreCase(M.Name()))
269                                ||(author.equalsIgnoreCase(M.Name())))
270                        &&((forum==null)||(forum.authorizationCheck(M, ForumJournalFlags.READ))))
271                        {
272                                if(parms.containsKey("REPLY"))
273                                {
274                                        if((forum!=null)&&(!forum.authorizationCheck(M, ForumJournalFlags.REPLY)))
275                                                return "Reply not submitted -- Unauthorized.";
276                                        final String text=httpReq.getUrlParameter("NEWTEXT"+fieldSuffix);
277                                        if((text==null)||(text.length()==0))
278                                                messages.append("Reply to #"+cardinalNumber+" not submitted -- No text!<BR>");
279                                        else
280                                        {
281                                                CMLib.database().DBWriteJournalReply(journalName,entry.key(),from,"","",clearWebMacros(text));
282                                                CMLib.journals().clearJournalSummaryStats(forum);
283                                                JournalInfo.clearJournalCache(httpReq, journalName);
284                                                messages.append("Reply to #"+cardinalNumber+" submitted<BR>");
285                                        }
286                                }
287                                else
288                                if(parms.containsKey("EMAIL"))
289                                {
290                                        if((forum!=null)&&(!forum.authorizationCheck(M, ForumJournalFlags.REPLY)))
291                                                return "Email not submitted -- Unauthorized.";
292                                        final String replyMsg=httpReq.getUrlParameter("NEWTEXT"+fieldSuffix);
293                                        if(replyMsg.length()==0)
294                                                messages.append("Email to #"+cardinalNumber+" not submitted -- No text!<BR>");
295                                        else
296                                        {
297                                                final String toName=entry.from();
298                                                final MOB toM=CMLib.players().getLoadPlayer(toName);
299                                                if((toM==null)||(toM.playerStats()==null))
300                                                        messages.append("Player '"+toName+"' does not exist.<BR>");
301                                                else
302                                                {
303                                                        if(toM.playerStats().getEmail().indexOf('@')<0)
304                                                                messages.append("Warning: Player '"+toName+"' has no email address..<BR>");
305                                                        CMLib.smtp().emailOrJournal(M.Name(),
306                                                                                                                M.Name(),
307                                                                                                                toM.Name(),
308                                                                                                                "RE: "+entry.subj(),
309                                                                                                                clearWebMacros(replyMsg));
310                                                        JournalInfo.clearJournalCache(httpReq, journalName);
311                                                        messages.append("Email to #"+cardinalNumber+" queued<BR>");
312                                                }
313                                        }
314                                }
315                                if(parms.containsKey("DELETE")||parms.containsKey("DELETEREPLY"))
316                                {
317                                        if((forum!=null)&&(!forum.authorizationCheck(M, ForumJournalFlags.ADMIN)))
318                                                return "Delete not authorized.";
319                                        if(parms.containsKey("DELETEREPLY")
320                                        &&(entry.parent()!=null)
321                                        &&(entry.parent().length()>0))
322                                        {
323                                                CMLib.database().DBDeleteJournal(journalName,entry.key());
324                                                // this constitutes a threaded reply -- update the counter
325                                                final JournalEntry parentEntry=CMLib.database().DBReadJournalEntry(journalName, entry.parent());
326                                                if(parentEntry!=null)
327                                                        CMLib.database().DBUpdateMessageReplies(parentEntry.key(),parentEntry.replies()-1);
328                                                JournalInfo.clearJournalCache(httpReq, journalName);
329                                                httpReq.addFakeUrlParameter("JOURNALMESSAGE",entry.parent());
330                                                httpReq.addFakeUrlParameter("JOURNALPARENT","");
331                                                if(cardinalNumber==0)
332                                                        cardinalNumber=entry.cardinal();
333                                                if(cardinalNumber==0)
334                                                        messages.append("Reply deleted.<BR>");
335                                                else
336                                                        messages.append("Reply #"+cardinalNumber+" deleted.<BR>");
337                                        }
338                                        else
339                                        {
340                                                JournalEntry nextEntry = null;
341                                                if(doThemAll && keepProcessing)
342                                                {
343                                                        cardinalNumber++;
344                                                        nextEntry = JournalInfo.getNextEntry(msgs, msgKey);
345                                                        while((nextEntry!=null) && (!CMLib.journals().canReadMessage(entry,srch,M,parms.containsKey("NOPRIV"))))
346                                                                nextEntry = JournalInfo.getNextEntry(msgs, nextEntry.key());
347                                                        cardinalNumber--;
348                                                }
349                                                CMLib.database().DBDeleteJournal(journalName,entry.key());
350                                                if(cardinalNumber==0)
351                                                        cardinalNumber=entry.cardinal();
352                                                if(cardinalNumber==0)
353                                                        messages.append("Message deleted.<BR>");
354                                                else
355                                                        messages.append("Message #"+cardinalNumber+" deleted.<BR>");
356                                                JournalInfo.clearJournalCache(httpReq, journalName);
357                                                httpReq.addFakeUrlParameter("JOURNALMESSAGE","");
358                                                if(nextEntry != null)
359                                                {
360                                                        cardinalNumber++;
361                                                        msgKey=nextEntry.key();
362                                                        CMLib.journals().clearJournalSummaryStats(forum);
363                                                        continue;
364                                                }
365                                        }
366                                        CMLib.journals().clearJournalSummaryStats(forum);
367                                }
368                                else
369                                if(parms.containsKey("EDIT"))
370                                {
371                                        if((entry.to().equals(M.Name()))
372                                        ||((forum!=null)&&(!forum.authorizationCheck(M, ForumJournalFlags.ADMIN)))
373                                        ||CMSecurity.isAllowedAnywhere(M,CMSecurity.SecFlag.JOURNALS))
374                                        {
375                                                final String text=httpReq.getUrlParameter("NEWTEXT"+fieldSuffix);
376                                                if((text==null)||(text.length()==0))
377                                                        messages.append("Edit to #"+cardinalNumber+" not submitted -- No text!<BR>");
378                                                else
379                                                {
380                                                        long attributes=0;
381                                                        if((forum!=null)&&(forum.authorizationCheck(M, ForumJournalFlags.ADMIN)))
382                                                        {
383                                                                String ISSTUCKY=httpReq.getUrlParameter("ISSTICKY"+fieldSuffix);
384                                                                if(ISSTUCKY==null)
385                                                                        ISSTUCKY=httpReq.getUrlParameter("ISSTUCKY"+fieldSuffix);
386                                                                if((ISSTUCKY!=null)&&(ISSTUCKY.equalsIgnoreCase("on")))
387                                                                        attributes|=JournalEntry.ATTRIBUTE_STUCKY;
388                                                                final String ISPROTECTED=httpReq.getUrlParameter("ISPROTECTED"+fieldSuffix);
389                                                                if((ISPROTECTED!=null)&&(ISPROTECTED.equalsIgnoreCase("on")))
390                                                                        attributes|=JournalEntry.ATTRIBUTE_PROTECTED;
391                                                        }
392                                                        CMLib.database().DBUpdateJournal(entry.key(), entry.subj(), clearWebMacros(text), attributes);
393                                                        if(cardinalNumber==0)
394                                                                cardinalNumber=entry.cardinal();
395                                                        if(cardinalNumber==0)
396                                                                messages.append("Message modified.<BR>");
397                                                        else
398                                                                messages.append("Message #"+cardinalNumber+" modified.<BR>");
399                                                        JournalInfo.clearJournalCache(httpReq, journalName);
400                                                        if((entry.parent()!=null)&&(entry.parent().length()>0))
401                                                        {
402                                                                httpReq.addFakeUrlParameter("JOURNALMESSAGE",entry.parent());
403                                                                httpReq.addFakeUrlParameter("JOURNALPARENT","");
404                                                        }
405                                                        CMLib.journals().clearJournalSummaryStats(forum);
406                                                }
407                                        }
408                                        else
409                                                return "Edit not authorized.";
410                                }
411                                else
412                                if(CMSecurity.isAllowedAnywhere(M,CMSecurity.SecFlag.JOURNALS))
413                                {
414                                        if(parms.containsKey("TRANSFER"))
415                                        {
416                                                if((forum!=null)&&(!forum.authorizationCheck(M, ForumJournalFlags.ADMIN)))
417                                                        return "Email not submitted -- Unauthorized.";
418                                                String journal=httpReq.getUrlParameter("NEWJOURNAL"+fieldSuffix);
419                                                if((journal==null) || (journal.length()==0))
420                                                        messages.append("Transfer #"+cardinalNumber+" not completed -- No journal!<BR>");
421
422                                                String realName=null;
423                                                if(journal!=null)
424                                                {
425                                                        final List<String> users=new ArrayList<String>();
426                                                        //if(forum != null)
427                                                        //      users.addAll(CMParms.parseAny(forum.getFlag(ForumJournalFlags.ASSIGN),':',true));
428                                                        if(!users.contains("ALL"))
429                                                                users.add("ALL");
430                                                        final boolean isPlayer=CMLib.players().playerExists(CMStrings.capitalizeAndLower(journal));
431                                                        if(journal.equals("FROM")||users.contains(journal)||isPlayer)
432                                                        {
433                                                                if(journal.equals("FROM"))
434                                                                        entry.to(entry.from());
435                                                                else
436                                                                if(isPlayer)
437                                                                        entry.to(CMStrings.capitalizeAndLower(journal));
438                                                                else
439                                                                        entry.to(journal);
440                                                                CMLib.journals().clearJournalSummaryStats(forum);
441                                                                JournalInfo.clearJournalCache(httpReq, journalName);
442                                                                CMLib.database().DBUpdateJournal(journalName, entry);
443                                                                journal=null;
444                                                                messages.append("Message #"+cardinalNumber+" transferred<BR>");
445                                                        }
446                                                        else
447                                                        {
448                                                                for(final Enumeration<JournalsLibrary.CommandJournal> e=CMLib.journals().commandJournals();e.hasMoreElements();)
449                                                                {
450                                                                        final JournalsLibrary.CommandJournal CMJ=e.nextElement();
451                                                                        if(journal.equalsIgnoreCase(CMJ.NAME())
452                                                                        ||journal.equalsIgnoreCase(CMJ.NAME()+"s")
453                                                                        ||journal.equalsIgnoreCase(CMJ.JOURNAL_NAME()))
454                                                                        {
455                                                                                realName=CMJ.JOURNAL_NAME();
456                                                                                break;
457                                                                        }
458                                                                }
459                                                        }
460                                                }
461                                                if(journal != null)
462                                                {
463                                                        if(realName==null)
464                                                                realName=CMLib.database().DBGetRealJournalName(journal);
465                                                        if(realName==null)
466                                                                realName=CMLib.database().DBGetRealJournalName(journal.toUpperCase());
467                                                        if(realName==null)
468                                                                messages.append("The journal '"+journal+"' does not presently exist.  Aborted.<BR>");
469                                                        else
470                                                        {
471                                                                CMLib.journals().clearJournalSummaryStats(forum);
472                                                                CMLib.database().DBDeleteJournal(journalName,entry.key());
473                                                                if(journalName.toUpperCase().startsWith("SYSTEM_"))
474                                                                        entry.update(System.currentTimeMillis());
475                                                                CMLib.database().DBWriteJournal(realName,entry);
476                                                                CMLib.journals().clearJournalSummaryStats(forum);
477                                                                JournalInfo.clearJournalCache(httpReq, journalName);
478                                                                httpReq.addFakeUrlParameter("JOURNALMESSAGE","");
479                                                                messages.append("Message #"+cardinalNumber+" transferred<BR>");
480                                                        }
481                                                }
482                                        }
483                                }
484                                else
485                                        messages.append("You are not allowed to perform this function on message #"+cardinalNumber+".<BR>");
486                        }
487                        if(keepProcessing)
488                        {
489                                cardinalNumber++;
490                                entry = JournalInfo.getNextEntry(msgs, msgKey);
491                                while((entry!=null) && (!CMLib.journals().canReadMessage(entry,srch,M,parms.containsKey("NOPRIV"))))
492                                        entry = JournalInfo.getNextEntry(msgs, entry.key());
493                                if(entry==null)
494                                        keepProcessing=false;
495                                else
496                                        msgKey=entry.key();
497                        }
498                }
499                return messages.toString();
500        }

502        public String fixForumString(final String s)
503        {
504                if(s==null)
505                        return "";
506                final int x=s.toUpperCase().indexOf("<P>");
507                final int y=s.toUpperCase().lastIndexOf("</P>");
508                if((x>=0)&&(y>x))
509                {
510                        return s.substring(0,x)+s.substring(x+3,y)+s.substring(y+4);
511                }
512                return s;
513        }
