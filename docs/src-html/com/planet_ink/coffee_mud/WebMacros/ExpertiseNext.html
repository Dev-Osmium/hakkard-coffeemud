037@SuppressWarnings({"unchecked","rawtypes"})
038public class ExpertiseNext extends StdWebMacro
039{
040        @Override
041        public String name()
042        {
043                return "ExpertiseNext";
044        }
045
046        @Override
047        public String runMacro(HTTPRequest httpReq, String parm, HTTPResponse httpResp)
048        {
049                final java.util.Map<String,String> parms=parseParms(parm);
050                final String last=httpReq.getUrlParameter("EXPERTISE");
051                if(parms.containsKey("RESET"))
052                {
053                        if(last!=null)
054                                httpReq.removeUrlParameter("EXPERTISE");
055                        return "";
056                }
057                String lastID="";
058                DVector experts=(DVector)httpReq.getRequestObjects().get("SORTED_EXPERTISE");
059                ExpertiseLibrary.ExpertiseDefinition E=null;
060                if(experts==null)
061                {
062                        experts=new DVector(2);
063                        String Ename=null;
064                        String Vname=null;
065                        int x=0;
066                        for(final Enumeration e=CMLib.expertises().definitions();e.hasMoreElements();)
067                        {
068                                E=(ExpertiseLibrary.ExpertiseDefinition)e.nextElement();
069                                Ename=E.name();
070                                x=Ename.lastIndexOf(' ');
071                                if((x>0)&&(CMath.isRomanNumeral(Ename.substring(x).trim())))
072                                        Ename=Ename.substring(0,x)+" "+((char)('a'+CMath.convertFromRoman(Ename.substring(x).trim())));
073                                boolean added=false;
074                                for(int v=0;v<experts.size();v++)
075                                {
076                                        Vname=(String)experts.elementAt(v,1);
077                                        if(Vname.compareTo(Ename)>0)
078                                        {
079                                                experts.insertElementAt(v,Ename,E);
080                                                added=true;
081                                                break;
082                                        }
083                                }
084                                if(!added)
085                                        experts.addElement(Ename,E);
086                        }
087                        httpReq.getRequestObjects().put("SORTED_EXPERTISE",experts);
088                }
089                Integer qualLevel=null;
090                final String levelName=httpReq.getUrlParameter("LEVEL");
091                final int levelCheck=((levelName!=null)&&(levelName.length()>0))?CMath.s_int(levelName):-1;
092                final String className=httpReq.getUrlParameter("CLASS");
093                Hashtable<String,Integer> expertsAllows=null;
094                if((className!=null)&&(className.length()>0))
095                {
096                        expertsAllows=(Hashtable)httpReq.getRequestObjects().get("ALLOWS-"+className.toUpperCase().trim());
097                        if(expertsAllows==null)
098                        {
099                                expertsAllows=new Hashtable<String,Integer>();
100                                httpReq.getRequestObjects().put("ALLOWS-"+className.toUpperCase().trim(),expertsAllows);
101                                final List<QualifyingID> DV=CMLib.ableMapper().getClassAllowsList(className);
102                                if(DV!=null)
103                                {
104                                        for(final QualifyingID qID : DV)
105                                        {
106                                                final String xpertise=qID.ID();
107                                                E=CMLib.expertises().getDefinition(xpertise);
108                                                if(E!=null)
109                                                {
110                                                        qualLevel=Integer.valueOf(qID.qualifyingLevel());
111                                                        int minLevel=E.getMinimumLevel();
112                                                        if((qualLevel!=null)&&(minLevel<qualLevel.intValue()))
113                                                                minLevel=qualLevel.intValue();
114                                                        expertsAllows.put(xpertise,Integer.valueOf(minLevel));
115                                                }
116                                        }
117                                }
118                        }
119                }
120                for(final Iterator<Object> e=experts.getDimensionList(2).iterator();e.hasNext();)
121                {
122                        E=(ExpertiseLibrary.ExpertiseDefinition)e.next();
123                        if(E!=null)
124                        {
125                                if(expertsAllows!=null)
126                                {
127                                        qualLevel=expertsAllows.get(E.ID());
128                                        if(qualLevel==null)
129                                                continue;
130                                        if((levelCheck>=0)&&(levelCheck!=qualLevel.intValue()))
131                                                continue;
132                                }
133                                else
134                                if((levelCheck>=0)&&(levelCheck!=E.getMinimumLevel()))
135                                        continue;
136                                if((last==null)||((last.length()>0)&&(last.equals(lastID))&&(!E.ID().equals(lastID))))
137                                {
138                                        httpReq.addFakeUrlParameter("EXPERTISE",E.ID());
139                                        return "";
140                                }
141                                lastID=E.ID();
142                        }
143                }
144                httpReq.addFakeUrlParameter("EXPERTISE","");
145                if(parms.containsKey("EMPTYOK"))
146                        return "<!--EMPTY-->";
147                return " @break@";
148        }
