050        public void updateINIFile(final List<String> page)
051        {
052                final StringBuffer buf=new StringBuffer("");
053                for(int p=0;p<page.size();p++)
054                        buf.append((page.get(p))+"\r\n");
055                new CMFile(CMProps.getVar(CMProps.Str.INIPATH),null,CMFile.FLAG_FORCEALLOW).saveText(buf);
056        }

058        public boolean modified(final Set<String> H, final String s)
059        {
060                if(s.endsWith("*"))
061                {
062                        for (final String string : H)
063                        {
064                                if(string.startsWith(s.substring(0,s.length()-1)))
065                                        return true;
066                        }
067                }
068                return H.contains(s);
069        }

071        @Override
072        public String runMacro(final HTTPRequest httpReq, final String parm, final HTTPResponse httpResp)
073        {
074                final MOB authM=Authenticate.getAuthenticatedMob(httpReq);
075                if((authM==null)||(!CMSecurity.isASysOp(authM)))
076                        return " @break@ ";
077                final java.util.Map<String,String> parms=parseParms(parm);
078                if(parms==null)
079                        return "";
080                final List<String> page=CMProps.loadEnumerablePage(CMProps.getVar(CMProps.Str.INIPATH));
081                if(parms.containsKey("ADDKEY"))
082                {
083                        String key=parms.get("KEY");
084                        if((key==null)||(key.trim().length()==0))
085                                return "";
086                        key=key.trim().toUpperCase();
087                        final CMProps ipage=CMProps.loadPropPage(CMProps.getVar(CMProps.Str.INIPATH));
088                        if((ipage==null)||(!ipage.isLoaded()))
089                                return "";
090                        if(ipage.containsKey(key))
091                                return "";
092                        int where=0;
093                        if(parms.containsKey("NEAR"))
094                        {
095                                boolean found=false;
096                                String near=parms.get("NEAR");
097                                if(near.endsWith("*"))
098                                        near=near.substring(0,near.length()-1);
099                                for(int p=0;p<page.size();p++)
100                                {
101                                        final String s=page.get(p).trim();
102                                        final int x=s.indexOf(near);
103                                        if(x==0)
104                                                found=true;
105                                        else
106                                        if((x>0)&&(!Character.isLetter(s.charAt(x-1))))
107                                                found=true;
108                                        if((!s.startsWith("#"))&&(!s.startsWith("!"))&&(found))
109                                        {
110                                                where=p;
111                                                break;
112                                        }
113                                }
114                        }
115                        if(where>=0)
116                                page.add(where,key+"=");
117                        else
118                                page.add(key+"=");
119                        Log.sysOut("INIModify","Key '"+key+"' added.");
120                        updateINIFile(page);
121                        return "";
122                }
123                else
124                if(parms.containsKey("DELKEY"))
125                {
126                        String key=parms.get("KEY");
127                        if((key==null)||(key.trim().length()==0))
128                                return "";
129                        key=key.trim().toUpperCase();
130                        for(int p=0;p<page.size();p++)
131                        {
132                                final String s=page.get(p).trim();
133                                if(s.startsWith("!")||s.startsWith("#"))
134                                        continue;
135                                int x=s.indexOf('=');
136                                if(x<0)
137                                        x=s.indexOf(':');
138                                if(x<0)
139                                        continue;
140                                final String thisKey=s.substring(0,x).trim().toUpperCase();
141                                if(thisKey.equals(key))
142                                {
143                                        page.remove(p);
144                                        Log.sysOut("INIModify","Key '"+thisKey+"' removed.");
145                                        updateINIFile(page);
146                                        break;
147                                }
148                        }
149                        return "";
150                }
151                else
152                if(parms.containsKey("UPDATEFACTIONPRELOAD"))
153                {
154                        final String factionID=parms.get("FACTION");
155                        if((factionID!=null)&&(factionID.length()>0))
156                        {
157                                final CMProps ipage=CMProps.loadPropPage(CMProps.getVar(CMProps.Str.INIPATH));
158                                if((ipage==null)||(!ipage.isLoaded()))
159                                        return "";
160                                for(int p=0;p<page.size();p++)
161                                {
162                                        final String s=page.get(p).trim();
163                                        if(s.startsWith("!")||s.startsWith("#"))
164                                                continue;
165                                        int x=s.indexOf('=');
166                                        if(x<0)
167                                                x=s.indexOf(':');
168                                        if(x<0)
169                                                continue;
170                                        final String thisKey=s.substring(0,x).trim().toUpperCase();
171                                        if(thisKey.equals("FACTIONS"))
172                                        {
173                                                final StringBuilder newVal=new StringBuilder("");
174                                                final String oldVal=CMProps.getVar(CMProps.Str.PREFACTIONS);
175                                                final List<String> oldList=CMParms.parseSemicolons(oldVal,true);
176                                                boolean done=false;
177                                                for(final String facID : oldList)
178                                                {
179                                                        if(facID.equalsIgnoreCase(factionID))
180                                                        {
181                                                                done=true;
182                                                        }
183                                                        else
184                                                        {
185                                                                if(newVal.length()>0)
186                                                                        newVal.append("; ");
187                                                                newVal.append(facID);
188                                                        }
189                                                }
190                                                if(!done)
191                                                {
192                                                        if(newVal.length()>0)
193                                                                newVal.append("; ");
194                                                        newVal.append(factionID);
195                                                }
196                                                if(!oldVal.equals(newVal.toString()))
197                                                {
198                                                        Log.sysOut("INIModify","Key '"+thisKey+"' modified.");
199                                                        page.set(p,thisKey+"="+newVal.toString());
200                                                        CMProps.setVar(CMProps.Str.PREFACTIONS, newVal.toString());
201                                                        updateINIFile(page);
202                                                }
203                                                break;
204                                        }
205                                }
206                        }
207                        return "";
208                }
209                else
210                if(parms.containsKey("UPDATE"))
211                {
212                        final Set<String> modified=new HashSet<String>();
213                        final List<String> iniBuildVars=CMParms.parseCommas(CMStrings.s_uppercase(httpReq.getUrlParameter("INIBUILDVARS")), true);
214                        if(iniBuildVars.contains("CHANNELS"))
215                                httpReq.addFakeUrlParameter("CHANNELS", buildChannelsVar(httpReq));
216                        if(iniBuildVars.contains("COMMANDJOURNALS"))
217                                httpReq.addFakeUrlParameter("COMMANDJOURNALS", buildCommandJournalsVar(httpReq));
218                        if(iniBuildVars.contains("FORUMJOURNALS"))
219                                httpReq.addFakeUrlParameter("FORUMJOURNALS", buildForumJournalsVar(httpReq));
220                        if(iniBuildVars.contains("ICHANNELS"))
221                                httpReq.addFakeUrlParameter("ICHANNELS", buildIChannelsVar(httpReq));
222                        if(iniBuildVars.contains("IMC2CHANNELS"))
223                                httpReq.addFakeUrlParameter("IMC2CHANNELS", buildIMC2ChannelsVar(httpReq));
224
225                        CMProps ipage=CMProps.loadPropPage(CMProps.getVar(CMProps.Str.INIPATH));
226                        if((ipage==null)||(!ipage.isLoaded()))
227                                return "";
228                        for(int p=0;p<page.size();p++)
229                        {
230                                final String s=page.get(p).trim();
231                                if(s.startsWith("!")||s.startsWith("#"))
232                                        continue;
233                                int x=s.indexOf('=');
234                                if(x<0)
235                                        x=s.indexOf(':');
236                                if(x<0)
237                                        continue;
238                                final String thisKey=s.substring(0,x).trim().toUpperCase();
239
240                                if(httpReq.isUrlParameter(thisKey)
241                                &&(ipage.containsKey(thisKey))
242                                &&(!modified.contains(thisKey))
243                                &&(!httpReq.getUrlParameter(thisKey).equals(ipage.getStr(thisKey))))
244                                {
245                                        modified.add(thisKey);
246                                        Log.sysOut("INIModify","Key '"+thisKey+"' modified.");
247                                        page.set(p,thisKey+"="+httpReq.getUrlParameter(thisKey));
248                                }
249                        }
250                        if(modified.size()>0)
251                        {
252                                if(modified.contains("JSCRIPTS")) return ""; // never modified through this
253                                updateINIFile(page);
254                                ipage=CMProps.loadPropPage(CMProps.getVar(CMProps.Str.INIPATH));
255                                if((ipage==null)||(!ipage.isLoaded()))
256                                        return "";
257                                ipage.resetSystemVars();
258                                if(modified(modified,"SYSOPMASK"))
259                                        CMSecurity.setSysOp(ipage.getStr("SYSOPMASK"));
260                                if(modified(modified,"GROUP_*"))
261                                        CMSecurity.parseGroups(ipage);
262                                if(modified(modified,"START")||(modified(modified,"START_*")))
263                                        CMLib.login().initStartRooms(ipage);
264                                if(modified(modified,"DEATH")||(modified(modified,"DEATH_*")))
265                                        CMLib.login().initDeathRooms(ipage);
266                                if(modified(modified,"MORGUE")||(modified(modified,"MORGUE_*")))
267                                        CMLib.login().initBodyRooms(ipage);
268                                if(modified(modified,"FACTIONS"))
269                                        CMLib.factions().reloadFactions(CMProps.getVar(CMProps.Str.PREFACTIONS));
270                                if(modified(modified,"CHANNELS")
271                                ||(modified(modified,"ICHANNELS"))
272                                ||(modified(modified,"COMMANDJOURNALS"))
273                                ||(modified(modified,"FORUMJOURNALS"))
274                                ||(modified(modified,"IMC2CHANNELS")))
275                                {
276                                        final String normalChannels=ipage.getStr("CHANNELS");
277                                        final String i3Channels=ipage.getBoolean("RUNI3SERVER") ? ipage.getStr("ICHANNELS") : "";
278                                        final String imc2Channels=ipage.getBoolean("RUNIMC2CLIENT") ? ipage.getStr("IMC2CHANNELS") : "";
279                                        CMLib.channels().loadChannels(normalChannels,i3Channels,imc2Channels);
280                                        CMLib.journals().loadCommandJournals(ipage.getStr("COMMANDJOURNALS"));
281                                        CMLib.journals().loadForumJournals(ipage.getStr("FORUMJOURNALS"));
282                                }
283                                CMLib.time().globalClock().initializeINIClock(ipage);
284                        }
285                        return "";
286                }
287                return "";
288        }

290        protected String getChannelsValue(final HTTPRequest httpReq, final String index)
291        {
292                final String name=httpReq.getUrlParameter("CHANNEL_"+index+"_NAME");
293                final String mask=httpReq.getUrlParameter("CHANNEL_"+index+"_MASK");
294                final String colors=httpReq.getUrlParameter("CHANNEL_"+index+"_COLORS");
295                if((name!=null)&&(name.trim().length()>0)
296                &&(!name.trim().equalsIgnoreCase("auction")))
297                {
298                        final StringBuilder str=new StringBuilder("");
299                        str.append(name.trim().replace(',',' ').toUpperCase()).append(" ");
300                        if(colors.trim().length()>0)
301                                str.append(colors.trim().replace(',',' ').toUpperCase()).append(" ");
302                        String flagid="";
303                        for(int i=0;httpReq.isUrlParameter("CHANNEL_"+index+"_FLAG_"+flagid);flagid=""+(++i))
304                        {
305                                final String flagName=httpReq.getUrlParameter("CHANNEL_"+index+"_FLAG_"+flagid);
306                                final ChannelsLibrary.ChannelFlag flag=(ChannelsLibrary.ChannelFlag)CMath.s_valueOf(ChannelsLibrary.ChannelFlag.values(), flagName);
307                                if(flag != null)
308                                        str.append(flag.name()).append(" ");
309                        }
310                        if(mask.trim().length()>0)
311                                str.append(mask.trim().replace(',',' ')).append(" ");
312                        str.setLength(str.length()-1);
313                        return str.toString();
314                }
315                return null;
316        }

318        protected void addChannelsVar(final HTTPRequest httpReq, final String index, final StringBuilder str)
319        {
320                final String firstPart=getChannelsValue(httpReq,index);
321                if(firstPart!=null)
322                {
323                        final String i3Name=httpReq.getUrlParameter("CHANNEL_"+index+"_I3NAME");
324                        final String imc2Name=httpReq.getUrlParameter("CHANNEL_"+index+"_IMC2NAME");
325                        if(((i3Name!=null)&&(i3Name.trim().length()>0))
326                        ||((imc2Name!=null)&&(imc2Name.trim().length()>0)))
327                                return;
328                        if(str.length()>0)
329                                str.append(", ");
330                        str.append(firstPart);
331                }
332        }

334        protected String buildChannelsVar(final HTTPRequest httpReq)
335        {
336                final StringBuilder str=new StringBuilder("");
337                for(int index=0;httpReq.isUrlParameter("CHANNEL_"+index+"_NAME");index++)
338                        addChannelsVar(httpReq,Integer.toString(index),str);
339                addChannelsVar(httpReq,"",str);
340                return str.toString();
341        }

391        protected void addCommandJournalsVar(final HTTPRequest httpReq, final String index, final StringBuilder str)
392        {
393                final String name=httpReq.getUrlParameter("COMMANDJOURNAL_"+index+"_NAME");
394                final String mask=httpReq.getUrlParameter("COMMANDJOURNAL_"+index+"_MASK");
395                if((name!=null)&&(name.trim().length()>0)
396                &&(!name.trim().equalsIgnoreCase("auction")))
397                {
398                        if(str.length()>0)
399                                str.append(", ");
400                        str.append(name.trim().replace(',',' ').toUpperCase()).append(" ");
401                        for(final JournalsLibrary.CommandJournalFlags flag : JournalsLibrary.CommandJournalFlags.values())
402                        {
403                                final String val=httpReq.getUrlParameter("COMMANDJOURNAL_"+index+"_FLAG_"+flag.name());
404                                if((val!=null)&&(val.trim().length()>0))
405                                        str.append(flag.name()).append("=").append(val.replace(',', ' ')).append(" ");
406                        }
407                        if(mask.trim().length()>0)
408                                str.append(mask.trim().replace(',',' ')).append(" ");
409                        str.setLength(str.length()-1);
410                }
411        }
