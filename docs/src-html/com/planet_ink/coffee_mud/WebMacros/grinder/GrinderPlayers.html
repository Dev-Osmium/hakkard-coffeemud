045        public static String titleList(MOB E, HTTPRequest httpReq, java.util.Map<String,String> parms)
046        {
047                if(E.playerStats()==null)
048                        return "";
049                E.playerStats().getTitles().clear();
050                if(httpReq.isUrlParameter("TITLE0"))
051                {
052                        int num=0;
053                        while(httpReq.isUrlParameter("TITLE"+num))
054                        {
055                                final String aff=httpReq.getUrlParameter("TITLE"+num);
056                                if(aff.trim().length()>0)
057                                        E.playerStats().getTitles().add(aff.trim());
058                                num++;
059                        }
060                }
061                return "";
062        }

064        public static String setBasics(HTTPRequest httpReq, MOB M)
065        {
066                for(int i=0;i<BASICS.values().length;i++)
067                {
068                        String err = setBasic(httpReq, M, BASICS.values()[i]);
069                        if((err != null) && (err.trim().length()>0))
070                                return err;
071                }
072                return "";
073        }

087                                M.setDescription(old);

089                        case LASTDATETIME:
090                                if (M.playerStats() != null)

337                                                        M.playerStats().setAccountExpiration(System.currentTimeMillis());

342                                break;

347                                {

380                                M.playerStats().getTitles().add(old);
381                                b++;
382                        }
383                }
384                return "";
385        }

387        public static String classList(MOB M, HTTPRequest httpReq, java.util.Map<String,String> parms)
388        {
389                if(httpReq.isUrlParameter("CHARCLASS1"))
390                {
391                        final StringBuffer classList=new StringBuffer("");
392                        final StringBuffer levelsList=new StringBuffer("");
393                        int num=1;
394                        String aff=httpReq.getUrlParameter("CHARCLASS"+num);
395                        int totalLevel=0;
396                        while(aff!=null)
397                        {
398                                if(aff.length()>0)
399                                {
400                                        final CharClass C=CMClass.getCharClass(aff);
401                                        if(C==null)
402                                                return "Unknown class '"+aff+"'.";
403                                        classList.append(C.ID()+";");
404                                        String lvl=httpReq.getUrlParameter("CHARCLASSLVL"+num);
405                                        if(lvl==null)
406                                                lvl="0";
407                                        totalLevel+=CMath.s_int(lvl);
408                                        levelsList.append(lvl+";");
409                                }
410                                num++;
411                                aff=httpReq.getUrlParameter("CHARCLASS"+num);
412                        }
413                        M.baseCharStats().setMyClasses(classList.toString());
414                        M.baseCharStats().setMyLevels(levelsList.toString());
415                        M.basePhyStats().setLevel(totalLevel);
416                }
417                return "";
418        }

420        public static String editPlayer(MOB whom, HTTPRequest httpReq, java.util.Map<String,String> parms, MOB M)
421        {
422                if(!CMProps.getBoolVar(CMProps.Bool.MUDSTARTED))
423                        return CMProps.getVar(CMProps.Str.MUDSTATUS);
424
425                final Vector<Item> allitems=new Vector<Item>();
426                while(M.numItems()>0)
427                {
428                        final Item I=M.getItem(0);
429                        allitems.addElement(I);
430                        M.delItem(I);
431                }
432
433                for(MOB.Attrib a : MOB.Attrib.values())
434                {
435                        if(httpReq.isUrlParameter(a.getName()))
436                        {
437                                String old=httpReq.getUrlParameter(a.getName());
438                                if(old==null)
439                                        old="";
440                                if(old.equalsIgnoreCase("on"))
441                                        M.setAttribute(a,true);
442                                else
443                                        M.setAttribute(a,false);
444                        }
445                }
446                for(final int i : CharStats.CODES.ALLCODES())
447                {
448                        final CharStats C=M.charStats();
449                        final String stat=CharStats.CODES.NAME(i);
450                        if(httpReq.isUrlParameter(stat))
451                        {
452                                String old=httpReq.getUrlParameter(stat);
453                                if(old==null)
454                                        old="";
455                                if(!stat.equalsIgnoreCase("GENDER"))
456                                        C.setStat(i,CMath.s_int(old));
457                                else
458                                if(old.length()>0)
459                                        C.setStat(i,old.charAt(0));
460                        }
461                }
462                for(final int i : CharStats.CODES.ALLCODES())
463                {
464                        final CharStats C=M.baseCharStats();
465                        final String stat=CharStats.CODES.NAME(i);
466                        if(httpReq.isUrlParameter("BASE"+stat))
467                        {
468                                String old=httpReq.getUrlParameter("BASE"+stat);
469                                if(old==null)
470                                        old="";
471                                if(!stat.equalsIgnoreCase("GENDER"))
472                                        C.setStat(i,CMath.s_int(old));
473                                else
474                                if(old.length()>0)
475                                        C.setStat(i,old.charAt(0));
476                        }
477                }
478                GrinderPlayers.setBasics(httpReq,M);
479                if(httpReq.isUrlParameter("RACE"))
480                {
481                        final String old=httpReq.getUrlParameter("RACE");
482                        if((old!=null)&&(CMClass.getRace(old)!=null))
483                                M.baseCharStats().setMyRace(CMClass.getRace(old));
484                }
485                if(httpReq.isUrlParameter("DEITY"))
486                {
487                        final String old=httpReq.getUrlParameter("DEITY");
488                        if((old!=null)&&(CMLib.map().getDeity(old)!=null))
489                                M.setWorshipCharID(CMLib.map().getDeity(old).Name());
490                }
491                if(httpReq.isUrlParameter("ALIGNMENT"))
492                {
493                        final String old=httpReq.getUrlParameter("ALIGNMENT");
494                        final Faction F=CMLib.factions().getFaction(CMLib.factions().AlignID());
495                        if((F!=null)&&(old!=null)&&(old.length()>0))
496                        {
497                                for(final Faction.Align v : Faction.Align.values())
498                                {
499                                        if((v!=Faction.Align.INDIFF)&&(v.toString().equalsIgnoreCase(old)))
500                                                CMLib.factions().setAlignment(M,v);
501                                }
502                        }
503                }
504                String error=GrinderExits.dispositions(M,httpReq,parms);
505                if(error.length()>0)
506                        return error;
507                error=GrinderMobs.senses(M,httpReq,parms);
508                if(error.length()>0)
509                        return error;
510                error=titleList(M,httpReq,parms);
511                if(error.length()>0)
512                        return error;
513                error=GrinderAreas.doAffects(M,httpReq,parms);
514                if(error.length()>0)
515                        return error;
516                error=GrinderAreas.doBehavs(M,httpReq,parms);
517                if(error.length()>0)
518                        return error;
519                error=GrinderMobs.factions(M,httpReq,parms);
520                if(error.length()>0)
521                        return error;
522                error=GrinderMobs.abilities(M,httpReq,parms);
523                if(error.length()>0)
524                        return error;
525                error=GrinderMobs.items(M,allitems,httpReq);
526                if(error.length()>0)
527                        return error;
528                error=GrinderMobs.expertiseList(M,httpReq,parms);
529                if(error.length()>0)
530                        return error;
531                error=GrinderMobs.clans(M,httpReq,parms);
532                if(error.length()>0)
533                        return error;
534                error=classList(M,httpReq,parms);
535                if(error.length()>0)
536                        return error;
537                M.recoverPhyStats();
538                M.recoverCharStats();
539                M.recoverMaxState();
540                M.recoverPhyStats();
541                M.recoverCharStats();
542                M.recoverMaxState();
543                if(M.location()!=null)
544                        M.location().recoverRoomStats();
545                CMLib.database().DBUpdatePlayer(M);
546                Log.sysOut("Grinder",whom.Name()+" modified player "+M.Name());
547                return "";
548        }
