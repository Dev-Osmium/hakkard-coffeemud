036public class GrinderClanGovernments
037{
038        public String name()
039        {
040                return "GrinderClanGovernments";
041        }
042
043        public String runMacro(HTTPRequest httpReq, String parm)
044        {
045                final String last=httpReq.getUrlParameter("GOVERNMENT");
046                if(last==null)
047                        return " @break@";
048                if(last.length()>0)
049                {
050                        ClanGovernment G=null;
051                        if(CMath.isInteger(last))
052                        {
053                                final int lastID=CMath.s_int(last);
054                                G=CMLib.clans().getStockGovernment(lastID);
055                        }
056                        else
057                        if((httpReq.getUrlParameter("NAME")==null)||httpReq.getUrlParameter("NAME").length()==0)
058                                return " @break@";
059                        else
060                        {
061                                final Set<Integer> usedTypeIDs=new HashSet<Integer>();
062                                for(final ClanGovernment G2 : CMLib.clans().getStockGovernments())
063                                        usedTypeIDs.add(Integer.valueOf(G2.getID()));
064                                G=CMLib.clans().createGovernment(httpReq.getUrlParameter("NAME"));
065                                for(int i=0;i<CMLib.clans().getStockGovernments().length;i++)
066                                {
067                                        if(!usedTypeIDs.contains(Integer.valueOf(i)))
068                                        {
069                                                httpReq.addFakeUrlParameter("GOVERNMENT", Integer.toString(i));
070                                                G.setID(i);
071                                                break;
072                                        }
073                                }
074                        }
075
076                        String str=null;
077                        str=httpReq.getUrlParameter("NAME");
078                        if(str!=null)
079                                G.setName(str);
080                        str=httpReq.getUrlParameter("CATEGORY");
081                        if(str!=null)
082                                G.setCategory(str);
083                        str=httpReq.getUrlParameter("ACCEPTPOS");
084                        if(str!=null)
085                                G.setAcceptPos(CMath.s_int(str));
086                        str=httpReq.getUrlParameter("AUTOROLE");
087                        if(str!=null)
088                                G.setAutoRole(CMath.s_int(str));
089                        str=httpReq.getUrlParameter("SHORTDESC");
090                        if(str!=null)
091                                G.setShortDesc(str);
092                        str=httpReq.getUrlParameter("REQUIREDMASK");
093                        if(str!=null)
094                        {
095                                G.setRequiredMaskStr(str);
096                        }
097                        str=httpReq.getUrlParameter("ENTRYSCRIPT");
098                        if(str!=null)
099                        {
100                                G.setEntryScript(str);
101                        }
102                        str=httpReq.getUrlParameter("EXITSCRIPT");
103                        if(str!=null)
104                        {
105                                G.setExitScript(str);
106                        }
107                        str=httpReq.getUrlParameter("ISPUBLIC");
108                        G.setPublic((str==null)?false:str.equalsIgnoreCase("on"));
109                        str=httpReq.getUrlParameter("ISFAMILYONLY");
110                        G.setFamilyOnly((str==null)?false:str.equalsIgnoreCase("on"));
111                        str=httpReq.getUrlParameter("XPLEVELFORMULA");
112                        if(str!=null)
113                                G.setXpCalculationFormulaStr(str);
114                        str=httpReq.getUrlParameter("OVERRIDEMINMEMBERS");
115                        if(str!=null)
116                                G.setOverrideMinMembers((str.trim().length()==0)?null:Integer.valueOf(CMath.s_int(str)));
117                        str=httpReq.getUrlParameter("CONQUESTENABLED");
118                        G.setConquestEnabled((str==null)?false:str.equalsIgnoreCase("on"));
119                        str=httpReq.getUrlParameter("CONQUESTITEMLOYALTY");
120                        G.setConquestItemLoyalty((str==null)?false:str.equalsIgnoreCase("on"));
121                        str=httpReq.getUrlParameter("CONQUESTDEITYBASIS");
122                        G.setConquestByWorship((str==null)?false:str.equalsIgnoreCase("on"));
123                        str=httpReq.getUrlParameter("ISRIVALROUS");
124                        G.setRivalrous((str==null)?false:str.equalsIgnoreCase("on"));
125                        str=httpReq.getUrlParameter("MAXVOTEDAYS");
126                        if(str!=null)
127                                G.setMaxVoteDays(CMath.s_int(str));
128                        str=httpReq.getUrlParameter("VOTEQUORUMPCT");
129                        if(str!=null)
130                                G.setVoteQuorumPct(CMath.s_int(str));
131                        str=httpReq.getUrlParameter("AUTOPROMOTEBY");
132                        if(str!=null)
133                                G.setAutoPromoteBy((Clan.AutoPromoteFlag)CMath.s_valueOf(Clan.AutoPromoteFlag.values(), str));
134                        str=httpReq.getUrlParameter("LONGDESC");
135                        if(str!=null)
136                                G.setLongDesc(str);
137                        final String old=httpReq.getUrlParameter("VOTEFUNCS");
138                        final Set<String> voteFuncs=new HashSet<String>();
139                        if((old!=null)&&(old.length()>0))
140                        {
141                                voteFuncs.add(old);
142                                int x=1;
143                                while(httpReq.getUrlParameter("VOTEFUNCS"+x)!=null)
144                                {
145                                        voteFuncs.add(httpReq.getUrlParameter("VOTEFUNCS"+x));
146                                        x++;
147                                }
148                        }
149
150                        final List<ClanPosition> posList=new Vector<ClanPosition>();
151                        String posDexStr="0";
152                        int posDex=0;
153                        while(httpReq.isUrlParameter("GPOSID_"+posDexStr) && httpReq.getUrlParameter("GPOSID_"+posDexStr).trim().length()>0)
154                        {
155                                final String oldID=httpReq.getUrlParameter("GPOSID_"+posDexStr);
156                                final String oldName=httpReq.getUrlParameter("GPOSNAME_"+posDexStr);
157                                final String oldPluralName=httpReq.getUrlParameter("GPOSPLURALNAME_"+posDexStr);
158                                final int oldRoleID=CMath.s_int(httpReq.getUrlParameter("GPOSROLEID_"+posDexStr));
159                                final int oldRank=CMath.s_int(httpReq.getUrlParameter("GPOSRANK_"+posDexStr));
160                                final int oldMax=CMath.s_int(httpReq.getUrlParameter("GPOSMAX_"+posDexStr));
161                                final String oldMask=httpReq.getUrlParameter("GPOSINNERMASK_"+posDexStr);
162                                final String oldIsPublicStr=httpReq.getUrlParameter("GPOSISPUBLIC_"+posDexStr);
163                                final boolean oldIsPublic=oldIsPublicStr==null?false:oldIsPublicStr.equalsIgnoreCase("on");
164                                final Clan.Authority powerFuncs[]=new Clan.Authority[Clan.Function.values().length];
165                                for(int f=0;f<Clan.Function.values().length;f++)
166                                        powerFuncs[f]=Clan.Authority.CAN_NOT_DO;
167                                String authDexStr="";
168                                int authDex=0;
169                                while(httpReq.getUrlParameter("GPOSPOWER_"+posDexStr+"_"+authDexStr)!=null)
170                                {
171                                        final Clan.Function auth = (Clan.Function)CMath.s_valueOf(Clan.Function.values(),httpReq.getUrlParameter("GPOSPOWER_"+posDexStr+"_"+authDexStr));
172                                        powerFuncs[auth.ordinal()]=Clan.Authority.CAN_DO;
173                                        authDex++;
174                                        authDexStr=Integer.toString(authDex);
175                                }
176                                for(final String s : voteFuncs)
177                                {
178                                        final Clan.Function auth = (Clan.Function)CMath.s_valueOf(Clan.Function.values(),s);
179                                        powerFuncs[auth.ordinal()]=Clan.Authority.MUST_VOTE_ON;
180                                }
181                                final ClanPosition P=(ClanPosition)CMClass.getCommon("DefaultClanPosition");
182                                P.setID(oldID);
183                                P.setRoleID(oldRoleID);
184                                P.setRank(oldRank);
185                                P.setName(oldName);
186                                P.setPluralName(oldPluralName);
187                                P.setMax(oldMax);
188                                P.setInnerMaskStr(oldMask);
189                                P.setFunctionChart(powerFuncs);
190                                P.setPublic(oldIsPublic);
191                                posList.add(P);
192                                posDex++;
193                                posDexStr=Integer.toString(posDex);
194                        }
195                        G.setPositions(posList.toArray(new ClanPosition[0]));
196                        GrinderRaces.setDynAbilities(G,httpReq);
197                        GrinderRaces.setDynEffects(G,httpReq);
198                }
199
200                return "";
201        }
202}
