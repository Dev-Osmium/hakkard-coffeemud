044        public static DVector cabilities(HTTPRequest httpReq)
045        {
046                final DVector theclasses=new DVector(9);
047                if(httpReq.isUrlParameter("CABLES1"))
048                {
049                        int num=1;
050                        String behav=httpReq.getUrlParameter("CABLES"+num);
051                        while(behav!=null)
052                        {
053                                if(behav.length()>0)
054                                {
055                                        String prof=httpReq.getUrlParameter("CABPOF"+num);
056                                        if(prof==null)
057                                                prof="0";
058                                        String qual=httpReq.getUrlParameter("CABQUA"+num);
059                                        if(qual==null) 
060                                                qual="";// null means unchecked
061                                        String levl=httpReq.getUrlParameter("CABLVL"+num);
062                                        if(levl==null)
063                                                levl="0";
064                                        String secr=httpReq.getUrlParameter("CABSCR"+num);
065                                        if(secr==null) 
066                                                secr="";// null means unchecked
067                                        Object parm=httpReq.getUrlParameter("CABPRM"+num);
068                                        if(parm==null)
069                                                parm="";
070                                        Object prereq=httpReq.getUrlParameter("CABPRE"+num);
071                                        if(prereq==null)
072                                                prereq="";
073                                        Object mask=httpReq.getUrlParameter("CABMSK"+num);
074                                        if(mask==null)
075                                                mask="";
076                                        String maxp=httpReq.getUrlParameter("CABMPOF"+num);
077                                        if(maxp==null)
078                                                maxp="100";
079                                        theclasses.addElement(behav,levl,prof,qual,secr,parm,prereq,mask,maxp);
080                                }
081                                num++;
082                                behav=httpReq.getUrlParameter("CABLES"+num);
083                        }
084                }
085                return theclasses;
086        }

088        public static String modifyCharClass(HTTPRequest httpReq, java.util.Map<String,String> parms, CharClass oldC, CharClass C)
089        {
090                final String replaceCommand=httpReq.getUrlParameter("REPLACE");
091                if((replaceCommand != null)
092                && (replaceCommand.length()>0)
093                && (replaceCommand.indexOf('=')>0))
094                {
095                        final int eq=replaceCommand.indexOf('=');
096                        final String field=replaceCommand.substring(0,eq);
097                        final String value=replaceCommand.substring(eq+1);
098                        httpReq.addFakeUrlParameter(field, value);
099                        httpReq.addFakeUrlParameter("REPLACE","");
100                }
101                String old;
102                // names are numerous
103                DVector DV=new DVector(2);
104                int num=0;
105                while(httpReq.isUrlParameter("NAME"+(++num)))
106                {
107                        if(CMath.isInteger(httpReq.getUrlParameter("NAMELEVEL"+(num))))
108                        {
109                                final int minLevel = CMath.s_int(httpReq.getUrlParameter("NAMELEVEL"+(num)));
110                                final String name=httpReq.getUrlParameter("NAME"+(num));
111                                if((name!=null)&&(name.length()>0))
112                                {
113                                        if(DV.size()==0)
114                                                DV.addElement(Integer.valueOf(minLevel),name);
115                                        else
116                                        {
117                                                boolean added=false;
118                                                for(int n=0;n<DV.size();n++)
119                                                {
120                                                        if(minLevel<((Integer)DV.elementAt(n,1)).intValue())
121                                                        {
122                                                                DV.insertElementAt(n,Integer.valueOf(minLevel),name);
123                                                                added=true;
124                                                                break;
125                                                        }
126                                                        else
127                                                        if(minLevel==((Integer)DV.elementAt(n,1)).intValue())
128                                                        {
129                                                                added=true;
130                                                                break;
131                                                        }
132                                                }
133                                                if(!added)
134                                                        DV.addElement(Integer.valueOf(minLevel),name);
135                                        }
136                                }
137                        }
138                }
139                if(DV.size()==0)
140                        DV.addElement(Integer.valueOf(0),C.ID());
141                C.setStat("NUMNAME",""+DV.size());
142                for(int l=0;l<DV.size();l++)
143                {
144                        C.setStat("NAME"+l, (String)DV.elementAt(l,2));
145                        C.setStat("NAMELEVEL"+l, ((Integer)DV.elementAt(l,1)).toString());
146                }
147
148                old=httpReq.getUrlParameter("");
149                C.setStat("",(old==null)?"":old);
150
151                old=httpReq.getUrlParameter("BASE");
152                C.setStat("BASE",(old==null)?"BASECLASS":old);
153                old=httpReq.getUrlParameter("HITPOINTSFORMULA");
154                C.setStat("HITPOINTSFORMULA",(old==null)?"1":old);
155                old=httpReq.getUrlParameter("MANAFORMULA");
156                C.setStat("MANAFORMULA",(old==null)?"1":old);
157                old=httpReq.getUrlParameter("MOVEMENTFORMULA");
158                C.setStat("MOVEMENTFORMULA",(old==null)?"1":old);
159                old=httpReq.getUrlParameter("LVLPRAC");
160                C.setStat("LVLPRAC",(old==null)?"0":old);
161                old=httpReq.getUrlParameter("LVLATT");
162                C.setStat("LVLATT",(old==null)?"0":old);
163                old=httpReq.getUrlParameter("ATTATT");
164                C.setStat("ATTATT",(old==null)?"0":old);
165                old=httpReq.getUrlParameter("FSTTRAN");
166                C.setStat("FSTTRAN",(old==null)?"0":old);
167                old=httpReq.getUrlParameter("FSTPRAC");
168                C.setStat("FSTPRAC",(old==null)?"0":old);
169                old=httpReq.getUrlParameter("LVLDAM");
170                C.setStat("LVLDAM",(old==null)?"10":old);
171                old=httpReq.getUrlParameter("MAXNCS");
172                C.setStat("MAXNCS",(old==null)?"0":old);
173                old=httpReq.getUrlParameter("MAXCRS");
174                C.setStat("MAXCRS",(old==null)?"0":old);
175                old=httpReq.getUrlParameter("MONEY");
176                C.setStat("MONEY",(old==null)?"":old);
177                old=httpReq.getUrlParameter("MAXCMS");
178                C.setStat("MAXCMS",(old==null)?"0":old);
179                old=httpReq.getUrlParameter("MAXLGS");
180                C.setStat("MAXLGS",(old==null)?"0":old);
181                old=httpReq.getUrlParameter("SUBRUL");
182                C.setStat("SUBRUL", (old==null)?"BASE":old);
183                old=httpReq.getUrlParameter("LEVELCAP");
184                C.setStat("LEVELCAP",(old==null)?"-1":old);
185                old=httpReq.getUrlParameter("ARMOR");
186                C.setStat("ARMOR",(old==null)?"0":old);
187                old=httpReq.getUrlParameter("STRLMT");
188                C.setStat("STRLMT",(old==null)?"STRLMT":old);
189                old=httpReq.getUrlParameter("STRBON");
190                C.setStat("STRBON",(old==null)?"STRBON":old);
191                old=httpReq.getUrlParameter("QUAL");
192                C.setStat("QUAL",(old==null)?"":old);
193                old=httpReq.getUrlParameter("PLAYER");
194                C.setStat("PLAYER",(old==null)?"0":old);
195                C.setStat("ESTATS",GrinderRaces.getPStats('E',httpReq));
196                C.setStat("CSTATS",GrinderRaces.getCStats('S',httpReq));
197                C.setStat("ASTATS",GrinderRaces.getCStats('A',httpReq));
198                C.setStat("ASTATE",GrinderRaces.getCState('A',httpReq));
199                C.setStat("STARTASTATE",GrinderRaces.getCState('S',httpReq));
200                old=httpReq.getUrlParameter("GENHELP");
201                C.setStat("HELP", ((old==null)?"":old));
202                old=httpReq.getUrlParameter("RACQUAL");
203                C.setStat("RACQUAL",(old==null)?"All":old);
204                String id="";
205                List<String> V=new Vector<String>();
206                for(int i=0;httpReq.isUrlParameter("NOWEAPS"+id);id=""+(++i))
207                        V.add(httpReq.getUrlParameter("NOWEAPS"+id));
208                C.setStat("GETWEP",CMParms.toListString(V));
209                int x=0;
210                final List<Pair<String,Integer>> minStats=new LinkedList<Pair<String,Integer>>();
211                while(httpReq.getUrlParameter("MINSTAT"+x)!=null)
212                {
213                        final String minStat=httpReq.getUrlParameter("MINSTAT"+x);
214                        final String statMin=httpReq.getUrlParameter("STATMIN"+x);
215                        if((minStat!=null)&&(minStat.length()>0)&&(CMath.isInteger(statMin)))
216                                minStats.add(new Pair<String,Integer>(minStat,Integer.valueOf(CMath.s_int(statMin))));
217                        x++;
218                }
219                C.setStat("NUMMINSTATS", ""+minStats.size());
220                for(int m=0;m<minStats.size();m++)
221                {
222                        C.setStat("GETMINSTAT"+m, minStats.get(m).first);
223                        C.setStat("GETSTATMIN"+m, minStats.get(m).second.toString());
224                }
225                final List<Item> Ivs=GrinderRaces.itemList(oldC.outfit(null),'O',httpReq,false);
226                C.setStat("NUMOFT",""+Ivs.size());
227                for(int l=0;l<Ivs.size();l++)
228                {
229                        C.setStat("GETOFTID"+l,((Environmental)Ivs.get(l)).ID());
230                        C.setStat("GETOFTPARM"+l,((Environmental)Ivs.get(l)).text());
231                }
232                C.setStat("DISFLAGS",""+CMath.s_long(httpReq.getUrlParameter("DISFLAGS")));
233                num=0;
234                DV.clear();
235                while(httpReq.isUrlParameter("SSET"+(++num)))
236                {
237                        if(CMath.isInteger(httpReq.getUrlParameter("SSETLEVEL"+(num))))
238                        {
239                                final int minLevel = CMath.s_int(httpReq.getUrlParameter("SSETLEVEL"+(num)));
240                                final String name=httpReq.getUrlParameter("SSET"+(num));
241                                if((name!=null)&&(name.length()>0))
242                                {
243                                        if(DV.size()==0)
244                                                DV.addElement(Integer.valueOf(minLevel),name);
245                                        else
246                                        {
247                                                boolean added=false;
248                                                for(int n=0;n<DV.size();n++)
249                                                {
250                                                        if(minLevel<((Integer)DV.elementAt(n,1)).intValue())
251                                                        {
252                                                                DV.insertElementAt(n,Integer.valueOf(minLevel),name);
253                                                                added=true;
254                                                                break;
255                                                        }
256                                                        else
257                                                        if(minLevel==((Integer)DV.elementAt(n,1)).intValue())
258                                                        {
259                                                                added=true;
260                                                                break;
261                                                        }
262                                                }
263                                                if(!added)
264                                                        DV.addElement(Integer.valueOf(minLevel),name);
265                                        }
266                                }
267                        }
268                }
269                C.setStat("NUMSSET",""+DV.size());
270                for(int l=0;l<DV.size();l++)
271                {
272                        final String sec=(String)DV.elementAt(l,2);
273                        V=CMParms.parseCommas(sec, true);
274                        C.setStat("SSET"+l, CMParms.combineQuoted(V,0));
275                        C.setStat("SSETLEVEL"+l, ((Integer)DV.elementAt(l,1)).toString());
276                }
277                id="";
278                V=new Vector<String>();
279                for(int i=0;httpReq.isUrlParameter("WEAPMATS"+id);id=""+(++i))
280                {
281                        if(CMath.isInteger(httpReq.getUrlParameter("WEAPMATS"+id)))
282                                V.add(httpReq.getUrlParameter("WEAPMATS"+id));
283                }
284                C.setStat("NUMWMAT",""+V.size());
285                C.setStat("GETWMAT",CMParms.toListString(V));
286                old=httpReq.getUrlParameter("ARMORMINOR");
287                C.setStat("ARMORMINOR",(old==null)?"-1":old);
288                old=httpReq.getUrlParameter("STATCLASS");
289                C.setStat("STATCLASS",(old==null)?"":old);
290                old=httpReq.getUrlParameter("EVENTCLASS");
291                C.setStat("EVENTCLASS",(old==null)?"":old);
292                DV=cabilities(httpReq);
293                C.setStat("NUMCABLE", ""+DV.size());
294                for(int i=0;i<DV.size();i++)
295                {
296                        C.setStat("GETCABLELVL"+i, (String)DV.elementAt(i,2));
297                        C.setStat("GETCABLEPROF"+i, (String)DV.elementAt(i,3));
298                        C.setStat("GETCABLEGAIN"+i, ((String)DV.elementAt(i,4)).equalsIgnoreCase("on")?"false":"true");
299                        C.setStat("GETCABLESECR"+i, ((String)DV.elementAt(i,5)).equalsIgnoreCase("on")?"true":"false");
300                        if(DV.elementAt(i,6) instanceof String)
301                                C.setStat("GETCABLEPARM"+i, (String)DV.elementAt(i,6));
302                        if(DV.elementAt(i,7) instanceof String)
303                                C.setStat("GETCABLEPREQ"+i, (String)DV.elementAt(i,7));
304                        if(DV.elementAt(i,8) instanceof String)
305                                C.setStat("GETCABLEMASK"+i, (String)DV.elementAt(i,8));
306                        C.setStat("GETCABLEMAXP"+i, (String)DV.elementAt(i,9));
307                        // CABLE MUST BE LAST
308                        C.setStat("GETCABLE"+i, (String)DV.elementAt(i,1));
309                }
310                return "";
311        }
