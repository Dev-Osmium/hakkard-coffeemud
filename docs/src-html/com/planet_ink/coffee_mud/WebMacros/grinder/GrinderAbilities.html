045        public static String modifyAbility(HTTPRequest httpReq, java.util.Map<String,String> parms, Ability oldA, Ability A)
046        {
047                final String replaceCommand=httpReq.getUrlParameter("REPLACE");
048                if((replaceCommand != null)
049                && (replaceCommand.length()>0)
050                && (replaceCommand.indexOf('=')>0))
051                {
052                        final int eq=replaceCommand.indexOf('=');
053                        final String field=replaceCommand.substring(0,eq);
054                        final String value=replaceCommand.substring(eq+1);
055                        httpReq.addFakeUrlParameter(field, value);
056                        httpReq.addFakeUrlParameter("REPLACE","");
057                }
058                String old;
059                old=httpReq.getUrlParameter("NAME");
060                A.setStat("NAME",(old==null)?"NAME":old);
061                int x1=CMath.s_int(httpReq.getUrlParameter("CLASSIFICATION_ACODE"));
062                final int x2=CMath.s_int(httpReq.getUrlParameter("CLASSIFICATION_DOMAIN"));
063                A.setStat("CLASSIFICATION",""+((x2<<5)+x1));
064                old=httpReq.getUrlParameter("TRIGSTR");
065                A.setStat("TRIGSTR",(old==null)?"TRIGSTR":old.toUpperCase().trim());
066                old=httpReq.getUrlParameter("MINRANGE");
067                A.setStat("MINRANGE",(old==null)?"0":old);
068                old=httpReq.getUrlParameter("MAXRANGE");
069                A.setStat("MAXRANGE",(old==null)?"":old);
070                old=httpReq.getUrlParameter("TICKSBETWEENCASTS");
071                A.setStat("TICKSBETWEENCASTS",(old==null)?"0":old);
072                old=httpReq.getUrlParameter("TICKSOVERRIDE");
073                A.setStat("TICKSOVERRIDE",(old==null)?"0":old);
074                old=httpReq.getUrlParameter("DISPLAY");
075                A.setStat("DISPLAY",(old==null)?"DISPLAY":old);
076                old=httpReq.getUrlParameter("TICKAFFECTS");
077                A.setStat("TICKAFFECTS",(old==null)?"":""+old.equalsIgnoreCase("on"));
078                old=httpReq.getUrlParameter("AUTOINVOKE");
079                A.setStat("AUTOINVOKE",(old==null)?"":""+old.equalsIgnoreCase("on"));
080                final Vector<String> V=new Vector<String>();
081                if(httpReq.isUrlParameter("ABILITY_FLAGS"))
082                {
083                        String id="";
084                        int num=0;
085                        for(;httpReq.isUrlParameter("ABILITY_FLAGS"+id);id=""+(++num))
086                                V.addElement(httpReq.getUrlParameter("ABILITY_FLAGS"+id));
087                }
088                A.setStat("FLAGS",CMParms.toListString(V));
089                old=httpReq.getUrlParameter("GENHELP");
090                A.setStat("HELP", old==null?"":old);
091                old=httpReq.getUrlParameter("OVERRIDEMANA");
092                x1=CMath.s_int(old);
093                if(((x1>0)&&(x1<Ability.COST_PCT)))
094                        old=httpReq.getUrlParameter("CUSTOMOVERRIDEMANA");
095                A.setStat("OVERRIDEMANA",(old==null)?"-1":old);
096                V.clear();
097                if(httpReq.isUrlParameter("USAGEMASK"))
098                {
099                        String id="";
100                        int num=0;
101                        for(;httpReq.isUrlParameter("USAGEMASK"+id);id=""+(++num))
102                                V.addElement(httpReq.getUrlParameter("USAGEMASK"+id));
103                }
104                A.setStat("USAGEMASK",CMParms.toListString(V));
105                V.clear();
106                if(httpReq.isUrlParameter("MATLIST"))
107                {
108                        String id="";
109                        int num=0;
110                        for(;httpReq.isUrlParameter("MATLIST"+id);id=""+(++num))
111                                V.addElement(httpReq.getUrlParameter("MATLIST"+id));
112                }
113                A.setStat("MATLIST",CMParms.toListString(V));
114                V.clear();
115                if(httpReq.isUrlParameter("CANAFFECTMASK"))
116                {
117                        String id="";
118                        int num=0;
119                        for(;httpReq.isUrlParameter("CANAFFECTMASK"+id);id=""+(++num))
120                                V.addElement(httpReq.getUrlParameter("CANAFFECTMASK"+id));
121                }
122                A.setStat("CANAFFECTMASK",CMParms.toListString(V));
123                V.clear();
124                if(httpReq.isUrlParameter("CANTARGETMASK"))
125                {
126                        String id="";
127                        int num=0;
128                        for(;httpReq.isUrlParameter("CANTARGETMASK"+id);id=""+(++num))
129                                V.addElement(httpReq.getUrlParameter("CANTARGETMASK"+id));
130                }
131                A.setStat("CANTARGETMASK",CMParms.toListString(V));
132                old=httpReq.getUrlParameter("CANMEND");
133                A.setStat("CANMEND",(old==null)?"false":Boolean.toString(old.equalsIgnoreCase("on")));
134                old=httpReq.getUrlParameter("CANREFIT");
135                A.setStat("CANREFIT",(old==null)?"false":Boolean.toString(old.equalsIgnoreCase("on")));
136                old=httpReq.getUrlParameter("CANBUNDLE");
137                A.setStat("CANBUNDLE",(old==null)?"false":Boolean.toString(old.equalsIgnoreCase("on")));
138                old=httpReq.getUrlParameter("CANSIT");
139                A.setStat("CANSIT",(old==null)?"false":Boolean.toString(old.equalsIgnoreCase("on")));
140                old=httpReq.getUrlParameter("SOUND");
141                A.setStat("SOUND",(old==null)?"":old);
142                old=httpReq.getUrlParameter("VERB");
143                A.setStat("VERB",(old==null)?"":old);
144                old=httpReq.getUrlParameter("FILENAME");
145                A.setStat("FILENAME",(old==null)?"":old);
146                old=httpReq.getUrlParameter("VQUALITY");
147                A.setStat("QUALITY",(old==null)?"":old);
148                old=httpReq.getUrlParameter("HERESTATS");
149                A.setStat("HERESTATS",(old==null)?"":old);
150                old=httpReq.getUrlParameter("SCRIPT");
151                A.setStat("SCRIPT",(old==null)?"":old);
152                old=httpReq.getUrlParameter("CASTMASK");
153                A.setStat("CASTMASK",(old==null)?"":old);
154                old=httpReq.getUrlParameter("TARGETMASK");
155                A.setStat("TARGETMASK",(old==null)?"":old);
156                old=httpReq.getUrlParameter("FIZZLEMSG");
157                A.setStat("FIZZLEMSG",(old==null)?"":old);
158                old=httpReq.getUrlParameter("AUTOCASTMSG");
159                A.setStat("AUTOCASTMSG",(old==null)?"":old);
160                old=httpReq.getUrlParameter("CASTMSG");
161                A.setStat("CASTMSG",(old==null)?"":old);
162                old=httpReq.getUrlParameter("POSTCASTMSG");
163                A.setStat("POSTCASTMSG",(old==null)?"":old);
164                old=httpReq.getUrlParameter("ATTACKCODE");
165                A.setStat("ATTACKCODE",(old==null)?"0":old);
166                old=httpReq.getUrlParameter("POSTCASTDAMAGE");
167                A.setStat("POSTCASTDAMAGE",(old==null)?"":old);
168                V.clear();
169                if(httpReq.isUrlParameter("POSTCASTAFFECT"))
170                {
171                        String id="";
172                        int num=0;
173                        for(;httpReq.isUrlParameter("POSTCASTAFFECT"+id);id=""+(++num))
174                                V.addElement(httpReq.getUrlParameter("POSTCASTAFFECT"+id));
175                }
176                A.setStat("POSTCASTAFFECT",CMParms.toSemicolonListString(V));
177                V.clear();
178                if(httpReq.isUrlParameter("POSTCASTABILITY"))
179                {
180                        String id="";
181                        int num=0;
182                        for(;httpReq.isUrlParameter("POSTCASTABILITY"+id);id=""+(++num))
183                                V.addElement(httpReq.getUrlParameter("POSTCASTABILITY"+id));
184                }
185                A.setStat("POSTCASTABILITY",CMParms.toSemicolonListString(V));
186                if(A instanceof Language)
187                {
188                        ((Language)A).translationLists(A.ID()).clear();
189                        if(httpReq.isUrlParameter("WORDLIST1"))
190                        {
191                                int x=1;
192                                while(httpReq.isUrlParameter("WORDLIST"+x))
193                                {
194                                        ((Language)A).translationLists(A.ID())
195                                        .add(CMParms.parseCommas(httpReq.getUrlParameter("WORDLIST"+x), true).toArray(new String[0]));
196                                        x++;
197                                }
198                        }
199                        for(int i=((Language)A).translationLists(A.ID()).size()-1;i>=0;i--)
200                                if(((Language)A).translationLists(A.ID()).get(i).length==0)
201                                        ((Language)A).translationLists(A.ID()).remove(i);
202                                else
203                                        break;
204                        ((Language)A).translationHash(A.ID()).clear();
205                        if(httpReq.isUrlParameter("HASHWORD1"))
206                        {
207                                int x=1;
208                                while(httpReq.isUrlParameter("HASHWORD"+x))
209                                {
210                                        final String word=httpReq.getUrlParameter("HASHWORD"+x).toUpperCase().trim();
211                                        final String def=httpReq.getUrlParameter("HASHWORDDEF"+x);
212                                        if((def!=null)&&(def.length()>0)&&(word.length()>0))
213                                                ((Language)A).translationHash(A.ID()).put(word,def);
214                                        x++;
215                                }
216                        }
217                }
218                return "";
219        }
