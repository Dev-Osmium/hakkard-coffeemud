038        public static String getAreaList(Enumeration<Area> a, Area pickedA, MOB mob, boolean noInstances)
039        {
040                final StringBuffer AreaList=new StringBuffer("");
041                final boolean anywhere=(CMSecurity.isAllowedAnywhere(mob,CMSecurity.SecFlag.CMDROOMS)||CMSecurity.isAllowedAnywhere(mob,CMSecurity.SecFlag.CMDAREAS));
042                final boolean everywhere=(CMSecurity.isASysOp(mob)||CMSecurity.isAllowedEverywhere(mob,CMSecurity.SecFlag.CMDROOMS)||CMSecurity.isAllowedEverywhere(mob,CMSecurity.SecFlag.CMDAREAS));
043                for(;a.hasMoreElements();)
044                {
045                        final Area A=a.nextElement();
046                        if((everywhere||(A.amISubOp(mob.Name())&&anywhere))
047                        &&((!noInstances)||(!CMath.bset(A.flags(), Area.FLAG_INSTANCE_CHILD))))
048                        {
049                                if((pickedA!=null)&&(pickedA==A))
050                                        AreaList.append("<OPTION SELECTED VALUE=\""+A.Name()+"\">"+A.name());
051                                else
052                                        AreaList.append("<OPTION VALUE=\""+A.Name()+"\">"+A.name());
053                        }
054                }
055                return AreaList.toString();
056        }

058        public static String doBehavs(PhysicalAgent E, HTTPRequest httpReq, java.util.Map<String,String> parms)
059        {
060                E.delAllBehaviors();
061                if(httpReq.isUrlParameter("BEHAV1"))
062                {
063                        int num=1;
064                        String behav=httpReq.getUrlParameter("BEHAV"+num);
065                        String theparm=httpReq.getUrlParameter("BDATA"+num);
066                        while((behav!=null)&&(theparm!=null))
067                        {
068                                if(behav.length()>0)
069                                {
070                                        final Behavior B=CMClass.getBehavior(behav);
071                                        if(B==null)
072                                                return "Unknown behavior '"+behav+"'.";
073                                        B.setParms(theparm);
074                                        E.addBehavior(B);
075                                        B.startBehavior(E);
076                                }
077                                num++;
078                                behav=httpReq.getUrlParameter("BEHAV"+num);
079                                theparm=httpReq.getUrlParameter("BDATA"+num);
080                        }
081                }
082                return "";
083        }

085        public static String doAffects(Physical P, HTTPRequest httpReq, java.util.Map<String,String> parms)
086        {
087                P.delAllEffects(false);
088                if(httpReq.isUrlParameter("AFFECT1"))
089                {
090                        int num=1;
091                        String aff=httpReq.getUrlParameter("AFFECT"+num);
092                        String theparm=httpReq.getUrlParameter("ADATA"+num);
093                        while((aff!=null)&&(theparm!=null))
094                        {
095                                if(aff.length()>0)
096                                {
097                                        final Ability B=CMClass.getAbility(aff);
098                                        if(B==null)
099                                                return "Unknown Effect '"+aff+"'.";
100                                        B.setMiscText(theparm);
101                                        P.addNonUninvokableEffect(B);
102                                }
103                                num++;
104                                aff=httpReq.getUrlParameter("AFFECT"+num);
105                                theparm=httpReq.getUrlParameter("ADATA"+num);
106                        }
107                }
108                return "";
109        }

111        public static String modifyArea(HTTPRequest httpReq, java.util.Map<String,String> parms)
112        {
113                final Vector<Area> areasNeedingUpdates=new Vector<Area>();
114                final String last=httpReq.getUrlParameter("AREA");
115                if((last==null)||(last.length()==0))
116                        return "Old area name not defined!";
117                Area A=CMLib.map().getArea(last);
118                if(A==null)
119                        return "Old Area not defined!";
120                areasNeedingUpdates.addElement(A);
121
122                boolean redoAllMyDamnRooms=false;
123                Vector<Room> allMyDamnRooms=null;
124                String oldName=null;
125
126                // class!
127                final String className=httpReq.getUrlParameter("CLASSES");
128                if((className==null)||(className.length()==0))
129                        return "Please select a class type for this area.";
130                if(!className.equalsIgnoreCase(CMClass.classID(A)))
131                {
132                        allMyDamnRooms=new Vector<Room>();
133                        for(final Enumeration<Room> r=A.getProperMap();r.hasMoreElements();)
134                                allMyDamnRooms.addElement(r.nextElement());
135                        final Area oldA=A;
136                        A=CMClass.getAreaType(className);
137                        if(A==null)
138                                return "The class you chose does not exist.  Choose another.";
139                        CMLib.map().delArea(oldA);
140                        CMLib.map().addArea(A);
141                        A.setName(oldA.Name());
142                        redoAllMyDamnRooms=true;
143                        areasNeedingUpdates.remove(oldA);
144                        areasNeedingUpdates.addElement(A);
145                }
146
147                // name
148                String name=httpReq.getUrlParameter("NAME");
149                if((name==null)||(name.length()==0))
150                        return "Please enter a name for this area.";
151                name=name.trim();
152                if(!name.equals(A.Name().trim()))
153                {
154                        if((CMLib.map().getArea(name)!=null)||(CMLib.map().getShip(name)!=null))
155                                return "The name you chose is already in use.  Please enter another.";
156                        allMyDamnRooms=new Vector<Room>();
157                        for(final Enumeration<Room> r=A.getCompleteMap();r.hasMoreElements();)
158                                allMyDamnRooms.addElement(r.nextElement());
159                        CMLib.map().delArea(A);
160                        oldName=A.Name();
161                        CMLib.database().DBDeleteArea(A);
162                        A=CMClass.getAreaType(A.ID());
163                        A.setName(name);
164                        CMLib.map().addArea(A);
165                        CMLib.database().DBCreateArea(A);
166                        redoAllMyDamnRooms=true;
167                        httpReq.addFakeUrlParameter("AREA",A.Name());
168                }
169
170                // climate
171                if(httpReq.isUrlParameter("CLIMATE"))
172                {
173                        int climate=CMath.s_int(httpReq.getUrlParameter("CLIMATE"));
174                        if(climate>=0)
175                        {
176                                for(int i=1;;i++)
177                                {
178                                        if(httpReq.isUrlParameter("CLIMATE"+(Integer.toString(i))))
179                                        {
180                                                final int newVal=CMath.s_int(httpReq.getUrlParameter("CLIMATE"+(Integer.toString(i))));
181                                                if(newVal<0)
182                                                {
183                                                        climate=-1;
184                                                        break;
185                                                }
186                                                climate=climate|newVal;
187                                        }
188                                        else
189                                                break;
190                                }
191                        }
192                        A.setClimateType(climate);
193                }
194                else
195                        A.setClimateType(-1);
196
197                // atmosphere
198                if(httpReq.isUrlParameter("ATMOSPHERE"))
199                        A.setAtmosphere(CMath.s_int(httpReq.getUrlParameter("ATMOSPHERE")));
200
201                // tech level
202                if(httpReq.isUrlParameter("THEME"))
203                        A.setTheme(CMath.s_int(httpReq.getUrlParameter("THEME")));
204
205                // modify subop list
206                for(final Enumeration<String> s=A.subOps();s.hasMoreElements();)
207                        A.delSubOp(s.nextElement());
208                for(int i=1;;i++)
209                {
210                        if(httpReq.isUrlParameter("SUBOP"+(Integer.toString(i))))
211                                A.addSubOp(httpReq.getUrlParameter("SUBOP"+(Integer.toString(i))));
212                        else
213                                break;
214                }
215
216                int num=1;
217                if(httpReq.isUrlParameter("BLURBFLAG1"))
218                {
219                        final Vector<String> prics=new Vector<String>();
220                        String DOUBLE=httpReq.getUrlParameter("BLURBFLAG"+num);
221                        String MASK=httpReq.getUrlParameter("BLURB"+num);
222                        while((DOUBLE!=null)&&(MASK!=null))
223                        {
224                                if(DOUBLE.trim().length()>0)
225                                        prics.addElement((DOUBLE.toUpperCase().trim()+" "+MASK).trim());
226                                num++;
227                                DOUBLE=httpReq.getUrlParameter("BLURBFLAG"+num);
228                                MASK=httpReq.getUrlParameter("BLURB"+num);
229                        }
230                        for(final Enumeration<String> f=A.areaBlurbFlags();f.hasMoreElements();)
231                                A.delBlurbFlag(f.nextElement());
232                        for(int v=0;v<prics.size();v++)
233                                A.addBlurbFlag(prics.elementAt(v));
234                }
235                // description
236                String desc=httpReq.getUrlParameter("DESCRIPTION");
237                if(desc==null)
238                        desc="";
239                A.setDescription(CMLib.coffeeFilter().safetyFilter(desc));
240
241                // image
242                String img=httpReq.getUrlParameter("IMAGE");
243                if(img==null)
244                        img="";
245                A.setImage(CMLib.coffeeFilter().safetyFilter(img));
246                
247                // playerlevel
248                String plvl=httpReq.getUrlParameter("PLAYERLEVEL");
249                if(plvl!=null)
250                        A.setPlayerLevel(CMath.s_int(plvl));
251
252                // gridy
253                final String gridy=httpReq.getUrlParameter("GRIDY");
254                if((gridy!=null)&&(A instanceof GridZones))
255                        ((GridZones)A).setYGridSize(CMath.s_int(gridy));
256                // gridx
257                final String gridx=httpReq.getUrlParameter("GRIDX");
258                if((gridx!=null)&&(A instanceof GridZones))
259                        ((GridZones)A).setXGridSize(CMath.s_int(gridx));
260
261                // author
262                String author=httpReq.getUrlParameter("AUTHOR");
263                if(author==null)
264                        author="";
265                A.setAuthorID(CMLib.coffeeFilter().safetyFilter(author));
266
267                // currency
268                String currency=httpReq.getUrlParameter("CURRENCY");
269                if(currency==null)
270                        currency="";
271                A.setCurrency(CMLib.coffeeFilter().safetyFilter(currency));
272
273                // SHOPPREJ
274                String SHOPPREJ=httpReq.getUrlParameter("SHOPPREJ");
275                if(SHOPPREJ==null)
276                        SHOPPREJ="";
277                A.setPrejudiceFactors(CMLib.coffeeFilter().safetyFilter(SHOPPREJ));
278
279                // BUDGET
280                String BUDGET=httpReq.getUrlParameter("BUDGET");
281                if(BUDGET==null)
282                        BUDGET="";
283                A.setBudget(CMLib.coffeeFilter().safetyFilter(BUDGET));
284
285                // DEVALRATE
286                String DEVALRATE=httpReq.getUrlParameter("DEVALRATE");
287                if(DEVALRATE==null)
288                        DEVALRATE="";
289                A.setDevalueRate(CMLib.coffeeFilter().safetyFilter(DEVALRATE));
290
291                // INVRESETRATE
292                String INVRESETRATE=httpReq.getUrlParameter("INVRESETRATE");
293                if(INVRESETRATE==null)
294                        INVRESETRATE="0";
295                A.setInvResetRate(CMath.s_int(CMLib.coffeeFilter().safetyFilter(INVRESETRATE)));
296
297                // IGNOREMASK
298                String IGNOREMASK=httpReq.getUrlParameter("IGNOREMASK");
299                if(IGNOREMASK==null)
300                        IGNOREMASK="";
301                A.setIgnoreMask(CMLib.coffeeFilter().safetyFilter(IGNOREMASK));
302
303                if(A instanceof AutoGenArea)
304                {
305                        String AGXMLPATH=httpReq.getUrlParameter("AGXMLPATH");
306                        if(AGXMLPATH==null)
307                                AGXMLPATH="";
308                        ((AutoGenArea) A).setGeneratorXmlPath(CMLib.coffeeFilter().safetyFilter(AGXMLPATH));
309
310                        String AGAUTOVAR=httpReq.getUrlParameter("AGAUTOVAR");
311                        if(AGAUTOVAR==null)
312                                AGAUTOVAR="";
313                        ((AutoGenArea) A).setAutoGenVariables(CMLib.coffeeFilter().safetyFilter(AGAUTOVAR));
314                }
315
316                // PRICEFACTORS
317                num=1;
318                if(httpReq.isUrlParameter("IPRIC1"))
319                {
320                        final Vector<String> prics=new Vector<String>();
321                        String DOUBLE=httpReq.getUrlParameter("IPRIC"+num);
322                        String MASK=httpReq.getUrlParameter("IPRICM"+num);
323                        while((DOUBLE!=null)&&(MASK!=null))
324                        {
325
326                                if(CMath.isNumber(DOUBLE))
327                                        prics.addElement((DOUBLE+" "+MASK).trim());
328                                num++;
329                                DOUBLE=httpReq.getUrlParameter("IPRIC"+num);
330                                MASK=httpReq.getUrlParameter("IPRICM"+num);
331                        }
332                        ((Economics)A).setItemPricingAdjustments(CMParms.toStringArray(prics));
333                }
334
335                // modify Parent Area list
336                while(A.getParents().hasMoreElements())
337                        A.removeParent(A.getParents().nextElement());
338                for(int i=1;;i++)
339                {
340                        if(httpReq.isUrlParameter("PARENT"+(Integer.toString(i))))
341                        {
342                                final Area parent=CMLib.map().getArea(httpReq.getUrlParameter("PARENT"+(Integer.toString(i))));
343                                if(parent!=null)
344                                {
345                                        if(A.canParent(parent))
346                                        {
347                                                A.addParent(parent);
348                                                parent.addChild(A);
349                                                areasNeedingUpdates.addElement(parent);
350                                        }
351                                        else
352                                                return "The area, '"+parent.Name()+"', cannot be added as a parent, as this would create a circular reference.";
353                                }
354                        }
355                        else
356                                break;
357                }
358
359                // modify Child Area list
360                while(A.getChildren().hasMoreElements())
361                        A.removeChild(A.getChildren().nextElement());
362                for(int i=1;;i++)
363                {
364                        if(httpReq.isUrlParameter("CHILDREN"+(Integer.toString(i))))
365                        {
366                                final Area child=CMLib.map().getArea(httpReq.getUrlParameter("CHILDREN"+(Integer.toString(i))));
367                                if(child!=null)
368                                {
369                                        if(A.canChild(child))
370                                        {
371                                                A.addChild(child);
372                                                child.addParent(A);
373                                                areasNeedingUpdates.addElement(child);
374                                        }
375                                        else
376                                                return "The area, '"+child.Name()+"', cannot be added as a child, as this would create a circular reference.";
377                                }
378                                else
379                                        break;
380                        }
381                }
382
383                String error=GrinderAreas.doAffects(A,httpReq,parms);
384                if(error.length()>0)
385                        return error;
386                error=GrinderAreas.doBehavs(A,httpReq,parms);
387                if(error.length()>0)
388                        return error;
389
390                if((redoAllMyDamnRooms)&&(allMyDamnRooms!=null))
391                        CMLib.map().renameRooms(A,oldName,allMyDamnRooms);
392
393                for(int i=0;i<areasNeedingUpdates.size();i++) // will always include A
394                {
395                        final Area A2=areasNeedingUpdates.elementAt(i);
396                        if(CMLib.flags().isSavable(A2))
397                        {
398                                CMLib.database().DBUpdateArea(A2.Name(),A2);
399                                CMLib.coffeeMaker().addAutoPropsToAreaIfNecessary(A2);
400                        }
401                }
402                return "";
403        }
