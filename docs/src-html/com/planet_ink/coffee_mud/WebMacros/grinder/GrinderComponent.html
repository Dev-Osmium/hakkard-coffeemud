036public class GrinderComponent
037{
038        public String name()
039        {
040                return "GrinderComponent";
041        }
042
043        public String runMacro(HTTPRequest httpReq, String parm)
044        {
045                final String last=httpReq.getUrlParameter("COMPONENT");
046                if(last==null)
047                        return " @break@";
048                if(last.length()>0)
049                {
050                        final String fixedCompID=last.replace(' ','_').toUpperCase();
051                        final List<AbilityComponent> set=new Vector<AbilityComponent>();
052                        int posDex=1;
053                        while(httpReq.isUrlParameter(fixedCompID+"_PIECE_CONNECTOR_"+posDex) && httpReq.getUrlParameter(fixedCompID+"_PIECE_CONNECTOR_"+posDex).trim().length()>0)
054                        {
055                                final String mask=httpReq.getUrlParameter(fixedCompID+"_PIECE_MASK_"+posDex);
056                                final String str=httpReq.getUrlParameter(fixedCompID+"_PIECE_STRING_"+posDex);
057                                final String amt=httpReq.getUrlParameter(fixedCompID+"_PIECE_AMOUNT_"+posDex);
058                                final String conn=httpReq.getUrlParameter(fixedCompID+"_PIECE_CONNECTOR_"+posDex);
059                                final String loc=httpReq.getUrlParameter(fixedCompID+"_PIECE_LOCATION_"+posDex);
060                                final String type=httpReq.getUrlParameter(fixedCompID+"_PIECE_TYPE_"+posDex);
061                                final String stype=httpReq.getUrlParameter(fixedCompID+"_PIECE_STYPE_"+posDex);
062                                final String consumed=httpReq.getUrlParameter(fixedCompID+"_PIECE_CONSUMED_"+posDex);
063                                if(!conn.equalsIgnoreCase("DELETE"))
064                                {
065                                        final AbilityComponent able=(AbilityComponent)CMClass.getCommon("DefaultAbilityComponent");
066                                        able.setAmount(CMath.s_int(amt));
067                                        if(posDex==1)
068                                                able.setConnector(AbilityComponent.CompConnector.AND);
069                                        else
070                                                able.setConnector(AbilityComponent.CompConnector.valueOf(conn));
071                                        able.setConsumed((consumed!=null)&&(consumed.equalsIgnoreCase("on")||consumed.equalsIgnoreCase("checked")));
072                                        able.setLocation(AbilityComponent.CompLocation.valueOf(loc));
073                                        able.setMask(mask);
074                                        able.setType(AbilityComponent.CompType.valueOf(type), str,stype);
075                                        set.add(able);
076                                }
077                                posDex++;
078                        }
079
080                        if(CMLib.ableComponents().getAbilityComponentMap().containsKey(last.toUpperCase().trim()))
081                        {
082                                final List<AbilityComponent> xset=CMLib.ableComponents().getAbilityComponentMap().get(last.toUpperCase().trim());
083                                xset.clear();
084                                xset.addAll(set);
085                        }
086                        else
087                                CMLib.ableComponents().getAbilityComponentMap().put(last.toUpperCase().trim(),set);
088                        CMLib.ableComponents().alterAbilityComponentFile(last.toUpperCase().trim(),false);
089                }
090                return "";
091        }
092}
