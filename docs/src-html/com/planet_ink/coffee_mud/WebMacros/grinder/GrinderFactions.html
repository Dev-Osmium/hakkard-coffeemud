040{
041        public String name()
042        {
043                return "GrinderFactions";

046        public static String modifyFaction(HTTPRequest httpReq, java.util.Map<String,String> parms, Faction F)
047        {
048                final String replaceCommand=httpReq.getUrlParameter("REPLACE");
049                if((replaceCommand != null)
050                && (replaceCommand.length()>0)
051                && (replaceCommand.indexOf('=')>0))
052                {
053                        final int eq=replaceCommand.indexOf('=');
054                        final String field=replaceCommand.substring(0,eq);
055                        final String value=replaceCommand.substring(eq+1);
056                        httpReq.addFakeUrlParameter(field, value);
057                        httpReq.addFakeUrlParameter("REPLACE","");
058                }
059                String old;
060
061                old=httpReq.getUrlParameter("NAME");
062                F.setName(old==null?("NAME"):old);
063
064                old=httpReq.getUrlParameter("SHOWINSCORE");
065                F.setShowInScore((old!=null)&&(old.equalsIgnoreCase("on")));
066
067                old=httpReq.getUrlParameter("SHOWINFACTIONS");
068                F.setShowInFactionsCommand((old!=null)&&(old.equalsIgnoreCase("on")));
069
070                old=httpReq.getUrlParameter("SHOWINEDITOR");
071                F.setShowInEditor((old!=null)&&(old.equalsIgnoreCase("on")));
072
073                old=httpReq.getUrlParameter("SHOWINREPORTS");
074                F.setShowInSpecialReported((old!=null)&&(old.equalsIgnoreCase("on")));
075
076                int num=0;
077                for(final Enumeration<Faction.FRange> e=F.ranges();e.hasMoreElements();)
078                        F.delRange(e.nextElement());
079                while(httpReq.getUrlParameter("RANGENAME"+num)!=null)
080                {
081                        old=httpReq.getUrlParameter("RANGENAME"+num);
082                        String code=httpReq.getUrlParameter("RANGECODE"+num);
083                        if(old.length()>0)
084                        {
085                                if(code.length()==0)
086                                        code=CMStrings.replaceAll(old.toUpperCase().trim()," ","_");
087                                final int low=CMath.s_int(httpReq.getUrlParameter("RANGELOW"+num));
088                                int high=CMath.s_int(httpReq.getUrlParameter("RANGEHIGH"+num));
089                                if(high<low)
090                                        high=low;
091                                final String flag=httpReq.getUrlParameter("RANGEFLAG"+num);
092                                F.addRange(low+";"+high+";"+old+";"+code+";"+flag);
093                        }
094                        num++;
095                }
096
097                old=httpReq.getUrlParameter("PLAYERCHOICETEXT");
098                F.setChoiceIntro(old==null?"":old);
099
100                final String[] prefixes={"AUTOVALUE","DEFAULTVALUE","PLAYERCHOICE"};
101                for(int i=0;i<prefixes.length;i++)
102                {
103                        final String prefix=prefixes[i];
104                        final Vector<String> V=new Vector<String>();
105                        num=0;
106                        while(httpReq.getUrlParameter(prefix+num)!=null)
107                        {
108                                final String value=httpReq.getUrlParameter(prefix+num);
109                                if(value.length()>0)
110                                {
111                                        final String mask=httpReq.getUrlParameter(prefix+"MASK"+num);
112                                        V.addElement((CMath.s_long(value)+" "+mask).trim());
113                                }
114                                num++;
115                        }
116                        switch(i)
117                        {
118                        case 0:
119                                F.setAutoDefaults(V);
120                                break;
121                        case 1:
122                                F.setDefaults(V);
123                                break;
124                        case 2:
125                                F.setChoices(V);
126                                break;
127                        }
128                }
129
130                F.clearChangeEvents();
131                num=0;
132                while(httpReq.getUrlParameter("CHANGESTRIGGER"+num)!=null)
133                {
134                        old=httpReq.getUrlParameter("CHANGESTRIGGER"+num);
135                        if(old.length()>0)
136                        {
137                                final String ctparms=httpReq.getUrlParameter("CHANGESTPARM"+num);
138                                if(ctparms.trim().length()>0)
139                                        old+="("+ctparms.trim()+")";
140                                old+=";";
141                                old+=Faction.FactionChangeEvent.CHANGE_DIRECTION_DESCS[CMath.s_int(httpReq.getUrlParameter("CHANGESDIR"+num))];
142                                old+=";";
143                                old+=CMath.toPct(httpReq.getUrlParameter("CHANGESFACTOR"+num));
144                                old+=";";
145                                String id="";
146                                int x=0;
147                                for(;httpReq.isUrlParameter("CHANGESFLAGS"+num+"_"+id);id=""+(++x))
148                                        old+=" "+httpReq.getUrlParameter("CHANGESFLAGS"+num+"_"+id).toUpperCase();
149                                int v=CMath.s_int(httpReq.getUrlParameter("XP"+num));
150                                if(v!=0)
151                                        old+=" XP="+v;
152                                v=CMath.s_int(httpReq.getUrlParameter("RPXP"+num));
153                                if(v!=0)
154                                        old+=" RPXP="+v;
155                                old+=";";
156                                old+=httpReq.getUrlParameter("CHANGESMASK"+num);
157                                F.createChangeEvent(old);
158                        }
159                        num++;
160                }
161
162                for(final Enumeration<Faction.FZapFactor> e=F.factors();e.hasMoreElements();)
163                        F.delFactor(e.nextElement());
164                num=0;
165                while(httpReq.getUrlParameter("ADJFACTOR"+num)!=null)
166                {
167                        old=httpReq.getUrlParameter("ADJFACTOR"+num);
168                        if(old.length()>0)
169                        {
170                                final double gain=CMath.s_pct(httpReq.getUrlParameter("ADJFACTORGAIN"+num));
171                                final double loss=CMath.s_pct(httpReq.getUrlParameter("ADJFACTORLOSS"+num));
172                                F.addFactor(gain,loss,old);
173                        }
174                        num++;
175                }
176
177                num=0;
178                for(final Enumeration<String> e=F.relationFactions();e.hasMoreElements();)
179                        F.delRelation(e.nextElement());
180                while(httpReq.getUrlParameter("RELATIONS"+num)!=null)
181                {
182                        old=httpReq.getUrlParameter("RELATIONS"+num);
183                        if(old.length()>0)
184                                F.addRelation(old,CMath.s_pct(httpReq.getUrlParameter("RELATIONSAMT"+num)));
185                        num++;
186                }
187
188                num=0;
189                final TriadVector<String,String,String> affBehav=new TriadVector<String,String,String>();
190                final HashSet<String> affBehavKeepers=new HashSet<String>();
191                // its done this strange way to minimize impact on mob recalculations.
192                while(httpReq.getUrlParameter("AFFBEHAV"+num)!=null)
193                {
194                        old=httpReq.getUrlParameter("AFFBEHAV"+num);
195                        if(old.length()>0)
196                        {
197                                final String parm=""+httpReq.getUrlParameter("AFFBEHAVPARM"+num);
198                                final String mask=""+httpReq.getUrlParameter("AFFBEHAVMASK"+num);
199                                final String[] oldParms=F.getAffectBehav(old);
200                                if((oldParms==null)||(!oldParms[0].equals(parm))||(!oldParms[1].equals(mask)))
201                                        affBehav.addElement(old.toUpperCase().trim(),parm,mask);
202                                else
203                                        affBehavKeepers.add(old.toUpperCase().trim());
204                        }
205                        num++;
206                }
207                for(final Enumeration<String> e=F.affectsBehavs();e.hasMoreElements();)
208                {
209                        old=e.nextElement();
210                        if(!affBehavKeepers.contains(old.toUpperCase().trim()))
211                                F.delAffectBehav(old);
212                }
213                for(int d=0;d<affBehav.size();d++)
214                {
215                        F.delAffectBehav(affBehav.get(d).first);
216                        F.addAffectBehav(affBehav.get(d).first,affBehav.get(d).second,affBehav.get(d).third);
217                }
218
219                num=0;
220                for(final Enumeration<Faction.FAbilityUsage> e=F.abilityUsages();e.hasMoreElements();)
221                        F.delAbilityUsage(e.nextElement());
222                while(httpReq.getUrlParameter("ABILITYUSE"+num)!=null)
223                {
224                        old=httpReq.getUrlParameter("ABILITYUSE"+num);
225                        if(old.length()>0)
226                        {
227                                final int usedType=CMLib.factions().getAbilityFlagType(old);
228                                if(usedType>0)
229                                {
230                                        int x=-1;
231                                        while(httpReq.isUrlParameter("ABILITYUSE"+num+"_"+(++x)))
232                                        {
233                                                final String s=httpReq.getUrlParameter("ABILITYUSE"+num+"_"+x);
234                                                if(s.length()>0)
235                                                {
236                                                        old+=" "+s.toUpperCase().trim();
237                                                }
238                                        }
239                                }
240                                old+=";"+CMath.s_int(httpReq.getUrlParameter("ABILITYMIN"+num));
241                                old+=";"+CMath.s_int(httpReq.getUrlParameter("ABILITYMAX"+num));
242                                F.addAbilityUsage(old);
243                        }
244                        num++;
245                }
246
247                num=0;
248                for(final Enumeration<Faction.FReactionItem> e=F.reactions();e.hasMoreElements();)
249                        F.delReaction(e.nextElement());
250                while(httpReq.getUrlParameter("REACTIONRANGE"+num)!=null)
251                {
252                        old=httpReq.getUrlParameter("REACTIONRANGE"+num);
253                        final String old1=httpReq.getUrlParameter("REACTIONMASK"+num);
254                        final String old2=httpReq.getUrlParameter("REACTIONABC"+num);
255                        final String old3=httpReq.getUrlParameter("REACTIONPARM"+num);
256                        if(old.length()>0)
257                                F.addReaction(old,old1,old2,old3);
258                        num++;
259                }
260                old=httpReq.getUrlParameter("USELIGHTREACTIONS");
261                F.setLightReactions((old!=null)&&(old.equalsIgnoreCase("on")));
262
263                old=httpReq.getUrlParameter("RATEMODIFIER");
264                F.setRateModifier(old==null?0.0:CMath.s_pct(old));
265
266                old=httpReq.getUrlParameter("AFFECTONEXP");
267                F.setExperienceFlag(old);
268
269                return "";
270        }
