042        protected int minZ = 0;
043        protected int maxZ = 0;
044        protected int zFix = 0;

046        public GrinderMap()
047        {
048                areaMap = new Vector<GrinderRoom>();
049                hashRooms = new Hashtable<String,GrinderRoom>();
050                for (final Enumeration<Area> q = CMLib.map().areas(); q.hasMoreElements(); )
051                {
052                        final Area A = q.nextElement();
053                        // for now, skip hidden areas.  Areas are often hidden if they aren't linked
054                        // to the world (ie under construction or Archon only)
055                        if ((CMLib.flags().isHidden(A))
056                        ||(CMath.bset(A.flags(),Area.FLAG_INSTANCE_CHILD)))
057                                        continue;
058                        for (final Enumeration<String> r = A.getProperRoomnumbers().getRoomIDs(); r.hasMoreElements(); )
059                        {
060                                final String roomID=r.nextElement();
061                                if((roomID.length() > 0)&&(roomID.indexOf("#(")<0))
062                                {
063                                        final GrinderRoom GR = new GrinderRoom(roomID);
064                                        areaMap.add(GR);
065                                        hashRooms.put(GR.roomID, GR);
066                                }
067                        }
068                }
069        }

078        {
079                if(areaMap==null)

083                hashRooms=null;
084                rebuildGrid();

087        @Override
088        public void rebuildGrid()
089        {
090                if (areaMap == null) return;
091                // build grid!
092                int xoffset = 0;
093                int yoffset = 0;
094
095                for (int x = 0; x < areaMap.size(); x++)
096                {
097                        final GrinderRoom GR =  areaMap.get(x);
098                        if (GR.xy[0] < xoffset)
099                        {
100                                xoffset = GR.xy[0];
101                                if (debug) Log.sysOut("GR-REGRID", "xoffset set0: " + xoffset);
102                                if((debug) && (GR.xy[0]>0) )
103                                        Log.sysOut("GR-REGRID", "positive GRx: " + GR.xy[0]);
104                        }
105                        if (GR.xy[1] < yoffset)
106                                yoffset = GR.xy[1];
107                }
108
109                xoffset = xoffset * -1;
110                yoffset = yoffset * -1;
111
112                if (debug) 
113                        Log.sysOut("GR-REGRID", "xoffset set : " + xoffset);
114                if (debug) 
115                        Log.sysOut("GR-REGRID", "Xbound  set : " + Xbound);
116
117                Xbound = 0;
118                Ybound = 0;
119                for (int x = 0; x < areaMap.size(); x++)
120                {
121                        final GrinderRoom room = areaMap.get(x);
122                        room.xy[0] = room.xy[0] + xoffset;
123                        if (room.xy[0] > Xbound) Xbound = room.xy[0];
124                        room.xy[1] = room.xy[1] + yoffset;
125                        if (room.xy[1] > Ybound) Ybound = room.xy[1];
126                }
127                if (debug) 
128                        Log.sysOut("GR-REGRID", "Xbound  set2: " + Xbound);
129                grid = new GrinderRoom[Xbound + 1][Ybound + 1][maxZ + 1];
130                if (debug) 
131                        Log.sysOut("GR-REGRID", "GrinderRoom Grid Created: (x,y,z) " +
132                                          (Xbound + 1) + "," + (Ybound + 1) + "," + (maxZ + 1));
133                for (int y = 0; y < areaMap.size(); y++)
134                {
135                        final GrinderRoom room = areaMap.get(y);
136                        // this was hardcoded to look for below zero, but my zFix math
137                        // often ended up with minZ >= 1
138                        if(room.z<minZ)
139                        {
140                                grid[room.xy[0]][room.xy[1]][(room.z + zFix)] = room;
141                                if (debug) 
142                                        Log.sysOut("GR-REGRID", "Room outside z range: " + room.z);
143                        }
144                        else
145                        {
146                                if ((debug) && ((room.z > maxZ) || (room.z < minZ)))
147                                {
148                                        Log.sysOut("GR-HTML", "Room.z error: " + room.z + " outside " + maxZ + "-" + minZ + "(" +
149                                                                room.roomID + ")");
150                                }
151                                grid[room.xy[0]][room.xy[1]][room.z] = room;
152                        }
153                }
154        }

156        protected GrinderRoom getProcessedRoomAt(Hashtable<String,GrinderRoom> processed, int x, int y, int z)
157        {
158                for (final Enumeration<GrinderRoom> e = processed.elements(); e.hasMoreElements(); )
159                {
160                        final GrinderRoom room = e.nextElement();
161                        if ( (room.xy[0] == x) && (room.xy[1] == y) && (room.z == z))
162                                return room;
163                }
164                return null;
165        }

167        @Override
168        public GrinderRoom getRoom(String ID)
169        {
170                if ( (hashRooms != null) && (hashRooms.containsKey(ID)))
171                        return hashRooms.get(ID);
172
173                if (areaMap != null)
174                {
175                        for (int r = 0; r < areaMap.size(); r++)
176                        {
177                                final GrinderRoom room = areaMap.get(r);
178                                if (room.roomID.equalsIgnoreCase(ID))
179                                        return room;
180                        }
181                }
182                return null;
183        }

185        protected boolean isEmptyCluster(Hashtable<String,GrinderRoom> processed, int x, int y, int z)
186        {
187                for (final Enumeration<GrinderRoom> e = processed.elements(); e.hasMoreElements(); )
188                {
189                        final GrinderRoom room = e.nextElement();
190                        if ( ( ( (room.xy[0] > x - CLUSTERSIZE) && (room.xy[0] < x + CLUSTERSIZE))
191                                  && ( (room.xy[1] > y - CLUSTERSIZE) && (room.xy[1] < y + CLUSTERSIZE)))
192                                || ( (room.xy[0] == x) && (room.xy[1] == y)) && (room.z == z))
193                                return false;
194                }
195                return true;
196        }

198        protected void findEmptyCluster(Hashtable<String,GrinderRoom> processed, Vector<Integer> XYZ)
199        {
200                final int x = XYZ.elementAt(0).intValue();
201                final int y = XYZ.elementAt(1).intValue();
202                final int z = XYZ.elementAt(2).intValue();
203                int spacing = CLUSTERSIZE;
204                while (true)
205                {
206                        for (int i = 0; i < 8; i++)
207                        {
208                                int yadjust = 0;
209                                int xadjust = 0;
210                                switch (i)
211                                {
212                                case 0:
213                                        xadjust = 1;
214                                        break;
215                                case 1:
216                                        xadjust = 1;
217                                        yadjust = 1;
218                                        break;
219                                case 2:
220                                        yadjust = 1;
221                                        break;
222                                case 3:
223                                        xadjust = 1;
224                                        xadjust = -1;
225                                        break;
226                                case 4:
227                                        xadjust = -1;
228                                        break;
229                                case 5:
230                                        xadjust = -1;
231                                        yadjust = -1;
232                                        break;
233                                case 6:
234                                        yadjust = -1;
235                                        break;
236                                case 7:
237                                        yadjust = -1;
238                                        xadjust = 1;
239                                        break;
240                                }
241                                // I'm letting EmptyCluster always search the current Z level
242                                if (isEmptyCluster(processed, x + (spacing * xadjust), y + (spacing * yadjust), z))
243                                {
244                                        XYZ.setElementAt(Integer.valueOf(x + (spacing * xadjust)), 0);
245                                        XYZ.setElementAt(Integer.valueOf(y + (spacing * yadjust)), 1);
246                                        XYZ.setElementAt(Integer.valueOf(z), 2);
247                                        return;
248                                }
249                        }
250                        spacing += 1;
251                }
252        }

254        @Override
255        public void placeRooms()
256        {
257                if (areaMap == null)
258                        return;
259                if (areaMap.size() == 0)
260                        return;
261
262                for (int i = 0; i < areaMap.size(); i++)
263                {
264                        final GrinderRoom room = areaMap.get(i);
265                        room.xy=new int[2];
266                        for (int d = 0; d < Directions.NUM_DIRECTIONS(); d++)
267                        {
268                                final GrinderDir dir = room.doors[d];
269                                if (dir != null)
270                                        dir.positionedAlready = false;
271                        }
272                }
273
274                final Hashtable<String,GrinderRoom> processed = new Hashtable<String,GrinderRoom>();
275                boolean doneSomething = true;
276
277                while ( (areaMap.size() > processed.size()) && (doneSomething))
278                {
279                        doneSomething = false;
280                        for (int i = 0; i < areaMap.size(); i++)
281                        {
282                                final GrinderRoom room = areaMap.get(i);
283                                if (!processed.containsKey(room.roomID))
284                                {
285                                        placeRoom(room, 0, 0, processed, true, true, 0, 0);
286                                        doneSomething = true;
287                                }
288                        }
289                }
290
291                // For sanity, we rehash all the Z levels into positive numbers
292                // some overhead, but worthwhile
293                for (int x = 0; x < areaMap.size(); x++)
294                {
295                        final GrinderRoom GR = areaMap.get(x);
296                        if(GR.xy==null)
297                                Log.errOut("GrinderMap",GR.roomID+" not assigned an XY!");
298                        if (GR.z < minZ)
299                        {
300                                if (debug) Log.sysOut("GR-PLACERS", "minZ changed: " + minZ + " to " + GR.z);
301                                minZ = GR.z;
302                        }
303                        if (GR.z > maxZ)
304                        {
305                                if (debug) Log.sysOut("GR-PLACERS", "maxZ changed: " + maxZ + " to " + GR.z);
306                                maxZ = GR.z;
307                        }
308                }
309
310                zFix = maxZ - minZ;
311                if ((zFix + minZ) > 0)
312                        zFix -= (0 - (minZ + zFix)) * -1;
313
314                if (debug) 
315                        Log.sysOut("GR-PLACERS", "zFix set    : " + zFix);
316                if (debug) 
317                        Log.sysOut("GR-PLACERS", "areaMap size: " + areaMap.size());
318                int updatedCount = 0;
319                for (int x = 0; x < areaMap.size(); x++)
320                {
321                        final GrinderRoom GR = areaMap.get(x);
322                        final int oldZ = GR.z;
323                        GR.z += zFix;
324                        areaMap.set(x,GR);
325                        if (GR.z!=oldZ) updatedCount++;
326                }
327                if (debug) 
328                        Log.sysOut("GR-PLACERS", "maybe update: " + updatedCount);
329                if (debug) 
330                        Log.sysOut("GR-PLACERS", "maxZ changed: " + maxZ + " to " + (maxZ + zFix));
331                maxZ += zFix;
332                if (debug) 
333                        Log.sysOut("GR-PLACERS", "minZ changed: " + minZ + " to " + (minZ + zFix));
334                minZ += zFix;
335
336                if (areaMap.size() > processed.size())
337                {
338                        Log.errOut("GrinderMap",
339                                           areaMap.size() - processed.size() +
340                                           " room(s) were not placed.");
341                }
342        }

556        @Override
557        protected GrinderRoom getRoomInDir(GrinderRoom room, int d)
558        {
559                switch (d)
560                {
561                case Directions.NORTH:
562                        if (room.xy[1] > 0)
563                                return grid[room.xy[0]][room.xy[1] - 1][room.z];
564                        break;
565                case Directions.SOUTH:
566                        if (room.xy[1] < Ybound)
567                                return grid[room.xy[0]][room.xy[1] + 1][room.z];
568                        break;
569                case Directions.NORTHWEST:
570                        if((room.xy[1]>0)&&(room.xy[0]>0))
571                                return grid[room.xy[0]-1][room.xy[1]-1][room.z];
572                        break;
573                case Directions.SOUTHWEST:
574                        if((room.xy[1]<Ybound)&&(room.xy[0]>0))
575                                return grid[room.xy[0]-1][room.xy[1]+1][room.z];
576                        break;
577                case Directions.NORTHEAST:
578                        if((room.xy[1]>0)&&(room.xy[0]<Xbound))
579                                return grid[room.xy[0]+1][room.xy[1]-1][room.z];
580                        break;
581                case Directions.SOUTHEAST:
582                        if((room.xy[1]<Ybound)&&(room.xy[0]<Xbound))
583                                return grid[room.xy[0]+1][room.xy[1]+1][room.z];
584                        break;
585                case Directions.EAST:
586                        if (room.xy[0] < Xbound)
587                                return grid[room.xy[0] + 1][room.xy[1]][room.z];
588                        break;
589                case Directions.WEST:
590                        if (room.xy[0] > 0)
591                                return grid[room.xy[0] - 1][room.xy[1]][room.z];
592                        break;
593                case Directions.UP:
594                        if (room.z > maxZ)
595                                return grid[room.xy[0]][room.xy[1]][room.z - 1];
596                        break;
597                case Directions.DOWN:
598                        if (room.z < minZ)
599                                return grid[room.xy[0]][room.xy[1]][room.z + 1];
600                        break;
601                }
602                return null;
603        }

605        public void placeRoom(GrinderRoom room,
606                                                  int favoredX,
607                                                  int favoredY,
608                                                  Hashtable<String,GrinderRoom> processed,
609                                                  boolean doNotDefer,
610                                                  boolean passTwo,
611                                                  int depth,
612                                                  int zLevel)
613        {
614                if (room == null)
615                        return;
616                if (depth > 500)
617                        return;
618
619                final GrinderRoom anythingAt = getProcessedRoomAt(processed, favoredX, favoredY, zLevel);
620                if (anythingAt != null)
621                {
622                        // maybe someone else will take care of it?
623                        if (!doNotDefer)
624                        {
625                                for (int r = 0; r < areaMap.size(); r++)
626                                {
627                                        final GrinderRoom roomToBlame = areaMap.get(r);
628                                        if (roomToBlame != room)
629                                        {
630                                                for (int rd = 0; rd < Directions.NUM_DIRECTIONS(); rd++)
631                                                {
632                                                        final GrinderDir RD = roomToBlame.doors[rd];
633                                                        if ( (RD != null)
634                                                        && (RD.room != null)
635                                                        && (!RD.positionedAlready)
636                                                        && (RD.room.equals(room.roomID)))
637                                                                return;
638                                                }
639                                        }
640                                }
641                        }
642                        // nope; nobody can.  It's up to this!
643                        final Vector<Integer> XYZ = new Vector<Integer>();
644                        XYZ.addElement(Integer.valueOf(0));
645                        XYZ.addElement(Integer.valueOf(0));
646                        XYZ.addElement(Integer.valueOf(0));
647                        findEmptyCluster(processed, XYZ);
648                        room.xy[0] = XYZ.elementAt(0).intValue();
649                        room.xy[1] = XYZ.elementAt(1).intValue();
650                        room.z = XYZ.elementAt(2).intValue();
651                }
652                else
653                {
654                        room.xy[0] = favoredX;
655                        room.xy[1] = favoredY;
656                        room.z = zLevel;
657                }
658
659                // once done, is never undone.  A room is
660                // considered processed only once!
661                processed.put(room.roomID, room);
662
663                for (int d = 0; d < Directions.NUM_DIRECTIONS(); d++)
664                {
665                        String roomID = null;
666                        if (room.doors[d] != null)
667                                roomID = room.doors[d].room;
668
669                        if ( (roomID != null)
670                        && (roomID.length() > 0)
671                        && (processed.get(roomID) == null)
672                        && (passTwo || ( (d != Directions.UP) && (d != Directions.DOWN))))
673                        {
674                                final GrinderRoom nextRoom = getRoom(roomID);
675                                if (nextRoom != null)
676                                {
677                                        int newFavoredX = room.xy[0];
678                                        int newFavoredY = room.xy[1];
679                                        int newZLevel = room.z;
680                                        switch (d)
681                                        {
682                                        case Directions.NORTH:
683                                                newFavoredY--;
684                                                break;
685                                        case Directions.SOUTH:
686                                                newFavoredY++;
687                                                break;
688                                        case Directions.EAST:
689                                                newFavoredX++;
690                                                break;
691                                        case Directions.WEST:
692                                                newFavoredX--;
693                                                break;
694                                        case Directions.NORTHEAST:
695                                                newFavoredY--;
696                                                newFavoredX++;
697                                                break;
698                                        case Directions.NORTHWEST:
699                                                newFavoredY--;
700                                                newFavoredX--;
701                                                break;
702                                        case Directions.SOUTHEAST:
703                                                newFavoredY++;
704                                                newFavoredX++;
705                                                break;
706                                        case Directions.SOUTHWEST:
707                                                newFavoredY++;
708                                                newFavoredX--;
709                                                break;
710                                        case Directions.UP:
711                                                newZLevel++;
712                                                break;
713                                        case Directions.DOWN:
714                                                newZLevel--;
715                                                break;
716                                        }
717                                        room.doors[d].positionedAlready = true;
718                                        placeRoom(nextRoom, newFavoredX, newFavoredY, processed, false,
719                                                          passTwo, depth + 1, newZLevel);
720                                }
721                        }
722                }
723        }
