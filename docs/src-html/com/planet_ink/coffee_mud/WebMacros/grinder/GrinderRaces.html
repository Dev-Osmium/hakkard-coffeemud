044        public static String getPStats(char c, HTTPRequest httpReq)
045        {
046                boolean changes = false;
047                final PhyStats adjPStats=(PhyStats)CMClass.getCommon("DefaultPhyStats");
048                adjPStats.setAllValues(0);
049                if(httpReq.isUrlParameter(c+"ESTATS1"))
050                {
051                        int num=1;
052                        String behav=httpReq.getUrlParameter(c+"ESTATS"+num);
053                        while(behav!=null)
054                        {
055                                if((behav.length()>0) && (new XVector<String>(adjPStats.getStatCodes()).contains(behav.toUpperCase().trim())))
056                                {
057                                        String prof=httpReq.getUrlParameter(c+"ESTATSV"+num);
058                                        if(prof==null)
059                                                prof="0";
060                                        if(CMath.s_int(prof)!=0)
061                                        {
062                                                adjPStats.setStat(behav.toUpperCase().trim(), prof);
063                                                changes = true;
064                                        }
065                                }
066                                num++;
067                                behav=httpReq.getUrlParameter(c+"ESTATS"+num);
068                        }
069                }
070                if(!changes)
071                        return "";
072                return CMLib.coffeeMaker().getPhyStatsStr(adjPStats);
073        }

135        public static List<Item> itemList(List<? extends Item> items, char c, HTTPRequest httpReq, boolean one)
136        {
137                if(items==null)
138                        items=new Vector<Item>();
139                final Vector<Item> classes=new Vector<Item>();
140                List<Item> itemlist=null;
141                if(httpReq.isUrlParameter(c+"ITEM1"))
142                {
143                        itemlist=RoomData.getItemCache();
144                        for(int i=1;;i++)
145                        {
146                                final String MATCHING=httpReq.getUrlParameter(c+"ITEM"+i);
147                                if(MATCHING==null)
148                                        break;
149                                Item I2=RoomData.getItemFromAnywhere(itemlist,MATCHING);
150                                if(I2==null)
151                                {
152                                        I2=RoomData.getItemFromAnywhere(items,MATCHING);
153                                        if(I2!=null)
154                                                RoomData.contributeItems(new XVector<Item>(I2));
155                                }
156                                if(I2!=null)
157                                        classes.addElement(I2);
158                                if(one)
159                                        break;
160                        }
161                }
162                return classes;
163        }

165        public static void setDynAbilities(Modifiable M, HTTPRequest httpReq)
166        {
167                final boolean supportsRoles=CMParms.contains(M.getStatCodes(), "GETRABLEROLE");
168                final DVector theclasses=new DVector(6);
169                if(httpReq.isUrlParameter("RABLES1"))
170                {
171                        int num=1;
172                        String behav=httpReq.getUrlParameter("RABLES"+num);
173                        while(behav!=null)
174                        {
175                                if(behav.length()>0)
176                                {
177                                        String prof=httpReq.getUrlParameter("RABPOF"+num);
178                                        if(prof==null)
179                                                prof="0";
180                                        String qual=httpReq.getUrlParameter("RABQUA"+num);
181                                        if(qual==null)
182                                                qual="";
183                                        String levl=httpReq.getUrlParameter("RABLVL"+num);
184                                        if(levl==null)
185                                                levl="0";
186                                        String parm=httpReq.getUrlParameter("RABPRM"+num);
187                                        if(parm==null)
188                                                parm="";
189                                        String roles=null;
190                                        if(supportsRoles)
191                                                roles=httpReq.getUrlParameter("RABROL"+num);
192                                        if(roles==null) 
193                                                roles="";
194                                        theclasses.addElement(behav,prof,qual,levl,parm,roles);
195                                }
196                                num++;
197                                behav=httpReq.getUrlParameter("RABLES"+num);
198                        }
199                }
200                M.setStat("NUMRABLE", ""+theclasses.size());
201                for(int i=0;i<theclasses.size();i++)
202                {
203                        M.setStat("GETRABLE"+i, (String)theclasses.get(i,1));
204                        M.setStat("GETRABLEPROF"+i, (String)theclasses.get(i,2));
205                        M.setStat("GETRABLEQUAL"+i, ((String)theclasses.get(i,3)).equalsIgnoreCase("on")?"true":"false");
206                        M.setStat("GETRABLELVL"+i, (String)theclasses.get(i,4));
207                        M.setStat("GETRABLEPARM"+i, (String)theclasses.get(i,5));
208                        if(supportsRoles)
209                                M.setStat("GETRABLEROLE"+i, (String)theclasses.get(i,6));
210                }
211        }

275        public static QuadVector<String,Integer,Integer,Boolean> cabilities(HTTPRequest httpReq)
276        {
277                final QuadVector<String,Integer,Integer,Boolean> theclasses=new QuadVector<String,Integer,Integer,Boolean>();
278                if(httpReq.isUrlParameter("CABLES1"))
279                {
280                        int num=1;
281                        String behav=httpReq.getUrlParameter("CABLES"+num);
282                        while(behav!=null)
283                        {
284                                if(behav.length()>0)
285                                {
286                                        String prof=httpReq.getUrlParameter("CABPOF"+num);
287                                        if(prof==null)
288                                                prof="0";
289                                        String qual=httpReq.getUrlParameter("CABQUA"+num);
290                                        if(qual==null) 
291                                                qual="";// null means unchecked
292                                        String levl=httpReq.getUrlParameter("CABLVL"+num);
293                                        if(levl==null)
294                                                levl="0";
295                                        theclasses.addElement(behav,Integer.valueOf(CMath.s_int(prof)),Integer.valueOf(CMath.s_int(levl)),qual.equals("on")?Boolean.TRUE:Boolean.FALSE);
296                                }
297                                num++;
298                                behav=httpReq.getUrlParameter("CABLES"+num);
299                        }
300                }
301                return theclasses;
302        }

304        public static String modifyRace(HTTPRequest httpReq, java.util.Map<String,String> parms, Race oldR, Race R)
305        {
306                final String replaceCommand=httpReq.getUrlParameter("REPLACE");
307                if((replaceCommand != null)
308                && (replaceCommand.length()>0)
309                && (replaceCommand.indexOf('=')>0))
310                {
311                        final int eq=replaceCommand.indexOf('=');
312                        final String field=replaceCommand.substring(0,eq);
313                        final String value=replaceCommand.substring(eq+1);
314                        httpReq.addFakeUrlParameter(field, value);
315                        httpReq.addFakeUrlParameter("REPLACE","");
316                }
317                String old;
318
319                old=httpReq.getUrlParameter("NAME");
320                R.setStat("NAME",(old==null)?"NAME":old);
321                old=httpReq.getUrlParameter("CAT");
322                R.setStat("CAT",(old==null)?"CAT":old);
323                old=httpReq.getUrlParameter("XPADJ");
324                R.setStat("XPADJ",""+(int)Math.round(100.0*CMath.s_pct((old==null)?"0":old)));
325                old=httpReq.getUrlParameter("VWEIGHT");
326                R.setStat("VWEIGHT",(old==null)?"VWEIGHT":old);
327                old=httpReq.getUrlParameter("BWEIGHT");
328                R.setStat("BWEIGHT",(old==null)?"BWEIGHT":old);
329                old=httpReq.getUrlParameter("VHEIGHT");
330                R.setStat("VHEIGHT",(old==null)?"VHEIGHT":old);
331                old=httpReq.getUrlParameter("MHEIGHT");
332                R.setStat("MHEIGHT",(old==null)?"MHEIGHT":old);
333                old=httpReq.getUrlParameter("FHEIGHT");
334                R.setStat("FHEIGHT",(old==null)?"FHEIGHT":old);
335                old=httpReq.getUrlParameter("LEAVESTR");
336                R.setStat("LEAVE",(old==null)?"LEAVESTR":old);
337                old=httpReq.getUrlParameter("ARRIVESTR");
338                R.setStat("ARRIVE",(old==null)?"ARRIVESTR":old);
339                old=httpReq.getUrlParameter("HEALTHRACE");
340                R.setStat("HEALTHRACE",(old==null)?"HEALTHRACE":old);
341                old=httpReq.getUrlParameter("WEAPONRACE");
342                R.setStat("WEAPONRACE",(old==null)?"WEAPONRACE":old);
343                old=httpReq.getUrlParameter("EVENTRACE");
344                R.setStat("EVENTRACE",(old==null)?"EVENTRACE":old);
345                old=httpReq.getUrlParameter("CANRIDE");
346                R.setStat("CANRIDE",(old==null)?"false":(""+old.equalsIgnoreCase("on")));
347                old=httpReq.getUrlParameter("GENHELP");
348                R.setStat("HELP", ((old==null)?"":old));
349                final StringBuffer bodyOld=new StringBuffer("");
350                for(int i=0;i<Race.BODYPARTSTR.length;i++)
351                {
352                        old=httpReq.getUrlParameter("BODYPART"+i);
353                        bodyOld.append((old==null)?"":old).append(";");
354                }
355                R.setStat("BODY",bodyOld.toString());
356                old=httpReq.getUrlParameter("WEARID");
357                long mask=0;
358                if(old!=null)
359                {
360                        mask|=CMath.s_long(old);
361                        for(int i=1;;i++)
362                        {
363                                if(httpReq.isUrlParameter("WEARID"+(Integer.toString(i))))
364                                        mask|=CMath.s_long(httpReq.getUrlParameter("WEARID"+(Integer.toString(i))));
365                                else
366                                        break;
367                        }
368                }
369                R.setStat("WEAR",""+mask);
370                R.setStat("AVAIL",""+CMath.s_long(httpReq.getUrlParameter("PLAYABLEID")));
371                String bodyKill=httpReq.getUrlParameter("BODYKILL");
372                if(bodyKill != null)
373                        R.setStat("BODYKILL",""+bodyKill.equalsIgnoreCase("on"));
374                R.setStat("DISFLAGS",""+CMath.s_long(httpReq.getUrlParameter("DISFLAGS")));
375                R.setStat("ESTATS",getPStats('E',httpReq));
376                R.setStat("CSTATS",getCStats('S',httpReq));
377                R.setStat("ASTATS",getCStats('A',httpReq));
378                R.setStat("ASTATE",getCState('A',httpReq));
379                R.setStat("STARTASTATE",getCState('S',httpReq));
380                final StringBuffer commaList = new StringBuffer("");
381                int val=0;
382                for(int i=0;i<Race.AGE_DESCS.length;i++)
383                {
384                        final int lastVal=val;
385                        val=CMath.s_int(httpReq.getUrlParameter("AGE"+i));
386                        if(val<lastVal)
387                                val=lastVal;
388                        if(i>0)
389                                commaList.append(",");
390                        commaList.append(val);
391                }
392                R.setStat("AGING",commaList.toString());
393                List<Item> V=itemList(oldR.myResources(),'R',httpReq,false);
394                R.setStat("NUMRSC",""+V.size());
395                for(int l=0;l<V.size();l++)
396                {
397                        R.setStat("GETRSCID"+l,((Environmental)V.get(l)).ID());
398                        R.setStat("GETRSCPARM"+l,((Environmental)V.get(l)).text());
399                }
400                V=itemList(oldR.outfit(null),'O',httpReq,false);
401                R.setStat("NUMOFT",""+V.size());
402                for(int l=0;l<V.size();l++)
403                {
404                        R.setStat("GETOFTID"+l,((Environmental)V.get(l)).ID());
405                        R.setStat("GETOFTPARM"+l,((Environmental)V.get(l)).text());
406                }
407                V=itemList(new XVector<Item>(oldR.myNaturalWeapon()),'W',httpReq,true);
408                if(V.size()==0)
409                        R.setStat("WEAPONCLASS","StdWeapon");
410                else
411                {
412                        R.setStat("WEAPONCLASS",((Environmental)V.get(0)).ID());
413                        R.setStat("WEAPONXML",((Environmental)V.get(0)).text());
414                }
415                int breathe=CMath.s_int(httpReq.getUrlParameter("BREATHES"));
416                final List<Integer> l=new Vector<Integer>();
417                if(breathe>=0)
418                {
419                        l.add(Integer.valueOf(breathe));
420                        for(int i=1;;i++)
421                        {
422                                if(httpReq.isUrlParameter("BREATHES"+(Integer.toString(i))))
423                                {
424                                        breathe=CMath.s_int(httpReq.getUrlParameter("BREATHES"+(Integer.toString(i))));
425                                        if(breathe<0)
426                                        {
427                                                l.clear();
428                                                break;
429                                        }
430                                        l.add(Integer.valueOf(breathe));
431                                }
432                                else
433                                        break;
434                        }
435                }
436                R.setStat("BREATHES", CMParms.toListString(l));
437
438                setDynAbilities(R,httpReq);
439                setDynEffects(R,httpReq);
440                setDynImmunities(R,httpReq);
441
442                QuadVector<String,Integer,Integer,Boolean> CV=cabilities(httpReq);
443                R.setStat("NUMCABLE", ""+CV.size());
444                for(int i=0;i<CV.size();i++)
445                {
446                        R.setStat("GETCABLE"+i, CV.elementAt(i).first);
447                        R.setStat("GETCABLEPROF"+i, CV.elementAt(i).second.toString());
448                        R.setStat("GETCABLELVL"+i, CV.elementAt(i).third.toString());
449                        R.setStat("GETCABLEGAIN"+i, Boolean.valueOf(!CV.elementAt(i).fourth.booleanValue()).toString());
450                }
451                return "";
452        }
