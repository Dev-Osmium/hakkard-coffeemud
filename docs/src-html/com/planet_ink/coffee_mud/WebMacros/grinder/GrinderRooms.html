040        public static void happilyAddMob(MOB M, Room R)
041        {
042                M.setStartRoom(R);
043                M.setLocation(R);
044                M.phyStats().setRejuv(5000);
045                M.recoverCharStats();
046                M.recoverPhyStats();
047                M.recoverMaxState();
048                M.resetToMaxState();
049                M.bringToLife(R,true);
050                R.recoverRoomStats();
051        }
052
053        public static void happilyAddItem(Item I, Room R)
054        {
055                if(I.subjectToWearAndTear())
056                        I.setUsesRemaining(100);
057                I.recoverPhyStats();
058                R.addItem(I);
059                R.recoverRoomStats();
060        }

062        public static String editRoom(HTTPRequest httpReq, java.util.Map<String,String> parms, MOB whom, Room R)
063        {
064                if(R==null)
065                        return "Old Room not defined!";
066                boolean redoAllMyDamnRooms=false;
067                final Room oldR=R;
068
069                // class!
070                final String className=httpReq.getUrlParameter("CLASSES");
071                if((className==null)||(className.length()==0))
072                        return "Please select a class type for this room.";
073                synchronized(("SYNC"+R.roomID()).intern())
074                {
075                        R=CMLib.map().getRoom(R);
076
077                        final boolean singleMobMode=CMath.s_bool(httpReq.getUrlParameter("SINGLEMOB"));
078                        final String delMOB=singleMobMode?httpReq.getUrlParameter("DELMOB"):null;
079
080                        CMLib.map().resetRoom(R);
081                        final Room copyRoom=(Room)R.copyOf();
082                        boolean skipImage=false;
083
084                        if(!className.equalsIgnoreCase(CMClass.classID(R)))
085                        {
086                                R=CMClass.getLocale(className);
087                                if(R==null)
088                                        return "The class you chose does not exist.  Choose another.";
089                                oldR.delAllEffects(true);
090                                CMLib.threads().deleteTick(oldR,-1);
091                                R.setRoomID(oldR.roomID());
092                                final Area area=oldR.getArea();
093                                if(area!=null)
094                                        area.delProperRoom(oldR);
095                                R.setArea(area);
096                                for(int d=0;d<R.rawDoors().length;d++)
097                                        R.rawDoors()[d]=oldR.rawDoors()[d];
098                                for(int d=Directions.NUM_DIRECTIONS()-1;d>=0;d--)
099                                {
100                                        final Exit E=oldR.getRawExit(d);
101                                        if(E!=null)
102                                                R.setRawExit(d,(Exit)E.copyOf());
103                                }
104                                redoAllMyDamnRooms=true;
105                                if(R.image().equalsIgnoreCase(CMLib.protocol().getDefaultMXPImage(oldR)))
106                                {
107                                        R.setImage(null);
108                                        skipImage=true;
109                                }
110                        }
111
112                        // name
113                        final String name=httpReq.getUrlParameter("NAME");
114                        if((name==null)||(name.length()==0))
115                                return "Please enter a name for this room.";
116                        R.setDisplayText(name);
117
118                        // description
119                        String desc=httpReq.getUrlParameter("DESCRIPTION");
120                        if(desc==null)
121                                desc="";
122                        R.setDescription(desc);
123
124                        // climate
125                        /*
126                        if(httpReq.isUrlParameter("CLIMATE"))
127                        {
128                                int climate=CMath.s_int(httpReq.getUrlParameter("CLIMATE"));
129                                if(climate>=0)
130                                {
131                                        for(int i=1;;i++)
132                                        {
133                                                if(httpReq.isUrlParameter("CLIMATE"+(Integer.toString(i))))
134                                                {
135                                                        int newVal=CMath.s_int(httpReq.getUrlParameter("CLIMATE"+(Integer.toString(i))));
136                                                        if(newVal<0)
137                                                        {
138                                                                climate=-1;
139                                                                break;
140                                                        }
141                                                        climate=climate|newVal;
142                                                }
143                                                else
144                                                        break;
145                                        }
146                                }
147                                R.setClimateType(climate);
148                        }
149                        else
150                                R.setClimateType(-1);
151                        */
152                        // atmosphere
153                        //if(httpReq.isUrlParameter("ATMOSPHERE"))
154                        //      R.setAtmosphere(CMath.s_int(httpReq.getUrlParameter("ATMOSPHERE")));
155
156                        // image
157                        if(!skipImage)
158                        {
159                                String img=httpReq.getUrlParameter("IMAGE");
160                                if(img==null)
161                                        img="";
162                                R.setImage(img);
163                        }
164
165                        if(R instanceof GridLocale)
166                        {
167                                String x=httpReq.getUrlParameter("XGRID");
168                                if(x==null)
169                                        x="";
170                                ((GridLocale)R).setXGridSize(CMath.s_int(x));
171                                String y=httpReq.getUrlParameter("YGRID");
172                                if(y==null)
173                                        y="";
174                                ((GridLocale)R).setYGridSize(CMath.s_int(y));
175                                ((GridLocale)R).clearGrid(null);
176                        }
177
178                        String error=GrinderAreas.doAffects(R,httpReq,parms);
179                        if(error.length()>0)
180                                return error;
181                        error=GrinderAreas.doBehavs(R,httpReq,parms);
182                        if(error.length()>0)
183                                return error;
184
185                        // here's where you resolve items and mobs
186                        final List<MOB> allmobs=new ArrayList<MOB>();
187                        int skip=0;
188                        while(oldR.numInhabitants()>(skip))
189                        {
190                                final MOB M=oldR.fetchInhabitant(skip);
191                                if(M.isSavable())
192                                {
193                                        if(!allmobs.contains(M))
194                                                allmobs.add(M);
195                                        oldR.delInhabitant(M);
196                                }
197                                else
198                                if(oldR!=R)
199                                {
200                                        oldR.delInhabitant(M);
201                                        R.bringMobHere(M,true);
202                                }
203                                else
204                                        skip++;
205                        }
206                        final List<Item> allitems=new ArrayList<Item>();
207                        while(oldR.numItems()>0)
208                        {
209                                final Item I=oldR.getItem(0);
210                                if(!allitems.contains(I))
211                                        allitems.add(I);
212                                oldR.delItem(I);
213                        }
214
215                        if(httpReq.isUrlParameter("MOB1"))
216                        {
217                                for(int i=1;;i++)
218                                {
219                                        final String MATCHING=httpReq.getUrlParameter("MOB"+i);
220                                        if(MATCHING==null)
221                                                break;
222                                        else
223                                        if(RoomData.isAllNum(MATCHING))
224                                        {
225                                                final MOB M=RoomData.getMOBFromCode(allmobs,MATCHING);
226                                                if(M!=null)
227                                                {
228                                                        if(MATCHING.equalsIgnoreCase(delMOB))
229                                                                continue;
230                                                        happilyAddMob(M,R);
231                                                }
232                                                else
233                                                {
234                                                        final StringBuffer str=new StringBuffer("!!!No MOB?!!!!");
235                                                        str.append(" Got: "+MATCHING);
236                                                }
237                                        }
238                                        else
239                                        if(MATCHING.startsWith("CATALOG-"))
240                                        {
241                                                final MOB M=RoomData.getMOBFromCatalog(MATCHING);
242                                                if(M!=null)
243                                                        happilyAddMob((MOB)M.copyOf(),R);
244                                        }
245                                        else
246                                        if(MATCHING.indexOf('@')>0)
247                                        {
248                                                for (final MOB M2 : RoomData.getMOBCache())
249                                                {
250                                                        if(MATCHING.equals(""+M2))
251                                                        {
252                                                                happilyAddMob((MOB)M2.copyOf(),R);
253                                                                break;
254                                                        }
255                                                }
256                                        }
257                                        else
258                                        for(final Enumeration<MOB> m=CMClass.mobTypes();m.hasMoreElements();)
259                                        {
260                                                final MOB M2=m.nextElement();
261                                                if((CMClass.classID(M2).equals(MATCHING)))
262                                                {
263                                                        happilyAddMob((MOB)M2.copyOf(),R);
264                                                        break;
265                                                }
266                                        }
267                                }
268                        }
269                        else
270                                return "No MOB Data!";
271
272                        if(httpReq.isUrlParameter("ITEM1"))
273                        {
274                                final List<Item> items=new ArrayList<Item>();
275                                final List<String> cstrings=new ArrayList<String>();
276                                for(int i=1;;i++)
277                                {
278                                        final String MATCHING=httpReq.getUrlParameter("ITEM"+i);
279                                        if(MATCHING==null)
280                                                break;
281                                        Item I2=RoomData.getItemFromAnywhere(allitems,MATCHING);
282                                        if(I2!=null)
283                                        {
284                                                if(!RoomData.isAllNum(MATCHING))
285                                                        I2=(Item)I2.copyOf();
286                                                if(I2!=null)
287                                                {
288                                                        I2.unWear();
289                                                        //if(worn) I2.wearEvenIfImpossible(M);
290                                                        happilyAddItem(I2,R);
291                                                        items.add(I2);
292                                                        I2.setContainer(null);
293                                                        final String CONTAINER=httpReq.getUrlParameter("ITEMCONT"+i);
294                                                        cstrings.add((CONTAINER==null)?"":CONTAINER);
295                                                }
296                                        }
297                                }
298                                for(int i=0;i<cstrings.size();i++)
299                                {
300                                        final String CONTAINER=cstrings.get(i);
301                                        if(CONTAINER.length()==0)
302                                                continue;
303                                        final Item I2=items.get(i);
304                                        final Item C2=(Item)CMLib.english().fetchEnvironmental(items,CONTAINER,true);
305                                        if(C2 instanceof Container)
306                                                I2.setContainer((Container)C2);
307                                }
308                        }
309                        else
310                                return "No Item Data!";
311
312                        for(int i=0;i<allitems.size();i++)
313                        {
314                                final Item I=allitems.get(i);
315                                if(!R.isContent(I))
316                                        I.destroy();
317                        }
318                        for(int i=0;i<R.numItems();i++)
319                        {
320                                final Item I=R.getItem(i);
321                                if((I.container()!=null)&&(!R.isContent(I.container())))
322                                        I.setContainer(null);
323                        }
324                        for(int m=0;m<allmobs.size();m++)
325                        {
326                                final MOB M=allmobs.get(m);
327                                if(!R.isInhabitant(M))
328                                        M.destroy();
329                        }
330
331                        if(redoAllMyDamnRooms)
332                        {
333                                try
334                                {
335                                        for(final Enumeration<Room> r=CMLib.map().rooms();r.hasMoreElements();)
336                                        {
337                                                final Room R2=r.nextElement();
338                                                for(int d=0;d<R2.rawDoors().length;d++)
339                                                {
340                                                        if(R2.rawDoors()[d]==oldR)
341                                                        {
342                                                                R2.rawDoors()[d]=R;
343                                                                if(R2 instanceof GridLocale)
344                                                                        ((GridLocale)R2).buildGrid();
345                                                        }
346                                                }
347                                        }
348                                }
349                                catch(final NoSuchElementException e)
350                                {
351                                }
352                                try
353                                {
354                                        for(final Enumeration<MOB> e=CMLib.players().players();e.hasMoreElements();)
355                                        {
356                                                final MOB M=e.nextElement();
357                                                if(M.getStartRoom()==oldR)
358                                                        M.setStartRoom(R);
359                                                else
360                                                if(M.location()==oldR)
361                                                        M.setLocation(R);
362                                        }
363                                }
364                                catch(final NoSuchElementException e)
365                                {
366                                }
367                        }
368                        R.getArea().fillInAreaRoom(R);
369                        CMLib.database().DBUpdateRoom(R);
370                        CMLib.database().DBUpdateMOBs(R);
371                        CMLib.database().DBUpdateItems(R);
372                        R.startItemRejuv();
373                        if(oldR!=R)
374                        {
375                                oldR.destroy();
376                                R.getArea().addProperRoom(R);
377                        }
378                        if(!copyRoom.sameAs(R))
379                                Log.sysOut("Grinder",whom.Name()+" modified room "+R.roomID()+".");
380                        copyRoom.destroy();
381                }
382                return "";
383        }

391        public static Room createLonelyRoom(Area A, Room linkTo, int dir, boolean copyThisOne)
392        {
393                Room newRoom=null;
394                final String newRoomID=A.getNewRoomID(linkTo,dir);
395                if(newRoomID.length()==0)
396                        return null;
397                if((copyThisOne)&&(linkTo!=null))
398                {
399                        CMLib.map().resetRoom(linkTo);
400                        newRoom=(Room)linkTo.copyOf();
401                        for(int d=Directions.NUM_DIRECTIONS()-1;d>=0;d--)
402                        {
403                                newRoom.rawDoors()[d]=null;
404                                newRoom.setRawExit(d,null);
405                        }
406                }
407                else
408                {
409                        newRoom=CMClass.getLocale("StdRoom");
410                        newRoom.setDisplayText(CMLib.lang().L("Title of @x1",newRoomID));
411                        newRoom.setDescription(CMLib.lang().L("Description of @x1",newRoomID));
412                }
413                newRoom.setRoomID(newRoomID);
414                newRoom.setArea(A);
415                if(linkTo!=null)
416                {
417                        newRoom.rawDoors()[Directions.getOpDirectionCode(dir)]=linkTo;
418                        newRoom.setRawExit(Directions.getOpDirectionCode(dir),CMClass.getExit("StdOpenDoorway"));
419                }
420                CMLib.database().DBCreateRoom(newRoom);
421                CMLib.database().DBUpdateExits(newRoom);
422                if(newRoom.numInhabitants()>0)
423                        CMLib.database().DBUpdateMOBs(newRoom);
424                if(newRoom.numItems()>0)
425                        CMLib.database().DBUpdateItems(newRoom);
426                newRoom.getArea().fillInAreaRoom(newRoom);
427                return newRoom;
428        }

430        public static String createRoom(Room R, int dir, boolean copyThisOne)
431        {
432                R.clearSky();
433                if(R instanceof GridLocale)
434                        ((GridLocale)R).clearGrid(null);
435                final Room newRoom=createLonelyRoom(R.getArea(),R,dir,copyThisOne);
436                R.rawDoors()[dir]=newRoom;
437                if(R.getRawExit(dir)==null)
438                        R.setRawExit(dir,CMClass.getExit("StdOpenDoorway"));
439                CMLib.database().DBUpdateExits(R);
440                R.getArea().fillInAreaRoom(R);
441                return "";
442        }

444        public static Room createGridRoom(Area A, String roomID, Room copyThisOne, RoomnumberSet deferredExitSaves, boolean autoLink)
445        {
446                Room R=null;
447                if(copyThisOne!=null)
448                {
449                        R=(Room)copyThisOne.copyOf();
450                        R.setRoomID(roomID);
451                }
452                else
453                {
454                        R=CMClass.getLocale("StdRoom");
455                        R.setRoomID(roomID);
456                        R.setDisplayText(CMLib.lang().L("Title of @x1",R.roomID()));
457                        R.setDescription(CMLib.lang().L("Description of @x1",R.roomID()));
458                }
459                R.setArea(A);
460                CMLib.database().DBCreateRoom(R);
461                if(R.numInhabitants()>0)
462                        CMLib.database().DBUpdateMOBs(R);
463                if(R.numItems()>0)
464                        CMLib.database().DBUpdateItems(R);
465                R.getArea().fillInAreaRoom(R);
466                if((autoLink)&&(R.getArea() instanceof GridZones))
467                {
468                        final GridZones GZ=(GridZones)R.getArea();
469                        final GridZones.XYVector Rxy=GZ.getRoomXY(R);
470                        boolean resaveMyExits=false;
471                        if((Rxy.x>=0)&&(Rxy.y>=0))
472                        {
473                                Room R2=null;
474
475                                for(int d=Directions.NUM_DIRECTIONS()-1;d>=0;d--)
476                                {
477                                        final int[] xy=Directions.adjustXYByDirections(Rxy.x,Rxy.y,d);
478                                        R2=GZ.getGridChild(xy[0],xy[1]);
479                                        if((R2!=null)&&(R!=R2))
480                                        {
481                                                final int opD=Directions.getOpDirectionCode(d);
482                                                if(R2.rawDoors()[opD]==null)
483                                                {
484                                                        R2.rawDoors()[opD]=R;
485                                                        if(R2.getRawExit(opD)==null)
486                                                                R2.setRawExit(opD,CMClass.getExit("StdOpenDoorway"));
487                                                        if(deferredExitSaves!=null)
488                                                        {
489                                                                if(!deferredExitSaves.contains(R2.roomID()))
490                                                                        deferredExitSaves.add(R2.roomID());
491                                                        }
492                                                        else
493                                                                CMLib.database().DBUpdateExits(R2);
494                                                }
495                                                if(R.rawDoors()[d]==null)
496                                                {
497                                                        R.rawDoors()[d]=R2;
498                                                        if(R.getRawExit(d)==null)
499                                                                R.setRawExit(d,CMClass.getExit("StdOpenDoorway"));
500                                                        resaveMyExits=true;
501                                                }
502                                        }
503                                }
504                        }
505                        if(resaveMyExits)
506                        {
507                                if(deferredExitSaves!=null)
508                                {
509                                        if(!deferredExitSaves.contains(R.roomID()))
510                                                deferredExitSaves.add(R.roomID());
511                                }
512                                else
513                                {
514                                        CMLib.database().DBUpdateExits(R);
515                                        R.getArea().fillInAreaRoom(R);
516                                }
517                        }
518                }
519                return R;
520        }
