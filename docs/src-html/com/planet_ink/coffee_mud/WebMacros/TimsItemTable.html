037public class TimsItemTable extends StdWebMacro
038{
039        @Override
040        public String name()
041        {
042                return "TimsItemTable";
043        }
044
045        @Override
046        public boolean isAdminMacro()
047        {
048                return true;
049        }
050
051        @Override
052        public String runMacro(HTTPRequest httpReq, String parm, HTTPResponse httpResp)
053        {
054                long endTime=System.currentTimeMillis()+(1000*60*10);
055                final int min=CMath.s_int((httpReq.getUrlParameter("MIN")));
056                if(min>0)
057                        endTime=System.currentTimeMillis()+(1000*60*((long)min));
058                final java.util.Map<String,String> parms=parseParms(parm);
059                final StringBuffer str=new StringBuffer("<TABLE WIDTH=100% BORDER=1>");
060                if(parms.containsKey("WORLD"))
061                {
062                        str.append("<TR><TD>Name</TD><TD>LVL</TD><TD>TVLV</TD><TD>DIFF</TD><TD>DIFF%</TD><TD>ARM</TD><TD>ATT</TD><TD>DAM</TD><TD>ADJ</TD><TD>CAST</TD><TD>RESIST</TD></TR>");
063                        final Vector<Environmental> onesDone=new Vector<Environmental>();
064                        for(final Enumeration<Area> a=CMLib.map().areas();a.hasMoreElements();)
065                        {
066                                final Area A=a.nextElement();
067                                for(final Enumeration<Room> r=A.getCompleteMap();r.hasMoreElements();)
068                                {
069                                        final Room R=r.nextElement();
070                                        if((endTime>0)&&(System.currentTimeMillis()>endTime))
071                                                break;
072                                        for(int i=0;i<R.numItems();i++)
073                                        {
074                                                final Item I=R.getItem(i);
075                                                if((endTime>0)&&(System.currentTimeMillis()>endTime))
076                                                        break;
077                                                if(!doneBefore(onesDone,I))
078                                                        str.append(addRow(I));
079                                        }
080                                        if((endTime>0)&&(System.currentTimeMillis()>endTime))
081                                                break;
082                                        for(int m=0;m<R.numInhabitants();m++)
083                                        {
084                                                if((endTime>0)&&(System.currentTimeMillis()>endTime))
085                                                        break;
086                                                final MOB M=R.fetchInhabitant(m);
087                                                if(M==null)
088                                                        continue;
089                                                for(int i=0;i<M.numItems();i++)
090                                                {
091                                                        final Item I=M.getItem(i);
092                                                        if((endTime>0)&&(System.currentTimeMillis()>endTime))
093                                                                break;
094                                                        if(!doneBefore(onesDone,I))
095                                                                str.append(addRow(I));
096                                                }
097                                                if((endTime>0)&&(System.currentTimeMillis()>endTime))
098                                                        break;
099                                                if(!(M instanceof ShopKeeper))
100                                                        continue;
101                                                final ShopKeeper SK=(ShopKeeper)M;
102                                                for(final Iterator<Environmental> i=SK.getShop().getStoreInventory();i.hasNext();)
103                                                {
104                                                        final Environmental E=i.next();
105                                                        if((endTime>0)&&(System.currentTimeMillis()>endTime))
106                                                                break;
107                                                        if((E instanceof Item)
108                                                        &&(!doneBefore(onesDone,(Item)E)))
109                                                                str.append(addRow((Item)E));
110                                                }
111                                        }
112                                }
113                        }
114                }
115                else
116                if(parms.containsKey("CRAFTED"))
117                {
118                        str.append("<TR><TD>Name</TD><TD>LVL</TD><TD>TVLV</TD><TD>DIFF</TD><TD>DIFF%</TD><TD>ARM</TD><TD>ATT</TD><TD>DAM</TD><TD>ADJ</TD><TD>CAST</TD><TD>RESIST</TD></TR>");
119                        final List<ItemCraftor> allCraftingSkills=new SVector<ItemCraftor>();
120                        for(final Enumeration<Ability> e=CMClass.abilities();e.hasMoreElements();)
121                        {
122                                final Ability A=e.nextElement();
123                                if(A instanceof ItemCraftor)
124                                        allCraftingSkills.add((ItemCraftor)A.copyOf());
125                        }
126                        final List<ItemCraftor> craftingSkills=new SVector<ItemCraftor>();
127                        for(final ItemCraftor skill : allCraftingSkills)
128                        {
129                                final int ii=CMLib.ableMapper().lowestQualifyingLevel(skill.ID());
130                                for(int i=0;i<=craftingSkills.size();i++)
131                                {
132                                        if(i>=craftingSkills.size())
133                                        {
134                                                craftingSkills.add(skill);
135                                                break;
136                                        }
137                                        else
138                                        if(ii<CMLib.ableMapper().lowestQualifyingLevel(craftingSkills.get(i).ID()))
139                                        {
140                                                craftingSkills.add(i,skill);
141                                                break;
142                                        }
143                                }
144                        }
145                        for(final ItemCraftor skill : craftingSkills)
146                        {
147                                final List<ItemCraftor.ItemKeyPair> pairs=skill.craftAllItemSets(false);
148                                if(pairs!=null)
149                                {
150                                        for(final ItemCraftor.ItemKeyPair IP : pairs)
151                                        {
152                                                if(IP.item instanceof Weapon)
153                                                        str.append(addRow(IP.item));
154                                                else
155                                                if(IP.item instanceof Armor)
156                                                        str.append(addRow(IP.item));
157                                        }
158                                }
159                        }
160                }
161                else
162                if(parms.containsKey("ITEMS"))
163                {
164                        final int[] materials={RawMaterial.RESOURCE_STEEL,RawMaterial.RESOURCE_IRON,RawMaterial.RESOURCE_OAK,RawMaterial.RESOURCE_LEATHER,RawMaterial.RESOURCE_COTTON};
165                        str.append("<TR><TD>Type</TD><TD>Level</TD>");
166                        for (final int material : materials)
167                                str.append("<TD>"+RawMaterial.CODES.NAME(material)+"</TD>");
168                        str.append("</TR>");
169                        //str.append("<TD>Attack</TD>");
170                        //str.append("<TD>Damage</TD>");
171                        final Wearable.CODES codes = Wearable.CODES.instance();
172                        for(int level=1;level<=91;level++)
173                        {
174                                str.append("<TR><TD>Armor</TD><TD>"+level+"</TD>");
175                                for (final int material : materials)
176                                {
177                                        int armorBonus=0;
178                                        for(final long wornCode : codes.all())
179                                        {
180                                                if((wornCode == Wearable.IN_INVENTORY)
181                                                || (wornCode == Wearable.WORN_HELD)
182                                                || (wornCode == Wearable.WORN_WIELD)
183                                                || (wornCode == Wearable.WORN_MOUTH))
184                                                        continue;
185                                                final Armor A=CMClass.getArmor("GenArmor");
186                                                A.setRawProperLocationBitmap(wornCode);
187                                                A.setMaterial(RawMaterial.RESOURCE_STEEL);
188                                                if((CharClass.ARMOR_WEARMASK & wornCode) > 0)
189                                                        A.setMaterial(material);
190                                                A.basePhyStats().setLevel(level);
191                                                A.basePhyStats().setWeight(5);
192                                                A.recoverPhyStats();
193                                                CMLib.itemBuilder().balanceItemByLevel(A);
194                                                armorBonus+=A.basePhyStats().armor();
195                                        }
196                                        str.append("<TD>"+armorBonus+"</TD>");
197                                }
198                                str.append("</TR>");
199                        }
200                        for(int level=1;level<=91;level++)
201                        {
202                                str.append("<TR><TD>Weapons</TD><TD>"+level+"</TD>");
203                                for (final int material : materials)
204                                {
205                                        final Weapon W=CMClass.getWeapon("GenWeapon");
206                                        W.setWeaponClassification(Weapon.CLASS_SWORD);
207                                        W.setWeaponDamageType(Weapon.TYPE_SLASHING);
208                                        W.setMaterial(material);
209                                        W.setRawProperLocationBitmap(Wearable.WORN_WIELD|Wearable.WORN_HELD);
210                                        W.setRawLogicalAnd(false);
211                                        W.basePhyStats().setLevel(level);
212                                        W.basePhyStats().setWeight(8);
213                                        W.recoverPhyStats();
214                                        CMLib.itemBuilder().balanceItemByLevel(W);
215                                        str.append("<TD>"+W.basePhyStats().attackAdjustment()+"/"+W.basePhyStats().damage()+"</TD>");
216                                }
217                                str.append("</TR>");
218                        }
219                }
220                else
221                        str.append("<TR><TD><FONT COLOR=RED>Unknown parms: "+CMParms.toKeyValueSlashListString(parms)+".  Try WORLD or ITEMS.</TD></TR>");
222                return clearWebMacros(str)+"</TABLE>";
223        }
224
225        public boolean doneBefore(Vector<Environmental> V, Item I)
226        {
227                if(I==null)
228                        return true;
229                if((!(I instanceof Armor))&&(!(I instanceof Weapon)))
230                        return true;
231                if(I.displayText().length()==0)
232                        return true;
233                for(int i=0;i<V.size();i++)
234                {
235                        if(I.sameAs(V.elementAt(i)))
236                                return true;
237                }
238                V.addElement(I);
239                return false;
240        }
241
242        public String addRow(Item I)
243        {
244                final StringBuffer row=new StringBuffer("");
245                final int lvl=I.phyStats().level();
246                row.append("<TR>");
247                row.append("<TD>"+I.name()+"</TD>");
248                row.append("<TD>"+lvl+"</TD>");
249                final int[] castMul=new int[1];
250                final Ability[] RET=CMLib.itemBuilder().getTimsAdjResCast(I,castMul);
251                final Ability ADJ=RET[0];
252                final Ability RES=RET[1];
253                final Ability CAST=RET[2];
254                final int tlvl=CMLib.itemBuilder().timsLevelCalculator(I,ADJ,RES,CAST,castMul[0]);
255                row.append("<TD>"+tlvl+"</TD>");
256                int diff=tlvl-lvl; if(diff<0) diff=diff*-1;
257                row.append("<TD>"+diff+"</TD>");
258                int pct=0;
259                if((lvl<0)&&(tlvl>=0))
260                        pct=(int)Math.round(CMath.div(tlvl+(lvl*-1),1)*100.0);
261                else
262                if((tlvl<=0)&&(lvl>0))
263                        pct=(int)Math.round(CMath.div((tlvl-lvl),-1)*100.0);
264                else
265                if((tlvl<0)&&(lvl==0))
266                        pct=(int)Math.round(CMath.div(tlvl,-1)*100.0);
267                else
268                if(lvl==0)
269                        pct=(int)Math.round(CMath.div(tlvl,1)*100.0);
270                else
271                        pct=(int)Math.round(CMath.div(tlvl,lvl)*100.0);
272                row.append("<TD>"+pct+"%</TD>");
273
274                if(!(I instanceof Weapon))
275                        row.append("<TD>"+I.basePhyStats().armor()+"</TD><TD>&nbsp;</TD><TD>&nbsp;</TD>");
276                else
277                {
278                        row.append("<TD>&nbsp;</TD><TD>"+I.basePhyStats().attackAdjustment()+"</TD>");
279                        row.append("<TD>"+I.basePhyStats().damage()+"</TD>");
280                }
281                if(ADJ!=null)
282                        row.append("<TD>"+ADJ.text()+"</TD>");
283                else row.append("<TD>&nbsp;</TD>");
284                if(CAST!=null)
285                        row.append("<TD>"+CAST.text()+"</TD>");
286                else row.append("<TD>&nbsp;</TD>");
287                if(RES!=null)
288                        row.append("<TD>"+RES.text()+"</TD>");
289                else row.append("<TD>&nbsp;</TD>");
290                row.append("</TR>");
291                return row.toString();
292        }
