036@SuppressWarnings({"unchecked","rawtypes"})
037public class AbilityRecipeData extends StdWebMacro
038{
039        @Override
040        public String name()
041        {
042                return "AbilityRecipeData";
043        }
044
045        // valid parms include help, ranges, quality, target, alignment, domain,
046        // qualifyQ, auto
047        @Override
048        public String runMacro(HTTPRequest httpReq, String parm, HTTPResponse httpResp)
049        {
050                final java.util.Map<String,String> parms=parseParms(parm);
051
052                final String replaceCommand=httpReq.getUrlParameter("REPLACE");
053                if((replaceCommand != null)
054                && (replaceCommand.length()>0)
055                && (replaceCommand.indexOf('=')>0))
056                {
057                        final int eq=replaceCommand.indexOf('=');
058                        final String field=replaceCommand.substring(0,eq);
059                        final String value=replaceCommand.substring(eq+1);
060                        httpReq.addFakeUrlParameter(field, value);
061                        httpReq.addFakeUrlParameter("REPLACE","");
062                }
063
064                final String last=httpReq.getUrlParameter("ABILITY");
065                if(last==null)
066                        return " @break@";
067
068                final String rownum=httpReq.getUrlParameter("ABILITYRECIPEROW");
069
070                if(last.length()>0)
071                {
072                        final Ability A=CMClass.getAbility(last);
073                        if((A!=null)
074                        &&(A instanceof CraftorAbility)
075                        &&(((CraftorAbility)A).parametersFile()!=null)
076                        &&(((CraftorAbility)A).parametersFile().length()>0)
077                        &&(((CraftorAbility)A).parametersFormat()!=null)
078                        &&(((CraftorAbility)A).parametersFormat().length()>0))
079                        {
080                                AbilityParameters.AbilityRecipeData recipeData =
081                                        (AbilityParameters.AbilityRecipeData)httpReq.getRequestObjects().get("ABILITYRECIPEDATA-"+last);
082                                if(recipeData == null)
083                                {
084                                        recipeData = CMLib.ableParms().parseRecipe(((CraftorAbility)A).parametersFile(),((CraftorAbility)A).parametersFormat());
085                                        if(recipeData.parseError() != null)
086                                        {
087                                                Log.errOut(ID(),recipeData.parseError());
088                                                return " @break@";
089                                        }
090                                        httpReq.getRequestObjects().put("ABILITYRECIPEDATA-"+last,recipeData);
091                                }
092                                final StringBuffer str=new StringBuffer("");
093                                final String sfont=(parms.containsKey("FONT"))?("<FONT "+(parms.get("FONT"))+">"):"";
094                                final String efont=(parms.containsKey("FONT"))?"</FONT>":"";
095                                final String hsfont=(parms.containsKey("HFONT"))?("<FONT "+(parms.get("HFONT"))+">"):"";
096                                final String hefont=(parms.containsKey("HFONT"))?"</FONT>":"";
097
098                                if(parms.containsKey("SAVETOVFS"))
099                                {
100                                        if(httpReq.isUrlParameter("SAVETOVFS"))
101                                                str.append(CMath.s_bool(httpReq.getUrlParameter("SAVETOVFS"))?"CHECKED":"");
102                                        else
103                                                str.append(recipeData.wasVFS()?"CHECKED":"");
104                                }
105                                else
106                                if(parms.containsKey("ROWTABLE")&&(CMath.isInteger(rownum)))
107                                {
108                                        final int row = CMath.s_int(rownum);
109                                        DVector dataRow = null;
110                                        final int classFieldIndex = recipeData.getClassFieldIndex();
111                                        if((row>0)&&((row-1)<recipeData.dataRows().size()))
112                                                dataRow = recipeData.dataRows().get(row-1);
113                                        else
114                                                dataRow=recipeData.newRow(httpReq.getUrlParameter("CLASSFIELD"));
115                                        str.append("\n\r<TABLE WIDTH=100% BORDER=1 CELLSPACING=0 CELLPADDING=0>");
116                                        for(int c=0;c<dataRow.size();c++)
117                                        {
118                                                final AbilityParameters.AbilityParmEditor editor =
119                                                        CMLib.ableParms().getEditors().get(dataRow.elementAt(c,1));
120                                                final String oldVal = (String)dataRow.elementAt(c,2);
121                                                if(!editor.ID().equalsIgnoreCase("N_A"))
122                                                {
123                                                        str.append("\n\r<TR>");
124                                                        str.append("<TD WIDTH=20%>" + hsfont + editor.prompt() + hefont + "</TD>");
125                                                        if(c==classFieldIndex)
126                                                                str.append("<TD>" + sfont + editor.webValue(httpReq,parms,oldVal,"DATA_"+row+"_"+c) + efont + "</TD>");
127                                                        else
128                                                                str.append("<TD>" + sfont + editor.webField(httpReq,parms,oldVal,"DATA_"+row+"_"+c) + efont + "</TD>");
129                                                        str.append("</TR>");
130                                                }
131                                        }
132                                        str.append("\n\r</TABLE>");
133                                        if(classFieldIndex>=0)
134                                        {
135                                                final String oldVal = (String)dataRow.elementAt(classFieldIndex,2);
136                                                final AbilityParameters.AbilityParmEditor editor =
137                                                        CMLib.ableParms().getEditors().get(dataRow.elementAt(classFieldIndex,1));
138                                                str.append("<INPUT TYPE=HIDDEN NAME=CLASSFIELD VALUE=\""+editor.webValue(httpReq,parms,oldVal,"CLASSFIELD")+"\">");
139                                        }
140                                }
141                                else
142                                if(parms.containsKey("ADDROW"))
143                                {
144                                        AbilityParameters.AbilityParmEditor classFieldEditor = null;
145                                        final int cfIndex = recipeData.getClassFieldIndex();
146                                        if(recipeData.dataRows().size()==0)
147                                        {
148                                                final DVector editRow = new DVector(2);
149                                                for(int c=0;c<recipeData.columns().size();c++)
150                                                {
151                                                        if(recipeData.columns().get(c) instanceof List)
152                                                                editRow.addElement(recipeData.columns().get(c),"");
153                                                }
154                                                final List<String> o=(List)editRow.elementAt(cfIndex,1);
155                                                classFieldEditor = CMLib.ableParms().getEditors().get(o.get(0).toString());
156                                        }
157                                        else
158                                        for(int row=0;row<recipeData.dataRows().size();row++)
159                                        {
160                                                if(cfIndex>=0)
161                                                {
162                                                        final DVector dataRow = recipeData.dataRows().get(row);
163                                                        classFieldEditor = CMLib.ableParms().getEditors().get(dataRow.elementAt(cfIndex,1));
164                                                }
165                                        }
166                                        if(classFieldEditor != null)
167                                                str.append(classFieldEditor.webField(httpReq,parms,classFieldEditor.defaultValue(),"NEWCLASSFIELD"));
168                                }
169                                else
170                                if(parms.containsKey("SAVEROW")&&(CMath.isInteger(rownum)))
171                                {
172                                        DVector dataRow = null;
173                                        final int row = CMath.s_int(rownum);
174                                        if((row-1>=0)&&(row-1<recipeData.dataRows().size()))
175                                                dataRow = recipeData.dataRows().get(row-1);
176                                        else
177                                        {
178                                                dataRow=recipeData.newRow(httpReq.getUrlParameter("CLASSFIELD"));
179                                                recipeData.dataRows().add(dataRow);
180                                        }
181                                        for(int c=0;c<dataRow.size();c++)
182                                        {
183                                                final AbilityParameters.AbilityParmEditor editor =
184                                                        CMLib.ableParms().getEditors().get(dataRow.elementAt(c,1));
185                                                final String oldVal = (String)dataRow.elementAt(c,2);
186                                                String newVal = editor.webValue(httpReq,parms,oldVal,"DATA_"+row+"_"+c);
187                                                if(newVal != null)
188                                                        newVal = newVal.replace('\'', '`');
189                                                dataRow.setElementAt(c,2,newVal);
190                                        }
191                                        final MOB M = Authenticate.getAuthenticatedMob(httpReq);
192                                        if(M==null)
193                                                return " @break@";
194                                        final boolean saveToVFS = CMath.s_bool(httpReq.getUrlParameter("SAVETOVFS"));
195                                        if(CMSecurity.isAllowedAnywhere(M,CMSecurity.SecFlag.CMDRECIPES))
196                                                CMLib.ableParms().resaveRecipeFile(M,recipeData.recipeFilename(),recipeData.dataRows(),recipeData.columns(), saveToVFS);
197                                        else
198                                                return " @break@";
199                                }
200                                else
201                                if(parms.containsKey("DELROW")&&(CMath.isInteger(rownum)))
202                                {
203                                        final int row = CMath.s_int(rownum);
204                                        if((row-1>=0)&&(row-1<recipeData.dataRows().size()))
205                                                recipeData.dataRows().remove(row-1);
206                                        else
207                                                return " @break@";
208                                        final MOB M = Authenticate.getAuthenticatedMob(httpReq);
209                                        if(M==null)
210                                                return " @break@";
211                                        final boolean saveToVFS = CMath.s_bool(httpReq.getUrlParameter("SAVETOVFS"));
212                                        if(CMSecurity.isAllowedAnywhere(M,CMSecurity.SecFlag.CMDRECIPES))
213                                                CMLib.ableParms().resaveRecipeFile(M,recipeData.recipeFilename(),recipeData.dataRows(),recipeData.columns(), saveToVFS);
214                                        else
215                                                return " @break@";
216                                }
217                                else
218                                if(parms.containsKey("TABLE"))
219                                {
220                                        str.append("\n\r<TABLE WIDTH=100% BORDER=1 CELLSPACING=0 CELLPADDING=0>");
221                                        //int currLenTotal = 0;
222                                        //for(int l=0;l<recipeData.columnLengths().length;l++)
223                                        //    currLenTotal+=recipeData.columnLengths()[l];
224                                        str.append("\n\r<TR>");
225                                        str.append("<TD WIDTH=1%>" + hsfont + "#" + hefont + "</TD>");
226                                        for(int c=0;c<recipeData.columnHeaders().length;c++)
227                                        {
228                                                str.append("<TD WIDTH="+Math.round(CMath.div(recipeData.columnLengths()[c],72) * 100.0)+"%>");
229                                                str.append(hsfont + recipeData.columnHeaders()[c] + hefont);
230                                                str.append("</TD>");
231                                        }
232                                        str.append("</TR>");
233                                        for(int r=0;r<recipeData.dataRows().size();r++)
234                                        {
235                                                final DVector dataRow = recipeData.dataRows().get(r);
236                                                str.append("\n\r<TR>");
237                                                str.append("<TD>");
238                                                str.append("<A HREF=\"javascript:Select("+(r+1)+")\">" + sfont + "<B><FONT COLOR=YELLOW>"+(r+1)+"</FONT></B>");
239                                                str.append(efont + "</A>");
240                                                str.append("</TD>");
241                                                for(int c=0;c<dataRow.size();c++)
242                                                {
243                                                        str.append("<TD>" + sfont);
244                                                        String val = (String)dataRow.elementAt(c,2);
245                                                        final AbilityParameters.AbilityParmEditor editor =
246                                                                CMLib.ableParms().getEditors().get(dataRow.elementAt(c,1));
247                                                        val = editor.webTableField(httpReq, parms, val);
248                                                        str.append(CMStrings.limit(val,(int)Math.round(CMath.div(recipeData.columnLengths()[c],36) * 100.0)));
249                                                        str.append(efont + "</TD>");
250                                                }
251                                                str.append("</A></TR>");
252                                        }
253                                        str.append("\n\r</TABLE>");
254                                }
255                                String strstr=str.toString();
256                                if(strstr.endsWith(", "))
257                                        strstr=strstr.substring(0,strstr.length()-2);
258                                return clearWebMacros(strstr);
259                        }
260                }
261                return "";
262        }
