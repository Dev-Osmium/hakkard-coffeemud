035@SuppressWarnings("rawtypes")
036public class AbilityNext extends StdWebMacro
037{
038        @Override
039        public String name()
040        {
041                return "AbilityNext";
042        }
043
044        @Override
045        @SuppressWarnings("unchecked")
046        public String runMacro(HTTPRequest httpReq, String parm, HTTPResponse httpResp)
047        {
048                final java.util.Map<String,String> parms=parseParms(parm);
049                final String last=httpReq.getUrlParameter("ABILITY");
050                if(parms.containsKey("RESET"))
051                {
052                        if(last!=null)
053                                httpReq.removeUrlParameter("ABILITY");
054                        return "";
055                }
056                final String ableType=httpReq.getUrlParameter("ABILITYTYPE");
057                if((ableType!=null)&&(ableType.length()>0))
058                        parms.put(ableType,ableType);
059                final String domainType=httpReq.getUrlParameter("DOMAIN");
060                if((domainType!=null)&&(domainType.length()>0))
061                        parms.put("DOMAIN",domainType);
062                long flags=0;
063                final String flagString=httpReq.getUrlParameter("FLAGS");
064                if((flagString!=null)&&(flagString.length()>0))
065                {
066                        final List<String> V=CMParms.parseSquiggles(flagString.toUpperCase());
067                        for(int i=0;i<Ability.FLAG_DESCS.length;i++)
068                        {
069                                if(V.contains(Ability.FLAG_DESCS[i]))
070                                        flags=flags|(CMath.pow(2,i));
071                        }
072                }
073
074                String lastID="";
075                final String className=httpReq.getUrlParameter("CLASS");
076                final boolean genericOnly =parms.containsKey("GENERIC");
077                final boolean parmsEditable=parms.containsKey("PARMSEDITABLE");
078                final boolean unqualifiedOK=parms.containsKey("UNQUALIFIEDOK");
079                final String levelName=httpReq.getUrlParameter("LEVEL");
080                final boolean notFlag =parms.containsKey("NOT");
081                final boolean allFlag =parms.containsKey("ALL");
082                final boolean domainFlag=parms.containsKey("DOMAIN");
083                final String domain=parms.get("DOMAIN");
084                boolean containsACodeMask=false;
085                for (final String element : Ability.ACODE_DESCS)
086                {
087                        if(parms.containsKey(element))
088                        {
089                                containsACodeMask = true;
090                                break;
091                        }
092                }
093
094                final Enumeration<Ability> a;
095                if(!parms.containsKey("SORTEDBYNAME"))
096                        a=CMClass.abilities();
097                else
098                if(httpReq.getRequestObjects().containsKey("ABILITIESSORTEDBYNAME"))
099                        a=((Vector)httpReq.getRequestObjects().get("ABILITIESSORTEDBYNAME")).elements();
100                else
101                {
102                        final Vector<Ability> fullList=new Vector<Ability>();
103                        for(final Enumeration<Ability> aa=CMClass.abilities();aa.hasMoreElements();)
104                                fullList.add(aa.nextElement());
105                        final Ability[] aaray=fullList.toArray(new Ability[0]);
106                        Arrays.sort(aaray, new Comparator<Ability>()
107                        {
108                                @Override 
109                                public int compare(Ability o1, Ability o2)
110                                {
111                                        return o1.Name().compareToIgnoreCase(o2.Name());
112                                }
113                        });
114                        fullList.clear();
115                        fullList.addAll(Arrays.asList(aaray));
116                        httpReq.getRequestObjects().put("ABILITIESSORTEDBYNAME",fullList);
117                        a=fullList.elements();
118                }
119                for(;a.hasMoreElements();)
120                {
121                        final Ability A=a.nextElement();
122                        boolean okToShow=true;
123                        final int classType=A.classificationCode()&Ability.ALL_ACODES;
124                        if(genericOnly)
125                                okToShow=A.isGeneric();
126                        else
127                        if(parmsEditable)
128                        {
129                                okToShow=((A instanceof CraftorAbility)
130                                                &&(((CraftorAbility)A).parametersFile()!=null)
131                                                &&(((CraftorAbility)A).parametersFile().length()>0)
132                                                &&(((CraftorAbility)A).parametersFormat()!=null)
133                                                &&(((CraftorAbility)A).parametersFormat().length()>0));
134                        }
135
136                        if((className!=null)&&(className.length()>0))
137                        {
138                                final int level=CMLib.ableMapper().getQualifyingLevel(className,true,A.ID());
139                                if((level<0)&&(!unqualifiedOK))
140                                        okToShow=false;
141                                else
142                                if(CMLib.ableMapper().getSecretSkill(className,false,A.ID()))
143                                        okToShow=false;
144                                else
145                                if((flags>0)&&((A.flags()&flags)!=flags))
146                                        okToShow=false;
147                                else
148                                {
149                                        if((levelName!=null)&&(levelName.length()>0)&&(CMath.s_int(levelName)!=level))
150                                                okToShow=false;
151                                }
152                        }
153                        else
154                        if(!allFlag)
155                        {
156                                final int level=CMLib.ableMapper().getQualifyingLevel("Archon",true,A.ID());
157                                if((level<0)&&(!unqualifiedOK))
158                                        okToShow=false;
159                                else
160                                if(CMLib.ableMapper().getAllSecretSkill(A.ID()))
161                                        okToShow=false;
162                                else
163                                if((flags>0)&&((A.flags()&flags)!=flags))
164                                        okToShow=false;
165                                else
166                                {
167                                        if((levelName!=null)&&(levelName.length()>0)&&(CMath.s_int(levelName)!=level))
168                                                okToShow=false;
169                                }
170                        }
171                        if(okToShow)
172                        {
173                                if((domainFlag)&&(!domain.equalsIgnoreCase(Ability.DOMAIN_DESCS[(A.classificationCode()&Ability.ALL_DOMAINS)>>5])))
174                                        okToShow=false;
175                                if(containsACodeMask&&(!parms.containsKey(Ability.ACODE_DESCS[classType])))
176                                        okToShow=false;
177                        }
178                        if(notFlag)
179                                okToShow=!okToShow;
180                        if(okToShow)
181                        {
182                                if((last==null)||((last.length()>0)&&(last.equals(lastID))&&(!A.ID().equals(lastID))))
183                                {
184                                        httpReq.addFakeUrlParameter("ABILITY",A.ID());
185                                        return "";
186                                }
187                                lastID=A.ID();
188                        }
189                }
190                httpReq.addFakeUrlParameter("ABILITY","");
191                if(parms.containsKey("EMPTYOK"))
192                        return "<!--EMPTY-->";
193                return " @break@";
194        }
195}
