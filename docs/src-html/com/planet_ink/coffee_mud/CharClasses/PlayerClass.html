103        private boolean isSkill(int classCode)
104        {
105                switch(classCode&Ability.ALL_ACODES)
106                {
107                case Ability.ACODE_COMMON_SKILL:
108                case Ability.ACODE_DISEASE:
109                case Ability.ACODE_POISON:
110                case Ability.ACODE_SKILL:
111                case Ability.ACODE_THIEF_SKILL:
112                case Ability.ACODE_TRAP:
113                case Ability.ACODE_LANGUAGE:
114                case Ability.ACODE_PROPERTY:
115                case Ability.ACODE_TECH:
116                        return true;
117                case Ability.ACODE_CHANT:
118                case Ability.ACODE_PRAYER:
119                case Ability.ACODE_SPELL:
120                case Ability.ACODE_SUPERPOWER:
121                        return false;
122                }
123                return true;
124        }

126        private List<String> makeRequirements(LinkedList<List<String>> prevSets, Ability A)
127        {
128                for(final Iterator<List<String>> i=prevSets.descendingIterator();i.hasNext();)
129                {
130                        final List<String> prevSet=i.next();
131                        final List<String> reqSet=new Vector<String>();
132                        for(final String prevID : prevSet)
133                        {
134                                final Ability pA=CMClass.getAbility(prevID);
135                                if((pA!=null)&&(A.classificationCode()==pA.classificationCode()))
136                                        reqSet.add(pA.ID());
137                        }
138                        if(reqSet.size()==0)
139                        {
140                                for(final String prevID : prevSet)
141                                {
142                                        final Ability pA=CMClass.getAbility(prevID);
143                                        if((A.classificationCode()&Ability.ALL_ACODES)==(pA.classificationCode()&Ability.ALL_ACODES))
144                                                reqSet.add(pA.ID());
145                                }
146                        }
147                        if(reqSet.size()==0)
148                        {
149                                final boolean aIsSkill=isSkill(A.classificationCode());
150                                for(final String prevID : prevSet)
151                                {
152                                        final Ability pA=CMClass.getAbility(prevID);
153                                        if(aIsSkill==isSkill(pA.classificationCode()))
154                                                reqSet.add(pA.ID());
155                                }
156                        }
157                        if(reqSet.size()>0)
158                                return reqSet;
159                }
160                return new Vector<String>();
161        }

163        @Override
164        public void startCharacter(MOB mob, boolean isBorrowedClass, boolean verifyOnly)
165        {
166                if(!loaded())
167                {
168                        setLoaded(true);
169                        final LinkedList<CharClass> charClassesOrder=new LinkedList<CharClass>();
170                        final HashSet<String> names=new HashSet<String>();
171                        for(final Enumeration<CharClass> c=CMClass.charClasses();c.hasMoreElements();)
172                        {
173                                final CharClass C=c.nextElement();
174                                if(C.baseClass().equals(C.ID()) 
175                                && (!C.baseClass().equalsIgnoreCase("Archon"))
176                                && (!C.baseClass().equalsIgnoreCase("PlayerClass"))
177                                && (!C.baseClass().equalsIgnoreCase("Qualifier"))
178                                && (!C.baseClass().equalsIgnoreCase("StdCharClass")))
179                                {
180                                        names.add(C.ID());
181                                        charClassesOrder.add(C);
182                                }
183                        }
184                        for(final Enumeration<CharClass> c=CMClass.charClasses();c.hasMoreElements();)
185                        {
186                                final CharClass C=c.nextElement();
187                                if(!names.contains(C.ID()) && names.contains(C.baseClass()))
188                                        charClassesOrder.add(C);
189                        }
190                        for(final Enumeration<CharClass> c=CMClass.charClasses();c.hasMoreElements();)
191                        {
192                                final CharClass C=c.nextElement();
193                                if(C.baseClass().equals("Commoner") && (!names.contains(C.ID())))
194                                        charClassesOrder.add(C);
195                        }
196
197                        for(final CharClass C : charClassesOrder)
198                        {
199                                final LinkedList<List<String>> prevSets=new LinkedList<List<String>>();
200                                for(int lvl=1;lvl<CMProps.getIntVar(CMProps.Int.LASTPLAYERLEVEL);lvl++)
201                                {
202                                        final List<String> curSet=CMLib.ableMapper().getLevelListings(C.ID(), false, lvl);
203                                        for(final String ID : curSet)
204                                        {
205                                                final String defaultParam=CMLib.ableMapper().getDefaultParm(C.ID(), true, ID);
206                                                if(CMLib.ableMapper().getQualifyingLevel(ID(), false, ID)<0)
207                                                {
208                                                        final Ability A=CMClass.getAbility(ID);
209                                                        if(A==null)
210                                                        {
211                                                                Log.errOut("Unknonwn class: "+ID);
212                                                                continue;
213                                                        }
214                                                        List<String> reqSet=makeRequirements(prevSets,A);
215                                                        int level=0;
216                                                        if(!this.leveless() && (!CMSecurity.isDisabled(DisFlag.LEVELS)))
217                                                                level=CMLib.ableMapper().lowestQualifyingLevel(A.ID());
218                                                        if(level<0)
219                                                                level=0;
220                                                        CMLib.ableMapper().addCharAbilityMapping(ID(), 0, ID, 0, defaultParam, false, false, reqSet, "");
221                                                }
222                                        }
223                                        if(curSet.size()>0)
224                                                prevSets.add(curSet);
225                                }
226                        }
227                }
228                super.startCharacter(mob, false, verifyOnly);
229        }

231        @Override
232        public void grantAbilities(MOB mob, boolean isBorrowedClass)
233        {
234                super.grantAbilities(mob,isBorrowedClass);
235                if(mob.playerStats()==null)
236                {
237                        final List<AbilityMapper.AbilityMapping> V=CMLib.ableMapper().getUpToLevelListings(ID(),
238                                                                                                mob.charStats().getClassLevel(ID()),
239                                                                                                false,
240                                                                                                false);
241                        for(final AbilityMapper.AbilityMapping able : V)
242                        {
243                                final Ability A=CMClass.getAbility(able.abilityID());
244                                if((A!=null)
245                                &&(!CMLib.ableMapper().getAllQualified(ID(),true,A.ID()))
246                                &&(!CMLib.ableMapper().getDefaultGain(ID(),true,A.ID())))
247                                        giveMobAbility(mob,A,CMLib.ableMapper().getDefaultProficiency(ID(),true,A.ID()),CMLib.ableMapper().getDefaultParm(ID(),true,A.ID()),isBorrowedClass);
248                        }
249                }
250        }
