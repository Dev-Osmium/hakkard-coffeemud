225        @Override
226        public boolean okMessage(final Environmental myHost, final CMMsg msg)
227        {
228                // no xp from combat
229                if((msg.sourceMinor()==CMMsg.TYP_EXPCHANGE)
230                &&(msg.source()==myHost)
231                &&(msg.source().charStats().getCurrentClass()==this)
232                &&(msg.target() instanceof MOB)
233                &&(((MOB)msg.target()).amDead()||(((MOB)msg.target()).curState().getHitPoints()<=0))
234                &&(msg.value()>0))
235                {
236                        msg.setValue(0);
237                }
238                return super.okMessage(myHost,msg);
239        }

266        public static void visitationBonusMessage(Environmental host, CMMsg msg)
267        {
268                if((msg.target() instanceof Room)
269                &&(msg.source()==host)
270                &&(!msg.source().isMonster())
271                &&(msg.targetMinor()==CMMsg.TYP_ENTER)
272                &&(msg.source().playerStats()!=null))
273                {
274                        final Room R=(Room)msg.target();
275                        if(((R.roomID().length()>0)
276                                ||((R.getGridParent()!=null)&&(R.getGridParent().roomID().length()>0)))
277                        &&(!CMath.bset(R.getArea().flags(),Area.FLAG_INSTANCE_CHILD))
278                        &&(!msg.source().playerStats().hasVisited(R))
279                        )
280                        {
281                                MOB M=null;
282                                boolean bookDealer=false;
283                                for(int m=0;m<R.numInhabitants();m++)
284                                {
285                                        M=R.fetchInhabitant(m);
286                                        if((M instanceof ShopKeeper)
287                                        &&(M.getStartRoom()==R))
288                                        {
289                                                if((((ShopKeeper)M).isSold(ShopKeeper.DEAL_BOOKS))
290                                                ||(((ShopKeeper)M).isSold(ShopKeeper.DEAL_READABLES)))
291                                                {
292                                                        bookDealer=true;
293                                                }
294                                        }
295                                }
296                                if(bookDealer)
297                                {
298                                        final int xpGain=50;
299                                        if(CMLib.leveler().postExperience((MOB)host,null,null,xpGain,true))
300                                                msg.addTrailerMsg(CMClass.getMsg((MOB)host,null,null,CMMsg.MSG_OK_VISUAL,CMLib.lang().L("^HYou have discovered a new place of books and gain @x1 experience.^?",""+xpGain),CMMsg.NO_EFFECT,null,CMMsg.NO_EFFECT,null));
301                                }
302                        }
303                }
304        }

306        @Override
307        public void executeMsg(final Environmental myHost, final CMMsg msg)
308        {
309                if((!(myHost instanceof MOB))
310                ||(((MOB)myHost).charStats().getCurrentClass()!=this))
311                {
312                        super.executeMsg(myHost, msg);
313                        return;
314                }
315
316                if(msg.source()==myHost)
317                {
318                        Scholar.visitationBonusMessage(myHost,msg);
319                        
320                        if((msg.targetMinor()==CMMsg.TYP_TEACH)
321                        &&(msg.target() instanceof MOB))
322                                CMLib.leveler().postExperience(msg.source(), null, null, 100, false);
323                        else
324                        if(((msg.targetMinor()==CMMsg.TYP_WRITE)
325                                ||(msg.targetMinor()==CMMsg.TYP_REWRITE)
326                                ||(msg.targetMinor()==CMMsg.TYP_WROTE))
327                        &&(msg.target() instanceof Item)
328                        &&(msg.targetMessage()!=null)
329                        &&(msg.targetMessage().length()>0))
330                        {
331                                if((msg.tool() instanceof Ability)
332                                &&(msg.targetMinor()==CMMsg.TYP_WROTE)
333                                &&(msg.tool().ID().equals("Skill_Map")
334                                        ||msg.tool().ID().equals("Thief_TreasureMap")
335                                        ||msg.tool().ID().equals("Skill_SeaMapping")))
336                                {
337                                        final Map<String,Object> persMap = Resources.getPersonalMap(msg.source(), true);
338                                        if(persMap != null)
339                                        {
340                                                int xp = 10;
341                                                long[] xpTrap = (long[])persMap.get("SCHOLAR_MAPXP");
342                                                if(xpTrap == null)
343                                                {
344                                                        xpTrap = new long[2];
345                                                        persMap.put("SCHOLAR_MAPXP", xpTrap);
346                                                }
347                                                if(System.currentTimeMillis() > xpTrap[1])
348                                                {
349                                                        xpTrap[0]=0;
350                                                        final Room R=msg.source().getStartRoom();
351                                                        if(R!=null)
352                                                        {
353                                                                final Area A=R.getArea();
354                                                                if((A!=null)&&(A.getTimeObj()!=null))
355                                                                        xpTrap[1]=System.currentTimeMillis() + (CMProps.getMillisPerMudHour() * A.getTimeObj().getHoursInDay());
356                                                                else
357                                                                        xpTrap[1]=System.currentTimeMillis() + (TimeManager.MILI_MINUTE * 10);
358                                                        }
359                                                        else
360                                                                xpTrap[1]=System.currentTimeMillis() + (TimeManager.MILI_MINUTE * 10);
361                                                }
362                                                final long maxLevel = msg.source().getExpNeededLevel() / 10;
363                                                if(xpTrap[0] < maxLevel)
364                                                {
365                                                        xpTrap[0]+=xp;
366                                                        CMLib.leveler().postExperience(msg.source(), null, null, xp, false);
367                                                }
368                                        }
369                                        CMLib.leveler().postExperience(msg.source(), null, null, 10, false);
370                                }
371                                else
372                                if((msg.tool() instanceof Ability)
373                                &&(msg.targetMinor()==CMMsg.TYP_WROTE)
374                                &&(msg.tool().ID().equals("Dissertating")))
375                                {
376                                        CMLib.leveler().postExperience(msg.source(), null, null, 25, false);
377                                }
378                                else
379                                {
380                                        final String msgStr =msg.targetMessage().trim();
381                                        int numChars = msgStr.length()-CMStrings.countChars(msgStr, ' ');
382                                        if(numChars > 10)
383                                        {
384                                                final Map<String,Object> persMap = Resources.getPersonalMap(msg.source(), true);
385                                                if(persMap != null)
386                                                {
387                                                        int xp = numChars/10;
388                                                        long[] xpTrap = (long[])persMap.get("SCHOLAR_WRITEXP");
389                                                        if(xpTrap == null)
390                                                        {
391                                                                xpTrap = new long[2];
392                                                                persMap.put("SCHOLAR_WRITEXP", xpTrap);
393                                                        }
394                                                        if(System.currentTimeMillis() > xpTrap[1])
395                                                        {
396                                                                xpTrap[0]=0;
397                                                                xpTrap[1]=System.currentTimeMillis() + (TimeManager.MILI_MINUTE * 10);
398                                                        }
399                                                        final long maxLevel = msg.source().getExpNeededLevel() / 24;
400                                                        if(xpTrap[0] < maxLevel)
401                                                        {
402                                                                if(100-xpTrap[0]<xp)
403                                                                        xp=(int)(100-xpTrap[0]);
404                                                                xpTrap[0]+=xp;
405                                                                CMLib.leveler().postExperience(msg.source(), null, null, xp, false);
406                                                        }
407                                                }
408                                        }
409                                }
410                        }
411                }
412                if((msg.tool() instanceof Ability)
413                &&(myHost instanceof MOB))
414                {
415                        Ability A;
416                        if((myHost == msg.source())
417                        &&((A=msg.source().fetchAbility(msg.tool().ID()))!=null)
418                        &&(A.isSavable())
419                        &&(msg.source().isPlayer())
420                        &&(CMParms.contains(FAV_DOMAINS, ((Ability)msg.tool()).classificationCode()&Ability.ALL_DOMAINS)))
421                        {
422                                final Map<String,Object> persMap = msg.source().playerStats().getClassVariableMap(this);
423                                if(persMap != null)
424                                {
425                                        final String key = "LAST_DATE_FOR_"+A.ID().toUpperCase().trim();
426                                        long[] lastTime = (long[])persMap.get(key);
427                                        if(lastTime == null)
428                                        {
429                                                lastTime = new long[1];
430                                                persMap.put(key, lastTime);
431                                        }
432                                        final Area homeA=CMLib.map().areaLocation(msg.source().getStartRoom());
433                                        final TimeClock homeL = (homeA == null) ? null : homeA.getTimeObj();
434                                        if((homeL!=null)
435                                        &&((homeL.toHoursSinceEpoc() - lastTime[0])>homeL.getHoursInDay()))
436                                        {
437                                                lastTime[0] = homeL.toHoursSinceEpoc(); 
438                                                CMLib.leveler().postExperience(msg.source(), null, null, 25, false);
439                                        }
440                                }
441                        }
442                        
443                        if((((MOB)myHost).getVictim()!=msg.source())
444                        &&(msg.source().getVictim()!=myHost)
445                        &&(CMLib.dice().rollPercentage()<25)
446                        &&((A=msg.source().fetchAbility(msg.tool().ID()))!=null)
447                        &&(((MOB)myHost).fetchAbility(msg.tool().ID())!=null)
448                        &&(((MOB)myHost).getGroupMembers(new TreeSet<MOB>()).contains(msg.source())))
449                        {
450                                final Ability A1=(Ability)msg.tool();
451                                if((A1!=null)&&(A1.isSavable()))
452                                        A1.helpProficiency(msg.source(), 0);
453                        }
454                }
455                super.executeMsg(myHost, msg);
456        }
