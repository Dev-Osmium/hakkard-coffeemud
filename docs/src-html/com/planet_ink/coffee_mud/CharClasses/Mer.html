254        @Override
255        public boolean okMessage(final Environmental myHost, final CMMsg msg)
256        {
257                if(!(myHost instanceof MOB))
258                        return super.okMessage(myHost,msg);
259                final MOB myChar=(MOB)myHost;
260                if(!super.okMessage(myChar, msg))
261                        return false;
262
263                if(msg.amISource(myChar)
264                &&(!myChar.isMonster())
265                &&(msg.sourceMinor()==CMMsg.TYP_CAST_SPELL)
266                &&(msg.tool() instanceof Ability)
267                &&((((Ability)msg.tool()).classificationCode()&Ability.ALL_ACODES)==Ability.ACODE_CHANT)
268                &&(myChar.isMine(msg.tool()))
269                &&(isQualifyingAuthority(myChar,(Ability)msg.tool()))
270                &&(CMLib.dice().rollPercentage()<50))
271                {
272                        if(((Ability)msg.tool()).appropriateToMyFactions(myChar))
273                                return true;
274                        myChar.tell(L("Extreme emotions disrupt your chant."));
275                        return false;
276                }
277                return true;
278        }

323                RawMaterial.RESOURCE_SALTWATER,
324                RawMaterial.RESOURCE_FRESHWATER
325        };

327        private final Map<int[],int[]> oldSets = new Hashtable<int[],int[]>(); 
328        
329        @Override
330        public void affectCharStats(final MOB affected, final CharStats affectableStats)
331        {
332                super.affectCharStats(affected, affectableStats);
333                // this is necessary because the race happens AFTER the breathe is modified,
334                // so we're actually adding water breathing to baseStat race (human, whatever)
335                affected.eachEffect(new EachApplicable<Ability>()
336                {
337                        @Override
338                        public void apply(Ability a)
339                        {
340                                if(a.ID().startsWith("Druid_ShapeShift"))
341                                        a.affectCharStats(affected, affectableStats);
342                        }
343                });
344                
345                if(!CMParms.contains(affectableStats.getBreathables(),breathableStuff))
346                {
347                        final int[] newSet = oldSets.get(affectableStats.getBreathables());
348                        if(newSet != null) 
349                        {
350                                affectableStats.setBreathables(newSet);
351                        }
352                        else
353                        {
354                                int[] newerSet=Arrays.copyOf(affectableStats.getBreathables(),affectableStats.getBreathables().length+breathableStuff.length);
355                                for(int i=breathableStuff.length-1;i>=0;i--)
356                                        newerSet[newerSet.length-(i+1)]=breathableStuff[i];
357                                if(oldSets.size()>100)
358                                        oldSets.clear();
359                                oldSets.put(affectableStats.getBreathables(), newerSet);
360                                Arrays.sort(newerSet);
361                                affectableStats.setBreathables(newerSet);
362                        }
363                }
364        }

366        @Override
367        public void affectCharState(final MOB affected, final CharState affectableState)
368        {
369                super.affectCharState(affected,affectableState);
370                if(affected.location()!=null)
371                {
372                        final Room room=affected.location();
373                        if(affected.charStats().getClassLevel(this)>=5)
374                        {
375                                final TidePhase phase = room.getArea().getTimeObj().getTidePhase(room);
376                                affectableState.setMovement(affectableState.getMovement()
377                                                                                        +(int)Math.round(CMath.mul(CMath.div(affectableState.getMovement(),8.0),phase.getFactor())));
378                                affectableState.setHitPoints(affectableState.getHitPoints()
379                                                                                        +(int)Math.round(CMath.mul(CMath.div(affectableState.getHitPoints(),8.0),phase.getFactor())));
380                                affectableState.setMana(affectableState.getMana()
381                                                                                        -(int)Math.round(CMath.mul(CMath.div(affectableState.getMana(),4.0),phase.getFactor())));
382                        }
383                }
384        }

386        @Override
387        public void affectPhyStats(final Physical affected, final PhyStats affectableStats)
388        {
389                super.affectPhyStats(affected,affectableStats);
390                if((affected instanceof MOB)&&(((MOB)affected).location()!=null))
391                {
392                        final MOB mob=(MOB)affected;
393                        final Room room=mob.location();
394                        final int classLevel=mob.charStats().getClassLevel(this);
395                        if(classLevel>=5)
396                        {
397                                final TidePhase phase = room.getArea().getTimeObj().getTidePhase(room);
398                                affectableStats.setArmor(affectableStats.armor() // - is good
399                                                                                 -(int)Math.round(CMath.mul(classLevel,phase.getFactor())));
400                                affectableStats.setAttackAdjustment(affectableStats.attackAdjustment() // - is bad
401                                                                                 -(int)Math.round(CMath.mul(classLevel,phase.getFactor())));
402                        }
403                }
404        }

406        @Override
407        public void grantAbilities(MOB mob, boolean isBorrowedClass)
408        {
409                super.grantAbilities(mob,isBorrowedClass);
410                if(mob.playerStats()==null)
411                {
412                        final List<AbilityMapper.AbilityMapping> V=CMLib.ableMapper().getUpToLevelListings(ID(),
413                                                                                                mob.charStats().getClassLevel(ID()),
414                                                                                                false,
415                                                                                                false);
416                        for(final AbilityMapper.AbilityMapping able : V)
417                        {
418                                final Ability A=CMClass.getAbility(able.abilityID());
419                                if((A!=null)
420                                &&((A.classificationCode()&Ability.ALL_ACODES)==Ability.ACODE_CHANT)
421                                &&(!CMLib.ableMapper().getDefaultGain(ID(),true,A.ID())))
422                                        giveMobAbility(mob,A,CMLib.ableMapper().getDefaultProficiency(ID(),true,A.ID()),CMLib.ableMapper().getDefaultParm(ID(),true,A.ID()),isBorrowedClass);
423                        }
424                }
425        }

441        @Override
442        public int classDurationModifier(MOB myChar,
443                                                                         Ability skill,
444                                                                         int duration)
445        {
446                if(myChar==null)
447                        return duration;
448                if((((skill.classificationCode()&Ability.ALL_DOMAINS)==Ability.DOMAIN_CRAFTINGSKILL)
449                        ||((skill.classificationCode()&Ability.ALL_DOMAINS)==Ability.DOMAIN_BUILDINGSKILL))
450                &&(myChar.charStats().getCurrentClass().ID().equals(ID()))
451                &&(!skill.ID().equals("FoodPrep"))
452                &&(!skill.ID().equals("Cooking"))
453                &&(!skill.ID().equals("Herbalism"))
454                &&(!skill.ID().equals("Masonry"))
455                &&(!skill.ID().equals("Landscaping")))
456                        return duration*2;
457
458                return duration;
459        }
