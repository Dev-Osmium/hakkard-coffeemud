439        @Override
440        public void grantAbilities(MOB mob, boolean isBorrowedClass)
441        {
442                super.grantAbilities(mob,isBorrowedClass);
443                if(!grantSomeSpells())
444                        return;
445
446                final Vector<String> grantable=new Vector<String>();
447
448                final int level=mob.charStats().getClassLevel(this);
449                int numSpells=3;
450                if(level<8)
451                        numSpells=3;
452                else
453                if(level<19)
454                        numSpells=2;
455                else
456                        numSpells=1;
457                for(final Enumeration<Ability> a=CMClass.abilities();a.hasMoreElements();)
458                {
459                        final Ability A=a.nextElement();
460                        if((CMLib.ableMapper().getQualifyingLevel(ID(),true,A.ID())==level)
461                        &&((CMLib.ableMapper().getQualifyingLevel(ID(),true,A.ID())<=25)
462                        &&(!CMLib.ableMapper().getSecretSkill(ID(),true,A.ID()))
463                        &&(!CMLib.ableMapper().getDefaultGain(ID(),true,A.ID()))
464                        &&((A.classificationCode()&Ability.ALL_ACODES)==Ability.ACODE_SPELL)))
465                        {
466                                if (!grantable.contains(A.ID()))
467                                        grantable.addElement(A.ID());
468                        }
469                }
470                for(int a=0;a<mob.numAbilities();a++)
471                {
472                        final Ability A=mob.fetchAbility(a);
473                        if(grantable.contains(A.ID()))
474                        {
475                                grantable.remove(A.ID());
476                                numSpells--;
477                        }
478                }
479                for(int i=0;i<numSpells;i++)
480                {
481                        if(grantable.size()==0)
482                                break;
483                        final String AID=grantable.elementAt(CMLib.dice().roll(1,grantable.size(),-1));
484                        if(AID!=null)
485                        {
486                                grantable.removeElement(AID);
487                                giveMobAbility(mob,
488                                                           CMClass.getAbility(AID),
489                                                           CMLib.ableMapper().getDefaultProficiency(ID(),true,AID),
490                                                           CMLib.ableMapper().getDefaultParm(ID(),true,AID),
491                                                           isBorrowedClass);
492                        }
493                }
494        }
