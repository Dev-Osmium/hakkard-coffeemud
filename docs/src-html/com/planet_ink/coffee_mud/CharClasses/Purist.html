241        @Override
242        public boolean okMessage(final Environmental myHost, final CMMsg msg)
243        {
244                if(!(myHost instanceof MOB))
245                        return super.okMessage(myHost,msg);
246                final MOB myChar=(MOB)myHost;
247                if(!super.okMessage(myChar, msg))
248                        return false;
249
250                if(msg.amISource(myChar)
251                &&(!myChar.isMonster())
252                &&(msg.sourceMinor()==CMMsg.TYP_CAST_SPELL)
253                &&(msg.tool() instanceof Ability)
254                &&((((Ability)msg.tool()).classificationCode()&Ability.ALL_ACODES)==Ability.ACODE_PRAYER)
255                &&(myChar.isMine(msg.tool()))
256                &&(isQualifyingAuthority(myChar,(Ability)msg.tool())))
257                {
258                        final int alignment = myChar.fetchFaction(CMLib.factions().AlignID());
259                        final int pct = CMLib.factions().getPercent(CMLib.factions().AlignID(), alignment);
260                        if(pct < 50)
261                        {
262                                myChar.tell(L("Your impurity disrupts the prayer."));
263                                return false;
264                        }
265                        final int hq=holyQuality((Ability)msg.tool());
266                        if(hq==0)
267                        {
268                                myChar.tell(L("You most certainly should not be casting that."));
269                                return false;
270                        }
271                }
272
273                if((msg.amITarget(myChar))
274                &&(msg.targetMinor()==CMMsg.TYP_DAMAGE)
275                &&((msg.sourceMinor()==CMMsg.TYP_COLD)
276                        ||(msg.sourceMinor()==CMMsg.TYP_WATER)))
277                {
278                        final int recovery=myChar.charStats().getClassLevel(this);
279                        msg.setValue(msg.value()-recovery);
280                }
281                else
282                if((msg.amITarget(myChar))
283                &&(msg.targetMinor()==CMMsg.TYP_DAMAGE)
284                &&(msg.sourceMinor()==CMMsg.TYP_FIRE))
285                {
286                        final int recovery=msg.value();
287                        msg.setValue(msg.value()+recovery);
288                }
289                else
290                if(msg.amISource(myChar)
291                &&(!msg.amITarget(myChar))
292                &&(msg.tool() instanceof Ability)
293                &&(msg.targetMinor()==CMMsg.TYP_DAMAGE)
294                &&((((Ability)msg.tool()).classificationCode()&Ability.ALL_ACODES)==Ability.ACODE_PRAYER)
295                &&(myChar.isMine(msg.tool()))
296                &&(isQualifyingAuthority(myChar,(Ability)msg.tool()))
297                &&(holyQuality((Ability)msg.tool())==1000))
298                        msg.setValue(msg.value()*2);
299                return true;
300        }

302        @Override
303        public List<Item> outfit(MOB myChar)
304        {
305                if(outfitChoices==null)
306                {
307                        final Weapon w=CMClass.getWeapon("SmallMace");
308                        if(w == null)
309                                return new Vector<Item>();
310                        outfitChoices=new Vector<Item>();
311                        outfitChoices.add(w);
312                }
313                return outfitChoices;
314        }
