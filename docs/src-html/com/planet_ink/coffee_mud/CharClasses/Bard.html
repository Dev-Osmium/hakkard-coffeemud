199        public static int bardAdjustExperienceGain(MOB host, MOB mob, MOB victim, int amount, double rate)
200        {
201                double theAmount=amount;
202                if((mob!=null)&&(victim!=null)&&(theAmount>10.0))
203                {
204                        if(host == mob)
205                        {
206                                final Set<MOB> H=mob.getGroupMembers(new HashSet<MOB>());
207                                final double origAmount=theAmount;
208                                for (final Object element : H)
209                                {
210                                        final MOB mob2=(MOB)element;
211                                        if((mob2!=mob)
212                                        &&(mob2!=victim)
213                                        &&(mob2.location()!=null)
214                                        &&(mob2.location()==mob.location()))
215                                        {
216                                                if(!mob2.isMonster())
217                                                        theAmount+=(origAmount/rate);
218                                                else
219                                                if(!CMLib.flags().isAnimalIntelligence(mob2))
220                                                        theAmount+=1.0;
221                                        }
222                                }
223                        }
224                        else
225                        if((!host.isMonster())&&(!mob.isMonster()))
226                                theAmount = 1.1 * theAmount;
227                }
228                return (int)Math.round(theAmount);
229        }

244        public static void visitationBonusMessage(Environmental host, CMMsg msg)
245        {
246                if((msg.target() instanceof Room)
247                &&(msg.source()==host)
248                &&(!msg.source().isMonster())
249                &&(msg.targetMinor()==CMMsg.TYP_ENTER)
250                &&(msg.source().playerStats()!=null))
251                {
252                        final Room R=(Room)msg.target();
253                        final MOB mob=msg.source();
254                        final Physical hostP=(Physical)host;
255                        if(((R.roomID().length()>0)
256                                ||((R.getGridParent()!=null)&&(R.getGridParent().roomID().length()>0)))
257                        &&(!CMath.bset(R.getArea().flags(),Area.FLAG_INSTANCE_CHILD))
258                        &&(!msg.source().playerStats().hasVisited(R))
259                        )
260                        {
261                                final Area A=R.getArea();
262                                MOB M=null;
263                                boolean pub=false;
264                                for(int m=0;m<R.numInhabitants();m++)
265                                {
266                                        M=R.fetchInhabitant(m);
267                                        if((M instanceof ShopKeeper)
268                                        &&(M.getStartRoom()==R))
269                                        {
270                                                final List<Ability> V2=new Vector<Ability>();
271                                                for(final Iterator<Environmental> i=((ShopKeeper)M).getShop().getStoreInventory();i.hasNext();)
272                                                {
273                                                        final Environmental O=i.next();
274                                                        if(O instanceof Potion)
275                                                        {
276                                                                V2.addAll(((Potion)O).getSpells());
277                                                                for(int v=V2.size()-1;v>=0;v--)
278                                                                {
279                                                                        if((V2.get(v).classificationCode()&Ability.ALL_ACODES)!=Ability.ACODE_POISON)
280                                                                                V2.get(v);
281                                                                }
282                                                        }
283                                                        if((O instanceof Drink)
284                                                        &&(O instanceof Item))
285                                                        {
286                                                                V2.addAll(CMLib.flags().domainAffects((Item)O,Ability.ACODE_POISON));
287                                                                final String name=" "+O.Name().toLowerCase();
288                                                                if(name.endsWith(" beer")
289                                                                ||name.endsWith(" liquor")
290                                                                ||name.endsWith(" ale")
291                                                                ||name.endsWith(" whiskey")
292                                                                ||name.endsWith(" wine"))
293                                                                        pub=true;
294                                                        }
295                                                        for(int v=0;v<V2.size();v++)
296                                                                pub=pub||CMath.bset(V2.get(v).flags(),Ability.FLAG_INTOXICATING);
297                                                }
298                                        }
299                                }
300                                if(pub)
301                                {
302                                        final int xpGain=50;
303                                        if(CMLib.leveler().postExperience((MOB)host,null,null,xpGain,true))
304                                                msg.addTrailerMsg(CMClass.getMsg((MOB)host,null,null,CMMsg.MSG_OK_VISUAL,CMLib.lang().L("^HYou have discovered a new pub, you gain @x1 experience.^?",""+xpGain),CMMsg.NO_EFFECT,null,CMMsg.NO_EFFECT,null));
305                                }
306                                if((!mob.playerStats().hasVisited(A))&&(mob.soulMate()==null))
307                                {
308                                        if(mob.playerStats().addRoomVisit(R))
309                                        {
310                                                CMLib.players().bumpPrideStat(mob,AccountStats.PrideStat.ROOMS_EXPLORED,1);
311                                                int xp=(int)Math.round(100.0*CMath.div(A.getAreaIStats()[Area.Stats.AVG_LEVEL.ordinal()],hostP.phyStats().level()));
312                                                if(xp>250)
313                                                        xp=250;
314                                                if((xp>0)&&CMLib.leveler().postExperience((MOB)host,null,null,xp,true))
315                                                        msg.addTrailerMsg(CMClass.getMsg((MOB)host,null,null,CMMsg.MSG_OK_VISUAL,CMLib.lang().L("^HYou have discovered '@x1', you gain @x2 experience.^?",A.name(),""+xp),CMMsg.NO_EFFECT,null,CMMsg.NO_EFFECT,null));
316                                        }
317                                }
318                                else
319                                {
320                                        final int pctBefore=mob.playerStats().percentVisited((MOB)host,A);
321                                        if(mob.playerStats().addRoomVisit(R))
322                                        {
323                                                CMLib.players().bumpPrideStat(mob,AccountStats.PrideStat.ROOMS_EXPLORED,1);
324                                                final int pctAfter=mob.playerStats().percentVisited((MOB)host,A);
325                                                if((pctBefore<50)&&(pctAfter>=50))
326                                                {
327                                                        int xp=(int)Math.round(50.0*CMath.div(A.getAreaIStats()[Area.Stats.AVG_LEVEL.ordinal()],hostP.phyStats().level()));
328                                                        if(xp>125)
329                                                                xp=125;
330                                                        if((xp>0)&&CMLib.leveler().postExperience((MOB)host,null,null,xp,true))
331                                                                msg.addTrailerMsg(CMClass.getMsg((MOB)host,null,null,CMMsg.MSG_OK_VISUAL,CMLib.lang().L("^HYou have familiarized yourself with '@x1', you gain @x2 experience.^?",A.name(),""+xp),CMMsg.NO_EFFECT,null,CMMsg.NO_EFFECT,null));
332                                                }
333                                                else
334                                                if((pctBefore<90)&&(pctAfter>=90))
335                                                {
336                                                        int xp=(int)Math.round(100.0*CMath.div(A.getAreaIStats()[Area.Stats.AVG_LEVEL.ordinal()],hostP.phyStats().level()));
337                                                        if(xp>250)
338                                                                xp=250;
339                                                        if((xp>0)&&CMLib.leveler().postExperience((MOB)host,null,null,xp,true))
340                                                                msg.addTrailerMsg(CMClass.getMsg((MOB)host,null,null,CMMsg.MSG_OK_VISUAL,CMLib.lang().L("^HYou have explored '@x1', you gain @x2 experience.^?",A.name(),""+xp),CMMsg.NO_EFFECT,null,CMMsg.NO_EFFECT,null));
341                                                }
342                                        }
343                                }
344                        }
345                }
346        }

348        @Override
349        public void grantAbilities(MOB mob, boolean isBorrowedClass)
350        {
351                super.grantAbilities(mob,isBorrowedClass);
352                if(mob.playerStats()==null)
353                {
354                        final List<AbilityMapper.AbilityMapping> V=CMLib.ableMapper().getUpToLevelListings(ID(),
355                                                                                                                         mob.charStats().getClassLevel(ID()),
356                                                                                                                         false,
357                                                                                                                         false);
358                        for(final AbilityMapper.AbilityMapping able : V)
359                        {
360                                final Ability A=CMClass.getAbility(able.abilityID());
361                                if((A!=null)
362                                &&((A.classificationCode()&Ability.ALL_ACODES)==Ability.ACODE_SONG)
363                                &&(!CMLib.ableMapper().getDefaultGain(ID(),true,A.ID())))
364                                        giveMobAbility(mob,A,CMLib.ableMapper().getDefaultProficiency(ID(),true,A.ID()),CMLib.ableMapper().getDefaultParm(ID(),true,A.ID()),isBorrowedClass);
365                        }
366                }
367        }
