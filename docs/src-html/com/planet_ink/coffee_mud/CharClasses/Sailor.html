260        @Override
261        public boolean tick(final Tickable ticking, final int tickID)
262        {
263                if((tickID==Tickable.TICKID_MOB)&&(ticking instanceof MOB))
264                {
265                        final MOB mob=(MOB)ticking;
266                        if(mob.charStats().getCurrentClass().ID().equals(ID()))
267                        {
268                                int exp=0;
269                                for(final Enumeration<Ability> a=mob.effects();a.hasMoreElements();)
270                                {
271                                        final Ability A=a.nextElement();
272                                        if((A!=null)
273                                        &&(!A.isAutoInvoked())
274                                        &&(mob.isMine(A))
275                                        &&((A.classificationCode()&Ability.ALL_ACODES)==Ability.ACODE_COMMON_SKILL))
276                                        {
277                                                exp++;
278                                                if((A.ID().equalsIgnoreCase("Trawling"))//||(A.ID().equalsIgnoreCase("Fishing"))))
279                                                ||(A.ID().equalsIgnoreCase("MasterTrawling")))//||(A.ID().equalsIgnoreCase("MasterFishing"))))
280                                                        exp++;
281                                        }
282                                }
283                                if(exp>0)
284                                        CMLib.leveler().postExperience(mob,null,null,exp,true);
285                        }
286                }
287                return super.tick(ticking,tickID);
288        }

290        protected void giveExploreXP(final MOB mob, final Room R, int amt)
291        {
292                if(((R.roomID().length()>0)
293                        ||((R.getGridParent()!=null)&&(R.getGridParent().roomID().length()>0)))
294                &&(!CMath.bset(R.getArea().flags(),Area.FLAG_INSTANCE_CHILD))
295                &&(!mob.playerStats().hasVisited(R))
296                &&(mob.soulMate()==null)
297                &&(CMLib.flags().isWaterySurfaceRoom(R))
298                )
299                {
300                        if(mob.playerStats().addRoomVisit(R))
301                        {
302                                CMLib.players().bumpPrideStat(mob,AccountStats.PrideStat.ROOMS_EXPLORED,1);
303                                if(mob.playerStats().hasVisited(R))
304                                {
305                                        CMLib.leveler().postExperience(mob, null, null, amt, false);
306                                }
307                        }
308                }
309        }

311        @Override
312        public void executeMsg(final Environmental myHost, final CMMsg msg)
313        {
314                super.executeMsg(myHost, msg);
315                
316                if((msg.target() instanceof Room)&&(msg.targetMinor()==CMMsg.TYP_ENTER))
317                {
318                        if((msg.source()==myHost)
319                        &&(msg.source().riding() !=null)
320                        &&(!msg.source().isMonster())
321                        &&(msg.source().playerStats()!=null)
322                        &&(msg.source().riding().rideBasis() == Rideable.RIDEABLE_WATER))
323                                giveExploreXP(msg.source(), (Room)msg.target(), 5);
324                        if((msg.source().riding() instanceof SailingShip)
325                        &&(msg.source().Name().equals(msg.source().riding().Name()))
326                        &&(myHost instanceof MOB)
327                        &&(((MOB)myHost).playerStats()!=null)
328                        &&(((MOB)myHost).location()!=null)
329                        &&(((MOB)myHost).location().getArea() == ((BoardableShip)msg.source().riding()).getShipArea()))
330                                giveExploreXP((MOB)myHost, (Room)msg.target(), 10);
331                }
332                
333                if((msg.sourceMinor()==CMMsg.TYP_EXPCHANGE)
334                &&(msg.value()>0)
335                &&(msg.source().charStats().getCurrentClass() == this)
336                &&(CMLib.map().areaLocation(msg.source()) instanceof BoardableShip)
337                &&(msg.source() != msg.target()))
338                {
339                        if(msg.target() instanceof MOB)
340                                msg.setValue(msg.value() * 2);
341                        else
342                        if(msg.target() == null)
343                        {
344                                BoardableShip shipArea = (BoardableShip)CMLib.map().areaLocation(msg.source());
345                                Room R=CMLib.map().roomLocation(shipArea.getShipItem());
346                                if(R!=null)
347                                {
348                                        for(final Enumeration<Item> i=R.items();i.hasMoreElements();)
349                                        {
350                                                Item I=i.nextElement();
351                                                if((I instanceof BoardableShip)
352                                                &&(I.fetchEffect("Sinking")!=null)
353                                                &&(I!=shipArea.getShipItem()))
354                                                {
355                                                        msg.setValue(msg.value() * 2);
356                                                        break;
357                                                }
358                                        }
359                                }
360                        }
361                }
362        }

384        @Override
385        public List<Item> outfit(MOB myChar)
386        {
387                if(outfitChoices==null)
388                {
389                        outfitChoices=new Vector<Item>();
390                        final Weapon w=CMClass.getWeapon("GenWeapon");
391                        if(w != null)
392                        {
393                                w.setName("a belaying pin");
394                                w.setDisplayText("a belaying pin lies here.");
395                                w.setMaterial(RawMaterial.RESOURCE_WOOD);
396                                w.setWeaponClassification(Weapon.CLASS_BLUNT);
397                                w.setWeaponDamageType(Weapon.TYPE_BASHING);
398                                w.setBaseValue(0);
399                                w.basePhyStats().setDamage(4);
400                                w.basePhyStats().setWeight(1);
401                                w.basePhyStats().setAttackAdjustment(0);
402                                w.recoverPhyStats();
403                                outfitChoices.add(w);
404                        }
405                }
406                return outfitChoices;
407        }
