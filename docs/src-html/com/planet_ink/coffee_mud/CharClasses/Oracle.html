241        protected int numNonQualified(MOB mob)
242        {
243                int numNonQualified=0;
244                for(int a=0;a<mob.numAbilities();a++)
245                {
246                        final Ability A=mob.fetchAbility(a);
247                        boolean qualifies=false;
248                        for(int c=0;c<mob.charStats().numClasses();c++)
249                        {
250                                final CharClass C=mob.charStats().getMyClass(c);
251                                if(CMLib.ableMapper().getQualifyingLevel(C.ID(),true,A.ID())>0)
252                                        qualifies=true;
253                        }
254                        if((!qualifies)
255                        &&(CMLib.ableMapper().getQualifyingLevel(mob.baseCharStats().getMyRace().ID(),true,A.ID())<0)
256                        &&(CMLib.ableMapper().qualifiesByAnyCharClass(A.ID())))
257                                numNonQualified++;
258                }
259                return numNonQualified;
260        }
261
262        protected int maxNonQualified(MOB mob)
263        {
264                int level=mob.charStats().getClassLevel(this)-30;
265                level++;
266                return level;
267        }

269        @Override
270        public void grantAbilities(MOB mob, boolean isBorrowedClass)
271        {
272                super.grantAbilities(mob,isBorrowedClass);
273
274                // if he already has one, don't give another!
275                if((mob.playerStats()!=null)
276                &&(mob.charStats().getClassLevel(this)>=30))
277                {
278                        if(numNonQualified(mob)>=maxNonQualified(mob))
279                                return;
280
281                        Ability newOne=null;
282                        int tries=0;
283                        while((newOne==null)&&((++tries)<100))
284                        {
285                                final CharClass C=CMClass.randomCharClass();
286                                if((C!=null)
287                                &&(!C.ID().equals(ID()))
288                                &&(!C.ID().equalsIgnoreCase("Archon"))
289                                &&(mob.charStats().getClassLevel(C)<0))
290                                {
291                                        int tries2=0;
292                                        while((newOne==null)&&((++tries2)<10000))
293                                        {
294                                                final Ability A=CMClass.randomAbility();
295                                                if( A != null )
296                                                {
297                                                        final int lql=CMLib.ableMapper().lowestQualifyingLevel(A.ID());
298                                                        if((lql<25)
299                                                        &&(lql>0))
300                                                        {
301                                                                if((!CMLib.ableMapper().getSecretSkill(C.ID(),true,A.ID()))
302                                                                &&(CMLib.ableMapper().getQualifyingLevel(ID(),true,A.ID())<0)
303                                                                &&(CMLib.ableMapper().availableToTheme(A.ID(),Area.THEME_FANTASY,true))
304                                                                &&(CMLib.ableMapper().qualifiesByAnyCharClass(A.ID()))
305                                                                &&(A.isAutoInvoked()||((A.triggerStrings()!=null)&&(A.triggerStrings().length>0)))
306                                                                &&(mob.fetchAbility(A.ID())==null))
307                                                                {
308                                                                        final DVector prereqs=CMLib.ableMapper().getUnmetPreRequisites(mob,A);
309                                                                        if((prereqs==null)||(prereqs.size()==0))
310                                                                                newOne=A;
311                                                                }
312                                                        }
313                                                }
314                                        }
315                                }
316                        }
317                        if(newOne!=null)
318                                mob.addAbility(newOne);
319                }
320        }

322        @Override
323        public List<Item> outfit(MOB myChar)
324        {
325                if(outfitChoices==null)
326                {
327                        final Weapon w=CMClass.getWeapon("SmallMace");
328                        if(w == null)
329                                return new Vector<Item>();
330                        outfitChoices=new Vector<Item>();
331                        outfitChoices.add(w);
332                }
333                return outfitChoices;
334        }

336        @Override
337        public void level(MOB mob, List<String> newAbilityIDs)
338        {
339                if(CMSecurity.isDisabled(CMSecurity.DisFlag.LEVELS))
340                        return;
341                if((!mob.isMonster())&&(mob.charStats().getClassLevel(this)>=30))
342                {
343                        if((newAbilityIDs.size()==0)&&(numNonQualified(mob)>=maxNonQualified(mob)))
344                                mob.tell(L("^NYou have learned no new secrets this level, as you already know ^H@x1/@x2^? secret skills.^N",""+numNonQualified(mob),""+maxNonQualified(mob)));
345                }
346        }
