233        @Override
234        public void grantAbilities(MOB mob, boolean isBorrowedClass)
235        {
236                super.grantAbilities(mob,isBorrowedClass);
237                if(mob.playerStats()==null)
238                {
239                        final List<AbilityMapper.AbilityMapping> V=CMLib.ableMapper().getUpToLevelListings(ID(),
240                                                                                                mob.charStats().getClassLevel(ID()),
241                                                                                                false,
242                                                                                                false);
243                        for(final AbilityMapper.AbilityMapping able : V)
244                        {
245                                final Ability A=CMClass.getAbility(able.abilityID());
246                                if((A!=null)
247                                &&(!CMLib.ableMapper().getAllQualified(ID(),true,A.ID()))
248                                &&(!CMLib.ableMapper().getDefaultGain(ID(),true,A.ID())))
249                                        giveMobAbility(mob,A,CMLib.ableMapper().getDefaultProficiency(ID(),true,A.ID()),CMLib.ableMapper().getDefaultParm(ID(),true,A.ID()),isBorrowedClass);
250                        }
251                }
252        }

274        @Override
275        public boolean okMessage(final Environmental myHost, final CMMsg msg)
276        {
277                if(!(myHost instanceof MOB))
278                        return super.okMessage(myHost,msg);
279                final MOB myChar=(MOB)myHost;
280                if((msg.amISource(myChar))
281                &&(msg.sourceMinor()==CMMsg.TYP_CAST_SPELL)
282                &&(!CMLib.flags().isGood(myChar))
283                &&((msg.tool()==null)||((CMLib.ableMapper().getQualifyingLevel(ID(),true,msg.tool().ID())>0)
284                                                                &&(myChar.isMine(msg.tool()))))
285                &&(CMLib.dice().rollPercentage()>myChar.charStats().getStat(CharStats.STAT_WISDOM)*2))
286                {
287                        myChar.location().show(myChar,null,CMMsg.MSG_OK_VISUAL,L("<S-NAME> watch(es) <S-HIS-HER> angry god absorb <S-HIS-HER> magical energy!"));
288                        return false;
289                }
290                return super.okMessage(myChar, msg);
291        }
