222        @Override
223        public boolean tick(final Tickable ticking, final int tickID)
224        {
225                if((tickID==Tickable.TICKID_MOB)&&(ticking instanceof MOB))
226                {
227                        final MOB mob=(MOB)ticking;
228                        if(mob.charStats().getCurrentClass().ID().equals(ID()))
229                        {
230                                int exp=0;
231                                for(final Enumeration<Ability> a=mob.effects();a.hasMoreElements();)
232                                {
233                                        final Ability A=a.nextElement();
234                                        if((A!=null)
235                                        &&(!A.isAutoInvoked())
236                                        &&(mob.isMine(A))
237                                        &&((A.classificationCode()&Ability.ALL_ACODES)==Ability.ACODE_COMMON_SKILL))
238                                                exp++;
239                                }
240                                if(exp>0)
241                                        CMLib.leveler().postExperience(mob,null,null,exp,true);
242
243                                if((CMLib.flags().isSleeping(mob))
244                                &&(mob.playerStats()!=null)
245                                &&(CMLib.dice().roll(1, 10, 0)==1))
246                                {
247                                        final Room R=mob.location();
248                                        final LegalBehavior legalBehavior=CMLib.law().getLegalBehavior(R);
249                                        final Area legalArea=CMLib.law().getLegalObject(R);
250                                        if(this.isRightOutsideAnOccupiedCell(R, legalBehavior, legalArea))
251                                                CMLib.leveler().postExperience(mob,null,null,5,false);
252                                }
253                        }
254                }
255                return super.tick(ticking,tickID);
256        }

258        public boolean isRightOutsideACell(final Room R, final LegalBehavior legalBehavior, final Area legalArea)
259        {
260                if(R==null)
261                        return false;
262                final List<Room> rooms=new ArrayList<Room>(3);
263                for(int d=0;d<Directions.NUM_DIRECTIONS();d++)
264                {
265                        final Room R2=R.getRoomInDir(d);
266                        final Exit E2=R.getExitInDir(d);
267                        if((R2!=null)
268                        &&(E2!=null))
269                                rooms.add(R2);
270                }
271                return legalBehavior.isJailRoom(legalArea, rooms);
272        }
273
274        public boolean isRightOutsideAnOccupiedCell(final Room R, final LegalBehavior legalBehavior, final Area legalArea)
275        {
276                if(R==null)
277                        return false;
278                final List<Room> rooms=new ArrayList<Room>(3);
279                for(int d=0;d<Directions.NUM_DIRECTIONS();d++)
280                {
281                        final Room R2=R.getRoomInDir(d);
282                        final Exit E2=R.getExitInDir(d);
283                        if((R2!=null)
284                        &&(E2!=null)
285                        &&(R2.numInhabitants()>0))
286                                rooms.add(R2);
287                }
288                if(rooms.size()==0)
289                        return false;
290                return legalBehavior.isJailRoom(legalArea, rooms);
291        }

293        protected boolean isAnOfficerOfTheCourt(final MOB mob, final LegalBehavior legalBehavior, final Area legalArea)
294        {
295                if(mob.isMonster())
296                {
297                        if(legalBehavior.isAnyOfficer(legalArea, mob)
298                        ||legalBehavior.isJudge(legalArea, mob))
299                                return true;
300                        final Room R=mob.location();
301                        if(R!=null)
302                        {
303                                if((mob.getStartRoom()==R)
304                                &&(isRightOutsideACell(R,legalBehavior,legalArea)))
305                                        return true;
306                        }
307                }
308                return false;
309        }

311        public boolean isInACell(final MOB M)
312        {
313                if(M==null)
314                        return false;
315                final Room R=M.location();
316                if(R==null)
317                        return false;
318                final LegalBehavior legalBehavior=CMLib.law().getLegalBehavior(R);
319                final Area legalArea=CMLib.law().getLegalObject(R);
320                if((legalBehavior != null)
321                &&(legalArea != null)
322                &&(legalBehavior.isJailRoom(legalArea, new XVector<Room>(R)))
323                &&(legalBehavior.hasWarrant(legalArea, M)))
324                        return true;
325                return false;
326        }

328        @Override
329        public boolean okMessage(final Environmental host, final CMMsg msg)
330        {
331                if(!super.okMessage(host, msg))
332                        return false;
333                if(((msg.targetMajor()&CMMsg.MASK_MALICIOUS)>0)
334                &&(msg.amITarget(host))
335                &&(msg.source().isMonster())
336                &&(host instanceof MOB)
337                &&(((MOB)host).charStats().getCurrentClass().ID().equals(ID()))
338                &&(!((MOB)host).isInCombat())
339                &&(msg.source().getVictim()!=host))
340                {
341                        final LegalBehavior legalBehavior=CMLib.law().getLegalBehavior(msg.source().location());
342                        final Area legalArea=CMLib.law().getLegalObject(msg.source().location());
343                        if((legalBehavior != null)
344                        &&(isAnOfficerOfTheCourt(msg.source(),legalBehavior, legalArea))
345                        &&((!legalBehavior.hasWarrant(legalArea, (MOB)host))))
346                        {
347                                msg.source().tell(L("You may not assault this fellow officer of the court."));
348                                final MOB mob=(MOB)host;
349                                if(mob.getVictim()==msg.source())
350                                {
351                                        mob.makePeace(true);
352                                        mob.setVictim(null);
353                                }
354                                return false;
355                        }
356                }
357                return true;
358        }

360        @Override
361        public void executeMsg(final Environmental host, final CMMsg msg)
362        {
363                if((msg.source()==host)
364                &&(msg.target() instanceof MOB)
365                &&(msg.target()!=msg.source())
366                &&(((MOB)host).charStats().getCurrentClass().ID().equals(ID())))
367                {
368                        if((msg.tool() instanceof Ability)
369                        &&(CMath.bset(((Ability)msg.tool()).flags(), Ability.FLAG_TORTURING))
370                        &&((MOB)host).isMine(msg.tool())
371                        &&(CMLib.map().getStartArea(host)!=null)
372                        &&(msg.sourceMinor()!=CMMsg.TYP_TEACH)
373                        &&(((MOB)host).charStats().getClassLevel(this)>0))
374                        {
375                                final CMMsg msg2=CMClass.getMsg((MOB)msg.target(),null,null,CMMsg.MSG_NOISE,L("<S-NAME> scream(s) in agony, AAAAAAARRRRGGGHHH!!@x1",CMLib.protocol().msp("scream.wav",40)));
376                                if(((MOB)msg.target()).location().okMessage(msg.target(),msg2))
377                                {
378                                        final MOB mob=(MOB)host;
379                                        final int baseAmt = 10 + CMLib.ableMapper().qualifyingLevel(msg.source(), (Ability)msg.tool());
380                                        final int xp=(int)Math.round(baseAmt*CMath.div(((MOB)msg.target()).phyStats().level(),((MOB)host).charStats().getClassLevel(this)));
381                                        @SuppressWarnings("unchecked")
382                                        Map<String, int[]> mudHourMOBXPMap = (Map<String, int[]>)((mob.playerStats()==null)?null:mob.playerStats().getClassVariableMap(this).get("MUDHOURMOBXPMAP"));
383                                        if(mudHourMOBXPMap == null)
384                                        {
385                                                mudHourMOBXPMap = new Hashtable<String, int[]>();
386                                                if(mob.playerStats() != null)
387                                                        mob.playerStats().getClassVariableMap(this).put("MUDHOURMOBXPMAP",mudHourMOBXPMap);
388                                        }
389                                        int[] done=mudHourMOBXPMap.get(host.Name()+"/"+msg.tool().ID());
390                                        if (done == null)
391                                        {
392                                                done = new int[3];
393                                                mudHourMOBXPMap.put(host.Name() + "/" + msg.tool().ID(), done);
394                                        }
395                                        if(Calendar.getInstance().get(Calendar.SECOND)!=done[2])
396                                        {
397                                                final TimeClock clock =CMLib.map().getStartArea(host).getTimeObj();
398                                                if(done[0]!=clock.getHourOfDay())
399                                                        done[1]=0;
400                                                done[0]=clock.getHourOfDay();
401                                                done[2]=Calendar.getInstance().get(Calendar.SECOND);
402
403                                                if(done[1]<(90+(50*((MOB)host).phyStats().level())))
404                                                {
405                                                        done[1]+=xp;
406                                                        CMLib.leveler().postExperience((MOB)host,null,null,xp,true);
407                                                        msg2.addTrailerMsg(CMClass.getMsg((MOB)host,null,null,CMMsg.MSG_OK_VISUAL,L("The sweet screams of your victim earns you @x1 experience points.",""+xp),CMMsg.NO_EFFECT,null,CMMsg.NO_EFFECT,null));
408                                                }
409                                                else
410                                                        msg2.addTrailerMsg(CMClass.getMsg((MOB)host,null,null,CMMsg.MSG_OK_VISUAL,L("The screams of this victim bore you now."),CMMsg.NO_EFFECT,null,CMMsg.NO_EFFECT,null));
411                                                msg.addTrailerMsg(msg2);
412                                        }
413                                }
414                        }
415                        else
416                        if((msg.sourceMinor()==CMMsg.TYP_GIVE)
417                        &&(msg.tool() instanceof Food)
418                        &&(msg.target() instanceof MOB)
419                        &&(isInACell((MOB)msg.target())))
420                        {
421                                boolean qualifyingFood=false;
422                                if(((Food)msg.tool()).fetchEffect("Poison_Rotten")!=null)
423                                        qualifyingFood=true;
424                                else
425                                {
426                                        final ItemCraftor foodChkA=(ItemCraftor)msg.source().fetchAbility("GaolFood");
427                                        if(foodChkA != null)
428                                        {
429                                                final List<List<String>> V = foodChkA.fetchMyRecipes(msg.source());
430                                                for(final List<String> item : V)
431                                                {
432                                                        if(item.get(0).toLowerCase().indexOf(foodChkA.name().toLowerCase())>0)
433                                                                qualifyingFood=true;
434                                                }
435                                        }
436                                }
437                                if(qualifyingFood)
438                                {
439                                        if(msg.source().playerStats()!=null)
440                                        {
441                                                @SuppressWarnings("unchecked")
442                                                Map<String,Long> map=(Map<String,Long>)msg.source().playerStats().getClassVariableMap(this).get("GAOLER_FEEDS");
443                                                if(map == null)
444                                                {
445                                                        map=new TreeMap<String,Long>();
446                                                        msg.source().playerStats().getClassVariableMap(this).put("GAOLER_FEEDS", map);
447                                                }
448                                                for(final Iterator<String> i=map.keySet().iterator();i.hasNext();)
449                                                {
450                                                        final Long L=map.get(i.next());
451                                                        if(System.currentTimeMillis() > L.longValue())
452                                                                i.remove();
453                                                }
454                                                if(map.containsKey(""+msg.target()))
455                                                        qualifyingFood=false;
456                                                else
457                                                {
458                                                        TimeClock C=CMLib.time().localClock(msg.source());
459                                                        if(C==null)
460                                                                C=CMLib.time().globalClock();
461                                                        map.put(""+msg.target(), new Long(System.currentTimeMillis() + (CMProps.getMillisPerMudHour() * C.getHoursInDay())));
462                                                }
463                                        }
464                                        if(qualifyingFood)
465                                                CMLib.leveler().postExperience((MOB)host,null,null,msg.source().phyStats().level(),false);
466                                }
467                        }
468                }
469        }
