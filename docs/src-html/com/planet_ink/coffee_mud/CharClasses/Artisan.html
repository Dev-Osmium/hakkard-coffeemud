275        @Override
276        public void executeMsg(final Environmental myHost, final CMMsg msg)
277        {
278                if((msg.source() == myHost)
279                &&(msg.targetMinor() == CMMsg.TYP_ITEMGENERATED)
280                &&(msg.target() != null)
281                &&(msg.tool() instanceof Ability)
282                &&(msg.value() > 0)
283                &&(((((Ability)msg.tool()).classificationCode() & Ability.ALL_DOMAINS) == Ability.DOMAIN_CRAFTINGSKILL)
284                        ||((((Ability)msg.tool()).classificationCode() & Ability.ALL_DOMAINS) == Ability.DOMAIN_EPICUREAN)
285                        ||((((Ability)msg.tool()).classificationCode() & Ability.ALL_DOMAINS) == Ability.DOMAIN_BUILDINGSKILL)))
286                {
287                                CMLib.leveler().postExperience(msg.source(),null,null,msg.value(),false);
288                }
289                else
290                if((msg.source() == myHost)
291                &&(msg.targetMinor() == CMMsg.TYP_RECIPELEARNED)
292                &&(msg.target() != null)
293                &&(msg.tool() instanceof Ability)
294                &&((((Ability)msg.tool()).classificationCode() & Ability.ALL_DOMAINS) == Ability.DOMAIN_CRAFTINGSKILL)
295                &&(msg.value() > 0))
296                        CMLib.leveler().postExperience(msg.source(),null,null,msg.value(),false);
297        }

299        @Override
300        public boolean tick(final Tickable ticking, final int tickID)
301        {
302                if((tickID==Tickable.TICKID_MOB)&&(ticking instanceof MOB))
303                {
304                        final MOB mob=(MOB)ticking;
305                        if(ID().equals(mob.charStats().getCurrentClass().ID())&&(CMLib.dice().rollPercentage()<20))
306                        {
307                                int exp=0;
308                                for(final Enumeration<Ability> a=mob.effects();a.hasMoreElements();)
309                                {
310                                        final Ability A=a.nextElement();
311                                        if((A!=null)
312                                        &&(!A.isAutoInvoked())
313                                        &&((A.classificationCode()&Ability.ALL_ACODES)==Ability.ACODE_COMMON_SKILL)
314                                        &&(mob.isMine(A)))
315                                                exp++;
316                                }
317                                if(exp>0)
318                                {
319                                        exp=exp/2;
320                                        if(exp<1)
321                                                exp=1;
322                                        CMLib.leveler().postExperience(mob,null,null,exp,false);
323                                }
324                        }
325                }
326                return super.tick(ticking,tickID);
327        }
