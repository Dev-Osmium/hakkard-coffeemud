358        @Override
359        public boolean qualifiesForThisClass(MOB mob, boolean quiet)
360        {
361                if(CMSecurity.isDisabled(CMSecurity.DisFlag.STDCLASSES) && (!isGeneric()))
362                        return false;
363
364                final String multiClassRule=CMProps.getVar(CMProps.Str.MULTICLASS);
365                String multiClassFirstRule=multiClassRule;
366                String multiClassSecondRule="";
367                final int x=multiClassRule.indexOf('-');
368                if(x>0)
369                {
370                        multiClassFirstRule=multiClassRule.substring(0, x);
371                        multiClassSecondRule=multiClassRule.substring(x+1);
372                }
373
374                final String changeToBaseClassID=baseClass();
375                final String changeToClassID=ID();
376                final SubClassRule changeToSubClassRule = getSubClassRule();
377
378                String canOnlyBeClassID="";
379                String canOnlyBeBaseClassID="";
380                if(multiClassSecondRule.length()>0)
381                {
382                        if(multiClassSecondRule.startsWith("GRP-"))
383                        {
384                                final CharClass possibleClass=CMClass.findCharClass(multiClassSecondRule.substring(4));
385                                if(possibleClass != null)
386                                        canOnlyBeBaseClassID=possibleClass.ID();
387                                multiClassSecondRule="NO";
388                        }
389                        else
390                        {
391                                final CharClass possibleClass=CMClass.findCharClass(multiClassSecondRule);
392                                if(possibleClass != null)
393                                        canOnlyBeClassID=possibleClass.ID();
394                                multiClassSecondRule="NO";
395                        }
396                }
397
398                if(mob == null)
399                {
400                        if(multiClassFirstRule.equals("SUB")||multiClassSecondRule.equals("SUB"))
401                        {
402                                if((changeToBaseClassID.equals(changeToClassID))||(changeToSubClassRule==SubClassRule.ANY))
403                                        return true;
404                                return false;
405                        }
406                        else
407                                return true;
408                }
409
410                final CharClass curClass = mob.baseCharStats().getCurrentClass();
411                final String currentClassID=curClass.ID();
412                final String currentBaseClassID=curClass.baseClass();
413
414                for(final Pair<String,Integer> minReq : getMinimumStatRequirements())
415                {
416                        final int statCode=CharStats.CODES.findWhole(minReq.first, true);
417                        if(statCode >= 0)
418                        {
419                                final CharStats cStats = (CharStats)mob.baseCharStats().copyOf();
420                                cStats.getMyRace().affectCharStats(mob, cStats);
421                                if(cStats.getStat(statCode) < minReq.second.intValue())
422                                {
423                                        if(!quiet)
424                                                mob.tell(L("You need at least a @x1 @x2 to become a @x3.",minReq.second.toString(),CMStrings.capitalizeAndLower(CharStats.CODES.NAME(statCode)),name()));
425                                        return false;
426                                }
427                        }
428                }
429                final Race R=mob.baseCharStats().getMyRace();
430                if(!isAllowedRace(R))
431                {
432                        if(!quiet)
433                        {
434                                mob.tell(L("You need to be a @x1 to be a @x2.",getRaceQualDesc(),name()));
435                        }
436                        return false;
437                }
438                if((!mob.isMonster())&&(mob.basePhyStats().level()>0))
439                {
440                        if(currentClassID.equals(changeToClassID))
441                        {
442                                if(!quiet)
443                                        mob.tell(L("But you are already a @x1!",name()));
444                                return false;
445                        }
446                        if(currentClassID.equalsIgnoreCase("StdCharClass")) // this is the starting character rule
447                        {
448                                if((canOnlyBeClassID.length()>0)&&(!changeToClassID.equals(canOnlyBeClassID)))
449                                        return false;
450                                if((canOnlyBeBaseClassID.length()>0)&&(!changeToBaseClassID.equals(canOnlyBeBaseClassID)))
451                                        return false;
452                                if((multiClassRule.equals("NO"))||(multiClassRule.equals("MULTI")))
453                                        return true;
454                                if((multiClassRule.equals("SUB")||multiClassSecondRule.equals("BASE"))
455                                &&(changeToBaseClassID.equals(changeToClassID)||(changeToSubClassRule==SubClassRule.ANY)))
456                                        return true;
457                                if((multiClassSecondRule.equals("SUBONLY"))
458                                &&((!changeToBaseClassID.equals(changeToClassID))||(changeToSubClassRule==SubClassRule.ANY)))
459                                        return true;
460                                if(multiClassFirstRule.equals("APP")&&(getSubClassRule()==SubClassRule.ANY))
461                                        return true;
462                                if(!quiet)
463                                        mob.tell(L("You can't train to be a @x1!",name()));
464                                return false;
465                        }
466                        else
467                        if(curClass.getSubClassRule()==SubClassRule.NONE)
468                        {
469                                if(!quiet)
470                                        mob.tell(L("You can't train to be a @x1!",name()));
471                                return false;
472                        }
473                        else
474                        if(curClass.getSubClassRule()==SubClassRule.ANY) // if you are an apprentice
475                        {
476                                if(multiClassFirstRule.equals("NO")
477                                ||multiClassSecondRule.equals("NO")
478                                ||multiClassFirstRule.equals("MULTI")
479                                ||multiClassSecondRule.equals("MULTI"))
480                                        return true;
481                                if(multiClassFirstRule.equals("SUB")||multiClassSecondRule.equals("SUB"))
482                                {
483                                        if((changeToBaseClassID.equals(changeToClassID))||(changeToBaseClassID.equals(currentBaseClassID)))
484                                                return true;
485                                        if(!quiet)
486                                                mob.tell(L("You must be a @x1 type to become a @x2.",changeToBaseClassID,name()));
487                                }
488                                return false;
489                        }
490                        else
491                        {
492                                if(multiClassFirstRule.equals("MULTI")||multiClassSecondRule.equals("MULTI"))
493                                        return true;
494                                else
495                                if(multiClassFirstRule.equals("NO")||(multiClassSecondRule.equals("NO")))
496                                {
497                                        if(!quiet)
498                                                mob.tell(L("You should be happy to be a @x1.",curClass.name()));
499                                }
500                                else
501                                if(multiClassFirstRule.equals("SUB")|| multiClassSecondRule.equals("SUB"))
502                                {
503                                        if(currentBaseClassID.equals(changeToBaseClassID)||(curClass.getSubClassRule()==SubClassRule.ANY))
504                                                return true;
505                                        boolean doesBaseHaveAnAny=false;
506                                        for(final Enumeration<CharClass> c=CMClass.charClasses();c.hasMoreElements();)
507                                        {
508                                                final CharClass C=c.nextElement();
509                                                if((C.baseClass().equals(currentBaseClassID))
510                                                &&(C.getSubClassRule()==SubClassRule.ANY))
511                                                {
512                                                        doesBaseHaveAnAny=true;
513                                                        break;
514                                                }
515                                        }
516                                        if(doesBaseHaveAnAny)
517                                        {
518                                                if((changeToBaseClassID.equals(changeToClassID))||(changeToBaseClassID.equals(currentBaseClassID)))
519                                                        return true;
520                                        }
521                                        if(!quiet)
522                                                mob.tell(L("You must be a @x1 type to become a @x2.",changeToBaseClassID,name()));
523                                }
524                        }
525                        return false;
526                }
527                return true;
528        }

554        @Override
555        public String getStatQualDesc()
556        {
557                final Pair<String,Integer>[] reqs=getMinimumStatRequirements();
558                if(reqs.length==0)
559                        return "None";
560                final StringBuilder str=new StringBuilder("");
561                for(int x=0;x<reqs.length;x++)
562                {
563                        final Pair<String,Integer> req=reqs[x];
564                        if(x>0)
565                                str.append(", ");
566                        str.append(CMStrings.capitalizeAndLower(req.first)).append(" ").append(req.second.toString()).append("+");
567                }
568                return str.toString();
569        }

571        @Override
572        public String getRaceQualDesc()
573        {
574                this.checkRaceQualifierChanges();
575                if(this.cachedRaceQualList != null)
576                        return this.cachedRaceQualList;
577
578                final XVector<Race> availRaces = new XVector<Race>();
579                final CharCreationLibrary loginLib = CMLib.login();
580                if((!CMClass.races().hasMoreElements())||(CMLib.login()==null))
581                        return "None";
582                for(final Enumeration<Race> r=CMClass.races();r.hasMoreElements();)
583                {
584                        final Race R=r.nextElement();
585                        if(loginLib.isAvailableRace(R))
586                                availRaces.add(R);
587                }
588                if(availRaces.size()==0)
589                {
590                        this.cachedRaceQualList=L("All");
591                }
592                else
593                {
594                        final XVector<Race> qualRaces = new XVector<Race>();
595                        for(Race R : availRaces)
596                        {
597                                if(isAllowedRace(R))
598                                        qualRaces.add(R);
599                        }
600                        if(qualRaces.size()==availRaces.size())
601                        {
602                                this.cachedRaceQualList=L("All");
603                        }
604                        else
605                        {
606                                final StringBuilder str=new StringBuilder();
607                                ArrayList<String> names = new ArrayList<String>(availRaces.size());
608                                //if(qualRaces.size()<=(availRaces.size()/2))
609                                {
610                                        for(Race R : qualRaces)
611                                                names.add(R.name());
612                                        str.append(CMLib.english().toEnglishStringList(names.toArray(new String[0])));
613                                }
614                                /*
615                                else
616                                {
617                                        str.append(L("All except "));
618                                        for(Race R : availRaces)
619                                        {
620                                                if(!qualRaces.contains(R))
621                                                        names.add(R.name());
622                                        }
623                                        str.append(CMLib.english().toEnglishStringList(names.toArray(new String[0])));
624                                }
625                                */
626                                this.cachedRaceQualList=str.toString();
627                        }
628                }
629                return this.cachedRaceQualList;
630        }

675                        String formula=getHitPointsFormula();
676                        final int x=formula.indexOf("*(1?");
677                        if(x>0)
678                        {
679                                final int y=formula.indexOf(')',x+1);
680                                if(y>x)
681                                        formula=formula.substring(0, x)+"d"+formula.substring(x+4,y)+formula.substring(y+1);
682                        }
683                        formula=CMStrings.replaceAlls(formula, hitPointDescReplacePairs);
684                        hitPointsDesc=CMProps.getIntVar(CMProps.Int.STARTHP)+" +["+formula+" /lvl]";
685                }
686                return hitPointsDesc;
687        }

826                if((((sourceCode&CMMsg.MINOR_MASK)==CMMsg.TYP_WEAPONATTACK)||((sourceCode&CMMsg.MINOR_MASK)==CMMsg.TYP_THROW))

835                                        return (filterResult == DoubleFilterer.Result.ALLOWED) ? true : false;
836                                }
837                        }
838                        if((((requiredWeaponMaterials()!=null)&&(!requiredWeaponMaterials().contains(Integer.valueOf(((Weapon)E).material()&RawMaterial.MATERIAL_MASK))))
839                                ||((disallowedWeaponClasses(mob)!=null)&&(disallowedWeaponClasses(mob).contains(Integer.valueOf(((Weapon)E).weaponClassification())))))
840                        &&(CMLib.dice().rollPercentage()>(mob.charStats().getStat(getAttackAttribute())*2))
841                        &&(mob.fetchWieldedItem()!=null))
842                        {
843                                mob.location().show(mob,null,CMMsg.MSG_OK_ACTION,L("<S-NAME> fumble(s) horribly with @x1.",E.name()));
844                                return false;
845                        }
846                }
847                return true;

883        @Override
884        public void grantAbilities(MOB mob, boolean isBorrowedClass)
885        {
886                if(CMSecurity.isAllowedEverywhere(mob,CMSecurity.SecFlag.ALLSKILLS)

891                        final Map<String,Ability> alreadyAff=new HashMap<String,Ability>();
892                        for(final Enumeration<Ability> a=mob.effects();a.hasMoreElements();)
893                        {
894                                final Ability A=a.nextElement();

907                                                A2.setProficiency(CMLib.ableMapper().getMaxProficiency(mob,true,A.ID()));
908                                        else
909                                                A.autoInvocation(mob, false);
910                                        alreadyAble.put(A.ID(),A);
911                                }
912                        }
913                        for(final Enumeration<Ability> a=CMClass.abilities();a.hasMoreElements();)
914                        {
915                                final Ability A=a.nextElement();
916                                final int lvl=CMLib.ableMapper().lowestQualifyingLevel(A.ID());
917                                if((lvl>=0)
918                                &&(CMLib.ableMapper().qualifiesByAnyCharClass(A.ID()))

920                                        giveMobAbility(mob,A,100,"",true,false);

922                        for(final Enumeration e=CMLib.expertises().definitions();e.hasMoreElements();)
923                                mob.addExpertise(((ExpertiseLibrary.ExpertiseDefinition)e.nextElement()).ID());
924                        alreadyAble.clear();
925                        alreadyAff.clear();
926                }
927                else
928                {
929                        final PairList<Ability,AbilityMapping> onesToAdd=new PairVector<Ability,AbilityMapping>();
930                        for(final Enumeration<Ability> a=CMClass.abilities();a.hasMoreElements();)
931                        {
932                                final Ability A=a.nextElement();
933                                final AbilityMapping mapping = CMLib.ableMapper().getQualifyingMapping(ID(), true, A.ID());
934                                if((mapping != null)

936                                &&(mapping.qualLevel()<=mob.baseCharStats().getClassLevel(this))

941                                        ||(extraMask.length()==0)
942                                        ||(CMLib.masking().maskCheck(extraMask,mob,true)))
943                                                onesToAdd.add(A,mapping);
944                                }
945                        }
946                        for(int v=0;v<onesToAdd.size();v++)
947                        {
948                                final Ability A=onesToAdd.get(v).first;
949                                final AbilityMapping map=onesToAdd.get(v).second;

951                        }
952                }

1321                                break;

1323                        case CMMsg.TYP_WEAR:
1324                        case CMMsg.TYP_HOLD:

1326                                if(!CMLib.utensils().armorCheck(msg.source(),(Item)msg.target(),allowedArmorLevel()))
