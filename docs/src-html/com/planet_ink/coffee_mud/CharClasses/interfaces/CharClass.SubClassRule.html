034/**
035 * This class represents a player or mobs character class.  One of more of these
036 * objects are associated with each mob through the mob interfaces charStats() object.
037 * @see com.planet_ink.coffee_mud.MOBS.interfaces.MOB#charStats()
038 * @see com.planet_ink.coffee_mud.Common.interfaces.CharStats
039 * @author Bo Zimmerman
040 */
041public interface CharClass extends Tickable, StatsAffecting, MsgListener, CMObject, Modifiable
042{
043        /**
044         * Returns the generally displayable name of this class.  Usually deferred to
045         * by name(int), which is more often called.
046         * @see CharClass#name(int)
047         * @see CharClass#nameSet()
048         */
049        @Override
050        public String name();
051
052        /**
053         * Returns the displayable name of this class, when the mob is the
054         * given class level.  Usually defers to name()
055         * @see CharClass#name()
056         * @see CharClass#nameSet()
057         * @param classLevel the level to look up a name for
058         * @return the displayable name of this class
059         */
060        public String name(int classLevel);
061
062        /**
063         * Returns all of the displayable names of this class.  Usually defers to name()
064         * @see CharClass#name()
065         * @see CharClass#name(int)
066         * @return all of the displayable names of this class
067         */
068        public String[] nameSet();
069
070        /**
071         * Returns the base-class of this class.  Typically only important in multi-classing
072         * systems that restrict class changing to those classes part of the same base class.
073         * True multi-classing systems don't need to worry about this value.  Can be the same
074         * as the ID() method.
075         * @return the base-class of this class
076         */
077        public String baseClass();
078
079        /**
080         * Returns one or a combination of the Area.THEME_*
081         * constants from the Area interface.  This bitmap
082         * then describes the types of areas, skills, and
083         * classes which can interact.
084         * This bitmap is also used to to tell whether
085         * the class is available for selection by users
086         * at char creation time, whether they can
087         * change to this class via spells, or whether
088         * the class is utterly unavailable to them.
089         * @see com.planet_ink.coffee_mud.Areas.interfaces.Area
090         * @return the availability/theme of this class
091         */
092        public int availabilityCode();
093
094        /**
095         * Returns whether this given mob qualifies for this class, and
096         * optionally gives them an error message. There are strange rules
097         * here.  If the mob is null, then this method returns true unless
098         * subclassing is on, and this class is not a subclass.  If the
099         * mob has a class of StdCharClass, then char creation rules apply,
100         * otherwise train rules apply.
101         * @param mob the mob to evaluate the worthiness of
102         * @param quiet false to give the mob error messages, true for silence
103         * @return whether the given mob is worthy of this class
104         */
105        public boolean qualifiesForThisClass(MOB mob, boolean quiet);
106
107        /**
108         * Returns any boot-time parameters that are required to fully define
109         * this instance of a charclass.  Charclasses are shared among mobs,
110         * but that doesn't mean mulitple instances of a single class can't
111         * generate different versions of themselves for the classloader, with
112         * different parameters.  GenCharClass is an example.
113         * @see CharClass#setClassParms(String)
114         * @see CharClass#isGeneric()
115         * @return any parameters used to define this class
116         */
117        public String classParms();
118
119        /**
120         * Sets any boot-time parameters that are required to fully define
121         * this instance of a charclass.  Charclasses are shared among mobs,
122         * but that doesn't mean mulitple instances of a single class can't
123         * generate different versions of themselves for the classloader, with
124         * different parameters.  GenCharClass is an example.
125         * @see CharClass#classParms()
126         * @see CharClass#isGeneric()
127         * @param parms any parameters used to define this class
128         */
129        public void setClassParms(String parms);
130
131        /**
132         * Returns whether this class is fully defined using the setParms
133         * method, as opposed to being defined by its Java code.
134         * @see CharClass#classParms()
135         * @see CharClass#makeGenCharClass()
136         * @see CharClass#setClassParms(String)
137         * @return whether this class is defined fully by parameters
138         */
139        public boolean isGeneric();
140
141        /**
142         * Converts this class into a generic one, if it is not already.
143         * If it is generic, this method returns itself.  Otherwise, the
144         * standard char class is converted to a generic one and returned.
145         * @see CharClass#isGeneric()
146         * @return a generic version of this class.
147         */
148        public CharClass makeGenCharClass();
149        /**
150         * Returns a read only set of security flags granted to all mobs/players
151         * who are this class, and the given class level or lower.
152         * @param classLevel the class level of the mob
153         * @return a vector of security flags
154         */
155        public CMSecurity.SecGroup getSecurityFlags(int classLevel);
156
157        /**
158         * This method should be called whenever a mob has this class added to
159         * their charStats list. Its purpose is to outfit the mob with any
160         * necessary abilities, or perform other necessary changes to the mob
161         * to reflect the class addition or change.  A character class is
162         * considered borrowed if its existence is derived from something else,
163         * or its skills/abilities should not be saved as a permanent feature
164         * of the mob.
165         * @param mob the mob being outfitted with this class
166         * @param isBorrowedClass whether the charclasses skills are borrowed(true) or permanent
167         * @param verifyOnly send true if no skills or changes are to be made
168         */
169        public void startCharacter(MOB mob, boolean isBorrowedClass, boolean verifyOnly);
170
171        /**
172         * Called when a class is no longer the current dominant class of a player or mob,
173         * usually during class training.
174         * @param mob the mob whose career to end or put aside
175         */
176        public void endCharacter(MOB mob);
177
178        /**
179         * Returns whether the given mob should share in the experience gained by the killer
180         * for having killed the killed.  Assumes the mob is in the same room, and requires
181         * the followers of the killer be passed in.   The class that determines this is usually that
182         * of the group leader.
183         * @param killer the killer mob
184         * @param killed who the killer mob killed
185         * @param mob the mob whose sharing capacity is being evaluated
186         * @param followers the killers followers
187         * @return whether the mob shares in the exp gains
188         */
189        public boolean isValidClassBeneficiary(MOB killer, MOB killed, MOB mob, Set<MOB> followers);
190
191        /**
192         * Returns whether the given mob should count in the division of experience gained by the killer
193         * for having killed the killed.  Assumes the mob is in the same room, and requires
194         * the followers of the killer be passed in.  The class that determines this is usually that
195         * of the group leader.
196         * @param killer the killer mob
197         * @param killed who the killer mob killed
198         * @param mob the mob whose sharing capacity is being evaluated
199         * @param followers the killers followers
200         * @return whether the mob shares in the exp gains
201         */
202        public boolean isValidClassDivider(MOB killer, MOB killed, MOB mob, Set<MOB> followers);
203
204        /**
205         * Returns whether this class, for the given mob of this class, should count in the division
206         * of experience gained by the killer for having killed the killed.  Assumes the mob is in the
207         * same room, and requires the followers of the killer be passed in.
208         * @param killer the killer mob
209         * @param killed who the killer mob killed
210         * @param mob the mob whose sharing capacity is being evaluated
211         * @param followers the killers followers
212         * @return whether the mob shares in the exp gains
213         */
214        public boolean canBeADivider(MOB killer, MOB killed, MOB mob, Set<MOB> followers);
215
216        /**
217         * Returns whether this class, for the given mob of this class, should share in the
218         * experience gained by the killer for having killed the killed.  Assumes the mob
219         * is in the same room, and requires the followers of the killer be passed in.
220         *
221         * @param killer the killer mob
222         * @param killed who the killer mob killed
223         * @param mob the mob whose sharing capacity is being evaluated
224         * @param followers the killers followers
225         * @return whether the mob shares in the exp gains
226         */
227        public boolean canBeABenificiary(MOB killer, MOB killed, MOB mob, Set<MOB> followers);
228
229        /**
230         * Typically called when a mob gains a level in this class, to allow the class to
231         * assign any new skills.  Can also be called just to populate a mob with class skills,
232         * so it should also confirm any lower level skills also.
233         * @see com.planet_ink.coffee_mud.MOBS.interfaces.MOB#addAbility(Ability)
234         * @param mob the mob to give abilities to.
235         * @param isBorrowedClass whether the skills are savable (false) or temporary (true)
236         */
237        public void grantAbilities(MOB mob, boolean isBorrowedClass);
238
239        /**
240         * This method is called whenever a player gains a level while a member of this class.  If
241         * there are any special things which need to be done to a player who gains a level, they
242         * can be done in this method.  By default, it doesn't do anything.
243         * @param mob the mob to level up
244         * @param gainedAbilityIDs the set of abilities/skill IDs gained during this leveling process
245         */
246        public void level(MOB mob, List<String> gainedAbilityIDs);
247
248        /**
249         * Whenever a player or mob of this race gains experience, this method gets a chance
250         * to modify the amount before the gain actually occurs.
251         * @param host the player or mob whose class is being queried
252         * @param mob the player or mob gaining experience
253         * @param victim if applicable, the mob or player who died to give the exp
254         * @param amount the amount of exp on track for gaining
255         * @return the adjusted amount of experience to gain
256         */
257        public int adjustExperienceGain(MOB host, MOB mob, MOB victim, int amount);
258
259        /**
260         * Whenever a uses a skill, this method can return any additional expertise
261         * granted explicitly by the class.
262         * @see com.planet_ink.coffee_mud.Libraries.interfaces.ExpertiseLibrary.Flag
263         * @param host the player or mob whose class is being queried
264         * @param expertiseCode the expertise code
265         * @param abilityID the Ability ID
266         * @return the inherent expertise level for this
267         */
268        public int addedExpertise(MOB host, ExpertiseLibrary.Flag expertiseCode, final String abilityID);
269
270        /**
271         * This method is called whenever a player loses a level while a member of this class.  If
272         * there are any special things which need to be done to a player who loses a level, they
273         * can be done in this method.  By default, it doesn't do anything.
274         * @param mob the mob to level down
275         */
276        public void unLevel(MOB mob);
277
278        /**
279         * Returns a vector of Item objects representing the standard
280         * clothing, weapons, or other objects commonly given to players
281         * of this class just starting out.
282         * @param myChar one who will receive the objects
283         * @return a vector of Item objects
284         */
285        public List<Item> outfit(MOB myChar);
286
287        /**
288         * This method is called whenever a player casts a spell which has a lasting
289         * effect on the target.  This method is called even if the class is not the
290         * players CURRENT class.
291         * @param myChar the caster or skill user
292         * @param skill the skill or spell that was cast.
293         * @param duration the default duration
294         * @return usually, it just returns default again
295         */
296        public int classDurationModifier(MOB myChar, Ability skill, int duration);
297
298        /**
299         * This method is called whenever a player casts a spell which has an affect
300         * that is level dependent.  This method is called even if the class is not the
301         * players CURRENT class.
302         * @param myChar the caster or skill user
303         * @param skill the skill or spell that was cast.
304         * @param level the default level
305         * @return usually, it just returns level again
306         */
307        public int classLevelModifier(MOB myChar, Ability skill, int level);
308        /**
309         * Returns the number of bonus practices received by members of
310         * this class when they gain a level.  This is over and above
311         * the normal formula applied during the leveling process.
312         * @return the number of bonus practices to grant
313         */
314        public int getBonusPracLevel();
315        /**
316         * Returns the number of bonus attack points received by members of
317         * this class when they gain a level.  This is over and above
318         * the normal formula applied during the leveling process.
319         * @return the number of bonus attack points to grant
320         */
321        public int getBonusAttackLevel();
322        /**
323         * Returns which of the CharStats.STAT_* constants should be
324         * used to calculate the standard attack prowess points given
325         * when a member of this class gains a level.
326         * @see com.planet_ink.coffee_mud.Common.interfaces.CharStats
327         * @return a CharStats.STAT_* attribute constant code
328         */
329        public int getAttackAttribute();
330        /**
331         * Returns the number of practices received by members of
332         * this class when they are first created.
333         * @return the number of initial practices to grant
334         */
335        public int getPracsFirstLevel();
336        /**
337         * Returns the number of training points received by members of
338         * this class when they are first created.
339         * @return the number of initial trains to grant
340         */
341        public int getTrainsFirstLevel();
342        /**
343         * Returns the number of levels which must be gained by a member
344         * of this class before they gain 1 more point of default damage
345         * per hit.
346         * @return the number of levels between damage gains
347         */
348        public int getLevelsPerBonusDamage();
349
350        /**
351         * Returns the formula used every time a player of this class gains
352         * a level as this class.  The total is added (or removed on unlevel)
353         * from the players movement score.
354         * Variables may be included in the formula, which are:
355         * {@literal @}x1: Players current class level
356         * {@literal @}x2: Players adjusted Strength     {@literal @}x3: Players Max adjusted Strength
357         * {@literal @}x4: Players adjusted Dexterity    {@literal @}x5: Players Max adjusted Dexterity
358         * {@literal @}x6: Players adjusted Constitution {@literal @}x7: Players Max adjusted Constitution
359         * {@literal @}x8: Players adjusted Wisdom       {@literal @}x9: Players adjusted Intelligence
360         * @see CharClass#getMovementDesc()
361         * @return the formula that causes a gain or loss in movement
362         */
363        public String getMovementFormula();
364
365        /**
366         * Returns a text description of the movement bonuses
367         * gained by members of this class.
368         * @see CharClass#getMovementFormula()
369         * @return a text description of the movement bonuses
370         */
371        public String getMovementDesc();
372
373        /**
374         * Returns the formula used every time a player of this class gains
375         * a level as this class.  The total is added (or removed on unlevel)
376         * from the players hit points score.
377         * Variables may be included in the formula, which are:
378         * \{@literal @}x1: Players current class level
379         * \{@literal @}x2: Players adjusted Strength     {@literal @}x3: Players Max adjusted Strength
380         * \{@literal @}x4: Players adjusted Dexterity    {@literal @}x5: Players Max adjusted Dexterity
381         * {@literal @}x6: Players adjusted Constitution {@literal @}x7: Players Max adjusted Constitution
382         * {@literal @}x8: Players adjusted Wisdom       {@literal @}x9: Players adjusted Intelligence
383         * @see CharClass#getHitPointDesc()
384         * @return the formula that causes a gain or loss in hit points
385         */
386        public String getHitPointsFormula();
387
388        /**
389         * Returns a text description of the hit point bonuses
390         * gained by members of this class.
391         * @see CharClass#getHitPointsFormula()
392         * @return a text description of the hit point bonuses
393         */
394        public String getHitPointDesc();
395
396        /**
397         * Returns the formula used every time a player of this class gains
398         * a level as this class.  The total is added (or removed on unlevel)
399         * from the players mana score.
400         * Variables may be included in the formula, which are:
401         * {@literal @}x1: Players current class level
402         * {@literal @}x2: Players adjusted Wisdom       {@literal @}x3: Players Max adjusted Wisdom
403         * {@literal @}x4: Players adjusted Intelligence {@literal @}x5: Players Max adjusted Intelligence
404         * {@literal @}x6: Players adjusted Attack Attr  {@literal @}x7: Players Max adjusted Attack Attr
405         * {@literal @}x8: Players adjusted Charisma     {@literal @}x9: Players adjusted Constitution
406         * @see CharClass#getManaDesc()
407         * @see CharClass#getAttackAttribute()
408         * @return the formula that causes a gain or loss in mana
409         */
410        public String getManaFormula();
411
412        /**
413         * Returns a text description of the mana bonuses
414         * gained by members of this class.
415         * @see CharClass#getManaFormula()
416         * @return a text description of the mana bonuses
417         */
418        public String getManaDesc();
419

622        /**
623         * This defines how this class fits into the SUB subclassing class system.
624         * A class may be one that can change to any class, or only another class
625         * if it is the same base class, or many no class at all.
626         * @see CharClass#getSubClassRule()
627         * @author Bo Zimmerman
628         *
629         */
630        public enum SubClassRule
