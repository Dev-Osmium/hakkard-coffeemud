277        @Override
278        public void executeMsg(final Environmental host, final CMMsg msg)
279        {
280                if(host instanceof MOB)
281                {
282                        final MOB myChar=(MOB)host;
283                        if(msg.amISource(myChar)
284                        &&(msg.tool() instanceof Ability)
285                        &&(!myChar.isMonster())
286                        &&(msg.sourceMinor()==CMMsg.TYP_PREINVOKE)
287                        &&(myChar.isMine(msg.tool()))
288                        &&(myChar.charStats().getClassLevel(this)>=30)
289                        &&(CMLib.ableMapper().getQualifyingLevel(ID(),true,msg.tool().ID())<1))
290                                invokable=new WeakReference(msg.tool());
291                }
292                super.executeMsg(host,msg);
293                Bard.visitationBonusMessage(host,msg);
294        }

296        @Override
297        public boolean okMessage(final Environmental myHost, final CMMsg msg)
298        {
299                if(!(myHost instanceof MOB))
300                        return super.okMessage(myHost,msg);
301
302                final MOB myChar=(MOB)myHost;
303                if((msg.tool() instanceof Ability)
304                &&(msg.targetMinor()!=CMMsg.TYP_TEACH))
305                {
306                        if(msg.amISource(myChar)
307                        &&(!myChar.isMonster())
308                        &&(msg.sourceMinor()!=CMMsg.TYP_PREINVOKE))
309                        {
310                                final WeakReference<Ability> curRef=invokable;
311                                if((curRef!=null)
312                                &&(msg.tool()==curRef.get()))
313                                {
314                                        curRef.clear();
315                                        final Ability A=((Ability)msg.tool());
316                                        final int[] usageCost=A.usageCost(myChar, false);
317                                        if(CMath.bset(A.usageType(),Ability.USAGE_MANA))
318                                                myChar.curState().adjMana(usageCost[Ability.USAGEINDEX_MANA]/4,myChar.maxState());
319                                        if(CMath.bset(A.usageType(),Ability.USAGE_MOVEMENT))
320                                                myChar.curState().adjMovement(usageCost[Ability.USAGEINDEX_MOVEMENT]/4,myChar.maxState());
321                                        if(CMath.bset(A.usageType(),Ability.USAGE_HITPOINTS))
322                                                myChar.curState().adjHitPoints(usageCost[Ability.USAGEINDEX_HITPOINTS]/4,myChar.maxState());
323                                }
324                        }
325                        else
326                        if(msg.amITarget(myChar))
327                        {
328                                if(((((Ability)msg.tool()).classificationCode()&Ability.ALL_ACODES)==Ability.ACODE_SPELL)
329                                &&((((Ability)msg.tool()).classificationCode()&Ability.ALL_DOMAINS)==Ability.DOMAIN_DIVINATION)
330                                &&(CMLib.dice().roll(1,100,0)<(myChar.charStats().getClassLevel(this)*4)))
331                                {
332                                        myChar.location().show(msg.source(),myChar,CMMsg.MSG_OK_ACTION,L("<T-NAME> fool(s) <S-NAMESELF>, causing <S-HIM-HER> to fizzle @x1.",msg.tool().name()));
333                                        return false;
334                                }
335                        }
336                }
337                return super.okMessage(myHost,msg);
338        }

340        @Override
341        public void grantAbilities(MOB mob, boolean isBorrowedClass)
342        {
343                super.grantAbilities(mob,isBorrowedClass);
344
345                // if he already has one, don't give another!
346                if(mob.playerStats()!=null)
347                {
348                        final int classLevel=mob.baseCharStats().getClassLevel(this);
349                        if(classLevel<2) 
350                                return;
351                        if((classLevel%2)!=0) 
352                                return;
353
354                        int maxSkills=classLevel/2;
355
356                        // now only give one, for current level, respecting alignment!
357                        // first, get a list of all skills you don't qualify for that you MIGHT have gained or will gain
358                        final List<Ability> choices=new Vector<Ability>();
359                        for(final Enumeration<Ability> a=CMClass.abilities();a.hasMoreElements();)
360                        {
361                                final Ability A=a.nextElement();
362                                final int lql=CMLib.ableMapper().lowestQualifyingLevel(A.ID());
363                                if((CMLib.ableMapper().qualifyingLevel(mob,A)<=0)
364                                &&(lql<25)
365                                &&(lql>0)
366                                &&(!CMLib.ableMapper().getSecretSkill(A.ID()))
367                                &&(CMLib.ableMapper().qualifiesByAnyCharClass(A.ID()))
368                                &&(CMLib.ableMapper().availableToTheme(A.ID(),Area.THEME_FANTASY,true))
369                                &&(!CMLib.ableMapper().qualifiesOnlyByClan(mob, A))
370                                &&(!CMLib.ableMapper().qualifiesOnlyByRace(mob, A))
371                                &&(A.isAutoInvoked()||((A.triggerStrings()!=null)&&(A.triggerStrings().length>0))))
372                                {
373                                        final DVector prereqs=CMLib.ableMapper().getUnmetPreRequisites(mob,A);
374                                        if((prereqs==null)||(prereqs.size()==0))
375                                                choices.add(A);
376                                }
377                        }
378                        
379                        // now count those you already have
380                        for(int a=choices.size()-1;a>=0;a--)
381                        {
382                                final Ability A=choices.get(a);
383                                if(mob.fetchAbility(A.ID())!=null)
384                                        maxSkills--;
385                        }
386                        if(maxSkills<1) // if that reduced you to 0, you are done.
387                                return;
388                        // now eliminate those you already have, and those that are
389                        // above your level, if you are <25
390                        for(int a=choices.size()-1;a>=0;a--)
391                        {
392                                final Ability A=choices.get(a);
393                                final int lql=CMLib.ableMapper().lowestQualifyingLevel(A.ID());
394                                if((mob.fetchAbility(ID())!=null)
395                                ||((lql!=classLevel)&&(lql!=classLevel-1)&&(classLevel<25)))
396                                        choices.remove(a);
397                        }
398                        if(choices.size()==0)
399                                return;
400                        final Ability A=choices.get(CMLib.dice().roll(1,choices.size(),-1));
401                        if(A!=null)
402                                giveMobAbility(mob,A,0,"",isBorrowedClass);
403                }
404                else
405                {
406                        final List<AbilityMapper.AbilityMapping> V=CMLib.ableMapper().getUpToLevelListings(ID(),
407                                                                                                mob.charStats().getClassLevel(ID()),
408                                                                                                false,
409                                                                                                false);
410                        for(final AbilityMapper.AbilityMapping able : V)
411                        {
412                                final Ability A=CMClass.getAbility(able.abilityID());
413                                if((A!=null)
414                                &&(!CMLib.ableMapper().getAllQualified(ID(),true,A.ID()))
415                                &&(!CMLib.ableMapper().getDefaultGain(ID(),true,A.ID())))
416                                        giveMobAbility(mob,A,CMLib.ableMapper().getDefaultProficiency(ID(),true,A.ID()),CMLib.ableMapper().getDefaultParm(ID(),true,A.ID()),isBorrowedClass);
417                        }
418                }
419        }
