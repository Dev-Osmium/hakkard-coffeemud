132        protected Set<Tickable> currentApprentices = new HashSet<Tickable>();
133
134        @Override
135        public void initializeClass()
136        {
137                super.initializeClass();
138                CMLib.ableMapper().addCharAbilityMapping(ID(),1,"Skill_Write",true);
139                CMLib.ableMapper().addCharAbilityMapping(ID(),1,"Specialization_Natural",false);
140                CMLib.ableMapper().addCharAbilityMapping(ID(),1,"Skill_Recall",25,true);
141                CMLib.ableMapper().addCharAbilityMapping(ID(),1,"Skill_Swim",false);
142                CMLib.ableMapper().addCharAbilityMapping(ID(),1,"Skill_Climb",true);
143                CMLib.ableMapper().addCharAbilityMapping(ID(),1,"ClanCrafting",false);
144        }

151
152        @Override
153        public boolean tick(final Tickable ticking, final int tickID)
154        {
155                if((tickID==Tickable.TICKID_MOB)
156                &&(ticking instanceof MOB)
157                &&(!((MOB)ticking).isMonster()))
158                {
159                        if(((MOB)ticking).baseCharStats().getCurrentClass().ID().equals(ID()))
160                        {
161                                if(!currentApprentices.contains(ticking))
162                                        currentApprentices.add(ticking);
163                        }
164                        else
165                        if(currentApprentices.contains(ticking))
166                        {
167                                currentApprentices.remove(ticking);
168                                ((MOB)ticking).tell(L("\n\r\n\r^ZYou are no longer an apprentice!!!!^N\n\r\n\r"));
169                                CMLib.leveler().postExperience((MOB)ticking,null,null,1000,false);
170                        }
171                }
172                return super.tick(ticking,tickID);
173        }

196        @Override
197        public void startCharacter(MOB mob, boolean isBorrowedClass, boolean verifyOnly)
198        {
199                super.startCharacter(mob, isBorrowedClass, verifyOnly);
200                if(!verifyOnly)
201                {
202                        if(mob.playerStats()!=null)
203                        {
204                                mob.playerStats().setBonusCommonSkillLimits(mob.playerStats().getBonusCommonSkillLimits()+1);
205                                mob.playerStats().setBonusCraftingSkillLimits(mob.playerStats().getBonusCraftingSkillLimits()+1);
206                                mob.playerStats().setBonusNonCraftingSkillLimits(mob.playerStats().getBonusNonCraftingSkillLimits()+1);
207                        }
208                }
209        }
210
211        @Override
212        public List<Item> outfit(MOB myChar)
213        {
214                if(outfitChoices==null)
215                {
216                        final Weapon w=CMClass.getWeapon("Dagger");
217                        if(w == null)
218                                return new Vector<Item>();
219                        outfitChoices=new Vector<Item>();
220                        outfitChoices.add(w);
221                }
222                return outfitChoices;
223        }
224
225        @Override
226        public int adjustExperienceGain(MOB host, MOB mob, MOB victim, int amount)
227        {
228                if((amount > 0)&&(!expless()))
229                {
230                        if(mob.charStats().getCurrentClass() == this)
231                        {
232                                if(mob.getExperience() + amount > mob.getExpNextLevel())
233                                {
234                                        amount = 0;
235                                }
236                        }
237                }
238                return amount;
239        }
