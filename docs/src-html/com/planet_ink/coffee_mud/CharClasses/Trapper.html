175        @Override
176        public boolean okMessage(final Environmental myHost, final CMMsg msg)
177        {
178                if(!(myHost instanceof MOB))
179                        return super.okMessage(myHost,msg);
180                final MOB myChar=(MOB)myHost;
181                if(msg.amISource(myChar)
182                &&(!myChar.isMonster())
183                &&(msg.tool() instanceof Ability)
184                &&(!CMath.bset(msg.sourceMajor(),CMMsg.MASK_ALWAYS))
185                &&(myChar.location()!=null)
186                &&(myChar.isMine(msg.tool()))
187                &&(msg.sourceMinor()!=CMMsg.TYP_TEACH))
188                {
189                        // animal trade must be here because execute of trade kills the mob object
190                        // also, an add trailer is done, which only hits if this msg is not cancelled,
191                        // so ALL GOOD
192                        if((msg.tool().ID().equalsIgnoreCase("AnimalTrading"))
193                        &&(msg.value()<0)
194                        &&(msg.target() instanceof MOB)
195                        &&(CMLib.flags().isAnimalIntelligence((MOB)msg.target()))
196                        &&(((MOB)msg.target()).getStartRoom()!=null)
197                        &&(CMLib.map().areaLocation(myChar)!=CMLib.map().getStartArea(msg.target())))
198                        {
199                                int xp=125;
200                                if((xp>0)&&CMLib.leveler().postExperience(myChar,null,null,xp,true))
201                                        msg.addTrailerMsg(CMClass.getMsg(myChar,null,null,CMMsg.MSG_OK_VISUAL,L("You gain @x1 experience for selling @x2.",""+xp,((MOB)msg.target()).name(myChar)),CMMsg.NO_EFFECT,null,CMMsg.NO_EFFECT,null));
202                        }
203                        if((((myChar.location().domainType()&Room.INDOORS)>0))
204                        ||(myChar.location().domainType()==Room.DOMAIN_OUTDOORS_CITY))
205                        {
206                                if((msg.tool().ID().equalsIgnoreCase("Thief_Hide"))
207                                ||(msg.tool().ID().equalsIgnoreCase("Thief_Sneak")))
208                                {
209                                        CharClass C=null;
210                                        CharClass chosenC=null;
211                                        for(int c=0;c<myChar.charStats().numClasses();c++)
212                                        {
213                                                C=myChar.charStats().getMyClass(c);
214                                                if(C==null)
215                                                        continue;
216                                                final int qlvl=CMLib.ableMapper().getQualifyingLevel(C.ID(),false,msg.tool().ID());
217                                                if((qlvl>=0)
218                                                &&(myChar.charStats().getClassLevel(C)>=qlvl)
219                                                &&((chosenC==null)||(chosenC.ID().equals(ID()))))
220                                                        chosenC=C;
221                                        }
222                                        if((chosenC!=null)&&(chosenC.ID().equals(ID())))
223                                        {
224                                                if(msg.tool().ID().equalsIgnoreCase("Thief_Hide"))
225                                                {
226                                                        myChar.tell(L("You don't know how to hide outside the wilderness."));
227                                                        return false;
228                                                }
229                                                else
230                                                if(msg.tool().ID().equalsIgnoreCase("Thief_Sneak"))
231                                                {
232                                                        myChar.tell(L("You don't know how to sneak outside the wilderness."));
233                                                        return false;
234                                                }
235                                        }
236                                }
237                        }
238                }
239                return super.okMessage(myHost,msg);
240        }
