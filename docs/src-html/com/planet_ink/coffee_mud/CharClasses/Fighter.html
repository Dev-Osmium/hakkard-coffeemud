245        @Override
246        public void grantAbilities(MOB mob, boolean isBorrowedClass)
247        {
248                super.grantAbilities(mob,isBorrowedClass);
249                if(mob.playerStats()==null)
250                {
251                        final List<AbilityMapper.AbilityMapping> V=CMLib.ableMapper().getUpToLevelListings(ID(),
252                                                                                                                         mob.charStats().getClassLevel(ID()),
253                                                                                                                         false,
254                                                                                                                         false);
255                        for(final AbilityMapper.AbilityMapping able : V)
256                        {
257                                final Ability A=CMClass.getAbility(able.abilityID());
258                                if((A!=null)
259                                &&(!CMLib.ableMapper().getAllQualified(ID(),true,A.ID()))
260                                &&(!CMLib.ableMapper().getDefaultGain(ID(),true,A.ID())))
261                                        giveMobAbility(mob,A,CMLib.ableMapper().getDefaultProficiency(ID(),true,A.ID()),CMLib.ableMapper().getDefaultParm(ID(),true,A.ID()),isBorrowedClass);
262                        }
263                }
264        }

280        public static void duelExperience(CharClass C, Environmental host, CMMsg msg)
281        {
282                if((msg.targetMinor()==CMMsg.TYP_DUELLOSS)
283                &&(msg.target()!=msg.tool())
284                &&(msg.target()!=host)
285                &&(host instanceof MOB)
286                &&(msg.tool()==host)
287                &&((msg.source().playerStats().getAccount()==null)
288                        ||(((MOB)host).playerStats().getAccount()==null)
289                        ||(((MOB)host).playerStats().getAccount()!=msg.source().playerStats().getAccount()))
290                )
291                {
292                        synchronized(duelWinners)
293                        {
294                                final long now=System.currentTimeMillis();
295                                final long timeout=now - (now - MILLIS_BETWEEN_DUEL_WINS);
296                                if(lastDuelWinner < timeout)
297                                {
298                                        for(final Iterator<String> m = duelWinners.keySet().iterator(); m.hasNext(); )
299                                        {
300                                                if(duelWinners.get(m.next())[0] < timeout)
301                                                        m.remove();
302                                        }
303                                }
304                                final long[] lastTime=duelWinners.get(host.Name());
305                                if((lastTime != null) && (lastTime[0] > timeout))
306                                        return;
307                                if(lastTime == null)
308                                        duelWinners.put(host.Name(), new long[]{now});
309                                else
310                                        lastTime[0] = now;
311                                lastDuelWinner = now;
312                        }
313                        ((MOB)host).tell(CMLib.lang().L("^YVictory!!^N"));
314                        CMLib.leveler().postExperience((MOB)host,null,null,300,false);
315                }
316        }

318        public static void conquestExperience(CharClass C, Environmental host, CMMsg msg)
319        {
320                if((msg.targetMinor()==CMMsg.TYP_AREAAFFECT)
321                &&(msg.target() instanceof Area)
322                &&(msg.targetMessage()!=null)
323                &&(msg.targetMessage().equalsIgnoreCase("CONQUEST"))
324                &&(host instanceof MOB)
325                &&(((MOB)host).charStats().getCurrentClass().ID().equals(C.ID()))
326                &&(((MOB)host).getClanRole(msg.source().Name())!=null)
327                &&(!((MOB)host).isMonster())
328                )
329                {
330                        final Area A=(Area)msg.target();
331                        int xp=(int)Math.round(50.0*CMath.div(A.getAreaIStats()[Area.Stats.AVG_LEVEL.ordinal()],((MOB)host).phyStats().level()));
332                        if(xp>500)
333                                xp=500;
334                        if(xp>0)
335                        {
336                                ((MOB)host).tell(CMLib.lang().L("^YVictory!!^N"));
337                                CMLib.leveler().postExperience((MOB)host,null,null,xp,false);
338                        }
339                }
340        }
