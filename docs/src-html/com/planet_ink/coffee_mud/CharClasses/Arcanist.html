204        @Override
205        public boolean okMessage(final Environmental myHost, final CMMsg msg)
206        {
207                if((myHost==null)
208                ||(!(myHost instanceof MOB)))
209                        return super.okMessage(myHost,msg);
210
211                final MOB mob=(MOB)myHost;
212                if((msg.amISource(mob))
213                &&(mob.charStats().getClassLevel(this)>4))
214                {
215                        if(((msg.sourceMinor()==CMMsg.TYP_BUY)
216                                ||(msg.sourceMinor()==CMMsg.TYP_LIST))
217                        &&(msg.tool() instanceof Potion))
218                        {
219                                mob.basePhyStats().setDisposition(mob.basePhyStats().disposition()|PhyStats.IS_BONUS);
220                                mob.recoverPhyStats();
221                                mob.recoverCharStats();
222                        }
223                        else
224                        if((mob.basePhyStats().disposition()&PhyStats.IS_BONUS)==PhyStats.IS_BONUS)
225                        {
226                                mob.basePhyStats().setDisposition(mob.basePhyStats().disposition()-PhyStats.IS_BONUS);
227                                mob.recoverPhyStats();
228                                mob.recoverCharStats();
229                        }
230                }
231                return super.okMessage(myHost,msg);
232        }

234        @Override
235        public void endCharacter(MOB mob)
236        {
237                final Vector<Ability> otherChoices=new Vector<Ability>();
238                for(int a=0;a<mob.numAbilities();a++)
239                {
240                        final Ability A2=mob.fetchAbility(a);
241                        if((A2!=null)
242                        &&(!A2.isSavable())
243                        &&((A2.classificationCode()&Ability.ALL_ACODES)==Ability.ACODE_SPELL))
244                                otherChoices.addElement(A2);
245                }
246                for(int a=0;a<otherChoices.size();a++)
247                        mob.delAbility(otherChoices.elementAt(a));
248        }
249
250        private void addAbilityToSpellcraftList(MOB mob, Ability A)
251        {
252                final Ability enabledA=mob.fetchAbility("Skill_Spellcraft");
253                if(enabledA!=null)
254                {
255                        final List<String> ables=CMParms.parseCommas(enabledA.text(), true);
256                        if(!ables.contains(A.ID()))
257                        {
258                                if(enabledA.text().length()==0)
259                                        enabledA.setMiscText(A.ID());
260                                else
261                                        enabledA.setMiscText(enabledA.text()+", "+A.ID());
262                                mob.addAbility(A);
263                        }
264                        else
265                        if(mob.isMine(A) && (A.proficiency()<75) && (!A.isSavable()))
266                                A.setProficiency(A.proficiency()+(mob.baseCharStats().getStat(CharStats.STAT_INTELLIGENCE)/3));
267                }
268        }
269
270        private void clearAbilityFromSpellcraftList(MOB mob, Ability A)
271        {
272                final Ability enabledA=mob.fetchAbility("Skill_Spellcraft");
273                if(enabledA!=null)
274                {
275                        final List<String> ables=CMParms.parseCommas(enabledA.text(), true);
276                        if(ables.contains(A.ID()))
277                        {
278                                if(!CMSecurity.isAllowed(mob, mob.location(), CMSecurity.SecFlag.ALLSKILLS))
279                                {
280                                        ables.remove(A.ID());
281                                        enabledA.setMiscText(CMParms.toListString(ables));
282                                        mob.delAbility(A);
283                                }
284                        }
285                }
286        }

288        @Override
289        public void executeMsg(final Environmental myHost, final CMMsg msg)
290        {
291                super.executeMsg(myHost,msg);
292                if((myHost==null)||(!(myHost instanceof MOB)))
293                        return;
294                final MOB mob=(MOB)myHost;
295                if(msg.amISource(mob))
296                {
297                        if(((msg.sourceMinor()==CMMsg.TYP_LOOK)||(msg.sourceMinor()==CMMsg.TYP_EXAMINE))
298                        &&(msg.target() instanceof Wand)
299                        &&(mob.charStats().getClassLevel(this)>=30))
300                        {
301                                final String message=L("<O-NAME> has @x1 charges remaining.",""+((Wand)msg.target()).usesRemaining());
302                                msg.addTrailerMsg(CMClass.getMsg(mob, null, msg.target(), CMMsg.MSG_OK_VISUAL, CMMsg.NO_EFFECT, CMMsg.NO_EFFECT, message));
303                        }
304                        else
305                        if((msg.tool()!=null)
306                        &&(msg.sourceMinor()!=CMMsg.TYP_TEACH))
307                        {
308                                if(msg.tool().ID().equals("Skill_Spellcraft"))
309                                {
310                                        if((msg.tool().text().length()>0)
311                                        &&(msg.target() instanceof MOB))
312                                        {
313                                                Ability A=((MOB)msg.target()).fetchAbility(msg.tool().text());
314                                                if(A==null)
315                                                        return;
316                                                final Ability myA=mob.fetchAbility(A.ID());
317                                                if(myA!=null)
318                                                {
319                                                        if((!A.isSavable())
320                                                        &&((A.classificationCode()&Ability.ALL_ACODES)==Ability.ACODE_SPELL)
321                                                        &&(CMLib.ableMapper().lowestQualifyingLevel(A.ID())<30))
322                                                                addAbilityToSpellcraftList(mob,A);
323                                                }
324                                                else
325                                                if(CMLib.ableMapper().lowestQualifyingLevel(A.ID())<30)
326                                                {
327                                                        final Vector<Ability> otherChoices=new Vector<Ability>();
328                                                        for(int a=0;a<mob.numAbilities();a++)
329                                                        {
330                                                                final Ability A2=mob.fetchAbility(a);
331                                                                if((A2!=null)
332                                                                &&(!A2.isSavable())
333                                                                &&((A2.classificationCode()&Ability.ALL_ACODES)==Ability.ACODE_SPELL))
334                                                                        otherChoices.addElement(A2);
335                                                        }
336                                                        A=(Ability)A.copyOf();
337                                                        A.setProficiency(0);
338                                                        A.setSavable(false);
339                                                        if(otherChoices.size()>(mob.charStats().getClassLevel(this)/3))
340                                                        {
341                                                                final Ability A2=otherChoices.elementAt(CMLib.dice().roll(1,otherChoices.size(),-1));
342                                                                clearAbilityFromSpellcraftList(mob,A2);
343                                                        }
344                                                        addAbilityToSpellcraftList(mob,A);
345                                                }
346                                        }
347                                }
348                                else
349                                if((msg.sourceMinor()!=CMMsg.TYP_PREINVOKE)
350                                &&(msg.tool().ID().equals("Spell_Scribe")
351                                ||msg.tool().ID().equals("Spell_EnchantWand")
352                                ||msg.tool().ID().equals("Spell_MagicItem")
353                                ||msg.tool().ID().equals("Spell_StoreSpell")
354                                ||msg.tool().ID().equals("Spell_WardArea")))
355                                {
356                                        final Ability A=mob.fetchAbility(msg.tool().text());
357                                        if((A!=null)&&(!A.isSavable()))
358                                                clearAbilityFromSpellcraftList(mob,A);
359                                }
360                                else
361                                if(msg.tool() instanceof Ability)
362                                {
363                                        final Ability A=mob.fetchAbility(msg.tool().ID());
364                                        if((A!=null)&&(!A.isSavable())
365                                        &&((A.classificationCode()&Ability.ALL_ACODES)==Ability.ACODE_SPELL))
366                                                clearAbilityFromSpellcraftList(mob,A);
367                                }
368                        }
369                }
370        }

372        @Override
373        public boolean tick(final Tickable ticking, final int tickID)
374        {
375                if((ticking instanceof MOB)
376                &&((((MOB)ticking).basePhyStats().disposition()&PhyStats.IS_BONUS)==PhyStats.IS_BONUS))
377                {
378                        ((MOB)ticking).basePhyStats().setDisposition(((MOB)ticking).basePhyStats().disposition()-PhyStats.IS_BONUS);
379                        ((MOB)ticking).recoverPhyStats();
380                }
381                return super.tick(ticking,tickID);
382        }

384        @Override
385        public void affectCharStats(final MOB affected, final CharStats affectableStats)
386        {
387                super.affectCharStats(affected,affectableStats);
388                affectableStats.setStat(CharStats.STAT_SAVE_MAGIC,affectableStats.getStat(CharStats.STAT_SAVE_MAGIC)+(affectableStats.getClassLevel(this)));
389                if(CMath.bset(affected.basePhyStats().disposition(),PhyStats.IS_BONUS))
390                        affectableStats.setStat(CharStats.STAT_CHARISMA,affectableStats.getStat(CharStats.STAT_CHARISMA)+30);
391        }
