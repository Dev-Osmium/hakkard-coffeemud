100        @Override
101        public void setMiscText(String newMiscText)
102        {
103                super.setMiscText(newMiscText);
104                List<String> flags=CMParms.parse(newMiscText.toUpperCase().trim());
105                if(flags.contains("PERMANENT")||flags.contains("UNEXTINGUISHABLE"))
106                        this.setAbilityCode(this.abilityCode|Burning.FIREFLAG_UNEXTINGUISHABLE);
107                if(flags.contains("PERSISTFLAGS"))
108                        this.setAbilityCode(this.abilityCode|Burning.FIREFLAG_PERSISTFLAGS);
109                if(flags.contains("DESTROY")||flags.contains("DESTROYHOST"))
110                        this.setAbilityCode(this.abilityCode|Burning.FIREFLAG_DESTROYHOST);
111                if(flags.contains("NODESTROY")||flags.contains("NODESTROYHOST"))
112                        this.setAbilityCode(this.abilityCode|Burning.FIREFLAG_NEVERDESTROYHOST);
113                if(flags.contains("NOSPREAD"))
114                        this.setAbilityCode(this.abilityCode|Burning.FIREFLAG_NOSPREAD);
115        }
116        @Override
117        public void unInvoke()
118        {
119                final Physical affected=this.affected;
120                super.unInvoke();
121                if(canBeUninvoked() && (affected != null) && (!affected.amDestroyed()))
122                {
123                        if(isFlagSet(FIREFLAG_DESTROYHOST))
124                                affected.destroy();
125                }
126        }

128        public boolean isFlagSet(int flag)
129        {
130                if(abilityCode()<0)
131                        return CMath.bset(-abilityCode(), flag);
132                else
133                        return CMath.bset(abilityCode(), flag);
134        }

136        @Override
137        public boolean tick(final Tickable ticking, final int tickID)
138        {
139                Physical affected=this.affected;
140                if((affected instanceof Item)
141                &&(((Item)affected).owner() instanceof Room)
142                &&(!CMath.bset(abilityCode(), Burning.FIREFLAG_UNEXTINGUISHABLE)))
143                {
144                        int unInvokeChance;
145                        if(abilityCode() < 0)
146                                unInvokeChance=((-abilityCode()) & FIREFLAG_WEATHERMASK);
147                        else
148                                unInvokeChance=-(abilityCode() & FIREFLAG_WEATHERMASK);
149                        String what=null;
150                        switch(((Room)(((Item)affected).owner())).getArea().getClimateObj().weatherType(((Room)(((Item)affected).owner()))))
151                        {
152                        case Climate.WEATHER_RAIN:
153                                what="rain";
154                                unInvokeChance+=10;
155                                break;
156                        case Climate.WEATHER_THUNDERSTORM:
157                                what="pounding rain";
158                                unInvokeChance+=15;
159                                break;
160                        case Climate.WEATHER_SLEET:
161                                what="sleet";
162                                unInvokeChance+=5;
163                                break;
164                        case Climate.WEATHER_BLIZZARD:
165                                what="swirling snow";
166                                unInvokeChance+=10;
167                                break;
168                        case Climate.WEATHER_SNOW:
169                                what="snow";
170                                unInvokeChance+=10;
171                                break;
172                        }
173                        if(CMLib.dice().rollPercentage()<unInvokeChance)
174                        {
175                                final Room R=((Room)(((Item)affected).owner()));
176                                if(R.numInhabitants()>0)
177                                        R.showHappens(CMMsg.MSG_OK_ACTION,L("The @x1 puts out @x2.",what,affected.name()));
178                                unInvoke();
179                                
180                                return false;
181                        }
182                }
183                if((tickDown<2)&&(affected!=null))
184                {
185                        if(affected instanceof Item)
186                        {
187                                final Environmental E=((Item)affected).owner();
188                                if(E==null)
189                                {
190                                        if(!isFlagSet(FIREFLAG_NEVERDESTROYHOST))
191                                                ((Item)affected).destroy();
192                                }
193                                else
194                                if(E instanceof Room)
195                                {
196                                        final Room room=(Room)E;
197                                        if((affected instanceof RawMaterial)
198                                        &&(!CMath.bset(abilityCode(), Burning.FIREFLAG_NOSPREAD))
199                                        &&(room.isContent((Item)affected)))
200                                        {
201                                                for(int i=0;i<room.numItems();i++)
202                                                {
203                                                        final Item I=room.getItem(i);
204                                                        if(I.name().equals(affected.name())
205                                                        &&(I!=affected)
206                                                        &&(I instanceof RawMaterial)
207                                                        &&(I.material()==((Item)affected).material()))
208                                                        {
209                                                                int durationOfBurn=CMLib.materials().getBurnDuration(I);
210                                                                if(durationOfBurn<=0) 
211                                                                        durationOfBurn=5;
212                                                                final Burning B=new Burning();
213                                                                if(isFlagSet(FIREFLAG_PERSISTFLAGS))
214                                                                        B.setAbilityCode(abilityCode());
215                                                                B.invoke(invoker,I,true,durationOfBurn);
216                                                                break;
217                                                        }
218                                                }
219                                        }
220                                        if(!(affected instanceof ClanItem))
221                                        {
222                                                switch(((Item)affected).material()&RawMaterial.MATERIAL_MASK)
223                                                {
224                                                case RawMaterial.MATERIAL_LIQUID:
225                                                case RawMaterial.MATERIAL_METAL:
226                                                case RawMaterial.MATERIAL_MITHRIL:
227                                                case RawMaterial.MATERIAL_ENERGY:
228                                                case RawMaterial.MATERIAL_GAS:
229                                                case RawMaterial.MATERIAL_PRECIOUS:
230                                                case RawMaterial.MATERIAL_ROCK:
231                                                case RawMaterial.MATERIAL_UNKNOWN:
232                                                        break;
233                                                default:
234                                                {
235                                                        if(CMLib.flags().isABonusItems(affected))
236                                                        {
237                                                                if(invoker==null)
238                                                                {
239                                                                        invoker=CMClass.getMOB("StdMOB");
240                                                                        invoker.setLocation(CMClass.getLocale("StdRoom"));
241                                                                        invoker.basePhyStats().setLevel(affected.phyStats().level());
242                                                                        invoker.phyStats().setLevel(affected.phyStats().level());
243                                                                }
244                                                                room.showHappens(CMMsg.MSG_OK_ACTION,L("@x1 EXPLODES!!!",affected.name()));
245                                                                for(int i=0;i<room.numInhabitants();i++)
246                                                                {
247                                                                        final MOB target=room.fetchInhabitant(i);
248                                                                        CMLib.combat().postDamage(invoker(),target,null,CMLib.dice().roll(affected.phyStats().level(),5,1),CMMsg.MASK_ALWAYS|CMMsg.TYP_FIRE,Weapon.TYPE_BURNING,L("The blast <DAMAGE> <T-NAME>!"));
249                                                                }
250                                                                if(!isFlagSet(FIREFLAG_NEVERDESTROYHOST))
251                                                                        ((Item)affected).destroy();
252                                                        }
253                                                        else
254                                                        {
255                                                                final Item ash=CMClass.getItem("GenResource");
256                                                                ash.setName(L("some ash"));
257                                                                ash.setDisplayText(L("a small pile of ash is here"));
258                                                                ash.setMaterial(RawMaterial.RESOURCE_ASH);
259                                                                ash.basePhyStats().setWeight(1);
260                                                                ash.recoverPhyStats();
261                                                                room.addItem(ash,ItemPossessor.Expire.Monster_EQ);
262                                                                ((RawMaterial)ash).rebundle();
263                                                                if((affected instanceof RawMaterial)
264                                                                &&(affected.basePhyStats().weight()>1)
265                                                                &&(CMLib.materials().getBurnDuration(affected)>0))
266                                                                {
267                                                                        affected.basePhyStats().setWeight(affected.basePhyStats().weight()-1);
268                                                                        affected.recoverPhyStats();
269                                                                        this.tickDown = CMLib.materials().getBurnDuration(affected);
270                                                                        CMLib.materials().adjustResourceName((Item)affected);
271                                                                        room.recoverRoomStats();
272                                                                        return super.tick(ticking,tickID);
273                                                                }
274                                                                room.showHappens(CMMsg.MSG_OK_VISUAL, L("@x1 is no longer burning.",affected.name()));
275                                                                if(!isFlagSet(FIREFLAG_NEVERDESTROYHOST))
276                                                                        ((Item)affected).destroy();
277                                                        }
278                                                        break;
279                                                }
280                                                }
281                                        }
282                                        ((Room)E).recoverRoomStats();
283                                }
284                                else
285                                if(E instanceof MOB)
286                                {
287                                        switch(((Item)affected).material()&RawMaterial.MATERIAL_MASK)
288                                        {
289                                        case RawMaterial.MATERIAL_LIQUID:
290                                        case RawMaterial.MATERIAL_METAL:
291                                        case RawMaterial.MATERIAL_ENERGY:
292                                        case RawMaterial.MATERIAL_MITHRIL:
293                                        case RawMaterial.MATERIAL_PRECIOUS:
294                                        case RawMaterial.MATERIAL_ROCK:
295                                        case RawMaterial.MATERIAL_UNKNOWN:
296                                                break;
297                                        default:
298                                                if(!isFlagSet(FIREFLAG_NEVERDESTROYHOST))
299                                                        ((Item)affected).destroy();
300                                                break;
301                                        }
302                                        ((MOB)E).location().recoverRoomStats();
303                                }
304                                return false;
305                        }
306                }
307                if(!super.tick(ticking,tickID))
308                        return false;
309
310                if(tickID!=Tickable.TICKID_MOB)
311                        return true;
312
313                affected=this.affected;
314                if(affected==null)
315                        return false;
316
317                if(affected instanceof Item)
318                {
319                        final Item I=(Item)affected;
320                        final Environmental owner=I.owner();
321                        if((owner instanceof MOB)&&(((MOB)owner).location()!=null))
322                        {
323                                if(!ouch((MOB)owner))
324                                        CMLib.commands().postDrop((MOB)owner,I,false,false,false);
325                        }
326                        if((I.subjectToWearAndTear())
327                        &&(I.usesRemaining()>1))
328                        {
329                                if(owner instanceof MOB)
330                                        CMLib.combat().postItemDamage((MOB)owner, I, null, 1, CMMsg.TYP_FIRE, null);
331                                else
332                                if(owner instanceof Room)
333                                {
334                                        final MOB M=CMLib.map().getFactoryMOB((Room)owner);
335                                        CMLib.combat().postItemDamage(M, I, null, 1, CMMsg.TYP_FIRE, null);
336                                        M.destroy();
337                                }
338                        }
339                }
340                
341                // might want to add the ability for it to spread
342                return true;
343        }

345        public boolean ouch(MOB mob)
346        {
347                if(CMLib.dice().rollPercentage()>(mob.charStats().getSave(CharStats.STAT_SAVE_FIRE)-50))
348                {
349                        if(affected instanceof Item)
350                        switch(((Item)affected).material()&RawMaterial.MATERIAL_MASK)
351                        {
352                        case RawMaterial.MATERIAL_LIQUID:
353                        case RawMaterial.MATERIAL_METAL:
354                        case RawMaterial.MATERIAL_MITHRIL:
355                        case RawMaterial.MATERIAL_PRECIOUS:
356                        case RawMaterial.MATERIAL_ENERGY:
357                        case RawMaterial.MATERIAL_GAS:
358                        case RawMaterial.MATERIAL_ROCK:
359                        case RawMaterial.MATERIAL_UNKNOWN:
360                                mob.tell(L("Ouch!! @x1 is HOT!",CMStrings.capitalizeAndLower(affected.name())));
361                                break;
362                        default:
363                                mob.tell(L("Ouch!! @x1 is on fire!",CMStrings.capitalizeAndLower(affected.name())));
364                                break;
365                        }
366                        CMLib.combat().postDamage(invoker,mob,this,CMLib.dice().roll(1,5,5),CMMsg.MASK_ALWAYS|CMMsg.TYP_FIRE,Weapon.TYPE_BURNING,null);
367                        return false;
368                }
369                return true;
370        }

372        @Override
373        public boolean okMessage(final Environmental myHost, final CMMsg msg)
374        {
375                if(!super.okMessage(myHost,msg))
376                        return false;
377
378                if((affected instanceof Item)
379                &&(msg.amITarget(affected))
380                &&((msg.targetMinor()==CMMsg.TYP_GET)
381                        ||(msg.targetMinor()==CMMsg.TYP_PUSH)
382                        ||(msg.targetMinor()==CMMsg.TYP_PULL)))
383                {
384                        if((msg.tool()==null)||(!(msg.tool() instanceof Item)))
385                                return ouch(msg.source());
386                        // the "oven" exception
387                        final Item container=(Item)affected;
388                        final Item target=(Item)msg.tool();
389                        if((target.owner()==container.owner())
390                        &&(target.container()==container))
391                        {
392                                switch(container.material()&RawMaterial.MATERIAL_MASK)
393                                {
394                                case RawMaterial.MATERIAL_METAL:
395                                case RawMaterial.MATERIAL_MITHRIL:
396                                case RawMaterial.MATERIAL_PRECIOUS:
397                                case RawMaterial.MATERIAL_ENERGY:
398                                case RawMaterial.MATERIAL_GAS:
399                                case RawMaterial.MATERIAL_ROCK:
400                                case RawMaterial.MATERIAL_UNKNOWN:
401                                        return true;
402                                default:
403                                        break;
404                                }
405                        }
406                        return ouch(msg.source());
407                }
408                return true;
409        }

411        @Override
412        public void executeMsg(final Environmental myHost, final CMMsg msg)
413        {
414                if((affected instanceof Item)
415                &&(msg.tool()==affected)
416                &&(msg.target() instanceof Container)
417                &&(msg.targetMinor()==CMMsg.TYP_PUT)
418                &&(!CMath.bset(abilityCode(), Burning.FIREFLAG_UNEXTINGUISHABLE)))
419                {
420                        final Item I=(Item)affected;
421                        final Item C=(Container)msg.target();
422                        if((C instanceof Drink)
423                           &&(((Drink)C).containsDrink()))
424                        {
425                                msg.addTrailerMsg(CMClass.getMsg(invoker,null,CMMsg.MSG_OK_VISUAL,L("@x1 is extinguished.",I.name())));
426                                I.delEffect(this);
427                        }
428                }
429                super.executeMsg(myHost,msg);
430        }

439        @Override
440        public boolean invoke(MOB mob, List<String> commands, Physical target, boolean auto, int asLevel)
441        {
442                if(!auto)
443                        return false;
444                if(target==null)
445                        return false;
446                if(target instanceof MOB)
447                {
448                        final MOB targM=(MOB)target;
449                        for(int i=0;i<10;i++)
450                        {
451                                Item I=targM.getRandomItem();
452                                if((I.container()==null)&&(CMLib.materials().getBurnDuration(I)>0))
453                                {
454                                        target=I;
455                                        break;
456                                }
457                        }
458                        // mobs don't burn, only their stuff
459                        if(target instanceof MOB)
460                        {
461                                int dmg=CMLib.dice().roll(1, targM.baseState().getHitPoints() / 10, 0);
462                                CMLib.combat().postDamage(mob, targM, this, dmg, CMMsg.MASK_ALWAYS | CMMsg.MASK_MALICIOUS | CMMsg.TYP_FIRE, Weapon.TYPE_BURNING, L("A fire <DAMAGE> <T-NAME>!"));
463                                return true;
464                        }
465                }
466                
467                if(target.fetchEffect("Burning")==null)
468                {
469                        if(((target instanceof Item)&&(((Item)target).material()==RawMaterial.RESOURCE_NOTHING))
470                        ||(target instanceof ClanItem))
471                                return false;
472                        if(mob!=null)
473                        {
474                                final Room room=mob.location();
475                                if(room!=null)
476                                {
477                                        final CMMsg msg=CMClass.getMsg(mob,target,CMMsg.MASK_ALWAYS|CMMsg.TYP_FIRE,null);
478                                        if(room.okMessage(mob,msg))
479                                                room.send(mob,msg);
480                                }
481                        }
482                        if(asLevel == 0)
483                                asLevel = CMLib.materials().getBurnDuration(target);
484                        if(asLevel < 0)
485                                asLevel = 0;
486                        beneficialAffect(mob,target,0,asLevel);
487                        target.recoverPhyStats();
488                        if(target instanceof Item)
489                        {
490                                final ItemPossessor owner=((Item)target).owner();
491                                if(owner!=null)
492                                {
493                                        owner.recoverPhyStats();
494                                        if(owner instanceof Room)
495                                                ((Room)owner).recoverRoomStats();
496                                        else
497                                        if(owner instanceof MOB)
498                                        {
499                                                final MOB ownerM=(MOB)owner;
500                                                final Room ownerRoom=ownerM.location();
501                                                if(ownerRoom != null)
502                                                        ownerRoom.recoverRoomStats();
503                                        }
504                                }
505                        }
506                }
507                return true;
508        }
