073        boolean                 temporarilyDisable      = false;
074        public Room             room                            = null;
075        boolean                 hitTheCeiling           = false;
076        int                             damageToTake            = 0;
077        protected int   fallTickDown            = 1;
078
079        protected boolean reversed()
080        {
081                return proficiency() == 100;
082        }
083
084        protected boolean isAirRoom(Room R)
085        {
086                if(R==null)
087                        return false;
088                if((R.domainType()==Room.DOMAIN_INDOORS_AIR)
089                ||(R.domainType()==Room.DOMAIN_OUTDOORS_AIR))
090                        return true;
091                return false;
092        }
093
094        protected boolean canFallFrom(Room fromHere, int direction)
095        {
096                if((fromHere==null)||(direction<0)||(direction>=Directions.NUM_DIRECTIONS()))
097                        return false;
098
099                final Room toHere=fromHere.getRoomInDir(direction);
100                if((toHere==null)
101                ||(fromHere.getExitInDir(direction)==null)
102                ||(!fromHere.getExitInDir(direction).isOpen()))
103                        return false;
104                if(CMLib.flags().isWaterySurfaceRoom(fromHere)&&CMLib.flags().isUnderWateryRoom(toHere))
105                        return false;
106                return true;
107        }
108
109        @Override
110        public void setMiscText(String newMiscText)
111        {
112                super.setMiscText(newMiscText);
113                if((newMiscText!=null) && (newMiscText.length()>0))
114                {
115                        for(final String parm : CMParms.parse(newMiscText.toUpperCase()))
116                        {
117                                if(parm.equals("REVERSED"))
118                                        this.setProficiency(100);
119                                else
120                                if(parm.equals("NORMAL"))
121                                        this.setProficiency(0);
122                        }
123                }
124        }
125        
126        protected boolean stopFalling(MOB mob)
127        {
128                final Room R=mob.location();
129                if(reversed())
130                {
131                        if(!hitTheCeiling)
132                        {
133                                hitTheCeiling=true;
134                                if(R!=null)
135                                        R.show(mob,null,CMMsg.MSG_OK_ACTION,L("<S-NAME> hit(s) the ceiling.@x1",CMLib.protocol().msp("splat.wav",50)));
136                                CMLib.combat().postDamage(mob,mob,this,damageToTake,CMMsg.MASK_ALWAYS|CMMsg.TYP_JUSTICE,-1,null);
137                        }
138                        return true;
139                }
140                hitTheCeiling=false;
141                unInvoke();
142                if(R!=null)
143                {
144                        if(isAirRoom(R))
145                                R.show(mob,null,CMMsg.MSG_OK_ACTION,L("<S-NAME> stop(s) falling.@x1",CMLib.protocol().msp("splat.wav",50)));
146                        else
147                        if(CMLib.flags().isWaterySurfaceRoom(R)||CMLib.flags().isUnderWateryRoom(R))
148                                R.show(mob,null,CMMsg.MSG_OK_ACTION,L("<S-NAME> hit(s) the water.@x1",CMLib.protocol().msp("splat.wav",50)));
149                        else
150                        {
151                                R.show(mob,null,CMMsg.MSG_OK_ACTION,L("<S-NAME> hit(s) the ground.@x1",CMLib.protocol().msp("splat.wav",50)));
152                                if(CMath.div(damageToTake, mob.maxState().getHitPoints())>0.1)
153                                {
154                                        LimbDamage damage = (LimbDamage)mob.fetchEffect("BrokenLimbs");
155                                        if(damage == null)
156                                        {
157                                                damage = (LimbDamage)CMClass.getAbility("BrokenLimbs");
158                                                damage.setAffectedOne(mob);
159                                        }
160                                        List<String> limbs = damage.unaffectedLimbSet();
161                                        if(limbs.size()>0)
162                                        {
163                                                if(mob.fetchEffect(damage.ID()) == null)
164                                                {
165                                                        mob.addEffect(damage);
166                                                        damage.makeLongLasting();
167                                                }
168                                                damage.damageLimb(limbs.get(CMLib.dice().roll(1, limbs.size(), -1)));
169                                        }
170                                }
171                        }
172                        CMLib.combat().postDamage(mob,mob,this,damageToTake,CMMsg.MASK_ALWAYS|CMMsg.TYP_JUSTICE,-1,null);
173                }
174                mob.delEffect(this);
175                return false;
176        }

178        @Override
179        public boolean tick(final Tickable ticking, final int tickID)
180        {
181                if(!super.tick(ticking,tickID))
182                        return false;
183
184                if(tickID!=Tickable.TICKID_MOB)
185                        return true;
186
187                if(affected==null)
188                        return false;
189                if(--fallTickDown>0)
190                        return true;
191                fallTickDown=1;
192
193                int direction=Directions.DOWN;
194                String addStr=L("down");
195                if(reversed())
196                {
197                        direction=Directions.UP;
198                        addStr=L("upwards");
199                }
200                if(affected instanceof MOB)
201                {
202                        final MOB mob=(MOB)affected;
203                        if(mob==null)
204                                return false;
205                        if(mob.location()==null)
206                                return false;
207
208                        if(CMLib.flags().isInFlight(mob))
209                        {
210                                damageToTake=0;
211                                unInvoke();
212                                return false;
213                        }
214                        else
215                        if(!canFallFrom(mob.location(),direction))
216                                return stopFalling(mob);
217                        else
218                        {
219                                if(mob.phyStats().weight()<1)
220                                {
221                                        mob.tell(L("\n\r\n\rYou are floating gently @x1.\n\r\n\r",addStr));
222                                }
223                                else
224                                {
225                                        mob.tell(L("\n\r\n\rYOU ARE FALLING @x1!!\n\r\n\r",addStr.toUpperCase()));
226                                        int damage = CMLib.dice().roll(1,(int)Math.round(CMath.mul(CMath.mul(mob.maxState().getHitPoints(),0.1),CMath.div(mob.baseWeight(),150.0))),0);
227                                        if(damage > (mob.maxState().getHitPoints()/3))
228                                                damage = (mob.maxState().getHitPoints()/3);
229                                        damageToTake=reversed()?damage:(damageToTake+damage);
230                                }
231                                temporarilyDisable=true;
232                                CMLib.tracking().walk(mob,direction,false,false);
233                                temporarilyDisable=false;
234                                if(!canFallFrom(mob.location(),direction))
235                                        return stopFalling(mob);
236                                return true;
237                        }
238                }
239                else
240                if(affected instanceof Item)
241                {
242                        final Item item=(Item)affected;
243                        if((room==null)
244                        &&(item.owner()!=null)
245                        &&(item.owner() instanceof Room))
246                                room=(Room)item.owner();
247
248                        if((room==null)
249                        ||((room!=null)&&(!room.isContent(item)))
250                        ||(!CMLib.flags().isGettable(item))
251                        ||(item.container()!=null)
252                        ||(CMLib.flags().isInFlight(item.ultimateContainer(null)))
253                        ||(room.getRoomInDir(direction)==null))
254                        {
255                                unInvoke();
256                                return false;
257                        }
258                        if(room.numItems()>100)
259                        {
260                                fallTickDown=CMLib.dice().roll(1,room.numItems()/50,0);
261                                if((--fallTickDown)>0)
262                                        return true;
263                        }
264                        final Room nextRoom=room.getRoomInDir(direction);
265                        if(canFallFrom(room,direction))
266                        {
267                                room.show(invoker,null,item,CMMsg.MSG_OK_ACTION,L("<O-NAME> falls @x1.",addStr));
268                                nextRoom.moveItemTo(item,ItemPossessor.Expire.Player_Drop);
269                                room=nextRoom;
270                                nextRoom.show(invoker,null,item,CMMsg.MSG_OK_ACTION,L("<O-NAME> falls in from @x1.",(reversed()?"below":"above")));
271                                return true;
272                        }
273                        if(reversed())
274                                return true;
275                        unInvoke();
276                        return false;
277                }
278
279                return false;
280        }

282        @Override
283        public boolean okMessage(final Environmental myHost, final CMMsg msg)
284        {
285                if(!super.okMessage(myHost,msg))
286                        return false;
287
288                if(temporarilyDisable)
289                        return true;
290                final MOB mob=msg.source();
291                if(affected instanceof MOB)
292                {
293                        if(msg.amISource((MOB)affected))
294                        {
295                                if(CMLib.flags().isInFlight(mob))
296                                {
297                                        damageToTake=0;
298                                        unInvoke();
299                                        return true;
300                                }
301                                if(msg.targetMajor(CMMsg.MASK_MOVE) 
302                                &&(!hitTheCeiling)
303                                &&(msg.sourceMinor()!=CMMsg.TYP_FLEE))
304                                {
305                                        msg.source().tell(L("You are too busy falling to do that right now."));
306                                        return false;
307                                }
308                        }
309                }
310                return true;
311        }

313        @Override
314        public void executeMsg(final Environmental myHost, final CMMsg msg)
315        {
316                super.executeMsg(myHost,msg);
317                if((affected!=null)
318                &&(affected instanceof MOB)
319                &&(msg.amISource((MOB)affected)))
320                {
321                        if(msg.sourceMinor()==CMMsg.TYP_RECALL)
322                        {
323                                damageToTake=0;
324                                unInvoke();
325                        }
326                        else
327                        if((msg.tool() instanceof Ability)
328                        &&(CMath.bset(((Ability)msg.tool()).flags(),Ability.FLAG_TRANSPORTING))
329                        &&(msg.sourceMinor()!=CMMsg.TYP_TEACH))
330                        {
331                                damageToTake=0;
332                                unInvoke();
333                        }
334                }
335        }

337        @Override
338        public void affectPhyStats(final Physical affected, final PhyStats affectableStats)
339        {
340                super.affectPhyStats(affected,affectableStats);
341                if((affectableStats.disposition()&PhyStats.IS_FLYING)==0)
342                        affectableStats.setDisposition(affectableStats.disposition()|PhyStats.IS_FALLING);
343        }
344
345        @Override
346        public void setAffectedOne(Physical P)
347        {
348                if(P instanceof Room)
349                        room=(Room)P;
350                else
351                        super.setAffectedOne(P);
352        }
353
354        @Override
355        public boolean invoke(MOB mob, List<String> commands, Physical target, boolean auto, int asLevel)
356        {
357                if(!auto)
358                        return false;
359                final Physical P=target;
360                if(P==null)
361                        return false;
362                if((P instanceof Item)&&(room==null))
363                        return false;
364                if(P.fetchEffect("Falling")==null)
365                {
366                        final Falling F=new Falling();
367                        F.setProficiency(proficiency());
368                        F.invoker=null;
369                        if(P instanceof MOB)
370                                F.invoker=(MOB)P;
371                        else
372                                F.invoker=CMClass.getMOB("StdMOB");
373                        F.setSavable(false);
374                        F.makeLongLasting();
375                        P.addEffect(F);
376                        if(!(P instanceof MOB))
377                                CMLib.threads().startTickDown(F,Tickable.TICKID_MOB,1);
378                        P.recoverPhyStats();
379
380                }
381                return true;
382        }

384        @Override
385        public void setStat(String code, String val)
386        {
387                if(code==null)
388                        return;
389                if(code.equalsIgnoreCase("DAMAGE"))
390                        this.damageToTake=CMath.s_int(val);
391                else
392                if(code.equalsIgnoreCase("REVERSED"))
393                        this.setProficiency(CMath.s_bool(val)?100:0);
394                else
395                if(code.equalsIgnoreCase("NORMAL"))
396                        this.setProficiency(CMath.s_bool(val)?0:100);
397                else
398                        super.setStat(code, val);
399        }
400        
401        @Override
402        public String getStat(String code)
403        {
404                if(code==null)
405                        return "";
406                if(code.equalsIgnoreCase("DAMAGE"))
407                        return ""+this.damageToTake;
408                else
409                if(code.equalsIgnoreCase("REVERSED"))
410                        return ""+(this.proficiency()==100);
411                else
412                if(code.equalsIgnoreCase("NORMAL"))
413                        return ""+(this.proficiency()==0);
414                else
415                        return super.getStat(code);
416        }
