058        @Override
059        public String displayText()
060        {
061                final List<String> affectedLimbs = affectedLimbNameSet();
062                if(affectedLimbs.size()==0)
063                        return "";
064                final List<String> limbConditionStrs = new XVector<String>(affectedLimbs);
065                for(int i=0;i<limbConditionStrs.size();i++)
066                {
067                        final String limb=limbConditionStrs.get(i);
068                        final int[] condition = setBones.get(limb);
069                        if(condition != null)
070                        {
071                                final double pct = CMath.div(condition[0], HEALING_TICKS) * 100.0;
072                                limbConditionStrs.set(i, limb + " (healing "+(100-Math.round(pct))+"%)");
073                        }
074                }
075                return "(Broken "+CMLib.english().toEnglishStringList(limbConditionStrs)+")";
076        }

146        @Override
147        public void executeMsg(final Environmental host, final CMMsg msg)
148        {
149                if(affected instanceof MOB)
150                {
151                        final MOB M = (MOB)affected;
152                        if((msg.target()==M)
153                        &&((msg.targetMinor()==CMMsg.TYP_LOOK)||(msg.targetMinor()==CMMsg.TYP_EXAMINE))
154                        &&(CMLib.flags().canBeSeenBy(M,msg.source())))
155                        {
156                                final String s=CMLib.utensils().niceCommaList(affectedLimbNameSet(),true);
157                                if(s.length()>0)
158                                        msg.addTrailerMsg(CMClass.getMsg(msg.source(),null,null,
159                                                                          CMMsg.MSG_OK_VISUAL,L("\n\r@x1 has broken @x2 @x3.\n\r",M.name(msg.source()),M.charStats().hisher(),s),
160                                                                          CMMsg.NO_EFFECT,null,
161                                                                          CMMsg.NO_EFFECT,null));
162                        }
163                        if((msg.sourceMinor()==CMMsg.TYP_DEATH)&&(msg.amISource(M)))
164                        {
165                                M.delEffect(this);
166                                M.recoverCharStats();
167                                M.recoverPhyStats();
168                                M.recoverMaxState();
169                        }
170                        if(msg.amITarget(M)&&(msg.targetMinor()==CMMsg.TYP_HEALING))
171                        {
172                                final int amount=msg.value();
173                                if((amount>0)
174                                &&(msg.tool() instanceof Ability)
175                                &&(CMath.bset(((Ability)msg.tool()).flags(),Ability.FLAG_HEALINGMAGIC))
176                                &&(!CMath.bset(((Ability)msg.tool()).flags(),Ability.FLAG_UNHOLY)))
177                                {
178                                        startAllLimbsHealing();
179                                        double healAmount = CMath.div(amount,M.maxState().getHitPoints());
180                                        if(healAmount < 0.05)
181                                                healAmount = 0.05;
182                                        progressLimbHealing(M, (int)Math.round(CMath.mul(HEALING_TICKS, healAmount)));
183                                }
184                        }
185                }
186                super.executeMsg(host,msg);
187        }

189        @Override
190        public void affectPhyStats(final Physical affected, final PhyStats affectableStats)
191        {
192                super.affectPhyStats(affected,affectableStats);
193                affectedLimbNameSet();
194                if(affected instanceof MOB)
195                {
196                        final MOB mob=(MOB)affected;
197                        if((brokenParts[Race.BODY_LEG]<0)&&(mob.getWearPositions(Wearable.WORN_LEGS)==0))
198                        {
199                                if((brokenParts[Race.BODY_ARM]<0)&&(mob.getWearPositions(Wearable.WORN_ARMS)==0))
200                                        affectableStats.setSensesMask(affectableStats.sensesMask()|PhyStats.CAN_NOT_MOVE);
201                                affectableStats.setDisposition(affectableStats.disposition()|PhyStats.IS_SITTING);
202                        }
203                }
204        }

239        @Override
240        public boolean okMessage(final Environmental host, final CMMsg msg)
241        {
242                if(!super.okMessage(host, msg))
243                        return false;
244                if(msg.source() == affected)
245                        switch(msg.sourceMinor())
246                        {
247                        case CMMsg.TYP_STAND:
248                                {
249                                        final MOB mob=msg.source();
250                                        if((brokenParts[Race.BODY_LEG]<0)&&(mob.getWearPositions(Wearable.WORN_LEGS)==0)&&(!CMLib.flags().isSleeping(mob)))
251                                        {
252                                                mob.tell(L("Your legs are broken!"));
253                                                return false;
254                                        }
255                                        break;
256                                }
257                        case CMMsg.TYP_ADVANCE:
258                        case CMMsg.TYP_RETREAT:
259                        {
260                                final MOB mob=msg.source();
261                                if((brokenParts[Race.BODY_LEG]<0)&&(!CMLib.flags().isSleeping(mob)))
262                                {
263                                        mob.tell(L("Your broken leg made movement too painful this time.  Try again."));
264                                        return false;
265                                }
266                                break;
267                        }
268                        case CMMsg.TYP_WIELD:
269                        {
270                                final MOB mob=msg.source();
271                                if((brokenParts[Race.BODY_HAND]<0)&&(mob.getWearPositions(Wearable.WORN_WIELD)==0))
272                                {
273                                        mob.tell(L("Your weapon hand is broken!"));
274                                        return false;
275                                }
276                                break;
277                        }
278                        case CMMsg.TYP_WEAPONATTACK:
279                        {
280                                final MOB mob=msg.source();
281                                if(((brokenParts[Race.BODY_HAND]<0)||(brokenParts[Race.BODY_ARM]<0))
282                                &&(msg.tool() == mob.fetchWieldedItem()))
283                                {
284                                        mob.tell(L("Your broken limbs make the attack impossible!"));
285                                        return false;
286                                }
287                                break;
288                        }
289                        }
290                return true;
291        }

293        @Override
294        public void unInvoke()
295        {
296                // undo the affects of this spell
297                if(!(affected instanceof MOB))
298                        return;
299                final MOB mob=(MOB)affected;
300
301                if((mob.isMonster())&&(mob.amDead())&&(!canBeUninvoked()))
302                        super.canBeUninvoked=true;
303
304                super.unInvoke();
305
306                if(canBeUninvoked())
307                        if((mob.location()!=null)&&(!mob.amDead()))
308                                mob.location().show(mob,null,CMMsg.MSG_OK_VISUAL,L("<S-YOUPOSS> broken limbs have been healed."));
309        }

311        public MOB getTarget(final MOB mob, final List<String> commands, final Environmental givenTarget, final boolean quiet)
312        {
313                String targetName=CMParms.combine(commands,0);
314                MOB target=null;
315                if((givenTarget!=null)&&(givenTarget instanceof MOB))
316                        target=(MOB)givenTarget;
317                else
318                if((targetName.length()==0)&&(mob.isInCombat())&&(abstractQuality()==Ability.QUALITY_MALICIOUS)&&(mob.getVictim()!=null))
319                        target=mob.getVictim();
320                else
321                if((targetName.length()==0)&&(abstractQuality()!=Ability.QUALITY_MALICIOUS))
322                        target=mob;
323                else
324                if(targetName.length()>0)
325                {
326                        target=mob.location().fetchInhabitant(targetName);
327                        if(target==null)
328                        {
329                                final Environmental t=mob.location().fetchFromRoomFavorItems(null,targetName);
330                                if((t!=null)&&(!(t instanceof MOB)))
331                                {
332                                        if(!quiet)
333                                                mob.tell(mob,target,null,L("You can't do that to <T-NAMESELF>."));
334                                        return null;
335                                }
336                        }
337                }
338
339                if(target!=null)
340                        targetName=target.name();
341
342                if((target==null)||((!CMLib.flags().canBeSeenBy(target,mob))&&((!CMLib.flags().canBeHeardMovingBy(target,mob))||(!target.isInCombat()))))
343                {
344                        if(!quiet)
345                        {
346                                if(targetName.trim().length()==0)
347                                        mob.tell(L("You don't see them here."));
348                                else
349                                        mob.tell(L("You don't see '@x1' here.",targetName));
350                        }
351                        return null;
352                }
353                return target;
354        }

380        @Override
381        public List<String> affectedLimbNameSet()
382        {
383                if(brokenLimbsList!=null)
384                        return brokenLimbsList;
385                brokenLimbsList=new Vector<String>();
386                if(affected==null)
387                        return brokenLimbsList;
388                if((!(affected instanceof MOB))&&(!(affected instanceof DeadBody)))
389                        return brokenLimbsList;
390                brokenLimbsList=CMParms.parseSemicolons(text(),true);
391                brokenParts=new int[Race.BODY_PARTS];
392                boolean right=false;
393                boolean left=false;
394                Integer code=null;
395                for(int v=0;v<brokenLimbsList.size();v++)
396                {
397                        final String s=brokenLimbsList.get(v).toUpperCase();
398                        left=s.startsWith("LEFT ");
399                        right=s.startsWith("RIGHT ");
400                        code=Race.BODYPARTHASH.get(s.substring(right?6:left?5:0).trim());
401                        if(code!=null)
402                                brokenParts[code.intValue()]--;
403                }
404                return brokenLimbsList;
405        }

407        public List<String> completeBrokenLimbNameSet(final Environmental E)
408        {
409                final Vector<String> V=new Vector<String>();
410                if(!(E instanceof MOB))
411                        return V;
412                final MOB M=(MOB)E;
413                final int[] limbs=M.charStats().getMyRace().bodyMask();
414                for(int i=0;i<limbs.length;i++)
415                {
416                        if((limbs[i]>0)&&(validBrokens[i]))
417                        {
418                                if(limbs[i]==1)
419                                        V.addElement(Race.BODYPARTSTR[i].toLowerCase());
420                                else
421                                if(limbs[i]==2)
422                                {
423                                        V.addElement("left "+Race.BODYPARTSTR[i].toLowerCase());
424                                        V.addElement("right "+Race.BODYPARTSTR[i].toLowerCase());
425                                }
426                                else
427                                for(int ii=0;ii<limbs[i];ii++)
428                                        V.addElement(Race.BODYPARTSTR[i].toLowerCase());
429                        }
430                }
431                return V;
432        }

434        protected void progressLimbHealing(final CMObject obj, final int thisMuch)
435        {
436                for(final String limb : setBones.keySet())
437                {
438                        final int[] ticksRemaining = setBones.get(limb);
439                        ticksRemaining[0] -= thisMuch;
440                        if(ticksRemaining[0] <= 0)
441                        {
442                                setBones.remove(limb);
443                                if(obj instanceof MOB)
444                                        restoreLimb(limb);
445                        }
446                }
447        }

459        @Override
460        public List<String> unaffectedLimbSet()
461        {
462                affectedLimbNameSet();
463                final List<String> remains=new Vector<String>();
464                if(!(affected instanceof MOB))
465                        return remains;
466                final MOB M=(MOB)affected;
467                final int[] limbs=new int[Race.BODY_PARTS];
468                final List<String> affected=affectedLimbNameSet();
469                for(int i=0;i<limbs.length;i++)
470                {
471                        limbs[i]=M.charStats().getBodyPart(i);
472                        if((limbs[i]>0)
473                        &&(validBrokens[i]))
474                        {
475                                if(brokenParts[i]>0)
476                                {
477                                        if(limbs[i]-brokenParts[i]==1)
478                                        {
479                                                if(!affected.contains(Race.BODYPARTSTR[i].toLowerCase()))
480                                                        remains.add(Race.BODYPARTSTR[i].toLowerCase());
481                                        }
482                                        else
483                                        if(limbs[i]-brokenParts[i]==2)
484                                        {
485                                                if(!affected.contains("left "+Race.BODYPARTSTR[i].toLowerCase()))
486                                                        remains.add("left "+Race.BODYPARTSTR[i].toLowerCase());
487                                                if(!affected.contains("right "+Race.BODYPARTSTR[i].toLowerCase()))
488                                                        remains.add("right "+Race.BODYPARTSTR[i].toLowerCase());
489                                        }
490                                        else
491                                        for(int ii=0;ii<limbs[i];ii++)
492                                                remains.add(Race.BODYPARTSTR[i].toLowerCase());
493                                }
494                                else
495                                for(int ii=0;ii<limbs[i];ii++)
496                                        remains.add(Race.BODYPARTSTR[i].toLowerCase());
497                        }
498                }
499                final Amputation A=(Amputation)M.fetchEffect("Amputation");
500                if(A!=null)
501                {
502                        final List<String> missing=A.affectedLimbNameSet();
503                        for(final String limb : missing)
504                                remains.remove(limb);
505                }
506                return remains;
507        }

509        @Override
510        public void restoreLimb(final String gone)
511        {
512                if (affected != null)
513                {
514                        if (affected instanceof MOB)
515                        {
516                                ((MOB)affected).location().show(((MOB)affected), null, CMMsg.MSG_OK_VISUAL, L("^G<S-YOUPOSS> @x1 is now mended.^?",gone));
517                        }
518                        else
519                        if ((affected instanceof DeadBody)
520                        && (((Item)affected).owner() instanceof Room))
521                        {
522                                ((Room)((Item)affected).owner()).showHappens(CMMsg.MSG_OK_VISUAL, L("^G@x1's @x2 is now mended.^?",affected.name(),gone));
523                        }
524                }
525
526                final List<String> theRest = affectedLimbNameSet();
527                if (theRest.contains(gone))
528                        theRest.remove(gone);
529                if((theRest.size()==0)&&(affected!=null))
530                        affected.delEffect(this);
531                else
532                {
533                        setMiscText(CMParms.combineWith(theRest, ';'));
534                }
535        }

560        @Override
561        public boolean invoke(final MOB mob, final List<String> commands, final Physical givenTarget, final boolean auto, final int asLevel)
562        {
563                String choice="";
564                if(givenTarget!=null)
565                {
566                        if((commands.size()>0)&&((commands.get(0)).equals(givenTarget.name())))
567                                commands.remove(0);
568                        choice=CMParms.combine(commands,0);
569                        commands.clear();
570                }
571                else
572                if(commands.size()>1)
573                {
574                        choice=CMParms.combine(commands,1);
575                        while(commands.size()>1)
576                                commands.remove(1);
577                }
578                if(choice.toUpperCase().startsWith("RIGHT "))
579                        choice=choice.substring(6).trim();
580                else
581                if(choice.toUpperCase().startsWith("LEFT "))
582                        choice=choice.substring(5).trim();
583                final MOB target=super.getTarget(mob,commands,givenTarget,false,true);
584                if(target==null)
585                        return false;
586                if(!auto)
587                {
588                        LegalBehavior B=null;
589                        if(mob.location()!=null)
590                                B=CMLib.law().getLegalBehavior(mob.location());
591                        List<LegalWarrant> warrants=new Vector<LegalWarrant>();
592                        if(B!=null)
593                                warrants=B.getWarrantsOf(CMLib.law().getLegalObject(mob.location()),target);
594                        if((warrants.size()==0)&&(!CMSecurity.isAllowed(mob,mob.location(),CMSecurity.SecFlag.ABOVELAW)))
595                        {
596                                mob.tell(L("You are not authorized by law to break anything on @x1 at this time.",target.Name()));
597                                return false;
598                        }
599                        final Item w=mob.fetchWieldedItem();
600                        if(!CMSecurity.isASysOp(mob))
601                        {
602                                Weapon ww=null;
603                                if((w==null)||(!(w instanceof Weapon)))
604                                {
605                                        mob.tell(L("You cannot break anything without a weapon!"));
606                                        return false;
607                                }
608                                ww=(Weapon)w;
609                                if((ww.weaponDamageType()==Weapon.TYPE_PIERCING)
610                                &&(ww.weaponDamageType()!=Weapon.TYPE_SLASHING)
611                                &&(ww.weaponDamageType()!=Weapon.TYPE_BASHING))
612                                {
613                                        mob.tell(L("You cannot break a limb with a @x1!",ww.name()));
614                                        return false;
615                                }
616                                if(mob.isInCombat()&&(mob.rangeToTarget()>0))
617                                {
618                                        mob.tell(L("You are too far away to try that!"));
619                                        return false;
620                                }
621                                if(!CMLib.flags().isBoundOrHeld(target))
622                                {
623                                        mob.tell(L("@x1 must be bound before you can break @x2 limbs.",target.charStats().HeShe(),target.charStats().hisher()));
624                                        return false;
625                                }
626                        }
627                }
628                if(!super.invoke(mob,commands,givenTarget,auto,asLevel))
629                        return false;
630                boolean success=proficiencyCheck(mob,0,auto);
631                if(success)
632                {
633                        final LimbDamage ampuA=(LimbDamage)target.fetchEffect("Amputation");
634                        final List<String> missingLimbs;
635                        if(ampuA != null)
636                                missingLimbs=ampuA.affectedLimbNameSet();
637                        else
638                                missingLimbs=new ArrayList<String>(1);
639
640                        BrokenLimbs brokenA=(BrokenLimbs)target.fetchEffect(ID());
641                        boolean newOne=false;
642                        if(brokenA==null)
643                        {
644                                brokenA=new BrokenLimbs();
645                                brokenA.setAffectedOne(target);
646                                newOne=true;
647                        }
648
649                        String brokeStr=null;
650                        final List<String> healthyLimbSet=brokenA.unaffectedLimbSet();
651                        for(final String missingLimb : missingLimbs)
652                                healthyLimbSet.remove(missingLimb);
653                        if(healthyLimbSet.size()==0)
654                        {
655                                if(!auto)
656                                        mob.tell(L("There is nothing left on @x1 that is unbroken!",target.name(mob)));
657                                return false;
658                        }
659
660                        if(choice.length()>0)
661                        {
662                                for(int i=0;i<healthyLimbSet.size();i++)
663                                {
664                                        if(CMLib.english().containsString(healthyLimbSet.get(i),choice))
665                                        {
666                                                brokeStr=healthyLimbSet.get(i);
667                                                break;
668                                        }
669                                }
670                                if(brokeStr==null)
671                                {
672                                        if(!auto)
673                                                mob.tell(L("There is nothing unbroken on @x1 called '@x2'!",target.name(mob),choice));
674                                        return false;
675                                }
676                        }
677
678                        if(brokeStr==null)
679                                brokeStr=healthyLimbSet.get(CMLib.dice().roll(1,healthyLimbSet.size(),-1));
680
681                        final String brokeName = brokeStr;
682
683                        final String str=auto?"":L("^F^<FIGHT^><S-NAME> break(s) <T-YOUPOSS> @x1!^</FIGHT^>^?",brokeName);
684                        final CMMsg msg=CMClass.getMsg(mob,target,this,CMMsg.MSK_MALICIOUS_MOVE|CMMsg.TYP_DELICATE_HANDS_ACT|(auto?CMMsg.MASK_ALWAYS:0),str);
685                        CMLib.color().fixSourceFightColor(msg);
686                        if(target.location().okMessage(target,msg))
687                        {
688                                final MOB vic=target.getVictim();
689                                final MOB vic2=mob.getVictim();
690                                target.location().send(target,msg);
691                                if(msg.value()<=0)
692                                {
693                                        if(newOne)
694                                        {
695                                                brokenA.makeLongLasting();
696                                                target.addEffect(brokenA);
697                                        }
698                                        brokenA.damageLimb(brokeStr);
699                                        target.recoverCharStats();
700                                        target.recoverPhyStats();
701                                        target.recoverMaxState();
702                                        target.location().recoverRoomStats();
703                                        target.setVictim(vic);
704                                        mob.setVictim(vic2);
705                                        success=true;
706                                }
707                        }
708                        else
709                                success=false;
710                }
711                else
712                        return maliciousFizzle(mob,target,L("<S-NAME> attempt(s) to break one of <T-YOUPOSS> limbs, but fail(s)."));
713                return success;
714        }
