110        @Override
111        public boolean tick(final Tickable ticking, final int tickID)
112        {
113                if(!super.tick(ticking,tickID))
114                        return false;
115                if(tickID==Tickable.TICKID_MOB)
116                {
117                        if(nextDirection==-999)
118                                return true;
119
120                        if((theTrail==null)
121                        ||(affected == null)
122                        ||(!(affected instanceof MOB)))
123                                return false;
124
125                        final MOB mob=(MOB)affected;
126
127                        if(nextDirection==999)
128                        {
129                                mob.tell(waterHere(mob,mob.location(),null));
130                                nextDirection=-2;
131                                unInvoke();
132                        }
133                        else
134                        if(nextDirection==-1)
135                        {
136                                if(waterHere(mob,mob.location(),null).length()==0)
137                                        mob.tell(L("The water trail dries up here."));
138                                nextDirection=-999;
139                                unInvoke();
140                        }
141                        else
142                        if(nextDirection>=0)
143                        {
144                                mob.tell(L("The water trail seems to continue @x1.",CMLib.directions().getDirectionName(nextDirection)));
145                                if(mob.isMonster())
146                                {
147                                        final Room nextRoom=mob.location().getRoomInDir(nextDirection);
148                                        if((nextRoom!=null)&&(nextRoom.getArea()==mob.location().getArea()))
149                                        {
150                                                final int dir=nextDirection;
151                                                nextDirection=-2;
152                                                CMLib.tracking().walk(mob,dir,false,false);
153                                        }
154                                        else
155                                                unInvoke();
156                                }
157                                else
158                                        nextDirection=-2;
159                        }
160
161                }
162                return true;
163        }

165        @Override
166        public void executeMsg(final Environmental myHost, final CMMsg msg)
167        {
168                super.executeMsg(myHost,msg);
169
170                if(!(affected instanceof MOB))
171                        return;
172
173                final MOB mob=(MOB)affected;
174                if((msg.amISource(mob))
175                &&(msg.amITarget(mob.location()))
176                &&(CMLib.flags().canBeSeenBy(mob.location(),mob))
177                &&(msg.targetMinor()==CMMsg.TYP_LOOK))
178                        nextDirection=CMLib.tracking().trackNextDirectionFromHere(theTrail,mob.location(),false);
179                else
180                if((affected!=null)
181                   &&(affected instanceof MOB)
182                   &&(msg.target()!=null)
183                   &&(msg.amISource((MOB)affected))
184                   &&((msg.sourceMinor()==CMMsg.TYP_LOOK)||(msg.sourceMinor()==CMMsg.TYP_EXAMINE)))
185                {
186                        if((msg.tool()!=null)&&(msg.tool().ID().equals(ID())))
187                        {
188                                final String str=waterHere((MOB)affected,msg.target(),null);
189                                if(str.length()>0)
190                                        ((MOB)affected).tell(str);
191                        }
192                        else
193                        if((msg.target()!=null)
194                        &&(waterHere((MOB)affected,msg.target(),null).length()>0)
195                        &&(msg.source()!=msg.target()))
196                        {
197                                final CMMsg msg2=CMClass.getMsg(msg.source(),msg.target(),this,CMMsg.MSG_LOOK,CMMsg.NO_EFFECT,CMMsg.NO_EFFECT,null);
198                                msg.addTrailerMsg(msg2);
199                        }
200                }
201        }

203        public String waterCheck(MOB mob, Item I, Item container, StringBuffer msg)
204        {
205                if(I==null)
206                        return "";
207                if(I.container()==container)
208                {
209                        if(((I instanceof Drink))
210                        &&(((Drink)I).containsDrink())
211                        &&(CMLib.flags().canBeSeenBy(I,mob)))
212                                msg.append(L("@x1 contains some sort of liquid.\n\r",I.name()));
213                }
214                else
215                if((I.container()!=null)&&(I.container().container()==container))
216                {
217                        if(msg.toString().indexOf(I.container().name()+" contains some sort of liquid.")<0)
218                                msg.append(L("@x1 contains some sort of liquid.\n\r",I.container().name()));
219                }
220                return msg.toString();
221        }

230        public String waterHere(MOB mob, Environmental E, Item container)
231        {
232                final StringBuffer msg=new StringBuffer("");
233                if(E==null)
234                        return msg.toString();
235                if((E instanceof Room)&&(CMLib.flags().canBeSeenBy(E,mob)))
236                {
237                        final Room room=(Room)E;
238                        if(CMLib.flags().isWateryRoom(room))
239                                msg.append(L("Your water-finding senses are saturated.  This is a very wet place.\n\r"));
240                        else
241                        if(CMath.bset(room.getClimateType(),Places.CLIMASK_WET))
242                                msg.append(L("Your water-finding senses are saturated.  This is a damp place.\n\r"));
243                        else
244                        if((room.getArea().getClimateObj().weatherType(room)==Climate.WEATHER_RAIN)
245                        ||(room.getArea().getClimateObj().weatherType(room)==Climate.WEATHER_THUNDERSTORM))
246                                msg.append(L("It is raining here! Your water-finding senses are saturated!\n\r"));
247                        else
248                        if(room.getArea().getClimateObj().weatherType(room)==Climate.WEATHER_HAIL)
249                                msg.append(L("It is hailing here! Your water-finding senses are saturated!\n\r"));
250                        else
251                        if(room.getArea().getClimateObj().weatherType(room)==Climate.WEATHER_SNOW)
252                                msg.append(L("It is snowing here! Your water-finding senses are saturated!\n\r"));
253                        else
254                        {
255                                for(int i=0;i<room.numItems();i++)
256                                {
257                                        final Item I=room.getItem(i);
258                                        waterCheck(mob,I,container,msg);
259                                }
260                                for(int m=0;m<room.numInhabitants();m++)
261                                {
262                                        final MOB M=room.fetchInhabitant(m);
263                                        if((M!=null)&&(M!=mob))
264                                                msg.append(waterHere(mob,M,null));
265                                }
266                        }
267                }
268                else
269                if((E instanceof Item)&&(CMLib.flags().canBeSeenBy(E,mob)))
270                {
271                        waterCheck(mob,(Item)E,container,msg);
272                        msg.append(waterHere(mob,((Item)E).owner(),(Item)E));
273                }
274                else
275                if((E instanceof MOB)&&(CMLib.flags().canBeSeenBy(E,mob)))
276                {
277                        for(int i=0;i<((MOB)E).numItems();i++)
278                        {
279                                final Item I=((MOB)E).getItem(i);
280                                final StringBuffer msg2=new StringBuffer("");
281                                waterCheck(mob,I,container,msg2);
282                                if(msg2.length()>0)
283                                        return L("@x1 is carrying some liquids.",E.name());
284                        }
285                        final ShopKeeper SK=CMLib.coffeeShops().getShopKeeper(E);
286                        if(SK!=null)
287                        {
288                                final StringBuffer msg2=new StringBuffer("");
289                                for(final Iterator<Environmental> i=SK.getShop().getStoreInventory();i.hasNext();)
290                                {
291                                        final Environmental E2=i.next();
292                                        if(E2 instanceof Item)
293                                                waterCheck(mob,(Item)E2,container,msg2);
294                                        if(msg2.length()>0)
295                                                return L("@x1 has some liquids in stock.",E.name());
296                                }
297                        }
298                }
299                return msg.toString();
300        }

302        @Override
303        public boolean invoke(MOB mob, List<String> commands, Physical givenTarget, boolean auto, int asLevel)
304        {
305                final List<Ability> V=CMLib.flags().flaggedAffects(mob,Ability.FLAG_TRACKING);
306                for(final Ability A : V) A.unInvoke();
307                if(V.size()>0)
308                {
309                        mob.tell(L("You stop tracking."));
310                        if(commands.size()==0)
311                                return true;
312                }
313
314                if(!super.invoke(mob,commands,givenTarget,auto,asLevel))
315                        return false;
316
317                final String here=waterHere(mob,mob.location(),null);
318                if(here.length()>0)
319                {
320                        mob.tell(here);
321                        return true;
322                }
323
324                final boolean success=proficiencyCheck(mob,0,auto);
325
326                final ArrayList<Room> rooms=new ArrayList<Room>();
327                TrackingLibrary.TrackingFlags flags;
328                flags = CMLib.tracking().newFlags()
329                                .plus(TrackingLibrary.TrackingFlag.NOEMPTYGRIDS)
330                                .plus(TrackingLibrary.TrackingFlag.NOAIR);
331                int range=60 + (2*super.getXLEVELLevel(mob))+(10*super.getXMAXRANGELevel(mob));
332                final List<Room> checkSet=CMLib.tracking().getRadiantRooms(mob.location(),flags,range);
333                for (final Room room : checkSet)
334                {
335                        final Room R=CMLib.map().getRoom(room);
336                        if(waterHere(mob,R,null).length()>0)
337                                rooms.add(R);
338                }
339
340                if(rooms.size()>0)
341                        theTrail=CMLib.tracking().findTrailToAnyRoom(mob.location(),rooms,flags,range);
342
343                if((success)&&(theTrail!=null))
344                {
345                        final CMMsg msg=CMClass.getMsg(mob,null,this,CMMsg.MSG_QUIETMOVEMENT,auto?L("<S-NAME> begin(s) sniffing around for water!"):L("<S-NAME> begin(s) sensing water."));
346                        if(mob.location().okMessage(mob,msg))
347                        {
348                                mob.location().send(mob,msg);
349                                final Ranger_FindWater newOne=(Ranger_FindWater)this.copyOf();
350                                if(mob.fetchEffect(newOne.ID())==null)
351                                        mob.addEffect(newOne);
352                                mob.recoverPhyStats();
353                                newOne.nextDirection=CMLib.tracking().trackNextDirectionFromHere(newOne.theTrail,mob.location(),false);
354                        }
355                }
356                else
357                        beneficialVisualFizzle(mob,null,L("<S-NAME> attempt(s) to find water, but fail(s)."));
358
359                return success;
360        }
