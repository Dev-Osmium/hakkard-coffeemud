038public class Ranger_WoodlandCreep extends StdAbility
039{
040        @Override
041        public String ID()
042        {
043                return "Ranger_WoodlandCreep";
044        }
045
046        private final static String localizedName = CMLib.lang().L("Woodland Creep");
047
048        @Override
049        public String name()
050        {
051                return localizedName;
052        }
053
054        private final static String localizedStaticDisplay = CMLib.lang().L("(Creeping through foliage)");
055
056        @Override
057        public String displayText()
058        {
059                return localizedStaticDisplay;
060        }
061
062        @Override
063        protected int canAffectCode()
064        {
065                return CAN_MOBS;
066        }
067
068        @Override
069        protected int canTargetCode()
070        {
071                return 0;
072        }
073
074        @Override
075        public int abstractQuality()
076        {
077                return Ability.QUALITY_OK_SELF;
078        }
079
080        private static final String[] triggerStrings =I(new String[] {"WCREEP"});
081        @Override
082        public String[] triggerStrings()
083        {
084                return triggerStrings;
085        }
086
087        @Override
088        public int classificationCode()
089        {
090                return Ability.ACODE_SKILL|Ability.DOMAIN_STEALTHY;
091        }
092
093        @Override
094        public int usageType()
095        {
096                return USAGE_MOVEMENT|USAGE_MANA;
097        }
098
099        protected int bonus=0;
100
101        @Override
102        public void affectPhyStats(final Physical affected, final PhyStats affectableStats)
103        {
104                super.affectPhyStats(affected,affectableStats);
105                affectableStats.setDisposition(affectableStats.disposition()|PhyStats.IS_HIDDEN);
106                affectableStats.setDisposition(affectableStats.disposition()|PhyStats.IS_SNEAKING);
107                affectableStats.setSpeed(0.5);
108        }
109
110        @Override
111        public void affectCharStats(final MOB affected, final CharStats affectableStats)
112        {
113                super.affectCharStats(affected,affectableStats);
114                affectableStats.setStat(CharStats.STAT_SAVE_DETECTION,proficiency()+25+bonus+affectableStats.getStat(CharStats.STAT_SAVE_DETECTION));
115        }
116
117        @Override
118        public void executeMsg(final Environmental myHost, final CMMsg msg)
119        {
120                if(!(affected instanceof MOB))
121                        return;
122
123                final MOB mob=(MOB)affected;
124
125                if((mob.location()!=null)
126                &&(mob.location().domainType()!=Room.DOMAIN_OUTDOORS_JUNGLE)
127                &&(mob.location().domainType()!=Room.DOMAIN_OUTDOORS_SWAMP)
128                &&(mob.location().domainType()!=Room.DOMAIN_OUTDOORS_WOODS))
129                {
130                        unInvoke();
131                        mob.recoverPhyStats();
132                }
133                if((msg.source()==affected)
134                &&(msg.sourceMajor(CMMsg.MASK_MALICIOUS))
135                &&(msg.source().isInCombat())
136                &&(msg.source().rangeToTarget()<=0))
137                {
138                        unInvoke();
139                        mob.recoverPhyStats();
140                }
141                return;
142        }
143
144        @Override
145        public boolean invoke(MOB mob, List<String> commands, Physical givenTarget, boolean auto, int asLevel)
146        {
147                if(!super.invoke(mob,commands,givenTarget,auto,asLevel))
148                        return false;
149
150                if(mob.fetchEffect(this.ID())!=null)
151                {
152                        mob.tell(L("You are already creeping around."));
153                        return false;
154                }
155
156                if(mob.isInCombat())
157                {
158                        mob.tell(L("Not while in combat!"));
159                        return false;
160                }
161
162                if((mob.location().domainType()!=Room.DOMAIN_OUTDOORS_JUNGLE)
163                &&(mob.location().domainType()!=Room.DOMAIN_OUTDOORS_SWAMP)
164                &&(mob.location().domainType()!=Room.DOMAIN_OUTDOORS_WOODS)
165                &&(!auto))
166                {
167                        mob.tell(L("You don't know how to creep around in a place like this."));
168                        return false;
169                }
170
171                final String str=L("You creep into some foliage.");
172                boolean success=proficiencyCheck(mob,0,auto);
173
174                if(!success)
175                        beneficialVisualFizzle(mob,null,L("<S-NAME> attempt(s) to creep into the foliage and fail(s)."));
176                else
177                {
178                        final CMMsg msg=CMClass.getMsg(mob,null,this,auto?CMMsg.MSG_OK_ACTION:(CMMsg.MSG_DELICATE_HANDS_ACT|CMMsg.MASK_MOVE),str,CMMsg.NO_EFFECT,null,CMMsg.NO_EFFECT,null);
179                        if(mob.location().okMessage(mob,msg))
180                        {
181                                mob.location().send(mob,msg);
182                                invoker=mob;
183                                beneficialAffect(mob, mob, asLevel, 0);
184                                final Ranger_WoodlandCreep newOne=(Ranger_WoodlandCreep)mob.fetchEffect(ID());
185                                newOne.bonus=getXLEVELLevel(mob)*2;
186                                newOne.makeLongLasting();
187                                mob.recoverPhyStats();
188                        }
189                        else
190                                success=false;
191                }
192                return success;
193        }
194}
