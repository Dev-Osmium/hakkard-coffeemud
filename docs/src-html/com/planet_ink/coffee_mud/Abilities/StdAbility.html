212        @Override
213        public ExpertiseLibrary.SkillCost getTrainingCost(final MOB mob)
214        {
215                int qualifyingLevel;
216                int playerLevel=1;
217                if(mob!=null)
218                {
219                        final Integer[] O=CMLib.ableMapper().getCostOverrides(mob,ID());
220                        if(O!=null)
221                        {
222                                Integer val=O[AbilityMapper.Cost.TRAIN.ordinal()];
223                                if(val!=null)
224                                        return CMLib.expertises().createNewSkillCost(ExpertiseLibrary.CostType.TRAIN,Double.valueOf(val.intValue()));
225                                val=O[AbilityMapper.Cost.PRAC.ordinal()];
226                                if(val!=null)
227                                        return CMLib.expertises().createNewSkillCost(ExpertiseLibrary.CostType.PRACTICE,Double.valueOf(val.intValue()));
228                        }
229                        qualifyingLevel=CMLib.ableMapper().qualifyingLevel(mob, this);
230                        playerLevel=mob.basePhyStats().level();
231                }
232                else
233                {
234                        qualifyingLevel=CMLib.ableMapper().lowestQualifyingLevel(ID());
235                        playerLevel=1;
236                }
237                if(qualifyingLevel<=0)
238                        qualifyingLevel=1;
239                final ExpertiseLibrary.SkillCostDefinition rawCost=getRawTrainingCost();
240                if(rawCost==null)
241                        return CMLib.expertises().createNewSkillCost(ExpertiseLibrary.CostType.TRAIN,Double.valueOf(1.0));
242                final double[] vars=new double[]{ qualifyingLevel,playerLevel};
243                final double value=CMath.parseMathExpression(rawCost.costDefinition(),vars);
244                return CMLib.expertises().createNewSkillCost(rawCost.type(),Double.valueOf(value));
245        }

331        protected int castingQuality(final MOB mob, final Physical target, final int abstractQuality)
332        {
333                if((target!=null)&&(target.fetchEffect(ID())!=null))
334                        return Ability.QUALITY_INDIFFERENT;
335                if(isAutoInvoked())
336                        return Ability.QUALITY_INDIFFERENT;
337                if((mob!=null)&&(target!=null)&&(mob.getVictim()==target))
338                {
339                        if((minRange()>0)&&(mob.rangeToTarget()<minRange()))
340                                return Ability.QUALITY_INDIFFERENT;
341                        if(mob.rangeToTarget()>maxRange())
342                                return Ability.QUALITY_INDIFFERENT;
343
344                }
345                switch(abstractQuality)
346                {
347                case Ability.QUALITY_BENEFICIAL_OTHERS:
348                        if(mob==target)
349                                return Ability.QUALITY_BENEFICIAL_SELF;
350                        return Ability.QUALITY_BENEFICIAL_OTHERS;
351                case Ability.QUALITY_MALICIOUS:
352                        return Ability.QUALITY_MALICIOUS;
353                case Ability.QUALITY_BENEFICIAL_SELF:
354                        if((target instanceof MOB)&&(mob!=target))
355                                return Ability.QUALITY_INDIFFERENT;
356                        return Ability.QUALITY_BENEFICIAL_SELF;
357                default:
358                        return Ability.QUALITY_INDIFFERENT;
359                }
360        }

604        @Override
605        public void startTickDown(final MOB invokerMOB, final Physical affected, int tickTime)
606        {
607                if(invokerMOB!=null)
608                        invoker=invokerMOB;
609
610                savable=false; // makes it so that the effect does not save!
611
612                if((classificationCode() != (Ability.ACODE_COMMON_SKILL|Ability.DOMAIN_CRAFTINGSKILL))
613                &&(classificationCode() != (Ability.ACODE_COMMON_SKILL|Ability.DOMAIN_GATHERINGSKILL))
614                &&(classificationCode() != (Ability.ACODE_COMMON_SKILL|Ability.DOMAIN_EPICUREAN))
615                &&(classificationCode() != (Ability.ACODE_COMMON_SKILL|Ability.DOMAIN_BUILDINGSKILL)))
616                        tickTime+=addedTickTime(invokerMOB,tickTime);
617                if(invoker()!=null)
618                {
619                        for(int c=0;c<invoker().charStats().numClasses();c++)
620                                tickTime=invoker().charStats().getMyClass(c).classDurationModifier(invoker(),this,tickTime);
621                }
622                if(affected instanceof MOB)
623                {
624                        final MOB mob=(MOB)affected;
625                        final Room room=mob.location();
626                        if(room==null)
627                                return;
628                        if(affected.fetchEffect(ID())==null)
629                                affected.addEffect(this);
630                        room.recoverRoomStats();
631                        if(invoker()!=affected)
632                        {
633                                for(int c=0;c<mob.charStats().numClasses();c++)
634                                        tickTime=mob.charStats().getMyClass(c).classDurationModifier(mob,this,tickTime);
635                        }
636                }
637                else
638                {
639                        if(affected.fetchEffect(this.ID())==null)
640                                affected.addEffect(this);
641
642                        if(affected instanceof Room)
643                                ((Room)affected).recoverRoomStats();
644                        else
645                                affected.recoverPhyStats();
646                        CMLib.threads().startTickDown(this,Tickable.TICKID_MOB,1);
647                }
648                tickDown=tickTime;
649        }

667        @Override
668        public int adjustedLevel(final MOB caster, final int asLevel)
669        {
670                if(caster==null)
671                        return 1;
672                final int lowestQualifyingLevel=CMLib.ableMapper().lowestQualifyingLevel(this.ID());
673                int adjLevel=lowestQualifyingLevel;
674                if(asLevel<=0)
675                {
676                        final int qualifyingLevel=CMLib.ableMapper().qualifyingLevel(caster,this);
677                        if((caster.isMonster())||(qualifyingLevel>=0))
678                                adjLevel+=(CMLib.ableMapper().qualifyingClassLevel(caster,this)-qualifyingLevel);
679                        else
680                                adjLevel=caster.phyStats().level()-lowestQualifyingLevel-25;
681                }
682                else
683                        adjLevel=asLevel;
684                if(adjLevel<lowestQualifyingLevel)
685                        adjLevel=lowestQualifyingLevel;
686                adjLevel += getPersonalLevelAdjustments(caster);
687                if(adjLevel<1)
688                        return 1;
689                int level=adjLevel+getXLEVELLevel(caster);
690                final CharStats CS=caster.charStats();
691                for(int c=0;c<CS.numClasses();c++)
692                        level=CS.getMyClass(c).classDurationModifier(invoker(),this,level);
693                return level;
694        }

696        protected int experienceLevels(final MOB caster, final int asLevel)
697        {
698                if(caster==null)
699                        return 1;
700                int adjLevel=1;
701                final int qualifyingLevel=CMLib.ableMapper().qualifyingLevel(caster,this);
702                final int lowestQualifyingLevel=CMLib.ableMapper().lowestQualifyingLevel(this.ID());
703                if(qualifyingLevel>=0)
704                {
705                        final int qualClassLevel=CMLib.ableMapper().qualifyingClassLevel(caster,this);
706                        if(qualClassLevel>=qualifyingLevel)
707                                adjLevel=(qualClassLevel-qualifyingLevel)+1;
708                        else
709                        if(caster.phyStats().level()>=qualifyingLevel)
710                                adjLevel=(caster.phyStats().level()-qualifyingLevel)+1;
711                        else
712                        if(caster.phyStats().level()>=lowestQualifyingLevel)
713                                adjLevel=(caster.phyStats().level()-lowestQualifyingLevel)+1;
714                }
715                else
716                if(caster.phyStats().level()>=lowestQualifyingLevel)
717                        adjLevel=(caster.phyStats().level()-lowestQualifyingLevel)+1;
718                if(asLevel>0)
719                        adjLevel=asLevel;
720                adjLevel += getPersonalLevelAdjustments(caster);
721                if(adjLevel<1)
722                        return 1;
723                return adjLevel+getXLEVELLevel(caster);
724        }
