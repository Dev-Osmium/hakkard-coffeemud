033/**
034 * TriggeredAffect is an ability interface to denote those properties
035 * that are typically non-removable inherent attributes of the things they affect.
036 */
037public interface TriggeredAffect extends Ability
038{
039        /** denotes a property whose affects are triggered always */
040        public final static int TRIGGER_ALWAYS=1;
041        /** denotes a property whose affects are triggered by entering the thing*/
042        public final static int TRIGGER_ENTER=2;
043        /** denotes a property whose affects are triggered by being hit by the thing*/
044        public final static int TRIGGER_BEING_HIT=4;
045        /** denotes a property whose affects are triggered by wearing/wielding the thing*/
046        public final static int TRIGGER_WEAR_WIELD=8;
047        /** denotes a property whose affects are triggered by getting the thing */
048        public final static int TRIGGER_GET=16;
049        /** denotes a property whose affects are triggered by using/eating/drinking the thing*/
050        public final static int TRIGGER_USE=32;
051        /** denotes a property whose affects are triggered by putting the thing somewhere*/
052        public final static int TRIGGER_PUT=64;
053        /** denotes a property whose affects are triggered by mounting the thing*/
054        public final static int TRIGGER_MOUNT=128;
055        /** denotes a property whose affects are triggered by putting something in or dropping the thing*/
056        public final static int TRIGGER_DROP_PUTIN=64;
057        /** denotes a property whose affects are triggered by hitting somethign with the thing*/
058        public final static int TRIGGER_HITTING_WITH=128;
059
060        /**
061         * This method returns a mask of TRIGGER_* constants denoting what triggers the properties
062         * @see TriggeredAffect#TRIGGER_ALWAYS
063         *
064         * @return  a mask of TRIGGER_* constants denoting what triggers the properties
065         */
066        public int triggerMask();
067}
