036/**
037 * This interface denotes an ability that also incidentally is capable
038 * of generating objects, usually items.  This is probably a common
039 * skill, but one could imagine an itemcraftor also being a spell that
040 * generates its own items.  Also the item generation is an incidental
041 * and internal aspect of the ability, these methods allow that
042 * functionality to be exposed for archon use.
043 */
044public interface ItemCraftor extends CraftorAbility
045{
046        /**
047         * Crafts a random item of a type supported by this class of
048         * the given resource code.
049         * Returns a vector containing the finished Item.  A second element is
050         * rare, but will occur when a key is required and also generated.
051         * @param material the rawmaterial code to make the item out of
052         * @return a vector of Item(s)
053         */
054        public ItemKeyPair craftAnyItem(int material);
055
056        /**
057         * Crafts every item of a type supported by this class of
058         * the given resource code.  Each finished item is represented
059         * as a vector in the returned vector. The item vector usually
060         * only contains the finished item, but a second element will
061         * occur when a key is required and also generated.
062         * @param material the rawmaterial code to make the item out of
063         * @param forceLevels forces crafted item to have a level if it otherwise doesn't
064         * @return a vector of vectors of item(s)
065         */
066        public List<ItemKeyPair> craftAllItemSets(int material, boolean forceLevels);
067
068        /**
069         * Crafts every item of a type supported by this class of
070         * every supported material.  Each finished item is represented
071         * as a vector in the returned vector. The item vector usually
072         * only contains the finished item, but a second element will
073         * occur when a key is required and also generated.
074         * @param forceLevels forces crafted item to have a level if it otherwise doesn't
075         * @return a vector of vectors of item vector(s)
076         */
077        public List<ItemKeyPair> craftAllItemSets(boolean forceLevels);
078
079        /**
080         * Crafts the item specified by the recipe name, of a supported
081         * material type which this class can produce.
082         * Returns a vector containing the finished Item.  A second element is
083         * rare, but will occur when a key is required and also generated.
084         * @param recipeName the name of the item to make
085         * @return a vector of Item(s)
086         */
087        public ItemKeyPair craftItem(String recipeName);
088
089        /**
090         * Crafts the item specified by the recipe name, of the specified
091         * material type which this class can produce or -1 for any material.
092         * Returns a vector containing the finished Item.  A second element is
093         * rare, but will occur when a key is required and also generated.
094         * @param recipeName the name of the item to make
095         * @param material the rawmaterial code to make the item out of, or -1
096         * @param forceLevels forces crafted item to have a level if it otherwise doesn't
097         * @param noSafety whether normal safeguards against creating broken items are overridden
098         * @return a vector of Item(s)
099         */
100        public ItemKeyPair craftItem(String recipeName, int material, boolean forceLevels, boolean noSafety);
101
102        /**
103         * For auto-crafting, this object represents an item,
104         * and (optionally) a key to go with it.
105         * @author Bo Zimmerman
106         */
107        public class ItemKeyPair
108        {
109                public Item item;
110                public DoorKey key;
111
112                public ItemKeyPair(Item item, DoorKey key)
113                {
114                        this.item = item;
115                        this.key = key;
116                }
117
118                public List<Item> asList()
119                {
120                        final List<Item> list = new LinkedList<Item>();
121                        if(item!=null)
122                                list.add(item);
123                        if(key != null)
124                                list.add(key);
125                        return list;
126                }
127
128        }
129
130        /**
131         * The base unlocalized brand added to crafted items
132         */
133        public final static String CRAFTING_BRAND_STR_PREFIX="This is the work of ";
134
135        /**
136         * The base unlocalized brand added to anonymously crafted items
137         */
138        public final static String CRAFTING_BRAND_STR_ANON=CRAFTING_BRAND_STR_PREFIX+"an anonymous craftsman.";
139
140        /**
141         * The base unlocalized brand added to named crafted items
142         */
143        public final static String CRAFTING_BRAND_STR_NAME=CRAFTING_BRAND_STR_PREFIX+"@x1.";
144
145        /**
146         * Returns whether the given item could have been crafted by this skill.
147         * @param I the item to examine
148         * @return true if the item is consistent with this crafting, or false otherwise
149         */
150        public boolean mayICraft(final Item I);
151
152        /**
153         * Returns true if mundane items can be demonstructed into recipes with this skill.
154         * @return true if mundane items can be demonstructed into recipes with this skill.
155         */
156        public boolean supportsDeconstruction();
157
158        /**
159         * Returns the ratio of the weight of material used to make an item with this
160         * skill versus the item weight when finished
161         * @param bundling true if the item being created is just a raw resource bundle
162         * @return the ratio of the weight of material used to make an item with this
163         */
164        public double getItemWeightMultiplier(boolean bundling);
165}
