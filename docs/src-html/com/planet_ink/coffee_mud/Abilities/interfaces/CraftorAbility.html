036/**
037 * This interface denotes an ability that also incidentally is capable
038 * of generating objects, usually items.  This is probably a common
039 * skill, but one could imagine an craftor also being a spell that
040 * generates its own items.  Also the generation is an incidental
041 * and internal aspect of the ability, these methods allow that
042 * functionality to be exposed for archon use.
043 */
044public interface CraftorAbility extends Ability
045{
046        /**
047         * A list containing an entry for each craftable recipe
048         * Each craftable recipe is also a list of strings.
049         * @return a vector of vectors
050         */
051        public List<List<String>> fetchRecipes();
052
053        /**
054         * A list containing an entry for each craftable recipe
055         * both standard, and extra recipes from given mob
056         * Each craftable recipe is also a list of strings.
057         * @param mob the mob to check for extra recipes
058         * @return a vector of vectors
059         */
060        public List<List<String>> fetchMyRecipes(final MOB mob);
061
062        /**
063         * A String containing the format of each entry in the parameter file
064         * in a recipe.
065         * @return a String showing the format of each entry in the parameter file
066         */
067        public String parametersFormat();
068
069        /**
070         * A String naming the file where the recipes are found
071         * @return a String naming the file where the recipes are found
072         */
073        public String parametersFile();
074
075        /**
076         * Returns a vector containing an entry for each craftable recipe
077         * whose name matches the given name.  Each entry is also a vector.
078         * @param recipeName the name of the recipe to craft
079         * @param beLoose whether to be specific or "loose" with name matching
080         * @return a vector of vectors
081         */
082        public List<List<String>> matchingRecipeNames(String recipeName, boolean beLoose);
083
084        /**
085         * Returns a list of Integer objects where each Integer
086         * is a fully qualified RawMaterial code.
087         * @see com.planet_ink.coffee_mud.Items.interfaces.RawMaterial
088         * @return a vector of integers
089         */
090        public List<Integer> myResources();
091
092        /**
093         * Given a raw recipe, returns a description of the required components to build it.
094         * @param mob the potential builder
095         * @param recipe the raw recipe description
096         * @return a descriptive string
097         */
098        public String getDecodedComponentsDescription(final MOB mob, final List<String> recipe);
099
100        /**
101         * Given a raw recipe, returns the raw name and level of the item built therefrom.
102         * @param recipe the raw recipe description
103         * @return a descriptive pair
104         */
105        public Pair<String,Integer> getDecodedItemNameAndLevel(final List<String> recipe);
106}
