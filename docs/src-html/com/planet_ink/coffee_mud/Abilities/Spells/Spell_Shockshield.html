036public class Spell_Shockshield extends Spell
037{
038
039        @Override
040        public String ID()
041        {
042                return "Spell_Shockshield";
043        }
044
045        private final static String localizedName = CMLib.lang().L("Shockshield");
046
047        @Override
048        public String name()
049        {
050                return localizedName;
051        }
052
053        private final static String localizedStaticDisplay = CMLib.lang().L("(Shockshield)");
054
055        @Override
056        public String displayText()
057        {
058                return localizedStaticDisplay;
059        }
060
061        @Override
062        public int abstractQuality()
063        {
064                return Ability.QUALITY_BENEFICIAL_OTHERS;
065        }
066
067        @Override
068        protected int canAffectCode()
069        {
070                return CAN_MOBS;
071        }
072
073        @Override
074        public int classificationCode()
075        {
076                return Ability.ACODE_SPELL|Ability.DOMAIN_EVOCATION;
077        }
078
079        @Override
080        public long flags()
081        {
082                return Ability.FLAG_AIRBASED;
083        }
084        final static String msgStr=CMLib.lang().L("The shock shield around <S-NAME> sparks and <DAMAGES> <T-NAME>!");
085        protected long oncePerTickTime=0;
086
087        @Override
088        public void unInvoke()
089        {
090                // undo the affects of this spell
091                if(!(affected instanceof MOB))
092                        return;
093                final MOB mob=(MOB)affected;
094
095                super.unInvoke();
096
097                if(canBeUninvoked())
098                        if((mob.location()!=null)&&(!mob.amDead()))
099                                mob.location().show(mob,null,CMMsg.MSG_OK_VISUAL,L("<S-YOUPOSS> electric shield fizzles out."));
100        }
101
102        @Override
103        public void executeMsg(final Environmental myHost, final CMMsg msg)
104        {
105                super.executeMsg(myHost,msg);
106                if(affected==null)
107                        return;
108                if(!(affected instanceof MOB))
109                        return;
110                final MOB mob=(MOB)affected;
111                if(msg.target()==null)
112                        return;
113                if(msg.source()==null)
114                        return;
115                final MOB source=msg.source();
116                if(source.location()==null)
117                        return;
118
119                if(msg.amITarget(mob)&&(!msg.amISource(mob)))
120                {
121                        if((CMath.bset(msg.targetMajor(),CMMsg.MASK_HANDS)||(msg.targetMajor(CMMsg.MASK_MOVE)))
122                                &&(msg.source().rangeToTarget()==0)
123                                &&(oncePerTickTime!=mob.lastTickedDateTime()))
124                        {
125                                if((CMLib.dice().rollPercentage()>(source.charStats().getStat(CharStats.STAT_DEXTERITY)*3)))
126                                {
127                                        final CMMsg msg2=CMClass.getMsg(mob,source,this,somanticCastCode(mob,source,true),null);
128                                        if(source.location().okMessage(source,msg2))
129                                        {
130                                                source.location().send(mob,msg2);
131                                                if(invoker==null)
132                                                        invoker=source;
133                                                if(msg2.value()<=0)
134                                                {
135                                                        final int damage = CMLib.dice().roll(1,(int)Math.round((invoker.phyStats().level()+super.getXLEVELLevel(invoker())+(2.0*super.getX1Level(invoker())))/3.0),1);
136                                                        CMLib.combat().postDamage(mob,source,this,damage,CMMsg.MASK_ALWAYS|CMMsg.MASK_MALICIOUS|CMMsg.TYP_ELECTRIC,Weapon.TYPE_STRIKING,msgStr);
137                                                }
138                                        }
139                                        oncePerTickTime=mob.lastTickedDateTime();
140                                }
141                        }
142                }
143                return;
144        }
145
146        @Override
147        public void affectPhyStats(final Physical affected, final PhyStats affectableStats)
148        {
149                super.affectPhyStats(affected,affectableStats);
150                if(affected==null)
151                        return;
152                if(!(affected instanceof MOB))
153                        return;
154                affectableStats.setArmor(affectableStats.armor()-(getXLEVELLevel(invoker())));
155        }
156
157        @Override
158        public boolean invoke(MOB mob, List<String> commands, Physical givenTarget, boolean auto, int asLevel)
159        {
160                final MOB target=this.getTarget(mob,commands,givenTarget);
161                if(target==null)
162                        return false;
163
164                if(!super.invoke(mob,commands,givenTarget,auto,asLevel))
165                        return false;
166
167                final boolean success=proficiencyCheck(mob,0,auto);
168
169                if(success)
170                {
171                        final CMMsg msg=CMClass.getMsg(mob,target,this,somanticCastCode(mob,target,auto),L((auto?"":"^S<S-NAME> incant(s) and wave(s) <S-HIS-HER> arms.  ")+"A field of sparks erupt(s) around <T-NAME>!^?")+CMLib.protocol().msp("lightning.wav",10));
172                        if(mob.location().okMessage(mob,msg))
173                        {
174                                mob.location().send(mob,msg);
175                                beneficialAffect(mob,target,asLevel,0);
176                        }
177                }
178                else
179                        return beneficialVisualFizzle(mob,target,L("<S-NAME> incant(s) and wave(s) <S-HIS-HER> arms, but only sparks emerge."));
180
181                // return whether it worked
182                return success;
183        }
184}
