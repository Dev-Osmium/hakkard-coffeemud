064        @Override
065        public String displayText()
066        {
067                if (spellbindings.size() == 0)
068                        return "";
069                final StringBuffer bindings = new StringBuffer("");
070                for (int i = 0; i < spellbindings.size(); i++)
071                        bindings.append(" " + ((String) spellbindings.elementAt(i, 1)));
072                return "(Bindings: " + bindings.toString() + ")";
073        }

105        protected DVector spellbindings=new DVector(2);
106        protected final static int COST_STATIC=50;
107
108        @Override
109        public void affectCharState(final MOB affected, final CharState affectableState)
110        {
111                super.affectCharState(affected,affectableState);
112                int total=0;
113                for(int i=0;i<spellbindings.size();i++)
114                {
115                        final DVector V=(DVector)spellbindings.elementAt(i,2);
116                        for(int x=0;x<V.size();x++)
117                                total+=((Integer)V.elementAt(x,2)).intValue();
118                }
119                total=(total+COST_STATIC)*spellbindings.size();
120                if(affectableState.getMana()>=total)
121                        affectableState.setMana(affectableState.getMana()-total);
122                else
123                        affectableState.setMana(0);
124        }
125
126        @Override
127        public String text()
128        {
129                if(spellbindings.size()==0)
130                        return super.text();
131                try
132                {
133                        final ByteArrayOutputStream bytes=new ByteArrayOutputStream();
134                        new ObjectOutputStream(bytes).writeObject(spellbindings);
135                        return CMParms.toSemicolonListString(bytes.toByteArray())+";";
136                }
137                catch(final Exception e)
138                {
139                        Log.errOut("Spell_Spellbinding","Spell bindings are corrupt for "+((affected!=null)?affected.Name():"someone")+".");
140                }
141                return super.text();
142        }
143
144        @Override
145        public void setMiscText(String text)
146        {
147                if(text.length()==0)
148                        spellbindings=new DVector(2);
149                else
150                {
151                        try
152                        {
153                                final ByteArrayInputStream bytes=new ByteArrayInputStream(CMParms.parseSemicolonByteList(text));
154                                spellbindings=(DVector)new ObjectInputStream(bytes).readObject();
155                        }
156                        catch(final Exception e)
157                        {
158                                Log.errOut("Spell_Spellbinding",e);
159                        }
160                }
161        }

163        @Override
164        public void executeMsg(final Environmental host, final CMMsg msg)
165        {
166                if((msg.source()==affected)
167                &&(msg.sourceMinor()==CMMsg.TYP_SPEAK)
168                &&(msg.sourceMessage()!=null)
169                &&(msg.sourceMessage().length()>0))
170                {
171                        final String s=CMStrings.getSayFromMessage(msg.sourceMessage());
172                        if(s!=null)
173                        {
174                                for(int v=0;v<spellbindings.size();v++)
175                                {
176                                        if(((String)spellbindings.elementAt(v,1)).equalsIgnoreCase(s))
177                                        {
178                                                boolean alreadyWanding=false;
179                                                final List<CMMsg> trailers =msg.trailerMsgs();
180                                                if(trailers!=null)
181                                                {
182                                                        for(final CMMsg msg2 : trailers)
183                                                        {
184                                                                if((msg2.targetMinor()==CMMsg.TYP_WAND_USE)
185                                                                &&(msg2.target() == msg.target()))
186                                                                        alreadyWanding=true;
187                                                        }
188                                                }
189                                                if(!alreadyWanding)
190                                                        msg.addTrailerMsg(CMClass.getMsg(msg.source(),msg.target(),this,CMMsg.MASK_ALWAYS|CMMsg.TYP_WAND_USE,L("The magic of '@x1' swells within you!",s),CMMsg.NO_EFFECT,null,CMMsg.NO_EFFECT,null));
191                                        }
192                                }
193                        }
194                }
195                else
196                if((msg.tool()==this)
197                &&(msg.source()==affected)
198                &&(msg.sourceMinor()==CMMsg.TYP_WAND_USE)
199                &&(msg.sourceMessage()!=null)
200                &&(msg.sourceMessage().length()>0))
201                {
202                        final String s=CMStrings.getSayFromMessage(msg.sourceMessage());
203                        if(s!=null)
204                        {
205                                for(int v=spellbindings.size()-1;v>=0;v--)
206                                {
207                                        if(((String)spellbindings.elementAt(v,1)).equalsIgnoreCase(s))
208                                        {
209                                                final DVector V2=(DVector)spellbindings.elementAt(v,2);
210                                                for(int v2=0;v2<V2.size();v2++)
211                                                {
212                                                        final Ability A=msg.source().fetchAbility((String)V2.elementAt(v2,1));
213                                                        final int curMana=msg.source().curState().getMana();
214                                                        msg.source().curState().setMana(1000);
215                                                        if(msg.target()!=null)
216                                                                A.invoke(msg.source(),CMParms.parse(msg.target().Name()),null,false,0);
217                                                        else
218                                                                A.invoke(msg.source(),new Vector<String>(),null,false,0);
219                                                        msg.source().curState().setMana(curMana);
220                                                }
221                                                if(canBeUninvoked())
222                                                        spellbindings.removeElementAt(v);
223                                        }
224                                }
225                        }
226                }
227                if((spellbindings.size()==0)&&(canBeUninvoked()))
228                        unInvoke();
229                super.executeMsg(host,msg);
230        }

232        @Override
233        public boolean invoke(MOB mob, List<String> commands, Physical givenTarget, boolean auto, int asLevel)
234        {
235                if(!super.invoke(mob,commands,givenTarget,auto,asLevel))
236                        return false;
237
238                final MOB target = mob;
239                Spell_Spellbinding priorBinding=(Spell_Spellbinding)target.fetchEffect(ID());
240                if(commands.size()<2)
241                {
242                        mob.tell(L("You must specify your trigger word, followed by a list of spells, seperated by spaces."));
243                        return false;
244                }
245                final String key=commands.get(0);
246                commands.remove(0);
247                final String combined=CMParms.combine(commands,0);
248                final DVector V=new DVector(2);
249                Ability A=mob.fetchAbility(combined);
250                if(A!=null)
251                {
252                        if(((A.classificationCode()&ALL_ACODES)!=ACODE_SPELL)
253                        ||(A.ID().equals(ID()))
254                        ||(A.usageCost(mob,true)[Ability.USAGEINDEX_MANA]>50))
255                        {
256                                mob.tell(L("You can't bind '@x1'.",A.ID()));
257                                return false;
258                        }
259                        V.addElement(A.ID(),Integer.valueOf(A.usageCost(mob,true)[Ability.USAGEINDEX_MANA]));
260                }
261                else
262                for(int v=0;v<commands.size();v++)
263                {
264                        A=mob.fetchAbility(commands.get(v));
265                        if((A==null)
266                        ||(A.ID().equals(ID()))
267                        ||((A.classificationCode()&ALL_ACODES)!=ACODE_SPELL)
268                        ||(A.usageCost(mob,true)[Ability.USAGEINDEX_MANA]>50))
269                        {
270                                mob.tell(L("You can't bind '@x1'.",(commands.get(v))));
271                                return false;
272                        }
273                        V.addElement(A.ID(),Integer.valueOf(A.usageCost(mob,true)[Ability.USAGEINDEX_MANA]));
274                }
275
276                int totalcost=0;
277                for(int v=0;v<V.size();v++)
278                        totalcost+=((Integer)V.elementAt(v,2)).intValue();
279                totalcost=(totalcost+COST_STATIC)*V.size();
280                final int curMana=mob.curState().getMana();
281                if(curMana<totalcost)
282                {
283                        mob.tell(L("You need @x1 mana to bind those spells.",""+totalcost));
284                        return false;
285                }
286                DVector thePriorKey=null;
287                if(priorBinding!=null)
288                {
289                        for(int x=0;x<priorBinding.spellbindings.size();x++)
290                        {
291                                if(((String)priorBinding.spellbindings.elementAt(x,1)).equalsIgnoreCase(key))
292                                {
293                                        thePriorKey = (DVector) priorBinding.spellbindings.elementAt(x, 2);
294                                }
295                        }
296                }
297
298                for(int v=0;v<V.size();v++)
299                {
300                        for(int v2=0;v2<V.size();v2++)
301                        {
302                                if((v!=v2)&&(((String)V.elementAt(v,1)).equals(V.elementAt(v2,1))))
303                                {
304                                        mob.tell(L("The same spell can not be bound to the same trigger more than once."));
305                                        return false;
306                                }
307                        }
308                }
309                if(thePriorKey!=null)
310                {
311                        for(int v=0;v<V.size();v++)
312                        {
313                                if(thePriorKey.contains(V.elementAt(v,1)))
314                                {
315                                        mob.tell(L("The same spell can not be bound to the same trigger more than once."));
316                                        return false;
317                                }
318                        }
319                }
320
321                final boolean success=proficiencyCheck(mob,0,auto);
322
323                if(mob.curState().getMana()>(curMana-totalcost))
324                        mob.curState().setMana(curMana-totalcost);
325
326                if(success)
327                {
328
329                        final CMMsg msg = CMClass.getMsg(mob, null, this, verbalCastCode(mob,target,auto),L(auto?"":"^S<S-NAME> shout(s) the magic of spellbinding!^?"));
330                        if(mob.location().okMessage(mob,msg))
331                        {
332                                mob.location().send(mob,msg);
333                                if(priorBinding==null)
334                                {
335                                        beneficialAffect(mob,target,asLevel,0);
336                                        priorBinding=(Spell_Spellbinding)target.fetchEffect(ID());
337                                        if(priorBinding==null)
338                                                return false;
339                                        priorBinding.makeLongLasting();
340                                }
341                                if(thePriorKey==null)
342                                        priorBinding.spellbindings.addElement(key,V);
343                                else
344                                for(int v=0;v<V.size();v++)
345                                        thePriorKey.addElement(V.elementAt(v,1),V.elementAt(v,2));
346                                target.recoverMaxState();
347                        }
348                }
349                else
350                        return beneficialWordsFizzle(mob,null,L("<S-NAME> whisper(s) about spellbinding, and the magic fizzles."));
351
352                // return whether it worked
353                return success;
354        }
