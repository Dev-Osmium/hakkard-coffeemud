036public class Spell_TimeStop extends Spell
037{
038
039        @Override
040        public String ID()
041        {
042                return "Spell_TimeStop";
043        }
044
045        private final static String localizedName = CMLib.lang().L("Time Stop");
046
047        @Override
048        public String name()
049        {
050                return localizedName;
051        }
052
053        private final static String localizedStaticDisplay = CMLib.lang().L("(Time is Stopped)");
054
055        @Override
056        public String displayText()
057        {
058                return localizedStaticDisplay;
059        }
060
061        @Override
062        protected int canAffectCode()
063        {
064                return CAN_ROOMS|CAN_MOBS;
065        }
066
067        @Override
068        protected int canTargetCode()
069        {
070                return 0;
071        }
072
073        @Override
074        protected int overrideMana()
075        {
076                return 100;
077        }
078
079        @Override
080        public int classificationCode()
081        {
082                return Ability.ACODE_SPELL|Ability.DOMAIN_ALTERATION;
083        }
084
085        @Override
086        public int abstractQuality()
087        {
088                return Ability.QUALITY_MALICIOUS;
089        }
090
091        protected Vector<MOB> fixed=new Vector<MOB>();
092
093        @Override
094        public void setMiscText(String newText)
095        {
096                super.setMiscText(newText);
097                fixed=new Vector<MOB>();
098        }
099
100        @Override
101        public void unInvoke()
102        {
103                // undo the affects of this spell
104                if((affected!=null)&&(canBeUninvoked()))
105                {
106                        if(affected instanceof Room)
107                        {
108                                final Room room=(Room)affected;
109                                room.showHappens(CMMsg.MSG_OK_VISUAL, L("Time starts moving again..."));
110                                if(invoker!=null)
111                                {
112                                        final Ability me=invoker.fetchEffect(ID());
113                                        if(me!=null)
114                                                me.unInvoke();
115                                }
116                                CMLib.threads().resumeTicking(room,-1);
117                                for(int i=0;i<fixed.size();i++)
118                                {
119                                        final MOB mob2=fixed.elementAt(i);
120                                        CMLib.threads().resumeTicking(mob2,-1);
121                                }
122                                fixed=new Vector<MOB>();
123                        }
124                        else
125                        if(affected instanceof MOB)
126                        {
127                                final MOB mob=(MOB)affected;
128                                CMLib.threads().resumeTicking(mob,-1);
129                                if(mob.location()!=null)
130                                {
131                                        mob.location().show(mob, null, CMMsg.MSG_OK_VISUAL, L("Time starts moving again..."));
132                                        final Ability me=mob.location().fetchEffect(ID());
133                                        if(me!=null)
134                                                me.unInvoke();
135                                }
136                        }
137                }
138                super.unInvoke();
139        }
140
141        @Override
142        public boolean okMessage(final Environmental myHost, final CMMsg msg)
143        {
144                if((affected!=null)
145                   &&(affected instanceof Room))
146                {
147                        switch(msg.targetMinor())
148                        {
149                        case CMMsg.TYP_ENTER:
150                        case CMMsg.TYP_LEAVE:
151                        case CMMsg.TYP_FLEE:
152                                if(msg.source()==invoker)
153                                        msg.source().tell(L("You cannot travel beyond the time stopped area."));
154                                else
155                                        msg.source().tell(L("Nothing just happened.  You didn't do that."));
156                                return false;
157                        default:
158                                if((msg.source() == invoker)
159                                &&(msg.target() != invoker)
160                                &&(msg.target() instanceof MOB)
161                                &&((CMath.bset(msg.sourceMajor(),CMMsg.MASK_MALICIOUS))
162                                ||(CMath.bset(msg.targetMajor(),CMMsg.MASK_MALICIOUS))
163                                ||(CMath.bset(msg.targetMajor(),CMMsg.MASK_MALICIOUS))))
164                                {
165                                        if(invoker.getVictim()==null)
166                                                invoker.setVictim((MOB)msg.target());
167                                }
168                                else
169                                if((msg.source()!=invoker)
170                                   &&(!CMath.bset(msg.sourceMajor(),CMMsg.MASK_ALWAYS))
171                                   &&(!CMath.bset(msg.targetMajor(),CMMsg.MASK_ALWAYS)))
172                                {
173                                        msg.source().tell(L("Time is stopped. Nothing just happened.  You didn't do that."));
174                                        return false;
175                                }
176                        }
177                }
178                return super.okMessage(myHost,msg);
179        }
180
181        @Override
182        public boolean invoke(MOB mob, List<String> commands, Physical givenTarget, boolean auto, int asLevel)
183        {
184                if(!super.invoke(mob,commands,givenTarget,auto,asLevel))
185                        return false;
186
187                final Physical target = mob.location();
188
189                if(target.fetchEffect(this.ID())!=null)
190                {
191                        mob.tell(mob,null,null,L("Time has already been stopped here!"));
192                        return false;
193                }
194
195                final boolean success=proficiencyCheck(mob,0,auto);
196
197                if(success)
198                {
199
200                        CMMsg msg = CMClass.getMsg(mob, target, this,somanticCastCode(mob,target,auto),L((auto?"T":"^S<S-NAME> speak(s) and gesture(s) and t")+"ime suddenly STOPS!^?"));
201                        if(mob.location().okMessage(mob,msg))
202                        {
203                                mob.location().send(mob,msg);
204                                if(msg.value()<=0)
205                                {
206                                        final Room room=mob.location();
207                                        fixed=new Vector<MOB>();
208                                        final Set<MOB> grpMembers = mob.getGroupMembers(new HashSet<MOB>());
209                                        for(int m=0;m<room.numInhabitants();m++)
210                                        {
211                                                final MOB mob2=room.fetchInhabitant(m);
212                                                if((mob2!=mob)&&(mob.mayIFight(mob2)))
213                                                {
214                                                        msg=CMClass.getMsg(mob,mob2,this,CMMsg.MASK_MALICIOUS|CMMsg.TYP_MIND,null);
215                                                        if(!grpMembers.contains(mob2))
216                                                        {
217                                                                if(room.okMessage(mob, msg))
218                                                                {
219                                                                        room.send(mob, msg);
220                                                                        if(msg.value()>0)
221                                                                                return false;
222                                                                }
223                                                                else
224                                                                        return beneficialVisualFizzle(mob,null,L("<S-NAME> incant(s) for awhile, but the spell fizzles."));
225                                                        }
226                                                }
227                                        }
228                                        CMLib.threads().suspendTicking(room,-1);
229                                        for(int m=0;m<room.numInhabitants();m++)
230                                        {
231                                                final MOB mob2=room.fetchInhabitant(m);
232                                                if(mob2!=mob)
233                                                {
234                                                        fixed.addElement(mob2);
235                                                        CMLib.threads().suspendTicking(mob2,-1);
236                                                }
237                                        }
238                                        beneficialAffect(mob,room,asLevel,2);
239                                }
240                        }
241                }
242                else
243                        return beneficialVisualFizzle(mob,null,L("<S-NAME> incant(s) for awhile, but the spell fizzles."));
244
245                // return whether it worked
246                return success;
247        }
