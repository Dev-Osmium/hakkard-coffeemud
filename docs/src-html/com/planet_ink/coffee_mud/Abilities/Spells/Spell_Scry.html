036public class Spell_Scry extends Spell
037{
038
039        @Override
040        public String ID()
041        {
042                return "Spell_Scry";
043        }
044
045        private final static String localizedName = CMLib.lang().L("Scry");
046
047        @Override
048        public String name()
049        {
050                return localizedName;
051        }
052
053        @Override
054        public String displayText()
055        {
056                return "";
057        }
058
059        @Override
060        public int enchantQuality()
061        {
062                return Ability.QUALITY_INDIFFERENT;
063        }
064
065        @Override
066        public int abstractQuality()
067        {
068                return Ability.QUALITY_INDIFFERENT;
069        }
070
071        @Override
072        protected int canAffectCode()
073        {
074                return CAN_MOBS;
075        }
076
077        @Override
078        public int classificationCode()
079        {
080                return Ability.ACODE_SPELL|Ability.DOMAIN_DIVINATION;
081        }
082
083        public static final DVector scries=new DVector(2);
084        private boolean recurse=false;
085
086        @Override
087        public void unInvoke()
088        {
089                // undo the affects of this spell
090                if(!(affected instanceof MOB))
091                        return;
092                final MOB mob=(MOB)affected;
093                if(canBeUninvoked())
094                        scries.removeElement(mob);
095                if((canBeUninvoked())&&(invoker!=null))
096                        invoker.tell(mob,null,null,L("Your knowledge of '<S-NAME>' fades."));
097                super.unInvoke();
098
099        }
100
101        @Override
102        public void executeMsg(final Environmental myHost, final CMMsg msg)
103        {
104                super.executeMsg(myHost,msg);
105                if((affected instanceof MOB)
106                &&(msg.amISource((MOB)affected))
107                &&((msg.sourceMinor()==CMMsg.TYP_LOOK)||(msg.sourceMinor()==CMMsg.TYP_EXAMINE))
108                &&(invoker!=null)
109                &&(msg.target()!=null)
110                &&((invoker.location()!=((MOB)affected).location())||(!(msg.target() instanceof Room)))
111                &&(!recurse))
112                {
113                        final CMMsg newAffect=CMClass.getMsg(invoker,msg.target(),msg.sourceMinor(),null);
114                        recurse=true;
115                        msg.target().executeMsg(msg.target(),newAffect);
116                        recurse=false;
117                }
118                else
119                if((affected instanceof MOB)
120                &&(invoker!=null)
121                &&(msg.source() != invoker)
122                &&(invoker.location()!=((MOB)affected).location())
123                &&(msg.othersCode()!=CMMsg.NO_EFFECT)
124                &&(msg.othersMessage()!=null)
125                &&(!CMath.bset(msg.sourceMajor(),CMMsg.MASK_CHANNEL))
126                &&(!recurse))
127                {
128                        recurse=true;
129                        invoker.executeMsg(invoker,msg);
130                        recurse=false;
131                }
132        }
133
134        @Override
135        public boolean invoke(MOB mob, List<String> commands, Physical givenTarget, boolean auto, int asLevel)
136        {
137                if((auto||mob.isMonster())&&((commands.size()<1)||((commands.get(0)).equals(mob.name()))))
138                {
139                        commands.clear();
140                        MOB M=null;
141                        int tries=0;
142                        while(((++tries)<100)&&(M==null))
143                        {
144                                final Room R=CMLib.map().getRandomRoom();
145                                if(R.numInhabitants()>0)
146                                        M=R.fetchRandomInhabitant();
147                                if((M!=null)&&(M.name().equals(mob.name())))
148                                        M=null;
149                        }
150                        if(M!=null)
151                                commands.add(M.Name());
152                }
153                if(commands.size()<1)
154                {
155                        final StringBuffer scryList=new StringBuffer("");
156                        for(int e=0;e<scries.size();e++)
157                        {
158                                if(scries.elementAt(e,2)==mob)
159                                        scryList.append(((e>0)?", ":"")+((MOB)scries.elementAt(e,1)).name());
160                        }
161                        if(scryList.length()>0)
162                                mob.tell(L("Cast on or revoke from whom?  You currently have @x1 on the following: @x2.",name(),scryList.toString()));
163                        else
164                                mob.tell(L("Cast on whom?"));
165                        return false;
166                }
167                final String mobName=CMParms.combine(commands,0).trim().toUpperCase();
168                MOB target=null;
169                if(givenTarget instanceof MOB)
170                        target=(MOB)givenTarget;
171                if(target!=null)
172                        target=mob.location().fetchInhabitant(mobName);
173                if(target==null)
174                {
175                        try
176                        {
177                                final List<MOB> targets=CMLib.map().findInhabitantsFavorExact(CMLib.map().rooms(), mob, mobName, false, 50);
178                                if(targets.size()>0)
179                                        target=targets.get(CMLib.dice().roll(1,targets.size(),-1));
180                        }
181                        catch(final NoSuchElementException nse)
182                        {
183                        }
184                }
185                if(target instanceof Deity)
186                        target=null;
187                Room newRoom=mob.location();
188                if((target!=null)&&(target.amActive())&&(!target.amDead()))
189                        newRoom=target.location();
190                else
191                {
192                        mob.tell(L("You can't seem to focus on '@x1'.",mobName));
193                        return false;
194                }
195
196                final Ability A=target.fetchEffect(ID());
197                if((A!=null)&&(A.invoker()==mob))
198                {
199                        A.unInvoke();
200                        return true;
201                }
202                else
203                if((A!=null)||(scries.contains(target)))
204                {
205                        mob.tell(L("You can't seem to focus on '@x1'.",mobName));
206                        return false;
207                }
208
209                if(!super.invoke(mob,commands,givenTarget,auto,asLevel))
210                        return false;
211
212                final boolean success=proficiencyCheck(mob,0,auto);
213
214                if(success)
215                {
216                        final CMMsg msg=CMClass.getMsg(mob,target,this,verbalCastCode(mob,target,auto),auto?"":L("^S<S-NAME> invoke(s) the name of '@x1'.^?",mobName));
217                        final CMMsg msg2=CMClass.getMsg(mob,target,this,verbalCastCode(mob,target,auto),null);
218                        if((mob.location().okMessage(mob,msg))&&((newRoom==mob.location())||(newRoom.okMessage(mob,msg2))))
219                        {
220                                mob.location().send(mob,msg);
221                                if(newRoom!=mob.location())
222                                        newRoom.send(target,msg2);
223                                if((msg.value()<=0)&&(msg2.value()<=0))
224                                {
225                                        scries.addElement(target,mob);
226                                        beneficialAffect(mob,target,asLevel,0);
227                                }
228                        }
229                        else
230                                mob.tell(L("You attempt to invoke scrying, but fizzle the spell."));
231
232                }
233                else
234                        beneficialVisualFizzle(mob,null,L("<S-NAME> attempt(s) to invoke scrying, but fizzle(s) the spell."));
235
236                // return whether it worked
237                return success;
238        }
