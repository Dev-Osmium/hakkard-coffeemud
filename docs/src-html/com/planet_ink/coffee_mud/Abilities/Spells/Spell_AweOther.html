035public class Spell_AweOther extends Spell
036{
037
038        @Override
039        public String ID()
040        {
041                return "Spell_AweOther";
042        }
043
044        private final static String localizedName = CMLib.lang().L("Awe Other");
045
046        @Override
047        public String name()
048        {
049                return localizedName;
050        }
051
052        @Override
053        public String displayText()
054        {
055                return L("(Awe of "+text()+")");
056        }
057
058        @Override
059        protected int canAffectCode()
060        {
061                return CAN_MOBS;
062        }
063
064        @Override
065        public int abstractQuality()
066        {
067                return Ability.QUALITY_MALICIOUS;
068        }
069
070        @Override
071        public int classificationCode()
072        {
073                return Ability.ACODE_SPELL|Ability.DOMAIN_ENCHANTMENT;
074        }
075
076        @Override
077        public long flags()
078        {
079                return super.flags() | Ability.FLAG_MINDALTERING;
080        }
081
082        @Override
083        public boolean okMessage(final Environmental myHost, final CMMsg msg)
084        {
085                if(((msg.targetMajor()&CMMsg.MASK_MALICIOUS)>0)
086                &&(!CMath.bset(msg.sourceMajor(),CMMsg.MASK_ALWAYS))
087                &&(msg.target()!=null)
088                &&(msg.target().Name().equalsIgnoreCase(text())))
089                {
090                        final MOB target=(MOB)msg.target();
091                        if((!target.isInCombat())
092                        &&(msg.source().getVictim()!=target)
093                        &&(msg.source().location()==target.location()))
094                        {
095                                msg.source().tell(L("You are too much in awe of @x1",target.name(msg.source())));
096                                if(target.getVictim()==msg.source())
097                                {
098                                        target.makePeace(true);
099                                        target.setVictim(null);
100                                }
101                                return false;
102                        }
103                }
104                return super.okMessage(myHost,msg);
105        }
106
107        @Override
108        public int castingQuality(final MOB mob, final Physical target)
109        {
110                if(mob!=null)
111                {
112                        if(target instanceof MOB)
113                        {
114                                if(((MOB)target).isInCombat())
115                                        return Ability.QUALITY_INDIFFERENT;
116                        }
117                }
118                return super.castingQuality(mob,target);
119        }
120
121        @Override
122        public void unInvoke()
123        {
124                // undo the affects of this spell
125                if(!(affected instanceof MOB))
126                        return;
127                final MOB mob=(MOB)affected;
128                super.unInvoke();
129
130                if(canBeUninvoked())
131                        if((mob.location()!=null)&&(!mob.amDead()))
132                                mob.location().show(mob,null,CMMsg.MSG_OK_VISUAL,L("<S-NAME> seem(s) less in awe of @x1.",text()));
133        }
134
135        @Override
136        public boolean invoke(final MOB mob, final List<String> commands, final Physical givenTarget, final boolean auto, final int asLevel)
137        {
138                if(commands.size()<2)
139                {
140                        mob.tell(L("Invoke awe on whom and of whom?"));
141                        return false;
142                }
143                final String aweWhom=CMParms.combine(commands,1);
144                final MOB target=getTarget(mob,new XVector<String>(commands.get(0)),givenTarget);
145                if(target==null)
146                        return false;
147                Room R=CMLib.map().roomLocation(target);
148                if(R==null)
149                        R=mob.location();
150
151                if(!super.invoke(mob,commands,givenTarget,auto,asLevel))
152                        return false;
153
154                final boolean success=proficiencyCheck(mob,0,auto);
155
156                if(success)
157                {
158                        invoker=mob;
159                        final CMMsg msg=CMClass.getMsg(mob,target,this,verbalCastCode(mob,target,auto),auto?"":L("^S<S-NAME> invoke(s) a spell on <T-NAMESELF>.^?"));
160                        if(R.okMessage(mob,msg))
161                        {
162                                R.send(mob,msg);
163                                if(maliciousAffect(mob,target,asLevel,0,-1)!=null)
164                                {
165                                        final Ability A=target.fetchEffect(ID());
166                                        if(A!=null)
167                                        {
168                                                A.setMiscText(CMStrings.capitalizeAndLower(aweWhom));
169                                                R.show(target,null,CMMsg.MSG_OK_VISUAL,L("<S-NAME> gain(s) a new awe of @x1!",CMStrings.capitalizeAndLower(aweWhom)));
170                                        }
171                                }
172                        }
173                }
174                else
175                        return beneficialWordsFizzle(mob,target,L("<S-NAME> attempt(s) to invoke a spell on <T-NAMESELF>, but fail(s) miserably."));
176
177                // return whether it worked
178                return success;
179        }
180}
