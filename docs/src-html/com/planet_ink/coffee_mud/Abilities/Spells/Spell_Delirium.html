093        protected Environmental getRandomOtherName(final Environmental likeThisOne)
094        {
095                if((invoker==null)||(invoker.location()==null))
096                        return likeThisOne;
097
098                if(likeThisOne instanceof Room)
099                {
100                        int tries=0;
101                        while((++tries)<1000)
102                        {
103                                final Room R=invoker.location().getArea().getRandomProperRoom();
104                                if(!R.displayText().equals(likeThisOne.displayText()))
105                                        return R;
106                        }
107                }
108                else
109                if(likeThisOne instanceof Exit)
110                {
111                        int tries=0;
112                        while((++tries)<1000)
113                        {
114                                final Room R=invoker.location().getArea().getRandomProperRoom();
115                                for(int d=Directions.NUM_DIRECTIONS()-1;d>=0;d--)
116                                {
117                                        final Exit x=R.getExitInDir(d);
118                                        if((x!=null)&&(!x.name().equals(likeThisOne.name())))
119                                                return x;
120                                }
121                        }
122                }
123                else
124                if(likeThisOne instanceof MOB)
125                {
126                        int tries=0;
127                        while((++tries)<1000)
128                        {
129                                final Room R=invoker.location().getArea().getRandomProperRoom();
130                                if((R!=null)&&(R.numInhabitants()>0))
131                                {
132                                        final MOB possible=R.fetchRandomInhabitant();
133                                        if((possible!=null)&&(!possible.name().equalsIgnoreCase(likeThisOne.name())))
134                                                return possible;
135                                }
136                        }
137                }
138                else
139                if(likeThisOne instanceof Item)
140                {
141                        int tries=0;
142                        while((++tries)<1000)
143                        {
144                                final Room R=invoker.location().getArea().getRandomProperRoom();
145                                if(R.numItems()>0)
146                                {
147                                        final Item possible=R.getRandomItem();
148                                        if((possible!=null)&&(!possible.name().equalsIgnoreCase(likeThisOne.name())))
149                                                return possible;
150                                }
151                                if(R.numInhabitants()>0)
152                                {
153                                        final MOB owner=R.fetchRandomInhabitant();
154                                        if((owner!=null)&&(owner.numItems()>0))
155                                        {
156                                                final Item possible=owner.getRandomItem();
157                                                if((possible!=null)&&(!possible.name().equalsIgnoreCase(likeThisOne.name())))
158                                                        return possible;
159                                        }
160                                }
161                        }
162                }
163                return null;
164        }
165
166        protected String getRand(final Environmental likeThis)
167        {
168                final Environmental E=this.getRandomOtherName(likeThis);
169                if(E==null)
170                {
171                        if(likeThis instanceof MOB)
172                                return "someone";
173                        return "something";
174                }
175                return E.name();
176        }

178        protected String process(final MOB mob, String str, final Environmental obj)
179        {
180                if(obj==null)
181                        return str;
182
183                int x=str.indexOf("<S-NAME>");
184                if(x>=0)
185                        str=str.substring(0,x)+getRand(obj)+str.substring(x+("<S-NAME>").length());
186                x=str.indexOf("<S-HIS-HER>");
187                if(x>=0)
188                        str=str.substring(0,x)+getRand(obj)+str.substring(x+("<S-HIS-HER>").length());
189                x=str.indexOf("<T-NAME>");
190                if(x>=0)
191                        str=str.substring(0,x)+getRand(obj)+str.substring(x+("<T-NAME>").length());
192                x=str.indexOf("<T-HIS-HER>");
193                if(x>=0)
194                        str=str.substring(0,x)+getRand(obj)+str.substring(x+("<T-HIS-HER>").length());
195                x=str.indexOf("<T-NAMESELF>");
196                if(x>=0)
197                        str=str.substring(0,x)+getRand(obj)+str.substring(x+("<T-NAMESELF>").length());
198                str=" "+str+" ";
199                x=str.toUpperCase().indexOf(" "+obj.name().toUpperCase()+" ");
200                if(x>=0)
201                        str=str.substring(0,x)+" "+getRand(obj)+" "+str.substring(x+(" "+obj.name()+" ").length());
202                x=str.toUpperCase().indexOf(" YOU ");
203                if(x>=0)
204                        str=str.substring(0,x)+" "+getRand(mob)+" "+str.substring(x+(" YOU ").length());
205                x=str.toUpperCase().indexOf(" "+mob.name().toUpperCase()+" ");
206                if(x>=0)
207                        str=str.substring(0,x)+" "+getRand(mob)+" "+str.substring(x+(" "+mob.name().toUpperCase()+" ").length());
208                final MOB victim=mob.getVictim();
209                if(victim!=null)
210                {
211                        x=str.toUpperCase().indexOf(" "+victim.name().toUpperCase()+" ");
212                        if(x>=0)
213                                str=str.substring(0,x)+" "+getRand(victim)+" "+str.substring(x+(" "+victim.name().toUpperCase()+" ").length());
214                }
215                return str.trim();
216        }

218        @Override
219        public boolean tick(final Tickable ticking, final int tickID)
220        {
221                if((tickID==Tickable.TICKID_MOB)
222                &&(affected instanceof MOB))
223                {
224                        final MOB mob=(MOB)affected;
225                        amountRemaining-=mob.charStats().getStat(CharStats.STAT_INTELLIGENCE);
226                        if(amountRemaining<0)
227                                unInvoke();
228                }
229                return super.tick(ticking,tickID);
230        }
231
232        @Override
233        public boolean okMessage(final Environmental myHost, final CMMsg msg)
234        {
235                if(!(affected instanceof MOB))
236                        return true;
237
238                final MOB mob=(MOB)affected;
239
240                final String othersMessage=msg.othersMessage();
241                String sourceMessage=msg.sourceMessage();
242                String targetMessage=msg.targetMessage();
243                if((msg.amITarget(mob))&&(targetMessage!=null))
244                {
245                        targetMessage=process(mob,process(mob,targetMessage,msg.target()),msg.target());
246                        if(!targetMessage.equals(msg.targetMessage()))
247                                msg.modify(msg.source(),msg.target(),msg.tool(),msg.sourceCode(),sourceMessage,msg.targetCode(),targetMessage,msg.othersCode(),othersMessage);
248                }
249                if((msg.amISource(mob))&&(sourceMessage!=null))
250                {
251                        sourceMessage=process(mob,process(mob,sourceMessage,msg.source()),msg.source());
252                        if(!sourceMessage.equals(msg.sourceMessage()))
253                                msg.modify(msg.source(),msg.target(),msg.tool(),msg.sourceCode(),sourceMessage,msg.targetCode(),targetMessage,msg.othersCode(),othersMessage);
254                }
255                return true;
256        }

258        @Override
259        public void unInvoke()
260        {
261                // undo the affects of this spell
262                if(!(affected instanceof MOB))
263                        return;
264                final MOB mob=(MOB)affected;
265                super.unInvoke();
266
267                if((mob.location()!=null)&&(!mob.amDead())&&(super.canBeUninvoked()))
268                        mob.location().show(mob,null,CMMsg.MSG_OK_VISUAL,L("<S-NAME> begin(s) to feel a bit less delirious."));
269        }

273        {
274                if(mob!=null)
275                {
276                        if((mob.isMonster())&&(mob.isInCombat()))
277                                return Ability.QUALITY_INDIFFERENT;
278                        if(target instanceof MOB)
279                        {
280                        }
281                }
282                return super.castingQuality(mob,target);
283        }

285        @Override
286        public boolean invoke(final MOB mob, final List<String> commands, final Physical givenTarget, final boolean auto, final int asLevel)
287        {
288                final MOB target=this.getTarget(mob,commands,givenTarget);
289                if(target==null)
290                        return false;
291
292                if(!super.invoke(mob,commands,givenTarget,auto,asLevel))
293                        return false;
294
295                final boolean success=proficiencyCheck(mob,0,auto);
296
297                if(success)
298                {
299                        invoker=mob;
300                        final CMMsg msg=CMClass.getMsg(mob,target,this,verbalCastCode(mob,target,auto),auto?"":L("^S<S-NAME> whisper(s) to <T-NAMESELF>.^?"));
301                        final CMMsg msg2=CMClass.getMsg(mob,target,this,CMMsg.MSK_CAST_MALICIOUS_VERBAL|CMMsg.TYP_MIND|(auto?CMMsg.MASK_ALWAYS:0),null);
302                        if((mob.location().okMessage(mob,msg))||(mob.location().okMessage(mob,msg2)))
303                        {
304                                mob.location().send(mob,msg);
305                                mob.location().send(mob,msg2);
306                                if(msg.value()<=0)
307                                {
308                                        amountRemaining=300;
309                                        maliciousAffect(mob,target,asLevel,0,-1);
310                                        target.location().show(target,null,CMMsg.MSG_OK_ACTION,L("<S-NAME> go(es) under the grip of delirium!!"));
311                                }
312                        }
313                }
314                else
315                        return maliciousFizzle(mob,target,L("<S-NAME> whisper(s) to <T-NAMESELF>, but the spell fades."));
316
317                // return whether it worked
318                return success;
319        }
