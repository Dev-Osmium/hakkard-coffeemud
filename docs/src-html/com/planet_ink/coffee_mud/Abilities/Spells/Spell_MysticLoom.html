107        @Override
108        public void unInvoke()
109        {
110                super.unInvoke();
111                if(canBeUninvoked())
112                {
113                        final Item theLoom=this.theLoom;
114                        final MOB theLoomer=this.theLoomer;
115                        final Ability skill=this.skill;
116                        if((theLoom!=null)
117                        &&(theLoom.owner()!=null)
118                        &&(theLoom.owner() instanceof Room)
119                        &&(((Room)theLoom.owner()).isContent(theLoom)))
120                        {
121                                final MOB actorM=(invoker!=null)? invoker : CMLib.map().deity();
122                                ((Room)theLoom.owner()).show(actorM,null,CMMsg.MSG_OK_VISUAL,L("The mystical loom fades away..."));
123                                theLoom.destroy();
124                        }
125                        this.theLoom=null;
126                        if(theLoomer != null)
127                        {
128                                if(skill != null)
129                                {
130                                        CMLib.threads().deleteTick(skill, Tickable.TICKID_MOB);
131                                        final Ability A=theLoomer.fetchEffect(skill.ID());
132                                        if(A!=null)
133                                        {
134                                                A.unInvoke();
135                                                CMLib.threads().deleteTick(A, Tickable.TICKID_MOB);
136                                        }
137                                }
138                                theLoomer.destroy();
139                                this.theLoomer=null;
140                        }
141                        if(skill != null)
142                                skill.destroy();
143                        this.skill=null;
144                }
145        }

147        protected String removePercent(final String thisStr)
148        {
149                int x=thisStr.indexOf("% ");
150                if(x>=0)
151                        return new StringBuffer(thisStr).replace(x,x+2,"").toString();
152                x=thisStr.indexOf(" %");
153                if(x>=0)
154                        return new StringBuffer(thisStr).replace(x,x+2,"").toString();
155                x=thisStr.indexOf('%');
156                if(x>=0)
157                        return new StringBuffer(thisStr).replace(x,x+1,"").toString();
158                return thisStr;
159        }

162        @Override
163        public boolean tick(final Tickable ticking, final int tickID)
164        {
165                if(tickID==Tickable.TICKID_MOB)
166                {
167                        final Item theLoom=this.theLoom;
168                        if((theLoom != null)
169                        &&(theLoom.owner() instanceof Room)
170                        &&((skill==null)
171                                ||((theLoomer != null)&&(theLoomer.fetchEffect(skill.ID())==null))))
172                        {
173                                final Room R=(Room)theLoom.owner();
174                                final int roomRange=1 + super.getXMAXRANGELevel(invoker());
175                                final List<Room> rooms;
176                                if(roomRange == 1)
177                                        rooms = new XVector<Room>(R);
178                                else
179                                {
180                                        final TrackingLibrary.TrackingFlags flags = CMLib.tracking().newFlags()
181                                                        .plus(TrackingLibrary.TrackingFlag.NOAIR)
182                                                        .plus(TrackingLibrary.TrackingFlag.OPENONLY);
183                                        rooms = CMLib.tracking().getRadiantRooms(R, flags, roomRange);
184                                }
185                                final List<Triad<Item,ItemPossessor,Container>> cloths = new LinkedList<Triad<Item,ItemPossessor,Container>>();
186                                for(final Room R1 : rooms)
187                                {
188                                        for(final Enumeration<Item> i=R1.items();i.hasMoreElements();)
189                                        {
190                                                final Item I=i.nextElement();
191                                                if(I instanceof RawMaterial)
192                                                {
193                                                        final Triad<Item,ItemPossessor,Container> t=new Triad<Item,ItemPossessor,Container>(I,R1,I.container());
194                                                        I.setContainer(null);
195                                                        if(I.owner() != R)
196                                                                R.moveItemTo(I);
197                                                        cloths.add(t);
198                                                }
199                                        }
200                                        for(final Enumeration<MOB> m=R1.inhabitants();m.hasMoreElements();)
201                                        {
202                                                final MOB M=m.nextElement();
203                                                if(M==null)
204                                                        continue;
205                                                for(final Enumeration<Item> i=M.items();i.hasMoreElements();)
206                                                {
207                                                        final Item I=i.nextElement();
208                                                        if(I instanceof RawMaterial)
209                                                        {
210                                                                final Triad<Item,ItemPossessor,Container> t=new Triad<Item,ItemPossessor,Container>(I,M,I.container());
211                                                                I.setContainer(null);
212                                                                if(I.owner() != R)
213                                                                        R.moveItemTo(I);
214                                                                cloths.add(t);
215                                                        }
216                                                }
217                                        }
218                                }
219                                if(skill != null)
220                                        CMLib.threads().deleteTick(skill, Tickable.TICKID_MOB);
221                                final Ability skill=CMClass.getAbility("Textiling");
222                                if(skill != null)
223                                {
224                                        theLoomer.setLocation(R);
225                                        final List<List<String>> recipes = ((CraftorAbility)skill).fetchRecipes();
226                                        for(int i=0;i<10;i++)
227                                        {
228                                                final List<String> recipe = recipes.get(CMLib.dice().roll(1, recipes.size(), -1));
229                                                final List<String> commands = new ArrayList<String>();
230                                                commands.add(removePercent(recipe.get(0)));
231                                                theLoomer.curState().setMovement(100);
232                                                theLoomer.curState().setMana(100);
233                                                theLoomer.curState().setHitPoints(20);
234                                                skill.invoke(theLoomer, commands, null, true, 0);
235                                                final Ability A=this.skill=theLoomer.fetchEffect(skill.ID());
236                                                if(A!=null)
237                                                {
238                                                        //System.out.println("GO: "+recipe.get(0));//BZ:DELME
239                                                        int tickDown = CMath.s_int(A.getStat("TICKDOWN"));
240                                                        if(tickDown > 0)
241                                                        {
242                                                                final double quickPct = getXTIMELevel(invoker()) * 0.05;
243                                                                tickDown-=(int)Math.round(CMath.mul(tickDown, quickPct));
244                                                                if(tickDown < 5)
245                                                                        tickDown=5;
246                                                                A.setStat("TICKDOWN", ""+tickDown);
247                                                        }
248                                                        CMLib.threads().startTickDown(A, Tickable.TICKID_MOB, 1);
249                                                        break;
250                                                }
251                                                else
252                                                {
253                                                        //System.out.println("FAIL: "+recipe.get(0));//BZ:DELME
254                                                }
255                                        }
256                                }
257                                final List<Item> spares=new LinkedList<Item>();
258                                for(final Enumeration<Item> i=R.items();i.hasMoreElements();)
259                                {
260                                        final Item I=i.nextElement();
261                                        if(I instanceof RawMaterial)
262                                        {
263                                                boolean found=false;
264                                                for(final Triad<Item,ItemPossessor,Container> T : cloths)
265                                                {
266                                                        if(T.first == I)
267                                                                found=true;
268                                                }
269                                                if(!found)
270                                                        spares.add(I);
271                                        }
272                                }
273                                Triad<Item,ItemPossessor,Container> lastT=null;
274                                for(final Triad<Item,ItemPossessor,Container> T : cloths)
275                                {
276                                        final ItemPossessor P;
277                                        final Container C;
278                                        final Item I;
279                                        if(!T.first.amDestroyed())
280                                        {
281                                                C=T.third;
282                                                P=T.second;
283                                                I=T.first;
284                                        }
285                                        else
286                                        if((lastT != null)
287                                        &&(spares.size()>0))
288                                        {
289                                                I=spares.remove(0);
290                                                C=lastT.third;
291                                                P=lastT.second;
292                                        }
293                                        else
294                                                continue;
295                                        if((I.owner()!=P)
296                                        ||(!P.isContent(I)))
297                                        {
298                                                I.setContainer(C);
299                                                if(I.owner()!=null)
300                                                        I.owner().delItem(I);
301                                                P.addItem(I, ItemPossessor.Expire.Player_Drop);
302                                        }
303                                        lastT=T;
304                                }
305                        }
306                }
307                return super.tick(ticking,tickID);
308        }

310        @Override
311        public boolean invoke(final MOB mob, final List<String> commands, final Physical givenTarget, final boolean auto, final int asLevel)
312        {
313                for(int i=0;i<mob.location().numItems();i++)
314                {
315                        final Item I=mob.location().getItem(i);
316                        if((I!=null)&&(I.fetchEffect(ID())!=null))
317                        {
318                                mob.tell(L("There is already a mystic loom here."));
319                                return false;
320                        }
321                }
322
323                if(!super.invoke(mob,commands,givenTarget,auto,asLevel))
324                        return false;
325
326                final Physical target = mob.location();
327
328                final boolean success=proficiencyCheck(mob,0,auto);
329
330                if(success)
331                {
332
333                        final CMMsg msg = CMClass.getMsg(mob, target, this,verbalCastCode(mob,target,auto),auto?L("A mystical loom appears!"):L("^S<S-NAME> evoke(s) a mystical loom!^?"));
334                        if(mob.location().okMessage(mob,msg))
335                        {
336                                mob.location().send(mob,msg);
337                                final Item I=CMClass.getItem("GenItem");
338                                I.setName(L("a mystic loom"));
339                                I.setDisplayText(L("a mystical loom sits here"));
340                                I.setDescription(L("It's a loom for spinning cloth.  Also, it's mystical looking."));
341                                I.setMaterial(RawMaterial.RESOURCE_WOOD);
342                                I.basePhyStats().setDisposition(I.basePhyStats().disposition()|PhyStats.IS_BONUS);
343                                CMLib.flags().setGettable(I,false);
344                                I.recoverPhyStats();
345                                mob.location().addItem(I);
346                                theLoom=I;
347                                final MOB M=CMClass.getFactoryMOB();
348                                M.setName(L("a mystic loom"));
349                                M.basePhyStats().setLevel(adjustedLevel(mob,asLevel));
350                                M.phyStats().setLevel(adjustedLevel(mob,asLevel));
351                                M.setDisplayText(L("a mystical loom sits here"));
352                                M.setDescription(L("It's a loom for spinning cloth.  Also, it's mystical looking."));
353                                theLoomer=M;
354                                beneficialAffect(mob,I,asLevel,75 + super.adjustedLevel(mob, asLevel) + (30 * super.getXTIMELevel(mob)));
355                        }
356                }
357                else
358                        return beneficialWordsFizzle(mob,null,L("<S-NAME> incant(s), but the magic fizzles."));
359
360                // return whether it worked
361                return success;
362        }
