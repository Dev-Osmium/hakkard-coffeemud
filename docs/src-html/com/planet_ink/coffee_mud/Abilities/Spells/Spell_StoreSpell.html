052        @Override
053        public String name()
054        {
055                if((affected!=null)&&(CMLib.flags().isInTheGame(affected,true)))
056                {
057                        if(spellName.length()==0)
058                        {
059                                spellName="unknown";
060                                final int x=text().indexOf('/');
061                                Ability A=null;
062                                if(x>0)
063                                {
064                                        A=CMClass.getAbility(text().substring(0,x));
065                                        if(A!=null)
066                                                spellName=A.name();
067                                }
068                        }
069                        return "Store Spell: "+spellName;
070                }
071                return Name();
072        }

101        public String getSpeakableName(String name)
102        {
103                name=CMStrings.removeColors(name.toUpperCase());
104                if(name.startsWith("A "))
105                        name=name.substring(2).trim();
106                if(name.startsWith("AN "))
107                        name=name.substring(3).trim();
108                if(name.startsWith("THE "))
109                        name=name.substring(4).trim();
110                if(name.startsWith("SOME "))
111                        name=name.substring(5).trim();
112                return name;
113        }

115        public void waveIfAble(MOB mob, Physical afftarget, String message, Item me)
116        {
117                if((mob.isMine(me))
118                &&(me.amBeingWornProperly())
119                &&(message!=null))
120                {
121                        Physical target=null;
122                        if((mob.location()!=null))
123                                target=afftarget;
124                        final String name=getSpeakableName(me.name());
125                        int x=message.toUpperCase().indexOf(name);
126                        if(x>=0)
127                        {
128                                message=message.substring(x+name.length());
129                                final int y=message.indexOf('\'');
130                                if(y>=0)
131                                        message=message.substring(0,y);
132                                message=message.trim();
133                                x=text().indexOf('/');
134                                int charges=0;
135                                Ability A=null;
136                                if(x>0)
137                                {
138                                        charges=CMath.s_int(text().substring(x+1));
139                                        A=CMClass.getAbility(text().substring(0,x));
140                                }
141                                if(A==null)
142                                        mob.tell(L("Something seems wrong with @x1.",me.name()));
143                                else
144                                if(charges<=0)
145                                {
146                                        mob.tell(L("@x1 seems spent.",me.name()));
147                                        me.delEffect(this);
148                                }
149                                else
150                                {
151                                        setMiscText(A.ID()+"/"+(charges-1));
152                                        A=(Ability)A.newInstance();
153                                        final Vector<String> V=new Vector<String>();
154                                        if(target!=null)
155                                                V.addElement(target.name());
156                                        V.addElement(message);
157                                        mob.location().show(mob,null,CMMsg.MSG_OK_VISUAL,L("@x1 glows brightly.",me.name()));
158                                        A.invoke(mob, V, target, true,0);
159                                }
160                        }
161                }
162        }

164        @Override
165        public void executeMsg(final Environmental myHost, final CMMsg msg)
166        {
167                final MOB mob=msg.source();
168
169                switch(msg.targetMinor())
170                {
171                case CMMsg.TYP_WAND_USE:
172                        if((msg.amITarget(affected))&&(affected instanceof Item)&&((msg.tool()==null)||(msg.tool() instanceof Physical)))
173                                waveIfAble(mob,(Physical)msg.tool(),msg.targetMessage(),(Item)affected);
174                        break;
175                case CMMsg.TYP_SPEAK:
176                        if((msg.sourceMinor()==CMMsg.TYP_SPEAK)
177                        &&(msg.targetMessage()!=null)
178                        &&(affected != null))
179                        {
180                                boolean alreadyWanding=false;
181                                final List<CMMsg> trailers =msg.trailerMsgs();
182                                if(trailers!=null)
183                                {
184                                        for(final CMMsg msg2 : trailers)
185                                        {
186                                                if((msg2.targetMinor()==CMMsg.TYP_WAND_USE)
187                                                &&(msg2.target() == affected))
188                                                        alreadyWanding=true;
189                                        }
190                                }
191                                if(!alreadyWanding)
192                                {
193                                        final String name=getSpeakableName(affected.name());
194                                        final int x=msg.targetMessage().toUpperCase().indexOf(name);
195                                        if(x>=0)
196                                        {
197                                                msg.addTrailerMsg(CMClass.getMsg(msg.source(),affected,msg.target(),CMMsg.NO_EFFECT,null,CMMsg.MASK_ALWAYS|CMMsg.TYP_WAND_USE,CMStrings.getSayFromMessage(msg.sourceMessage()),CMMsg.NO_EFFECT,null));
198                                        }
199                                }
200                        }
201                        break;
202                default:
203                        break;
204                }
205                super.executeMsg(myHost,msg);
206        }

208        @Override
209        public boolean invoke(MOB mob, List<String> commands, Physical givenTarget, boolean auto, int asLevel)
210        {
211                if(commands.size()<2)
212                {
213                        mob.tell(L("Store which spell onto what?"));
214                        return false;
215                }
216                final Physical target=mob.location().fetchFromMOBRoomFavorsItems(mob,null,commands.get(commands.size()-1),Wearable.FILTER_UNWORNONLY);
217                if((target==null)||(!CMLib.flags().canBeSeenBy(target,mob)))
218                {
219                        mob.tell(L("You don't see '@x1' here.",(commands.get(commands.size()-1))));
220                        return false;
221                }
222                if(!(target instanceof Item))
223                {
224                        mob.tell(L("You can't enchant '@x1'.",target.name(mob)));
225                        return false;
226                }
227
228                final Item item=(Item)target;
229
230                commands.remove(commands.size()-1);
231
232                final String spellName=CMParms.combine(commands,0).trim();
233                Spell wandThis=null;
234                for(final Enumeration<Ability> a=mob.allAbilities();a.hasMoreElements();)
235                {
236                        final Ability A=a.nextElement();
237                        if((A!=null)
238                        &&(A instanceof Spell)
239                        &&((!A.isSavable())||(CMLib.ableMapper().qualifiesByLevel(mob,A)))
240                        &&(A.name().toUpperCase().startsWith(spellName.toUpperCase()))
241                        &&(!A.ID().equals(this.ID())))
242                                wandThis=(Spell)A;
243                }
244                if(wandThis==null)
245                {
246                        mob.tell(L("You don't know how to enchant anything with '@x1'.",spellName));
247                        return false;
248                }
249                if((CMLib.ableMapper().lowestQualifyingLevel(wandThis.ID())>24)
250                ||(((StdAbility)wandThis).usageCost(null,true)[0]>45)
251                ||(CMath.bset(wandThis.flags(), Ability.FLAG_CLANMAGIC)))
252                {
253                        mob.tell(L("That spell is too powerful to store."));
254                        return false;
255                }
256                Ability A=item.fetchEffect(ID());
257                if((A!=null)&&(A.text().length()>0)&&(!A.text().startsWith(wandThis.ID()+"/")))
258                {
259                        mob.tell(L("'@x1' already has a different spell stored in it.",item.name()));
260                        return false;
261                }
262                else
263                if(A==null)
264                {
265                        A=(Ability)copyOf();
266                        A.setMiscText(wandThis.ID()+"/0");
267                }
268                int charges=0;
269                final int x=A.text().indexOf('/');
270                if(x>=0)
271                        charges=CMath.s_int(A.text().substring(x+1));
272                overridemana=-1;
273                int mana=usageCost(mob,true)[0]+wandThis.usageCost(mob,true)[0];
274                if(mana>mob.maxState().getMana())
275                        mana=mob.maxState().getMana();
276                overridemana=mana;
277
278                // lose all the mana!
279                if(!super.invoke(mob,commands,givenTarget,auto,asLevel))
280                        return false;
281                overridemana=-1;
282
283                final boolean success=proficiencyCheck(mob,0,auto);
284
285                if(success)
286                {
287                        setMiscText(wandThis.ID());
288                        final CMMsg msg=CMClass.getMsg(mob,target,this,verbalCastCode(mob,target,auto),L("^S<S-NAME> move(s) <S-HIS-HER> fingers around <T-NAMESELF>, incanting softly.^?"));
289                        if(mob.location().okMessage(mob,msg))
290                        {
291                                mob.location().send(mob,msg);
292                                if(target.fetchEffect(ID())==null)
293                                {
294                                        A.setInvoker(mob);
295                                        target.addNonUninvokableEffect(A);
296                                }
297                                A.setMiscText(wandThis.ID()+"/"+(charges+1));
298                                mob.location().show(mob,target,null,CMMsg.MSG_OK_VISUAL,L("<T-NAME> glow(s) softly."));
299                        }
300
301                }
302                else
303                        beneficialWordsFizzle(mob,target,L("<S-NAME> move(s) <S-HIS-HER> fingers around <T-NAMESELF>, incanting softly, and looking very frustrated."));
304
305                // return whether it worked
306                return success;
307        }
