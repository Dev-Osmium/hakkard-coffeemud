036public class Spell_MassFly extends Spell
037{
038
039        @Override
040        public String ID()
041        {
042                return "Spell_MassFly";
043        }
044
045        private final static String localizedName = CMLib.lang().L("Mass Fly");
046
047        @Override
048        public String name()
049        {
050                return localizedName;
051        }
052
053        @Override
054        public String displayText()
055        {
056                return "";
057        }
058
059        @Override
060        public int abstractQuality()
061        {
062                return Ability.QUALITY_BENEFICIAL_OTHERS;
063        }
064
065        @Override
066        protected int canAffectCode()
067        {
068                return 0;
069        }
070
071        @Override
072        public int classificationCode()
073        {
074                return Ability.ACODE_SPELL|Ability.DOMAIN_TRANSMUTATION;
075        }
076
077        @Override
078        public boolean invoke(MOB mob, List<String> commands, Physical givenTarget, boolean auto, int asLevel)
079        {
080                final Set<MOB> h=properTargets(mob,givenTarget,false);
081                if(h==null)
082                {
083                        mob.tell(L("There doesn't appear to be anyone here worth making fly."));
084                        return false;
085                }
086
087                if(!super.invoke(mob,commands,givenTarget,auto,asLevel))
088                        return false;
089
090                final boolean success=proficiencyCheck(mob,0,auto);
091
092                if(success)
093                {
094                        if(mob.location().show(mob,null,this,somanticCastCode(mob,null,auto),auto?"":L("^S<S-NAME> wave(s) <S-HIS-HER> arms and speak(s).^?")))
095                        {
096                                for (final Object element : h)
097                                {
098                                        final MOB target=(MOB)element;
099
100                                        final CMMsg msg=CMClass.getMsg(mob,target,this,somanticCastCode(mob,target,auto),null);
101                                        if(mob.location().okMessage(mob,msg))
102                                        {
103                                                mob.location().send(mob,msg);
104                                                if(mob.location()==target.location())
105                                                        target.location().show(target,null,CMMsg.MSG_OK_ACTION,L("<S-NAME> start(s) to fly around!"));
106                                                final Spell_Fly fly=new Spell_Fly();
107                                                fly.setProficiency(proficiency());
108                                                fly.beneficialAffect(mob,target,asLevel,0);
109                                        }
110                                }
111                        }
112                }
113                else
114                        return beneficialVisualFizzle(mob,null,L("<S-NAME> wave(s) <S-HIS-HER> arms and speak(s), but the spell fizzles."));
115
116                // return whether it worked
117                return success;
118        }
119}
