036public class Spell_WallOfFire extends Spell
037{
038
039        @Override
040        public String ID()
041        {
042                return "Spell_WallOfFire";
043        }
044
045        private final static String localizedName = CMLib.lang().L("Wall of Fire");
046
047        @Override
048        public String name()
049        {
050                return localizedName;
051        }
052
053        private final static String localizedStaticDisplay = CMLib.lang().L("(Wall of Fire)");
054
055        @Override
056        public String displayText()
057        {
058                return localizedStaticDisplay;
059        }
060
061        @Override
062        public int maxRange()
063        {
064                return adjustedMaxInvokerRange(10);
065        }
066
067        @Override
068        public int minRange()
069        {
070                return 1;
071        }
072
073        @Override
074        public int abstractQuality()
075        {
076                return Ability.QUALITY_OK_SELF;
077        }
078
079        @Override
080        public int enchantQuality()
081        {
082                return Ability.QUALITY_INDIFFERENT;
083        }
084
085        @Override
086        protected int canAffectCode()
087        {
088                return CAN_ITEMS;
089        }
090
091        @Override
092        protected int canTargetCode()
093        {
094                return 0;
095        }
096
097        @Override
098        public int classificationCode()
099        {
100                return Ability.ACODE_SPELL|Ability.DOMAIN_CONJURATION;
101        }
102
103        @Override
104        public long flags()
105        {
106                return Ability.FLAG_HEATING|Ability.FLAG_FIREBASED;
107        }
108
109        protected Item theWall=null;
110        protected String deathNotice="";
111
112        @Override
113        public boolean tick(final Tickable ticking, final int tickID)
114        {
115                if(tickID==Tickable.TICKID_MOB)
116                {
117                        if((invoker!=null)
118                           &&(theWall!=null)
119                           &&(invoker.location()!=null))
120                        {
121                                final Room room=invoker.location();
122                                if(!invoker.location().isContent(theWall))
123                                        unInvoke();
124                                else
125                                for(int m=0;m<room.numInhabitants();m++)
126                                {
127                                        final MOB mob=room.fetchInhabitant(m);
128                                        if((mob!=null)
129                                        &&(mob!=invoker)
130                                        &&(mob.isInCombat())
131                                        &&(mob.getVictim()==invoker)
132                                        &&(mob.rangeToTarget()==1))
133                                        {
134                                                final int damage = CMLib.dice().roll((int)Math.round((invoker.phyStats().level()+super.getXLEVELLevel(invoker())+(2.0*super.getX1Level(invoker())))/4.0),6,1);
135                                                CMLib.combat().postDamage(invoker,mob,this,damage,CMMsg.MASK_MALICIOUS|CMMsg.MASK_ALWAYS|CMMsg.TYP_FIRE,Weapon.TYPE_BURNING,L("The wall of fire flares and <DAMAGE> <T-NAME>!"));
136                                                if((!mob.isInCombat())&&(mob.isMonster())&&(mob!=invoker)&&(invoker!=null)&&(mob.location()==invoker.location())&&(mob.location().isInhabitant(invoker))&&(CMLib.flags().canBeSeenBy(invoker,mob)))
137                                                        CMLib.combat().postAttack(mob,invoker,mob.fetchWieldedItem());
138                                        }
139                                }
140                        }
141                }
142                return super.tick(ticking,tickID);
143        }
144
145        @Override
146        public void unInvoke()
147        {
148                super.unInvoke();
149                if(canBeUninvoked())
150                {
151                        if((theWall!=null)
152                        &&(theWall.owner()!=null)
153                        &&(theWall.owner() instanceof Room)
154                        &&(((Room)theWall.owner()).isContent(theWall)))
155                        {
156                                final MOB actorM=(invoker!=null)? invoker : CMLib.map().deity();
157                                ((Room)theWall.owner()).show(actorM,null,CMMsg.MSG_OK_VISUAL,deathNotice);
158                                final Item wall=theWall;
159                                theWall=null;
160                                wall.destroy();
161                        }
162                }
163        }
164
165        @Override
166        public boolean invoke(MOB mob, List<String> commands, Physical givenTarget, boolean auto, int asLevel)
167        {
168                if((!mob.isInCombat())||(mob.rangeToTarget()<1))
169                {
170                        mob.tell(L("You really should be in ranged combat to cast this."));
171                        return false;
172                }
173                for(int i=0;i<mob.location().numItems();i++)
174                {
175                        final Item I=mob.location().getItem(i);
176                        if((I!=null)&&(I.fetchEffect(ID())!=null))
177                        {
178                                mob.tell(L("There is already a wall of fire here."));
179                                return false;
180                        }
181                }
182
183                if(!super.invoke(mob,commands,givenTarget,auto,asLevel))
184                        return false;
185
186                final Physical target = mob.location();
187
188                final boolean success=proficiencyCheck(mob,0,auto);
189
190                if(success)
191                {
192
193                        final CMMsg msg = CMClass.getMsg(mob, target, this,verbalCastCode(mob,target,auto),L(auto?"A blazing wall of fire appears!":"^S<S-NAME> conjur(s) up a blazing wall of fire!^?")+CMLib.protocol().msp("fireball.wav",10));
194                        if(mob.location().okMessage(mob,msg))
195                        {
196                                mob.location().send(mob,msg);
197                                final Item I=CMClass.getItem("GenItem");
198                                I.setName(L("a wall of fire"));
199                                I.setDisplayText(L("a blazing wall of fire is burning here"));
200                                I.setDescription(L("The flames are high and hot."));
201                                I.setMaterial(RawMaterial.RESOURCE_NOTHING);
202                                I.basePhyStats().setDisposition(I.basePhyStats().disposition()|PhyStats.IS_LIGHTSOURCE);
203                                CMLib.flags().setGettable(I,false);
204                                I.recoverPhyStats();
205                                mob.location().addItem(I);
206                                theWall=I;
207                                deathNotice="The wall of fire flickers...";
208                                beneficialAffect(mob,I,asLevel,20);
209                        }
210                }
211                else
212                        return beneficialWordsFizzle(mob,null,L("<S-NAME> incant(s), but the magic fizzles."));
213
214                // return whether it worked
215                return success;
216        }
