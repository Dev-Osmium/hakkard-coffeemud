079        protected static List<ItemCraftor> craftingSkills=new Vector<ItemCraftor>();
080        protected static List<ItemCraftor> getCraftingSkills()
081        {
082                if(craftingSkills.size()==0)
083                {
084                        final Vector<Ability> V=new Vector<Ability>();
085                        for(final Enumeration<Ability> e=CMClass.abilities();e.hasMoreElements();)
086                        {
087                                final Ability A=e.nextElement();
088                                if(A instanceof ItemCraftor)
089                                        V.addElement((ItemCraftor)A.copyOf());
090                        }
091                        while(V.size()>0)
092                        {
093                                int lowest=Integer.MAX_VALUE;
094                                ItemCraftor lowestA=null;
095                                for(int i=0;i<V.size();i++)
096                                {
097                                        final ItemCraftor A=(ItemCraftor)V.elementAt(i);
098                                        final int ii=CMLib.ableMapper().lowestQualifyingLevel(A.ID());
099                                        if(ii<lowest)
100                                        {
101                                                lowest=ii;
102                                                lowestA=A;
103                                        }
104                                }
105                                if(lowestA==null)
106                                        lowestA=(ItemCraftor)V.firstElement();
107                                if(lowestA!=null)
108                                {
109                                        V.removeElement(lowestA);
110                                        craftingSkills.add(lowestA);
111                                }
112                                else
113                                        break;
114                        }
115                }
116                return craftingSkills;
117        }

119        protected List<Item> previousItems=null;
120
121        @Override
122        public void unInvoke()
123        {
124                final Physical affected = super.affected;
125                super.unInvoke();
126                if((previousItems == null) && (text().length()>0))
127                {
128                        previousItems=new XVector<Item>();
129                        CMLib.coffeeMaker().addItemsFromXML(text(), previousItems, null);
130                }
131                if(canBeUninvoked() && (affected instanceof Item) && (previousItems!=null) && (previousItems.size()>0))
132                {
133                        final Item item=(Item)affected;
134                        if(item.owner() instanceof Room)
135                        {
136                                ((Room)item.owner()).showHappens(CMMsg.MSG_OK_VISUAL, item, L("<S-NAME> reverts to its previous form."));
137                                for(final Item I : previousItems)
138                                        ((Room)item.owner()).addItem(I, Expire.Player_Drop);
139                                previousItems = null;
140                        }
141                        else
142                        if(item.owner() instanceof MOB)
143                        {
144                                ((MOB)item.owner()).tell(((MOB)item.owner()),item,null,L("<T-NAME> reverts to its previous form."));
145                                for(final Item I : previousItems)
146                                        ((MOB)item.owner()).addItem(I);
147                                previousItems = null;
148                        }
149                        affected.destroy();
150                }
151        }
152        
153        @Override
154        public boolean invoke(MOB mob, List<String> commands, Physical givenTarget, boolean auto, int asLevel)
155        {
156                // add something to disable traps
157                //
158                if(commands.size()<2)
159                {
160                        mob.tell(L("Polymorph what object into what?"));
161                        return false;
162                }
163                String itemName=commands.get(0);
164                final Item targetI=super.getTarget(mob, mob.location(), givenTarget, new XVector<String>(itemName), Wearable.FILTER_UNWORNONLY);
165                if(targetI==null)
166                {
167                        mob.tell(L("You don't seem to have a '@x1'.",itemName));
168                        return false;
169                }
170                if((targetI instanceof DeadBody)
171                ||(!CMLib.utensils().canBePlayerDestroyed(mob,targetI,false, false)))
172                {
173                        mob.tell(L("You can't polymorph that."));
174                        return false;
175                }
176                Vector<String> intoWhatV=new XVector<String>(commands);
177                intoWhatV.remove(0);
178                String intoWhat=CMParms.combineQuoted(commands, 1);
179                
180                if(targetI.fetchEffect(ID())!=null)
181                {
182                        mob.tell(mob,targetI,null,L("<T-NAME> is already polymorphed!"));
183                        return false;
184                }
185
186                Item intoI = null;
187                for(ItemCraftor A : getCraftingSkills())
188                {
189                        List<List<String>> L = A.matchingRecipeNames(intoWhat, false);
190                        if((L!=null)&&(L.size()>0))
191                        {
192                                ItemKeyPair what=A.craftItem(L.get(0).get(0),targetI.material(),true, false);
193                                if((what!=null)&&(what.item!=null))
194                                {
195                                        intoI=what.item;
196                                        break;
197                                }
198                        }
199                }
200                if(intoI == null)
201                {
202                        for(ItemCraftor A : getCraftingSkills())
203                        {
204                                List<List<String>> L = A.matchingRecipeNames(intoWhat, true);
205                                if((L!=null)&&(L.size()>0))
206                                {
207                                        ItemKeyPair what=A.craftItem(L.get(0).get(0),targetI.material(),true, false);
208                                        if((what!=null)&&(what.item!=null))
209                                        {
210                                                intoI=what.item;
211                                                break;
212                                        }
213                                }
214                        }
215                }
216                if(intoI == null)
217                        intoI = mob.findItem(intoWhat);
218                if(intoI == null)
219                        intoI = CMLib.map().findFirstRoomItem(mob.location().getArea().getCompleteMap(), mob, intoWhat, true, 5);
220                
221                if(intoI == null)
222                {
223                        mob.tell(L("You have no idea what a '@x1' is.  Perhaps if you saw one again?",intoWhat));
224                        return false;
225                }
226                
227                if((intoI instanceof ArchonOnly)
228                ||(!CMLib.flags().isGettable(intoI))
229                ||(intoI instanceof ClanItem)
230                ||(intoI.basePhyStats().weight() > mob.maxCarry())
231                ||(CMath.bset(intoI.phyStats().sensesMask(), PhyStats.SENSE_ITEMNOWISH)))
232                {
233                        mob.tell(L("You can't polymorph anything into @x1?",intoI.Name()));
234                        return false;
235                }
236                
237                if(intoI.basePhyStats().level()>this.adjustedLevel(mob, asLevel))
238                {
239                        mob.tell(L("You aren't experienced enough to polymorph anything into @x1?",intoI.Name()));
240                        return false;
241                }
242                
243                double pct= 0.3 + (0.2 * super.getXLEVELLevel(mob));
244                int weightDiff = (int)Math.round(CMath.mul(targetI.basePhyStats().weight(),pct));
245                if(intoI.basePhyStats().weight() < (targetI.basePhyStats().weight() - weightDiff))
246                {
247                        mob.tell(L("You can only polymorph an item into one no more than @x1 smaller.  @x2 is too small.",CMath.toPct(pct),intoI.Name()));
248                        return false;
249                }
250                if(intoI.basePhyStats().weight() > (targetI.basePhyStats().weight() + weightDiff))
251                {
252                        mob.tell(L("You can only polymorph an item into one no more than @x1 large.  @x2 is too big.",CMath.toPct(pct),intoI.Name()));
253                        return false;
254                }
255                intoI=(Item)intoI.copyOf();
256                if(intoI.material() != targetI.material())
257                {
258                        intoI.setMaterial(targetI.material());
259                        String oldMaterialName=RawMaterial.CODES.NAME(intoI.material());
260                        String newMaterialName=RawMaterial.CODES.NAME(targetI.material()).toLowerCase();
261                        intoI.setName(CMStrings.replaceWord(intoI.Name(), oldMaterialName, newMaterialName));
262                        intoI.setDisplayText(CMStrings.replaceWord(intoI.displayText(), oldMaterialName, newMaterialName));
263                        intoI.setDescription(CMStrings.replaceWord(intoI.description(), oldMaterialName, newMaterialName));
264                }
265                
266                CMLib.utensils().disenchantItem(intoI);
267                while(intoI.numBehaviors()>0)
268                {
269                        Behavior B=intoI.fetchBehavior(0);
270                        if(B!=null)
271                                intoI.delBehavior(B);
272                }
273                intoI.basePhyStats().setDisposition(intoI.basePhyStats().disposition() & (~PhyStats.IS_BONUS));
274                intoI.recoverPhyStats();
275                intoI.setBaseValue(0);
276                
277                // the reason this costs experience is to make it less valuable than Duplicate 
278                // but more valuable than Wish.
279                final int experienceToLose=getXPCOSTAdjustment(mob,5+intoI.basePhyStats().level());
280                CMLib.leveler().postExperience(mob,null,null,-experienceToLose,false);
281                mob.tell(L("The effort causes you to lose @x1 experience.",""+experienceToLose));
282
283                if(!super.invoke(mob,commands,givenTarget,auto,asLevel))
284                        return false;
285
286                final boolean success=proficiencyCheck(mob,0,auto);
287
288                if(success)
289                {
290                        invoker=mob;
291                        final CMMsg msg=CMClass.getMsg(mob,targetI,this,somanticCastCode(mob,targetI,auto),L("^S<S-NAME> wave(s) <S-HIS-HER> hands around <T-NAME> polymorphing it into @x1.^?",intoI.name()));
292                        if(mob.location().okMessage(mob,msg))
293                        {
294                                mob.location().send(mob,msg);
295
296                                if(msg.value()>0)
297                                        return false;
298                                Spell_PolymorphObject A=(Spell_PolymorphObject)super.beneficialAffect(mob, intoI, asLevel, 0);
299                                if(A!=null)
300                                {
301                                        final List<Item> items = new XVector<Item>();
302                                        items.add(targetI);
303                                        if(targetI instanceof Container)
304                                                items.addAll(((Container)targetI).getDeepContents());
305                                        A.setMiscText(CMLib.coffeeMaker().getItemsXML(items, new Hashtable<String,List<Item>>(),new HashSet<String>(),null).toString());
306                                        A.previousItems = items;
307                                        ItemPossessor possessor = targetI.owner();
308                                        if(possessor != null)
309                                        {
310                                                for(Item I : items)
311                                                        possessor.delItem(I);
312                                                possessor.addItem(intoI);
313                                        }
314                                        else
315                                                mob.addItem(intoI);
316                                }
317                                else
318                                        mob.addItem(intoI);
319                                mob.location().recoverRoomStats();
320                        }
321                }
322                else
323                {
324                        
325                        return beneficialVisualFizzle(mob,targetI,L("<S-NAME> attempt(s) to polymorph <T-NAME> into @x1, but flub(s) it.",intoWhat));
326                }
327
328                // return whether it worked
329                return success;
330        }
