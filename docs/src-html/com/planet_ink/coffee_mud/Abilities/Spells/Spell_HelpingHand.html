037{

085        protected MOB targetM = null;
086        protected int frustration = 0;
087        protected String targetName = "your target";
088        protected List<Room> trail = null;
089        
090        @Override
091        public void setMiscText(String text)
092        {
093                super.setMiscText(text);
094                targetM=null;
095                targetName = "your target";
096                frustration = 0;
097                if(text.length()>0)
098                {
099                        targetName = text;
100                        MOB M=CMLib.players().getPlayer(text);
101                        if((M!=null)&&(CMLib.flags().isInTheGame(M, true)))
102                                targetM=M;
103                }
104        }
105        
106        @Override
107        public void setAffectedOne(Physical P)
108        {
109                if(!(P instanceof MOB))
110                        super.setAffectedOne(P);
111                else
112                {
113                        frustration = 0;
114                        if(CMLib.flags().isInTheGame(P, true))
115                                targetM=(MOB)P;
116                }
117        }
118        
119        @Override
120        public void unInvoke()
121        {
122                final Physical affected=super.affected;
123                // undo the affects of this spell
124                if(affected==null)
125                        return;
126                if(canBeUninvoked())
127                {
128                        if((invoker()!=null)
129                        &&((targetM == null)||(!CMLib.flags().isInTheGame(targetM, true))||(targetM.location()!=invoker().location())))
130                                invoker().tell(L("The helping hand for @x1 has given up.",targetName));
131                        final Room R=CMLib.map().roomLocation(affected);
132                        if(R.isHere(affected))
133                                R.showHappens(CMMsg.MSG_OK_VISUAL,L("The helping hand vanishes."));
134                }
135                super.unInvoke();
136                if(canBeUninvoked() && (!affected.amDestroyed()))
137                        affected.destroy();
138        }

143                super.executeMsg(myHost, msg);
144        }

146        @Override
147        public boolean tick(final Tickable ticking, final int tickID)
148        {
149                if(!super.tick(ticking, tickID))
150                        return false;
151                final MOB targetM = this.targetM;
152                final MOB invokerM = this.invoker();
153                if((targetM == null)
154                ||(!CMLib.flags().isInTheGame(targetM, true))
155                ||(invokerM==null)
156                ||(!CMLib.flags().isInTheGame(invokerM, true))
157                ||(targetM.location()==invokerM.location()))
158                {
159                        unInvoke();
160                        return false;
161                }
162                if(affected instanceof Item)
163                {
164                        final Item handI=(Item)affected;
165                        Room handIR=CMLib.map().roomLocation(handI);
166                        if(targetM.location() != handIR)
167                        {
168                                if((frustration % 2) == 0)
169                                {
170                                        targetM.location().moveItemTo(handI);
171                                        if(frustration < 3)
172                                                targetM.location().show(CMLib.map().deity(), targetM, handI, CMMsg.MSG_OK_VISUAL, L("<O-NAME> floats into the room and gestures for <T-NAME> to follow it."));
173                                        else
174                                        if(frustration < 10)
175                                                targetM.location().show(CMLib.map().deity(), targetM, handI, CMMsg.MSG_OK_VISUAL, L("<O-NAME> floats into the room and urgently gestures for <T-NAME> to follow it."));
176                                        else
177                                        if(frustration < 20)
178                                                targetM.location().show(CMLib.map().deity(), targetM, handI, CMMsg.MSG_OK_VISUAL, L("<O-NAME> floats into the room and wildly gestures for <T-NAME> to follow it."));
179                                        else
180                                        if(frustration < 30)
181                                                targetM.location().show(CMLib.map().deity(), targetM, handI, CMMsg.MSG_OK_VISUAL, L("<O-NAME> floats into the room and INSISTS that <T-NAME> follow it."));
182                                        else
183                                        if(frustration < 40)
184                                                targetM.location().show(CMLib.map().deity(), targetM, handI, CMMsg.MSG_OK_VISUAL, L("<O-NAME> floats into the room, grabs <T-NAME> by the collar and DEMANDS that <T-NAME> follow it."));
185                                        else
186                                        {
187                                                targetM.location().show(CMLib.map().deity(), targetM, handI, CMMsg.MSG_OK_VISUAL, L("<O-NAME> floats into the room, slaps <T-NAME> in the face, and disappears."));
188                                                unInvoke();
189                                                return false;
190                                        }
191                                        handIR=targetM.location();
192                                }
193                                frustration++;
194                                // give the target a second to read
195                                return true;
196                        }
197                        if((trail == null) || (trail.size()==0) || (trail.get(0) != invokerM.location()))
198                                trail = CMLib.tracking().findTrailToRoom(handIR, invokerM.location(), null, 1000);
199                        int nextDirection = CMLib.tracking().trackNextDirectionFromHere(trail, handIR, false);
200                        if(nextDirection < 0)
201                        {
202                                trail = CMLib.tracking().findTrailToRoom(handIR, invokerM.location(), null, 1000);
203                                nextDirection = CMLib.tracking().trackNextDirectionFromHere(trail, handIR, false);
204                        }
205                        final Room nextRoom=handIR.getRoomInDir(nextDirection);
206                        final Exit nextExit=handIR.getExitInDir(nextDirection);
207                        if((nextDirection < 0)||(nextRoom==null)||(nextExit==null))
208                        {
209                                handIR.show(CMLib.map().deity(), targetM, handI, CMMsg.MSG_OK_VISUAL, L("<O-NAME> gives a thumbs-down."));
210                                unInvoke();
211                                return false;
212                        }
213                        if(!nextExit.isOpen())
214                        {
215                                final MOB doorOpenerM=CMClass.getFactoryMOB();
216                                doorOpenerM.setName(handI.Name());
217                                doorOpenerM.basePhyStats().setLevel(handI.basePhyStats().level());
218                                doorOpenerM.phyStats().setLevel(handI.basePhyStats().level());
219                                doorOpenerM.setLocation(handIR);
220                                if((nextExit.hasALock())&&(nextExit.isLocked()))
221                                {
222                                        CMMsg msg=CMClass.getMsg(doorOpenerM,nextExit,null,CMMsg.MSG_OK_VISUAL,CMMsg.MSG_OK_VISUAL,CMMsg.MSG_OK_VISUAL,null);
223                                        if(handIR.okMessage(doorOpenerM,msg))
224                                        {
225                                                msg=CMClass.getMsg(doorOpenerM,nextExit,null,CMMsg.MSG_OK_VISUAL,CMMsg.MSG_UNLOCK,CMMsg.MSG_OK_VISUAL,L("<S-NAME> pick(s) the lock on <T-NAMESELF>."));
226                                                if(handIR.okMessage(doorOpenerM,msg))
227                                                        CMLib.utensils().roomAffectFully(msg,handIR,nextDirection);
228                                        }
229                                }
230                                if(!nextExit.isOpen())
231                                {
232                                        final Vector<String> openCommandV=new ReadOnlyVector<String>(CMParms.parse("OPEN "+CMLib.directions().getDirectionName(nextDirection)));
233                                        doorOpenerM.doCommand(openCommandV,MUDCmdProcessor.METAFLAG_FORCED);
234                                }
235                                doorOpenerM.destroy();
236                        }
237                        handIR.show(CMLib.map().deity(), targetM, handI, CMMsg.MSG_OK_VISUAL, L("<O-NAME> floats @x1.",CMLib.directions().getDirectionName(nextDirection)));
238                        nextRoom.moveItemTo(handI);
239                        if((targetM.location() == handIR) && (targetM.isMonster()) && (invoker()!=null) && (invoker().getGroupMembers(new HashSet<MOB>()).contains(targetM)))
240                                CMLib.tracking().walk(targetM, nextDirection, targetM.isInCombat(), true);
241                }
242                return true;
243        }

245        @Override
246        public boolean invoke(MOB mob, List<String> commands, Physical givenTarget, boolean auto, int asLevel)
247        {
248                if(commands.size()<1)
249                {
250                        if(mob.isMonster() && (mob.getGroupMembers(new HashSet<MOB>()).size()>1))
251                        {
252                                for(MOB M : mob.getGroupMembers(new HashSet<MOB>()))
253                                {
254                                        if((M!=mob)&&(M.isPlayer()))
255                                        {
256                                                commands.add(M.Name());
257                                                break;
258                                        }
259                                }
260                                if(commands.size()<1)
261                                {
262                                        for(MOB M : mob.getGroupMembers(new HashSet<MOB>()))
263                                        {
264                                                if(M!=mob)
265                                                {
266                                                        commands.add(M.Name());
267                                                        break;
268                                                }
269                                        }
270                                }
271                        }
272                        if(commands.size()<1)
273                        {
274                                mob.tell(L("Provide a helping hand for whom?"));
275                                return false;
276                        }
277                }
278                
279                MOB target=null;
280                final String whomName=CMParms.combine(commands);
281                MOB M=CMLib.players().getPlayer(whomName);
282                if((M!=null)&&(CMLib.flags().isInTheGame(M,true)))
283                        target=M;
284                else
285                for(MOB M2 : mob.getGroupMembers(new HashSet<MOB>()))
286                {
287                        if((M2!=null)
288                        &&(CMLib.flags().isInTheGame(M2,true))
289                        &&(CMLib.english().containsString(M2.name(), whomName)||CMLib.english().containsString(M2.displayText(), whomName)))
290                        {
291                                target=M2;
292                                break;
293                        }
294                }
295                if(target==null)
296                {
297                        mob.tell(L("You can't seem to focus on @x1.",whomName));
298                        return false;
299                }
300
301                if(!super.invoke(mob,commands,givenTarget,auto,asLevel))
302                        return false;
303
304                final boolean success=proficiencyCheck(mob,0,auto);
305
306                if(success)
307                {
308                        final CMMsg msg=CMClass.getMsg(mob,target,this,somanticCastCode(mob,target,auto),auto?"":L("^S<S-NAME> wave(s) <S-HIS-HER> arms around, incanting.^?"));
309                        if(mob.location().okMessage(mob,msg))
310                        {
311                                mob.location().send(mob,msg);
312                                Item handI=CMClass.getItem("GenItem");
313                                handI.setName(L("a helping hand"));
314                                handI.setDisplayText(L("a helping hand is here to show @x1 the way.",target.name()));
315                                handI.setDescription(L("It seems to be here for @x1, and wants you to follow where it leads you.",target.name()));
316                                handI.basePhyStats().setLevel(adjustedLevel(mob,asLevel));
317                                handI.phyStats().setLevel(adjustedLevel(mob,asLevel));
318                                CMLib.flags().setGettable(handI, false);
319                                handI.basePhyStats().setDisposition(handI.basePhyStats().disposition()|PhyStats.IS_FLYING);
320                                mob.location().addItem(handI);
321                                final Ability A=beneficialAffect(mob,handI,asLevel,0);
322                                if(A!=null)
323                                {
324                                        A.setMiscText(target.Name());
325                                        A.setAffectedOne(target);
326                                }
327                                else
328                                        handI.destroy();
329                        }
330                }
331                else
332                        beneficialVisualFizzle(mob,target,L("<S-NAME> wave(s) <S-HIS-HER> arms around, incanting, but nothing happens."));
333
334                // return whether it worked
335                return success;
336        }
