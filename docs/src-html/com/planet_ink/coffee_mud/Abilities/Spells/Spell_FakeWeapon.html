036public class Spell_FakeWeapon extends Spell
037{
038
039        @Override
040        public String ID()
041        {
042                return "Spell_FakeWeapon";
043        }
044
045        private final static String localizedName = CMLib.lang().L("Fake Weapon");
046
047        @Override
048        public String name()
049        {
050                return localizedName;
051        }
052
053        @Override
054        protected int canAffectCode()
055        {
056                return CAN_ITEMS;
057        }
058
059        @Override
060        protected int canTargetCode()
061        {
062                return 0;
063        }
064
065        @Override
066        public int classificationCode()
067        {
068                return Ability.ACODE_SPELL|Ability.DOMAIN_ILLUSION;
069        }
070
071        @Override
072        public int abstractQuality()
073        {
074                return Ability.QUALITY_INDIFFERENT;
075        }
076
077        @Override
078        public void unInvoke()
079        {
080                Item item=null;
081                if(affected instanceof Item)
082                        item=(Item)affected;
083                super.unInvoke();
084                if((item != null)&&(super.canBeUninvoked()))
085                        item.destroy();
086        }
087
088        @Override
089        public boolean okMessage(final Environmental myHost, final CMMsg msg)
090        {
091                if((affected!=null)&&(affected instanceof Item))
092                {
093                        if((msg.tool()==affected)
094                        &&(msg.targetMinor()==CMMsg.TYP_DAMAGE))
095                        {
096                                int damageType=Weapon.TYPE_BURSTING;
097                                if(affected instanceof Weapon)
098                                        damageType=((Weapon)affected).weaponDamageType();
099                                if(msg.sourceMessage()!=null)
100                                        msg.setSourceMessage(CMLib.combat().replaceDamageTag(msg.sourceMessage(), msg.value(), damageType, CMMsg.View.SOURCE));
101                                if(msg.targetMessage()!=null)
102                                        msg.setTargetMessage(CMLib.combat().replaceDamageTag(msg.targetMessage(), msg.value(), damageType, CMMsg.View.TARGET));
103                                if(msg.othersMessage()!=null)
104                                        msg.setOthersMessage(CMLib.combat().replaceDamageTag(msg.othersMessage(), msg.value(), damageType, CMMsg.View.OTHERS));
105                                msg.setValue(0);
106                        }
107                        else
108                        if((msg.target()!=null)
109                        &&(!(msg.target() instanceof DeadBody))
110                        &&((msg.target()==affected)
111                                ||(msg.target()==((Item)affected).container())
112                                ||(msg.target()==((Item)affected).ultimateContainer(null))))
113                        {
114                                // what is this trying to prevent, because it's damn broad right now...
115                                if(((CMath.bset(msg.sourceMajor(),CMMsg.MASK_MAGIC))
116                                ||(CMath.bset(msg.targetMajor(),CMMsg.MASK_MAGIC))
117                                ||(CMath.bset(msg.othersMajor(),CMMsg.MASK_MAGIC))))
118                                {
119                                        Room room=null;
120                                        if(msg.source().location()!=null)
121                                                room=msg.source().location();
122                                        if(room==null)
123                                                room=CMLib.map().roomLocation(affected);
124                                        if(room!=null)
125                                                room.showHappens(CMMsg.MSG_OK_VISUAL,L("Magic energy fizzles around @x1 and is absorbed into the air.",affected.Name()));
126                                        return false;
127                                }
128                                else
129                                if(msg.tool() instanceof Ability)
130                                {
131                                        msg.source().tell(L("That doesn't appear to work on @x1",affected.name()));
132                                        return false;
133                                }
134                        }
135                }
136                return super.okMessage(myHost,msg);
137        }
138
139        @Override
140        public boolean invoke(MOB mob, List<String> commands, Physical givenTarget, boolean auto, int asLevel)
141        {
142                final String weaponName=CMParms.combine(commands,0);
143                final String[] choices={"sword","dagger","mace","staff","axe","hammer", "flail"};
144                int choice=-1;
145                for(int i=0;i<choices.length;i++)
146                {
147                        if(choices[i].equalsIgnoreCase(weaponName))
148                                choice=i;
149                }
150                if(choice<0)
151                {
152                        mob.tell(L("You must specify what kind of weapon to create: sword, dagger, mace, flail, staff, axe, or hammer."));
153                        return false;
154                }
155                if(!super.invoke(mob,commands,givenTarget,auto,asLevel))
156                        return false;
157
158                final boolean success=proficiencyCheck(mob,0,auto);
159                if(success)
160                {
161                        final CMMsg msg=CMClass.getMsg(mob,null,this,somanticCastCode(mob,null,auto),auto?"":L("^S<S-NAME> wave(s) <S-HIS-HER> arms around dramatically.^?"));
162                        if(mob.location().okMessage(mob,msg))
163                        {
164                                mob.location().send(mob,msg);
165                                final Weapon weapon=(Weapon)CMClass.getItem("GenWeapon");
166                                weapon.basePhyStats().setAttackAdjustment(100 +(10 * super.getXLEVELLevel(mob)));
167                                weapon.basePhyStats().setDamage(75+(3 * super.getXLEVELLevel(mob)));
168                                weapon.basePhyStats().setDisposition(weapon.basePhyStats().disposition()|PhyStats.IS_BONUS);
169                                weapon.setMaterial(RawMaterial.RESOURCE_COTTON);
170                                switch(choice)
171                                {
172                                case 0:
173                                        weapon.setName(L("a fancy sword"));
174                                        weapon.setDisplayText(L("a fancy sword sits here"));
175                                        weapon.setDescription(L("looks fit to cut something up!"));
176                                        weapon.setWeaponClassification(Weapon.CLASS_SWORD);
177                                        weapon.setWeaponDamageType(Weapon.TYPE_SLASHING);
178                                        break;
179                                case 1:
180                                        weapon.setName(L("a sharp dagger"));
181                                        weapon.setDisplayText(L("a sharp dagger sits here"));
182                                        weapon.setDescription(L("looks fit to cut something up!"));
183                                        weapon.setWeaponClassification(Weapon.CLASS_DAGGER);
184                                        weapon.setWeaponDamageType(Weapon.TYPE_PIERCING);
185                                        break;
186                                case 2:
187                                        weapon.setName(L("a large mace"));
188                                        weapon.setDisplayText(L("a large mace sits here"));
189                                        weapon.setDescription(L("looks fit to whomp on something with!"));
190                                        weapon.setWeaponClassification(Weapon.CLASS_BLUNT);
191                                        weapon.setWeaponDamageType(Weapon.TYPE_BASHING);
192                                        break;
193                                case 3:
194                                        weapon.setName(L("a quarterstaff"));
195                                        weapon.setDisplayText(L("a quarterstaff sits here"));
196                                        weapon.setDescription(L("looks like a reliable weapon"));
197                                        weapon.setWeaponClassification(Weapon.CLASS_STAFF);
198                                        weapon.setWeaponDamageType(Weapon.TYPE_BASHING);
199                                        break;
200                                case 4:
201                                        weapon.setName(L("a deadly axe"));
202                                        weapon.setDisplayText(L("a deadly axe sits here"));
203                                        weapon.setDescription(L("looks fit to shop something up!"));
204                                        weapon.setWeaponClassification(Weapon.CLASS_AXE);
205                                        weapon.setWeaponDamageType(Weapon.TYPE_SLASHING);
206                                        break;
207                                case 5:
208                                        weapon.setName(L("a large hammer"));
209                                        weapon.setDisplayText(L("a large hammer sits here"));
210                                        weapon.setDescription(L("looks fit to pound something into a pulp!"));
211                                        weapon.setWeaponClassification(Weapon.CLASS_HAMMER);
212                                        weapon.setWeaponDamageType(Weapon.TYPE_BASHING);
213                                        break;
214                                case 6:
215                                        weapon.setName(L("a large flail"));
216                                        weapon.setDisplayText(L("a large flail sits here"));
217                                        weapon.setDescription(L("looks fit to pound something into a pulp!"));
218                                        weapon.setWeaponClassification(Weapon.CLASS_FLAILED);
219                                        weapon.setWeaponDamageType(Weapon.TYPE_BASHING);
220                                        break;
221                                }
222                                weapon.basePhyStats().setWeight(0);
223                                weapon.setBaseValue(0);
224                                weapon.recoverPhyStats();
225                                mob.addItem(weapon);
226                                mob.location().show(mob,null,weapon,CMMsg.MSG_OK_ACTION,L("Suddenly, <S-NAME> own(s) <O-NAME>!"));
227                                beneficialAffect(mob,weapon,asLevel,0);
228                        }
229                }
230                else
231                        beneficialVisualFizzle(mob,null,L("<S-NAME> dramatically wave(s) <S-HIS-HER> arms around, but fizzle(s) the spell."));
232
233                // return whether it worked
234                return success;
235        }
