036public class Spell_Portal extends Spell
037{
038
039        @Override
040        public String ID()
041        {
042                return "Spell_Portal";
043        }
044
045        private final static String localizedName = CMLib.lang().L("Portal");
046
047        @Override
048        public String name()
049        {
050                return localizedName;
051        }
052
053        @Override
054        protected int canTargetCode()
055        {
056                return 0;
057        }
058
059        @Override
060        public int classificationCode()
061        {
062                return Ability.ACODE_SPELL|Ability.DOMAIN_EVOCATION;
063        }
064
065        @Override
066        public long flags()
067        {
068                return Ability.FLAG_TRANSPORTING;
069        }
070
071        @Override
072        protected int overrideMana()
073        {
074                return Ability.COST_ALL-90;
075        }
076
077        @Override
078        public int abstractQuality()
079        {
080                return Ability.QUALITY_INDIFFERENT;
081        }
082
083        Room newRoom=null;
084        Room oldRoom=null;
085
086        @Override
087        public void unInvoke()
088        {
089                if(canBeUninvoked())
090                {
091                        if(newRoom!=null)
092                        {
093                                newRoom.showHappens(CMMsg.MSG_OK_VISUAL,L("The swirling portal closes."));
094                                newRoom.rawDoors()[Directions.GATE]=null;
095                                newRoom.setRawExit(Directions.GATE,null);
096                        }
097                        if(oldRoom!=null)
098                        {
099                                oldRoom.showHappens(CMMsg.MSG_OK_VISUAL,L("The swirling portal closes."));
100                                oldRoom.rawDoors()[Directions.GATE]=null;
101                                oldRoom.setRawExit(Directions.GATE,null);
102                        }
103                }
104                super.unInvoke();
105        }
106
107        @Override
108        public boolean invoke(MOB mob, List<String> commands, Physical givenTarget, boolean auto, int asLevel)
109        {
110                newRoom=null;
111                oldRoom=null;
112
113                if((auto||mob.isMonster())&&((commands.size()<1)||((commands.get(0)).equals(mob.name()))))
114                {
115                        commands.clear();
116                        if(text().trim().length()>0)
117                                commands.add(text());
118                        else
119                                commands.add(CMLib.map().getRandomRoom().displayText());
120                }
121                if(commands.size()<1)
122                {
123                        mob.tell(L("Create a portal to where?"));
124                        return false;
125                }
126                if((mob.location().getRoomInDir(Directions.GATE)!=null)
127                ||(mob.location().getExitInDir(Directions.GATE)!=null))
128                {
129                        mob.tell(L("A portal cannot be created here."));
130                        return false;
131                }
132                final String areaName=CMParms.combine(commands,0).trim().toUpperCase();
133                oldRoom=null;
134                newRoom=null;
135                int tries=0;
136                while(((++tries)<10000))
137                {
138                        final Room room=CMLib.map().getRandomRoom();
139                        if((CMLib.flags().canAccess(mob,room))
140                        &&(CMLib.english().containsString(room.displayText(mob),areaName)))
141                        {
142                           newRoom=room;
143                           break;
144                        }
145                }
146                if(newRoom==null)
147                {
148                        try
149                        {
150                                final List<Room> rooms=CMLib.map().findRooms(CMLib.map().rooms(), mob, areaName, true, 10);
151                                if(rooms.size()>0)
152                                        newRoom=rooms.get(CMLib.dice().roll(1,rooms.size(),-1));
153                        }
154                        catch(final NoSuchElementException nse)
155                        {
156                        }
157                }
158
159                if(newRoom==null)
160                {
161                        mob.tell(L("You don't know of a place called '@x1'.",CMParms.combine(commands,0)));
162                        return false;
163                }
164
165                int profNeg=0;
166                for(int i=0;i<newRoom.numInhabitants();i++)
167                {
168                        final MOB t=newRoom.fetchInhabitant(i);
169                        if(t!=null)
170                        {
171                                int adjustment=t.phyStats().level()-(mob.phyStats().level()+(2*getXLEVELLevel(mob)));
172                                if(t.isMonster())
173                                        adjustment=adjustment*3;
174                                profNeg+=adjustment;
175                        }
176                }
177                profNeg+=newRoom.numItems()*20;
178
179                if(!super.invoke(mob,commands,givenTarget,auto,asLevel))
180                        return false;
181
182                final boolean success=proficiencyCheck(mob,-profNeg,auto);
183
184                if((success)
185                &&((newRoom.getRoomInDir(Directions.GATE)==null)
186                &&(newRoom.getExitInDir(Directions.GATE)==null)))
187                {
188                        final CMMsg msg=CMClass.getMsg(mob,mob.location(),this,verbalCastCode(mob,mob.location(),auto),L("^S<S-NAME> evoke(s) a blinding, swirling portal here.^?"));
189                        final CMMsg msg2=CMClass.getMsg(mob,newRoom,this,verbalCastCode(mob,newRoom,auto),L("A blinding, swirling portal appears here."));
190                        if((mob.location().okMessage(mob,msg))&&(newRoom.okMessage(mob,msg2)))
191                        {
192                                mob.location().send(mob,msg);
193                                newRoom=(Room)msg2.target();
194                                newRoom.send(mob,msg2);
195                                final Exit e=CMClass.getExit("GenExit");
196                                e.setDescription(L("A swirling portal to somewhere"));
197                                e.setDisplayText(L("A swirling portal to somewhere"));
198                                e.setDoorsNLocks(false,true,false,false,false,false);
199                                e.setExitParams("portal","close","open","closed.");
200                                e.setName(L("a swirling portal"));
201                                final Ability A1=CMClass.getAbility("Prop_RoomView");
202                                if(A1!=null)
203                                {
204                                        A1.setMiscText(CMLib.map().getExtendedRoomID(newRoom));
205                                        e.addNonUninvokableEffect(A1);
206                                }
207                                final Exit e2=(Exit)e.copyOf();
208                                final Ability A2=CMClass.getAbility("Prop_RoomView");
209                                if(A2!=null)
210                                {
211                                        A2.setMiscText(CMLib.map().getExtendedRoomID(mob.location()));
212                                        e2.addNonUninvokableEffect(A2);
213                                }
214                                mob.location().rawDoors()[Directions.GATE]=newRoom;
215                                newRoom.rawDoors()[Directions.GATE]=mob.location();
216                                mob.location().setRawExit(Directions.GATE,e);
217                                newRoom.setRawExit(Directions.GATE,e2);
218                                oldRoom=mob.location();
219                                beneficialAffect(mob,e,asLevel,15);
220                        }
221                }
222                else
223                        beneficialWordsFizzle(mob,null,L("<S-NAME> attempt(s) to evoke a portal, but fizzle(s) the spell."));
224
225                // return whether it worked
226                return success;
227        }
