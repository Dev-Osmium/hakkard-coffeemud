081        public int classificationCode()
082        {
083                return Ability.ACODE_SPELL|Ability.DOMAIN_DIVINATION;
084        }

086        @Override
087        public void unInvoke()
088        {
089                if(!(affected instanceof MOB))
090                        return;
091                final MOB mob=(MOB)affected;
092                if(canBeUninvoked())
093                        lastRoom=null;
094                super.unInvoke();
095                if(canBeUninvoked())
096                        mob.tell(L("Your senses are no longer sensitive to traps."));
097        }
098
099        public String trapCheck(Physical P)
100        {
101                if(P!=null)
102                {
103                        if(CMLib.utensils().fetchMyTrap(P)!=null)
104                                return L("@x1 is trapped.\n\r",P.name());
105                }
106                return "";
107        }
108
109        public String trapHere(MOB mob, Physical P)
110        {
111                final StringBuffer msg=new StringBuffer("");
112                if(P==null)
113                        return msg.toString();
114                if((P instanceof Room)&&(CMLib.flags().canBeSeenBy(P,mob)))
115                        msg.append(trapCheck(mob.location()));
116                else
117                if((P instanceof Container)&&(CMLib.flags().canBeSeenBy(P,mob)))
118                {
119                        final Container C=(Container)P;
120                        final List<Item> V=C.getDeepContents();
121                        for(int v=0;v<V.size();v++)
122                        {
123                                if(trapCheck(V.get(v)).length()>0)
124                                        msg.append(L("@x1 contains something trapped.",C.name()));
125                        }
126                }
127                else
128                if((P instanceof Item)&&(CMLib.flags().canBeSeenBy(P,mob)))
129                        msg.append(trapCheck(P));
130                else
131                if((P instanceof Exit)&&(CMLib.flags().canBeSeenBy(P,mob)))
132                {
133                        final Room room=mob.location();
134                        if(room!=null)
135                        {
136                                for(int d=Directions.NUM_DIRECTIONS()-1;d>=0;d--)
137                                {
138                                        if(room.getExitInDir(d)==P)
139                                        {
140                                                final Exit E2=room.getReverseExit(d);
141                                                final Room R2=room.getRoomInDir(d);
142                                                msg.append(trapCheck(P));
143                                                msg.append(trapCheck(E2));
144                                                msg.append(trapCheck(R2));
145                                                break;
146                                        }
147                                }
148                        }
149                }
150                else
151                if((P instanceof MOB)&&(CMLib.flags().canBeSeenBy(P,mob)))
152                {
153                        for(int i=0;i<((MOB)P).numItems();i++)
154                        {
155                                final Item I=((MOB)P).getItem(i);
156                                if(trapCheck(I).length()>0)
157                                        return P.name()+" is carrying something trapped.";
158                        }
159                        final ShopKeeper SK=CMLib.coffeeShops().getShopKeeper(P);
160                        if(SK!=null)
161                        {
162                                for(final Iterator<Environmental> i=SK.getShop().getStoreInventory();i.hasNext();)
163                                {
164                                        final Environmental E2=i.next();
165                                        if(E2 instanceof Item)
166                                        {
167                                                if(trapCheck((Item)E2).length()>0)
168                                                        return P.name()+" has something trapped in stock.";
169                                        }
170                                }
171                        }
172                }
173                return msg.toString();
174        }
175
176        @Override
177        public void executeMsg(final Environmental myHost, final CMMsg msg)
178        {
179                super.executeMsg(myHost,msg);
180                if((affected!=null)
181                &&(affected instanceof MOB)
182                &&(msg.target() instanceof Physical)
183                &&(msg.amISource((MOB)affected))
184                &&((msg.sourceMinor()==CMMsg.TYP_LOOK)||(msg.sourceMinor()==CMMsg.TYP_EXAMINE)))
185                {
186                        if((msg.tool()!=null)&&(msg.tool().ID().equals(ID())))
187                        {
188                                final String str=trapHere((MOB)affected,(Physical)msg.target());
189                                if(str.length()>0)
190                                        ((MOB)affected).tell(str);
191                        }
192                        else
193                        if((trapHere((MOB)affected,(Physical)msg.target()).length()>0)
194                        &&(msg.source()!=msg.target()))
195                        {
196                                final CMMsg msg2=CMClass.getMsg(msg.source(),msg.target(),this,CMMsg.MSG_LOOK,CMMsg.NO_EFFECT,CMMsg.NO_EFFECT,null);
197                                msg.addTrailerMsg(msg2);
198                        }
199                }
200        }
201
202        @Override
203        public int castingQuality(MOB mob, Physical target)
204        {
205                if(mob!=null)
206                {
207                        if(target instanceof MOB)
208                        {
209                                if(((MOB)target).isInCombat()||((MOB)target).isMonster())
210                                        return Ability.QUALITY_INDIFFERENT;
211                        }
212                }
213                return super.castingQuality(mob,target);
214        }
215
216        @Override
217        public boolean invoke(MOB mob, List<String> commands, Physical givenTarget, boolean auto, int asLevel)
218        {
219                MOB target=mob;
220                if((auto)&&(givenTarget!=null)&&(givenTarget instanceof MOB))
221                        target=(MOB)givenTarget;
222                if(target.fetchEffect(this.ID())!=null)
223                {
224                        mob.tell(target,null,null,L("<S-NAME> <S-IS-ARE> already detecting traps."));
225                        return false;
226                }
227                if(!super.invoke(mob,commands,givenTarget,auto,asLevel))
228                        return false;
229
230                final boolean success=proficiencyCheck(mob,0,auto);
231
232                if(success)
233                {
234                        final CMMsg msg=CMClass.getMsg(mob,target,this,verbalCastCode(mob,target,auto),auto?L("<T-NAME> gain(s) trap sensitivities!"):L("^S<S-NAME> incant(s) softly, and gain(s) sensitivity to traps!^?"));
235                        if(mob.location().okMessage(mob,msg))
236                        {
237                                mob.location().send(mob,msg);
238                                beneficialAffect(mob,target,asLevel,0);
239                        }
240                }
241                else
242                        beneficialVisualFizzle(mob,null,L("<S-NAME> incant(s) and open(s) <S-HIS-HER> eyes, but the spell fizzles."));
243
244                return success;
245        }
