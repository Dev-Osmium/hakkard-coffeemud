036public class Spell_AnalyzeDweomer extends Spell
037{
038
039        @Override
040        public String ID()
041        {
042                return "Spell_AnalyzeDweomer";
043        }
044
045        private final static String localizedName = CMLib.lang().L("Analyze Item");
046
047        @Override
048        public String name()
049        {
050                return localizedName;
051        }
052
053        @Override
054        public int abstractQuality()
055        {
056                return Ability.QUALITY_INDIFFERENT;
057        }
058
059        @Override
060        protected int canTargetCode()
061        {
062                return CAN_ITEMS;
063        }
064
065        @Override
066        public int classificationCode()
067        {
068                return Ability.ACODE_SPELL|Ability.DOMAIN_DIVINATION;
069        }
070
071        @Override
072        public boolean invoke(MOB mob, List<String> commands, Physical givenTarget, boolean auto, int asLevel)
073        {
074                final Item target=getTarget(mob,mob.location(),givenTarget,commands,Wearable.FILTER_ANY);
075                if(target==null)
076                        return false;
077
078                if(!super.invoke(mob,commands,givenTarget,auto,asLevel))
079                        return false;
080
081                final boolean success=proficiencyCheck(mob,0,auto);
082
083                if(success)
084                {
085                        final CMMsg msg=CMClass.getMsg(mob,target,this,verbalCastCode(mob,target,auto),auto?"":L("^S<S-NAME> analyze(s) the nature of <T-NAMESELF> carefully.^?"));
086                        if(mob.location().okMessage(mob,msg))
087                        {
088                                mob.location().send(mob,msg);
089                                final StringBuffer str=new StringBuffer("");
090                                if(target instanceof Armor)
091                                {
092                                        str.append(L("It is a kind of armor.  "));
093                                        if(!target.rawLogicalAnd())
094                                                str.append(L("It is worn on any one of the following: "));
095                                        else
096                                                str.append(L("It is worn on all of the following: "));
097                                        final Wearable.CODES codes = Wearable.CODES.instance();
098                                        for(final long wornCode : codes.all())
099                                        {
100                                                if(wornCode!=Wearable.IN_INVENTORY)
101                                                {
102                                                        if((codes.name(wornCode).length()>0)
103                                                        &&(((target.rawProperLocationBitmap()&wornCode)==wornCode)))
104                                                                str.append(codes.name(wornCode).toLowerCase()+" ");
105                                                }
106                                        }
107                                        str.append(".  ");
108                                }
109                                if((target instanceof Container)&&(((Container)target).capacity()>0))
110                                        str.append(L("It is a container.  "));
111                                if(target instanceof Coins)
112                                        str.append(L("It is currency. "));
113                                if(target instanceof Drink)
114                                        str.append(L("You can drink it. "));
115                                if(target instanceof Food)
116                                        str.append(L("You can eat it.  "));
117                                if(target instanceof Pill)
118                                        str.append(L("It is a magic pill.  "));
119                                if(target instanceof Potion)
120                                        str.append(L("It is a magic potion.  "));
121                                if(target instanceof Light)
122                                        str.append(L("It is a light source.  "));
123                                if(target instanceof com.planet_ink.coffee_mud.Items.interfaces.RoomMap)
124                                        str.append(L("It is a map.  "));
125                                if(target instanceof MiscMagic)
126                                        str.append(L("It has a magical aura.  "));
127                                if(target instanceof Scroll)
128                                {
129                                        if(target instanceof MiscMagic)
130                                                str.append(L("It is a magic scroll.  "));
131                                        else
132                                                str.append(L("It is an instructional document.  "));
133                                }
134                                if(target instanceof Wand)
135                                        str.append(L("It is a magic wand.  "));
136                                if(target instanceof Electronics)
137                                        str.append(L("It is some sort of high technology.  "));
138                                if(target instanceof InnKey)
139                                        str.append(L("It is an Inn key.  "));
140                                else
141                                if(target instanceof DoorKey)
142                                        str.append(L("It is a key.  "));
143                                if(target instanceof LandTitle)
144                                        str.append(L("It is a property title.  "));
145                                if(target.isReadable())
146                                        str.append(L("It is readable.  "));
147                                if(target instanceof DeadBody)
148                                        str.append(L("It is a corpse of a @x1.  ",((DeadBody)target).charStats().getMyRace().name()));
149                                if(target instanceof Weapon)
150                                {
151                                        final Weapon w=(Weapon)target;
152                                        str.append(L("It is a @x1 weapon.  ",Weapon.CLASS_DESCS[w.weaponClassification()].toLowerCase()));
153                                        str.append(L("It does @x1 damage.  ",Weapon.TYPE_DESCS[w.weaponDamageType()].toLowerCase()));
154                                        if(w.minRange()>0)
155                                                str.append(L("It has a minimum range of @x1.  ",""+w.minRange()));
156                                        if(w.maxRange()>w.minRange())
157                                                str.append(L("It has a maximum range of @x1.  ",""+w.maxRange()));
158                                }
159                                str.append(L("It is made of @x1.  ",RawMaterial.CODES.NAME(target.material()).toLowerCase()));
160                                final Command C=CMClass.getCommand("Affect");
161                                try
162                                {
163                                        final String affectStr=C.executeInternal(mob,0,target).toString();
164                                        if(affectStr.length()<5)
165                                                str.append(L("It is affected by: @x1",affectStr));
166                                }
167                                catch(final Exception e)
168                                {
169                                }
170                                if(mob.isMonster())
171                                        CMLib.commands().postSay(mob,null,str.toString().trim(),false,false);
172                                else
173                                        mob.tell(str.toString().trim());
174                        }
175
176                }
177                else
178                        beneficialVisualFizzle(mob,target,L("<S-NAME> analyze(s) the nature of <T-NAMESELF>, looking more frustrated every second."));
179
180                // return whether it worked
181                return success;
182        }
183}
