036public class Spell_Scribe extends Spell
037{
038
039        @Override
040        public String ID()
041        {
042                return "Spell_Scribe";
043        }
044
045        private final static String     localizedName   = CMLib.lang().L("Scribe");
046
047        @Override
048        public String name()
049        {
050                return localizedName;
051        }
052
053        @Override
054        protected int canTargetCode()
055        {
056                return CAN_ITEMS;
057        }
058
059        @Override
060        public int classificationCode()
061        {
062                return Ability.ACODE_SPELL | Ability.DOMAIN_EVOCATION;
063        }
064
065        @Override
066        protected int overrideMana()
067        {
068                return Ability.COST_ALL-50;
069        }
070
071        @Override
072        public long flags()
073        {
074                return Ability.FLAG_NOORDERING;
075        }
076
077        @Override
078        public int abstractQuality()
079        {
080                return Ability.QUALITY_INDIFFERENT;
081        }
082
083        @Override
084        public boolean invoke(MOB mob, List<String> commands, Physical givenTarget, boolean auto, int asLevel)
085        {
086                if(commands.size()<2)
087                {
088                        mob.tell(L("Scribe which spell onto what?"));
089                        return false;
090                }
091                final Physical target=mob.location().fetchFromMOBRoomFavorsItems(mob,null,commands.get(commands.size()-1),Wearable.FILTER_UNWORNONLY);
092                if((target==null)||(!CMLib.flags().canBeSeenBy(target,mob)))
093                {
094                        mob.tell(L("You don't see '@x1' here.",(commands.get(commands.size()-1))));
095                        return false;
096                }
097                if((!(target instanceof Scroll))
098                ||(!(target instanceof MiscMagic)))
099                {
100                        mob.tell(L("You can't scribe magic onto that."));
101                        return false;
102                }
103
104                commands.remove(commands.size()-1);
105                final Scroll scroll=(Scroll)target;
106
107                final String spellName=CMParms.combine(commands,0).trim();
108                Spell scrollThis=null;
109                for(final Enumeration<Ability> a=mob.allAbilities();a.hasMoreElements();)
110                {
111                        final Ability A=a.nextElement();
112                        if((A!=null)
113                        &&(A instanceof Spell)
114                        &&(A.name().equalsIgnoreCase(spellName))
115                        &&(!A.ID().equals(this.ID())))
116                                scrollThis=(Spell)A;
117                }
118                if(scrollThis==null)
119                {
120                        for(final Enumeration<Ability> a=mob.allAbilities();a.hasMoreElements();)
121                        {
122                                final Ability A=a.nextElement();
123                                if((A!=null)
124                                &&(A instanceof Spell)
125                                &&(CMLib.english().containsString(A.name(),spellName))
126                                &&(!A.ID().equals(this.ID())))
127                                        scrollThis=(Spell)A;
128                        }
129                }
130                
131                if(scrollThis==null)
132                {
133                        mob.tell(L("You don't know how to scribe '@x1'.",spellName));
134                        return false;
135                }
136                if(CMLib.ableMapper().lowestQualifyingLevel(scrollThis.ID())>24)
137                {
138                        mob.tell(L("That spell is too powerful to scribe."));
139                        return false;
140                }
141
142                int numSpells=(CMLib.ableMapper().qualifyingClassLevel(mob,this)-CMLib.ableMapper().qualifyingLevel(mob,this));
143                if(numSpells<0) 
144                        numSpells=1;
145                if(scroll.getSpells().size()>numSpells)
146                {
147                        mob.tell(L("You aren't powerful enough to scribe any more spells onto @x1.",scroll.name()));
148                        return false;
149                }
150
151                final List<Ability> spells=new XVector<Ability>(scroll.getSpells());
152                if(scroll.usesRemaining()==0)
153                        spells.clear();
154                else
155                for(final Ability spell: spells)
156                {
157                        if(spell.ID().equals(scrollThis.ID()))
158                        {
159                                mob.tell(L("That spell is already scribed onto @x1.",scroll.name()));
160                                return false;
161                        }
162                }
163
164                int level=25;
165                for(final Ability A : spells)
166                {
167                        int lvl=CMLib.ableMapper().qualifyingLevel(mob,A);
168                        if(lvl<0)
169                                lvl=CMLib.ableMapper().lowestQualifyingLevel(A.ID());
170                        level -= lvl;
171                }
172                if(level <= 0)
173                {
174                        mob.tell(L("You can only scribe on blank scrolls, or scroll with less than 25 levels of spells on it."));
175                        return false;
176                }
177                
178                int experienceToLose=5+CMLib.ableMapper().lowestQualifyingLevel(scrollThis.ID());
179                if((mob.getExperience()-experienceToLose)<0)
180                {
181                        mob.tell(L("You don't have enough experience to cast this spell."));
182                        return false;
183                }
184                // lose all the mana!
185                if(!super.invoke(mob,commands,givenTarget,auto,asLevel))
186                        return false;
187
188                experienceToLose=getXPCOSTAdjustment(mob,experienceToLose);
189                CMLib.leveler().postExperience(mob,null,null,-experienceToLose,false);
190                mob.tell(L("You lose @x1 experience points for the effort.",""+experienceToLose));
191
192                final boolean success=proficiencyCheck(mob,0,auto);
193
194                if(success)
195                {
196                        setMiscText(scrollThis.ID());
197                        final CMMsg msg=CMClass.getMsg(mob,target,this,verbalCastCode(mob,target,auto),L("^S<S-NAME> move(s) <S-HIS-HER> fingers around <T-NAMESELF>, incanting softly.^?"));
198                        if(mob.location().okMessage(mob,msg))
199                        {
200                                mob.location().send(mob,msg);
201                                if(scroll.usesRemaining()==0)
202                                        scroll.setSpellList("");
203                                if(scroll.getSpellList().trim().length()==0)
204                                        scroll.setSpellList(scrollThis.ID());
205                                else
206                                        scroll.setSpellList(scroll.getSpellList()+";"+scrollThis.ID());
207                                if((scroll.usesRemaining()==Integer.MAX_VALUE)||(scroll.usesRemaining()<0))
208                                        scroll.setUsesRemaining(0);
209                                scroll.setUsesRemaining(scroll.usesRemaining()+1);
210                        }
211
212                }
213                else
214                        beneficialWordsFizzle(mob,target,L("<S-NAME> move(s) <S-HIS-HER> fingers around <T-NAMESELF>, incanting softly, and looking very frustrated."));
215
216                // return whether it worked
217                return success;
218        }
