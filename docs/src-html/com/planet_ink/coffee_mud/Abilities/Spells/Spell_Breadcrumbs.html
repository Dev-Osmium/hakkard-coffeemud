036public class Spell_Breadcrumbs extends Spell
037{
038
039        @Override
040        public String ID()
041        {
042                return "Spell_Breadcrumbs";
043        }
044
045        private final static String localizedName = CMLib.lang().L("Breadcrumbs");
046
047        @Override
048        public String name()
049        {
050                return localizedName;
051        }
052
053        @Override
054        public int abstractQuality()
055        {
056                return Ability.QUALITY_OK_SELF;
057        }
058
059        @Override
060        protected int canAffectCode()
061        {
062                return CAN_MOBS;
063        }
064
065        @Override
066        public int classificationCode()
067        {
068                return Ability.ACODE_SPELL|Ability.DOMAIN_DIVINATION;
069        }
070
071        public Vector<Room> trail=null;
072
073        @Override
074        public void unInvoke()
075        {
076                // undo the affects of this spell
077                if(!(affected instanceof MOB))
078                        return;
079                final MOB mob=(MOB)affected;
080                super.unInvoke();
081
082                if(canBeUninvoked())
083                        mob.tell(L("Your breadcrumbs fade away."));
084                trail=null;
085        }
086
087        @Override
088        public String displayText()
089        {
090                final StringBuffer str=new StringBuffer(L("(Breadcrumb Trail: "));
091                if(trail!=null)
092                {
093                        synchronized(trail)
094                        {
095                                Room lastRoom=null;
096                                for(int v=trail.size()-1;v>=0;v--)
097                                {
098                                        final Room R=trail.elementAt(v);
099                                        if(lastRoom!=null)
100                                        {
101                                                int dir=-1;
102                                                for(int d=Directions.NUM_DIRECTIONS()-1;d>=0;d--)
103                                                {
104                                                        if(lastRoom.getRoomInDir(d)==R)
105                                                        {
106                                                                dir=d;
107                                                                break;
108                                                        }
109                                                }
110                                                if(dir>=0)
111                                                        str.append(CMLib.directions().getDirectionName(dir)+" ");
112                                                else
113                                                        str.append(L("Unknown "));
114                                        }
115                                        lastRoom=R;
116                                }
117                        }
118                }
119                return str.toString()+")";
120        }
121
122        @Override
123        public void executeMsg(final Environmental myHost, final CMMsg msg)
124        {
125                if(!(affected instanceof MOB))
126                        return;
127                final MOB mob=(MOB)affected;
128                if((msg.amISource(mob))
129                &&(trail!=null)
130                &&(msg.targetMinor()==CMMsg.TYP_ENTER)
131                &&(msg.target() instanceof Room))
132                {
133                        final Room newRoom=(Room)msg.target();
134                        boolean kill=false;
135                        int t=0;
136                        while(t<trail.size())
137                        {
138                                if(kill)
139                                        trail.removeElement(trail.elementAt(t));
140                                else
141                                {
142                                        final Room R=trail.elementAt(t);
143                                        if(R==newRoom)
144                                                kill=true;
145                                        t++;
146                                }
147                        }
148                        if(kill)
149                                return;
150                        for(int d=Directions.NUM_DIRECTIONS()-1;d>=0;d--)
151                        {
152                                final Room adjacentRoom=newRoom.getRoomInDir(d);
153                                if((adjacentRoom!=null)
154                                   &&(newRoom.getExitInDir(d)!=null))
155                                {
156                                        kill=false;
157                                        t=0;
158                                        while(t<trail.size())
159                                        {
160                                                if(kill)
161                                                        trail.removeElement(trail.elementAt(t));
162                                                else
163                                                {
164                                                        final Room R=trail.elementAt(t);
165                                                        if(R==adjacentRoom)
166                                                                kill=true;
167                                                        t++;
168                                                }
169                                        }
170                                }
171                        }
172                        trail.addElement(newRoom);
173                }
174        }
175
176        @Override
177        public boolean invoke(MOB mob, List<String> commands, Physical givenTarget, boolean auto, int asLevel)
178        {
179                MOB target=mob;
180                if(target==null)
181                        return false;
182                if((auto)&&(givenTarget!=null)&&(givenTarget instanceof MOB))
183                        target=(MOB)givenTarget;
184                if(target.fetchEffect(this.ID())!=null)
185                {
186                        mob.tell(target,null,null,L("<S-NAME> <S-IS-ARE> already dropping breadcrumbs."));
187                        return false;
188                }
189
190                if(!super.invoke(mob,commands,givenTarget,auto,asLevel))
191                        return false;
192
193                // now see if it worked
194                final boolean success=proficiencyCheck(mob,0,auto);
195                if(success)
196                {
197                        final CMMsg msg=CMClass.getMsg(mob,target,this,verbalCastCode(mob,target,auto),auto?L("<T-NAME> attain(s) mysterious breadcrumbs."):L("^S<S-NAME> invoke(s) the mystical breadcrumbs.^?"));
198                        if(mob.location().okMessage(mob,msg))
199                        {
200                                mob.location().send(mob,msg);
201                                trail=new Vector<Room>();
202                                trail.addElement(mob.location());
203                                beneficialAffect(mob,target,asLevel,0);
204                        }
205                }
206                else
207                        return beneficialWordsFizzle(mob,target,L("<S-NAME> attempt(s) to invoke breadcrumbs, but fail(s)."));
208
209                // return whether it worked
210                return success;
211        }
212}
