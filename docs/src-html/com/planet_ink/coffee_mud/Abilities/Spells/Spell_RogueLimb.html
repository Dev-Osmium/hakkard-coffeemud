036public class Spell_RogueLimb extends Spell
037{
038
039        @Override
040        public String ID()
041        {
042                return "Spell_RogueLimb";
043        }
044
045        private final static String localizedName = CMLib.lang().L("Rogue Limb");
046
047        @Override
048        public String name()
049        {
050                return localizedName;
051        }
052
053        private final static String localizedStaticDisplay = CMLib.lang().L("(Rogue Limb)");
054
055        @Override
056        public String displayText()
057        {
058                return localizedStaticDisplay;
059        }
060
061        @Override
062        public int abstractQuality()
063        {
064                return Ability.QUALITY_MALICIOUS;
065        }
066
067        @Override
068        protected int canTargetCode()
069        {
070                return CAN_MOBS;
071        }
072
073        @Override
074        protected int canAffectCode()
075        {
076                return CAN_MOBS;
077        }
078
079        @Override
080        public int classificationCode()
081        {
082                return Ability.ACODE_SPELL|Ability.DOMAIN_ENCHANTMENT;
083        }
084
085        public MOB rogueLimb=null;
086
087        @Override
088        public boolean tick(final Tickable ticking, final int tickID)
089        {
090                if((rogueLimb!=null)
091                &&(affected instanceof MOB))
092                {
093                        if(rogueLimb.location()!=((MOB)affected).location())
094                        {
095                                ((MOB)affected).location().bringMobHere(rogueLimb,false);
096                                rogueLimb.setVictim((MOB)affected);
097                        }
098                        if((rogueLimb.amFollowing()!=null)
099                        ||(rogueLimb.getVictim()!=affected)
100                        ||(!CMLib.flags().isAliveAwakeMobileUnbound(rogueLimb,true))
101                        ||(!CMLib.flags().isAliveAwakeMobileUnbound((MOB)affected,true))
102                        ||(!CMLib.flags().isInTheGame((MOB)affected,false))
103                        ||(!CMLib.flags().isInTheGame(rogueLimb,false)))
104                                unInvoke();
105                }
106                else
107                        unInvoke();
108                return super.tick(ticking,tickID);
109        }
110
111        @Override
112        public boolean okMessage(final Environmental myHost, final CMMsg msg)
113        {
114                if(!super.okMessage(myHost,msg))
115                        return false;
116                if((affected==null)
117                   ||(!(affected instanceof MOB))
118                   ||(rogueLimb==null))
119                        return true;
120                if(msg.amITarget(rogueLimb)
121                &&(CMLib.flags().isAliveAwakeMobileUnbound(rogueLimb,true))
122                &&(CMLib.flags().isAliveAwakeMobileUnbound((MOB)affected,true))
123                &&(msg.targetMinor()==CMMsg.TYP_DAMAGE))
124                        CMLib.combat().postDamage(rogueLimb,(MOB)affected,this,msg.value(),CMMsg.MASK_ALWAYS|msg.sourceCode(),Weapon.TYPE_NATURAL,null);
125                if(msg.amISource(rogueLimb)
126                &&(msg.sourceMinor()==CMMsg.TYP_DEATH))
127                {
128                        unInvoke();
129                        return false;
130                }
131                return true;
132        }
133
134        @Override
135        public void unInvoke()
136        {
137                if((affected!=null)
138                &&(affected instanceof MOB))
139                        ((MOB)affected).location().show(((MOB)affected),rogueLimb,null,CMMsg.MSG_OK_ACTION,L("<S-NAME> gain(s) control of <T-NAMESELF>."));
140                if((rogueLimb!=null)&&(super.canBeUninvoked()))
141                {
142                        rogueLimb.destroy();
143                        rogueLimb=null;
144                }
145                super.unInvoke();
146        }
147
148        @Override
149        public boolean invoke(MOB mob, List<String> commands, Physical givenTarget, boolean auto, int asLevel)
150        {
151                final MOB target=getTarget(mob,commands,givenTarget);
152                if(target==null)
153                        return false;
154
155                if(!super.invoke(mob,commands,givenTarget,auto,asLevel))
156                        return false;
157
158                final boolean success=proficiencyCheck(mob,0,auto);
159
160                if(success)
161                {
162                        final CMMsg msg=CMClass.getMsg(mob,target,this,verbalCastCode(mob,target,auto),auto?L("<T-NAME> lose(s) control of <T-HIS-HER> limb!"):L("^S<S-NAME> invoke(s) a powerful spell upon <T-NAMESELF>!^?"));
163                        if(mob.location().okMessage(mob,msg))
164                        {
165                                mob.location().send(mob,msg);
166                                if(msg.value()<=0)
167                                {
168                                        final Vector<Integer> limbs=new Vector<Integer>();
169                                        final Race theRace=target.charStats().getMyRace();
170                                        for(int i=0;i<Race.BODY_PARTS;i++)
171                                        {
172                                                if((target.charStats().getBodyPart(i)>0)
173                                                &&(i!=Race.BODY_TORSO))
174                                                        limbs.addElement(Integer.valueOf(i));
175                                        }
176                                        String limb=null;
177                                        if(limbs.size()==0)
178                                                limb="body part";
179                                        else
180                                                limb=(Race.BODYPARTSTR[limbs.elementAt(CMLib.dice().roll(1,limbs.size(),-1)).intValue()]).toLowerCase();
181                                        rogueLimb=CMClass.getMOB("GenMob");
182                                        rogueLimb.setName(L("@x1's @x2",target.name(),limb));
183                                        rogueLimb.setDisplayText(L("@x1 is misbehaving here.",rogueLimb.name()));
184                                        rogueLimb.basePhyStats().setAttackAdjustment(target.phyStats().attackAdjustment());
185                                        rogueLimb.basePhyStats().setArmor(target.phyStats().armor());
186                                        rogueLimb.baseCharStats().setMyRace(theRace);
187                                        int hp=100;
188                                        if(hp>(target.baseState().getHitPoints()/3))
189                                                hp=(target.baseState().getHitPoints()/3);
190                                        rogueLimb.basePhyStats().setDamage(1);
191                                        rogueLimb.baseState().setHitPoints(100);
192                                        rogueLimb.baseState().setMana(0);
193                                        rogueLimb.baseState().setMovement(100);
194                                        rogueLimb.basePhyStats().setSensesMask(PhyStats.CAN_SEE_DARK);
195                                        rogueLimb.setVictim(target);
196                                        rogueLimb.recoverMaxState();
197                                        rogueLimb.recoverCharStats();
198                                        rogueLimb.recoverPhyStats();
199                                        rogueLimb.resetToMaxState();
200                                        rogueLimb.setStartRoom(null);
201                                        rogueLimb.bringToLife(mob.location(),true);
202                                        CMLib.beanCounter().clearZeroMoney(rogueLimb,null);
203                                        rogueLimb.setMoneyVariation(0);
204                                        rogueLimb.setVictim(target);
205                                        maliciousAffect(mob,target,asLevel,0,-1);
206                                        rogueLimb.setVictim(target);
207                                }
208                        }
209                }
210                else
211                        mob.location().show(mob,target,CMMsg.MSG_OK_ACTION,L("^S<S-NAME> invoke(s) at <T-NAMESELF>, causing <T-NAME> to twitch, and nothing more."));
212
213                // return whether it worked
214                return success;
215        }
