036public class Spell_FleshStone extends Spell
037{
038
039        @Override
040        public String ID()
041        {
042                return "Spell_FleshStone";
043        }
044
045        private final static String     localizedName   = CMLib.lang().L("Flesh Stone");
046
047        @Override
048        public String name()
049        {
050                return localizedName;
051        }
052
053        private final static String     localizedStaticDisplay  = CMLib.lang().L("(Flesh to Stone)");
054
055        @Override
056        public String displayText()
057        {
058                return localizedStaticDisplay;
059        }
060
061        @Override
062        public int abstractQuality()
063        {
064                return Ability.QUALITY_MALICIOUS;
065        }
066
067        @Override
068        public int classificationCode()
069        {
070                return Ability.ACODE_SPELL | Ability.DOMAIN_TRANSMUTATION;
071        }
072
073        public Item                             statue          = null;
074        protected boolean               recurse         = false;
075        protected CharState             prevState       = null;
076
077        @Override
078        public boolean tick(final Tickable ticking, final int tickID)
079        {
080                if((tickID==Tickable.TICKID_MOB)
081                &&(statue!=null)
082                &&(affected instanceof MOB))
083                {
084                        final MOB mob=(MOB)affected;
085                        mob.makePeace(true);
086                        if((statue.owner()!=null)&&(statue.owner()!=mob.location()))
087                        {
088                                Room room=null;
089                                if(statue.owner() instanceof MOB)
090                                        room=((MOB)statue.owner()).location();
091                                else
092                                if(statue.owner() instanceof Room)
093                                        room=(Room)statue.owner();
094                                if((room!=null)&&(room!=mob.location()))
095                                        room.bringMobHere(mob,false);
096                        }
097                }
098                return super.tick(ticking,tickID);
099        }
100
101        @Override
102        public boolean okMessage(final Environmental myHost, final CMMsg msg)
103        {
104                if(affected instanceof MOB)
105                {
106                        final MOB mob=(MOB)affected;
107                        if(msg.source().getVictim()==mob)
108                                msg.source().setVictim(null);
109                        if(mob.isInCombat())
110                        {
111                                final MOB victim=mob.getVictim();
112                                if(victim!=null)
113                                        victim.makePeace(true);
114                                mob.makePeace(true);
115                        }
116                        mob.recoverMaxState();
117                        mob.resetToMaxState();
118                        mob.curState().setHunger(1000);
119                        mob.curState().setThirst(1000);
120                        mob.recoverCharStats();
121                        mob.recoverPhyStats();
122
123                        // when this spell is on a MOBs Affected list,
124                        // it should consistantly prevent the mob
125                        // from trying to do ANYTHING except sleep
126                        if(msg.amISource(mob))
127                        {
128                                if((!msg.sourceMajor(CMMsg.MASK_ALWAYS))
129                                &&(msg.sourceMajor()>0))
130                                {
131                                        mob.tell(L("Statues can't do that."));
132                                        return false;
133                                }
134                        }
135                }
136                if(!super.okMessage(myHost,msg))
137                        return false;
138
139                if(affected instanceof MOB)
140                {
141                        final MOB mob=(MOB)affected;
142                        if(msg.source().getVictim()==affected)
143                                msg.source().setVictim(null);
144                        if(mob.isInCombat())
145                        {
146                                final MOB victim=mob.getVictim();
147                                if(victim!=null)
148                                        victim.makePeace(true);
149                                mob.makePeace(true);
150                        }
151                }
152                return true;
153        }
154
155        @Override
156        public void affectPhyStats(final Physical affected, final PhyStats affectableStats)
157        {
158                super.affectPhyStats(affected,affectableStats);
159                // when this spell is on a MOBs Affected list,
160                // it should consistantly put the mob into
161                // a sleeping state, so that nothing they do
162                // can get them out of it.
163                if(affected instanceof MOB)
164                {
165                        //affectableStats.setReplacementName("a statue of "+affected.name());
166                        affectableStats.setSensesMask(affectableStats.sensesMask()|PhyStats.CAN_NOT_MOVE);
167                        affectableStats.setSensesMask(affectableStats.sensesMask()|PhyStats.CAN_NOT_HEAR);
168                        affectableStats.setSensesMask(affectableStats.sensesMask()|PhyStats.CAN_NOT_SMELL);
169                        affectableStats.setSensesMask(affectableStats.sensesMask()|PhyStats.CAN_NOT_SPEAK);
170                        affectableStats.setSensesMask(affectableStats.sensesMask()|PhyStats.CAN_NOT_TASTE);
171                        affectableStats.setDisposition(affectableStats.disposition()|PhyStats.IS_NOT_SEEN);
172                }
173        }
174
175        @Override
176        public void unInvoke()
177        {
178                // undo the affects of this spell
179                if((affected==null)||(!(affected instanceof MOB))||(recurse))
180                        return;
181                recurse=true;
182                final MOB mob=(MOB)affected;
183
184                super.unInvoke();
185                if(canBeUninvoked())
186                {
187                        if(statue!=null)
188                                statue.destroy();
189                        if((mob.location()!=null)&&(!mob.amDead()))
190                                mob.location().show(mob,null,CMMsg.MSG_OK_VISUAL,L("<S-YOUPOSS> flesh returns to normal."));
191                        if(prevState!=null)
192                                prevState.copyInto(mob.curState());
193                        CMLib.commands().postStand(mob,true, false);
194                        CMLib.utensils().confirmWearability(mob);
195                }
196                recurse=false;
197        }
198
199        @Override
200        public boolean invoke(MOB mob, List<String> commands, Physical givenTarget, boolean auto, int asLevel)
201        {
202                final MOB target=this.getTarget(mob,commands,givenTarget);
203                if(target==null)
204                        return false;
205
206                // if they can't hear the sleep spell, it
207                // won't happen
208                if((!auto)&&(!CMLib.flags().canBeHeardSpeakingBy(mob,target)))
209                {
210                        mob.tell(L("@x1 can't hear your words.",target.charStats().HeShe()));
211                        return false;
212                }
213
214                if(!super.invoke(mob,commands,givenTarget,auto,asLevel))
215                        return false;
216
217                int levelDiff=target.phyStats().level()-(mob.phyStats().level()+(2*getXLEVELLevel(mob)));
218                if(levelDiff<0)
219                        levelDiff=0;
220                boolean success=proficiencyCheck(mob,-(levelDiff*5),auto);
221
222                if(success)
223                {
224                        invoker=mob;
225                        final CMMsg msg=CMClass.getMsg(mob,target,this,verbalCastCode(mob,target,auto),auto?"":L("^S<S-NAME> incant(s) at <T-NAMESELF>.^?"));
226                        if(mob.location().okMessage(mob,msg))
227                        {
228                                mob.location().send(mob,msg);
229                                if(msg.value()<=0)
230                                {
231                                        int a=0;
232                                        while(a<target.numEffects()) // personal effects
233                                        {
234                                                final Ability A=target.fetchEffect(a);
235                                                final int s=target.numEffects();
236                                                if(A!=null)
237                                                        A.unInvoke();
238                                                if(target.numEffects()==s)
239                                                        a++;
240                                        }
241                                        CMLib.commands().postStand(target,true, false);
242                                        statue=CMClass.getItem("GenItem");
243                                        String name=target.name();
244                                        if(name.startsWith("A "))
245                                                name="a "+name.substring(2);
246                                        if(name.startsWith("An "))
247                                                name="an "+name.substring(3);
248                                        if(name.startsWith("The "))
249                                                name="the "+name.substring(4);
250                                        statue.setName(L("a statue of @x1",name));
251                                        statue.setDisplayText(L("a statue of @x1 stands here.",name));
252                                        statue.setDescription(L("It`s a hard granite statue, which looks exactly like @x1.",name));
253                                        statue.setMaterial(RawMaterial.RESOURCE_GRANITE);
254                                        statue.basePhyStats().setWeight(2000);
255                                        mob.location().show(target,null,CMMsg.MSG_OK_VISUAL,L("<S-NAME> turn(s) into stone!!"));
256                                        success=maliciousAffect(mob,target,asLevel,(mob.phyStats().level()+(2*getXLEVELLevel(mob))),-1)!=null;
257                                        target.makePeace(true);
258                                        if(mob.getVictim()==target)
259                                                mob.setVictim(null);
260                                        final Ability A=target.fetchEffect(ID());
261                                        if(success&&(A!=null))
262                                        {
263                                                mob.location().addItem(statue);
264                                                statue.addEffect(A);
265                                                A.setAffectedOne(target);
266                                                statue.recoverPhyStats();
267                                                ((Spell_FleshStone)A).prevState=(CharState)target.curState().copyOf();
268                                        }
269                                }
270                        }
271                }
272                else
273                        return maliciousFizzle(mob,target,L("<S-NAME> incant(s) at <T-NAMESELF>, but the spell fades."));
274
275                // return whether it worked
276                return success;
277        }
