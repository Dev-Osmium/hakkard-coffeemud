097        public int castingQuality(final MOB mob, final MOB target)
098        {
099                if((target!=null)&&(mob!=target))
100                        return Ability.QUALITY_INDIFFERENT;
101                if(mob.numFollowers()>0)
102                        return Ability.QUALITY_INDIFFERENT;
103                if(mob.isMonster())
104                        return Ability.QUALITY_INDIFFERENT;
105                return abstractQuality();
106        }
107
108        @Override
109        public int castingQuality(final MOB mob, final Physical target)
110        {
111                if(mob!=null)
112                {
113                        if(target instanceof MOB)
114                        {
115                                if((((MOB)target).numFollowers()>0)||(((MOB)target).isMonster()))
116                                        return Ability.QUALITY_INDIFFERENT;
117                        }
118                }
119                return super.castingQuality(mob,target);
120        }
121
122        @Override
123        public boolean invoke(final MOB mob, final List<String> commands, final Physical givenTarget, final boolean auto, final int asLevel)
124        {
125                if((mob.numFollowers()>0)||(mob.isMonster()))
126                {
127                        mob.tell(L("You cannot have any followers when casting this spell."));
128                        return false;
129                }
130
131                if(!super.invoke(mob,commands,givenTarget,auto,asLevel))
132                        return false;
133
134                final int experienceToLose=getXPCOSTAdjustment(mob,100);
135                CMLib.leveler().postExperience(mob,null,null,-experienceToLose,false);
136                mob.tell(L("The effort causes you to lose @x1 experience.",""+experienceToLose));
137
138                final boolean success=proficiencyCheck(mob,0,auto);
139
140                if(success)
141                {
142                        invoker=mob;
143                        final CMMsg msg=CMClass.getMsg(mob,null,this,verbalCastCode(mob,null,auto),auto?"":L("^S<S-NAME> call(s) for a familiar.^?"));
144                        if(mob.location().okMessage(mob,msg))
145                        {
146                                mob.location().send(mob,msg);
147                                final MOB target = determineMonster(mob, mob.phyStats().level());
148                                if(target.isInCombat())
149                                        target.makePeace(true);
150                                CMLib.commands().postFollow(target,mob,true);
151                                invoker=mob;
152                                if(target.amFollowing()!=mob)
153                                        mob.tell(L("@x1 seems unwilling to follow you.",target.name(mob)));
154                        }
155                }
156                else
157                        return beneficialWordsFizzle(mob,null,L("<S-NAME> call(s), but choke(s) on the words."));
158
159                // return whether it worked
160                return success;
161        }
