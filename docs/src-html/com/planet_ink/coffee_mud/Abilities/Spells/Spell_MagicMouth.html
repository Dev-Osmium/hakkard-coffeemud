036public class Spell_MagicMouth extends Spell
037{
038
039        @Override
040        public String ID()
041        {
042                return "Spell_MagicMouth";
043        }
044
045        private final static String localizedName = CMLib.lang().L("Magic Mouth");
046
047        @Override
048        public String name()
049        {
050                return localizedName;
051        }
052
053        @Override
054        protected int canAffectCode()
055        {
056                return CAN_ITEMS;
057        }
058
059        @Override
060        protected int canTargetCode()
061        {
062                return CAN_ITEMS;
063        }
064
065        @Override
066        public int classificationCode()
067        {
068                return Ability.ACODE_SPELL|Ability.DOMAIN_ALTERATION;
069        }
070
071        @Override
072        public int abstractQuality()
073        {
074                return Ability.QUALITY_INDIFFERENT;
075        }
076
077        Room myRoomContainer=null;
078        int myTrigger=CMMsg.TYP_ENTER;
079        String message="NO MESSAGE ENTERED";
080
081        boolean waitingForLook=false;
082
083        public void doMyThing()
084        {
085                myRoomContainer.showHappens(CMMsg.MSG_NOISE,L("\n\r\n\r@x1 says '@x2'.\n\r\n\r",affected.name(),message));
086                unInvoke();
087                return;
088        }
089
090        @Override
091        public void executeMsg(final Environmental myHost, final CMMsg msg)
092        {
093                super.executeMsg(myHost,msg);
094
095                if(affected==null)
096                {
097                        this.unInvoke();
098                        return;
099                }
100
101                if((msg.amITarget(myRoomContainer))
102                &&(!CMLib.flags().isSneaking(msg.source())))
103                {
104                        if((waitingForLook)
105                        &&(msg.targetMinor()==CMMsg.TYP_LOOK))
106                        {
107                                doMyThing();
108                                return;
109                        }
110                        else
111                        if(msg.targetMinor()==myTrigger)
112                                waitingForLook=true;
113                }
114                else
115                if(msg.amITarget(affected))
116                {
117                        if((msg.targetMinor()==myTrigger)
118                        &&(!CMLib.flags().isSneaking(msg.source())))
119                        {
120                                doMyThing();
121                                return;
122                        }
123                        switch(myTrigger)
124                        {
125                        case CMMsg.TYP_GET:
126                                if((msg.targetMinor()==CMMsg.TYP_OPEN)
127                                 ||(msg.targetMinor()==CMMsg.TYP_GIVE)
128                                 ||(msg.targetMinor()==CMMsg.TYP_DELICATE_HANDS_ACT)
129                                 ||(msg.targetMinor()==CMMsg.TYP_JUSTICE)
130                                 ||(msg.targetMinor()==CMMsg.TYP_GENERAL)
131                                 ||(msg.targetMinor()==CMMsg.TYP_LOCK)
132                                 ||(msg.targetMinor()==CMMsg.TYP_PULL)
133                                 ||(msg.targetMinor()==CMMsg.TYP_PUSH)
134                                 ||(msg.targetMinor()==CMMsg.TYP_UNLOCK))
135                                {
136                                        doMyThing();
137                                        return;
138                                }
139                        }
140                }
141
142        }
143
144        @Override
145        public boolean invoke(MOB mob, List<String> commands, Physical givenTarget, boolean auto, int asLevel)
146        {
147
148                if(commands.size()<3)
149                {
150                        mob.tell(L("You must specify:\n\r 1. What object you want the spell cast on.\n\r 2. Whether it is triggered by TOUCH, HOLD, WIELD, WEAR, or someone ENTERing the same room. \n\r 3. The message you wish the object to impart. "));
151                        return false;
152                }
153                final Physical target=mob.location().fetchFromMOBRoomFavorsItems(mob,null,(commands.get(0)),Wearable.FILTER_UNWORNONLY);
154                if((target==null)||(!CMLib.flags().canBeSeenBy(target,mob)))
155                {
156                        mob.tell(L("You don't see '@x1' here.",(commands.get(0))));
157                        return false;
158                }
159                if(target instanceof MOB)
160                {
161                        mob.tell(L("You can't can't cast this on @x1.",target.name(mob)));
162                        return false;
163                }
164
165                final String triggerStr=commands.get(1).trim().toUpperCase();
166
167                if(triggerStr.startsWith("HOLD"))
168                        myTrigger=CMMsg.TYP_HOLD;
169                else
170                if(triggerStr.startsWith("WIELD"))
171                        myTrigger=CMMsg.TYP_WIELD;
172                else
173                if(triggerStr.startsWith("WEAR"))
174                        myTrigger=CMMsg.TYP_WEAR;
175                else
176                if(triggerStr.startsWith("TOUCH"))
177                        myTrigger=CMMsg.TYP_GET;
178                else
179                if(triggerStr.startsWith("ENTER"))
180                        myTrigger=CMMsg.TYP_ENTER;
181                else
182                {
183                        mob.tell(L("You must specify the trigger event that will cause the mouth to speak.\n\r'@x1' is not correct, but you can try TOUCH, WEAR, WIELD, HOLD, or ENTER.\n\r",triggerStr));
184                        return false;
185                }
186
187                if(!super.invoke(mob,commands,givenTarget,auto,asLevel))
188                        return false;
189
190                final boolean success=proficiencyCheck(mob,0,auto);
191
192                if(success)
193                {
194                        final CMMsg msg=CMClass.getMsg(mob,target,this,verbalCastCode(mob,target,auto),L("^S<S-NAME> invoke(s) a spell upon <T-NAMESELF>.^?"));
195                        if(mob.location().okMessage(mob,msg))
196                        {
197                                mob.location().send(mob,msg);
198                                myRoomContainer=mob.location();
199                                message=CMParms.combine(commands,2);
200                                beneficialAffect(mob,target,asLevel,0);
201                        }
202
203                }
204                else
205                        beneficialWordsFizzle(mob,target,L("<S-NAME> attempt(s) to invoke a spell upon <T-NAMESELF>, but the spell fizzles."));
206
207                // return whether it worked
208                return success;
209        }
