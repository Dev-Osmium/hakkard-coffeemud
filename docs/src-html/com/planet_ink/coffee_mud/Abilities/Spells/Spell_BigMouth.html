079        @Override
080        public boolean okMessage(final Environmental myHost, final CMMsg msg)
081        {
082                if(!(affected instanceof MOB))
083                        return super.okMessage(myHost,msg);
084
085                final MOB mob=(MOB)affected;
086                if((msg.amISource(mob))
087                &&(msg.targetMinor()==CMMsg.TYP_EAT)
088                &&(msg.target() instanceof Physical)
089                &&(Stomach()!=null))
090                {
091                        final CMMsg maliciousNessMsg=CMClass.getMsg(msg.source(), msg.target(), CMMsg.MSG_OK_ACTION | CMMsg.MASK_MALICIOUS, null);
092                        final int targetWeight = (msg.target() instanceof MOB) ? ((MOB)msg.target()).baseWeight() : ((Physical)msg.target()).phyStats().weight();
093                        if((targetWeight<(mob.phyStats().weight()/3))
094                        &&(mob.location()!=null)
095                        &&(mob.location().okMessage(myHost, maliciousNessMsg)))
096                        {
097                                final int maxInhabitants=1+((mob.fetchAbility(ID())!=null)?super.getXLEVELLevel(mob):0);
098                                if((Stomach()!=null)&&(Stomach().numInhabitants()>maxInhabitants))
099                                {
100                                        mob.tell(L("Your stomach is too full."));
101                                        return false;
102                                }
103
104                                if(msg.target() instanceof MOB)
105                                {
106                                        final MOB target=(MOB)msg.target();
107                                        final boolean isHit=CMLib.combat().rollToHit(msg.source(),target);
108                                        if(!isHit)
109                                        {
110                                                mob.tell(L("You fail to eat @x1.",target.name(mob)));
111                                                if((!target.isInCombat())&&(target.isMonster())&&(target!=msg.source())
112                                                &&(target.location()==msg.source().location())&&(target.location().isInhabitant(msg.source()))
113                                                &&(CMLib.flags().canBeSeenBy(msg.source(),target)))
114                                                        CMLib.combat().postAttack(target,msg.source(),target.fetchWieldedItem());
115                                                return false;
116                                        }
117                                }
118                                else
119                                if(msg.target() instanceof Food)
120                                        return super.okMessage(myHost,msg);
121                                else
122                                if(!(msg.target() instanceof Item))
123                                        return super.okMessage(myHost,msg);
124                                else
125                                if((!CMLib.flags().isGettable((Item)msg.target()))||(msg.target().displayText().length()==0))
126                                {
127                                        mob.tell(L("You can not eat @x1.",((Item)msg.target()).name(mob)));
128                                        return false;
129                                }
130
131                                msg.modify(msg.source(),msg.target(),msg.tool(),
132                                                   msg.sourceCode()|CMMsg.MASK_ALWAYS|CMMsg.MASK_MALICIOUS,msg.sourceMessage(),
133                                                   CMMsg.MSG_NOISYMOVEMENT|CMMsg.MASK_MALICIOUS,msg.targetMessage(),
134                                                   msg.othersCode()|CMMsg.MASK_ALWAYS|CMMsg.MASK_MALICIOUS,msg.othersMessage());
135
136                        }
137                        else
138                        {
139                                mob.tell(L("@x1 is just too large for you to eat!",((Physical)msg.target()).name(mob)));
140                                return false;
141                        }
142                }
143                return super.okMessage(myHost,msg);
144        }

146        @Override
147        public void executeMsg(final Environmental myHost, final CMMsg msg)
148        {
149                if(!(affected instanceof MOB))
150                {
151                        super.executeMsg(myHost,msg);
152                        return;
153                }
154
155                final MOB mob=(MOB)affected;
156
157                if((msg.amISource(mob))
158                &&(msg.sourceMinor()==CMMsg.TYP_EAT)
159                &&(msg.target() instanceof Physical)
160                &&(Stomach()!=null))
161                {
162                        final int targetWeight = (msg.target() instanceof MOB) ? ((MOB)msg.target()).baseWeight() : ((Physical)msg.target()).phyStats().weight();
163                        if(targetWeight<(mob.phyStats().weight()/2))
164                        {
165                                if(msg.target() instanceof MOB)
166                                {
167                                        final MOB TastyMorsel=(MOB)msg.target();
168                                        final CMMsg msg2=CMClass.getMsg(mob,TastyMorsel,this,CMMsg.MSK_MALICIOUS_MOVE|CMMsg.TYP_JUSTICE,null);
169                                        if(TastyMorsel.location().okMessage(mob,msg2))
170                                        {
171                                                TastyMorsel.location().send(mob,msg2);
172                                                if(msg2.value()<=0)
173                                                {
174                                                        Stomach().bringMobHere(TastyMorsel,false);
175                                                        final CMMsg enterMsg=CMClass.getMsg(TastyMorsel,Stomach(),null,CMMsg.MSG_ENTER,L("<S-NAME> <S-IS-ARE> swallowed whole by @x1!",mob.name()),CMMsg.MSG_ENTER,null,CMMsg.MSG_ENTER,L("<S-NAME> slide(s) down the gullet into the stomach!"));
176                                                        Stomach().send(TastyMorsel,enterMsg);
177                                                }
178                                        }
179                                }
180                                if((msg.target() instanceof Item)
181                                &&(!(msg.target() instanceof Food)))
182                                        Stomach().moveItemTo((Item)msg.target(),ItemPossessor.Expire.Monster_EQ);
183                        }
184                }
185
186                if((msg.amISource(mob))
187                &&((msg.sourceMinor()==CMMsg.TYP_QUIT)||(msg.sourceMinor()==CMMsg.TYP_DEATH)))
188                        kill();
189
190                super.executeMsg(myHost,msg);
191        }

193        protected Room myStomach = null;
194        protected Room lastKnownRoom=null;
195        protected Room lastKnownLocation()
196        {
197                Room R=null;
198                if(affected instanceof MOB)
199                        R=((MOB)affected).location();
200                if(R==null)
201                        R=CMLib.map().roomLocation(invoker());
202                if(R==null)
203                        R=CMLib.map().roomLocation(affected);
204                if(R!=null)
205                        lastKnownRoom=R;
206                return lastKnownRoom;
207        }
208
209        protected Room Stomach()
210        {
211                if((myStomach==null)&&(affected!=null))
212                {
213                        myStomach = CMClass.getLocale("StdRoom");
214                        myStomach.setArea(CMLib.map().getRandomArea());
215                        myStomach.setName(L("The Stomach of @x1",affected.name()));
216                        myStomach.setDescription(L("You are in the stomach of @x1.  It is wet with digestive acids, and the walls are grinding you to a pulp.  You have been swallowed whole and are being digested.",affected.name()));
217                }
218                return myStomach;
219        }

221        protected int digestDown=4;
222
223        public void kill()
224        {
225                if((Stomach()==null)||(lastKnownLocation()==null))
226                        return;
227
228                // ===== move all inhabitants to the dragons location
229                // ===== loop through all inhabitants of the stomach
230                final int morselCount = Stomach().numInhabitants();
231                for (int x=morselCount-1;x>=0;x--)
232                {
233                        // ===== get the tasty morsels
234                        final MOB TastyMorsel = Stomach().fetchInhabitant(x);
235                        if(TastyMorsel!=null)
236                                lastKnownLocation().bringMobHere(TastyMorsel,false);
237                }
238
239                // =====move the inventory of the stomach to the room
240                final int itemCount = Stomach().numItems();
241                for (int y=itemCount-1;y>=0;y--)
242                {
243                        final Item PartiallyDigestedItem = Stomach().getItem(y);
244                        if (PartiallyDigestedItem!=null)
245                        {
246                                lastKnownLocation().addItem(PartiallyDigestedItem,ItemPossessor.Expire.Player_Drop);
247                                Stomach().delItem(PartiallyDigestedItem);
248                        }
249                }
250                if((morselCount>0)||(itemCount>0))
251                        lastKnownLocation().recoverRoomStats();
252        }

254        @Override
255        public boolean tick(final Tickable ticking, final int tickID)
256        {
257                if(!super.tick(ticking,tickID))
258                        return false;
259                if(invoker()==null)
260                        return true;
261                final MOB mob=invoker();
262                if((!mob.amDead())&&((--digestDown)<=0)&&(Stomach()!=null))
263                {
264                        digestDown=2;
265                        for (int x=0;x<Stomach().numInhabitants();x++)
266                        {
267                                // ===== get a tasty morsel
268                                final MOB TastyMorsel = Stomach().fetchInhabitant(x);
269                                if (TastyMorsel != null)
270                                {
271                                        final CMMsg DigestMsg=CMClass.getMsg(mob,
272                                                                                           TastyMorsel,
273                                                                                           null,
274                                                                                           CMMsg.MASK_ALWAYS|CMMsg.TYP_ACID,
275                                                                                           L("<S-NAME> digest(s) <T-NAMESELF>!!"));
276                                        // no OKaffectS, since the dragon is not in his own stomach.
277                                        Stomach().send(mob,DigestMsg);
278                                        int damage=(int)Math.round(CMath.div(TastyMorsel.curState().getHitPoints(),2));
279                                        if(damage<(TastyMorsel.phyStats().level()+6))
280                                                damage=TastyMorsel.curState().getHitPoints()*100;
281                                        CMLib.combat().postDamage(mob,TastyMorsel,null,damage,CMMsg.MASK_ALWAYS|CMMsg.TYP_ACID,Weapon.TYPE_MELTING,L("The stomach acid <DAMAGE> <T-NAME>!"));
282                                }
283                        }
284                }
285                else
286                if(mob.amDead())
287                        kill();
288
289                if((affected instanceof Room)
290                &&(((Room)affected).numInhabitants()==0))
291                        unInvoke();
292                return true;
293        }

295        @Override
296        public void unInvoke()
297        {
298                // undo the affects of this spell
299                if(affected==null)
300                        return;
301
302                final Environmental thang=affected;
303                super.unInvoke();
304
305                if(canBeUninvoked())
306                {
307                        if(thang instanceof MOB)
308                        {
309                                ((MOB)thang).tell(L("Your mouth shrinks to normal size."));
310                                if((Stomach()!=null)&&(Stomach().numInhabitants()>0))
311                                {
312                                        unInvoked=false;
313                                        final Spell_BigMouth A =(Spell_BigMouth)this.copyOf();
314                                        A.startTickDown(invoker,Stomach(),10000);
315                                }
316                        }
317                        else
318                        if(thang instanceof Room)
319                                kill();
320                }
321        }

323        @Override
324        public boolean invoke(MOB mob, List<String> commands, Physical givenTarget, boolean auto, int asLevel)
325        {
326                MOB target=mob;
327                if(target==null)
328                        return false;
329                if((auto)&&(givenTarget!=null)&&(givenTarget instanceof MOB))
330                        target=(MOB)givenTarget;
331                if(target.fetchEffect(this.ID())!=null)
332                {
333                        mob.tell(target,null,null,L("<S-NAME> <S-IS-ARE> already the owner of a huge mouth."));
334                        return false;
335                }
336
337                if(!super.invoke(mob,commands,givenTarget,auto,asLevel))
338                        return false;
339
340                final boolean success=proficiencyCheck(mob,0,auto);
341
342                if(success)
343                {
344                        invoker=mob;
345                        final CMMsg msg=CMClass.getMsg(mob,target,this,verbalCastCode(mob,target,auto),auto?"":L("^S<S-NAME> invoke(s) a spell.^?"));
346                        if(mob.location().okMessage(mob,msg))
347                        {
348                                mob.location().send(mob,msg);
349                                final Ability A=target.fetchEffect("Spell_ShrinkMouth");
350                                boolean isJustUnInvoking=false;
351                                if((A!=null)&&(A.canBeUninvoked()))
352                                {
353                                        A.unInvoke();
354                                        isJustUnInvoking=true;
355                                }
356                                if((!isJustUnInvoking)&&(msg.value()<=0))
357                                {
358                                        mob.location().show(target,null,CMMsg.MSG_OK_VISUAL,L("<S-NAME> feel(s) <S-HIS-HER> mouth grow to an enormous size!"));
359                                        beneficialAffect(mob,target,asLevel,4);
360                                }
361                        }
362                }
363                else
364                        return beneficialWordsFizzle(mob,target,L("<S-NAME> attempt(s) to invoke a spell, but fail(s) miserably."));
365
366                // return whether it worked
367                return success;
368        }
