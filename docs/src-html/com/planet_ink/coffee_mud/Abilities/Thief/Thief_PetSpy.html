036public class Thief_PetSpy extends ThiefSkill
037{
038        @Override
039        public String ID()
040        {
041                return "Thief_PetSpy";
042        }
043
044        private final static String     localizedName   = CMLib.lang().L("Pet Spy");
045
046        @Override
047        public String name()
048        {
049                return localizedName;
050        }
051
052        @Override
053        public int abstractQuality()
054        {
055                return Ability.QUALITY_INDIFFERENT;
056        }
057
058        @Override
059        public int classificationCode()
060        {
061                return Ability.ACODE_THIEF_SKILL | Ability.DOMAIN_ANIMALAFFINITY;
062        }
063
064        @Override
065        public String displayText()
066        {
067                return "";
068        }
069
070        @Override
071        public boolean isAutoInvoked()
072        {
073                return true;
074        }
075
076        @Override
077        public boolean canBeUninvoked()
078        {
079                return affected != invoker;
080        }
081
082        protected List<Integer> path    = new ArrayList<Integer>(1);
083
084        @Override
085        public boolean tick(final Tickable ticking, final int tickID)
086        {
087                if(!super.tick(ticking,tickID))
088                        return false;
089                
090                if((tickID==Tickable.TICKID_MOB)
091                &&(affected instanceof MOB)
092                &&(affected != invoker))
093                {
094                        final MOB M=(MOB)affected;
095                        final MOB mob=invoker();
096                        final Room R=CMLib.map().roomLocation(M);
097                        if((R==null)||(this.path==null)||(this.path.size()==0))
098                        {
099                                unInvoke();
100                                return false;
101                        }
102                        int direction = path.remove(0).intValue();
103                        final Room nextR=R.getRoomInDir(direction);
104                        if(nextR==null)
105                        {
106                                unInvoke();
107                                return false;
108                        }
109                        CMLib.tracking().walk(M, direction, false, true, false, true);
110                        if(M.location()==nextR)
111                        {
112                                final CMMsg msg2=CMClass.getMsg(mob,nextR,CMMsg.MSG_LOOK,null);
113                                nextR.executeMsg(mob,msg2);
114                        }
115                        else
116                                unInvoke();
117                }
118                return true;
119        }
120
121        @Override
122        public void unInvoke()
123        {
124                if(canBeUninvoked())
125                {
126                        Physical affected=this.affected;
127                        if(affected instanceof MOB)
128                        {
129                                if(path != null)
130                                        path.clear();
131                                final MOB invoker=this.invoker;
132                                final MOB M=(MOB)affected;
133                                if(invoker!=null)
134                                {
135                                        final Room R=CMLib.map().roomLocation(invoker);
136                                        if(R!=null)
137                                                CMLib.tracking().wanderIn(M, R);
138                                }
139                        }
140                }
141                super.unInvoke();
142        }
143
144        @Override
145        public void executeMsg(final Environmental myHost, final CMMsg msg)
146        {
147                super.executeMsg(myHost,msg);
148                if(msg.amISource(invoker)
149                &&(invoker == affected)
150                &&(msg.sourceMinor()==CMMsg.TYP_SPEAK)
151                &&(msg.sourceMessage()!=null)
152                &&(msg.target() instanceof MOB)
153                &&(((MOB)msg.target()).amFollowing()==msg.source())
154                &&(((MOB)msg.target()).fetchEffect("Prop_Familiar")!=null)
155                &&((msg.sourceMajor()&CMMsg.MASK_MAGIC)==0))
156                {
157                        final MOB mob=msg.source();
158                        final List<String> commands = CMParms.parseSpaces(CMStrings.getSayFromMessage(msg.sourceMessage()),true);
159                        if(commands.size()==0)
160                                return;
161                        if(!commands.get(0).equalsIgnoreCase("SPY"))
162                                return;
163                        commands.remove(0);
164                        final Room R=mob.location();
165                        if(R==null)
166                                return;
167                        MOB target=(MOB)msg.target();
168                        if(commands.size()<1)
169                        {
170                                R.show(target,mob,CMMsg.MSG_QUIETMOVEMENT,L("<S-NAME> look(s) at <T-NAME> expectantly."));
171                                return;
172                        }
173                        if(target.fetchEffect(ID())!=null)
174                        {
175                                mob.tell(L("@x1 is already on a spy mission!",target.name(mob)));
176                                return;
177                        }
178                        
179                        if(!super.proficiencyCheck(mob, 0, false))
180                        {
181                                R.show(target,mob,CMMsg.MSG_QUIETMOVEMENT,L("<S-NAME> refuse(s)."));
182                                return;
183                        }
184                        
185                        int dirs=0;
186                        for(String s : commands)
187                        {
188                                if(CMLib.directions().getProbableDirectionCode(s)>=0)
189                                        dirs++;
190                        }
191                        double pct=CMath.div(dirs, commands.size());
192                        final List<Integer> directions=new ArrayList<Integer>();
193                        int range=30 + (10 * super.getXLEVELLevel(mob));
194                        if(pct < .75)
195                        {
196                                final String roomName=CMParms.combine(commands,0).trim();
197                                final TrackingFlags flags=CMLib.tracking().newFlags();
198                                final List<Room> rooms=CMLib.tracking().getRadiantRooms(R, flags, range);
199                                Room R2=(Room)CMLib.english().fetchEnvironmental(rooms,roomName,true);
200                                if(R2 == null)
201                                        R2=(Room)CMLib.english().fetchEnvironmental(rooms,roomName,false);
202                                if(R2 == null)
203                                {
204                                        R.show(target,mob,CMMsg.MSG_QUIETMOVEMENT,L("<S-NAME> look(s) at <T-NAME> confusedly."));
205                                        return;
206                                }
207                                List<Room> trail=CMLib.tracking().findTrailToRoom(R, R2, flags, range, rooms);
208                                if((trail.size()==0)||(trail.get(trail.size()-1)==R))
209                                {
210                                        R.show(target,mob,CMMsg.MSG_QUIETMOVEMENT,L("<S-NAME> look(s) at <T-NAME> sadly."));
211                                        return;
212                                }
213                                Room R3=R;
214                                for(int r=trail.size()-2;r>=0;r--)
215                                {
216                                        final int dir=CMLib.map().getRoomDir(R3, trail.get(r));
217                                        if(dir < 0)
218                                        {
219                                                R.show(target,mob,CMMsg.MSG_QUIETMOVEMENT,L("<S-NAME> look(s) at <T-NAME> and frowns."));
220                                                return;
221                                        }
222                                        directions.add(Integer.valueOf(dir));
223                                }
224                        }
225                        else
226                        {
227                                for(String s : commands)
228                                {
229                                        dirs = CMLib.directions().getProbableDirectionCode(s);
230                                        if(dirs < 0)
231                                        {
232                                                R.show(target,mob,CMMsg.MSG_QUIETMOVEMENT,L("<S-NAME> look(s) at <T-NAME> confusedly."));
233                                                return;
234                                        }
235                                        directions.add(Integer.valueOf(dirs));
236                                        if(directions.size()>= ( range / 3) )
237                                                break;
238                                }
239                        }
240                        Thief_PetSpy affect = (Thief_PetSpy)beneficialAffect(mob,target,0,0);
241                        if(affect != null)
242                        {
243                                affect.path=directions;
244                        }
245                }
246        }
