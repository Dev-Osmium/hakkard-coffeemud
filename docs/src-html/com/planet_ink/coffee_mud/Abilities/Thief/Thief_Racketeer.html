092        @Override
093        public boolean okMessage(final Environmental myHost, final CMMsg msg)
094        {
095                if(!super.okMessage(myHost,msg))
096                {
097                        return false;
098                }
099                final MOB source=msg.source();
100                if((!msg.source().Name().equals(text()))
101                &&((msg.source().getClanRole(text())==null))
102                &&(msg.tool() instanceof Ability)
103                &&(msg.target()==affected)
104                &&((((Ability)msg.tool()).classificationCode()&Ability.ALL_ACODES)==Ability.ACODE_THIEF_SKILL)
105                &&(msg.sourceMinor()!=CMMsg.TYP_TEACH))
106                {
107                        if(invoker()==source)
108                        {
109                                source.tell(L("@x1 is currently under your protection.",((Physical)msg.target()).name(source)));
110                        }else
111                        {
112                                source.tell(L("@x1 is under @x2's protection.",((Physical)msg.target()).name(source),invoker().name(source)));
113                                invoker().tell(L("Word on the street is that @x1 is hassling @x2 who is under your protection.",source.name(invoker()),((Physical)msg.target()).name(invoker())));
114                        }
115                        return false;
116                }
117                return true;
118        }
119
120        @Override
121        public int castingQuality(MOB mob, Physical target)
122        {
123                if(mob!=null)
124                {
125                        if(mob.isInCombat())
126                                return Ability.QUALITY_INDIFFERENT;
127                        if(target instanceof PhysicalAgent)
128                        {
129                                final PhysicalAgent AE=(PhysicalAgent)target;
130                                if((CMLib.coffeeShops().getShopKeeper(target)==null)&&(AE.fetchBehavior("MoneyChanger")==null)
131                                &&(AE.fetchBehavior("ItemMender")==null)&&(AE.fetchBehavior("ItemIdentifier")==null)
132                                &&(AE.fetchBehavior("ItemRefitter")==null))
133                                        return Ability.QUALITY_INDIFFERENT;
134                                if(target.fetchEffect(ID())!=null)
135                                        return Ability.QUALITY_INDIFFERENT;
136                                if((target instanceof MOB)&&(!((MOB)target).mayIFight(mob)))
137                                        return Ability.QUALITY_INDIFFERENT;
138                        }
139                }
140                return super.castingQuality(mob,target);
141        }
142
143        @Override
144        public boolean invoke(MOB mob, List<String> commands, Physical givenTarget, boolean auto, int asLevel)
145        {
146                if((commands.size()<1)&&(givenTarget==null))
147                {
148                        mob.tell(L("Get protection money from whom?"));
149                        return false;
150                }
151                MOB target=null;
152                if((givenTarget!=null)&&(givenTarget instanceof MOB))
153                        target=(MOB)givenTarget;
154                else
155                        target=mob.location().fetchInhabitant(CMParms.combine(commands,0));
156                if((target==null)||(target.amDead())||(!CMLib.flags().canBeSeenBy(target,mob)))
157                {
158                        mob.tell(L("You don't see '@x1' here.",CMParms.combine(commands,1)));
159                        return false;
160                }
161                if(mob.isInCombat())
162                {
163                        mob.tell(L("You are too busy to racketeer right now."));
164                        return false;
165                }
166                if((CMLib.coffeeShops().getShopKeeper(target)==null)
167                &&(target.fetchBehavior("MoneyChanger")==null)
168                &&(target.fetchBehavior("ItemMender")==null)
169                &&(target.fetchBehavior("ItemIdentifier")==null)
170                &&(target.fetchBehavior("ItemRefitter")==null))
171                {
172                        mob.tell(L("You can't get protection money from @x1.",target.name(mob)));
173                        return false;
174                }
175                final Ability A=target.fetchEffect(ID());
176                if(A!=null)
177                {
178                        if(A.invoker()==mob)
179                                mob.tell(L("@x1 has already been extracted from today.",target.name(mob)));
180                        else
181                        {
182                                mob.tell(L("@x1 is already under @x2's protection.",target.name(mob),A.invoker().name(mob)));
183                                A.invoker().tell(L("Word on the street is that @x1 is trying to push into your business with @x2.",mob.name(A.invoker()),target.name()));
184                        }
185                        return false;
186                }
187                final int levelDiff=target.phyStats().level()-(mob.phyStats().level()+(2*getXLEVELLevel(mob)));
188                if(!target.mayIFight(mob))
189                {
190                        mob.tell(L("You cannot racketeer @x1.",target.charStats().himher()));
191                        return false;
192                }
193                if(!super.invoke(mob,commands,givenTarget,auto,asLevel))
194                        return false;
195                final double amount=CMLib.dice().roll(proficiency(),target.phyStats().level(),0);
196                final boolean success=proficiencyCheck(mob,-(levelDiff),auto);
197                final Room R=mob.location();
198                if((success)&&(R!=null))
199                {
200                        final CMMsg msg=CMClass.getMsg(mob,target,this,(auto?CMMsg.MASK_ALWAYS:0)|CMMsg.MSG_THIEF_ACT,L("<S-NAME> extract(s) @x1 of protection money from <T-NAME>.",CMLib.beanCounter().nameCurrencyShort(target,amount)));
201                        if(R.okMessage(mob,msg))
202                        {
203                                R.send(mob,msg);
204                                final TimeClock timeObj = R.getArea().getTimeObj();
205                                final int hoursInMonth =  timeObj.getHoursInDay() * timeObj.getDaysInMonth();
206                                final int tickDown = (int)(((CMProps.getMillisPerMudHour()) * hoursInMonth) / (CMProps.getTickMillis()) );
207                                beneficialAffect( mob, target, asLevel, tickDown);
208                                final Coins C=CMLib.beanCounter().makeBestCurrency(target,amount);
209                                if(C!=null)
210                                {
211                                        R.addItem(C,ItemPossessor.Expire.Player_Drop);
212                                        CMLib.commands().postGet(mob,null,C,true);
213                                }
214                        }
215                }else
216                        maliciousFizzle(mob,target,L("<T-NAME> seem(s) unintimidated by <S-NAME>."));
217                return success;
218        }
