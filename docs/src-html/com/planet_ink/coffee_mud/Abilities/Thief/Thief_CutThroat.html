056        {

058        }

097        protected String lastMOB="";
098        protected int controlCode=0;
099
100        @Override
101        public int abilityCode()
102        {
103                return controlCode;
104        }
105
106        @Override
107        public void setAbilityCode(int newCode)
108        {
109                super.setAbilityCode(newCode);
110                controlCode=newCode;
111        }
112
113        @Override
114        public void affectPhyStats(final Physical affected, final PhyStats affectableStats)
115        {
116                super.affectPhyStats(affected,affectableStats);
117                final int factor=(int)Math.round(CMath.div(adjustedLevel((MOB)affected,0),6.0))+2+abilityCode();
118                affectableStats.setDamage(affectableStats.damage()*factor);
119                affectableStats.setAttackAdjustment(affectableStats.attackAdjustment()+100+(10*getXLEVELLevel(invoker())));
120        }
121        
122        @Override
123        public void executeMsg(final Environmental myHost, final CMMsg msg)
124        {
125                super.executeMsg(myHost, msg);
126                if((msg.source()==invoker())
127                &&(msg.target() instanceof MOB)
128                &&(msg.targetMinor()==CMMsg.TYP_DAMAGE)
129                &&(msg.tool()==msg.source().fetchWieldedItem())
130                &&(((MOB)msg.target()).fetchEffect("Bleeding")==null))
131                {
132                        final Ability A2=CMClass.getAbility("Bleeding");
133                        if(A2!=null)
134                        {
135                                A2.invoke(msg.source(),(MOB)msg.target(),true,0);
136                                if((msg.trailerMsgs()==null)||(msg.trailerMsgs().size()==0))
137                                        msg.addTrailerMsg(CMClass.getMsg(msg.source(), msg.target(), msg.tool(), CMMsg.MSG_OK_ACTION, L("<S-NAME> cut(s) <T-YOUPOSS> throat with <O-NAME>! Blood start(s) running...")));
138                        }
139                }
140        }
141
142        @Override
143        public int castingQuality(MOB mob, Physical target)
144        {
145                if((mob!=null)&&(target!=null))
146                {
147                        if(!(target instanceof MOB))
148                                return Ability.QUALITY_INDIFFERENT;
149                        if(mob.isInCombat())
150                                return Ability.QUALITY_INDIFFERENT;
151                        if(CMLib.flags().canBeSeenBy(mob,(MOB)target))
152                                return Ability.QUALITY_INDIFFERENT;
153                        if(lastMOB.equals(target+""))
154                                return Ability.QUALITY_INDIFFERENT;
155                        final Item I=mob.fetchWieldedItem();
156                        if((I==null)||(!(I instanceof Weapon))||(((Weapon)I).weaponClassification()!=Weapon.CLASS_DAGGER))
157                                return Ability.QUALITY_INDIFFERENT;
158                }
159                return super.castingQuality(mob,target);
160        }
161
162        @Override
163        public boolean invoke(MOB mob, List<String> commands, Physical givenTarget, boolean auto, int asLevel)
164        {
165                if((commands.size()<1)&&(givenTarget==null))
166                {
167                        mob.tell(L("Cut whose throat?"));
168                        return false;
169                }
170                final MOB target=this.getTarget(mob,commands,givenTarget);
171                if(target==null)
172                        return false;
173
174                if(CMLib.flags().canBeSeenBy(mob,target))
175                {
176                        mob.tell(L("@x1 is watching you too closely to do that.",target.name(mob)));
177                        return false;
178                }
179                if(lastMOB.equals(target+""))
180                {
181                        mob.tell(target,null,null,L("@x1 is watching <S-HIS-HER> neck too closely to do that again.",target.name(mob)));
182                        return false;
183                }
184                if(mob.isInCombat())
185                {
186                        mob.tell(L("You are too busy to focus on cutting someone`s throat right now."));
187                        return false;
188                }
189
190                CMLib.commands().postDraw(mob,false,true);
191
192                final Item I=mob.fetchWieldedItem();
193                Weapon weapon=null;
194                if((I!=null)&&(I instanceof Weapon))
195                        weapon=(Weapon)I;
196                if(weapon==null)
197                {
198                        mob.tell(mob,target,null,L("Cut <T-HIS-HER> throat with what? You need to wield a dagger!"));
199                        return false;
200                }
201                if(weapon.weaponClassification()!=Weapon.CLASS_DAGGER)
202                {
203                        mob.tell(mob,target,weapon,L("You cannot cut a throat with <O-NAME>.  Try a dagger-type weapon."));
204                        return false;
205                }
206
207                if(!super.invoke(mob,commands,givenTarget,auto,asLevel))
208                        return false;
209
210                boolean success=proficiencyCheck(mob,0,auto);
211
212                final CMMsg msg=CMClass.getMsg(mob,target,this,(auto?CMMsg.MSG_OK_ACTION:CMMsg.MSG_THIEF_ACT),auto?"":L("<S-NAME> attempt(s) to cut <T-YOUPOSS> throat!"));
213                if(mob.location().okMessage(mob,msg))
214                {
215                        mob.location().send(mob,msg);
216                        Ability A=null;
217                        if(((!success)||(CMLib.flags().canBeSeenBy(mob,target))||(msg.value()>0))&&(!CMLib.flags().isSleeping(target)))
218                                mob.location().show(target,mob,CMMsg.MSG_OK_VISUAL,auto?"":L("<S-NAME> spot(s) <T-NAME>!"));
219                        else
220                        {
221                                setInvoker(mob);
222                                A=(Ability)this.copyOf();
223                                A.setSavable(false);
224                                mob.addEffect(A);
225                                mob.recoverPhyStats();
226                        }
227                        try
228                        {
229                                CMLib.combat().postAttack(mob,target,weapon);
230                        }
231                        finally
232                        {
233                                if(A!=null)
234                                        mob.delEffect(A);
235                                mob.recoverPhyStats();
236                        }
237                        lastMOB=""+target;
238                }
239                else
240                        success=false;
241                return success;
242        }
