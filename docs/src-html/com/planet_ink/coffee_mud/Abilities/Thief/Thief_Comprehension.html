036public class Thief_Comprehension extends ThiefSkill
037{
038        @Override
039        public String ID()
040        {
041                return "Thief_Comprehension";
042        }
043
044        private final static String localizedName = CMLib.lang().L("Linguistic Comprehension");
045
046        @Override
047        public String name()
048        {
049                return localizedName;
050        }
051
052        private final static String localizedStaticDisplay = CMLib.lang().L("(Lang. Comprehension)");
053
054        @Override
055        public String displayText()
056        {
057                return localizedStaticDisplay;
058        }
059
060        @Override
061        public int abstractQuality()
062        {
063                return Ability.QUALITY_OK_SELF;
064        }
065
066        @Override
067        protected int canAffectCode()
068        {
069                return CAN_MOBS;
070        }
071
072        private static final String[] triggerStrings =I(new String[] {"COMPREHEND","COMPREHENSION"});
073        @Override
074        public String[] triggerStrings()
075        {
076                return triggerStrings;
077        }
078
079        @Override
080        public boolean disregardsArmorCheck(MOB mob)
081        {
082                return true;
083        }
084
085        protected Vector<CMMsg> queue=new Vector<CMMsg>();
086
087        @Override
088        public int classificationCode()
089        {
090                return Ability.ACODE_THIEF_SKILL|Ability.DOMAIN_STREETSMARTS;
091        }
092
093        @Override
094        public void unInvoke()
095        {
096                // undo the affects of this spell
097                if(!(affected instanceof MOB))
098                        return;
099                final MOB mob=(MOB)affected;
100                super.unInvoke();
101
102                if(canBeUninvoked())
103                        mob.tell(L("You are no longer comprehending languages."));
104        }
105
106        @Override
107        public boolean tick(final Tickable ticking, final int tickID)
108        {
109                if(!super.tick(ticking,tickID))
110                        return false;
111                if((queue.size()>0)&&(affected instanceof MOB)&&(proficiencyCheck((MOB)affected,0,false)))
112                {
113                        final CMMsg msg=queue.firstElement();
114                        queue.removeElementAt(0);
115                        final MOB mob=(MOB)affected;
116                        if((mob.location()!=null)&&(mob.location().okMessage(mob,msg)))
117                                mob.location().send(mob,msg);
118                }
119                return true;
120        }
121
122        @Override
123        public void executeMsg(final Environmental myHost, final CMMsg msg)
124        {
125                super.executeMsg(myHost,msg);
126                if((affected instanceof MOB)
127                &&(!msg.amISource((MOB)affected))
128                &&((msg.sourceMinor()==CMMsg.TYP_SPEAK)
129                   ||(msg.sourceMinor()==CMMsg.TYP_TELL)
130                   ||(CMath.bset(msg.sourceMajor(),CMMsg.MASK_CHANNEL)))
131                &&(msg.tool() !=null)
132                &&(msg.sourceMessage()!=null)
133                &&(msg.tool() instanceof Ability)
134                &&((((Ability)msg.tool()).classificationCode()&Ability.ALL_ACODES)==Ability.ACODE_LANGUAGE)
135                &&(((MOB)affected).fetchEffect(msg.tool().ID())==null)
136                &&(!CMLib.flags().isAnimalIntelligence(msg.source()))
137                &&(msg.source().charStats().getMyRace().racialAbilities(msg.source()).find(msg.tool().ID())==null))
138                {
139                        final String str=CMStrings.getSayFromMessage(msg.sourceMessage());
140                        if(str!=null)
141                        {
142                                if(CMath.bset(msg.sourceMajor(),CMMsg.MASK_CHANNEL))
143                                        queue.addElement(CMClass.getMsg(msg.source(),null,null,CMMsg.NO_EFFECT,CMMsg.NO_EFFECT,msg.othersCode(),L("@x1 (translated from @x2)",CMStrings.substituteSayInMessage(msg.othersMessage(),str),((Ability)msg.tool()).name())));
144                                else
145                                if(msg.amITarget(affected)&&(msg.targetMessage()!=null))
146                                        queue.addElement(CMClass.getMsg(msg.source(),affected,null,CMMsg.NO_EFFECT,msg.targetCode(),CMMsg.NO_EFFECT,L("@x1 (translated from @x2)",CMStrings.substituteSayInMessage(msg.targetMessage(),str),((Ability)msg.tool()).name())));
147                                else
148                                if((msg.othersMessage()!=null)&&(msg.othersMessage().indexOf('\'')>0))
149                                {
150                                        String otherMes=msg.othersMessage();
151                                        if(msg.target()!=null)
152                                                otherMes=CMLib.coffeeFilter().fullOutFilter(((MOB)affected).session(),(MOB)affected,msg.source(),msg.target(),msg.tool(),otherMes,false);
153                                        queue.addElement(CMClass.getMsg(msg.source(),affected,null,CMMsg.NO_EFFECT,msg.othersCode(),CMMsg.NO_EFFECT,L("@x1 (translated from @x2)",CMStrings.substituteSayInMessage(otherMes,str),((Ability)msg.tool()).name())));
154                                }
155                        }
156                }
157        }
158
159        @Override
160        public boolean invoke(MOB mob, List<String> commands, Physical givenTarget, boolean auto, int asLevel)
161        {
162                MOB target=mob;
163                if(target==null)
164                        return false;
165                if((auto)&&(givenTarget!=null)&&(givenTarget instanceof MOB))
166                        target=(MOB)givenTarget;
167                if(target.fetchEffect(ID())!=null)
168                {
169                        mob.tell(L("You already have comprehension."));
170                        return false;
171                }
172
173                if(!super.invoke(mob,commands,givenTarget,auto,asLevel))
174                        return false;
175
176                final boolean success=proficiencyCheck(mob,0,auto);
177
178                if(success)
179                {
180                        invoker=mob;
181                        final CMMsg msg=CMClass.getMsg(mob,target,this,CMMsg.MSG_DELICATE_SMALL_HANDS_ACT,auto?L("<T-NAME> feel(s) more comprehending."):L("^S<S-NAME> listen(s) with intense comprehension.^?"));
182                        if(mob.location().okMessage(mob,msg))
183                        {
184                                mob.location().send(mob,msg);
185                                beneficialAffect(mob,target,asLevel,0);
186                        }
187                }
188                else
189                        return beneficialWordsFizzle(mob,target,L("<S-NAME> attempt(s) to listen with comprehension, but fail(s) miserably."));
190
191                // return whether it worked
192                return success;
193        }
194}
