036public class Thief_SlipItem extends ThiefSkill
037{
038        @Override
039        public String ID()
040        {
041                return "Thief_SlipItem";
042        }
043
044        private final static String localizedName = CMLib.lang().L("Slip Item");
045
046        @Override
047        public String name()
048        {
049                return localizedName;
050        }
051
052        @Override
053        protected int canAffectCode()
054        {
055                return 0;
056        }
057
058        @Override
059        protected int canTargetCode()
060        {
061                return CAN_MOBS;
062        }
063
064        @Override
065        public int abstractQuality()
066        {
067                return Ability.QUALITY_MALICIOUS;
068        }
069
070        @Override
071        public int classificationCode()
072        {
073                return Ability.ACODE_THIEF_SKILL|Ability.DOMAIN_STEALING;
074        }

098        public void setAbilityCode(int newCode)

100                code=newCode;

103        @Override
104        public int castingQuality(MOB mob, Physical target)
105        {
106                if(mob!=null)
107                {
108                        if(mob.isInCombat())
109                                return Ability.QUALITY_INDIFFERENT;
110                        if(target instanceof MOB)
111                        {
112                                if((((MOB)target).amDead())||(!CMLib.flags().canBeSeenBy(target,mob)))
113                                        return Ability.QUALITY_INDIFFERENT;
114                                final Item w=mob.fetchWieldedItem();
115                                if((w==null)||(w.minRange()>0)||(w.maxRange()>0))
116                                        return Ability.QUALITY_INDIFFERENT;
117                        }
118                }
119                return super.castingQuality(mob,target);
120        }
121
122        @Override
123        public boolean invoke(MOB mob, List<String> commands, Physical givenTarget, boolean auto, int asLevel)
124        {
125                String itemToSteal="all";
126                if(!auto)
127                {
128                        if(commands.size()<2)
129                        {
130                                mob.tell(L("Slip what off of whom?"));
131                                return false;
132                        }
133                        itemToSteal=commands.get(0);
134                }
135
136                MOB target=null;
137                if((givenTarget!=null)&&(givenTarget instanceof MOB))
138                        target=(MOB)givenTarget;
139                else
140                        target=mob.location().fetchInhabitant(CMParms.combine(commands,1));
141                if((target==null)||(target.amDead())||(!CMLib.flags().canBeSeenBy(target,mob)))
142                {
143                        mob.tell(L("You don't see '@x1' here.",CMParms.combine(commands,1)));
144                        return false;
145                }
146                if(mob.isInCombat())
147                {
148                        mob.tell(L("Not while you are fighting!"));
149                        return false;
150                }
151                int levelDiff=target.phyStats().level()-(mob.phyStats().level()+abilityCode()+(getXLEVELLevel(mob)*2));
152
153                if((!target.mayIFight(mob))||(levelDiff>15))
154                {
155                        mob.tell(L("You cannot slip anything off of @x1.",target.charStats().himher()));
156                        return false;
157                }
158                if(!super.invoke(mob,commands,givenTarget,auto,asLevel))
159                        return false;
160
161                final Item stolen=target.fetchItem(null,Wearable.FILTER_WORNONLY,itemToSteal);
162                if((stolen==null)||(!CMLib.flags().canBeSeenBy(stolen,mob)))
163                {
164                        mob.tell(L("@x1 doesn't seem to be wearing '@x2'.",target.name(mob),itemToSteal));
165                        return false;
166                }
167                if(stolen.amWearingAt(Wearable.WORN_WIELD))
168                {
169                        mob.tell(L("@x1 is wielding @x2! Try disarm!",target.name(mob),stolen.name()));
170                        return false;
171                }
172
173                // higher is good for the player, lower is good for the npc
174                // leveldiff is + when npc has advantage, and - when player does.
175                int discoverChance=(mob.charStats().getStat(CharStats.STAT_DEXTERITY)*3)
176                                                        -(target.charStats().getStat(CharStats.STAT_WISDOM)*5)
177                                                        -(levelDiff*5)
178                                                        +(getX1Level(mob)*5);
179                if(!CMLib.flags().canBeSeenBy(mob,target))
180                        discoverChance+=50;
181                if(discoverChance>95)
182                        discoverChance=95;
183                if(discoverChance<5)
184                        discoverChance=5;
185
186                if(levelDiff>0)
187                        levelDiff=-(levelDiff*((!CMLib.flags().canBeSeenBy(mob,target))?5:15));
188                else
189                        levelDiff=-(levelDiff*((CMLib.flags().canBeSeenBy(mob,target))?1:2));
190                final boolean success=proficiencyCheck(mob,levelDiff,auto);
191
192                if(!success)
193                {
194                        if(CMLib.dice().rollPercentage()>discoverChance)
195                        {
196                                final CMMsg msg=CMClass.getMsg(mob,target,this,CMMsg.MSG_NOISYMOVEMENT,auto?"":L("You fumble the attempt to slip @x1 off <T-NAME>; <T-NAME> spots you!",stolen.name()),CMMsg.MSG_NOISYMOVEMENT,auto?"":L("<S-NAME> tries to slip @x1 off you and fails!",stolen.name()),CMMsg.MSG_NOISYMOVEMENT,auto?"":L("<S-NAME> tries to slip @x1 off <T-NAME> and fails!",stolen.name()));
197                                if(mob.location().okMessage(mob,msg))
198                                        mob.location().send(mob,msg);
199                        }
200                        else
201                                mob.tell(auto?"":L("You fumble the attempt to slip @x1 off @x2.",stolen.name(),target.name(mob)));
202                }
203                else
204                {
205                        String str=null;
206                        if(!auto)
207                                if(!stolen.amWearingAt(Wearable.IN_INVENTORY))
208                                        str=L("<S-NAME> slip(s) @x1 off <T-NAMESELF>.",stolen.name());
209                                else
210                                        str=L("<S-NAME> attempt(s) to slip @x1 off <T-HIM-HER>, but it doesn't appear @x2 has that in <T-HIS-HER> inventory!",stolen.name(),target.charStats().heshe());
211
212                        final boolean alreadyFighting=(mob.getVictim()==target)||(target.getVictim()==mob);
213                        String hisStr=str;
214                        int hisCode=CMMsg.MSG_THIEF_ACT;
215                        if(CMLib.dice().rollPercentage()<discoverChance)
216                                hisStr=null;
217                        else
218                        {
219                                str+=" ^Z<T-NAME> spots you!^.^N";
220                                hisCode=hisCode|((target.mayIFight(mob))?CMMsg.MASK_MALICIOUS:0);
221                        }
222
223                        CMMsg msg=CMClass.getMsg(mob,target,this,CMMsg.MSG_THIEF_ACT,str,hisCode,hisStr,CMMsg.NO_EFFECT,null);
224                        if(mob.location().okMessage(mob,msg))
225                        {
226                                mob.location().send(mob,msg);
227
228                                if((!target.isMonster())&&(mob.isMonster())&&(!alreadyFighting))
229                                {
230                                        if(target.getVictim()==mob)
231                                                target.makePeace(true);
232                                        if(mob.getVictim()==target)
233                                                mob.makePeace(true);
234                                }
235                                else
236                                if(((hisStr==null)||mob.isMonster())
237                                &&(!alreadyFighting)
238                                &&(CMLib.dice().rollPercentage()>stolen.phyStats().level()))
239                                {
240                                        if(target.getVictim()==mob)
241                                                target.makePeace(true);
242                                }
243                                msg=CMClass.getMsg(target,stolen,null,CMMsg.MSG_REMOVE,CMMsg.MSG_REMOVE,CMMsg.MSG_NOISE,null);
244                                if(target.location().okMessage(target,msg))
245                                        target.location().send(mob,msg);
246                        }
247                }
248                return success;
249        }
