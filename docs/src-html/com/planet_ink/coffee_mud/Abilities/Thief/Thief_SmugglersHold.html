037{

090        protected Room fromRoom = null;
091        protected int direction = -1;
092        protected boolean messedUp = false;
093        protected boolean aborted = false;
094        protected boolean fullMode = false;
095        
096        @Override
097        public void executeMsg(final Environmental myHost, final CMMsg msg)
098        {
099                super.executeMsg(myHost, msg);
100                if(affected instanceof Exit)
101                {
102                        final Room R=msg.source().location();
103                        if((text().length()>0)&&(R!=null))
104                        {
105                                if(R.fetchInhabitant(text())==null)
106                                {
107                                        if(!fullMode)
108                                        {
109                                                fullMode=true;
110                                                affected.recoverPhyStats();
111                                        }
112                                }
113                                else
114                                {
115                                        if(fullMode)
116                                        {
117                                                fullMode=false;
118                                                affected.recoverPhyStats();
119                                        }
120                                }
121                        }
122                }
123        }
124        
125        protected void commonTell(MOB mob, String str)
126        {
127                if(mob.isMonster()&&(mob.amFollowing()!=null))
128                {
129                        if(str.startsWith("You"))
130                                str=L("I@x1",str.substring(3));
131                        CMLib.commands().postSay(mob,null,str,false,false);
132                }
133                else
134                        mob.tell(str);
135        }
136
137        protected void commonEmote(MOB mob, String str)
138        {
139                if(mob.isMonster()&&(mob.amFollowing()!=null))
140                        mob.location().show(mob,null,CMMsg.MSG_NOISYMOVEMENT|CMMsg.MASK_ALWAYS,str);
141                else
142                        mob.tell(mob,null,null,str);
143        }
144        
145        @Override
146        public void unInvoke()
147        {
148                if(canBeUninvoked())
149                {
150                        if(affected instanceof MOB)
151                        {
152                                final MOB mob=(MOB)affected;
153                                final Room room = CMLib.map().getRoom(fromRoom);
154                                final int direction=this.direction;
155                                if((messedUp)||(direction<0)||(fromRoom==null))
156                                {
157                                        commonTell(mob,L("You've ruined the smuggler's hold!"));
158                                }
159                                else
160                                {
161                                        synchronized(("SYNC"+room.roomID()).intern())
162                                        {
163                                                final Room R=CMClass.getLocale("WoodRoom");
164                                                R.setRoomID(room.getArea().getNewRoomID(room,direction));
165                                                if(R.roomID().length()==0)
166                                                {
167                                                        commonTell(mob,L("You've ruined the smuggler's hold!"));
168                                                }
169                                                else
170                                                {
171                                                        R.setArea(room.getArea());
172                                                        if(invoker==null)
173                                                                R.setDisplayText(L("The Smuggler's Hold"));
174                                                        else
175                                                                R.setDisplayText(L(invoker.name()+"'s Smuggler's Hold"));
176                                                        R.setDescription("");
177                                                        
178                                                        Exit newExit=CMClass.getExit("GenDoor");
179                                                        newExit.setName(L("a false wall"));
180                                                        newExit.setExitParams("wall","close","open","a false wall");
181                                                        Thief_SmugglersHold holder=(Thief_SmugglersHold)this.copyOf();
182                                                        holder.setMiscText(mob.Name());
183                                                        holder.fullMode=false;
184                                                        newExit.addNonUninvokableEffect(holder);
185                                                        newExit.recoverPhyStats();
186                                                        newExit.text();
187                                                        
188                                                        room.rawDoors()[direction]=R;
189                                                        room.setRawExit(direction, newExit);
190                                                        
191                                                        Exit opExit=CMClass.getExit("Open");
192                                                        R.rawDoors()[Directions.getOpDirectionCode(direction)]=room;
193                                                        R.setRawExit(Directions.getOpDirectionCode(direction), opExit);
194                                                        commonEmote(mob,L("<S-NAME> finish(es) the smuggler's hold!"));
195                                                }
196                                        }
197                                }
198                        }
199                }
200                super.unInvoke();
201        }

203        private static final int totalTicks=20;
204        
205        @Override
206        public boolean tick(final Tickable ticking, final int tickID)
207        {
208                if(!super.tick(ticking, tickID))
209                        return false;
210                final Physical affected=this.affected;
211                if(affected instanceof MOB)
212                {
213                        final MOB mob=(MOB)affected;
214                        if((mob.isInCombat())
215                        ||(CMLib.flags().isFalling(affected))
216                        ||(!CMLib.flags().isAliveAwakeMobileUnbound(mob,true)))
217                        {
218                                aborted=true;
219                                unInvoke();
220                                return false;
221                        }
222                        if(tickDown==4)
223                                mob.location().show(mob,null,CMMsg.MSG_NOISYMOVEMENT,L("<S-NAME> <S-IS-ARE> almost done building a smuggler's hold."));
224                        else
225                        if((tickDown%4)==0)
226                        {
227                                final int pct=(int)Math.round(CMath.div(totalTicks-tickDown,totalTicks)*100.0);
228                                mob.location().show(mob,null,this,CMMsg.MSG_NOISYMOVEMENT,L("<S-NAME> continue(s) building a smuggler's hold (@x1% completed).",""+pct),null,L("<S-NAME> continue(s) building a smuggler's hold."));
229                        }
230                }
231                return true;
232        }

234        @Override
235        public void affectPhyStats(Physical affectedEnv, PhyStats affectableStats)
236        {
237                if(affectedEnv instanceof MOB)
238                        affectableStats.setSensesMask(affectableStats.sensesMask()|PhyStats.CAN_NOT_TRACK);
239                else
240                if(affectedEnv instanceof Exit)
241                {
242                        if(fullMode)
243                                affectableStats.setDisposition(affectableStats.disposition()|PhyStats.IS_HIDDEN);
244                        else
245                                affectableStats.setDisposition(affectableStats.disposition()|PhyStats.IS_HIDDEN|PhyStats.IS_INVISIBLE);
246                }
247                super.affectPhyStats(affectedEnv, affectableStats);
248        }

250        @Override
251        public boolean invoke(MOB mob, List<String> commands, Physical givenTarget, boolean auto, int asLevel)
252        {
253                this.fromRoom = null;
254                this.direction = -1;
255                this.messedUp = false;
256                this.aborted = false;
257                
258                if((CMLib.flags().isSitting(mob)||CMLib.flags().isSleeping(mob)))
259                {
260                        mob.tell(L("You are on the floor!"));
261                        return false;
262                }
263
264                if(!CMLib.flags().isAliveAwakeMobileUnbound(mob,false))
265                        return false;
266                
267                final Room R=mob.location();
268                if(R==null)
269                        return false;
270                
271                final Item target;
272                if((R.getArea() instanceof BoardableShip)
273                &&(((BoardableShip)R.getArea()).getShipItem() instanceof BoardableShip))
274                {
275                        target=((BoardableShip)R.getArea()).getShipItem();
276                }
277                else
278                {
279                        mob.tell(L("You must be on a ship to build a smuggler's hold!"));
280                        return false;
281                }
282                
283                for(final Enumeration<Room> r=R.getArea().getProperMap();r.hasMoreElements();)
284                {
285                        final Room R2=r.nextElement();
286                        if((R2!=null)
287                        &&(R2.fetchEffect(ID())!=null))
288                        {
289                                mob.tell(L("Your ship already has a smuggler's hold!"));
290                                return false;
291                        }
292                }
293
294                if(commands.size()==0)
295                {
296                        mob.tell(L("You must specify which direction to build your hold in."));
297                        return false;
298                }
299                
300                String dirStr=CMParms.combine(commands,0);
301                final int direction=CMLib.directions().getGoodShipDirectionCode(dirStr);
302                if(direction<0)
303                {
304                        mob.tell(L("@x1 is not a valid direction to build your hold in.",dirStr));
305                        return false;
306                }
307                dirStr=CMLib.directions().getShipInDirectionName(direction);
308                
309                final Room R2=R.getRoomInDir(direction);
310                final Exit E2=R.getExitInDir(direction);
311                if((R2!=null)||(E2!=null))
312                {
313                        mob.tell(L("You can not build in that direction."));
314                        return false;
315                }
316                
317                if(!super.invoke(mob,commands,givenTarget,auto,asLevel))
318                        return false;
319
320                final boolean success=proficiencyCheck(mob,0,auto);
321                final CMMsg msg=CMClass.getMsg(mob,target,this,CMMsg.MSG_THIEF_ACT,auto?"":L("<S-NAME> start(s) building a smuggler's hold @x1!",dirStr));
322                if(mob.location().okMessage(mob,msg))
323                {
324                        mob.location().send(mob,msg);
325                        Thief_SmugglersHold A=(Thief_SmugglersHold)beneficialAffect(mob, mob, asLevel, totalTicks);
326                        if(A!=null)
327                        {
328                                A.fromRoom = mob.location();
329                                A.direction = direction;
330                                A.messedUp = !success;
331                                A.aborted = false;
332                        }
333                }
334                return success;
335        }
