103        @Override
104        public boolean tick(final Tickable ticking, final int tickID)
105        {
106                if((pathOut!=null)&&(tickID==Tickable.TICKID_MOB))
107                {
108                        if((pathOut.size()==0)||(theGroup==null)||(currRoom==null))
109                                unInvoke();
110                        else
111                        {
112                                currRoom.showHappens(CMMsg.MSG_OK_ACTION,L("@x1 goes by.",theNoun));
113                                currRoom=currRoom.getRoomInDir(pathOut.get(0).intValue());
114                                pathOut.remove(0);
115                                if(currRoom!=null)
116                                {
117                                        final String roomDesc=currRoom.displayText(null);
118                                        if((lastDesc==null)||(!roomDesc.equalsIgnoreCase(lastDesc)))
119                                        {
120                                                lastDesc=roomDesc;
121                                                for(int g=0;g<theGroup.size();g++)
122                                                {
123                                                        final MOB M=theGroup.get(g);
124                                                        if((M.playerStats()!=null)&&(!CMSecurity.isDisabled(CMSecurity.DisFlag.HYGIENE)))
125                                                                M.playerStats().adjHygiene(hygieneLoss);
126                                                        switch(CMLib.dice().roll(1,10,0))
127                                                        {
128                                                        case 1:
129                                                                M.tell(L("You think you are being taken through '@x1'.", roomDesc));
130                                                                break;
131                                                        case 2:
132                                                                M.tell(L("You might be going through '@x1' now.", roomDesc));
133                                                                break;
134                                                        case 3:
135                                                                M.tell(L("Now you are definitely going through '@x1'.", roomDesc));
136                                                                break;
137                                                        case 4:
138                                                                M.tell(L("You are being taken through '@x1', you think.", roomDesc));
139                                                                break;
140                                                        case 5:
141                                                                M.tell(L("Sounds like '@x1' now.", roomDesc));
142                                                                break;
143                                                        case 6:
144                                                                M.tell(L("Now this might be '@x1'.", roomDesc));
145                                                                break;
146                                                        case 7:
147                                                                M.tell(L("You're going through '@x1'.", roomDesc));
148                                                                break;
149                                                        case 8:
150                                                                M.tell(L("Sounds like this could be '@x1'.", roomDesc));
151                                                                break;
152                                                        case 9:
153                                                                M.tell(L("You are probably going through '@x1' now.", roomDesc));
154                                                                break;
155                                                        case 10:
156                                                                M.tell(L("Sounds like you are being taken through '@x1'.", roomDesc));
157                                                                break;
158                                                        }
159                                                }
160                                        }
161                                }
162                        }
163                }
164                return super.tick(ticking,tickID);
165        }

167        @Override
168        public boolean okMessage(final Environmental host, final CMMsg msg)
169        {
170                if(((msg.sourceMinor()==CMMsg.TYP_QUIT)
171                        ||(msg.sourceMinor()==CMMsg.TYP_SHUTDOWN)
172                        ||((msg.targetMinor()==CMMsg.TYP_EXPIRE)&&(storage.contains(msg.target())))
173                        ||(msg.sourceMinor()==CMMsg.TYP_ROOMRESET)))
174                {
175                        unInvoke();
176                }
177                return super.okMessage(host,msg);
178        }

180        @Override
181        public void unInvoke()
182        {
183                if(pathOut!=null)
184                {
185                        if(currRoom==null)
186                                currRoom=CMLib.map().getRandomRoom();
187                        if(theGroup!=null)
188                        for(int g=0;g<theGroup.size();g++)
189                        {
190                                final MOB M=theGroup.get(g);
191                                M.tell(L("You are told that it's safe and released."));
192                                currRoom.bringMobHere(M,false);
193                                CMLib.commands().postStand(M,true, false);
194                                CMLib.commands().postLook(M,true);
195                        }
196                        if(storage!=null)
197                        for(int s=0;s<storage.size();s++)
198                                storage.get(s).destroy();
199                        pathOut=null;
200                        currRoom=null;
201                        if(storage!=null)
202                                storage.clear();
203                        storage=null;
204                        if(theGroup!=null)
205                                theGroup.clear();
206                        theGroup=null;
207                }
208                super.unInvoke();
209        }

211        public void bringMOBSHere(Room newRoom, Vector<MOB> group, String enterStr, String leaveStr)
212        {
213                for(int g=group.size()-1;g>=0;g--)
214                {
215                        final MOB follower=group.elementAt(g);
216                        if(!bringMOBHere(newRoom,follower,enterStr,leaveStr))
217                                group.removeElementAt(g);
218                }
219        }
220
221        public void bringMOBSLikeHere(List<Room> rooms, Room newRoom, Vector<MOB> group, String enterStr, String leaveStr)
222        {
223                for(int g=group.size()-1;g>=0;g--)
224                {
225                        final MOB follower=group.elementAt(g);
226                        final Room R=(Room)newRoom.copyOf();
227                        if(!bringMOBHere(R,follower,enterStr,leaveStr))
228                                group.removeElementAt(g);
229                        else
230                                rooms.add(R);
231                }
232        }

234        public boolean bringMOBHere(Room newRoom, MOB follower, String leaveStr, String enterStr)
235        {
236                final Room thisRoom=follower.location();
237                final CMMsg enterMsg=CMClass.getMsg(follower,newRoom,this,CMMsg.MSG_ENTER,enterStr,CMMsg.MSG_ENTER,null,CMMsg.MSG_ENTER,L("You are joined by <S-NAME>."));
238                final CMMsg leaveMsg=CMClass.getMsg(follower,thisRoom,this,CMMsg.MSG_LEAVE,leaveStr,CMMsg.MSG_LEAVE,null,CMMsg.MSG_LEAVE,leaveStr);
239                if(thisRoom.okMessage(follower,leaveMsg)&&newRoom.okMessage(follower,enterMsg))
240                {
241                        if(follower.isInCombat())
242                        {
243                                CMLib.commands().postFlee(follower,("NOWHERE"));
244                                follower.makePeace(true);
245                        }
246                        thisRoom.send(follower,leaveMsg);
247                        newRoom.bringMobHere(follower,false);
248                        thisRoom.delInhabitant(follower);
249                        newRoom.send(follower,enterMsg);
250                        follower.basePhyStats().setDisposition(follower.basePhyStats().disposition()|PhyStats.IS_SITTING);
251                        follower.phyStats().setDisposition(follower.phyStats().disposition()|PhyStats.IS_SITTING);
252                        //follower.tell(L("\n\r\n\r"));
253                        //CMLib.commands().postLook(follower,true);
254                        return true;
255                }
256                return false;
257        }

259        public Room makeNewRoom(Area area, String display, String description)
260        {
261                final Room R=CMClass.getLocale("MagicShelter");
262                R.setDisplayText(display);
263                R.setDescription(description);
264                R.setRoomID("");
265                R.setArea(area);
266                final Ability A=CMClass.getAbility("Thief_Bind");
267                if(A!=null)
268                {
269                        final Item I=CMClass.getBasicItem("StdItem");
270                        I.setName(display);
271                        A.setAffectedOne(I);
272                        R.addNonUninvokableEffect(A);
273                }
274                return R;
275        }

277        @Override
278        public boolean invoke(MOB mob, List<String> commands, Physical givenTarget, boolean auto, int asLevel)
279        {
280                MOB target=mob;
281                if((auto)&&(givenTarget!=null)&&(givenTarget instanceof MOB))
282                        target=(MOB)givenTarget;
283                if(target.isInCombat())
284                {
285                        mob.tell(target,null,null,L("Not while <S-NAME> <S-IS-ARE> fighting."));
286                        return false;
287                }
288                final Room thisRoom=target.location();
289                if(thisRoom==null)
290                        return false;
291
292                if((!auto)&&(thisRoom.domainType()!=Room.DOMAIN_OUTDOORS_CITY))
293                {
294                        mob.tell(L("You must be out on a street to contact your underground connections."));
295                        return false;
296                }
297                final Area A=CMLib.map().areaLocation(target);
298                if((!CMLib.law().isACity(A))
299                &&(!auto))
300                {
301                        mob.tell(L("You can only use this skill in cities."));
302                        return false;
303                }
304
305                final TrackingLibrary.TrackingFlags flags=CMLib.tracking().newFlags();
306                flags.plus(TrackingLibrary.TrackingFlag.NOEMPTYGRIDS)
307                         .plus(TrackingLibrary.TrackingFlag.NOAIR)
308                         .plus(TrackingLibrary.TrackingFlag.NOWATER);
309                final List<Room> trail=CMLib.tracking().getRadiantRooms(thisRoom,flags,30+(2*getXLEVELLevel(mob)));
310                final Vector<Room> finalTos=new Vector<Room>();
311                Room R=null;
312                for(int c=0;c<trail.size();c++)
313                {
314                        R=trail.get(c);
315                        if((R.getArea()!=A)
316                        &&(!CMath.bset(R.domainType(),Room.INDOORS))
317                        &&(CMLib.flags().canAccess(target,R)))
318                        {
319                                Room checkR=null;
320                                for(int d=Directions.NUM_DIRECTIONS()-1;d>=0;d--)
321                                {
322                                        checkR=R.getRoomInDir(d);
323                                        if((checkR!=null)
324                                        &&(checkR.getArea()==A)
325                                        &&(!CMath.bset(checkR.domainType(),Room.INDOORS))
326                                        &&(trail.indexOf(checkR)<c))
327                                        {
328                                                finalTos.addElement(R);
329                                                break;
330                                        }
331                                }
332                        }
333                }
334                final List<List<Integer>> allTrails=CMLib.tracking().findAllTrails(thisRoom,finalTos,trail);
335                for(int a=allTrails.size()-1;a>=0;a--)
336                {
337                        final List<Integer> thisTrail=allTrails.get(a);
338                        R=thisRoom;
339                        for(int t=0;t<thisTrail.size();t++)
340                        {
341                                R=R.getRoomInDir(thisTrail.get(t).intValue());
342                                if((R==null)||(CMath.bset(R.domainType(),Room.INDOORS)))
343                                {
344                                        allTrails.remove(a);
345                                        break;
346                                }
347                        }
348                }
349                if(allTrails.size()==0)
350                {
351                        mob.tell(L("Your informants tell you that there's no way they can get you out of here."));
352                        return false;
353                }
354                final List<Integer> theTrail=allTrails.get(CMLib.dice().roll(1,allTrails.size(),-1));
355
356                if(!super.invoke(mob,commands,givenTarget,auto,asLevel))
357                        return false;
358
359                final boolean success=proficiencyCheck(mob,0,auto);
360
361                final CMMsg msg=CMClass.getMsg(mob,null,this,auto?CMMsg.MASK_ALWAYS:CMMsg.MSG_DELICATE_HANDS_ACT,CMMsg.MSG_DELICATE_SMALL_HANDS_ACT,CMMsg.MSG_DELICATE_SMALL_HANDS_ACT,auto?"":L("<S-NAME> contact(s) <S-HIS-HER> underground connections here."));
362                if(!success)
363                        return beneficialVisualFizzle(mob,null,auto?"":L("<S-NAME> can't seem to contact <S-HIS-HER> underground connections here."));
364                else
365                if(mob.location().okMessage(mob,msg))
366                {
367                        mob.location().send(mob,msg);
368                        beneficialAffect(mob,target,asLevel,0);
369                        final Thief_UndergroundConnections underA=(Thief_UndergroundConnections)target.fetchEffect(ID());
370                        if(underA!=null)
371                        {
372                                underA.currRoom=thisRoom;
373                                final Set<MOB> H=target.getGroupMembers(new HashSet<MOB>());
374                                final Vector<MOB> group=new Vector<MOB>();
375                                group.addElement(target);
376                                for (final Object element : H)
377                                {
378                                        final MOB M=(MOB)element;
379                                        if((M!=null)
380                                        &&(M.location()==thisRoom)
381                                        &&(CMLib.flags().isInTheGame(M,true))
382                                        &&(!group.contains(M)))
383                                                group.addElement(M);
384                                }
385                                underA.theGroup=group;
386                                final Area area=thisRoom.getArea();
387                                final List<Room> rooms=new Vector<Room>();
388                                switch(CMLib.dice().roll(1,4,0))
389                                {
390                                case 1:
391                                {
392                                        underA.theNoun="a strange horse drawn cart";
393                                        thisRoom.showHappens(CMMsg.MSG_OK_ACTION,L("A horse drawn wagon filled with hay pulls up.\n\rTwo large farmers jump out."));
394                                        final Room destR=makeNewRoom(area,"A bundle of straw","Your eyes, mouth, and ears are filled with the stuff. This trip better be short!");
395                                        rooms.add(destR);
396                                        bringMOBSHere(destR,group,"","<S-NAME> <S-IS-ARE> picked up and stuffed into the piles of hay");
397                                        break;
398                                }
399                                case 2:
400                                {
401                                        underA.theNoun="a large group of teamsters carrying potato sacks";
402                                        thisRoom.showHappens(CMMsg.MSG_OK_ACTION,L("A group of large teamsters with big empty potato sacks approaches."));
403                                        final Room destR=makeNewRoom(area,"A potato sack","Once you get past the musty rotten smell, this is really uncomfortable!");
404                                        bringMOBSLikeHere(rooms,destR,group,"","<S-NAME> <S-IS-ARE> picked up and stuffed into a sack");
405                                        break;
406                                }
407                                case 3:
408                                {
409                                        underA.theNoun="a pullcart full of plague victims";
410                                        thisRoom.showHappens(CMMsg.MSG_OK_ACTION,L("A pullcart full of the decaying bodies of plague victims pulls up."));
411                                        final Room destR=makeNewRoom(area,"A pile of bodies","There is a finger in your ear, and you aren't sure whose it is.");
412                                        rooms.add(destR);
413                                        bringMOBSHere(destR,group,"","<S-NAME> <S-IS-ARE> picked up and stuffed into the bottom of the pile");
414                                        break;
415                                }
416                                case 4:
417                                {
418                                        underA.theNoun="a horse drawn cart full of empty vinegar barrels";
419                                        thisRoom.showHappens(CMMsg.MSG_OK_ACTION,L("A horse drawn cart full of empty vinegar barrels pulls up."));
420                                        final Room destR=makeNewRoom(area,"A vinegar barrel","You feel thoroughly pickled!");
421                                        bringMOBSLikeHere(rooms,destR,group,"","<S-NAME> <S-IS-ARE> picked up and stuffed into an empty barrel.");
422                                        break;
423                                }
424                                }
425                                underA.hygieneLoss=(int)PlayerStats.HYGIENE_DELIMIT/theTrail.size();
426                                underA.storage=rooms;
427                                underA.pathOut=theTrail;
428                        }
429                }
430                return success;
431        }
