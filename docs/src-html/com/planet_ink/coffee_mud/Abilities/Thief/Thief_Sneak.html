036public class Thief_Sneak extends ThiefSkill
037{
038        @Override
039        public String ID()
040        {
041                return "Thief_Sneak";
042        }
043
044        private final static String localizedName = CMLib.lang().L("Sneak");
045
046        @Override
047        public String name()
048        {
049                return localizedName;
050        }
051
052        @Override
053        protected int canAffectCode()
054        {
055                return 0;
056        }
057
058        @Override
059        protected int canTargetCode()
060        {
061                return 0;
062        }
063
064        @Override
065        public int abstractQuality()
066        {
067                return Ability.QUALITY_INDIFFERENT;
068        }
069
070        @Override
071        public int classificationCode()
072        {
073                return Ability.ACODE_THIEF_SKILL | Ability.DOMAIN_STEALTHY;
074        }
075
076        private static final String[] triggerStrings = I(new String[] { "SNEAK" });
077
078        @Override
079        public String[] triggerStrings()
080        {
081                return triggerStrings;
082        }
083
084        @Override
085        public int usageType()
086        {
087                return USAGE_MOVEMENT;
088        }
089
090        @Override
091        public boolean invoke(MOB mob, List<String> commands, Physical givenTarget, boolean auto, int asLevel)
092        {
093                String dir=CMParms.combine(commands,0);
094                if(commands.size()>0)
095                        dir=commands.get(commands.size()-1);
096                Physical target=givenTarget;
097                int dirCode=CMLib.directions().getGoodDirectionCode(dir);
098                if(dirCode<0)
099                {
100                        if(target == null)
101                                target=mob.location().fetchFromRoomFavorExits(CMParms.combine(commands,0));
102                        if(target instanceof Exit)
103                                dirCode = CMLib.map().getExitDir(mob.location(), (Exit)target);
104                        if((dirCode<0)&&(target != null))
105                        {
106                                if(target instanceof Rideable)
107                                {
108                                        if(target instanceof Exit) // it's a portal .. so we just assume you can climb "in" it
109                                        {
110                                                
111                                        }
112                                        else
113                                        if(((Rideable)target).rideBasis()!=Rideable.RIDEABLE_LADDER)
114                                        {
115                                                mob.tell(L("You can not sneak into '@x1'.",target.name(mob)));
116                                                return false;
117                                        }
118                                        else // ordinary ladder item, just convert to an UP
119                                        {
120                                                target=null;
121                                                dirCode=Directions.UP;
122                                        }
123                                }
124                                else
125                                {
126                                        mob.tell(L("You can not sneak into '@x1'.",target.name(mob)));
127                                        return false;
128                                }
129                        }
130                        else
131                        {
132                                target = null; // it's an ordinary exit
133                        }
134                }
135                
136                if((dirCode<0)&&(!(target instanceof Rideable)))
137                {
138                        mob.tell(L("Sneak where?"));
139                        return false;
140                }
141
142                if((dirCode>=0)
143                &&((mob.location().getRoomInDir(dirCode)==null)
144                        ||(mob.location().getExitInDir(dirCode)==null)))
145                {
146                        mob.tell(L("Sneak where?"));
147                        return false;
148                }
149
150                final MOB highestMOB=getHighestLevelMOB(mob,null);
151                int levelDiff=(mob.phyStats().level()+(super.getXLEVELLevel(mob)*2))-getMOBLevel(highestMOB);
152
153                if(!super.invoke(mob,commands,givenTarget,auto,asLevel))
154                        return false;
155
156                boolean success=false;
157                String msgStr;
158                if(dirCode<0)
159                        msgStr=L("You quietly sneak into <T-NAME>.");
160                else
161                        msgStr=L("You quietly sneak @x1.",CMLib.directions().getDirectionName(dirCode));
162                final CMMsg msg=CMClass.getMsg(mob,target,this,auto?CMMsg.MSG_OK_VISUAL:CMMsg.MSG_DELICATE_HANDS_ACT,msgStr,CMMsg.NO_EFFECT,null,CMMsg.NO_EFFECT,null);
163                if(mob.location().okMessage(mob,msg))
164                {
165                        mob.location().send(mob,msg);
166                        if(levelDiff<0)
167                                levelDiff=levelDiff*8;
168                        else
169                                levelDiff=levelDiff*10;
170                        success=proficiencyCheck(mob,levelDiff,auto);
171                        if(success)
172                        {
173                                mob.basePhyStats().setDisposition(mob.basePhyStats().disposition()|PhyStats.IS_SNEAKING);
174                                mob.recoverPhyStats();
175                        }
176                        if(dirCode>=0)
177                                CMLib.tracking().walk(mob,dirCode,false,false);
178                        else
179                        if(target instanceof Rideable)
180                                CMLib.commands().forceStandardCommand(mob, "Enter", new XVector<String>("ENTER",mob.location().getContextName(target)));
181                        if(success)
182                        {
183
184                                final int disposition=mob.basePhyStats().disposition();
185                                if((disposition&PhyStats.IS_SNEAKING)>0)
186                                {
187                                        mob.basePhyStats().setDisposition(disposition-PhyStats.IS_SNEAKING);
188                                        mob.recoverPhyStats();
189                                }
190                                Ability toHide=mob.fetchAbility("Thief_Hide");
191                                if(toHide==null)
192                                        toHide=mob.fetchAbility("Ranger_Hide");
193                                if(toHide!=null)
194                                        toHide.invoke(mob,new Vector<String>(),null,false,asLevel);
195                        }
196                        if(CMLib.flags().isSneaking(mob))
197                                mob.phyStats().setDisposition(mob.phyStats().disposition()-PhyStats.IS_SNEAKING);
198                }
199                return success;
200        }
