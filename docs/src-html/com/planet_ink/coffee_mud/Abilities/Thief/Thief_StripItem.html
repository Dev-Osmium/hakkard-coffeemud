036public class Thief_StripItem extends ThiefSkill
037{
038        @Override
039        public String ID()
040        {
041                return "Thief_StripItem";
042        }
043
044        private final static String localizedName = CMLib.lang().L("Strip Item");
045
046        @Override
047        public String name()
048        {
049                return localizedName;
050        }
051
052        @Override
053        protected int canAffectCode()
054        {
055                return 0;
056        }
057
058        @Override
059        protected int canTargetCode()
060        {
061                return CAN_MOBS;
062        }
063
064        @Override
065        public int abstractQuality()
066        {
067                return Ability.QUALITY_MALICIOUS;
068        }
069
070        @Override
071        public int classificationCode()
072        {
073                return Ability.ACODE_THIEF_SKILL|Ability.DOMAIN_STEALING;
074        }
075
076        private static final String[] triggerStrings =I(new String[] {"STRIPITEM"});
077        @Override
078        public String[] triggerStrings()
079        {
080                return triggerStrings;
081        }
082
083        @Override
084        public int usageType()
085        {
086                return USAGE_MOVEMENT;
087        }
088
089        public int code=0;
090
091        @Override
092        public int abilityCode()
093        {
094                return code;
095        }
096
097        @Override
098        public void setAbilityCode(int newCode)
099        {
100                code=newCode;
101        }
102
103        @Override
104        public int castingQuality(MOB mob, Physical target)
105        {
106                if(mob!=null)
107                {
108                        if(!(target instanceof MOB))
109                                return Ability.QUALITY_INDIFFERENT;
110                        if(((MOB)target).amDead()||(!CMLib.flags().canBeSeenBy(target,mob)))
111                                return Ability.QUALITY_INDIFFERENT;
112                        if(!((MOB)target).mayIFight(mob))
113                                return Ability.QUALITY_INDIFFERENT;
114                }
115                return super.castingQuality(mob,target);
116        }
117
118        @Override
119        public boolean invoke(MOB mob, List<String> commands, Physical givenTarget, boolean auto, int asLevel)
120        {
121                String itemToSteal="all";
122                if(!auto)
123                {
124                        if(commands.size()<2)
125                        {
126                                mob.tell(L("Strip what off of whom?"));
127                                return false;
128                        }
129                        itemToSteal=commands.get(0);
130                }
131
132                MOB target=null;
133                if((givenTarget!=null)&&(givenTarget instanceof MOB))
134                        target=(MOB)givenTarget;
135                else
136                        target=mob.location().fetchInhabitant(CMParms.combine(commands,1));
137                if((target==null)||(target.amDead())||(!CMLib.flags().canBeSeenBy(target,mob)))
138                {
139                        mob.tell(L("You don't see '@x1' here.",CMParms.combine(commands,1)));
140                        return false;
141                }
142                int levelDiff=target.phyStats().level()-(mob.phyStats().level()+abilityCode()+(getXLEVELLevel(mob)*2));
143                if((!target.mayIFight(mob))||(levelDiff>15))
144                {
145                        mob.tell(L("You cannot strip anything off of @x1.",target.charStats().himher()));
146                        return false;
147                }
148                if(!super.invoke(mob,commands,givenTarget,auto,asLevel))
149                        return false;
150
151                final Item stolen=target.fetchItem(null,Wearable.FILTER_WORNONLY,itemToSteal);
152                if((stolen==null)||(!CMLib.flags().canBeSeenBy(stolen,mob)))
153                {
154                        mob.tell(L("@x1 doesn't seem to be wearing '@x2'.",target.name(mob),itemToSteal));
155                        return false;
156                }
157                if(stolen.amWearingAt(Wearable.WORN_WIELD))
158                {
159                        mob.tell(L("@x1 is wielding @x2! Try disarm!",target.name(mob),stolen.name()));
160                        return false;
161                }
162
163                if(levelDiff>0)
164                        levelDiff=-(levelDiff*((!CMLib.flags().canBeSeenBy(mob,target))?5:15));
165                else
166                        levelDiff=-(levelDiff*((CMLib.flags().canBeSeenBy(mob,target))?1:2));
167                final boolean success=proficiencyCheck(mob,levelDiff,auto);
168
169                if(!success)
170                {
171                        if((target.isMonster())&&(mob.getVictim()==null))
172                                mob.setVictim(target);
173                        final CMMsg msg=CMClass.getMsg(mob,target,this,CMMsg.MSG_NOISYMOVEMENT,auto?"":L("You fumble the attempt to strip @x1 off <T-NAME>; <T-NAME> spots you!",stolen.name()),CMMsg.MSG_NOISYMOVEMENT,auto?"":L("<S-NAME> tries to strip @x1 off you and fails!",stolen.name()),CMMsg.MSG_NOISYMOVEMENT,auto?"":L("<S-NAME> tries to strip @x1 off <T-NAME> and fails!",stolen.name()));
174                        if(mob.location().okMessage(mob,msg))
175                                mob.location().send(mob,msg);
176                }
177                else
178                {
179                        String str=null;
180                        if(!auto)
181                                str=L("<S-NAME> strip(s) @x1 off <T-NAMESELF>.",stolen.name());
182
183                        final boolean alreadyFighting=(mob.getVictim()==target)||(target.getVictim()==mob);
184                        final String hisStr=str;
185                        final int hisCode=CMMsg.MSG_THIEF_ACT | ((target.mayIFight(mob))?CMMsg.MASK_MALICIOUS:0);
186
187                        CMMsg msg=CMClass.getMsg(mob,target,this,CMMsg.MSG_THIEF_ACT,str,hisCode,hisStr,CMMsg.NO_EFFECT,null);
188                        if(mob.location().okMessage(mob,msg))
189                        {
190                                mob.location().send(mob,msg);
191
192                                if((!target.isMonster())&&(mob.isMonster())&&(!alreadyFighting))
193                                {
194                                        if(target.getVictim()==mob)
195                                                target.makePeace(true);
196                                        if(mob.getVictim()==target)
197                                                mob.makePeace(true);
198                                }
199                                else
200                                if(((hisStr==null)||mob.isMonster())
201                                &&(!alreadyFighting)
202                                &&(CMLib.dice().rollPercentage()>stolen.phyStats().level()))
203                                {
204                                        if(target.getVictim()==mob)
205                                                target.makePeace(true);
206                                }
207                                msg=CMClass.getMsg(target,stolen,null,CMMsg.MSG_REMOVE,CMMsg.MSG_REMOVE,CMMsg.MSG_NOISE,null);
208                                if(target.location().okMessage(target,msg))
209                                        target.location().send(mob,msg);
210                        }
211                }
212                return success;
213        }
