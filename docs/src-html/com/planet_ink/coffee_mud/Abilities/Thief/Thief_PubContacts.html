096        public Triad<Item,Double,String> cheapestAlcoholHere(MOB mob, Room room)
097        {
098                double lowestPrice=Integer.MAX_VALUE;
099                Item lowestItem=null;
100                String currency="";
101                for(int m=0;m<room.numInhabitants();m++)
102                {
103                        final MOB M=room.fetchInhabitant(m);
104                        if((M!=null)&&(M!=mob))
105                        {
106                                if(CMLib.flags().canBeSeenBy(M,mob))
107                                {
108                                        final ShopKeeper SK=CMLib.coffeeShops().getShopKeeper(M);
109                                        if(SK!=null)
110                                        {
111                                                for(final Iterator<Environmental> i=SK.getShop().getStoreInventory();i.hasNext();)
112                                                {
113                                                        final Environmental E=i.next();
114                                                        if((E instanceof Item)&&(CMLib.flags().isAlcoholic((Item)E)))
115                                                        {
116                                                                double moneyPrice=0;
117                                                                ShopKeeper.ShopPrice price=CMLib.coffeeShops().sellingPrice(M,mob,E,SK,SK.getShop(), true);
118                                                                if(price.experiencePrice>0)
119                                                                        moneyPrice=(100 * price.experiencePrice);
120                                                                else
121                                                                if(price.questPointPrice>0)
122                                                                        moneyPrice=(100 * price.questPointPrice);
123                                                                else
124                                                                {
125                                                                        moneyPrice=price.absoluteGoldPrice;
126                                                                }
127                                                                if(moneyPrice < lowestPrice)
128                                                                {
129                                                                        lowestPrice=moneyPrice;
130                                                                        lowestItem=(Item)E;
131                                                                        currency=CMLib.beanCounter().getCurrency(M);
132                                                                }
133                                                        }
134                                                }
135                                        }
136                                }
137                        }
138                }
139                if(lowestItem == null)
140                        return null;
141                return new Triad<Item,Double,String>(lowestItem,Double.valueOf(lowestPrice),currency);
142        }

144        @Override
145        public void unInvoke()
146        {
147                final Physical affected=this.affected;
148                super.unInvoke();
149                if((this.unInvoked)&&(affected!=null))
150                {
151                        final MOB mob=(MOB)affected;
152                        final Room R=mob.location();
153                        if(R!=null)
154                        {
155                                if((pubRoom != R)&&(!success))
156                                        mob.tell(L("You stop trying to make pub contacts."));
157                                else
158                                if((pubRoom==R)&&(!success))
159                                        mob.tell(L("No one is interested in talking to you."));
160                                else
161                                {
162                                        final Ability A=mob.fetchAbility(ID());
163                                        final String roomID=CMLib.map().getExtendedRoomID(R);
164                                        int level=0;
165                                        if((A!=null)&&(roomID.length()>0))
166                                        {
167                                                final List<String> roomIDs=CMParms.parseSemicolons(A.text(),true);
168                                                for(int i=0;i<roomIDs.size();i++)
169                                                {
170                                                        if(roomIDs.get(i).equals(roomID))
171                                                                level++;
172                                                }
173                                                if(level<3)
174                                                {
175                                                        roomIDs.add(roomID);
176                                                        A.setMiscText(CMParms.combineWith(roomIDs, ';'));
177                                                }
178                                        }
179                                        final Map<Room,List<Item>> allShips=new HashMap<Room,List<Item>>(CMLib.map().numShips());
180                                        for(final Enumeration<BoardableShip> ship=CMLib.map().ships();ship.hasMoreElements();)
181                                        {
182                                                final BoardableShip S=ship.nextElement();
183                                                if((S!=null)
184                                                &&(S instanceof Item)
185                                                &&((level==3)||(CMLib.flags().canBeSeenBy(S, mob))))
186                                                {
187                                                        final Room sR=CMLib.map().roomLocation(S);
188                                                        if((sR!=null)
189                                                        &&(CMLib.flags().isWaterySurfaceRoom(sR))
190                                                        &&(CMLib.flags().canAccess(mob, R)))
191                                                        {
192                                                                if(!allShips.containsKey(sR))
193                                                                        allShips.put(sR,new ArrayList<Item>(1));
194                                                                allShips.get(sR).add((Item)S);
195                                                        }
196                                                }
197                                        }
198                                        final TrackingLibrary.TrackingFlags flags=CMLib.tracking().newFlags();
199                                        int range = baseWaterRange + super.getXLEVELLevel(mob)+super.getXMAXRANGELevel(mob);
200                                        final List<Room> nearby=CMLib.tracking().findTrailToAnyRoom(R, TrackingFlag.WATERSURFACEONLY.myFilter, flags, range);
201                                        Room shore=null;
202                                        Room notShore=null;
203                                        for(int n=nearby.size()-1;(n>=0) && (shore==null);n--)
204                                        {
205                                                final Room R2=nearby.get(n);
206                                                if(CMLib.flags().isWateryRoom(R2))
207                                                {
208                                                        if(notShore==null)
209                                                                notShore=R2;
210                                                        if(!CMLib.flags().isUnderWateryRoom(R2.getRoomInDir(Directions.DOWN)))
211                                                        {
212                                                                shore=R2;
213                                                                notShore=R2;
214                                                        }
215                                                        else
216                                                        {
217                                                                for(int d=0;d<Directions.NUM_DIRECTIONS();d++)
218                                                                {
219                                                                        Room R3=R2.getRoomInDir(d);
220                                                                        if(R3!=null)
221                                                                        {
222                                                                                switch(R3.domainType())
223                                                                                {
224                                                                                case Room.DOMAIN_INDOORS_AIR:
225                                                                                case Room.DOMAIN_OUTDOORS_AIR:
226                                                                                case Room.DOMAIN_INDOORS_UNDERWATER:
227                                                                                case Room.DOMAIN_OUTDOORS_UNDERWATER:
228                                                                                case Room.DOMAIN_INDOORS_WATERSURFACE:
229                                                                                case Room.DOMAIN_OUTDOORS_WATERSURFACE:
230                                                                                        break;
231                                                                                default:
232                                                                                        shore=R3;
233                                                                                        notShore=R2;
234                                                                                        break;
235                                                                                }
236                                                                        }
237                                                                }
238                                                        }
239                                                }
240                                        }
241                                        if(shore==null)
242                                                shore=notShore;
243                                        
244                                        if((shore==null)||(notShore==null))
245                                        {
246                                                mob.tell(L("No one really knows anything."));
247                                                return;
248                                        }
249                                        flags.add(TrackingLibrary.TrackingFlag.WATERSURFACEONLY);
250                                        int radius=50 + (10*(super.getXLEVELLevel(mob)+super.getXMAXRANGELevel(mob)));
251                                        final List<Room> ocean = CMLib.tracking().getRadiantRooms(notShore, flags, radius);
252                                        final Map<Room,List<Room>> trails=new Hashtable<Room,List<Room>>();
253                                        int farthest=0;
254                                        int totalShips=0;
255                                        for(Room R2 : ocean)
256                                        {
257                                                if(allShips.containsKey(R2))
258                                                {
259                                                        List<Room> trail = CMLib.tracking().findTrailToRoom(notShore, R2, flags, radius, ocean);
260                                                        if((trail != null)&&(trail.size()>0))
261                                                        {
262                                                                trails.put(R2,trail);
263                                                                totalShips += allShips.get(R2).size();
264                                                                if(trail.size()>farthest)
265                                                                        farthest=trail.size();
266                                                        }
267                                                }
268                                        }
269                                        String roomName = notShore.displayText(mob);
270                                        if(shore != notShore)
271                                                roomName+=L(" (just off @x1)",shore.displayText(mob));
272                                        StringBuilder shipList=new StringBuilder(L("^NYour contacts tell you about @x1 ships within @x2 of ^W@x3^N: \n\r",
273                                                                                        ""+totalShips,""+farthest,roomName));
274                                        for(Room R2 : trails.keySet())
275                                        {
276                                                List<Room> trail = trails.get(R2);
277                                                int distance=trail.size();
278                                                List<Item> ships=allShips.get(R2);
279                                                for(Item I : ships)
280                                                {
281                                                        switch(level)
282                                                        {
283                                                        case 0:
284                                                                shipList.append("^H"+I.name()+"^N").append("\n\r");
285                                                                break;
286                                                        case 1:
287                                                                shipList.append("^H"+I.name()+"^N").append(L(" (distance: @x2)",""+distance)).append("\n\r");
288                                                                break;
289                                                        default:
290                                                                shipList.append("^H"+I.name()+"^N").append(", directions: ^W");
291                                                                Room lastRoom=notShore;
292                                                                for(int r=trail.size()-2;r>=0;r--)
293                                                                {
294                                                                        int dir=CMLib.map().getRoomDir(lastRoom, trail.get(r));
295                                                                        lastRoom=trail.get(r);
296                                                                        shipList.append(CMLib.directions().getDirectionChar(dir)).append(" ");
297                                                                }
298                                                                shipList.append("^N\n\r");
299                                                                break;
300                                                        }
301                                                }
302                                        }
303                                        mob.tell(shipList.toString());
304                                }
305                        }
306                }
307        }

309        @Override
310        public boolean tick(final Tickable ticking, final int tickID)
311        {
312                if(!super.tick(ticking, tickID))
313                        return false;
314                
315                final Physical affected=this.affected;
316                if(affected instanceof MOB)
317                {
318                        final MOB M=(MOB)affected;
319                        if((pubRoom!=null)&&(pubRoom!=M.location()))
320                        {
321                                success=false;
322                                pubRoom=null;
323                                unInvoke();
324                        }
325                        else
326                        if((super.tickDown % 3)==0)
327                        {
328                                Room R=CMLib.map().roomLocation(M);
329                                if(R.isInhabitant(M))
330                                {
331                                        switch(CMLib.dice().roll(1, 5, -1))
332                                        {
333                                        case 0:
334                                                R.show(M, null, CMMsg.MSG_NOISYMOVEMENT, L("<S-NAME> continue(s) making contacts."));
335                                                break;
336                                        case 1:
337                                                R.show(M, null, CMMsg.MSG_NOISYMOVEMENT, L("<S-NAME> continue(s) drinking and carefully socializing."));
338                                                break;
339                                        case 2:
340                                                R.show(M, null, CMMsg.MSG_NOISYMOVEMENT, L("<S-NAME> continue(s) offering drinks and making the right friends."));
341                                                break;
342                                        case 3:
343                                                R.show(M, null, CMMsg.MSG_NOISYMOVEMENT, L("<S-NAME> continue(s) descreetly making friends."));
344                                                break;
345                                        case 4:
346                                                R.show(M, null, CMMsg.MSG_NOISYMOVEMENT, L("<S-NAME> continue(s) drinking and carefully socializing."));
347                                                break;
348                                        }
349                                }
350                        }
351                }
352                return true;
353        }

355        @Override
356        public boolean invoke(MOB mob, List<String> commands, Physical givenTarget, boolean auto, int asLevel)
357        {
358                final Room R=mob.location();
359                if(R==null)
360                        return false;
361                double money=0.0;
362                String moneyStr="money";
363                if(!auto)
364                {
365                        final Triad<Item,Double,String> alco = cheapestAlcoholHere(mob,R);
366                        if(alco == null)
367                        {
368                                mob.tell(L("You can only establish contacts at a pub."));
369                                return false;
370                        }
371                        TrackingLibrary.TrackingFlags flags=CMLib.tracking().newFlags();
372                        int range = baseWaterRange + super.getXLEVELLevel(mob)+super.getXMAXRANGELevel(mob);
373                        List<Room> nearby=CMLib.tracking().findTrailToAnyRoom(R, TrackingFlag.WATERSURFACEONLY.myFilter, flags, range);
374                        if((nearby==null)||(nearby.size()==0))
375                        {
376                                mob.tell(L("There's no sea or river nearby, so no one here will know anything."));
377                                return false;
378                        }
379                        double pct=0.5 + (CMath.mul(CMath.div(10-super.getXLOWCOSTLevel(mob),2.0), 0.1));
380                        money = CMath.mul(pct,alco.second.doubleValue()*6.0);
381                        moneyStr = CMLib.beanCounter().abbreviatedPrice(alco.third, money);
382                        if(CMLib.beanCounter().getTotalAbsoluteValue(mob, alco.third) < money)
383                        {
384                                mob.tell(L("You need at least @x1 to buy enough drinks to loosen tongues.",moneyStr));
385                                return false;
386                        }
387                }
388                
389                if(!super.invoke(mob,commands,givenTarget,auto,asLevel))
390                        return false;
391
392                final boolean success=proficiencyCheck(mob,0,auto);
393
394                if(money > 0.0)
395                        CMLib.beanCounter().subtractMoney(mob, money);
396                
397                final CMMsg msg=CMClass.getMsg(mob,null,this,CMMsg.MSG_THIEF_ACT,L("<S-NAME> drop(s) @x1 on drinks and start(s) socializing.",moneyStr));
398                if(mob.location().okMessage(mob,msg))
399                {
400                        mob.location().send(mob,msg);
401                        int ticks=12 - super.getXLEVELLevel(mob);
402                        if(!success)
403                                ticks /= 2;
404                        if(ticks<1)
405                                ticks=1;
406                        Thief_PubContacts pub = (Thief_PubContacts)this.beneficialAffect(mob, mob, asLevel, ticks);
407                        if(pub != null)
408                        {
409                                pub.pubRoom=R;
410                                pub.success=success;
411                        }
412                        
413                }
414                return success;
415        }
