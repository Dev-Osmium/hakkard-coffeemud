036public class Thief_ConcealDoor extends ThiefSkill
037{
038        @Override
039        public String ID()
040        {
041                return "Thief_ConcealDoor";
042        }
043
044        private final static String localizedName = CMLib.lang().L("Conceal Door");
045
046        @Override
047        public String name()
048        {
049                return localizedName;
050        }
051
052        @Override
053        protected int canAffectCode()
054        {
055                return Ability.CAN_ITEMS;
056        }
057
058        @Override
059        protected int canTargetCode()
060        {
061                return Ability.CAN_ITEMS;
062        }
063
064        @Override
065        public int abstractQuality()
066        {
067                return Ability.QUALITY_INDIFFERENT;
068        }
069
070        @Override
071        public int classificationCode()
072        {
073                return Ability.ACODE_THIEF_SKILL|Ability.DOMAIN_STEALTHY;
074        }
075
076        private static final String[] triggerStrings =I(new String[] {"DOORCONCEAL","DCONCEAL","CONCEALDOOR"});
077        @Override
078        public String[] triggerStrings()
079        {
080                return triggerStrings;
081        }

083        @Override
084        public int usageType()
085        {
086                return USAGE_MOVEMENT|USAGE_MANA;
087        }
088
089        public int code=Integer.MIN_VALUE;
090
091        @Override
092        public int abilityCode()
093        {
094                if(code<0)
095                        code=CMath.s_int(text());
096                return code;
097        }
098
099        @Override
100        public void setAbilityCode(int newCode)
101        {
102                code=newCode;
103                super.miscText=""+newCode;
104        }
105
106        @Override
107        public void affectPhyStats(Physical host, PhyStats stats)
108        {
109                super.affectPhyStats(host,stats);
110                if((host instanceof Exit)&&(!((Exit)host).isOpen()))
111                {
112                        stats.setDisposition(stats.disposition()|PhyStats.IS_HIDDEN);
113                        // cant affect level because will make it unpickable, therefore unopenable
114                        // need some other way to designate its hiddenitude.
115                        //stats.setLevel(stats.level()+abilityCode());
116                }
117        }
118
119        @Override
120        public void executeMsg(final Environmental host, final CMMsg msg)
121        {
122                super.executeMsg(host,msg);
123                if((msg.target()==affected)
124                &&(msg.targetMinor()==CMMsg.TYP_OPEN)
125                &&(super.canBeUninvoked()))
126                {
127                        unInvoke();
128                        if(affected!=null)
129                        {
130                                affected.delEffect(this);
131                                affected.recoverPhyStats();
132                        }
133                }
134                else
135                if(canBeUninvoked() && (invoker()!=null) && (!msg.source().isMonster()) && (msg.source()!=invoker()) && (msg.sourceMinor()==CMMsg.TYP_ENTER) && (affected!=null))
136                {
137                        if(!CMLib.flags().isInTheGame(invoker(), true))
138                        {
139                                unInvoke();
140                                if(affected!=null)
141                                {
142                                        affected.delEffect(this);
143                                        affected.recoverPhyStats();
144                                }
145                        }
146                        else
147                        {
148                                final Set<MOB> grp=invoker().getGroupMembers(new HashSet<MOB>());
149                                if(!grp.contains(msg.source()))
150                                {
151                                        unInvoke();
152                                        if(affected!=null)
153                                        {
154                                                affected.delEffect(this);
155                                                affected.recoverPhyStats();
156                                        }
157                                }
158                        }
159                }
160        }
161
162        @Override
163        public boolean invoke(MOB mob, List<String> commands, Physical givenTarget, boolean auto, int asLevel)
164        {
165                if((commands.size()<1)&&(givenTarget==null))
166                {
167                        mob.tell(L("Which door would you like to conceal?"));
168                        return false;
169                }
170                Environmental chkE=null;
171                final String typed=CMParms.combine(commands,0);
172                if(CMLib.directions().getGoodDirectionCode(typed)<0)
173                        chkE=mob.location().fetchFromMOBRoomItemExit(mob,null,typed,Wearable.FILTER_WORNONLY);
174                else
175                        chkE=mob.location().getExitInDir(CMLib.directions().getGoodDirectionCode(typed));
176                if((!(chkE instanceof Exit))||(!CMLib.flags().canBeSeenBy(chkE,mob)))
177                {
178                        mob.tell(L("You don't see any doors called '@x1' here.",typed));
179                        return false;
180                }
181                final Exit X=(Exit)chkE;
182                if(!X.hasADoor())
183                {
184                        mob.tell(mob,X,null,L("<T-NAME> is not a door!"));
185                        return false;
186                }
187                if((!auto)&&(X.phyStats().level()>((adjustedLevel(mob,asLevel)*2))))
188                {
189                        mob.tell(L("You aren't good enough to conceal that door."));
190                        return false;
191                }
192                if(X.isOpen())
193                {
194                        mob.tell(mob,X,null,L("You'd better close <T-NAME> first."));
195                        return false;
196                }
197
198                if(!super.invoke(mob,commands,givenTarget,auto,asLevel))
199                        return false;
200
201                final boolean success=proficiencyCheck(mob,0,auto);
202
203                if(success)
204                {
205                        final CMMsg msg=CMClass.getMsg(mob,X,this,CMMsg.MSG_THIEF_ACT,L("<S-NAME> conceal(s) <T-NAME>."),CMMsg.MSG_THIEF_ACT,null,CMMsg.MSG_THIEF_ACT,null);
206                        if(mob.location().okMessage(mob,msg))
207                        {
208                                mob.location().send(mob,msg);
209                                final Ability A=(Ability)super.copyOf();
210                                A.setInvoker(mob);
211                                A.setAbilityCode((adjustedLevel(mob,asLevel)*2)-X.phyStats().level());
212                                final Room R=mob.location();
213                                Room R2=null;
214                                for(int d=Directions.NUM_DIRECTIONS()-1;d>=0;d--)
215                                {
216                                        if(R.getExitInDir(d)==X)
217                                        {
218                                                R2=R.getRoomInDir(d);
219                                                break;
220                                        }
221                                }
222                                if((CMLib.law().doesOwnThisLand(mob,R))
223                                ||((R2!=null)&&(CMLib.law().doesOwnThisLand(mob,R2))))
224                                {
225                                        X.addNonUninvokableEffect(A);
226                                        CMLib.database().DBUpdateExits(mob.location());
227                                }
228                                else
229                                        A.startTickDown(mob,X,15*(adjustedLevel(mob,asLevel)));
230                                X.recoverPhyStats();
231                        }
232                }
233                else
234                        beneficialVisualFizzle(mob,X,L("<S-NAME> attempt(s) to conceal <T-NAME>, but obviously fail(s)."));
235                return success;
236        }
