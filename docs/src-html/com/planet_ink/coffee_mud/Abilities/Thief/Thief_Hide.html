110        @Override
111        public void executeMsg(final Environmental myHost, final CMMsg msg)
112        {
113                if(!(affected instanceof MOB))
114                        return;
115
116                final MOB mob=(MOB)affected;
117
118                if(msg.amISource(mob))
119                {
120                        if(((msg.sourceMinor()==CMMsg.TYP_ENTER)
121                                ||(msg.sourceMinor()==CMMsg.TYP_LEAVE)
122                                ||(msg.sourceMinor()==CMMsg.TYP_FLEE)
123                                ||(msg.sourceMinor()==CMMsg.TYP_RECALL))
124                        &&(!msg.sourceMajor(CMMsg.MASK_ALWAYS))
125                        &&(msg.sourceMajor()>0))
126                        {
127                                unInvoke();
128                                mob.recoverPhyStats();
129                                mob.recoverCharStats();
130                        }
131                        else
132                        if((abilityCode()==0)
133                        &&(!msg.sourceMajor(CMMsg.MASK_ALWAYS))
134                        &&(msg.othersMinor()!=CMMsg.TYP_LOOK)
135                        &&(msg.othersMinor()!=CMMsg.TYP_EXAMINE)
136                        &&(msg.othersMajor()>0))
137                        {
138                                if(msg.othersMajor(CMMsg.MASK_SOUND))
139                                {
140                                        unInvoke();
141                                        mob.recoverPhyStats();
142                                        mob.recoverCharStats();
143                                }
144                                else
145                                switch(msg.othersMinor())
146                                {
147                                case CMMsg.TYP_SPEAK:
148                                case CMMsg.TYP_CAST_SPELL:
149                                        {
150                                                unInvoke();
151                                                mob.recoverPhyStats();
152                                                mob.recoverCharStats();
153                                        }
154                                        break;
155                                case CMMsg.TYP_OPEN:
156                                case CMMsg.TYP_CLOSE:
157                                case CMMsg.TYP_LOCK:
158                                case CMMsg.TYP_UNLOCK:
159                                case CMMsg.TYP_PUSH:
160                                case CMMsg.TYP_PULL:
161                                        if(((msg.target() instanceof Exit)
162                                                ||((msg.target() instanceof Item)
163                                                   &&(!msg.source().isMine(msg.target())))))
164                                        {
165                                                unInvoke();
166                                                mob.recoverPhyStats();
167                                                mob.recoverCharStats();
168                                        }
169                                        break;
170                                }
171                        }
172                }
173                return;
174        }
175
176        @Override
177        public void affectCharStats(final MOB affected, final CharStats affectableStats)
178        {
179                super.affectCharStats(affected,affectableStats);
180                affectableStats.setStat(CharStats.STAT_SAVE_DETECTION,proficiency()+bonus+affectableStats.getStat(CharStats.STAT_SAVE_DETECTION));
181        }
182
183        @Override
184        public void affectPhyStats(final Physical affected, final PhyStats affectableStats)
185        {
186                super.affectPhyStats(affected,affectableStats);
187                affectableStats.setDisposition(affectableStats.disposition()|PhyStats.IS_HIDDEN);
188                if(CMLib.flags().isSneaking(affected))
189                        affectableStats.setDisposition(affectableStats.disposition()-PhyStats.IS_SNEAKING);
190        }
191
192        @Override
193        public boolean invoke(MOB mob, List<String> commands, Physical givenTarget, boolean auto, int asLevel)
194        {
195                final Room room=mob.location();
196                if((mob.fetchEffect(this.ID())!=null)||(room==null))
197                {
198                        mob.tell(L("You are already hiding."));
199                        return false;
200                }
201
202                if(mob.isInCombat())
203                {
204                        mob.tell(L("Not while in combat!"));
205                        return false;
206                }
207
208                if(!super.invoke(mob,commands,givenTarget,auto,asLevel))
209                        return false;
210
211                final MOB highestMOB=getHighestLevelMOB(mob,null);
212                final int levelDiff=(mob.phyStats().level()+(2*getXLEVELLevel(mob)))-getMOBLevel(highestMOB);
213
214                final String str=L("You creep into a shadow and remain completely still.");
215
216                boolean success=(highestMOB==null)||proficiencyCheck(mob,levelDiff*10,auto);
217
218                if(!success)
219                {
220                        if(highestMOB!=null)
221                                beneficialVisualFizzle(mob,highestMOB,L("<S-NAME> attempt(s) to hide from <T-NAMESELF> and fail(s)."));
222                        else
223                                beneficialVisualFizzle(mob,null,L("<S-NAME> attempt(s) to hide and fail(s)."));
224                }
225                else
226                {
227                        final CMMsg msg=CMClass.getMsg(mob,null,this,auto?CMMsg.MSG_OK_ACTION:(CMMsg.MSG_DELICATE_HANDS_ACT|CMMsg.MASK_MOVE),str,CMMsg.NO_EFFECT,null,CMMsg.NO_EFFECT,null);
228                        if(room.okMessage(mob,msg))
229                        {
230                                room.send(mob,msg);
231                                invoker=mob;
232                                final Ability newOne=(Ability)this.copyOf();
233                                ((Thief_Hide)newOne).bonus=getXLEVELLevel(mob)*2;
234                                if(mob.fetchEffect(newOne.ID())==null)
235                                        mob.addEffect(newOne);
236                                mob.recoverPhyStats();
237                        }
238                        else
239                                success=false;
240                }
241                return success;
242        }
