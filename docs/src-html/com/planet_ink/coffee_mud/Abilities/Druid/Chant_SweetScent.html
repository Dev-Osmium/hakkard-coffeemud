036public class Chant_SweetScent extends Chant
037{
038        @Override
039        public String ID()
040        {
041                return "Chant_SweetScent";
042        }
043
044        private final static String     localizedName   = CMLib.lang().L("Sweet Scent");
045
046        @Override
047        public String name()
048        {
049                return localizedName;
050        }
051
052        @Override
053        public int abstractQuality()
054        {
055                return Ability.QUALITY_INDIFFERENT;
056        }
057
058        @Override
059        public int classificationCode()
060        {
061                return Ability.ACODE_CHANT | Ability.DOMAIN_PLANTCONTROL;
062        }
063
064        @Override
065        protected int canAffectCode()
066        {
067                return Ability.CAN_ITEMS;
068        }
069
070        @Override
071        protected int canTargetCode()
072        {
073                return Ability.CAN_ITEMS;
074        }
075
076        @Override
077        public boolean tick(final Tickable ticking, final int tickID)
078        {
079                if(!super.tick(ticking,tickID))
080                        return false;
081                if((affected!=null)&&(affected instanceof Item))
082                {
083                        final Item I=(Item)affected;
084                        if(I.owner() instanceof Room)
085                        {
086                                final Room room=(Room)I.owner();
087                                final ArrayList<Room> rooms=new ArrayList<Room>();
088                                TrackingLibrary.TrackingFlags flags;
089                                flags = CMLib.tracking().newFlags()
090                                                .plus(TrackingLibrary.TrackingFlag.OPENONLY);
091                                int range=10 + super.getXLEVELLevel(invoker())+(2*super.getXMAXRANGELevel(invoker()));
092                                CMLib.tracking().getRadiantRooms(room,rooms,flags,null,range,null);
093                                for(int i=0;i<room.numInhabitants();i++)
094                                {
095                                        final MOB M=room.fetchInhabitant(i);
096                                        if((M!=null)
097                                        &&(CMLib.flags().isAnimalIntelligence(M))
098                                        &&(CMLib.flags().canSmell(M)))
099                                                M.tell(M,I,null,L("<T-NAME> smell(s) absolutely intoxicating!"));
100                                }
101                                for(int r=0;r<rooms.size();r++)
102                                {
103                                        final Room R=rooms.get(r);
104                                        if(R!=room)
105                                        {
106                                                final int dir=CMLib.tracking().radiatesFromDir(R,rooms);
107                                                if(dir>=0)
108                                                {
109                                                        for(int i=0;i<R.numInhabitants();i++)
110                                                        {
111                                                                final MOB M=R.fetchInhabitant(i);
112                                                                if((M!=null)
113                                                                &&(CMLib.flags().isAnimalIntelligence(M))
114                                                                &&(!M.isInCombat())
115                                                                &&((!M.isMonster())||(CMLib.flags().isMobile(M)))
116                                                                &&(CMLib.flags().canSmell(M)))
117                                                                {
118                                                                        M.tell(M,null,null,L("You smell something irresistable @x1.",CMLib.directions().getInDirectionName(dir)));
119                                                                        if(CMLib.dice().rollPercentage()>M.charStats().getSave(CharStats.STAT_SAVE_MIND))
120                                                                                CMLib.tracking().walk(M,dir,false,false);
121                                                                }
122                                                        }
123                                                }
124                                        }
125
126                                }
127                        }
128                }
129                return true;
130        }
131
132        @Override
133        public void executeMsg(final Environmental myHost, final CMMsg msg)
134        {
135                super.executeMsg(myHost,msg);
136                if((msg.amITarget(affected))
137                &&(msg.targetMinor()==CMMsg.TYP_SNIFF)
138                &&(CMLib.flags().canSmell(msg.source())))
139                        msg.source().tell(msg.source(),affected,null,L("<T-NAME> smell(s) absolutely intoxicating!"));
140        }
141
142        @Override
143        public boolean invoke(MOB mob, List<String> commands, Physical givenTarget, boolean auto, int asLevel)
144        {
145                if(
146                        (CMLib.flags().isWateryRoom(mob.location()))
147                ||(mob.location().domainType()==Room.DOMAIN_OUTDOORS_AIR)
148                ||(mob.location().domainType()==Room.DOMAIN_INDOORS_AIR)
149                   )
150                {
151                        mob.tell(L("This magic will not work here."));
152                        return false;
153                }
154
155                final Item target=getTarget(mob,mob.location(),givenTarget,commands,Wearable.FILTER_UNWORNONLY);
156                if(target==null)
157                        return false;
158                if(!Druid_MyPlants.isMyPlant(target,mob))
159                {
160                        mob.tell(L("@x1 is not one of your plants!",target.name(mob)));
161                        return false;
162                }
163
164                if(!super.invoke(mob,commands,givenTarget,auto,asLevel))
165                        return false;
166
167                // now see if it worked
168                final boolean success=proficiencyCheck(mob,0,auto);
169                if(success)
170                {
171                        final CMMsg msg=CMClass.getMsg(mob,target,this,verbalCastCode(mob,target,auto),auto?"":L("^S<S-NAME> chant(s) to <T-NAMESELF>.^?"));
172                        if(mob.location().okMessage(mob,msg))
173                        {
174                                mob.location().send(mob,msg);
175                                beneficialAffect(mob,target,asLevel,0);
176                        }
177                }
178                else
179                        return beneficialWordsFizzle(mob,target,L("<S-NAME> chant(s) to the <T-NAMESELF>, but nothing happens."));
180
181                // return whether it worked
182                return success;
183        }
184}
