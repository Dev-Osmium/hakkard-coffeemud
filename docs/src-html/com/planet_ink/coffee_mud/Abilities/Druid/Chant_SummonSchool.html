036public class Chant_SummonSchool extends Chant
037{
038        @Override
039        public String ID()
040        {
041                return "Chant_SummonSchool";
042        }

044        private final static String     localizedName   = CMLib.lang().L("Summon School");
045
046        @Override
047        public String name()
048        {
049                return localizedName;
050        }
051
052        private final static String     localizedStaticDisplay  = CMLib.lang().L("(Summon School)");
053
054        @Override
055        public String displayText()
056        {
057                return localizedStaticDisplay;
058        }
059
060        @Override
061        public int abstractQuality()
062        {
063                return Ability.QUALITY_BENEFICIAL_SELF;
064        }
065
066        @Override
067        protected int canAffectCode()
068        {
069                return CAN_MOBS;
070        }
071
072        @Override
073        public int classificationCode()
074        {
075                return Ability.ACODE_CHANT | Ability.DOMAIN_ANIMALAFFINITY;
076        }

078        private final Random    randomizer              = new Random(System.currentTimeMillis());
079        protected int                   numberOfImages  = 0;
080        protected boolean               notAgain                = false;
081        protected String                schoolMemberName= "";
082
083        @Override
084        public boolean okMessage(final Environmental myHost, final CMMsg msg)
085        {
086                if(!(affected instanceof MOB))
087                        return true;
088
089                final MOB mob=(MOB)affected;
090
091                if((msg.amITarget(mob))
092                &&(msg.targetMinor()==CMMsg.TYP_WEAPONATTACK)
093                &&(mob!=msg.source()))
094                {
095                        if(numberOfImages <= 0)
096                        {
097                                unInvoke();
098                                return true;
099                        }
100                        int intAdjustment = (msg.source().charStats().getMaxStat(CharStats.STAT_INTELLIGENCE) 
101                                                                        - mob.charStats().getStat(CharStats.STAT_INTELLIGENCE))/2;
102                        if(intAdjustment < 1)
103                                intAdjustment = 1;
104
105                        final int numberOfTargets = numberOfImages + intAdjustment;
106                        if(randomizer.nextInt() % numberOfTargets >= intAdjustment)
107                        {
108                                if(mob.location().show(mob,msg.source(),CMMsg.MSG_NOISYMOVEMENT,L("<T-NAME> attack(s) a member of <S-YOUPOSS> school, making <T-HIM-HER> swim away!")))
109                                        numberOfImages--;
110                                return false;
111                        }
112                }
113                if(numberOfImages<=0)
114                        unInvoke();
115                return true;
116        }
117
118        @Override
119        public void executeMsg(final Environmental myHost, final CMMsg msg)
120        {
121                super.executeMsg(myHost,msg);
122
123                if(!(affected instanceof MOB))
124                        return;
125
126                if(notAgain)
127                        return;
128
129                final MOB mob=(MOB)affected;
130                Room R=mob.location();
131                if(R==null)
132                        return;
133                if(!CMLib.flags().isUnderWateryRoom(R))
134                {
135                        unInvoke();
136                }
137                
138                if(msg.amISource(mob))
139                {
140                        if((
141                                (CMath.bset(msg.othersMajor(),CMMsg.MASK_EYES))
142                                ||(CMath.bset(msg.othersMajor(),CMMsg.MASK_MOVE))
143                                ||(CMath.bset(msg.othersMajor(),CMMsg.MASK_MOUTH))
144                                ||(CMath.bset(msg.othersMajor(),CMMsg.MASK_HANDS)))
145                        &&(msg.othersMessage()!=null)
146                        &&(msg.targetMinor()!=CMMsg.TYP_DAMAGE)
147                        &&(msg.othersMessage().length()>0))
148                        {
149                                notAgain=true;
150                                if(numberOfImages<=0)
151                                        unInvoke();
152                                else
153                                {
154                                        for(int x=0;x<numberOfImages;x++)
155                                                msg.addTrailerMsg(CMClass.getMsg(mob,msg.target(),msg.tool(),CMMsg.MSG_OK_VISUAL,msg.othersMessage()));
156                                }
157                        }
158                }
159                else
160                if((msg.amITarget(mob.location())&&(!msg.amISource(mob))&&((msg.targetMinor()==CMMsg.TYP_LOOK)||(msg.targetMinor()==CMMsg.TYP_EXAMINE)))
161                &&((CMLib.flags().canBeSeenBy(mob,msg.source()))&&(mob.displayText(msg.source()).length()>0)))
162                {
163                        final StringBuffer Say=new StringBuffer("");
164                        final boolean compress=msg.source().isAttributeSet(MOB.Attrib.COMPRESS);
165                        for(int i=0;i<numberOfImages;i++)
166                        {
167                                Say.append("^M");
168                                if(compress)
169                                        Say.append(CMLib.flags().getDispositionBlurbs(mob,mob)+"^M ");
170                                if(mob.displayText(msg.source()).length()>0)
171                                        Say.append(CMStrings.endWithAPeriod(CMStrings.capitalizeFirstLetter(mob.displayText(msg.source()))));
172                                else
173                                        Say.append(CMStrings.endWithAPeriod(CMStrings.capitalizeFirstLetter(mob.name())));
174                                if(!compress)
175                                        Say.append(CMLib.flags().getDispositionBlurbs(mob,msg.source())+"^N\n\r");
176                                else
177                                        Say.append("^N");
178                        }
179                        if(Say.toString().length()>0)
180                        {
181                                final CMMsg msg2=CMClass.getMsg(msg.source(),null,this,CMMsg.MSG_OK_VISUAL,Say.toString(),CMMsg.NO_EFFECT,null,CMMsg.NO_EFFECT,null);
182                                msg.addTrailerMsg(msg2);
183                        }
184                }
185                notAgain=false;
186        }
187
188        @Override
189        public void unInvoke()
190        {
191                // undo the affects of this spell
192                if(!(affected instanceof MOB))
193                        return;
194                final MOB mob=(MOB)affected;
195                if(canBeUninvoked())
196                        numberOfImages=0;
197                super.unInvoke();
198
199                if(canBeUninvoked() && (mob.location()!=null))
200                        mob.location().show(mob,null,null,CMMsg.MSG_OK_ACTION, (L("<S-YOUPOSS> school swims away.")));
201        }
202
203        @Override
204        public boolean invoke(MOB mob, List<String> commands, Physical givenTarget, boolean auto, int asLevel)
205        {
206                MOB target=mob;
207                if((auto)&&(givenTarget!=null)&&(givenTarget instanceof MOB))
208                        target=(MOB)givenTarget;
209                if(target.fetchEffect(ID())!=null)
210                {
211                        mob.tell(target,null,null,L("<S-NAME> already <S-HAS-HAVE> a school here."));
212                        return false;
213                }
214
215                if(!Druid_ShapeShift.isShapeShifted(mob))
216                {
217                        mob.tell(L("You must be in your animal form to call a mate."));
218                        return false;
219                }
220                
221                Room R=mob.location();
222                if(R==null)
223                        return false;
224                if(!CMLib.flags().isUnderWateryRoom(R))
225                {
226                        mob.tell(L("You can only summon a school underwater."));
227                        return false;
228                }
229                
230                String raceName = mob.charStats().getMyRace().name();
231                String raceCat = mob.charStats().getMyRace().racialCategory();
232                if((raceCat.equalsIgnoreCase("Fish"))||(raceCat.equalsIgnoreCase("Sea Mammal")))
233                {
234                        // yay
235                }
236                else
237                {
238                        mob.tell(L("You are not in the proper form to call on a school here."));
239                        return false;
240                }
241
242                if(!super.invoke(mob,commands,givenTarget,auto,asLevel))
243                        return false;
244
245                final boolean success=proficiencyCheck(mob,0,auto);
246
247                if(success)
248                {
249                        invoker=mob;
250                        if(raceName.equalsIgnoreCase("ClownFish"))
251                                numberOfImages = CMLib.dice().roll(1,adjustedLevel(mob,asLevel),100);
252                        else
253                                numberOfImages = CMLib.dice().roll(1,(int)(Math.round(CMath.div(adjustedLevel(mob,asLevel),3.0))),2);
254                        final CMMsg msg=CMClass.getMsg(mob,target,this,verbalCastCode(mob,target,auto),L((auto?"":"^S<S-NAME> chant(s) to the waters, and ")+"suddenly @x1 @x2(s) arrive, forming up as a school around <S-NAME>.^?",""+numberOfImages,raceName.toLowerCase()));
255                        if(mob.charStats().getStat(CharStats.STAT_INTELLIGENCE)<5)
256                                mob.charStats().setStat(CharStats.STAT_INTELLIGENCE, 5);
257                        if(mob.location().okMessage(mob,msg))
258                        {
259                                mob.recoverCharStats();
260                                mob.location().send(mob,msg);
261                                beneficialAffect(mob,target,asLevel,0);
262                        }
263                        else
264                                mob.recoverCharStats();
265                }
266                else
267                {
268                        numberOfImages = 0;
269                        return beneficialWordsFizzle(mob,target,L("<S-NAME> chant(s) with great force, but nothing more happens."));
270                }
271                // return whether it worked
272                return success;
273        }
