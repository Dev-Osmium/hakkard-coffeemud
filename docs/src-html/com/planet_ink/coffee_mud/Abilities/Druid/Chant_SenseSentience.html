036public class Chant_SenseSentience extends Chant
037{
038        @Override
039        public String ID()
040        {
041                return "Chant_SenseSentience";
042        }
043
044        private final static String localizedName = CMLib.lang().L("Sense Sentience");
045
046        @Override
047        public String name()
048        {
049                return localizedName;
050        }
051
052        @Override
053        public int classificationCode()
054        {
055                return Ability.ACODE_CHANT|Ability.DOMAIN_BREEDING;
056        }
057
058        @Override
059        public int abstractQuality()
060        {
061                return Ability.QUALITY_OK_SELF;
062        }
063
064        @Override
065        protected int canTargetCode()
066        {
067                return 0;
068        }
069
070        @Override
071        protected int canAffectCode()
072        {
073                return 0;
074        }
075
076        @Override
077        public boolean invoke(MOB mob, List<String> commands, Physical givenTarget, boolean auto, int asLevel)
078        {
079                if(!super.invoke(mob,commands,givenTarget,auto,asLevel))
080                        return false;
081
082                final boolean success=proficiencyCheck(mob,0,auto);
083
084                if(success)
085                {
086                        final CMMsg msg=CMClass.getMsg(mob,null,this,verbalCastCode(mob,null,auto),auto?"":L("^S<S-NAME> chant(s) softly to <S-HIM-HERSELF>!^?"));
087                        if(mob.location().okMessage(mob,msg))
088                        {
089                                mob.location().send(mob,msg);
090                                final StringBuffer lines=new StringBuffer("^x");
091                                lines.append(CMStrings.padRight(L("Name"),25)+"| ");
092                                lines.append(CMStrings.padRight(L("Location"),17)+"^.^N\n\r");
093                                TrackingLibrary.TrackingFlags flags;
094                                flags = CMLib.tracking().newFlags()
095                                                .plus(TrackingLibrary.TrackingFlag.AREAONLY);
096                                int range=50 + super.getXLEVELLevel(mob)+(2*super.getXMAXRANGELevel(mob));
097                                final List<Room> checkSet=CMLib.tracking().getRadiantRooms(mob.location(),flags,range);
098                                if(!checkSet.contains(mob.location()))
099                                        checkSet.add(mob.location());
100                                for (final Room room : checkSet)
101                                {
102                                        final Room R=CMLib.map().getRoom(room);
103                                        if((((R.domainType()&Room.INDOORS)==0)
104                                                &&(R.domainType()!=Room.DOMAIN_OUTDOORS_CITY)
105                                                &&(R.domainType()!=Room.DOMAIN_OUTDOORS_SPACEPORT))
106                                        ||(R==mob.location()))
107                                        for(int m=0;m<R.numInhabitants();m++)
108                                        {
109                                                final MOB M=R.fetchInhabitant(m);
110                                                if((M!=null)&&(M.charStats().getStat(CharStats.STAT_INTELLIGENCE)>=2))
111                                                {
112                                                        lines.append("^!"+CMStrings.padRight(M.name(mob),25)+"^?| ");
113                                                        lines.append(R.displayText(mob));
114                                                        lines.append("\n\r");
115                                                }
116                                        }
117                                }
118                                mob.tell(lines.toString()+"^.");
119                        }
120                }
121                else
122                        beneficialVisualFizzle(mob,null,L("<S-NAME> chant(s) softly to <S-HIM-HERSELF>, but the magic fades."));
123
124                return success;
125        }
126}
