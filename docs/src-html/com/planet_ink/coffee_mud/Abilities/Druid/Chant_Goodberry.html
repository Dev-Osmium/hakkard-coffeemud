035public class Chant_Goodberry extends Chant
036{
037        @Override
038        public String ID()
039        {
040                return "Chant_Goodberry";
041        }
042
043        private final static String     localizedName   = CMLib.lang().L("Goodberry");
044
045        @Override
046        public String name()
047        {
048                return localizedName;
049        }
050
051        @Override
052        public int classificationCode()
053        {
054                return Ability.ACODE_CHANT | Ability.DOMAIN_PLANTCONTROL;
055        }
056
057        @Override
058        public int abstractQuality()
059        {
060                return Ability.QUALITY_INDIFFERENT;
061        }
062
063        @Override
064        protected int canAffectCode()
065        {
066                return 0;
067        }
068
069        @Override
070        protected int canTargetCode()
071        {
072                return CAN_ITEMS;
073        }
074
075        public boolean checkDo(Item newTarget, Item originaltarget, Environmental owner)
076        {
077                if((newTarget!=null)
078                &&(newTarget instanceof Food)
079                &&(!(newTarget instanceof Pill))
080                &&(isBerry(newTarget))
081                &&(newTarget.container()==originaltarget.container())
082                &&(newTarget.name().equals(originaltarget.name())))
083                {
084                        final Pill newItem=(Pill)CMClass.getItem("GenPill");
085                        newItem.setName(newTarget.name());
086                        newItem.setDisplayText(newTarget.displayText());
087                        newItem.setDescription(newTarget.description());
088                        newItem.setMaterial(RawMaterial.RESOURCE_BERRIES);
089                        newItem.setNourishment(0);
090                        newItem.basePhyStats().setDisposition(PhyStats.IS_GLOWING);
091                        newItem.setSpellList(";Prayer_CureLight;");
092                        newItem.recoverPhyStats();
093                        newItem.setMiscText(newItem.text());
094                        final Container location=newTarget.container();
095                        newTarget.destroy();
096                        if(owner instanceof MOB)
097                                ((MOB)owner).addItem(newItem);
098                        else
099                        if(owner instanceof Room)
100                                ((Room)owner).addItem(newItem,ItemPossessor.Expire.Player_Drop);
101                        newItem.setContainer(location);
102                        return true;
103                }
104                return false;
105        }
106
107        public boolean isBerry(Item I)
108        {
109                return CMParms.contains(RawMaterial.CODES.BERRIES(), I.material());
110        }
111
112        @Override
113        public boolean invoke(MOB mob, List<String> commands, Physical givenTarget, boolean auto, int asLevel)
114        {
115                final Item target=getTarget(mob,mob.location(),givenTarget,commands,Wearable.FILTER_UNWORNONLY);
116                if(target==null)
117                        return false;
118
119                final Environmental owner=target.owner();
120                if(owner==null)
121                        return false;
122
123                if(!super.invoke(mob,commands,givenTarget,auto,asLevel))
124                        return false;
125
126                final boolean success=proficiencyCheck(mob,0,auto);
127
128                if((!(target instanceof Food))
129                ||(!isBerry(target)))
130                {
131                        mob.tell(L("This magic will not work on @x1.",target.name(mob)));
132                        return false;
133                }
134
135                if(success)
136                {
137                        int numAffected=CMLib.dice().roll(1,adjustedLevel(mob,asLevel)/7,1);
138                        final CMMsg msg=CMClass.getMsg(mob,target,this,verbalCastCode(mob,target,auto),auto?"":L("^S<S-NAME> chant(s) to <T-NAMESELF>.^?"));
139                        if(mob.location().okMessage(mob,msg))
140                        {
141                                mob.location().send(mob,msg);
142                                mob.location().show(mob,target,CMMsg.MSG_OK_ACTION,L("<T-NAME> begin to glow!"));
143                                if(owner instanceof MOB)
144                                {
145                                        for(int i=0;i<((MOB)owner).numItems();i++)
146                                        {
147                                                final Item newTarget=((MOB)owner).getItem(i);
148                                                if((newTarget!=null)&&(checkDo(newTarget,target,owner)))
149                                                {
150                                                        if((--numAffected)==0)
151                                                                break;
152                                                        i=-1;
153                                                }
154                                        }
155                                }
156                                if(owner instanceof Room)
157                                {
158                                        for(int i=0;i<((Room)owner).numItems();i++)
159                                        {
160                                                final Item newTarget=((Room)owner).getItem(i);
161                                                if((newTarget!=null)&&(checkDo(newTarget,target,owner)))
162                                                {
163                                                        if((--numAffected)==0)
164                                                                break;
165                                                        i=-1;
166                                                }
167                                        }
168                                }
169                        }
170                }
171                else
172                        beneficialWordsFizzle(mob,target,L("<S-NAME> chant(s) to <T-NAMESELF>, but nothing happens."));
173
174                // return whether it worked
175                return success;
176        }
177}
