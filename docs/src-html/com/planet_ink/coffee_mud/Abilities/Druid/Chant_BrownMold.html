036public class Chant_BrownMold extends Chant
037{
038        @Override
039        public String ID()
040        {
041                return "Chant_BrownMold";
042        }
043
044        private final static String localizedName = CMLib.lang().L("Brown Mold");
045
046        @Override
047        public String name()
048        {
049                return localizedName;
050        }
051
052        private final static String localizedStaticDisplay = CMLib.lang().L("(Brown Mold)");
053
054        @Override
055        public String displayText()
056        {
057                return localizedStaticDisplay;
058        }
059
060        @Override
061        public int classificationCode()
062        {
063                return Ability.ACODE_CHANT|Ability.DOMAIN_PLANTGROWTH;
064        }
065
066        @Override
067        public int abstractQuality()
068        {
069                return Ability.QUALITY_BENEFICIAL_SELF;
070        }
071
072        @Override
073        public int enchantQuality()
074        {
075                return Ability.QUALITY_INDIFFERENT;
076        }
077
078        @Override
079        protected int canAffectCode()
080        {
081                return CAN_MOBS;
082        }
083
084        @Override
085        protected int canTargetCode()
086        {
087                return 0;
088        }
089
090        @Override
091        public boolean tick(final Tickable ticking, final int tickID)
092        {
093                if(tickID==Tickable.TICKID_MOB)
094                {
095                        if((affected!=null)
096                        &&(affected instanceof MOB)
097                        &&(invoker!=null))
098                        {
099                                final MOB mob=(MOB)affected;
100                                if(((mob.amFollowing()==null)
101                                ||(mob.amDead())
102                                ||(!mob.isInCombat())
103                                ||((invoker!=null)&&(mob.location()!=invoker.location()))))
104                                        unInvoke();
105                        }
106                }
107                return super.tick(ticking,tickID);
108        }
109
110        @Override
111        public boolean okMessage(final Environmental myHost, final CMMsg msg)
112        {
113                if((affected!=null)
114                &&(affected instanceof MOB)
115                &&(msg.amISource((MOB)affected)))
116                {
117                        if(msg.sourceMinor()==CMMsg.TYP_DEATH)
118                        {
119                                unInvoke();
120                                return false;
121                        }
122                }
123                return super.okMessage(myHost,msg);
124        }
125
126        @Override
127        public void unInvoke()
128        {
129                final MOB mob=(MOB)affected;
130                super.unInvoke();
131                if((canBeUninvoked())&&(mob!=null))
132                {
133                        if(mob.location()!=null)
134                                mob.location().show(mob,null,CMMsg.MSG_OK_VISUAL,L("<S-NAME> wither(s) away."));
135                        if(mob.amDead())
136                                mob.setLocation(null);
137                        mob.destroy();
138                }
139        }
140
141        @Override
142        public void executeMsg(final Environmental myHost, final CMMsg msg)
143        {
144                super.executeMsg(myHost,msg);
145                if((affected!=null)
146                &&(affected instanceof MOB)
147                &&(msg.amISource((MOB)affected)||msg.amISource(((MOB)affected).amFollowing())||(msg.source()==invoker()))
148                &&(msg.sourceMinor()==CMMsg.TYP_QUIT))
149                {
150                        unInvoke();
151                        if(msg.source().playerStats()!=null)
152                                msg.source().playerStats().setLastUpdated(0);
153                }
154        }
155
156        @Override
157        public int castingQuality(MOB mob, Physical target)
158        {
159                if(mob!=null)
160                {
161                        if(!mob.isInCombat())
162                                return Ability.QUALITY_INDIFFERENT;
163                }
164                return super.castingQuality(mob,target);
165        }
166
167        @Override
168        public boolean invoke(MOB mob, List<String> commands, Physical givenTarget, boolean auto, int asLevel)
169        {
170                if(!mob.isInCombat())
171                {
172                        mob.tell(L("Only the anger of combat can summon the brown mold."));
173                        return false;
174                }
175                final int material=RawMaterial.RESOURCE_HEMP;
176
177                if(!super.invoke(mob,commands,givenTarget,auto,asLevel))
178                        return false;
179
180                final boolean success=proficiencyCheck(mob,0,auto);
181
182                if(success)
183                {
184                        invoker=mob;
185                        final CMMsg msg=CMClass.getMsg(mob,null,this,verbalCastCode(mob,null,auto),auto?"":L("^S<S-NAME> chant(s) and summon(s) a brown mold!^?"));
186                        if(mob.location().okMessage(mob,msg))
187                        {
188                                mob.location().send(mob,msg);
189                                final MOB target = determineMonster(mob, material);
190                                beneficialAffect(mob,target,asLevel,0);
191                        }
192                }
193                else
194                        return beneficialWordsFizzle(mob,null,L("<S-NAME> chant(s), but nothing happens."));
195
196                // return whether it worked
197                return success;
198        }
199
200        public MOB determineMonster(MOB caster, int material)
201        {
202                final MOB victim=caster.getVictim();
203                final MOB newMOB=CMClass.getMOB("GenMOB");
204                final int level=20;
205                newMOB.basePhyStats().setLevel(level);
206                newMOB.basePhyStats().setAbility(25);
207                newMOB.baseCharStats().setMyRace(CMClass.getRace("Mold"));
208                final String name="a brown mold";
209                newMOB.setName(name);
210                newMOB.setDisplayText(L("@x1 looks scary!",name));
211                newMOB.setDescription("");
212                CMLib.factions().setAlignment(newMOB,Faction.Align.NEUTRAL);
213                final Ability A=CMClass.getAbility("Fighter_Rescue");
214                A.setProficiency(100);
215                newMOB.addAbility(A);
216                newMOB.setVictim(victim);
217                newMOB.basePhyStats().setAbility(newMOB.basePhyStats().ability()*2);
218                newMOB.basePhyStats().setSensesMask(newMOB.basePhyStats().sensesMask()|PhyStats.CAN_SEE_DARK);
219                newMOB.setLocation(caster.location());
220                newMOB.basePhyStats().setRejuv(PhyStats.NO_REJUV);
221                newMOB.basePhyStats().setDamage(25);
222                newMOB.basePhyStats().setAttackAdjustment(60);
223                newMOB.basePhyStats().setArmor(-super.getX1Level(caster));
224                newMOB.baseCharStats().setStat(CharStats.STAT_GENDER,'N');
225                newMOB.addNonUninvokableEffect(CMClass.getAbility("Prop_ModExperience"));
226                newMOB.setMiscText(newMOB.text());
227                newMOB.recoverCharStats();
228                newMOB.recoverPhyStats();
229                newMOB.recoverMaxState();
230                newMOB.resetToMaxState();
231                newMOB.bringToLife(caster.location(),true);
232                CMLib.beanCounter().clearZeroMoney(newMOB,null);
233                newMOB.setMoneyVariation(0);
234                newMOB.setStartRoom(null); // keep before postFollow for Conquest
235                CMLib.commands().postFollow(newMOB,caster,true);
236                if(newMOB.amFollowing()!=caster)
237                        caster.tell(L("@x1 seems unwilling to follow you.",newMOB.name()));
238                else
239                {
240                        if(newMOB.getVictim()!=victim)
241                                newMOB.setVictim(victim);
242                        newMOB.location().showOthers(newMOB,victim,CMMsg.MSG_OK_ACTION,L("<S-NAME> start(s) attacking <T-NAMESELF>!"));
243                }
244                return(newMOB);
245        }
