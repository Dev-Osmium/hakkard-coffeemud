036public class Chant_SummonLightning extends Chant
037{
038        @Override
039        public String ID()
040        {
041                return "Chant_SummonLightning";
042        }
043
044        @Override
045        public String name()
046        {
047                return renderedMundane?"lightning":"Summon Lightning";
048        }
049
050        @Override
051        public int abstractQuality()
052        {
053                return Ability.QUALITY_MALICIOUS;
054        }
055
056        @Override
057        public int maxRange()
058        {
059                return adjustedMaxInvokerRange(10);
060        }
061
062        @Override
063        protected int canAffectCode()
064        {
065                return 0;
066        }
067
068        @Override
069        protected int canTargetCode()
070        {
071                return CAN_MOBS;
072        }
073
074        @Override
075        public int classificationCode()
076        {
077                return Ability.ACODE_CHANT|Ability.DOMAIN_WEATHER_MASTERY;
078        }
079
080        @Override
081        public long flags()
082        {
083                return Ability.FLAG_WEATHERAFFECTING|Ability.FLAG_AIRBASED;
084        }
085
086        @Override
087        public int castingQuality(MOB mob, Physical target)
088        {
089                 if(mob!=null)
090                 {
091                         final Room R=mob.location();
092                         if(R!=null)
093                         {
094                                 if((R.domainType()&Room.INDOORS)>0)
095                                         return Ability.QUALITY_INDIFFERENT;
096                                 final Area A=R.getArea();
097                                 if(A.getClimateObj().weatherType(mob.location())!=Climate.WEATHER_THUNDERSTORM)
098                                         return Ability.QUALITY_INDIFFERENT;
099                         }
100                 }
101                 return super.castingQuality(mob,target);
102        }
103
104        @Override
105        public boolean invoke(MOB mob, List<String> commands, Physical givenTarget, boolean auto, int asLevel)
106        {
107                if(((mob.location().domainType()&Room.INDOORS)>0)&&(!auto))
108                {
109                        mob.tell(L("You must be outdoors for this chant to work."));
110                        return false;
111                }
112                if(mob.location().getArea().getClimateObj().weatherType(mob.location())!=Climate.WEATHER_THUNDERSTORM)
113                {
114                        mob.tell(L("This chant requires a thunderstorm!"));
115                        return false;
116                }
117                final MOB target=this.getTarget(mob,commands,givenTarget);
118                if(target==null)
119                        return false;
120
121                if(!super.invoke(mob,commands,givenTarget,auto,asLevel))
122                        return false;
123
124                final boolean success=proficiencyCheck(mob,0,auto);
125
126                if(success)
127                {
128                        final CMMsg msg=CMClass.getMsg(mob,target,this,verbalCastCode(mob,target,auto),L(auto?"^JA lightning bolt streaks out of the sky!^?":"^S<S-NAME> chant(s) to <T-NAMESELF>.  Suddenly a lightning bolt streaks from the sky!^?")+CMLib.protocol().msp("lightning.wav",40));
129                        final CMMsg msg2=CMClass.getMsg(mob,target,this,verbalCastMask(mob,target,auto)|CMMsg.TYP_ELECTRIC,null);
130                        if((mob.location().okMessage(mob,msg))&&((mob.location().okMessage(mob,msg2))))
131                        {
132                                mob.location().send(mob,msg);
133                                mob.location().send(mob,msg2);
134                                final int maxDie =  adjustedLevel( mob, asLevel ) + ( 2 * super.getX1Level( mob ) );
135                                int damage = CMLib.dice().roll(maxDie,8,maxDie);
136                                if((msg.value()>0)||(msg2.value()>0))
137                                        damage = (int)Math.round(CMath.div(damage,2.0));
138                                if(target.location()==mob.location())
139                                        CMLib.combat().postDamage(mob,target,this,damage,CMMsg.MASK_ALWAYS|CMMsg.TYP_ELECTRIC,Weapon.TYPE_STRIKING,L("The bolt <DAMAGE> <T-NAME>!"));
140                        }
141                }
142                else
143                        return maliciousFizzle(mob,target,L("<S-NAME> chant(s) at <T-NAMESELF>, but the magic fades."));
144
145                // return whether it worked
146                return success;
147        }
148}
