037public class Chant extends StdAbility
038{
039        @Override
040        public String ID()
041        {
042                return "Chant";
043        }
044
045        private final static String     localizedName   = CMLib.lang().L("a Druidic Chant");
046
047        @Override
048        public String name()
049        {
050                return localizedName;
051        }
052
053        @Override
054        public String displayText()
055        {
056                return "(" + name() + ")";
057        }
058
059        protected boolean               renderedMundane = false;
060
061        /** codes: -1=do nothing, 1=wind, 2=rain, 4=hot, 8=cold, 16=calm */
062        public final static int WEATHERQUE_NADA=0;
063        public final static int WEATHERQUE_WIND=1;
064        public final static int WEATHERQUE_RAIN=2;
065        public final static int WEATHERQUE_HOT=4;
066        public final static int WEATHERQUE_COLD=8;
067        public final static int WEATHERQUE_CALM=16;
068
069        @Override
070        protected int verbalCastCode(MOB mob, Physical target, boolean auto)
071        {
072                if(renderedMundane)
073                {
074                        int affectType=CMMsg.MSG_CAST_VERBAL_SPELL;
075                        affectType=CMMsg.MSG_NOISE|CMMsg.MASK_MOUTH;
076                        if(abstractQuality()==Ability.QUALITY_MALICIOUS)
077                                affectType=affectType|CMMsg.MASK_MALICIOUS;
078                        if(auto)
079                                affectType=affectType|CMMsg.MASK_ALWAYS;
080                        return affectType;
081                }
082                return super.verbalCastCode(mob,target,auto);
083        }
084        
085        private static final String[]   triggerStrings  = I(new String[] { "CHANT", "CH" });
086
087        @Override
088        public int abstractQuality()
089        {
090                return Ability.QUALITY_OK_SELF;
091        }
092
093        @Override
094        public String[] triggerStrings()
095        {
096                return triggerStrings;
097        }
098
099        @Override
100        protected int canAffectCode()
101        {
102                return Ability.CAN_MOBS;
103        }
104
105        @Override
106        protected int canTargetCode()
107        {
108                return Ability.CAN_MOBS;
109        }
110
111        @Override
112        public void setMiscText(String newText)
113        {
114                if(newText.equalsIgnoreCase("render mundane"))
115                        renderedMundane=true;
116                else
117                        super.setMiscText(newText);
118        }
119
120        @Override
121        public int classificationCode()
122        {
123                return renderedMundane ? Ability.ACODE_SKILL : Ability.ACODE_CHANT;
124        }
125
126        // codes: -1=do nothing, 1=wind, 2=rain, 4=hot, 8=cold, 16=calm
127        public int weatherQue(Room R)
128        {
129                if(R==null)
130                        return WEATHERQUE_NADA;
131                if((R.domainType()&Room.INDOORS)>0)
132                        return WEATHERQUE_NADA;
133                switch(R.getArea().getClimateObj().weatherType(R))
134                {
135                case Climate.WEATHER_BLIZZARD:
136                case Climate.WEATHER_THUNDERSTORM:
137                case Climate.WEATHER_HEAT_WAVE:
138                        return WEATHERQUE_NADA;
139                case Climate.WEATHER_CLEAR:
140                        return WEATHERQUE_WIND | WEATHERQUE_RAIN | WEATHERQUE_HOT | WEATHERQUE_COLD;
141                case Climate.WEATHER_CLOUDY:
142                        return WEATHERQUE_WIND | WEATHERQUE_RAIN;
143                case Climate.WEATHER_DROUGHT:
144                        return WEATHERQUE_RAIN | WEATHERQUE_COLD;
145                case Climate.WEATHER_DUSTSTORM:
146                        return WEATHERQUE_RAIN | WEATHERQUE_CALM | WEATHERQUE_COLD;
147                case Climate.WEATHER_HAIL:
148                        return WEATHERQUE_HOT | WEATHERQUE_CALM;
149                case Climate.WEATHER_RAIN:
150                        return WEATHERQUE_WIND | WEATHERQUE_RAIN;
151                case Climate.WEATHER_SLEET:
152                        return WEATHERQUE_HOT;
153                case Climate.WEATHER_SNOW:
154                        return WEATHERQUE_WIND;
155                case Climate.WEATHER_WINDY:
156                        return WEATHERQUE_RAIN;
157                case Climate.WEATHER_WINTER_COLD:
158                        return WEATHERQUE_RAIN;
159                default:
160                        return WEATHERQUE_CALM;
161                }
162        }
163        
164        protected static boolean chantAlignmentCheck(StdAbility A, MOB mob, boolean renderedMundane, boolean auto)
165        {
166                if((!auto)
167                &&(!mob.isMonster())
168                &&(!A.disregardsArmorCheck(mob))
169                &&(mob.isMine(A))
170                &&(!renderedMundane)
171                &&(CMLib.dice().rollPercentage()<50))
172                {
173                        if(!A.appropriateToMyFactions(mob))
174                        {
175                                mob.tell(A.L("Extreme emotions disrupt your chant."));
176                                return false;
177                        }
178                        else
179                        if(!CMLib.utensils().armorCheck(mob,CharClass.ARMOR_LEATHER))
180                        {
181                                mob.location().show(mob,null,CMMsg.MSG_OK_VISUAL,A.L("<S-NAME> watch(es) <S-HIS-HER> armor absorb <S-HIS-HER> magical energy!"));
182                                return false;
183                        }
184                }
185                return true;
186        }
187
188        @Override
189        public boolean invoke(MOB mob, List<String> commands, Physical givenTarget, boolean auto, int asLevel)
190        {
191                if(!super.invoke(mob,commands,givenTarget,auto,asLevel))
192                        return false;
193
194                if(!chantAlignmentCheck(this,mob,renderedMundane,auto))
195                        return false;
196                return true;
197        }
