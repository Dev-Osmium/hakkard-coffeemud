036public class Chant_Whirlpool extends Chant
037{
038        @Override
039        public String ID()
040        {
041                return "Chant_Whirlpool";
042        }
043
044        private final static String     localizedName   = CMLib.lang().L("Whirlpool");
045
046        @Override
047        public String name()
048        {
049                return localizedName;
050        }
051
052        @Override
053        protected int canAffectCode()
054        {
055                return Ability.CAN_ROOMS;
056        }
057
058        @Override
059        protected int canTargetCode()
060        {
061                return 0;
062        }
063
064        @Override
065        public int classificationCode()
066        {
067                return Ability.ACODE_CHANT | Ability.DOMAIN_WATERCONTROL;
068        }
069
070        @Override
071        public long flags()
072        {
073                return Ability.FLAG_TRANSPORTING;
074        }
075
076        @Override
077        public int abstractQuality()
078        {
079                return Ability.QUALITY_INDIFFERENT;
080        }
081
082        @Override
083        protected int overrideMana()
084        {
085                return Ability.COST_ALL - 99;
086        }
087        
088        private volatile Room theWhirlpool = null;
089
090        @Override
091        public void unInvoke()
092        {
093                Room R=null;
094                if(affected instanceof Room)
095                        R=(Room)affected;
096                super.unInvoke();
097                if((R!=null) && canBeUninvoked())
098                {
099                        if(theWhirlpool instanceof GridLocale)
100                        {
101                                ((GridLocale)theWhirlpool).clearGrid(R);
102                                theWhirlpool.getArea().destroy();
103                                theWhirlpool = null;
104                        }
105                }
106        }
107
108        public boolean canEnterTheWhirlpool(MOB M)
109        {
110                if(invoker()!=null)
111                        return invoker().mayIFight(M);
112                return true;
113        }
114        
115        public boolean canEnterTheWhirlpool(Rideable R)
116        {
117                for(final Enumeration<Rider> r2 = R.riders();r2.hasMoreElements();)
118                {
119                        Rider R2=r2.nextElement();
120                        if((R2 instanceof MOB)&&(!canEnterTheWhirlpool((MOB)R2)))
121                                return false;
122                }
123                return true;
124        }
125        
126        public boolean canEnterTheWhirlpool(BoardableShip S)
127        {
128                if(S instanceof PrivateProperty)
129                {
130                        if(invoker()!=null)
131                        {
132                                if(!CMLib.law().canAttackThisProperty(invoker(), (PrivateProperty)S))
133                                        return false;
134                                
135                        }
136                }
137                return true;
138        }
139        
140        public boolean canEverEnterThePool(MOB M)
141        {
142                if((invoker!=null)
143                &&((invoker==M)||(!invoker.mayIFight(M))))
144                        return false;
145                boolean enterThePool = false;
146                if((M.riding()!=null)&&(!CMLib.flags().isFlying(M.riding())))
147                {
148                        if(M.riding() instanceof BoardableShip)
149                                enterThePool = canEnterTheWhirlpool((BoardableShip)M.riding());
150                        else
151                        if(M.riding().rideBasis()==CMMsg.TYP_WATER)
152                                enterThePool = canEnterTheWhirlpool(M.riding());
153                }
154                if(canEnterTheWhirlpool(M))
155                {
156                        enterThePool = true;
157                }
158                return enterThePool;
159        }
160
161        public boolean canEverEnterThePool(Item I)
162        {
163                if(I instanceof BoardableShip)
164                        return canEnterTheWhirlpool((BoardableShip)I);
165                else
166                if((I instanceof Rideable)
167                &&(((Rideable)I).rideBasis()==CMMsg.TYP_WATER))
168                        return canEnterTheWhirlpool((Rideable)I);
169                if(I.container()!=null)
170                        return false;
171                return true;
172        }
173
174        @Override
175        public boolean tick(final Tickable ticking, final int tickID)
176        {
177                if(!super.tick(ticking, tickID))
178                        return false;
179                if(affected instanceof Room)
180                {
181                        final Room R=(Room)affected;
182                        for(final Enumeration<Item> r=R.items();r.hasMoreElements();)
183                        {
184                                final Item I=r.nextElement();
185                                if((I!=null)&&(canEverEnterThePool(I)))
186                                {
187                                        if(I instanceof BoardableShip)
188                                        {
189                                                final Area A=((BoardableShip)I).getShipArea();
190                                                if(A!=null)
191                                                {
192                                                        for(final Enumeration<Room> r2=A.getProperMap();r2.hasMoreElements();)
193                                                        {
194                                                                Room R2=r2.nextElement();
195                                                                if((R2!=null)&&((R2.domainType()&Room.INDOORS)==0))
196                                                                        R2.showHappens(CMMsg.MSG_OK_ACTION, L("@x1 is swept into a massive whirlpool!",I.Name()));
197                                                        }
198                                                }
199                                        }
200                                        R.showHappens(CMMsg.MSG_OK_ACTION, L("@x1 is swept into a massive whirlpool!",I.Name()));
201                                        final GridLocale grid = (GridLocale)theWhirlpool;
202                                        final Room targetRoom = grid.getGridChild(grid.xGridSize()-1, 0);
203                                        targetRoom.moveItemTo(I);
204                                }
205                        }
206                        for(final Enumeration<MOB> r=R.inhabitants();r.hasMoreElements();)
207                        {
208                                final MOB M=r.nextElement();
209                                if((M!=null)&&(canEverEnterThePool(M)))
210                                {
211                                        if(!M.isMonster())
212                                                M.tell(L("^XYou're swept into a terrible whirlpool!"));
213                                        final GridLocale grid = (GridLocale)theWhirlpool;
214                                        final Room targetRoom = grid.getGridChild(grid.xGridSize()-1, 0);
215                                        targetRoom.bringMobHere(M, false);
216                                }
217                        }
218                }
219                return true;
220        }
221        
222        @Override
223        public boolean okMessage(final Environmental myHost, final CMMsg msg)
224        {
225                if(!super.okMessage(myHost, msg))
226                        return false;
227                if((msg.target()==affected)
228                &&(msg.tool() instanceof Exit)
229                &&(msg.targetMinor()==CMMsg.TYP_ENTER)
230                &&(!CMLib.flags().isFlying(msg.source())))
231                {
232                        if(this.canEverEnterThePool(msg.source()))
233                        {
234                                final MOB M=msg.source();
235                                if(theWhirlpool instanceof GridLocale)
236                                {
237                                        if(!M.isMonster())
238                                                M.tell(L("^XYou've stumbled into a terrible whirlpool!"));
239                                        final GridLocale grid = (GridLocale)theWhirlpool;
240                                        final Room targetRoom = grid.getGridChild(grid.xGridSize()-1, 0);
241                                        msg.setTarget(targetRoom);
242                                }
243                        }
244                }
245                return true;
246        }
247        
248        @Override
249        public void executeMsg(final Environmental myHost, final CMMsg msg)
250        {
251                if((msg.target()==affected)
252                &&((msg.targetMinor()==CMMsg.TYP_LOOK)||(msg.targetMinor()==CMMsg.TYP_EXAMINE)))
253                {
254                        msg.addTrailerMsg(CMClass.getMsg(msg.source(),null,null,CMMsg.MSG_OK_VISUAL,L("A MASSIVE WHIRLPOOL IS HERE!!!"),CMMsg.NO_EFFECT,null,CMMsg.NO_EFFECT,null));
255                }
256        }
257        
258        @Override
259        public boolean invoke(MOB mob, List<String> commands, Physical givenTarget, boolean auto, int asLevel)
260        {
261                final Room target=mob.location();
262                if(!CMLib.flags().isWaterySurfaceRoom(target))
263                {
264                        mob.tell(L("This magic only works in water surfaces."));
265                        return false;
266                }
267
268                Room waterBelow = target.getRoomInDir(Directions.DOWN);
269                if(!CMLib.flags().isUnderWateryRoom(waterBelow))
270                {
271                        mob.tell(L("This magic only works above the watery deeps."));
272                        return false;
273                }
274                
275                if(target.fetchEffect(ID())!=null)
276                {
277                        mob.tell(L("There is already a whirlpool here! Geesh, how did you miss it?!"));
278                        return false;
279                }
280                
281                if(!super.invoke(mob,commands,givenTarget,auto,asLevel))
282                        return false;
283
284                // now see if it worked
285                final boolean success=proficiencyCheck(mob,0,auto);
286                if(success)
287                {
288                        final CMMsg msg=CMClass.getMsg(mob,target,this,verbalCastCode(mob,null,auto),auto?"":L("^S<S-NAME> chant(s) treacherously to the waters.^?"));
289                        if(target.okMessage(mob,msg))
290                        {
291                                target.send(mob,msg);
292                                target.showHappens(CMMsg.MSG_OK_ACTION, L("A MASSIVE WHIRLPOOL FORMS HERE!"));
293                                Chant_Whirlpool W=(Chant_Whirlpool)beneficialAffect(mob,target,asLevel,0);
294                                if(W!=null)
295                                {
296                                        W.theWhirlpool = CMClass.getLocale("Whirlpool");
297                                        W.theWhirlpool.setDisplayText(L("You are caught in a massive whirlpool"));
298                                        W.theWhirlpool.setDescription(L("The only way out appear to be to fight against the currents."));
299                                        final Area tempArea=CMClass.getAreaType("StdArea");
300                                        tempArea.setName("A Whirlpool");
301                                        W.theWhirlpool.setArea(tempArea);
302                                        for(int d=0;d<Directions.NUM_DIRECTIONS();d++)
303                                        {
304                                                if(d != Directions.DOWN)
305                                                {
306                                                        W.theWhirlpool.rawDoors()[d] = target.rawDoors()[d];
307                                                        W.theWhirlpool.setRawExit(d, target.getRawExit(d));
308                                                }
309                                        }
310                                        ((GridLocale)W.theWhirlpool).buildGrid();
311                                }
312                                target.recoverPhyStats();
313                        }
314                }
315                else
316                        return beneficialWordsFizzle(mob,null,L("<S-NAME> chant(s) treacherously to the waters, but nothing happens."));
317
318                // return whether it worked
319                return success;
320        }
