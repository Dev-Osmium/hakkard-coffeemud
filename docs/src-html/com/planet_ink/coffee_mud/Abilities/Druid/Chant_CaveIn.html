036public class Chant_CaveIn extends Chant
037{
038        @Override
039        public String ID()
040        {
041                return "Chant_CaveIn";
042        }
043
044        private final static String localizedName = CMLib.lang().L("Cave-In");
045
046        @Override
047        public String name()
048        {
049                return localizedName;
050        }
051
052        @Override
053        public int classificationCode()
054        {
055                return Ability.ACODE_CHANT|Ability.DOMAIN_ROCKCONTROL;
056        }
057
058        @Override
059        public int abstractQuality()
060        {
061                return Ability.QUALITY_MALICIOUS;
062        }
063
064        @Override
065        protected int canAffectCode()
066        {
067                return 0;
068        }
069
070        @Override
071        protected int canTargetCode()
072        {
073                return Ability.CAN_MOBS|CAN_EXITS;
074        }
075
076        public int amountRemaining=0;
077
078        @Override
079        public long flags()
080        {
081                return Ability.FLAG_PARALYZING;
082        }
083
084        @Override
085        public boolean okMessage(final Environmental host, final CMMsg msg)
086        {
087                if((affected instanceof Exit)
088                &&((msg.targetMinor()==CMMsg.TYP_ENTER)||(msg.targetMinor()==CMMsg.TYP_LEAVE)||(msg.targetMinor()==CMMsg.TYP_OPEN))
089                &&(msg.source().phyStats().height()>=0)
090                &&((msg.tool()==affected)||(msg.target()==affected)))
091                {
092                        msg.source().tell(L("This exit is blocked by rubble, and can not be moved through."));
093                        return false;
094                }
095                else
096                if((affected instanceof MOB)
097                &&(msg.amISource((MOB)affected)))
098                {
099                        final MOB mob=(MOB)affected;
100                        if(msg.sourceMinor()==CMMsg.TYP_STAND)
101                                return false;
102                        if((!msg.sourceMajor(CMMsg.MASK_ALWAYS))
103                        &&((msg.sourceMajor(CMMsg.MASK_HANDS))
104                        ||(msg.sourceMajor(CMMsg.MASK_MOVE))))
105                        {
106                                mob.location().show(mob,null,null,CMMsg.MSG_OK_ACTION,L("<S-NAME> struggle(s) to get out from under the rocks."));
107                                amountRemaining-=(mob.charStats().getStat(CharStats.STAT_STRENGTH)*4);
108                                if(amountRemaining<0)
109                                        unInvoke();
110                                else
111                                        return false;
112                        }
113                }
114                return super.okMessage(host,msg);
115        }
116
117        @Override
118        public void affectPhyStats(final Physical affected, final PhyStats affectableStats)
119        {
120                super.affectPhyStats(affected,affectableStats);
121                if(affected instanceof MOB)
122                {
123                        affectableStats.setDisposition(affectableStats.disposition()|PhyStats.IS_BOUND);
124                        affectableStats.setDisposition(affectableStats.disposition()|PhyStats.IS_SITTING);
125                }
126        }
127
128        @Override
129        public int castingQuality(MOB mob, Physical target)
130        {
131                if(mob!=null)
132                {
133                        if((mob.location().domainType()!=Room.DOMAIN_INDOORS_CAVE)
134                        &&((mob.location().getAtmosphere()&RawMaterial.MATERIAL_ROCK)==0))
135                                return Ability.QUALITY_INDIFFERENT;
136                        if(mob.isMonster())
137                                return Ability.QUALITY_INDIFFERENT;
138                }
139                return super.castingQuality(mob,target);
140        }
141
142        @Override
143        public boolean invoke(MOB mob, List<String> commands, Physical givenTarget, boolean auto, int asLevel)
144        {
145                Physical target=null;
146                if(mob.isMonster()&&(givenTarget instanceof MOB))
147                        target=mob.location();
148                else
149                if((commands.size()>0)&&(givenTarget==null))
150                {
151                        final int dir=CMLib.directions().getGoodDirectionCode(CMParms.combine(commands,0));
152                        if((dir>=0)&&(dir!=Directions.UP)&&(mob.location().getExitInDir(dir)!=null))
153                                target=mob.location().getExitInDir(dir);
154                }
155                if(target==null)
156                        target=getAnyTarget(mob,commands,givenTarget,Wearable.FILTER_ANY);
157                if(target==null)
158                        return false;
159                if((target instanceof Item)||(target instanceof Room))
160                {
161                        mob.tell(L("This chant can only target exits or creatures."));
162                        return false;
163                }
164                if((!auto)
165                &&(mob.location().domainType()!=Room.DOMAIN_INDOORS_CAVE)
166                &&((mob.location().getAtmosphere()&RawMaterial.MATERIAL_ROCK)==0))
167                {
168                        mob.tell(L("This chant only works in caves."));
169                        return false;
170                }
171
172                if(!super.invoke(mob,commands,givenTarget,auto,asLevel))
173                        return false;
174
175                boolean success=proficiencyCheck(mob,0,auto);
176                if(success)
177                {
178                        final CMMsg msg=CMClass.getMsg(mob,target,this,verbalCastCode(mob,target,auto)|CMMsg.MASK_MALICIOUS,auto?"":L("^S<S-NAME> chant(s) at <T-NAMESELF>!^?"));
179                        if(mob.location().okMessage(mob,msg))
180                        {
181                                mob.location().send(mob,msg);
182                                amountRemaining=200;
183                                if(target instanceof Exit)
184                                        mob.location().show(mob,target,CMMsg.MSG_OK_VISUAL,L("A cave-in causes rubble to fall, blocking <T-NAME>!"));
185                                else
186                                if(target instanceof MOB)
187                                {
188                                        mob.location().show(mob,target,CMMsg.MSG_OK_VISUAL,L("A cave-in drops rocks on <T-NAME>!"));
189                                        final int maxDie =  (adjustedLevel( mob, asLevel )+(2*super.getX1Level(mob))) / 2;
190                                        int damage = CMLib.dice().roll(maxDie,3,maxDie);
191                                        if(msg.value()>0)
192                                                damage = (int)Math.round(CMath.div(damage,1.5));
193
194                                        if(((MOB)target).location()==mob.location())
195                                                CMLib.combat().postDamage(mob,(MOB)target,this,damage,CMMsg.MASK_ALWAYS|CMMsg.TYP_CAST_SPELL,Weapon.TYPE_BASHING,L("The falling rubble <DAMAGE> <T-NAME>!"));
196                                }
197                                if(msg.value()<=0)
198                                        success=maliciousAffect(mob,target,asLevel,(target instanceof Exit)?0:10,0)!=null;
199                        }
200                }
201                else
202                        return maliciousFizzle(mob,target,L("<S-NAME> chant(s) at <T-NAMESELF>, but nothing happens."));
203
204                // return whether it worked
205                return success;
206        }
207}
