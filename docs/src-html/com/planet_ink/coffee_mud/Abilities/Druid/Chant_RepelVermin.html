036public class Chant_RepelVermin extends Chant
037{
038        @Override
039        public String ID()
040        {
041                return "Chant_RepelVermin";
042        }
043
044        private final static String localizedName = CMLib.lang().L("Repel Vermin");
045
046        @Override
047        public String name()
048        {
049                return localizedName;
050        }
051
052        private final static String localizedStaticDisplay = CMLib.lang().L("(Repel Vermin)");
053
054        @Override
055        public String displayText()
056        {
057                return localizedStaticDisplay;
058        }
059
060        @Override
061        public int abstractQuality()
062        {
063                return Ability.QUALITY_BENEFICIAL_OTHERS;
064        }
065
066        @Override
067        protected int canAffectCode()
068        {
069                return CAN_MOBS;
070        }
071
072        @Override
073        protected int canTargetCode()
074        {
075                return 0;
076        }
077
078        @Override
079        public int classificationCode()
080        {
081                return Ability.ACODE_CHANT|Ability.DOMAIN_ANIMALAFFINITY;
082        }
083
084        protected int pointsRemaining = 8;
085        
086        @Override
087        public void setAffectedOne(Physical affected)
088        {
089                if(super.affected != affected)
090                        pointsRemaining = 8 + (adjustedLevel(invoker(),0)/5);
091                super.setAffectedOne(affected);
092        }
093        
094        @Override
095        public boolean okMessage(final Environmental myHost, final CMMsg msg)
096        {
097                if(!super.okMessage(myHost, msg))
098                        return false;
099
100                if((msg.target() == affected) 
101                && msg.isTarget(CMMsg.MASK_MALICIOUS)
102                && CMLib.flags().isVermin(msg.source())
103                &&(affected instanceof MOB)
104                &&(pointsRemaining >= 0))
105                {
106                        final MOB mob=(MOB)affected;
107                        final MOB verminM=msg.source();
108                        final Room R=verminM.location();
109                        if((R!=null)&&(R==mob.location()))
110                        {
111                                if((msg.isSource(CMMsg.TYP_ADVANCE))
112                                && (--pointsRemaining >=0))
113                                {
114                                        R.show(verminM, affected, CMMsg.MSG_OK_ACTION, L("<S-NAME> struggle(s) against <T-YOUPOSS> anti-vermin field."));
115                                        return false;
116                                }
117                                else
118                                if(verminM.getVictim() == null)
119                                {
120                                        verminM.setVictim(mob);
121                                        if(mob.getVictim()==verminM)
122                                        {
123                                                if(mob.rangeToTarget() > 0)
124                                                        verminM.setRangeToTarget(mob.rangeToTarget());
125                                                else
126                                                {
127                                                        verminM.setRangeToTarget(R.maxRange());
128                                                        mob.setRangeToTarget(R.maxRange());
129                                                }
130                                        }
131                                        else
132                                                verminM.setRangeToTarget(R.maxRange());
133                                        if((--pointsRemaining) < 0)
134                                        {
135                                                unInvoke();
136                                                return true;
137                                        }
138                                        R.show(verminM, affected, CMMsg.MSG_OK_ACTION, L("<S-NAME> <S-IS-ARE> repelled by <T-YOUPOSS> anti-vermin field."));
139                                        return false;
140                                }
141                                else
142                                if((verminM.getVictim() == affected) && (verminM.rangeToTarget() <= 0))
143                                {
144                                        verminM.setRangeToTarget(R.maxRange());
145                                        if(mob.getVictim()==verminM)
146                                                mob.setRangeToTarget(R.maxRange());
147                                        if((--pointsRemaining) < 0)
148                                        {
149                                                unInvoke();
150                                                return true;
151                                        }
152                                        R.show(verminM, affected, CMMsg.MSG_OK_ACTION, L("<S-NAME> <S-IS-ARE> repelled by <T-YOUPOSS> anti-vermin field."));
153                                        return false;
154                                }
155                                else
156                                if((mob.getVictim()==verminM)&&(mob.rangeToTarget() <= 0))
157                                {
158                                        mob.setRangeToTarget(R.maxRange());
159                                        if((--pointsRemaining) < 0)
160                                        {
161                                                unInvoke();
162                                                return true;
163                                        }
164                                        R.show(verminM, affected, CMMsg.MSG_OK_ACTION, L("<S-NAME> <S-IS-ARE> repelled by <T-YOUPOSS> anti-vermin field."));
165                                        return false;
166                                }
167                        }
168                }
169                else
170                if(msg.isSource(CMMsg.TYP_ADVANCE)
171                &&(msg.source() == affected)
172                && (msg.source().getVictim()==msg.target())
173                && (CMLib.flags().isVermin((MOB)msg.target()))
174                && (pointsRemaining >=0)
175                && (msg.source().rangeToTarget() == 1))
176                {
177                        final MOB verminM=msg.source().getVictim();
178                        if(verminM != null)
179                        {
180                                final Room R=verminM.location();
181                                if(R!=null)
182                                {
183                                        final CMMsg msg2=CMClass.getMsg(verminM,msg.source(),CMMsg.MSG_RETREAT,L("<S-NAME> <S-IS-ARE> pushed back by <T-YOUPOSS> anti-vermin field."));
184                                        if(R.okMessage(verminM,msg2))
185                                                R.send(verminM,msg2);
186                                }
187                        }
188                }
189                else
190                if((msg.target() == affected) 
191                && (msg.isTarget(CMMsg.MASK_MALICIOUS)||msg.isTarget(CMMsg.TYP_DAMAGE))
192                &&(affected instanceof MOB)
193                &&(pointsRemaining >= 0)
194                &&(msg.tool() instanceof Ability)
195                &&(msg.sourceMinor()!=CMMsg.TYP_TEACH)
196                &&(msg.tool().ID().indexOf("Insect")>=0))
197                {
198                        final Room R=((MOB)affected).location();
199                        if(R!=null)
200                        {
201                                R.show(msg.source(),affected,null,CMMsg.MSG_OK_VISUAL,L("The insect magic from <S-NAME> is repelled by <T-YOUPOSS> anti-vermin field!"));
202                                return false;
203                        }
204                        if((--pointsRemaining) < 0)
205                        {
206                                unInvoke();
207                                return true;
208                        }
209                }
210                if(pointsRemaining < 0)
211                {
212                        unInvoke();
213                        return true;
214                }
215                return true;
216        }
217
218        @Override
219        public void unInvoke()
220        {
221                final Physical affected=super.affected;
222                super.unInvoke();
223                if(canBeUninvoked())
224                {
225                        if(affected instanceof MOB)
226                                ((MOB)affected).tell(L("Your anti-vermin field fades."));
227                }
228        }
229
230        @Override
231        public boolean invoke(MOB mob, List<String> commands, Physical givenTarget, boolean auto, int asLevel)
232        {
233                final MOB target=getTarget(mob,commands,givenTarget);
234                if(target==null)
235                        return false;
236                
237                if(!super.invoke(mob,commands,givenTarget,auto,asLevel))
238                        return false;
239
240                final boolean success=proficiencyCheck(mob,0,auto);
241
242                if(success)
243                {
244
245                        final CMMsg msg = CMClass.getMsg(mob, target, this,verbalCastCode(mob,target,auto),auto?L("An anti-vermin field surrounds <T-NAME>!"):L("^S<S-NAME> chant(s) for an anti-vermin field around <T-NAMESELF>!^?"));
246                        if(mob.location().okMessage(mob,msg))
247                        {
248                                mob.location().send(mob,msg);
249                                beneficialAffect(mob,target,asLevel,0);
250                        }
251                }
252                else
253                        return beneficialWordsFizzle(mob,target,L("<S-NAME> chant(s) for a field at <T-NAMESELF>, but the magic fizzles."));
254
255                // return whether it worked
256                return success;
257        }
