037public class Chant_FindDriftwood extends Chant_FindPlant
038{
039        @Override
040        public String ID()
041        {
042                return "Chant_FindDriftwood";
043        }
044
045        private final static String     localizedName   = CMLib.lang().L("Find Driftwood");
046
047        @Override
048        public String name()
049        {
050                return localizedName;
051        }
052
053        @Override
054        public int classificationCode()
055        {
056                return Ability.ACODE_CHANT | Ability.DOMAIN_NATURELORE;
057        }
058
059        @Override
060        public String displayText()
061        {
062                return L("(Finding Diftwood)");
063        }
064
065        @Override
066        public long flags()
067        {
068                return Ability.FLAG_TRACKING;
069        }
070
071        @Override
072        public int abstractQuality()
073        {
074                return Ability.QUALITY_OK_SELF;
075        }
076
077        private final int[]     myMats  = { RawMaterial.MATERIAL_WOODEN };
078
079        private Item theDriftwood = null;
080        private Room theDriftroom = null;
081        
082        @Override
083        protected int[] okMaterials()
084        {
085                return myMats;
086        }
087
088        @Override
089        protected int[] okResources()
090        {
091                return null;
092        }
093
094        public Chant_FindDriftwood()
095        {
096                super();
097
098                lookingFor = "driftwood";
099        }
100        
101        @Override
102        public String itsHere(MOB mob, Room R)
103        {
104                if(R==null)
105                        return "";
106                if((theDriftwood != null)
107                &&(theDriftwood.owner() instanceof Room)
108                &&(CMLib.map().roomLocation(theDriftwood) == R))
109                        return "There seems to be "+lookingFor+" around here.\n\r";
110                return "";
111        }
112
113        @Override
114        public void executeMsg(final Environmental myHost, final CMMsg msg)
115        {
116                super.executeMsg(myHost, msg);
117                if((msg.target() == theDriftwood)
118                &&(msg.targetMinor()==CMMsg.TYP_GET))
119                {
120                        msg.addTrailerRunnable(new Runnable()
121                        {
122                                @Override
123                                public void run()
124                                {
125                                        unInvoke();
126                                }
127                        });
128                }
129        }
130        
131        @Override
132        protected boolean findWhatImLookingFor(MOB mob, String s)
133        {
134                TrackingLibrary.TrackingFlags flags = getTrackingFlags();
135                int limit = 50 - (super.getXLEVELLevel(mob) + super.getXMAXRANGELevel(mob));
136                final List<Room> checkSet=CMLib.tracking().getRadiantRooms(mob.location(),flags,limit);
137                if((checkSet == null) || (checkSet.size() < limit))
138                {
139                        mob.tell(L("You don't sense any driftwood around here.  Perhaps no wrecks have occurred?"));
140                        return false;
141                }
142                final int[] choices = new int[]{
143                        RawMaterial.RESOURCE_OAK,
144                        RawMaterial.RESOURCE_WOOD,
145                        RawMaterial.RESOURCE_PINE,
146                        RawMaterial.RESOURCE_BALSA,
147                        RawMaterial.RESOURCE_MAPLE,
148                        RawMaterial.RESOURCE_REDWOOD,
149                        RawMaterial.RESOURCE_HICKORY,
150                        RawMaterial.RESOURCE_IRONWOOD,
151                        RawMaterial.RESOURCE_YEW,
152                        RawMaterial.RESOURCE_TEAK,
153                        RawMaterial.RESOURCE_CEDAR,
154                        RawMaterial.RESOURCE_ELM,
155                        RawMaterial.RESOURCE_CHERRYWOOD,
156                        RawMaterial.RESOURCE_BEECHWOOD,
157                        RawMaterial.RESOURCE_WILLOW,
158                        RawMaterial.RESOURCE_SYCAMORE,
159                        RawMaterial.RESOURCE_SPRUCE,
160                        RawMaterial.RESOURCE_MESQUITE,
161                        RawMaterial.RESOURCE_BAMBOO,
162                        RawMaterial.RESOURCE_REED,
163                };
164                this.whatImLookingFor = choices[CMLib.dice().roll(1, choices.length, -1)];
165                limit = 150 - (2*super.getXLEVELLevel(mob));
166                if(checkSet.size() > limit)
167                        theDriftroom=checkSet.get(limit + CMLib.dice().roll(1,checkSet.size()-limit,-1));
168                else
169                        theDriftroom=checkSet.get(checkSet.size()-1);
170                theDriftwood=CMLib.materials().makeItemResource(this.whatImLookingFor);
171                final int amount=CMLib.dice().roll(1,4,3);
172                theDriftwood.basePhyStats().setWeight(amount);
173                theDriftwood.phyStats().setWeight(amount);
174                CMLib.materials().adjustResourceName(theDriftwood);
175                theDriftroom.addItem(theDriftwood, Expire.Player_Drop);
176                theTrail = null;
177                return true;
178        }
179        
180        @Override
181        protected List<Room> makeTheTrail(MOB mob, MOB target, Room mobRoom)
182        {
183                if(theDriftroom == null)
184                {
185                        theTrail = null;
186                        theDriftroom = null;
187                        if(theDriftwood != null)
188                                theDriftwood.destroy();
189                        theDriftwood = null;
190                        return null;
191                }
192                                
193                TrackingLibrary.TrackingFlags flags = getTrackingFlags();
194                List<Room> rooms=new XVector<Room>(theDriftroom);
195                int limit = 50 - (super.getXLEVELLevel(mob));
196                if(rooms.size()>0)
197                        theTrail=CMLib.tracking().findTrailToAnyRoom(mobRoom,rooms,flags,limit);
198                return theTrail;
199        }
200        
201        @Override
202        public void unInvoke()
203        {
204                Physical affected = this.affected;
205                super.unInvoke();
206                if(theDriftwood != null)
207                {
208                        if((theDriftwood.owner() instanceof Room)
209                        &&(CMLib.map().roomLocation(affected) != CMLib.map().roomLocation(theDriftwood)))
210                        {
211                                theDriftwood.destroy();
212                        }
213                }
214                theDriftwood = null;
215                theDriftroom = null;
216                theTrail = null;
217        }
218
219        @Override
220        protected TrackingLibrary.TrackingFlags getTrackingFlags()
221        {
222                TrackingLibrary.TrackingFlags flags;
223                flags = CMLib.tracking().newFlags().plus(TrackingLibrary.TrackingFlag.WATERSURFACEONLY);
224                return flags;
225        }
226        
227        @Override
228        public boolean invoke(MOB mob, List<String> commands, Physical givenTarget, boolean auto, int asLevel)
229        {
230                if(commands==null)
231                        commands=new ArrayList<String>(1);
232                if(commands.size()==0)
233                        commands.add("driftwood");
234                final Room R=mob.location();
235                if(R==null)
236                        return false;
237                if(!CMLib.flags().isWaterySurfaceRoom(R))
238                {
239                        mob.tell(L("You must be on the surface of the water to find driftwood."));
240                        return false;
241                }
242                return super.invoke(mob, commands, givenTarget, auto, asLevel);
243        }
