096        @Override

098        {
099                final MOB mob=(MOB)affected;
100                super.unInvoke();

102                {
103                        if(mob.amDead())
104                                mob.setLocation(null);

109        @Override
110        public void executeMsg(final Environmental myHost, final CMMsg msg)
111        {
112                super.executeMsg(myHost,msg);
113                if((affected!=null)
114                &&(affected instanceof MOB)
115                &&(msg.amISource((MOB)affected)
116                        ||msg.amISource(((MOB)affected).amFollowing())
117                        ||(msg.source()==invoker()))
118                &&(msg.sourceMinor()==CMMsg.TYP_QUIT))
119                {
120                        unInvoke();
121                        if(msg.source().playerStats()!=null)
122                                msg.source().playerStats().setLastUpdated(0);
123                }
124        }

126        public Vector<Integer> outdoorChoices(final Room R)
127        {
128                final Vector<Integer> choices=new Vector<Integer>();
129                if(R==null)
130                        return choices;
131                for(int d=Directions.NUM_DIRECTIONS()-1;d>=0;d--)
132                {
133                        final Room room=R.getRoomInDir(d);
134                        final Exit exit=R.getExitInDir(d);
135                        final Exit opExit=R.getReverseExit(d);
136                        if((room!=null)
137                        &&((room.domainType()&Room.INDOORS)==0)
138                        &&(room.domainType()!=Room.DOMAIN_OUTDOORS_AIR)
139                        &&((exit!=null)&&(exit.isOpen()))
140                        &&(opExit!=null)&&(opExit.isOpen()))
141                                choices.addElement(Integer.valueOf(d));
142                }
143                return choices;
144        }

146        @Override
147        public int castingQuality(final MOB mob, final Physical target)
148        {
149                if(mob!=null)
150                {
151                        final Room R=mob.location();
152                        if(R!=null)
153                        {
154                                if((R.domainType()&Room.INDOORS)>0)
155                                        return Ability.QUALITY_INDIFFERENT;
156                                final Vector<Integer> choices=outdoorChoices(mob.location());
157                                if(choices.size()==0)
158                                        return Ability.QUALITY_INDIFFERENT;
159                        }
160                }
161                return super.castingQuality(mob,target);
162        }

164        @Override
165        public boolean invoke(final MOB mob, final List<String> commands, final Physical givenTarget, final boolean auto, final int asLevel)
166        {
167                if((mob.location().domainType()&Room.INDOORS)>0)
168                {
169                        mob.tell(L("You must be outdoors for this chant to work."));
170                        return false;
171                }
172                final Vector<Integer> choices=outdoorChoices(mob.location());
173                int fromDir=-1;
174                if(choices.size()==0)
175                {
176                        mob.tell(L("You must be further outdoors to summon an animal."));
177                        return false;
178                }
179                fromDir=choices.elementAt(CMLib.dice().roll(1,choices.size(),-1)).intValue();
180                final Room newRoom=mob.location().getRoomInDir(fromDir);
181                final int opDir=Directions.getOpDirectionCode(fromDir);
182
183                if(!super.invoke(mob,commands,givenTarget,auto,asLevel))
184                        return false;
185
186                final boolean success=proficiencyCheck(mob,0,auto);
187
188                if(success)
189                {
190                        invoker=mob;
191                        final CMMsg msg=CMClass.getMsg(mob,null,this,verbalCastCode(mob,null,auto),auto?"":L("^S<S-NAME> chant(s) and summon(s) a companion from the Java Plane.^?"));
192                        final Room room=mob.location();
193                        if(room.okMessage(mob,msg))
194                        {
195                                room.send(mob,msg);
196                                final MOB target = determineMonster(mob, adjustedLevel(mob,asLevel));
197                                target.bringToLife(newRoom,true);
198                                CMLib.beanCounter().clearZeroMoney(target,null);
199                                target.setMoneyVariation(0);
200                                newRoom.showOthers(target,null,CMMsg.MSG_OK_ACTION,L("<S-NAME> appears!"));
201                                newRoom.recoverRoomStats();
202                                target.setStartRoom(null);
203                                if(target.isInCombat())
204                                        target.makePeace(true);
205                                CMLib.tracking().walk(target,opDir,false,false);
206                                if(target.location()==room)
207                                {
208                                        if(target.isInCombat())
209                                                target.makePeace(true);
210                                        CMLib.commands().postFollow(target,mob,true);
211                                        beneficialAffect(mob,target,asLevel,0);
212                                        if(target.amFollowing()!=mob)
213                                                mob.tell(L("@x1 seems unwilling to follow you.",target.name(mob)));
214                                }
215                                else
216                                {
217                                        if(target.amDead())
218                                                target.setLocation(null);
219                                        target.destroy();
220                                }
221                                invoker=mob;
222                        }
223                        else
224                                return beneficialWordsFizzle(mob,null,L("<S-NAME> chant(s) and summon(s), but nothing happens."));
225                }
226                else
227                        return beneficialWordsFizzle(mob,null,L("<S-NAME> chant(s) and summon(s), but nothing happens."));
228
229                // return whether it worked
230                return success;
231        }
