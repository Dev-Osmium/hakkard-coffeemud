080        @Override
081        public void unInvoke()
082        {
083                final Room plantsR=plantsLocationR;
084                final Item plantsI=littlePlantsI;
085                if(plantsR==null)
086                        return;
087                if(plantsI==null)
088                        return;
089                if(canBeUninvoked())
090                        plantsR.showHappens(CMMsg.MSG_OK_VISUAL,L("@x1 wither@x2 away.",plantsI.name(),(plantsI.name().startsWith("s")?"":"s")));
091                super.unInvoke();
092                if(canBeUninvoked())
093                {
094                        this.littlePlantsI=null;
095                        plantsI.destroy();
096                        plantsR.recoverRoomStats();
097                        this.plantsLocationR=null;
098                }
099        }

118        public Item buildPlant(MOB mob, Room room)
119        {
120                final Item newItem=CMClass.getItem("GenItem");
121                newItem.setMaterial(RawMaterial.RESOURCE_GREENS);
122                switch(CMLib.dice().roll(1,5,0))
123                {
124                case 1:
125                        newItem.setName(L("some happy flowers"));
126                        newItem.setDisplayText(L("some happy flowers are growing here."));
127                        newItem.setDescription(L("Happy flowers with little red and yellow blooms."));
128                        break;
129                case 2:
130                        newItem.setName(L("some happy weeds"));
131                        newItem.setDisplayText(L("some happy weeds are growing here."));
132                        newItem.setDescription(L("Long stalked little plants with tiny bulbs on top."));
133                        break;
134                case 3:
135                        newItem.setName(L("a pretty fern"));
136                        newItem.setDisplayText(L("a pretty fern is growing here."));
137                        newItem.setDescription(L("Like a tiny bush, this dark green plant is lovely."));
138                        break;
139                case 4:
140                        newItem.setName(L("a patch of sunflowers"));
141                        newItem.setDisplayText(L("a patch of sunflowers is growing here."));
142                        newItem.setDescription(L("Happy flowers with little yellow blooms."));
143                        break;
144                case 5:
145                        newItem.setName(L("a patch of bluebonnets"));
146                        newItem.setDisplayText(L("a patch of bluebonnets is growing here."));
147                        newItem.setDescription(L("Happy flowers with little blue and purple blooms."));
148                        break;
149                }
150                final Chant_SummonPlants newChant=new Chant_SummonPlants();
151                newItem.basePhyStats().setLevel(10+(10*newChant.getX1Level(mob)));
152                newItem.basePhyStats().setWeight(1);
153                newItem.setSecretIdentity(mob.Name());
154                newItem.setMiscText(newItem.text());
155                room.addItem(newItem);
156                Druid_MyPlants.addNewPlant(mob, newItem);
157                newItem.setExpirationDate(0);
158                room.showHappens(CMMsg.MSG_OK_ACTION,CMLib.lang().L("Suddenly, @x1 sprout(s) up here.",newItem.name()));
159                newChant.plantsLocationR=room;
160                newChant.littlePlantsI=newItem;
161                if(CMLib.law().doesOwnThisLand(mob,room))
162                {
163                        newChant.setInvoker(mob);
164                        newChant.setMiscText(mob.Name());
165                        newItem.addNonUninvokableEffect(newChant);
166                }
167                else
168                        newChant.beneficialAffect(mob,newItem,0,(newChant.adjustedLevel(mob,0)*240)+450);
169                room.recoverPhyStats();
170                return newItem;
171        }

173        public Item buildMyThing(MOB mob, Room room)
174        {
175                final Area A=room.getArea();
176                final boolean bonusWorthy=
177                          (Druid_MyPlants.myPlant(room,mob,0)==null)
178                        &&((room.getGridParent()==null)
179                                ||(CMLib.flags().matchedAffects(mob,room.getGridParent(),-1,Ability.ACODE_CHANT,-1).size()==0));
180                
181                final List<Room> V=Druid_MyPlants.myAreaPlantRooms(mob,room.getArea());
182                int pct=0;
183                if(A.getAreaIStats()[Area.Stats.VISITABLE_ROOMS.ordinal()]>10)
184                        pct=(int)Math.round(100.0*CMath.div(V.size(),A.getAreaIStats()[Area.Stats.VISITABLE_ROOMS.ordinal()]));
185                final Item I=buildMyPlant(mob,room);
186                if((I!=null)
187                &&((mob.charStats().getCurrentClass().baseClass().equalsIgnoreCase("Druid"))||(CMSecurity.isASysOp(mob))))
188                {
189                        if(!CMLib.law().isACity(A))
190                        {
191                                if(pct>0)
192                                {
193                                        final int newPct=(int)Math.round(100.0*CMath.div(V.size(),A.getAreaIStats()[Area.Stats.VISITABLE_ROOMS.ordinal()]));
194                                        if((newPct>=50)&&(A.fetchEffect("Chant_DruidicConnection")==null))
195                                        {
196                                                final Ability A2=CMClass.getAbility("Chant_DruidicConnection");
197                                                if(A2!=null)
198                                                        A2.invoke(mob,A,true,0);
199                                        }
200                                }
201                        }
202                        else
203                        if((bonusWorthy)&&(!mob.isMonster()))
204                        {
205                                long[] num=plantBonuses.get(mob.Name()+"/"+room.getArea().Name());
206                                if((num==null)||(System.currentTimeMillis()-num[1]>(room.getArea().getTimeObj().getDaysInMonth()*room.getArea().getTimeObj().getHoursInDay()*CMProps.getMillisPerMudHour())))
207                                {
208                                        num=new long[2];
209                                        plantBonuses.remove(mob.Name()+"/"+room.getArea().Name());
210                                        plantBonuses.put(mob.Name()+"/"+room.getArea().Name(),num);
211                                        num[1]=System.currentTimeMillis();
212                                }
213                                if(V.size()>=num[0])
214                                {
215                                        num[0]++;
216                                        if(num[0]<19)
217                                        {
218                                                mob.tell(L("You have made this city greener."));
219                                                CMLib.leveler().postExperience(mob,null,null,(int)num[0],false);
220                                        }
221                                }
222                        }
223                }
224                return I;
225        }

227        protected Item buildMyPlant(MOB mob, Room room)
228        {
229                return buildPlant(mob, room);
230        }

232        public boolean rightPlace(MOB mob,boolean auto)
233        {
234                final Room R=mob.location();
235                if(R==null)
236                        return false;
237                
238                if((!auto)&&(R.domainType()&Room.INDOORS)>0)
239                {
240                        mob.tell(L("You must be outdoors for this chant to work."));
241                        return false;
242                }
243                
244                if((R.domainType()==Room.DOMAIN_OUTDOORS_CITY)
245                   ||(R.domainType()==Room.DOMAIN_OUTDOORS_SPACEPORT)
246                   ||(R.domainType()==Room.DOMAIN_OUTDOORS_UNDERWATER)
247                   ||(R.domainType()==Room.DOMAIN_OUTDOORS_AIR)
248                   ||(R.domainType()==Room.DOMAIN_OUTDOORS_WATERSURFACE))
249                {
250                        mob.tell(L("This magic will not work here."));
251                        return false;
252                }
253                return true;
254        }

256        @Override
257        public int castingQuality(MOB mob, Physical target)
258        {
259                if(mob!=null)
260                {
261                        if(!rightPlace(mob,false))
262                                return Ability.QUALITY_INDIFFERENT;
263                        final Item myPlant=Druid_MyPlants.myPlant(mob.location(),mob,0);
264                        if(myPlant==null)
265                                return super.castingQuality(mob, target,Ability.QUALITY_BENEFICIAL_SELF);
266                }
267                return super.castingQuality(mob,target);
268        }
269
270        @Override
271        public boolean invoke(MOB mob, List<String> commands, Physical givenTarget, boolean auto, int asLevel)
272        {
273                if(!rightPlace(mob,auto))
274                        return false;
275
276                if(!super.invoke(mob,commands,givenTarget,auto,asLevel))
277                        return false;
278
279                // now see if it worked
280                final boolean success=proficiencyCheck(mob,0,auto);
281                if(success)
282                {
283                        final CMMsg msg=CMClass.getMsg(mob,null,this,verbalCastCode(mob,null,auto),auto?"":L("^S<S-NAME> chant(s) to the ground.^?"));
284                        if(mob.location().okMessage(mob,msg))
285                        {
286                                mob.location().send(mob,msg);
287                                buildMyThing(mob,mob.location());
288                        }
289                }
290                else
291                        return beneficialWordsFizzle(mob,null,L("<S-NAME> chant(s) to the ground, but nothing happens."));
292
293                // return whether it worked
294                return success;
295        }
