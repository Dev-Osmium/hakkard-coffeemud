131        {

147        @Override
148        public void executeMsg(final Environmental host, final CMMsg msg)
149        {
150                super.executeMsg(host,msg);
151                if((affected==invoker)
152                &&(msg.amISource(invoker))
153                &&(!unInvoked))
154                {
155                        if(((msg.sourceMinor()==CMMsg.TYP_SPEAK)
156                                ||((msg.sourceMinor()==CMMsg.TYP_CAST_SPELL)&&(CMath.bset(msg.sourceCode(), CMMsg.MASK_SOUND))))
157                        &&(!(msg.tool() instanceof Song))
158                        &&(!(msg.tool() instanceof Play))
159                        &&(!(msg.tool() instanceof Dance))
160                        &&(!msg.sourceMajor(CMMsg.MASK_CHANNEL)))
161                        {
162                                if(msg.source().location()!=null)
163                                        msg.source().location().show(msg.source(),null,CMMsg.MSG_NOISE,L("<S-NAME> stop(s) singing."));
164                                unInvoke();
165                        }
166                        else
167                        if((msg.target() instanceof Armor)
168                        &&(msg.targetMinor()==CMMsg.TYP_WEAR)
169                        &&(CMath.bset(((Armor)msg.target()).rawProperLocationBitmap(),Wearable.WORN_MOUTH)))
170                        {
171                                if(msg.source().location()!=null)
172                                        msg.source().location().show(msg.source(),null,CMMsg.MSG_NOISE,L("<S-NAME> stop(s) singing."));
173                                unInvoke();
174                        }
175                }
176        }

193        @SuppressWarnings("unchecked")
194        protected List<String> getLyrics()
195        {
196                Map<String,List<String>> lyricMap=(Map<String,List<String>>)Resources.getResource("SYSTEM_SONG_LYRICS");
197                if(lyricMap==null)
198                {
199                        lyricMap=new TreeMap<String,List<String>>();
200                        List<String> lines=Resources.getFileLineVector(new CMFile(Resources.buildResourcePath("skills/songlyrics.txt"),null).text());
201                        if(lines.size()>0)
202                        {
203                                List<String> current=null;
204                                for(String line : lines)
205                                {
206                                        line=line.trim();
207                                        if(line.startsWith("[")&&(line.endsWith("]")))
208                                        {
209                                                String ID=line.substring(1,line.length()-1);
210                                                Ability A=CMClass.getAbility(ID);
211                                                if(A==null)
212                                                        A=CMClass.getAbility("Song_"+ID);
213                                                if(A!=null)
214                                                {
215                                                        current=new ArrayList<String>();
216                                                        lyricMap.put(A.ID(),current);
217                                                }
218                                                else
219                                                {
220                                                        Log.errOut("Song","Unknown song class: "+ID);
221                                                        current=null;
222                                                }
223                                        }
224                                        else
225                                        if((current!=null)&&(line.length()>0))
226                                                current.add(line);
227                                }
228                        }
229                }
230                return lyricMap.get(ID());
231        }

233        @Override
234        public boolean tick(final Tickable ticking, final int tickID)
235        {
236                if((!super.tick(ticking,tickID))||(!(affected instanceof MOB)))
237                        return false;
238
239                if(skipSimpleStandardSongTickToo())
240                        return true;
241
242                final MOB mob=(MOB)affected;
243                if((affected==invoker())&&(invoker()!=null)&&(invoker().location()!=originRoom))
244                {
245                        final List<Room> V=getInvokerScopeRoomSet(null);
246                        commonRoomSet.clear();
247                        commonRoomSet.addAll(V);
248                        originRoom=invoker().location();
249                }
250                else
251                if((abstractQuality()==Ability.QUALITY_MALICIOUS)
252                &&(!maliciousButNotAggressiveFlag())
253                &&(!mob.amDead())
254                &&(mob.isMonster())
255                &&(!mob.isInCombat())
256                &&(mob.amFollowing()==null)
257                &&((!(mob instanceof Rideable))||(((Rideable)mob).numRiders()==0))
258                &&(!CMLib.flags().isATrackingMonster(mob))
259                &&(CMLib.flags().isAliveAwakeMobile(mob,true)))
260                {
261                        if((mob.location()!=originRoom)
262                        &&(CMLib.flags().isMobile(mob)))
263                        {
264                                final int dir=this.getCorrectDirToOriginRoom(mob.location(),commonRoomSet.indexOf(mob.location()));
265                                if(dir>=0)
266                                        CMLib.tracking().walk(mob,dir,false,false);
267                        }
268                        else
269                        if((mob.location().isInhabitant(invoker()))
270                        &&(CMLib.flags().canBeSeenBy(invoker(),mob)))
271                                CMLib.combat().postAttack(mob,invoker(),mob.fetchWieldedItem());
272                }
273
274                if((invoker==null)
275                ||(invoker.fetchEffect(ID())==null)
276                ||(commonRoomSet==null))
277                        return unsingMe(mob,null);
278                if(!commonRoomSet.contains(mob.location()))
279                {
280                        final List<Room> V=getInvokerScopeRoomSet(null);
281                        commonRoomSet.clear();
282                        commonRoomSet.addAll(V);
283                        if(!commonRoomSet.contains(mob.location()))
284                                return unsingMe(mob,null);
285                }
286                
287                if(invoker==affected)
288                {
289                        final List<String> lyrics=this.getLyrics();
290                        final MOB invoker=this.invoker;
291                        if((lyrics!=null) && (invoker==affected))
292                        {
293                                final String line = lyrics.get((int)((System.currentTimeMillis()/CMProps.getTickMillis()) % (lyrics.size())));
294                                final Room R=invoker.location();
295                                if(R!=null)
296                                {
297                                        R.show(invoker, null, this, CMMsg.MSG_SPEAK, L("<S-NAME> sing(s) '@x1'",line));
298                                }
299                        }
300                }
301
302                if(skipStandardSongTick())
303                        return true;
304
305                if((invoker==null)
306                ||(!CMLib.flags().isAliveAwakeMobile(invoker,true))
307                ||(!CMLib.flags().canBeHeardSpeakingBy(invoker,mob)))
308                        return unsingMe(mob,null);
309                return true;
310        }

312        protected void unsingAll(MOB mob, MOB invoker)
313        {
314                if(mob!=null)
315                {
316                        for(int a=mob.numEffects()-1;a>=0;a--)
317                        {
318                                final Ability A=mob.fetchEffect(a);
319                                if((A instanceof Song)
320                                &&((invoker==null)||(A.invoker()==null)||(A.invoker()==invoker)))
321                                        ((Song)A).unsingMe(mob,invoker);
322                        }
323                }
324        }

326        protected void unsingAllByThis(MOB mob, MOB invoker)
327        {
328                if(mob!=null)
329                {
330                        for(int a=mob.numEffects()-1;a>=0;a--)
331                        {
332                                final Ability A=mob.fetchEffect(a);
333                                if((A instanceof Song)
334                                &&(!A.ID().equals(ID()))
335                                &&((invoker==null)||(A.invoker()==null)||(A.invoker()==invoker)))
336                                        ((Song)A).unsingMe(mob,invoker);
337                        }
338                }
339        }

341        protected boolean unsingMe(MOB mob, MOB invoker)
342        {
343                if(mob==null)
344                        return false;
345                final Ability A=mob.fetchEffect(ID());
346                if((A instanceof Song)
347                &&((invoker==null)||(A.invoker()==null)||(A.invoker()==invoker)))
348                {
349                        final Song S=(Song)A;
350                        if(S.timeOut==0)
351                        {
352                                S.timeOut = System.currentTimeMillis()
353                                                  + (CMProps.getTickMillis() * (((invoker()!=null)&&(invoker()!=mob))?super.getXTIMELevel(invoker()):0));
354                        }
355                        if(System.currentTimeMillis() >= S.timeOut)
356                        {
357                                A.unInvoke();
358                                return false;
359                        }
360                }
361                return true;
362        }

364        protected List<Room> getInvokerScopeRoomSet(MOB backupMob)
365        {
366                if((invoker()==null)
367                ||(invoker().location()==null))
368                {
369                        if((backupMob!=null)&&(backupMob.location()!=null))
370                                return new XVector<Room>(backupMob.location());
371                        return new Vector<Room>();
372                }
373                final int depth=getXMAXRANGELevel(invoker()) / 2; // decreased because fireball
374                if(depth==0)
375                        return new XVector<Room>(invoker().location());
376                final Vector<Room> rooms=new Vector<Room>();
377                // needs to be area-only, because of the aggro-tracking rule
378                TrackingLibrary.TrackingFlags flags;
379                flags = CMLib.tracking().newFlags()
380                                .plus(TrackingLibrary.TrackingFlag.OPENONLY)
381                                .plus(TrackingLibrary.TrackingFlag.AREAONLY)
382                                .plus(TrackingLibrary.TrackingFlag.NOAIR);
383                CMLib.tracking().getRadiantRooms(invoker().location(), rooms,flags, null, depth, null);
384                if(!rooms.contains(invoker().location()))
385                        rooms.addElement(invoker().location());
386                return rooms;
387        }

389        protected int getCorrectDirToOriginRoom(Room R, int v)
390        {
391                if(v<0)
392                        return -1;
393                int dir=-1;
394                Room R2=null;
395                Exit E2=null;
396                int lowest=v;
397                for(int d=Directions.NUM_DIRECTIONS()-1;d>=0;d--)
398                {
399                        R2=R.getRoomInDir(d);
400                        E2=R.getExitInDir(d);
401                        if((R2!=null)&&(E2!=null)&&(E2.isOpen()))
402                        {
403                                final int dx=commonRoomSet.indexOf(R2);
404                                if((dx>=0)&&(dx<lowest))
405                                {
406                                        lowest=dx;
407                                        dir=d;
408                                }
409                        }
410                }
411                return dir;
412        }

414        protected String getCorrectMsgString(Room R, String str, int v)
415        {
416                String msgStr=null;
417                if(R==originRoom)
418                        msgStr=str;
419                else
420                {
421                        final int dir=this.getCorrectDirToOriginRoom(R,v);
422                        if(dir>=0)
423                                msgStr=L("^SYou hear the @x1 being sung @x2!^?",songOf(),CMLib.directions().getInDirectionName(dir));
424                        else
425                                msgStr=L("^SYou hear the @x1 being sung nearby!^?",songOf());
426                }
427                return msgStr;
428        }

430        public Set<MOB> sendMsgAndGetTargets(MOB mob, Room R, CMMsg msg, Environmental givenTarget, boolean auto)
431        {
432                if(originRoom==R)
433                        R.send(mob,msg);
434                else
435                        R.sendOthers(mob,msg);
436                if(R!=originRoom)
437                        mob.setLocation(R);
438                final Set<MOB> h=properTargets(mob,givenTarget,auto);
439                if(R!=originRoom)
440                {
441                        R.delInhabitant(mob);
442                        mob.setLocation(originRoom);
443                }
444                if(h==null)
445                        return null;
446                if(R==originRoom)
447                {
448                        if(!h.contains(mob))
449                                h.add(mob);
450                }
451                else
452                        h.remove(mob);
453                return h;
454        }

456        @Override
457        public boolean invoke(MOB mob, List<String> commands, Physical givenTarget, boolean auto, int asLevel)
458        {
459                timeOut=0;
460                if(!super.invoke(mob,commands,givenTarget,auto,asLevel))
461                        return false;
462
463                if((!auto)
464                &&(!mob.isMonster())
465                &&(!disregardsArmorCheck(mob))
466                &&(!CMLib.utensils().armorCheck(mob,CharClass.ARMOR_LEATHER))
467                &&(mob.isMine(this))
468                &&(mob.location()!=null)
469                &&(CMLib.dice().rollPercentage()<50))
470                {
471                        mob.location().show(mob,null,CMMsg.MSG_OK_VISUAL,L("<S-NAME> hit(s) a foul note on @x1 due to <S-HIS-HER> armor!",name()));
472                        return false;
473                }
474
475                if(skipStandardSongInvoke())
476                        return true;
477
478                if((!auto)&&(!CMLib.flags().canSpeak(mob)))
479                {
480                        mob.tell(L("You can't sing!"));
481                        return false;
482                }
483
484                final boolean success=proficiencyCheck(mob,0,auto);
485                unsingAllByThis(mob,mob);
486                if(success)
487                {
488                        invoker=mob;
489                        originRoom=mob.location();
490                        commonRoomSet=getInvokerScopeRoomSet(null);
491                        String str=auto?L("^SThe @x1 begins to play!^?",songOf()):L("^S<S-NAME> begin(s) to sing the @x1.^?",songOf());
492                        if((!auto)&&(mob.fetchEffect(this.ID())!=null))
493                                str=L("^S<S-NAME> start(s) the @x1 over again.^?",songOf());
494                        for(int v=0;v<commonRoomSet.size();v++)
495                        {
496                                final Room R=commonRoomSet.get(v);
497                                final String msgStr=getCorrectMsgString(R,str,v);
498                                final CMMsg msg=CMClass.getMsg(mob,null,this,verbalCastCode(mob,null,auto),msgStr);
499                                if(R.okMessage(mob,msg))
500                                {
501                                        final Set<MOB> h=this.sendMsgAndGetTargets(mob, R, msg, givenTarget, auto);
502                                        if(h==null)
503                                                continue;
504                                        final Song newOne=(Song)this.copyOf();
505                                        for (final Object element : h)
506                                        {
507                                                final MOB follower=(MOB)element;
508                                                final Room R2=follower.location();
509
510                                                // malicious songs must not affect the invoker!
511                                                int affectType=CMMsg.MSG_CAST_VERBAL_SPELL;
512                                                if(auto)
513                                                        affectType=affectType|CMMsg.MASK_ALWAYS;
514                                                if((castingQuality(mob,follower)==Ability.QUALITY_MALICIOUS)&&(follower!=mob))
515                                                        affectType=affectType|CMMsg.MASK_MALICIOUS;
516
517                                                if((CMLib.flags().canBeHeardSpeakingBy(invoker,follower)&&(follower.fetchEffect(this.ID())==null)))
518                                                {
519                                                        CMMsg msg2=CMClass.getMsg(mob,follower,this,affectType,null);
520                                                        final CMMsg msg3=msg2;
521                                                        if((mindAttack())&&(follower!=mob))
522                                                                msg2=CMClass.getMsg(mob,follower,this,CMMsg.MSK_CAST_MALICIOUS_VERBAL|CMMsg.TYP_MIND|(auto?CMMsg.MASK_ALWAYS:0),null);
523                                                        if((R.okMessage(mob,msg2))&&(R.okMessage(mob,msg3)))
524                                                        {
525                                                                R2.send(follower,msg2);
526                                                                if(msg2.value()<=0)
527                                                                {
528                                                                        R2.send(follower,msg3);
529                                                                        if((msg3.value()<=0)&&(follower.fetchEffect(newOne.ID())==null))
530                                                                        {
531                                                                                newOne.setSavable(false);
532                                                                                if(follower!=mob)
533                                                                                        follower.addEffect((Ability)newOne.copyOf());
534                                                                                else
535                                                                                        follower.addEffect(newOne);
536                                                                        }
537                                                                }
538                                                        }
539                                                }
540                                        }
541                                        R.recoverRoomStats();
542                                }
543                        }
544                }
545                else
546                        mob.location().show(mob,null,CMMsg.MSG_NOISE,L("<S-NAME> hit(s) a foul note."));
547
548                return success;
549        }
