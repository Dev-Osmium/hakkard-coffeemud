148        @Override
149        public boolean tick(final Tickable ticking, final int tickID)
150        {
151                if((!super.tick(ticking,tickID))||(!(affected instanceof MOB)))
152                        return false;
153
154                final MOB mob=(MOB)affected;
155                if((affected==invoker())&&(invoker()!=null)&&(invoker().location()!=originRoom))
156                {
157                        final Vector<Room> V=getInvokerScopeRoomSet(null);
158                        commonRoomSet.clear();
159                        commonRoomSet.addAll(V);
160                        originRoom=invoker().location();
161                }
162                else
163                if((abstractQuality()==Ability.QUALITY_MALICIOUS)
164                &&(!maliciousButNotAggressiveFlag())
165                &&(!mob.amDead())
166                &&(mob.isMonster())
167                &&(!mob.isInCombat())
168                &&(CMLib.flags().isAliveAwakeMobile(mob,true))
169                &&(mob.amFollowing()==null)
170                &&((!(mob instanceof Rideable))||(((Rideable)mob).numRiders()==0))
171                &&(!CMLib.flags().isATrackingMonster(mob)))
172                {
173                        if((mob.location()!=originRoom)
174                        &&(CMLib.flags().isMobile(mob)))
175                        {
176                                final int dir=this.getCorrectDirToOriginRoom(mob.location(),commonRoomSet.indexOf(mob.location()));
177                                if(dir>=0)
178                                        CMLib.tracking().walk(mob,dir,false,false);
179                        }
180                        else
181                        if((mob.location().isInhabitant(invoker()))
182                        &&(CMLib.flags().canBeSeenBy(invoker(),mob)))
183                                CMLib.combat().postAttack(mob,invoker(),mob.fetchWieldedItem());
184                }
185
186                if((invoker==null)
187                ||(invoker.fetchEffect(ID())==null)
188                ||(commonRoomSet==null))
189                        return undanceMe(mob,null);
190                if(!commonRoomSet.contains(mob.location()))
191                {
192                        final List<Room> V=getInvokerScopeRoomSet(null);
193                        commonRoomSet.clear();
194                        commonRoomSet.addAll(V);
195                        if(!commonRoomSet.contains(mob.location()))
196                                return undanceMe(mob,null);
197                }
198
199                if(skipStandardDanceTick())
200                        return true;
201
202                if((invoker==null)
203                ||(CMLib.flags().isSitting(invoker()))
204                ||(!CMLib.flags().isAliveAwakeMobile(mob,true))
205                ||(!CMLib.flags().isAliveAwakeMobile(invoker(),true))
206                ||(mob.riding() instanceof MOB)
207                ||(!CMLib.flags().canBeSeenBy(invoker,mob)))
208                        return undanceMe(mob,null);
209
210                if(invokerManaCost<0)
211                        invokerManaCost=usageCost(invoker(),false)[1];
212                if(!mob.curState().adjMovement(-(invokerManaCost/15),mob.maxState()))
213                {
214                        mob.tell(L("The dancing exhausts you."));
215                        undanceAll(mob,null);
216                        return false;
217                }
218                return true;
219        }

221        protected void undanceAll(MOB mob, MOB invoker)
222        {
223                if(mob!=null)
224                {
225                        for(int a=mob.numEffects()-1;a>=0;a--)
226                        {
227                                final Ability A=mob.fetchEffect(a);
228                                if((A instanceof Dance)
229                                &&((invoker==null)||(A.invoker()==null)||(A.invoker()==invoker)))
230                                        ((Dance)A).undanceMe(mob,invoker);
231                        }
232                }
233        }

250        protected boolean undanceMe(MOB mob, MOB invoker)
251        {
252                if(mob==null)
253                        return false;
254                final Ability A=mob.fetchEffect(ID());
255                if((A instanceof Dance)
256                &&((invoker==null)||(A.invoker()==null)||(A.invoker()==invoker)))
257                {
258                        final Dance D=(Dance)A;
259                        if(D.timeOut==0)
260                                D.timeOut = System.currentTimeMillis()
261                                                  + (CMProps.getTickMillis() * (((invoker()!=null)&&(invoker()!=mob))?super.getXTIMELevel(invoker()):0));
262                        if(System.currentTimeMillis() >= D.timeOut)
263                        {
264                                A.unInvoke();
265                                return false;
266                        }
267                }
268                return true;
269        }

271        protected Vector<Room> getInvokerScopeRoomSet(MOB backupMob)
272        {
273                if((invoker()==null)
274                ||(invoker().location()==null))
275                {
276                        if((backupMob!=null)&&(backupMob.location()!=null))
277                                 return new XVector<Room>(backupMob.location());
278                        return new Vector<Room>();
279                }
280                final int depth=super.getXMAXRANGELevel(invoker()) / 2; // decreased because fireball
281                if(depth==0)
282                        return new XVector<Room>(invoker().location());
283                final Vector<Room> rooms=new Vector<Room>();
284                // needs to be area-only, because of the aggro-tracking rule
285                TrackingLibrary.TrackingFlags flags;
286                flags = CMLib.tracking().newFlags()
287                                .plus(TrackingLibrary.TrackingFlag.OPENONLY)
288                                .plus(TrackingLibrary.TrackingFlag.AREAONLY)
289                                .plus(TrackingLibrary.TrackingFlag.NOAIR);
290                CMLib.tracking().getRadiantRooms(invoker().location(), rooms, flags, null, depth, null);
291                if(!rooms.contains(invoker().location()))
292                        rooms.addElement(invoker().location());
293                return rooms;
294        }

296        protected int getCorrectDirToOriginRoom(Room R, int v)
297        {
298                if(v<0)
299                        return -1;
300                int dir=-1;
301                Room R2=null;
302                Exit E2=null;
303                int lowest=v;
304                for(int d=Directions.NUM_DIRECTIONS()-1;d>=0;d--)
305                {
306                        R2=R.getRoomInDir(d);
307                        E2=R.getExitInDir(d);
308                        if((R2!=null)&&(E2!=null)&&(E2.isOpen()))
309                        {
310                                final int dx=commonRoomSet.indexOf(R2);
311                                if((dx>=0)&&(dx<lowest))
312                                {
313                                        lowest=dx;
314                                        dir=d;
315                                }
316                        }
317                }
318                return dir;
319        }

321        protected String getCorrectMsgString(Room R, String str, int v)
322        {
323                String msgStr=null;
324                if(R==originRoom)
325                        msgStr=str;
326                else
327                {
328                        final int dir=this.getCorrectDirToOriginRoom(R,v);
329                        if(dir>=0)
330                                msgStr=L("^SYou see the @x1 being performed @x2!^?",danceOf(),CMLib.directions().getInDirectionName(dir));
331                        else
332                                msgStr=L("^SYou see the @x1 being performed nearby!^?",danceOf());
333                }
334                return msgStr;
335        }

337        public Set<MOB> sendMsgAndGetTargets(MOB mob, Room R, CMMsg msg, Environmental givenTarget, boolean auto)
338        {
339                if(originRoom==R)
340                        R.send(mob,msg);
341                else
342                        R.sendOthers(mob,msg);
343                if(R!=originRoom)
344                        mob.setLocation(R);
345                final Set<MOB> h=properTargets(mob,givenTarget,auto);
346                if(R!=originRoom)
347                {
348                        R.delInhabitant(mob);
349                        mob.setLocation(originRoom);
350                }
351                if(h==null)
352                        return null;
353                if(R==originRoom)
354                {
355                        if(!h.contains(mob))
356                                h.add(mob);
357                }
358                else
359                        h.remove(mob);
360                return h;
361        }

363        @Override
364        public void executeMsg(final Environmental host, final CMMsg msg)
365        {
366                if((invoker()!=null)
367                &&(!unInvoked)
368                &&(affected==invoker())
369                &&(msg.amISource(invoker()))
370                &&(msg.target() instanceof Armor)
371                &&(msg.targetMinor()==CMMsg.TYP_WEAR))
372                {
373                        if(msg.source().location()!=null)
374                                msg.source().location().show(msg.source(),null,CMMsg.MSG_NOISE,L("<S-NAME> stop(s) dancing."));
375                        unInvoke();
376                }
377                super.executeMsg(host,msg);
378        }

381        public int castingQuality(MOB mob, Physical target)
382        {
383                if(mob!=null)
384                {
385                        for(final Enumeration<Ability> a=mob.effects();a.hasMoreElements();)
386                        {
387                                final Ability A=a.nextElement();
388                                if(A instanceof Dance)
389                                        return Ability.QUALITY_INDIFFERENT;
390                        }
391                }
392                return super.castingQuality(mob,target);
393        }

395        @Override
396        public boolean invoke(MOB mob, List<String> commands, Physical givenTarget, boolean auto, int asLevel)
397        {
398                timeOut=0;
399                if(!super.invoke(mob,commands,givenTarget,auto,asLevel))
400                        return false;
401
402                if((!auto)
403                &&(!mob.isMonster())
404                &&(!disregardsArmorCheck(mob))
405                &&(!CMLib.utensils().armorCheck(mob,CharClass.ARMOR_LEATHER))
406                &&(mob.isMine(this))
407                &&(mob.location()!=null)
408                &&(CMLib.dice().rollPercentage()<50))
409                {
410                        mob.location().show(mob,null,CMMsg.MSG_OK_VISUAL,L("<S-NAME> fumble(s) the @x1 due to <S-HIS-HER> armor!",name()));
411                        return false;
412                }
413
414                if(skipStandardDanceInvoke())
415                        return true;
416
417                if((!auto)&&(!CMLib.flags().isAliveAwakeMobile(mob,false)))
418                        return false;
419
420                final boolean success=proficiencyCheck(mob,0,auto);
421
422                undanceAll(mob,null);
423
424                if(success)
425                {
426                        invoker=mob;
427                        originRoom=mob.location();
428                        commonRoomSet=getInvokerScopeRoomSet(null);
429                        String str=auto?L("^SThe @x1 begins!^?",danceOf()):L("^S<S-NAME> begin(s) to dance the @x1.^?",danceOf());
430                        if((!auto)&&(mob.fetchEffect(this.ID())!=null))
431                                str=L("^S<S-NAME> start(s) the @x1 over again.^?",danceOf());
432
433                        for(int v=0;v<commonRoomSet.size();v++)
434                        {
435                                final Room R=commonRoomSet.elementAt(v);
436                                final String msgStr=getCorrectMsgString(R,str,v);
437                                final CMMsg msg=CMClass.getMsg(mob,null,this,somanticCastCode(mob,null,auto),msgStr);
438                                if(R.okMessage(mob,msg))
439                                {
440                                        final Set<MOB> h=this.sendMsgAndGetTargets(mob, R, msg, givenTarget, auto);
441                                        if(h==null)
442                                                continue;
443                                        invoker=mob;
444                                        final Dance newOne=(Dance)this.copyOf();
445                                        newOne.invoker=mob;
446                                        newOne.invokerManaCost=-1;
447
448                                        for (final Object element : h)
449                                        {
450                                                final MOB follower=(MOB)element;
451                                                final Room R2=follower.location();
452
453                                                // malicious dances must not affect the invoker!
454                                                int affectType=CMMsg.MSG_CAST_SOMANTIC_SPELL;
455                                                if((castingQuality(mob,follower)==Ability.QUALITY_MALICIOUS)&&(follower!=mob))
456                                                        affectType=affectType|CMMsg.MASK_MALICIOUS;
457                                                if(auto)
458                                                        affectType=affectType|CMMsg.MASK_ALWAYS;
459
460                                                if((R2!=null)&&(CMLib.flags().canBeSeenBy(invoker,follower)&&(follower.fetchEffect(this.ID())==null)))
461                                                {
462                                                        CMMsg msg2=CMClass.getMsg(mob,follower,this,affectType,null);
463                                                        final CMMsg msg3=msg2;
464                                                        if((mindAttack())&&(follower!=mob))
465                                                                msg2=CMClass.getMsg(mob,follower,this,CMMsg.MSK_CAST_MALICIOUS_SOMANTIC|CMMsg.TYP_MIND|(auto?CMMsg.MASK_ALWAYS:0),null);
466                                                        if((R.okMessage(mob,msg2))&&(R.okMessage(mob,msg3)))
467                                                        {
468                                                                R2.send(follower,msg2);
469                                                                if(msg2.value()<=0)
470                                                                {
471                                                                        R2.send(follower,msg3);
472                                                                        if((msg3.value()<=0)&&(follower.fetchEffect(newOne.ID())==null))
473                                                                        {
474                                                                                //possiblyUndance(follower,null,false);
475                                                                                newOne.setSavable(false);
476                                                                                if(follower!=mob)
477                                                                                        follower.addEffect((Ability)newOne.copyOf());
478                                                                                else
479                                                                                        follower.addEffect(newOne);
480                                                                        }
481                                                                }
482                                                        }
483                                                }
484                                        }
485                                        R.recoverRoomStats();
486                                }
487                        }
488                }
489                else
490                        mob.location().show(mob,null,CMMsg.MSG_NOISE,L("<S-NAME> make(s) a false step."));
491
492                return success;
493        }
