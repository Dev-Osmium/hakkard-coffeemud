089        protected CMMsg lastMsg=null;

092        @Override
093        public void executeMsg(final Environmental affecting, final CMMsg msg)
094        {
095                super.executeMsg(affecting,msg);
096                if((affecting instanceof MOB)&&(!CMLib.flags().isAliveAwakeMobileUnbound((MOB)affecting,true)))
097                        return;
098                if(disabled)
099                        return;
100
101                if(((!(affecting instanceof MOB))||(!msg.amISource((MOB)affecting)))
102                &&((text().length()==0)||(text().equalsIgnoreCase(msg.source().Name())))
103                &&((msg.sourceMinor()!=CMMsg.TYP_EMOTE)
104                        ||(msg.tool() instanceof Social)))
105                        lastMsg=msg;
106        }

109        {
110                //String src=msg.sourceMessage();
111                String trg=msg.targetMessage();
112                String oth=msg.othersMessage();

151        @Override
152        public boolean tick(final Tickable ticking, final int tickID)
153        {
154                super.tick(ticking,tickID);
155                CMMsg msg=lastMsg;
156                if(msg==null)
157                        return true;
158                lastMsg=null;
159                if(((affected instanceof MOB)&&(!CMLib.flags().isAliveAwakeMobileUnbound((MOB)affected,true))))
160                        return true;
161                msg=(CMMsg)msg.copyOf();
162                final MOB sMOB=msg.source();
163                if(msg.sourceMinor()==CMMsg.TYP_EMOTE)
164                {
165                        if(affected instanceof MOB)
166                                msg.modify((MOB)affected,msg.target(),msg.tool(),
167                                                   msg.sourceCode(),msg.sourceMessage(),
168                                                   msg.targetCode(),msg.targetMessage(),
169                                                   msg.othersCode(),msg.othersMessage());
170                        else
171                        {
172                                final MOB newSMOB=CMClass.getFactoryMOB();
173                                newSMOB.baseCharStats().setStat(CharStats.STAT_GENDER,'N');
174                                newSMOB.setName(affected.name());
175                                newSMOB.recoverCharStats();
176                                msg.modify(newSMOB,msg.source(),msg.tool(),
177                                                   msg.sourceCode(),msg.sourceMessage(),
178                                                   msg.targetCode(),msg.targetMessage(),
179                                                   msg.othersCode(),msg.othersMessage());
180                        }
181                }
182                else
183                if(msg.tool() instanceof Social)
184                {
185                        MOB target=null;
186                        if(msg.target() instanceof MOB)
187                                target=msg.source();
188                        if(affected instanceof MOB)
189                                msg.modify((MOB)affected,target,msg.tool(),
190                                                   msg.sourceCode(),msg.sourceMessage(),
191                                                   msg.targetCode(),msg.targetMessage(),
192                                                   msg.othersCode(),msg.othersMessage());
193                        else
194                        {
195                                final MOB newSMOB=CMClass.getFactoryMOB();
196                                newSMOB.baseCharStats().setStat(CharStats.STAT_GENDER,'N');
197                                newSMOB.setName(affected.name());
198                                newSMOB.recoverCharStats();
199                                msg.modify(newSMOB,target,msg.tool(),
200                                                   msg.sourceCode(),msg.sourceMessage(),
201                                                   msg.targetCode(),msg.targetMessage(),
202                                                   msg.othersCode(),msg.othersMessage());
203                        }
204                }
205                else
206                        return true;
207                disabled=true;
208                if((sMOB.location()!=null)
209                &&(sMOB.location().okMessage(sMOB,msg)))
210                {
211                        if(msg.source().location()==null)
212                        {
213                                msg.source().setLocation(sMOB.location());
214                                sMOB.location().send(msg.source(),msg);
215                                msg.source().setLocation(null);
216                        }
217                        else
218                                sMOB.location().send(msg.source(),msg);
219                }
220                disabled=false;
221                return true;
222        }

224        @Override
225        public boolean invoke(MOB mob, List<String> commands, Physical givenTarget, boolean auto, int asLevel)
226        {
227                final MOB target=this.getTarget(mob,commands,givenTarget);
228                if(target==null)
229                        return false;
230
231                if(!super.invoke(mob,commands,givenTarget,auto,asLevel))
232                        return false;
233
234                final boolean success=proficiencyCheck(mob,0,auto);
235
236                if(success)
237                {
238                        final CMMsg msg=CMClass.getMsg(mob,target,this,CMMsg.MSG_QUIETMOVEMENT|(auto?CMMsg.MASK_ALWAYS:0),auto?"":L("<S-NAME> begin(s) mimicing <T-NAMESELF>."));
239                        if(mob.location().okMessage(mob,msg))
240                        {
241                                mob.location().send(mob,msg);
242                                beneficialAffect(mob,mob,asLevel,0);
243                                final Ability A=mob.fetchEffect(ID());
244                                if((A!=null)&&(target!=mob))
245                                        A.setMiscText(target.Name());
246                        }
247                }
248                else
249                        return beneficialVisualFizzle(mob,target,L("<S-NAME> attempt(s) to mimic <T-NAMESELF>, but fail(s)."));
250                return success;
251        }
