036public class Song_Thanks extends Song
037{
038        @Override
039        public String ID()
040        {
041                return "Song_Thanks";
042        }
043
044        private final static String localizedName = CMLib.lang().L("Thanks");
045
046        @Override
047        public String name()
048        {
049                return localizedName;
050        }
051
052        @Override
053        public int abstractQuality()
054        {
055                return Ability.QUALITY_MALICIOUS;
056        }
057
058        @Override
059        protected boolean skipStandardSongInvoke()
060        {
061                return true;
062        }
063
064        @Override
065        protected boolean maliciousButNotAggressiveFlag()
066        {
067                return true;
068        }
069
070        @Override
071        public long flags()
072        {
073                return super.flags() | Ability.FLAG_MINDALTERING;
074        }
075
076        @Override
077        public int castingQuality(final MOB mob, final Physical target)
078        {
079                if(mob!=null)
080                {
081                        if(mob.isInCombat()&&(mob.isMonster()))
082                                return Ability.QUALITY_INDIFFERENT;
083                }
084                return super.castingQuality(mob,target);
085        }
086
087        @Override
088        public boolean tick(final Tickable ticking, final int tickID)
089        {
090                if(!super.tick(ticking,tickID))
091                        return false;
092
093                final MOB mob=(MOB)affected;
094                if(mob==null)
095                        return true;
096                if(mob==invoker)
097                        return true;
098                if(invoker==null)
099                        return true;
100                if(mob.location()!=invoker.location())
101                        return true;
102                //if(!mob.isMonster()) return true;
103                if((CMLib.dice().rollPercentage()<6)
104                   &&(CMLib.dice().rollPercentage()>mob.charStats().getSave(CharStats.STAT_SAVE_MIND))
105                   &&(CMLib.dice().rollPercentage()>mob.charStats().getSave(CharStats.STAT_SAVE_MAGIC))
106                   &&(CMLib.flags().canMove(mob))
107                   &&(CMLib.flags().canBeSeenBy(invoker,mob))
108                   &&(CMLib.beanCounter().getTotalAbsoluteNativeValue(mob)>(1.0+super.getXLEVELLevel(invoker()))))
109                {
110                        switch(CMLib.dice().roll(1,10,0))
111                        {
112                        case 1:
113                                CMLib.commands().postSay(mob,invoker,L("Thank you @x1!",invoker.name()),false,false);
114                                break;
115                        case 2:
116                                CMLib.commands().postSay(mob,invoker,L("Thanks for being you, @x1!",invoker.name()),false,false);
117                                break;
118                        case 3:
119                                CMLib.commands().postSay(mob,invoker,L("Thanks @x1!",invoker.name()),false,false);
120                                break;
121                        case 4:
122                                CMLib.commands().postSay(mob,invoker,L("You are great, @x1!  Thanks!",invoker.name()),false,false);
123                                break;
124                        case 5:
125                                CMLib.commands().postSay(mob,invoker,L("I appreciate you, @x1!",invoker.name()),false,false);
126                                break;
127                        case 6:
128                                CMLib.commands().postSay(mob,invoker,L("Keep it up, @x1! Thanks!",invoker.name()),false,false);
129                                break;
130                        case 7:
131                                CMLib.commands().postSay(mob,invoker,L("Thanks a lot, @x1!",invoker.name()),false,false);
132                                break;
133                        case 8:
134                                CMLib.commands().postSay(mob,invoker,L("Thank you dearly, @x1!",invoker.name()),false,false);
135                                break;
136                        case 9:
137                                CMLib.commands().postSay(mob,invoker,L("Thank you always, @x1!",invoker.name()),false,false);
138                                break;
139                        case 10:
140                                CMLib.commands().postSay(mob,invoker,L("You're the best, @x1! Thanks!",invoker.name()),false,false);
141                                break;
142                        }
143                        final Coins C=CMLib.beanCounter().makeBestCurrency(mob,CMath.mul(1.0,super.getXLEVELLevel(invoker())));
144                        if(C!=null)
145                        {
146                                CMLib.beanCounter().subtractMoney(mob,CMath.mul(1.0,super.getXLEVELLevel(invoker())));
147                                mob.addItem(C);
148                                mob.doCommand(CMParms.parse("GIVE \""+C.name()+"\" \""+invoker.name()+"\""),MUDCmdProcessor.METAFLAG_FORCED);
149                                if(!C.amDestroyed())
150                                        C.putCoinsBack();
151                        }
152                }
153                return true;
154        }
155
156        @Override
157        public boolean invoke(final MOB mob, final List<String> commands, final Physical givenTarget, final boolean auto, final int asLevel)
158        {
159                timeOut=0;
160                if(!super.invoke(mob,commands,givenTarget,auto,asLevel))
161                        return false;
162
163                if((!auto)&&(!CMLib.flags().canSpeak(mob)))
164                {
165                        mob.tell(L("You can't sing!"));
166                        return false;
167                }
168
169                final boolean success=proficiencyCheck(mob,0,auto);
170                unsingAllByThis(mob,mob);
171                if(success)
172                {
173                        invoker=mob;
174                        originRoom=mob.location();
175                        commonRoomSet=getInvokerScopeRoomSet(null);
176                        String str=auto?L("^SThe @x1 begins to play!^?",songOf()):L("^S<S-NAME> begin(s) to sing the @x1.^?",songOf());
177                        if((!auto)&&(mob.fetchEffect(this.ID())!=null))
178                                str=L("^S<S-NAME> start(s) the @x1 over again.^?",songOf());
179
180                        for(int v=0;v<commonRoomSet.size();v++)
181                        {
182                                final Room R=commonRoomSet.get(v);
183                                final String msgStr=getCorrectMsgString(R,str,v);
184                                final CMMsg msg=CMClass.getMsg(mob,null,this,verbalCastCode(mob,null,auto),msgStr);
185                                if(mob.location().okMessage(mob,msg))
186                                {
187                                        final Set<MOB> h=this.sendMsgAndGetTargets(mob, R, msg, givenTarget, auto);
188                                        if(h==null)
189                                                continue;
190                                        final Song newOne=(Song)this.copyOf();
191
192                                        for (final Object element : h)
193                                        {
194                                                final MOB follower=(MOB)element;
195
196                                                // malicious songs must not affect the invoker!
197                                                int affectType=CMMsg.MSG_CAST_VERBAL_SPELL;
198                                                if(auto)
199                                                        affectType=affectType|CMMsg.MASK_ALWAYS;
200
201                                                if((CMLib.flags().canBeHeardSpeakingBy(invoker,follower)&&(follower.fetchEffect(this.ID())==null)))
202                                                {
203                                                        final CMMsg msg2=CMClass.getMsg(mob,follower,this,affectType,null);
204                                                        final CMMsg msg3=msg2;
205                                                        if((R.okMessage(mob,msg2))&&(R.okMessage(mob,msg3)))
206                                                        {
207                                                                follower.location().send(follower,msg2);
208                                                                if(msg2.value()<=0)
209                                                                {
210                                                                        follower.location().send(follower,msg3);
211                                                                        if((msg3.value()<=0)&&(follower.fetchEffect(newOne.ID())==null))
212                                                                        {
213                                                                                if(follower!=mob)
214                                                                                        follower.addEffect((Ability)newOne.copyOf());
215                                                                                else
216                                                                                        follower.addEffect(newOne);
217                                                                        }
218                                                                }
219                                                        }
220                                                }
221                                        }
222                                        R.recoverRoomStats();
223                                }
224                        }
225                }
226                else
227                        mob.location().show(mob,null,CMMsg.MSG_NOISE,L("<S-NAME> hit(s) a foul note."));
228
229                return success;
230        }
