104        public String composition()
105        {
106                String comp="";
107                if(trail!=null)
108                {
109                        String t=trail.toString();
110                        final int[] counts=new int[stuff.length];
111                        int x=t.indexOf(';');
112                        while(x>=0)
113                        {
114                                final int q=CMath.s_int(t.substring(0,x));
115                                t=t.substring(x+1);
116                                if(q>=0)
117                                {
118                                        for(int i=0;i<stuff.length;i++)
119                                        {
120                                                if(CMath.s_int(stuff[i][0])==q)
121                                                        counts[i]++;
122                                        }
123                                }
124                                x=t.indexOf(';');
125                        }
126                        int wa=-1;
127                        for(int i=0;i<stuff.length;i++)
128                        {
129                                if(CMath.s_int(stuff[i][0])==CMMsg.TYP_WEAPONATTACK)
130                                {
131                                        wa=i;
132                                        break;
133                                }
134
135                        }
136                        if(wa>=0)
137                                counts[wa]=counts[wa]/25;
138
139                        final Vector<Integer> V=new Vector<Integer>();
140                        while(V.size()<counts.length)
141                        {
142                                int high=-1;
143                                int which=-1;
144                                for(int i=0;i<counts.length;i++)
145                                {
146                                        if((counts[i]>high)&&(!V.contains(Integer.valueOf(i))))
147                                        {
148                                                high=counts[i];
149                                                which=i;
150                                        }
151                                }
152                                if(which>=0)
153                                        V.addElement(Integer.valueOf(which));
154                        }
155                        final Vector<Integer> V2=new Vector<Integer>();
156                        for(int i=0;i<3;i++)
157                        {
158                                final Integer ref=V.elementAt(i);
159                                Integer which=null;
160                                while((which==null)||(V2.contains(which)))
161                                {
162                                        final Integer w=V.elementAt(CMLib.dice().roll(1,V.size(),-1));
163                                        if(counts[w.intValue()]==counts[ref.intValue()])
164                                                which=w;
165                                }
166                                V2.addElement(which);
167                                comp+=which.intValue()+";"+counts[which.intValue()]+";";
168                        }
169                }
170                return comp;
171        }

173        @Override
174        public void affectPhyStats(final Physical affected, final PhyStats affectableStats)
175        {
176                if((whom!=null)&&(song!=null))
177                {
178                        final Hashtable<Integer,Integer> H=getSongBenefits(song);
179                        for(final Enumeration<Integer> e=H.keys();e.hasMoreElements();)
180                        {
181                                final Integer I=e.nextElement();
182                                final String[] chk=stuff[I.intValue()];
183                                if((chk!=null)&&(chk[1].startsWith("e")))
184                                {
185                                        int ticks=H.get(I).intValue();
186                                        if(ticks<=0)
187                                                ticks=1;
188                                        switch(chk[2].charAt(0))
189                                        {
190                                        case 'a':
191                                                if(ticks>25)
192                                                        ticks=25;
193                                                affectableStats.setAttackAdjustment(affectableStats.attackAdjustment()+ticks+getXLEVELLevel(invoker()));
194                                                break;
195                                        default:
196                                                break;
197                                        }
198                                }
199                        }
200                }
201        }

226                                        affectableStats.setStat(stat,affectableStats.getStat(stat)+ticks+getXLEVELLevel(invoker()));

286        {
287                if((whom!=null)
288                &&(song==null)
289                &&(msg.amISource(whom))

292                        if(trail==null)

297                }
298                super.executeMsg(myHost,msg);

312        @Override
313        public boolean tick(final Tickable ticking, final int tickID)
314        {
315                final MOB mob=(MOB)affected;
316                if(mob==null)
317                        return false;
318                if(song==null)
319                {
320                        if((whom==null)
321                        ||(commonRoomSet==null)
322                        ||(!commonRoomSet.contains(whom.location()))
323                        ||(CMLib.flags().isSleeping(invoker))
324                        ||(!CMLib.flags().canBeSeenBy(whom,invoker)))
325                                return unsingMe(mob,null);
326                }
327
328                if((whom!=null)&&(song!=null)&&(affected==invoker())
329                   &&(CMLib.dice().rollPercentage()<10))
330                {
331                        final Hashtable<Integer,Integer> H=getSongBenefits(song);
332                        final Vector<Integer> V=new Vector<Integer>();
333                        for(final Enumeration<Integer> e=H.keys();e.hasMoreElements();)
334                                V.addElement(e.nextElement());
335                        final Integer I=V.elementAt(CMLib.dice().roll(1,V.size(),-1));
336                        final String[] chk=stuff[I.intValue()];
337                        invoker().location().show(invoker(),this,whom,CMMsg.MSG_SPEAK,L("<S-NAME> sing(s) '@x1'.",chk[3]));
338                }
339
340                if(!super.tick(ticking,tickID))
341                        return false;
342
343                return true;
344        }

346        public Hashtable<Integer,Integer> getSongBenefits(String s)
347        {
348                if(benefits!=null)
349                        return benefits;
350                benefits=new Hashtable<Integer,Integer>();
351
352                int x=s.indexOf(';');
353                while(x>=0)
354                {
355                        final int code=CMath.s_int(s.substring(0,x));
356                        s=s.substring(x+1);
357                        x=s.indexOf(';');
358                        if(x>=0)
359                        {
360                                final int tick=CMath.s_int(s.substring(0,x));
361                                s=s.substring(x+1);
362                                benefits.put(Integer.valueOf(code),Integer.valueOf(tick));
363                        }
364                        x=s.indexOf(';');
365                }
366                return benefits;
367        }

369        @Override
370        public String text()
371        {
372                final StringBuffer x=new StringBuffer("");
373                for(final Enumeration<String> e=getSongs().keys();e.hasMoreElements();)
374                {
375                        final String key=e.nextElement();
376                        final String notkey=getSongs().get(key);
377                        x.append(key+"|~|"+notkey+"[|]");
378                }
379                miscText=x.toString();
380                return x.toString();
381        }
382
383        public Hashtable<String,String> getSongs()
384        {
385                if(songs!=null)
386                        return songs;
387                String t=miscText;
388                int x=t.indexOf("|~|");
389                songs=new Hashtable<String,String>();
390                while(x>=0)
391                {
392                        final String n=t.substring(0,x);
393                        t=t.substring(x+3);
394                        x=t.indexOf("[|]");
395                        String y="";
396                        if(x>=0)
397                        {
398                                y=t.substring(0,x);
399                                t=t.substring(x+3);
400                                songs.put(n,y);
401                        }
402                        x=t.indexOf("|~|");
403                }
404                return songs;
405        }

407        @Override
408        public boolean invoke(final MOB mob, final List<String> commands, final Physical givenTarget, final boolean auto, final int asLevel)
409        {
410                timeOut=0;
411                if(auto)
412                        return false;
413
414                final Hashtable<String,String> H=getSongs();
415                if(commands.size()==0)
416                {
417                        final Song_Ode A=(Song_Ode)mob.fetchEffect(ID());
418                        if((A!=null)&&(A.whom!=null)&&(A.song==null))
419                        {
420                                final String str=L("^S<S-NAME> finish(es) composing the @x1.^?",A.songOf());
421                                final CMMsg msg=CMClass.getMsg(mob,null,this,(auto?CMMsg.MASK_ALWAYS:0)|CMMsg.MSG_DELICATE_SMALL_HANDS_ACT,str);
422                                if(mob.location().okMessage(mob,msg))
423                                {
424                                        mob.location().send(mob,msg);
425                                        mob.delEffect(A);
426                                        getSongs().put(A.whom.name(),A.composition());
427                                        whom=null;
428                                        return true;
429                                }
430                                return false;
431                        }
432
433                        final StringBuffer str=new StringBuffer("");
434                        for(final Enumeration<String> e=H.keys();e.hasMoreElements();)
435                                str.append(e.nextElement()+" ");
436                        mob.tell(L("Compose or sing an ode about whom?"));
437                        if(str.length()>0)
438                                mob.tell(L("You presently have odes written about: @x1.",str.toString().trim()));
439                        return false;
440                }
441                String name=CMParms.combine(commands,0);
442                for(final Enumeration<String> e=H.keys();e.hasMoreElements();)
443                {
444                        final String key=e.nextElement();
445                        if(CMLib.english().containsString(key,name))
446                        {
447                                invoker=mob;
448                                originRoom=mob.location();
449                                commonRoomSet=getInvokerScopeRoomSet(null);
450                                name=key;
451                                song=H.get(name);
452                                benefits=null;
453                                whom=mob.location().fetchInhabitant(name);
454                                if((whom==null)||(!whom.name().equals(name)))
455                                        whom=CMLib.players().getPlayer(name);
456                                if((whom==null)||(!whom.name().equals(name)))
457                                {
458                                        whom=CMClass.getMOB("StdMOB");
459                                        whom.setName(name);
460                                        whom.setLocation(mob.location());
461                                }
462                                return super.invoke(mob,commands,givenTarget,auto,asLevel);
463                        }
464                }
465
466                final MOB target=getTarget(mob,commands,givenTarget);
467                if(target==null)
468                        return false;
469                if(target==mob)
470                {
471                        mob.tell(L("You may not compose an ode about yourself!"));
472                        return false;
473                }
474
475                final boolean success=proficiencyCheck(mob,0,auto);
476                if(success)
477                {
478                        unsingAll(mob,mob);
479                        invoker=mob;
480                        originRoom=mob.location();
481                        commonRoomSet=getInvokerScopeRoomSet(null);
482                        whom=target;
483                        final String str=L("^S<S-NAME> begin(s) to compose an @x1.^?",songOf());
484                        final CMMsg msg=CMClass.getMsg(mob,null,this,(auto?CMMsg.MASK_ALWAYS:0)|CMMsg.MSG_DELICATE_SMALL_HANDS_ACT,str);
485                        if(mob.location().okMessage(mob,msg))
486                        {
487                                mob.location().send(mob,msg);
488                                invoker=mob;
489                                final Song_Ode newOne=(Song_Ode)copyOf();
490                                newOne.whom=target;
491                                newOne.trail=new StringBuffer("");
492                                newOne.song=null;
493                                mob.addEffect(newOne);
494                        }
495                }
496                else
497                        mob.location().show(mob,null,CMMsg.MSG_NOISE,L("<S-NAME> lose(s) <S-HIS-HER> inspiration."));
498                return success;
499        }
