035public class Song_Lullibye extends Song
036{
037        @Override
038        public String ID()
039        {
040                return "Song_Lullibye";
041        }
042
043        private final static String localizedName = CMLib.lang().L("Lullaby");
044
045        @Override
046        public String name()
047        {
048                return localizedName;
049        }
050
051        @Override
052        public int abstractQuality()
053        {
054                return Ability.QUALITY_MALICIOUS;
055        }
056
057        boolean asleep=false;
058
059        @Override
060        public void affectPhyStats(final Physical affected, final PhyStats affectableStats)
061        {
062                super.affectPhyStats(affected,affectableStats);
063                if(invoker==null)
064                        return;
065                if(affected==invoker)
066                        return;
067                if(asleep)
068                        affectableStats.setDisposition(affectableStats.disposition()|PhyStats.IS_SLEEPING);
069        }
070
071        @Override
072        public boolean tick(final Tickable ticking, final int tickID)
073        {
074                if(!super.tick(ticking,tickID))
075                        return false;
076
077                final MOB mob=(MOB)affected;
078                if(mob==null)
079                        return true;
080                if(mob==invoker)
081                        return true;
082                final boolean oldasleep=asleep;
083                if(CMLib.dice().rollPercentage()>(50-(2*getXLEVELLevel(invoker()))))
084                        asleep=true;
085                else
086                        asleep=false;
087                if(asleep!=oldasleep)
088                {
089                        if(oldasleep)
090                        {
091                                if(CMLib.flags().isSleeping(mob))
092                                        mob.phyStats().setDisposition(mob.phyStats().disposition()-PhyStats.IS_SLEEPING);
093                                mob.location().show(mob,null,CMMsg.MSG_QUIETMOVEMENT,L("<S-NAME> wake(s) up."));
094                        }
095                        else
096                        {
097                                mob.location().show(mob,null,CMMsg.MSG_NOISYMOVEMENT,L("<S-NAME> fall(s) asleep."));
098                                mob.phyStats().setDisposition(mob.phyStats().disposition()|PhyStats.IS_SLEEPING);
099                        }
100                }
101
102                return true;
103        }
104
105        @Override
106        public boolean okMessage(final Environmental myHost, final CMMsg msg)
107        {
108                if(!super.okMessage(myHost,msg))
109                        return false;
110
111                if(msg.source()==invoker)
112                        return true;
113
114                if(msg.source()!=affected)
115                        return true;
116
117                if((!msg.sourceMajor(CMMsg.MASK_ALWAYS))
118                &&((msg.targetMinor()==CMMsg.TYP_STAND)||(msg.sourceMinor()==CMMsg.TYP_SIT))&&(asleep))
119                        return false;
120                return true;
121        }
122}
