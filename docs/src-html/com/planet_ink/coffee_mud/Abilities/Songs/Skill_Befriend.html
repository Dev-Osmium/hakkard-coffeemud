091        @Override
092        public boolean okMessage(final Environmental myHost, final CMMsg msg)
093        {
094                if (msg.amITarget(affected))
095                {
096                        switch (msg.targetMinor())
097                        {
098                        case CMMsg.TYP_VALUE:
099                        case CMMsg.TYP_SELL:
100                        case CMMsg.TYP_BUY:
101                        case CMMsg.TYP_LIST:
102                                if(affected instanceof MOB)
103                                {
104                                        MOB M=(MOB)affected;
105                                        final MOB srcM=msg.source();
106                                        if(srcM==M.amFollowing())
107                                        {
108                                                srcM.recoverCharStats();
109                                                srcM.charStats().setStat(CharStats.STAT_CHARISMA, srcM.charStats().getStat(CharStats.STAT_CHARISMA) + 10);
110                                                msg.addTrailerRunnable(new Runnable()
111                                                {
112                                                        @Override
113                                                        public void run()
114                                                        {
115                                                                srcM.recoverCharStats();
116                                                        }
117                                                });
118                                        }
119                                        else
120                                        if((text().length()>0)&&(text().startsWith(srcM.Name())))
121                                        {
122                                                String name = text();
123                                                int amt=10;
124                                                int x=text().indexOf('=');
125                                                if(x>0)
126                                                {
127                                                        name = text().substring(0,x).trim();
128                                                        amt=CMath.s_int(text().substring(x+1).trim());
129                                                }
130                                                if(name.equals(srcM.Name()))
131                                                {
132                                                        final int chaAmt = amt;
133                                                        final ExtAbility A=(ExtAbility)CMClass.getAbility("ExtAbility");
134                                                        A.setAbilityID(ID()+"_Stat");
135                                                        A.setStatsAffector(new StatsAffecting()
136                                                        {
137                                                                final int amt=chaAmt;
138                                                                @Override
139                                                                public void affectPhyStats(final Physical affected, final PhyStats affectableStats)
140                                                                {
141                                                                }
142
143                                                                @Override
144                                                                public void affectCharStats(final MOB affectedMob, final CharStats affectableStats)
145                                                                {
146                                                                        affectableStats.setStat(CharStats.STAT_CHARISMA, affectableStats.getStat(CharStats.STAT_CHARISMA)+amt);
147                                                                }
148
149                                                                @Override
150                                                                public void affectCharState(MOB affectedMob, CharState affectableMaxState)
151                                                                {
152                                                                }
153                                                        });
154                                                        final CMMsg thisMsg = msg;
155                                                        A.setMsgListener(new MsgListener()
156                                                        {
157                                                                final CMMsg msg2=thisMsg;
158                                                                final Ability extA = A;
159                                                                @Override
160                                                                public void executeMsg(final Environmental myHost, final CMMsg msg)
161                                                                {
162                                                                        if((msg != msg2)&&(myHost instanceof MOB))
163                                                                        {
164                                                                                ((MOB)myHost).delEffect(extA);
165                                                                                ((MOB)myHost).recoverCharStats();
166                                                                        }
167                                                                }
168
169                                                                @Override
170                                                                public boolean okMessage(final Environmental myHost, final CMMsg msg)
171                                                                {
172                                                                        if((msg != msg2)&&(myHost instanceof MOB))
173                                                                        {
174                                                                                ((MOB)myHost).delEffect(extA);
175                                                                                ((MOB)myHost).recoverCharStats();
176                                                                        }
177                                                                        return true;
178                                                                }
179                                                        });
180                                                        srcM.addEffect(A);
181                                                        srcM.recoverCharStats();
182                                                        msg.addTrailerRunnable(new Runnable()
183                                                        {
184                                                                final Ability extA = A;
185                                                                final MOB mob=srcM;
186                                                                @Override
187                                                                public void run()
188                                                                {
189                                                                        mob.delEffect(extA);
190                                                                        mob.recoverCharStats();
191                                                                }
192                                                        });
193                                                }
194                                        }
195                                }
196                                break;
197                        case CMMsg.TYP_NOFOLLOW:
198                                if(affected instanceof MOB)
199                                {
200                                        MOB M=(MOB)affected;
201                                        final MOB srcM=msg.source();
202                                        if(srcM==M.amFollowing())
203                                                unInvoke();
204                                }
205                                break;
206                        default:
207                                break;
208                        }
209                }
210                return super.okMessage(myHost, msg);
211        }

213        @Override 
214        public boolean tick(final Tickable ticking, final int tickID)
215        {
216                if(!super.tick(ticking, tickID))
217                        return false;
218                if((ticking instanceof MOB)
219                &&((text().length()==0)||(text().indexOf('=')<0)))
220                {
221                        final MOB mob=(MOB)ticking;
222                        if((mob.amFollowing()==null)
223                        ||(mob.amFollowing().isMonster())
224                        ||(!CMLib.flags().isInTheGame(mob.amFollowing(), true)))
225                        {
226                                if(mob.getStartRoom()==null)
227                                        mob.destroy();
228                                else
229                                if(mob.getStartRoom() != mob.location())
230                                        CMLib.tracking().wanderAway(mob, false, true);
231                                return false;
232                        }
233                }
234                return true;
235        }
236
237        @Override
238        public boolean invoke(MOB mob, List<String> commands, Physical givenTarget, boolean auto, int asLevel)
239        {
240                if(commands.size()<1)
241                {
242                        mob.tell(L("You must specify someone to befriend!"));
243                        return false;
244                }
245                final MOB target=getTarget(mob,commands,givenTarget);
246                if(target==null)
247                        return false;
248
249                if(target==mob)
250                {
251                        mob.tell(L("You are already your own friend."));
252                        return false;
253                }
254                if(target.phyStats().level()>mob.phyStats().level()+(mob.phyStats().level()/10))
255                {
256                        mob.tell(L("@x1 is a bit too powerful to befriend.",target.charStats().HeShe()));
257                        return false;
258                }
259                if(!CMLib.flags().isMobile(target))
260                {
261                        mob.tell(L("You can only befriend fellow travellers."));
262                        return false;
263                }
264
265                if(CMLib.coffeeShops().getShopKeeper(target)!=null)
266                {
267                        mob.tell(L("You cann't befriend a merchant."));
268                        return false;
269                }
270
271                if(!target.isMonster())
272                {
273                        mob.tell(L("You need to ask @x1",target.charStats().himher()));
274                        return false;
275                }
276
277                if(target.amFollowing()!=null)
278                {
279                        mob.tell(target,null,null,L("<S-NAME> is already someone elses friend."));
280                        return false;
281                }
282
283                if(!target.charStats().getMyRace().racialCategory().equals(mob.charStats().getMyRace().racialCategory()))
284                {
285                        mob.tell(target,null,null,L("<S-NAME> is not a fellow @x1.",mob.charStats().getMyRace().racialCategory()));
286                        return false;
287                }
288
289                final Faction F=CMLib.factions().getFaction(CMLib.factions().AlignID());
290                if(F!=null)
291                {
292                        final int his=target.fetchFaction(F.factionID());
293                        final int mine=target.fetchFaction(F.factionID());
294                        if(F.fetchRange(his)!=F.fetchRange(mine))
295                        {
296                                mob.tell(target,null,null,L("<S-NAME> is not @x1, like yourself.",F.fetchRangeName(mine)));
297                                return false;
298                        }
299                }
300
301                if((!auto)&&(!CMLib.flags().canSpeak(mob)))
302                {
303                        mob.tell(L("You can't speak!"));
304                        return false;
305                }
306
307                // if they can't hear the sleep spell, it
308                // won't happen
309                if((!auto)&&(!CMLib.flags().canBeHeardSpeakingBy(mob,target)))
310                {
311                        mob.tell(L("@x1 can't hear your words.",target.charStats().HeShe()));
312                        return false;
313                }
314
315                if(!super.invoke(mob,commands,givenTarget,auto,asLevel))
316                        return false;
317
318                int levelDiff=mob.phyStats().level()-target.phyStats().level();
319                if(levelDiff>0)
320                        levelDiff=(-(levelDiff*levelDiff))/(1+super.getXLEVELLevel(mob));
321                else
322                        levelDiff=(levelDiff*(-levelDiff))/(1+super.getXLEVELLevel(mob));
323
324                final boolean success=proficiencyCheck(mob,levelDiff,auto);
325                if(success)
326                {
327                        final CMMsg msg=CMClass.getMsg(mob,target,this,CMMsg.MSG_NOISYMOVEMENT|(auto?CMMsg.MASK_ALWAYS:0),L("<S-NAME> befriend(s) <T-NAME>."));
328                        if(mob.location().okMessage(mob,msg))
329                        {
330                                mob.location().send(mob,msg);
331                                CMLib.commands().postFollow(target,mob,false);
332                                CMLib.combat().makePeaceInGroup(mob);
333                                if(target.amFollowing()!=mob)
334                                        mob.tell(L("@x1 seems unwilling to be your friend.",target.name(mob)));
335                                else
336                                {
337                                        Ability A=super.beneficialAffect(mob, target, asLevel, Ability.TICKS_FOREVER);
338                                        if(A!=null)
339                                        {
340                                                A.makeNonUninvokable();
341                                                A.makeLongLasting();
342                                                A.setSavable(true);
343                                        }
344                                }
345                        }
346                }
347                else
348                        return beneficialVisualFizzle(mob,target,L("<S-NAME> attempt(s) to befriend <T-NAMESELF>, but fail(s)."));
349
350                return success;
351        }
