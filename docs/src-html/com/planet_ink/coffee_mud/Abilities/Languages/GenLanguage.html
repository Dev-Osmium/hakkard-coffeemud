058        private static final Hashtable<String,Object[]> vars=new Hashtable<String,Object[]>();
059        private static final int V_NAME=0;//S
060        private static final int V_WSETS=1;//L<S[]>
061        private static final int V_HSETS=2;//H<S,S>
062        private static final int V_HELP=3;//S
063        private static final int NUM_VS=4;//S
064
065        private static final Object[] makeEmpty()
066        {
067                final Object[] O=new Object[NUM_VS];
068                O[V_NAME]="a language";
069                O[V_WSETS]=new Vector<String[]>();
070                O[V_HSETS]=new Hashtable<String,String>();
071                O[V_HELP]="<ABILITY>This language is not yet documented.";
072                return O;
073        }
074
075        private static final Object V(final String ID, final int varNum)
076        {
077                if(vars.containsKey(ID))
078                        return vars.get(ID)[varNum];
079                final Object[] O=makeEmpty();
080                vars.put(ID,O);
081                return O[varNum];
082        }
083
084        private static final void SV(final String ID,final int varNum,final Object O)
085        {
086                if(vars.containsKey(ID))
087                        vars.get(ID)[varNum]=O;
088                else
089                {
090                        final Object[] O2=makeEmpty();
091                        vars.put(ID,O2);
092                        O2[varNum]=O;
093                }
094        }

174        @Override
175        @SuppressWarnings("unchecked")
176        public String getStat(String code)
177        {
178                int num=0;
179                int numDex=code.length();
180                while((numDex>0)&&(Character.isDigit(code.charAt(numDex-1))))
181                        numDex--;
182                if(numDex<code.length())
183                {
184                        num=CMath.s_int(code.substring(numDex));
185                        code=code.substring(0,numDex);
186                }
187                switch(getCodeNum(code))
188                {
189                case 0:
190                        return ID();
191                case 1:
192                        return text();
193                case 2:
194                        return (String) V(ID, V_NAME);
195                case 3:
196                        if (num == 0)
197                        {
198                                final List<String[]> words = (List<String[]>) V(ID, V_WSETS);
199                                final StringBuilder str = new StringBuilder("");
200                                for (final String[] wset : words)
201                                {
202                                        if (str.length() > 0)
203                                                str.append("/");
204                                        str.append(CMParms.toListString(wset));
205                                }
206                                return str.toString();
207                        }
208                        else
209                        if (num <= ((List<String[]>) V(ID, V_WSETS)).size())
210                                return CMParms.toListString(((List<String[]>) V(ID, V_WSETS)).get(num - 1));
211                        else
212                                return "";
213                case 4:
214                        return CMParms.toKeyValueSlashListString((Map<String, String>) V(ID, V_HSETS));
215                case 5:
216                        return (String) V(ID, V_HELP);
217                default:
218                        if(code.equalsIgnoreCase("allxml"))
219                                return getAllXML();
220                        break;
221                }
222                return "";
223        }

225        @Override
226        @SuppressWarnings("unchecked")
227        public void setStat(String code, final String val)
228        {
229                int num=0;
230                int numDex=code.length();
231                while((numDex>0)&&(Character.isDigit(code.charAt(numDex-1))))
232                        numDex--;
233                if(numDex<code.length())
234                {
235                        num=CMath.s_int(code.substring(numDex));
236                        code=code.substring(0,numDex);
237                }
238                switch(getCodeNum(code))
239                {
240                case 0:
241                        if(val.trim().length()>0)
242                        {
243                                V(ID,V_NAME); // force creation, if necc
244                                final Object[] O=vars.get(ID);
245                                vars.remove(ID);
246                                vars.put(val,O);
247                                if(num!=9)
248                                        CMClass.delClass(CMObjectType.ABILITY,this);
249                                ID=val;
250                                if(num!=9)
251                                        CMClass.addClass(CMObjectType.ABILITY,this);
252                        }
253                        break;
254                case 1:
255                        setMiscText(val);
256                        break;
257                case 2:
258                        SV(ID, V_NAME, val);
259                        if (ID.equalsIgnoreCase("GenLanguage"))
260                                break;
261                        break;
262                case 3:
263                        if (num == 0)
264                        {
265                                final String[] allSets = val.split("/");
266                                final List<String[]> wordSets = new Vector<String[]>();
267                                for (final String wordList : allSets)
268                                        wordSets.add(CMParms.parseCommas(wordList, true).toArray(new String[0]));
269                                SV(ID, V_WSETS, wordSets);
270                        }
271                        else
272                        if ((num == ((List<String[]>) V(ID, V_WSETS)).size()) && (val.length() == 0))
273                                ((List<String[]>) V(ID, V_WSETS)).remove(num - 1);
274                        else
275                        if (num <= ((List<String[]>) V(ID, V_WSETS)).size())
276                                ((List<String[]>) V(ID, V_WSETS)).set(num - 1, CMParms.parseCommas(val, true).toArray(new String[0]));
277                        else
278                        if ((num == ((List<String[]>) V(ID, V_WSETS)).size() + 1) && (val.length() > 0))
279                                ((List<String[]>) V(ID, V_WSETS)).add(CMParms.parseCommas(val, true).toArray(new String[0]));
280                        break;
281                case 4:
282                        SV(ID, V_HSETS, CMParms.parseKeyValueSlashListString(val));
283                        break;
284                case 5:
285                        SV(ID, V_HELP, val);
286                        break;
287                default:
288                        if(code.equalsIgnoreCase("allxml")&&ID.equalsIgnoreCase("GenLanguage"))
289                                parseAllXML(val);
290                        break;
291                }
292        }

294        @Override
295        public boolean sameAs(final Environmental E)
296        {
297                if(!(E instanceof GenLanguage))
298                        return false;
299                if(!((GenLanguage)E).ID().equals(ID))
300                        return false;
301                if(!((GenLanguage)E).text().equals(text()))
302                        return false;
303                return true;
304        }

306        private void parseAllXML(final String xml)
307        {
308                final List<XMLLibrary.XMLTag> V=CMLib.xml().parseAllXML(xml);
309                if((V==null)||(V.size()==0))
310                        return;
311                for(int c=0;c<getStatCodes().length;c++)
312                {
313                        if(getStatCodes()[c].equals("CLASS"))
314                                ID=CMLib.xml().restoreAngleBrackets(CMLib.xml().getValFromPieces(V, getStatCodes()[c]));
315                        else
316                        if(!getStatCodes()[c].equals("TEXT"))
317                                setStat(getStatCodes()[c],CMLib.xml().restoreAngleBrackets(CMLib.xml().getValFromPieces(V, getStatCodes()[c])));
318                }
319        }
320
321        private String getAllXML()
322        {
323                final StringBuffer str=new StringBuffer("");
324                for(int c=0;c<getStatCodes().length;c++)
325                {
326                        if(!getStatCodes()[c].equals("TEXT"))
327                        {
328                                str.append("<"+getStatCodes()[c]+">"
329                                                +CMLib.xml().parseOutAngleBrackets(getStat(getStatCodes()[c]))
330                                                +"</"+getStatCodes()[c]+">");
331                        }
332                }
333                return str.toString();
334        }
