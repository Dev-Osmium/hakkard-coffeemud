180        protected String fixCase(String like,String make)
181        {
182                final StringBuffer s=new StringBuffer(make);
183                char lastLike=' ';
184                for(int x=0;x<make.length();x++)
185                {
186                        if(x<like.length())
187                                lastLike=like.charAt(x);
188                        s.setCharAt(x,fixCase(lastLike,make.charAt(x)));
189                }
190                return s.toString();
191        }
192
193        protected char fixCase(char like,char make)
194        {
195                if(Character.isUpperCase(like))
196                        return Character.toUpperCase(make);
197                return Character.toLowerCase(make);
198        }

200        @Override
201        public String translate(String language, String word)
202        {
203                if(translationHash(language).containsKey(word.toUpperCase()))
204                        return fixCase(word,translationHash(language).get(word.toUpperCase()));
205                final MOB M=CMLib.players().getPlayerAllHosts(word);
206                if(M!=null)
207                        return word;
208                final List<String[]> translationVector=translationLists(language);
209                if(translationVector.size()>0)
210                {
211                        String[] choices=null;
212                        try
213                        {
214                                choices = translationVector.get(word.length() - 1);
215                        }
216                        catch (final Exception e)
217                        {
218                        }
219                        if(choices==null)
220                                choices=translationVector.get(translationVector.size()-1);
221                        return choices[CMath.abs(word.toLowerCase().hashCode()) % choices.length];
222                }
223                return word;
224        }

226        protected int numChars(String words)
227        {
228                int num=0;
229                for(int i=0;i<words.length();i++)
230                {
231                        if(Character.isLetter(words.charAt(i)))
232                                num++;
233                }
234                return num;
235        }

237        public String messChars(String language, String words, int numToMess)
238        {
239                numToMess=numToMess/2;
240                if(numToMess==0)
241                        return words;
242                final StringBuffer w=new StringBuffer(words);
243                while(numToMess>0)
244                {
245                        final int x=CMLib.dice().roll(1,words.length(),-1);
246                        final char c=words.charAt(x);
247                        if(Character.isLetter(c))
248                        {
249                                if(vowels.indexOf(c)>=0)
250                                        w.setCharAt(x,fixCase(c,vowels.charAt(CMLib.dice().roll(1,vowels.length(),-1))));
251                                else
252                                        w.setCharAt(x,fixCase(c,consonants.charAt(CMLib.dice().roll(1,consonants.length(),-1))));
253                                numToMess--;
254                        }
255                }
256                return w.toString();
257        }

259        public String scrambleAll(String language, String str, int numToMess)
260        {
261                final StringBuffer newStr=new StringBuffer("");
262                int start=0;
263                int end=0;
264                int state=-1;
265                while(start<=str.length())
266                {
267                        char c='\0';
268                        if(end>=str.length())
269                                c=' ';
270                        else
271                                c=str.charAt(end);
272                        switch(state)
273                        {
274                        case -1:
275                                if(Character.isLetter(c))
276                                {
277                                        state = 0;
278                                        end++;
279                                }
280                                else
281                                {
282                                        newStr.append(c);
283                                        end++;
284                                        start = end;
285                                }
286                                break;
287                        case 0:
288                                if(Character.isLetter(c))
289                                        end++;
290                                else
291                                if(Character.isDigit(c))
292                                {
293                                        newStr.append(str.substring(start, end + 1));
294                                        end++;
295                                        start = end;
296                                        state = 1;
297                                }
298                                else
299                                {
300                                        newStr.append(translate(language, str.substring(start, end)) + c);
301                                        end++;
302                                        start = end;
303                                        state = -1;
304                                }
305                                break;
306                        case 1:
307                                if(Character.isLetterOrDigit(c))
308                                {
309                                        newStr.append(c);
310                                        end++;
311                                        start = end;
312                                }
313                                else
314                                {
315                                        newStr.append(c);
316                                        end++;
317                                        start = end;
318                                        state = -1;
319                                }
320                                break;
321                        }
322                }
323                return newStr.toString();
324        }

326        protected Language getMyTranslator(String id, Physical P, Language winner)
327        {
328                if(P==null)
329                        return winner;
330                for(final Enumeration<Ability> a=P.effects();a.hasMoreElements();)
331                {
332                        final Ability A=a.nextElement();
333                        if((A instanceof Language)
334                        && ((Language)A).translatesLanguage(id)
335                        && ((winner==null)
336                                ||((Language)A).getProficiency(id) > winner.getProficiency(id)))
337                        {
338                                winner = (Language)A;
339                        }
340                }
341                return winner;
342        }

344        protected Language getAnyTranslator(String id, MOB mob)
345        {
346                Language winner = null;
347                winner = getMyTranslator(id,mob,winner);
348                winner = getMyTranslator(id,mob.location(),winner);
349                for(int i=0;i<mob.numItems();i++)
350                        winner=getMyTranslator(id,mob.getItem(i),winner);
351                return winner;

354        protected boolean processSourceMessage(CMMsg msg, String str, int numToMess)
355        {
356                String smsg=CMStrings.getSayFromMessage(msg.sourceMessage());
357                if(smsg != null)
358                {
359                        if(numToMess>0)
360                                smsg=messChars(ID(),smsg,numToMess);
361                        msg.modify(msg.source(),
362                                           msg.target(),
363                                           this,
364                                           msg.sourceCode(),
365                                           CMStrings.substituteSayInMessage(msg.sourceMessage(),smsg),
366                                           msg.targetCode(),
367                                           msg.targetMessage(),
368                                           msg.othersCode(),
369                                           msg.othersMessage());
370                }
371                return true;
372        }

407        @Override
408        public boolean okMessage(final Environmental myHost, final CMMsg msg)
409        {
410                if((affected instanceof MOB)
411                &&(beingSpoken(ID())))
412                {
413                        if((msg.source()==affected)
414                        &&(msg.sourceMessage()!=null)
415                        &&(msg.tool()==null)
416                        &&((msg.sourceMinor()==CMMsg.TYP_SPEAK)
417                           ||(msg.sourceMinor()==CMMsg.TYP_TELL)
418                           ||(CMath.bset(msg.sourceMajor(),CMMsg.MASK_CHANNEL))))
419                        {
420                                String str=CMStrings.getSayFromMessage(msg.othersMessage());
421                                if(str==null)
422                                        str=CMStrings.getSayFromMessage(msg.targetMessage());
423                                if(str!=null)
424                                {
425                                        final int numToMess=(int)Math.round(CMath.mul(numChars(str),CMath.div(100-getProficiency(ID()),100)));
426                                        if(!processSourceMessage(msg, str, numToMess))
427                                                return false;
428                                        if(!processNonSourceMessages(msg,str,numToMess))
429                                                return false;
430                                        if(CMLib.flags().isAliveAwakeMobile((MOB)affected,true))
431                                                helpProficiency((MOB)affected, 0);
432                                }
433                        }
434                        else
435                        if((msg.sourceMinor()==CMMsg.TYP_WRITE)
436                        &&(msg.source()==affected)
437                        &&(msg.target() instanceof Item)
438                        &&(((Item)msg.target()).isReadable())
439                        &&(msg.targetMessage()!=null)
440                        &&(msg.targetMessage().length()>0))
441                        {
442                                if(!tryLinguisticWriting(msg))
443                                        return false;
444                        }
445                        else
446                        if((msg.target()==affected)
447                        &&(msg.source()!=affected)
448                        &&(msg.sourceMinor()!=CMMsg.NO_EFFECT))
449                        {
450                                switch(msg.targetMinor())
451                                {
452                                case CMMsg.TYP_ORDER:
453                                case CMMsg.TYP_BUY:
454                                case CMMsg.TYP_BID:
455                                case CMMsg.TYP_SELL:
456                                case CMMsg.TYP_LIST:
457                                case CMMsg.TYP_VIEW:
458                                case CMMsg.TYP_WITHDRAW:
459                                case CMMsg.TYP_DEPOSIT:
460                                {
461                                        // yes, this means that a mob speaking Common to a marketing player will get failed,
462                                        // however, remember that the LISTer language doesn't matter, only the responding (this) language.
463                                        // also, think about muds where there is no Common (an interesting mud!)
464                                        if((!CMSecurity.isAllowed(msg.source(),msg.source().location(),CMSecurity.SecFlag.ORDER))
465                                        &&(!CMSecurity.isAllowed(msg.source(),msg.source().location(),CMSecurity.SecFlag.CMDMOBS)||(!((MOB)msg.target()).isMonster()))
466                                        &&(!CMSecurity.isAllowed(msg.source(),msg.source().location(),CMSecurity.SecFlag.CMDROOMS)||(!((MOB)msg.target()).isMonster())))
467                                        {
468                                                final Language L;
469                                                if((msg.tool() instanceof Language)&&(msg.tool().ID().equals(ID())))
470                                                        L=(Language)msg.tool();
471                                                else
472                                                        L=getAnyTranslator(ID(),msg.source());
473                                                if((L==null)
474                                                ||(!L.beingSpoken(ID()))
475                                                ||((CMLib.dice().rollPercentage()*2)>(L.getProficiency(ID())+getProficiency(ID()))))
476                                                {
477                                                        msg.setTargetCode(CMMsg.TYP_SPEAK);
478                                                        msg.setSourceCode(CMMsg.TYP_SPEAK);
479                                                        msg.setOthersCode(CMMsg.TYP_SPEAK);
480                                                        String reply=null;
481                                                        if((L==null)||(!L.beingSpoken(ID())))
482                                                                reply="<S-NAME> <S-IS-ARE> speaking "+name()+" and <T-NAME> would not understand <S-HIM-HER>.";
483                                                        else
484                                                                reply="<T-NAME> <T-IS-ARE> having trouble understanding <S-YOUPOSS> pronunciation.";
485                                                        msg.addTrailerMsg(CMClass.getMsg((MOB)msg.target(),msg.source(),null,CMMsg.MSG_OK_VISUAL,reply));
486                                                }
487                                        }
488                                        break;
489                                }
490                                default:
491                                        break;
492                                }
493                        }
494                }
495                if((affected instanceof Item)
496                &&(!canBeUninvoked())
497                &&(msg.target()==affected)
498                &&(msg.targetMinor()==CMMsg.TYP_WASREAD)
499                &&(msg.othersCode()==CMMsg.NO_EFFECT)
500                &&((msg.othersMessage()==null)||(!msg.othersMessage().equals(CANCEL_WORD)))
501                &&(!(affected instanceof LandTitle))
502                &&(CMLib.flags().canBeSeenBy(this,msg.source()))
503                &&(msg.targetMessage()!=null)
504                &&(msg.targetMessage().length()>0)
505                &&(((Item)affected).isReadable())
506                )
507                {
508                        String str=msg.targetMessage();
509                        int numToMess=numChars(str);
510                        if(numToMess>0)
511                        {
512                                final Language L=(Language)msg.source().fetchEffect(ID());
513                                if(L!=null)
514                                        numToMess=(int)Math.round(CMath.mul(numChars(str),CMath.div(100-L.getProficiency(ID()),100)));
515                                final String original=messChars(ID(),str,numToMess);
516                                str=scrambleAll(ID(),str,numToMess);
517                                msg.setSourceMessage(L("It says '@x1'.  ",str.trim()));
518                                if((L!=null)&&(!original.equals(str)))
519                                {
520                                        msg.setSourceMessage(msg.sourceMessage()+(L("\n\rIt says '@x1' (translated from @x2).",original,L.writtenName())));
521                                        msg.setTargetMessage(original);
522                                }
523                        }
524                }
525                return super.okMessage(myHost,msg);
526        }

528        @Override
529        public boolean canBeLearnedBy(MOB teacher, MOB student)
530        {
531                if(!super.canBeLearnedBy(teacher,student))
532                        return false;
533                if(student==null)
534                        return true;
535                final AbilityComponents.AbilityLimits remainders = CMLib.ableComponents().getSpecialSkillRemainder(student, this);
536                if(remainders.languageSkills()<=0)
537                {
538                        if(teacher != null)
539                                teacher.tell(L("@x1 can not learn any more languages.",student.name(teacher)));
540                        student.tell(L("You have learned the maximum @x1 languages, and may not learn any more.",""+remainders.maxLanguageSkills()));
541                        return false;
542                }
543                return true;
544        }

561        @Override
562        public boolean invoke(MOB mob, List<String> commands, Physical givenTarget, boolean auto, int asLevel)
563        {
564                if(!auto)
565                {
566                        boolean alreadySpeaking=false;
567                        boolean found=false;
568                        for(final Enumeration<Ability> a=mob.effects();a.hasMoreElements();)
569                        {
570                                final Ability A=a.nextElement();
571                                if((A!=null)&&(A instanceof Language))
572                                {
573                                        if(mob.isMonster())
574                                                A.setProficiency(100);
575                                        if(A.ID().equals(ID()))
576                                        {
577                                                found=true;
578                                                alreadySpeaking = ((Language)A).beingSpoken(A.ID());
579                                                ((Language)A).setBeingSpoken(ID(),true);
580                                        }
581                                        else
582                                                ((Language)A).setBeingSpoken(ID(),false);
583                                }
584                        }
585                        isAnAutoEffect=false;
586                        if(found)
587                        {
588                                if(alreadySpeaking)
589                                        mob.tell(L("You were already speaking @x1.",name()));
590                                else
591                                        mob.tell(L("You are now speaking @x1.",name()));
592                        }
593                        else
594                                mob.tell(L("You are now speaking Common."));
595                }
596                else
597                        setBeingSpoken(ID(),true);
598                return true;
599        }

602        {
603                if((msg.othersMessage()!=null)&&(msg.othersMessage().indexOf('\'')>0))
604                {
605                        String otherMes=msg.othersMessage();
606                        if(msg.target()!=null)
607                                otherMes=CMLib.coffeeFilter().fullOutFilter(null,(MOB)affected,msg.source(),msg.target(),msg.tool(),otherMes,false);
608                        msg.addTrailerMsg(CMClass.getMsg(msg.source(),affected,null,CMMsg.NO_EFFECT,null,msg.othersCode(),L("@x1 (translated from @x2)",CMStrings.substituteSayInMessage(otherMes,sourceWords),name()),CMMsg.NO_EFFECT,null));
609                        return true;
610                }
611                return false;
612        }

637        @Override
638        public void executeMsg(final Environmental myHost, final CMMsg msg)
639        {
640                super.executeMsg(myHost,msg);
641
642                if((affected instanceof MOB)
643                &&(!msg.amISource((MOB)affected))
644                &&((msg.sourceMinor()==CMMsg.TYP_SPEAK)
645                   ||(msg.sourceMinor()==CMMsg.TYP_TELL)
646                   ||(CMath.bset(msg.sourceMajor(),CMMsg.MASK_CHANNEL)))
647                &&(msg.sourceMessage()!=null)
648                &&(msg.tool() instanceof Language)
649                &&(msg.tool().ID().equals(ID())))
650                {
651                        String str=CMStrings.getSayFromMessage(msg.sourceMessage());
652                        if(str!=null)
653                        {
654                                final int numToMess=(int)Math.round(CMath.mul(numChars(str),CMath.div(100-getProficiency(ID()),100)));
655                                if(numToMess>0)
656                                        str=messChars(ID(),str,numToMess);
657                                if(!translateChannelMessage(msg,str))
658                                {
659                                        if(!translateTargetMessage(msg,str))
660                                                translateOthersMessage(msg, str);
661                                }
662                        }
663                }
664                else
665                if((affected instanceof MOB)
666                &&(msg.source()==affected)
667                &&(beingSpoken(ID()))
668                &&(msg.target() instanceof Item)
669                &&(msg.sourceMinor()==CMMsg.TYP_WRITE)
670                &&(((Item)msg.target()).isReadable())
671                &&(msg.targetMessage()!=null)
672                &&((msg.targetMessage().length()>0)||(msg.target().ID().endsWith("Book"))))
673                {
674                        final Item I = (Item)msg.target();
675                        Ability L=null;
676                        for(int i=I.numEffects()-1;i>=0;i--) // reverse enumeration
677                        {
