081
082        @Override
083        public boolean translatesLanguage(String language)
084        {
085                return ID().equalsIgnoreCase(language)
086                                || "Dwarven".equalsIgnoreCase(language)
087                                || "Goblinese".equalsIgnoreCase(language)
088                                || "Drowish".equalsIgnoreCase(language)
089                                || "Gnomish".equalsIgnoreCase(language);
090        }
091
092        @Override
093        public int getProficiency(String language) 
094        {
095                if(ID().equalsIgnoreCase(language))
096                        return proficiency();
097                if(translatesLanguage(language))
098                        return proficiency() / 5;
099                return 0;
100        }
101
102        @Override
103        public Map<String, String> translationHash(String language)
104        {
105                return drowish.translationHash(drowish.ID());
106        }
