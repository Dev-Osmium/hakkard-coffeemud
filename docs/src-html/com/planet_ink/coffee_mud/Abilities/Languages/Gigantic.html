080        public String tup(String msg)
081        {
082                if(msg==null)
083                        return msg;
084                return msg.toUpperCase();
085        }
086
087        @Override
088        public boolean okMessage(final Environmental myHost, final CMMsg msg)
089        {
090                if((beingSpoken(ID()))
091                &&(affected instanceof MOB)
092                &&(msg.amISource((MOB)affected))
093                &&(msg.sourceMessage()!=null)
094                &&(msg.tool()==null)
095                &&((msg.sourceMinor()==CMMsg.TYP_SPEAK)
096                   ||(msg.sourceMinor()==CMMsg.TYP_TELL)
097                   ||(CMath.bset(msg.sourceMajor(),CMMsg.MASK_CHANNEL))))
098                {
099                        msg.modify(msg.source(),msg.target(),msg.tool(),
100                                           msg.sourceCode(),tup(msg.sourceMessage()),
101                                           msg.targetCode(),tup(msg.targetMessage()),
102                                           msg.othersCode(),tup(msg.othersMessage()));
103                }
104                return super.okMessage(myHost,msg);
105        }
