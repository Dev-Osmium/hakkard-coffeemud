037public class Paladin_PaladinsMount extends PaladinSkill
038{
039        @Override
040        public String ID()
041        {
042                return "Paladin_PaladinsMount";
043        }
044
045        private final static String localizedName = CMLib.lang().L("Paladin`s Mount");
046
047        @Override
048        public String name()
049        {
050                return localizedName;
051        }
052
053        @Override
054        public int classificationCode()
055        {
056                return Ability.ACODE_SKILL|Ability.DOMAIN_ANIMALAFFINITY;
057        }
058        
059        protected boolean pass=false;
060        
061        public Paladin_PaladinsMount()
062        {
063                super();
064        }
065
066        @Override
067        public boolean tick(final Tickable ticking, final int tickID)
068        {
069                if(!super.tick(ticking,tickID))
070                        return false;
071                final Physical affected=super.affected;
072                final MOB invoker = super.invoker();
073                if((affected instanceof Rideable)
074                &&(affected instanceof MOB)
075                &&(invoker!=null)
076                &&(affected != invoker))
077                {
078                        final MOB affectedMob=(MOB)affected;
079                        if((!((Rideable)affected).amRiding(invoker))
080                        ||(affectedMob.amDead())
081                        ||(CMLib.flags().isEvil(affectedMob))
082                        ||(!CMLib.flags().isGood(invoker)))
083                        {
084                                affected.delEffect(this);
085                                affected.recoverPhyStats();
086                                affectedMob.recoverCharStats();
087                                affectedMob.recoverMaxState();
088                        }
089                }
090                else
091                if((affected == invoker)
092                &&(invoker!=null)
093                &&(CMLib.flags().isGood(invoker))
094                &&(invoker.riding()!=null)
095                &&(invoker.riding() instanceof MOB)
096                &&(((MOB)invoker.riding()).charStats().getMyRace().racialCategory().equals("Equine"))
097                &&(!CMLib.flags().isEvil(invoker.riding()))
098                &&(invoker.riding().fetchEffect(ID())==null))
099                {
100                        invoker.riding().addEffect((Paladin_PaladinsMount)copyOf());
101                        invoker.riding().recoverPhyStats();
102                        ((MOB)invoker.riding()).recoverCharStats();
103                        ((MOB)invoker.riding()).recoverMaxState();
104                }
105                return true;
106        }
107
108        @Override
109        public void affectCharStats(final MOB affected, final CharStats affectableStats)
110        {
111                super.affectCharStats(affected,affectableStats);
112                if((invoker!=null)
113                &&(affected!=invoker))
114                {
115                        affectableStats.setStat(CharStats.STAT_SAVE_DISEASE,affectableStats.getStat(CharStats.STAT_SAVE_DISEASE)+50+proficiency()+(5*getXLEVELLevel(invoker())));
116                        affectableStats.setStat(CharStats.STAT_SAVE_POISON,affectableStats.getStat(CharStats.STAT_SAVE_POISON)+50+proficiency()+(5*getXLEVELLevel(invoker())));
117                        final int amount=(int)Math.round(CMath.mul(CMath.div(proficiency(),100.0),affected.phyStats().level()+(2*getXLEVELLevel(invoker))));
118                        for(final int i : CharStats.CODES.SAVING_THROWS())
119                                affectableStats.setStat(i,affectableStats.getStat(i)+amount);
120                }
121        }
122        
123        @Override
124        public boolean okMessage(final Environmental myHost, final CMMsg msg)
125        {
126                if(!super.okMessage(myHost,msg))
127                        return false;
128                if((invoker==null)
129                ||(!(CMLib.flags().isGood(invoker)))
130                ||(affected==null)
131                ||(affected==invoker)
132                ||(!(affected instanceof MOB)))
133                        return true;
134
135                if(msg.target()==affected)
136                {
137                        if(msg.tool() instanceof Ability)
138                        {
139                                if((CMath.bset(msg.targetMajor(),CMMsg.MASK_MALICIOUS))
140                                &&(msg.targetMinor()==CMMsg.TYP_CAST_SPELL)
141                                &&(!CMath.bset(((Ability)msg.tool()).flags(),Ability.FLAG_HOLY))
142                                &&(CMath.bset(((Ability)msg.tool()).flags(),Ability.FLAG_UNHOLY))
143                                &&(proficiencyCheck(invoker,0,false)))
144                                {
145                                        msg.source().location().show((MOB)msg.target(),null,CMMsg.MSG_OK_VISUAL,L("The holy field around <S-NAME> protect(s) <S-HIM-HER> from the evil magic attack of @x1.",msg.source().name()));
146                                        return false;
147                                }
148                                final String str1=msg.tool().ID().toUpperCase();
149                                if(((str1.indexOf("SPOOK")>=0)||(str1.indexOf("NIGHTMARE")>=0)||(str1.indexOf("FEAR")>=0))
150                                &&(proficiencyCheck(invoker,0,false)))
151                                {
152                                        final MOB mob=(MOB)msg.target();
153                                        mob.location().show(invoker,mob,CMMsg.MSG_OK_VISUAL,L("<T-YOUPOSS> courage protects you from the @x1 attack.",msg.tool().name()));
154                                        return false;
155                                }
156                        }
157                        if(((msg.targetMinor()==CMMsg.TYP_POISON)||(msg.targetMinor()==CMMsg.TYP_DISEASE))&&(proficiencyCheck(invoker,0,false)))
158                                return false;
159                }
160                return true;
161        }
