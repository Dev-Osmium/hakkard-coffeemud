128        public int baseRejuvTime(int level)
129        {
130                if(level>=30)
131                        level=29;
132                int ticks=(int)Math.round((30.0-(CMath.mul(level,.75)))*30.0);
133                if(ticks<1)
134                        ticks=1;
135                return ticks;
136        }

156        public boolean doesSaveVsTraps(MOB target)
157        {
158                int save=target.charStats().getSave(CharStats.STAT_SAVE_TRAPS);
159                if(invoker()!=null)
160                {
161                        save += target.phyStats().level();
162                        save -= invoker().phyStats().level();
163                }
164                return (CMLib.dice().rollPercentage()<=save);
165        }

167        public boolean isLocalExempt(MOB target)
168        {
169                if(target==null)
170                        return false;
171                final Room R=target.location();
172                if((!canBeUninvoked())
173                &&(!isABomb())
174                &&(R!=null))
175                {
176                        if((CMLib.law().getLandTitle(R)!=null)
177                        &&(CMLib.law().doesHavePriviledgesHere(target,R)))
178                                return true;
179
180                        if((target.isMonster())
181                        &&(target.getStartRoom()!=null)
182                        &&(target.getStartRoom().getArea()==R.getArea()))
183                                return true;
184                }
185                return false;
186        }

249        @Override
250        public void setMiscText(String text)
251        {
252                text=text.trim();
253                if(text.startsWith("`"))
254                {
255                        final int x=text.indexOf("` ",1);
256                        if(x>=0)
257                        {
258                                invokerName=text.substring(1,x);
259                                text=text.substring(x+2).trim();
260                        }
261                }
262                while(text.startsWith("\""))
263                {
264                        final int x=text.indexOf("\"",1);
265                        if(x>=0)
266                        {
267                                newMessaging.add(text.substring(1,x));
268                                text=text.substring(x+1).trim();
269                        }
270                        else
271                                break;
272                }
273                if(text.startsWith(":"))
274                {
275                        final int x=text.indexOf(':');
276                        final int y=text.indexOf(':',x+1);
277                        if((x>=0)&&(y>x)&&(CMath.isInteger(text.substring(x+1,y).trim())))
278                        {
279                                setAbilityCode(CMath.s_int(text.substring(x+1,y).trim()));
280                                text=text.substring(y+1).trim();
281                        }
282                }
283                super.setMiscText(text);
284        }

307        @Override
308        public boolean okMessage(final Environmental myHost, final CMMsg msg)
309        {
310                if((!disabled())&&(affected instanceof Item))
311                {
312                        if((msg.tool()==affected)
313                        &&(msg.targetMinor()==CMMsg.TYP_GIVE)
314                        &&(msg.targetMessage()!=null)
315                        &&(msg.target() instanceof MOB)
316                        &&(!msg.source().getGroupMembers(new HashSet<MOB>()).contains(msg.target())))
317                        {
318                                msg.source().tell((MOB)msg.target(),msg.tool(),null,L("<S-NAME> can't accept <T-NAME>."));
319                                return false;
320                        }
321                }
322                if((!sprung)
323                && CMath.bset(canAffectCode(),Ability.CAN_ROOMS)
324                && getTravelThroughFlag()
325                && msg.amITarget(affected)
326                && (affected instanceof Room)
327                && (msg.tool() instanceof Exit))
328                {
329                        final Room room=(Room)affected;
330                        if ((msg.targetMinor()==CMMsg.TYP_LEAVE)||(msg.targetMinor()==CMMsg.TYP_FLEE))
331                        {
332                                final int movingInDir=CMLib.map().getExitDir(room, (Exit)msg.tool());
333                                if((movingInDir!=Directions.DOWN)&&(movingInDir!=Directions.UP))
334                                {
335                                        final PairVector<MOB,Integer> safeDirs=getSafeDirs();
336                                        synchronized(safeDirs)
337                                        {
338                                                for(final Iterator<Pair<MOB,Integer>> i=safeDirs.iterator();i.hasNext();)
339                                                {
340                                                        final Pair<MOB,Integer> p=i.next();
341                                                        if(p.first == msg.source())
342                                                        {
343                                                                i.remove();
344                                                                if(movingInDir==p.second.intValue())
345                                                                        return true;
346                                                                spring(msg.source());
347                                                                return !sprung();
348                                                        }
349                                                }
350                                        }
351                                }
352                        }
353                }
354                return super.okMessage(myHost,msg);
355        }

369        @Override
370        public void executeMsg(final Environmental myHost, final CMMsg msg)
371        {
372                if(!sprung)
373                if(CMath.bset(canAffectCode(),Ability.CAN_EXITS))
374                {
375                        if(msg.amITarget(affected))
376                        {
377                                if((affected instanceof Exit)
378                                &&(((Exit)affected).hasADoor())
379                                &&(((Exit)affected).hasALock())
380                                &&(((Exit)affected).isLocked()))
381                                {
382                                        if(msg.targetMinor()==CMMsg.TYP_UNLOCK)
383                                                spring(msg.source());
384                                }
385                                else
386                                if((affected instanceof Container)
387                                &&(((Container)affected).hasADoor())
388                                &&(((Container)affected).hasALock())
389                                &&(((Container)affected).isLocked()))
390                                {
391                                        if(msg.targetMinor()==CMMsg.TYP_UNLOCK)
392                                                spring(msg.source());
393                                }
394                                else
395                                if(msg.targetMinor()==CMMsg.TYP_OPEN)
396                                        spring(msg.source());
397                        }
398                }
399                else
400                if(CMath.bset(canAffectCode(),Ability.CAN_ITEMS))
401                {
402                        if(isABomb())
403                        {
404                                if(msg.amITarget(affected))
405                                {
406                                        if((msg.targetMinor()==CMMsg.TYP_HOLD)
407                                        &&(msg.source().isMine(affected)))
408                                        {
409                                                msg.source().tell(msg.source(),affected,null,L("You activate <T-NAME>."));
410                                                activateBomb();
411                                        }
412                                }
413                        }
414                        else
415                        if(msg.amITarget(affected))
416                        {
417                                if(((msg.targetMinor()==CMMsg.TYP_GET)||(msg.targetMinor()==CMMsg.TYP_PUSH)||(msg.targetMinor()==CMMsg.TYP_PULL))
418                                &&(!msg.source().isMine(affected)))
419                                        spring(msg.source());
420                        }
421                }
422                else
423                if(CMath.bset(canAffectCode(), Ability.CAN_ROOMS)
424                && msg.amITarget(affected)
425                &&(msg.targetMinor()==CMMsg.TYP_ENTER))
426                {
427                        if(getTravelThroughFlag())
428                        {
429                                if ((affected instanceof Room)
430                                && (msg.tool() instanceof Exit))
431                                {
432                                        final Room room=(Room)affected;
433                                        final int movingInDir=CMLib.map().getExitDir(room, (Exit)msg.tool());
434                                        if((movingInDir!=Directions.DOWN)&&(movingInDir!=Directions.UP))
435                                        {
436                                                final PairVector<MOB,Integer> safeDirs=getSafeDirs();
437                                                synchronized(safeDirs)
438                                                {
439                                                        final int dex=safeDirs.indexOfFirst(msg.source());
440                                                        if(dex>=0)
441                                                                safeDirs.remove(dex);
442                                                        while(safeDirs.size()>room.numInhabitants()+1)
443                                                                safeDirs.remove(0);
444                                                        safeDirs.add(new Pair<MOB,Integer>(msg.source(),Integer.valueOf(movingInDir)));
445                                                }
446                                        }
447                                }
448                        }
449                        else
450                        if(!msg.source().isMine(affected))
451                                spring(msg.source());
452                }
453                super.executeMsg(myHost,msg);
454        }

456        @Override
457        public boolean maySetTrap(MOB mob, int asLevel)
458        {
459                if(mob==null)
460                        return false;
461                if(trapLevel()<0)
462                        return false;
463                if(asLevel<0)

466                        return true;

470        @Override
471        public boolean canReSetTrap(MOB mob)
472        {
473                final Physical P=affected;
474                if(P==null)
475                {
476                        if(mob!=null)
477                                mob.tell(L("This trap is not presently set."));
478                        return false;
479                }
480                if(mob!=null)
481                {
482                        if((!maySetTrap(mob,mob.phyStats().level()))
483                        &&(!mob.charStats().getCurrentClass().leveless())
484                        &&(!CMSecurity.isDisabled(CMSecurity.DisFlag.LEVELS)))
485                        {
486                                mob.tell(L("You are not high enough level (@x1) to set that trap.",""+trapLevel()));
487                                return false;
488                        }
489                }
490                final Trap T=(Trap)P.fetchEffect(ID());
491                if(T!=this)
492                {
493                        if(mob!=null)
494                                mob.tell(L("This trap is not presently set on @x1.",P.name()));
495                        return false;
496                }
497                
498                if(T.invoker() != mob)
499                {
500                        if(mob!=null)
501                                mob.tell(L("The trap was not set by you."));
502                        return false;
503                }
504                
505                return true;
506        }

508        @Override
509        public boolean canSetTrapOn(MOB mob, Physical P)
510        {
511                if(mob!=null)
512                {
513                        if((!maySetTrap(mob,mob.phyStats().level()))
514                        &&(!mob.charStats().getCurrentClass().leveless())
515                        &&(!CMSecurity.isDisabled(CMSecurity.DisFlag.LEVELS)))
516                        {
517                                mob.tell(L("You are not high enough level (@x1) to set that trap.",""+trapLevel()));
518                                return false;
519                        }
520                }
521                if(P.fetchEffect(ID())!=null)
522                {
523                        if(mob!=null)
524                                mob.tell(L("This trap is already set on @x1.",P.name()));
525                        return false;
526                }
527                if(!canAffect(P))
528                {
529                        if(mob!=null)
530                                mob.tell(L("You can't set '@x1' on @x2.",name(),P.name()));
531                        return false;
532                }
533                if((canAffectCode()&Ability.CAN_EXITS)==Ability.CAN_EXITS)
534                {
535                        if((P instanceof Item)&&(!(P instanceof Container)))
536                        {
537                                if(mob!=null)
538                                        mob.tell(L("@x1 has no lid, so '@x2' cannot be set on it.",P.name(),name()));
539                                return false;
540                        }
541                        if(((P instanceof Exit)&&(!(((Exit)P).hasADoor()))))
542                        {
543                                if(mob!=null)
544                                        mob.tell(L("@x1 has no door, so '@x2' cannot be set on it.",P.name(),name()));
545                                return false;
546                        }
547                        if(((P instanceof Container)&&(!(((Container)P).hasADoor()))))
548                        {
549                                if(mob!=null)
550                                        mob.tell(L("@x1 has no lid, so '@x2' cannot be set on it.",P.name(),name()));
551                                return false;
552                        }
553                }
554                return true;
555        }

563        @Override
564        public Trap setTrap(MOB mob, Physical P, int trapBonus, int qualifyingClassLevel, boolean perm)
565        {
566                if(P==null)
567                        return null;
568                final int rejuv=baseRejuvTime(qualifyingClassLevel+trapBonus);
569                final Trap T=(Trap)copyOf();
570                T.setReset(rejuv);
571                T.setInvoker(mob);
572                T.setSavable(false);
573                T.setAbilityCode(trapBonus);
574                P.addEffect(T);
575                if(perm)
576                {
577                        T.setSavable(true);
578                        T.makeNonUninvokable();
579                }
580                else
581                if(!isABomb())
582                        CMLib.threads().startTickDown(T,Tickable.TICKID_TRAP_DESTRUCTION,baseDestructTime(qualifyingClassLevel+trapBonus));
583                return T;
584        }

594        @Override
595        public boolean tick(final Tickable ticking, final int tickID)
596        {
597                if((unInvoked)&&(canBeUninvoked()))
598                        return false;
599
600                if(tickID==Tickable.TICKID_TRAP_DESTRUCTION)
601                {
602                        if(canBeUninvoked())
603                                disable();
604                        return false;
605                }
606                else
607                if((tickID==Tickable.TICKID_TRAP_RESET)&&(getReset()>0))
608                {
609                        if((--tickDown)<=0)
610                        {
611                                if((isABomb())
612                                &&(affected instanceof Item)
613                                &&(((Item)affected).owner()!=null))
614                                {
615                                        final Item I=(Item)affected;
616                                        if(I.owner() instanceof MOB)
617                                                spring((MOB)I.owner());
618                                        else
619                                        if(I.owner() instanceof Room)
620                                        {
621                                                final Room R=(Room)I.owner();
622                                                for(int i=R.numInhabitants()-1;i>=0;i--)
623                                                {
624                                                        final MOB M=R.fetchInhabitant(i);
625                                                        if(M!=null)
626                                                                spring(M);
627                                                }
628                                        }
629                                        disable();
630                                        unInvoke();
631                                        I.destroy();
632                                        return false;
633                                }
634                                sprung=false;
635                                disabled=false;
636                                return false;
637                        }
638                }
639                return true;
640        }

690        @Override
691        public boolean invoke(MOB mob, List<String> commands, Physical givenTarget, boolean auto, int asLevel)
692        {
693                final MOB target=super.getTarget(mob, commands, givenTarget);
694                if(target == null)
695                        return false;
696                if(!super.proficiencyCheck(mob, 0, auto))
697                        return true;
698                if(!super.invoke(mob, commands, target, auto, asLevel))
699                        return false;
700                StdTrap T=(StdTrap)copyOf();
701                T.setInvoker(mob);
702                T.setAffectedOne(mob);
703                T.spring(target);
704                return true;
705        }
