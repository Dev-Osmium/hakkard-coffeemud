036public class Trap_WeaponSwing extends StdTrap
037{
038        @Override
039        public String ID()
040        {
041                return "Trap_WeaponSwing";
042        }
043
044        private final static String     localizedName   = CMLib.lang().L("weapon swing");
045
046        @Override
047        public String name()
048        {
049                return localizedName;
050        }
051
052        @Override
053        protected int canAffectCode()
054        {
055                return Ability.CAN_EXITS | Ability.CAN_ITEMS;
056        }
057
058        @Override
059        protected int canTargetCode()
060        {
061                return 0;
062        }
063
064        @Override
065        protected int trapLevel()
066        {
067                return 9;
068        }
069
070        @Override
071        public String requiresToSet()
072        {
073                return "a melee weapon";
074        }
075
076        protected Item getPoison(MOB mob)
077        {
078                if(mob==null)
079                        return null;
080                if(mob.location()==null)
081                        return null;
082                for(int i=0;i<mob.location().numItems();i++)
083                {
084                        final Item I=mob.location().getItem(i);
085                        if((I!=null)
086                        &&(I instanceof Weapon)
087                        &&(((Weapon)I).weaponClassification()!=Weapon.CLASS_RANGED))
088                                return I;
089                }
090                return null;
091        }
092
093        @Override
094        public List<Item> getTrapComponents()
095        {
096                final Vector<Item> V=new Vector<Item>();
097                V.addElement(CMClass.getWeapon("Sword"));
098                return V;
099        }
100
101        @Override
102        public Trap setTrap(MOB mob, Physical P, int trapBonus, int qualifyingClassLevel, boolean perm)
103        {
104                if(P==null)
105                        return null;
106                final Item I=getPoison(mob);
107                setMiscText("3/a club");
108                if(I!=null)
109                {
110                        setMiscText(""+I.basePhyStats().damage()+"/"+I.name());
111                        I.destroy();
112                }
113                return super.setTrap(mob,P,trapBonus,qualifyingClassLevel,perm);
114        }
115
116        @Override
117        public boolean canSetTrapOn(MOB mob, Physical P)
118        {
119                if(!super.canSetTrapOn(mob,P))
120                        return false;
121                if(mob!=null)
122                {
123                        final Item I=getPoison(mob);
124                        if(I==null)
125                        {
126                                mob.tell(L("You'll need to set down a melee weapon first."));
127                                return false;
128                        }
129                }
130                return true;
131        }
132
133        @Override
134        public void spring(MOB target)
135        {
136                if((target!=invoker())
137                &&(target.location()!=null))
138                {
139                        final int x=text().indexOf('/');
140                        int dam=3;
141                        String name="a club";
142                        if(x>=0)
143                        {
144                                dam=CMath.s_int(text().substring(0,x));
145                                name=text().substring(x+1);
146                        }
147                        if((!canInvokeTrapOn(invoker(),target))
148                        ||(isLocalExempt(target))
149                        ||(invoker().getGroupMembers(new HashSet<MOB>()).contains(target))
150                        ||(target==invoker())
151                        ||(doesSaveVsTraps(target)))
152                                target.location().show(target,null,null,CMMsg.MASK_ALWAYS|CMMsg.MSG_NOISE,L("<S-NAME> avoid(s) setting off @x1 trap!",name));
153                        else
154                        if(target.location().show(target,target,this,CMMsg.MASK_ALWAYS|CMMsg.MSG_NOISE,L("<S-NAME> <S-IS-ARE> struck by @x1 trap!",name)))
155                        {
156                                super.spring(target);
157                                final int damage=CMLib.dice().roll(trapLevel()+abilityCode(),dam,1);
158                                CMLib.combat().postDamage(invoker(),target,this,damage,CMMsg.NO_EFFECT,-1,null);
159                                if((canBeUninvoked())&&(affected instanceof Item))
160                                        disable();
161                        }
162                }
163        }
164}
