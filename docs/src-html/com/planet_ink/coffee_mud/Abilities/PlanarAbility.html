159        @Override
160        public void setMiscText(String newText)
161        {
162                super.setMiscText(newText);
163                clearVars();
164                if(newText.length()>0)
165                {
166                        this.planeVars=getPlane(newText);
167                        if(this.planeVars==null)
168                                throw new IllegalArgumentException("Unknown: "+newText);
169                        this.roomsDone=new WeakArrayList<Room>();
170                        this.planarPrefix=planeVars.get(PlanarVar.PREFIX.toString());
171                        this.recoverRate = CMath.s_int(planeVars.get(PlanarVar.RECOVERRATE.toString()));
172                        this.fatigueRate = CMath.s_int(planeVars.get(PlanarVar.FATIGUERATE.toString()));
173                        this.recoverTick=1;
174                        if((planarPrefix!=null)&&(planarPrefix.indexOf(',')>0))
175                        {
176                                List<String> choices=CMParms.parseCommas(planarPrefix, true);
177                                planarPrefix=choices.get(CMLib.dice().roll(1, choices.size(), -1));
178                        }
179                        if(affected instanceof Area)
180                        {
181                                planeArea=(Area)affected;
182                                int medianLevel=planeArea.getPlayerLevel();
183                                if(medianLevel == 0)
184                                        medianLevel=planeArea.getAreaIStats()[Area.Stats.MED_LEVEL.ordinal()];
185                                planarLevel=medianLevel;
186                        }
187                        String specflags = planeVars.get(PlanarVar.SPECFLAGS.toString());
188                        if(specflags != null)
189                        {
190                                for(String s : CMParms.parse(specflags))
191                                {
192                                        PlanarSpecFlag flag=(PlanarSpecFlag)CMath.s_valueOf(PlanarSpecFlag.class, s);
193                                        if(flag == null)
194                                                Log.errOut("Spell_Planeshift","Unknown spec flag "+s);
195                                        else
196                                        {
197                                                if(this.specFlags==null)
198                                                        this.specFlags=new HashSet<PlanarSpecFlag>();
199                                                this.specFlags.add(flag);
200                                        }
201                                }
202                        }
203                        String areablurbs = planeVars.get(PlanarVar.AREABLURBS.toString());
204                        if((areablurbs!=null)&&(areablurbs.length()>0))
205                        {
206                                Map<String,String> blurbSets=CMParms.parseEQParms(areablurbs);
207                                for(String key : blurbSets.keySet())
208                                        planeArea.addBlurbFlag(key.toUpperCase().trim().replace(' ', '_')+" "+blurbSets.get(key));
209                        }
210                        String behaves = planeVars.get(PlanarVar.BEHAVE.toString());
211                        if(behaves!=null)
212                                this.behavList=CMParms.parseSpaceParenList(behaves);
213                        String reffects = planeVars.get(PlanarVar.REFFECT.toString());
214                        if(reffects!=null)
215                                this.reffectList=CMParms.parseSpaceParenList(reffects);
216                        String enables = planeVars.get(PlanarVar.ENABLE.toString());
217                        if(enables!=null)
218                        {
219                                List<Pair<String,String>> enableAs=CMParms.parseSpaceParenList(enables);
220                                Integer perLevel=new Integer(CMProps.getIntVar(CMProps.Int.LASTPLAYERLEVEL));
221                                Integer numSkills=Integer.valueOf(Integer.MAX_VALUE);
222                                for(final Iterator<Pair<String,String>> p=enableAs.iterator();p.hasNext();)
223                                {
224                                        Pair<String,String> P=p.next();
225                                        if(P.first.toLowerCase().equals("number"))
226                                        {
227                                                p.remove();
228                                                String parms=P.second;
229                                                int x=parms.indexOf('/');
230                                                if(x<0)
231                                                        numSkills=Integer.valueOf(CMath.s_int(parms.trim()));
232                                                else
233                                                {
234                                                        numSkills=Integer.valueOf(CMath.s_int(parms.substring(0,x).trim()));
235                                                        perLevel=Integer.valueOf(CMath.s_int(parms.substring(x+1).trim()));
236                                                }
237                                        }
238                                }
239                                if(enableAs.size()>0)
240                                {
241                                        PairList<String,String> addThese = new PairVector<String,String>();
242                                        for(final Iterator<Pair<String,String>> p = enableAs.iterator();p.hasNext();)
243                                        {
244                                                Pair<String,String> P=p.next();
245                                                Ability A=CMClass.getAbility(P.first);
246                                                if(A==null)
247                                                {
248                                                        p.remove();
249                                                        boolean foundSomething=false;
250                                                        long flag=CMParms.indexOf(Ability.FLAG_DESCS, P.first);
251                                                        if(flag >=0 )
252                                                                flag=CMath.pow(2, flag);
253                                                        int domain=CMParms.indexOf(Ability.DOMAIN_DESCS, P.first);
254                                                        if(domain > 0)
255                                                                domain = domain << 5;
256                                                        int acode=CMParms.indexOf(Ability.ACODE_DESCS, P.first);
257                                                        for(final Enumeration<Ability> a=CMClass.abilities();a.hasMoreElements();)
258                                                        {
259                                                                A=a.nextElement();
260                                                                if((A.Name().toUpperCase().equals(P.first))
261                                                                ||((flag>0)&&(CMath.bset(A.flags(),flag)))
262                                                                ||((domain>0)&&(A.classificationCode()&Ability.ALL_DOMAINS)==domain)
263                                                                ||((acode>=0)&&(A.classificationCode()&Ability.ALL_ACODES)==acode)
264                                                                )
265                                                                {
266                                                                        if(!addThese.containsFirst(A.ID().toUpperCase()))
267                                                                        {
268                                                                                addThese.add(A.ID().toUpperCase(), P.second);
269                                                                                foundSomething=true;
270                                                                        }
271                                                                }
272                                                        }
273                                                        if(!foundSomething)
274                                                                Log.errOut("Spell_Planeshift","Unknown skill type/domain/flag: "+P.first);
275                                                }
276                                        }
277                                        enableAs.addAll(addThese);
278                                        if(enableAs.size()>0)
279                                                this.enableList=new Pair<Pair<Integer,Integer>,List<Pair<String,String>>>(new Pair<Integer,Integer>(numSkills,perLevel),enableAs);
280                                }
281                        }
282                        String bonusDamageStat = planeVars.get(PlanarVar.BONUSDAMAGESTAT.toString());
283                        if(bonusDamageStat!=null)
284                        {
285                                this.bonusDmgStat=CMParms.indexOf(CharStats.CODES.BASENAMES(), bonusDamageStat.toUpperCase().trim());
286                        }
287                        String reqWeapons = planeVars.get(PlanarVar.REQWEAPONS.toString());
288                        if(reqWeapons != null)
289                                this.reqWeapons = new HashSet<String>(CMParms.parse(reqWeapons.toUpperCase().trim()));
290                        String atmosphere = planeVars.get(PlanarVar.ATMOSPHERE.toString());
291                        if(atmosphere!=null)
292                        {
293                                if(atmosphere.length()==0)
294                                        this.planeArea.setAtmosphere(Integer.MIN_VALUE);
295                                else
296                                {
297                                        int atmo=RawMaterial.CODES.FIND_IgnoreCase(atmosphere);
298                                        this.planeArea.setAtmosphere(atmo);
299                                }
300                        }
301                        String absorb = planeVars.get(PlanarVar.ABSORB.toString());
302                        if(absorb != null)
303                                reEffect(planeArea,"Prop_AbsorbDamage",absorb);
304                        TimeClock C=(TimeClock)CMLib.time().globalClock().copyOf();
305                        C.setDayOfMonth(1);
306                        C.setYear(1);
307                        C.setMonth(1);
308                        C.setHourOfDay(0);
309                        String hours = planeVars.get(PlanarVar.HOURS.toString());
310                        if((hours != null)&&(CMath.isInteger(hours)))
311                        {
312                                double mul=CMath.div(CMath.s_int(hours),CMLib.time().globalClock().getHoursInDay());
313                                if(mul != 1.0)
314                                {
315                                        int newHours = (int)Math.round(CMath.mul(C.getHoursInDay(),mul));
316                                        C.setHoursInDay(newHours);
317                                        C.setDawnToDusk((int)Math.round(CMath.mul(C.getDawnToDusk()[0],mul))
318                                                                        , (int)Math.round(CMath.mul(C.getDawnToDusk()[1],mul))
319                                                                        , (int)Math.round(CMath.mul(C.getDawnToDusk()[2],mul))
320                                                                        , (int)Math.round(CMath.mul(C.getDawnToDusk()[3],mul)));
321                                }
322                        }
323                        planeArea.setTimeObj(C);
324                        planeArea.addNonUninvokableEffect(CMClass.getAbility("Prop_NoTeleportOut"));
325                        planeArea.addNonUninvokableEffect(CMClass.getAbility("Prop_NoTeleport"));
326                        planeArea.addNonUninvokableEffect(CMClass.getAbility("Prop_NoRecall"));
327                        String aeffects = planeVars.get(PlanarVar.AEFFECT.toString());
328                        if(aeffects!=null)
329                        {
330                                List<Pair<String,String>> affectList=CMParms.parseSpaceParenList(aeffects);
331                                if(affectList!=null)
332                                {
333                                        for(Pair<String,String> p : affectList)
334                                        {
335                                                if(planeArea.fetchBehavior(p.first)==null)
336                                                {
337                                                        Behavior B=CMClass.getBehavior(p.first);
338                                                        if(B==null)
339                                                        {
340                                                                Ability A=CMClass.getAbility(p.first);
341                                                                if(A==null)
342                                                                        Log.errOut("Spell_Planeshift","Unknown behavior : "+p.first);
343                                                                else
344                                                                {
345                                                                        A.setMiscText(p.second);
346                                                                        planeArea.addNonUninvokableEffect(A);
347                                                                }
348                                                        }
349                                                        else
350                                                        {
351                                                                B.setParms(p.second);
352                                                                planeArea.addBehavior(B);
353                                                        }
354                                                }
355                                        }
356                                }
357                        }
358                }
359        }

361        protected void reEffect(final Physical M, String ID, String parms)
362        {
363                if(M!=null)
364                {
365                        Ability A=M.fetchEffect(ID);
366                        if(A!=null)
367                                M.delEffect(A);
368                        else
369                                A=CMClass.getAbility(ID);
370                        if(A!=null)
371                        {
372                                M.addNonUninvokableEffect(A);
373                                A.setMiscText((parms+" "+A.text()).trim());
374                        }
375                }

378        public synchronized void fixRoom(Room room)
379        {
380                try
381                {
382                        room.toggleMobility(false);
383                        CMLib.threads().suspendResumeRecurse(room, false, true);
384                        for(int i=0;i<Directions.NUM_DIRECTIONS();i++)
385                        {
386                                Room R=room.rawDoors()[i];
387                                if((R!=null)&&(R.getArea()!=planeArea))
388                                        room.rawDoors()[i]=null;
389                        }
390                        if(planeVars.containsKey(PlanarVar.ATMOSPHERE.toString()))
391                                room.setAtmosphere(planeArea.getAtmosphere());
392                        int eliteLevel=0;
393                        if(planeVars.containsKey(PlanarVar.ELITE.toString()))
394                                eliteLevel=CMath.s_int(planeVars.get(PlanarVar.ELITE.toString()));
395                        if(planeVars.containsKey(PlanarVar.ROOMCOLOR.toString()))
396                        {
397                                String prefix="";
398                                String displayText = room.displayText();
399                                if(displayText.toUpperCase().startsWith("<VARIES>"))
400                                {
401                                        prefix="<VARIES>";
402                                        displayText=displayText.substring(prefix.length());
403                                }
404                                String color=planeVars.get(PlanarVar.ROOMCOLOR.toString());
405                                if(color.startsWith("UP "))
406                                {
407                                        color=color.substring(3).trim();
408                                        displayText=displayText.toUpperCase();
409                                }
410                                room.setDisplayText(prefix+color+displayText+"^N");
411                        }
412                        if(planeVars.containsKey(PlanarVar.ROOMADJS.toString()))
413                        {
414                                String wordStr=planeVars.get(PlanarVar.ROOMADJS.toString());
415                                String prefix="";
416                                String desc = room.description();
417                                if(wordStr.startsWith("UP "))
418                                {
419                                        wordStr=wordStr.substring(3).trim();
420                                        desc=desc.toUpperCase();
421                                }
422                                int chance=30;
423                                int x=wordStr.indexOf(' ');
424                                if((x>0)&&(CMath.isInteger(wordStr.substring(0, x))))
425                                {
426                                        chance=CMath.s_int(wordStr.substring(0, x));
427                                        wordStr=wordStr.substring(x+1).trim();
428                                }
429                                String[] words= wordStr.split(",");
430                                if(desc.toUpperCase().startsWith("<VARIES>"))
431                                {
432                                        prefix="<VARIES>";
433                                        desc=desc.substring(prefix.length());
434                                }
435                                room.setDescription(prefix+CMLib.english().insertAdjectives(desc, words, chance));
436                        }
437                        if(this.reffectList!=null)
438                        {
439                                for(Pair<String,String> p : this.reffectList)
440                                {
441                                        if(room.fetchBehavior(p.first)==null)
442                                        {
443                                                Behavior B=CMClass.getBehavior(p.first);
444                                                if(B==null)
445                                                {
446                                                        Ability A=CMClass.getAbility(p.first);
447                                                        if(A==null)
448                                                                Log.errOut("Spell_Planeshift","Unknown behavior : "+p.first);
449                                                        else
450                                                        {
451                                                                A.setMiscText(p.second);
452                                                                room.addNonUninvokableEffect(A);
453                                                        }
454                                                }
455                                                else
456                                                {
457                                                        B.setParms(p.second);
458                                                        room.addBehavior(B);
459                                                }
460                                        }
461                                }
462                        }
463                        
464                        if(CMLib.law().getLandTitle(room)!=null)
465                        {
466                                List<Physical> destroyMe=new ArrayList<Physical>();
467                                Set<Rider> protSet = new HashSet<Rider>();
468                                for(final Enumeration<MOB> m=room.inhabitants();m.hasMoreElements();)
469                                {
470                                        final MOB M=m.nextElement();
471                                        if((M!=null)&&(M.isPlayer()))
472                                        {
473                                                protSet.add(M);
474                                                M.getGroupMembersAndRideables(protSet);
475                                        }
476                                }
477                                for(final Enumeration<MOB> m=room.inhabitants();m.hasMoreElements();)
478                                {
479                                        final MOB M=m.nextElement();
480                                        if((M!=null) && (!M.isPlayer()) && (!protSet.contains(M)))
481                                                destroyMe.add(M);
482                                }
483                                for(final Enumeration<Item> i=room.items();i.hasMoreElements();)
484                                {
485                                        final Item I=i.nextElement();
486                                        if((I!=null)&&(!protSet.contains(I)))
487                                                destroyMe.add(I);
488                                }
489                                for(Physical P : destroyMe)
490                                        P.destroy();
491                        }
492                        int allLevelAdj=CMath.s_int(planeVars.get(PlanarVar.LEVELADJ.toString()));
493                        List<Item> delItems=new ArrayList<Item>(0);
494                        for(final Enumeration<Item> i=room.items();i.hasMoreElements();)
495                        {
496                                final Item I=i.nextElement();
497                                if(I==null)
498                                        continue;
499                                if((I instanceof Exit)&&((I instanceof BoardableShip)))
500                                {
501                                        for(int x=0;x<100;x++)
502                                        {
503                                                Room R2=((Exit)I).lastRoomUsedFrom(room);
504                                                if((R2!=null)&&(R2.getArea()!=planeArea))
505                                                {
506                                                        delItems.add(I);
507                                                        break;
508                                                }
509                                        }
510                                }
511                                else
512                                if(I instanceof Exit)
513                                        I.setReadableText("");
514                                else
515                                if((invoker!=null)&&((I instanceof Weapon)||(I instanceof Armor)))
516                                {
517                                        int newILevelAdj = (planarLevel - I.phyStats().level());
518                                        int newILevel=invoker.phyStats().level() - newILevelAdj + allLevelAdj;
519                                        if(newILevel <= 0)
520                                                newILevel = 1;
521                                        CMLib.itemBuilder().itemFix(I, newILevel, null);
522                                        I.basePhyStats().setLevel(newILevel);
523                                        I.phyStats().setLevel(newILevel);
524                                        CMLib.itemBuilder().balanceItemByLevel(I);
525                                        if((I instanceof Weapon)
526                                        &&(this.reqWeapons!=null)
527                                        &&(this.reqWeapons.contains("MAGICAL")))
528                                        {
529                                                I.basePhyStats().setDisposition(I.basePhyStats().disposition()|PhyStats.IS_BONUS);
530                                                I.phyStats().setDisposition(I.phyStats().disposition()|PhyStats.IS_BONUS);
531                                        }
532                                        I.text();
533                                }
534                        }
535                        for(Item I : delItems)
536                                I.destroy();
537                        for(final Enumeration<MOB> m=room.inhabitants();m.hasMoreElements();)
538                        {
539                                final MOB M=m.nextElement();
540                                if((M!=null)
541                                &&(invoker!=null)
542                                &&(M.isMonster())
543                                &&(M.getStartRoom()!=null)
544                                &&(M.getStartRoom().getArea()==planeArea))
545                                {
546                                        if(planeVars.containsKey(PlanarVar.MIXRACE.toString()))
547                                        {
548                                                String mixRace = planeVars.get(PlanarVar.MIXRACE.toString());
549                                                Race firstR=CMClass.getRace(mixRace);
550                                                if(firstR==null)
551                                                        Log.errOut("PlanarAbility","Unknown mixrace: "+mixRace);
552                                                else
553                                                {
554                                                        final Race secondR=M.charStats().getMyRace();
555                                                        Race R=CMLib.utensils().getMixedRace(firstR.ID(),secondR.ID(), false);
556                                                        if(R!=null)
557                                                        {
558                                                                M.baseCharStats().setMyRace(R);
559                                                                M.charStats().setMyRace(R);
560                                                                M.charStats().setWearableRestrictionsBitmap(M.charStats().getWearableRestrictionsBitmap()|M.charStats().getMyRace().forbiddenWornBits());
561                                                        }
562                                                }
563                                        }
564                                        
565                                        if(planeVars.containsKey(PlanarVar.ATMOSPHERE.toString()))
566                                                M.baseCharStats().setBreathables(new int[]{room.getAtmosphere()});
567                                        int newLevelAdj = (planarLevel - M.phyStats().level());
568                                        int newLevel = invoker.phyStats().level() - newLevelAdj + allLevelAdj;
569                                        if(newLevel <= 0)
570                                                newLevel = 1;
571                                        if((planarPrefix!=null)&&(planarPrefix.length()>0))
572                                        {
573                                                String oldName=M.Name();
574                                                int x;
575                                                if(oldName.toLowerCase().indexOf(planarPrefix.toLowerCase())<0)
576                                                {
577                                                        if(CMLib.english().startsWithAnArticle(M.Name()))
578                                                        {
579                                                                String Name = M.Name().substring(M.Name().indexOf(' ')).trim();
580                                                                M.setName(CMLib.english().startWithAorAn(planarPrefix+" "+Name));
581                                                        }
582                                                        else
583                                                        {
584                                                                M.setName(CMStrings.capitalizeFirstLetter(planarPrefix)+" "+M.Name());
585                                                        }
586                                                        if((x=M.displayText().toLowerCase().indexOf(oldName.toLowerCase()))>=0)
587                                                        {
588                                                                M.setDisplayText(M.displayText().substring(0,x)+M.Name()+M.displayText().substring(x+oldName.length()));
589                                                        }
590                                                        else
591                                                        if(CMLib.english().startsWithAnArticle(M.displayText()))
