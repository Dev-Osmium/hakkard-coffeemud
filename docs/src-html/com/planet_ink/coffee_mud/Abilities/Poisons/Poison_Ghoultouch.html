036public class Poison_Ghoultouch extends Poison
037{
038        @Override
039        public String ID()
040        {
041                return "Poison_Ghoultouch";
042        }
043
044        private final static String localizedName = CMLib.lang().L("Ghoultouch");
045
046        @Override
047        public String name()
048        {
049                return localizedName;
050        }
051
052        private static final String[] triggerStrings =I(new String[] {"POISONGHOUL"});
053        @Override
054        public String[] triggerStrings()
055        {
056                return triggerStrings;
057        }
058
059        @Override
060        public long flags()
061        {
062                return Ability.FLAG_PARALYZING|Ability.FLAG_UNHOLY;
063        }
064
065        @Override
066        protected int POISON_TICKS()
067        {
068                return 7;
069        } // 0 means no adjustment!
070
071        @Override
072        protected int POISON_DELAY()
073        {
074                return 1;
075        }
076
077        @Override
078        protected String POISON_DONE()
079        {
080                return "Your muscles relax again.";
081        }
082
083        @Override
084        protected String POISON_START()
085        {
086                return "^G<S-NAME> become(s) stiff and immobile!^?";
087        }
088
089        @Override
090        protected String POISON_AFFECT()
091        {
092                return "";
093        }
094
095        @Override
096        protected String POISON_CAST()
097        {
098                return "^F^<FIGHT^><S-NAME> poison(s) <T-NAMESELF>!^</FIGHT^>^?";
099        }
100
101        @Override
102        protected String POISON_FAIL()
103        {
104                return "<S-NAME> attempt(s) to poison <T-NAMESELF>, but fail(s).";
105        }
106
107        @Override
108        protected int POISON_DAMAGE()
109        {
110                return 0;
111        }
112
113        @Override
114        public void affectPhyStats(final Physical affected, final PhyStats affectableStats)
115        {
116                if(affected instanceof MOB)
117                        affectableStats.setSensesMask(affectableStats.sensesMask()|PhyStats.CAN_NOT_MOVE);
118        }
119
120        @Override
121        public void affectCharStats(final MOB affected, final CharStats affectableStats)
122        {
123                affectableStats.setStat(CharStats.STAT_CONSTITUTION,affectableStats.getStat(CharStats.STAT_CONSTITUTION)-1);
124                if(affectableStats.getStat(CharStats.STAT_CONSTITUTION)<=0)
125                        affectableStats.setStat(CharStats.STAT_CONSTITUTION,1);
126        }
127}
