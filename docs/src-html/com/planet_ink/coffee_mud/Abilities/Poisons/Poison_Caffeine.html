036public class Poison_Caffeine extends Poison {
037        @Override
038        public String ID()
039        {
040                return "Poison_Caffeine";
041        }
042
043        private final static String localizedName = CMLib.lang().L("Poison_Hyper");
044
045        @Override
046        public String name()
047        {
048                return localizedName;
049        }
050
051        private final static String localizedStaticDisplay = CMLib.lang().L("(CAFFEINATED!!)");
052
053        @Override
054        public String displayText()
055        {
056                return localizedStaticDisplay;
057        }
058
059        private static final String[] triggerStrings =I(new String[] {"POISONHYPER"});
060        @Override
061        public String[] triggerStrings()
062        {
063                return triggerStrings;
064        }
065
066        @Override
067        protected int POISON_TICKS()
068        {
069                return 30;
070        } // 0 means no adjustment!
071
072        @Override
073        protected int POISON_DELAY()
074        {
075                return 5;
076        }
077
078        @Override
079        protected String POISON_DONE()
080        {
081                return "The caffeine runs its course.";
082        }
083
084        @Override
085        protected String POISON_START()
086        {
087                return "^G<S-NAME> seem(s) wired!^?";
088        }
089
090        @Override
091        protected String POISON_AFFECT()
092        {
093                return "^G<S-NAME> twitch(es) spastically.";
094        }
095
096        @Override
097        protected String POISON_CAST()
098        {
099                return "^F^<FIGHT^><S-NAME> caffeinate(s) <T-NAMESELF>!^</FIGHT^>^?";
100        }
101
102        @Override
103        protected String POISON_FAIL()
104        {
105                return "<S-NAME> attempt(s) to caffinate <T-NAMESELF>, but fail(s).";
106        }
107
108        @Override
109        protected int POISON_DAMAGE()
110        {
111                return 0;
112        }
113
114        @Override
115        public void affectCharStats(final MOB affected, final CharStats affectableStats)
116        {
117                affectableStats.setStat(CharStats.STAT_DEXTERITY,affectableStats.getStat(CharStats.STAT_DEXTERITY)+1);
118        }
119
120        @Override
121        public void affectPhyStats(final Physical affected, final PhyStats affectableStats)
122        {
123                super.affectPhyStats(affected,affectableStats);
124                affectableStats.setSpeed(affectableStats.speed() + 0.25);
125                int oldDisposition=affectableStats.disposition();
126                oldDisposition=oldDisposition&(~(PhyStats.IS_SLEEPING|PhyStats.IS_SNEAKING|PhyStats.IS_SITTING|PhyStats.IS_CUSTOM));
127                affectableStats.setDisposition(oldDisposition);
128        }
129
130        @Override
131        public long flags()
132        {
133                return super.flags() | Ability.FLAG_MINDALTERING;
134        }
135
136        @Override
137        public boolean okMessage(final Environmental myHost, final CMMsg msg)
138        {
139                if(!(affected instanceof MOB))
140                                return true;
141
142                final MOB mob=(MOB)affected;
143                if(msg.amISource(mob)&&((msg.sourceMinor()==CMMsg.TYP_SIT)||(msg.sourceMinor()==CMMsg.TYP_SLEEP)))
144                {
145                        mob.tell(L("You're too caffeinated for that!"));
146                        return false;
147                }
148                return true;
149        }
150}
