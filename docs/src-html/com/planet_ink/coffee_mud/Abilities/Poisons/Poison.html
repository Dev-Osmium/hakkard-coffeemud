172        protected boolean catchIt(MOB mob, Physical target)
173        {
174                MOB poisoner=invoker;
175                if(poisoner==null)
176                        poisoner=mob;
177                if((poisoner==null)&&(target instanceof MOB))
178                        poisoner=(MOB)target;
179                if((target!=null)
180                &&(target instanceof MOB)
181                &&(target.fetchEffect(ID())==null)
182                &&(poisoner!=null)
183                &&((poisoner==target)||(!poisoner.isPlayer())||(poisoner.mayIFight((MOB)target))))
184                {
185                        final MOB targetMOB=(MOB)target;
186                        if(POISON_START_TARGETONLY().length()>0)
187                                targetMOB.tell(POISON_START_TARGETONLY());
188                        final String startMsg=CMStrings.replaceAll(POISON_START(),"<S-","<T-");
189                        final CMMsg msg=CMClass.getMsg(poisoner,targetMOB,this,CMMsg.MASK_ALWAYS|CMMsg.MASK_MALICIOUS|CMMsg.TYP_POISON,startMsg);
190                        if((POISON_START_TARGETONLY().length()>0)
191                        ||((targetMOB.location()!=null)
192                                &&(targetMOB.location().okMessage(targetMOB,msg))))
193                        {
194                                targetMOB.location().send(targetMOB,msg);
195                                if((POISON_AFFECTTARGET()&&(msg.value()<=0)))
196                                {
197                                        final MOB pvic=poisoner.getVictim();
198                                        final MOB tvic = (target instanceof MOB) ? ((MOB)target).getVictim() : null;
199                                        maliciousAffect(poisoner,target,
200                                                                        (affected instanceof Item)?affected.phyStats().level():0,
201                                                                        POISON_TICKS(),-1);
202                                        if(POISON_MAKE_PEACE())
203                                        {
204                                                if((pvic == null)
205                                                &&(poisoner.getVictim() == target))
206                                                        poisoner.makePeace(true);
207                                                if((target instanceof MOB)
208                                                &&(tvic == null)
209                                                &&(((MOB)target).getVictim()==poisoner))
210                                                        ((MOB)target).makePeace(true);
211                                        }
212                                        return true;
213                                }
214                                return false;
215                        }
216                }
217                return false;
218        }

220        @Override
221        public boolean tick(final Tickable ticking, final int tickID)
222        {
223                if(!super.tick(ticking,tickID))
224                        return false;
225
226                if(!(affected instanceof MOB))
227                        return true;
228
229                final MOB mob=(MOB)affected;
230                if(mob==null)
231                        return false;
232                if((--poisonTick)<=0)
233                {
234                        poisonTick=POISON_DELAY();
235                        if(POISON_AFFECT().length()>0)
236                                mob.location().show(mob,null,CMMsg.MSG_OK_VISUAL,POISON_AFFECT()+CMLib.protocol().msp("poisoned.wav",10));
237                        final MOB invoker=(invoker()!=null) ? invoker() : mob;
238                        if(POISON_DAMAGE()!=0)
239                        {
240                                CMLib.combat().postDamage(invoker,mob,this,POISON_DAMAGE(),CMMsg.MASK_ALWAYS|CMMsg.TYP_POISON,-1,null);
241                                if((!mob.isInCombat())&&(mob!=invoker)&&(mob.location()!=null)&&(mob.location().isInhabitant(invoker))&&(CMLib.flags().canBeSeenBy(invoker,mob)))
242                                        CMLib.combat().postAttack(mob,invoker,mob.fetchWieldedItem());
243                        }
244                }
245                return true;
246        }

248        @Override
249        public void affectCharStats(final MOB affected, final CharStats affectableStats)
250        {
251                if((affected==null)||(!ID().equals("Poison")))
252                        return;
253                affectableStats.setStat(CharStats.STAT_CONSTITUTION,affectableStats.getStat(CharStats.STAT_CONSTITUTION)-5);
254                affectableStats.setStat(CharStats.STAT_STRENGTH,affectableStats.getStat(CharStats.STAT_STRENGTH)-5);
255                if(affectableStats.getStat(CharStats.STAT_CONSTITUTION)<=0)
256                        affectableStats.setStat(CharStats.STAT_CONSTITUTION,1);
257                if(affectableStats.getStat(CharStats.STAT_STRENGTH)<=0)
258                        affectableStats.setStat(CharStats.STAT_STRENGTH,1);
259        }

280        @Override
281        public void executeMsg(final Environmental myHost, final CMMsg msg)
282        {
283                if(affected==null)
284                        return;
285                if(affected instanceof Item)
286                {
287                        if(!processing)
288                        {
289                                final Item myItem=(Item)affected;
290                                if(myItem.owner()==null)
291                                        return;
292                                try
293                                {
294                                        processing=true;
295                                        if(msg.amITarget(myItem))
296                                        {
297                                                switch(msg.sourceMinor())
298                                                {
299                                                case CMMsg.TYP_DRINK:
300                                                        if(myItem instanceof Drink)
301                                                        {
302                                                                catchIt(msg.source(),msg.source());
303                                                                if(((((Drink)myItem).liquidRemaining()<0)
304                                                                        ||((((Drink)myItem).liquidRemaining()-((Drink)myItem).thirstQuenched())<=0))
305                                                                &&(!((Drink)myItem).disappearsAfterDrinking()))
306                                                                        affected.delEffect(this);
307                                                        }
308                                                        break;
309                                                case CMMsg.TYP_EAT:
310                                                        if(myItem instanceof Food)
311                                                                catchIt(msg.source(),msg.source());
312                                                        break;
313                                                }
314                                        }
315                                        else
316                                        if(msg.tool()==affected)
317                                        {
318                                                switch(msg.targetMinor())
319                                                {
320                                                case CMMsg.TYP_DAMAGE:
321                                                        if((msg.source()!=msg.target())
322                                                        &&(myItem instanceof Weapon)
323                                                        &&(msg.target() instanceof MOB))
324                                                        {
325                                                                tickDown--;
326                                                                catchIt(msg.source(),(MOB)msg.target());
327                                                        }
328                                                        break;
329                                                case CMMsg.TYP_FILL:
330                                                        if((msg.target() instanceof Drink)
331                                                        &&(affected instanceof Drink))
332                                                        {
333                                                                if((msg.target() instanceof Item)
334                                                                &&((!CMLib.flags().isGettable((Item)msg.target()))
335                                                                        ||(((Drink)msg.target()).liquidRemaining()>9999)))
336                                                                        invoke(msg.source(), (Physical)msg.target(), true, 0);
337                                                                else
338                                                                if(msg.target() instanceof Physical)
339                                                                        ((Physical)msg.target()).addEffect((Ability)this.copyOf());
340                                                                if((((Drink)affected).liquidRemaining()-((Drink)msg.target()).amountTakenToFillMe((Drink)affected))<=0)
341                                                                        affected.delEffect(this);
342                                                        }
343                                                        break;
344                                                }
345                                        }
346                                }
347                                finally
348                                {
349                                        processing=false;
350                                }
351                        }
352                }
353                super.executeMsg(myHost,msg);
354        }

356        @Override
357        public boolean invoke(MOB mob, List<String> commands, Physical givenTarget, boolean auto, int asLevel)
358        {
359                final Environmental target=this.getAnyTarget(mob,commands,givenTarget,Wearable.FILTER_UNWORNONLY);
360                if(target==null)
361                        return false;
362                if(!super.invoke(mob,commands,givenTarget,auto,asLevel))
363                        return false;
364                boolean success=proficiencyCheck(mob,0,auto);
365                if(success)
366                {
367                        final String str=auto?"":POISON_CAST();
368                        final CMMsg msg=CMClass.getMsg(mob,target,this,CMMsg.MSK_MALICIOUS_MOVE|CMMsg.TYP_POISON|(auto?CMMsg.MASK_ALWAYS:0),str);
369                        CMLib.color().fixSourceFightColor(msg);
370                        Room R=mob.location();
371                        if((target instanceof MOB)&&(((MOB)target).location()!=null))
372                                R=((MOB)target).location();
373                        if(R.okMessage(mob,msg))
374                        {
375                                R.send(mob,msg);
376                                if(msg.value()<=0)
377                                {
378                                        if(target instanceof MOB)
379                                        {
380                                                if(POISON_START_TARGETONLY().length()>0)
381                                                        ((MOB)target).tell(POISON_START_TARGETONLY());
382                                                else
383                                                        R.show((MOB)target,null,CMMsg.MSG_OK_VISUAL,POISON_START());
384                                                if(POISON_AFFECTTARGET())
385                                                        success=maliciousAffect(mob,(MOB)target,asLevel,POISON_TICKS(),-1)!=null;
386                                                else
387                                                        success=true;
388                                        }
389                                        else
390                                        if(target instanceof Physical)
391                                                success=maliciousAffect(mob,(Physical)target,asLevel,100,-1)!=null;
392                                }
393                                else
394                                        success=false;
395                        }
396                        else
397                                success=false;
398                }
399                else
400                        return maliciousFizzle(mob,target,POISON_FAIL());
401
402                return success;
403
404        }
