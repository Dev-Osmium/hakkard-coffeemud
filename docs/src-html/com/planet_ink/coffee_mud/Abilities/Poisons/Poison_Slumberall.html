036public class Poison_Slumberall extends Poison
037{
038        @Override
039        public String ID()
040        {
041                return "Poison_Slumberall";
042        }
043
044        private final static String localizedName = CMLib.lang().L("Slumberall");
045
046        @Override
047        public String name()
048        {
049                return localizedName;
050        }
051
052        private static final String[] triggerStrings =I(new String[] {"POISONSLEEP"});
053        @Override
054        public String[] triggerStrings()
055        {
056                return triggerStrings;
057        }
058
059        @Override
060        protected int POISON_TICKS()
061        {
062                return 50;
063        } // 0 means no adjustment!
064
065        @Override
066        protected int POISON_DELAY()
067        {
068                return 1;
069        }
070
071        @Override
072        protected String POISON_DONE()
073        {
074                return "You don't feel so drowsy anymore.";
075        }
076
077        @Override
078        protected String POISON_START()
079        {
080                return null;
081        }
082
083        @Override
084        protected String POISON_AFFECT()
085        {
086                return "";
087        }
088
089        @Override
090        protected String POISON_CAST()
091        {
092                return "^F^<FIGHT^><S-NAME> poison(s) <T-NAMESELF>!^</FIGHT^>^?";
093        }
094
095        @Override
096        protected String POISON_FAIL()
097        {
098                return "<S-NAME> attempt(s) to poison <T-NAMESELF>, but fail(s).";
099        }
100
101        @Override
102        protected int POISON_DAMAGE()
103        {
104                return 0;
105        }
106
107        protected boolean fallenYet=false;
108
109        @Override
110        public void affectPhyStats(final Physical affected, final PhyStats affectableStats)
111        {
112                if((affected instanceof MOB)&&(fallenYet))
113                        affectableStats.setDisposition(affectableStats.disposition()|PhyStats.IS_SLEEPING);
114        }
115
116        @Override
117        public void unInvoke()
118        {
119                if((affected instanceof MOB)&&(super.canBeUninvoked()))
120                {
121                        final MOB mob=(MOB)affected;
122                        CMLib.commands().postStand(mob,true, false);
123                }
124                super.unInvoke();
125        }
126
127        @Override
128        public boolean okMessage(final Environmental myHost, final CMMsg msg)
129        {
130                if(!(affected instanceof MOB))
131                        return true;
132
133                final MOB mob=(MOB)affected;
134
135                // when this spell is on a MOBs Affected list,
136                // it should consistantly prevent the mob
137                // from trying to do ANYTHING except sleep
138                if(msg.amITarget(mob)&&(fallenYet)&&(msg.targetMinor()==CMMsg.TYP_DAMAGE))
139                        unInvoke();
140                else
141                if((msg.amISource(mob))
142                &&(fallenYet)
143                &&(!msg.sourceMajor(CMMsg.MASK_ALWAYS))
144                &&(msg.sourceMajor()>0)
145                &&(msg.sourceMinor()!=CMMsg.TYP_SLEEP))
146                {
147                        mob.tell(L("You are way too drowsy."));
148                        return false;
149                }
150                return super.okMessage(myHost,msg);
151        }
152
153        @Override
154        public boolean tick(final Tickable ticking, final int tickID)
155        {
156                if(!super.tick(ticking,tickID))
157                        return false;
158
159                if(!(affected instanceof MOB))
160                        return true;
161
162                final MOB mob=(MOB)affected;
163                if(mob==null)
164                        return false;
165                if((!fallenYet)&&(mob.location()!=null))
166                {
167                        fallenYet=true;
168                        mob.location().show(mob,null,CMMsg.MSG_SLEEP,L("<S-NAME> fall(s) asleep!"));
169                        mob.recoverPhyStats();
170                }
171                return true;
172        }
173}
