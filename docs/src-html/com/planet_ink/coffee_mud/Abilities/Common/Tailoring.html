115        @Override
116        public void unInvoke()
117        {
118                if(canBeUninvoked())
119                {
120                        if(affected instanceof MOB)
121                        {
122                                final MOB mob=(MOB)affected;
123                                if((buildingI!=null)&&(!aborted))
124                                {
125                                        if(messedUp)
126                                        {
127                                                if(activity == CraftingActivity.MENDING)
128                                                        messedUpCrafting(mob);
129                                                else
130                                                if(activity == CraftingActivity.LEARNING)
131                                                {
132                                                        commonEmote(mob,L("<S-NAME> fail(s) to learn how to make @x1.",buildingI.name()));
133                                                        buildingI.destroy();
134                                                }
135                                                else
136                                                if(activity == CraftingActivity.REFITTING)
137                                                        commonEmote(mob,L("<S-NAME> mess(es) up refitting @x1.",buildingI.name()));
138                                                else
139                                                        commonEmote(mob,L("<S-NAME> mess(es) up knitting @x1.",buildingI.name()));
140                                        }
141                                        else
142                                        {
143                                                if(activity == CraftingActivity.MENDING)
144                                                {
145                                                        buildingI.setUsesRemaining(100);
146                                                        CMLib.achievements().possiblyBumpAchievement(mob, AchievementLibrary.Event.MENDER, 1, this);
147                                                }
148                                                else
149                                                if(activity==CraftingActivity.LEARNING)
150                                                {
151                                                        deconstructRecipeInto(mob, buildingI, recipeHolder );
152                                                        buildingI.destroy();
153                                                }
154                                                else
155                                                if(activity == CraftingActivity.REFITTING)
156                                                {
157                                                        buildingI.basePhyStats().setHeight(0);
158                                                        buildingI.recoverPhyStats();
159                                                }
160                                                else
161                                                {
162                                                        dropAWinner(mob,buildingI);
163                                                        CMLib.achievements().possiblyBumpAchievement(mob, AchievementLibrary.Event.CRAFTING, 1, this);
164                                                }
165                                        }
166                                }
167                                buildingI=null;
168                                activity = CraftingActivity.CRAFTING;
169                        }
170                }
171                super.unInvoke();
172        }

180        protected boolean masterCraftCheck(final Item I)
181        {
182                if(I.basePhyStats().level()>30)
183                        return false;
184                if(I.name().toUpperCase().startsWith("DESIGNER")||(I.name().toUpperCase().indexOf(" DESIGNER ")>0))
185                        return false;
186                return true;
187        }

189        @Override
190        public boolean mayICraft(final Item I)
191        {
192                if(I==null)
193                        return false;
194                if(!super.mayBeCrafted(I))
195                        return false;
196                if((I.material()&RawMaterial.MATERIAL_MASK)!=RawMaterial.MATERIAL_CLOTH)
197                        return false;
198                if(CMLib.flags().isDeadlyOrMaliciousEffect(I))
199                        return false;
200                if(!masterCraftCheck(I))
201                        return (isANativeItem(I.Name()));
202                if(I.baseGoldValue()>I.basePhyStats().level())
203                        return (isANativeItem(I.Name()));
204                if(I instanceof Rideable)
205                {
206                        final Rideable R=(Rideable)I;
207                        final int rideType=R.rideBasis();
208                        switch(rideType)
209                        {
210                        case Rideable.RIDEABLE_ENTERIN:
211                        case Rideable.RIDEABLE_SIT:
212                                return true;
213                        default:
214                                return false;
215                        }
216                }
217                if(I instanceof Armor)
218                {
219                        final long nonWearablePlaces=Wearable.WORN_FEET|Wearable.WORN_LEFT_FINGER|Wearable.WORN_RIGHT_FINGER|Wearable.WORN_EARS|Wearable.WORN_EYES;
220                        if((I.rawProperLocationBitmap()&nonWearablePlaces)>0)
221                                return (isANativeItem(I.Name()));
222                        return true;
223                }
224                if(I instanceof Weapon)
225                {
226                        if(I.basePhyStats().damage()!=0)
227                                return false;
228                        if(I.basePhyStats().attackAdjustment()!=0)
229                                return (isANativeItem(I.Name()));
230                        return true;
231                }
232                return (isANativeItem(I.Name()));
233        }

267        @Override
268        protected boolean autoGenInvoke(final MOB mob, List<String> commands, Physical givenTarget, final boolean auto, 
269                                                                 final int asLevel, int autoGenerate, boolean forceLevels, List<Item> crafted)
270        {
271                if(super.checkStop(mob, commands))
272                        return true;
273
274                if(super.checkInfo(mob, commands))
275                        return true;
276                
277                final PairVector<EnhancedExpertise,Integer> enhancedTypes=enhancedTypes(mob,commands);
278                int recipeLevel = 1;
279                randomRecipeFix(mob,addRecipes(mob,loadRecipes()),commands,autoGenerate);
280                if(commands.size()==0)
281                {
282                        commonTell(mob,L("Knit what? Enter \"knit list\" for a list, \"knit info <item>\", \"knit refit <item>\" to resize,"
283                                                + " \"knit learn <item>\", \"knit scan\", \"knit mend <item>\", or \"knit stop\" to cancel."));
284                        return false;
285                }
286                if((!auto)
287                &&(commands.size()>0)
288                &&((commands.get(0)).equalsIgnoreCase("bundle")))
289                {
290                        bundling=true;
291                        if(super.invoke(mob,commands,givenTarget,auto,asLevel))
292                                return super.bundle(mob,commands);
293                        return false;
294                }
295                final List<List<String>> recipes=addRecipes(mob,loadRecipes());
296                final String str=commands.get(0);
297                String startStr=null;
298                bundling=false;
299                int duration=4;
300                if(str.equalsIgnoreCase("list"))
301                {
302                        String mask=CMParms.combine(commands,1);
303                        boolean allFlag=false;
304                        if(mask.equalsIgnoreCase("all"))
305                        {
306                                allFlag=true;
307                                mask="";
308                        }
309                        final StringBuffer buf=new StringBuffer("");
310                        int toggler=1;
311                        final int toggleTop=2;
312                        final int[] cols={
313                                CMLib.lister().fixColWidth(28,mob.session()),
314                                CMLib.lister().fixColWidth(3,mob.session()),
315                                CMLib.lister().fixColWidth(5,mob.session())
316                        };
317                        for(int r=0;r<toggleTop;r++)
318                                buf.append((r>0?" ":"")+CMStrings.padRight(L("Item"),cols[0])+" "+CMStrings.padRight(L("Lvl"),cols[1])+" "+CMStrings.padRight(L("Cloth"),cols[2]));
319                        buf.append("\n\r");
320                        final List<List<String>> listRecipes=((mask.length()==0) || mask.equalsIgnoreCase("all")) ? recipes : super.matchingRecipeNames(recipes, mask, true);
321                        for(int r=0;r<listRecipes.size();r++)
322                        {
323                                final List<String> V=listRecipes.get(r);
324                                if(V.size()>0)
325                                {
326                                        final String item=replacePercent(V.get(RCP_FINALNAME),"");
327                                        final int level=CMath.s_int(V.get(RCP_LEVEL));
328                                        final String wood=getComponentDescription(mob,V,RCP_WOOD);
329                                        if(wood.length()>5)
330                                        {
331                                                if(toggler>1)
332                                                        buf.append("\n\r");
333                                                toggler=toggleTop;
334                                        }
335                                        if((level<=xlevel(mob))||allFlag)
336                                        {
337                                                buf.append(CMStrings.padRight(item,cols[0])+" "+CMStrings.padRight(""+level,cols[1])+" "+CMStrings.padRightPreserve(""+wood,cols[2])+((toggler!=toggleTop)?" ":"\n\r"));
338                                                if(++toggler>toggleTop)
339                                                        toggler=1;
340                                        }
341                                }
342                        }
343                        if(toggler!=1)
344                                buf.append("\n\r");
345                        commonTell(mob,buf.toString());
346                        enhanceList(mob);
347                        return true;
348                }
349                else
350                if(((commands.get(0))).equalsIgnoreCase("learn"))
351                {
352                        return doLearnRecipe(mob, commands, givenTarget, auto, asLevel);
353                }
354                else
355                if(str.equalsIgnoreCase("scan"))
356                        return publicScan(mob,commands);
357                else
358                if(str.equalsIgnoreCase("mend"))
359                {
360                        buildingI=null;
361                        activity = CraftingActivity.CRAFTING;
362                        messedUp=false;
363                        final Vector<String> newCommands=CMParms.parse(CMParms.combine(commands,1));
364                        buildingI=getTargetItemFavorMOB(mob,mob.location(),givenTarget,newCommands,Wearable.FILTER_UNWORNONLY);
365                        if(!canMend(mob,buildingI,false))
366                                return false;
367                        activity = CraftingActivity.MENDING;
368                        if(!super.invoke(mob,commands,givenTarget,auto,asLevel))
369                                return false;
370                        startStr=L("<S-NAME> start(s) mending @x1.",buildingI.name());
371                        displayText=L("You are mending @x1",buildingI.name());
372                        verb=L("mending @x1",buildingI.name());
373                }
374                else
375                if(str.equalsIgnoreCase("refit"))
376                {
377                        buildingI=null;
378                        activity = CraftingActivity.CRAFTING;
379                        messedUp=false;
380                        final Vector<String> newCommands=CMParms.parse(CMParms.combine(commands,1));
381                        buildingI=getTargetItemFavorMOB(mob,mob.location(),givenTarget,newCommands,Wearable.FILTER_UNWORNONLY);
382                        if(buildingI==null)
383                                return false;
384                        if((buildingI.material()&RawMaterial.MATERIAL_MASK)!=RawMaterial.MATERIAL_CLOTH)
385                        {
386                                commonTell(mob,L("That's not made of cloth.  It can't be refitted."));
387                                return false;
388                        }
389                        if(!(buildingI instanceof Armor))
390                        {
391                                commonTell(mob,L("You don't know how to refit that sort of thing."));
392                                return false;
393                        }
394                        if(buildingI.phyStats().height()==0)
395                        {
396                                commonTell(mob,L("@x1 is already the right size.",buildingI.name(mob)));
397                                return false;
398                        }
399                        activity = CraftingActivity.REFITTING;
400                        if(!super.invoke(mob,commands,givenTarget,auto,asLevel))
401                                return false;
402                        startStr=L("<S-NAME> start(s) refitting @x1.",buildingI.name());
403                        displayText=L("You are refitting @x1",buildingI.name());
404                        verb=L("refitting @x1",buildingI.name());
405                }
406                else
407                {
408                        buildingI=null;
409                        activity = CraftingActivity.CRAFTING;
410                        messedUp=false;
411                        aborted=false;
412                        int amount=-1;
413                        if((commands.size()>1)&&(CMath.isNumber(commands.get(commands.size()-1))))
414                        {
415                                amount=CMath.s_int(commands.get(commands.size()-1));
416                                commands.remove(commands.size()-1);
417                        }
418                        final String recipeName=CMParms.combine(commands,0);
419                        List<String> foundRecipe=null;
420                        final List<List<String>> matches=matchingRecipeNames(recipes,recipeName,true);
421                        for(int r=0;r<matches.size();r++)
422                        {
423                                final List<String> V=matches.get(r);
424                                if(V.size()>0)
425                                {
426                                        final int level=CMath.s_int(V.get(RCP_LEVEL));
427                                        if((autoGenerate>0)||(level<=xlevel(mob)))
428                                        {
429                                                foundRecipe=V;
430                                                recipeLevel=level;
431                                                break;
432                                        }
433                                }
434                        }
435                        if(foundRecipe==null)
436                        {
437                                commonTell(mob,L("You don't know how to knit a '@x1'.  Try \"@x2 list\" for a list.",recipeName,triggerStrings()[0].toLowerCase()));
438                                return false;
439                        }
440
441                        final String woodRequiredStr = foundRecipe.get(RCP_WOOD);
442                        final int[] compData = new int[CF_TOTAL];
443                        final String realRecipeName=replacePercent(foundRecipe.get(RCP_FINALNAME),"");
444                        final List<Object> componentsFoundList=getAbilityComponents(mob, woodRequiredStr, "make "+CMLib.english().startWithAorAn(realRecipeName),autoGenerate,compData,1);
445                        if(componentsFoundList==null)
446                                return false;
447                        int woodRequired=CMath.s_int(woodRequiredStr);
448                        woodRequired=adjustWoodRequired(woodRequired,mob);
449
450                        if(amount>woodRequired)
451                                woodRequired=amount;
452                        final String misctype=foundRecipe.get(RCP_MISCTYPE);
453                        bundling=misctype.equalsIgnoreCase("BUNDLE");
454                        final int[] pm={RawMaterial.MATERIAL_CLOTH};
455                        final int[][] data=fetchFoundResourceData(mob,
456                                                                                                        woodRequired,"cloth",pm,
457                                                                                                        0,null,null,
458                                                                                                        bundling,
459                                                                                                        autoGenerate,
460                                                                                                        enhancedTypes);
461                        if(data==null)
462                                return false;
463                        fixDataForComponents(data,woodRequiredStr,(autoGenerate>0) && (woodRequired==0),componentsFoundList, 1);
464                        woodRequired=data[0][FOUND_AMT];
465                        if(!super.invoke(mob,commands,givenTarget,auto,asLevel))
466                                return false;
467                        final MaterialLibrary.DeadResourceRecord deadMats;
468                        if((componentsFoundList.size() > 0)||(autoGenerate>0))
469                                deadMats = new MaterialLibrary.DeadResourceRecord();
470                        else
471                                deadMats = CMLib.materials().destroyResources(mob.location(),woodRequired,data[0][FOUND_CODE],data[1][FOUND_CODE],null,null);
472                        final MaterialLibrary.DeadResourceRecord deadComps = CMLib.ableComponents().destroyAbilityComponents(componentsFoundList);
473                        final int lostValue=autoGenerate>0?0:(deadMats.lostValue + deadComps.lostValue);
474                        buildingI=CMClass.getItem(foundRecipe.get(RCP_CLASSTYPE));
475                        if(buildingI==null)
476                        {
477                                commonTell(mob,L("There's no such thing as a @x1!!!",foundRecipe.get(RCP_CLASSTYPE)));
478                                return false;
479                        }
480                        duration=getDuration(CMath.s_int(foundRecipe.get(RCP_TICKS)),mob,CMath.s_int(foundRecipe.get(RCP_LEVEL)),4);
481                        buildingI.setMaterial(super.getBuildingMaterial(woodRequired, data, compData));
482                        String itemName=determineFinalName(foundRecipe.get(RCP_FINALNAME),buildingI.material(),deadMats,deadComps);
483                        if(bundling)
484                                itemName="a "+woodRequired+"# "+itemName;
485                        else
486                        if(itemName.endsWith("s"))
487                                itemName="some "+itemName;
488                        else
489                                itemName=CMLib.english().startWithAorAn(itemName);
490                        buildingI.setName(itemName);
491                        startStr=L("<S-NAME> start(s) knitting @x1.",buildingI.name());
492                        displayText=L("You are knitting @x1",buildingI.name());
493                        verb=L("knitting @x1",buildingI.name());
494                        playSound="scissor.wav";
495                        buildingI.setDisplayText(L("@x1 lies here",itemName));
496                        buildingI.setDescription(itemName+". ");
497                        buildingI.basePhyStats().setWeight(getStandardWeight(woodRequired+compData[CF_AMOUNT],bundling));
498                        final int hardness=RawMaterial.CODES.HARDNESS(buildingI.material())-1;
499                        buildingI.setBaseValue(CMath.s_int(foundRecipe.get(RCP_VALUE)));
500                        buildingI.basePhyStats().setLevel(CMath.s_int(foundRecipe.get(RCP_LEVEL))+(hardness*3));
501                        setBrand(mob, buildingI);
502                        final int capacity=CMath.s_int(foundRecipe.get(RCP_CAPACITY));
503                        final long canContain=getContainerType(foundRecipe.get(RCP_CONTAINMASK));
504                        final int armordmg=CMath.s_int(foundRecipe.get(RCP_ARMORDMG));
505                        final String spell=(foundRecipe.size()>RCP_SPELL)?foundRecipe.get(RCP_SPELL).trim():"";
506                        if(bundling)
507                                buildingI.setBaseValue(lostValue);
508                        addSpells(buildingI,spell,deadMats.lostProps,deadComps.lostProps);
509                        if(buildingI instanceof Weapon)
510                        {
511                                ((Weapon)buildingI).setWeaponClassification(Weapon.CLASS_NATURAL);
512                                setWeaponTypeClass((Weapon)buildingI,misctype);
513                                buildingI.basePhyStats().setDamage(armordmg);
514                                ((Weapon)buildingI).setRawProperLocationBitmap(Wearable.WORN_WIELD|Wearable.WORN_HELD);
515                                ((Weapon)buildingI).setRawLogicalAnd((capacity>1));
516                        }
517                        if((buildingI instanceof Armor)&&(!(buildingI instanceof FalseLimb)))
518                        {
519                                if((capacity>0)&&(buildingI instanceof Container))
520                                {
521                                        ((Container)buildingI).setCapacity(capacity+woodRequired);
522                                        ((Container)buildingI).setContainTypes(canContain);
523                                }
524                                ((Armor)buildingI).basePhyStats().setArmor(0);
525                                if(armordmg!=0)
526                                        ((Armor)buildingI).basePhyStats().setArmor(armordmg+(baseYield()+abilityCode()-1)+hardness);
527                                setWearLocation(buildingI,misctype,0);
528                        }
529                        if(buildingI instanceof Rideable)
530                        {
531                                setRideBasis((Rideable)buildingI,misctype);
532                        }
533                        buildingI.recoverPhyStats();
534                        buildingI.text();
535                        buildingI.recoverPhyStats();
536                }
537
538                messedUp=!proficiencyCheck(mob,0,auto);
539
540                if(bundling)
541                {
542                        messedUp=false;
543                        duration=1;
544                        verb=L("bundling @x1",RawMaterial.CODES.NAME(buildingI.material()).toLowerCase());
545                        startStr=L("<S-NAME> start(s) @x1.",verb);
546                        displayText=L("You are @x1",verb);
547                }
548
549                if(autoGenerate>0)
550                {
551                        crafted.add(buildingI);
552                        return true;
553                }
554
555                final CMMsg msg=CMClass.getMsg(mob,buildingI,this,getActivityMessageType(),startStr);
556                if(mob.location().okMessage(mob,msg))
557                {
558                        mob.location().send(mob,msg);
559                        buildingI=(Item)msg.target();
560                        beneficialAffect(mob,mob,asLevel,duration);
561                        enhanceItem(mob,buildingI,recipeLevel,enhancedTypes);
562                }
563                else
564                if(bundling)
565                {
566                        messedUp=false;
567                        aborted=false;
568                        unInvoke();
569                }
570                return true;
571        }
