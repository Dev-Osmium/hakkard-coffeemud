089        @Override
090        public boolean tick(final Tickable ticking, final int tickID)
091        {
092                if((affected!=null)&&(affected instanceof MOB)&&(tickID==Tickable.TICKID_MOB))
093                {
094                        final MOB mob=(MOB)affected;
095                        if((buildingI==null)
096                        ||((fireRequired)&&(getRequiredFire(mob,0)==null))
097                        ||(theSpell==null))
098                        {
099                                aborted=true;
100                                unInvoke();
101                        }
102                        else
103                        if(tickUp==0)
104                        {
105                                if((theSpell.classificationCode()&Ability.ALL_ACODES)==Ability.ACODE_PRAYER)
106                                {
107                                        commonEmote(mob,L("<S-NAME> start(s) praying for @x1.",buildingI.name()));
108                                        displayText=L("You are praying for @x1",buildingI.name());
109                                        verb=L("praying for @x1",buildingI.name());
110                                }
111                                else
112                                {
113                                        commonEmote(mob,L("<S-NAME> start(s) brewing @x1.",buildingI.name()));
114                                        displayText=L("You are brewing @x1",buildingI.name());
115                                        verb=L("brewing @x1",buildingI.name());
116                                        playSound="hotspring.wav";
117                                }
118                        }
119                }
120                return super.tick(ticking,tickID);
121        }

154        @Override
155        public boolean mayICraft(final Item I)
156        {
157                if(I==null)
158                        return false;
159                if(!super.mayBeCrafted(I))
160                        return false;
161                if(!(I instanceof Potion))
162                        return false;
163                final Potion P=(Potion)I;
164                if((P.liquidType()==RawMaterial.RESOURCE_LIQUOR)
165                ||(P.liquidType()==RawMaterial.RESOURCE_POISON))
166                        return false;
167                final List<Ability> spells=P.getSpells();
168                if((spells == null)||(spells.size()==0))
169                        return false;
170                for(final Ability A : spells)
171                        if(((A.classificationCode()&Ability.ALL_ACODES)!=Ability.ACODE_SPELL)
172                        &&((A.classificationCode()&Ability.ALL_ACODES)!=Ability.ACODE_PRAYER))
173                                return false;
174                return true;
175        }

177        @Override
178        public void unInvoke()
179        {
180                if(canBeUninvoked())
181                {
182                        if(affected instanceof MOB)
183                        {
184                                final MOB mob=(MOB)affected;
185                                if((buildingI!=null)&&(!aborted))
186                                {
187                                        if(messedUp)
188                                        {
189                                                if(activity==CraftingActivity.LEARNING)
190                                                        commonEmote(mob,L("<S-NAME> fail(s) to learn how to make @x1.",buildingI.name()));
191                                                else
192                                                if(oldName.length()>0)
193                                                        commonTell(mob,L("Something went wrong! @x1 explodes!",(Character.toUpperCase(oldName.charAt(0))+oldName.substring(1))));
194                                                buildingI.destroy();
195                                        }
196                                        else
197                                        if(activity==CraftingActivity.LEARNING)
198                                        {
199                                                deconstructRecipeInto(mob, buildingI, recipeHolder );
200                                                buildingI.destroy();
201                                        }
202                                        else
203                                                mob.addItem(buildingI);
204                                }
205                                buildingI=null;
206                        }
207                }
208                super.unInvoke();
209        }

211        protected int spellLevel(final MOB mob, final Ability A)
212        {
213                int lvl=CMLib.ableMapper().qualifyingLevel(mob,A);
214                if(lvl<0)
215                        lvl=CMLib.ableMapper().lowestQualifyingLevel(A.ID());
216                switch(lvl)
217                {
218                case 0:
219                        return lvl;
220                case 1:
221                        return lvl;
222                case 2:
223                        return lvl + 1;
224                case 3:
225                        return lvl + 1;
226                case 4:
227                        return lvl + 2;
228                case 5:
229                        return lvl + 2;
230                case 6:
231                        return lvl + 3;
232                case 7:
233                        return lvl + 3;
234                case 8:
235                        return lvl + 4;
236                case 9:
237                        return lvl + 4;
238                default:
239                        return lvl + 5;
240                }
241        }

249        protected Item buildItem(final Ability theSpell, final int level)
250        {
251                buildingI=CMClass.getItem("GenPotion");
252                ((Potion)buildingI).setSpellList(theSpell.ID());
253                buildingI.setName(L("a potion of @x1",theSpell.name().toLowerCase()));
254                buildingI.setDisplayText(L("a potion of @x1 sits here.",theSpell.name().toLowerCase()));
255                buildingI.setDescription("");
256                buildingI.basePhyStats().setLevel(level);
257                buildingI.phyStats().setLevel(level);
258                buildingI.recoverPhyStats();
259                buildingI.text();
260                return buildingI;
261        }

269        @Override
270        protected boolean autoGenInvoke(final MOB mob, final List<String> commands, final Physical givenTarget, final boolean auto, final int asLevel, final int autoGenerate, final boolean forceLevels, final List<Item> crafted)
271        {
272                if(autoGenerate>0)
273                {
274                        final Ability theSpell=super.getCraftableSpellRecipeSpell(commands);
275                        if(theSpell==null)
276                                return false;
277                        final int level=spellLevel(mob,theSpell);
278                        buildingI=buildItem(theSpell, level);
279                        crafted.add(buildingI);
280                        if(forceLevels)
281                        {
282                                final int minLevel=CMLib.ableMapper().lowestQualifyingLevel(theSpell.ID());
283                                buildingI.basePhyStats().setLevel(minLevel);
284                                buildingI.phyStats().setLevel(buildingI.basePhyStats().level());
285                        }
286                        return true;
287                }
288                if(super.checkStop(mob, commands))
289                        return true;
290                randomRecipeFix(mob,addRecipes(mob,loadRecipes()),commands,0);
291                if(commands.size()<1)
292                {
293                        commonTell(mob,L("Brew what? Enter \"brew list\" for a list, or \"brew stop\" to cancel."));
294                        return false;
295                }
296                final int[] cols={
297                                CMLib.lister().fixColWidth(25,mob.session()),
298                        };
299                final List<List<String>> recipes=addRecipes(mob,loadRecipes());
300                final String pos=commands.get(commands.size()-1);
301                if(((commands.get(0))).equalsIgnoreCase("list"))
302                {
303                        String mask=CMParms.combine(commands,1);
304                        boolean allFlag=false;
305                        if(mask.equalsIgnoreCase("all"))
306                        {
307                                allFlag=true;
308                                mask="";
309                        }
310                        final StringBuffer buf=new StringBuffer(L("Potions you know how to brew:\n\r"));
311                        buf.append(CMStrings.padRight(L("Spell"),cols[0])+" "+CMStrings.padRight(L("Spell"),cols[0])+" "+CMStrings.padRight(L("Spell"),cols[0]));
312                        int toggler=1;
313                        final int toggleTop=3;
314                        final List<List<String>> listRecipes=((mask.length()==0) || mask.equalsIgnoreCase("all")) ? recipes : super.matchingRecipeNames(recipes, mask, true);
315                        for(int r=0;r<listRecipes.size();r++)
316                        {
317                                final List<String> V=listRecipes.get(r);
318                                if(V.size()>0)
319                                {
320                                        final String spell=V.get(0);
321                                        final Ability A=mob.fetchAbility(spell);
322                                        if((A!=null)
323                                        &&((spellLevel(mob,A)>=0)||(allFlag))
324                                        &&((xlevel(mob)>=spellLevel(mob,A))||(allFlag)))
325                                        {
326                                                buf.append(CMStrings.padRight(A.name(),cols[0])+((toggler!=toggleTop)?" ":"\n\r"));
327                                                if(++toggler>toggleTop)
328                                                        toggler=1;
329                                        }
330                                }
331                        }
332                        if(toggler!=1)
333                                buf.append("\n\r");
334                        commonTell(mob,buf.toString());
335                        return true;
336                }
337                else
338                if((!auto)&&(commands.size()<2))
339                {
340                        commonEmote(mob,L("You must specify what magic you wish to brew, and the container to brew it in."));
341                        return false;
342                }
343                else
344                {
345                        buildingI=getTarget(mob,null,givenTarget,CMParms.parse(pos),Wearable.FILTER_UNWORNONLY);
346                        commands.remove(pos);
347                        if(buildingI==null)
348                                return false;
349                        if(!mob.isMine(buildingI))
350                        {
351                                commonTell(mob,L("You'll need to pick that up first."));
352                                return false;
353                        }
354                        if(!(buildingI instanceof Container))
355                        {
356                                commonTell(mob,L("There's nothing in @x1 to brew!",buildingI.name(mob)));
357                                return false;
358                        }
359                        if(!(buildingI instanceof Drink))
360                        {
361                                commonTell(mob,L("You can't drink out of a @x1.",buildingI.name(mob)));
362                                return false;
363                        }
364                        if(((Drink)buildingI).liquidRemaining()==0)
365                        {
366                                commonTell(mob,L("The @x1 contains no liquid base.  Water is probably fine.",buildingI.name(mob)));
367                                return false;
368                        }
369                        if(buildingI.material()!=RawMaterial.RESOURCE_GLASS)
370                        {
371                                commonTell(mob,L("You can only brew into glass containers."));
372                                return false;
373                        }
374                        activity = CraftingActivity.CRAFTING;
375                        final String recipeName=CMParms.combine(commands,0);
376                        theSpell=null;
377                        int theSpellLevel=1;
378                        String ingredient="";
379                        for(int r=0;r<recipes.size();r++)
380                        {
381                                final List<String> V=recipes.get(r);
382                                if(V.size()>0)
383                                {
384                                        final String spell=V.get(0);
385                                        final Ability A=mob.fetchAbility(spell);
386                                        if((A!=null)
387                                        &&(xlevel(mob)>=spellLevel(mob,A))
388                                        &&(A.name().equalsIgnoreCase(recipeName)))
389                                        {
390                                                theSpell=A;
391                                                theSpellLevel=spellLevel(mob, A);
392                                                ingredient=V.get(1);
393                                        }
394                                }
395                        }
396                        if(theSpell==null)
397                        {
398                                commonTell(mob,L("You don't know how to brew '@x1'.  Try \"brew list\" for a list.",recipeName));
399                                return false;
400                        }
401                        int experienceToLose=10;
402                        if((theSpell.classificationCode()&Ability.ALL_ACODES)==Ability.ACODE_PRAYER)
403                        {
404                                fireRequired=false;
405                                experienceToLose+=CMLib.ableMapper().qualifyingLevel(mob,theSpell)*10;
406                                experienceToLose-=CMLib.ableMapper().qualifyingClassLevel(mob,theSpell)*5;
407                        }
408                        else
409                        {
410                                fireRequired=true;
411                                final Item fire=getRequiredFire(mob,0);
412                                if(fire==null)
413                                        return false;
414                                experienceToLose+=CMLib.ableMapper().qualifyingLevel(mob,theSpell)*10;
415                                experienceToLose-=CMLib.ableMapper().qualifyingClassLevel(mob,theSpell)*5;
416                        }
417                        final int resourceType=RawMaterial.CODES.FIND_IgnoreCase(ingredient);
418
419                        boolean found=false;
420                        final List<Item> V=((Container)buildingI).getContents();
421                        if(resourceType>0)
422                        {
423                                if(((Drink)buildingI).liquidType()==resourceType)
424                                {
425                                        found=true;
426                                        if(V.size()>0)
427                                        {
428                                                commonTell(mob,L("The extraneous stuff from the @x1 must be removed before starting.",buildingI.name(mob)));
429                                                return false;
430                                        }
431                                }
432                                else
433                                for(int i=0;i<V.size();i++)
434                                {
435                                        final Item I=V.get(i);
436                                        if(I.material()==resourceType)
437                                                found=true;
438                                        else
439                                        {
440                                                commonTell(mob,L("The @x1 must be removed from the @x2 before starting.",I.name(mob),buildingI.name(mob)));
441                                                return false;
442                                        }
443                                }
444                                if(!found)
445                                {
446                                        commonTell(mob,L("This potion requires @x1.  Please place some inside the @x2 and try again.",ingredient,buildingI.name(mob)));
447                                        return false;
448                                }
449                        }
450                        if(experienceToLose<10)
451                                experienceToLose=10;
452
453                        if(!super.invoke(mob,commands,givenTarget,auto,asLevel))
454                                return false;
455
456                        playSound=null;
457                        experienceToLose=getXPCOSTAdjustment(mob,experienceToLose);
458                        CMLib.leveler().postExperience(mob,null,null,-experienceToLose,false);
459                        commonTell(mob,L("You lose @x1 experience points for the effort.",""+experienceToLose));
460                        oldName=buildingI.name();
461                        buildingI.destroy();
462                        buildingI=buildItem(theSpell, theSpellLevel);
463                        setBrand(mob, buildingI);
464
465                        int duration=CMLib.ableMapper().qualifyingLevel(mob,theSpell)*5;
466                        if(duration<10)
467                                duration=10;
468                        messedUp=!proficiencyCheck(mob,0,auto);
469
470                        final CMMsg msg=CMClass.getMsg(mob,buildingI,this,getActivityMessageType(),null);
471                        if(mob.location().okMessage(mob,msg))
472                        {
473                                mob.location().send(mob,msg);
474                                buildingI=(Item)msg.target();
475                                beneficialAffect(mob,mob,asLevel,duration);
476                        }
477                }
478                return true;
479        }
