109        @Override
110        public void unInvoke()
111        {
112                if(canBeUninvoked())
113                {
114                        if(affected instanceof MOB)
115                        {
116                                final MOB mob=(MOB)affected;
117                                if((buildingI!=null)&&(!aborted))
118                                {
119                                        if(messedUp)
120                                        {
121                                                if(activity == CraftingActivity.LEARNING)
122                                                        commonEmote(mob,L("<S-NAME> fail(s) to learn how to make @x1.",buildingI.name()));
123                                                else
124                                                        commonTell(mob,L("You've ruined @x1!",buildingI.name(mob)));
125                                                buildingI.destroy();
126                                        }
127                                        else
128                                        if(activity==CraftingActivity.LEARNING)
129                                        {
130                                                deconstructRecipeInto(mob, buildingI, recipeHolder );
131                                                buildingI.destroy();
132                                        }
133                                        else
134                                        {
135                                                dropAWinner(mob,buildingI);
136                                                CMLib.achievements().possiblyBumpAchievement(mob, AchievementLibrary.Event.CRAFTING, 1, this);
137                                        }
138                                }
139                                buildingI=null;
140                        }
141                }
142                super.unInvoke();
143        }

151        @Override
152        public boolean mayICraft(final Item I)
153        {
154                if(I==null)
155                        return false;
156                if(!super.mayBeCrafted(I))
157                        return false;
158                if(!CMLib.flags().isDeadlyOrMaliciousEffect(I))
159                        return false;
160                if(I instanceof Ammunition)
161                        return false;
162                if(I instanceof Rideable)
163                        return true;
164                if(I instanceof Shield)
165                        return false;
166                if(I instanceof Weapon)
167                        return (isANativeItem(I.Name()));
168                if(I instanceof Armor)
169                        return true;
170                if(I instanceof FalseLimb)
171                        return true;
172                if(I.rawProperLocationBitmap()==Wearable.WORN_HELD)
173                        return true;
174                return (isANativeItem(I.Name()));
175        }

182        @Override
183        protected boolean canMend(final MOB mob, final Environmental E, final boolean quiet)
184        {
185                if(!super.canMend(mob,E,quiet))
186                        return false;
187                if((!(E instanceof Item))
188                ||(!mayICraft((Item)E)))
189                {
190                        if(!quiet)
191                                commonTell(mob,L("That's not a torturesmithing item."));
192                        return false;
193                }
194                return true;
195        }

209        @Override
210        protected boolean autoGenInvoke(final MOB mob, final List<String> commands, final Physical givenTarget, final boolean auto,
211                                                                 final int asLevel, final int autoGenerate, final boolean forceLevels, final List<Item> crafted)
212        {
213                if(super.checkStop(mob, commands))
214                        return true;
215
216                if(super.checkInfo(mob, commands))
217                        return true;
218
219                randomRecipeFix(mob,addRecipes(mob,loadRecipes()),commands,autoGenerate);
220                if(commands.size()==0)
221                {
222                        commonTell(mob,L("Make what? Enter \"@x1 list\" for a list, \"@x2 info <item>\", \"@x2 learn <item>\" to gain recipes,"
223                                                        + " or \"@x3 stop\" to cancel.", triggerStrings()[0].toLowerCase(), triggerStrings()[0].toLowerCase(),
224                                                        triggerStrings()[0].toLowerCase()));
225                        return false;
226                }
227                final List<List<String>> recipes=addRecipes(mob,loadRecipes());
228                final String str=commands.get(0);
229                String startStr=null;
230                bundling=false;
231                int duration=4;
232                @SuppressWarnings("unused")
233                int recipeLevel=1;
234                if(str.equalsIgnoreCase("list"))
235                {
236                        String mask=CMParms.combine(commands,1);
237                        boolean allFlag=false;
238                        if(mask.equalsIgnoreCase("all"))
239                        {
240                                allFlag=true;
241                                mask="";
242                        }
243                        final StringBuffer buf=new StringBuffer(L("@x1 Lvl Material required\n\r",CMStrings.padRight(L("Item"),16)));
244                        final List<List<String>> listRecipes=((mask.length()==0) || mask.equalsIgnoreCase("all")) ? recipes : super.matchingRecipeNames(recipes, mask, true);
245                        for(int r=0;r<listRecipes.size();r++)
246                        {
247                                final List<String> V=listRecipes.get(r);
248                                if(V.size()>0)
249                                {
250                                        final String item=replacePercent(V.get(RCP_FINALNAME),"");
251                                        final int level=CMath.s_int(V.get(RCP_LEVEL));
252                                        String mat=V.get(RCP_MATERIAL);
253                                        final String wood=getComponentDescription(mob,V,RCP_WOOD);
254                                        if(wood.length()>5)
255                                                mat="";
256                                        if((level<=xlevel(mob))||allFlag)
257                                                buf.append(CMStrings.padRight(item,16)+" "+CMStrings.padRight(""+level,3)+" "+wood+" "+mat.toLowerCase()+"\n\r");
258                                }
259                        }
260                        commonTell(mob,buf.toString());
261                        return true;
262                }
263                else
264                if(((commands.get(0))).equalsIgnoreCase("learn"))
265                {
266                        return doLearnRecipe(mob, commands, givenTarget, auto, asLevel);
267                }
268
269                activity = CraftingActivity.CRAFTING;
270                buildingI=null;
271                messedUp=false;
272                int amount=-1;
273                if((commands.size()>1)&&(CMath.isNumber(commands.get(commands.size()-1))))
274                {
275                        amount=CMath.s_int(commands.get(commands.size()-1));
276                        commands.remove(commands.size()-1);
277                }
278                final String recipeName=CMParms.combine(commands,0);
279                List<String> foundRecipe=null;
280                final List<List<String>> matches=matchingRecipeNames(recipes,recipeName,true);
281                for(int r=0;r<matches.size();r++)
282                {
283                        final List<String> V=matches.get(r);
284                        if(V.size()>0)
285                        {
286                                final int level=CMath.s_int(V.get(RCP_LEVEL));
287                                if((autoGenerate>0)||(level<=xlevel(mob)))
288                                {
289                                        foundRecipe=V;
290                                        recipeLevel=level;
291                                        break;
292                                }
293                        }
294                }
295                if(foundRecipe==null)
296                {
297                        commonTell(mob,L("You don't know how to make a '@x1'.  Try \"@x2 list\" for a list.",recipeName,triggerStrings[0].toLowerCase()));
298                        return false;
299                }
300
301                final String woodRequiredStr = foundRecipe.get(RCP_WOOD);
302                final int[] compData = new int[CF_TOTAL];
303                final String realRecipeName=replacePercent(foundRecipe.get(RCP_FINALNAME),"");
304                final List<Object> componentsFoundList=getAbilityComponents(mob, woodRequiredStr, "make "+CMLib.english().startWithAorAn(realRecipeName),autoGenerate,compData,1);
305                if(componentsFoundList==null)
306                        return false;
307                int woodRequired=CMath.s_int(woodRequiredStr);
308                woodRequired=adjustWoodRequired(woodRequired,mob);
309
310                if(amount>woodRequired)
311                        woodRequired=amount;
312                final String misctype=foundRecipe.get(RCP_MISCTYPE);
313                final String materialtype=foundRecipe.get(RCP_MATERIAL);
314                int[] pm=null;
315                if(materialtype.equalsIgnoreCase("wood"))
316                {
317                        pm=new int[1];
318                        pm[0]=RawMaterial.MATERIAL_WOODEN;
319                }
320                else
321                if(materialtype.equalsIgnoreCase("metal"))
322                {
323                        pm=new int[2];
324                        pm[0]=RawMaterial.MATERIAL_METAL;
325                        pm[1]=RawMaterial.MATERIAL_MITHRIL;
326                }
327                else
328                if(materialtype.equalsIgnoreCase("cloth"))
329                {
330                        pm=new int[1];
331                        pm[0]=RawMaterial.MATERIAL_CLOTH;
332                }
333                bundling=misctype.equalsIgnoreCase("BUNDLE");
334                final int[][] data=fetchFoundResourceData(mob,
335                                                                                                woodRequired,"wood or cloth",pm,
336                                                                                                0,null,null,
337                                                                                                bundling,
338                                                                                                autoGenerate,
339                                                                                                null);
340                if(data==null)
341                        return false;
342                woodRequired=data[0][FOUND_AMT];
343                if(!super.invoke(mob,commands,givenTarget,auto,asLevel))
344                        return false;
345                final MaterialLibrary.DeadResourceRecord deadMats;
346                if((componentsFoundList.size() > 0)||(autoGenerate>0))
347                        deadMats = new MaterialLibrary.DeadResourceRecord();
348                else
349                        deadMats = CMLib.materials().destroyResources(mob.location(),data[0][FOUND_AMT],data[0][FOUND_CODE],data[1][FOUND_CODE],null,null);
350                final MaterialLibrary.DeadResourceRecord deadComps = CMLib.ableComponents().destroyAbilityComponents(componentsFoundList);
351                final int lostValue=autoGenerate>0?0:(deadMats.lostValue + deadComps.lostValue);
352                buildingI=CMClass.getItem(foundRecipe.get(RCP_CLASSTYPE));
353                if(buildingI==null)
354                {
355                        commonTell(mob,L("There's no such thing as a @x1!!!",foundRecipe.get(RCP_CLASSTYPE)));
356                        return false;
357                }
358                duration=getDuration(CMath.s_int(foundRecipe.get(RCP_TICKS)),mob,CMath.s_int(foundRecipe.get(RCP_LEVEL)),4);
359                buildingI.setMaterial(super.getBuildingMaterial(woodRequired, data, compData));
360                String itemName=determineFinalName(foundRecipe.get(RCP_FINALNAME),buildingI.material(),deadMats,deadComps);
361                if(bundling)
362                        itemName="a "+woodRequired+"# "+itemName;
363                else
364                        itemName=CMLib.english().startWithAorAn(itemName);
365                buildingI.setName(itemName);
366                startStr=L("<S-NAME> start(s) making @x1.",buildingI.name());
367                displayText=L("You are making @x1",buildingI.name());
368                verb=L("making @x1",buildingI.name());
369                playSound="hammer.wav";
370                buildingI.setDisplayText(L("@x1 lies here",itemName));
371                buildingI.setDescription(itemName+". ");
372                buildingI.basePhyStats().setWeight(woodRequired);
373                buildingI.setBaseValue(CMath.s_int(foundRecipe.get(RCP_VALUE))+(woodRequired*(RawMaterial.CODES.VALUE(data[0][FOUND_CODE]))));
374                buildingI.basePhyStats().setLevel(CMath.s_int(foundRecipe.get(RCP_LEVEL)));
375                setBrand(mob, buildingI);
376                final int capacity=CMath.s_int(foundRecipe.get(RCP_CAPACITY));
377                final int armordmg=CMath.s_int(foundRecipe.get(RCP_ARMORDMG));
378                final int hardness=RawMaterial.CODES.HARDNESS(buildingI.material())-3;
379                final String spell=(foundRecipe.size()>RCP_SPELL)?foundRecipe.get(RCP_SPELL).trim():"";
380                addSpells(buildingI,spell,deadMats.lostProps,deadComps.lostProps);
381                if(buildingI instanceof Container)
382                {
383                        ((Container)buildingI).setCapacity(capacity+woodRequired);
384                        if(misctype.equalsIgnoreCase("LID"))
385                                ((Container)buildingI).setDoorsNLocks(true,false,true,false,false,false);
386                        else
387                        if(misctype.equalsIgnoreCase("LOCK"))
388                        {
389                                ((Container)buildingI).setDoorsNLocks(true,false,true,true,false,true);
390                                ((Container)buildingI).setKeyName(Double.toString(Math.random()));
391                        }
392                        else
393                                ((Container)buildingI).setContainTypes(getContainerType(misctype));
394                }
395                if(buildingI instanceof Rideable)
396                {
397                        setRideBasis((Rideable)buildingI,misctype);
398                        if(capacity==0)
399                                ((Rideable)buildingI).setRiderCapacity(1);
400                        else
401                        if(capacity<5)
402                                ((Rideable)buildingI).setRiderCapacity(capacity);
403                }
404                if((buildingI instanceof Armor)&&(!(buildingI instanceof FalseLimb)))
405                {
406                        ((Armor)buildingI).basePhyStats().setArmor(0);
407                        if(armordmg!=0)
408                                ((Armor)buildingI).basePhyStats().setArmor(armordmg+(baseYield()+abilityCode()-1));
409                        setWearLocation(buildingI,misctype,hardness);
410                }
411                if(buildingI instanceof Drink)
412                {
413                        if(CMLib.flags().isGettable(buildingI))
414                        {
415                                ((Drink)buildingI).setLiquidHeld(capacity*50);
416                                ((Drink)buildingI).setThirstQuenched(250);
417                                if((capacity*50)<250)
418                                        ((Drink)buildingI).setThirstQuenched(capacity*50);
419                                ((Drink)buildingI).setLiquidRemaining(0);
420                        }
421                }
422                if((buildingI instanceof Wand)
423                &&(foundRecipe.get(RCP_CAPACITY).trim().length()>0))
424                {
425                        ((Wand)buildingI).setMaxUses(capacity);
426                }
427                if(bundling)
428                        buildingI.setBaseValue(lostValue);
429                buildingI.recoverPhyStats();
430                buildingI.text();
431                buildingI.recoverPhyStats();
432
433                messedUp=!proficiencyCheck(mob,0,auto);
434
435                if(autoGenerate>0)
436                {
437                        crafted.add(buildingI);
438                        return true;
439                }
440
441                final CMMsg msg=CMClass.getMsg(mob,buildingI,this,getActivityMessageType(),startStr);
442                if(mob.location().okMessage(mob,msg))
443                {
444                        mob.location().send(mob,msg);
445                        buildingI=(Item)msg.target();
446                        beneficialAffect(mob,mob,asLevel,duration);
447                }
448                return true;
449        }
