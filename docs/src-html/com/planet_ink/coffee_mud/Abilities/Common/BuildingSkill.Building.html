204                {
205                        if((affected!=null)&&(affected instanceof MOB)&&(!helping))
206                        {
207                                final MOB mob=(MOB)affected;
208                                if(!aborted)
209                                {

216                                                this.buildComplete(mob, recipe, room, dir, designTitle, designDescription);

222                                                                mob.location().send(mob,msg);
223                                                }
224                                        }

275        protected void demolishRoom(final MOB mob, final Room room)
276        {
277                final LandTitle title=CMLib.law().getLandTitle(room);
278                if(title==null)
279                        return;
280                Room returnToRoom=null;
281                Room backupToRoom1=null;
282                Room backupToRoom2=null;
283                for(int d=0;d<Directions.NUM_DIRECTIONS();d++)
284                {
285                        final Room R=room.getRoomInDir(d);
286                        if(CMLib.law().doesOwnThisLand(mob, R))
287                        {
288                                returnToRoom=R;
289                                break;
290                        }
291                        final LandTitle adjacentTitle=CMLib.law().getLandTitle(R);
292                        if((adjacentTitle==null)||(adjacentTitle.getOwnerName().length()>0))
293                                backupToRoom1=R;
294                        else
295                        if(R.roomID().length()>0)
296                                backupToRoom2=R;
297                }
298                if(returnToRoom==null)
299                        returnToRoom=backupToRoom1;
300                if(returnToRoom==null)
301                        returnToRoom=backupToRoom2;
302                if(returnToRoom==null)
303                        returnToRoom=mob.getStartRoom();
304                if(returnToRoom==null)
305                        returnToRoom=room.getArea().getRandomProperRoom();
306                if(returnToRoom==null)
307                        returnToRoom=room.getArea().getRandomMetroRoom();
308                if(returnToRoom==null)
309                        returnToRoom=CMLib.map().getRandomRoom();
310                final Room theRoomToReturnTo=returnToRoom;
311                room.eachInhabitant(new EachApplicable<MOB>()
312                {
313                        @Override
314                        public void apply(final MOB a)
315                        {
316                                theRoomToReturnTo.bringMobHere(a, false);
317                        }
318                });
319                room.eachItem(new EachApplicable<Item>()
320                {
321                        @Override
322                        public void apply(final Item a)
323                        {
324                                theRoomToReturnTo.addItem(a,Expire.Player_Drop);
325                        }
326                });
327                title.setOwnerName("");
328                title.updateLot(null); // this is neat -- this will obliterate leaf rooms around this one.
329                if((theRoomToReturnTo!=null)
330                &&(theRoomToReturnTo.rawDoors()[Directions.UP]==room)
331                &&(theRoomToReturnTo.getRawExit(Directions.UP)!=null))
332                {
333                        theRoomToReturnTo.getRawExit(Directions.UP).destroy();
334                        theRoomToReturnTo.setRawExit(Directions.UP, null);
335                }
336                CMLib.map().obliterateMapRoom(room);
337        }

339        private void removeEffects(final PhysicalAgent E, String extraProp)
340        {
341                extraProp=extraProp.trim();
342                if(extraProp.length()>0)
343                {
344                        final List<String> spells = CMParms.parseAny(extraProp, ")", true);
345                        for(String spellName : spells)
346                        {
347                                final int x=spellName.indexOf('(');
348                                if(x>0)
349                                        spellName=spellName.substring(0,x);
350                                final Ability A=E.fetchEffect(spellName);
351                                if(A!=null)
352                                {
353                                        A.unInvoke();
354                                        E.delEffect(A);
355                                }
356                                else
357                                {
358                                        final Behavior B=E.fetchBehavior(spellName);
359                                        if(B!=null)
360                                                E.delBehavior(B);
361                                }
362                        }
363                }
364        }

366        private void addEffects(final PhysicalAgent E, final Room R2, String extraProp)
367        {
368                extraProp=extraProp.trim();
369                if(extraProp.length()>0)
370                {
371                        final List<String> spells = CMParms.parseAny(extraProp, ")", true);
372                        for(String spellName : spells)
373                        {
374                                String parms="";
375                                final int x=spellName.indexOf('(');
376                                if(x>0)
377                                {
378                                        parms=spellName.substring(x+1);
379                                        spellName=spellName.substring(0,x);
380                                }
381                                if(parms.trim().equalsIgnoreCase("@remove"))
382                                {
383                                        final Ability A=E.fetchEffect(spellName);
384                                        if(A!=null)
385                                        {
386                                                A.unInvoke();
387                                                E.delEffect(A);
388                                        }
389                                        else
390                                        {
391                                                final Behavior B=E.fetchBehavior(spellName);
392                                                if(B!=null)
393                                                        E.delBehavior(B);
394                                        }
395                                        continue;
396                                }
397                                final Ability A=CMClass.getAbility(spellName);
398                                if(A!=null)
399                                {
400
401                                        if(parms.length()>0)
402                                                A.setMiscText(parms);
403                                        else
404                                        if(A.ID().equals("Prop_RoomView"))
405                                                A.setMiscText(CMLib.map().getExtendedRoomID(R2));
406                                        E.addNonUninvokableEffect(A);
407                                }
408                                else
409                                {
410                                        final Behavior B=CMClass.getBehavior(spellName);
411                                        if(parms.length()>0)
412                                                B.setParms(parms);
413                                        E.addBehavior(B);
414                                }
415                        }
416                }
417        }

465        protected Room buildNewRoomType(Room room, String newLocale, final String extraProp, int dimension)
466        {
467                Room R=null;
468                synchronized(("SYNC"+room.roomID()).intern())
469                {
470                        if((dimension == 0)&&(room instanceof GridLocale))
471                                dimension = ((GridLocale)room).xGridSize();
472                        room=CMLib.map().getRoom(room);
473                        if(newLocale.equalsIgnoreCase("IndoorWaterSurface")
474                        &&((room.domainType()&Room.INDOORS)==0))
475                                newLocale="WaterSurface";
476                        R=CMClass.getLocale(newLocale);
477                        R.setRoomID(room.roomID());
478                        R.setDisplayText(room.displayText());
479                        R.setDescription(room.description());
480                        if(R.image().equalsIgnoreCase(CMLib.protocol().getDefaultMXPImage(room)))
481                                R.setImage(null);
482                        if(R instanceof GridLocale)
483                        {
484                                ((GridLocale)R).setXGridSize(dimension);
485                                ((GridLocale)R).setYGridSize(dimension);
486                        }
487
488                        final Area area=room.getArea();
489                        if(area!=null)
490                                area.delProperRoom(room);
491                        R.setArea(area);
492                        for(int a=room.numEffects()-1;a>=0;a--)
493                        {
494                                final Ability A=room.fetchEffect(a);
495                                if(A!=null)
496                                {
497                                        room.delEffect(A);
498                                        R.addEffect(A);
499                                }
500                        }
501                        for(int i=room.numItems()-1;i>=0;i--)
502                        {
503                                final Item I=room.getItem(i);
504                                if(I!=null)
505                                {
506                                        room.delItem(I);
507                                        R.addItem(I);
508                                }
509                        }
510                        for(int m=room.numInhabitants()-1;m>=0;m--)
511                        {
512                                final MOB M=room.fetchInhabitant(m);
513                                if(M!=null)
514                                {
515                                        room.delInhabitant(M);
516                                        R.addInhabitant(M);
517                                        M.setLocation(R);
518                                }
519                        }
520                        CMLib.threads().deleteTick(room,-1);
521                        for(int d=0;d<R.rawDoors().length;d++)
522                        {
523                                if((R.rawDoors()[d]==null)
524                                ||(R.rawDoors()[d].roomID().length()>0))
525                                        R.rawDoors()[d]=room.rawDoors()[d];
526                        }
527                        for(int d=Directions.NUM_DIRECTIONS()-1;d>=0;d--)
528                        {
529                                if((R.rawDoors()[d]==null)
530                                ||(R.rawDoors()[d].roomID().length()>0))
531                                {
532                                        if(room.getRawExit(d)!=null)
533                                                R.setRawExit(d, (Exit)room.getRawExit(d).copyOf());
534                                }
535                        }
536                        final LandTitle title = CMLib.law().getLandTitle(room);
537                        if((title!=null)&&(title.gridLayout()))
538                        {
539                                final PairVector<Room,int[]> rooms=CMLib.tracking().buildGridList(R, title.getOwnerName(), 100);
540                                for(int dir=0;dir<Directions.NUM_DIRECTIONS();dir++)
541                                {
542                                        if(dir==Directions.GATE)
543                                                continue;
544                                        Room R3=R.getRoomInDir(dir);
545                                        if(R3 == null)
546                                        {
547                                                R3=CMLib.tracking().getCalculatedAdjacentRoom(rooms, R3, dir);
548                                                if(R3!=null)
549                                                {
550                                                        R.rawDoors()[dir]=R3;
551                                                        R3.rawDoors()[Directions.getOpDirectionCode(dir)]=R;
552                                                        CMLib.database().DBUpdateExits(R3);
553                                                }
554                                        }
555                                }
556                        }
557
558                        R.clearSky();
559                        R.startItemRejuv();
560                        try
561                        {
562                                boolean rebuild=false;
563                                for(final Enumeration<Room> r=CMLib.map().rooms();r.hasMoreElements();)
564                                {
565                                        final Room R2=r.nextElement();
566                                        rebuild=false;
567                                        for(int d=0;d<R2.rawDoors().length;d++)
568                                        {
569                                                if(R2.rawDoors()[d]==room)
570                                                {
571                                                        rebuild=true;
572                                                        R2.rawDoors()[d]=R;
573                                                }
574                                        }
575                                        if((rebuild)&&(R2 instanceof GridLocale))
576                                                ((GridLocale)R2).buildGrid();
577                                }
578                        }
579                        catch (final NoSuchElementException e)
580                        {
581                        }
582                        try
583                        {
584                                for(final Enumeration<MOB> e=CMLib.players().players();e.hasMoreElements();)
585                                {
586                                        final MOB M=e.nextElement();
587                                        if(M.getStartRoom()==room)
588                                                M.setStartRoom(R);
589                                        else
590                                        if(M.location()==room)
591                                                M.setLocation(R);
592                                }
593                        }
594                        catch (final NoSuchElementException e)
595                        {
596                        }
597                        if((R.domainType()==Room.DOMAIN_INDOORS_WATERSURFACE)
598                        ||(R.domainType()==Room.DOMAIN_OUTDOORS_WATERSURFACE))
599                        {
600                                if(dimension > 0)
601                                {
602                                        final Room R2=CMClass.getLocale("UnderWater");
603                                        R2.setRoomID(R.getArea().getNewRoomID(R,Directions.DOWN));
604                                        R2.setDisplayText(L("Under the water"));
605                                        R2.setDescription(L("You are swimming around under the water."));
606                                        R2.setArea(R.getArea());
607                                        R2.rawDoors()[Directions.UP]=R;
608                                        R2.setRawExit(Directions.UP,CMClass.getExit("Open"));
609                                        R.clearSky();
610                                        R.rawDoors()[Directions.DOWN]=R2;
611                                        R.setRawExit(Directions.DOWN,CMClass.getExit("Open"));
612                                        final LandTitle titleA=CMLib.law().getLandTitle(R);
613                                        if((titleA!=null)&&(CMLib.law().getLandTitle(R2)==null))
614                                        {
615                                                final LandTitle A2=(LandTitle)titleA.copyOf();
616                                                R2.addNonUninvokableEffect((Ability)A2);
617                                        }
618                                        final Ability capacityA = R.fetchEffect("Prop_ReqCapacity");
619                                        if(capacityA != null)
620                                        {
621                                                final Ability A2=(Ability)capacityA.copyOf();
622                                                R2.addNonUninvokableEffect(A2);
623                                        }
624                                        if(CMSecurity.isDebugging(CMSecurity.DbgFlag.PROPERTY))
625                                                Log.debugOut(ID(),R2.roomID()+" created for water.");
626                                        CMLib.database().DBCreateRoom(R2);
627                                        CMLib.database().DBUpdateExits(R2);
628                                }
629                        }
630
631                        R.getArea().fillInAreaRoom(R);
632                        addEffects(R,R,extraProp);
633                        if(CMSecurity.isDebugging(CMSecurity.DbgFlag.PROPERTY))
634                                Log.debugOut(ID(),R.roomID()+" updated.");
635                        CMLib.database().DBUpdateRoom(R);
636                        CMLib.database().DBUpdateExits(R);
637                        room.destroy();
638                }
639                return R;
640        }

642        protected void buildDoor(final String[] recipe, Room room, final int dir)
643        {
644                synchronized(("SYNC"+room.roomID()).intern())
645                {
646                        final String localeName = recipe[DAT_CLASS];
647                        String doorName = recipe[DAT_MISC];
648                        final String spells = recipe[DAT_PROPERTIES];
649                        //int size = CMath.s_int(recipe[DAT_MISC]);
650                        String closeWord=null;
651                        String openWord=null;
652                        String closedWord=null;
653                        String displayText="";
654                        if(doorName.indexOf("|")>0)
655                        {
656                                final List<String> split=CMParms.parseAny(doorName, '|',false);
657                                if(split.get(0).length()>0)
658                                        doorName=split.get(0);
659                                if((split.size()>1)&&(split.get(1).length()>0))
660                                        openWord=split.get(1);
661                                if((split.size()>2)&&(split.get(2).length()>0))
662                                        closeWord=split.get(2);
663                                if((split.size()>3)&&(split.get(3).length()>0))
664                                        closedWord=split.get(3);
665                                if((split.size()>4)&&(split.get(4).length()>0))
666                                        displayText=split.get(4);
667                        }
668                        if(closeWord == null)

1087                }

1089                CMLib.database().DBUpdateRoom(R);
1090                CMLib.database().DBUpdateExits(R);
