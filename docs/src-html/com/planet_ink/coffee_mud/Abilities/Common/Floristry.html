036public class Floristry extends CommonSkill
037{
038        @Override
039        public String ID()
040        {
041                return "Floristry";
042        }
043
044        private final static String localizedName = CMLib.lang().L("Floristry");
045
046        @Override
047        public String name()
048        {
049                return localizedName;
050        }
051
052        private static final String[] triggerStrings =I(new String[] {"FLORISTRY"});
053        @Override
054        public String[] triggerStrings()
055        {
056                return triggerStrings;
057        }
058
059        @Override
060        public int classificationCode()
061        {
062                return Ability.ACODE_COMMON_SKILL|Ability.DOMAIN_NATURELORE;
063        }
064
065        public String parametersFormat()
066        {
067                return "FLOWER_NAME";
068        }
069
070        protected Item          found           = null;
071        protected boolean       messedUp        = false;
072        protected String        foundName       = null;
073        protected boolean       lastAuto        = false;
074
075        @Override
076        protected boolean canBeDoneSittingDown()
077        {
078                return true;
079        }
080
081        public Floristry()
082        {
083                super();
084                displayText=L("You are evaluating...");
085                verb=L("evaluating");
086        }
087
088        @Override
089        public void unInvoke()
090        {
091                if(canBeUninvoked())
092                {
093                        if((affected!=null)
094                        &&(affected instanceof MOB)
095                        &&(!aborted)
096                        &&(!helping))
097                        {
098                                final MOB mob=(MOB)affected;
099                                if(messedUp)
100                                        commonTell(mob,L("You lose your concentration on @x1.",found.name()));
101                                else
102                                {
103                                        final List<String> flowerList=Resources.getFileLineVector(Resources.getFileResource("skills/floristry.txt",true));
104                                        Item origFound=found;
105                                        while(found != null)
106                                        {
107                                                if(found.phyStats().weight()>1)
108                                                        found=(Item)CMLib.materials().unbundle(found, 1, null);
109                                                String flower=null;
110                                                while((flowerList.size()>2)&&((flower==null)||(flower.trim().length()==0)))
111                                                        flower=flowerList.get(CMLib.dice().roll(1,flowerList.size(),-1)).trim().toLowerCase();
112        
113                                                if(found.rawSecretIdentity().length()>0)
114                                                {
115                                                        flower=found.rawSecretIdentity();
116                                                        found.setSecretIdentity("");
117                                                }
118        
119                                                commonTell(mob,L("@x1 appears to be @x2.",found.name(),flower));
120                                                String name=found.Name();
121                                                name=name.substring(0,name.length()-8).trim();
122                                                if(name.startsWith("a pound of"))
123                                                        name="some"+name.substring(10);
124                                                if(name.length()>0)
125                                                        found.setName(name+" "+flower);
126                                                else
127                                                        found.setName(L("some @x1",flower));
128                                                found.setDisplayText(L("@x1 are here",found.Name()));
129                                                found.setDescription("");
130                                                found.text();
131                                                if((!isLimitedToOne()) && (foundName!=null))
132                                                {
133                                                        final Item tempFound=found;
134                                                        if((origFound!=null)
135                                                        &&(!origFound.amDestroyed())
136                                                        &&(origFound!=found))
137                                                                found=origFound;
138                                                        else
139                                                                found=mob.fetchItem(null, Wearable.FILTER_UNWORNONLY, "$"+foundName+"$");
140                                                        if((found != null)
141                                                        &&((found.material()==RawMaterial.RESOURCE_HERBS))
142                                                        &&((found.Name().toUpperCase().endsWith(" HERBS"))
143                                                                ||(found.Name().equalsIgnoreCase("herbs"))
144                                                                ||(found.Name().toUpperCase().endsWith("BUNDLE")))
145                                                        &&(proficiencyCheck(mob,0,lastAuto)))
146                                                        {
147                                                                if(origFound==tempFound)
148                                                                        origFound=found;
149                                                                continue;
150                                                        }
151                                                        found=null;
152                                                }
153                                                else
154                                                        found=null;
155                                        }
156                                }
157                        }
158                }
159                super.unInvoke();
160        }
161
162        protected boolean isLimitedToOne()
163        {
164                return true;
165        }
166        
167        protected int duration(final MOB mob)
168        {
169                return getDuration(15,mob,1,2);
170        }
171        
172        @Override
173        public boolean invoke(MOB mob, List<String> commands, Physical givenTarget, boolean auto, int asLevel)
174        {
175                if(super.checkStop(mob, commands))
176                        return true;
177                if(commands.size()<1)
178                {
179                        commonTell(mob,L("You must specify what flower you want to identify."));
180                        return false;
181                }
182                String finalName=CMParms.combine(commands,0);
183                Item target=mob.fetchItem(null,Wearable.FILTER_UNWORNONLY,finalName);
184                if((target==null)||(!CMLib.flags().canBeSeenBy(target,mob)))
185                {
186                        commonTell(mob,L("You don't seem to have a '@x1'.",(commands.get(0))));
187                        return false;
188                }
189                commands.remove(commands.get(0));
190
191                if((target.material()!=RawMaterial.RESOURCE_FLOWERS)
192                ||((!target.Name().toUpperCase().endsWith(" FLOWERS"))
193                   &&(!target.Name().equalsIgnoreCase("flowers"))
194                   &&(!target.Name().toUpperCase().endsWith("BUNDLE")))
195                ||(!(target instanceof RawMaterial))
196                ||(!target.isGeneric()))
197                {
198                        commonTell(mob,L("You can only identify unknown flowers."));
199                        return false;
200                }
201                if(isLimitedToOne() && target.basePhyStats().weight()>1)
202                        target=(Item)CMLib.materials().unbundle(target, 1, null);
203                if(!super.invoke(mob,commands,givenTarget,auto,asLevel))
204                        return false;
205                verb=L("studying @x1",target.name());
206                displayText=L("You are @x1",verb);
207                found=target;
208                messedUp=false;
209                if(!proficiencyCheck(mob,0,auto))
210                        messedUp=true;
211                final int duration=duration(mob);
212                final CMMsg msg=CMClass.getMsg(mob,null,this,getActivityMessageType(),L("<S-NAME> stud(ys) @x1.",target.name()));
213                if(mob.location().okMessage(mob,msg))
214                {
215                        mob.location().send(mob,msg);
216                        this.foundName=target.Name();
217                        this.lastAuto=auto;
218                        beneficialAffect(mob,mob,asLevel,duration);
219                }
220                return true;
221        }
222}
