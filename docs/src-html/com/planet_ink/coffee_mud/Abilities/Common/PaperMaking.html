126        @Override
127        public void unInvoke()
128        {
129                if(canBeUninvoked())
130                {
131                        if(affected instanceof MOB)
132                        {
133                                final MOB mob=(MOB)affected;
134                                if((buildingI!=null)&&(!aborted))
135                                {
136                                        if(messedUp)
137                                                commonTell(mob,L("<S-NAME> mess(es) up making @x1.",buildingI.name(mob)));
138                                        else
139                                        {
140                                                dropAWinner(mob,buildingI);
141                                                CMLib.achievements().possiblyBumpAchievement(mob, AchievementLibrary.Event.CRAFTING, 1, this);
142                                                if(key!=null)
143                                                {
144                                                        dropAWinner(mob,key);
145                                                        if(buildingI instanceof Container)
146                                                                key.setContainer((Container)buildingI);
147                                                }
148                                        }
149                                }
150                                buildingI=null;
151                                key=null;
152                        }
153                }
154                super.unInvoke();
155        }

169        @Override
170        protected boolean autoGenInvoke(final MOB mob, final List<String> commands, final Physical givenTarget, final boolean auto,
171                                                                 final int asLevel, final int autoGenerate, final boolean forceLevels, final List<Item> crafted)
172        {
173                final List<String> originalCommands = new XVector<String>(commands);
174                if(super.checkStop(mob, commands))
175                        return true;
176                final Session session=mob.session();
177
178                @SuppressWarnings("unused")
179                int recipeLevel=1;
180                randomRecipeFix(mob,addRecipes(mob,loadRecipes()),commands,autoGenerate);
181                if(commands.size()==0)
182                {
183                        commonTell(mob,L("Papermake what? Enter \"papermake list\" for a list, or \"papermake stop\" to cancel."));
184                        return false;
185                }
186                if((!auto)
187                &&(commands.size()>0)
188                &&((commands.get(0)).equalsIgnoreCase("bundle")))
189                {
190                        bundling=true;
191                        if(super.invoke(mob,commands,givenTarget,auto,asLevel))
192                                return super.bundle(mob,commands);
193                        return false;
194                }
195                final List<List<String>> recipes=addRecipes(mob,loadRecipes());
196                final String str=commands.get(0);
197                String startStr=null;
198                int duration=4;
199                if(str.equalsIgnoreCase("list"))
200                {
201                        String mask=CMParms.combine(commands,1);
202                        boolean allFlag=false;
203                        if(mask.equalsIgnoreCase("all"))
204                        {
205                                allFlag=true;
206                                mask="";
207                        }
208                        final int[] cols={
209                                CMLib.lister().fixColWidth(22,mob.session()),
210                                CMLib.lister().fixColWidth(3,mob.session())
211                        };
212                        final StringBuffer buf=new StringBuffer(L("@x1 @x2 Material required\n\r",CMStrings.padRight(L("Item"),cols[0]),CMStrings.padRight(L("Lvl"),cols[1])));
213                        final List<List<String>> listRecipes=((mask.length()==0) || mask.equalsIgnoreCase("all")) ? recipes : super.matchingRecipeNames(recipes, mask, true);
214                        for(int r=0;r<listRecipes.size();r++)
215                        {
216                                final List<String> V=listRecipes.get(r);
217                                if(V.size()>0)
218                                {
219                                        final String item=replacePercent(V.get(RCP_FINALNAME),"");
220                                        final int level=CMath.s_int(V.get(RCP_LEVEL));
221                                        String material=V.get(RCP_WOODTYPE);
222                                        String wood=getComponentDescription(mob,V,RCP_WOOD);
223                                        if(!CMath.isInteger(wood))
224                                        {
225                                                material="";
226                                                wood=wood.toLowerCase();
227                                        }
228                                        else
229                                        if(CMath.s_int(wood)>1)
230                                                material="pounds of "+material;
231                                        else
232                                                material="pound of "+material;
233                                        if((level<=xlevel(mob))||allFlag)
234                                                buf.append(CMStrings.padRight(item,cols[0])+" "+CMStrings.padRight(""+level,cols[1])+" "+wood+" "+material.toLowerCase()+"\n\r");
235                                }
236                        }
237                        commonTell(mob,buf.toString());
238                        return true;
239                }
240                activity = CraftingActivity.CRAFTING;
241                buildingI=null;
242                key=null;
243                messedUp=false;
244                String statue=null;
245                if((commands.size()>1)&&(commands.get(commands.size()-1)).startsWith("STATUE="))
246                {
247                        statue=((commands.get(commands.size()-1)).substring(7)).trim();
248                        if(statue.length()==0)
249                                statue=null;
250                        else
251                                commands.remove(commands.size()-1);
252                }
253                String materialDesc="";
254                final String recipeName=CMParms.combine(commands,0);
255                List<String> foundRecipe=null;
256                final List<List<String>> matches=matchingRecipeNames(recipes,recipeName,true);
257                for(int r=0;r<matches.size();r++)
258                {
259                        final List<String> V=matches.get(r);
260                        if(V.size()>0)
261                        {
262                                final int level=CMath.s_int(V.get(RCP_LEVEL));
263                                if((autoGenerate>0)||(level<=xlevel(mob)))
264                                {
265                                        foundRecipe=V;
266                                        recipeLevel=level;
267                                        materialDesc=foundRecipe.get(RCP_WOODTYPE);
268                                        if(materialDesc.equalsIgnoreCase("WOOD"))
269                                                materialDesc="WOODEN";
270                                        break;
271                                }
272                        }
273                }
274
275                if(materialDesc.length()==0)
276                        materialDesc="WOODEN";
277
278                if(foundRecipe==null)
279                {
280                        commonTell(mob,L("You don't know how to make a '@x1'.  Try \"make list\" for a list.",recipeName));
281                        return false;
282                }
283
284                final String woodRequiredStr = foundRecipe.get(RCP_WOOD);
285                final int[] compData = new int[CF_TOTAL];
286                final String realRecipeName=replacePercent(foundRecipe.get(RCP_FINALNAME),"");
287                final List<Object> componentsFoundList=getAbilityComponents(mob, woodRequiredStr, "make "+CMLib.english().startWithAorAn(realRecipeName),autoGenerate,compData,1);
288                if(componentsFoundList==null)
289                        return false;
290                int woodRequired=CMath.s_int(woodRequiredStr);
291                woodRequired=adjustWoodRequired(woodRequired,mob);
292
293                final int[][] data=fetchFoundResourceData(mob,
294                                                                                                woodRequired,materialDesc,null,
295                                                                                                0,null,null,
296                                                                                                false,
297                                                                                                autoGenerate,
298                                                                                                null);
299                if(data==null)
300                        return false;
301                woodRequired=data[0][FOUND_AMT];
302
303                final String misctype=(foundRecipe.size()>RCP_MISCTYPE)?foundRecipe.get(RCP_MISCTYPE).trim():"";
304                if((misctype.equalsIgnoreCase("statue"))
305                &&(session!=null)
306                &&((statue==null)||(statue.trim().length()==0)))
307                {
308                        final Ability me=this;
309                        final Physical target=givenTarget;
310                        if(autoGenerate>0)
311                                statue=mob.Name();
312                        else
313                        session.prompt(new InputCallback(InputCallback.Type.PROMPT,"",0)
314                        {
315                                @Override
316                                public void showPrompt()
317                                {
318                                        session.promptPrint(L("What is this of?\n\r: "));
319                                }
320
321                                @Override
322                                public void timedOut()
323                                {
324                                }
325
326                                @Override
327                                public void callBack()
328                                {
329                                        final String of=this.input;
330                                        if((of.trim().length()==0)||(of.indexOf('<')>=0))
331                                                return;
332                                        final Vector<String> newCommands=new XVector<String>(originalCommands);
333                                        newCommands.add("STATUE="+of);
334                                        me.invoke(mob, newCommands, target, auto, asLevel);
335                                }
336                        });
337                        return false;
338                }
339
340                if(!super.invoke(mob,commands,givenTarget,auto,asLevel))
341                        return false;
342
343                final MaterialLibrary.DeadResourceRecord deadMats;
344                if((componentsFoundList.size() > 0)||(autoGenerate>0))
345                        deadMats = new MaterialLibrary.DeadResourceRecord();
346                else
347                        deadMats = CMLib.materials().destroyResources(mob.location(),data[0][FOUND_AMT],data[0][FOUND_CODE],0,null,null);
348                final MaterialLibrary.DeadResourceRecord deadComps = CMLib.ableComponents().destroyAbilityComponents(componentsFoundList);
349                buildingI=CMClass.getItem(foundRecipe.get(RCP_CLASSTYPE));
350                if(buildingI==null)
351                {
352                        commonTell(mob,L("There's no such thing as a @x1!!!",foundRecipe.get(RCP_CLASSTYPE)));
353                        return false;
354                }
355                duration=getDuration(CMath.s_int(foundRecipe.get(RCP_TICKS)),mob,CMath.s_int(foundRecipe.get(RCP_LEVEL)),4);
356                String itemName=foundRecipe.get(RCP_FINALNAME).toLowerCase();
357                itemName=CMLib.english().startWithAorAn(itemName);
358                buildingI.setName(itemName);
359                startStr=L("<S-NAME> start(s) making @x1.",buildingI.name());
360                displayText=L("You are making @x1",buildingI.name());
361                verb=L("making @x1",buildingI.name());
362                playSound="crumple.wav";
363                buildingI.setDisplayText(L("@x1 lies here",itemName));
364                buildingI.setDescription(itemName+". ");
365                int weight = getStandardWeight(woodRequired+compData[CF_AMOUNT],bundling) / 10;
366                if(weight < 1)
367                        weight = 1;
368                buildingI.basePhyStats().setWeight(weight);
369                buildingI.setBaseValue(CMath.s_int(foundRecipe.get(RCP_VALUE))+(woodRequired*(RawMaterial.CODES.VALUE(data[0][FOUND_CODE]))));
370                buildingI.setMaterial(super.getBuildingMaterial(woodRequired, data, compData));
371                final String spell=(foundRecipe.size()>RCP_SPELL)?foundRecipe.get(RCP_SPELL).trim():"";
372                addSpells(buildingI,spell,deadMats.lostProps,deadComps.lostProps);
373                setBrand(mob, buildingI);
374                if(((data[0][FOUND_CODE]&RawMaterial.MATERIAL_MASK)==RawMaterial.MATERIAL_WOODEN)
375                ||(data[0][FOUND_CODE]==RawMaterial.RESOURCE_RICE))
376                        buildingI.setMaterial(RawMaterial.RESOURCE_PAPER);
377                if(buildingI instanceof Recipe)
378                        ((Recipe)buildingI).setTotalRecipePages(CMath.s_int(woodRequiredStr));
379                final int capacity;
380                if(buildingI instanceof Book)
381                {
382                        capacity=0;
383                        final String pgs=foundRecipe.get(RCP_CAPACITY);
384                        final int x=pgs.indexOf('/');
385                        if(x<0)
386                                ((Book)buildingI).setMaxPages(CMath.s_int(pgs));
387                        else
388                        {
389                                ((Book)buildingI).setMaxPages(CMath.s_int(pgs.substring(0, x)));
390                                ((Book)buildingI).setMaxCharsPerPage(CMath.s_int(pgs.substring(x+1)));
391                        }
392                }
393                else
394                        capacity=CMath.s_int(foundRecipe.get(RCP_CAPACITY));
395                buildingI.basePhyStats().setLevel(CMath.s_int(foundRecipe.get(RCP_LEVEL)));
396                buildingI.recoverPhyStats();
397                buildingI.text();
398                buildingI.recoverPhyStats();
399
400                if((buildingI instanceof RawMaterial)
401                &&(misctype.length()>0))
402                        ((RawMaterial)buildingI).setSubType(misctype.toUpperCase().trim());
403                else
404                if((misctype.equalsIgnoreCase("statue"))
405                &&(statue!=null)
406                &&(statue.trim().length()>0))
407                {
408                        if(buildingI.Name().indexOf('%')>0)
409                        {
410                                buildingI.setName(CMStrings.replaceAll(buildingI.Name(), "%", statue.trim()));
411                                buildingI.setDisplayText(CMStrings.replaceAll(buildingI.displayText(), "%", statue.trim()));
412                                buildingI.setDescription(CMStrings.replaceAll(buildingI.description(), "%", statue.trim()));
413                        }
414                        else
415                        {
416                                buildingI.setName(L("@x1 of @x2",itemName,statue.trim()));
417                                buildingI.setDisplayText(L("@x1 of @x2 is here",itemName,statue.trim()));
418                                buildingI.setDescription(L("@x1 of @x2. ",itemName,statue.trim()));
419                        }
420                        verb=L("making @x1",buildingI.name());
421                }
422                else
423                if((buildingI instanceof Container)
424                &&(!(buildingI instanceof Armor)))
425                {
426                        if(capacity>0)
427                        {
428                                ((Container)buildingI).setCapacity(capacity+woodRequired);
429                                ((Container)buildingI).setContainTypes(Container.CONTAIN_ANYTHING);
430                        }
431                        if(misctype.equalsIgnoreCase("LID"))
432                                ((Container)buildingI).setDoorsNLocks(true,false,true,false,false,false);
433                        else
434                        if(misctype.equalsIgnoreCase("LOCK"))
435                        {
436                                ((Container)buildingI).setDoorsNLocks(true,false,true,true,false,true);
437                                ((Container)buildingI).setKeyName(Double.toString(Math.random()));
438                                key=(DoorKey)CMClass.getItem("GenKey");
439                                key.setKey(((Container)buildingI).keyName());
440                                key.setName(L("a key"));
441                                key.setDisplayText(L("a small key sits here"));
442                                key.setDescription(L("looks like a key to @x1",buildingI.name()));
443                                key.recoverPhyStats();
444                                key.text();
445                        }
446                }
447
448                messedUp=!proficiencyCheck(mob,0,auto);
449
450                if(autoGenerate>0)
451                {
452                        if(key!=null)
453                                crafted.add(key);
454                        crafted.add(buildingI);
455                        return true;
456                }
457
458                final CMMsg msg=CMClass.getMsg(mob,buildingI,this,getActivityMessageType(),startStr);
459                if(mob.location().okMessage(mob,msg))
460                {
461                        mob.location().send(mob,msg);
462                        buildingI=(Item)msg.target();
463                        beneficialAffect(mob,mob,asLevel,duration);
464                }
465                return true;
466        }
