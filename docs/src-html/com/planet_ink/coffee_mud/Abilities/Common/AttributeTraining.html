036public class AttributeTraining extends CommonSkill
037{
038        @Override
039        public String ID()
040        {
041                return "AttributeTraining";
042        }
043
044        private final static String localizedName = CMLib.lang().L("Attribute Training");
045
046        @Override
047        public String name()
048        {
049                return localizedName;
050        }
051
052        private static final String[] triggerStrings =I(new String[] {"ATTRIBUTETRAINING","ATTRIBTRAIN","ATRAIN"});
053        @Override
054        public String[] triggerStrings()
055        {
056                return triggerStrings;
057        }
058
059        @Override
060        public int classificationCode()
061        {
062                return Ability.ACODE_COMMON_SKILL|Ability.DOMAIN_EDUCATIONLORE;
063        }
064
065        protected final static int costMultiplier=2;
066        protected final static int costAdder=0;
067        
068        protected Physical trained=null;
069        protected int attribute=-1;
070        protected boolean messedUp=false;
071        
072        public AttributeTraining()
073        {
074                super();
075                displayText=L("You are training...");
076                verb=L("training");
077        }
078
079        @Override
080        public boolean tick(final Tickable ticking, final int tickID)
081        {
082                if((affected!=null)
083                &&(affected instanceof MOB)
084                &&(tickID==Tickable.TICKID_MOB))
085                {
086                        final MOB mob=(MOB)affected;
087                        if((trained==null)||(mob.location()==null))
088                        {
089                                messedUp=true;
090                                unInvoke();
091                        }
092                        if((trained instanceof MOB)&&(!mob.location().isInhabitant((MOB)trained)))
093                        {
094                                messedUp=true;
095                                unInvoke();
096                        }
097                        if((trained instanceof Item)&&(!mob.location().isContent((Item)trained)))
098                        {
099                                messedUp=true;
100                                unInvoke();
101                        }
102                        final String attribName=CMStrings.capitalizeAndLower(CharStats.CODES.DESC(attribute));
103
104                        if(CMLib.dice().rollPercentage()<25)
105                                verb=L("@x1 training with @x2",attribName,trained.name());
106                        else
107                        switch(attribute)
108                        {
109                        case CharStats.STAT_CHARISMA:
110                                verb=L("praticing charm, decorum, and grooming with @x2",attribName,trained.name());
111                                break;
112                        case CharStats.STAT_CONSTITUTION:
113                                verb=L("doing cardio-vascular training with @x2",attribName,trained.name());
114                                break;
115                        case CharStats.STAT_DEXTERITY:
116                                verb=L("doing eye-hand coordination exercises with @x2",attribName,trained.name());
117                                break;
118                        case CharStats.STAT_STRENGTH:
119                                verb=L("doing weight repetitions with @x2",attribName,trained.name());
120                                break;
121                        case CharStats.STAT_INTELLIGENCE:
122                                verb=L("studying ancient knowledge with @x2",attribName,trained.name());
123                                break;
124                        case CharStats.STAT_WISDOM:
125                                verb=L("praticing wit and awareness with @x2",attribName,trained.name());
126                                break;
127                        }
128                }
129                return super.tick(ticking,tickID);
130        }
131
132        @Override
133        public void unInvoke()
134        {
135                if(canBeUninvoked())
136                {
137                        if(affected instanceof MOB)
138                        {
139                                final MOB mob=(MOB)affected;
140                                if((trained!=null)&&(!aborted))
141                                {
142                                        MOB follower=null;
143                                        if(trained instanceof MOB)
144                                                follower=(MOB)trained;
145                                        if((messedUp)||(follower==null)||(attribute<0))
146                                                commonTell(mob,L("You've failed to train @x1!",trained.name()));
147                                        else
148                                        {
149                                                final String s=CMStrings.capitalizeAndLower(CharStats.CODES.DESC(attribute));
150                                                mob.location().show(mob,null,getActivityMessageType(),L("<S-NAME> manage(s) to complete @x2 training with @x1.",follower.name(),s));
151                                                final String attribName=CMStrings.capitalizeAndLower(CharStats.CODES.DESC(attribute));
152                                                if(follower.isMonster())
153                                                {
154                                                        Ability adjA=follower.fetchEffect("Prop_StatAdjuster");
155                                                        if(adjA == null)
156                                                        {
157                                                                adjA=CMClass.getAbility("Prop_StatAdjuster");
158                                                                follower.addNonUninvokableEffect(adjA);
159                                                        }
160                                                        final int oldVal=CMParms.getParmInt(adjA.text(), CMStrings.limit(CharStats.CODES.NAME(attribute),3), 0);
161                                                        int trainsRequired=CMLib.login().getTrainingCost(follower, attribute, false)*costMultiplier;
162                                                        if(trainsRequired>=0)
163                                                                trainsRequired+=costAdder;
164                                                        commonTell(mob,L("The training cost @x1 @x2 training points.",follower.name(),""+trainsRequired));
165                                                        if(trainsRequired > follower.getTrains())
166                                                                follower.setTrains(0);
167                                                        else
168                                                                follower.setTrains(follower.getTrains()-trainsRequired);
169                                                        String oldStr=adjA.text();
170                                                        oldStr = CMParms.delParmLong(oldStr, CMStrings.limit(CharStats.CODES.NAME(attribute),3));
171                                                        oldStr = CMStrings.replaceAll(oldStr, "  ", " ");
172                                                        adjA.setMiscText(oldStr.trim()+" "+CMStrings.limit(CharStats.CODES.NAME(attribute),3)+"="+(oldVal+1));
173                                                }
174                                                else
175                                                {
176                                                        CMLib.commands().forceStandardCommand(follower, "Train", new XVector<String>(attribName));
177                                                }
178                                                follower.recoverCharStats();
179                                                follower.recoverPhyStats();
180                                                follower.recoverMaxState();
181                                                if(trained instanceof CagedAnimal)
182                                                {
183                                                        follower.text();
184                                                        ((CagedAnimal)trained).cageMe(follower);
185                                                }
186                                        }
187                                }
188                        }
189                }
190                super.unInvoke();
191        }

193        @Override
194        public boolean invoke(MOB mob, List<String> commands, Physical givenTarget, boolean auto, int asLevel)
195        {
196                if(super.checkStop(mob, commands))
197                        return true;
198
199                verb=L("training");
200                trained=null;
201                String valid="Attributes include:  ";
202                final StringBuffer thingsToTrainFor=new StringBuffer("");
203                for(final int i: CharStats.CODES.BASECODES())
204                        thingsToTrainFor.append(CharStats.CODES.DESC(i)+", ");
205                if(thingsToTrainFor.length()>2)
206                        valid+=thingsToTrainFor.toString().substring(0,thingsToTrainFor.length()-2)+".";
207                if(commands.size()<2)
208                {
209                        commonTell(mob,L("Train whom in what attribute? @x1",valid));
210                        return false;
211                }
212                attribute=-1;
213                String attribName="";
214                final String what=commands.get(commands.size()-1);
215                commands.remove(commands.size()-1);
216                for(final int i: CharStats.CODES.BASECODES())
217                {
218                        if(CharStats.CODES.DESC(i).startsWith(what.toUpperCase()))
219                        {
220                                attribute=i;
221                                attribName=CMStrings.capitalizeAndLower(CharStats.CODES.DESC(i));
222                        }
223                }
224                if(attribute<0)
225                {
226                        commonTell(mob,L("Train whom in what attribute? @x1",valid));
227                        return false;
228                }
229
230                final String str=CMParms.combine(commands,0);
231                MOB M=super.getTarget(mob, commands, givenTarget);
232                trained=M;
233                if(M!=null)
234                {
235                        if(!CMLib.flags().canBeSeenBy(M,mob))
236                        {
237                                commonTell(mob,L("You don't see anyone called '@x1' here.",str));
238                                return false;
239                        }
240                        if(!M.isMonster())
241                        {
242                                commonTell(mob,L("@x1 is perfectly capable of training on their own.",M.name(mob)));
243                                return false;
244                        }
245                        if(CMLib.flags().isAnimalIntelligence(M))
246                        {
247                                commonTell(mob,L("You can't train with @x1.",M.name(mob)));
248                                return false;
249                        }
250                        if((!CMLib.flags().canMove(M))&&(CMLib.flags().isBoundOrHeld(M)))
251                        {
252                                commonTell(mob,L("@x1 doesn't seem able to train.",M.name(mob)));
253                                return false;
254                        }
255                        if(!mob.getGroupMembers(new HashSet<MOB>()).contains(M))
256                        {
257                                commonTell(mob,L("@x1 doesn't seem willing to train with you.",M.name(mob)));
258                                return false;
259                        }
260                        int curStat=M.baseCharStats().getRacialStat(M, attribute);
261                        int trainsRequired=CMLib.login().getTrainingCost(M, attribute, false)*costMultiplier;
262                        if(trainsRequired>=0)
263                                trainsRequired+=costAdder;
264                        if(trainsRequired<0)
265                                return false;
266                        final int teachStat=mob.charStats().getStat(attribute);
267                        if(curStat>=teachStat)
268                        {
269                                commonTell(mob,L("@x1 can only train with someone whose score is higher than their own.",M.name(mob)));
270                                return false;
271                        }
272                        if(M.getTrains()<trainsRequired)
273                        {
274                                if(trainsRequired>1)
275                                {
276                                        commonTell(mob,L("@x1 requires @x1 training points to do that.",M.name(mob),""+trainsRequired));
277                                        return false;
278                                }
279                                else
280                                if(trainsRequired==1)
281                                {
282                                        commonTell(mob,L("@x1 requires @x1 training point to do that.",M.name(mob),""+trainsRequired));
283                                        return false;
284                                }
285                        }
286                        trained=M;
287                }
288                else
289                        return false;
290
291                if(!super.invoke(mob,commands,givenTarget,auto,asLevel))
292                        return false;
293                
294                messedUp=!proficiencyCheck(mob,-trained.phyStats().level()+(2*getXLEVELLevel(mob)),auto);
295                final int duration=getDuration(45,mob,trained.phyStats().level(),10);
296                verb=L("@x1 training with @x2",attribName,M.name());
297                final CMMsg msg=CMClass.getMsg(mob,null,this,getActivityMessageType(),L("<S-NAME> start(s) @x2 training with @x1.",M.name(),attribName.toLowerCase()));
298                if(mob.location().okMessage(mob,msg))
299                {
300                        mob.location().send(mob,msg);
301                        beneficialAffect(mob,mob,asLevel,duration);
302                }
303                return true;
304        }
