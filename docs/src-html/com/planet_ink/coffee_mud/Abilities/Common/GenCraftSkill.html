152        {
153                if(vars.containsKey(ID))

156                vars.put(ID,O);

293        @Override
294        public void setStat(String code, String val)
295        {
296                int num=0;
297                int numDex=code.length();
298                while((numDex>0)&&(Character.isDigit(code.charAt(numDex-1))))
299                        numDex--;
300                if(numDex<code.length())
301                {
302                        num=CMath.s_int(code.substring(numDex));
303                        code=code.substring(0,numDex);
304                }
305                switch(getCodeNum(code))
306                {
307                case 0:
308                        if(val.trim().length()>0)
309                        {
310                                V(ID,V_NAME); // force creation, if necc
311                                final Object[] O=vars.get(ID);
312                                vars.remove(ID);
313                                vars.put(val,O);
314                                if(num!=9)
315                                        CMClass.delClass(CMObjectType.ABILITY,this);
316                                ID=val;
317                                if(num!=9)
318                                        CMClass.addClass(CMObjectType.ABILITY,this);
319                        }
320                        break;
321                case 1:
322                        setMiscText(val);
323                        break;
324                case 2:
325                        SV(ID, V_NAME, val);
326                        if (ID.equalsIgnoreCase("GenCraftSkill"))
327                                break;
328                        break;
329                case 3:
330                        SV(ID, V_HELP, val);
331                        break;
332                case 4:
333                        SV(ID, V_TRIG, CMParms.parseCommas(val, true).toArray(new String[0]));
334                        break;
335                case 5:
336                        SV(ID, V_FNAM, val);
337                        break;
338                case 6:
339                        SV(ID, V_RSCS, val.toUpperCase().replace(',', '|'));
340                        break;
341                case 7:
342                        SV(ID, V_VERB, val);
343                        break;
344                case 8:
345                        SV(ID, V_CNMN, Boolean.valueOf(CMath.s_bool(val)));
346                        break;
347                case 9:
348                        SV(ID, V_CNRF, Boolean.valueOf(CMath.s_bool(val)));
349                        break;
350                case 10:
351                        SV(ID, V_CNBN, Boolean.valueOf(CMath.s_bool(val)));
352                        break;
353                case 11:
354                        SV(ID, V_SOND, val);
355                        break;
356                case 12:
357                        SV(ID, V_CNST, Boolean.valueOf(CMath.s_bool(val)));
358                        break;
359                default:
360                        if(code.equalsIgnoreCase("allxml")&&ID.equalsIgnoreCase("GenCraftSkill"))
361                                parseAllXML(val);
362                        break;
363                }
364        }

378        private void parseAllXML(String xml)
379        {
380                final List<XMLLibrary.XMLTag> V=CMLib.xml().parseAllXML(xml);
381                if((V==null)||(V.size()==0))
382                        return;
383                for(int c=0;c<getStatCodes().length;c++)
384                {
385                        if(getStatCodes()[c].equals("CLASS"))
386                                ID=CMLib.xml().restoreAngleBrackets(CMLib.xml().getValFromPieces(V, getStatCodes()[c]));
387                        else
388                        if(!getStatCodes()[c].equals("TEXT"))
389                                setStat(getStatCodes()[c],CMLib.xml().restoreAngleBrackets(CMLib.xml().getValFromPieces(V, getStatCodes()[c])));
390                }
391        }

393        private String getAllXML()
394        {
395                final StringBuffer str=new StringBuffer("");
396                for(int c=0;c<getStatCodes().length;c++)
397                {
398                        if(!getStatCodes()[c].equals("TEXT"))
399                        {
400                                str.append("<"+getStatCodes()[c]+">"
401                                                +CMLib.xml().parseOutAngleBrackets(getStat(getStatCodes()[c]))
402                                                +"</"+getStatCodes()[c]+">");
403                        }
404                }
405                return str.toString();
406        }

427        @Override
428        public void unInvoke()
429        {
430                if(canBeUninvoked())
431                {
432                        if(affected instanceof MOB)
433                        {
434                                final MOB mob=(MOB)affected;
435                                if((buildingI!=null)&&(!aborted))
436                                {
437                                        if(messedUp)
438                                        {
439                                                if(activity == CraftingActivity.MENDING)
440                                                        messedUpCrafting(mob);
441                                                else
442                                                if(activity == CraftingActivity.REFITTING)
443                                                        commonEmote(mob,L("<S-NAME> mess(es) up refitting @x1.",buildingI.name()));
444                                                else
445                                                        commonEmote(mob,L("<S-NAME> mess(es) up @x1 @x2.",((String)V(ID,V_VERB)),buildingI.name()));
446                                        }
447                                        else
448                                        {
449                                                if(activity == CraftingActivity.MENDING)
450                                                {
451                                                        buildingI.setUsesRemaining(100);
452                                                        CMLib.achievements().possiblyBumpAchievement(mob, AchievementLibrary.Event.MENDER, 1, this);
453                                                }
454                                                else
455                                                if(activity == CraftingActivity.REFITTING)
456                                                {
457                                                        buildingI.basePhyStats().setHeight(0);
458                                                        buildingI.recoverPhyStats();
459                                                }
460                                                else
461                                                {
462                                                        dropAWinner(mob,buildingI);
463                                                        CMLib.achievements().possiblyBumpAchievement(mob, AchievementLibrary.Event.CRAFTING, 1, this);
464                                                        if(key!=null)
465                                                        {
466                                                                dropAWinner(mob,key);
467                                                                if(buildingI instanceof Container)
468                                                                        key.setContainer((Container)buildingI);
469                                                        }
470                                                }
471                                        }
472                                }
473                                buildingI=null;
474                                key=null;
475                                activity = CraftingActivity.CRAFTING;
476                        }
477                }
478                super.unInvoke();
479        }

487        @Override
488        public boolean mayICraft(final MOB mob, final Item I)
489        {
490                if(I==null)
491                        return false;
492                if(!super.isMadeOfSupportedResource(I))
493                        return false;
494                if(!super.mayBeCrafted(I))
495                        return false;
496                final List<List<String>> recipes=addRecipes(mob,loadRecipes());
497                for(final List<String> recipe : recipes)
498                {
499                        String finalName=recipe.get(RCP_FINALNAME);
500                        finalName=CMStrings.replaceAll(finalName, "%", ".*").toLowerCase();
501                        if(Pattern.matches(finalName, I.Name().toLowerCase()))
502                                return true;
503                }
504                return false;
505        }

512        @Override
513        protected boolean canMend(MOB mob, Environmental E, boolean quiet)
514        {
515                final Boolean canMendB=(Boolean)V(ID,V_CNMN);
516                if(!canMendB.booleanValue())
517                        return false;
518                if(!super.canMend(mob,E,quiet))
519                        return false;
520                final Item IE=(Item)E;
521                if(mayICraft(mob, IE))
522                        return true;
523                if(!super.isMadeOfSupportedResource(IE))
524                {
525                        if(!quiet)
526                        {
527                                commonTell(mob,L("That can't be mended with this skill."));
528                        }
529                        return false;
530                }
531                return true;
532        }

546        @Override
547        protected boolean autoGenInvoke(final MOB mob, List<String> commands, Physical givenTarget, final boolean auto, 
548                                                                 final int asLevel, int autoGenerate, boolean forceLevels, List<Item> crafted)
549        {
550                if(super.checkStop(mob, commands))
551                        return true;
552
553                if(super.checkInfo(mob, commands))
554                        return true;
555                
556                final PairVector<EnhancedExpertise,Integer> enhancedTypes=enhancedTypes(mob,commands);
557                int recipeLevel = 1;
558                randomRecipeFix(mob,addRecipes(mob,loadRecipes()),commands,autoGenerate);
559                final String noun=CMStrings.capitalizeAndLower(triggerStrings()[0]);
560                final String verbing=V(ID,V_VERB).toString();
561                final Boolean canMendB=(Boolean)V(ID,V_CNMN);
562                final Boolean canRefitB=(Boolean)V(ID,V_CNRF);
563                final Boolean canBundleB=(Boolean)V(ID,V_CNBN);
564                if(commands.size()==0)
565                {
566                        final StringBuilder features=new StringBuilder(noun+" what? Enter \""+noun.toLowerCase()+" list\" for a list");
567                        features.append(", \""+noun.toLowerCase()+" info\" to details");
568                        if(canMendB.booleanValue())
569                                features.append(", \""+noun.toLowerCase()+" mend <item>\" to mend broken items, \""+noun.toLowerCase()+" scan\" to scan for mendable items");
570                        if(canRefitB.booleanValue())
571                                features.append(", \""+noun.toLowerCase()+" refit <item>\" to resize wearables");
572                        if(canBundleB.booleanValue())
573                                features.append(", \""+noun.toLowerCase()+" bundle\" to make bundles");
574                        features.append(", or \""+noun.toLowerCase()+" stop\" to cancel.");
575                        commonTell(mob,features.toString());
576                        return false;
577                }
578                if((!auto)
579                &&(commands.size()>0)
580                &&((commands.get(0)).equalsIgnoreCase("bundle"))
581                &&(canBundleB.booleanValue()))
582                {
583                        bundling=true;
584                        if(super.invoke(mob,commands,givenTarget,auto,asLevel))
585                                return super.bundle(mob,commands);
586                        return false;
587                }
588                final List<List<String>> recipes=addRecipes(mob,loadRecipes());
589                final String str=commands.get(0);
590                String startStr=null;
591                int duration=4;
592                bundling=false;
593                if(str.equalsIgnoreCase("list"))
594                {
595                        String mask=CMParms.combine(commands,1);
596                        boolean allFlag=false;
597                        if(mask.equalsIgnoreCase("all"))
598                        {
599                                allFlag=true;
600                                mask="";
601                        }
602                        final StringBuffer buf=new StringBuffer(L("Item <S-NAME> <S-IS-ARE> skilled at @x1:\n\r",verbing));
603                        int toggler=1;
604                        final int toggleTop=2;
605                        final int[] cols={
606                                CMLib.lister().fixColWidth(29,mob.session()),
607                                CMLib.lister().fixColWidth(3,mob.session()),
608                                CMLib.lister().fixColWidth(4,mob.session())
609                        };
610                        for(int r=0;r<toggleTop;r++)
611                                buf.append((r>0?" ":"")+CMStrings.padRight(L("Item"),cols[0])+" "+CMStrings.padRight(L("Lvl"),cols[1])+" "+CMStrings.padRight(L("Mats"),cols[2]));
612                        buf.append("\n\r");
613                        final List<List<String>> listRecipes=((mask.length()==0) || mask.equalsIgnoreCase("all")) ? recipes : super.matchingRecipeNames(recipes, mask, true);
614                        for(int r=0;r<listRecipes.size();r++)
615                        {
616                                final List<String> V=listRecipes.get(r);
617                                if(V.size()>0)
618                                {
619                                        final String item=replacePercent(V.get(RCP_FINALNAME),"");
620                                        final int level=CMath.s_int(V.get(RCP_LEVEL));
621                                        final String mats=getComponentDescription(mob,V,RCP_AMOUNTMATS);
622                                        if(mats.length()>5)
623                                        {
624                                                if(toggler>1)
625                                                        buf.append("\n\r");
626                                                toggler=toggleTop;
627                                        }
628                                        if((level<=xlevel(mob))||allFlag)
629                                        {
630                                                buf.append(CMStrings.padRight(item,cols[0])+" "+CMStrings.padRight(""+level,cols[1])+" "+CMStrings.padRightPreserve(""+mats,cols[2])+((toggler!=toggleTop)?" ":"\n\r"));
631                                                if(++toggler>toggleTop)
632                                                        toggler=1;
633                                        }
634                                }
635                        }
636                        if(toggler!=1)
637                                buf.append("\n\r");
638                        commonTell(mob,buf.toString());
639                        enhanceList(mob);
640                        return true;
641                }
642                if(str.equalsIgnoreCase("scan") && canMendB.booleanValue())
643                        return publicScan(mob,commands);
644                else
645                if(str.equalsIgnoreCase("mend") && canMendB.booleanValue())
646                {
647                        buildingI=null;
648                        activity = CraftingActivity.CRAFTING;
649                        key=null;
650                        messedUp=false;
651                        final Vector<String> newCommands=CMParms.parse(CMParms.combine(commands,1));
652                        buildingI=getTargetItemFavorMOB(mob,mob.location(),givenTarget,newCommands,Wearable.FILTER_UNWORNONLY);
653                        if(!canMend(mob, buildingI,false))
654                                return false;
655                        activity = CraftingActivity.MENDING;
656                        if(!super.invoke(mob,commands,givenTarget,auto,asLevel))
657                                return false;
658                        startStr=L("<S-NAME> start(s) mending @x1.",buildingI.name());
659                        displayText=L("You are mending @x1",buildingI.name());
660                        verb=L("mending @x1",buildingI.name());
661                }
662                else
663                if(str.equalsIgnoreCase("refit") && canRefitB.booleanValue())
664                {
665                        buildingI=null;
666                        activity = CraftingActivity.CRAFTING;
667                        messedUp=false;
668                        final Vector<String> newCommands=CMParms.parse(CMParms.combine(commands,1));
669                        buildingI=getTargetItemFavorMOB(mob,mob.location(),givenTarget,newCommands,Wearable.FILTER_UNWORNONLY);
670                        if(buildingI==null)
671                                return false;
672                        if((!this.mayICraft(mob, buildingI))&&(!super.isMadeOfSupportedResource(buildingI)))
673                        {
674                                commonTell(mob,L("That's can't be refitted with this skill."));
675                                return false;
676                        }
677                        if(!(buildingI instanceof Armor))
678                        {
679                                commonTell(mob,L("You don't know how to refit that sort of thing."));
680                                return false;
681                        }
682                        if(buildingI.phyStats().height()==0)
683                        {
684                                commonTell(mob,L("@x1 is already the right size.",buildingI.name(mob)));
685                                return false;
686                        }
687                        activity = CraftingActivity.REFITTING;
688                        if(!super.invoke(mob,commands,givenTarget,auto,asLevel))
689                                return false;
690                        startStr=L("<S-NAME> start(s) refitting @x1.",buildingI.name());
691                        displayText=L("You are refitting @x1",buildingI.name());
692                        verb=L("refitting @x1",buildingI.name());
693                }
694                else
695                {
696                        buildingI=null;
697                        activity = CraftingActivity.CRAFTING;
698                        aborted=false;
699                        key=null;
700                        messedUp=false;
701                        int amount=-1;
702                        if((commands.size()>1)&&(CMath.isNumber(commands.get(commands.size()-1))))
703                        {
704                                amount=CMath.s_int(commands.get(commands.size()-1));
705                                commands.remove(commands.size()-1);
706                        }
707                        final String recipeName=CMParms.combine(commands,0);
708                        List<String> foundRecipe=null;
709                        final List<List<String>> matches=matchingRecipeNames(recipes,recipeName,true);
710                        for(int r=0;r<matches.size();r++)
711                        {
712                                final List<String> V=matches.get(r);
713                                if(V.size()>0)
714                                {
715                                        final int level=CMath.s_int(V.get(RCP_LEVEL));
716                                        if((autoGenerate>0)||(level<=xlevel(mob)))
717                                        {
718                                                foundRecipe=V;
719                                                recipeLevel=level;
720                                                break;
721                                        }
722                                }
723                        }
724                        if(foundRecipe==null)
725                        {
726                                commonTell(mob,L("You don't know how to @x1 a '@x2'.  Try \"@x3 list\" for a list.",noun.toLowerCase(),recipeName,noun.toLowerCase()));
727                                return false;
728                        }
729
730                        final String requiredMats = foundRecipe.get(RCP_AMOUNTMATS);
731                        final int[] compData = new int[CF_TOTAL];
732                        final String realRecipeName=replacePercent(foundRecipe.get(RCP_FINALNAME),"");
733                        final List<Object> componentsFoundList=getAbilityComponents(mob, requiredMats, "make "+CMLib.english().startWithAorAn(realRecipeName),autoGenerate,compData,1);
734                        if(componentsFoundList==null)
735                                return false;
736                        int numRequired=CMath.isInteger(requiredMats)?CMath.s_int(requiredMats):0;
737                        numRequired=adjustWoodRequired(numRequired,mob);
738
739                        if(amount>numRequired)
740                                numRequired=amount;
741                        final String misctype=foundRecipe.get(RCP_MISCTYPE);
742                        final Integer[] ipm=super.supportedResourcesMap();
743                        final int[] pm=new int[ipm.length];
744                        for(int i=0;i<ipm.length;i++) pm[i]=ipm[i].intValue();
745                        bundling=misctype.equalsIgnoreCase("BUNDLE");
746                        final int[][] data=fetchFoundResourceData(mob,
747                                                                                                        numRequired,"material",pm,
748                                                                                                        0,null,null,
749                                                                                                        bundling,
750                                                                                                        autoGenerate,
751                                                                                                        enhancedTypes);
752                        if(data==null)
753                                return false;
754                        fixDataForComponents(data,requiredMats,autoGenerate>0,componentsFoundList, 1);
755                        numRequired=data[0][FOUND_AMT];
756                        if(!super.invoke(mob,commands,givenTarget,auto,asLevel))
757                                return false;
758                        final MaterialLibrary.DeadResourceRecord deadMats;
759                        if((componentsFoundList.size() > 0)||(autoGenerate>0))
760                                deadMats = new MaterialLibrary.DeadResourceRecord();
761                        else
762                                deadMats = CMLib.materials().destroyResources(mob.location(),numRequired,data[0][FOUND_CODE],0,null,null);
763                        final MaterialLibrary.DeadResourceRecord deadComps = CMLib.ableComponents().destroyAbilityComponents(componentsFoundList);
764                        final int lostValue=autoGenerate>0?0:(deadMats.lostValue + deadComps.lostValue);
765                        buildingI=CMClass.getItem(foundRecipe.get(RCP_CLASSTYPE));
766                        if(buildingI==null)
767                        {
768                                commonTell(mob,L("There's no such thing as a @x1!!!",foundRecipe.get(RCP_CLASSTYPE)));
769                                return false;
770                        }
771                        duration=getDuration(CMath.s_int(foundRecipe.get(RCP_TICKS)),mob,CMath.s_int(foundRecipe.get(RCP_LEVEL)),4);
772                        buildingI.setMaterial(getBuildingMaterial(numRequired,data,compData));
773                        String itemName=determineFinalName(foundRecipe.get(RCP_FINALNAME),buildingI.material(),deadMats,deadComps);
774                        if(bundling)
775                                itemName="a "+numRequired+"# "+itemName;
776                        else
777                                itemName=CMLib.english().startWithAorAn(itemName);
778                        buildingI.setName(itemName);
779                        startStr=L("<S-NAME> start(s) @x1 @x2.",verbing,buildingI.name());
780                        displayText=L("You are @x1 @x2",verbing,buildingI.name());
781                        playSound=(String)V(ID,V_SOND);
782                        verb=verbing+" "+buildingI.name();
783                        buildingI.setDisplayText(L("@x1 lies here",itemName));
784                        buildingI.setDescription(itemName+". ");
785                        buildingI.basePhyStats().setWeight(getStandardWeight(numRequired, bundling));
786                        buildingI.setBaseValue(CMath.s_int(foundRecipe.get(RCP_VALUE)));
787                        final int hardness=RawMaterial.CODES.HARDNESS(buildingI.material())-3;
788                        buildingI.basePhyStats().setLevel(CMath.s_int(foundRecipe.get(RCP_LEVEL))+(hardness));
789                        if(buildingI.basePhyStats().level()<1)
790                                buildingI.basePhyStats().setLevel(1);
791                        setBrand(mob, buildingI);
792                        final int capacity=CMath.s_int(foundRecipe.get(RCP_CAPACITY));
793                        final long canContain=getContainerType(foundRecipe.get(RCP_CONTAINMASK));
794                        final int armordmg=CMath.s_int(foundRecipe.get(RCP_ARMORDMG));
795                        if(bundling)
796                                buildingI.setBaseValue(lostValue);
797                        final String spell=(foundRecipe.size()>RCP_SPELL)?foundRecipe.get(RCP_SPELL).trim():"";
798                        addSpells(buildingI,spell,deadMats.lostProps,deadComps.lostProps);
799                        key=null;
800                        if((buildingI instanceof Container)
801                        &&(!(buildingI instanceof Armor)))
802                        {
803                                if(capacity>0)
804                                {
805                                        ((Container)buildingI).setCapacity(capacity+numRequired);
806                                        ((Container)buildingI).setContainTypes(canContain);
807                                }
808                                if(misctype.equalsIgnoreCase("LID"))
809                                        ((Container)buildingI).setDoorsNLocks(true,false,true,false,false,false);
810                                else
811                                if(misctype.equalsIgnoreCase("LOCK"))
812                                {
813                                        ((Container)buildingI).setDoorsNLocks(true,false,true,true,false,true);
814                                        ((Container)buildingI).setKeyName(Double.toString(Math.random()));
815                                        key=(DoorKey)CMClass.getItem("GenKey");
816                                        key.setKey(((Container)buildingI).keyName());
817                                        key.setName(L("a key"));
818                                        key.setDisplayText(L("a small key sits here"));
819                                        key.setDescription(L("looks like a key to @x1",buildingI.name()));
820                                        key.recoverPhyStats();
821                                        key.text();
822                                }
823                        }
824                        if(buildingI instanceof Drink)
825                        {
826                                if(CMLib.flags().isGettable(buildingI))
827                                {
828                                        ((Drink)buildingI).setLiquidHeld(capacity*50);
829                                        ((Drink)buildingI).setThirstQuenched(250);
830                                        if((capacity*50)<250)
831                                                ((Drink)buildingI).setThirstQuenched(capacity*50);
832                                        ((Drink)buildingI).setLiquidRemaining(0);
833                                }
834                        }
835                        if(buildingI instanceof Rideable)
836                        {
837                                setRideBasis((Rideable)buildingI,misctype);
838                        }
839                        if(buildingI instanceof Wand)
840                        {
841                                if(foundRecipe.get(RCP_CAPACITY).trim().length()>0)
842                                        ((Wand)buildingI).setMaxUses(capacity);
843                        }
844                        else
845                        if(buildingI instanceof Weapon)
846                        {
847                                ((Weapon)buildingI).setRawLogicalAnd((capacity>1));
848                        }
849                        if(buildingI instanceof Weapon)
850                        {
851                                ((Weapon)buildingI).setWeaponClassification(Weapon.CLASS_BLUNT);
852                                setWeaponTypeClass((Weapon)buildingI,misctype,Weapon.TYPE_SLASHING);
853                                buildingI.basePhyStats().setAttackAdjustment((baseYield()+abilityCode()+(hardness*5)-1));
854                                buildingI.basePhyStats().setDamage(armordmg+hardness);
855                                ((Weapon)buildingI).setRawProperLocationBitmap(Wearable.WORN_WIELD|Wearable.WORN_HELD);
856                                if(!(buildingI instanceof Container))
857                                        buildingI.basePhyStats().setAttackAdjustment(buildingI.basePhyStats().attackAdjustment()+(int)canContain);
858                        }
859                        if((buildingI instanceof Armor)&&(!(buildingI instanceof FalseLimb)))
860                        {
861                                ((Armor)buildingI).basePhyStats().setArmor(0);
862                                if(armordmg!=0)
863                                        ((Armor)buildingI).basePhyStats().setArmor(armordmg+(baseYield()+abilityCode()-1));
864                                setWearLocation(buildingI,misctype,hardness);
865                        }
866                        if(buildingI instanceof Light)
867                        {
868                                ((Light)buildingI).setDuration(capacity);
869                                if((buildingI instanceof Container)
870                                &&(!misctype.equals("SMOKE")))
871                                {
872                                        ((Light)buildingI).setDuration(200);
873                                        ((Container)buildingI).setCapacity(0);
874                                }
875                        }
876                        buildingI.recoverPhyStats();
877                        buildingI.text();
878                        buildingI.recoverPhyStats();
879                }
880
881                messedUp=!proficiencyCheck(mob,0,auto);
882
883                if(bundling)
884                {
885                        messedUp=false;
886                        duration=1;
887                        verb=L("bundling @x1",RawMaterial.CODES.NAME(buildingI.material()).toLowerCase());
888                        startStr=L("<S-NAME> start(s) @x1.",verb);
889                        displayText=L("You are @x1",verb);
890                }
891
892                if(autoGenerate>0)
893                {
894                        if(key!=null)
895                                crafted.add(key);
896                        crafted.add(buildingI);
897                        return true;
898                }
899
900                final CMMsg msg=CMClass.getMsg(mob,buildingI,this,getActivityMessageType(),startStr);
901                if(mob.location().okMessage(mob,msg))
902                {
903                        mob.location().send(mob,msg);
904                        buildingI=(Item)msg.target();
905                        beneficialAffect(mob,mob,asLevel,duration);
906                        enhanceItem(mob,buildingI,recipeLevel,enhancedTypes);
907                }
908                else
909                if(bundling)
910                {
911                        messedUp=false;
912                        aborted=false;
913                        unInvoke();
914                }
915                return true;
916        }
