082        protected String fixColor(String name, char colorChar, String colorWord)
083        {
084                final int end=name.indexOf("^?");
085                if((end>0)&&(end<=name.length()-3))
086                {
087                        final int start=name.substring(0,end).indexOf('^');
088                        if((start>=0)&&(start<(end-3)))
089                        {
090                                name=name.substring(0,start)
091                                         +name.substring(end+3);
092                        }
093                }
094                final Vector<String> V=CMParms.parse(name);
095                for(int v=0;v<V.size();v++)
096                {
097                        final String word=V.elementAt(v);
098                        if((word.equalsIgnoreCase("an")) || (word.equalsIgnoreCase("a")))
099                        {
100                                final String properPrefix=CMLib.english().properIndefiniteArticle(colorWord);
101                                V.insertElementAt(colorWord,v+1);
102                                if(word.toLowerCase().equals(word))
103                                        V.set(v,properPrefix.toLowerCase());
104                                else
105                                        V.set(v,CMStrings.capitalizeAndLower(properPrefix));
106                                return CMParms.combine(V,0);
107                        }
108                        else
109                        if((word.equalsIgnoreCase("of"))
110                        ||(word.equalsIgnoreCase("some"))
111                        ||(word.equalsIgnoreCase("the")))
112                        {
113                                V.insertElementAt(colorWord,v+1);
114                                return CMParms.combine(V,0);
115                        }
116                }
117                V.insertElementAt(colorWord,0);
118                return CMParms.combine(V,0);
119        }

121        @Override
122        public void unInvoke()
123        {
124                if(canBeUninvoked())
125                {
126                        if((affected!=null)
127                        &&(affected instanceof MOB)
128                        &&(!aborted)
129                        &&(!helping))
130                        {
131                                final MOB mob=(MOB)affected;
132                                if(writing.length()==0)
133                                        commonEmote(mob,L("<S-NAME> mess(es) up the lacquering."));
134                                else
135                                {
136                                        char colorCode='^';
137                                        for(int i=0;i<writing.length();i++)
138                                        {
139                                                if((writing.charAt(i)=='^')
140                                                &&(i<writing.length()-1)
141                                                &&(writing.charAt(i+1)!='?')
142                                                &&(writing.charAt(i+1)!=ColorLibrary.COLORCODE_BACKGROUND)
143                                                &&(writing.charAt(i+1)!=ColorLibrary.COLORCODE_FANSI256)
144                                                &&(writing.charAt(i+1)!=ColorLibrary.COLORCODE_BANSI256))
145                                                {
146                                                        colorCode=writing.charAt(i+1);
147                                                        break;
148                                                }
149                                        }
150                                        final StringBuffer desc=new StringBuffer(found.description());
151                                        for(int x=0;x<(desc.length()-1);x++)
152                                        {
153                                                if((desc.charAt(x)=='^')
154                                                &&(desc.charAt(x+1)!='?')
155                                                &&(desc.charAt(x+1)!=ColorLibrary.COLORCODE_BACKGROUND)
156                                                &&(desc.charAt(x+1)!=ColorLibrary.COLORCODE_FANSI256)
157                                                &&(desc.charAt(x+1)!=ColorLibrary.COLORCODE_BANSI256))
158                                                        desc.setCharAt(x+1, colorCode);
159                                        }
160                                        final String d=desc.toString();
161                                        if(!d.endsWith("^?"))
162                                                desc.append("^?");
163                                        if(!d.startsWith("^"+colorCode))
164                                                desc.insert(0,"^"+colorCode);
165                                        found.setDescription(desc.toString());
166                                        found.setName(fixColor(found.Name(), colorCode, writing));
167                                        found.setDisplayText(fixColor(found.displayText(), colorCode, writing));
168                                        found.text();
169                                }
170                        }
171                }
172                super.unInvoke();
173        }

175        @Override
176        public boolean invoke(MOB mob, List<String> commands, Physical givenTarget, boolean auto, int asLevel)
177        {
178                if(super.checkStop(mob, commands))
179                        return true;
180                final List<List<String>> recipes = addRecipes(mob,super.loadRecipes("lacquering.txt"));
181                writing=CMParms.combine(commands,0).toLowerCase();
182                List<String> finalRecipe = null; 
183                if(writing.equalsIgnoreCase("list"))
184                {
185                        final StringBuilder colors=new StringBuilder(L("^NColors you can choose: "));
186                        for(final List<String> list : recipes)
187                {
188                                final String name=list.get(0);
189                                final int level=CMath.s_int(list.get(1));
190                                if(level <= adjustedLevel(mob,asLevel))
191                                        colors.append(name).append(", ");
192                        }
193                        commonTell(mob,colors.substring(0,colors.length()-2)+".\n\r");
194                        return false;
195                }
196                if(commands.size()<2)
197                {
198                        commonTell(mob,L("You must specify what you want to lacqer, and color to it to be, or LIST."));
199                        return false;
200                }
201                Item target=mob.fetchItem(null,Wearable.FILTER_UNWORNONLY,commands.get(0));
202                if((target==null)||(!CMLib.flags().canBeSeenBy(target,mob)))
203                        target=mob.location().findItem(null, commands.get(0));
204                if((target!=null)&&(CMLib.flags().canBeSeenBy(target,mob)))
205                {
206                        final Set<MOB> followers=mob.getGroupMembers(new TreeSet<MOB>());
207                        boolean ok=false;
208                        for(final MOB M : followers)
209                        {
210                                if(target.secretIdentity().indexOf(getBrand(M))>=0)
211                                        ok=true;
212                        }
213                        if(!ok)
214                        {
215                                commonTell(mob,L("You aren't allowed to work on '@x1'.",(commands.get(0))));
216                                return false;
217                        }
218                }
219                if((target==null)||(!CMLib.flags().canBeSeenBy(target,mob)))
220                {
221                        commonTell(mob,L("You don't seem to have a '@x1'.",(commands.get(0))));
222                        return false;
223                }
224                commands.remove(commands.get(0));
225
226                if((((target.material()&RawMaterial.MATERIAL_MASK)!=RawMaterial.MATERIAL_GLASS)
227                        &&((target.material()&RawMaterial.MATERIAL_MASK)!=RawMaterial.MATERIAL_METAL)
228                        &&((target.material()&RawMaterial.MATERIAL_MASK)!=RawMaterial.MATERIAL_MITHRIL)
229                        &&((target.material()&RawMaterial.MATERIAL_MASK)!=RawMaterial.MATERIAL_SYNTHETIC)
230                        &&((target.material()&RawMaterial.MATERIAL_MASK)!=RawMaterial.MATERIAL_PRECIOUS)
231                        &&((target.material()&RawMaterial.MATERIAL_MASK)!=RawMaterial.MATERIAL_ROCK)
232                        &&((target.material()&RawMaterial.MATERIAL_MASK)!=RawMaterial.MATERIAL_WOODEN))
233                ||(!target.isGeneric()))
234                {
235                        commonTell(mob,L("You can't lacquer that material."));
236                        return false;
237                }
238
239                writing=CMParms.combine(commands,0).toLowerCase();
240                
241                for(final List<String> list : recipes)
242                {
243                        final String name=list.get(0);
244                        final int level=CMath.s_int(list.get(1));
245                        if(name.equalsIgnoreCase(writing)
246                        &&(level<=adjustedLevel(mob,asLevel)))
247                        {
248                                finalRecipe=list;
249                                break;
250                        }
251                }
252                if(finalRecipe == null)
253                {
254                        commonTell(mob,L("You can't lacquer anything '@x1'. Try LACQUER LIST for a list.",writing));
255                        return false;
256                }
257                
258                if(!super.invoke(mob,commands,givenTarget,auto,asLevel))
259                        return false;
260                verb=L("lacquering @x1 @x2",target.name(),writing);
261                displayText=L("You are @x1",verb);
262                found=target;
263                if(!proficiencyCheck(mob,0,auto))
264                        writing="";
265                final int duration=getDuration(60,mob,1,12);
266                final CMMsg msg=CMClass.getMsg(mob,target,this,getActivityMessageType(),L("<S-NAME> start(s) lacquering <T-NAME> @x1.",writing));
267                if(mob.location().okMessage(mob,msg))
268                {
269                        writing = finalRecipe.get(RCP_COLOR);
270                        mob.location().send(mob,msg);
271                        beneficialAffect(mob,mob,asLevel,duration);
272                }
273                return true;
274        }
