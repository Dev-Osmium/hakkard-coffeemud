084        @Override
085        public void unInvoke()
086        {
087                if(canBeUninvoked())
088                {
089                        if((affected!=null)
090                        &&(affected instanceof MOB)
091                        &&(!aborted)
092                        &&(!helping))
093                        {
094                                final MOB mob=(MOB)affected;
095                                if((foundI==null)||(targetI==null))
096                                        commonTell(mob,L("You mess up your transcribing."));
097                                else
098                                {
099                                        MOB factM=CMClass.getFactoryMOB(mob.Name(), mob.phyStats().level(), mob.location());
100                                        try
101                                        {
102                                                String tmsg="";
103                                                final CMMsg rmsg=CMClass.getMsg(mob,foundI,this,CMMsg.TYP_READ,null,pageNum,null);
104                                                foundI.executeMsg(foundI, rmsg);
105                                                for(CMMsg m2 : rmsg.trailerMsgs())
106                                                {
107                                                        if((m2.source()==mob)
108                                                        &&(m2.target()==foundI)
109                                                        &&(m2.targetMessage().length()>0)
110                                                        &&(m2.sourceMinor()==CMMsg.TYP_WASREAD))
111                                                                tmsg+=m2.targetMessage();
112                                                }
113                                                if((foundI instanceof Recipe)
114                                                &&(targetI instanceof Recipe)
115                                                &&(CMClass.getAbilityPrototype(((Recipe)targetI).getCommonSkillID())==null))
116                                                        ((Recipe)targetI).setCommonSkillID(((Recipe)foundI).getCommonSkillID());
117                                                        
118                                                final CMMsg msg=CMClass.getMsg(mob,targetI,this,CMMsg.TYP_WRITE,
119                                                                L("<S-NAME> transcribe(s) from @x1 into <T-NAME>.",foundI.name(mob)),
120                                                                tmsg,
121                                                                L("<S-NAME> transcribe(s) from @x1 into <T-NAME>.",foundI.name(mob)));
122                                                if(mob.location().okMessage(mob,msg))
123                                                        mob.location().send(mob,msg);
124                                        }
125                                        finally
126                                        {
127                                                factM.destroy();
128                                        }
129                                }
130                        }
131                }
132                super.unInvoke();
133        }

141        protected Item getBrandedItem(final MOB mob, final String itemName, boolean from)
142        {
143                Item I=mob.fetchItem(null,Wearable.FILTER_UNWORNONLY,itemName);
144                if((I==null)||(!CMLib.flags().canBeSeenBy(I,mob)))
145                        I=mob.location().findItem(null, itemName);
146                if((I!=null)&&(CMLib.flags().canBeSeenBy(I,mob)))
147                {
148                        /*
149                        final Set<MOB> followers=mob.getGroupMembers(new TreeSet<MOB>());
150                        boolean ok=false;
151                        for(final MOB M : followers)
152                        {
153                                if(I.secretIdentity().indexOf(getBrand(M))>=0)
154                                        ok=true;
155                        }
156                        if(!ok)
157                        {
158                                if(from)
159                                        commonTell(mob,L("You aren't allowed to copy from '@x1'.",I.name(mob)));
160                                else
161                                        commonTell(mob,L("You aren't allowed to copy to '@x1'.",I.name(mob)));
162                                return null;
163                        }
164                        */
165                }
166                if((I==null)||(!CMLib.flags().canBeSeenBy(I,mob)))
167                {
168                        commonTell(mob,L("You don't seem to have a '@x1'.",itemName));
169                        return null;
170                }
171                if((I.material()&RawMaterial.MATERIAL_MASK)!=RawMaterial.MATERIAL_PAPER)
172                {
173                        if(from)
174                                commonTell(mob,L("You can't transcribe something like @x1.",I.name(mob)));
175                        else
176                                commonTell(mob,L("You can't transcribe onto something like @x1.",I.name(mob)));
177                        return null;
178                }
179                if((!CMLib.flags().isReadable(I))
180                ||(I instanceof Scroll))
181                {
182                        commonTell(mob,L("@x1 isn't even readable!",CMStrings.capitalizeAndLower(I.name(mob))));
183                        return null;
184                }
185                
186                if(!I.isGeneric())
187                {
188                        commonTell(mob,L("You aren't able to transcribe @x1.",I.name(mob)));
189                        return null;
190                }
191                return I;
192        }

194        @Override
195        public boolean invoke(MOB mob, List<String> commands, Physical givenTarget, boolean auto, int asLevel)
196        {
197                if(super.checkStop(mob, commands))
198                        return true;
199                if(commands.size()<1)
200                        return error(mob);
201                foundI = null;
202                targetI = null;
203                pageNum="";
204                if((commands.size()>2)&&(CMath.isInteger(commands.get(commands.size()-1))))
205                        pageNum=commands.remove(commands.size()-1);
206                String copyFromName = commands.get(0);
207                String copyToName = CMParms.combine(commands,1);
208                Item copyFromI=this.getBrandedItem(mob, copyFromName, true);
209                if(copyFromI == null)
210                        return false;
211                Item copyToI=this.getBrandedItem(mob, copyToName, false);
212                if(copyToI == null)
213                        return false;
214                if((copyToI instanceof Recipe)
215                &&(((Recipe)copyToI).getTotalRecipePages() <= ((Recipe)copyToI).getRecipeCodeLines().length))
216                {
217                        commonTell(mob,L("@x1 is full.",copyToI.name(mob)));
218                        return false;
219                }
220                if((copyToI instanceof Recipe) != (copyFromI instanceof Recipe))
221                {
222                        commonTell(mob,L("@x1 can not be copied to @x2.",copyFromI.name(mob),copyToI.name(mob)));
223                        return false;
224                }
225                if((copyFromI instanceof Recipe)
226                &&(copyToI instanceof Recipe)
227                &&(CMClass.getAbilityPrototype(((Recipe)copyFromI).getCommonSkillID())!=null)
228                &&(CMClass.getAbilityPrototype(((Recipe)copyToI).getCommonSkillID())!=null)
229                &&(CMClass.getAbilityPrototype(((Recipe)copyToI).getCommonSkillID())!=CMClass.getAbilityPrototype(((Recipe)copyFromI).getCommonSkillID())))
230                {
231                        commonTell(mob,L("@x1 can not be copied to @x2, as it would break up the recipe types.",copyFromI.name(mob),copyToI.name(mob)));
232                        return false;
233                }
234                final Ability write=mob.fetchAbility("Skill_Write");
235                if(write==null)
236                {
237                        commonTell(mob,L("You must know how to write."));
238                        return false;
239                }
240                
241                if(!super.invoke(mob,commands,givenTarget,auto,asLevel))
242                        return false;
243                foundI=copyFromI;
244                targetI=copyToI;
245                verb=L("transcribing @x1 into @x2",foundI.name(),targetI.name());
246                displayText=L("You are @x1",verb);
247                if((!proficiencyCheck(mob,0,auto))||(!write.proficiencyCheck(mob,0,auto)))
248                        foundI = null;
249                final int duration=getDuration(30,mob,1,1);
250                final CMMsg msg=CMClass.getMsg(mob,copyFromI,this,getActivityMessageType(),
251                                L("<S-NAME> start(s) transcribing <T-NAME> into @x1.",targetI.name()),
252                                pageNum,
253                                L("<S-NAME> start(s) transcribing <T-NAME> into @x1.",targetI.name()));
254                if(mob.location().okMessage(mob,msg))
255                {
256                        mob.location().send(mob,msg);
257                        if(duration==1)
258                        {
259                                Transcribing B=(Transcribing)beneficialAffect(mob,mob,asLevel,duration);
260                                if(B!=null)
261                                {
262                                        B.tickDown=0;
263                                        B.tickUp=0;
264                                        B.unInvoke();
265                                }
266                        }
267                        else
268                                beneficialAffect(mob,mob,asLevel,duration);
269                }
270                return true;
271        }
