112
113        protected Map<String,int[]> proficiencies = new Hashtable<String,int[]>();
114
115        @Override
116        public void setMiscText(final String text)
117        {
118                proficiencies.clear();
119                final Map<String,String> map= CMParms.parseStrictEQParms(text);
120                for(final String key : map.keySet())
121                        proficiencies.put(key, new int[] {CMath.s_int(map.get(key))});
122        }
123
124        @Override
125        public String text()
126        {
127                proficiency();
128                final Map<String,String> strMap=new HashMap<String,String>();
129                for(final String key : proficiencies.keySet())
130                        strMap.put(key, ""+proficiencies.get(key)[0]);
131                return CMParms.toEqListString(strMap);
132        }
133
134        @Override
135        public int proficiency()
136        {
137                final MusicalInstrument instrument = this.instrument;
138                if(instrument != null)
139                {
140                        final String typeName = instrument.getInstrumentTypeName();
141                        final int[] profSave;
142                        if(proficiencies.size()==0)
143                        {
144                                profSave = new int[] {super.proficiency};
145                                proficiencies.put(typeName, profSave);
146                        }
147                        else
148                        if(!proficiencies.containsKey(typeName))
149                        {
150                                profSave = new int[] {0};
151                                proficiencies.put(typeName, profSave);
152                        }
153                        else
154                                profSave = proficiencies.get(typeName);
155                        if(lastInstrType.equals(typeName))
156                                profSave[0] = super.proficiency();
157                        else
158                        {
159                                super.proficiency = profSave[0];
160                                lastInstrType = typeName;
161                        }
162                }
163                return super.proficiency();
164        }
165
166        @Override
167        public void affectPhyStats(final Physical affected, final PhyStats affectableStats)
168        {
169                if((affected instanceof MOB)&&(instrument != null))
170                        affectableStats.addAmbiance("playing "+instrument.name());
171                super.affectPhyStats(affected, affectableStats);
172        }
173
174        public static boolean usingInstrument(final MusicalInstrument I, final MOB mob)
175        {
176                if((I==null)||(mob==null))
177                        return false;
178                if(I instanceof Rideable)
179                {
180                        return (((Rideable)I).amRiding(mob)
181                                        &&(mob.fetchFirstWornItem(Wearable.WORN_WIELD)==null)
182                                        &&(mob.fetchHeldItem()==null));
183                }
184                return mob.isMine(I)&&(!I.amWearingAt(Wearable.IN_INVENTORY));
185        }
186
187        @Override
188        public boolean tick(final Tickable ticking, final int tickID)
189        {
190                final Physical affected=this.affected;
191                if(!(affected instanceof MOB))
192                        return false;
193                if(tickID==Tickable.TICKID_MOB)
194                {
195                        final MOB mob=(MOB)affected;
196                        if(!usingInstrument(this.instrument,mob))
197                        {
198                                aborted=true;
199                                unInvoke();
200                                return false;
201                        }
202                        this.activityRoom = mob.location();
203                }
204                return super.tick(ticking, tickID);
205        }
206
207        @Override
208        public boolean invoke(final MOB mob, final List<String> commands, final Physical givenTarget, final boolean auto, final int asLevel)
209        {
210                if(super.checkStop(mob, commands))
211                        return true;
212
213                if((commands.size()>0)&&(commands.get(0).equalsIgnoreCase("LIST")))
214                {
215                        final StringBuilder str = new StringBuilder(L("You have some musical proficiency with: "));
216                        if(proficiencies.size()==0)
217                                str.append(L("Nothing!"));
218                        else
219                        {
220                                for(final String s : proficiencies.keySet())
221                                {
222                                        str.append(CMStrings.capitalizeAndLower(s)+" ("+proficiencies.get(s)[0]+"%), ");
223                                }
224                                str.delete(str.length()-2,str.length());
225                        }
226                        commonTell(mob,str.toString());
227                        return false;
228                }
229
230                if(mob.fetchEffect(ID())!=null)
231                {
232                        commonTell(mob,L("You are already playing an instrument.  Use PLAYINSTRUMENT STOP to stop."));
233                        return false;
234                }
235
236                MusicalInstrument target=null;
237                if((mob.riding()!=null)&&(mob.riding() instanceof MusicalInstrument))
238                {
239                        if(!usingInstrument((MusicalInstrument)mob.riding(),mob))
240                        {
241                                commonTell(mob,L("You need to free your hands to play @x1.",mob.riding().name()));
242                                return false;
243                        }
244                        target=(MusicalInstrument)mob.riding();
245                }
246                if(target==null)
247                {
248                        for(int i=0;i<mob.numItems();i++)
249                        {
250                                final Item I=mob.getItem(i);
251                                if((I!=null)
252                                &&(I instanceof MusicalInstrument)
253                                &&(I.container()==null)
254                                &&(usingInstrument((MusicalInstrument)I,mob)))
255                                {
256                                        target = (MusicalInstrument) I;
257                                        break;
258                                }
259                        }
260                }
261                if(target==null)
262                {
263                        commonTell(mob,L("You need an instrument to play one!"));
264                        return false;
265                }
266
267                if(!super.invoke(mob,commands,givenTarget,auto,asLevel))
268                        return false;
269                this.instrument=target; // necc for the proficiency checks
270                verb=L("playing @x1",target.name());
271                displayText=L("You are @x1",verb);
272                if(!proficiencyCheck(mob,0,auto))
273                {
274                        if(proficiency() < 25)
275                        {
276                                verb=L("making foul noises with @x1",target.name());
277                                displayText=L("You are @x1",verb);
278                        }
279                        else
280                        if(proficiency() < 50)
281                        {
282                                verb=L("trying to play @x1 and failing",target.name());
283                                displayText=L("You are @x1",verb);
284                        }
285                        else
286                        if(proficiency() < 75)
287                        {
288                                verb=L("playing @x1, but terribly",target.name());
289                                displayText=L("You are @x1",verb);
290                        }
291                        else
292                        {
293                                verb=L("playing @x1 pretty badly",target.name());
294                                displayText=L("You are @x1",verb);
295                        }
296                }
297                else
298                if(proficiency() == 100)
299                {
300                        verb=L("playing @x1 beautifully",target.name());
301                        displayText=L("You are @x1",verb);
302                }
303
304                final int duration=30;//getDuration(30,mob,1,3);
305                final CMMsg msg=CMClass.getMsg(mob,target,this,getActivityMessageType(),L("<S-NAME> start(s) playing <T-NAME>."));
306                if(mob.location().okMessage(mob,msg))
307                {
308                        instrument = target;
309                        mob.location().send(mob,msg);
310                        final PlayInstrument pA = (PlayInstrument)beneficialAffect(mob,mob,asLevel,duration);
311                        if(pA != null)
312                        {
313                                pA.proficiencies = this.proficiencies;
314                                pA.lastInstrType = this.lastInstrType;
315                                pA.instrument = this.instrument;
316                        }
317                }
318                return true;
319        }
