092        {
093                if((affected!=null)&&(affected instanceof MOB)&&(tickID==Tickable.TICKID_MOB))

116        @Override
117        public void unInvoke()
118        {
119                if(canBeUninvoked())
120                {
121                        if(affected instanceof MOB)
122                        {
123                                final MOB mob=(MOB)affected;
124                                if((buildingI!=null)&&(!aborted))
125                                {
126                                        if(messedUp)
127                                        {
128                                                if(activity == CraftingActivity.MENDING)
129                                                        messedUpCrafting(mob);
130                                                else
131                                                if(activity == CraftingActivity.LEARNING)
132                                                {
133                                                        commonEmote(mob,L("<S-NAME> fail(s) to learn how to make @x1.",buildingI.name()));
134                                                        buildingI.destroy();
135                                                }
136                                                else
137                                                if(activity == CraftingActivity.REFITTING)
138                                                        commonEmote(mob,L("<S-NAME> mess(es) up refitting @x1.",buildingI.name()));
139                                                else
140                                                        commonEmote(mob,L("<S-NAME> mess(es) up cobbling @x1.",buildingI.name()));
141                                        }
142                                        else
143                                        {
144                                                if(activity == CraftingActivity.MENDING)
145                                                {
146                                                        buildingI.setUsesRemaining(100);
147                                                        CMLib.achievements().possiblyBumpAchievement(mob, AchievementLibrary.Event.MENDER, 1, this);
148                                                }
149                                                else
150                                                if(activity==CraftingActivity.LEARNING)
151                                                {
152                                                        deconstructRecipeInto(mob, buildingI, recipeHolder );
153                                                        buildingI.destroy();
154                                                }
155                                                else
156                                                if(activity == CraftingActivity.REFITTING)
157                                                {
158                                                        buildingI.basePhyStats().setHeight(0);
159                                                        buildingI.recoverPhyStats();
160                                                }
161                                                else
162                                                {
163                                                        dropAWinner(mob,buildingI);
164                                                        CMLib.achievements().possiblyBumpAchievement(mob, AchievementLibrary.Event.CRAFTING, 1, this);
165                                                }
166                                        }
167                                }
168                                buildingI=null;
169                                activity = CraftingActivity.CRAFTING;
170                        }
171                }
172                super.unInvoke();
173        }

175        @Override
176        public boolean mayICraft(final Item I)
177        {
178                if(I==null)
179                        return false;
180                if(!super.mayBeCrafted(I))
181                        return false;
182                if(CMLib.flags().isDeadlyOrMaliciousEffect(I))
183                        return false;
184                if(!(I instanceof Armor))
185                        return false;
186                if(!I.fitsOn(Wearable.WORN_FEET))
187                        return false;
188                return true;
189        }

197        @Override
198        protected boolean canMend(MOB mob, Environmental E, boolean quiet)
199        {
200                if(!super.canMend(mob,E,quiet))
201                        return false;
202                if((!(E instanceof Item))
203                ||(!mayICraft((Item)E)))
204                {
205                        if(!quiet)
206                                commonTell(mob,L("That's not a cobbled item."));
207                        return false;
208                }
209                return true;
210        }

224        @Override
225        protected boolean autoGenInvoke(final MOB mob, List<String> commands, Physical givenTarget, final boolean auto, 
226                                                                 final int asLevel, int autoGenerate, boolean forceLevels, List<Item> crafted)
227        {
228                if(super.checkStop(mob, commands))
229                        return true;
230
231                if(super.checkInfo(mob, commands))
232                        return true;
233                
234                final PairVector<EnhancedExpertise,Integer> enhancedTypes=enhancedTypes(mob,commands);
235                int recipeLevel = 1;
236                randomRecipeFix(mob,addRecipes(mob,loadRecipes()),commands,autoGenerate);
237                if(commands.size()==0)
238                {
239                        commonTell(mob,L("Make what? Enter \"cobble list\" for a list, \"cobble info <item>\", \"cobble refit <item>\" to resize, "
240                                                        + "\"cobble learn <item>\", \"cobble scan\", or \"cobble mend <item>\", \"cobble stop\" to cancel."));
241                        return false;
242                }
243                if((!auto)
244                &&(commands.size()>0)
245                &&((commands.get(0)).equalsIgnoreCase("bundle")))
246                {
247                        bundling=true;
248                        if(super.invoke(mob,commands,givenTarget,auto,asLevel))
249                                return super.bundle(mob,commands);
250                        return false;
251                }
252                final List<List<String>> recipes=addRecipes(mob,loadRecipes());
253                final String str=commands.get(0);
254                String startStr=null;
255                bundling=false;
256                int duration=4;
257                final int[] cols={
258                        CMLib.lister().fixColWidth(29,mob.session()),
259                        CMLib.lister().fixColWidth(3,mob.session()),
260                        CMLib.lister().fixColWidth(3,mob.session()),
261                };
262                if(str.equalsIgnoreCase("list"))
263                {
264                        String mask=CMParms.combine(commands,1);
265                        boolean allFlag=false;
266                        if(mask.equalsIgnoreCase("all"))
267                        {
268                                allFlag=true;
269                                mask="";
270                        }
271                        final StringBuffer buf=new StringBuffer("");
272                        int toggler=1;
273                        final int toggleTop=2;
274                        for(int r=0;r<toggleTop;r++)
275                                buf.append((r>0?" ":"")+CMStrings.padRight(L("Item"),cols[0])+" "+CMStrings.padRight(L("Lvl"),cols[1])+" "+CMStrings.padRight(L("Amt"),cols[2]));
276                        buf.append("\n\r");
277                        final List<List<String>> listRecipes=((mask.length()==0) || mask.equalsIgnoreCase("all")) ? recipes : super.matchingRecipeNames(recipes, mask, true);
278                        for(int r=0;r<listRecipes.size();r++)
279                        {
280                                final List<String> V=listRecipes.get(r);
281                                if(V.size()>0)
282                                {
283                                        final String item=replacePercent(V.get(RCP_FINALNAME),"");
284                                        final int level=CMath.s_int(V.get(RCP_LEVEL));
285                                        final String wood=getComponentDescription(mob,V,RCP_WOOD);
286                                        if(wood.length()>5)
287                                        {
288                                                if(toggler>1)
289                                                        buf.append("\n\r");
290                                                toggler=toggleTop;
291                                        }
292                                        if((level<=xlevel(mob))||allFlag)
293                                        {
294                                                buf.append(CMStrings.padRight(item,cols[0])+" "+CMStrings.padRight(""+level,cols[1])+" "+CMStrings.padRightPreserve(""+wood,cols[2])+((toggler!=toggleTop)?" ":"\n\r"));
295                                                if(++toggler>toggleTop)
296                                                        toggler=1;
297                                        }
298                                }
299                        }
300                        if(toggler!=1)
301                                buf.append("\n\r");
302                        commonTell(mob,buf.toString());
303                        enhanceList(mob);
304                        return true;
305                }
306                else
307                if(((commands.get(0))).equalsIgnoreCase("learn"))
308                {
309                        return doLearnRecipe(mob, commands, givenTarget, auto, asLevel);
310                }
311                else
312                if(str.equalsIgnoreCase("scan"))
313                        return publicScan(mob,commands);
314                else
315                if(str.equalsIgnoreCase("mend"))
316                {
317                        buildingI=null;
318                        activity = CraftingActivity.CRAFTING;
319                        messedUp=false;
320                        final Vector<String> newCommands=CMParms.parse(CMParms.combine(commands,1));
321                        buildingI=getTargetItemFavorMOB(mob,mob.location(),givenTarget,newCommands,Wearable.FILTER_UNWORNONLY);
322                        if(!canMend(mob, buildingI,false))
323                                return false;
324                        activity = CraftingActivity.MENDING;
325                        if(!super.invoke(mob,commands,givenTarget,auto,asLevel))
326                                return false;
327                        startStr=L("<S-NAME> start(s) mending @x1.",buildingI.name());
328                        displayText=L("You are mending @x1",buildingI.name());
329                        verb=L("mending @x1",buildingI.name());
330                }
331                else
332                if(str.equalsIgnoreCase("refit"))
333                {
334                        buildingI=null;
335                        activity = CraftingActivity.CRAFTING;
336                        messedUp=false;
337                        final Vector<String> newCommands=CMParms.parse(CMParms.combine(commands,1));
338                        buildingI=getTargetItemFavorMOB(mob,mob.location(),givenTarget,newCommands,Wearable.FILTER_UNWORNONLY);
339                        if(buildingI==null)
340                                return false;
341                        if(!buildingI.fitsOn(Wearable.WORN_FEET))
342                        {
343                                commonTell(mob,L("That's not footwear.  That can't be refitted."));
344                                return false;
345                        }
346                        if(!(buildingI instanceof Armor))
347                        {
348                                commonTell(mob,L("You don't know how to refit that sort of thing."));
349                                return false;
350                        }
351                        if(buildingI.phyStats().height()==0)
352                        {
353                                commonTell(mob,L("@x1 is already the right size.",buildingI.name(mob)));
354                                return false;
355                        }
356                        activity = CraftingActivity.REFITTING;
357                        if(!super.invoke(mob,commands,givenTarget,auto,asLevel))
358                                return false;
359                        startStr=L("<S-NAME> start(s) refitting @x1.",buildingI.name());
360                        displayText=L("You are refitting @x1",buildingI.name());
361                        verb=L("refitting @x1",buildingI.name());
362                }
363                else
364                {
365                        buildingI=null;
366                        activity = CraftingActivity.CRAFTING;
367                        messedUp=false;
368                        aborted=false;
369                        int amount=-1;
370                        if((commands.size()>1)&&(CMath.isNumber(commands.get(commands.size()-1))))
371                        {
372                                amount=CMath.s_int(commands.get(commands.size()-1));
373                                commands.remove(commands.size()-1);
374                        }
375                        final String recipeName=CMParms.combine(commands,0);
376                        List<String> foundRecipe=null;
377                        final List<List<String>> matches=matchingRecipeNames(recipes,recipeName,true);
378                        for(int r=0;r<matches.size();r++)
379                        {
380                                final List<String> V=matches.get(r);
381                                if(V.size()>0)
382                                {
383                                        final int level=CMath.s_int(V.get(RCP_LEVEL));
384                                        if((autoGenerate>0)||(level<=xlevel(mob)))
385                                        {
386                                                foundRecipe=V;
387                                                recipeLevel=level;
388                                                break;
389                                        }
390                                }
391                        }
392                        if(foundRecipe==null)
393                        {
394                                commonTell(mob,L("You don't know how to make a '@x1'.  Try \"cobble list\" for a list.",recipeName));
395                                return false;
396                        }
397
398                        final String woodRequiredStr = foundRecipe.get(RCP_WOOD);
399                        final int[] compData = new int[CF_TOTAL];
400                        final String realRecipeName=replacePercent(foundRecipe.get(RCP_FINALNAME),"");
401                        final List<Object> componentsFoundList=getAbilityComponents(mob, woodRequiredStr, "make "+CMLib.english().startWithAorAn(realRecipeName),autoGenerate,compData,1);
402                        if(componentsFoundList==null)
403                                return false;
404                        int woodRequired=CMath.s_int(woodRequiredStr);
405                        woodRequired=adjustWoodRequired(woodRequired,mob);
406
407                        if(amount>woodRequired)
408                                woodRequired=amount;
409                        final String misctype=foundRecipe.get(RCP_MISCTYPE);
410                        final int[] pm={RawMaterial.MATERIAL_METAL,RawMaterial.MATERIAL_MITHRIL,RawMaterial.MATERIAL_CLOTH,RawMaterial.MATERIAL_WOODEN,RawMaterial.MATERIAL_LEATHER};
411                        bundling=misctype.equalsIgnoreCase("BUNDLE");
412                        final int[][] data=fetchFoundResourceData(mob,
413                                                                                                        woodRequired,"metal",pm,
414                                                                                                        0,null,null,
415                                                                                                        bundling,
416                                                                                                        autoGenerate,
417                                                                                                        enhancedTypes);
418                        if(data==null)
419                                return false;
420                        fixDataForComponents(data,woodRequiredStr,(autoGenerate>0) && (woodRequired==0),componentsFoundList, 1);
421                        woodRequired=data[0][FOUND_AMT];
422
423                        if(!super.invoke(mob,commands,givenTarget,auto,asLevel))
424                                return false;
425                        final MaterialLibrary.DeadResourceRecord deadMats;
426                        if((componentsFoundList.size() > 0)||(autoGenerate>0))
427                                deadMats = new MaterialLibrary.DeadResourceRecord();
428                        else
429                                deadMats = CMLib.materials().destroyResources(mob.location(),woodRequired,data[0][FOUND_CODE],0,null,null);
430                        final MaterialLibrary.DeadResourceRecord deadComps = CMLib.ableComponents().destroyAbilityComponents(componentsFoundList);
431                        final int lostValue=autoGenerate>0?0:(deadMats.lostValue + deadComps.lostValue);
432                        buildingI=CMClass.getItem(foundRecipe.get(RCP_CLASSTYPE));
433                        if(buildingI==null)
434                        {
435                                commonTell(mob,L("There's no such thing as a @x1!!!",foundRecipe.get(RCP_CLASSTYPE)));
436                                return false;
437                        }
438                        duration=getDuration(CMath.s_int(foundRecipe.get(RCP_TICKS)),mob,CMath.s_int(foundRecipe.get(RCP_LEVEL)),4);
439                        buildingI.setMaterial(getBuildingMaterial(woodRequired,data,compData));
440                        String itemName=determineFinalName(foundRecipe.get(RCP_FINALNAME),buildingI.material(),deadMats,deadComps);
441                        if(itemName.endsWith("s"))
442                                itemName="some "+itemName;
443                        else
444                                itemName=CMLib.english().startWithAorAn(itemName);
445                        buildingI.setName(itemName);
446                        startStr=L("<S-NAME> start(s) cobbling @x1.",buildingI.name());
447                        displayText=L("You are cobbling @x1",buildingI.name());
448                        verb=L("cobbling @x1",buildingI.name());
449                        playSound="sanding.wav";
450                        buildingI.setDisplayText(L("@x1 lies here",itemName));
451                        buildingI.setDescription(itemName+". ");
452                        buildingI.basePhyStats().setWeight(getStandardWeight(woodRequired+compData[CF_AMOUNT],bundling));
453                        buildingI.setBaseValue(CMath.s_int(foundRecipe.get(RCP_VALUE)));
454                        final int hardness=RawMaterial.CODES.HARDNESS(buildingI.material())-6;
455                        buildingI.basePhyStats().setLevel(CMath.s_int(foundRecipe.get(RCP_LEVEL))+(hardness*3));
456                        if(buildingI.basePhyStats().level()<1)
457                                buildingI.basePhyStats().setLevel(1);
458                        final int capacity=CMath.s_int(foundRecipe.get(RCP_CAPACITY));
459                        final long canContain=getContainerType(foundRecipe.get(RCP_CONTAINMASK));
460                        final int armordmg=CMath.s_int(foundRecipe.get(RCP_ARMORDMG));
461                        setBrand(mob, buildingI);
462                        final String spell=(foundRecipe.size()>RCP_SPELL)?foundRecipe.get(RCP_SPELL).trim():"";
463                        if(bundling)
464                                buildingI.setBaseValue(lostValue);
465                        addSpells(buildingI,spell,deadMats.lostProps,deadComps.lostProps);
466                        if((buildingI instanceof Armor)&&(!(buildingI instanceof FalseLimb)))
467                        {
468                                ((Armor)buildingI).basePhyStats().setArmor(0);
469                                if(armordmg!=0)
470                                        ((Armor)buildingI).basePhyStats().setArmor(armordmg+(baseYield()+abilityCode()-1));
471                                setWearLocation(buildingI,misctype,hardness);
472                        }
473                        if(buildingI instanceof Container)
474                        {
475                                if(capacity>0)
476                                {
477                                        ((Container)buildingI).setCapacity(capacity+woodRequired);
478                                        ((Container)buildingI).setContainTypes(canContain);
479                                }
480                        }
481                        buildingI.recoverPhyStats();
482                        buildingI.text();
483                        buildingI.recoverPhyStats();
484                }
485
486                messedUp=!proficiencyCheck(mob,0,auto);
487
488                if(bundling)
489                {
490                        messedUp=false;
491                        duration=1;
492                        verb=L("bundling @x1",RawMaterial.CODES.NAME(buildingI.material()).toLowerCase());
493                        startStr=L("<S-NAME> start(s) @x1.",verb);
494                        displayText=L("You are @x1",verb);
495                }
496
497                if(autoGenerate>0)
498                {
499                        crafted.add(buildingI);
500                        return true;
501                }
502
503                final CMMsg msg=CMClass.getMsg(mob,buildingI,this,getActivityMessageType(),startStr);
504                if(mob.location().okMessage(mob,msg))
505                {
506                        mob.location().send(mob,msg);
507                        buildingI=(Item)msg.target();
508                        beneficialAffect(mob,mob,asLevel,duration);
509                        enhanceItem(mob,buildingI,recipeLevel,enhancedTypes);
510                }
511                else
512                if(bundling)
513                {
514                        messedUp=false;
515                        aborted=false;
516                        unInvoke();
517                }
518                return true;
519        }
