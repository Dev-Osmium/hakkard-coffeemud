100        @Override
101        public boolean tick(final Tickable ticking, final int tickID)
102        {
103                if((affected!=null)&&(affected instanceof Room))
104                {
105                        final Room R=(Room)affected;
106                        if(R!=null)
107                        {
108                                if(R.getArea().getClimateObj().canSeeTheSun(R))
109                                        goodticks++;
110                                final MOB mob=invoker();
111                                if(tickUp==6)
112                                {
113                                        if((found==null)
114                                        ||(mob==null)
115                                        ||(mob.location()==null))
116                                        {
117                                                commonTell(mob,L("Your @x1 crop has failed.\n\r",foundShortName));
118                                                unInvoke();
119                                        }
120                                }
121                                else
122                                if((tickUp > 10)&&(goodticks < (tickUp/2)))
123                                {
124                                        found=null;
125                                        commonTell(mob,L("Your @x1 crop has failed due to lack of sunlight.\n\r",foundShortName));
126                                        unInvoke();
127                                }
128                        }
129                }
130                else
131                        goodticks=0;
132                return super.tick(ticking,tickID);
133        }

135        protected boolean isCompost(final Item I)
136        {
137                return ((I!=null) &&(I.rawSecretIdentity().equals("compost")));
138        }

140        protected int deCompost(final Item I, int doubleRemain)
141        {
142                int amount=0;
143                if(I.phyStats().weight()<=doubleRemain)
144                {
145                        amount+=I.phyStats().weight();
146                        doubleRemain-=I.phyStats().weight();
147                        I.destroy();
148                }
149                else
150                {
151                        if(I.basePhyStats().weight()<=doubleRemain)
152                                I.destroy();
153                        else
154                                I.basePhyStats().setWeight(I.basePhyStats().weight()-doubleRemain);
155                        amount+=doubleRemain;
156                        doubleRemain=0;
157                }
158                return amount;
159        }

161        @Override
162        public void unInvoke()
163        {
164                final boolean isaborted=aborted;
165                final Environmental aff=affected;
166                if(canBeUninvoked())
167                {
168                        if((affected!=null)&&(affected==room))
169                        {
170                                if((found!=null)&&(!isaborted))
171                                {
172                                        int amount=CMLib.dice().roll(1,7,0)*(baseYield()+abilityCode());
173                                        int origAmount = amount; 
174                                        int doubleRemain = amount * 10;
175                                        for(final Enumeration<Item> i=room.items();i.hasMoreElements();)
176                                        {
177                                                final Item I=i.nextElement();
178                                                if((I instanceof PackagedItems)
179                                                &&(((PackagedItems)I).areAllItemsTheSame()))
180                                                {
181                                                        Item I2=((PackagedItems)I).peekFirstItem();
182                                                        while(isCompost(I2))
183                                                        {
184                                                                int amt=deCompost(I2,doubleRemain);
185                                                                if(I2.amDestroyed())
186                                                                        ((PackagedItems)I).setNumberOfItemsInPackage(((PackagedItems)I).numberOfItemsInPackage()-1);
187                                                                if(amt != 0)
188                                                                {
189                                                                        amount += amt * origAmount;
190                                                                        doubleRemain -=amt;
191                                                                }
192                                                                I2.destroy();
193                                                                if(((PackagedItems)I).numberOfItemsInPackage()==0)
194                                                                {
195                                                                        I.destroy();
196                                                                        break;
197                                                                }
198                                                                I2=((PackagedItems)I).peekFirstItem();
199                                                        }
200                                                        I2.destroy();
201                                                }
202                                                else
203                                                if(isCompost(I))
204                                                {
205                                                        int amt=deCompost(I,doubleRemain);
206                                                        if(amt != 0)
207                                                        {
208                                                                amount += amt * origAmount;
209                                                                doubleRemain -=amt;
210                                                        }
211                                                }
212                                                if(doubleRemain <= 0)
213                                                        break;
214                                        }
215                                        String s="s";
216                                        if(amount==1)
217                                                s="";
218                                        room.showHappens(CMMsg.MSG_OK_VISUAL,L("@x1 pound@x2 of @x3 have grown here.",""+amount,s,foundShortName));
219                                        for(int i=0;i<amount;i++)
220                                        {
221                                                final Item newFound=(Item)found.copyOf();
222                                                if(!dropAWinner(null,room,newFound))
223                                                        break;
224                                        }
225                                }
226                        }
227                }
228                super.unInvoke();
229                if((canBeUninvoked)
230                &&(aff!=null)
231                &&(aff instanceof MOB)
232                &&(aff!=room)
233                &&(!isaborted)
234                &&(room!=null))
235                {
236                        final Farming F=((Farming)copyOf());
237                        F.unInvoked=false;
238                        F.tickUp=0;
239                        F.tickDown=50;
240                        F.startTickDown(invoker,room,50);
241                }
242        }

244        public boolean isPotentialCrop(Room R, int code)
245        {
246                if(R==null)
247                        return false;
248                if(R.resourceChoices()==null)
249                        return false;
250                for(int i=0;i<R.resourceChoices().size();i++)
251                {
252                        if(R.resourceChoices().get(i).intValue()==code)
253                                return true;
254                }
255                if(R.myResource()==RawMaterial.RESOURCE_DIRT)
256                        return true;
257                return false;
258        }

260        private boolean plantable(MOB mob, Item I2)
261        {
262                if((I2!=null)
263                &&(I2 instanceof RawMaterial)
264                &&(CMLib.flags().canBeSeenBy(I2,mob))
265                &&(I2.container()==null)
266                &&((I2 instanceof Food)||(((RawMaterial)I2).getSubType().equals(RawMaterial.ResourceSubType.SEED.name())))
267                &&(((I2.material()&RawMaterial.MATERIAL_MASK)==RawMaterial.MATERIAL_VEGETATION)
268                        ||(I2.material()==RawMaterial.RESOURCE_COTTON)
269                        ||(I2.material()==RawMaterial.RESOURCE_HEMP)
270                        ||((I2.material()&RawMaterial.MATERIAL_MASK)==RawMaterial.MATERIAL_WOODEN)))
271                        return true;
272                return false;
273        }

275        @Override
276        public boolean invoke(MOB mob, List<String> commands, Physical givenTarget, boolean auto, int asLevel)
277        {
278                final Room R=mob.location();
279                if(R==null)
280                        return false;
281                if(super.checkStop(mob, commands))
282                        return true;
283                bundling=false;
284                if((!auto)
285                &&(commands.size()>0)
286                &&((commands.get(0)).equalsIgnoreCase("bundle")))
287                {
288                        bundling=true;
289                        if(super.invoke(mob,commands,givenTarget,auto,asLevel))
290                                return super.bundle(mob,commands);
291                        return false;
292                }
293
294                verb=L("planting");
295                if((!auto)&&(!R.getArea().getClimateObj().canSeeTheSun(R)))
296                {
297                        commonTell(mob,L("You need clear sunlight to do your farming.  Check the time and weather."));
298                        return false;
299                }
300                if((!auto)
301                &&(R.domainType()!=Room.DOMAIN_OUTDOORS_HILLS)
302                &&(R.domainType()!=Room.DOMAIN_OUTDOORS_PLAINS)
303                &&(R.domainType()!=Room.DOMAIN_OUTDOORS_WOODS)
304                &&(R.domainType()!=Room.DOMAIN_OUTDOORS_JUNGLE)
305                &&(R.domainType()!=Room.DOMAIN_OUTDOORS_SWAMP)
306                &&(R.myResource()!=RawMaterial.RESOURCE_DIRT))
307                {
308                        commonTell(mob,L("The land is not suitable for farming here."));
309                        return false;
310                }
311                if((!auto)&&(R.getArea().getClimateObj().weatherType(R)==Climate.WEATHER_DROUGHT))
312                {
313                        commonTell(mob,L("The current drought conditions make planting useless."));
314                        return false;
315                }
316                if(R.fetchEffect(ID())!=null)
317                {
318                        commonTell(mob,L("It looks like a crop is already growing here."));
319                        return false;
320                }
321                if(mob.isMonster()
322                &&(!auto)
323                &&(!CMLib.flags().isAnimalIntelligence(mob))
324                &&(commands.size()==0))
325                {
326                        Item mine=null;
327                        for(int i=0;i<R.numItems();i++)
328                        {
329                                final Item I2=R.getItem(i);
330                                if(plantable(mob,I2))
331                                {
332                                        mine=I2;
333                                        commands.add(RawMaterial.CODES.NAME(I2.material()));
334                                        break;
335                                }
336                        }
337                        if(mine==null)
338                        {
339                                for(int i=0;i<mob.numItems();i++)
340                                {
341                                        final Item I2=mob.getItem(i);
342                                        if(plantable(mob,I2))
343                                        {
344                                                commands.add(RawMaterial.CODES.NAME(I2.material()));
345                                                mine=(Item)I2.copyOf();
346                                                if(R.findItem(null,R.getContextName(I2))==null)
347                                                        R.addItem(mine,ItemPossessor.Expire.Resource);
348                                                break;
349                                        }
350                                }
351                        }
352                        if(mine==null)
353                        {
354                                commonTell(mob,L("You don't have anything you can plant."));
355                                return false;
356                        }
357                }
358                else
359                if(commands.size()==0)
360                {
361                        commonTell(mob,L("Grow what?"));
362                        return false;
363                }
364                int code=-1;
365                final String what=CMParms.combine(commands,0).toUpperCase();
366                final RawMaterial.CODES codes = RawMaterial.CODES.instance();
367                for(final int cd : codes.all())
368                {
369                        final String str=codes.name(cd).toUpperCase();
370                        if((str.equals(what))
371                        &&(((cd&RawMaterial.MATERIAL_MASK)==RawMaterial.MATERIAL_VEGETATION)
372                          ||(cd==RawMaterial.RESOURCE_COTTON)
373                          ||(cd==RawMaterial.RESOURCE_HEMP)
374                          ||((cd&RawMaterial.MATERIAL_MASK)==RawMaterial.MATERIAL_WOODEN)))
375                        {
376                                code=cd;
377                                foundShortName=CMStrings.capitalizeAndLower(str);
378                                break;
379                        }
380                }
381                if(code<0)
382                {
383                        for(final int cd : codes.all())
384                        {
385                                final String str=codes.name(cd).toUpperCase();
386                                if((str.toUpperCase().startsWith(what)||(what.startsWith(str)))
387                                &&(((cd&RawMaterial.MATERIAL_MASK)==RawMaterial.MATERIAL_VEGETATION)
388                                  ||(cd==RawMaterial.RESOURCE_COTTON)
389                                  ||(cd==RawMaterial.RESOURCE_HEMP)
390                                  ||((cd&RawMaterial.MATERIAL_MASK)==RawMaterial.MATERIAL_WOODEN)))
391                                {
392                                        code=cd;
393                                        foundShortName=CMStrings.capitalizeAndLower(str);
394                                        break;
395                                }
396                        }
397                }
398                if(code<0)
399                {
400                        commonTell(mob,L("You've never heard of '@x1'.",CMParms.combine(commands,0)));
401                        return false;
402                }
403
404                Item mine=null;
405                for(int i=0;i<R.numItems();i++)
406                {
407                        final Item I=R.getItem(i);
408                        if(plantable(mob,I)
409                        &&(I.material()==code))
410                        {
411                                mine = I;
412                                break;
413                        }
414                }
415                if(mine==null)
416                {
417                        commonTell(mob,L("You'll need to have some @x1 to seed from on the ground first.",foundShortName));
418                        return false;
419                }
420                final String mineName=mine.name();
421                mine=(Item)CMLib.materials().unbundle(mine,-1,null);
422                if(mine==null)
423                {
424                        commonTell(mob,L("'@x1' is not suitable for use as a seed crop.",mineName));
425                        return false;
426                }
427                if(!(isPotentialCrop(R,code)))
428                {
429                        commonTell(mob,L("'@x1' does not seem to be taking root here.",mineName));
430                        return false;
431                }
432
433                found=null;
434                if(!super.invoke(mob,commands,givenTarget,auto,asLevel))
435                        return false;
436
437                if((proficiencyCheck(mob,0,auto))
438                &&(isPotentialCrop(R,code)))
439                {
440                        found=(Item)CMLib.materials().makeResource(code,Integer.toString(R.domainType()),false,null, "");
441                        if((found!=null)
442                        &&(found.material()==RawMaterial.RESOURCE_HERBS)
443                        &&(mine.material()==found.material()))
444                        {
445                                found.setName(mine.name());
446                                found.setDisplayText(mine.displayText());
447                                found.setDescription(mine.description());
448                                found.text();
449                        }
450                }
451
452                mine.destroy();
453                final int duration=getDuration(mob,1);
454                final CMMsg msg=CMClass.getMsg(mob,found,this,getActivityMessageType(),L("<S-NAME> start(s) planting @x1.",foundShortName));
455                verb=L("planting @x1",foundShortName);
456                displayText=L("You are planting @x1",foundShortName);
457                room=R;
458                if(R.okMessage(mob,msg))
459                {
460                        R.send(mob,msg);
461                        found=(Item)msg.target();
462                        beneficialAffect(mob,mob,asLevel,duration);
463                }
464                return true;
465        }
