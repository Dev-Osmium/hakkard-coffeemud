037public class Taxidermy extends CraftingSkill
038{
039        @Override
040        public String ID()
041        {
042                return "Taxidermy";
043        }
044
045        private final static String localizedName = CMLib.lang().L("Taxidermy");
046
047        @Override
048        public String name()
049        {
050                return localizedName;
051        }
052
053        private static final String[] triggerStrings =I(new String[] {"STUFF","TAXIDERMY"});
054        @Override
055        public String[] triggerStrings()
056        {
057                return triggerStrings;
058        }
059
060        @Override
061        public String supportedResourceString()
062        {
063                return "BODIES";
064        }
065
066        public String parametersFormat()
067        {
068                return "POSE_NAME\nPOSE_DESCRIPTION\n...\n";
069        }
070
071        protected final static String CRAFTING_RACE_STR_PREFIX="This statue was once ";
072        protected final static String CRAFTING_RACE_STR=CRAFTING_RACE_STR_PREFIX+"@x1.";
073        
074        protected String foundShortName="";
075
076        public Taxidermy()
077        {
078                super();
079                displayText=L("You are stuffing...");
080                verb=L("stuffing");
081        }
082
083        @Override
084        public void unInvoke()
085        {
086                if(canBeUninvoked())
087                {
088                        if(affected instanceof MOB)
089                        {
090                                final MOB mob=(MOB)affected;
091                                if((buildingI!=null)&&(!aborted))
092                                {
093                                        if(messedUp)
094                                                commonTell(mob,L("You've messed up stuffing @x1!",foundShortName));
095                                        else
096                                        {
097                                                dropAWinner(mob,buildingI);
098                                                CMLib.achievements().possiblyBumpAchievement(mob, AchievementLibrary.Event.CRAFTING, 1, this);
099                                        }
100                                }
101                        }
102                }
103                super.unInvoke();
104        }
105
106        protected static String getStatueRace(final Item buildingI)
107        {
108                if(buildingI != null)
109                {
110                        final int x=buildingI.secretIdentity().indexOf(CRAFTING_RACE_STR_PREFIX);
111                        if(x>=0)
112                        {
113                                int y=buildingI.secretIdentity().indexOf('.',x+CRAFTING_RACE_STR_PREFIX.length());
114                                if(y>=0)
115                                {
116                                        return buildingI.secretIdentity().substring(x,y);
117                                }
118                        }
119                }
120                return "";
121        }
122
123        @Override
124        protected List<List<String>> loadRecipes()
125        {
126                final String filename="taxidermy.txt";
127                @SuppressWarnings("unchecked")
128                List<List<String>> V=(List<List<String>>)Resources.getResource("PARSED_RECIPE: "+filename);
129                if(V==null)
130                {
131                        V=new Vector<List<String>>();
132                        final StringBuffer str=new CMFile(Resources.buildResourcePath("skills")+filename,null,CMFile.FLAG_LOGERRORS).text();
133                        final List<String> strV=Resources.getFileLineVector(str);
134                        List<String> V2=null;
135                        boolean header=true;
136                        for(int v=0;v<strV.size();v++)
137                        {
138                                final String s=strV.get(v);
139                                if(header)
140                                {
141                                        if((V2!=null)&&(V2.size()>0))
142                                                V.add(V2);
143                                        V2=new Vector<String>();
144                                }
145                                if(s.length()==0)
146                                        header=true;
147                                else
148                                if(V2 != null)
149                                {
150                                        V2.add(s);
151                                        header=false;
152                                }
153                        }
154                        if((V2!=null)&&(V2.size()>0))
155                                V.add(V2);
156                        if(V.size()==0)
157                                Log.errOut("Taxidermy","Poses not found!");
158                        Resources.submitResource("PARSED_RECIPE: "+filename,V);
159                }
160                return V;
161        }
162
163        @Override
164        public boolean invoke(MOB mob, List<String> commands, Physical givenTarget, boolean auto, int asLevel)
165        {
166                if(super.checkStop(mob, commands))
167                        return true;
168                final List<List<String>> POSES=loadRecipes();
169                String pose=null;
170                if(CMParms.combine(commands,0).equalsIgnoreCase("list"))
171                {
172                        final StringBuffer str=new StringBuffer(L("^xTaxidermy Poses^?^.\n"));
173                        for(int p=0;p<POSES.size();p++)
174                        {
175                                final List<String> PP=POSES.get(p);
176                                if(PP.size()>1)
177                                        str.append((PP.get(0))+"\n");
178                        }
179                        mob.tell(str.toString());
180                        return true;
181                }
182                else
183                if(commands.size()>0)
184                {
185                        for(int p=0;p<POSES.size();p++)
186                        {
187                                final List<String> PP=POSES.get(p);
188                                if((PP.size()>1)&&(PP.get(0).equalsIgnoreCase(commands.get(0))))
189                                {
190                                        commands.remove(0);
191                                        pose=PP.get(CMLib.dice().roll(1,PP.size()-1,0));
192                                        break;
193                                }
194                        }
195                }
196
197                verb=L("stuffing");
198                final String str=CMParms.combine(commands,0);
199                final Item I=mob.location().findItem(null,str);
200                if((I==null)||(!CMLib.flags().canBeSeenBy(I,mob)))
201                {
202                        commonTell(mob,L("You don't see anything called '@x1' here.",str));
203                        return false;
204                }
205                foundShortName=I.Name();
206                if((!(I instanceof DeadBody))
207                ||(((DeadBody)I).isPlayerCorpse())
208                ||(((DeadBody)I).getMobName().length()==0))
209                {
210                        commonTell(mob,L("You don't know how to stuff @x1.",I.name(mob)));
211                        return false;
212                }
213                for(int i=0;i<mob.location().numItems();i++)
214                {
215                        final Item I2=mob.location().getItem(i);
216                        if(I2.container()==I)
217                        {
218                                commonTell(mob,L("You need to remove the contents of @x1 first.",I2.name(mob)));
219                                return false;
220                        }
221                }
222                int woodRequired=I.basePhyStats().weight()/5;
223                final int[] pm={RawMaterial.MATERIAL_CLOTH};
224                final int[][] data=fetchFoundResourceData(mob,
225                                                                                        woodRequired,"cloth stuffing",pm,
226                                                                                        0,null,null,
227                                                                                        false,
228                                                                                        0,
229                                                                                        null);
230                if(data==null)
231                        return false;
232                woodRequired=data[0][FOUND_AMT];
233
234                activity = CraftingActivity.CRAFTING;
235                buildingI=null;
236                if(!super.invoke(mob,commands,givenTarget,auto,asLevel))
237                        return false;
238                CMLib.materials().destroyResourcesValue(mob.location(),woodRequired,data[0][FOUND_CODE],0,null);
239                messedUp=!proficiencyCheck(mob,0,auto);
240                if(buildingI!=null)
241                        foundShortName=I.Name();
242                int duration=15+(woodRequired/3);
243                if(duration>65)
244                        duration=65;
245                duration=getDuration(duration,mob,1,10);
246                buildingI=CMClass.getItem("GenItem");
247                buildingI.basePhyStats().setWeight(woodRequired);
248                final String name=((DeadBody)I).getMobName();
249                final String desc=((DeadBody)I).getMobDescription();
250                buildingI.setMaterial(super.getBuildingMaterial(woodRequired, data, new int[CF_TOTAL]));
251                buildingI.setName(L("the stuffed body of @x1",name));
252                final CharStats C=(I instanceof DeadBody)?((DeadBody)I).charStats():null;
253                if((pose==null)||(C==null))
254                        buildingI.setDisplayText(L("the stuffed body of @x1 stands here",name));
255                else
256                {
257                        pose=CMStrings.replaceAll(pose,"<S-NAME>",buildingI.name());
258                        pose=CMStrings.replaceAll(pose,"<S-HIS-HER>",C.hisher());
259                        pose=CMStrings.replaceAll(pose,"<S-HIM-HER>",C.himher());
260                        pose=CMStrings.replaceAll(pose,"<S-HIM-HERSELF>",C.himher()+"self");
261                        buildingI.setDisplayText(pose);
262                }
263                buildingI.setDescription(desc);
264                setBrand(mob, buildingI);
265                if(C!=null)
266                        buildingI.setSecretIdentity((buildingI.secretIdentity()+"  "+L(CRAFTING_RACE_STR,C.getMyRace().name())).trim());
267                buildingI.recoverPhyStats();
268                displayText=L("You are stuffing @x1",I.name());
269                verb=L("stuffing @x1",I.name());
270                playSound="scissor.wav";
271                final CMMsg msg=CMClass.getMsg(mob,buildingI,this,getActivityMessageType(),L("<S-NAME> start(s) stuffing @x1.",I.name()));
272                if(mob.location().okMessage(mob,msg))
273                {
274                        mob.location().send(mob,msg);
275                        buildingI=(Item)msg.target();
276                        beneficialAffect(mob,mob,asLevel,duration);
277                }
278                I.destroy();
279                return true;
280        }
