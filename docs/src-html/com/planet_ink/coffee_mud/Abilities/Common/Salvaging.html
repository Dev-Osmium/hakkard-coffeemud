036public class Salvaging extends CommonSkill
037{
038        @Override
039        public String ID()
040        {
041                return "Salvaging";
042        }
043
044        private final static String     localizedName   = CMLib.lang().L("Salvaging");
045
046        @Override
047        public String name()
048        {
049                return localizedName;
050        }
051
052        private static final String[]   triggerStrings  = I(new String[] { "SALVAGE", "SALVAGING" });
053
054        @Override
055        public String[] triggerStrings()
056        {
057                return triggerStrings;
058        }
059
060        @Override
061        protected ExpertiseLibrary.SkillCostDefinition getRawTrainingCost()
062        {
063                return CMProps.getNormalSkillGainCost(ID());
064        }
065
066        @Override
067        public int classificationCode()
068        {
069                return Ability.ACODE_COMMON_SKILL | Ability.DOMAIN_NATURELORE;
070        }
071
072        protected Item          found                   = null;
073        protected int           amount                  = 0;
074        protected String        oldItemName             = "";
075        protected boolean       messedUp                = false;
076
077        public Salvaging()
078        {
079                super();
080                displayText = L("You are salvaging...");
081                verb = L("salvaging");
082        }
083
084        @Override
085        public boolean tick(final Tickable ticking, final int tickID)
086        {
087                if((affected!=null)
088                &&(affected instanceof MOB)
089                &&(tickID==Tickable.TICKID_MOB))
090                {
091                        if(found==null)
092                        {
093                                messedUp=true;
094                                unInvoke();
095                        }
096                }
097                return super.tick(ticking,tickID);
098        }
099        
100        protected void finishSalvage(final MOB mob, final Item found, final int amount)
101        {
102                final CMMsg msg=CMClass.getMsg(mob,found,this,getCompletedActivityMessageType(),null);
103                msg.setValue(amount);
104                if(mob.location().okMessage(mob, msg))
105                {
106                        String s="s";
107                        if(msg.value()==1)
108                                s="";
109                        msg.modify(L("<S-NAME> manage(s) to salvage @x1 pound@x2 of @x3.",""+msg.value(),s,RawMaterial.CODES.NAME(found.material()).toLowerCase()));
110                        mob.location().send(mob, msg);
111                        for(int i=0;i<msg.value();i++)
112                        {
113                                final Item newFound=(Item)found.copyOf();
114                                if(!dropAWinner(mob,newFound))
115                                        break;
116                        }
117                }
118        }
119        
120        @Override
121        public void unInvoke()
122        {
123                if(canBeUninvoked())
124                {
125                        if(affected instanceof MOB)
126                        {
127                                final MOB mob=(MOB)affected;
128                                if((found!=null)&&(!aborted)&&(mob.location()!=null))
129                                {
130                                        if(messedUp)
131                                                commonTell(mob,L("You've messed up salvaging @x1!",oldItemName));
132                                        else
133                                        {
134                                                Item baseShip=found;
135                                                int finalAmount=amount*(baseYield()+abilityCode());
136                                                finishSalvage(mob,baseShip, finalAmount);
137                                                if((baseShip.material()&RawMaterial.MATERIAL_MASK)!=RawMaterial.MATERIAL_METAL)
138                                                {
139                                                        Item metalFound=CMLib.materials().makeItemResource(RawMaterial.RESOURCE_IRON);
140                                                        int metalAmount = Math.round(CMath.sqrt(finalAmount));
141                                                        finishSalvage(mob,metalFound, metalAmount);
142                                                }
143                                                if((baseShip.material()&RawMaterial.MATERIAL_MASK)!=RawMaterial.MATERIAL_CLOTH)
144                                                {
145                                                        Item clothFound=CMLib.materials().makeItemResource(RawMaterial.RESOURCE_COTTON);
146                                                        int metalAmount = Math.round(CMath.sqrt(finalAmount));
147                                                        int clothAmount = Math.round(CMath.sqrt(metalAmount));
148                                                        finishSalvage(mob,clothFound, clothAmount);
149                                                }
150                                        }
151                                }
152                        }
153                }
154                super.unInvoke();
155        }
156
157        @Override
158        public boolean invoke(MOB mob, List<String> commands, Physical givenTarget, boolean auto, int asLevel)
159        {
160                if(super.checkStop(mob, commands))
161                        return true;
162                verb=L("salvaging");
163                final String str=CMParms.combine(commands,0);
164                final Item I=mob.location().findItem(null,str);
165                if((I==null)||(!CMLib.flags().canBeSeenBy(I,mob)))
166                {
167                        commonTell(mob,L("You don't see anything called '@x1' here.",str));
168                        return false;
169                }
170                boolean okMaterial=true;
171                oldItemName=I.Name();
172                switch(I.material()&RawMaterial.MATERIAL_MASK)
173                {
174                case RawMaterial.MATERIAL_FLESH:
175                case RawMaterial.MATERIAL_LIQUID:
176                case RawMaterial.MATERIAL_PAPER:
177                case RawMaterial.MATERIAL_ENERGY:
178                case RawMaterial.MATERIAL_GAS:
179                case RawMaterial.MATERIAL_VEGETATION:
180                {
181                        okMaterial = false;
182                        break;
183                }
184                }
185                if(!okMaterial)
186                {
187                        commonTell(mob,L("You don't know how to salvage @x1.",I.name(mob)));
188                        return false;
189                }
190
191                if(I instanceof RawMaterial)
192                {
193                        commonTell(mob,L("@x1 already looks like salvage.",I.name(mob)));
194                        return false;
195                }
196
197                if(CMLib.flags().isEnchanted(I))
198                {
199                        commonTell(mob,L("@x1 is enchanted, and can't be salvaged.",I.name(mob)));
200                        return false;
201                }
202
203                final LandTitle t=CMLib.law().getLandTitle(mob.location());
204                if((t!=null)&&(!CMLib.law().doesHavePriviledgesHere(mob,mob.location())))
205                {
206                        mob.tell(L("You are not allowed to salvage anything here."));
207                        return false;
208                }
209
210                if((!(I instanceof SailingShip))
211                ||((((SailingShip)I).subjectToWearAndTear())&&(((SailingShip)I).usesRemaining()>0))
212                ||(((SailingShip)I).getShipArea()==null))
213                {
214                        mob.tell(L("You can only salvage large sunk sailing ships, which @x1 is not.",I.Name()));
215                        return false;
216                }
217                SailingShip ship=(SailingShip)I;
218                Area shipArea=ship.getShipArea();
219                
220                int totalWeight=I.phyStats().weight();
221                final Vector<Item> itemsToMove=new Vector<Item>();
222                for(final Enumeration<Room> r=shipArea.getProperMap();r.hasMoreElements();)
223                {
224                        final Room R=r.nextElement();
225                        if(R!=null)
226                        {
227                                if(R.numInhabitants()>0)
228                                {
229                                        mob.tell(L("There are still people aboard!"));
230                                        return false;
231                                }
232                                for(final Enumeration<Item> i=R.items();i.hasMoreElements();)
233                                {
234                                        final Item I2=i.nextElement();
235                                        if((I2!=null)&&(CMLib.flags().isGettable(I2))&&(I2.container()==null))
236                                                itemsToMove.add(I2);
237                                }
238                        }
239                }
240                found=null;
241                if(!super.invoke(mob,commands,givenTarget,auto,asLevel))
242                        return false;
243                int duration=getDuration(45,mob,1,10);
244                amount=I.phyStats().weight();
245                messedUp=!proficiencyCheck(mob,0,auto);
246                found=CMLib.materials().makeItemResource(I.material());
247                playSound="ripping.wav";
248                final CMMsg msg=CMClass.getMsg(mob,I,this,getActivityMessageType(),L("<S-NAME> start(s) salvaging @x1.",I.name()));
249                if(mob.location().okMessage(mob,msg))
250                {
251                        mob.location().send(mob,msg);
252                        for(Item I2 : itemsToMove)
253                                mob.location().moveItemTo(I2);
254                        I.destroy();
255                        mob.location().recoverPhyStats();
256                        duration += CMath.sqrt(totalWeight/5);
257                        beneficialAffect(mob,mob,asLevel,duration);
258                }
259                return true;
260        }
