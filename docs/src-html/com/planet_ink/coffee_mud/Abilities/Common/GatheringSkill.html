080        protected static int fixResourceRequirement(final int resource, int amt)
081        {
082                if(amt<=0)
083                        return amt;
084                switch(resource)
085                {
086                case RawMaterial.RESOURCE_MITHRIL:
087                        amt=amt/2;
088                        break;
089                case RawMaterial.RESOURCE_ADAMANTITE:
090                        amt=amt/3;
091                        break;
092                case RawMaterial.RESOURCE_BALSA:
093                        amt=amt/2;
094                        break;
095                case RawMaterial.RESOURCE_IRONWOOD:
096                        amt=amt*2;
097                        break;
098                }
099                if(amt<=0)
100                        amt=1;
101                return amt;
102        }

104        public List<Integer> myResources()
105        {
106                if(supportedResources.containsKey(ID()))
107                        return supportedResources.get(ID());
108                String mask=supportedResourceString();
109                final List<Integer> maskV=new Vector<Integer>();
110                String str=mask;
111                while(mask.length()>0)
112                {
113                        str=mask;
114                        final int x=mask.indexOf('|');
115                        if(x>=0)
116                        {
117                                str=mask.substring(0,x);
118                                mask=mask.substring(x+1);
119                        }
120                        else
121                                mask="";
122                        if(str.length()>0)
123                        {
124                                boolean found=false;
125                                if(str.startsWith("_"))
126                                {
127                                        final int rsc=RawMaterial.CODES.FIND_IgnoreCase(str.substring(1));
128                                        if(rsc>=0)
129                                        {
130                                                maskV.add(Integer.valueOf(rsc));
131                                                found=true;
132                                        }
133                                }
134                                if(!found)
135                                {
136                                        final List<Integer> notResources=new ArrayList<Integer>();
137                                        final int y=str.indexOf('-');
138                                        if(y>0)
139                                        {
140                                                final List<String> restV=CMParms.parseAny(str.substring(y+1),"-",true);
141                                                str=str.substring(0,y);
142                                                for(final String sv : restV)
143                                                {
144                                                        final int code=RawMaterial.CODES.FIND_CaseSensitive(sv);
145                                                        if(code >= 0)
146                                                                notResources.add(Integer.valueOf(code));
147                                                }
148                                        }
149                                        final RawMaterial.Material m=RawMaterial.Material.findIgnoreCase(str);
150                                        if(m!=null)
151                                        {
152                                                final List<Integer> rscs=new XVector<Integer>(RawMaterial.CODES.COMPOSE_RESOURCES(m.mask()));
153                                                maskV.addAll(rscs);
154                                                maskV.removeAll(notResources);
155                                                found=rscs.size()>0;
156                                        }
157                                }
158                                if(!found)
159                                {
160                                        final int rsc=RawMaterial.CODES.FIND_IgnoreCase(str);
161                                        if(rsc>=0)
162                                                maskV.add(Integer.valueOf(rsc));
163                                }
164                        }
165                }
166                supportedResources.put(ID(),maskV);
167                return maskV;
168        }

170        public boolean bundle(final MOB mob, final List<String> what)
171        {
172                if((what.size()<3)
173                ||((!CMath.isNumber(what.get(1)))&&(!what.get(1).equalsIgnoreCase("ALL"))))
174                {
175                        commonTell(mob,L("You must specify an amount to bundle, followed by what resource to bundle."));
176                        return false;
177                }
178                int amount=CMath.s_int(what.get(1));
179                if(what.get(1).equalsIgnoreCase("ALL"))
180                        amount=Integer.MAX_VALUE;
181                if(amount<=0)
182                {
183                        commonTell(mob,L("@x1 is not an appropriate amount.",""+amount));
184                        return false;
185                }
186                int numHere=0;
187                final Room R=mob.location();
188                if(R==null)
189                        return false;
190                String name=CMParms.combine(what,2);
191                int foundResource=-1;
192                String foundSubType=null;
193                String foundSecret=null;
194                Item foundAnyway=null;
195                final List<RawMaterial> allFound=new ArrayList<RawMaterial>();
196                final List<Integer> maskV=myResources();
197                final Hashtable<String,Ability> foundAblesH=new Hashtable<String,Ability>();
198                Ability A=null;
199                long lowestNonZeroFoodNumber=Long.MAX_VALUE;
200                int count=name.lastIndexOf('.');
201                if(count > 0)
202                {
203                        final int x=count;
204                        count=CMath.s_int(name.substring(count+1))-1;
205                        if(count>=0)
206                                name=name.substring(0, x);
207                }
208                for(int i=0;i<R.numItems();i++)
209                {
210                        final Item I=R.getItem(i);
211                        if(CMLib.english().containsString(I.Name(),name)
212                        &&(count--<=0))
213                        {
214                                if(foundAnyway==null)
215                                        foundAnyway=I;
216                                if((I instanceof RawMaterial)
217                                &&(!CMLib.flags().isOnFire(I))
218                                &&(!CMLib.flags().isEnchanted(I))
219                                &&(I.container()==null)
220                                &&((foundSubType==null)
221                                        ||(((RawMaterial)I).getSubType().equals(foundSubType)))
222                                &&((foundSecret==null)
223                                        ||(((RawMaterial)I).rawSecretIdentity().equals(foundSecret)))
224                                &&((I.material()==foundResource)
225                                        ||((foundResource<0)&&maskV.contains(Integer.valueOf(I.material())))))
226                                {
227                                        if((I instanceof Decayable)
228                                        &&(((Decayable)I).decayTime()>0)
229                                        &&(((Decayable)I).decayTime()<lowestNonZeroFoodNumber))
230                                                lowestNonZeroFoodNumber=((Decayable)I).decayTime();
231                                        for(final Enumeration<Ability> a=I.effects();a.hasMoreElements();)
232                                        {
233                                                A=a.nextElement();
234                                                if((A!=null)
235                                                &&(!A.canBeUninvoked())
236                                                &&(!foundAblesH.containsKey(A.ID())))
237                                                        foundAblesH.put(A.ID(),A);
238                                        }
239                                        foundResource=I.material();
240                                        numHere+=I.phyStats().weight();
241                                        foundSubType=((RawMaterial)I).getSubType();
242                                        foundSecret=I.rawSecretIdentity();
243                                        allFound.add((RawMaterial)I);
244                                }
245                        }
246                }
247                if((numHere==0)||(foundResource<0))
248                {
249                        if(foundAnyway!=null)
250                                commonTell(mob,L("You can't bundle @x1 with this skill.",foundAnyway.name()));
251                        else
252                                commonTell(mob,L("You don't see any @x1 on the ground here.",name));
253                        return false;
254                }
255                if(amount==Integer.MAX_VALUE)
256                        amount=numHere;
257                if(numHere<amount)
258                {
259                        commonTell(mob,L("You only see @x1 pounds of @x2 on the ground here.",""+numHere,name));
260                        return false;
261                }
262                if(allFound.size()==1)
263                {
264                        commonTell(mob,L("It appears that @x1 is already bundled as much as it can be.",allFound.get(0).Name()));
265                        return false;
266                }
267                if(lowestNonZeroFoodNumber==Long.MAX_VALUE)
268                        lowestNonZeroFoodNumber=0;
269                final Item I=(Item)CMLib.materials().makeResource(foundResource,Integer.toString(mob.location().domainType()),true,foundSecret,foundSubType);
270                if(I==null)
271                {
272                        commonTell(mob,L("You could not bundle @x1 due to @x2 being an invalid resource code.  Bug it!",name,""+foundResource));
273                        return false;
274                }
275                I.basePhyStats().setWeight(amount);
276                CMLib.materials().adjustResourceName(I);
277                I.setDisplayText(L("@x1 is here.",I.name()));
278                if(R.show(mob,null,I,getActivityMessageType(),L("<S-NAME> create(s) <O-NAME>.")))
279                {
280                        int amountToGo=amount;
281                        for(final RawMaterial I2 : allFound)
282                        {
283                                if(I2.basePhyStats().weight()<amountToGo)
284                                {
285                                        amountToGo-= I2.basePhyStats().weight();
286                                        I2.destroy();
287                                }
288                                else
289                                {
290                                        I2.basePhyStats().setWeight(I2.basePhyStats().weight()-amountToGo);
291                                        I2.recoverPhyStats();
292                                        amountToGo=0;
293                                        CMLib.materials().adjustResourceName(I2);
294                                        break;
295                                }
296                        }
297                        if((!I.amDestroyed())&&(!R.isContent(I)))
298                                R.addItem(I,ItemPossessor.Expire.Player_Drop);
299                }
300                if(I instanceof Decayable)
301                        ((Decayable)I).setDecayTime(lowestNonZeroFoodNumber);
302                for(final Enumeration<String> e=foundAblesH.keys();e.hasMoreElements();)
303                        I.addNonUninvokableEffect((Ability)((Environmental)foundAblesH.get(e.nextElement())).copyOf());
304                R.recoverRoomStats();
305                return true;
306        }
