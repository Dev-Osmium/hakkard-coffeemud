077        @Override
078        public boolean tick(final Tickable ticking, final int tickID)
079        {
080                if((affected!=null)&&(affected instanceof MOB)&&(tickID==Tickable.TICKID_MOB))
081                {
082                        if((buildingI==null)
083                        ||(theSpell==null))
084                        {
085                                aborted=true;
086                                unInvoke();
087                        }
088                }
089                return super.tick(ticking,tickID);
090        }

104        @Override
105        public void unInvoke()
106        {
107                if(canBeUninvoked())
108                {
109                        if(affected instanceof MOB)
110                        {
111                                final MOB mob=(MOB)affected;
112                                if((buildingI!=null)&&(!aborted))
113                                {
114                                        if(messedUp)
115                                                commonTell(mob,L("You got writer`s block! Your dissertation on @x1 fails!",buildingI.name()));
116                                        else
117                                        {
118                                                int theSpellLevel=spellLevel(mob,theSpell);
119                                                if(fromTheScroll != null)
120                                                        eraseFromScrollItem(fromTheScroll,theSpell,theSpellLevel);
121                                                final Item oldBuildingI=buildingI;
122                                                buildingI=buildScrollItem(buildingI, theSpell, theSpellLevel);
123                                                if(buildingI.secretIdentity().length()==0)
124                                                        setBrand(mob, buildingI);
125                                                if((buildingI != null) && (oldBuildingI != buildingI) && (oldBuildingI.owner()!=null))
126                                                {
127                                                        oldBuildingI.owner().addItem(buildingI);
128                                                        oldBuildingI.destroy();
129                                                }
130                                                final Room R=mob.location();
131                                                if(R!=null)
132                                                        R.send(mob, CMClass.getMsg(mob,buildingI,this,CMMsg.MSG_WROTE, null, CMMsg.MSG_WROTE, theSpell.ID(),-1,null));
133
134                                        }
135                                }
136                                buildingI=null;
137                        }
138                }
139                super.unInvoke();
140        }

142        protected void eraseFromScrollItem(Scroll buildingI, Ability theSpell, int level)
143        {
144                if(buildingI == null)
145                        return;
146                StringBuilder newList=new StringBuilder();
147                buildingI.setBaseValue(buildingI.baseGoldValue() - (100*CMLib.ableMapper().lowestQualifyingLevel(theSpell.ID())));
148                if(buildingI.baseGoldValue()<=0)
149                        buildingI.setBaseValue(1);
150                for(Ability A : buildingI.getSpells())
151                {
152                        if(!A.ID().equalsIgnoreCase(theSpell.ID()))
153                                newList.append(A.ID()).append(";");
154                }
155                buildingI.setSpellList(newList.toString());
156                this.setName(buildingI);
157                if(buildingI.usesRemaining()>1)
158                        buildingI.setUsesRemaining(buildingI.usesRemaining()-1);
159                buildingI.text();
160        }

162        protected int spellLevel(MOB mob, Ability A)
163        {
164                int lvl=CMLib.ableMapper().qualifyingLevel(mob,A);
165                if(lvl<0)
166                        lvl=CMLib.ableMapper().lowestQualifyingLevel(A.ID());
167                switch(lvl)
168                {
169                case 0:
170                        return lvl;
171                case 1:
172                        return lvl;
173                case 2:
174                        return lvl + 1;
175                case 3:
176                        return lvl + 1;
177                case 4:
178                        return lvl + 2;
179                case 5:
180                        return lvl + 2;
181                case 6:
182                        return lvl + 3;
183                case 7:
184                        return lvl + 3;
185                case 8:
186                        return lvl + 4;
187                case 9:
188                        return lvl + 4;
189                default:
190                        return lvl + 5;
191                }
192        }

200        protected void setName(Scroll buildingI)
201        {
202                int x=buildingI.Name().indexOf(L("on the study of"));
203                if(x>0)
204                {
205                        buildingI.setName(buildingI.Name().substring(0,x).trim());
206                        buildingI.setDisplayText(L("@x1 sits here.",buildingI.Name()));
207                        buildingI.setDescription("");
208                }
209                if(buildingI.getSpells().size()>0)
210                {
211                        Ability theSpell=buildingI.getSpells().get(0);
212                        buildingI.setName(L("@x1 on the study of @x2",buildingI.Name(),theSpell.Name()));
213                        buildingI.setDisplayText(L("@x1 sits here.",buildingI.Name()));
214                        String x1="";
215                        String x2="";
216                        switch(CMLib.dice().roll(1, 4, 0))
217                        {
218                        case 1:
219                                x1=L("short");
220                                break;
221                        case 2:
222                                x1=L("lengthy");
223                                break;
224                        case 3:
225                                x1=L("wordy");
226                                break;
227                        case 4:
228                                x1=L("verbose");
229                                break;
230                        }
231                        switch(CMLib.dice().roll(1, 4, 0))
232                        {
233                        case 1:
234                                x2=L("practical");
235                                break;
236                        case 2:
237                                x2=L("theoretical");
238                                break;
239                        case 3:
240                                x2=L("advanced");
241                                break;
242                        case 4:
243                                x2=L("standard");
244                                break;
245                        }
246                        buildingI.setDescription(L("a @x1 thesis on the @x2 application of @x3",x1,x2,theSpell.Name()));
247                }
248        }

250        protected Scroll buildScrollItem(Item oldBuildingI, Ability theSpell, int level)
251        {
252                Scroll buildingI=(Scroll)CMClass.getItem("GenDissertation");
253                StringBuilder newList=new StringBuilder(theSpell.ID());
254                buildingI.setSpellList(newList.toString());
255                setName(buildingI);
256                if(buildingI.basePhyStats().level() < level)
257                {
258                        buildingI.basePhyStats().setLevel(level);
259                        buildingI.phyStats().setLevel(level);
260                        buildingI.recoverPhyStats();
261                }
262                buildingI.setBaseValue(buildingI.baseGoldValue() + (100*CMLib.ableMapper().lowestQualifyingLevel(theSpell.ID())));
263                buildingI.setDescription("");
264                buildingI.setUsesRemaining(1);
265                buildingI.text();
266                return buildingI;
267        }

275        @Override
276        protected boolean autoGenInvoke(final MOB mob, List<String> commands, Physical givenTarget, final boolean auto, 
277                                                                 final int asLevel, int autoGenerate, boolean forceLevels, List<Item> crafted)
278        {
279                if(super.checkStop(mob, commands))
280                        return true;
281
282                if(autoGenerate>0)
283                {
284                        final Ability theSpell=mob.fetchRandomAbility();
285                        if(theSpell==null) 
286                                return false;
287                        final int level=spellLevel(mob,theSpell);
288                        buildingI=buildScrollItem(null, theSpell, level);
289                        crafted.add(buildingI);
290                        return true;
291                }
292                randomRecipeFix(mob,addRecipes(mob,loadRecipes()),commands,0);
293                if(commands.size()<1)
294                {
295                        commonEmote(mob,L("You must specify what skill to write about, and the paper to write the dissertation on."));
296                        return false;
297                }
298                final String pos=commands.get(commands.size()-1);
299                if((!auto)&&(commands.size()<2))
300                {
301                        commonEmote(mob,L("You must specify what skill to write about, and the paper to write the dissertation on."));
302                        return false;
303                }
304                else
305                {
306                        buildingI=getTarget(mob,null,givenTarget,CMParms.parse(pos),Wearable.FILTER_UNWORNONLY);
307                        commands.remove(pos);
308                        if(buildingI==null)
309                                return false;
310                        if(!mob.isMine(buildingI))
311                        {
312                                commonTell(mob,L("You'll need to pick that up first."));
313                                return false;
314                        }
315                        if((((buildingI.material()&RawMaterial.MATERIAL_MASK)!=RawMaterial.MATERIAL_PAPER))
316                        &&(buildingI.material()!=RawMaterial.RESOURCE_HIDE)
317                        &&(buildingI.material()!=RawMaterial.RESOURCE_HEMP)
318                        &&(buildingI.material() != RawMaterial.RESOURCE_SILK))
319                        {
320                                commonTell(mob,L("@x1 isn't even made of paper or silk!",buildingI.name(mob)));
321                                return false;
322                        }
323                        if(((buildingI instanceof MiscMagic))
324                        ||(buildingI instanceof Recipe)
325                        ||(!buildingI.isGeneric()))
326                        {
327                                commonTell(mob,L("There's can't write a dissertation on @x1!",buildingI.name(mob)));
328                                return false;
329                        }
330                        if(buildingI instanceof Scroll)
331                        {
332                                if(((Scroll)buildingI).getSpells().size()>0)
333                                {
334                                        commonTell(mob,L("You can only write on blank scrolls."));
335                                        return false;
336                                }
337                        }
338                        else
339                        if(buildingI.readableText().length()>0)
340                        {
341                                commonTell(mob,L("You can only write on blank paper."));
342                                return false;
343                        }
344                        String recipeName=CMParms.combine(commands,0);
345                        theSpell=null;
346                        fromTheScroll=null;
347                        String ingredient="";
348                        {
349                                Ability A=(Ability)CMLib.english().fetchEnvironmental(mob.abilities(), recipeName, true);
350                                if(A==null)
351                                        A=(Ability)CMLib.english().fetchEnvironmental(mob.abilities(), recipeName, false);
352                                if((A!=null)
353                                &&(A.name().equalsIgnoreCase(recipeName)))
354                                {
355                                        if(xlevel(mob)>=spellLevel(mob,A))
356                                                theSpell=A;
357                                        else
358                                        {
359                                                commonTell(mob,L("You aren't ready to write a dissertation on '@x1' yet.",recipeName));
360                                                return false;
361                                        }
362                                }
363                        }
364                        int manaToLose=10;
365                        int experienceToLose=0;
366                        if(theSpell==null)
367                        {
368                                int x=CMParms.indexOfIgnoreCase(commands, "from");
369                                if((x>0)&&(x<commands.size()-1))
370                                {
371                                        recipeName=CMParms.combine(commands,0,x);
372                                        String otherScrollName=CMParms.combine(commands,x+1,commands.size());
373                                        Item scrollFromI=getTarget(mob,null,givenTarget,CMParms.parse(otherScrollName),Wearable.FILTER_UNWORNONLY);
374                                        if(scrollFromI==null)
375                                                return false;
376                                        if(!mob.isMine(scrollFromI))
377                                        {
378                                                commonTell(mob,L("You'll need to pick that up first."));
379                                                return false;
380                                        }
381                                        if((!(scrollFromI instanceof Scroll))
382                                        ||(scrollFromI instanceof MiscMagic))
383                                        {
384                                                commonTell(mob,L("@x1 is not a scroll!",scrollFromI.name(mob)));
385                                                return false;
386                                        }
387                                        if((!(scrollFromI instanceof Scroll))
388                                        ||(((Scroll)scrollFromI).getSpells().size()==0))
389                                        {
390                                                commonTell(mob,L("@x1 has nothing on it!",scrollFromI.name(mob)));
391                                                return false;
392                                        }
393                                        ingredient="";
394                                        for(Ability A : ((Scroll)scrollFromI).getSpells())
395                                        {
396                                                if((A!=null)
397                                                &&(A.name().equalsIgnoreCase(recipeName)))
398                                                {
399                                                        if(xlevel(mob)>=spellLevel(mob,A))
400                                                                theSpell=A;
401                                                        else
402                                                        {
403                                                                commonTell(mob,L("You aren't ready to write a dissertation on '@x1' yet.",recipeName));
404                                                                return false;
405                                                        }
406                                                }
407                                        }
408                                        if(theSpell==null)
409                                        {
410                                                commonTell(mob,L("You can't copy the dissertation on '@x1' from the scroll @x2!",recipeName,scrollFromI.name(mob)));
411                                                return false;
412                                        }
413                                        fromTheScroll=(Scroll)scrollFromI;
414                                }
415                                else
416                                if(theSpell==null)
417                                {
418                                        commonTell(mob,L("You don't know how to write a dissertation on '@x1'.  Try \"SKILLS\" for a list.",recipeName));
419                                        return false;
420                                }
421                                manaToLose+=spellLevel(mob,theSpell)*10;
422                        }
423                        else
424                        {
425                                manaToLose+=CMLib.ableMapper().qualifyingLevel(mob,theSpell)*10;
426                                manaToLose-=CMLib.ableMapper().qualifyingClassLevel(mob,theSpell)*5;
427                                experienceToLose+=10+CMLib.ableMapper().qualifyingLevel(mob,theSpell);
428                                experienceToLose-=CMLib.ableMapper().qualifyingClassLevel(mob,theSpell);
429                                if(experienceToLose < CMLib.ableMapper().qualifyingLevel(mob,theSpell))
430                                        experienceToLose = CMLib.ableMapper().qualifyingLevel(mob,theSpell);
431                        }
432                        
433                        final int resourceType=(ingredient.length()==0) ? -1 : RawMaterial.CODES.FIND_IgnoreCase(ingredient);
434
435                        int[][] data = null;
436                        if(resourceType>0)
437                        {
438                                final int[] pm={resourceType};
439                                data=fetchFoundResourceData(mob,
440                                                                                        1,ingredient,pm,
441                                                                                        0,null,null,
442                                                                                        bundling,
443                                                                                        -1,
444                                                                                        null);
445                                if(data==null) 
446                                        return false;
447                        }
448                        if(manaToLose<10) 
449                                manaToLose=10;
450
451                        if(mob.curState().getMana()<manaToLose)
452                        {
453                                commonTell(mob,L("You need at least @x1 mana to accomplish that.",""+manaToLose));
454                        }
455                        
456                        if(!super.invoke(mob,commands,givenTarget,auto,asLevel))
457                                return false;
458                        
459                        mob.curState().adjMana(-manaToLose, mob.maxState());
460
461                        if((resourceType>0)&&(data != null))
462                                CMLib.materials().destroyResourcesValue(mob.location(),data[0][FOUND_AMT],data[0][FOUND_CODE],0,null);
463
464                        playSound=null;
465                        if(experienceToLose > 0)
466                        {
467                                experienceToLose=getXPCOSTAdjustment(mob,experienceToLose);
468                                CMLib.leveler().postExperience(mob,null,null,-experienceToLose,false);
469                                commonTell(mob,L("You lose @x1 experience points for the effort.",""+experienceToLose));
470                        }
471                        
472                        int duration=getDuration(100+(CMLib.ableMapper().qualifyingLevel(mob,theSpell)*10),mob,CMLib.ableMapper().lowestQualifyingLevel(theSpell.ID()),10);
473                        if(duration<10)
474                                duration=10;
475                        messedUp=!proficiencyCheck(mob,0,auto);
476
477                        String msgStr;
478                        if(fromTheScroll != null)
479                        {
480                                msgStr=L("<S-NAME> start(s) copying a dissertation on @x1 from @x2 to @x3.",theSpell.name(),fromTheScroll.name(),buildingI.name());
481                                displayText=L("You are copying a dissertation on @x1 from @x2 to @x3",theSpell.name(),fromTheScroll.name(),buildingI.name());
482                                verb=L("copying a dissertation on @x1 from @x2 to @x3",theSpell.name(),fromTheScroll.name(),buildingI.name());
483                        }
484                        else
485                        {
486                                msgStr=L("<S-NAME> start(s) writing a dissertation on @x1 onto @x2.",theSpell.name(),buildingI.name());
487                                displayText=L("You are writing a dissertation on @x1 onto @x2",theSpell.name(),buildingI.name());
488                                verb=L("writing a dissertation on @x1 onto @x2",theSpell.name(),buildingI.name());
489                        }
490                        final CMMsg msg=CMClass.getMsg(mob,buildingI,this,getActivityMessageType(),msgStr);
491                        if(mob.location().okMessage(mob,msg))
492                        {
493                                mob.location().send(mob,msg);
494                                buildingI=(Item)msg.target();
495                                beneficialAffect(mob,mob,asLevel,duration);
496                        }
497                }
498                return true;
499        }
