035public class PlantLore extends CommonSkill
036{
037        @Override
038        public String ID()
039        {
040                return "PlantLore";
041        }
042
043        private final static String     localizedName   = CMLib.lang().L("Plant Lore");
044
045        @Override
046        public String name()
047        {
048                return localizedName;
049        }
050
051        private static final String[]   triggerStrings  = I(new String[] { "PLANTLORE", "PSPECULATE" });
052
053        @Override
054        public String[] triggerStrings()
055        {
056                return triggerStrings;
057        }
058
059        @Override
060        public int classificationCode()
061        {
062                return Ability.ACODE_COMMON_SKILL | Ability.DOMAIN_NATURELORE;
063        }
064
065        protected boolean success=false;
066        public PlantLore()
067        {
068                super();
069                displayText=L("You are observing plant growth...");
070                verb=L("observing plant growths");
071        }
072
073        @Override
074        public boolean tick(final Tickable ticking, final int tickID)
075        {
076                if((affected!=null)&&(affected instanceof MOB)&&(tickID==Tickable.TICKID_MOB))
077                {
078                        final MOB mob=(MOB)affected;
079                        if(tickUp==6)
080                        {
081                                if(success==false)
082                                {
083                                        final StringBuffer str=new StringBuffer(L("Your growth observation attempt failed.\n\r"));
084                                        commonTell(mob,str.toString());
085                                        unInvoke();
086                                }
087                        }
088                }
089                return super.tick(ticking,tickID);
090        }
091
092        @Override
093        public void unInvoke()
094        {
095                if(canBeUninvoked())
096                {
097                        if((affected!=null)&&(affected instanceof MOB)&&(!helping))
098                        {
099                                final MOB mob=(MOB)affected;
100                                final Room room=mob.location();
101                                if((success)&&(!aborted)&&(room!=null))
102                                {
103                                        if((room.domainType()&Room.INDOORS)==0)
104                                        {
105                                                final StringBuffer str=new StringBuffer("");
106                                                final Vector<Room> V=new Vector<Room>();
107                                                TrackingLibrary.TrackingFlags flags;
108                                                flags = CMLib.tracking().newFlags()
109                                                                .plus(TrackingLibrary.TrackingFlag.OPENONLY)
110                                                                .plus(TrackingLibrary.TrackingFlag.AREAONLY)
111                                                                .plus(TrackingLibrary.TrackingFlag.NOAIR);
112                                                int range=2+(getXLEVELLevel(mob)/2)+super.getXMAXRANGELevel(mob);
113                                                CMLib.tracking().getRadiantRooms(room,V,flags,null,range,null);
114                                                for(int v=0;v<V.size();v++)
115                                                {
116                                                        final Room R=V.elementAt(v);
117                                                        final int material=R.myResource()&RawMaterial.MATERIAL_MASK;
118                                                        final int resource=R.myResource()&RawMaterial.RESOURCE_MASK;
119                                                        if(!RawMaterial.CODES.IS_VALID(resource))
120                                                                continue;
121                                                        if((material!=RawMaterial.MATERIAL_VEGETATION)
122                                                        &&(resource!=RawMaterial.RESOURCE_COTTON)
123                                                        &&(resource!=RawMaterial.RESOURCE_HEMP)
124                                                        &&(resource!=RawMaterial.RESOURCE_SAP)
125                                                        &&(material!=RawMaterial.MATERIAL_WOODEN))
126                                                                continue;
127                                                        final String resourceStr=RawMaterial.CODES.NAME(resource);
128                                                        if(R==room)
129                                                                str.append(L("You think this spot would be good for @x1.\n\r",resourceStr.toLowerCase()));
130                                                        else
131                                                        {
132                                                                int isAdjacent=-1;
133                                                                for(int d=Directions.NUM_DIRECTIONS()-1;d>=0;d--)
134                                                                {
135                                                                        final Room room2=room.getRoomInDir(d);
136                                                                        if(room2==R)
137                                                                                isAdjacent=d;
138                                                                }
139                                                                if(isAdjacent>=0)
140                                                                        str.append(L("There looks like @x1 @x2.\n\r",resourceStr.toLowerCase(),CMLib.directions().getInDirectionName(isAdjacent)));
141                                                                else
142                                                                {
143                                                                        int d=CMLib.tracking().radiatesFromDir(R,V);
144                                                                        if(d>=0)
145                                                                        {
146                                                                                d=Directions.getOpDirectionCode(d);
147                                                                                str.append(L("There looks like @x1 far @x2.\n\r",resourceStr.toLowerCase(),CMLib.directions().getInDirectionName(d)));
148                                                                        }
149                                                                }
150
151                                                        }
152                                                }
153                                                commonTell(mob,str.toString());
154                                        }
155                                        else
156                                                commonTell(mob,L("You don't find any good plant life around here."));
157                                }
158                        }
159                }
160                super.unInvoke();
161        }
162
163        @Override
164        public boolean invoke(MOB mob, List<String> commands, Physical givenTarget, boolean auto, int asLevel)
165        {
166                if(super.checkStop(mob, commands))
167                        return true;
168                verb=L("observing plant growth");
169                success=false;
170                if(!super.invoke(mob,commands,givenTarget,auto,asLevel))
171                        return false;
172                if(proficiencyCheck(mob,0,auto))
173                        success=true;
174                final int duration=getDuration(45,mob,1,5);
175                final CMMsg msg=CMClass.getMsg(mob,null,this,getActivityMessageType(),L("<S-NAME> start(s) observing the growth in this area."));
176                if(mob.location().okMessage(mob,msg))
177                {
178                        mob.location().send(mob,msg);
179                        beneficialAffect(mob,mob,asLevel,duration);
180                }
181                return true;
182        }
183}
