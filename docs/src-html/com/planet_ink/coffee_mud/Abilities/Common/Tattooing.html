036public class Tattooing extends CommonSkill
037{
038        @Override
039        public String ID()
040        {
041                return "Tattooing";
042        }
043
044        private final static String     localizedName   = CMLib.lang().L("Tattooing");
045
046        @Override
047        public String name()
048        {
049                return localizedName;
050        }
051
052        private static final String[]   triggerStrings  = I(new String[] { "TATTOO", "TATTOOING" });
053
054        @Override
055        public int classificationCode()
056        {
057                return Ability.ACODE_COMMON_SKILL | Ability.DOMAIN_ARTISTIC;
058        }
059
060        @Override
061        public String[] triggerStrings()
062        {
063                return triggerStrings;
064        }
065
066        @Override
067        public long flags()
068        {
069                return super.flags() | Ability.FLAG_TORTURING;
070        }
071
072        protected String        writing = "";
073        protected MOB           target  = null;
074        protected int           oldHP   = 1;
075        protected String        bodyPart= "";
076
077        public Tattooing()
078        {
079                super();
080                displayText=L("You are tattooing...");
081                verb=L("tattooing");
082        }
083
084        @Override
085        public void unInvoke()
086        {
087                if(canBeUninvoked())
088                {
089                        if((affected!=null)
090                        &&(affected instanceof MOB)
091                        &&(!aborted)
092                        &&(!helping)
093                        &&(target!=null))
094                        {
095                                final MOB mob=(MOB)affected;
096                                if(writing.length()==0)
097                                        commonEmote(mob,L("<S-NAME> mess(es) up the tattoo on @x1.",target.name()));
098                                else
099                                {
100                                        commonEmote(mob,L("<S-NAME> complete(s) the tattoo on @x1.",target.name()));
101                                        target.addTattoo(writing);
102                                }
103                                if((bodyPart!=null)&&(bodyPart.length()>0))
104                                {
105                                        Ability injuryA=CMClass.getAbility("Injury");
106                                        if(injuryA!=null)
107                                        {
108                                                injuryA.invoke(mob,new XVector<String>(),target,true,0);
109                                                injuryA=target.fetchEffect("Injury");
110                                                if(injuryA!=null)
111                                                {
112                                                        ((LimbDamage)injuryA).damageLimb(bodyPart);
113                                                }
114                                        }
115                                }
116                        }
117                }
118                super.unInvoke();
119        }
120
121        @Override
122        public boolean tick(final Tickable ticking, final int tickID)
123        {
124                if((affected!=null)&&(affected instanceof MOB)&&(tickID==Tickable.TICKID_MOB))
125                {
126                        final MOB mob=(MOB)affected;
127                        if((target==null)
128                        ||(mob.location()!=target.location())
129                        ||(!CMLib.flags().canBeSeenBy(target,mob)))
130                        {
131                                aborted = true;
132                                unInvoke();
133                                return false;
134                        }
135                        else
136                        if(target!=null)
137                                target.curState().setHitPoints(oldHP);
138                }
139                return super.tick(ticking,tickID);
140        }
141
142        @Override
143        public boolean invoke(final MOB mob, final List<String> commands, final Physical givenTarget, final boolean auto, final int asLevel)
144        {
145                if(commands.size()<3)
146                {
147                        commonTell(mob,L("You must specify whom you want to tattoo, what body part to tattoo, and what the tattoo looks like. Use 'REMOVE' as the description to remove a tattoo."));
148                        return false;
149                }
150                final String whom=commands.get(0);
151                commands.remove(0);
152                final String part=commands.get(0);
153                commands.remove(0);
154                final String message=CMParms.combine(commands,0);
155                commands.clear();
156                commands.add(whom);
157
158                int partNum=-1;
159                final StringBuffer allParts=new StringBuffer("");
160                final long[] tattoable={Wearable.WORN_ARMS,
161                                                                Wearable.WORN_LEGS,
162                                                                Wearable.WORN_HANDS,
163                                                                Wearable.WORN_HEAD,
164                                                                Wearable.WORN_FEET,
165                                                                Wearable.WORN_LEFT_WRIST,
166                                                                Wearable.WORN_RIGHT_WRIST,
167                                                                Wearable.WORN_NECK,
168                                                                Wearable.WORN_BACK,
169                                                                Wearable.WORN_TORSO};
170                final Wearable.CODES codes = Wearable.CODES.instance();
171                for(int i=0;i<codes.total();i++)
172                {
173                        for (final long element : tattoable)
174                        {
175                                if(codes.get(i)==element)
176                                {
177                                        if(codes.name(i).equalsIgnoreCase(part))
178                                                partNum=i;
179                                        allParts.append(", "+CMStrings.capitalizeAndLower(codes.name(i).toLowerCase()));
180                                        break;
181                                }
182                        }
183                }
184                if(partNum<0)
185                {
186                        commonTell(mob,L("'@x1' is not a valid location.  Valid locations include: @x2",part,allParts.toString().substring(2)));
187                        return false;
188                }
189                final long wornCode=codes.get(partNum);
190                final String wornName=codes.name(partNum);
191
192                final MOB target=super.getTarget(mob,commands,givenTarget);
193                if(target==null)
194                        return false;
195
196                if((!auto)&&(CMLib.flags().isSleeping(target)))
197                        CMLib.commands().postStand(target, true, true);
198
199                if((!mob.getGroupMembers(new HashSet<MOB>()).contains(target))
200                &&(mob!=target)
201                &&(!CMLib.flags().isBoundOrHeld(target))
202                &&(!CMLib.flags().isSleeping(target)))
203                {
204                        commonTell(mob,target,null,L("<T-NAME> seem(s) unwilling to cooperate."));
205                        return false;
206                }
207
208                if(target.getWearPositions(wornCode)<=0)
209                {
210                        commonTell(mob,L("That location is not available for tattooing."));
211                        return false;
212                }
213                if(target.freeWearPositions(wornCode,(short)(Short.MIN_VALUE+1),(short)0)<=0)
214                {
215                        commonTell(mob,L("That location is currently covered by something."));
216                        return false;
217                }
218                if(target.curState().getHitPoints() < target.maxState().getHitPoints())
219                {
220                        commonTell(mob,L("You need to wait until @x1 is at full health.",target.name(mob)));
221                        return false;
222                }
223
224                int numTattsDone=0;
225                Tattoo tatToRemove=null;
226                for(final Enumeration<Tattoo> e=target.tattoos();e.hasMoreElements();)
227                {
228                        final Tattoo T=e.nextElement();
229                        if(T.getTattooName().startsWith(wornName.toUpperCase()+":"))
230                        {
231                                numTattsDone++;
232                                if(T.getTattooName().substring(wornName.length()+1).toUpperCase().startsWith("A TATTOO OF"))
233                                        tatToRemove=T;
234                        }
235                }
236                if("REMOVE".startsWith(message.toUpperCase()))
237                {
238                        if(tatToRemove==null)
239                        {
240                                commonTell(mob,L("There is no tattoo there to remove."));
241                                return false;
242                        }
243                }
244                else
245                if(numTattsDone>=target.getWearPositions(codes.get(partNum)))
246                {
247                        commonTell(mob,L("That location is already completely decorated."));
248                        return false;
249                }
250
251                if(!super.invoke(mob,commands,givenTarget,auto,asLevel))
252                        return false;
253                writing=wornName.toUpperCase()+":A tattoo of "+message;
254                verb=L("tattooing @x1 on the @x2",target.name(),wornName);
255                displayText=L("You are @x1",verb);
256                if(!proficiencyCheck(mob,0,auto))
257                        writing="";
258                final int duration=getDuration(35,mob,1,6);
259                String str=L("<S-NAME> start(s) tattooing @x1 on <T-YOUPOSS> @x2.",message,wornName.toLowerCase());
260                if("REMOVE".startsWith(message.toUpperCase()))
261                        str=L("<S-NAME> remove(s) the tattoo on <T-YOUPOSS> @x1.",wornName.toLowerCase());
262
263                final CMMsg msg=CMClass.getMsg(mob,target,this,getActivityMessageType(),str);
264                if(mob.location().okMessage(mob,msg))
265                {
266                        final int percentOff=target.maxState().getHitPoints()/8;
267                        if(target.curState().getHitPoints() > (percentOff*2))
268                        {
269                                CMLib.combat().postDamage(mob, target, this, percentOff, CMMsg.MASK_ALWAYS|CMMsg.TYP_JUSTICE, Weapon.TYPE_PIERCING, null);
270                                CMLib.combat().postDamage(mob, target, this, percentOff, CMMsg.MASK_ALWAYS|CMMsg.TYP_JUSTICE, Weapon.TYPE_PIERCING, null);
271                        }
272                        mob.location().send(mob,msg);
273                        if("REMOVE".startsWith(message.toUpperCase()))
274                                target.delTattoo(tatToRemove);
275                        else
276                        {
277                                //target.curState().adjHitPoints(-percentOff, target.maxState());
278                                final List<Integer> bodyPartNums = new ArrayList<Integer>();
279                                for(int i=0;i<Race.BODY_WEARVECTOR.length;i++)
280                                {
281                                        if((Race.BODY_WEARVECTOR[i] == wornCode)
282                                        &&(!bodyPartNums.contains(Integer.valueOf(i)))
283                                        &&(target.charStats().getBodyPart(i)>0))
284                                                bodyPartNums.add(Integer.valueOf(i));
285                                }
286                                String bodyPartName="";
287                                if(bodyPartNums.size()>0)
288                                {
289                                        final Integer pNum=bodyPartNums.get(CMLib.dice().roll(1, bodyPartNums.size(), -1));
290                                        bodyPartName=Race.BODYPARTSTR[pNum.intValue()].toLowerCase();
291                                }
292                                beneficialAffect(mob,mob,asLevel,duration);
293                                final Tattooing A=(Tattooing)mob.fetchEffect(ID());
294                                if(A!=null)
295                                {
296                                        A.target=target;
297                                        A.oldHP=target.curState().getHitPoints();
298                                        A.bodyPart=bodyPartName;
299                                }
300                        }
301                }
302                return true;
303        }
