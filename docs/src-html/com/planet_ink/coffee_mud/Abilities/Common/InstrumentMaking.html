125        @Override
126        public boolean mayICraft(final Item I)
127        {
128                if(I==null)
129                        return false;
130                if(!super.mayBeCrafted(I))
131                        return false;
132                if(CMLib.flags().isDeadlyOrMaliciousEffect(I))
133                        return false;
134                if(isANativeItem(I.Name()))
135                        return true;
136                if(I instanceof MusicalInstrument)
137                        return true;
138                return false;
139        }

141        @Override
142        public void unInvoke()
143        {
144                if(canBeUninvoked())
145                {
146                        if(affected instanceof MOB)
147                        {
148                                final MOB mob=(MOB)affected;
149                                if((buildingI!=null)&&(!aborted))
150                                {
151                                        if(messedUp)
152                                        {
153                                                if(activity == CraftingActivity.LEARNING)
154                                                        commonEmote(mob,L("<S-NAME> fail(s) to learn how to make @x1.",buildingI.name()));
155                                                else
156                                                        commonEmote(mob,L("<S-NAME> mess(es) up making @x1.",buildingI.name()));
157                                                buildingI.destroy();
158                                        }
159                                        else
160                                        if(activity==CraftingActivity.LEARNING)
161                                        {
162                                                deconstructRecipeInto(mob, buildingI, recipeHolder );
163                                                buildingI.destroy();
164                                        }
165                                        else
166                                        {
167                                                dropAWinner(mob,buildingI);
168                                                CMLib.achievements().possiblyBumpAchievement(mob, AchievementLibrary.Event.CRAFTING, 1, this);
169                                        }
170                                }
171                                buildingI=null;
172                        }
173                }
174                super.unInvoke();
175        }

189        @Override
190        protected boolean autoGenInvoke(final MOB mob, final List<String> commands, final Physical givenTarget, final boolean auto,
191                                                                 final int asLevel, final int autoGenerate, final boolean forceLevels, final List<Item> crafted)
192        {
193                if(super.checkStop(mob, commands))
194                        return true;
195
196                if(super.checkInfo(mob, commands))
197                        return true;
198
199                @SuppressWarnings("unused")
200                int recipeLevel=1;
201                randomRecipeFix(mob,addRecipes(mob,loadRecipes()),commands,autoGenerate);
202                if(commands.size()==0)
203                {
204                        commonTell(mob,L("Make what Instrument? Enter \"instrumentmake list\" for a list, \"instrumentmake info <item>\","
205                                                        + " \"instrumentmake learn <item>\" to gain recipes, or \"instrumentmake stop\" to cancel."));
206                        return false;
207                }
208                if((!auto)
209                &&(commands.size()>0)
210                &&((commands.get(0)).equalsIgnoreCase("bundle")))
211                {
212                        bundling=true;
213                        if(super.invoke(mob,commands,givenTarget,auto,asLevel))
214                                return super.bundle(mob,commands);
215                        return false;
216                }
217                final List<List<String>> recipes=addRecipes(mob,loadRecipes());
218                final String str=commands.get(0);
219                bundling=false;
220                String startStr=null;
221                int duration=4;
222                final boolean archon=CMSecurity.isASysOp(mob)||CMSecurity.isAllowed(mob,mob.location(),CMSecurity.SecFlag.ALLSKILLS);
223                if(str.equalsIgnoreCase("list"))
224                {
225                        String mask=CMParms.combine(commands,1);
226                        boolean allFlag=false;
227                        if(mask.equalsIgnoreCase("all"))
228                        {
229                                allFlag=true;
230                                mask="";
231                        }
232                        final int[] cols={
233                                CMLib.lister().fixColWidth(23,mob.session()),
234                                CMLib.lister().fixColWidth(3,mob.session()),
235                                CMLib.lister().fixColWidth(10,mob.session())
236                        };
237                        final StringBuffer buf=new StringBuffer(L("@x1 @x2 @x3 Material required\n\r",CMStrings.padRight(L("Item"),cols[0]),CMStrings.padRight(L("Lvl"),cols[1]),CMStrings.padRight(L("Type"),cols[2])));
238                        final List<List<String>> listRecipes=((mask.length()==0) || mask.equalsIgnoreCase("all")) ? recipes : super.matchingRecipeNames(recipes, mask, true);
239                        for(int r=0;r<listRecipes.size();r++)
240                        {
241                                final List<String> V=listRecipes.get(r);
242                                if(V.size()>0)
243                                {
244                                        final String item=replacePercent(V.get(RCP_FINALNAME),"");
245                                        final int level=CMath.s_int(V.get(RCP_LEVEL));
246                                        String type=V.get(RCP_MATERIAL);
247                                        final String wood=getComponentDescription(mob,V,RCP_WOOD);
248                                        if(wood.length()>5)
249                                                type="";
250                                        final String race=V.get(RCP_RACES).trim();
251                                        final String itype=CMStrings.capitalizeAndLower(V.get(RCP_TYPE).toLowerCase()).trim();
252                                        if(((level<=xlevel(mob))||allFlag)
253                                        &&((race.length()==0)||archon||((" "+race+" ").toUpperCase().indexOf(" "+mob.charStats().getMyRace().ID().toUpperCase()+" ")>=0)))
254                                                buf.append(CMStrings.padRight(item,cols[0])+" "+CMStrings.padRight(""+level,cols[1])+" "+CMStrings.padRight(itype,cols[2])+" "+wood+" "+type+"\n\r");
255                                }
256                        }
257                        commonTell(mob,buf.toString());
258                        return true;
259                }
260                else
261                if(((commands.get(0))).equalsIgnoreCase("learn"))
262                {
263                        return doLearnRecipe(mob, commands, givenTarget, auto, asLevel);
264                }
265                activity = CraftingActivity.CRAFTING;
266                buildingI=null;
267                int amount=-1;
268                if((commands.size()>1)&&(CMath.isNumber(commands.get(commands.size()-1))))
269                {
270                        amount=CMath.s_int(commands.get(commands.size()-1));
271                        commands.remove(commands.size()-1);
272                }
273                final String recipeName=CMParms.combine(commands,0);
274                List<String> foundRecipe=null;
275                final List<List<String>> matches=matchingRecipeNames(recipes,recipeName,true);
276                for(int r=0;r<matches.size();r++)
277                {
278                        final List<String> V=matches.get(r);
279                        if(V.size()>0)
280                        {
281                                final String race=V.get(RCP_RACES).trim();
282                                final int level=CMath.s_int(V.get(RCP_LEVEL));
283                                if(((autoGenerate>0)||(level<=xlevel(mob)))
284                                &&((autoGenerate>0)||(race.length()==0)||archon||((" "+race+" ").toUpperCase().indexOf(" "+mob.charStats().getMyRace().ID().toUpperCase()+" ")>=0)))
285                                {
286                                        foundRecipe=V;
287                                        recipeLevel=level;
288                                        break;
289                                }
290                        }
291                }
292                if(foundRecipe==null)
293                {
294                        commonTell(mob,L("You don't know how to make a '@x1'.  Try \"instrumentmake list\" for a list.",recipeName));
295                        return false;
296                }
297
298                final String woodRequiredStr = foundRecipe.get(RCP_WOOD);
299                final int[] compData = new int[CF_TOTAL];
300                final String realRecipeName=replacePercent(foundRecipe.get(RCP_FINALNAME),"");
301                final List<Object> componentsFoundList=getAbilityComponents(mob, woodRequiredStr, "make "+CMLib.english().startWithAorAn(realRecipeName),autoGenerate,compData,1);
302                if(componentsFoundList==null)
303                        return false;
304                int woodRequired=CMath.s_int(woodRequiredStr);
305                woodRequired=adjustWoodRequired(woodRequired,mob);
306
307                if(amount>woodRequired)
308                        woodRequired=amount;
309                final String materialRequired=foundRecipe.get(RCP_MATERIAL);
310                final String misctype=foundRecipe.get(RCP_MISCTYPE);
311                final int[] pm={RawMaterial.MATERIAL_METAL,RawMaterial.MATERIAL_MITHRIL};
312                if(!materialRequired.toUpperCase().startsWith("METAL"))
313                {
314                        pm[0]=RawMaterial.MATERIAL_WOODEN;
315                        pm[1]=RawMaterial.MATERIAL_WOODEN;
316                }
317                bundling=misctype.equalsIgnoreCase("BUNDLE");
318                final int[][] data=fetchFoundResourceData(mob,
319                                                                                                woodRequired,"material",pm,
320                                                                                                0,null,null,
321                                                                                                bundling,
322                                                                                                autoGenerate,
323                                                                                                null);
324                if(data==null)
325                        return false;
326                woodRequired=data[0][FOUND_AMT];
327                if(!super.invoke(mob,commands,givenTarget,auto,asLevel))
328                        return false;
329                final MaterialLibrary.DeadResourceRecord deadMats;
330                if((componentsFoundList.size() > 0)||(autoGenerate>0))
331                        deadMats = new MaterialLibrary.DeadResourceRecord();
332                else
333                        deadMats = CMLib.materials().destroyResources(mob.location(),woodRequired,data[0][FOUND_CODE],0,null,null);
334                final MaterialLibrary.DeadResourceRecord deadComps = CMLib.ableComponents().destroyAbilityComponents(componentsFoundList);
335                final int lostValue=autoGenerate>0?0:(deadMats.lostValue + deadComps.lostValue);
336                buildingI=CMClass.getItem(foundRecipe.get(RCP_CLASSTYPE));
337                if(buildingI==null)
338                {
339                        commonTell(mob,L("There's no such thing as a @x1!!!",foundRecipe.get(RCP_CLASSTYPE)));
340                        return false;
341                }
342                duration=getDuration(CMath.s_int(foundRecipe.get(RCP_TICKS)),mob,CMath.s_int(foundRecipe.get(RCP_LEVEL)),4);
343                buildingI.setMaterial(super.getBuildingMaterial(woodRequired, data, compData));
344                String itemName=determineFinalName(foundRecipe.get(RCP_FINALNAME),buildingI.material(),deadMats,deadComps);
345                if(bundling)
346                        itemName="a "+woodRequired+"# "+itemName;
347                else
348                        itemName=CMLib.english().startWithAorAn(itemName);
349                buildingI.setName(itemName);
350                startStr=L("<S-NAME> start(s) making @x1.",buildingI.name());
351                displayText=L("You are making @x1",buildingI.name());
352                verb=L("making @x1",buildingI.name());
353                playSound="sanding.wav";
354                buildingI.setDisplayText(L("@x1 lies here",itemName));
355                buildingI.setDescription(itemName+". ");
356                buildingI.basePhyStats().setWeight(getStandardWeight(woodRequired+compData[CF_AMOUNT],bundling));
357                buildingI.setBaseValue(CMath.s_int(foundRecipe.get(RCP_VALUE)));
358                buildingI.basePhyStats().setLevel(CMath.s_int(foundRecipe.get(RCP_LEVEL)));
359                if(buildingI.basePhyStats().level()<1)
360                        buildingI.basePhyStats().setLevel(1);
361                final String type=foundRecipe.get(RCP_TYPE);
362                for(final InstrumentType iType : InstrumentType.values())
363                {
364                        if(type.equalsIgnoreCase(iType.name()))
365                                ((MusicalInstrument)buildingI).setInstrumentType(iType);
366                }
367                setBrand(mob, buildingI);
368                if(buildingI instanceof Rideable)
369                {
370                        ((Rideable)buildingI).setRideBasis(Rideable.RIDEABLE_SIT);
371                        ((Rideable)buildingI).setRiderCapacity(CMath.s_int(misctype));
372                        if(((Rideable)buildingI).riderCapacity()<=0)
373                                ((Rideable)buildingI).setRiderCapacity(1);
374                }
375                else
376                if(!(buildingI instanceof FalseLimb))
377                {
378                        setWearLocation(buildingI,misctype,0);
379                }
380                if(bundling)
381                        buildingI.setBaseValue(lostValue);
382                buildingI.recoverPhyStats();
383                buildingI.text();
384                buildingI.recoverPhyStats();
385
386                messedUp=!proficiencyCheck(mob,0,auto);
387
388                if(bundling)
389                {
390                        messedUp=false;
391                        duration=1;
392                        verb=L("bundling @x1",RawMaterial.CODES.NAME(buildingI.material()).toLowerCase());
393                        startStr=L("<S-NAME> start(s) @x1.",verb);
394                        displayText=L("You are @x1",verb);
395                }
396
397                if(autoGenerate>0)
398                {
399                        crafted.add(buildingI);
400                        return true;
401                }
402
403                final CMMsg msg=CMClass.getMsg(mob,buildingI,this,getActivityMessageType(),startStr);
404                if(mob.location().okMessage(mob,msg))
405                {
406                        mob.location().send(mob,msg);
407                        buildingI=(Item)msg.target();
408                        beneficialAffect(mob,mob,asLevel,duration);
409                }
410                else
411                if(bundling)
412                {
413                        messedUp=false;
414                        aborted=false;
415                        unInvoke();
416                }
417                return true;
418        }
