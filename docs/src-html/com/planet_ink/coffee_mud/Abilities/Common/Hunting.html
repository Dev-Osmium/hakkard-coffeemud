080        public Room nearByRoom()
081        {
082                final Vector<Integer> possibilities=new Vector<Integer>();
083                for(int d=Directions.NUM_DIRECTIONS()-1;d>=0;d--)
084                {
085                        if(d!=Directions.UP)
086                        {
087                                final Room room=activityRoom.getRoomInDir(d);
088                                final Exit exit=activityRoom.getExitInDir(d);
089                                if((room!=null)&&(exit!=null)&&(exit.isOpen()))
090                                        possibilities.addElement(Integer.valueOf(d));
091                        }
092                }
093                if(possibilities.size()>0)
094                {
095                        final int dir=possibilities.elementAt(CMLib.dice().roll(1,possibilities.size(),-1)).intValue();
096                        return activityRoom.getRoomInDir(dir);
097                }
098                return null;
099        }

101        public void moveFound()
102        {
103                if(found.location()==null)
104                        return;
105
106                final List<Integer> possibilities=new ArrayList<Integer>();
107                for(int d=Directions.NUM_DIRECTIONS()-1;d>=0;d--)
108                {
109                        final Room room=found.location().getRoomInDir(d);
110                        if((room == null)
111                        ||(CMLib.flags().isAiryRoom(room)&&(!CMLib.flags().isFlying(found))))
112                        {
113                                final Exit exit=found.location().getExitInDir(d);
114                                if((exit!=null)
115                                &&(exit.isOpen())
116                                &&(CMLib.flags().canBreatheHere(found, room)))
117                                        possibilities.add(Integer.valueOf(d));
118                        }
119                }
120                if(possibilities.size()>0)
121                {
122                        final int dir=possibilities.get(CMLib.dice().roll(1,possibilities.size(),-1)).intValue();
123                        CMLib.tracking().walk(found,dir,true,false);
124                }
125        }

127        @Override
128        public boolean tick(final Tickable ticking, final int tickID)
129        {
130                if((affected!=null)&&(affected instanceof MOB)&&(tickID==Tickable.TICKID_MOB))
131                {
132                        final MOB mob=(MOB)affected;
133                        activityRoom=mob.location();
134                        if((found!=null)&&(found.amDead()))
135                        {
136                                found.setLocation(null);
137                                found.destroy();
138                                unInvoke();
139                        }
140                        else
141                        if((found!=null)
142                        &&(found.location()!=null)
143                        &&(CMLib.flags().isAliveAwakeMobile(found,true))
144                        &&(!found.isInCombat()))
145                        {
146                                if(found.location()==mob.location())
147                                {
148                                        if((mob.isMonster())
149                                        &&(CMLib.flags().isAliveAwakeMobile(mob,true))
150                                        &&(CMLib.flags().canBeSeenBy(found,mob))
151                                        &&(!mob.isInCombat()))
152                                                CMLib.combat().postAttack(mob,found,mob.fetchWieldedItem());
153                                        else
154                                                moveFound();
155                                }
156                        }
157
158                        if(tickUp==0)
159                        {
160                                if(found!=null)
161                                {
162                                        if(CMLib.flags().isWateryRoom(mob.location()))
163                                                commonTell(mob,L("You have found some @x1 signs!",foundShortName));
164                                        else
165                                                commonTell(mob,L("You have found some @x1 tracks!",foundShortName));
166                                        commonTell(mob,L("You need to find the @x1 nearby before the trail goes cold!",foundShortName));
167                                        displayText=L("You are hunting for @x1",found.name());
168                                        verb=L("hunting for @x1",found.name());
169                                        found.basePhyStats().setLevel(mob.basePhyStats().level());
170                                        found.recoverPhyStats();
171                                        final Ability A=CMClass.getAbility("Prop_ModExperience");
172                                        A.setMiscText("=20%");
173                                        found.addNonUninvokableEffect(A);
174                                        found.bringToLife(nearByRoom(),true);
175                                        CMLib.beanCounter().clearZeroMoney(found,null);
176                                        found.setMoneyVariation(0);
177                                }
178                                else
179                                {
180                                        final StringBuffer str=new StringBuffer(L("You can't seem to find any game around here.\n\r"));
181                                        final int d=lookingForMat(RawMaterial.MATERIAL_FLESH,mob.location());
182                                        if(d<0)
183                                                str.append(L("You might try elsewhere."));
184                                        else
185                                                str.append(L("You might try @x1.",CMLib.directions().getInDirectionName(d)));
186                                        commonTell(mob,str.toString());
187                                        unInvoke();
188                                }
189
190                        }
191                        else
192                        if(mob.isMonster()
193                        &&(CMLib.dice().rollPercentage()>50)
194                        &&(CMLib.flags().isMobile(mob))
195                        &&(CMLib.flags().isAliveAwakeMobile(mob,true))
196                        &&(CMLib.flags().canSenseEnteringLeaving(found,mob)))
197                        {
198                                for(int d=Directions.NUM_DIRECTIONS()-1;d>=0;d--)
199                                {
200                                        final Room R=mob.location().getRoomInDir(d);
201                                        if((R!=null)&&(R==found.location()))
202                                        {
203                                                CMLib.tracking().walk(mob,d,false,false);
204                                                break;
205                                        }
206                                }
207                        }
208                }
209                return super.tick(ticking,tickID);
210        }

212        @Override
213        public void unInvoke()
214        {
215                if(canBeUninvoked())
216                {
217                        if(affected instanceof MOB)
218                        {
219                                final MOB mob=(MOB)affected;
220                                if((found!=null)&&(!found.amDead())&&(found.location()!=null)&&(!found.isInCombat()))
221                                {
222                                        if(found.location()==mob.location())
223                                                moveFound();
224                                        found.location().delInhabitant(found);
225                                        found.setLocation(null);
226                                        found.destroy();
227                                        mob.location().show(mob,null,getActivityMessageType(),L("<S-NAME> <S-HAS-HAVE> lost the trail."));
228                                }
229                        }
230                }
231                super.unInvoke();
232        }

234        @Override
235        public boolean invoke(MOB mob, List<String> commands, Physical givenTarget, boolean auto, int asLevel)
236        {
237                if(super.checkStop(mob, commands))
238                        return true;
239                verb=L("hunting");
240                found=null;
241                activityRoom=null;
242                final Room R=mob.location();
243                if(R==null)
244                        return false;
245
246                if(!super.invoke(mob,commands,givenTarget,auto,asLevel))
247                        return false;
248                final int resourceType=R.myResource();
249                final String resourceName = RawMaterial.CODES.NAME(resourceType).toUpperCase();
250                final int domainType = R.domainType()&(~Room.INDOORS);
251                final String domainName;
252                if((R.domainType()&Room.INDOORS)==0)
253                        domainName = (domainType < Room.DOMAIN_OUTDOOR_DESCS.length) ? Room.DOMAIN_OUTDOOR_DESCS[domainType].toUpperCase() : "";
254                else
255                        domainName = (domainType < Room.DOMAIN_INDOORS_DESCS.length) ? Room.DOMAIN_INDOORS_DESCS[domainType].toUpperCase() : "";
256
257                if((proficiencyCheck(mob,0,auto))
258                &&(nearByRoom()!=null))
259                {
260                        final List<List<String>> recipes = loadRecipes("hunting.txt");
261                        if((recipes != null)
262                        &&(recipes.size()>0))
263                        {
264                                int totalWeights = 0;
265                                List<List<String>> subset=new ArrayList<List<String>>();
266                                for(final List<String> subl : recipes)
267                                {
268                                        if(subl.size()<4)
269                                                continue;
270                                        if(subl.get(RCP_RESOURCE).toUpperCase().equals(resourceName)
271                                        &&((subl.get(RCP_DOMAIN).equals("*"))
272                                                ||(subl.get(RCP_DOMAIN).equals(domainName))
273                                                ||(subl.get(RCP_DOMAIN).equals(R.ID()))))
274                                        {
275                                                subset.add(subl);
276                                                totalWeights += CMath.s_int(subl.get(RCP_FREQ));
277                                        }
278                                }
279                                List<String> winl = null;
280                                if(totalWeights > 0)
281                                {
282                                        int winner=CMLib.dice().roll(1, totalWeights, -1);
283                                        int current = 0;
284                                        for(final List<String> subl : subset)
285                                        {
286                                                current += CMath.s_int(subl.get(RCP_FREQ));
287                                                if(winner < current)
288                                                {
289                                                        winl = subl;
290                                                        break;
291                                                }
292                                        }
293                                        if(winl == null)
294                                                winl=subset.get(subset.size()-1);
295                                }
296                                if(winl != null)
297                                {
298                                        String mobID = winl.get(RCP_MOB);
299                                        MOB stdM = CMClass.getMOB(mobID);
300                                        if(stdM == null)
301                                                stdM = CMLib.catalog().getCatalogMob(mobID);
302                                        if(stdM == null)
303                                        {
304                                                commonTell(mob,L("There are no signs of life here."));
305                                                return false;
306                                        }
307                                        MOB genM=stdM;
308                                        if(!genM.isGeneric())
309                                        {
310                                                if(genM.baseCharStats().getMyRace().useRideClass())
311                                                        genM=CMClass.getMOB("GenRideable");
312                                                else
313                                                        genM=CMClass.getMOB("GenMob");
314                                                genM.setBaseCharStats((CharStats)stdM.baseCharStats().copyOf());
315                                                genM.setBasePhyStats((PhyStats)stdM.basePhyStats().copyOf());
316                                                genM.setBaseState((CharState)stdM.baseState().copyOf());
317                                                for(final GenericBuilder.GenMOBCode stat : GenericBuilder.GenMOBCode.values())
318                                                {
319                                                        if(stat != GenericBuilder.GenMOBCode.ABILITY) // because this screws up gen hit points
320                                                                genM.setStat(stat.name(), CMLib.coffeeMaker().getGenMobStat((MOB)stdM,stat.name()));
321                                                }
322                                                genM.recoverCharStats();
323                                                genM.recoverPhyStats();
324                                                genM.recoverMaxState();
325                                                genM.resetToMaxState();
326                                        }
327                                        final int moblevel = Math.round(CMath.sqrt(mob.phyStats().level())); // pity on artisans
328                                        final int addlevel = (moblevel < CMProps.getIntVar(CMProps.Int.EXPRATE)) ? 0 : (moblevel - CMProps.getIntVar(CMProps.Int.EXPRATE));
329                                        genM.basePhyStats().setLevel(genM.basePhyStats().level() + addlevel);
330                                        genM.recoverCharStats();
331                                        genM.recoverPhyStats();
332                                        genM.recoverMaxState();
333                                        genM.resetToMaxState();
334                                        CMLib.leveler().fillOutMOB(genM,genM.basePhyStats().level());
335                                        
336                                        found=genM;
337                                        foundShortName=found.name();
338                                        int x=0;
339                                        if((x=foundShortName.lastIndexOf(' '))>=0)
340                                                foundShortName=foundShortName.substring(x).trim().toLowerCase();
341                                        found.setLocation(null);
342                                        if(found instanceof Rideable)
343                                        {
344                                                if((CMLib.flags().canBreatheThis(found, RawMaterial.RESOURCE_FRESHWATER)
345                                                        || CMLib.flags().canBreatheThis(found, RawMaterial.RESOURCE_SALTWATER))
346                                                &&(!CMLib.flags().canBreatheThis(found, RawMaterial.RESOURCE_AIR)))
347                                                        ((Rideable)found).setRideBasis(Rideable.RIDEABLE_WATER);
348                                                else
349                                                        ((Rideable)found).setRideBasis(Rideable.RIDEABLE_LAND);
350                                        }
351                                }
352                        }
353                }
354                final int duration=10+mob.phyStats().level()+(super.getXTIMELevel(mob)*2);
355                final CMMsg msg=CMClass.getMsg(mob,found,this,getActivityMessageType(),L("<S-NAME> start(s) hunting."));
356                if(R.okMessage(mob,msg))
357                {
358                        R.send(mob,msg);
359                        found=(MOB)msg.target();
360                        beneficialAffect(mob,mob,asLevel,duration);
361                }
362                return true;
363        }
