227        public boolean isPossibleCatalog(final MOB mob, final Item I, final Environmental fullyE, final boolean quiet)
228        {
229                if((I.material()&RawMaterial.MATERIAL_MASK)!=RawMaterial.MATERIAL_PAPER)
230                {
231                        if(!quiet)
232                                commonTell(mob,L("You can't write in @x1.",I.name(mob)));
233                        return false;
234                }
235                if(!CMLib.flags().isReadable(I))
236                {
237                        if(!quiet)
238                                commonTell(mob,L("@x1 is not even readable!",CMStrings.capitalizeFirstLetter(I.name(mob))));
239                        return false;
240                }
241                if(I instanceof Recipe)
242                {
243                        commonTell(mob,L("@x1 isn't a catalog!",CMStrings.capitalizeAndLower(I.name(mob))));
244                        return false;
245                }
246                /*
247                String brand = getBrand(I);
248                if((brand==null)||(brand.length()==0))
249                {
250                        if(!quiet)
251                                commonTell(mob,L("You aren't permitted to add catalog entries to @x1.",I.name(mob)));
252                        return false;
253                }
254                */
255                
256                if(fullyE != null)
257                {
258                        if(!Titling.getCatalogType(I).equals(Titling.getCatalogEntryType(fullyE)))
259                        {
260                                if(!quiet)
261                                        commonTell(mob,L("@x1 is not a proper catalog for @x2!",CMStrings.capitalizeFirstLetter(I.name(mob)),fullyE.name()));
262                        }
263                }
264                return true;
265        }

267        public Item findCatalogBook(final MOB mob, final Physical itemI)
268        {
269                Item catalogI=null;
270                for(final Enumeration<Item> i=mob.items();i.hasMoreElements();)
271                {
272                        final Item I=i.nextElement();
273                        if((I.container()==null)
274                        &&(!I.amWearingAt(Item.IN_INVENTORY))
275                        &&(isPossibleCatalog(mob, I, itemI, true)))
276                                catalogI=I;
277                }
278                if(catalogI==null)
279                {
280                        for(final Enumeration<Item> i=mob.items();i.hasMoreElements();)
281                        {
282                                final Item I=i.nextElement();
283                                if((I.container()==null)
284                                &&(isPossibleCatalog(mob, I, itemI, true)))
285                                        catalogI=I;
286                        }
287                }
288                if(catalogI==null)
289                {
290                        for(final Enumeration<Item> i=mob.items();i.hasMoreElements();)
291                        {
292                                final Item I=i.nextElement();
293                                if(isPossibleCatalog(mob, I, itemI, true))
294                                        catalogI=I;
295                        }
296                }
297                return catalogI;
298        }

300        @Override
301        public boolean invoke(MOB mob, List<String> commands, Physical givenTarget, boolean auto, int asLevel)
302        {
303                if(super.checkStop(mob, commands))
304                        return true;
305                if(commands.size()<1)
306                {
307                        commonTell(mob,L("You must specify what you want to catalog, and, optionally, where to add the entry.  If you do "
308                                        + "not specify the catalog, one prepared with the Titling skill will be automatically selected from your inventory."));
309                        return false;
310                }
311                Physical physP=null;
312                Item catalogI=null;
313                String itemName=CMParms.combine(commands,0);
314                if(commands.size()==1) // this always means the item is alone
315                {
316                        physP=this.getAnyTarget(mob, mob.location(), true, commands, givenTarget, Wearable.FILTER_UNWORNONLY);
317                        if((physP==null)||(!CMLib.flags().canBeSeenBy(physP,mob)))
318                                physP=null;
319                        if(physP!=null)
320                        {
321                                catalogI=this.findCatalogBook(mob, physP);
322                                if(catalogI==null)
323                                {
324                                        commonTell(mob,L("You need to specify a proper catalog for '@x1'.",physP.name(mob)));
325                                        return false;
326                                }
327                        }
328                }
329                else
330                {
331                        physP=this.getAnyTarget(mob, mob.location(), true, commands, givenTarget, Wearable.FILTER_UNWORNONLY, true);
332                        if((physP==null)||(!CMLib.flags().canBeSeenBy(physP,mob)))
333                                physP=null;
334                        if(physP!=null)
335                        {
336                                catalogI=this.findCatalogBook(mob, physP);
337                                if(catalogI==null)
338                                {
339                                        commonTell(mob,L("You need to specify a proper catalog for '@x1'.",physP.name(mob)));
340                                        return false;
341                                }
342                        }
343                        if((physP==null)||(catalogI==null))
344                        {
345                                itemName=commands.get(0);
346                                physP=this.getAnyTarget(mob, mob.location(), true, new XVector<String>(commands.get(0)), givenTarget, Wearable.FILTER_UNWORNONLY);
347                                if((physP==null)||(!CMLib.flags().canBeSeenBy(physP,mob)))
348                                        physP=null;
349                                if(physP!=null)
350                                {
351                                        List<String> cmds2=new ArrayList<String>(commands);
352                                        cmds2.remove(0);
353                                        catalogI=this.getTarget(mob, null, givenTarget, cmds2, Wearable.FILTER_UNWORNONLY);
354                                        if(catalogI!=null)
355                                        {
356                                                if(!isPossibleCatalog(mob, catalogI, null, false))
357                                                        return false;
358                                                String cat=Titling.getCatalogType(catalogI);
359                                                if((cat.length()>0)&&(!isPossibleCatalog(mob, catalogI, physP, false)))
360                                                        return false;
361                                        }
362                                }
363                        }
364                }
365                if(physP==null)
366                {
367                        commonTell(mob,L("You don't seem to have a '@x1'.",itemName));
368                        return false;
369                }
370                if(!(physP instanceof Item))
371                {
372                        commonTell(mob,L("You can't catalog @x1",physP.name()));
373                        return false;
374                }
375                if(catalogI==null)
376                {
377                        commonTell(mob,L("You don't seem to have a proper catalog."));
378                        return false;
379                }
380                
381                final Ability writeA=mob.fetchAbility("Skill_Write");
382                if(writeA==null)
383                {
384                        commonTell(mob,L("You must know how to write to entitle."));
385                        return false;
386                }
387
388                if(!super.invoke(mob,commands,givenTarget,auto,asLevel))
389                        return false;
390                verb=L("cataloging @x1 into @x2",physP.name(),catalogI.name());
391                displayText=L("You are @x1",verb);
392                this.found=physP;
393                this.catalogI=catalogI;
394                if((!proficiencyCheck(mob,0,auto))
395                ||(!writeA.proficiencyCheck(mob,super.getXLEVELLevel(mob)*10,auto)))
396                        this.catalogI=null;
397                final int duration=getDuration(15,mob,1,1);
398                final CMMsg msg=CMClass.getMsg(mob,physP,this,getActivityMessageType(),L("<S-NAME> start(s) cataloging <T-NAME> into @x1.",catalogI.name()));
399                if(mob.location().okMessage(mob,msg))
400                {
401                        mob.location().send(mob,msg);
402                        beneficialAffect(mob,mob,asLevel,duration);
403                }
404                return true;
405        }
