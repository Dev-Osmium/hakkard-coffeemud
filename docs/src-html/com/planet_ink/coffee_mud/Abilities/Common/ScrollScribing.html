116        @Override
117        public void unInvoke()
118        {
119                if(canBeUninvoked())
120                {
121                        if(affected instanceof MOB)
122                        {
123                                final MOB mob=(MOB)affected;
124                                if((buildingI!=null)&&(!aborted))
125                                {
126                                        if(messedUp)
127                                        {
128                                                commonTell(mob,L("Something went wrong! @x1 explodes!",buildingI.name()));
129                                                buildingI.destroy();
130                                        }
131                                        else
132                                        {
133                                                final int theSpellLevel=spellLevel(mob,theSpell);
134                                                if(fromTheScroll != null)
135                                                        eraseFromScrollItem(fromTheScroll,theSpell,theSpellLevel);
136                                                buildingI=buildScrollItem((Scroll)buildingI, theSpell, theSpellLevel);
137                                                if(buildingI.secretIdentity().length()==0)
138                                                        setBrand(mob, buildingI);
139
140                                        }
141                                }
142                                buildingI=null;
143                        }
144                }
145                super.unInvoke();
146        }

148        protected int spellLevel(final MOB mob, final Ability A)
149        {
150                int lvl=CMLib.ableMapper().qualifyingLevel(mob,A);
151                if(lvl<0)
152                        lvl=CMLib.ableMapper().lowestQualifyingLevel(A.ID());
153                switch(lvl)
154                {
155                case 0:
156                        return lvl;
157                case 1:
158                        return lvl;
159                case 2:
160                        return lvl + 1;
161                case 3:
162                        return lvl + 1;
163                case 4:
164                        return lvl + 2;
165                case 5:
166                        return lvl + 2;
167                case 6:
168                        return lvl + 3;
169                case 7:
170                        return lvl + 3;
171                case 8:
172                        return lvl + 4;
173                case 9:
174                        return lvl + 4;
175                default:
176                        return lvl + 5;
177                }
178        }

192        protected void eraseFromScrollItem(final Scroll buildingI, final Ability theSpell, final int level)
193        {
194                if(buildingI == null)
195                        return;
196                final StringBuilder newList=new StringBuilder();
197                for(final Ability A : buildingI.getSpells())
198                {
199                        if(!A.ID().equalsIgnoreCase(theSpell.ID()))
200                                newList.append(A.ID()).append(";");
201                        if(buildingI.isGeneric())
202                        {
203                                final String testName=L(" OF @x1",A.Name().toUpperCase());
204                                if(buildingI.Name().toUpperCase().endsWith(testName))
205                                        buildingI.setName(buildingI.Name().substring(0,buildingI.Name().length()-testName.length()));
206                        }
207                }
208                buildingI.setSpellList(newList.toString());
209                if(buildingI.isGeneric())
210                {
211                        if(buildingI.getSpells().size()==1)
212                        {
213                                if(buildingI.name().toUpperCase().endsWith(L(" SCROLL")))
214                                        buildingI.setName(L("@x1 of @x2",buildingI.name(),buildingI.getSpells().get(0).name().toLowerCase()));
215                                buildingI.setDisplayText(L("@x1 sits here.",buildingI.Name()));
216                        }
217                }
218                if(buildingI.usesRemaining()>0)
219                        buildingI.setUsesRemaining(buildingI.usesRemaining()-1);
220                buildingI.text();
221        }

223        protected Scroll buildScrollItem(Scroll buildingI, final Ability theSpell, final int level)
224        {
225                if(buildingI == null)
226                        buildingI=(Scroll)CMClass.getItem("GenScroll");
227                final StringBuilder newList=new StringBuilder();
228                if(buildingI.usesRemaining()==0)
229                {
230                        for(final Ability A : buildingI.getSpells())
231                                this.eraseFromScrollItem(buildingI, A, -1);
232                }
233                else
234                {
235                        for(final Ability A : buildingI.getSpells())
236                        {
237                                newList.append(A.ID()).append(";");
238                                final String testName=L(" OF @x1",A.Name().toUpperCase());
239                                if(buildingI.Name().toUpperCase().endsWith(testName))
240                                        buildingI.setName(buildingI.Name().substring(0,buildingI.Name().length()-testName.length()));
241                        }
242                }
243                newList.append(theSpell.ID());
244                buildingI.setSpellList(newList.toString());
245                if(buildingI.getSpells().size()==1)
246                {
247                        if(buildingI.name().toUpperCase().endsWith(L(" SCROLL")))
248                                buildingI.setName(L("@x1 of @x2",buildingI.name(),theSpell.name().toLowerCase()));
249                        buildingI.setDisplayText(L("@x1 sits here.",buildingI.Name()));
250                        buildingI.basePhyStats().setLevel(level);
251                        buildingI.phyStats().setLevel(level);
252                        buildingI.recoverPhyStats();
253                }
254                else
255                if(buildingI.basePhyStats().level() < level)
256                {
257                        buildingI.basePhyStats().setLevel(level);
258                        buildingI.phyStats().setLevel(level);
259                        buildingI.recoverPhyStats();
260                }
261                buildingI.setDescription("");
262                buildingI.setUsesRemaining(buildingI.usesRemaining()+1);
263                buildingI.text();
264                return buildingI;
265        }

279        @Override
280        protected boolean autoGenInvoke(final MOB mob, final List<String> commands, final Physical givenTarget, final boolean auto,
281                                                                 final int asLevel, final int autoGenerate, final boolean forceLevels, final List<Item> crafted)
282        {
283                if(super.checkStop(mob, commands))
284                        return true;
285
286                if(autoGenerate>0)
287                {
288                        final Ability theSpell=super.getCraftableSpellRecipeSpell(commands);
289                        if(theSpell==null)
290                                return false;
291                        final int level=spellLevel(mob,theSpell);
292                        buildingI=buildScrollItem(null, theSpell, level);
293                        crafted.add(buildingI);
294                        return true;
295                }
296                randomRecipeFix(mob,addRecipes(mob,loadRecipes()),commands,0);
297                if(commands.size()<1)
298                {
299                        commonTell(mob,L("Enscribe what? Enter \"enscribe list\" for a list, or \"enscribe stop\" to cancel."));
300                        return false;
301                }
302                final List<List<String>> recipes=addRecipes(mob,loadRecipes());
303                final String pos=commands.get(commands.size()-1);
304                if(((commands.get(0))).equalsIgnoreCase("list"))
305                {
306                        String mask=CMParms.combine(commands,1);
307                        boolean allFlag=false;
308                        if(mask.equalsIgnoreCase("all"))
309                        {
310                                allFlag=true;
311                                mask="";
312                        }
313                        final StringBuffer buf=new StringBuffer(L("Scrolls you know how to enscribe:\n\r"));
314                        final int colWidth=CMLib.lister().fixColWidth(22,mob.session());
315                        final int col2Width=CMLib.lister().fixColWidth(12,mob.session());
316                        final String headerPart="^H"+CMStrings.padRight(L("Spell"),colWidth)+" "+CMStrings.padRight(L("Materials"),col2Width);
317                        buf.append(headerPart).append("^w| ^H").append(headerPart).append("^.^N\n\r");
318                        int toggler=1;
319                        final int toggleTop=2;
320                        for(int r=0;r<recipes.size();r++)
321                        {
322                                final List<String> V=recipes.get(r);
323                                if(V.size()>0)
324                                {
325                                        final String spell=V.get(0);
326                                        final String matts=CMStrings.capitalizeAndLower(V.get(1));
327                                        final Ability A=mob.fetchAbility(spell);
328                                        if((A!=null)
329                                        &&((spellLevel(mob,A)>=0)||(allFlag))
330                                        &&((xlevel(mob)>=spellLevel(mob,A))||(allFlag))
331                                        &&((mask.length()==0)||mask.equalsIgnoreCase("all")||CMLib.english().containsString(spell,mask)))
332                                        {
333                                                buf.append(CMStrings.padRight(A.name(),colWidth)).append(" ");
334                                                buf.append(CMStrings.padRight(matts,col2Width)+((toggler!=toggleTop)?"^w| ^N":"\n\r"));
335                                                if(++toggler>toggleTop)
336                                                        toggler=1;
337                                        }
338                                }
339                        }
340                        if(toggler!=1)
341                                buf.append("\n\r");
342                        commonTell(mob,buf.toString());
343                        return true;
344                }
345                else
346                if((!auto)&&(commands.size()<2))
347                {
348                        commonEmote(mob,L("You must specify what magic you wish to enscribe, and the paper to enscribe it in.  You can specify the word \"from\" and another scroll name to move a spell from one scroll to another."));
349                        return false;
350                }
351                else
352                {
353                        buildingI=getTarget(mob,null,givenTarget,CMParms.parse(pos),Wearable.FILTER_UNWORNONLY);
354                        commands.remove(pos);
355                        if(buildingI==null)
356                                return false;
357                        if(!mob.isMine(buildingI))
358                        {
359                                commonTell(mob,L("You'll need to pick that up first."));
360                                return false;
361                        }
362                        if((((buildingI.material()&RawMaterial.MATERIAL_MASK)!=RawMaterial.MATERIAL_PAPER))
363                        &&(buildingI.material() != RawMaterial.RESOURCE_SILK))
364                        {
365                                commonTell(mob,L("@x1 isn't even made of paper or silk!",buildingI.name(mob)));
366                                return false;
367                        }
368                        if((!(buildingI instanceof Scroll))
369                        ||(!buildingI.isGeneric())
370                        ||(!(buildingI instanceof MiscMagic))
371                        ||(buildingI instanceof Recipe))
372                        {
373                                commonTell(mob,L("There's can't enscribe magic on @x1!",buildingI.name(mob)));
374                                return false;
375                        }
376                        if(((Scroll)buildingI).getSpells().size()>0)
377                        {
378                                int level=25;
379                                for(final Ability A : ((Scroll)buildingI).getSpells())
380                                {
381                                        int lvl=CMLib.ableMapper().qualifyingLevel(mob,A);
382                                        if(lvl<0)
383                                                lvl=CMLib.ableMapper().lowestQualifyingLevel(A.ID());
384                                        level -= lvl;
385                                }
386                                if(level <= 0)
387                                {
388                                        commonTell(mob,L("You can only scribe on blank scrolls, or scroll with enough free space on it."));
389                                        return false;
390                                }
391                        }
392                        String recipeName=CMParms.combine(commands,0);
393                        theSpell=null;
394                        fromTheScroll=null;
395                        String ingredient="";
396                        for(int r=0;r<recipes.size();r++)
397                        {
398                                final List<String> V=recipes.get(r);
399                                if(V.size()>0)
400                                {
401                                        final String spell=V.get(0);
402                                        final Ability A=mob.fetchAbility(spell);
403                                        if((A!=null)
404                                        &&(xlevel(mob)>=spellLevel(mob,A))
405                                        &&(A.name().equalsIgnoreCase(recipeName)))
406                                        {
407                                                theSpell=A;
408                                                ingredient=V.get(1);
409                                        }
410                                }
411                        }
412                        int manaToLose=10;
413                        int experienceToLose=0;
414                        if(theSpell==null)
415                        {
416                                final int x=CMParms.indexOfIgnoreCase(commands, "from");
417                                if((x>0)&&(x<commands.size()-1))
418                                {
419                                        recipeName=CMParms.combine(commands,0,x);
420                                        final String otherScrollName=CMParms.combine(commands,x+1,commands.size());
421                                        final Item scrollFromI=getTarget(mob,null,givenTarget,CMParms.parse(otherScrollName),Wearable.FILTER_UNWORNONLY);
422                                        if(scrollFromI==null)
423                                                return false;
424                                        if(!mob.isMine(scrollFromI))
425                                        {
426                                                commonTell(mob,L("You'll need to pick that up first."));
427                                                return false;
428                                        }
429                                        if(!(scrollFromI instanceof Scroll))
430                                        {
431                                                commonTell(mob,L("@x1 is not a scroll!",scrollFromI.name(mob)));
432                                                return false;
433                                        }
434                                        if((!(scrollFromI instanceof Scroll))||(((Scroll)scrollFromI).getSpells().size()==0))
435                                        {
436                                                commonTell(mob,L("@x1 has no spells on it!",scrollFromI.name(mob)));
437                                                return false;
438                                        }
439                                        if(scrollFromI.usesRemaining() <=0)
440                                        {
441                                                commonTell(mob,L("@x1 has no magical charge left in it.",scrollFromI.name(mob)));
442                                                return false;
443                                        }
444                                        ingredient="";
445                                        for(final Ability A : ((Scroll)scrollFromI).getSpells())
446                                        {
447                                                if((A!=null)
448                                                &&(xlevel(mob)>=spellLevel(mob,A))
449                                                &&(A.name().equalsIgnoreCase(recipeName)))
450                                                        theSpell=A;
451                                        }
452                                        if(theSpell==null)
453                                        {
454                                                commonTell(mob,L("You can't enscribe the spell '@x1' from the scroll @x2!",recipeName,scrollFromI.name(mob)));
455                                                return false;
456                                        }
457                                        fromTheScroll=(Scroll)scrollFromI;
458                                }
459                                else
460                                if(theSpell==null)
461                                {
462                                        commonTell(mob,L("You don't know how to enscribe '@x1'.  Try \"enscribe list\" for a list.",recipeName));
463                                        return false;
464                                }
465                                manaToLose+=spellLevel(mob,theSpell)*10;
466                        }
467                        else
468                        {
469                                manaToLose+=CMLib.ableMapper().qualifyingLevel(mob,theSpell)*10;
470                                manaToLose-=CMLib.ableMapper().qualifyingClassLevel(mob,theSpell)*5;
471                                experienceToLose+=10+CMLib.ableMapper().qualifyingLevel(mob,theSpell);
472                                experienceToLose-=CMLib.ableMapper().qualifyingClassLevel(mob,theSpell);
473                                if(experienceToLose < CMLib.ableMapper().qualifyingLevel(mob,theSpell))
474                                        experienceToLose = CMLib.ableMapper().qualifyingLevel(mob,theSpell);
475                        }
476
477                        if(buildingI!=null)
478                        {
479                                if(((Scroll)buildingI).usesRemaining()>0)
480                                {
481                                        for(final Ability spell: ((Scroll)buildingI).getSpells())
482                                        {
483                                                if(spell.ID().equals(theSpell.ID()))
484                                                {
485                                                        mob.tell(L("That spell is already scribed onto @x1.",buildingI.name()));
486                                                        return false;
487                                                }
488                                        }
489                                }
490                        }
491                        if((CMath.bset(theSpell.flags(), Ability.FLAG_CLANMAGIC)))
492                        {
493                                mob.tell(L("That spell cannot be scribed onto a scroll."));
494                                return false;
495                        }
496
497                        final int resourceType=(ingredient.length()==0) ? -1 : RawMaterial.CODES.FIND_IgnoreCase(ingredient);
498
499                        int[][] data = null;
500                        if(resourceType>0)
501                        {
502                                final int[] pm={resourceType};
503                                data=fetchFoundResourceData(mob,
504                                                                                        1,ingredient,pm,
505                                                                                        0,null,null,
506                                                                                        bundling,
507                                                                                        -1,
508                                                                                        null);
509                                if(data==null)
510                                        return false;
511                        }
512                        if(manaToLose<10)
513                                manaToLose=10;
514
515                        if(mob.curState().getMana()<manaToLose)
516                        {
517                                commonTell(mob,L("You need at least @x1 mana to accomplish that.",""+manaToLose));
518                        }
519
520                        if(!super.invoke(mob,commands,givenTarget,auto,asLevel))
521                                return false;
522
523                        mob.curState().adjMana(-manaToLose, mob.maxState());
524
525                        if((resourceType>0)&&(data != null))
526                                CMLib.materials().destroyResourcesValue(mob.location(),data[0][FOUND_AMT],data[0][FOUND_CODE],0,null);
527
528                        playSound=null;
529                        if(experienceToLose > 0)
530                        {
531                                experienceToLose=getXPCOSTAdjustment(mob,experienceToLose);
532                                CMLib.leveler().postExperience(mob,null,null,-experienceToLose,false);
533                                commonTell(mob,L("You lose @x1 experience points for the effort.",""+experienceToLose));
534                        }
535
536                        int duration=getDuration(CMLib.ableMapper().qualifyingLevel(mob,theSpell)*5,mob,CMLib.ableMapper().lowestQualifyingLevel(theSpell.ID()),10);
537                        if(duration<10)
538                                duration=10;
539                        messedUp=!proficiencyCheck(mob,0,auto);
540
541                        String msgStr;
542                        if(fromTheScroll != null)
543                        {
544                                msgStr=L("<S-NAME> start(s) transscribing @x1 from @x2 to @x3.",theSpell.name(),fromTheScroll.name(),buildingI.name());
545                                displayText=L("You are transscribing @x1 from @x2 to @x3",theSpell.name(),fromTheScroll.name(),buildingI.name());
546                                verb=L("transscribing @x1 from @x2 to @x3",theSpell.name(),fromTheScroll.name(),buildingI.name());
547                        }
548                        else
549                        {
550                                msgStr=L("<S-NAME> start(s) scribing @x1 onto @x2.",theSpell.name(),buildingI.name());
551                                displayText=L("You are scribing @x1 onto @x2",theSpell.name(),buildingI.name());
552                                verb=L("scribing @x1 onto @x2",theSpell.name(),buildingI.name());
553                        }
554                        final CMMsg msg=CMClass.getMsg(mob,buildingI,this,getActivityMessageType(),msgStr);
555                        if(mob.location().okMessage(mob,msg))
556                        {
557                                mob.location().send(mob,msg);
558                                buildingI=(Item)msg.target();
559                                beneficialAffect(mob,mob,asLevel,duration);
560                        }
561                }
562                return true;
563        }
