115        @Override
116        public void unInvoke()
117        {
118                if(canBeUninvoked())
119                {
120                        if(affected instanceof MOB)
121                        {
122                                final MOB mob=(MOB)affected;
123                                if((buildingI!=null)&&(!aborted))
124                                {
125                                        if(messedUp)
126                                        {
127                                                if(activity == CraftingActivity.MENDING)
128                                                        messedUpCrafting(mob);
129                                                else
130                                                if(activity == CraftingActivity.LEARNING)
131                                                {
132                                                        commonEmote(mob,L("<S-NAME> fail(s) to learn how to make @x1.",buildingI.name()));
133                                                        buildingI.destroy();
134                                                }
135                                                else
136                                                if(activity == CraftingActivity.REFITTING)
137                                                        commonEmote(mob,L("<S-NAME> mess(es) up refitting @x1.",buildingI.name()));
138                                                else
139                                                        commonEmote(mob,L("<S-NAME> mess(es) up making @x1.",buildingI.name()));
140                                        }
141                                        else
142                                        {
143                                                if(activity == CraftingActivity.MENDING)
144                                                {
145                                                        buildingI.setUsesRemaining(100);
146                                                        CMLib.achievements().possiblyBumpAchievement(mob, AchievementLibrary.Event.MENDER, 1, this);
147                                                }
148                                                else
149                                                if(activity==CraftingActivity.LEARNING)
150                                                {
151                                                        deconstructRecipeInto(mob, buildingI, recipeHolder );
152                                                        buildingI.destroy();
153                                                }
154                                                else
155                                                if(activity == CraftingActivity.REFITTING)
156                                                {
157                                                        buildingI.basePhyStats().setHeight(0);
158                                                        buildingI.recoverPhyStats();
159                                                }
160                                                else
161                                                {
162                                                        dropAWinner(mob,buildingI);
163                                                        CMLib.achievements().possiblyBumpAchievement(mob, AchievementLibrary.Event.CRAFTING, 1, this);
164                                                }
165                                        }
166                                }
167                                buildingI=null;
168                                activity = CraftingActivity.CRAFTING;
169                        }
170                }
171                super.unInvoke();
172        }

174        @Override
175        public boolean mayICraft(final Item I)
176        {
177                if(I==null)
178                        return false;
179                if(!super.mayBeCrafted(I))
180                        return false;
181                if((I.material()&RawMaterial.MATERIAL_MASK)!=RawMaterial.MATERIAL_LEATHER)
182                        return false;
183                if(CMLib.flags().isDeadlyOrMaliciousEffect(I))
184                        return false;
185                if(I.basePhyStats().level()<31)
186                        return (isANativeItem(I.Name()));
187                if(I instanceof Armor)
188                {
189                        final long noWearLocations=Wearable.WORN_LEFT_FINGER|Wearable.WORN_RIGHT_FINGER|Wearable.WORN_EARS;
190                        if((I.rawProperLocationBitmap() & noWearLocations)>0)
191                                return (isANativeItem(I.Name()));
192                        return true;
193                }
194                if(I instanceof Rideable)
195                {
196                        final Rideable R=(Rideable)I;
197                        final int rideType=R.rideBasis();
198                        switch(rideType)
199                        {
200                        case Rideable.RIDEABLE_SLEEP:
201                        case Rideable.RIDEABLE_SIT:
202                        case Rideable.RIDEABLE_TABLE:
203                                return true;
204                        default:
205                                return false;
206                        }
207                }
208                if(I instanceof Shield)
209                        return true;
210                if(I instanceof Weapon)
211                {
212                        final Weapon W=(Weapon)I;
213                        if(((W instanceof AmmunitionWeapon)&&((AmmunitionWeapon)W).requiresAmmunition())
214                        ||(W.weaponClassification()==Weapon.CLASS_FLAILED))
215                                return true;
216                        return (isANativeItem(I.Name()));
217                }
218                if(I instanceof Container)
219                        return true;
220                if((I instanceof Drink)&&(!(I instanceof Potion)))
221                        return true;
222                if(I instanceof FalseLimb)
223                        return true;
224                if(I.rawProperLocationBitmap()==Wearable.WORN_HELD)
225                        return true;
226                return (isANativeItem(I.Name()));
227        }

249        @Override
250        protected List<List<String>> loadRecipes()
251        {
252                final String filename=parametersFile();
253                @SuppressWarnings("unchecked")
254                List<List<String>> recipes=(List<List<String>>)Resources.getResource("PARSED_RECIPE: "+filename);
255                if(recipes==null)
256                {
257                        final StringBuffer str=new CMFile(Resources.buildResourcePath("skills")+filename,null,CMFile.FLAG_LOGERRORS).text();
258                        recipes=loadList(str);
259                        if(recipes.size()==0)
260                                Log.errOut("LeatherWorking","Recipes not found!");
261                        else
262                        {
263                                final List<List<String>> newRecipes=new Vector<List<String>>();
264                                for(int r=0;r<recipes.size();r++)
265                                {
266                                        final List<String> V=recipes.get(r);
267                                        if(V.size()>0)
268                                        {
269                                                final String name=V.get(RCP_FINALNAME);
270                                                final int baseLevel=CMath.s_int(V.get(RCP_LEVEL))+2;
271                                                for(final Stage s : Stage.values())
272                                                {
273                                                        final List<String> V1=new XVector<String>(V);
274                                                        V1.set(RCP_FINALNAME,s.name()+" "+name);
275                                                        final int level=baseLevel+s.recipeLevel;
276                                                        V1.set(RCP_LEVEL,""+level);
277                                                        for(int i=0;i<=newRecipes.size();i++)
278                                                        {
279                                                                if(newRecipes.size()==i)
280                                                                {
281                                                                        newRecipes.add(V1);
282                                                                        break;
283                                                                }
284                                                                else
285                                                                if(CMath.s_int(newRecipes.get(i).get(RCP_LEVEL))>level)
286                                                                {
287                                                                        newRecipes.add(i,V1);
288                                                                        break;
289                                                                }
290                                                        }
291                                                }
292                                        }
293                                }
294                                recipes.clear();
295                                recipes=newRecipes;
296                        }
297                        Resources.submitResource("PARSED_RECIPE: "+filename,recipes);
298                }
299                return recipes;
300        }

314        @Override
315        protected boolean autoGenInvoke(final MOB mob, List<String> commands, Physical givenTarget, final boolean auto, 
316                                                                 final int asLevel, int autoGenerate, boolean forceLevels, List<Item> crafted)
317        {
318                if(super.checkStop(mob, commands))
319                        return true;
320
321                if(super.checkInfo(mob, commands))
322                        return true;
323                
324                final PairVector<EnhancedExpertise,Integer> enhancedTypes=enhancedTypes(mob,commands);
325                int recipeLevel = 1;
326                randomRecipeFix(mob,addRecipes(mob,loadRecipes()),commands,autoGenerate);
327                if(commands.size()==0)
328                {
329                        commonTell(mob,L("Make what? Enter \"mleatherwork list\" for a list, \"mleatherworkd info <item>\", \"mleatherwork refit <item>\" to resize,"
330                                                        + " \"mleatherwork learn <item>\", \"mleatherwork scan\", \"mleatherwork mend <item>\", or \"mleatherwork stop\" to cancel."));
331                        return false;
332                }
333                if((!auto)
334                &&(commands.size()>0)
335                &&((commands.get(0)).equalsIgnoreCase("bundle")))
336                {
337                        bundling=true;
338                        if(super.invoke(mob,commands,givenTarget,auto,asLevel))
339                                return super.bundle(mob,commands);
340                        return false;
341                }
342                final List<List<String>> recipes=addRecipes(mob,loadRecipes());
343                final String str=commands.get(0);
344                playSound="scissor.wav";
345                String startStr=null;
346                bundling=false;
347                int multiplier=4;
348                int duration=4;
349                if(str.equalsIgnoreCase("list"))
350                {
351                        String mask=CMParms.combine(commands,1);
352                        boolean allFlag=false;
353                        if(mask.equalsIgnoreCase("all"))
354                        {
355                                allFlag=true;
356                                mask="";
357                        }
358                        final StringBuffer buf=new StringBuffer("");
359                        int toggler=1;
360                        final int toggleTop=2;
361                        final int[] cols={
362                                CMLib.lister().fixColWidth(30,mob.session()),
363                                CMLib.lister().fixColWidth(3,mob.session()),
364                                CMLib.lister().fixColWidth(3,mob.session())
365                        };
366                        for(int r=0;r<toggleTop;r++)
367                                buf.append((r>0?" ":"")+CMStrings.padRight(L("Item"),cols[0])+" "+CMStrings.padRight(L("Lvl"),cols[1])+" "+CMStrings.padRight(L("Amt"),cols[2]));
368                        buf.append("\n\r");
369                        final List<List<String>> listRecipes=((mask.length()==0) || mask.equalsIgnoreCase("all")) ? recipes : super.matchingRecipeNames(recipes, mask, true);
370                        for(int r=0;r<listRecipes.size();r++)
371                        {
372                                final List<String> V=listRecipes.get(r);
373                                if(V.size()>0)
374                                {
375                                        final String item=replacePercent(V.get(RCP_FINALNAME),"");
376                                        final int level=CMath.s_int(V.get(RCP_LEVEL));
377                                        final String wood=getComponentDescription(mob,V,RCP_WOOD);
378                                        if(wood.length()>5)
379                                        {
380                                                if(toggler>1)
381                                                        buf.append("\n\r");
382                                                toggler=toggleTop;
383                                        }
384                                        if((level<=xlevel(mob))||allFlag)
385                                        {
386                                                buf.append(CMStrings.padRight(item,cols[0])+" "+CMStrings.padRight(""+(level),cols[1])+" "+CMStrings.padRightPreserve(""+wood,cols[2])+((toggler!=toggleTop)?" ":"\n\r"));
387                                                if(++toggler>toggleTop)
388                                                        toggler=1;
389                                        }
390                                }
391                        }
392                        if(toggler!=1)
393                                buf.append("\n\r");
394                        commonTell(mob,buf.toString());
395                        enhanceList(mob);
396                        return true;
397                }
398                else
399                if(((commands.get(0))).equalsIgnoreCase("learn"))
400                {
401                        return doLearnRecipe(mob, commands, givenTarget, auto, asLevel);
402                }
403                else
404                if(str.equalsIgnoreCase("scan"))
405                        return publicScan(mob,commands);
406                else
407                if(str.equalsIgnoreCase("mend"))
408                {
409                        buildingI=null;
410                        activity = CraftingActivity.CRAFTING;
411                        messedUp=false;
412                        final Vector<String> newCommands=CMParms.parse(CMParms.combine(commands,1));
413                        buildingI=getTargetItemFavorMOB(mob,mob.location(),givenTarget,newCommands,Wearable.FILTER_UNWORNONLY);
414                        if(!canMend(mob,buildingI,false))
415                                return false;
416                        activity = CraftingActivity.MENDING;
417                        if(!super.invoke(mob,commands,givenTarget,auto,asLevel))
418                                return false;
419                        startStr=L("<S-NAME> start(s) mending @x1.",buildingI.name());
420                        displayText=L("You are mending @x1",buildingI.name());
421                        verb=L("mending @x1",buildingI.name());
422                }
423                else
424                if(str.equalsIgnoreCase("refit"))
425                {
426                        buildingI=null;
427                        activity = CraftingActivity.CRAFTING;
428                        messedUp=false;
429                        final Vector<String> newCommands=CMParms.parse(CMParms.combine(commands,1));
430                        buildingI=getTargetItemFavorMOB(mob,mob.location(),givenTarget,newCommands,Wearable.FILTER_UNWORNONLY);
431                        if(buildingI==null)
432                                return false;
433                        if((buildingI.material()&RawMaterial.MATERIAL_MASK)!=RawMaterial.MATERIAL_LEATHER)
434                        {
435                                commonTell(mob,L("That's not made of leather.  That can't be refitted."));
436                                return false;
437                        }
438                        if(!(buildingI instanceof Armor))
439                        {
440                                commonTell(mob,L("You don't know how to refit that sort of thing."));
441                                return false;
442                        }
443                        if(buildingI.phyStats().height()==0)
444                        {
445                                commonTell(mob,L("@x1 is already the right size.",buildingI.name(mob)));
446                                return false;
447                        }
448                        activity = CraftingActivity.REFITTING;
449                        if(!super.invoke(mob,commands,givenTarget,auto,asLevel))
450                                return false;
451                        startStr=L("<S-NAME> start(s) refitting @x1.",buildingI.name());
452                        displayText=L("You are refitting @x1",buildingI.name());
453                        verb=L("refitting @x1",buildingI.name());
454                }
455                else
456                {
457                        buildingI=null;
458                        activity = CraftingActivity.CRAFTING;
459                        messedUp=false;
460                        aborted=false;
461                        int amount=-1;
462                        if((commands.size()>1)&&(CMath.isNumber(commands.get(commands.size()-1))))
463                        {
464                                amount=CMath.s_int(commands.get(commands.size()-1));
465                                commands.remove(commands.size()-1);
466                        }
467                        final String recipeName=CMParms.combine(commands,0);
468                        List<String> foundRecipe=null;
469                        final List<List<String>> matches=matchingRecipeNames(recipes,recipeName,true);
470                        for(int r=0;r<matches.size();r++)
471                        {
472                                final List<String> V=matches.get(r);
473                                if(V.size()>0)
474                                {
475                                        final String name=V.get(RCP_FINALNAME);
476                                        final int level=CMath.s_int(V.get(RCP_LEVEL));
477                                        if(level<=xlevel(mob))
478                                        {
479                                                final int x=name.indexOf(' ');
480                                                final Stage stage=(Stage)CMath.s_valueOf(Stage.class, name.substring(0,x));
481                                                if(stage == null)
482                                                        multiplier=1;
483                                                else
484                                                        multiplier=stage.multiplier;
485                                                foundRecipe=V;
486                                                recipeLevel=level;
487                                                break;
488                                        }
489                                }
490                        }
491                        if(foundRecipe==null)
492                        {
493                                commonTell(mob,L("You don't know how to make a '@x1'.  Try \"mleatherwork list\" for a list.",recipeName));
494                                return false;
495                        }
496
497                        final String woodRequiredStr = foundRecipe.get(RCP_WOOD);
498                        final int[] compData = new int[CF_TOTAL];
499                        final String realRecipeName=replacePercent(foundRecipe.get(RCP_FINALNAME),"");
500                        final List<Object> componentsFoundList=getAbilityComponents(mob, woodRequiredStr, "make "+CMLib.english().startWithAorAn(realRecipeName),autoGenerate,compData,1);
501                        if(componentsFoundList==null)
502                                return false;
503                        int woodRequired=CMath.s_int(woodRequiredStr);
504                        woodRequired=adjustWoodRequired(woodRequired,mob);
505
506                        if(amount>woodRequired)
507                                woodRequired=amount;
508                        final int[] pm={RawMaterial.MATERIAL_LEATHER};
509                        final int[] pm1={RawMaterial.MATERIAL_METAL,RawMaterial.MATERIAL_MITHRIL};
510                        final String misctype=foundRecipe.get(RCP_MISCTYPE);
511                        bundling=misctype.equalsIgnoreCase("BUNDLE");
512                        final int[][] data=fetchFoundResourceData(mob,
513                                                                                                        woodRequired,"leather",pm,
514                                                                                                        (multiplier==6)?1:0,
515                                                                                                        (multiplier==6)?"metal":null,
516                                                                                                        (multiplier==6)?pm1:null,
517                                                                                                        bundling,
518                                                                                                        autoGenerate,
519                                                                                                        enhancedTypes);
520                        if(data==null)
521                                return false;
522                        fixDataForComponents(data,woodRequiredStr,(autoGenerate>0) && (woodRequired==0),componentsFoundList, 1);
523                        woodRequired=data[0][FOUND_AMT];
524                        if(!super.invoke(mob,commands,givenTarget,auto,asLevel))
525                                return false;
526                        final MaterialLibrary.DeadResourceRecord deadMats;
527                        if((componentsFoundList.size() > 0)||(autoGenerate>0))
528                                deadMats = new MaterialLibrary.DeadResourceRecord();
529                        else
530                                deadMats = CMLib.materials().destroyResources(mob.location(),woodRequired,data[0][FOUND_CODE],data[1][FOUND_CODE],null,null);
531                        final MaterialLibrary.DeadResourceRecord deadComps = CMLib.ableComponents().destroyAbilityComponents(componentsFoundList);
532                        final int lostValue=autoGenerate>0?0:(deadMats.lostValue + deadComps.lostValue);
533                        buildingI=CMClass.getItem(foundRecipe.get(RCP_CLASSTYPE));
534                        if(buildingI==null)
535                        {
536                                commonTell(mob,L("There's no such thing as a @x1!!!",foundRecipe.get(RCP_CLASSTYPE)));
537                                return false;
538                        }
539                        duration=getDuration(multiplier*CMath.s_int(foundRecipe.get(RCP_TICKS)),mob,30,4);
540                        buildingI.setMaterial(super.getBuildingMaterial(woodRequired, data, compData));
541                        String itemName=determineFinalName(foundRecipe.get(RCP_FINALNAME),buildingI.material(),deadMats,deadComps);
542                        if(bundling)
543                                itemName="a "+woodRequired+"# "+itemName;
544                        else
545                        if(itemName.endsWith("s"))
546                                itemName="some "+itemName;
547                        else
548                                itemName=CMLib.english().startWithAorAn(itemName);
549                        buildingI.setName(itemName);
550                        startStr=L("<S-NAME> start(s) making @x1.",buildingI.name());
551                        displayText=L("You are making @x1",buildingI.name());
552                        verb=L("making @x1",buildingI.name());
553                        buildingI.setDisplayText(L("@x1 lies here",itemName));
554                        buildingI.setDescription(itemName+". ");
555                        buildingI.basePhyStats().setWeight(getStandardWeight(woodRequired+compData[CF_AMOUNT],bundling));
556                        buildingI.setBaseValue(CMath.s_int(foundRecipe.get(RCP_VALUE))*multiplier);
557                        setBrand(mob, buildingI);
558                        final int hardness=RawMaterial.CODES.HARDNESS(buildingI.material())-2;
559                        buildingI.basePhyStats().setLevel(CMath.s_int(foundRecipe.get(RCP_LEVEL))+hardness);
560                        final int capacity=CMath.s_int(foundRecipe.get(RCP_CAPACITY));
561                        final long canContain=getContainerType(foundRecipe.get(RCP_CONTAINMASK));
562                        int armordmg=CMath.s_int(foundRecipe.get(RCP_ARMORDMG));
563                        if(armordmg!=0)
564                                armordmg=armordmg+(multiplier-1);
565                        if(bundling)
566                                buildingI.setBaseValue(lostValue);
567                        final String spell=(foundRecipe.size()>RCP_SPELL)?foundRecipe.get(RCP_SPELL).trim():"";
568                        addSpells(buildingI,spell,deadMats.lostProps,deadComps.lostProps);
569                        if(buildingI instanceof Wand)
570                        {
571                                if(foundRecipe.get(RCP_CAPACITY).trim().length()>0)
572                                        ((Wand)buildingI).setMaxUses(capacity);
573                        }
574                        else
575                        if(buildingI instanceof Weapon)
576                        {
577                                ((Weapon)buildingI).setRawLogicalAnd((capacity>1));
578                        }
579                        if(buildingI instanceof Weapon)
580                        {
581                                ((Weapon)buildingI).basePhyStats().setAttackAdjustment(baseYield()+abilityCode()+(hardness*5)-1);
582                                ((Weapon)buildingI).setWeaponClassification(Weapon.CLASS_FLAILED);
583                                setWeaponTypeClass((Weapon)buildingI,misctype,Weapon.TYPE_SLASHING);
584                                buildingI.basePhyStats().setDamage(armordmg+hardness);
585                                ((Weapon)buildingI).setRawProperLocationBitmap(Wearable.WORN_WIELD|Wearable.WORN_HELD);
586                        }
587                        if((buildingI instanceof Armor)&&(!(buildingI instanceof FalseLimb)))
588                        {
589                                if((capacity>0)&&(buildingI instanceof Container))
590                                {
591                                        ((Container)buildingI).setCapacity(capacity+woodRequired);
592                                        ((Container)buildingI).setContainTypes(canContain);
593                                }
594                                ((Armor)buildingI).basePhyStats().setArmor(0);
595                                if(armordmg!=0)
596                                        ((Armor)buildingI).basePhyStats().setArmor(armordmg+(baseYield()+abilityCode()-1)+hardness);
597                                setWearLocation(buildingI,misctype,0);
598                        }
599                        if(buildingI instanceof Drink)
600                        {
601                                if(CMLib.flags().isGettable(buildingI))
602                                {
603                                        ((Drink)buildingI).setLiquidRemaining(0);
604                                        ((Drink)buildingI).setLiquidHeld(capacity*50);
605                                        ((Drink)buildingI).setThirstQuenched(250);
606                                        if((capacity*50)<250)
607                                                ((Drink)buildingI).setThirstQuenched(capacity*50);
608                                }
609                        }
610                        buildingI.recoverPhyStats();
611                        buildingI.text();
612                        buildingI.recoverPhyStats();
613                }
614
615                messedUp=!proficiencyCheck(mob,0,auto);
616
617                if(bundling)
618                {
619                        messedUp=false;
620                        duration=1;
621                        verb=L("bundling @x1",RawMaterial.CODES.NAME(buildingI.material()).toLowerCase());
622                        startStr=L("<S-NAME> start(s) @x1.",verb);
623                        displayText=L("You are @x1",verb);
624                }
625
626                if(autoGenerate>0)
627                {
628                        crafted.add(buildingI);
629                        return true;
630                }
631
632                final CMMsg msg=CMClass.getMsg(mob,buildingI,this,getActivityMessageType(),startStr);
633                if(mob.location().okMessage(mob,msg))
634                {
635                        mob.location().send(mob,msg);
636                        buildingI=(Item)msg.target();
637                        beneficialAffect(mob,mob,asLevel,duration);
638                        enhanceItem(mob,buildingI,recipeLevel,enhancedTypes);
639                }
640                else
641                if(bundling)
642                {
643                        messedUp=false;
644                        aborted=false;
645                        unInvoke();
646                }
647                return true;
648        }
