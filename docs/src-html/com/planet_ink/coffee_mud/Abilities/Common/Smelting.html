111        @Override
112        public void unInvoke()
113        {
114                if(canBeUninvoked())
115                {
116                        if(affected instanceof MOB)
117                        {
118                                final MOB mob=(MOB)affected;
119                                if((buildingI!=null)&&(!aborted))
120                                {
121                                        if(messedUp)
122                                        {
123                                                if(activity == CraftingActivity.LEARNING)
124                                                {
125                                                        commonEmote(mob,L("<S-NAME> fail(s) to learn how to smelt @x1.",buildingI.name()));
126                                                        buildingI.destroy();
127                                                }
128                                                else
129                                                        commonEmote(mob,L("<S-NAME> mess(es) up smelting @x1.",buildingI.name()));
130                                        }
131                                        else
132                                        {
133                                                if(activity==CraftingActivity.LEARNING)
134                                                {
135                                                        deconstructRecipeInto(mob, buildingI, recipeHolder );
136                                                        buildingI.destroy();
137                                                }
138                                                else
139                                                {
140                                                        dropAWinner(mob,buildingI);
141                                                        CMLib.achievements().possiblyBumpAchievement(mob, AchievementLibrary.Event.CRAFTING, 1, this);
142                                                }
143                                        }
144                                }
145                                buildingI=null;
146                                activity = CraftingActivity.CRAFTING;
147                        }
148                }
149                super.unInvoke();
150        }

163        @Override
164        public boolean mayICraft(final Item I)
165        {
166                if(I==null)
167                        return false;
168                if(!super.mayBeCrafted(I))
169                        return false;
170                if(((I.material()&RawMaterial.MATERIAL_MASK)!=RawMaterial.MATERIAL_METAL)
171                &&((I.material()&RawMaterial.MATERIAL_MASK)!=RawMaterial.MATERIAL_MITHRIL))
172                        return false;
173                if(!(I instanceof RawMaterial))
174                        return false;
175                if(((RawMaterial)I).getSubType().length()==0)
176                        return false;
177                if(CMLib.flags().isDeadlyOrMaliciousEffect(I))
178                        return false;
179                if(!masterCraftCheck(I))
180                        return (isANativeItem(I.Name()));
181                if(I.baseGoldValue()>I.basePhyStats().level())
182                        return (isANativeItem(I.Name()));
183                return (isANativeItem(I.Name()));
184        }

218        @Override
219        protected boolean autoGenInvoke(final MOB mob, final List<String> commands, final Physical givenTarget, final boolean auto,
220                                                                 final int asLevel, final int autoGenerate, final boolean forceLevels, final List<Item> crafted)
221        {
222                if(super.checkStop(mob, commands))
223                        return true;
224
225                if(super.checkInfo(mob, commands))
226                        return true;
227
228                final PairVector<EnhancedExpertise,Integer> enhancedTypes=enhancedTypes(mob,commands);
229                int recipeLevel = 1;
230                randomRecipeFix(mob,addRecipes(mob,loadRecipes()),commands,autoGenerate);
231                if(commands.size()==0)
232                {
233                        commonTell(mob,L("Make what? Enter \"smelt list\" for a list, \"smelt info <item>\","
234                                                + " \"smelt learn <item>\", or \"smelt stop\" to cancel."));
235                        return false;
236                }
237                bundling=false;
238                if((!auto)
239                &&(commands.size()>0)
240                &&((commands.get(0)).equalsIgnoreCase("bundle")))
241                {
242                        bundling=true;
243                        if(super.invoke(mob,commands,givenTarget,auto,asLevel))
244                                return super.bundle(mob,commands);
245                        return false;
246                }
247                final List<List<String>> recipes=addRecipes(mob,loadRecipes());
248                final String str=commands.get(0);
249                String startStr=null;
250                int duration=4;
251                if(str.equalsIgnoreCase("list"))
252                {
253                        String mask=CMParms.combine(commands,1);
254                        boolean allFlag=false;
255                        if(mask.equalsIgnoreCase("all"))
256                        {
257                                allFlag=true;
258                                mask="";
259                        }
260                        final StringBuffer buf=new StringBuffer("");
261                        final int[] cols={
262                                CMLib.lister().fixColWidth(25,mob.session()),
263                                CMLib.lister().fixColWidth(3,mob.session()),
264                                CMLib.lister().fixColWidth(45,mob.session())
265                        };
266                        buf.append(CMStrings.padRight(L("Item"),cols[0])+" "+CMStrings.padRight(L("Lvl"),cols[1])+" "+CMStrings.padRight(L("Resources"),cols[2]));
267                        buf.append("\n\r");
268                        final List<List<String>> listRecipes=((mask.length()==0) || mask.equalsIgnoreCase("all")) ? recipes : super.matchingRecipeNames(recipes, mask, true);
269                        for(int r=0;r<listRecipes.size();r++)
270                        {
271                                final List<String> V=listRecipes.get(r);
272                                if(V.size()>0)
273                                {
274                                        final String item=replacePercent(V.get(RCP_FINALNAME),"");
275                                        final int level=CMath.s_int(V.get(RCP_LEVEL));
276                                        String wood=getComponentDescription(mob,V,RCP_WOOD);
277                                        if(wood.length()<5)
278                                                wood=L("@x1 pounds of metal",wood);
279                                        if(wood.length()>cols[2])
280                                        {
281                                                final int x=wood.lastIndexOf(' ',cols[2]);
282                                                wood=wood.substring(0,x)+"\n\r"+CMStrings.repeat(' ',cols[0]+cols[1]+2)+wood.substring(x+1);
283                                        }
284                                        if((level<=xlevel(mob))||allFlag)
285                                                buf.append(CMStrings.padRight(item,cols[0])+" "+CMStrings.padRight(""+level,cols[1])+" "+wood+"\n\r");
286                                }
287                        }
288                        commonTell(mob,buf.toString());
289                        enhanceList(mob);
290                        return true;
291                }
292                else
293                if(((commands.get(0))).equalsIgnoreCase("learn"))
294                {
295                        return doLearnRecipe(mob, commands, givenTarget, auto, asLevel);
296                }
297                else
298                {
299                        buildingI=null;
300                        activity = CraftingActivity.CRAFTING;
301                        messedUp=false;
302                        aborted=false;
303                        int amount=-1;
304                        if((commands.size()>1)&&(CMath.isNumber(commands.get(commands.size()-1))))
305                        {
306                                amount=CMath.s_int(commands.get(commands.size()-1));
307                                commands.remove(commands.size()-1);
308                        }
309                        final String recipeName=CMParms.combine(commands,0);
310                        List<String> foundRecipe=null;
311                        final List<List<String>> matches=matchingRecipeNames(recipes,recipeName,true);
312                        for(int r=0;r<matches.size();r++)
313                        {
314                                final List<String> V=matches.get(r);
315                                if(V.size()>0)
316                                {
317                                        final int level=CMath.s_int(V.get(RCP_LEVEL));
318                                        if((autoGenerate>0)||(level<=xlevel(mob)))
319                                        {
320                                                foundRecipe=V;
321                                                recipeLevel=level;
322                                                break;
323                                        }
324                                }
325                        }
326                        if(foundRecipe==null)
327                        {
328                                commonTell(mob,L("You don't know how to smelt '@x1'.  Try \"@x2 list\" for a list.",recipeName,triggerStrings()[0].toLowerCase()));
329                                return false;
330                        }
331                        final String realRecipeName=replacePercent(foundRecipe.get(RCP_FINALNAME),"");
332                        final String woodRequiredStr = foundRecipe.get(RCP_WOOD);
333                        final int[] compData = new int[CF_TOTAL];
334                        final List<Object> componentsFoundList=getAbilityComponents(mob, woodRequiredStr, "smelt "+CMLib.english().startWithAorAn(realRecipeName),autoGenerate,compData,amount);
335                        if(componentsFoundList==null)
336                                return false;
337                        int woodRequired=CMath.s_int(woodRequiredStr);
338                        woodRequired=adjustWoodRequired(woodRequired,mob);
339                        if(amount>woodRequired)
340                                woodRequired*=amount;
341
342                        final int[] pm={RawMaterial.MATERIAL_METAL|RawMaterial.MATERIAL_MITHRIL};
343                        final int[][] data=fetchFoundResourceData(mob,
344                                                                                                        woodRequired,"metal",pm,
345                                                                                                        0,null,null,
346                                                                                                        bundling,
347                                                                                                        autoGenerate,
348                                                                                                        enhancedTypes);
349                        if(data==null)
350                                return false;
351                        fixDataForComponents(data,woodRequiredStr,(autoGenerate>0) && (woodRequired==0),componentsFoundList, amount);
352                        woodRequired=data[0][FOUND_AMT];
353                        if(!bundling)
354                        {
355                                fireRequired=true;
356                                final Item fire=getRequiredFire(mob,autoGenerate);
357                                if(fire==null)
358                                        return false;
359                        }
360                        else
361                                fireRequired=false;
362
363                        if(!super.invoke(mob,commands,givenTarget,auto,asLevel))
364                                return false;
365                        final MaterialLibrary.DeadResourceRecord deadMats;
366                        if((componentsFoundList.size() > 0)||(autoGenerate>0))
367                                deadMats = new MaterialLibrary.DeadResourceRecord();
368                        else
369                                deadMats = CMLib.materials().destroyResources(mob.location(),woodRequired,data[0][FOUND_CODE],data[1][FOUND_CODE],null,null);
370                        final MaterialLibrary.DeadResourceRecord deadComps = CMLib.ableComponents().destroyAbilityComponents(componentsFoundList);
371                        final int lostValue=autoGenerate>0?0:(deadMats.lostValue + deadComps.lostValue);
372                        buildingI=CMClass.getItem(foundRecipe.get(RCP_CLASSTYPE));
373                        if(buildingI==null)
374                        {
375                                commonTell(mob,L("There's no such thing as @x1!!!",foundRecipe.get(RCP_CLASSTYPE)));
376                                return false;
377                        }
378                        duration=getDuration(CMath.s_int(foundRecipe.get(RCP_TICKS)),mob,CMath.s_int(foundRecipe.get(RCP_LEVEL)),4);
379                        final String materialName=foundRecipe.get(RCP_FINALRSC);
380                        final String subType=foundRecipe.get(RCP_SUBTYPE);
381                        int resourceType = super.getBuildingMaterial(woodRequired, data, compData);
382                        if(materialName.length()>0)
383                        {
384                                int resource=RawMaterial.CODES.FIND_CaseSensitive(materialName);
385                                if(resource < 0)
386                                        resource=RawMaterial.CODES.FIND_IgnoreCase(materialName);
387                                if(resource < 0)
388                                        resource=RawMaterial.CODES.FIND_StartsWith(materialName);
389                                if(resource > 0)
390                                        resourceType=resource;
391                        }
392                        buildingI.setMaterial(resourceType);
393                        String itemName=determineFinalName(foundRecipe.get(RCP_FINALNAME),buildingI.material(),deadMats,deadComps);
394                        if(bundling)
395                                itemName="a "+woodRequired+"# "+itemName;
396                        else
397                        if(itemName.endsWith("s"))
398                                itemName="some "+itemName;
399                        else
400                                itemName="a pound of "+itemName;
401                        buildingI.setName(itemName);
402                        buildingI.setDisplayText(L("@x1 lies here",itemName));
403                        buildingI.setDescription(itemName+". ");
404                        buildingI.basePhyStats().setWeight(getStandardWeight(compData[CF_AMOUNT],bundling));
405                        buildingI.setBaseValue(CMath.s_int(foundRecipe.get(RCP_VALUE)));
406                        buildingI.basePhyStats().setLevel(CMath.s_int(foundRecipe.get(RCP_LEVEL)));
407                        final String spell=(foundRecipe.size()>RCP_SPELL)?foundRecipe.get(RCP_SPELL).trim():"";
408                        if(bundling)
409                                buildingI.setBaseValue(lostValue);
410                        addSpells(buildingI,spell,deadMats.lostProps,deadComps.lostProps);
411                        if(buildingI instanceof RawMaterial)
412                        {
413                                ((RawMaterial)buildingI).setSubType(subType.toUpperCase().trim());
414                                buildingI.setSecretIdentity(itemName);
415                        }
416                        else
417                                setBrand(mob, buildingI);
418                        buildingI.recoverPhyStats();
419                        CMLib.materials().adjustResourceName(buildingI);
420                        buildingI.text();
421                        buildingI.recoverPhyStats();
422                        startStr=L("<S-NAME> start(s) smelting @x1.",buildingI.name());
423                        displayText=L("You are smelting @x1",buildingI.name());
424                        verb=L("smelting @x1",buildingI.name());
425                        playSound="sizzling.wav";
426                }
427
428                messedUp=!proficiencyCheck(mob,0,auto);
429
430                if(bundling)
431                {
432                        messedUp=false;
433                        duration=1;
434                        verb=L("bundling @x1",RawMaterial.CODES.NAME(buildingI.material()).toLowerCase());
435                        startStr=L("<S-NAME> start(s) @x1.",verb);
436                        displayText=L("You are @x1",verb);
437                }
438
439                if(autoGenerate>0)
440                {
441                        crafted.add(buildingI);
442                        return true;
443                }
444
445                final CMMsg msg=CMClass.getMsg(mob,buildingI,this,getActivityMessageType(),startStr);
446                if(mob.location().okMessage(mob,msg))
447                {
448                        mob.location().send(mob,msg);
449                        buildingI=(Item)msg.target();
450                        beneficialAffect(mob,mob,asLevel,duration);
451                        enhanceItem(mob,buildingI,recipeLevel,enhancedTypes);
452                }
453                else
454                if(bundling)
455                {
456                        messedUp=false;
457                        aborted=false;
458                        unInvoke();
459                }
460                return true;
461        }
