185        @Override
186        public boolean isSold(int mask)
187        {
188                if(mask==0)
189                        return whatIsSoldMask==0;
190                if((whatIsSoldMask&255)==mask)
191                        return true;
192                return CMath.bset(whatIsSoldMask>>8, CMath.pow(2,mask-1));
193        }
194
195        @Override
196        public void addSoldType(int mask)
197        {
198                if(mask==0)
199                        whatIsSoldMask=0;
200                else
201                {
202                        if((whatIsSoldMask>0)&&(whatIsSoldMask<256))
203                                whatIsSoldMask=(CMath.pow(2,whatIsSoldMask-1)<<8);
204
205                        for(int c=0;c<ShopKeeper.DEAL_CONFLICTS.length;c++)
206                        {
207                                for(int c1=0;c1<ShopKeeper.DEAL_CONFLICTS[c].length;c1++)
208                                {
209                                        if(ShopKeeper.DEAL_CONFLICTS[c][c1]==mask)
210                                        {
211                                                for(c1=0;c1<ShopKeeper.DEAL_CONFLICTS[c].length;c1++)
212                                                        if((ShopKeeper.DEAL_CONFLICTS[c][c1]!=mask)
213                                                        &&(isSold(ShopKeeper.DEAL_CONFLICTS[c][c1])))
214                                                                addSoldType(-ShopKeeper.DEAL_CONFLICTS[c][c1]);
215                                                break;
216                                        }
217                                }
218                        }
219
220                        if(mask>0)
221                                whatIsSoldMask|=(CMath.pow(2,mask-1)<<8);
222                        else
223                                whatIsSoldMask=CMath.unsetb(whatIsSoldMask,(CMath.pow(2,(-mask)-1)<<8));
224                }
225        }

356        public MOB deriveMerchant(MOB roomHelper)
357        {
358                if(affected ==null)
359                        return null;
360                if(affected instanceof MOB)
361                        return (MOB)affected;
362                if(affected instanceof Item)
363                {
364                        if(((Item)affected).owner() instanceof MOB)
365                                return (MOB)((Item)affected).owner();
366                        if(CMLib.flags().isGettable((Item)affected))
367                                return null;
368                }
369                Room room=CMLib.map().roomLocation(affected);
370                if((affected instanceof Area)&&(roomHelper!=null))
371                        room=roomHelper.location();
372                if(room==null)
373                        return null;
374                if(staticMOB==null)
375                {
376                        staticMOB=CMClass.getMOB("StdMOB");
377                        if((affected instanceof Room)
378                        ||(affected instanceof Exit))
379                                staticMOB.setName(L("the shopkeeper"));
380                        else
381                        if(affected instanceof Area)
382                                staticMOB.setName(L("the shop"));
383                        else
384                                staticMOB.setName(affected.Name());
385                }
386                staticMOB.setStartRoom(room);
387                staticMOB.setLocation(room);
388                if(finalBudget() != null)
389                {
390                        if( CMLib.beanCounter().getTotalAbsoluteNativeValue( staticMOB ) < finalBudget().first.longValue() )
391                                staticMOB.setMoney(finalBudget().first.intValue());
392                }
393                return staticMOB;
394        }

396        @Override
397        public boolean okMessage(final Environmental myHost, final CMMsg msg)
398        {
399                final MOB merchantM=deriveMerchant(msg.source());
400                if(merchantM==null)
401                        return super.okMessage(myHost,msg);
402
403                final MOB shopperM=msg.source();
404                if((msg.source()==merchantM)
405                &&(msg.targetMinor()==CMMsg.TYP_GET)
406                &&(msg.target() instanceof Item))
407                {
408                        final Item newitem=(Item)msg.target();
409                        if((newitem.numberOfItems()>(merchantM.maxItems()-(merchantM.numItems()+shop.totalStockSizeIncludingDuplicates())))
410                        &&(!merchantM.isMine(this)))
411                        {
412                                merchantM.tell(L("You can't carry that many items."));
413                                return false;
414                        }
415                }
416
417                if(msg.amITarget(merchantM)||(msg.amITarget(affected)))
418                {
419                        switch(msg.targetMinor())
420                        {
421                        case CMMsg.TYP_VALUE:
422                        case CMMsg.TYP_SELL:
423                        {
424                                if(!merchantM.isMonster())
425                                {
426                                        shopperM.tell(shopperM,null,null,L("You'll have to talk to <S-NAME> about that."));
427                                        return false;
428                                }
429                                if(!CMLib.coffeeShops().ignoreIfNecessary(msg.source(),finalIgnoreMask(),merchantM))
430                                        return false;
431                                final double budgetRemaining=CMLib.beanCounter().getTotalAbsoluteValue(merchantM,CMLib.beanCounter().getCurrency(merchantM));
432                                final double budgetMax=budgetRemaining*100;
433                                if(CMLib.coffeeShops().standardSellEvaluation(merchantM,msg.source(),msg.tool(),this,budgetRemaining,budgetMax,msg.targetMinor()==CMMsg.TYP_SELL))
434                                        return super.okMessage(myHost,msg);
435                                return false;
436                        }
437                        case CMMsg.TYP_BUY:
438                        case CMMsg.TYP_VIEW:
439                        {
440                                if(!CMLib.coffeeShops().ignoreIfNecessary(msg.source(),finalIgnoreMask(),merchantM))
441                                        return false;
442                                if((msg.targetMinor()==CMMsg.TYP_BUY)&&(msg.tool()!=null)&&(!msg.tool().okMessage(myHost,msg)))
443                                        return false;
444                                if(CMLib.coffeeShops().standardBuyEvaluation(merchantM,msg.source(),msg.tool(),this,msg.targetMinor()==CMMsg.TYP_BUY))
445                                        return super.okMessage(myHost,msg);
446                                return false;
447                        }
448                        case CMMsg.TYP_LIST:
449                                CMLib.coffeeShops().ignoreIfNecessary(msg.source(),finalIgnoreMask(),merchantM);
450                                break;
451                        default:
452                                break;
453                        }
454                }
455                else
456                if(msg.amISource(merchantM)&&(msg.sourceMinor()==CMMsg.TYP_DEATH))
457                {
458                        Item I=(Item)getShop().removeStock("all",merchantM);
459                        while(I!=null)
460                        {
461                                merchantM.addItem(I);
462                                I=(Item)getShop().removeStock("all",merchantM);
463                        }
464                        merchantM.recoverPhyStats();
465                }
466                return super.okMessage(myHost,msg);
467        }

469        public boolean putUpForSale(MOB source, MOB merchantM, Environmental tool)
470        {
471                if((tool!=null)
472                &&(!tool.ID().endsWith("ClanApron"))
473                &&(merchantM.isMonster())
474                &&((CMSecurity.isAllowed(source,merchantM.location(),CMSecurity.SecFlag.ORDER)
475                        ||(CMLib.law().doesHavePriviledgesHere(source,merchantM.getStartRoom()))
476                        ||(CMSecurity.isAllowed(source,merchantM.location(),CMSecurity.SecFlag.CMDMOBS)&&(merchantM.isMonster()))
477                        ||(CMSecurity.isAllowed(source,merchantM.location(),CMSecurity.SecFlag.CMDROOMS)&&(merchantM.isMonster())))
478                        ||((CMLib.law().getLegalBehavior(merchantM.getStartRoom())!=null)
479                                &&(source.getClanRole(CMLib.law().getLegalBehavior(merchantM.getStartRoom()).rulingOrganization())!=null)))
480                &&((doISellThis(tool))||(isSold(DEAL_INVENTORYONLY))))
481                {
482                        CMLib.commands().postSay(merchantM,source,L("OK, I will now sell @x1.",tool.name()),false,false);
483                        getShop().addStoreInventory(tool,1,-1);
484                        if(affected instanceof Area)
485                                CMLib.database().DBUpdateArea(affected.Name(),(Area)affected);
486                        else
487                        if(affected instanceof Exit)
488                                CMLib.database().DBUpdateExits(merchantM.location());
489                        else
490                        if(affected instanceof Room)
491                                CMLib.database().DBUpdateRoom(merchantM.location());
492                        return true;
493                }
494                return false;
495        }

497        public boolean canPossiblyVend(Environmental E, Environmental what)
498        {
499                if(!(what instanceof Item))
500                        return false;
501                final Item whatI=(Item)what;
502                if((E instanceof Container)
503                &&(!(((Container)E).owner() instanceof MOB))
504                &&(((Container)E).canContain(whatI))
505                &&(((Container)E).capacity()>whatI.phyStats().weight()))
506                        return true;
507                return false;
508        }

510        @Override
511        public void executeMsg(final Environmental myHost, final CMMsg msg)
512        {
513                final MOB merchantM=deriveMerchant(msg.source());
514                if(merchantM==null)
515                {
516                        super.executeMsg(myHost,msg);
517                        return;
518                }
519
520                if(msg.amITarget(merchantM)||(msg.amITarget(affected)))
521                {
522                        final MOB mob=msg.source();
523                        switch(msg.targetMinor())
524                        {
525                        case CMMsg.TYP_GIVE:
526                                if(!putUpForSale(msg.source(),merchantM,msg.tool()))
527                                        super.executeMsg(myHost,msg);
528                                break;
529                        case CMMsg.TYP_PUT:
530                                if((canPossiblyVend(affected,msg.tool()))
531                                &&(putUpForSale(msg.source(),merchantM,msg.tool())))
532                                        return;
533                                super.executeMsg(myHost,msg);
534                                break;
535                        case CMMsg.TYP_VALUE:
536                                super.executeMsg(myHost,msg);
537                                if(merchantM.isMonster())
538                                        CMLib.commands().postSay(merchantM,mob,L("I'll give you @x1 for @x2.",CMLib.beanCounter().nameCurrencyShort(merchantM,CMLib.coffeeShops().pawningPrice(merchantM,mob,msg.tool(),this, getShop()).absoluteGoldPrice),msg.tool().name()),true,false);
539                                break;
540                        case CMMsg.TYP_VIEW:
541                                super.executeMsg(myHost,msg);
542                                if((msg.tool() instanceof Physical)
543                                &&(getShop().doIHaveThisInStock(msg.tool().Name(),mob)))
544                                {
545
546                                        CMLib.commands().postSay(merchantM,msg.source(),L("Interested in @x1? Here is some information for you:\n\rLevel @x2\n\rDescription: @x3",msg.tool().name(),""+((Physical)msg.tool()).phyStats().level(),msg.tool().description()),true,false);
547                                }
548                                break;
549                        case CMMsg.TYP_SELL: // sell TO -- this is a shopkeeper purchasing from a player
550                        {
551                                super.executeMsg(myHost,msg);
552                                CMLib.coffeeShops().transactPawn(merchantM,msg.source(),this,msg.tool());
553                                break;
554                        }
555                        case CMMsg.TYP_BUY:
556                        {
557                                super.executeMsg(myHost,msg);
558                                final MOB mobFor=CMLib.coffeeShops().parseBuyingFor(msg.source(),msg.targetMessage());
559                                if((msg.tool()!=null)
560                                &&(getShop().doIHaveThisInStock("$"+msg.tool().Name()+"$",mobFor))
561                                &&(merchantM.location()!=null))
562                                {
563                                        final Environmental item=getShop().getStock("$"+msg.tool().Name()+"$",mobFor);
564                                        if(item!=null)
565                                                CMLib.coffeeShops().transactMoneyOnly(merchantM,msg.source(),this,item,!merchantM.isMonster());
566
567                                        final List<Environmental> products=getShop().removeSellableProduct("$"+msg.tool().Name()+"$",mobFor);
568                                        if(products.size()==0)
569                                                break;
570                                        final Environmental product=products.get(0);
571                                        if(product instanceof Item)
572                                        {
573                                                if(!CMLib.coffeeShops().purchaseItems((Item)product,products,merchantM,mobFor))
574                                                        return;
575                                        }
576                                        else
577                                        if(product instanceof MOB)
578                                        {
579                                                if(CMLib.coffeeShops().purchaseMOB((MOB)product,merchantM,this,mobFor))
580                                                {
581                                                        msg.modify(msg.source(),msg.target(),product,msg.sourceCode(),msg.sourceMessage(),msg.targetCode(),msg.targetMessage(),msg.othersCode(),msg.othersMessage());
582                                                        product.executeMsg(myHost,msg);
583                                                }
584                                        }
585                                        else
586                                        if(product instanceof Ability)
587                                                CMLib.coffeeShops().purchaseAbility((Ability)product,merchantM,this,mobFor);
588                                }
589                                break;
590                        }
591                        case CMMsg.TYP_LIST:
592                        {
593                                super.executeMsg(myHost,msg);
594                                final Vector<Environmental> inventory=new XVector<Environmental>(getShop().getStoreInventory());
595                                final String forMask=CMLib.coffeeShops().getListForMask(msg.targetMessage());
596                                final String s=CMLib.coffeeShops().getListInventory(merchantM,mob,inventory,0,this,forMask);
597                                if(s.length()>0)
598                                        mob.tell(s);
599                                break;
600                        }
601                        default:
602                                super.executeMsg(myHost,msg);
603                                break;
604                        }
605                }
606                else
607                if((msg.targetMinor()==CMMsg.TYP_DROP)
608                &&(myHost==affected)
609                &&((affected instanceof Room)
610                        ||(affected instanceof Exit)
611                        ||((affected instanceof Item)&&(!canPossiblyVend(affected,msg.tool()))))
612                &&(putUpForSale(msg.source(),merchantM,msg.target())))
613                        return;
614                else
615                if((msg.targetMinor()==CMMsg.TYP_THROW)
616                &&(myHost==affected)
617                &&(affected instanceof Area)
618                &&(msg.target() instanceof Room)
619                &&(((Room)msg.target()).domainType()==Room.DOMAIN_OUTDOORS_AIR)
620                &&(msg.source().location().getRoomInDir(Directions.UP)==msg.target())
621                &&(putUpForSale(msg.source(),merchantM,msg.tool())))
622                        return;
623                else
624                        super.executeMsg(myHost,msg);
625        }

627        @Override
628        public boolean invoke(MOB mob, List<String> commands, Physical givenTarget, boolean auto, int asLevel)
629        {
630                if(commands.size()==0)
631                {
632                        commonTell(mob,L("Market what? Enter \"market list\" for a list or \"market item value\" to sell something."));
633                        return false;
634                }
635                if(CMParms.combine(commands,0).equalsIgnoreCase("list"))
636                {
637                        final CMMsg msg=CMClass.getMsg(mob,mob,CMMsg.MSG_LIST,L("<S-NAME> review(s) <S-HIS-HER> inventory."));
638                        if(mob.location().okMessage(mob,msg))
639                                mob.location().send(mob,msg);
640                        return true;
641                }
642                if((commands.get(0)).equalsIgnoreCase("remove")
643                ||(commands.get(0)).equalsIgnoreCase("delete"))
644                {
645                        if(commands.size()==1)
646                        {
647                                commonTell(mob,L("Remove what item from the marketing list?"));
648                                return false;
649                        }
650                        final String itemName=CMParms.combine(commands,1);
651                        Item I=(Item)getShop().removeStock(itemName,mob);
652                        if(I==null)
653                        {
654                                commonTell(mob,L("'@x1' is not on the list.",itemName));
655                                return false;
656                        }
657                        final String iname=I.name();
658                        while(I!=null)
659                        {
660                                mob.addItem(I);
661                                I=(Item)getShop().removeStock(itemName,mob);
662                        }
663                        getShop().delAllStoreInventory(I);
664                        mob.recoverCharStats();
665                        mob.recoverPhyStats();
666                        mob.recoverMaxState();
667                        mob.tell(L("@x1 has been removed from your inventory list.",iname));
668                        return true;
669                }
670
671                Environmental target=null;
672                double val=-1;
673                if(commands.size()>1)
674                {
675                        final String s=commands.get(commands.size()-1);
676                        if(CMath.isInteger(s))
677                        {
678                                val=CMath.s_int( s );
679                                if(val>0)
680                                        commands.remove(s);
681                        }
682                        else
683                        {
684                                final long numberCoins=CMLib.english().numPossibleGold(mob,s);
685                                if(numberCoins>0)
686                                {
687                                        final String currency=CMLib.english().numPossibleGoldCurrency(mob,s);
688                                        final double denom=CMLib.english().numPossibleGoldDenomination(mob,currency,s);
689                                        if(denom>0.0)
690                                        {
691                                                val=CMath.mul(numberCoins,denom);
692                                                if(val>0)
693                                                        commands.remove(s);
694                                        }
695                                }
696                        }
697                }
698
699                String itemName=CMParms.combine(commands,0);
700                final Vector<Item> V=new Vector<Item>();
701                boolean allFlag=commands.get(0).equalsIgnoreCase("all");
702                if(itemName.toUpperCase().startsWith("ALL."))
703                {
704                        allFlag=true;
705                        itemName="ALL "+itemName.substring(4);
706                }
707                if(itemName.toUpperCase().endsWith(".ALL"))
708                {
709                        allFlag=true;
710                        itemName="ALL "+itemName.substring(0,itemName.length()-4);
711                }
712                int addendum=1;
713                String addendumStr="";
714                boolean doBugFix = true;
715                while(doBugFix || allFlag)
716                {
717                        doBugFix=false;
718                        final Item I=mob.fetchItem(null,Wearable.FILTER_UNWORNONLY,itemName+addendumStr);
719                        if(I==null)
720                                break;
721                        if(target==null)
722                                target=I;
723                        else
724                        if(!target.sameAs(I))
725                                break;
726                        if(CMLib.flags().canBeSeenBy(I,mob))
727                                V.addElement(I);
728                        addendumStr="."+(++addendum);
729                }
730
731                if(V.size()==0)
732                {
733                        commonTell(mob,L("You don't seem to be carrying '@x1'.",itemName));
734                        return false;
735                }
736
737                if((getShop().numberInStock(target)<=0)&&(val<=0))
738                {
739                        commonTell(mob,L("You failed to specify a price for '@x1'.",itemName));
740                        return false;
741                }
742
743                if(!super.invoke(mob,commands,givenTarget,auto,asLevel))
744                        return false;
745
746                if(!proficiencyCheck(mob,0,auto))
747                {
748                        commonTell(mob,target,null,L("You fail to put <T-NAME> up for sale."));
749                        return false;
750                }
751
752                final CMMsg msg=CMClass.getMsg(mob,target,CMMsg.MSG_SELL,L("<S-NAME> put(s) <T-NAME> up for sale."));
753                if(mob.location().okMessage(mob,msg))
754                {
755                        mob.location().send(mob,msg);
756                        for(int i=0;i<V.size();i++)
757                        {
758                                final Item I=V.elementAt(i);
759                                if(val<=0)
760                                        getShop().addStoreInventory(I);
761                                else
762                                        getShop().addStoreInventory(I,1,(int)Math.round(val));
763                                mob.delItem(I);
764                        }
765                }
766                mob.location().recoverRoomStats();
767                mob.recoverPhyStats();
768                return true;
769        }
