104        @Override
105        public void unInvoke()
106        {
107                if(canBeUninvoked())
108                {
109                        if(affected instanceof MOB)
110                        {
111                                final MOB mob=(MOB)affected;
112                                if((buildingI!=null)&&(!aborted)&&(!helping))
113                                {
114                                        if(messedUp)
115                                        {
116                                                if(activity == CraftingActivity.MENDING)
117                                                        messedUpCrafting(mob);
118                                                else
119                                                if(activity == CraftingActivity.LEARNING)
120                                                {
121                                                        commonEmote(mob,L("<S-NAME> fail(s) to learn how to make @x1.",buildingI.name()));
122                                                        buildingI.destroy();
123                                                }
124                                                else
125                                                        commonEmote(mob,L("<S-NAME> mess(es) up making @x1.",buildingI.name()));
126                                        }
127                                        else
128                                        {
129                                                if(activity == CraftingActivity.MENDING)
130                                                {
131                                                        buildingI.setUsesRemaining(100);
132                                                        CMLib.achievements().possiblyBumpAchievement(mob, AchievementLibrary.Event.MENDER, 1, this);
133                                                }
134                                                else
135                                                if(activity==CraftingActivity.LEARNING)
136                                                {
137                                                        deconstructRecipeInto(mob, buildingI, recipeHolder );
138                                                        buildingI.destroy();
139                                                }
140                                                else
141                                                {
142                                                        dropAWinner(mob,buildingI);
143                                                        CMLib.achievements().possiblyBumpAchievement(mob, AchievementLibrary.Event.CRAFTING, 1, this);
144                                                }
145                                        }
146                                }
147                                buildingI=null;
148                                activity = CraftingActivity.CRAFTING;
149                        }
150                }
151                super.unInvoke();
152        }

165        @Override
166        public boolean mayICraft(final Item I)
167        {
168                if(I==null)
169                        return false;
170                if(!super.mayBeCrafted(I))
171                        return false;
172                if((I.material()&RawMaterial.MATERIAL_MASK)!=RawMaterial.MATERIAL_WOODEN)
173                        return false;
174                if(I instanceof Ammunition)
175                        return true;
176                if(!(I instanceof Weapon))
177                        return false;
178                if((((Weapon)I).weaponClassification()==Weapon.CLASS_RANGED)
179                ||(((Weapon)I).weaponClassification()==Weapon.CLASS_THROWN))
180                        return true;
181                return false;
182        }

190        @Override
191        protected boolean canMend(MOB mob, Environmental E, boolean quiet)
192        {
193                if(!super.canMend(mob,E,quiet))
194                        return false;
195                if((!(E instanceof Item))
196                ||(!mayICraft((Item)E)))
197                {
198                        if(!quiet)
199                                commonTell(mob,L("That's not a @x1 item.",name()));
200                        return false;
201                }
202                return true;
203        }

241        @Override
242        protected boolean autoGenInvoke(final MOB mob, List<String> commands, Physical givenTarget, final boolean auto, 
243                                                                 final int asLevel, int autoGenerate, boolean forceLevels, List<Item> crafted)
244        {
245                if(super.checkStop(mob, commands))
246                        return true;
247
248                if(super.checkInfo(mob, commands))
249                        return true;
250                
251                final PairVector<EnhancedExpertise,Integer> enhancedTypes=enhancedTypes(mob,commands);
252                int recipeLevel = 1;
253                randomRecipeFix(mob,addRecipes(mob,loadRecipes()),commands,autoGenerate);
254                if(commands.size()==0)
255                {
256                        commonTell(mob,L("Make what? Enter \"@x1 list\" for a list, \"@x1 scan\", \"@x1 info <item>\", \"@x1 learn <item>\","
257                                                        + " \"@x1 mend <item>\", or \"@x1 stop\" to cancel.",commandWord()));
258                        return false;
259                }
260                if((!auto)
261                &&(commands.size()>0)
262                &&((commands.get(0)).equalsIgnoreCase("bundle")))
263                {
264                        bundling=true;
265                        if(super.invoke(mob,commands,givenTarget,auto,asLevel))
266                                return super.bundle(mob,commands);
267                        return false;
268                }
269                final List<List<String>> recipes=addRecipes(mob,loadRecipes());
270                final String str=commands.get(0);
271                String startStr=null;
272                bundling=false;
273                int duration=4;
274                if(str.equalsIgnoreCase("list"))
275                {
276                        String mask=CMParms.combine(commands,1);
277                        boolean allFlag=false;
278                        if(mask.equalsIgnoreCase("all"))
279                        {
280                                allFlag=true;
281                                mask="";
282                        }
283                        final int toggleTop=getNumberOfColumns();
284                        int toggler=1;
285                        final StringBuffer buf=new StringBuffer("");
286                        final int[] cols={
287                                CMLib.lister().fixColWidth(27,mob.session()),
288                                CMLib.lister().fixColWidth(3,mob.session()),
289                                CMLib.lister().fixColWidth(5,mob.session())
290                        };
291                        for(int r=0;r<toggleTop;r++)
292                                buf.append((r>0?" ":"")+CMStrings.padRight(L("Item"),cols[0])+" "+CMStrings.padRight(L("Lvl"),cols[1])+" "+CMStrings.padRight(L("Wood"),cols[2]));
293                        buf.append("\n\r");
294                        final List<List<String>> listRecipes=((mask.length()==0) || mask.equalsIgnoreCase("all")) ? recipes : super.matchingRecipeNames(recipes, mask, true);
295                        for(int r=0;r<listRecipes.size();r++)
296                        {
297                                final List<String> V=listRecipes.get(r);
298                                if(V.size()>0)
299                                {
300                                        final String item=replacePercent(V.get(RCP_FINALNAME),"");
301                                        final int level=CMath.s_int(V.get(RCP_LEVEL));
302                                        final String wood=getComponentDescription(mob,V,RCP_WOOD);
303                                        if(wood.length()>5)
304                                        {
305                                                if(toggler>1)
306                                                        buf.append("\n\r");
307                                                toggler=toggleTop;
308                                        }
309                                        if((level<=xlevel(mob))||allFlag)
310                                        {
311                                                buf.append(CMStrings.padRight(item,cols[0])+" "+
312                                                                        CMStrings.padRight(""+level,cols[1])+" "+
313                                                                        CMStrings.padRightPreserve(""+wood,cols[2])+
314                                                                        ((toggler!=toggleTop)?" ":"\n\r"));
315                                                if(++toggler>toggleTop)
316                                                        toggler=1;
317                                        }
318                                }
319                        }
320                        buf.append(L("\n\rSome items may require additional material."));
321                        commonTell(mob,buf.toString());
322                        enhanceList(mob);
323                        return true;
324                }
325                else
326                if(((commands.get(0))).equalsIgnoreCase("learn"))
327                {
328                        return doLearnRecipe(mob, commands, givenTarget, auto, asLevel);
329                }
330                else
331                if(str.equalsIgnoreCase("scan"))
332                        return publicScan(mob,commands);
333                else
334                if(str.equalsIgnoreCase("mend"))
335                {
336                        buildingI=null;
337                        activity = CraftingActivity.CRAFTING;
338                        messedUp=false;
339                        final Vector<String> newCommands=CMParms.parse(CMParms.combine(commands,1));
340                        buildingI=getTargetItemFavorMOB(mob,mob.location(),givenTarget,newCommands,Wearable.FILTER_UNWORNONLY);
341                        if(!canMend(mob,buildingI,false))
342                                return false;
343                        activity = CraftingActivity.MENDING;
344                        if(!super.invoke(mob,commands,givenTarget,auto,asLevel))
345                                return false;
346                        startStr=L("<S-NAME> start(s) mending @x1.",buildingI.name());
347                        displayText=L("You are mending @x1",buildingI.name());
348                        verb=L("mending @x1",buildingI.name());
349                }
350                else
351                {
352                        buildingI=null;
353                        activity = CraftingActivity.CRAFTING;
354                        messedUp=false;
355                        aborted=false;
356                        int amount=-1;
357                        if((commands.size()>1)&&(CMath.isNumber(commands.get(commands.size()-1))))
358                        {
359                                amount=CMath.s_int(commands.get(commands.size()-1));
360                                commands.remove(commands.size()-1);
361                        }
362                        final String recipeName=CMParms.combine(commands,0);
363                        List<String> foundRecipe=null;
364                        final List<List<String>> matches=matchingRecipeNames(recipes,recipeName,true);
365                        for(int r=0;r<matches.size();r++)
366                        {
367                                final List<String> V=matches.get(r);
368                                if(V.size()>0)
369                                {
370                                        final int level=CMath.s_int(V.get(RCP_LEVEL));
371                                        if((autoGenerate>0)||(level<=xlevel(mob)))
372                                        {
373                                                foundRecipe=V;
374                                                recipeLevel=level;
375                                                break;
376                                        }
377                                }
378                        }
379                        if(foundRecipe==null)
380                        {
381                                commonTell(mob,L("You don't know how to make a '@x1'.  Try \"@x2 list\" for a list.",recipeName,commandWord()));
382                                return false;
383                        }
384
385                        final String woodRequiredStr = foundRecipe.get(RCP_WOOD);
386                        final int[] compData = new int[CF_TOTAL];
387                        final String realRecipeName=replacePercent(foundRecipe.get(RCP_FINALNAME),"");
388                        final List<Object> componentsFoundList=getAbilityComponents(mob, woodRequiredStr, "make "+CMLib.english().startWithAorAn(realRecipeName),autoGenerate,compData,1);
389                        if(componentsFoundList==null)
390                                return false;
391                        int woodRequired=CMath.s_int(woodRequiredStr);
392                        woodRequired=adjustWoodRequired(woodRequired,mob);
393
394                        if((amount>woodRequired)&&(woodRequired>0))
395                                woodRequired=amount;
396                        final String otherRequired=getOtherRscRequired(foundRecipe.get(RCP_EXTRAREQ));
397                        final int otherAmtRequired=getOtherRscAmtRequired(mob,foundRecipe.get(RCP_EXTRAREQ));
398                        final int[] pm={RawMaterial.MATERIAL_WOODEN};
399                        final int[][] data=fetchFoundResourceData(mob,
400                                                                                                          woodRequired,"wood",pm,
401                                                                                                          otherAmtRequired,otherRequired,null,
402                                                                                                          false,
403                                                                                                          autoGenerate,
404                                                                                                          enhancedTypes);
405                        if(data==null)
406                                return false;
407                        fixDataForComponents(data,woodRequiredStr,(autoGenerate>0) && (woodRequired==0),componentsFoundList, 1);
408                        woodRequired=data[0][FOUND_AMT];
409                        if(((data[1][FOUND_CODE]&RawMaterial.MATERIAL_MASK)==RawMaterial.MATERIAL_METAL)
410                        ||((data[1][FOUND_CODE]&RawMaterial.MATERIAL_MASK)==RawMaterial.MATERIAL_MITHRIL))
411                        {
412                                Item fire=null;
413                                if(autoGenerate<=0)
414                                {
415                                        for(int i=0;i<mob.location().numItems();i++)
416                                        {
417                                                final Item I2=mob.location().getItem(i);
418                                                if((I2!=null)&&(I2.container()==null)&&(CMLib.flags().isOnFire(I2)))
419                                                {
420                                                        fire=I2;
421                                                        break;
422                                                }
423                                        }
424                                        if((fire==null)||(!mob.location().isContent(fire)))
425                                        {
426                                                commonTell(mob,L("You'll need to build a fire first."));
427                                                return false;
428                                        }
429                                }
430                        }
431                        final String spell=(foundRecipe.size()>RCP_SPELL)?foundRecipe.get(RCP_SPELL).trim():"";
432                        bundling=spell.equalsIgnoreCase("BUNDLE");
433                        if(!super.invoke(mob,commands,givenTarget,auto,asLevel))
434                                return false;
435                        final MaterialLibrary.DeadResourceRecord deadMats;
436                        if((componentsFoundList.size() > 0)||(autoGenerate>0))
437                                deadMats = new MaterialLibrary.DeadResourceRecord();
438                        else
439                                deadMats = CMLib.materials().destroyResources(mob.location(),woodRequired,data[0][FOUND_CODE],0,null,null);
440                        final MaterialLibrary.DeadResourceRecord deadComps = CMLib.ableComponents().destroyAbilityComponents(componentsFoundList);
441                        final int lostValue=autoGenerate>0?0:(deadMats.lostValue + deadComps.lostValue);
442                        buildingI=CMClass.getItem(foundRecipe.get(RCP_CLASSTYPE));
443                        if(buildingI==null)
444                        {
445                                commonTell(mob,L("There's no such thing as a @x1!!!",foundRecipe.get(RCP_CLASSTYPE)));
446                                return false;
447                        }
448                        duration=getDuration(CMath.s_int(foundRecipe.get(RCP_TICKS)),mob,CMath.s_int(foundRecipe.get(RCP_LEVEL)),4);
449                        buildingI.setMaterial(getBuildingMaterial(woodRequired,data,compData));
450                        int hardness=RawMaterial.CODES.HARDNESS(buildingI.material())-3;
451                        String itemName=determineFinalName(foundRecipe.get(RCP_FINALNAME),buildingI.material(),deadMats,deadComps);
452                        itemName=CMLib.english().startWithAorAn(itemName);
453                        buildingI.setName(itemName);
454                        startStr=L("<S-NAME> start(s) making @x1.",buildingI.name());
455                        displayText=L("You are making @x1",buildingI.name());
456                        verb=L("making @x1",buildingI.name());
457                        playSound="sanding.wav";
458                        buildingI.setDisplayText(L("@x1 lies here",itemName));
459                        buildingI.setDescription(itemName+". ");
460                        buildingI.basePhyStats().setWeight(getStandardWeight(woodRequired + compData[CF_AMOUNT],bundling));
461                        buildingI.setBaseValue(CMath.s_int(foundRecipe.get(RCP_VALUE)));
462                        final int level=CMath.s_int(foundRecipe.get(RCP_LEVEL));
463                        buildingI.basePhyStats().setLevel(level+hardness);
464                        setBrand(mob, buildingI);
465                        final String ammotype=foundRecipe.get(RCP_AMMOTYPE);
466                        final int capacity=CMath.s_int(foundRecipe.get(RCP_AMOCAPACITY));
467                        final int maxrange=CMath.s_int(foundRecipe.get(RCP_MAXRANGE));
468                        final int armordmg=CMath.s_int(foundRecipe.get(RCP_ARMORDMG));
469                        if(bundling)
470                                buildingI.setBaseValue(lostValue);
471                        addSpells(buildingI,spell,deadMats.lostProps,deadComps.lostProps);
472                        if(buildingI instanceof Weapon)
473                        {
474                                if(buildingI instanceof AmmunitionWeapon)
475                                {
476                                        if(ammotype.length()>0)
477                                        {
478                                                ((AmmunitionWeapon)buildingI).setAmmoCapacity(capacity);
479                                                ((AmmunitionWeapon)buildingI).setAmmoRemaining(0);
480                                                ((AmmunitionWeapon)buildingI).setAmmunitionType(ammotype);
481                                        }
482                                }
483                                buildingI.basePhyStats().setAttackAdjustment((baseYield()+abilityCode()-1+(hardness*5)));
484                                buildingI.basePhyStats().setDamage(armordmg+hardness);
485                                ((Weapon)buildingI).setRanges(((Weapon)buildingI).minRange(),maxrange);
486                        }
487                        else
488                        if((ammotype.length()>0)&&(buildingI instanceof Ammunition))
489                        {
490                                ((Ammunition)buildingI).setAmmunitionType(ammotype);
491                                ((Ammunition)buildingI).setAmmoRemaining(capacity);
492                        }
493                        buildingI.recoverPhyStats();
494                        buildingI.text();
495                        buildingI.recoverPhyStats();
496                }
497
498                messedUp=!proficiencyCheck(mob,0,auto);
499
500                if(bundling)
501                {
502                        messedUp=false;
503                        duration=1;
504                        verb=L("bundling @x1",RawMaterial.CODES.NAME(buildingI.material()).toLowerCase());
505                        startStr=L("<S-NAME> start(s) @x1.",verb);
506                        displayText=L("You are @x1",verb);
507                }
508
509                if(autoGenerate>0)
510                {
511                        crafted.add(buildingI);
512                        return true;
513                }
514
515                final CMMsg msg=CMClass.getMsg(mob,buildingI,this,getActivityMessageType(),startStr);
516                if(mob.location().okMessage(mob,msg))
517                {
518                        mob.location().send(mob,msg);
519                        buildingI=(Item)msg.target();
520                        beneficialAffect(mob,mob,asLevel,duration);
521                        enhanceItem(mob,buildingI,recipeLevel,enhancedTypes);
522                }
523                else
524                if(bundling)
525                {
526                        messedUp=false;
527                        aborted=false;
528                        unInvoke();
529                }
530                return true;
531        }
