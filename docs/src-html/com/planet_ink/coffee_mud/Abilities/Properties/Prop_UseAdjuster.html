080        @Override
081        public int triggerMask()
082        {
083                if(affected instanceof Weapon)
084                        return TriggeredAffect.TRIGGER_HITTING_WITH;
085                else
086                if((affected instanceof Food)||(affected instanceof Drink))
087                        return TriggeredAffect.TRIGGER_USE;
088                else
089                if(affected instanceof Armor)
090                        return TriggeredAffect.TRIGGER_BEING_HIT;
091                else
092                if(affected instanceof Container)
093                        return TriggeredAffect.TRIGGER_DROP_PUTIN;
094                else
095                        return TriggeredAffect.TRIGGER_USE;
096        }

101        @Override
102        public void setMiscText(String newText)
103        {
104                super.setMiscText(newText);
105                if((parameters != null)
106                &&(parameters.length>0)
107                &&(parameters[0].length()>0))
108                {
109                        final List<String> chk=CMParms.parse(parameters[0].toUpperCase().trim());
110                        reversed = chk.contains("REVERSED") || chk.contains("REVERSE");
111                }
112                parameters=CMLib.masking().separateMaskStrs(text());
113        }

115        public void adjCharState(final MOB mob, Object[] changes, CharState charState)
116        {
117                if(changes==null)
118                        return;
119                if(multiplyCharStates)
120                {
121                        for(int c=0;c<changes.length;c+=2)
122                        {
123                                switch(((Integer)changes[c]).intValue())
124                                {
125                                case CharState.STAT_HITPOINTS:
126                                        charState.adjHitPoints((int)Math.round(CMath.mul(charState.getHitPoints(), CMath.div(((Integer) changes[c + 1]).intValue(),100))-charState.getHitPoints()),mob.maxState());
127                                        break;
128                                case CharState.STAT_HUNGER:
129                                        charState.adjHunger(((Integer) changes[c + 1]).intValue(),mob.maxState().maxHunger(mob.baseWeight()));
130                                        break;
131                                case CharState.STAT_THIRST:
132                                        charState.adjThirst(((Integer) changes[c + 1]).intValue(),mob.maxState().maxThirst(mob.baseWeight()));
133                                        break;
134                                case CharState.STAT_MANA:
135                                        charState.adjMana((int)Math.round(CMath.mul(charState.getMana(), CMath.div(((Integer) changes[c + 1]).intValue(),100))-charState.getMana()),mob.maxState());
136                                        break;
137                                case CharState.STAT_MOVE:
138                                        charState.adjMovement((int)Math.round(CMath.mul(charState.getMovement(), CMath.div(((Integer) changes[c + 1]).intValue(),100)))-charState.getMovement(),mob.maxState());
139                                        break;
140                                }
141                        }
142                }
143                else
144                {
145                        for(int c=0;c<changes.length;c+=2)
146                        {
147                                switch(((Integer)changes[c]).intValue())
148                                {
149                                case CharState.STAT_HITPOINTS:
150                                        charState.adjHitPoints( ((Integer) changes[c + 1]).intValue(), mob.maxState());
151                                        break;
152                                case CharState.STAT_HUNGER:
153                                        charState.adjHunger( ((Integer) changes[c + 1]).intValue(), mob.maxState().maxHunger(mob.baseWeight()));
154                                        break;
155                                case CharState.STAT_THIRST:
156                                        charState.adjThirst( ((Integer) changes[c + 1]).intValue(), mob.maxState().maxThirst(mob.baseWeight()));
157                                        break;
158                                case CharState.STAT_MANA:
159                                        charState.adjMana( ((Integer) changes[c + 1]).intValue(), mob.maxState());
160                                        break;
161                                case CharState.STAT_MOVE:
162                                        charState.adjMovement( ((Integer) changes[c + 1]).intValue(), mob.maxState());
163                                        break;
164                                }
165                        }
166                }
167                if(firstTime)
168                {
169                        firstTime=false;
170                        charState.copyInto(mob.curState());
171                }
172        }

174        @Override
175        public void executeMsg(final Environmental myHost, final CMMsg msg)
176        {
177                if(processing)
178                        return;
179                processing=true;
180                try
181                {
182                        if(affected==null)
183                                return;
184                        final Item myItem=(Item)affected;
185                        final ItemPossessor owner=myItem.owner();
186                        if(owner==null)
187                                return;
188                        if(!(owner instanceof MOB))
189                                return;
190                        if(msg.amISource((MOB)owner))
191                        {
192                                switch(msg.sourceMinor())
193                                {
194                                case CMMsg.TYP_EAT:
195                                case CMMsg.TYP_DRINK:
196                                        if(msg.amITarget(myItem))
197                                        {
198                                                if(canApply(owner))
199                                                {
200                                                        adjCharStats((MOB)owner, charStatsChanges,msg.source().baseCharStats());
201                                                        phyStuff(phyStatsChanges, msg.source().basePhyStats());
202                                                        adjCharState((MOB)owner, charStateChanges,msg.source().curState());
203                                                }
204                                        }
205                                        break;
206                                case CMMsg.TYP_DAMAGE:
207                                        if((myItem instanceof Weapon)
208                                        &&(msg.tool()==myItem)
209                                        &&(msg.value()>0)
210                                        &&(msg.source()!=msg.target())
211                                        &&(msg.target() instanceof MOB))
212                                        {
213                                                if(reversed)
214                                                {
215                                                        if(canApply((MOB)msg.target()))
216                                                        {
217                                                                adjCharStats((MOB)msg.target(), charStatsChanges,((MOB)msg.target()).baseCharStats());
218                                                                phyStuff(phyStatsChanges, ((MOB)msg.target()).basePhyStats());
219                                                                adjCharState((MOB)msg.target(), charStateChanges,((MOB)msg.target()).curState());
220                                                        }
221                                                }
222                                                else
223                                                if(canApply(owner))
224                                                {
225                                                        adjCharStats(msg.source(), charStatsChanges,msg.source().baseCharStats());
226                                                        phyStuff(phyStatsChanges, msg.source().basePhyStats());
227                                                        adjCharState(msg.source(), charStateChanges,msg.source().curState());
228                                                }
229                                        }
230                                        break;
231                                case CMMsg.TYP_PUT:
232                                case CMMsg.TYP_INSTALL:
233                                        if((myItem instanceof Container)
234                                        &&(!(myItem instanceof Drink))
235                                        &&(msg.amITarget(myItem)))
236                                        {
237                                                if(canApply(owner))
238                                                {
239                                                        adjCharStats((MOB)owner, charStatsChanges,msg.source().baseCharStats());
240                                                        phyStuff(phyStatsChanges, msg.source().basePhyStats());
241                                                        adjCharState((MOB)owner, charStateChanges,msg.source().curState());
242                                                }
243                                        }
244                                        break;
245                                }
246                        }
247                        else
248                        if((msg.targetMinor()==CMMsg.TYP_DAMAGE)
249                        &&(msg.target() == owner)
250                        &&(myItem instanceof Armor)
251                        &&(msg.value()>0)
252                        &&(msg.source() != owner)
253                        &&(myItem.amBeingWornProperly()))
254                        {
255                                if(reversed)
256                                {
257                                        if(canApply(msg.source()))
258                                        {
259                                                adjCharStats(msg.source(), charStatsChanges,msg.source().baseCharStats());
260                                                phyStuff(phyStatsChanges, msg.source().basePhyStats());
261                                                adjCharState(msg.source(), charStateChanges,msg.source().curState());
262                                        }
263                                }
264                                else
265                                if(canApply(owner))
266                                {
267                                        adjCharStats((MOB)owner, charStatsChanges,((MOB)owner).baseCharStats());
268                                        phyStuff(phyStatsChanges, ((MOB)owner).basePhyStats());
269                                        adjCharState((MOB)owner, charStateChanges, ((MOB)owner).curState());
270                                }
271                        }
272                }
273                finally
274                {
275                        processing=false;
276                }
277        }
