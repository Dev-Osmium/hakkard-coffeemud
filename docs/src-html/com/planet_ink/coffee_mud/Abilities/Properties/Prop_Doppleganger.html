035public class Prop_Doppleganger extends Property
036{
037        @Override
038        public String ID()
039        {
040                return "Prop_Doppleganger";
041        }
042
043        @Override
044        public String name()
045        {
046                return "Doppleganger";
047        }
048
049        @Override
050        protected int canAffectCode()
051        {
052                return Ability.CAN_MOBS|Ability.CAN_ITEMS;
053        }
054        //protected boolean lastLevelChangers=true;
055        protected Physical lastOwner=null;
056        private int maxLevel=Integer.MAX_VALUE;
057        private int minLevel=Integer.MIN_VALUE;
058        protected int lastLevel=Integer.MIN_VALUE;
059        protected int levelAdd=0;
060        protected double levelPct=1.0;
061        protected boolean matchPlayersOnly=false;
062        protected boolean matchPlayersFollowersOnly=false;
063        protected int asMaterial=-1;
064
065        @Override
066        public long flags()
067        {
068                return Ability.FLAG_ADJUSTER;
069        }

071        @Override
072        public String accountForYourself()
073        {
074                return "Level Changer";
075        }

077        @Override
078        public void setMiscText(String text)
079        {
080                super.setMiscText(text);
081                levelAdd=0;
082                levelPct=1.0;
083                asMaterial=-1;
084                maxLevel=Integer.MAX_VALUE;
085                minLevel=Integer.MIN_VALUE;
086                if(CMath.isInteger(text))
087                        levelAdd=CMath.s_int(text);
088                else
089                if(CMath.isPct(text))
090                        levelPct=CMath.s_pct(text);
091                else
092                {
093                        maxLevel=CMParms.getParmInt(text,"MAX",Integer.MAX_VALUE);
094                        minLevel=CMParms.getParmInt(text,"MIN",Integer.MIN_VALUE);
095                        levelAdd=CMParms.getParmInt(text, "LEVELADD", 0);
096                        levelPct=CMParms.getParmInt(text, "LEVELPCT", 100)/100.0;
097                        matchPlayersFollowersOnly=CMParms.getParmBool(text, "PLAYERSNFOLS", false);
098                        matchPlayersOnly=CMParms.getParmBool(text, "PLAYERSONLY", false);
099                        final String asMat = CMParms.getParmStr(text, "ASMATERIAL", "");
100                        if((asMat!=null)&&(asMat.trim().length()>0))
101                                asMaterial = RawMaterial.CODES.FIND_IgnoreCase(asMat);
102                }
103        }
104
105        @Override
106        public void executeMsg(final Environmental myHost, final CMMsg msg)
107        {
108                if((affected instanceof Item)
109                &&((((Item)affected).owner()!=lastOwner)||((lastOwner!=null)&&(lastOwner.phyStats().level()!=lastLevel)))
110                &&(((Item)affected).owner() instanceof MOB))
111                {
112                        final Item I=(Item)affected;
113                        lastOwner=I.owner();
114                        lastLevel=lastOwner.phyStats().level();
115                        int level=(int)Math.round(CMath.mul(((MOB)lastOwner).phyStats().level(),levelPct))+levelAdd;
116                        if(level<minLevel)
117                                level=minLevel;
118                        if(level>maxLevel)
119                                level=maxLevel;
120                        I.basePhyStats().setLevel(level);
121                        I.phyStats().setLevel(level);
122                        int oldMaterial=I.material();
123                        if(asMaterial != -1)
124                                I.setMaterial(asMaterial);
125                        CMLib.itemBuilder().balanceItemByLevel(I);
126                        I.setMaterial(oldMaterial);
127                        level=((MOB)lastOwner).phyStats().level();
128                        if(level<minLevel)
129                                level=minLevel;
130                        if(level>maxLevel)
131                                level=maxLevel;
132                        I.basePhyStats().setLevel(level);
133                        I.phyStats().setLevel(level);
134                        lastOwner.recoverPhyStats();
135                        final Room R=((MOB)lastOwner).location();
136                        if(R!=null)
137                                R.recoverRoomStats();
138                }
139                super.executeMsg(myHost,msg);
140        }
141
142        public boolean qualifies(MOB mob, Room R)
143        {
144                if((mob==affected)||(mob==null))
145                        return false;
146                if(mob.fetchEffect(ID())!=null)
147                        return false;
148                if(mob.isMonster())
149                {
150                        if(matchPlayersFollowersOnly)
151                        {
152                                final MOB folM=mob.amUltimatelyFollowing();
153                                return (folM!=null)&& (!folM.isMonster());
154                        }
155                        return (!matchPlayersOnly);
156                }
157
158                if((!CMSecurity.isAllowed(mob,R,CMSecurity.SecFlag.CMDMOBS))
159                &&(!CMSecurity.isAllowed(mob,R,CMSecurity.SecFlag.CMDROOMS))
160                &&(!CMLib.flags().isUnattackable(mob)))
161                        return true;
162                return false;
163        }
164
165        @Override
166        public boolean okMessage(final Environmental myHost, final CMMsg msg)
167        {
168                if((affected instanceof MOB)
169                &&(((msg.target() instanceof Room)&&(msg.sourceMinor()==CMMsg.TYP_ENTER))
170                   ||(msg.sourceMinor()==CMMsg.TYP_LIFE)))
171                //&&(lastLevelChangers))
172                {
173                        //lastLevelChangers=false;
174                        final MOB mob=(MOB)affected;
175                        final Room R=(msg.target() instanceof Room)?((Room)msg.target()):msg.source().location();
176                        if((R!=null)
177                        &&(CMLib.flags().isAliveAwakeMobile(mob,true))
178                        &&(mob.curState().getHitPoints()>=mob.maxState().getHitPoints()))
179                        {
180                                int total=0;
181                                int num=0;
182                                final MOB victim=mob.getVictim();
183                                if(qualifies(victim,R))
184                                {
185                                        total+=victim.phyStats().level();
186                                        num++;
187                                }
188                                final MOB entrant=msg.source();
189                                if(qualifies(entrant,R))
190                                {
191                                        total+=entrant.phyStats().level();
192                                        num++;
193                                }
194                                for(int i=0;i<R.numInhabitants();i++)
195                                {
196                                        final MOB M=R.fetchInhabitant(i);
197                                        if((M!=null)
198                                        &&(M!=mob)
199                                        &&((M.getVictim()==mob)||(victim==null))
200                                        &&((M!=victim)&&(M!=entrant))
201                                        &&(qualifies(M,R)))
202                                        {
203                                                total+=M.phyStats().level();
204                                                num++;
205                                        }
206                                }
207                                if(num>0)
208                                {
209                                        int level=(int)Math.round(CMath.mul(CMath.div(total,num),levelPct))+levelAdd;
210                                        if(level<minLevel)
211                                                level=minLevel;
212                                        if(level>maxLevel)
213                                                level=maxLevel;
214                                        if(level!=mob.basePhyStats().level())
215                                        {
216                                                mob.basePhyStats().setLevel(level);
217                                                mob.basePhyStats().setArmor(CMLib.leveler().getLevelMOBArmor(mob));
218                                                mob.basePhyStats().setAttackAdjustment(CMLib.leveler().getLevelAttack(mob));
219                                                mob.basePhyStats().setDamage(CMLib.leveler().getLevelMOBDamage(mob));
220                                                mob.basePhyStats().setSpeed(1.0+(CMath.div(level,100)*4.0));
221                                                mob.baseState().setHitPoints(CMLib.leveler().getPlayerHitPoints(mob));
222                                                mob.baseState().setMana(CMLib.leveler().getLevelMana(mob));
223                                                mob.baseState().setMovement(CMLib.leveler().getLevelMove(mob));
224                                                mob.recoverPhyStats();
225                                                mob.recoverCharStats();
226                                                mob.recoverMaxState();
227                                                mob.resetToMaxState();
228                                        }
229                                }
230                        }
231                }
232                return super.okMessage(myHost,msg);
233        }
