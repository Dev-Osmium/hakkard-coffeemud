035public class Prop_Socials extends Property
036{
037        @Override
038        public String ID()
039        {
040                return "Prop_Socials";
041        }
042
043        @Override
044        public String name()
045        {
046                return "Local Social creating property";
047        }
048
049        @Override
050        protected int canAffectCode()
051        {
052                return Ability.CAN_MOBS|Ability.CAN_ITEMS|Ability.CAN_ROOMS|Ability.CAN_EXITS|Ability.CAN_AREAS;
053        }
054
055        protected final Map<String,List<Social>> socials=new TreeMap<String,List<Social>>();
056
057        protected boolean wornOnly=false;
058        protected MaskingLibrary.CompiledZMask mask=null;
059
060        @Override
061        public void setMiscText(String newText)
062        {
063                super.setMiscText(newText);
064                socials.clear();
065                wornOnly=false;
066                mask=null;
067                final List<String> socialsV=CMParms.parseAny(newText, ';', false);
068                final List<String> lines = new Vector<String>();
069                for(final String social : socialsV)
070                {
071                        boolean forgive=false;
072                        final String load=CMParms.getParmStr(social, "LOAD", "");
073                        final String maskStr=CMParms.getParmStr(social, "MASK", "");
074                        final String wornonly=CMParms.getParmStr(social, "WORNONLY", "");
075                        if(load.length()>0)
076                        {
077                                final List<String> flines=Resources.getFileLineVector(Resources.getFileResource(social.substring(4).trim().substring(1).trim(),true));
078                                if((flines!=null)&&(flines.size()>0))
079                                        lines.addAll(flines);
080                                forgive=true;
081                        }
082                        if(maskStr.length()>0)
083                        {
084                                mask=CMLib.masking().getPreCompiledMask(maskStr);
085                                forgive=true;
086                        }
087                        if(wornonly.length()>0)
088                        {
089                                wornOnly=CMath.s_bool(wornonly)||wornonly.equalsIgnoreCase("T")||wornonly.toUpperCase().startsWith("Y");
090                                forgive=true;
091                        }
092
093                        final String name=CMParms.getParmStr(social, "NAME", "");
094                        final String target=CMParms.getParmStr(social, "TARGET", "");
095                        final String srcCode=CMParms.getParmStr(social, "SRCCODE", CMParms.getParmStr(social, "SOURCECODE", "M"));
096                        final String othCode=CMParms.getParmStr(social, "OTHCODE", CMParms.getParmStr(social, "TGTCODE", CMParms.getParmStr(social, "OTHERCODE", CMParms.getParmStr(social, "TARGETCODE", "V"))));
097                        final String youSee=CMParms.getParmStr(social, "YOUSEE", "");
098                        final String othersSee=CMParms.getParmStr(social, "OTHSEE", CMParms.getParmStr(social, "OTHERSEE", CMParms.getParmStr(social, "OTHERSSEE", "")));
099                        final String targetSee=CMParms.getParmStr(social, "TARGSEE", CMParms.getParmStr(social, "TARGETSEE", CMParms.getParmStr(social, "TGTSEE", "")));
100                        final String seeNoTargetSee=CMParms.getParmStr(social, "NOTARGSEE", CMParms.getParmStr(social, "NOTARGETSEE", CMParms.getParmStr(social, "NOTGTSEE", "")));
101                        final String mspFile=CMParms.getParmStr(social, "MSPFILE", "");
102                        if(name.length()==0)
103                        {
104                                if(!forgive)
105                                        Log.errOut("Prop_Socials","NAME not found in: "+social);
106                        }
107                        else
108                        if((srcCode.length()==0)||(srcCode.length()>1))
109                        {
110                                if(!forgive)
111                                        Log.errOut("Prop_Socials","Bad SRCCODE letter '"+srcCode+"': "+social);
112                        }
113                        else
114                        if((othCode.length()==0)||(othCode.length()>1))
115                        {
116                                if(!forgive)
117                                        Log.errOut("Prop_Socials","Bad OTHCODE letter '"+othCode+"': "+social);
118                        }
119                        else
120                        if(youSee.length()==0)
121                        {
122                                if(!forgive)
123                                        Log.errOut("Prop_Socials","Missing YOUSEE: "+social);
124                        }
125                        else
126                        {
127                                final StringBuilder tabLine=new StringBuilder("");
128                                tabLine.append(srcCode).append(othCode).append("\t");
129                                tabLine.append(name.toUpperCase().trim());
130                                if(target.trim().length()>0)
131                                        tabLine.append(" ").append(target.toUpperCase().trim());
132                                tabLine.append("\t");
133                                tabLine.append(youSee).append("\t");
134                                tabLine.append(othersSee).append("\t");
135                                tabLine.append(targetSee).append("\t");
136                                tabLine.append(seeNoTargetSee).append("\t");
137                                tabLine.append(mspFile);
138                                lines.add(tabLine.toString());
139                        }
140                }
141                CMLib.socials().putSocialsInHash(socials, lines);
142        }
143
144        @Override
145        public boolean okMessage(final Environmental myHost, final CMMsg msg)
146        {
147                if(affected==null)
148                        return true;
149                if((msg.targetMinor()==CMMsg.TYP_HUH)
150                &&(socials.size()>0)
151                &&(msg.targetMessage()!=null)
152                &&((mask==null)||(CMLib.masking().maskCheck(mask, msg.source(), true)))
153                &&((!(affected instanceof Item))
154                        ||((msg.source()==((Item)affected).owner())
155                           &&((!wornOnly)||(!((Item)affected).amWearingAt(Wearable.IN_INVENTORY)))
156                           )
157                        )
158                )
159                {
160                        final Vector<String> V=CMParms.parse(msg.targetMessage());
161                        Social S=CMLib.socials().fetchSocialFromSet(socials, V, true, true);
162                        if(S==null)
163                                S=CMLib.socials().fetchSocialFromSet(socials, V, false, true);
164                        if(S!=null)
165                        {
166                                S.invoke(msg.source(),V,null,false);
167                                return false;
168                        }
169                }
170                return super.okMessage(myHost,msg);
171        }
