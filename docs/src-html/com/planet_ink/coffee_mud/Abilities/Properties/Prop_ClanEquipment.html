097        @Override
098        public String accountForYourself()
099        {
100                // My slightly complicated way of showing the clan effect when ID'd
101                final StringBuffer id=new StringBuffer(L("@x1 @x2 Bonus: ",clanType,clanName));
102                if((affected instanceof Weapon)&&(!(affected instanceof Wand))&&(TypeOfEffect<1000))
103                {
104                        id.append("Does "+(1*PowerLevel)+"-"+(6*PowerLevel)+" additional "+type.toLowerCase()+" damage.");
105                }
106                else
107                if((affected instanceof Armor)&&(TypeOfEffect<1000)&&(!(affected instanceof Shield)))
108                {
109                        id.append("Raises "+type.toLowerCase()+" resistance by "+(PowerLevel*5)+".");
110                }
111                if((affected instanceof Armor)&&(TypeOfEffect<1000)&&(affected instanceof Shield))
112                {
113                        id.append("Causes "+(1*PowerLevel)+"-"+(3*PowerLevel)+" "+type.toLowerCase()+" damage to melee attackers.");
114                }
115                if((affected instanceof Wand)&&(TypeOfEffect<1000))
116                {
117                        id.append("Does "+(1*PowerLevel)+"-"+(6*PowerLevel)+" "+type.toLowerCase()+" damage when the user says `"+secretWord+"` to the target.");
118                }
119                return id.toString();
120        }

228        public boolean useAsWand(MOB mob, int level)
229        {
230                int manaRequired=50;
231                // For simplicity, there's no charges BUT use costs a flat 10% mana
232                manaRequired=(int)CMath.div(mob.maxState().getMana(),10);
233                manaRequired-=(5*level);
234                if(manaRequired<5)
235                        manaRequired=5;
236
237                if(manaRequired>mob.curState().getMana())
238                {
239                        mob.tell(L("You don't have enough mana."));
240                        return false;
241                }
242                mob.curState().adjMana(-manaRequired,mob.maxState());
243                return true;
244        }

246        public static String getWandWord(String from)
247        {
248                int hash=from.hashCode();
249                if(hash<0)
250                {
251                        hash=hash*-1;
252                }
253                return CMProps.getListFileChoiceFromIndexedListByHash(CMProps.ListFile.MAGIC_WORDS,hash);
254        }

256        public boolean checkWave(MOB mob, String message, Wand me)
257        {
258                if((mob.isMine(me))
259                &&(me.amBeingWornProperly()))
260                {
261                        final int x=message.toUpperCase().indexOf(secretWord.toUpperCase());
262                        return (x>=0);
263                }
264                return false;
265        }

270        public void waveIfAble(MOB mob, Environmental afftarget, String message, Wand me)
271        {
272                if((mob.isMine(me))&&(afftarget!=null)&&(afftarget instanceof MOB)&&(!me.amWearingAt(Wearable.IN_INVENTORY)))
273                {
274                        MOB target=null;
275                        if((mob.location()!=null))
276                        {
277                                target=(MOB)afftarget;
278                        }
279                        final int x=message.toUpperCase().indexOf(secretWord.toUpperCase());
280                        if(x>=0)
281                        {
282                                message=message.substring(x+secretWord.length());
283                                final int y=message.indexOf('\'');
284                                if(y>=0)
285                                {
286                                        message=message.substring(0,y);
287                                }
288                                message=message.trim();
289                                final Ability wandUse=mob.fetchAbility("Skill_WandUse");
290                                if((wandUse==null)||(!wandUse.proficiencyCheck(mob,0,false)))
291                                {
292                                        mob.tell(L("@x1 glows faintly for a moment, then fades.",me.name()));
293                                }
294                                else
295                                {
296                                        wandUse.setInvoker(mob);
297                                        if(useAsWand(mob,wandUse.abilityCode()))
298                                        {
299                                                mob.location().show(mob,null,CMMsg.MSG_OK_VISUAL,L("@x1 glows brightly.",me.name()));
300                                                final int flameDamage=CMLib.dice().roll(1,6*PowerLevel,1*PowerLevel);
301                                                CMLib.combat().postDamage(mob,target,null,flameDamage,CMMsg.MASK_MALICIOUS|CMMsg.MASK_ALWAYS|TypeOfEffect,WeaponType,
302                                                                L("^F^<FIGHT^>The magic of @x1 <DAMAGE> <T-NAME>!^</FIGHT^>^?",clanName));
303                                                wandUse.helpProficiency(mob, 0);
304                                                return;
305                                        }
306                                }
307                        }
308                }
309        }

311        public static void initAdjustments(CharStats adjCharStats)
312        {
313                // ensure we get no NULL errors
314                for(final int i : CharStats.CODES.SAVING_THROWS())
315                        adjCharStats.setStat(i,0);

318        @Override
319        public void affectCharStats(MOB affectedMOB, CharStats affectedStats)
320        {
321                if(EQadjCharStats==null)
322                {
323                        setMiscText(text());
324                        /*
325                         * ************************* Armor-based Resistances
326                         * *************************
327                         */
328                }
329                if((affected!=null)&&(affected instanceof Armor)&&(!(affected instanceof Shield))&&(activated)
330                                &&(!((Armor)affected).amWearingAt(Wearable.IN_INVENTORY)))
331                {
332                        for(final int i : CharStats.CODES.SAVING_THROWS())
333                                affectedStats.setStat(i,affectedStats.getStat(i)+EQadjCharStats.getStat(i));
334                }
335                super.affectCharStats(affectedMOB,affectedStats);
336        }

338        @Override
339        public void executeMsg(final Environmental myHost, final CMMsg msg)
340        {
341                super.executeMsg(myHost,msg);
342                if(((System.currentTimeMillis()-LastChecked)>TimeManager.MILI_HOUR)&&(affected!=null))
343                {
344                        if((clanName!=null)
345                        &&(clanName.length()>0)
346                        &&(CMLib.clans().getClanAnyHost(clanName)==null))
347                                affected.delEffect(this);
348                        LastChecked=System.currentTimeMillis();
349                }
350                MOB mob=null;
351                MOB source=null;
352                if((affected!=null)&&(affected instanceof Item))
353                {
354                        if((((Item)affected).owner()!=null)&&((Item)affected).owner() instanceof MOB)
355                        {
356                                mob=(MOB)((Item)affected).owner();
357                        }
358                }
359                // if held by the wrong clan, it is inactive.
360                if((mob!=null)&&(mob.getClanRole(clanName)!=null))
361                        activated=true;
362                else
363                        activated=false;
364                if(!activated)
365                        return;
366                source=msg.source();
367                /*
368                 * ********************** Weapon addtl. Damage
369                 * **********************
370                 */
371                if((msg.source().location()!=null)&&(msg.targetMinor()==CMMsg.TYP_DAMAGE)&&(msg.value()>0)&&(msg.tool()==affected)&&(!notAgain)
372                                &&(msg.target() instanceof MOB)&&(msg.tool() instanceof Weapon)&&(!(msg.tool() instanceof Wand))&&(TypeOfEffect<1000)
373                                &&(!((MOB)msg.target()).amDead()))
374                {
375                        final double flameDamage=CMLib.dice().roll(1,6*PowerLevel,1*PowerLevel);
376                        final String str=L("^F^<FIGHT^>The magic of @x1 <DAMAGE> <T-NAME>!^</FIGHT^>^?",clanName);
377                        CMLib.combat().postDamage(msg.source(),(MOB)msg.target(),null,(int)Math.round(flameDamage),
378                                        CMMsg.MASK_MALICIOUS|CMMsg.MASK_ALWAYS|TypeOfEffect,WeaponType,str);
379                }
380                /*
381                 * ********************** Shield Thorns Damage **********************
382                 */
383                if((mob!=null)&&(msg.amITarget(mob))&&(affected instanceof Shield)&&(!((Shield)affected).amWearingAt(Wearable.IN_INVENTORY))
384                &&(TypeOfEffect<1000)&&(!msg.amISource(mob)))
385                {
386                        if((CMLib.dice().rollPercentage()>32+msg.source().charStats().getStat(CharStats.STAT_DEXTERITY))
387                        &&(msg.source().rangeToTarget()==0)
388                        &&((lastMessage==null)||(lastMessage.indexOf("The magic around")<0))
389                        &&((msg.targetMajor(CMMsg.MASK_HANDS))||(msg.targetMajor(CMMsg.MASK_MOVE))))
390                        {
391                                final CMMsg msg2=CMClass.getMsg(mob,source,this,CMMsg.MSG_CAST_ATTACK_VERBAL_SPELL,null);
392                                if((source!=null)&&(source.location().okMessage(source,msg2)))
393                                {
394                                        source.location().send(source,msg2);
395                                        if(msg2.value()<=0)
396                                        {
397                                                final int damage=CMLib.dice().roll(1,3*PowerLevel,1*PowerLevel);
398                                                CMLib.combat().postDamage(mob,source,this,damage,CMMsg.MASK_MALICIOUS|CMMsg.MASK_ALWAYS|TypeOfEffect,WeaponType,
399                                                                L("^F^<FIGHT^>The magic of @x1 around <S-NAME> <DAMAGE> <T-NAME>!^</FIGHT^>^?",clanName));
400                                                if((!source.isInCombat())&&(source.isMonster())&&(source!=mob)&&(source.location()==mob.location())&&(source.location().isInhabitant(mob))&&(CMLib.flags().canBeSeenBy(mob,source)))
401                                                        CMLib.combat().postAttack(source,mob,source.fetchWieldedItem());
402                                        }
403                                }
404                        }
405                }
406                /*
407                 * ************************* Staff/Wand Message Watch
408                 * *************************
409                 */
410                if(affected instanceof Wand)
411                {
412                        switch(msg.targetMinor())
413                        {
414                        case CMMsg.TYP_WAND_USE:
415                                if(msg.amITarget(this)&&((msg.tool()==null)||(msg.tool() instanceof Physical)))
416                                        waveIfAble(mob,msg.tool(),msg.targetMessage(),(Wand)affected);
417                                break;
418                        case CMMsg.TYP_SPEAK:
419                                if(msg.sourceMinor()==CMMsg.TYP_SPEAK)
420                                {
421                                        boolean alreadyWanding=false;
422                                        final List<CMMsg> trailers =msg.trailerMsgs();
423                                        if(trailers!=null)
424                                        {
425                                                for(final CMMsg msg2 : trailers)
426                                                {
427                                                        if((msg2.targetMinor()==CMMsg.TYP_WAND_USE)
428                                                        &&(msg2.target() == this))
429                                                                alreadyWanding=true;
430                                                }
431                                        }
432                                        final String said=CMStrings.getSayFromMessage(msg.sourceMessage());
433                                        if((!alreadyWanding)
434                                        &&(said != null)
435                                        &&(checkWave(mob,said,(Wand)affected)))
436                                                msg.addTrailerMsg(CMClass.getMsg(msg.source(),this,msg.target(),CMMsg.NO_EFFECT,null,CMMsg.MASK_ALWAYS|CMMsg.TYP_WAND_USE,said,CMMsg.NO_EFFECT,null));
437                                }
438                                break;
439                        default:
440                                break;
441                        }
442                }
443        }
