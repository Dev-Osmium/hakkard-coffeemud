036public class Prop_RoomsForSale extends Prop_RoomForSale
037{
038        @Override
039        public String ID()
040        {
041                return "Prop_RoomsForSale";
042        }
043
044        @Override
045        public String name()
046        {
047                return "Putting a cluster of rooms up for sale";
048        }
049
050        protected String        uniqueLotID     = null;
051
052        protected void fillCluster(Room R, List<Room> V)
053        {
054                V.add(R);
055                for(int d=Directions.NUM_DIRECTIONS()-1;d>=0;d--)
056                {
057                        final Room R2=R.getRoomInDir(d);
058                        if((R2!=null)&&(R2.roomID().length()>0)&&(!V.contains(R2)))
059                        {
060                                final Ability A=R2.fetchEffect(ID());
061                                if((R2.getArea()==R.getArea())&&(A!=null))
062                                        fillCluster(R2,V);
063                                else
064                                {
065                                        V.remove(R); // purpose here is to put the "front" door up front.
066                                        V.add(0,R);
067                                }
068                        }
069                }
070        }
071
072        @Override
073        public List<Room> getAllTitledRooms()
074        {
075                final List<Room> V=new ArrayList<Room>();
076                Room R=null;
077                if(affected instanceof Room)
078                        R=(Room)affected;
079                else
080                        R=CMLib.map().getRoom(landPropertyID());
081                if(R!=null)
082                        fillCluster(R,V);
083                return V;
084        }
085
086        @Override
087        public String getTitleID()
088        {
089                if(affected instanceof Room)
090                        return "LAND_TITLE_FOR#"+CMLib.map().getExtendedRoomID((Room)affected);
091                else
092                {
093                        final Room R=CMLib.map().getRoom(landPropertyID());
094                        if(R!=null)
095                                return "LAND_TITLE_FOR#"+CMLib.map().getExtendedRoomID(R);
096                }
097                return "";
098        }
099
100        // update title, since it may affect room clusters, worries about EVERYONE
101        @Override
102        public void updateTitle()
103        {
104                final List<Room> V=getAllTitledRooms();
105                final String owner=getOwnerName();
106                final int price=getPrice();
107                final boolean rental=rentalProperty();
108                final boolean gridLayout = gridLayout();
109                final int back=backTaxes();
110                String uniqueID="ROOMS_PROPERTY_"+this;
111                if(V.size()>0)
112                        uniqueID="ROOMS_PROPERTY_"+CMLib.map().getExtendedRoomID(V.get(0));
113                for(int v=0;v<V.size();v++)
114                {
115                        Room R=V.get(v);
116                        synchronized(("SYNC"+R.roomID()).intern())
117                        {
118                                R=CMLib.map().getRoom(R);
119                                final LandTitle A=(LandTitle)R.fetchEffect(ID());
120                                if((A!=null)
121                                &&((!A.getOwnerName().equals(owner))
122                                   ||(A.getPrice()!=price)
123                                   ||(A.backTaxes()!=back)
124                                   ||(A.rentalProperty()!=rental)))
125                                {
126                                        A.setOwnerName(owner);
127                                        A.setPrice(price);
128                                        A.setBackTaxes(back);
129                                        A.setRentalProperty(rental);
130                                        A.setGridLayout(gridLayout);
131                                        CMLib.database().DBUpdateRoom(R);
132                                }
133                                if(A instanceof Prop_RoomsForSale)
134                                        ((Prop_RoomsForSale)A).uniqueLotID=uniqueID;
135                        }
136                }
137        }
138
139        @Override
140        public String getUniqueLotID()
141        {
142                if(uniqueLotID==null)
143                        updateTitle();
144                return uniqueLotID;
145        }
146
147        // update lot, since its called by the savethread, ONLY worries about itself
148        @Override
149        public void updateLot(Set<String> optPlayerList)
150        {
151                if(affected instanceof Room)
152                {
153                        lastItemNums=updateLotWithThisData((Room)affected,this,false,scheduleReset,optPlayerList,lastItemNums);
154                        if((lastDayDone!=((Room)affected).getArea().getTimeObj().getDayOfMonth())
155                        &&(CMProps.getBoolVar(CMProps.Bool.MUDSTARTED)))
156                        {
157                                final Room R=(Room)affected;
158                                lastDayDone=R.getArea().getTimeObj().getDayOfMonth();
159                                final List<Room> V=getAllTitledRooms();
160                                for(int v=0;v<V.size();v++)
161                                {
162                                        final Room R2=V.get(v);
163                                        final Prop_RoomForSale PRFS=(Prop_RoomForSale)R2.fetchEffect(ID());
164                                        if(PRFS!=null)
165                                                PRFS.lastDayDone=R.getArea().getTimeObj().getDayOfMonth();
166                                }
167                                if((getOwnerName().length()>0)&&rentalProperty()&&(R.roomID().length()>0))
168                                {
169                                        if(doRentalProperty(R.getArea(),R.roomID(),getOwnerName(),getPrice()))
170                                        {
171                                                setOwnerName("");
172                                                updateTitle();
173                                                lastItemNums=updateLotWithThisData((Room)affected,this,false,scheduleReset,optPlayerList,lastItemNums);
174                                        }
175                                }
176                        }
177                        scheduleReset=false;
178                }
179        }
180}
