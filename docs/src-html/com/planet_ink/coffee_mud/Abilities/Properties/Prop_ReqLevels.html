035@SuppressWarnings({"unchecked","rawtypes"})
036public class Prop_ReqLevels extends Property implements TriggeredAffect
037{
038        @Override
039        public String ID()
040        {
041                return "Prop_ReqLevels";
042        }
043
044        @Override
045        public String name()
046        {
047                return "Level Limitations";
048        }
049
050        @Override
051        protected int canAffectCode()
052        {
053                return Ability.CAN_ROOMS|Ability.CAN_AREAS|Ability.CAN_EXITS;
054        }
055
056        private boolean                 noFollow        = false;
057        private boolean                 noSneak         = false;
058        private boolean                 allFlag         = false;
059        private final boolean   sysopFlag       = false;
060        private String                  message         = L("You are not allowed to go that way.");
061        private int[]                   lvls            = new int[0];
062
063        @Override
064        public long flags()
065        {
066                return Ability.FLAG_ZAPPER;
067        }
068
069        @Override
070        public int triggerMask()
071        {
072                return TriggeredAffect.TRIGGER_ENTER;
073        }
074
075        @Override
076        public void setMiscText(String txt)
077        {
078                noFollow=false;
079                noSneak=false;
080                message = L("You are not allowed to go that way.");
081                lvls = new int[0];
082                final Vector<String> parms=CMParms.parse(txt.toUpperCase());
083                String s;
084                for(final Enumeration<String> p=parms.elements();p.hasMoreElements();)
085                {
086                        s=p.nextElement();
087                        if("NOFOLLOW".startsWith(s))
088                                noFollow=true;
089                        else
090                        if(s.startsWith("NOSNEAK"))
091                                noSneak=true;
092                        else
093                        if("ALL".equals(s))
094                                allFlag=true;
095                        else
096                        if("SYSOP".equals(s))
097                                noSneak=true;
098                }
099                message=CMParms.getParmStr(txt, "MESSAGE", message);
100                int lastPlace=0;
101                int x=0;
102                final String text=txt.trim();
103                if(text.length()>0)
104                {
105                        final List<Integer> vals=new LinkedList<Integer>();
106                        while(x>=0)
107                        {
108                                x=text.indexOf('>',lastPlace);
109                                if(x<0)
110                                        x=text.indexOf('<',lastPlace);
111                                if(x<0)
112                                        x=text.indexOf('=',lastPlace);
113                                if(x>=0)
114                                {
115                                        final char primaryChar=text.charAt(x);
116                                        x++;
117                                        boolean andEqual=false;
118                                        if(text.charAt(x)=='=')
119                                        {
120                                                andEqual=true;
121                                                x++;
122                                        }
123                                        lastPlace=x;
124                                        String cmpString="";
125                                        while((x<text.length())
126                                        &&(((text.charAt(x)==' ')&&(cmpString.length()==0))
127                                                ||(Character.isDigit(text.charAt(x)))))
128                                        {
129                                                if(Character.isDigit(text.charAt(x)))
130                                                        cmpString+=text.charAt(x);
131                                                x++;
132                                        }
133                                        if(cmpString.length()>0)
134                                        {
135                                                final int cmpLevel=CMath.s_int(cmpString);
136                                                vals.add(Integer.valueOf(primaryChar));
137                                                vals.add(Integer.valueOf(cmpLevel));
138                                                if(andEqual)
139                                                {
140                                                        vals.add(Integer.valueOf('='));
141                                                        vals.add(Integer.valueOf(cmpLevel));
142                                                }
143                                        }
144                                }
145                        }
146                        if(vals.size()>0)
147                        {
148                                lvls=new int[vals.size()];
149                                int i=0;
150                                for(Integer I : vals)
151                                        lvls[i++]=I.intValue();
152                        }
153                }
154                super.setMiscText(txt);
155        }
156
157        public boolean passesMuster(MOB mob, Environmental R)
158        {
159                if(mob==null)
160                        return false;
161                if(CMLib.flags().isATrackingMonster(mob))
162                        return true;
163                if(CMLib.flags().isSneaking(mob)&&(!noSneak))
164                        return true;
165
166                if((allFlag)
167                ||(text().length()==0)
168                ||(!(R instanceof Room))
169                ||(CMSecurity.isAllowed(mob,(Room)R,CMSecurity.SecFlag.GOTO)))
170                        return true;
171
172                if(sysopFlag)
173                        return false;
174
175                final int lvl=mob.phyStats().level();
176                for(int i=0;i<lvls.length;i+=2)
177                {
178                        switch(lvls[i])
179                        {
180                        case '=':
181                                if(lvl == lvls[i+1])
182                                        return true;
183                                break;
184                        case '>':
185                                if(lvl > lvls[i+1])
186                                        return true;
187                                break;
188                        case '<':
189                                if(lvl < lvls[i+1])
190                                        return true;
191                                break;
192                        }
193                }
194                
195
196                return false;
197        }
198
199        @Override
200        public boolean okMessage(final Environmental myHost, final CMMsg msg)
201        {
202                if((affected!=null)
203                &&(((msg.target() instanceof Room)&&(msg.targetMinor()==CMMsg.TYP_ENTER))
204                        ||((msg.target() instanceof Rideable)&&(msg.targetMinor()==CMMsg.TYP_SIT)))
205                &&(!CMLib.flags().isFalling(msg.source()))
206                &&((msg.amITarget(affected))||(msg.tool()==affected)||(affected instanceof Area)))
207                {
208                        final HashSet<MOB> H=new HashSet<MOB>();
209                        if(noFollow)
210                                H.add(msg.source());
211                        else
212                        {
213                                msg.source().getGroupMembers(H);
214                                final HashSet<MOB> H2=new XHashSet(H);
215                                for(final Iterator e=H2.iterator();e.hasNext();)
216                                        ((MOB)e.next()).getRideBuddies(H);
217                        }
218                        for(final Iterator e=H.iterator();e.hasNext();)
219                        {
220                                final Environmental E=(Environmental)e.next();
221                                if((E instanceof MOB)
222                                &&(passesMuster((MOB)E,msg.target())))
223                                        return super.okMessage(myHost,msg);
224                        }
225                        msg.source().tell(message);
226                        return false;
227                }
228                return super.okMessage(myHost,msg);
229        }
