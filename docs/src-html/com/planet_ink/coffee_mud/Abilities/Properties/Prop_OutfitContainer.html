035public class Prop_OutfitContainer extends Property
036{
037        @Override
038        public String ID()
039        {
040                return "Prop_OutfitContainer";
041        }
042
043        @Override
044        public String name()
045        {
046                return "Remembers whats put in, and wear the contents when worn";
047        }
048
049        @Override
050        protected int canAffectCode()
051        {
052                return Ability.CAN_ITEMS;
053        }
054
055        @Override
056        public boolean bubbleAffect()
057        {
058                return true;
059        }
060        
061        protected List<Item> outfitContents = new SLinkedList<Item>();
062        protected boolean fixedYet = false;
063
064        @Override
065        public void setMiscText(String text)
066        {
067                super.setMiscText(text);
068                fixedYet = false;
069                this.outfitContents.clear();
070        }
071        
072        protected void fixContentsFromText()
073        {
074                final Item affected = (this.affected instanceof Item)? (Item)this.affected : null;
075                if((!fixedYet)
076                &&(affected != null)
077                &&(affected.owner() instanceof MOB))
078                {
079                        this.outfitContents.clear();
080                        final MOB mob=(MOB)affected.owner();
081                        if(mob.numItems() > 0)
082                        {
083                                fixedYet = true;
084                                if(super.miscText.length()>1)
085                                {
086                                        final int xmlStart = super.miscText.indexOf(';');
087                                        if(xmlStart >=0)
088                                        {
089                                                final List<XMLTag> tags=CMLib.xml().parseAllXML(miscText.substring(xmlStart+1));
090                                                final List<String> itemNames = new ArrayList<String>(tags.size());
091                                                for(XMLTag tag : tags)
092                                                        itemNames.add(tag.value());
093                                                Collections.sort(itemNames);
094                                                String lastName="";
095                                                int ctr=1;
096                                                for(String name : itemNames)
097                                                {
098                                                        if(name.equals(lastName))
099                                                        {
100                                                                ctr++;
101                                                                name = "$"+name+"$."+ctr;
102                                                        }
103                                                        else
104                                                                name = "$"+name+"$";
105                                                        Item I=mob.findItem(name);
106                                                        if(I!=null)
107                                                        {
108                                                                if((I.container() == affected)
109                                                                ||(!I.amWearingAt(Item.IN_INVENTORY))
110                                                                ||((I.container()!=null)&&(!I.ultimateContainer(null).amWearingAt(Item.IN_INVENTORY))))
111                                                                        this.outfitContents.add(I);
112                                                        }
113                                                }
114                                        }
115                                }
116                        }
117                }
118        }
119        
120        @Override
121        public String text()
122        {
123                fixContentsFromText();
124                StringBuilder str=new StringBuilder(";");
125                for(Item I : this.outfitContents)
126                        str.append("<I>"+CMLib.xml().parseOutAngleBrackets(I.Name())+"</I>");
127                return str.toString();
128        }

130        @Override
131        public String accountForYourself()
132        {
133                StringBuilder str=new StringBuilder("");
134                return str.toString();
135        }
136        
137        @Override
138        public boolean okMessage(final Environmental myHost, final CMMsg msg)
139        {
140                if(!super.okMessage(myHost,msg))
141                        return false;
142                final Item affected = (this.affected instanceof Item)? (Item)this.affected : null;
143                if((msg.target() == affected) 
144                && (msg.othersMessage()!=null)
145                && (msg.othersMessage().length()>0))
146                {
147                        this.fixContentsFromText();
148                        final Item item = (Item)msg.tool();
149                        if (msg.tool() instanceof Item)
150                        {
151                                if (msg.targetMinor() == CMMsg.TYP_PUT)
152                                {
153                                        if(!outfitContents.contains(item))
154                                        {
155                                                outfitContents.add(item);
156                                        }
157                                }
158                                else
159                                if((msg.targetMinor() == CMMsg.TYP_GET)
160                                &&(item.container()==affected))
161                                {
162                                        if(outfitContents.contains(item))
163                                        {
164                                                outfitContents.remove(item);
165                                        }
166                                }
167                        }
168                        if((msg.targetMinor() == CMMsg.TYP_WEAR)
169                        ||(msg.targetMinor() == CMMsg.TYP_HOLD)
170                        ||(msg.targetMinor() == CMMsg.TYP_WIELD))
171                        {
172                                fixContentsFromText();
173                                final List<Item> thingsToWear = new ArrayList<Item>(outfitContents.size());
174                                for(Item I : outfitContents)
175                                {
176                                        if(I.amWearingAt(Item.IN_INVENTORY))
177                                                thingsToWear.add(I);
178                                }
179                                if(thingsToWear.size()>0)
180                                {
181                                        final MOB mob=msg.source();
182                                        final List<Item> thingsToRemove = new ArrayList<Item>(outfitContents.size());
183                                        List<Item> thingsToReDo = new ArrayList<Item>(1);
184                                        for(Item I : thingsToWear)
185                                        {
186                                                if(I.canWear(msg.source(), 0))
187                                                        continue;
188                                                long cantWearAt=I.whereCantWear(mob);
189                                                final Item alreadyWearingI=(cantWearAt==0)?null:mob.fetchFirstWornItem(cantWearAt);
190                                                if((alreadyWearingI != null)
191                                                &&(!thingsToRemove.contains(alreadyWearingI))
192                                                &&(!thingsToWear.contains(alreadyWearingI)))
193                                                {
194                                                        thingsToRemove.add(alreadyWearingI);
195                                                        continue;
196                                                }
197                                                
198                                                boolean done=false;
199                                                for(final Enumeration<Item> i=msg.source().items();i.hasMoreElements();)
200                                                {
201                                                        final Item chkI=i.nextElement();
202                                                        if(chkI==null)
203                                                                continue;
204                                                        if((thingsToRemove.contains(chkI))
205                                                        ||(thingsToWear.contains(chkI))
206                                                        ||(chkI.amWearingAt(Item.IN_INVENTORY))
207                                                        ||((chkI.rawProperLocationBitmap() & I.rawProperLocationBitmap())==0))
208                                                                continue;
209                                                        long oldWornCode = chkI.rawWornCode();
210                                                        chkI.setRawWornCode(Item.IN_INVENTORY);
211                                                        if(I.canWear(msg.source(), 0))
212                                                        {
213                                                                chkI.setRawWornCode(oldWornCode);
214                                                                thingsToRemove.add(I);
215                                                                done=true;
216                                                                break;
217                                                        }
218                                                }
219                                                if(!done)
220                                                        thingsToReDo.add(I);
221                                        }
222                                        //for(Item I : thingsToReDo)
223                                        //{
224                                                //TODO: I have no idea .. but it will be layer related..
225                                        //}
226                                        
227                                        for(Item I : thingsToRemove)
228                                                CMLib.commands().postRemove(msg.source(), I, true);
229                                        for(Item I : thingsToWear)
230                                        {
231                                                if(I.container()==affected)
232                                                {
233                                                        CMLib.commands().postGet(msg.source(), affected, I, true);
234                                                        I.setContainer(null);
235                                                }
236                                                CMLib.commands().postWear(msg.source(),I, false);
237                                        }
238                                        return false;
239                                }
240                        }
241                }
242                return true;
243        }

245        @Override
246        public void executeMsg(final Environmental myHost, final CMMsg msg)
247        {
248                final Container affected = (this.affected instanceof Container)? (Container)this.affected : null;
249                super.executeMsg(myHost, msg);
250                if((msg.targetMinor() == CMMsg.TYP_REMOVE)
251                &&(msg.target() instanceof Item)
252                &&(outfitContents.contains(msg.target())))
253                {
254                        final Item item=(Item)msg.target();
255                        final MOB mob=msg.source();
256                        if(item.container() == null)
257                        {
258                                msg.addTrailerRunnable(new Runnable()
259                                {
260                                        final Item I=item;
261                                        final MOB M = mob;
262                                        @Override
263                                        public void run()
264                                        {
265                                                CMLib.commands().postPut(M, affected, I, true);
266                                                
267                                        }
268                                });
269                        }
270                }
271        }
