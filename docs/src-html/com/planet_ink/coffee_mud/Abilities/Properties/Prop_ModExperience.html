038public class Prop_ModExperience extends Property
039{
040        @Override
041        public String ID()
042        {
043                return "Prop_ModExperience";
044        }
045
046        @Override
047        public String name()
048        {
049                return "Modifying Experience Gained";
050        }
051
052        @Override
053        protected int canAffectCode()
054        {
055                return Ability.CAN_MOBS | Ability.CAN_ITEMS | Ability.CAN_AREAS | Ability.CAN_ROOMS;
056        }
057
058        protected String                        operationFormula        = "";
059        protected boolean                       selfXP                          = false;
060        protected CompiledFormula       operation                       = null;
061        protected CompiledZMask         mask                            = null;
062
063        @Override
064        public String accountForYourself()
065        {
066                return "Modifies experience gained: " + operationFormula;
067        }
068
069        public int translateAmount(int amount, final String val)
070        {
071                if(amount<0)
072                        amount=-amount;
073                if(val.endsWith("%"))
074                        return (int)Math.round(CMath.mul(amount,CMath.div(CMath.s_int(val.substring(0,val.length()-1)),100)));
075                return CMath.s_int(val);
076        }
077
078        public String translateNumber(final String val)
079        {
080                if(val.endsWith("%"))
081                        return "( @x1 * (" + val.substring(0,val.length()-1) + " / 100) )";
082                return Integer.toString(CMath.s_int(val));
083        }
084
085        @Override
086        public void setMiscText(final String newText)
087        {
088                super.setMiscText(newText);
089                operation = null;
090                mask=null;
091                selfXP=false;
092                String s=newText.trim();
093                int x=s.indexOf(';');
094                if(x>=0)
095                {
096                        mask=CMLib.masking().getPreCompiledMask(s.substring(x+1).trim());
097                        s=s.substring(0,x).trim();
098                }
099                x=s.indexOf("SELF");
100                if(x>=0)
101                {
102                        selfXP=true;
103                        s=s.substring(0,x)+s.substring(x+4);
104                }
105                operationFormula="Amount "+s;
106                final List<String> ops = new ArrayList<String>();
107                int paren=0;
108                final StringBuilder curr=new StringBuilder("");
109                for(int i=0;i<s.length();i++)
110                {
111                        if(paren > 0)
112                        {
113                                if(s.charAt('i')=='(')
114                                {
115                                        if(paren == 0)
116                                        {
117                                                if(curr.length()>0)
118                                                        ops.add(curr.toString().trim());
119                                                curr.setLength(0);
120                                        }
121                                        paren++;
122                                }
123                                else
124                                if(s.charAt('i')==')')
125                                        paren--;
126                                curr.append(s.charAt(i));
127                        }
128                        else
129                        switch(s.charAt(i))
130                        {
131                        case '=':
132                        case '+':
133                        case '-':
134                        case '*':
135                        case '/':
136                                if(curr.length()>0)
137                                        ops.add(curr.toString().trim());
138                                curr.setLength(0);
139                                curr.append(s.charAt(i));
140                                break;
141                        default:
142                                curr.append(s.charAt(i));
143                                break;
144                        }
145                }
146                if(curr.length()>0)
147                        ops.add(curr.toString().trim());
148                StringBuilder finalOps = new StringBuilder("");
149                for(final String op : ops)
150                {
151                        if(op.startsWith("="))
152                                finalOps = new StringBuilder(translateNumber(op.substring(1)).trim());
153                        else
154                        if(op.startsWith("(")&&(op.endsWith(")")))
155                                finalOps = new StringBuilder(op);
156                        else
157                        if(op.startsWith("+")||op.startsWith("-")||op.startsWith("*")||op.startsWith("/"))
158                        {
159                                if(finalOps.length()==0)
160                                        finalOps.append("@x1");
161                                finalOps.append(" ").append(op.charAt(0)).append(" ");
162                                finalOps.append(translateNumber(op.substring(1)).trim());
163                        }
164                        else
165                                finalOps=new StringBuilder(translateNumber(s.trim()));
166                }
167                if(finalOps.length()>0)
168                        operation = CMath.compileMathExpression(finalOps.toString());
169                operationFormula=CMStrings.replaceAll(operationFormula, "@x1", "Amount");
170        }
171
172        @Override
173        public boolean okMessage(final Environmental myHost, final CMMsg msg)
174        {
175                if(operation == null)
176                        setMiscText(text());
177                if(((msg.sourceMinor()==CMMsg.TYP_EXPCHANGE)
178                        ||(msg.sourceMinor()==CMMsg.TYP_RPXPCHANGE))
179                &&(operation != null)
180                &&((((msg.target()==affected)||(selfXP && (msg.source()==affected)))
181                                &&(affected instanceof MOB))
182                   ||((affected instanceof Item)
183                           &&(msg.source()==((Item)affected).owner())
184                           &&(((Item)affected).amBeingWornProperly()))
185                   ||(affected instanceof Room)
186                   ||(affected instanceof Area)))
187                {
188                        if(mask!=null)
189                        {
190                                if(affected instanceof Item)
191                                {
192                                        if((msg.target()==null)||(!(msg.target() instanceof MOB))||(!CMLib.masking().maskCheck(mask,msg.target(),true)))
193                                                return super.okMessage(myHost,msg);
194                                }
195                                else
196                                if(!CMLib.masking().maskCheck(mask,msg.source(),true))
197                                        return super.okMessage(myHost,msg);
198                        }
199                        msg.setValue((int)Math.round(CMath.parseMathExpression(operation, new double[]{msg.value()}, 0.0)));
200                }
201                return super.okMessage(myHost,msg);
202        }
