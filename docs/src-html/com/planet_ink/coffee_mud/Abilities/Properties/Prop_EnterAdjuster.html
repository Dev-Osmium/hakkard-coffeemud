038public class Prop_EnterAdjuster extends Property implements TriggeredAffect
039{
040        @Override
041        public String ID()
042        {
043                return "Prop_EnterAdjuster";
044        }
045
046        @Override
047        public String name()
048        {
049                return "Room entering adjuster";
050        }
051
052        @Override
053        protected int canAffectCode()
054        {
055                return Ability.CAN_EXITS | Ability.CAN_ROOMS;
056        }
057
058        protected MaskingLibrary.CompiledZMask mask=CMLib.masking().createEmptyMask();
059        protected String[] parameters=new String[]{"",""};
060
061        @Override
062        public long flags()
063        {
064                return Ability.FLAG_ADJUSTER;
065        }
066
067        @Override
068        public int triggerMask()
069        {
070                return TriggeredAffect.TRIGGER_ENTER;
071        }
072
073        @Override
074        public void setMiscText(String newText)
075        {
076                super.setMiscText(newText);
077                parameters=CMLib.masking().separateMaskStrs(text());
078                if(parameters[1].trim().length()==0)
079                        mask=CMLib.masking().createEmptyMask();
080                else
081                        mask=CMLib.masking().getPreCompiledMask(parameters[1]);
082        }
083
084        @Override
085        public String accountForYourself()
086        {
087                return Prop_HaveAdjuster.fixAccoutingsWithMask("Affects those who enter: "+parameters[0],parameters[1]);
088        }
089
090        @Override
091        public boolean okMessage(final Environmental myHost, final CMMsg msg)
092        {
093                if((affected!=null)
094                &&(((msg.targetMinor()==CMMsg.TYP_ENTER)&&((affected instanceof Room)||(affected instanceof Exit)))
095                   ||((msg.targetMinor()==CMMsg.TYP_SIT)&&(affected==msg.target())&&(affected instanceof Rideable)))
096                &&((mask==null)||(CMLib.masking().maskCheck(mask,msg.source(),true))))
097                {
098                        final MOB mob=msg.source();
099                        final Vector<Ability> theSpells=new Vector<Ability>();
100                        int del=parameters[0].indexOf(';');
101                        while(del>=0)
102                        {
103                                final String thisOne=parameters[0].substring(0,del);
104                                if((thisOne.length()>0)&&(!thisOne.equals(";")))
105                                {
106                                        Ability A=CMClass.getAbility(thisOne);
107                                        if((A!=null)&&(!CMLib.ableMapper().classOnly("Archon",A.ID())))
108                                        {
109                                                A=(Ability)A.copyOf();
110                                                theSpells.addElement(A);
111                                        }
112                                }
113                                parameters[0]=parameters[0].substring(del+1);
114                                del=parameters[0].indexOf(';');
115                        }
116                        if((parameters[0].length()>0)&&(!parameters[0].equals(";")))
117                        {
118                                Ability A=CMClass.getAbility(parameters[0]);
119                                if(A!=null)
120                                {
121                                        A=(Ability)A.copyOf();
122                                        theSpells.addElement(A);
123                                }
124                        }
125                        for(int i=0;i<theSpells.size();i++)
126                        {
127                                final Ability thisOne=(Ability)theSpells.elementAt(i).copyOf();
128                                thisOne.invoke(mob,mob,true,0);
129                        }
130
131                        mob.basePhyStats().setAbility(mob.basePhyStats().ability()+CMParms.getParmPlus(parameters[0],"abi"));
132                        mob.basePhyStats().setArmor(mob.basePhyStats().armor()+CMParms.getParmPlus(parameters[0],"arm"));
133                        mob.basePhyStats().setAttackAdjustment(mob.basePhyStats().attackAdjustment()+CMParms.getParmPlus(parameters[0],"att"));
134                        mob.basePhyStats().setDamage(mob.basePhyStats().damage()+CMParms.getParmPlus(parameters[0],"dam"));
135                        mob.basePhyStats().setDisposition(mob.basePhyStats().disposition()|CMParms.getParmPlus(parameters[0],"dis"));
136                        mob.basePhyStats().setLevel(mob.basePhyStats().level()+CMParms.getParmPlus(parameters[0],"lev"));
137                        mob.basePhyStats().setRejuv(mob.basePhyStats().rejuv()+CMParms.getParmPlus(parameters[0],"rej"));
138                        mob.basePhyStats().setSensesMask(mob.basePhyStats().sensesMask()|CMParms.getParmPlus(parameters[0],"sen"));
139                        mob.basePhyStats().setSpeed(mob.basePhyStats().speed()+CMParms.getParmPlus(parameters[0],"spe"));
140                        mob.basePhyStats().setWeight(mob.basePhyStats().weight()+CMParms.getParmPlus(parameters[0],"wei"));
141                        mob.basePhyStats().setHeight(mob.basePhyStats().height()+CMParms.getParmPlus(parameters[0],"hei"));
142
143                        mob.baseCharStats().setStat(CharStats.STAT_CHARISMA,mob.baseCharStats().getStat(CharStats.STAT_CHARISMA)+CMParms.getParmPlus(parameters[0],"cha"));
144                        mob.baseCharStats().setStat(CharStats.STAT_CONSTITUTION,mob.baseCharStats().getStat(CharStats.STAT_CONSTITUTION)+CMParms.getParmPlus(parameters[0],"con"));
145                        mob.baseCharStats().setStat(CharStats.STAT_DEXTERITY,mob.baseCharStats().getStat(CharStats.STAT_DEXTERITY)+CMParms.getParmPlus(parameters[0],"dex"));
146                        String val=CMParms.getParmStr(parameters[0],"gen","").toUpperCase();
147                        if((val.length()>0)&&((val.charAt(0)=='M')||(val.charAt(0)=='F')||(val.charAt(0)=='N')))
148                                mob.baseCharStats().setStat(CharStats.STAT_GENDER,val.charAt(0));
149                        mob.baseCharStats().setStat(CharStats.STAT_INTELLIGENCE,mob.baseCharStats().getStat(CharStats.STAT_INTELLIGENCE)+CMParms.getParmPlus(parameters[0],"int"));
150                        val=CMParms.getParmStr(parameters[0],"cla","").toUpperCase();
151                        if(val.length()>0)
152                        {
153                                final CharClass C=CMClass.findCharClass(val);
154                                if((C!=null)&&(C.availabilityCode()!=0))
155                                        mob.baseCharStats().setCurrentClass(C);
156                        }
157                        val=CMParms.getParmStr(parameters[0],"rac","").toUpperCase();
158                        if((val.length()>0)&&(CMClass.getRace(val)!=null))
159                        {
160                                final int oldCat=mob.baseCharStats().ageCategory();
161                                mob.baseCharStats().setMyRace(CMClass.getRace(val));
162                                mob.baseCharStats().getMyRace().startRacing(mob,false);
163                                if(mob.baseCharStats().getStat(CharStats.STAT_AGE)>0)
164                                        mob.baseCharStats().setStat(CharStats.STAT_AGE,mob.baseCharStats().getMyRace().getAgingChart()[oldCat]);
165                        }
166                        mob.baseCharStats().setStat(CharStats.STAT_STRENGTH,mob.baseCharStats().getStat(CharStats.STAT_STRENGTH)+CMParms.getParmPlus(parameters[0],"str"));
167                        mob.baseCharStats().setStat(CharStats.STAT_WISDOM,mob.baseCharStats().getStat(CharStats.STAT_WISDOM)+CMParms.getParmPlus(parameters[0],"wis"));
168                        if(CMParms.getParmPlus(parameters[0],"lev")!=0)
169                                mob.baseCharStats().setClassLevel(mob.baseCharStats().getCurrentClass(),mob.baseCharStats().getClassLevel(mob.baseCharStats().getCurrentClass())+CMParms.getParmPlus(parameters[0],"lev"));
170
171                        mob.baseState().setHitPoints(mob.curState().getHitPoints()+CMParms.getParmPlus(parameters[0],"hit"));
172                        mob.curState().setHunger(mob.curState().getHunger()+CMParms.getParmPlus(parameters[0],"hun"));
173                        mob.curState().setMana(mob.curState().getMana()+CMParms.getParmPlus(parameters[0],"man"));
174                        mob.curState().setMovement(mob.curState().getMovement()+CMParms.getParmPlus(parameters[0],"mov"));
175                        mob.curState().setThirst(mob.curState().getThirst()+CMParms.getParmPlus(parameters[0],"thi"));
176
177                        mob.setPractices(mob.getPractices()+CMParms.getParmPlus(parameters[0],"prac"));
178                        mob.setTrains(mob.getTrains()+CMParms.getParmPlus(parameters[0],"trai"));
179                        final int qp=CMParms.getParmPlus(parameters[0],"ques");
180                        if(qp!=0)
181                                mob.setQuestPoint(mob.getQuestPoint()+qp);
182                        if(qp>0)
183                                CMLib.players().bumpPrideStat(mob,PrideStat.QUESTPOINTS_EARNED, qp);
184                        final int newMoney=CMParms.getParmPlus(parameters[0],"coin");
185                        if(newMoney!=0)
186                                CMLib.beanCounter().setMoney(mob,CMLib.beanCounter().getMoney(mob)+newMoney);
187                        final int exp=CMParms.getParmPlus(parameters[0],"expe");
188                        if(exp>0)
189                                CMLib.leveler().postExperience(mob,null,null,exp,false);
190                        mob.recoverCharStats();
191                        mob.recoverPhyStats();
192                        mob.recoverMaxState();
193                        CMLib.utensils().confirmWearability(mob);
194                }
195                return super.okMessage(myHost,msg);
196        }
197}
