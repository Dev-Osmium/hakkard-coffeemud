089        @Override
090        public void setMiscText(String txt)
091        {
092                super.setMiscText(txt);
093                peopleCap=Integer.MAX_VALUE;
094                playerCap=Integer.MAX_VALUE;
095                mobCap=Integer.MAX_VALUE;
096                itemCap=Integer.MAX_VALUE;
097                maxWeight=Integer.MAX_VALUE;
098                roomLimit=Integer.MAX_VALUE;
099                indoorOnly=false;
100                if(txt.length()==0)
101                        peopleCap=2;
102                else
103                if(CMath.isNumber(txt))
104                        peopleCap=CMath.s_int(txt);
105                else
106                {
107                        peopleCap=CMParms.getParmInt(txt,"people",peopleCap);
108                        playerCap=CMParms.getParmInt(txt,"players",playerCap);
109                        mobCap=CMParms.getParmInt(txt,"mobs",mobCap);
110                        itemCap=CMParms.getParmInt(txt,"items",itemCap);
111                        roomLimit=CMParms.getParmInt(txt,"rooms",roomLimit);
112                        maxWeight=CMParms.getParmInt(txt,"weight",maxWeight);
113                        indoorOnly=CMParms.getParmBool(txt,"indoor",indoorOnly);
114                        containersOk=CMParms.getParmBool(txt,"droponly",containersOk)||CMParms.getParmBool(txt,"containersok",containersOk);
115                }
116        }

118        protected int getRoomWeight(Room R)
119        {
120                if(R==null)
121                        return Integer.MAX_VALUE/2;
122                int soFar=0;
123                for(int i=0;i<R.numItems();i++)
124                {
125                        final Item I = R.getItem(i);
126                        if (I != null)
127                                soFar += I.phyStats().weight();
128                }
129                return soFar;
130        }
131
132        protected int getRoomItemCount(Room R)
133        {
134                if(R==null)
135                        return Integer.MAX_VALUE/2;
136                int soFar=0;
137                for(int i=0;i<R.numItems();i++)
138                {
139                        final Item I=R.getItem(i);
140                        if((I!=null)&&(I.container()==null))
141                                soFar++;
142                }
143                return soFar;
144        }

146        protected void overflowCheck()
147        {
148                final Physical affected=this.affected;
149                if(affected!=null)
150                {
151                        final List<Room> doRooms = new LinkedList<Room>();
152                        if(affected instanceof Area)
153                                doRooms.addAll(Collections.list(((Area)affected).getProperMap()));
154                        else
155                                doRooms.add(CMLib.map().roomLocation(affected));
156                        for(final Iterator<Room> r=doRooms.iterator();r.hasNext();)
157                        {
158                                try
159                                {
160                                        final Room R=r.next();
161                                        if(R!=null)
162                                        {
163                                                if((peopleCap<Integer.MAX_VALUE)
164                                                &&((!indoorOnly)||((R.domainType()&Room.INDOORS)==Room.INDOORS)))
165                                                {
166                                                }
167                                                if((!indoorOnly)||((R.domainType()&Room.INDOORS)==Room.INDOORS))
168                                                {
169                                                        if(itemCap<Integer.MAX_VALUE)
170                                                        {
171                                                                final int roomItemCount=getRoomItemCount(R);
172                                                                if(roomItemCount>itemCap)
173                                                                {
174                                                                        int totOver=roomItemCount-itemCap;
175                                                                        Room targetRoom = null;
176                                                                        int smallestCount=Integer.MAX_VALUE/2;
177                                                                        for(int d=0;d<Directions.NUM_DIRECTIONS();d++)
178                                                                        {
179                                                                                final Room R2=R.getRoomInDir(d);
180                                                                                final Exit E2=R.getExitInDir(d);
181                                                                                if((R2!=null)&&(E2!=null)
182                                                                                &&(E2.isOpen())&&(!CMLib.flags().isAiryRoom(R2)))
183                                                                                {
184                                                                                        int rct=this.getRoomItemCount(R2);
185                                                                                        if(rct < smallestCount)
186                                                                                        {
187                                                                                                targetRoom=R2;
188                                                                                                smallestCount=rct;
189                                                                                        }
190                                                                                }
191                                                                        }
192                                                                        if(targetRoom != null)
193                                                                        {
194                                                                                for(int ri=R.numItems()-1;ri>=0 && totOver>0;ri--,totOver--)
195                                                                                {
196                                                                                        Item I=R.getItem(ri);
197                                                                                        if((I!=null)&&(I.container()==null))
198                                                                                                targetRoom.moveItemTo(I);
199                                                                                }
200                                                                        }
201                                                                }
202                                                        }
203                                                        if(maxWeight<Integer.MAX_VALUE)
204                                                        {
205                                                                final int roomItemWeight=getRoomWeight(R);
206                                                                if(roomItemWeight>maxWeight)
207                                                                {
208                                                                        int totOver=roomItemWeight-maxWeight;
209                                                                        Room targetRoom = null;
210                                                                        int smallestCount=Integer.MAX_VALUE/2;
211                                                                        for(int d=0;d<Directions.NUM_DIRECTIONS();d++)
212                                                                        {
213                                                                                final Room R2=R.getRoomInDir(d);
214                                                                                final Exit E2=R.getExitInDir(d);
215                                                                                if((R2!=null)&&(E2!=null)
216                                                                                &&(E2.isOpen())&&(!CMLib.flags().isAiryRoom(R2)))
217                                                                                {
218                                                                                        int rct=this.getRoomWeight(R2);
219                                                                                        if(rct < smallestCount)
220                                                                                        {
221                                                                                                targetRoom=R2;
222                                                                                                smallestCount=rct;
223                                                                                        }
224                                                                                }
225                                                                        }
226                                                                        if(targetRoom != null)
227                                                                        {
228                                                                                for(int ri=R.numItems()-1;ri>=0 && totOver>0;ri--,totOver--)
229                                                                                {
230                                                                                        Item I=R.getItem(ri);
231                                                                                        if((I!=null)&&(I.container()==null))
232                                                                                        {
233                                                                                                targetRoom.moveItemTo(I);
234                                                                                                totOver -= I.phyStats().weight();
235                                                                                        }
236                                                                                }
237                                                                        }
238                                                                }
239                                                        }
240                                                }
241                                        }
242                                }
243                                catch(Exception e)
244                                {
245                                        Log.errOut("Prop_ReqCapacity",e);
246                                }
247                        }
248                }
249        }

251        @Override
252        public boolean okMessage(final Environmental myHost, final CMMsg msg)
253        {
254                final Physical affected=this.affected;
255                if(affected!=null)
256                {
257                        if(System.currentTimeMillis() > lastCheck)
258                        {
259                                this.lastCheck=System.currentTimeMillis() + (30 * 60 * 1000);
260                                CMLib.threads().executeRunnable(new Runnable()
261                                {
262                                        @Override
263                                        public void run()
264                                        {
265                                                overflowCheck();
266                                        }
267                                });
268                        }
269                        
270                        switch(msg.targetMinor())
271                        {
272                        case CMMsg.TYP_ENTER:
273                                if((msg.target() instanceof Room)
274                                &&(peopleCap<Integer.MAX_VALUE)
275                                &&((!indoorOnly)||((((Room)msg.target()).domainType()&Room.INDOORS)==Room.INDOORS))
276                                &&((msg.amITarget(affected))||(msg.tool()==affected)||(affected instanceof Area)))
277                                {
278                                        if(((Room)msg.target()).numInhabitants()>=peopleCap)
279                                        {
280                                                msg.source().tell(L("No more people can fit in there."));
281                                                if(msg.source().isMonster())
282                                                {
283                                                        final MOB M=msg.source().amUltimatelyFollowing();
284                                                        if((M!=null)&&(!M.isMonster())&&(M.location()==(Room)msg.target()))
285                                                                M.tell(L("No more people can fit in here."));
286                                                }
287                                                return false;
288                                        }
289                                        if(((Room)msg.target()).numPCInhabitants()>=playerCap)
290                                        {
291                                                msg.source().tell(L("No more players can fit in there."));
292                                                return false;
293                                        }
294                                        if(msg.source().isMonster()
295                                        && (((Room)msg.target()).numInhabitants()-((Room)msg.target()).numPCInhabitants())>=mobCap)
296                                        {
297                                                msg.source().tell(L("No more MOBs can fit in there."));
298                                                final MOB M=msg.source().amUltimatelyFollowing();
299                                                if((!M.isMonster())&&(M.location()==(Room)msg.target()))
300                                                        M.tell(L("No more people can fit in here."));
301                                                return false;
302                                        }
303                                }
304                                break;
305                        case CMMsg.TYP_PUSH:
306                        case CMMsg.TYP_PULL:
307                                if((msg.tool() != affected)
308                                &&(msg.tool() != myHost))
309                                        break;
310                                //$FALL-THROUGH$
311                        case CMMsg.TYP_DROP:
312                        case CMMsg.TYP_ITEMGENERATED:
313                                if((msg.target() instanceof Item)
314                                &&(msg.source().location()!=null)
315                                &&((!msg.targetMajor(CMMsg.MASK_INTERMSG))||(!containersOk))) // intermsgs are PUTs on the ground
316                                {
317                                        final Item targetI=(Item)msg.target();
318                                        Room R=null;
319                                        if(affected instanceof Room)
320                                        {
321                                                R=(Room)affected;
322                                                if((msg.source().location() != R)
323                                                &&(msg.targetMinor()!=CMMsg.TYP_PUSH)
324                                                &&(msg.targetMinor()!=CMMsg.TYP_PULL))
325                                                        break;
326                                        }
327                                        else
328                                        if(myHost instanceof Room)
329                                        {
330                                                R=(Room)myHost;
331                                                if((msg.source().location() != R)
332                                                &&(msg.targetMinor()!=CMMsg.TYP_PUSH)
333                                                &&(msg.targetMinor()!=CMMsg.TYP_PULL))
334                                                        break;
335                                        }
336                                        else
337                                                R=msg.source().location();
338                                        if((!indoorOnly)||((R.domainType()&Room.INDOORS)==Room.INDOORS))
339                                        {
340                                                if(itemCap<Integer.MAX_VALUE)
341                                                {
342                                                        int soFar=0;
343                                                        int rawResources=0;
344                                                        for(int i=0;i<R.numItems();i++)
345                                                        {
346                                                                final Item I=R.getItem(i);
347                                                                if(I instanceof RawMaterial)
348                                                                        rawResources++;
349                                                                if((I!=null)&&(I.container()==null))
350                                                                        soFar++;
351                                                        }
352                                                        if(soFar>=itemCap)
353                                                        {
354                                                                msg.source().tell(L("There is no more room in here to drop @x1.",msg.target().Name()));
355                                                                if((rawResources>0)&&(CMath.div(rawResources,itemCap)>0.5))
356                                                                        msg.source().tell(L("You should consider bundling up some of those resources."));
357                                                                return false;
358                                                        }
359                                                }
360                                                if(maxWeight<Integer.MAX_VALUE)
361                                                {
362                                                        int soFar=getRoomWeight(R);
363                                                        if((soFar+targetI.phyStats().weight())>=maxWeight)
364                                                        {
365                                                                msg.source().tell(L("There is no room in here to put @x1.",targetI.Name()));
366                                                                return false;
367                                                        }
368                                                }
369                                        }
370                                }
371                                break;
372                        }
373                }
374                return super.okMessage(myHost,msg);
375        }
