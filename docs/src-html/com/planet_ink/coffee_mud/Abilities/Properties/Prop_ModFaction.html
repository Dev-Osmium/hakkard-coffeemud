035public class Prop_ModFaction extends Property
036{
037        @Override
038        public String ID()
039        {
040                return "Prop_ModFaction";
041        }
042
043        @Override
044        public String name()
045        {
046                return "Modifying Faction Gained";
047        }
048
049        @Override
050        protected int canAffectCode()
051        {
052                return Ability.CAN_MOBS | Ability.CAN_ITEMS | Ability.CAN_AREAS | Ability.CAN_ROOMS;
053        }
054
055        protected String        operationFormula        = "";
056        protected String        factionID                       = "";
057        protected boolean       reactions                       = false;
058        protected boolean       gainonly                        = false;
059        protected boolean       lossonly                        = false;
060
061        protected CMath.CompiledFormula operation                       = null;
062        protected MaskingLibrary.CompiledZMask                  mask                            = null;
063        
064        @Override
065        public String accountForYourself()
066        {
067                final Faction F=(factionID.length()>0)?CMLib.factions().getFaction(factionID):null;
068                final String gainOrLoss=(gainonly)?"gained ":lossonly?"lost ":"";
069                final String factionName;
070                if(factionID.length()==0)
071                        factionName = "any faction";
072                else
073                if (reactions)
074                        factionName = "certain factions";
075                else
076                if(F==null)
077                        factionName = "some faction";
078                else
079                        factionName = F.name();
080                return "Modifies "+gainOrLoss+"faction with "+factionName+": "+operationFormula;
081        }
082
083        public int translateAmount(int amount, String val)
084        {
085                if(amount<0)
086                        amount=-amount;
087                if(val.endsWith("%"))
088                        return (int)Math.round(CMath.mul(amount,CMath.div(CMath.s_int(val.substring(0,val.length()-1)),100)));
089                return CMath.s_int(val);
090        }
091
092        public String translateNumber(String val)
093        {
094                if(val.endsWith("%"))
095                        return "@x1 * (" + val.substring(0,val.length()-1) + " / 100)";
096                return Integer.toString(CMath.s_int(val));
097        }
098
099        @Override
100        public void setMiscText(String newText)
101        {
102                super.setMiscText(newText);
103                operation = null;
104                factionID = "";
105                mask=null;
106                gainonly=false;
107                lossonly=false;
108                reactions=false;
109                String s=newText.trim();
110                int x=s.indexOf(';');
111                if(x>=0)
112                {
113                        mask=CMLib.masking().getPreCompiledMask(s.substring(x+1).trim());
114                        s=s.substring(0,x).trim();
115                }
116                x=s.indexOf(':');
117                if(x>=0)
118                {
119                        factionID=s.substring(0,x).trim();
120                        if(factionID.startsWith("+"))
121                        {
122                                gainonly=true;
123                                factionID=factionID.substring(1).trim();
124                        }
125                        else
126                        if(factionID.startsWith("-"))
127                        {
128                                lossonly=true;
129                                factionID=factionID.substring(1).trim();
130                        }
131                        s=s.substring(x+1).trim();
132                }
133                if(factionID.trim().equalsIgnoreCase("REACTION"))
134                {
135                        factionID="";
136                        reactions=true;
137                }
138
139                operationFormula="Amount "+s;
140                if(s.startsWith("="))
141                        operation = CMath.compileMathExpression(translateNumber(s.substring(1)).trim());
142                else
143                if(s.startsWith("+"))
144                        operation = CMath.compileMathExpression("@x1 + "+translateNumber(s.substring(1)).trim());
145                else
146                if(s.startsWith("-"))
147                        operation = CMath.compileMathExpression("@x1 - "+translateNumber(s.substring(1)).trim());
148                else
149                if(s.startsWith("*"))
150                        operation = CMath.compileMathExpression("@x1 * "+translateNumber(s.substring(1)).trim());
151                else
152                if(s.startsWith("/"))
153                        operation = CMath.compileMathExpression("@x1 / "+translateNumber(s.substring(1)).trim());
154                else
155                if(s.startsWith("(")&&(s.endsWith(")")))
156                {
157                        operationFormula="Amount ="+s;
158                        operation = CMath.compileMathExpression(s);
159                }
160                else
161                        operation = CMath.compileMathExpression(translateNumber(s.trim()));
162                operationFormula=CMStrings.replaceAll(operationFormula, "@x1", "Amount");
163        }
164
165        @Override
166        public boolean okMessage(final Environmental myHost, final CMMsg msg)
167        {
168                if((msg.sourceMinor()==CMMsg.TYP_FACTIONCHANGE)
169                &&(operation != null)
170                &&(((msg.target()==affected)&&(affected instanceof MOB))
171                   ||((affected instanceof Item)
172                           &&(msg.source()==((Item)affected).owner())
173                           &&(((Item)affected).amBeingWornProperly()))
174                   ||(affected instanceof Room)
175                   ||(affected instanceof Area))
176                &&(msg.value()!=Integer.MAX_VALUE)
177                &&(msg.value()!=Integer.MIN_VALUE)
178                &&((!gainonly)||(msg.value()>0))
179                &&((!lossonly)||(msg.value()<0))
180                &&((factionID.length()==0)||(msg.othersMessage().equalsIgnoreCase(factionID)))
181                )
182                {
183                        if(reactions)
184                        {
185                                final Faction F=CMLib.factions().getFaction(msg.othersMessage());
186                                if((F==null)||(!F.reactions().hasMoreElements()))
187                                        return super.okMessage(myHost,msg);
188                        }
189                        if(mask!=null)
190                        {
191                                if(affected instanceof Item)
192                                {
193                                        if((msg.target()==null)||(!(msg.target() instanceof MOB))||(!CMLib.masking().maskCheck(mask,msg.target(),true)))
194                                                return super.okMessage(myHost,msg);
195                                }
196                                else
197                                if(!CMLib.masking().maskCheck(mask,msg.source(),true))
198                                        return super.okMessage(myHost,msg);
199                        }
200                        msg.setValue((int)Math.round(CMath.parseMathExpression(operation, new double[]{msg.value()}, 0.0)));
201                }
202                return super.okMessage(myHost,msg);
203        }
