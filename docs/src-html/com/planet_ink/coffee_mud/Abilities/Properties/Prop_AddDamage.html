035public class Prop_AddDamage extends Property implements TriggeredAffect
036{
037        @Override
038        public String ID()
039        {
040                return "Prop_AddDamage";
041        }
042
043        @Override
044        public String name()
045        {
046                return "Additional Damage";
047        }
048
049        @Override
050        protected int canAffectCode()
051        {
052                return Ability.CAN_MOBS|Ability.CAN_ITEMS;
053        }
054
055        protected int weaponDamageType=Weapon.TYPE_NATURAL;
056        protected int typeOfEffect=CMMsg.TYP_WEAPONATTACK;
057        protected double pctDamage=0.0;
058        protected int bonusDamage=0;
059        protected volatile boolean norecurse=false;
060
061        @Override
062        public int triggerMask()
063        {
064                return TriggeredAffect.TRIGGER_HITTING_WITH;
065        }
066
067        @Override
068        public String accountForYourself()
069        {
070                final String id="Does extra damage of the following amount and types: "+text();
071                return id;
072        }
073
074        @Override
075        public void setMiscText(String newMiscText)
076        {
077                super.setMiscText(newMiscText);
078                final List<String> parms=CMParms.parse(newMiscText.toUpperCase());
079                for(String s : parms)
080                {
081                        if(s.startsWith("+"))
082                                s=s.substring(1);
083                        if(CMath.isPct(s))
084                                pctDamage=CMath.s_pct(s);
085                        else
086                        if(CMath.isInteger(s))
087                                bonusDamage=CMath.s_int(s);
088                        else
089                        {
090                                boolean done=false;
091                                for(int i=0;i<Weapon.TYPE_DESCS.length;i++)
092                                {
093                                        final String type=Weapon.TYPE_DESCS[i];
094                                        if(type.equals(s))
095                                        {
096                                                weaponDamageType=i;
097                                                done=true;
098                                                break;
099                                        }
100                                }
101                                if(!done)
102                                for(int i=0;i<CMMsg.TYPE_DESCS.length;i++)
103                                {
104                                        final String type=CMMsg.TYPE_DESCS[i];
105                                        if(type.equals(s))
106                                        {
107                                                typeOfEffect=i;
108                                                done=true;
109                                                break;
110                                        }
111                                }
112                                if(!done)
113                                for(int i=0;i<Weapon.TYPE_DESCS.length;i++)
114                                {
115                                        final String type=Weapon.TYPE_DESCS[i];
116                                        if(type.startsWith(s))
117                                        {
118                                                weaponDamageType=i;
119                                                done=true;
120                                                break;
121                                        }
122                                }
123                                if(!done)
124                                for(int i=0;i<Weapon.TYPE_DESCS.length;i++)
125                                {
126                                        final String type=Weapon.TYPE_DESCS[i];
127                                        if(type.startsWith(s))
128                                        {
129                                                typeOfEffect=i;
130                                                done=true;
131                                                break;
132                                        }
133                                }
134                                if(!done)
135                                for(int i=0;i<RawMaterial.CODES.NAMES().length;i++)
136                                {
137                                        final String type=RawMaterial.CODES.NAMES()[i];
138                                        if(type.equals(s))
139                                        {
140                                                done=true; // just eat it
141                                                break;
142                                        }
143                                }
144                                if((!done)&&(!s.equals("ALL")))
145                                        Log.errOut("Prop_AddDamage","Unknown weapon type/attack: "+s+" in "+CMLib.map().getDescriptiveExtendedRoomID(CMLib.map().roomLocation(affected)));
146                        }
147                }
148        }
149
150        protected final int getDamage(final CMMsg msg)
151        {
152                final int dmg = (int)CMath.round(CMath.mul(msg.value(), pctDamage)) + bonusDamage;
153                if(dmg < 0)
154                        return 0;
155                return dmg;
156        }
157
158        @Override
159        public void executeMsg(final Environmental myHost, final CMMsg msg)
160        {
161                super.executeMsg(myHost,msg);
162                MOB mob=null;
163                if(affected instanceof Item)
164                {
165                        if(((Item)affected).owner() instanceof MOB)
166                        {
167                                mob=(MOB)((Item)affected).owner();
168                        }
169                        else
170                                return;
171                }
172                else
173                if(affected instanceof MOB)
174                        mob=(MOB)affected;
175                else
176                        return;
177                if((msg.targetMinor()==CMMsg.TYP_DAMAGE)
178                &&(msg.value()>0)
179                &&(msg.target() instanceof MOB)
180                &&(!((MOB)msg.target()).amDead())
181                &&(msg.tool()!=this)
182                &&(msg.source().location()!=null))
183                {
184                        if(((affected instanceof Armor)||(affected instanceof Shield))
185                        &&(msg.amITarget(mob))
186                        &&(!msg.amISource(mob))
187                        &&(CMLib.dice().rollPercentage()>32+msg.source().charStats().getStat(CharStats.STAT_DEXTERITY))
188                        &&(msg.source().rangeToTarget()==0)
189                        &&((msg.targetMajor(CMMsg.MASK_HANDS))||(msg.targetMajor(CMMsg.MASK_MOVE))))
190                        {
191                                final CMMsg msg2=CMClass.getMsg(mob,msg.source(),this,CMMsg.MSG_CAST_ATTACK_VERBAL_SPELL,null);
192                                if(msg.source().location().okMessage(msg.source(),msg2))
193                                {
194                                        msg.source().location().send(msg.source(),msg2);
195                                        if(msg2.value()<=0)
196                                        {
197                                                final int damage=getDamage(msg);
198                                                CMLib.combat().postDamage(mob,msg.source(),affected,damage,CMMsg.MASK_MALICIOUS|CMMsg.MASK_ALWAYS|typeOfEffect,weaponDamageType,
199                                                         "^F^<FIGHT^><S-YOUPOSS> <O-NAME> <DAMAGE> <T-NAME>!^</FIGHT^>^?");
200                                        }
201                                }
202                        }
203                        else
204                        if((msg.tool()==affected)
205                        &&(!msg.amITarget(mob))
206                        &&(msg.amISource(mob))
207                        &&(!(msg.tool() instanceof Wand)))
208                        {
209                                final int damage=getDamage(msg);
210                                final String str=L("^F^<FIGHT^><S-YOUPOSS> <O-NAME> <DAMAGE> <T-NAME>!^</FIGHT^>^?");
211                                synchronized(this)
212                                {
213                                        if(!norecurse)
214                                        {
215                                                norecurse=true;
216                                                try
217                                                {
218                                                        CMLib.combat().postDamage(msg.source(),(MOB)msg.target(),affected,Math.round(damage),
219                                                        CMMsg.MASK_MALICIOUS|CMMsg.MASK_ALWAYS|typeOfEffect,weaponDamageType,str);
220                                                }
221                                                finally
222                                                {
223                                                        norecurse=false;
224                                                }
225                                        }
226                                }
227                        }
228                        else
229                        if((mob==affected)
230                        &&(!msg.amITarget(mob))
231                        &&(msg.amISource(mob))
232                        &&(msg.tool() instanceof Weapon)
233                        &&(!(msg.tool() instanceof Wand)))
234                        {
235                                final int damage=getDamage(msg);
236                                final String str=L("^F^<FIGHT^><S-NAME> <DAMAGE> <T-NAME>!^</FIGHT^>^?");
237                                CMLib.combat().postDamage(mob,(MOB)msg.target(),this,Math.round(damage),
238                                          CMMsg.MASK_MALICIOUS|CMMsg.MASK_ALWAYS|typeOfEffect,weaponDamageType,str);
239                        }
240                }
241        }
