074        @Override
075        public int getPrice()
076        {
077                if(text().length()==0)
078                        return 100000;
079                final String s=text();
080                int index=s.length();
081                while((--index)>=0)
082                {
083                        if((!Character.isDigit(s.charAt(index)))
084                        &&(!Character.isWhitespace(s.charAt(index))))
085                                break;
086                }
087                int price=CMath.s_int(s.substring(index+1).trim());
088
089                if(price<=0)
090                        price=100000;
091                return price;
092        }

146        @Override
147        public int backTaxes()
148        {
149                if(text().indexOf('/')<0)
150                        return 0;
151                final int x=text().indexOf("TAX",text().indexOf('/'));
152                if(x<0)
153                        return 0;
154                final String s=CMParms.parse(text().substring(x+3)).firstElement();
155                return CMath.s_int(s.substring(0,s.length()-1));
156        }

185        @Override
186        public void updateTitle()
187        {
188                if(affected instanceof Area)
189                        CMLib.database().DBUpdateArea(((Area)affected).name(),(Area)affected);
190                else
191                if(affected instanceof Room)
192                        Log.errOut("Prop_AreaForSale","Prop_AreaForSale goes on an Area, NOT "+CMLib.map().getDescriptiveExtendedRoomID((Room)affected));
193                else
194                {
195                        final Area A=CMLib.map().getArea(landPropertyID());
196                        if(A!=null)
197                                CMLib.database().DBUpdateArea(A.Name(),A);
198                }
199        }

246        @Override
247        public void executeMsg(final Environmental myHost, final CMMsg msg)
248        {
249                super.executeMsg(myHost,msg);
250                if(((msg.sourceMinor()==CMMsg.TYP_SHUTDOWN)
251                                ||((msg.targetMinor()==CMMsg.TYP_EXPIRE)&&(msg.target() instanceof Room))
252                                ||(msg.sourceMinor()==CMMsg.TYP_ROOMRESET))
253                &&(affected instanceof Area)
254                &&((System.currentTimeMillis()-lastMobSave)>360000))
255                {
256                        lastMobSave=System.currentTimeMillis();
257                        final List<Room> V=getAllTitledRooms();
258                        for(int v=0;v<V.size();v++)
259                        {
260                                Room R=V.get(v);
261                                synchronized(("SYNC"+R.roomID()).intern())
262                                {
263                                        R=CMLib.map().getRoom(R);
264                                        lastMobSave=System.currentTimeMillis();
265                                        final Vector<MOB> mobs=new Vector<MOB>();
266                                        for(int m=0;m<R.numInhabitants();m++)
267                                        {
268                                                final MOB M=R.fetchInhabitant(m);
269                                                if((M!=null)
270                                                &&(M.isSavable())
271                                                &&(M.getStartRoom()==R)
272                                                &&((M.basePhyStats().rejuv()==0)||(M.basePhyStats().rejuv()==PhyStats.NO_REJUV)))
273                                                        mobs.addElement(M);
274                                        }
275                                        if(!CMSecurity.isSaveFlag(CMSecurity.SaveFlag.NOPROPERTYMOBS))
276                                                CMLib.database().DBUpdateTheseMOBs(R,mobs);
277                                }
278                        }
279                        lastMobSave=System.currentTimeMillis();
280                }
281        }
282
283        @Override
284        public List<Room> getAllTitledRooms()
285        {
286                final List<Room> V=new Vector<Room>();
287                Area A=null;
288                if(affected instanceof Area)
289                        A=(Area)affected;
290                else
291                if(affected instanceof Room)
292                        V.add((Room)affected);
293                else
294                        A=CMLib.map().getArea(landPropertyID());
295                if(A!=null)
296                {
297                        for(final Enumeration<Room> e=A.getProperMap();e.hasMoreElements();)
298                                V.add(e.nextElement());
299                }
300                return V;
301        }
302
303        @Override
304        public List<Room> getConnectedPropertyRooms()
305        {
306                return getAllTitledRooms();
307        }

309        // update lot, since its called by the savethread, ONLY worries about itself
310        @Override
311        public void updateLot(Set<String> optPlayerList)
312        {
313                if(((System.currentTimeMillis()-lastCall)>360000)
314                &&(CMProps.getBoolVar(CMProps.Bool.MUDSTARTED)))
315                {
316                        final List<Room> V=getAllTitledRooms();
317                        for(int v=0;v<V.size();v++)
318                        {
319                                final Room R=V.get(v);
320                                lastCall=System.currentTimeMillis();
321                                final Integer lastItemNum=lastItemNums.get(R);
322                                lastItemNums.put(R,Integer.valueOf(Prop_RoomForSale.updateLotWithThisData(R,this,false,false,optPlayerList,(lastItemNum==null)?-1:lastItemNum.intValue())));
323                        }
324                        lastCall=System.currentTimeMillis();
325                        Area A=null;
326                        if(affected instanceof Area)
327                                A=(Area)affected;
328                        else
329                                A=CMLib.map().getArea(landPropertyID());
330                        if((A!=null)&&(lastDayDone!=A.getTimeObj().getDayOfMonth()))
331                        {
332                                lastDayDone=A.getTimeObj().getDayOfMonth();
333                                if((getOwnerName().length()>0)&&rentalProperty())
334                                {
335                                        if(Prop_RoomForSale.doRentalProperty(A,A.Name(),getOwnerName(),getPrice()))
336                                        {
337                                                setOwnerName("");
338                                                CMLib.database().DBUpdateArea(A.Name(),A);
339                                        }
340                                }
341                        }
342                }
343        }
