037public class Prop_InstantDeath extends Property
038{
039        @Override
040        public String ID()
041        {
042                return "Prop_InstantDeath";
043        }
044
045        @Override
046        public long flags()
047        {
048                return super.flags()|Ability.FLAG_POTENTIALLY_DEADLY;
049        }
050
051        @Override
052        protected int canAffectCode()
053        {
054                return Ability.CAN_ROOMS|Ability.CAN_AREAS|Ability.CAN_ITEMS|Ability.CAN_MOBS;
055        }
056
057        protected CompiledZMask mask=null;
058        protected volatile boolean[] killTrigger={false};
059
060        public Prop_InstantDeath()
061        {
062                super();
063        }
064
065        @Override
066        public void setMiscText(String newMiscText)
067        {
068                super.setMiscText(newMiscText);
069                final String maskStr=CMParms.getParmStr(newMiscText,"mask","");
070                mask=null;
071                if((maskStr!=null)&&(maskStr.trim().length()>0))
072                        mask=CMLib.masking().getPreCompiledMask(maskStr);
073        }
074
075        @Override
076        public String accountForYourself()
077        {
078                return "instant killing";
079        }
080
081        public Set<MOB> getEveryoneHere(MOB spareMe, Room R)
082        {
083                final Set<MOB> V=new HashSet<MOB>();
084                if(R==null)
085                        return V;
086                for(int i=0;i<R.numInhabitants();i++)
087                {
088                        final MOB M=R.fetchInhabitant(i);
089                        if((spareMe!=null)&&(spareMe==M))
090                                continue;
091                        if((M!=null)
092                        &&(!CMSecurity.isAllowed(M,R,CMSecurity.SecFlag.IMMORT))
093                        &&((mask==null)||(CMLib.masking().maskCheck(mask, M, false))))
094                                V.add(M);
095                }
096                return V;
097                //CMLib.combat().postDeath(null,M,null);
098        }
099
100        protected MOB getTickersMOB(Tickable ticking)
101        {
102                if(ticking==null)
103                        return null;
104
105                if(ticking instanceof MOB)
106                        return (MOB)ticking;
107                else
108                if(ticking instanceof Item)
109                {
110                        if(((Item)ticking).owner() != null)
111                        {
112                                if(((Item)ticking).owner() instanceof MOB)
113                                        return (MOB)((Item)ticking).owner();
114                        }
115
116                }
117                return null;
118        }
119
120        protected Room getTickersRoom(Tickable ticking)
121        {
122                if(ticking==null)
123                        return null;
124
125                if(ticking instanceof Room)
126                        return (Room)ticking;
127
128                final MOB mob=getTickersMOB(ticking);
129                if(mob!=null)
130                        return mob.location();
131
132                if(ticking instanceof Item)
133                {
134                        if(((Item)ticking).owner() != null)
135                        {
136                                if(((Item)ticking).owner() instanceof Room)
137                                        return (Room)((Item)ticking).owner();
138                        }
139
140                }
141                return null;
142        }
143
144        public Set<MOB> getDeadMOBsFrom(Environmental whoE)
145        {
146                if(whoE instanceof MOB)
147                {
148                        final MOB mob=(MOB)whoE;
149                        final Room room=mob.location();
150                        if(room!=null)
151                                return getEveryoneHere(mob,room);
152                }
153                else
154                if(whoE instanceof Item)
155                {
156                        final Item item=(Item)whoE;
157                        final Environmental E=item.owner();
158                        if(E!=null)
159                        {
160                                final Room room=getTickersRoom(whoE);
161                                if(room!=null)
162                                {
163                                        if((E instanceof MOB)&&((mask==null)||(CMLib.masking().maskCheck(mask, E, false))))
164                                                return new XHashSet<MOB>((MOB)E);
165                                        else
166                                        if(E instanceof Room)
167                                                return getEveryoneHere(null,(Room)E);
168                                        room.recoverRoomStats();
169                                }
170                        }
171                }
172                else
173                if(whoE instanceof Room)
174                        return getEveryoneHere(null,(Room)whoE);
175                else
176                if(whoE instanceof Area)
177                {
178                        final Set<MOB> allMobs=new HashSet<MOB>();
179                        for(final Enumeration<Room> r=((Area)whoE).getMetroMap();r.hasMoreElements();)
180                        {
181                                final Room R=r.nextElement();
182                                allMobs.addAll(getEveryoneHere(null,R));
183                        }
184                }
185                return new HashSet<MOB>();
186        }
187
188        @Override
189        public boolean tick(final Tickable ticking, final int tickID)
190        {
191                if(tickID!=Tickable.TICKID_MISCELLANEOUS)
192                        return super.tick(ticking, tickID);
193                while(killTrigger[0])
194                {
195                        final LinkedList<MOB> killThese=new LinkedList<MOB>();
196                        synchronized(killTrigger)
197                        {
198                                killThese.addAll(getDeadMOBsFrom(affected));
199                                killTrigger[0]=false;
200                        }
201                        for(final MOB M : killThese)
202                        {
203                                CMLib.combat().postDeath(null, M, null);
204                        }
205                }
206                return false;
207        }
208
209        @Override
210        public void executeMsg(final Environmental affecting, final CMMsg msg)
211        {
212                super.executeMsg(affecting,msg);
213                if(msg.amITarget(affecting))
214                {
215                        boolean activated=false;
216                        if(affecting instanceof MOB)
217                        {
218                                if((msg.targetMajor(CMMsg.MASK_MALICIOUS))
219                                &&(!msg.source().isMonster()))
220                                        activated=true;
221                        }
222                        else
223                        if((affecting instanceof Food)
224                        ||(affecting instanceof Drink))
225                        {
226                                if((msg.targetMinor()==CMMsg.TYP_EAT)
227                                ||(msg.targetMinor()==CMMsg.TYP_DRINK))
228                                        activated=true;
229                        }
230                        else
231                        if((affecting instanceof Armor)
232                        ||(affecting instanceof Weapon))
233                        {
234                                if((msg.targetMinor()==CMMsg.TYP_WEAR)
235                                ||(msg.targetMinor()==CMMsg.TYP_HOLD)
236                                ||(msg.targetMinor()==CMMsg.TYP_WIELD))
237                                        activated=true;
238                        }
239                        else
240                        if(affecting instanceof Item)
241                        {
242                                if((msg.targetMinor()==CMMsg.TYP_GET)||(msg.targetMinor()==CMMsg.TYP_PUSH)||(msg.targetMinor()==CMMsg.TYP_PULL))
243                                        activated=true;
244                        }
245                        else
246                                activated=true;
247                        if(activated)
248                        {
249                                synchronized(killTrigger)
250                                {
251                                        killTrigger[0]=true;
252                                        if(!CMLib.threads().isTicking(this, Tickable.TICKID_MISCELLANEOUS))
253                                                CMLib.threads().startTickDown(this, Tickable.TICKID_MISCELLANEOUS, 500,1);
254                                }
255                        }
256                }
257        }
