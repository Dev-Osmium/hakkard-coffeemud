058        protected void fillCluster(final Room R, final List<Room> V)
059        {
060                V.add(R);
061                for(int d=Directions.NUM_DIRECTIONS()-1;d>=0;d--)
062                {
063                        final Room R2=R.getRoomInDir(d);
064                        if((R2!=null)&&(R2.roomID().length()>0)&&(!V.contains(R2)))
065                        {
066                                final Ability A=R2.fetchEffect(ID());
067                                if((R2.getArea()==R.getArea())&&(A!=null))
068                                        fillCluster(R2,V);
069                                else
070                                {
071                                        V.remove(R); // purpose here is to put the "front" door up front.
072                                        V.add(0,R);
073                                }
074                        }
075                }
076        }

078        @Override
079        public List<Room> getConnectedPropertyRooms()
080        {
081                final List<Room> V=new ArrayList<Room>();
082                Room R=null;
083                if(affected instanceof Room)
084                        R=(Room)affected;
085                else
086                        R=CMLib.map().getRoom(landPropertyID());
087                if(R!=null)
088                {
089                        fillCluster(R,V);
090                        String uniqueID="LOTS_PROPERTY_"+this;
091                        if(V.size()>0)
092                                uniqueID="LOTS_PROPERTY_"+CMLib.map().getExtendedRoomID(V.get(0));
093                        for(final Iterator<Room> r=V.iterator();r.hasNext();)
094                        {
095                                Ability A=null;
096                                R=r.next();
097                                if(R!=null)
098                                        A=R.fetchEffect(ID());
099                                if(A instanceof Prop_LotsForSale)
100                                        ((Prop_LotsForSale)A).uniqueLotID=uniqueID;
101                        }
102                }
103                else
104                        uniqueLotID="";
105                return V;
106
107        }

109        protected boolean isRetractableLink(final Map<Room,Boolean> recurseChkRooms, final Room fromRoom, final Room theRoom)
110        {
111                if(theRoom==null)
112                        return true;
113
114                if((theRoom.roomID().length()>0)
115                &&((CMLib.law().getLandTitle(theRoom)==null)
116                        ||(CMLib.law().getLandTitle(theRoom).getOwnerName().length()>0)))
117                {
118                        if(recurseChkRooms != null)
119                                recurseChkRooms.put(theRoom, Boolean.valueOf(false));
120                        return false;
121                }
122
123                for(int d=Directions.NUM_DIRECTIONS()-1;d>=0;d--)
124                {
125                        final Room R=theRoom.rawDoors()[d];
126                        if(R!=null)
127                        {
128                                if((recurseChkRooms != null)
129                                &&(recurseChkRooms.containsKey(R)))
130                                        return recurseChkRooms.get(theRoom).booleanValue();
131                                if((R!=fromRoom)
132                                &&(R.roomID().length()>0))
133                                {
134                                        if((CMLib.law().getLandTitle(R)==null)||(CMLib.law().getLandTitle(R).getOwnerName().length()>0))
135                                        {
136                                                if(recurseChkRooms != null)
137                                                        recurseChkRooms.put(theRoom, Boolean.valueOf(false));
138                                                return false;
139                                        }
140                                        if((recurseChkRooms != null)
141                                        &&(!isRetractableLink(recurseChkRooms,theRoom,R)))
142                                        {
143                                                recurseChkRooms.put(theRoom, Boolean.valueOf(false));
144                                                return false;
145                                        }
146                                }
147                        }
148                }
149                if(recurseChkRooms != null)
150                        recurseChkRooms.put(theRoom, Boolean.valueOf(true));
151                return true;
152        }

176        protected boolean retractRooms(final Room R, final List<Runnable> postWork)
177        {
178                boolean updateExits=false;
179                boolean foundOne=false;
180                boolean didAnything = false;
181                final Map<Room,Boolean> checkedRetractRooms;
182                if(super.gridLayout())
183                        checkedRetractRooms = new Hashtable<Room,Boolean>();
184                else
185                        checkedRetractRooms = null;
186                for(int d=0;d<Directions.NUM_DIRECTIONS();d++)
187                {
188                        if(d==Directions.GATE)
189                                continue;
190                        final Room R2=R.rawDoors()[d];
191                        if((R2!=null)&&((!R2.isSavable())||(R2.roomID().length()==0)))
192                                continue;
193                        Exit E=R.getRawExit(d);
194                        if(checkedRetractRooms != null)
195                                checkedRetractRooms.clear();
196                        if((R2!=null)&&(R2.rawDoors()[Directions.getOpDirectionCode(d)]==R))
197                                foundOne=true;
198                        else
199                        if((R2!=null)
200                        &&(isRetractableLink(checkedRetractRooms,R,R2)))
201                        {
202                                R.rawDoors()[d]=null;
203                                R.setRawExit(d,null);
204                                updateExits=true;
205                                postWork.add(new Runnable()
206                                {
207                                        final Room room=R2;
208
209                                        @Override
210                                        public void run()
211                                        {
212                                                CMLib.map().obliterateMapRoom(room);
213                                        }
214                                });
215                                didAnything=true;
216                        }
217                        else
218                        if((E!=null)&&(E.hasALock())&&(E.isGeneric()))
219                        {
220                                E.setKeyName("");
221                                E.setDoorsNLocks(E.hasADoor(),E.isOpen(),E.defaultsClosed(),false,false,false);
222                                updateExits=true;
223                                if(R2!=null)
224                                {
225                                        E=R2.getRawExit(Directions.getOpDirectionCode(d));
226                                        if((E!=null)&&(E.hasALock())&&(E.isGeneric()))
227                                        {
228                                                E.setKeyName("");
229                                                E.setDoorsNLocks(E.hasADoor(),E.isOpen(),E.defaultsClosed(),false,false,false);
230                                                postWork.add(new Runnable()
231                                                {
232                                                        final Room room=R2;
233                                                        @Override
234                                                        public void run()
235                                                        {
236                                                                CMLib.database().DBUpdateExits(room);
237                                                                R2.getArea().fillInAreaRoom(room);
238                                                        }
239                                                });
240                                                didAnything=true;
241                                        }
242                                }
243                        }
244                }
245                if(checkedRetractRooms != null)
246                        checkedRetractRooms.clear();
247                if(!foundOne)
248                {
249                        CMLib.map().obliterateMapRoom(R);
250                        didAnything=true;
251                }
252                else
253                if(updateExits)
254                {
255                        CMLib.database().DBUpdateExits(R);
256                        R.getArea().fillInAreaRoom(R);
257                        didAnything=true;
258                }
259                return didAnything;
260        }

262        public boolean expandRooms(final Room R, final List<Runnable> postWork)
263        {
264                int numberOfPeers = -1;//getConnectedPropertyRooms().size();
265                final boolean doGrid=super.gridLayout();
266                long roomLimit = Long.MAX_VALUE;
267                final Set<Room> updateExits=new HashSet<Room>();
268                Prop_ReqCapacity cap = null;
269                boolean didAnything = false;
270                for(int d=0;d<Directions.NUM_DIRECTIONS();d++)
271                {
272                        if((d==Directions.UP)||(d==Directions.DOWN)||(d==Directions.GATE))
273                                continue;
274                        final Room chkR=R.getRoomInDir(d);
275                        if((chkR==null)&&(numberOfPeers < 0))
276                        {
277                                final List<Room> allRooms = getConnectedPropertyRooms();
278                                if(allRooms.size()>0)
279                                {
280                                        cap = (Prop_ReqCapacity)allRooms.get(0).fetchEffect("Prop_ReqCapacity");
281                                        if(cap != null)
282                                        {
283                                                roomLimit = cap.roomLimit;
284                                        }
285                                }
286                                numberOfPeers = allRooms.size();
287                        }
288                        if((chkR==null)&&(numberOfPeers < roomLimit))
289                        {
290                                numberOfPeers++;
291                                final Room R2=CMClass.getLocale(CMClass.classID(R));
292                                R2.setRoomID(R.getArea().getNewRoomID(R,d));
293                                if(R2.roomID().length()==0)
294                                        continue;
295                                R2.setArea(R.getArea());
296                                final LandTitle oldTitle=CMLib.law().getLandTitle(R);
297                                final LandTitle newTitle;
298                                if((oldTitle!=null)&&(CMLib.law().getLandTitle(R2)==null))
299                                {
300                                        newTitle = oldTitle.generateNextRoomTitle();
301                                        R2.addNonUninvokableEffect((Ability)newTitle);
302                                }
303                                else
304                                        newTitle=null;
305                                R.rawDoors()[d]=R2;
306                                R.setRawExit(d,CMClass.getExit("Open"));
307                                R2.rawDoors()[Directions.getOpDirectionCode(d)]=R;
308                                R2.setRawExit(Directions.getOpDirectionCode(d),CMClass.getExit("Open"));
309                                updateExits.add(R);
310                                if(doGrid)
311                                {
312                                        final PairVector<Room,int[]> rooms=CMLib.tracking().buildGridList(R2, this.getOwnerName(), 100);
313                                        for(int dir=0;dir<Directions.NUM_DIRECTIONS();dir++)
314                                        {
315                                                if(dir==Directions.GATE)
316                                                        continue;
317                                                Room R3=R2.getRoomInDir(dir);
318                                                if(R3 == null)
319                                                {
320                                                        R3=CMLib.tracking().getCalculatedAdjacentRoom(rooms, R3, dir);
321                                                        if(R3!=null)
322                                                        {
323                                                                R2.rawDoors()[dir]=R3;
324                                                                R3.rawDoors()[Directions.getOpDirectionCode(dir)]=R2;
325                                                                updateExits.add(R3);
326                                                        }
327                                                }
328                                        }
329                                }
330                                updateExits.add(R2);
331                                if(CMSecurity.isDebugging(CMSecurity.DbgFlag.PROPERTY))
332                                        Log.debugOut("Lots4Sale",R2.roomID()+" created and put up for sale.");
333                                if(cap != null)
334                                        R2.addNonUninvokableEffect((Ability)cap.copyOf());
335                                postWork.add(new Runnable()
336                                {
337                                        final Room room = R2;
338                                        final LandTitle title=newTitle;
339
340                                        @Override
341                                        public void run()
342                                        {
343                                                CMLib.database().DBCreateRoom(room);
344                                                if(title!=null)
345                                                        CMLib.law().colorRoomForSale(room,title,true);
346                                                room.getArea().fillInAreaRoom(room);
347                                        }
348                                });
349                                didAnything=true;
350                        }
351                }
352                if(updateExits.size()>0)
353                {
354                        didAnything=true;
355                        R.getArea().fillInAreaRoom(R);
356                        postWork.add(new Runnable()
357                        {
358                                final Set<Room> updateExits2=new SHashSet<Room>(updateExits);
359
360                                @Override
361                                public void run()
362                                {
363                                        for(final Room xR : updateExits2)
364                                        {
365                                                CMLib.database().DBUpdateExits(xR);
366                                        }
367                                }
368                        });
369                }
370                return didAnything;
371        }

373        @Override
374        public void updateLot(final Set<String> optPlayerList)
375        {
376                final Environmental EV=affected;
377                if(!(EV instanceof Room))
378                        return;
379                Room R=(Room)EV;
380                boolean didAnything=false;
381                try
382                {
383                        final List<Runnable> postWork=new ArrayList<Runnable>();
384                        synchronized(("SYNC"+R.roomID()).intern())
385                        {
386                                R=CMLib.map().getRoom(R);
387                                lastItemNums=updateLotWithThisData(R,this,true,scheduleReset,optPlayerList,lastItemNums);
388
389                                if(getOwnerName().length()==0)
390                                {
391                                        didAnything = retractRooms(R,postWork) || didAnything;
392                                }
393                                else
394                                if(canGenerateAdjacentRooms(R))
395                                {
396                                        didAnything = expandRooms(R,postWork) || didAnything;
397                                }
398                        }
399                        for(final Runnable run : postWork)
400                                run.run();
401                        scheduleReset=false;
402                }
403                finally
404                {
405                        if(didAnything)
406                                getConnectedPropertyRooms(); // recalculates the unique id for this lot of rooms
407                }
408        }
