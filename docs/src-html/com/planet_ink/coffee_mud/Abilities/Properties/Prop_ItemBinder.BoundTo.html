035public class Prop_ItemBinder extends Property implements TriggeredAffect
036{
037        @Override
038        public String ID()
039        {
040                return "Prop_ItemBinder";
041        }
042
043        @Override
044        public String name()
045        {
046                return "Allows an item to be bound to player, account, or clan.";
047        }
048
049        @Override
050        protected int canAffectCode()
051        {
052                return Ability.CAN_ITEMS;
053        }
054        
055        protected enum BoundTo
056        {
057                CHARACTER,
058                ACCOUNT,
059                GROUP,
060                CLAN
061        }
062
063        protected BoundTo to = BoundTo.CHARACTER;
064        
065        protected enum BoundOn
066        {
067                PICKUP,
068                EQUIP
069        }
070
071        protected BoundOn on = BoundOn.PICKUP;
072        
073        protected String boundToName = "";
074        
075        protected String msgStr=defaultMessage();
076        
077        @Override
078        public void setMiscText(String text)
079        {
080                String boundTo=CMParms.getParmStr(text, "TO", "CHARACTER");
081                to = (BoundTo)CMath.s_valueOf(BoundTo.class,boundTo.toUpperCase().trim());
082                for(BoundTo b : BoundTo.values())
083                {
084                        if(b.name().startsWith(boundTo.toUpperCase().trim()))
085                                to=b;
086                }
087                if(to == null)
088                        to = BoundTo.CHARACTER;
089                
090                String boundOn=CMParms.getParmStr(text, "ON", "PICKUP");
091                on = (BoundOn)CMath.s_valueOf(BoundOn.class,boundOn.toUpperCase().trim());
092                for(BoundOn b : BoundOn.values())
093                {
094                        if(b.name().startsWith(boundTo.toUpperCase().trim()))
095                                on=b;
096                }
097                if(on == null)
098                        on = BoundOn.PICKUP;
099                
100                boundToName = CMStrings.deEscape(CMParms.getParmStr(text, "BOUND", ""));
101                msgStr=CMParms.getParmStr(text,"MESSAGE",defaultMessage());
102                super.setMiscText(text);
103        }
104        
105        protected String defaultMessage()
106        {
107                return "<O-NAME> flashes and flies out of <S-HIS-HER> hands!";
108        }
109
110        @Override
111        public long flags()
112        {
113                return Ability.FLAG_ZAPPER;
114        }
115
116        @Override
117        public int triggerMask()
118        {
119                return TriggeredAffect.TRIGGER_GET;
120        }
121
122        @Override
123        public String accountForYourself()
124        {
125                if(boundToName.length()>0)
126                        return "Bound to: "+boundToName;
127                return "Binds to "+this.to.toString().toLowerCase()+" on "+this.on.toString().toLowerCase();
128        }
129
130        @Override
131        public boolean okMessage(final Environmental myHost, final CMMsg msg)
132        {
133                if(affected==null)
134                        return true;
135
136                final MOB mob=msg.source();
137                if(mob.location()==null)
138                        return true;
139
140                if(msg.amITarget(affected))
141                {
142                        switch(msg.targetMinor())
143                        {
144                        case CMMsg.TYP_HOLD:
145                                break;
146                        case CMMsg.TYP_WEAR:
147                                break;
148                        case CMMsg.TYP_WIELD:
149                                break;
150                        case CMMsg.TYP_GET:
151                        case CMMsg.TYP_EAT:
152                        case CMMsg.TYP_DRINK:
153                                if(this.boundToName.length()>0)
154                                {
155                                        boolean zap = true;
156                                        switch(to)
157                                        {
158                                        case CHARACTER:
159                                                if(this.boundToName.equals(msg.source().Name()))
160                                                        zap=false;
161                                                break;
162                                        case GROUP:
163                                        {
164                                                final List<String> names=CMParms.parse(this.boundToName);
165                                                if(names.contains(msg.source().Name()))
166                                                        zap=false;
167                                                break;
168                                        }
169                                        case ACCOUNT:
170                                                if((msg.source().playerStats()!=null)
171                                                &&(msg.source().playerStats().getAccount()!=null)
172                                                &&(this.boundToName.equals(msg.source().playerStats().getAccount().getAccountName())))
173                                                        zap=false;
174                                                break;
175                                        case CLAN:
176                                        {
177                                                final Pair<Clan,Integer> pI=msg.source().getClanRole(this.boundToName);
178                                                if(pI!=null)
179                                                        zap=false;
180                                                break;
181                                        }
182                                        }
183                                        if(zap)
184                                        {
185                                                mob.location().show(mob,null,affected,CMMsg.MSG_OK_ACTION,msgStr);
186                                                return false;
187                                        }
188                                }
189                                break;
190                        default:
191                                break;
192                        }
193                }
194                else
195                if((this.boundToName.length()>0)
196                &&(msg.tool()==affected)
197                &&(msg.sourceMinor()==CMMsg.TYP_SELL))
198                {
199                        msg.source().tell(L("@x1 is bound to @x2, and can not be sold.",affected.name(),this.boundToName));
200                        return false;
201                }
202                return true;
203        }
204        
205        final String getBindyName(final MOB mob)
206        {
207                switch(to)
208                {
209                case CHARACTER:
210                        return mob.Name();
211                case GROUP:
212                {
213                        final Set<MOB> grp = mob.getGroupMembers(new HashSet<MOB>());
214                        final List<String> grpMemberNames = new ArrayList<String>();
215                        for(MOB M : grp)
216                                grpMemberNames.add(M.Name());
217                        return CMParms.combineQuoted(grpMemberNames, 0);
218                }
219                case ACCOUNT:
220                        if((mob.playerStats()!=null)
221                        &&(mob.playerStats().getAccount()!=null))
222                                return mob.playerStats().getAccount().getAccountName();
223                        break;
224                case CLAN:
225                {
226                        final Clan C=CMLib.clans().findRivalrousClan(mob);
227                        if(C!=null)
228                                return C.getName();
229                        break;
230                }
231                }
232                return "";
233        }
234        
235        @Override
236        public void executeMsg(final Environmental myHost, final CMMsg msg)
237        {
238                if(msg.amITarget(affected))
239                {
240                        switch(msg.targetMinor())
241                        {
242                        case CMMsg.TYP_HOLD:
243                        case CMMsg.TYP_WEAR:
244                        case CMMsg.TYP_WIELD:
245                        case CMMsg.TYP_EAT:
246                        case CMMsg.TYP_DRINK:
247                                if((this.boundToName.length()==0)
248                                &&(on == BoundOn.EQUIP))
249                                {
250                                        this.boundToName = getBindyName(msg.source());
251                                        if(this.boundToName.length()>0)
252                                                super.miscText = text()+" BOUND=\""+CMStrings.escape(this.boundToName)+"\"";
253                                }
254                                break;
255                        case CMMsg.TYP_GET:
256                                if((this.boundToName.length()==0)
257                                &&(on == BoundOn.PICKUP))
258                                {
259                                        this.boundToName = getBindyName(msg.source());
260                                        if(this.boundToName.length()>0)
261                                                super.miscText = text()+" BOUND=\""+CMStrings.escape(this.boundToName)+"\"";
262                                }
263                                break;
264                        default:
265                                break;
266                        }
267                }
268        }
269}
