036public class Fighter_CoupDeGrace extends FighterSkill
037{
038        @Override
039        public String ID()
040        {

042        }
043
044        private final static String localizedName = CMLib.lang().L("Coup de Grace");
045
046        @Override
047        public String name()
048        {
049                return localizedName;
050        }
051
052        private static final String[] triggerStrings =I(new String[] {"COUP","COUPDEGRACE"});
053        @Override
054        public int abstractQuality()
055        {
056                return Ability.QUALITY_MALICIOUS;
057        }
058
059        @Override
060        public String[] triggerStrings()
061        {
062                return triggerStrings;
063        }
064

066        protected int canAffectCode()
067        {

074                return Ability.CAN_MOBS;
075        }
076

078        public int maxRange()
079        {
080                return adjustedMaxInvokerRange(0);
081        }

092                return USAGE_MOVEMENT;

095        @Override
096        protected int overrideMana()
097        {
098                return 150;
099        }

101        @Override
102        public int castingQuality(MOB mob, Physical target)
103        {
104                if((mob!=null)&&(target!=null))
105                {
106                        if(!mob.isInCombat())
107                                return Ability.QUALITY_INDIFFERENT;
108                        if(mob.rangeToTarget()>0)
109                                return Ability.QUALITY_INDIFFERENT;
110                        final Item w=mob.fetchWieldedItem();
111                        Weapon ww=null;
112                        if((w==null)||(!(w instanceof Weapon)))
113                                return Ability.QUALITY_INDIFFERENT;
114                        ww=(Weapon)w;
115                        if((ww.weaponDamageType()!=Weapon.TYPE_SLASHING)
116                        &&(ww.weaponDamageType()!=Weapon.TYPE_PIERCING))
117                                 return Ability.QUALITY_INDIFFERENT;
118                        if(mob.curState().getMovement()<overrideMana())
119                                return Ability.QUALITY_INDIFFERENT;
120                        if(!CMLib.flags().isSleeping(mob.getVictim()))
121                                return Ability.QUALITY_INDIFFERENT;
122                }
123                return super.castingQuality(mob,target);
124        }
125
126        @Override
127        public boolean invoke(MOB mob, List<String> commands, Physical givenTarget, boolean auto, int asLevel)
128        {
129                if(!mob.isInCombat())
130                {
131                        mob.tell(L("You must be in combat to do this!"));
132                        return false;
133                }
134                if(mob.isInCombat()&&(mob.rangeToTarget()>0))
135                {
136                        mob.tell(L("You are too far away to try that!"));
137                        return false;
138                }
139                final Item w=mob.fetchWieldedItem();
140                Weapon ww=null;
141                if(!auto)
142                {
143                        if((w==null)||(!(w instanceof Weapon)))
144                        {
145                                mob.tell(L("You cannot coup de grace without a weapon!"));
146                                return false;
147                        }
148                        ww=(Weapon)w;
149                        if((ww.weaponDamageType()!=Weapon.TYPE_SLASHING)
150                        &&(ww.weaponDamageType()!=Weapon.TYPE_PIERCING))
151                        {
152                                mob.tell(L("You cannot coup de grace with a @x1!",ww.name()));
153                                return false;
154                        }
155                        if(mob.curState().getMovement()<150)
156                        {
157                                mob.tell(L("You don't have the energy to try it."));
158                                return false;
159                        }
160                        if(!CMLib.flags().isSleeping(mob.getVictim()))
161                        {
162                                if(CMLib.flags().isSitting(mob.getVictim()))
163                                        mob.tell(L("@x1 is on the ground, but not prone!",mob.getVictim().charStats().HeShe()));
164                                else
165                                        mob.tell(L("@x1 is not prone!",mob.getVictim().charStats().HeShe()));
166                                return false;
167                        }
168                }
169
170                final MOB target=mob.getVictim();
171                final int dmg=target.curState().getHitPoints();
172                if((!super.invoke(mob,commands,givenTarget,auto,asLevel))||(ww==null))
173                        return false;
174
175                int levelDiff=target.phyStats().level()-(mob.phyStats().level()+(2*getXLEVELLevel(mob)));
176                if(levelDiff>0)
177                        levelDiff=levelDiff*3;
178                else
179                        levelDiff=0;
180                final int chance=(-levelDiff)+(-(target.charStats().getStat(CharStats.STAT_CONSTITUTION)*2));
181                final boolean hit=(auto)||CMLib.combat().rollToHit(mob,target);
182                final boolean success=proficiencyCheck(mob,chance,auto)&&(hit);
183                if((success)&&((dmg<50)||(dmg<(target.maxState().getHitPoints()/4))))
184                {
185                        final CMMsg msg=CMClass.getMsg(mob,target,this,CMMsg.MSK_MALICIOUS_MOVE|CMMsg.TYP_JUSTICE|(auto?CMMsg.MASK_ALWAYS:0),null);
186                        if(mob.location().okMessage(mob,msg))
187                        {
188                                mob.location().send(mob,msg);
189                                target.curState().setHitPoints(1);
190                                CMLib.combat().postDamage(mob,target,ww,dmg,CMMsg.MSG_WEAPONATTACK,ww.weaponClassification(),auto?"":L("^F^<FIGHT^><S-NAME> rear(s) back and Coup-de-Grace(s) <T-NAME>!^</FIGHT^>^?@x1",CMLib.protocol().msp("decap.wav",30)));
191                                mob.location().recoverRoomStats();
192                        }
193                }
194                else
195                {
196                        final String str=auto?"":L("<S-NAME> attempt(s) a Coup-de-Grace and fail(s)!");
197                        final CMMsg msg=CMClass.getMsg(mob,target,null,CMMsg.MASK_MALICIOUS|CMMsg.MSG_OK_ACTION,str);
198                        if(mob.location().okMessage(mob,msg))
199                                mob.location().send(mob,msg);
200                }
201                return success;
202        }
