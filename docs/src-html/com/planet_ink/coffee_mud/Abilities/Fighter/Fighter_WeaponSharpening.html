103        protected final static int TICKS_TO_SHARPEN=8;
104        protected Item weapon=null;
105        protected int damageBonus = 1;
106
107        @Override
108        public void setMiscText(String newMiscText)
109        {
110                super.setMiscText(newMiscText);
111                if(newMiscText.length()>0)
112                        damageBonus=CMath.s_int(newMiscText);
113        }
114
115        @Override
116        public boolean tick(final Tickable ticking, final int tickID)
117        {
118                if(affected instanceof MOB)
119                {
120                        final MOB mob=(MOB)affected;
121                        if((weapon==null)||(weapon.owner()!=affected)||(weapon.amDestroyed())||(!CMLib.flags().isInTheGame(mob, true))||(mob.location()==null))
122                        {
123                                weapon=null;
124                                unInvoke();
125                                return false;
126                        }
127                        if((this.tickDown % 2)==0)
128                        {
129                                mob.location().show(mob, weapon, CMMsg.MSG_HANDS, L("<S-NAME> continue(s) sharpening <T-NAME> (@x1).",CMath.toPct(CMath.div((TICKS_TO_SHARPEN-tickDown+1),TICKS_TO_SHARPEN))));
130                        }
131                }
132                else
133                if(affected instanceof Item)
134                {
135                        final Item weapon=(Item)affected;
136                        if((weapon==null)
137                        ||(weapon.amDestroyed())
138                        ||(weapon.subjectToWearAndTear()&&(weapon.usesRemaining()<95)))
139                        {
140                                this.weapon=null;
141                                unInvoke();
142                                return false;
143                        }
144                }
145                return super.tick(ticking, tickID);
146        }
147
148        @Override
149        public void unInvoke()
150        {
151                if(canBeUninvoked())
152                {
153                        if((affected instanceof MOB)
154                        &&(weapon != null)
155                        &&(!weapon.amDestroyed()))
156                        {
157                                final MOB mob=(MOB)affected;
158                                beneficialAffect(mob,weapon,0,0);
159                                final Ability A=weapon.fetchEffect(ID());
160                                if(A!=null)
161                                {
162                                        A.setMiscText(text());
163                                        A.makeLongLasting();
164                                }
165                                weapon.recoverPhyStats();
166                                if(mob.location()!=null)
167                                        mob.location().recoverRoomStats();
168                                mob.tell(mob,weapon,null,L("You have finished sharpening <T-NAME>."));
169                                weapon = null;
170                        }
171                        else
172                        if((affected instanceof Item)
173                        &&(!((Item)affected).amDestroyed())
174                        &&(((Item)affected).owner() instanceof MOB))
175                        {
176                                final MOB M=(MOB)((Item)affected).owner();
177                                if((!M.amDead())&&(CMLib.flags().isInTheGame(M,true))&&(!((Item)affected).amWearingAt(Wearable.IN_INVENTORY)))
178                                        M.tell(M,affected,null,L("<T-NAME> no longer seem(s) quite as sharp."));
179                        }
180                }
181                super.unInvoke();
182        }
183
184        @Override
185        public void affectPhyStats(final Physical affected, final PhyStats stats)
186        {
187                if((affected instanceof Item)&&(damageBonus>0)&&(((Item)affected).owner() instanceof MOB))
188                {
189                        stats.setDamage(stats.damage()+damageBonus);
190                        stats.addAmbiance("^w*^?");
191                }
192        }

194        @Override
195        public boolean invoke(MOB mob, List<String> commands, Physical givenTarget, boolean auto, int asLevel)
196        {
197                if(mob.fetchEffect(ID())!=null)
198                {
199                        mob.tell(L("You are already sharpening something."));
200                        return false;
201                }
202                final Item weapon=super.getTarget(mob,null,givenTarget,null,commands,Wearable.FILTER_ANY);
203                if(weapon==null)
204                        return false;
205                if(!(weapon instanceof Weapon))
206                {
207                        mob.tell(mob,weapon,null,L("<T-NAME> is not a weapon."));
208                        return false;
209                }
210                if(weapon.fetchEffect(ID())!=null)
211                {
212                        mob.tell(mob,weapon,null,L("<T-NAME> is already sharp."));
213                        return false;
214                }
215                boolean isSharpenable;
216                switch(((Weapon)weapon).weaponClassification())
217                {
218                case Weapon.CLASS_AXE:
219                case Weapon.CLASS_DAGGER:
220                case Weapon.CLASS_EDGED:
221                case Weapon.CLASS_SWORD:
222                case Weapon.CLASS_POLEARM:
223                        isSharpenable=true;
224                        break;
225                case Weapon.CLASS_FLAILED:
226                case Weapon.CLASS_RANGED:
227                case Weapon.CLASS_THROWN:
228                        switch(((Weapon)weapon).weaponDamageType())
229                        {
230                        case Weapon.TYPE_PIERCING:
231                        case Weapon.TYPE_SLASHING:
232                                isSharpenable=true;
233                                break;
234                        default:
235                                isSharpenable=false;
236                                break;
237                        }
238                        break;
239                default:
240                        isSharpenable=false;
241                        break;
242                }
243                if(!isSharpenable)
244                {
245                        mob.tell(mob,weapon,null,L("<T-NAME> can not be sharpened with this skill."));
246                        return false;
247                }
248                if((weapon.subjectToWearAndTear())&&(weapon.usesRemaining()<95))
249                {
250                        mob.tell(mob,weapon,null,L("<T-NAME> needs repairing first."));
251                        return false;
252                }
253                if((!auto)&&(mob.isInCombat()))
254                {
255                        mob.tell(L("You are a bit too busy to do that right now."));
256                        return false;
257                }
258                final int bonus=(int)Math.round(CMath.mul(0.10+(0.10*getXLEVELLevel(mob)),weapon.phyStats().damage()));
259                if(bonus<1)
260                {
261                        mob.tell(mob,weapon,null,L("<T-NAME> is too weak of a weapon to provide any more benefit from sharpening."));
262                        return false;
263                }
264
265                if(!super.invoke(mob,commands,givenTarget,auto,asLevel))
266                        return false;
267
268                final boolean success=proficiencyCheck(mob,0,auto);
269                if(success)
270                {
271                        final String str=auto?L("<T-NAME> looks sharper!"):L("<S-NAME> start(s) sharpening <T-NAMESELF>.");
272                        final CMMsg msg=CMClass.getMsg(mob,weapon,this,CMMsg.MSG_NOISYMOVEMENT,str);
273                        if(mob.location().okMessage(mob,msg))
274                        {
275                                displayString="Sharpening "+weapon.name();
276                                mob.location().send(mob,msg);
277                                if(auto)
278                                {
279                                        beneficialAffect(mob,weapon,asLevel,0);
280                                        final Ability A=weapon.fetchEffect(ID());
281                                        if(A!=null)
282                                        {
283                                                A.setMiscText(""+bonus);
284                                                A.makeLongLasting();
285                                        }
286                                        weapon.recoverPhyStats();
287                                        mob.location().recoverRoomStats();
288                                }
289                                else
290                                {
291                                        beneficialAffect(mob,mob,asLevel,TICKS_TO_SHARPEN);
292                                        final Fighter_WeaponSharpening A=(Fighter_WeaponSharpening)mob.fetchEffect(ID());
293                                        if(A != null)
294                                        {
295                                                A.weapon=weapon;
296                                                A.setMiscText(""+bonus);
297                                        }
298                                }
299                        }
300                }
301                else
302                        return beneficialVisualFizzle(mob,weapon,L("<S-NAME> attempt(s) to tweak <T-NAME>, but just can't get it quite right."));
303                return success;
304        }
