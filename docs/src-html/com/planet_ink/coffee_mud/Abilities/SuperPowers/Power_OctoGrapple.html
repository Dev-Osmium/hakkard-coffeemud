036public class Power_OctoGrapple extends SuperPower
037{
038        @Override
039        public String ID()
040        {
041                return "Power_OctoGrapple";
042        }
043
044        private final static String localizedName = CMLib.lang().L("Octo-Grapple");
045
046        @Override
047        public String name()
048        {
049                return localizedName;
050        }
051
052        @Override
053        public String displayText()
054        {
055                if(affected==invoker)
056                        return "(Grappling)";
057                return "(Grappled)";
058        }
059
060        private static final String[] triggerStrings =I(new String[] {"GRAPPLE"});
061        @Override
062        public int abstractQuality()
063        {
064                return Ability.QUALITY_MALICIOUS;
065        }
066
067        @Override
068        public String[] triggerStrings()
069        {
070                return triggerStrings;
071        }
072
073        @Override
074        protected int canAffectCode()
075        {
076                return 0;
077        }
078
079        @Override
080        protected int canTargetCode()
081        {
082                return Ability.CAN_MOBS;
083        }
084
085        @Override
086        public int classificationCode()
087        {
088                return Ability.ACODE_SKILL;
089        }
090
091        @Override
092        public long flags()
093        {
094                return Ability.FLAG_BINDING;
095        }
096
097        @Override
098        public int usageType()
099        {
100                return USAGE_MOVEMENT;
101        }
102
103        @Override
104        public boolean okMessage(final Environmental myHost, final CMMsg msg)
105        {
106                if(!(affected instanceof MOB))
107                        return true;
108
109                final MOB mob=(MOB)affected;
110
111                // when this spell is on a MOBs Affected list,
112                // it should consistantly prevent the mob
113                // from trying to do ANYTHING except sleep
114                if((msg.amISource(mob))
115                &&(!msg.sourceMajor(CMMsg.MASK_ALWAYS))
116                &&(mob!=invoker))
117                {
118                        if((msg.sourceMajor(CMMsg.MASK_EYES))
119                        ||(msg.sourceMajor(CMMsg.MASK_HANDS))
120                        ||(msg.sourceMajor(CMMsg.MASK_MOUTH))
121                        ||(msg.sourceMajor(CMMsg.MASK_MOVE)))
122                        {
123                                if(msg.sourceMessage()!=null)
124                                {
125                                        if(mob.location().show(mob,null,CMMsg.MSG_OK_ACTION,L("<S-NAME> struggle(s) against the grappling arms.")))
126                                        {
127                                                if(CMLib.dice().rollPercentage()<mob.charStats().getStat(CharStats.STAT_STRENGTH))
128                                                {
129                                                        unInvoke();
130                                                        if((mob.fetchEffect(ID())==null)&&(invoker!=null)&&(invoker!=mob))
131                                                        {
132                                                                final Ability A=mob.fetchEffect(ID());
133                                                                if(A!=null)
134                                                                        A.unInvoke();
135                                                        }
136                                                }
137                                        }
138                                }
139                                return false;
140                        }
141                }
142                return super.okMessage(myHost,msg);
143        }
144
145        @Override
146        public void affectPhyStats(final Physical affected, final PhyStats affectableStats)
147        {
148                super.affectPhyStats(affected,affectableStats);
149                // when this spell is on a MOBs Affected list,
150                // it should consistantly put the mob into
151                // a sleeping state, so that nothing they do
152                // can get them out of it.
153                affectableStats.setDisposition(affectableStats.sensesMask()|PhyStats.IS_BOUND);
154        }
155
156        @Override
157        public void affectCharStats(final MOB affected, final CharStats affectableStats)
158        {
159                super.affectCharStats(affected,affectableStats);
160                if(affected==invoker)
161                        affectableStats.alterBodypart(Race.BODY_ARM,-2);
162        }
163
164        @Override
165        public void unInvoke()
166        {
167                // undo the affects of this spell
168                if(!(affected instanceof MOB))
169                        return;
170                final MOB mob=(MOB)affected;
171
172                super.unInvoke();
173
174                if(canBeUninvoked())
175                {
176                        if((!mob.amDead())&&(CMLib.flags().isInTheGame(mob,false)))
177                        {
178                                if(mob==invoker)
179                                {
180                                        if(mob.location()!=null)
181                                                mob.location().show(mob,null,CMMsg.MSG_OK_ACTION,L("<S-NAME> release(s) <S-HIS-HER> grapple."));
182                                        else
183                                                mob.tell(L("You release your grapple."));
184                                }
185                                else
186                                {
187                                        if(mob.location()!=null)
188                                                mob.location().show(mob,null,CMMsg.MSG_OK_ACTION,L("<S-NAME> <S-IS-ARE> released from the grapple"));
189                                        else
190                                                mob.tell(L("You are released from the grapple."));
191                                }
192                                CMLib.commands().postStand(mob,true, false);
193                        }
194                }
195        }
196
197        @Override
198        public boolean invoke(MOB mob, List<String> commands, Physical givenTarget, boolean auto, int asLevel)
199        {
200                final MOB target=this.getTarget(mob,commands,givenTarget);
201                if(target==null)
202                        return false;
203
204                if((!auto)&&(mob.baseWeight()<(target.baseWeight()-200)))
205                {
206                        mob.tell(L("@x1 is too big to grapple!",target.name(mob)));
207                        return false;
208                }
209
210                if(!super.invoke(mob,commands,givenTarget,auto,asLevel))
211                        return false;
212
213                int levelDiff=target.phyStats().level()-(mob.phyStats().level()+(2*getXLEVELLevel(mob)));
214                if(levelDiff>0)
215                        levelDiff=levelDiff*10;
216                else
217                        levelDiff=0;
218                // now see if it worked
219                final boolean hit=(auto)||CMLib.combat().rollToHit(mob,target);
220                boolean success=proficiencyCheck(mob,(-levelDiff)+(-(((target.charStats().getStat(CharStats.STAT_STRENGTH)-mob.charStats().getStat(CharStats.STAT_STRENGTH))*5))),auto)&&(hit);
221                success=success&&(target.charStats().getBodyPart(Race.BODY_ARM)>2);
222                if(success)
223                {
224                        invoker=mob;
225                        final CMMsg msg=CMClass.getMsg(mob,target,this,CMMsg.MSK_MALICIOUS_MOVE|CMMsg.TYP_JUSTICE|(auto?CMMsg.MASK_ALWAYS:0),auto?L("<T-NAME> get(s) grappled!"):L("^F^<FIGHT^><S-NAME> grab(s) <T-NAMESELF> with <S-HIS-HER> huge metallic arms!^</FIGHT^>^?"));
226                        CMLib.color().fixSourceFightColor(msg);
227                        if((mob.location().okMessage(mob,msg))&&(msg.value()<=0))
228                        {
229                                mob.location().send(mob,msg);
230                                success=maliciousAffect(mob,target,asLevel,10,-1)!=null;
231                                success=maliciousAffect(mob,mob,asLevel,10,-1)!=null;
232                        }
233                }
234                else
235                        return maliciousFizzle(mob,target,L("<S-NAME> attempt(s) to grab <T-NAMESELF>, but fail(s)."));
236
237                // return whether it worked
238                return success;
239        }
