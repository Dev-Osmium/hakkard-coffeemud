125        @Override
126        public boolean okMessage(final Environmental myHost, final CMMsg msg)
127        {
128                if((affected!=null)
129                &&(affected instanceof MOB)
130                &&(msg.amISource((MOB)affected))
131                &&((msg.sourceMinor()==CMMsg.TYP_DEATH)
132                        ||(msg.sourceMinor()==CMMsg.TYP_QUIT)
133                        ||(msg.sourceMinor()==CMMsg.TYP_RECALL)
134                        ||(msg.sourceMinor()==CMMsg.TYP_LEAVE)
135                        ||(destR==null)
136                        ||(msg.source().location()!=boxR)))
137                {
138                        destR=null;
139                        unInvoke();
140                }
141                return super.okMessage(myHost,msg);
142        }
143
144        @Override
145        public void unInvoke()
146        {
147                final Physical affected=this.affected;
148                final Room R=this.destR;
149                final Room boxR=this.boxR;
150                super.unInvoke();
151                if((affected instanceof MOB)&&(R!=null))
152                {
153                        final MOB mob=(MOB)affected;
154                        if(mob.location()==boxR)
155                        {
156                                mob.tell(L("You feel yourself being unloaded from the ship."));
157                                final CMMsg leaveMsg=CMClass.getMsg(mob,R,this,CMMsg.MSG_LEAVE|CMMsg.MASK_ALWAYS,L("<S-NAME> slip(s) out of a cargo box just unloaded from one of the ships."));
158                                if(R.okMessage(mob,leaveMsg))
159                                        R.send(mob,leaveMsg);
160                                if(!R.isInhabitant(mob))
161                                        R.bringMobHere(mob, false);
162                                CMLib.commands().postStand(mob, true, false);
163                        }
164                }
165                if(boxR!=null)
166                {
167                        if(R==null)
168                                CMLib.map().emptyRoom(boxR, CMLib.map().getStartRoom(affected), true);
169                        else
170                                CMLib.map().emptyRoom(boxR, R, true);
171                        boxR.destroy();
172                }
173                this.destR=null;
174                this.boxR=null;
175        }

177        @Override
178        public boolean tick(final Tickable ticking, final int tickID)
179        {
180                if(!super.tick(ticking, tickID))
181                        return false;
182                
183                if((affected instanceof MOB)
184                &&(((MOB)affected).location()==boxR)
185                &&(destR!=null))
186                {
187                        final MOB mob=(MOB)affected;
188                        if(++tickUp==2)
189                        {
190                                mob.tell(L("^xYou feel yourself being picked up and loaded onto a ship.^.^?"));
191                        }
192                        else
193                        if(tickUp>2)
194                        {
195                                switch(CMLib.dice().roll(1,50,0))
196                                {
197                                case 1:
198                                        mob.tell(L("You feel the ship rocking gently in the waves."));
199                                        break;
200                                case 2:
201                                        mob.tell(L("You hear rats crawling around on your crate."));
202                                        break;
203                                case 3:
204                                        mob.tell(L("You hear the sounds of some crewmen talking outside."));
205                                        break;
206                                case 4:
207                                case 5:
208                                        mob.tell(L("You are extremely bored."));
209                                        break;
210                                case 6:
211                                case 7:
212                                        mob.tell(L("It is dark and cramped in here."));
213                                        break;
214                                }
215                        }
216                }
217                else
218                {
219                        destR=null;
220                        unInvoke();
221                }
222                return true;
223        }

225        @Override
226        public boolean invoke(MOB mob, List<String> commands, Physical givenTarget, boolean auto, int asLevel)
227        {
228                final Room R=mob.location();
229                if(R==null)
230                        return false;
231
232                final RFilter SHORE_ONLY=TrackingFlag.SHOREONLY.myFilter;
233                if(SHORE_ONLY.isFilteredOut(R, R, null, 0))
234                {
235                        mob.tell(L("You must be on a shore to stow-away."));
236                        return false;
237                }
238                
239                final Ability seaChartA=mob.fetchAbility("Skill_SeaCharting");
240                Room destR=null;
241                if(commands.size()>0)
242                {
243                        if(seaChartA!=null)
244                        {
245                                List<String> chartRooms=CMParms.parseAny(seaChartA.text(),';',true);
246                                String name=CMParms.combine(commands,0);
247                                if(CMath.isInteger(name))
248                                {
249                                        int x=CMath.s_int(name);
250                                        if((x<1)||(x>chartRooms.size()))
251                                        {
252                                                mob.tell(L("@x1 is not a valid number.  Check your sea charts!"));
253                                                return false;
254                                        }
255                                        destR=CMLib.map().getRoom(chartRooms.get(x-1));
256                                }
257                                else
258                                {
259                                        for(String roomID : chartRooms)
260                                        {
261                                                final Room R2=CMLib.map().getRoom(roomID);
262                                                if((R2!=null)&&(CMLib.english().containsString(R2.displayText(mob), name)))
263                                                {
264                                                        destR=R2;
265                                                        break;
266                                                }
267                                        }
268                                        if(destR==null)
269                                        {
270                                                for(String roomID : chartRooms)
271                                                {
272                                                        final Room R2=CMLib.map().getRoom(roomID);
273                                                        if((R2!=null)&&(CMLib.english().containsString(R2.description(mob), name)))
274                                                        {
275                                                                destR=R2;
276                                                                break;
277                                                        }
278                                                }
279                                        }
280                                        if(destR==null)
281                                        {
282                                                mob.tell(L("You have not charted a room called '@x1'.",name));
283                                                return false;
284                                        }
285                                }
286                        }
287                        else
288                        {
289                                mob.tell(L("You cannot specify a destination unless you have Sea Charting."));
290                                return false;
291                        }
292                }
293                
294                TrackingFlags flags=CMLib.tracking().newFlags().plus(TrackingFlag.WATERSURFACEORSHOREONLY);
295                int radius=50 + (10*(super.getXLEVELLevel(mob)+super.getXMAXRANGELevel(mob)));
296                List<Room> rooms=CMLib.tracking().getRadiantRooms(R, flags, radius);
297                boolean success=proficiencyCheck(mob,0,auto);
298                if((destR==null)||(!rooms.contains(destR))||(!success))
299                {
300                        for(final Iterator<Room> i=rooms.iterator();i.hasNext();)
301                        {
302                                final Room R2=i.next();
303                                if((R2 == R)
304                                ||(CMLib.map().getRoomDir(R, R2)>=0)
305                                ||(SHORE_ONLY.isFilteredOut(R2, R2, null, 0)))
306                                        i.remove();
307                        }
308                        if(rooms.size()==0)
309                        {
310                                mob.tell(L("There isn't enough shipping traffic here."));
311                                return false;
312                        }
313                        destR=rooms.get(CMLib.dice().roll(1, rooms.size(), -1));
314                }
315                
316                if(!super.invoke(mob, commands, givenTarget, auto, asLevel))
317                        return false;
318                
319                final CMMsg leaveMsg=CMClass.getMsg(mob,R,this,CMMsg.MSG_LEAVE|(auto?CMMsg.MASK_ALWAYS:0),auto?L("<S-NAME> stow(s) away!"):L("<S-NAME> slip(s) into a cargo box!"));
320                if(R.okMessage(mob,leaveMsg))
321                {
322                        R.send(mob,leaveMsg);
323                        Room boxR=CMClass.getLocale("WoodRoom");
324                        boxR.setDisplayText(L("You are squeezed into a dark cramped shipping box."));
325                        boxR.addNonUninvokableEffect(CMClass.getAbility("Prop_Crawlspace"));
326                        boxR.addNonUninvokableEffect(CMClass.getAbility("Prop_RoomDark"));
327                        final Ability consA=CMClass.getAbility("Prop_ReqCapacity");
328                        consA.setMiscText("people=1");
329                        boxR.addNonUninvokableEffect(consA);
330                        boxR.bringMobHere(mob, false);
331                        CMLib.commands().forceStandardCommand(mob, "Sit",new XVector<String>(""));
332                        Skill_Stowaway stow=(Skill_Stowaway)super.beneficialAffect(mob, mob, asLevel, (int)((4 * 60 * 1000) / CMProps.getTickMillis())  / (1+super.getXTIMELevel(mob)));
333                        stow.destR=destR;
334                        stow.boxR=boxR;
335                        stow.tickUp=0;
336                }
337                return true;
338        }
