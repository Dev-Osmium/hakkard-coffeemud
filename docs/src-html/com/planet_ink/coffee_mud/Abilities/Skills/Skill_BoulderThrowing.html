094        protected volatile int  tickDownThisShipCombatRound     = CombatLibrary.TICKS_PER_SHIP_COMBAT;
095
096        @Override
097        public boolean tick(final Tickable ticking, final int tickID)
098        {
099                if(tickID==Tickable.TICKID_MOB)
100                {
101                        if(tickDownThisShipCombatRound>0)
102                                tickDownThisShipCombatRound--;
103                        if(affected instanceof MOB)
104                        {
105                                final MOB mob=(MOB)affected;
106                                final Room R=(mob!=null)?mob.location():null;
107                                final Area A=(R != null) ? R.getArea() : null;
108                                final Item shipItem=(A instanceof BoardableShip) ? ((BoardableShip)A).getShipItem() : null;
109                                if((mob!=null)
110                                &&(mob.isMonster())
111                                &&(shipItem instanceof SailingShip)
112                                &&(!mob.isInCombat()))
113                                {
114                                        final Item ammoI=mob.fetchHeldItem();
115                                        if(ammoI instanceof Ammunition)
116                                        {
117                                                if(tickDownThisShipCombatRound==0)
118                                                {
119                                                        if(((SailingShip)shipItem).isInCombat())
120                                                                CMLib.commands().forceStandardCommand(mob, "Throw", new XVector<String>("THROW",ammoI.Name()));
121                                                }
122                                        }
123                                        else
124                                        {
125                                                if(ammoI!=null)
126                                                        CMLib.commands().postRemove(mob, ammoI, false);
127                                                Item holdI=null;
128                                                for(final Enumeration<Item> i=mob.items();i.hasMoreElements();)
129                                                {
130                                                        final Item I=i.nextElement();
131                                                        if((I instanceof Ammunition)
132                                                        &&(I.container()==null)
133                                                        &&(I.amWearingAt(Item.IN_INVENTORY)))
134                                                        {
135                                                                holdI=I;
136                                                                break;
137                                                        }
138                                                }
139                                                if(holdI!=null)
140                                                        CMLib.commands().forceStandardCommand(mob, "Hold", new XVector<String>("HOLD",holdI.Name()));
141                                                else
142                                                if(R!=null)
143                                                {
144                                                        for(final Enumeration<Item> i=R.items();i.hasMoreElements();)
145                                                        {
146                                                                final Item I=i.nextElement();
147                                                                if((I instanceof Ammunition)
148                                                                &&(I.container()==null)
149                                                                &&(CMLib.flags().isGettable(I)))
150                                                                {
151                                                                        holdI=I;
152                                                                        break;
153                                                                }
154                                                        }
155                                                        if(holdI!=null)
156                                                                CMLib.commands().postGet(mob, null, holdI, false);
157                                                }
158                                        }
159                                }
160                        }
161                }
162                return super.tick(ticking,tickID);
163        }

165        private AmmunitionWeapon boulderThrower=null;
166        
167        @Override
168        public boolean okMessage(final Environmental myHost, final CMMsg msg)
169        {
170                if(!(affected instanceof MOB))
171                        return true;
172
173                final MOB mob=(MOB)affected;
174                if((msg.source()==mob)
175                &&(msg.sourceMinor()==CMMsg.TYP_COMMANDFAIL)
176                &&(msg.targetMessage()!=null))
177                {
178                        final List<String> cmd =CMParms.parse(msg.targetMessage());
179                        final Room R=mob.location();
180                        if(cmd.get(0).toUpperCase().startsWith("TH") && (R!=null))
181                        {
182                                final CMObject O = CMLib.english().findCommand(mob, cmd);
183                                if((O instanceof Command) && (((Command)O).ID().equals("Throw")))
184                                {
185                                        final Area A=R.getArea();
186                                        final Item shipItem=(A instanceof BoardableShip) ? ((BoardableShip)A).getShipItem() : null;
187                                        final Room shipR=(shipItem != null)?CMLib.map().roomLocation(shipItem):null;
188                                        
189                                        final String str;
190                                        if(cmd.size()>2)
191                                        {
192                                                str=cmd.get(cmd.size()-1);
193                                                cmd.remove(str);
194                                        }
195                                        else
196                                        if((shipItem instanceof SailingShip)
197                                        &&(shipR!=null)
198                                        &&(((SailingShip)shipItem).getCombatant()!=null))
199                                                str=shipR.getContextName(((SailingShip)shipItem).getCombatant());
200                                        else
201                                                return true;
202                                        
203                                        final String what=CMParms.combine(cmd,1);
204                                        Item item=mob.fetchItem(null,Wearable.FILTER_WORNONLY,what);
205                                        if(item==null)
206                                                item=mob.findItem(null,what);
207                                        if((item==null)||(!CMLib.flags().canBeSeenBy(item,mob)))
208                                                return true;
209                                        if((!item.amWearingAt(Wearable.WORN_HELD))&&(!item.amWearingAt(Wearable.WORN_WIELD)))
210                                                return true;
211                                        final Item target;
212                                        if((shipItem != null)&&(shipR!=null))
213                                                target=shipR.findItem(null,str);
214                                        else
215                                                target=R.findItem(null,str);
216                                        if((target instanceof BoardableShip)
217                                        &&(CMLib.flags().canBeSeenBy(target, mob)))
218                                        {
219                                                if(tickDownThisShipCombatRound>0)
220                                                        msg.setSourceMessage(L("You've already thrown at a ship this round."));
221                                                else
222                                                if(mob.charStats().getStat(CharStats.STAT_STRENGTH)<10)
223                                                        msg.setSourceMessage(L("You aren't strong enough."));
224                                                else
225                                                if((!(item instanceof Ammunition))||(((Ammunition)item).ammunitionRemaining()<1))
226                                                        msg.setSourceMessage(L("You can't throw that at another ship."));
227                                                else
228                                                if((shipItem instanceof SailingShip)
229                                                &&((R.domainType()&Room.INDOORS)==0))
230                                                {
231                                                        final SailingShip sailShip=(SailingShip)shipItem;
232                                                        if(sailShip.isInCombat() && (sailShip.getCombatant()==target))
233                                                        {
234                                                                if(this.boulderThrower==null)
235                                                                {
236                                                                        boulderThrower=(AmmunitionWeapon)CMClass.getWeapon("GenSiegeWeapon");
237                                                                        boulderThrower.setAmmoCapacity(1);
238                                                                        boulderThrower.setWeaponClassification(Weapon.CLASS_THROWN);
239                                                                        boulderThrower.setWeaponDamageType(Weapon.TYPE_BASHING);
240                                                                }
241                                                                final AmmunitionWeapon weapon=boulderThrower;
242                                                                final Ammunition ammo=(Ammunition)item;
243                                                                boulderThrower.setOwner(null);
244                                                                weapon.setAmmunitionType(ammo.ammunitionType());
245                                                                weapon.setName(ammo.Name());
246                                                                weapon.setRanges(0, 1+(mob.charStats().getStat(CharStats.STAT_STRENGTH)-10)/2);
247                                                                final int distance=sailShip.rangeToTarget();
248                                                                boolean wasHit=true;
249                                                                if((weapon.maxRange() < distance)||(weapon.minRange() > distance))
250                                                                {
251                                                                        msg.setSourceMessage(L("Your target is presently at distance of @x1, but your range is @x2 to @x3.",
252                                                                                                                ""+distance,""+weapon.minRange(),""+weapon.maxRange()));
253                                                                }
254                                                                else
255                                                                {
256                                                                        if(ammo.ammunitionType().equals("catapult-boulders"))
257                                                                                weapon.basePhyStats().setDamage(10 + super.getXLEVELLevel(mob));
258                                                                        else
259                                                                        if(ammo.ammunitionType().equals("ballista-bolts"))
260                                                                                weapon.basePhyStats().setDamage(5 + super.getXLEVELLevel(mob));
261                                                                        else
262                                                                        if(ammo.ammunitionType().equals("trebuchet-shots"))
263                                                                                weapon.basePhyStats().setDamage(15 + super.getXLEVELLevel(mob));
264                                                                        else
265                                                                                weapon.basePhyStats().setDamage(1 + (super.getXLEVELLevel(mob)/2));
266                                                                        weapon.phyStats().setDamage(weapon.basePhyStats().damage());
267                                                                        weapon.setAmmoRemaining(1);
268                                                                        weapon.setAmmoRemaining(ammo.ammunitionRemaining()-1);
269                                                                        final LinkedList<Ability> removeThese=new LinkedList<Ability>();
270                                                                        for(final Enumeration<Ability> a=weapon.effects();a.hasMoreElements();)
271                                                                        {
272                                                                                final Ability A1=a.nextElement();
273                                                                                if((A1!=null)&&(!A1.isSavable())&&(A1.invoker()==null))
274                                                                                        removeThese.add(A1);
275                                                                        }
276                                                                        for(final Ability A1 : removeThese)
277                                                                                weapon.delEffect(A1);
278                                                                        for(final Enumeration<Ability> a=ammo.effects();a.hasMoreElements();)
279                                                                        {
280                                                                                Ability A1=a.nextElement();
281                                                                                if((A!=null)&&(A.isSavable())&&(weapon.fetchEffect(A1.ID())==null))
282                                                                                {
283                                                                                        A1=(Ability)A1.copyOf();
284                                                                                        A1.setInvoker(null);
285                                                                                        A1.setSavable(false);
286                                                                                        weapon.addEffect(A1);
287                                                                                }
288                                                                        }
289                                                                        tickDownThisShipCombatRound=CombatLibrary.TICKS_PER_SHIP_COMBAT;
290                                                                        for(int i=0;i<distance;i++)
291                                                                                wasHit = wasHit && super.proficiencyCheck(mob, -50+mob.charStats().getStat(CharStats.STAT_DEXTERITY)+(super.getXLEVELLevel(mob)*2), false);
292                                                                        CMLib.combat().postShipAttack(mob, shipItem, target, weapon, wasHit);
293                                                                        if(ammo.ammunitionRemaining()<=0)
294                                                                                ammo.destroy();
295                                                                }
296                                                        }
297                                                        else
298                                                                msg.setSourceMessage(L("Your ship must first be in combat with @x1.",target.name(mob)));
299                                                }
300                                                else
301                                                        msg.setSourceMessage(L("You must be on the deck of a ship to throw at another ship."));
302                                        }
303                                }
304                        }
305                }
306                return true;
307        }

313                &&(msg.tool()==this.boulderThrower))
314                        this.boulderThrower.executeMsg(myHost, msg);
315                super.executeMsg(myHost, msg);
