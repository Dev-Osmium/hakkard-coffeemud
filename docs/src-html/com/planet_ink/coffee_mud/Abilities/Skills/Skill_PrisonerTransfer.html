105        @Override
106        public void executeMsg(final Environmental myHost, final CMMsg msg)
107        {
108                super.executeMsg(myHost,msg);
109                if(enRoute.get()!=null)
110                {
111                        if(affected instanceof MOB)
112                        {
113                                final MOB prisonerM=enRoute.get();
114                                final MOB copM=(MOB)affected;
115                                if((msg.sourceMinor()==CMMsg.TYP_ENTER)
116                                &&((msg.source()==copM)||(msg.source()==enRoute.get()))
117                                &&(invoker()!=null)
118                                &&(invoker().location()!=null)
119                                &&(copM!=null)
120                                &&(prisonerM!=null))
121                                {
122                                        if(!CMLib.flags().isInTheGame(invoker(), true))
123                                        {
124                                                enRoute= new WeakReference<MOB>(null);
125                                                unInvoke();
126                                                return;
127                                        }
128                                        if((msg.target()!=invoker().location()))
129                                        {
130                                                return;
131                                        }
132                                        final Ability me=this;
133                                        msg.addTrailerRunnable(new Runnable()
134                                        {
135                                                @Override
136                                                public void run()
137                                                {
138                                                        synchronized(me)
139                                                        {
140                                                                if(copM.location()!=invoker().location())
141                                                                        return;
142                                                                if(copM.numFollowers()==0)
143                                                                {
144                                                                        enRoute= new WeakReference<MOB>(null);
145                                                                        unInvoke();
146                                                                        return;
147                                                                }
148                                                                if(prisonerM.location()!=invoker().location())
149                                                                        return;
150                                                                final Room R=copM.location();
151                                                                final LegalBehavior B=CMLib.law().getLegalBehavior(R);
152                                                                final Area legalA=CMLib.law().getLegalObject(R);
153                                                                if((B==null)||(legalA==null))
154                                                                {
155                                                                        enRoute= new WeakReference<MOB>(null);
156                                                                        unInvoke();
157                                                                        return;
158                                                                }
159                                                                CMLib.commands().postSay(copM, invoker(), L("OK, @x1's is all yours.",prisonerM.charStats().heshe()));
160                                                                final List<LegalWarrant> warrants=B.getWarrantsOf(legalA, prisonerM);
161                                                                for(final LegalWarrant W : warrants)
162                                                                {
163                                                                        if((W!=null)
164                                                                        &&(W.arrestingOfficer()==copM))
165                                                                        {
166                                                                                W.setState(Law.STATE_SEEKING);
167                                                                                W.setArrestingOfficer(legalA, invoker());
168                                                                        }
169                                                                }
170                                                                final CMMsg msg1=CMClass.getMsg(prisonerM,copM,null,CMMsg.MSG_NOFOLLOW,L("<S-NAME> stop(s) following <T-NAMESELF>."));
171                                                                R.send(prisonerM,msg1);
172                                                                if(prisonerM.amFollowing()!=null)
173                                                                        prisonerM.setFollowing(null);
174                                                                final CMMsg msg2=CMClass.getMsg(prisonerM,invoker(),null,CMMsg.MSG_FOLLOW,L("<S-NAME> start(s) following <T-NAMESELF>."));
175                                                                R.send(prisonerM,msg2);
176                                                                if(prisonerM.amFollowing()!=invoker())
177                                                                        prisonerM.setFollowing(invoker());
178                                                                for(final Enumeration<Ability> e=prisonerM.effects();e.hasMoreElements();)
179                                                                {
180                                                                        final Ability eA=e.nextElement();
181                                                                        if((eA!=null)
182                                                                        &&(eA.ID().equals("Skill_HandCuff")))
183                                                                                eA.setInvoker(invoker());
184                                                                }
185                                                                CMLib.tracking().wanderAway(copM, false, true);
186                                                                unInvoke();
187                                                        }
188                                                }
189                                        });
190                                }
191                        }
192                }
193                else
194                if((msg.sourceMinor()==CMMsg.TYP_LEGALSTATE)
195                &&(msg.tool()==affected)
196                &&(msg.value()==Law.STATE_MOVING) // moving to judge
197                &&(invoker()!=null)
198                &&(msg.othersMessage()!=null)
199                &&(affected instanceof MOB)
200                &&(((MOB)affected).mayIFight(msg.source())) // mobs, or pvp players, or otherwise at war
201                &&(msg.source().location()==((MOB)affected).location())
202                &&(invoker().location()!=null)
203                &&(CMLib.flags().isInTheGame(invoker(), true)))
204                {
205                        final MOB meM=(MOB)affected;
206                        final Room R=msg.source().location();
207                        final LegalBehavior B=CMLib.law().getLegalBehavior(R);
208                        final Area legalA=CMLib.law().getLegalObject(R);
209                        if((B!=null)
210                        &&(legalA!=null)
211                        &&(legalA.inMyMetroArea(invoker().location().getArea()))
212                        &&(B.isAnyOfficer(legalA, meM)))
213                        {
214                                final Room transR=invoker().location();
215                                if((transR != null)
216                                &&(!transR.amDestroyed())
217                                &&(transR.isInhabitant(invoker())))
218                                {
219                                        final List<LegalWarrant> warrants=B.getWarrantsOf(legalA, msg.source());
220                                        for(final LegalWarrant W : warrants)
221                                        {
222                                                if((W!=null)
223                                                &&(W.arrestingOfficer()==meM))
224                                                {
225                                                        W.setJail(transR);
226                                                        enRoute = new WeakReference<MOB>(W.criminal());
227                                                }
228                                        }
229                                }
230                        }
231                }
232        }

252        @Override
253        public boolean invoke(final MOB mob, final List<String> commands, final Physical givenTarget, final boolean auto, final int asLevel)
254        {
255                final Room R=mob.location();
256                if(R==null)
257                        return false;
258
259                final MOB target=this.getTarget(mob,commands,givenTarget, false, true);
260                if(target==null)
261                        return false;
262
263                if(mob.isInCombat())
264                {
265                        mob.tell(L("Not while you are fighting!"));
266                        return false;
267                }
268                else
269                if(!CMLib.flags().isAliveAwakeMobileUnbound(target, false)
270                ||(!CMLib.flags().canBeHeardSpeakingBy(mob, target)))
271                {
272                        mob.tell(L("@x1 doesn't seem to be paying any attention to you.",target.name(mob)));
273                        return false;
274                }
275
276                final LegalBehavior B=CMLib.law().getLegalBehavior(R);
277                final Area legalA=CMLib.law().getLegalObject(R);
278                if((B==null)
279                ||(legalA==null)
280                ||(!B.isAnyOfficer(legalA, target)))
281                {
282                        mob.tell(L("@x1 is not an officer here.",target.name(mob)));
283                        return false;
284                }
285
286                if(!B.isElligibleOfficer(legalA, target))
287                {
288                        mob.tell(L("@x1 is too busy to talk to you right now.",target.name(mob)));
289                        return false;
290                }
291
292                final String currency=CMLib.beanCounter().getCurrency(target);
293                double amt=CMLib.beanCounter().getTotalAbsoluteNativeValue(target) / 10.0;
294                if(amt < target.phyStats().level()/2)
295                        amt=target.phyStats().level()/2;
296                final String amtStr=CMLib.beanCounter().nameCurrencyLong(target, amt);
297                if(CMLib.beanCounter().getTotalAbsoluteValue(mob, currency) < amt)
298                {
299                        CMLib.commands().postSay(mob, L("That kind of paperwork is tricky, I couldn't do it for less than @x1.",amtStr));
300                        return false;
301                }
302
303                /*
304                final Ability collectA=mob.fetchAbility("Skill_CollectBounty");
305                if(collectA == null)
306                {
307                        mob.tell(L("You need to know how to collect bounties to do that."));
308                        return false;
309                }
310                */
311
312                if(!super.invoke(mob,commands,givenTarget,auto,asLevel))
313                        return false;
314
315                final boolean success=proficiencyCheck(mob,0,auto);
316                if(success)
317                {
318                        if(R.show(mob, target, this, CMMsg.MSG_SPEAK, auto ? "" : L("<S-NAME> make(s) an arrangement with <T-NAME>.")))
319                        {
320                                final List<Coins> V=CMLib.beanCounter().makeAllCurrency(currency,amt);
321                                boolean allOK=true;
322                                for(int i=0;i<V.size();i++)
323                                {
324                                        final Coins C=V.get(i);
325                                        final CMMsg newMsg=CMClass.getMsg(mob,target,C,CMMsg.MSG_GIVE,L("<S-NAME> give(s) @x1 to <T-NAMESELF>.",C.Name()));
326                                        if(R.okMessage(mob,newMsg))
327                                                R.send(mob,newMsg);
328                                        else
329                                                allOK=false;
330                                }
331                                if(allOK)
332                                {
333                                        final Ability oldA=target.fetchEffect(ID());
334                                        if(oldA!=null)
335                                        {
336                                                oldA.unInvoke();
337                                                target.delEffect(oldA);
338                                        }
339                                        final int duration = (int)(CMProps.getTicksPerMudHour() * CMLib.time().globalClock().getHoursInDay() * (1+super.getXLEVELLevel(mob)));
340                                        final Ability A=beneficialAffect(mob, target, asLevel, duration);
341                                        if(A!=null)
342                                        {
343                                                CMLib.beanCounter().subtractMoney(mob, amt);
344                                                ((Skill_PrisonerTransfer)A).enRoute= new WeakReference<MOB>(null);
345                                                //CMLib.map().addGlobalHandler(A, CMMsg.TYP_LEGALSTATE);
346                                        }
347                                }
348                        }
349                }
350                else
351                        return beneficialWordsFizzle(mob,target,L("<S-NAME> attempt(s) to purchase a prisoner trasnfer from <T-NAME>, but fail(s)."));
352
353                // return whether it worked
354                return success;
355        }
