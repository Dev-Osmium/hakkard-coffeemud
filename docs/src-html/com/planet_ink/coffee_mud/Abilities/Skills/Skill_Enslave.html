112        public MOB getMaster()
113        {
114                if(masterMOB==null)
115                {
116                        masterMOB=CMLib.players().getLoadPlayer(masterName);
117                        if(masterMOB!=null)
118                        {
119                                oldLeige=masterMOB.getLiegeID();
120                                oldClans=new Vector<Pair<Clan,Integer>>();
121                                for(final Pair<Clan,Integer> p : masterMOB.clans())
122                                        oldClans.add(p);
123                        }
124                }
125                return masterMOB;
126        }
127
128        public void unMaster(MOB mob)
129        {
130                if((masterMOB!=null) && (mob!=null))
131                {
132                        mob.setLiegeID(oldLeige);
133                        mob.setClan("", Integer.MIN_VALUE);
134                        for(final Pair<Clan,Integer> p : oldClans)
135                                mob.setClan(p.first.clanID(),p.second.intValue());
136                }
137        }

139        @Override
140        public void executeMsg(final Environmental myHost, final CMMsg msg)
141        {
142                // undo the affects of this spell
143                if(!(affected instanceof MOB))
144                        return;
145                final MOB mob=(MOB)affected;
146                if((msg.amITarget(mob))
147                &&(msg.tool() instanceof Social)
148                &&(msg.tool().Name().equals("WHIP <T-NAME>")
149                        ||msg.tool().Name().equals("BEAT <T-NAME>")))
150                        speedDown=SPEEDMAX;
151                else
152                if(msg.amITarget(mob)
153                &&(msg.targetMinor()==CMMsg.TYP_DAMAGE)
154                &&((msg.value())>0))
155                {
156                        masterAnger+=10;
157                        CMLib.combat().postPanic(mob,msg);
158                }
159                else
160                if((msg.sourceMinor()==CMMsg.TYP_SPEAK)
161                &&(msg.targetMinor()!=CMMsg.TYP_ORDER)
162                &&(msg.sourceMessage()!=null)
163                &&(msg.sourceMessage().length()>0))
164                {
165                        if(STEPS!=null)
166                        {
167                                if((msg.target()==null)||(msg.target() instanceof MOB))
168                                {
169                                        final String response=CMStrings.getSayFromMessage(msg.sourceMessage());
170                                        if(response!=null)
171                                        {
172                                                if((msg.target()==mob)
173                                                &&(msg.source().Name().equals(mob.getLiegeID())))
174                                                {
175                                                        final Vector<String> V=CMParms.parse(response.toUpperCase());
176                                                        if(V.contains("STOP")||V.contains("CANCEL"))
177                                                        {
178                                                                CMLib.commands().postSay(mob,msg.source(),L("Yes master."),false,false);
179                                                                STEPS=null;
180                                                                return;
181                                                        }
182                                                }
183                                                STEPS.sayResponse(msg.source(),(MOB)msg.target(),response);
184                                        }
185                                }
186                        }
187                        else
188                        if((msg.amITarget(mob))
189                        &&(mob.getLiegeID().length()>0))
190                        {
191                                if((msg.tool()==null)
192                                ||((msg.tool() instanceof Ability)
193                                        &&((((Ability)msg.tool()).classificationCode()&Ability.ALL_ACODES)==Ability.ACODE_LANGUAGE)
194                                        &&(mob.fetchAbility(msg.tool().ID())!=null)
195                                        &&(msg.sourceMinor()!=CMMsg.TYP_TEACH)))
196                                {
197                                        if(!msg.source().Name().equals(mob.getLiegeID()))
198                                        {
199                                                final String response=CMStrings.getSayFromMessage(msg.sourceMessage());
200                                                if(response!=null)
201                                                {
202                                                        if((response.toUpperCase().startsWith("I COMMAND YOU TO "))
203                                                        ||(response.toUpperCase().startsWith("I ORDER YOU TO ")))
204                                                                CMLib.commands().postSay(mob,msg.source(),L("I don't take orders from you. "),false,false);
205                                                }
206                                        }
207                                        else
208                                        {
209                                                String response=CMStrings.getSayFromMessage(msg.sourceMessage());
210                                                if(response!=null)
211                                                {
212                                                        if(response.toUpperCase().startsWith("I COMMAND YOU TO "))
213                                                                response=response.substring(("I COMMAND YOU TO ").length());
214                                                        else
215                                                        if(response.toUpperCase().startsWith("I ORDER YOU TO "))
216                                                                response=response.substring(("I ORDER YOU TO ").length());
217                                                        else
218                                                        {
219                                                                CMLib.commands().postSay(mob,msg.source(),L("Master, please begin your instruction with the words 'I command you to '.  You can also tell me to 'stop' or 'cancel' any order you give."),false,false);
220                                                                return;
221                                                        }
222                                                        STEPS=CMLib.slavery().processRequest(msg.source(),mob,response);
223                                                        if((STEPS!=null)&&(STEPS.size()>0))
224                                                                CMLib.commands().postSay(mob,msg.source(),L("Yes master."),false,false);
225                                                        else
226                                                        {
227                                                                STEPS=null;
228                                                                CMLib.commands().postSay(mob,msg.source(),L("Huh? Wuh?"),false,false);
229                                                        }
230                                                }
231                                        }
232                                }
233                                else
234                                if((msg.tool() instanceof Ability)
235                                &&((((Ability)msg.tool()).classificationCode()&Ability.ALL_ACODES)==Ability.ACODE_LANGUAGE)
236                                &&(msg.sourceMinor()!=CMMsg.TYP_TEACH))
237                                        CMLib.commands().postSay(mob,msg.source(),L("I don't understand your words."),false,false);
238                        }
239                }
240                else
241                if((mob.location()!=null)&&(getMaster()!=null))
242                {
243                        final Room room=mob.location();
244                        if((room!=lastRoom)
245                        &&(CMLib.law().doesHavePriviledgesHere(getMaster(),room))
246                        &&(room.isInhabitant(mob)))
247                        {
248                                lastRoom=room;
249                                mob.basePhyStats().setRejuv(PhyStats.NO_REJUV);
250                                mob.setStartRoom(room);
251                        }
252                }
253                else
254                if((msg.sourceMinor()==CMMsg.TYP_SHUTDOWN)
255                ||((msg.targetMinor()==CMMsg.TYP_EXPIRE)&&((msg.target()==mob.location())||(msg.target()==mob)||(msg.target()==mob.amFollowing())))
256                ||((msg.sourceMinor()==CMMsg.TYP_QUIT)&&(msg.amISource(mob.amFollowing()))))
257                        mob.setFollowing(null);
258        }

260        @Override
261        public boolean tick(final Tickable ticking, final int tickID)
262        {
263                if(!(affected instanceof MOB))
264                        return super.tick(ticking,tickID);
265                if(tickID==Tickable.TICKID_MOB)
266                {
267                        final MOB mob=(MOB)ticking;
268                        if((speedDown>-500)&&((--speedDown)>=0))
269                        {
270                                for(int a=mob.numEffects()-1;a>=0;a--) // personal
271                                {
272                                        final Ability A=mob.fetchEffect(a);
273                                        if((A!=null)&&((A.classificationCode()&Ability.ALL_ACODES)==Ability.ACODE_COMMON_SKILL))
274                                        {
275                                                if(!A.tick(ticking,tickID))
276                                                        mob.delEffect(A);
277                                        }
278                                }
279                        }
280                        if((--hungerTickDown)<=0)
281                        {
282                                hungerTickDown=HUNGERTICKMAX;
283                                CMLib.combat().expendEnergy(mob,false);
284                                if((!mob.isInCombat())&&(CMLib.dice().rollPercentage()==1)&&(CMLib.dice().rollPercentage()<(masterAnger/10)))
285                                {
286                                        final MOB myMaster=getMaster();
287                                        if((myMaster!=null)&&(mob.location().isInhabitant(myMaster)))
288                                        {
289                                                mob.location().show(mob,myMaster,null,CMMsg.MSG_OK_ACTION,L("<S-NAME> rebel(s) against <T-NAMESELF>!"));
290                                                final MOB master=getMaster();
291                                                unMaster(mob);
292                                                setMiscText("");
293                                                mob.recoverCharStats();
294                                                mob.recoverPhyStats();
295                                                mob.resetToMaxState();
296                                                mob.setFollowing(null);
297                                                CMLib.combat().postAttack(mob,master,mob.fetchWieldedItem());
298                                        }
299                                        else
300                                        if(CMLib.dice().rollPercentage()<50)
301                                        {
302                                                mob.location().show(mob,myMaster,null,CMMsg.MSG_OK_ACTION,L("<S-NAME> escape(s) <T-NAMESELF>!"));
303                                                CMLib.tracking().beMobile(mob,true,true,false,false,null,null);
304                                        }
305                                }
306                                if(mob.curState().getHunger()<=0)
307                                {
308                                        Food f=null;
309                                        for(int i=0;i<mob.numItems();i++)
310                                        {
311                                                final Item I=mob.getItem(i);
312                                                if(I instanceof Food)
313                                                {
314                                                        f = (Food) I;
315                                                        break;
316                                                }
317                                        }
318                                        if(f==null)
319                                                CMLib.commands().postSay(mob,null,L("I am hungry."),false,false);
320                                        else
321                                        {
322                                                final Command C=CMClass.getCommand("Eat");
323                                                try
324                                                {
325                                                        C.execute(mob, CMParms.parse("EAT \"" + f.Name() + "$\""), MUDCmdProcessor.METAFLAG_ORDER);
326                                                }
327                                                catch (final Exception e)
328                                                {
329                                                }
330                                        }
331                                }
332                                if(mob.curState().getThirst()<=0)
333                                {
334                                        Drink d=null;
335                                        for(int i=0;i<mob.numItems();i++)
336                                        {
337                                                final Item I=mob.getItem(i);
338                                                if(I instanceof Drink)
339                                                {
340                                                        d = (Drink) I;
341                                                        break;
342                                                }
343                                        }
344                                        if(d==null)
345                                                CMLib.commands().postSay(mob,null,L("I am thirsty."),false,false);
346                                        else
347                                        {
348                                                final Command C=CMClass.getCommand("Drink");
349                                                try
350                                                {
351                                                        C.execute(mob, CMParms.parse("DRINK \"" + d.Name() + "$\""), MUDCmdProcessor.METAFLAG_ORDER);
352                                                }
353                                                catch (final Exception e)
354                                                {
355                                                }
356                                        }
357                                }
358                        }
359                        if(!mob.getLiegeID().equals(masterName))
360                        {
361                                mob.setLiegeID(masterName);
362                                final MOB myMaster=getMaster();
363                                if(myMaster!=null)
364                                {
365                                        for(final Pair<Clan,Integer> p : CMLib.clans().findRivalrousClans(myMaster))
366                                                mob.setClan(p.first.clanID(),p.first.getGovernment().getAcceptPos());
367                                }
368                        }
369                        if(STEPS==null)
370                        {
371                                // wait to be told to do something
372                        }
373                        else
374                        if((STEPS.size()==0)||(STEPS.isDone()))
375                        {
376                                if(mob.isInCombat())
377                                        return true; // let them finish fighting.
378                                if((STEPS!=null)&&((STEPS.size()==0)||(STEPS.isDone())))
379                                        mob.tell(L("You have completed your masters task."));
380                                else
381                                        mob.tell(L("You have been released from your masters task."));
382                                if((mob.isMonster())
383                                &&(!mob.amDead())
384                                &&(mob.location()!=null)
385                                &&(mob.location()!=mob.getStartRoom()))
386                                        CMLib.tracking().wanderAway(mob,true,true);
387                                unInvoke();
388                                STEPS=null;
389                                return !canBeUninvoked();
390                        }
391                        if(STEPS!=null)
392                        {
393                                STEPS.step();
394                        }
395                }
396                return super.tick(ticking,tickID);
397        }

399        @Override
400        public void unInvoke()
401        {
402                MOB mob=null;
403                if(affected instanceof MOB)
404                        mob=(MOB)affected;
405                super.unInvoke();
406                if(this.masterMOB!=null)
407                        unMaster(mob);
408        }

410        @Override
411        public boolean invoke(MOB mob, List<String> commands, Physical givenTarget, boolean auto, int asLevel)
412        {
413                if(mob.isMonster())
414                {
415                        mob.location().show(mob,null,CMMsg.MSG_NOISE,L("<S-NAME> sigh(s)."));
416                        CMLib.commands().postSay(mob,null,L("You know, if I had any ambitions, I would enslave myself so I could do interesting things!"),false,false);
417                        return false;
418                }
419
420                if(commands.size()<1)
421                {
422                        mob.tell(L("You need to specify a target to enslave."));
423                        return false;
424                }
425                final MOB target=getTarget(mob,commands,givenTarget,false,true);
426                if(target==null)
427                        return false;
428                if(target.charStats().getStat(CharStats.STAT_INTELLIGENCE)<5)
429                {
430                        mob.tell(L("@x1 would be too stupid to understand your instructions!",target.name(mob)));
431                        return false;
432                }
433
434                if((!CMLib.flags().isBoundOrHeld(target))
435                &&(target.fetchEffect(ID())==null)
436                &&(!CMSecurity.isAllowed(mob,target.location(), CMSecurity.SecFlag.CMDMOBS)))
437                {
438                        mob.tell(L("@x1 must be bound first.",target.name(mob)));
439                        return false;
440                }
441
442                if(!super.invoke(mob,commands,givenTarget,auto,asLevel))
443                        return false;
444
445                final boolean success=proficiencyCheck(mob,0,auto);
446
447                if(success)
448                {
449                        invoker=mob;
450                        final boolean peace1=!mob.isInCombat();
451                        final boolean peace2=!target.isInCombat();
452                        final CMMsg msg=CMClass.getMsg(mob,target,this,CMMsg.MSG_NOISE|CMMsg.MASK_MALICIOUS,auto?"":L("^S<S-NAME> enslave(s) <T-NAMESELF>!^?"));
453                        if(mob.location().okMessage(mob,msg))
454                        {
455                                mob.location().send(mob,msg);
456                                if(peace2)
457                                        target.makePeace(true);
458                                if(peace1)
459                                        mob.makePeace(true);
460                                Ability A=target.fetchEffect(ID());
461                                if(A==null)
462                                {
463                                        A=(Ability)copyOf();
464                                        target.addNonUninvokableEffect(A);
465                                }
466                                A.setMiscText(mob.Name());
467                        }
468                }
469                else
470                        return beneficialWordsFizzle(mob,target,L("<S-NAME> attempt(s) to enslave on <T-NAMESELF>, but fails."));
471
472                // return whether it worked
473                return success;
474        }
