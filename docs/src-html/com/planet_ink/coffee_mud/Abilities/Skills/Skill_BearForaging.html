116        public void performForage(final MOB mob, final Environmental E)
117        {
118                if(E instanceof MOB)
119                {
120                        final MOB M=(MOB)E;
121                        final Environmental foodE=this.isFoodHere(mob, E);
122                        if((foodE instanceof Item)&&(((Item)foodE).owner() == M))
123                        {
124                                final Item I=(Item)foodE;
125                                Item dropI=I;
126                                final List<Item> containerSet=this.containerSet(mob, I);
127                                if((containerSet != null)&&(containerSet.size()>0))
128                                        dropI=containerSet.get(0);
129                                CMMsg msg=CMClass.getMsg(mob,M,this,CMMsg.MASK_MALICIOUS|CMMsg.MSG_THIEF_ACT,L("<S-NAME> suddenly attack(s) <T-NAME> ripping @x1 away from <T-HIM-HER>.",dropI.Name()));
130                                if(mob.location().okMessage(mob,msg))
131                                {
132                                        mob.location().send(mob,msg);
133                                        msg=CMClass.getMsg(M,dropI,null,CMMsg.MSG_DROP,CMMsg.MSG_DROP,CMMsg.MSG_NOISE,null);
134                                        if(M.location().okMessage(M,msg))
135                                        {
136                                                M.location().send(mob,msg);
137                                                if(M.location().isContent(dropI))
138                                                        performForage(mob, foodE);
139                                        }
140                                }
141                        }
142                }
143                else
144                if(E instanceof Item)
145                {
146                        Item I=(Item)E;
147                        final List<Item> containerSet=this.containerSet(mob, I);
148                        if((containerSet != null)&&(containerSet.size()>0))
149                        {
150                                for(final Item containerI : containerSet)
151                                {
152                                        final CMMsg msg=CMClass.getMsg(mob,containerI,CMMsg.MASK_MALICIOUS|CMMsg.MSG_NOISYMOVEMENT,L("<S-NAME> rips through <T-NAME>."));
153                                        if(mob.location().okMessage(mob,msg))
154                                        {
155                                                mob.location().send(mob,msg);
156                                                if(containerI instanceof Container)
157                                                        ((Container)containerI).emptyPlease(false);
158                                                containerI.destroy();
159                                        }
160                                        else
161                                                return;
162                                }
163                        }
164                        if((I.container()==null)&&(I instanceof Food))
165                        {
166                                CMLib.commands().postGet(mob, null, I, true);
167                                final Command eatC=CMClass.getCommand("Eat");
168                                if(eatC != null)
169                                {
170                                        try
171                                        {
172                                                eatC.execute(mob, new XVector<String>("EAT",I.Name()), 0);
173                                        }
174                                        catch (IOException e)
175                                        {
176                                                Log.errOut(e);
177                                        }
178                                }
179                        }
180                }
181        }

183        protected List<Item> containerSet(final MOB mob, Item I)
184        {
185                final List<Item> containers=new ArrayList<Item>();
186                int tries=99;
187                while((I.container()!=null)&&(I.container()!=I)&&(--tries>0))
188                {
189                        I=I.container();
190                        if((!CMLib.utensils().canBePlayerDestroyed(mob,I,false,true))
191                        ||(CMLib.flags().isABonusItems(I))
192                        ||(I instanceof MiscMagic))
193                                return null;
194                        switch(I.material() & RawMaterial.MATERIAL_MASK)
195                        {
196                        case RawMaterial.MATERIAL_CLOTH:
197                        case RawMaterial.MATERIAL_GLASS:
198                        case RawMaterial.MATERIAL_LIQUID:
199                        case RawMaterial.MATERIAL_PAPER:
200                                containers.add(I);
201                                break;
202                        default:
203                                return null;
204                        }
205                }
206                return containers;
207        }

209        @Override
210        public boolean tick(final Tickable ticking, final int tickID)
211        {
212                if(!super.tick(ticking,tickID))
213                        return false;
214                if(tickID==Tickable.TICKID_MOB)
215                {
216                        if(nextDirection==-999)
217                                return true;
218
219                        if((theTrail==null)
220                        ||(affected == null)
221                        ||(!(affected instanceof MOB)))
222                                return false;
223
224                        final MOB mob=(MOB)affected;
225
226                        if(nextDirection==999)
227                        {
228                                final Environmental E=isFoodHere(mob,mob.location());
229                                if(E!=null)
230                                {
231                                        mob.tell(L("You found a meal!"));
232                                        performForage(mob,E);
233                                }
234                                else
235                                        mob.tell(L("The trail dries up here."));
236                                nextDirection=-2;
237                                unInvoke();
238                        }
239                        else
240                        if(nextDirection==-1)
241                        {
242                                final Environmental E=isFoodHere(mob,mob.location());
243                                if(E==null)
244                                        mob.tell(L("The trail dries up here."));
245                                else
246                                {
247                                        mob.tell(L("You found a meal!"));
248                                        performForage(mob,E);
249                                }
250                                nextDirection=-999;
251                                unInvoke();
252                        }
253                        else
254                        if(nextDirection>=0)
255                        {
256                                mob.tell(L("The way home seems to continue @x1.",CMLib.directions().getDirectionName(nextDirection)));
257                                if(mob.isMonster())
258                                {
259                                        final Room nextRoom=mob.location().getRoomInDir(nextDirection);
260                                        if((nextRoom!=null)&&(nextRoom.getArea()==mob.location().getArea()))
261                                        {
262                                                final int dir=nextDirection;
263                                                nextDirection=-2;
264                                                CMLib.tracking().walk(mob,dir,false,false);
265                                        }
266                                        else
267                                                unInvoke();
268                                }
269                                else
270                                        nextDirection=-2;
271                        }
272
273                }
274                return true;
275        }

277        @Override
278        public void executeMsg(final Environmental myHost, final CMMsg msg)
279        {
280                super.executeMsg(myHost,msg);
281
282                if(!(affected instanceof MOB))
283                        return;
284
285                final MOB mob=(MOB)affected;
286                if((msg.amISource(mob))
287                &&(msg.amITarget(mob.location()))
288                &&(CMLib.flags().canBeSeenBy(mob.location(),mob))
289                &&(msg.targetMinor()==CMMsg.TYP_LOOK))
290                        nextDirection=CMLib.tracking().trackNextDirectionFromHere(theTrail,mob.location(),false);
291                else
292                if((affected!=null)
293                &&(affected instanceof MOB)
294                &&(msg.target() instanceof Room)
295                &&(msg.amISource((MOB)affected))
296                &&((msg.sourceMinor()==CMMsg.TYP_LOOK)||(msg.sourceMinor()==CMMsg.TYP_EXAMINE)))
297                {
298                        if((msg.tool()!=null)&&(msg.tool().ID().equals(ID())))
299                        {
300                                final Environmental E=isFoodHere((MOB)affected,(Room)msg.target());
301                                if(E!=null)
302                                {
303                                        ((MOB)affected).tell(L("You sense a meal here."));
304                                        performForage(mob,E);
305                                        unInvoke();
306                                }
307                        }
308                        else
309                        {
310                                final Environmental E=isFoodHere((MOB)affected,(Room)msg.target());
311                                if(E!=null)
312                                {
313                                        final CMMsg msg2=CMClass.getMsg(msg.source(),msg.target(),this,CMMsg.MSG_LOOK,CMMsg.NO_EFFECT,CMMsg.NO_EFFECT,null);
314                                        msg.addTrailerMsg(msg2);
315                                        // targets the above
316                                }
317                        }
318                }
319        }

328        private Environmental isFoodHere(final MOB mob, final Environmental E)
329        {
330                if(E==null)
331                        return null;
332                if(E instanceof MOB)
333                {
334                        final MOB M=(MOB)E;
335                        for(final Enumeration<Item> i=M.items();i.hasMoreElements();)
336                        {
337                                final Item I=i.nextElement();
338                                if((I!=null)
339                                &&(I instanceof Food)
340                                &&(containerSet(mob,I)!=null))
341                                        return I;
342                        }
343                }
344                else
345                if(E instanceof Item)
346                {
347                        if((E instanceof Food)
348                        &&(containerSet(mob,(Item)E)!=null))
349                                return E;
350                }
351                else
352                if(E instanceof Room)
353                {
354                        final Room room=(Room)E;
355                        for(final Enumeration<Item> i=room.items();i.hasMoreElements();)
356                        {
357                                final Item I=i.nextElement();
358                                if((I!=null)
359                                &&(I instanceof Food)
360                                &&(containerSet(mob,I)!=null))
361                                        return I;
362                        }
363                        for(final Enumeration<MOB> m=room.inhabitants();m.hasMoreElements();)
364                        {
365                                final MOB M=m.nextElement();
366                                if((M!=null)
367                                &&(mob.mayIFight(M))
368                                &&(M!=mob))
369                                {
370                                        final Environmental I=this.isFoodHere(mob, M);
371                                        if(I!=null)
372                                                return M;
373                                }
374                        }
375                }
376                return null;
377        }

379        @Override
380        public boolean invoke(MOB mob, List<String> commands, Physical givenTarget, boolean auto, int asLevel)
381        {
382                final List<Ability> V=CMLib.flags().flaggedAffects(mob,Ability.FLAG_TRACKING);
383                for(final Ability A : V) 
384                        A.unInvoke();
385                if(V.size()>0)
386                {
387                        mob.tell(L("You stop tracking."));
388                        if(commands.size()==0)
389                                return true;
390                }
391
392                if(!super.invoke(mob,commands,givenTarget,auto,asLevel))
393                        return false;
394
395                final Environmental hereE=isFoodHere(mob, mob.location());
396                if(hereE!=null)
397                {
398                        mob.tell(L("You smell something RIGHT HERE!!"));
399                        performForage(mob,hereE);
400                        return true;
401                }
402
403                final boolean success=proficiencyCheck(mob,0,auto);
404
405                final ArrayList<Room> rooms=new ArrayList<Room>();
406                TrackingLibrary.TrackingFlags flags;
407                flags = CMLib.tracking().newFlags()
408                                .plus(TrackingLibrary.TrackingFlag.NOEMPTYGRIDS)
409                                .plus(TrackingLibrary.TrackingFlag.NOAIR);
410                int range=3 + (adjustedLevel(mob,asLevel)/15) + (super.getXMAXRANGELevel(mob));
411                final List<Room> checkSet=CMLib.tracking().getRadiantRooms(mob.location(),flags,range);
412                for (final Room room : checkSet)
413                {
414                        final Room R=CMLib.map().getRoom(room);
415                        if(isFoodHere(mob,R)!=null)
416                                rooms.add(R);
417                }
418
419                if(rooms.size()>0)
420                        theTrail=CMLib.tracking().findTrailToAnyRoom(mob.location(),rooms,flags,range);
421
422                if((success)&&(theTrail!=null))
423                {
424                        final CMMsg msg=CMClass.getMsg(mob,null,this,CMMsg.MSG_QUIETMOVEMENT,auto?L("<S-NAME> begin(s) foraging!"):L("<S-NAME> begin(s) foraging for someone elses food."));
425                        if(mob.location().okMessage(mob,msg))
426                        {
427                                mob.location().send(mob,msg);
428                                final Skill_BearForaging newOne=(Skill_BearForaging)this.copyOf();
429                                if(mob.fetchEffect(newOne.ID())==null)
430                                        mob.addEffect(newOne);
431                                mob.recoverPhyStats();
432                                newOne.nextDirection=CMLib.tracking().trackNextDirectionFromHere(newOne.theTrail,mob.location(),false);
433                        }
434                }
435                else
436                        beneficialVisualFizzle(mob,null,L("<S-NAME> attempt(s) to forage for someone elses food, but fail(s)."));
437
438                return success;
439        }
