175        public Behavior getSailor()
176        {
177                if(affected instanceof PhysicalAgent)
178                {
179                        PhysicalAgent agent=(PhysicalAgent)affected;
180                        if((sailor == null)||(agent.fetchBehavior("Sailor")!=sailor))
181                        {
182                                final Behavior B=agent.fetchBehavior("Sailor");
183                                if(B!=null)
184                                        agent.delBehavior(B);
185                                sailor = CMClass.getBehavior("Sailor");
186                                switch(type)
187                                {
188                                case REPAIRER:
189                                        if(agent instanceof MOB)
190                                        {
191                                                final MOB mob=(MOB)agent;
192                                                if(mob.fetchAbility("Shipwright")==null)
193                                                {
194                                                        Ability A=CMClass.getAbility("Shipwright");
195                                                        int prof=(mob.phyStats().level()+(5*super.getXLEVELLevel(invoker())))*2;
196                                                        if(prof>=100)
197                                                                prof=99;
198                                                        A.setProficiency(prof);
199                                                        A.setSavable(true);
200                                                        mob.addAbility(A);
201                                                }
202                                        }
203                                        sailor.setParms("FIGHTMOVER=false FIGHTTECH=false TICKBONUS="+abilityCode());
204                                        break;
205                                case DEFENDER:
206                                        sailor.setParms("DEFENDER=true FIGHTMOVER=false FIGHTTECH=false TICKBONUS="+abilityCode());
207                                        break;
208                                case TACTICIAN:
209                                        sailor.setParms("FIGHTMOVER=true FIGHTTECH=false TICKBONUS="+abilityCode());
210                                        break;
211                                case CAPTAIN:
212                                        sailor.setParms("PEACEMOVER=true WANDEROK=true FIGHTMOVER=false FIGHTTECH=false TICKBONUS="+abilityCode());
213                                        break;
214                                case TRAWLER:
215                                        if(agent instanceof MOB)
216                                        {
217                                                final MOB mob=(MOB)agent;
218                                                if(mob.fetchAbility("Trawling")==null)
219                                                {
220                                                        Ability A=CMClass.getAbility("Trawling");
221                                                        int prof=(mob.phyStats().level()+(5*super.getXLEVELLevel(invoker())))*2;
222                                                        if(prof>=100)
223                                                                prof=99;
224                                                        A.setProficiency(prof);
225                                                        A.setSavable(true);
226                                                        mob.addAbility(A);
227                                                }
228                                        }
229                                        sailor.setParms("FIGHTMOVER=false FIGHTTECH=false TICKBONUS="+abilityCode());
230                                        break;
231                                }
232                                sailor.setSavable(false);
233                                ((PhysicalAgent)affected).addBehavior(sailor);
234                        }
235                }
236                return sailor;
237        }

244                {
245                        PhysicalAgent agent=(PhysicalAgent)affected;
246                        final Behavior B=agent.fetchBehavior("Sailor");

248                                agent.delBehavior(B);

251                        {
252                                R.show((MOB)agent, null, CMMsg.MSG_OK_VISUAL, L("<S-NAME> quit(s)."));
253                                CMLib.tracking().wanderAway((MOB)agent, false, false);
254                        }
255                }
256                super.unInvoke();
257                affected.destroy();
258        }

280        @Override
281        public boolean tick(final Tickable ticking, final int tickID)
282        {
283                if(!super.tick(ticking,tickID))
284                        return false;
285                if(!(affected instanceof MOB))
286                        return false;
287                if((type == null)||(this.shipName.length()==0))
288                {
289                        final Physical affected=this.affected;
290                        if(affected instanceof MOB)
291                        {
292                                final MOB mob=(MOB)affected;
293                                final Room R=mob.location();
294                                if((mob.amDead())||(R==null)||(!CMLib.flags().isInTheGame(mob, true)))
295                                {
296                                        this.unInvoke();
297                                        return false;
298                                }
299                                
300                                if((R.getArea() instanceof BoardableShip)
301                                &&(((BoardableShip)R.getArea()).getShipItem() instanceof SailingShip))
302                                {
303                                        final Area shipArea=R.getArea();
304                                        final SailingShip ship = (SailingShip)((BoardableShip)shipArea).getShipItem();
305                                        if(this.type==null)
306                                        {
307                                                int numRooms=0;
308                                                int numCrew=0;
309                                                int numDecks=0;
310                                                int[] numTypes=new int[CrewType.values().length];
311                                                for(final Enumeration<Room> r=shipArea.getProperMap();r.hasMoreElements();)
312                                                {
313                                                        final Room R2=r.nextElement();
314                                                        switch(R2.domainType())
315                                                        {
316                                                        case Room.DOMAIN_INDOORS_AIR:
317                                                        case Room.DOMAIN_OUTDOORS_AIR:
318                                                                break;
319                                                        default:
320                                                                if(((R2.domainType()&Room.INDOORS)!=0))
321                                                                        numDecks++;
322                                                                numRooms++;
323                                                                break;
324                                                        }
325                                                        for(final Enumeration<MOB> m=R2.inhabitants();m.hasMoreElements();)
326                                                        {
327                                                                final MOB M=m.nextElement();
328                                                                if((M!=null)
329                                                                &&(M.isMonster())
330                                                                &&(isCrew(M,ship.Name())))
331                                                                {
332                                                                        numCrew++;
333                                                                        numTypes[getCrewType(M).ordinal()]++;
334                                                                }
335                                                        }
336                                                }
337                                                
338                                                int bonus= ( adjustedLevel(mob,0) / 10);
339                                                if(bonus > 0)
340                                                {
341                                                        int bonusDecks = bonus / 2;
342                                                        numRooms += bonus;
343                                                        numDecks += bonusDecks;
344                                                }
345                                                
346                                                CrewType nextType = null;
347                                                final int maxCaptains = 1;
348                                                final int maxTacticians = 1;
349                                                final int maxDefenders = numDecks;
350                                                final int maxTrawlers = numDecks;
351                                                final int maxRepairers = (int)Math.round(Math.ceil(CMath.div((numRooms-numDecks),2.0)));
352                                                
353                                                if(numCrew >= (maxRepairers + maxTrawlers + maxDefenders + maxTacticians + maxCaptains))
354                                                {
355                                                        CMLib.commands().postSay(mob, L("This ship already has the maximum crew."));
356                                                        unInvoke();
357                                                        return false;
358                                                }
359                                                
360                                                int attempts=10000;
361                                                while((nextType == null)&&(--attempts>0))
362                                                {
363                                                        nextType = CrewType.values()[CMLib.dice().roll(1, CrewType.values().length, -1)];
364                                                        if(numTypes[nextType.ordinal()]>0)
365                                                        {
366                                                                switch(nextType)
367                                                                {
368                                                                case TRAWLER:
369                                                                        if(numTypes[nextType.ordinal()]>=maxTrawlers)
370                                                                                nextType=null;
371                                                                        break;
372                                                                case DEFENDER:
373                                                                        if(numTypes[nextType.ordinal()]>=maxDefenders)
374                                                                                nextType=null;
375                                                                        break;
376                                                                case CAPTAIN:
377                                                                        if(numTypes[nextType.ordinal()]>=maxCaptains)
378                                                                                nextType=null;
379                                                                        break;
380                                                                case TACTICIAN:
381                                                                        if(numTypes[nextType.ordinal()]>=maxTacticians)
382                                                                                nextType=null;
383                                                                        break;
384                                                                case REPAIRER:
385                                                                        if(numTypes[nextType.ordinal()]>=maxRepairers)
386                                                                                nextType=null;
387                                                                        break;
388                                                                }
389                                                        }
390                                                }
391                                                
392                                                if(nextType == null)
393                                                {
394                                                        CMLib.commands().postSay(mob, L("This ship already has enough crew."));
395                                                        unInvoke();
396                                                        return false;
397                                                }
398                                                
399                                                this.type=nextType;
400                                                mob.recoverPhyStats();
401                                                R.recoverRoomStats();
402                                                mob.recoverPhyStats();
403                                        }
404                                        if(mob.amFollowing()!=null)
405                                        {
406                                                // I'm only done when I'm no longer following.
407                                        }
408                                        else
409                                        {
410                                                this.setMiscText(ship.Name()+";"+type.name());
411                                                this.setAbilityCode(super.getXLEVELLevel(mob));
412                                                this.makeNonUninvokable();
413                                                this.setSavable(true);
414                                                getSailor();
415                                                CMLib.commands().postSay(mob, L("I am ready for duty."));
416                                        }
417                                }
418                                else
419                                if(mob.amFollowing()==null)
420                                {
421                                        CMLib.commands().postSay(mob, L("I guess I'm fired."));
422                                        unInvoke();
423                                        return false;
424                                }
425                        }
426                }
427                else
428                        getSailor();
429                return true;
430        }

444        public boolean isPub(MOB mob, Room room)
445        {
446                for(int m=0;m<room.numInhabitants();m++)
447                {
448                        final MOB M=room.fetchInhabitant(m);
449                        if((M!=null)&&(M!=mob))
450                        {
451                                if(CMLib.flags().canBeSeenBy(M,mob))
452                                {
453                                        final ShopKeeper SK=CMLib.coffeeShops().getShopKeeper(M);
454                                        if(SK!=null)
455                                        {
456                                                for(final Iterator<Environmental> i=SK.getShop().getStoreInventory();i.hasNext();)
457                                                {
458                                                        final Environmental E=i.next();
459                                                        if((E instanceof Item)&&(CMLib.flags().isAlcoholic((Item)E)))
460                                                        {
461                                                                return true;
462                                                        }
463                                                }
464                                        }
465                                }
466                        }
467                }
468                return false;
469        }

471        @Override
472        public boolean invoke(MOB mob, List<String> commands, Physical givenTarget, boolean auto, int asLevel)
473        {
474                final Room R=mob.location();
475                if(R==null)
476                        return false;
477                double money=0.0;
478                String moneyStr="money";
479                int minLevel = mob.phyStats().level()-10;
480                if(minLevel < 1)
481                        minLevel=1;
482                int range=5;
483                if(minLevel + range > mob.phyStats().level())
484                        range = 1;
485                int level = CMLib.dice().roll(1, range, minLevel);
486                if(!auto)
487                {
488                        if(!this.isPub(mob, R))
489                        {
490                                mob.tell(L("You must in a pub to hire a sailor."));
491                                return false;
492                        }
493                        TrackingLibrary.TrackingFlags flags=CMLib.tracking().newFlags();
494                        int roomRange = baseWaterRange + super.getXLEVELLevel(mob)+super.getXMAXRANGELevel(mob);
495                        List<Room> nearby=CMLib.tracking().findTrailToAnyRoom(R, TrackingFlag.WATERSURFACEONLY.myFilter, flags, roomRange);
496                        if((nearby==null)||(nearby.size()==0))
497                        {
498                                mob.tell(L("There's no sea or river nearby, so no one here would be a sailor."));
499                                return false;
500                        }
501                        
502                        int medLevel = minLevel + (int)Math.round(CMath.ceiling(CMath.div(range, 2.0)));
503                        double amt = medLevel * 10.0;
504                        String currency=R.getArea().getCurrency();
505                        moneyStr = CMLib.beanCounter().abbreviatedPrice(currency, amt);
506                        if(CMLib.beanCounter().getTotalAbsoluteValue(mob, currency) < amt)
507                        {
508                                mob.tell(L("You need at least @x1 to hire a decent sailor here.",moneyStr));
509                                return false;
510                        }
511                        money=amt;
512                }
513                
514                if(mob.numFollowers() >= mob.maxFollowers())
515                {
516                        mob.tell(L("You have too many followers to gather up another one right now."));
517                        return false;
518                }
519
520                if(!super.invoke(mob,commands,givenTarget,auto,asLevel))
521                        return false;
522
523                boolean success=proficiencyCheck(mob,0,auto);
524                if(success)
525                {
526                        if(money > 0.0)
527                                CMLib.beanCounter().subtractMoney(mob, money);
528                        final MOB targetM=CMClass.getMOB("GenMob");
529                        final List<Race> races=CMLib.login().raceQualifies(Area.THEME_FANTASY);
530                        final Race raceR=races.get(CMLib.dice().roll(1, races.size(), -1));
531                        String name=CMLib.login().generateRandomName(1, 5);
532                        String raceName=raceR.name();
533                        switch(CMLib.dice().roll(1, 5, -1))
534                        {
535                        case 0:
536                                targetM.setName(name);
537                                break;
538                        case 1:
539                                targetM.setName(L("sailor @x1",name));
540                                break;
541                        case 2:
542                                targetM.setName(L("a sailor"));
543                                break;
544                        case 3:
545                                targetM.setName(L("an @x1",raceName));
546                                break;
547                        case 4:
548                                targetM.setName(L("a sailor @x1",raceName));
549                                break;
550                        }
551                        targetM.setDisplayText(L("@x1 stands here",targetM.Name()));
552                        targetM.setDescription("");
553                        targetM.basePhyStats().setAbility(CMProps.getMobHPBase());
554                        targetM.basePhyStats().setDisposition(targetM.basePhyStats().disposition()|PhyStats.IS_FLYING);
555                        targetM.basePhyStats().setLevel(level);
556                        targetM.basePhyStats().setRejuv(PhyStats.NO_REJUV);
557                        targetM.recoverPhyStats();
558                        targetM.baseCharStats().setMyRace(raceR);
559                        targetM.baseCharStats().setStat(CharStats.STAT_GENDER,(CMLib.dice().rollPercentage()>50)?'M':'F');
560                        targetM.baseCharStats().getMyRace().startRacing(targetM,false);
561                        targetM.recoverPhyStats();
562                        targetM.recoverCharStats();
563                        targetM.basePhyStats().setArmor(CMLib.leveler().getLevelMOBArmor(targetM));
564                        targetM.basePhyStats().setAttackAdjustment(CMLib.leveler().getLevelAttack(targetM));
565                        targetM.basePhyStats().setDamage(CMLib.leveler().getLevelMOBDamage(targetM));
566                        targetM.basePhyStats().setSpeed(CMLib.leveler().getLevelMOBSpeed(targetM));
567                        //targetM.addNonUninvokableEffect(CMClass.getAbility("Prop_ModExperience")); -- could be dangerous not having this, but 5-10 levels lower, so...
568                        targetM.recoverCharStats();
569                        targetM.recoverPhyStats();
570                        targetM.recoverMaxState();
571                        CMLib.factions().setAlignment(targetM,Faction.Align.NEUTRAL);
572                        targetM.resetToMaxState();
573                        targetM.text();
574                        targetM.bringToLife(R,true);
575                        CMLib.beanCounter().clearZeroMoney(targetM,null);
576                        targetM.setMoneyVariation(0);
577                        //targetM.location().showOthers(targetM,null,CMMsg.MSG_OK_ACTION,L("<S-NAME> appears!"));
578                        R.recoverRoomStats();
579                        targetM.setStartRoom(null);
580                        
581                        String str=auto?"":L("^S<S-NAME> offer(s) <T-NAME> @x1 to hire on as a crewmember..^?",moneyStr);
582                        final CMMsg msg=CMClass.getMsg(mob,targetM,this,CMMsg.MSG_NOISYMOVEMENT,str,CMMsg.MSG_NOISYMOVEMENT|(auto?CMMsg.MASK_ALWAYS:0),str,CMMsg.MSG_NOISYMOVEMENT,str);
583                        if(R.okMessage(mob,msg))
584                        {
585                                R.send(mob,msg);
586                                R.show(targetM, null, CMMsg.MSG_QUIETMOVEMENT,L("<S-NAME> sign(s) up to be a member of your crew.",targetM.name()));
587                                type=null;
588                                Skill_HireCrewmember A=(Skill_HireCrewmember)this.beneficialAffect(mob, targetM, asLevel, 0);
589                                A.type=null;
590                                CMLib.commands().postFollow(targetM, mob, false);
591                        }
592                        else
593                                targetM.destroy();
594                }
595                else
596                        return beneficialWordsFizzle(mob,null,L("<S-NAME> offer(s) @x1 to potential sailors, but no one is interested.",moneyStr));
597
598                // return whether it worked
599                return success;
600        }
