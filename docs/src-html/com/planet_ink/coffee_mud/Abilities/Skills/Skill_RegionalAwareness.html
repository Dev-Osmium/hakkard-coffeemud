200        public String[] getMiniMap(Room room, final int diameter, boolean openOnly)
201        {
202                final char[][] map=new char[diameter][diameter];
203                for(int i=0;i<diameter;i++)
204                {
205                        for(int i2=0;i2<diameter;i2++)
206                                map[i][i2]=' ';
207                }
208                final boolean amIndoors=((room.domainType()&Room.INDOORS)==Room.INDOORS);
209                final Room[][] rmap=new Room[diameter][diameter];
210                final ArrayList<Room> rooms=new ArrayList<Room>();
211                final HashSet<Room> closedPaths=new HashSet<Room>();
212                TrackingLibrary.TrackingFlags flags;
213                flags = CMLib.tracking().newFlags()
214                                        .plus(TrackingLibrary.TrackingFlag.NOEMPTYGRIDS)
215                                        .plus(TrackingLibrary.TrackingFlag.NOAIR);
216                if(openOnly)
217                        flags = flags.plus(TrackingLibrary.TrackingFlag.OPENONLY);
218
219                CMLib.tracking().getRadiantRooms(room,rooms,flags,null,diameter,null);
220                rmap[diameter/2][diameter/2]=room;
221                map[diameter/2][diameter/2]='*';
222                for(int i=0;i<rooms.size();i++)
223                {
224                        final Room R=rooms.get(i);
225                        if((closedPaths.contains(R))
226                        ||(R==room))
227                                continue;
228                        Room parentR=null;
229                        int parentDir=-1;
230                        int[] xy=null;
231                        for(int i2=0;(i2<diameter)&&(parentR==null);i2++)
232                        {
233                                for(int i3=0;(i3<diameter)&&(parentR==null);i3++)
234                                {
235                                        final Room R2=rmap[i2][i3];
236                                        if(R2!=null)
237                                        for(int d=Directions.NUM_DIRECTIONS()-1;d>=0;d--)
238                                        {
239                                                if((R2.getRoomInDir(d)==R)
240                                                &&(!closedPaths.contains(R2))
241                                                &&(R2.getExitInDir(d)!=null))
242                                                {
243                                                        parentR=R2;
244                                                        parentDir=d;
245                                                        xy=Directions.adjustXYByDirections(i3,i2,d);
246                                                        break;
247                                                }
248                                        }
249                                }
250                        }
251                        if(xy!=null)
252                        {
253                                if((parentDir<0)
254                                ||(xy[0]<0)||(xy[0]>=diameter)||(xy[1]<0)||(xy[1]>=diameter)
255                                ||(map[xy[1]][xy[0]]!=' '))
256                                        closedPaths.add(R);
257                                else
258                                {
259                                        map[xy[1]][xy[0]]=roomChar(R,!amIndoors);
260                                        rmap[xy[1]][xy[0]]=R;
261
262                                        if((R.domainType()&Room.INDOORS)==Room.INDOORS)
263                                                closedPaths.add(R);
264                                }
265                        }
266                }
267                final String[] miniMap=new String[diameter];
268                final StringBuffer str=new StringBuffer("");
269                char r=' ';
270                char c=' ';
271                for(int i2=0;i2<diameter;i2++)
272                {
273                        str.setLength(0);
274                        for(int i3=0;i3<diameter;i3++)
275                        {
276                                r=map[i2][i3];
277                                c=roomColor(rmap[i2][i3]);
278                                if(c!=' ')
279                                        str.append("^"+c+""+r);
280                                else
281                                        str.append(r);
282                        }
283                        miniMap[i2]=str.toString();
284                }
285                return miniMap;
286        }

288        @Override
289        public boolean invoke(MOB mob, List<String> commands, Physical givenTarget, boolean auto, int asLevel)
290        {
291                final Session sess = mob.session();
292                if(auto && (givenTarget instanceof Room) && (asLevel>0))
293                {
294                        final String[] miniMap=getMiniMap((Room)givenTarget, asLevel, false);
295                        if(commands!=null)
296                        {
297                                for(final String s : miniMap)
298                                        commands.add(s);
299                        }
300                        else
301                        if(sess != null)
302                        {
303                                for(final String s : miniMap)
304                                {
305                                        sess.setIdleTimers();
306                                        sess.colorOnlyPrintln(s);
307                                }
308                        }
309                        return true;
310                }
311
312                if((!auto)&&((mob.location().domainType()&Room.INDOORS)==Room.INDOORS))
313                {
314                        mob.tell(L("This only works outdoors."));
315                        return false;
316                }
317
318                if((!auto)
319                &&(!CMLib.flags().canBeSeenBy(mob.location(),mob)))
320                {
321                        mob.tell(L("You need to be able to see your surroundings to do that."));
322                        return false;
323                }
324
325                if(!super.invoke(mob,commands,givenTarget,auto,asLevel))
326                        return false;
327
328                final boolean success=proficiencyCheck(mob,0,auto);
329                if(success)
330                {
331                        final CMMsg msg=CMClass.getMsg(mob,null,this,CMMsg.MSG_LOOK,auto?"":L("<S-NAME> peer(s) at the horizon with a distant expression."));
332                        if(mob.location().okMessage(mob,msg))
333                        {
334                                mob.location().send(mob,msg);
335                                final String[] miniMap=getMiniMap(mob.location(), 2+(adjustedLevel(mob,asLevel)/10), true);
336                                if(sess != null)
337                                {
338                                        for(final String s : miniMap)
339                                        {
340                                                sess.setIdleTimers(); // prevents spam block
341                                                sess.colorOnlyPrintln(s+"\n\r");
342                                        }
343                                }
344                        }
345                }
346                else
347                        beneficialVisualFizzle(mob,null,L("<S-NAME> peer(s) around distantly, looking frustrated."));
348                return success;
349        }
