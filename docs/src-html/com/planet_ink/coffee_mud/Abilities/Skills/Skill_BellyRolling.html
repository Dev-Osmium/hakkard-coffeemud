035public class Skill_BellyRolling extends StdSkill
036{
037        @Override
038        public String ID()
039        {
040                return "Skill_BellyRolling";
041        }
042
043        private final static String localizedName = CMLib.lang().L("Belly Rolling");
044
045        @Override
046        public String name()
047        {
048                return localizedName;
049        }
050
051        @Override
052        public String displayText()
053        {
054                return "";
055        }
056
057        @Override
058        protected int canAffectCode()
059        {
060                return CAN_MOBS;
061        }
062
063        @Override
064        protected int canTargetCode()
065        {
066                return 0;
067        }
068
069        @Override
070        public int abstractQuality()
071        {
072                return Ability.QUALITY_BENEFICIAL_SELF;
073        }
074
075        @Override
076        public int classificationCode()
077        {
078                return Ability.ACODE_SKILL|Ability.DOMAIN_FOOLISHNESS;
079        }
080
081        @Override
082        public boolean isAutoInvoked()
083        {
084                return true;
085        }
086
087        @Override
088        public boolean canBeUninvoked()
089        {
090                return false;
091        }
092
093        protected boolean doneThisRound=false;
094
095        @Override
096        public boolean tick(final Tickable ticking, final int tickID)
097        {
098                if(tickID==Tickable.TICKID_MOB)
099                        doneThisRound=false;
100                return super.tick(ticking,tickID);
101        }
102
103        @Override
104        public boolean okMessage(final Environmental myHost, final CMMsg msg)
105        {
106                if(!(affected instanceof MOB))
107                        return true;
108
109                final MOB mob=(MOB)affected;
110
111                if(msg.amITarget(mob)
112                &&(msg.targetMinor()==CMMsg.TYP_WEAPONATTACK)
113                &&(CMLib.flags().isAliveAwakeMobile(mob,true))
114                &&(CMLib.flags().isSitting(mob))
115                &&(!doneThisRound)
116                &&(msg.tool() instanceof Weapon))
117                {
118                        // can't use -NAME for msg.source() lest sitting prevent it
119                        final CMMsg msg2=CMClass.getMsg(mob,msg.source(),null,CMMsg.MSG_SITMOVE,L("<S-NAME> roll(s) away from the attack by <T-NAMESELF>!"));
120                        if((proficiencyCheck(null,mob.charStats().getStat(CharStats.STAT_DEXTERITY)-50+(super.getXLEVELLevel(mob)*5),false))
121                        &&((msg.source().getVictim()==mob)||(msg.source().getVictim()==null))
122                        &&(mob.location().okMessage(mob,msg2)))
123                        {
124                                doneThisRound=true;
125                                mob.location().send(mob,msg2);
126                                helpProficiency(mob, 0);
127                                return false;
128                        }
129                }
130                return true;
131        }
132}
