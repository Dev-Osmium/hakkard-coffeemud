101        @Override
102        public void unInvoke()
103        {
104                // undo the affects of this spell
105                if(!(affected instanceof MOB))
106                        return;
107                final MOB mob=(MOB)affected;
108
109                super.unInvoke();
110
111                if(canBeUninvoked())
112                {
113                        if(text().length()>0)
114                                mob.tell(L("You start to have doubts about @x1.",text()));
115                        mob.setWorshipCharID(priorFaith);
116                }
117        }
118
119        @Override
120        public boolean tick(final Tickable ticking, final int tickID)
121        {
122                if((text().length()>0)&&(affected instanceof MOB)&&(!text().equals(((MOB)affected).getWorshipCharID())))
123                        ((MOB)affected).setWorshipCharID(text());
124                return super.tick(ticking,tickID);
125        }

127        @Override
128        public boolean invoke(final MOB mob, final List<String> commands, final Physical givenTarget, final boolean auto, final int asLevel)
129        {
130                final List<String> oldCommands = new XVector<String>(commands);
131                if(commands.size()==0)
132                {
133                        mob.tell(L("You must specify either a deity to convert yourself to, or a player to convert to your religion."));
134                        if(mob.isMonster())
135                                CMLib.commands().postSay(mob,null,L("I am unable to convert."),false,false);
136                        return false;
137                }
138
139                MOB target=mob;
140                Deity D=CMLib.map().getDeity(CMParms.combine(commands,0));
141                if(D==null)
142                {
143                        D=mob.getMyDeity();
144                        target=getTarget(mob,commands,givenTarget,false,true);
145                        if(target==null)
146                        {
147                                mob.tell(L("You've also never heard of a deity called '@x1'.",CMParms.combine(commands,0)));
148                                if(mob.isMonster())
149                                        CMLib.commands().postSay(mob,target,L("I've never heard of '@x1'.",CMParms.combine(commands,0)),false,false);
150                                return false;
151                        }
152                        if(D==null)
153                        {
154                                mob.tell(L("A faithless one cannot convert @x1.",target.name(mob)));
155                                if(mob.isMonster())
156                                        CMLib.commands().postSay(mob,target,L("I am faithless, and can not convert you."),false,false);
157                                return false;
158                        }
159                }
160                if((CMLib.flags().isAnimalIntelligence(target))
161                ||((target.isMonster())&&(target.phyStats().level()>mob.phyStats().level())))
162                {
163                        mob.tell(L("You can't convert @x1.",target.name(mob)));
164                        if(mob.isMonster())
165                                CMLib.commands().postSay(mob,target,L("I can not convert you."),false,false);
166                        return false;
167                }
168                if(target.getMyDeity()==D)
169                {
170                        mob.tell(L("@x1 already worships @x2.",target.name(mob),D.name()));
171                        if(mob.isMonster())
172                                CMLib.commands().postSay(mob,target,L("You already worship @x1.",D.Name()),false,false);
173                        return false;
174                }
175                if(!auto)
176                {
177                        convertStack.clear();
178                        if(convertStack.containsFirst(target))
179                        {
180                                final Long L=convertStack.getSecond(convertStack.indexOfFirst(target));
181                                if((System.currentTimeMillis()-L.longValue())>CMProps.getMillisPerMudHour()*5)
182                                        convertStack.removeElementFirst(target);
183                        }
184                        if(convertStack.containsFirst(target))
185                        {
186                                mob.tell(L("@x1 must wait to be converted again.",target.name(mob)));
187                                if(mob.isMonster())
188                                        CMLib.commands().postSay(mob,target,L("You must wait to be converted again."),false,false);
189                                return false;
190                        }
191                }
192
193                final boolean success=proficiencyCheck(mob,0,auto);
194                boolean targetMadeSave=(givenTarget == target) || CMLib.dice().roll(1,100,0)>(target.charStats().getSave(CharStats.STAT_FAITH));
195                //if(CMSecurity.isASysOp(mob))
196                //      targetMadeSave=false;
197                if((!target.isMonster())
198                &&(success)
199                &&(targetMadeSave)
200                &&(target.getMyDeity()!=null))
201                {
202                        mob.tell(L("@x1 is worshipping @x2.  @x3 must REBUKE @x4 first.",target.name(mob),target.getMyDeity().name(),target.charStats().HeShe(),target.getMyDeity().charStats().himher()));
203                        if(mob.isMonster())
204                                CMLib.commands().postSay(mob,target,L("You already worship @x1.",target.getMyDeity().Name()),false,false);
205                        return false;
206                }
207                if((success)
208                &&(targetMadeSave)
209                &&(!target.isMonster())
210                &&(target!=mob))
211                {
212                        if(givenTarget == target)
213                                targetMadeSave=!success;
214                        else
215                        {
216                                try
217                                {
218                                        final Session tsess = target.session();
219                                        final Deity tD=D;
220                                        final MOB t=target;
221                                        final Ability A=this;
222                                        tsess.prompt(new InputCallback(InputCallback.Type.CONFIRM, "N", 30000L) 
223                                        {
224                                                
225                                                final Session session = tsess;
226                                                final Deity D=tD;
227                                                final MOB target=t;
228                                                final Ability me=A;
229
230                                                @Override
231                                                public void showPrompt()
232                                                {
233                                                        session.promptPrint(L("\n\r@x1 is trying to convert you to the worship of @x2.  Is this what you want (N/y)?",mob.name(target),D.name()));
234                                                }
235        
236                                                @Override
237                                                public void timedOut()
238                                                {
239                                                }
240        
241                                                @Override
242                                                public void callBack()
243                                                {
244                                                        if(this.confirmed)
245                                                                me.invoke(mob, oldCommands, target, false, asLevel);
246                                                        else
247                                                        {
248                                                                if(CMLib.flags().isInTheGame(mob, true))
249                                                                        mob.location().show(mob,target,CMMsg.MSG_SPEAK,L("<S-YOUPOSS> attempt to convert <T-NAME> to the worship of @x1 is rejected.",D.name()));
250                                                        }
251                                                }
252                                        });
253                                        return false;
254                                }
255                                catch(final Exception e)
256                                {
257                                        return false;
258                                }
259                        }
260                }
261                if(!super.invoke(mob,commands,givenTarget,auto,asLevel))
262                        return false;
263
264                if((success)
265                &&((!targetMadeSave)||(target==mob)))
266                {
267                        Room dRoom=D.location();
268                        if(dRoom==mob.location())
269                                dRoom=null;
270                        if(target.getMyDeity()!=null)
271                        {
272                                final Ability A=target.fetchEffect(ID());
273                                if(A!=null)
274                                {
275                                        A.unInvoke();
276                                        target.delEffect(A);
277                                }
278                                final CMMsg msg2=CMClass.getMsg(target,D,this,CMMsg.MSG_REBUKE,null);
279                                if((mob.location().okMessage(mob,msg2))&&((dRoom==null)||(dRoom.okMessage(mob,msg2))))
280                                {
281                                        mob.location().send(target,msg2);
282                                        if(dRoom!=null)
283                                                dRoom.send(target,msg2);
284                                }
285                        }
286                        final CMMsg msg=CMClass.getMsg(mob,target,this,CMMsg.MSG_SPEAK,auto?L("<T-NAME> <T-IS-ARE> converted!"):L("<S-NAME> convert(s) <T-NAMESELF> to the worship of @x1.",D.name()));
287                        final CMMsg msg2=CMClass.getMsg(target,D,this,CMMsg.MSG_SERVE,null);
288                        if((mob.location().okMessage(mob,msg))
289                        &&(mob.location().okMessage(mob,msg2))
290                        &&((dRoom==null)||(dRoom.okMessage(mob,msg2))))
291                        {
292                                mob.location().send(mob,msg);
293                                mob.location().send(target,msg2);
294                                if(dRoom!=null)
295                                        dRoom.send(target,msg2);
296                                convertStack.addElement(target,Long.valueOf(System.currentTimeMillis()));
297                                if(mob!=target)
298                                {
299                                        if(target.isMonster())
300                                                CMLib.leveler().postExperience(mob,null,null,1,false);
301                                        else
302                                                CMLib.leveler().postExperience(mob,null,null,200,false);
303                                }
304                                if(target.isMonster())
305                                {
306                                        beneficialAffect(mob,target,asLevel,(int)(TimeManager.MILI_HOUR/CMProps.getTickMillis()));
307                                        final Skill_Convert A=(Skill_Convert)target.fetchEffect(ID());
308                                        if(A!=null)
309                                                A.priorFaith=target.getWorshipCharID();
310                                }
311                        }
312                }
313                else
314                {
315                        if((target.isMonster())&&(target.fetchEffect("Prayer_ReligiousDoubt")==null))
316                        {
317                                final Ability A=CMClass.getAbility("Prayer_ReligiousDoubt");
318                                if(A!=null)
319                                        A.invoke(mob,target,true,asLevel);
320                        }
321                        else
322                                beneficialWordsFizzle(mob,target,L("<S-NAME> attempt(s) to convert <T-NAMESELF>, but <S-IS-ARE> unconvincing."));
323                }
324
325                // return whether it worked
326                return success;
327        }
