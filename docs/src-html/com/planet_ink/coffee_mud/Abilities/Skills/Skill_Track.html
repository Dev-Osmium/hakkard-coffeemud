145        @Override
146        public boolean tick(final Tickable ticking, final int tickID)
147        {
148                if(!super.tick(ticking,tickID))
149                        return false;
150                if(tickID==Tickable.TICKID_MOB)
151                {
152                        if(nextDirection==-999)
153                                return true;
154
155                        if((theTrail==null)
156                        ||(affected == null)
157                        ||(!(affected instanceof MOB)))
158                                return false;
159
160                        final MOB mob=(MOB)affected;
161
162                        if(nextDirection==999)
163                        {
164                                mob.tell(L("The trail seems to pause here."));
165                                nextDirection=-2;
166                                unInvoke();
167                        }
168                        else
169                        if(nextDirection==-1)
170                        {
171                                mob.tell(L("The trail dries up here."));
172                                nextDirection=-999;
173                                unInvoke();
174                        }
175                        else
176                        if(nextDirection>=0)
177                        {
178                                mob.tell(L("The trail seems to continue @x1.",CMLib.directions().getDirectionName(nextDirection)));
179                                if((mob.isMonster())&&(mob.location()!=null))
180                                {
181                                        final Room oldRoom=mob.location();
182                                        final Room nextRoom=oldRoom.getRoomInDir(nextDirection);
183                                        final Exit nextExit=oldRoom.getExitInDir(nextDirection);
184                                        final int opDirection=Directions.getOpDirectionCode(nextDirection);
185                                        if((nextRoom!=null)&&(nextExit!=null))
186                                        {
187                                                boolean reclose=false;
188                                                boolean relock=false;
189                                                // handle doors!
190                                                if(nextExit.hasADoor()&&(!nextExit.isOpen()))
191                                                {
192                                                        if((nextExit.hasALock())&&(nextExit.isLocked()))
193                                                        {
194                                                                CMMsg msg=CMClass.getMsg(mob,nextExit,null,CMMsg.MSG_OK_VISUAL,CMMsg.MSG_OK_VISUAL,CMMsg.MSG_OK_VISUAL,null);
195                                                                if(oldRoom.okMessage(mob,msg))
196                                                                {
197                                                                        relock=true;
198                                                                        msg=CMClass.getMsg(mob,nextExit,null,CMMsg.MSG_OK_VISUAL,CMMsg.MSG_UNLOCK,CMMsg.MSG_OK_VISUAL,L("<S-NAME> unlock(s) <T-NAMESELF>."));
199                                                                        CMLib.utensils().roomAffectFully(msg,oldRoom,nextDirection);
200                                                                }
201                                                        }
202                                                        CMMsg msg=CMClass.getMsg(mob,nextExit,null,CMMsg.MSG_OK_VISUAL,CMMsg.MSG_OK_VISUAL,CMMsg.MSG_OK_VISUAL,null);
203                                                        if(oldRoom.okMessage(mob,msg))
204                                                        {
205                                                                reclose=true;
206                                                                msg=CMClass.getMsg(mob,nextExit,null,CMMsg.MSG_OK_VISUAL,CMMsg.MSG_OPEN,CMMsg.MSG_OK_VISUAL,L("<S-NAME> @x1(s) <T-NAMESELF>.",nextExit.openWord()));
207                                                                CMLib.utensils().roomAffectFully(msg,oldRoom,nextDirection);
208                                                        }
209                                                }
210                                                if(!nextExit.isOpen())
211                                                        unInvoke();
212                                                else
213                                                {
214                                                        final int dir=nextDirection;
215                                                        nextDirection=-2;
216                                                        CMLib.tracking().walk(mob,dir,false,false);
217                                                        if(mob.location()==nextRoom)
218                                                        {
219                                                                // backup follower mover for handcuffed followers
220                                                                final LinkedList<MOB> reMoveV=new LinkedList<MOB>();
221                                                                for(final Enumeration<Pair<MOB,Short>> e=mob.followers(); e.hasMoreElements();)
222                                                                {
223                                                                        final Pair<MOB,Short> F=e.nextElement();
224                                                                        if((F.first != null)
225                                                                        &&(F.first != mob)
226                                                                        &&(F.first.location()==oldRoom)
227                                                                        &&(F.first.location()!=nextRoom))
228                                                                                reMoveV.add(F.first);
229                                                                }
230                                                                for(final MOB M : reMoveV)
231                                                                {
232                                                                        if(CMLib.flags().isBoundOrHeld(M))
233                                                                                CMLib.tracking().walk(M,dir,false,false);
234                                                                }
235                                                                if(reclose)
236                                                                {
237                                                                        final Exit opExit=nextRoom.getExitInDir(opDirection);
238                                                                        if((opExit!=null)
239                                                                        &&(opExit.hasADoor())
240                                                                        &&(opExit.isOpen()))
241                                                                        {
242                                                                                CMMsg msg=CMClass.getMsg(mob,opExit,null,CMMsg.MSG_OK_VISUAL,CMMsg.MSG_OK_VISUAL,CMMsg.MSG_OK_VISUAL,null);
243                                                                                if(nextRoom.okMessage(mob,msg))
244                                                                                {
245                                                                                        msg=CMClass.getMsg(mob,opExit,null,CMMsg.MSG_OK_VISUAL,CMMsg.MSG_CLOSE,CMMsg.MSG_OK_VISUAL,L("<S-NAME> @x1(s) <T-NAMESELF>.",nextExit.closeWord()));
246                                                                                        CMLib.utensils().roomAffectFully(msg,nextRoom,opDirection);
247                                                                                }
248                                                                                if((opExit.hasALock())&&(relock))
249                                                                                {
250                                                                                        msg=CMClass.getMsg(mob,opExit,null,CMMsg.MSG_OK_VISUAL,CMMsg.MSG_OK_VISUAL,CMMsg.MSG_OK_VISUAL,null);
251                                                                                        if(nextRoom.okMessage(mob,msg))
252                                                                                        {
253                                                                                                msg=CMClass.getMsg(mob,opExit,null,CMMsg.MSG_OK_VISUAL,CMMsg.MSG_LOCK,CMMsg.MSG_OK_VISUAL,L("<S-NAME> lock(s) <T-NAMESELF>."));
254                                                                                                CMLib.utensils().roomAffectFully(msg,nextRoom,opDirection);
255                                                                                        }
256                                                                                }
257                                                                        }
258                                                                }
259                                                        }
260                                                }
261                                        }
262                                        else
263                                                unInvoke();
264                                }
265                                else
266                                        nextDirection=-2;
267                        }
268
269                }
270                return true;
271        }

273        @Override
274        public void executeMsg(final Environmental myHost, final CMMsg msg)
275        {
276                super.executeMsg(myHost,msg);
277
278                if(!(affected instanceof MOB))
279                        return;
280
281                final MOB mob=(MOB)affected;
282                if((msg.amISource(mob))
283                &&(msg.amITarget(mob.location()))
284                &&(msg.targetMinor()==CMMsg.TYP_LOOK))
285                        nextDirection=CMLib.tracking().trackNextDirectionFromHere(theTrail,mob.location(),false);
286        }

288        @Override
289        public boolean invoke(final MOB mob, final List<String> commands, Physical givenTarget, final boolean auto, final int asLevel)
290        {
291                tickStatus=Tickable.STATUS_MISC6;
292                if((!CMLib.flags().isAliveAwakeMobile(mob,false))||(mob.location()==null)||(!CMLib.flags().isInTheGame(mob,true)))
293                {
294                        tickStatus=Tickable.STATUS_NOT;
295                        return false;
296                }
297                tickStatus=Tickable.STATUS_MISC6+1;
298                final Room thisRoom=mob.location();
299
300                final List<Ability> V=CMLib.flags().flaggedAffects(mob,Ability.FLAG_TRACKING);
301                for(final Ability A : V)
302                        A.unInvoke();
303                if(V.size()>0)
304                {
305                        mob.tell(L("You stop tracking."));
306                        if((commands.size()==0)||(CMParms.combine(commands,0).equalsIgnoreCase("stop")))
307                        {
308                                tickStatus=Tickable.STATUS_NOT;
309                                return true;
310                        }
311                }
312
313                tickStatus=Tickable.STATUS_MISC6+2;
314                theTrail=null;
315                nextDirection=-2;
316
317                if(!super.invoke(mob,commands,givenTarget,auto,asLevel))
318                {
319                        tickStatus=Tickable.STATUS_NOT;
320                        return false;
321                }
322
323                tickStatus=Tickable.STATUS_MISC6+3;
324                int radius=50 + (10*(super.getXMAXRANGELevel(mob)+super.getXLEVELLevel(mob)));
325                boolean allowAir=true;
326                boolean allowWater=true;
327                if((commands.size()>1)
328                &&((commands.get(commands.size()-1)).toUpperCase().startsWith("RADIUS="))
329                &&(CMath.isInteger((commands.get(commands.size()-1)).substring(7))))
330                {
331                        radius=CMath.s_int((commands.get(commands.size()-1)).substring(7));
332                        commands.remove(commands.size()-1);
333                }
334                if((commands.size()>1)&&((commands.get(commands.size()-1)).equalsIgnoreCase("LANDONLY")))
335                {
336                        allowAir=false;
337                        allowWater=false;
338                        commands.remove(commands.size()-1);
339                }
340                if((commands.size()>1)&&((commands.get(commands.size()-1)).equalsIgnoreCase("NOAIR")))
341                {
342                        allowAir=false;
343                        commands.remove(commands.size()-1);
344                }
345                if((commands.size()>1)&&((commands.get(commands.size()-1)).equalsIgnoreCase("NOWATER")))
346                {
347                        allowWater=false;
348                        commands.remove(commands.size()-1);
349                }
350
351                final String mobName=CMParms.combine(commands,0);
352                if((givenTarget==null)&&(mobName.length()==0))
353                {
354                        mob.tell(L("Track whom?"));
355                        tickStatus=Tickable.STATUS_NOT;
356                        return false;
357                }
358
359                if(givenTarget==null)
360                        givenTarget=CMLib.map().getRoom(mobName);
361
362                if(givenTarget==null)
363                        givenTarget=CMLib.map().getArea(mobName);
364
365                tickStatus=Tickable.STATUS_MISC6+4;
366                if((givenTarget==null)
367                &&(thisRoom.fetchInhabitant(mobName)!=null))
368                {
369                        mob.tell(L("Try 'look'."));
370                        tickStatus=Tickable.STATUS_NOT;
371                        return false;
372                }
373
374                final ArrayList<Room> rooms=new ArrayList<Room>();
375                if(givenTarget instanceof Area)
376                        rooms.add(((Area)givenTarget).getRandomMetroRoom());
377                else
378                if(givenTarget instanceof Room)
379                        rooms.add((Room)givenTarget);
380                else
381                if((givenTarget instanceof MOB)&&(((MOB)givenTarget).location()!=null))
382                        rooms.add(((MOB)givenTarget).location());
383                else
384                if(mobName.length()>0)
385                {
386                        final Room R=CMLib.map().getRoom(mobName);
387                        if(R!=null)
388                                rooms.add(R);
389                }
390
391                final TrackingLibrary.TrackingFlags flags=CMLib.tracking().newFlags();
392                if(!(allowAir||allowWater))
393                        flags.plus(TrackingLibrary.TrackingFlag.NOEMPTYGRIDS);
394                if(!allowAir)
395                        flags.plus(TrackingLibrary.TrackingFlag.NOAIR);
396                if(!allowWater)
397                        flags.plus(TrackingLibrary.TrackingFlag.NOWATER);
398                tickStatus=Tickable.STATUS_MISC6+5;
399                if(rooms.size()<=0)
400                {
401                        try
402                        {
403                                final List<Room> checkSet=CMLib.tracking().getRadiantRooms(thisRoom,flags,radius);
404                                for (final Room room : checkSet)
405                                {
406                                        final Room R=CMLib.map().getRoom(room);
407                                        if(R!=null)
408                                        {
409                                                if(R.fetchInhabitant(mobName)!=null)
410                                                        rooms.add(R);
411                                        }
412                                }
413                        }
414                        catch (final NoSuchElementException nse)
415                        {
416                        }
417                }
418                tickStatus=Tickable.STATUS_MISC6+6;
419
420                tickStatus=Tickable.STATUS_MISC6+7;
421                final boolean success=proficiencyCheck(mob,0,auto);
422                if(rooms.size()>0)
423                {
424                        theTrail=null;
425                        tickStatus=Tickable.STATUS_MISC6+8;
426                        if((cacheCode==1)&&(rooms.size()==1))
427                                theTrail=cachedPaths.get(CMLib.map().getExtendedRoomID(thisRoom)+"->"+CMLib.map().getExtendedRoomID(rooms.get(0)));
428                        tickStatus=Tickable.STATUS_MISC6+9;
429                        if(theTrail==null)
430                                theTrail=CMLib.tracking().findTrailToAnyRoom(thisRoom,rooms,flags,radius);
431                        tickStatus=Tickable.STATUS_MISC6+10;
432                        if((cacheCode==1)&&(rooms.size()==1)&&(theTrail!=null))
433                                cachedPaths.put(CMLib.map().getExtendedRoomID(thisRoom)+"->"+CMLib.map().getExtendedRoomID(rooms.get(0)),theTrail);
434                }
435
436                tickStatus=Tickable.STATUS_MISC6+11;
437                if((success)&&(theTrail!=null))
438                {
439                        theTrail.add(thisRoom);
440
441                        final CMMsg msg=CMClass.getMsg(mob,null,this,CMMsg.MSG_QUIETMOVEMENT,mob.isMonster()?null:L("<S-NAME> begin(s) to track."));
442                        if(thisRoom.okMessage(mob,msg))
443                        {
444                                tickStatus=Tickable.STATUS_MISC6+12;
445                                thisRoom.send(mob,msg);
446                                invoker=mob;
447                                final Skill_Track newOne=(Skill_Track)copyOf();
448                                if(mob.fetchEffect(newOne.ID())==null)
449                                        mob.addEffect(newOne);
450                                mob.recoverPhyStats();
451                                tickStatus=Tickable.STATUS_MISC6+13;
452                                newOne.nextDirection=CMLib.tracking().trackNextDirectionFromHere(theTrail,thisRoom,false);
453                        }
454                        tickStatus=Tickable.STATUS_MISC6+14;
455                }
456                else
457                {
458                        tickStatus=Tickable.STATUS_NOT;
459                        return beneficialVisualFizzle(mob,null,L("<S-NAME> attempt(s) to track, but can't find the trail."));
460                }
461                tickStatus=Tickable.STATUS_NOT;
462                // return whether it worked
463                return success;
464        }
