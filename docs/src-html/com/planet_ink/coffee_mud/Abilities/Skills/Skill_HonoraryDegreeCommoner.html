102        private final String[][] getAllDegrees()
103        {
104                String[][] degrees = (String[][])Resources.getResource("SKILL_"+ID().toUpperCase());
105                if(degrees == null)
106                {
107                        final List<CharClass> classes=new ArrayList<CharClass>();
108                        for(final Enumeration<CharClass> c = CMClass.charClasses();c.hasMoreElements();)
109                        {
110                                final CharClass C=c.nextElement();
111                                if(C.baseClass().equals(getBaseClassID())
112                                &&(C.availabilityCode()!=0)
113                                &&((C.availabilityCode()&Area.THEME_SKILLONLYMASK)==0))
114                                        classes.add(C);
115                        }
116                        final List<String[]> nearFinal = new ArrayList<String[]>();
117                        for(CharClass c : classes)
118                                nearFinal.add(new String[]{c.ID(),c.name(),L("*, Honorary @x1",c.name())});
119                        degrees = nearFinal.toArray(new String[][]{});
120                        Resources.submitResource("SKILL_"+ID().toUpperCase(), degrees);
121                }
122                return degrees;
123        }

125        protected final List<String[]>  myClasses       = new ArrayList<String[]>();
126        protected final Set<String>             myTitles        = new HashSet<String>();
127        protected volatile int                  numSkills       = -1;
128        protected volatile CharClass    activatedC      = null;
129        protected volatile CharClass    elligibleC      = null;
130        protected String                                lastTitle       = "";
131
132        @Override
133        public boolean tick(final Tickable ticking, final int tickID)
134        {
135                if(tickID==Tickable.TICKID_MOB)
136                {
137                        final Physical P=affected;
138                        if(P instanceof MOB)
139                        {
140                                final MOB mob=(MOB)P;
141                                if(mob.numAbilities()!=numSkills)
142                                {
143                                        numSkills = mob.numAbilities();
144                                        this.myClasses.clear();
145                                        this.myTitles.clear();
146                                        final String[][] allDegrees = this.getAllDegrees();
147                                        final Ability studyingA=mob.fetchAbility("Studying");
148                                        final AbilityContainer collection=(studyingA==null)?mob:(AbilityContainer)studyingA;
149                                        final int[] counts=new int[allDegrees.length];
150                                        boolean fail=true;
151                                        while(fail)
152                                        {
153                                                fail=false;
154                                                try
155                                                {
156                                                        for(final Enumeration<Ability> a=collection.abilities();a.hasMoreElements();)
157                                                        {
158                                                                final Ability A=a.nextElement();
159                                                                if(CMLib.ableMapper().getAbleMap("EACH", A.ID())!=null)
160                                                                        continue;
161                                                                for(int ci=0;ci<allDegrees.length;ci++)
162                                                                {
163                                                                        if(CMLib.ableMapper().getAbleMap(allDegrees[ci][0], A.ID()) != null)
164                                                                                counts[ci]++;
165                                                                }
166                                                        }
167                                                }
168                                                catch(java.util.ConcurrentModificationException e)
169                                                {
170                                                        for(int i=0;i<counts.length;i++)
171                                                                counts[i]=0;
172                                                        fail=true;
173                                                }
174                                        }
175                                        final int threshold = 3;
176                                        synchronized(this.myClasses)
177                                        {
178                                                for(int ci=0;ci<allDegrees.length;ci++)
179                                                {
180                                                        if(counts[ci]>threshold)
181                                                        {
182                                                                final String[] degree = allDegrees[ci];
183                                                                this.myClasses.add(degree);
184                                                        }
185                                                }
186                                        }
187                                        synchronized(this.myTitles)
188                                        {
189                                                for(int ci=0;ci<allDegrees.length;ci++)
190                                                {
191                                                        if(counts[ci]>threshold)
192                                                        {
193                                                                final String[] degree = allDegrees[ci];
194                                                                this.myTitles.add(degree[DEG_TITLE]);
195                                                        }
196                                                }
197                                        }
198                                        final PlayerStats pStats = mob.playerStats();
199                                        if((pStats!=null)&&(pStats.getTitles()!=null))
200                                        {
201                                                for(int ci=0;ci<allDegrees.length;ci++)
202                                                {
203                                                        final String[] degree = allDegrees[ci];
204                                                        if(pStats.getTitles().contains(degree[DEG_TITLE]))
205                                                        {
206                                                                if(!this.myClasses.contains(degree))
207                                                                {
208                                                                        while(CMParms.numContains(pStats.getTitles(), degree[DEG_TITLE]) > 1)
209                                                                                pStats.getTitles().remove(degree[DEG_TITLE]);
210                                                                }
211                                                        }
212                                                        else
213                                                        if(this.myClasses.contains(degree))
214                                                                pStats.getTitles().add(degree[DEG_TITLE]);
215                                                }
216                                        }
217                                        this.lastTitle="";
218                                }
219                                final PlayerStats pStats = mob.playerStats();
220                                if((pStats!=null)
221                                &&(lastTitle != pStats.getActiveTitle()))
222                                {
223                                        synchronized(this)
224                                        {
225                                                this.lastTitle=pStats.getActiveTitle();
226                                                this.elligibleC = null;
227                                                if((this.lastTitle!=null)
228                                                &&(myTitles.contains(this.lastTitle)))
229                                                {
230                                                        for(final String[] degree : this.myClasses)
231                                                        {
232                                                                if(degree[DEG_TITLE].equals(this.lastTitle))
233                                                                {
234                                                                        this.elligibleC = CMClass.getCharClass(degree[DEG_CID]);
235                                                                        this.helpProficiency(mob,0);
236                                                                        break;
237                                                                }
238                                                        }
239                                                }
240                                        }
241                                }
242                        }
243                }
244                return super.tick(ticking,tickID);
245        }

247        @Override
248        public void affectCharStats(final MOB affectedMob, final CharStats affectableStats)
249        {
250                super.affectCharStats(affectedMob, affectableStats);
251                final CharClass C;
252                synchronized(this)
253                {
254                        C=this.activatedC;
255                }
256                if((C != null)&&(affectableStats.getCurrentClass()!=C))
257                {
258                        final int level = affectableStats.getCurrentClassLevel();
259                        affectableStats.setCurrentClass(C);
260                        affectableStats.setCurrentClassLevel(level);
261                }
262        }
263
264        private void activateDegree(final CMMsg msg)
265        {
266                synchronized(this)
267                {
268                        if(this.elligibleC != null)
269                        {
270                                if(!super.proficiencyCheck(msg.source(), 0, false))
271                                        return;
272                                
273                                this.activatedC = this.elligibleC;
274                                msg.source().recoverCharStats();
275                        }
276                        else
277                                return;
278                }
279                final Skill_HonoraryDegreeCommoner me = this;
280                final MOB mob=msg.source();
281                msg.addTrailerRunnable(new Runnable()
282                {
283                        @Override
284                        public void run()
285                        {
286                                synchronized(me)
287                                {
288                                        me.activatedC = null;
289                                        mob.recoverCharStats();
290                                }
291                        }
292                });
293        }
294
295        @Override
296        public boolean okMessage(final Environmental myHost, final CMMsg msg)
297        {
298                if(!(affected instanceof MOB))
299                        return true;
300
301                final MOB mob=(MOB)affected;
302                if(msg.amISource(mob))
303                {
304                        switch(msg.sourceMinor())
305                        {
306                        case CMMsg.TYP_WEAR:
307                        case CMMsg.TYP_WIELD:
308                        case CMMsg.TYP_GET:
309                        case CMMsg.TYP_HOLD:
310                        case CMMsg.TYP_TRAVEL:
311                        case CMMsg.TYP_ENTER:
312                        case CMMsg.TYP_LEAVE:
313                                this.activateDegree(msg);
314                                break;
315                        default:
316                                if(this.activatedC != null)
317                                {
318                                        synchronized(this)
319                                        {
320                                                this.activatedC = null;
321                                                mob.recoverCharStats();
322                                        }
323                                }
324                        }
325                }
326                return true;
327        }
