037{

096        private final String socialName="PRAY <T-NAME>";
097        private final int numToDo=10;
098        private final Set<MOB> doneMobs = new HashSet<MOB>();
099        private final Set<Room> botheredRooms = new HashSet<Room>();
100        
101        private static String[] FRIENDLY_SOCIALS = 
102                {"ANGELIC","APOLOGIZE","AWE","BEAM","BEARHUG","BKISS","BLUSH","BOW","BSCRATCH","CHEER","CLAP",
103                "COMFORT","CRY","CUDDLE","DANCE","EMBRACE","GIRN","GREET","GRIN","GROVEL","HANDSHAKE","HIGHFIVE",
104                "HSHAKE","HUG","KISS","LOVE","MASSAGE","NOD","NUZZLE","PAT","PATPAT","PET","PRAY","PURR","SALUTE",
105                "SERENADE","SMILE","SMOOCH","SNUGGLE","SPOON","SSMILE","STROKE","SUPPORT","SWEET","THANK","TICKLE","TOAST"};

107        @Override
108        public void unInvoke()
109        {
110                if(!(affected instanceof MOB))
111                        return;
112                // undo the affects of this spell
113                final MOB mob=(MOB)affected;
114                final MOB invoker=this.invoker();
115                super.unInvoke();
116                if((canBeUninvoked()&&(!mob.amDead())))
117                {
118                        if(doneMobs.size()>=numToDo)
119                                mob.location().show(mob,null,CMMsg.MSG_OK_VISUAL,L("<S-NAME> <S-HAS-HAVE> survived the judgement."));
120                        else
121                        if(super.tickDown < 2)
122                        {
123                                mob.location().show(mob,null,CMMsg.MSG_OK_VISUAL,L("<S-NAME> <S-IS-ARE> consumed by <S-HIS-HER> judgement!"));
124                                CMLib.combat().postDeath(invoker(), mob, null);
125                        }
126                        if(mob.isMonster() && (!mob.amDead()))
127                        {
128                                CMLib.commands().postStand(mob,true, false);
129                                CMLib.tracking().wanderAway(mob,false,true);
130                                if((invoker!=null)&&(invoker!=mob)&&(invoker.location()==mob.location()))
131                                        CMLib.combat().postAttack(mob, invoker, mob.fetchWieldedItem());
132                        }
133                        botheredRooms.clear();
134                        doneMobs.clear();
135                }
136        }
137        
138        @Override
139        public boolean tick(final Tickable ticking, final int tickID)
140        {
141                if(!super.tick(ticking, tickID))
142                        return false;
143                final Physical affected=this.affected;
144                if((affected instanceof MOB)&&(((MOB)affected).isMonster()))
145                {
146                        final MOB mob=(MOB)affected;
147                        final Room R=mob.location();
148                        if(mob.isInCombat())
149                                CMLib.combat().postPanic(mob, null);
150                        else
151                        if(!CMLib.flags().canMove(mob))
152                                unInvoke();
153                        else
154                        if(R!=null)
155                        {
156                                MOB doMOB=null;
157                                for(final Enumeration<MOB> m=R.inhabitants();m.hasMoreElements();)
158                                {
159                                        final MOB M=m.nextElement();
160                                        if((!doneMobs.contains(M)) && (M!=mob) && CMLib.flags().canBeSeenBy(M,mob))
161                                        {
162                                                doMOB=M;
163                                                break;
164                                        }
165                                }
166                                if(doMOB==null)
167                                {
168                                        if(!botheredRooms.contains(R))
169                                                botheredRooms.add(R);
170                                        CMLib.commands().postStand(mob,true, false);
171                                        if(!CMLib.tracking().beMobile(mob,true,true,false,true,null,botheredRooms))
172                                                CMLib.tracking().beMobile(mob,true,true,false,false,null,null);
173                                }
174                                else
175                                {
176                                        Social S=CMLib.socials().fetchSocial(socialName, true);
177                                        if(S==null)
178                                                unInvoke();
179                                        else
180                                                S.invoke(mob, new XVector<String>(S.baseName(),R.getContextName(doMOB)), doMOB, false);
181                                }
182                                        
183                        }
184                }
185                return true;
186        }
187        
188        @Override
189        public void executeMsg(final Environmental myHost, final CMMsg msg)
190        {
191                super.executeMsg(myHost, msg);
192                if((msg.source()==affected) 
193                && (msg.tool() instanceof Social)
194                && (msg.target() instanceof MOB)
195                && (msg.tool().Name().equals(socialName))
196                && (msg.source() != msg.target())
197                && (!doneMobs.contains(msg.target())))
198                {
199                        doneMobs.add((MOB)msg.target());
200                        if(doneMobs.size()>=numToDo)
201                                unInvoke();
202                        else
203                                msg.addTrailerMsg(CMClass.getMsg(msg.source(),null,null,CMMsg.MSG_OK_ACTION,L("^HThat`s @x1/@x2.",""+doneMobs.size(),""+numToDo),CMMsg.NO_EFFECT,null,CMMsg.NO_EFFECT,null));
204                }
205        }
206
207        @Override
208        public boolean invoke(MOB mob, List<String> commands, Physical givenTarget, boolean auto, int asLevel)
209        {
210                final MOB target=this.getTarget(mob,commands,givenTarget);
211                if(target==null)
212                        return false;
213
214                if(!super.invoke(mob,commands,givenTarget,auto,asLevel))
215                        return false;
216
217                boolean success=proficiencyCheck(mob,0,auto);
218
219                if(success && (!CMLib.flags().isGood(target)))
220                {
221                        final CMMsg msg=CMClass.getMsg(mob,target,this,verbalCastCode(mob,target,auto)|CMMsg.MASK_MALICIOUS,auto?L("<T-NAME> gain(s) an holy geas of judgement!"):L("^S<S-NAME> judge(s) <T-NAMESELF>, laying a geas upon <T-HIM-HER>.^?"));
222                        if(mob.location().okMessage(mob,msg))
223                        {
224                                mob.location().send(mob,msg);
225                                if(msg.value()<=0)
226                                {
227                                        final Ability A=maliciousAffect(mob,target,asLevel,0,-1);
228                                        if(A!=null)
229                                        {
230                                                int num=10 + super.getX1Level(mob) + super.getXLEVELLevel(mob);
231                                                String socialName=null;
232                                                String socialBaseName=null;
233                                                int tries=100;
234                                                while( ((--tries)>0) && (socialName==null))
235                                                {
236                                                        String name=FRIENDLY_SOCIALS[CMLib.dice().roll(1, FRIENDLY_SOCIALS.length, -1)];
237                                                        Social S=CMLib.socials().fetchSocial(name+" <T-NAME>", true);
238                                                        if(S!=null)
239                                                        {
240                                                                socialName=S.Name();
241                                                                socialBaseName=S.baseName().toLowerCase();
242                                                        }
243                                                }
244                                                if(socialName == null)
245                                                        A.unInvoke();
246                                                else
247                                                {
248                                                        A.setMiscText("NUM="+num+" SOCIAL=\""+socialName+"\"");
249                                                        target.tell(L("You have been placed under a powerful geas which you must fulfill before this spell expires, or you will die.  You must @x1 at least @x2 unique creatures to survive. BEGIN!",socialBaseName,""+num));
250                                                }
251                                        }
252                                        else
253                                                success=false;
254                                        target.recoverPhyStats();
255                                }
256                        }
257                }
258                else
259                        return maliciousFizzle(mob,target,L("<S-NAME> attempt(s) to curse <T-NAMESELF>, but nothing happens."));
260
261                // return whether it worked
262                return success;
263        }
