036public class Prayer_Fertilize extends Prayer
037{
038        @Override
039        public String ID()
040        {
041                return "Prayer_Fertilize";
042        }
043
044        private final static String localizedName = CMLib.lang().L("Fertilize");
045
046        @Override
047        public String name()
048        {
049                return localizedName;
050        }
051
052        @Override
053        protected int canTargetCode()
054        {
055                return 0;
056        }
057
058        @Override
059        protected int canAffectCode()
060        {
061                return Ability.CAN_ROOMS;
062        }
063
064        @Override
065        public int classificationCode()
066        {
067                return Ability.ACODE_PRAYER|Ability.DOMAIN_CREATION;
068        }
069
070        @Override
071        public long flags()
072        {
073                return Ability.FLAG_NEUTRAL;
074        }
075
076        @Override
077        public int abstractQuality()
078        {
079                return Ability.QUALITY_INDIFFERENT;
080        }
081
082        private volatile int oldResource = -1;
083        
084        @Override
085        public void unInvoke()
086        {
087                if(this.canBeUninvoked() && (affected instanceof Room))
088                {
089                        final Room R=(Room)affected;
090                        if((R!=null)&&(oldResource>0))
091                                R.setResource(oldResource);
092                }
093                super.unInvoke();
094        }
095
096        @Override
097        public boolean tick(final Tickable ticking, final int tickID)
098        {
099                if((affected!=null)&&(affected instanceof Room))
100                {
101                        final Room R=(Room)affected;
102                        if((R.myResource()&RawMaterial.MATERIAL_MASK)==RawMaterial.MATERIAL_VEGETATION)
103                        {
104                                oldResource=R.myResource();
105                                for(int m=0;m<R.numInhabitants();m++)
106                                {
107                                        final MOB M=R.fetchInhabitant(m);
108                                        if(M!=null)
109                                        {
110                                                Ability A=M.fetchEffect("Farming");
111                                                if(A==null)
112                                                        A=M.fetchEffect("Foraging");
113                                                if(A==null)
114                                                        A=M.fetchEffect("MasterFarming");
115                                                if(A==null)
116                                                        A=M.fetchEffect("MasterForaging");
117                                                if(A==null)
118                                                        A=M.fetchEffect("MasterGardening");
119                                                if(A==null)
120                                                        A=M.fetchEffect("Gardening");
121                                                if(A!=null)
122                                                        A.setAbilityCode(1);
123                                        }
124                                }
125                        }
126                }
127                return super.tick(ticking,tickID);
128
129        }
130
131        @Override
132        public boolean invoke(MOB mob, List<String> commands, Physical givenTarget, boolean auto, int asLevel)
133        {
134
135                final int type=mob.location().domainType();
136                if(((type&Room.INDOORS)>0)
137                        ||(type==Room.DOMAIN_OUTDOORS_AIR)
138                        ||(type==Room.DOMAIN_OUTDOORS_CITY)
139                        ||(type==Room.DOMAIN_OUTDOORS_SPACEPORT)
140                        ||(type==Room.DOMAIN_OUTDOORS_UNDERWATER)
141                        ||(type==Room.DOMAIN_OUTDOORS_WATERSURFACE))
142                {
143                        mob.tell(L("That magic won't work here."));
144                        return false;
145                }
146                if(!super.invoke(mob,commands,givenTarget,auto,asLevel))
147                        return false;
148
149                final boolean success=proficiencyCheck(mob,0,auto);
150
151                if(success)
152                {
153                        final CMMsg msg=CMClass.getMsg(mob,mob.location(),this,verbalCastCode(mob,mob.location(),auto),auto?"":L("^S<S-NAME> @x1 to make the land fruitful.^?",prayForWord(mob)));
154                        if(mob.location().okMessage(mob,msg))
155                        {
156                                mob.location().send(mob,msg);
157                                this.oldResource=mob.location().myResource();
158                                if(beneficialAffect( mob,
159                                                                  mob.location(),
160                                                                  asLevel,
161                                                                  CMLib.ableMapper().qualifyingClassLevel( mob, this ) *
162                                                                          (int)( ( CMProps.getMillisPerMudHour() *
163                                                                                          (mob.location().getArea().getTimeObj().getHoursInDay()) ) /
164                                                                                          CMProps.getTickMillis() ) )!=null)
165                                {
166                                        // the chant should be better than the prayer, so leave this part out -- 
167                                        // but keep the functionality around just in case we want it.
168                                        //mob.location().setResource(RawMaterial.RESOURCE_DIRT);
169                                }
170                        }
171
172                }
173                else
174                        beneficialWordsFizzle(mob,null,L("<S-NAME> @x1 to make the land fruitful, but nothing happens.",prayForWord(mob)));
175
176                // return whether it worked
177                return success;
178        }
179}
