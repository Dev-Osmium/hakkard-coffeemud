035public class Prayer_Avatar extends Prayer
036{
037        @Override
038        public String ID()
039        {
040                return "Prayer_Avatar";
041        }
042
043        private final static String     localizedName   = CMLib.lang().L("Avatar");
044
045        @Override
046        public String name()
047        {
048                return localizedName;
049        }
050
051        @Override
052        public int classificationCode()
053        {
054                return Ability.ACODE_PRAYER | Ability.DOMAIN_COMMUNING;
055        }
056
057        @Override
058        public int abstractQuality()
059        {
060                return Ability.QUALITY_BENEFICIAL_SELF;
061        }
062
063        @Override
064        public long flags()
065        {
066                return Ability.FLAG_UNHOLY;
067        }
068
069        @Override
070        public String displayText()
071        {
072                if ((invoker() != null) && (invoker().getWorshipCharID().length() > 0))
073                        return "(You are the AVATAR of " + invoker().getWorshipCharID() + ")";
074                return "(You are the AVATAR of the gods)";
075        }
076
077        @Override
078        protected int canAffectCode()
079        {
080                return Ability.CAN_MOBS;
081        }
082
083        @Override
084        protected int canTargetCode()
085        {
086                return 0;
087        }
088
089        @Override
090        public void unInvoke()
091        {
092                // undo the affects of this spell
093                if(!(affected instanceof MOB))
094                        return;
095                final MOB mob=(MOB)affected;
096
097                super.unInvoke();
098
099                if(canBeUninvoked())
100                        mob.tell(L("Your unholy alliance has been severed."));
101        }
102
103        @Override
104        public void affectCharState(MOB affectedMOB, CharState affectedState)
105        {
106                super.affectCharState(affectedMOB,affectedState);
107                affectedState.setHitPoints(affectedState.getHitPoints()+200);
108        }
109
110        @Override
111        public void affectPhyStats(Physical affected, PhyStats affectedStats)
112        {
113                super.affectPhyStats(affected,affectedStats);
114                final int xlvl=2+(int)Math.round(CMath.div(adjustedLevel(invoker(),0),1.5));
115                affectedStats.setArmor(affectedStats.armor()-(xlvl));
116                affectedStats.setSpeed(affectedStats.speed()+1.0+CMath.mul(0.33,super.getXLEVELLevel(invoker())));
117                affectedStats.setAttackAdjustment(affectedStats.attackAdjustment()+(xlvl*2));
118                if(affected instanceof MOB)
119                {
120                        final MOB mob=(MOB)affected;
121                        if(mob.getMyDeity()!=null)
122                                affectedStats.setName(L("@x1, the Avatar of @x2",mob.name(),mob.getMyDeity().name()));
123                        else
124                                affectedStats.setName(L("@x1, the Avatar",mob.name()));
125                }
126        }
127
128        @Override
129        public boolean tick(final Tickable ticking, final int tickID)
130        {
131                if(!(affected instanceof MOB))
132                        return super.tick(ticking,tickID);
133
134                if(!super.tick(ticking,tickID))
135                        return false;
136                final MOB mob=(MOB)affected;
137                if(mob.location()!=null)
138                {
139                        if(mob.isInCombat())
140                        {
141                                final MOB newvictim=mob.location().fetchRandomInhabitant();
142                                if(newvictim!=mob)
143                                        mob.setVictim(newvictim);
144                        }
145                        else
146                        {
147                                MOB attack=null;
148                                final Room R=mob.location();
149                                for(int m=0;m<R.numInhabitants();m++)
150                                {
151                                        final MOB M=R.fetchInhabitant(m);
152                                        if((M!=null)&&(M!=mob)&&(mob.mayPhysicallyAttack(M)))
153                                        {
154                                                attack = M;
155                                                break;
156                                        }
157                                }
158                                if(attack==null)
159                                {
160                                        int dir=-1;
161                                        final Vector<Integer> dirs=new Vector<Integer>();
162
163                                        for(int d=Directions.NUM_DIRECTIONS()-1;d>=0;d--)
164                                        {
165                                                final Room R2=R.getRoomInDir(d);
166                                                if((R2!=null)
167                                                &&(R.getExitInDir(d)!=null)
168                                                &&(R.getExitInDir(d).isOpen()))
169                                                        dirs.addElement(Integer.valueOf(d));
170                                        }
171                                        while(dirs.size()>0)
172                                        {
173                                                final int d=dirs.remove(CMLib.dice().roll(1, dirs.size(), -1)).intValue();
174                                                final Room R2=R.getRoomInDir(d);
175                                                if(R2!=null)
176                                                {
177                                                        if((dir<0)||(dir==Directions.UP))
178                                                                dir=d;
179                                                        for(int m=0;m<R2.numInhabitants();m++)
180                                                        {
181                                                                final MOB M=R2.fetchInhabitant(m);
182                                                                if((M!=null)&&(M!=mob)&&(mob.mayPhysicallyAttack(M)))
183                                                                {
184                                                                        attack = M;
185                                                                        break;
186                                                                }
187                                                        }
188                                                }
189                                        }
190                                        if(dir>=0)
191                                        {
192                                                final String godName=mob.getWorshipCharID().length()==0?"Your god":mob.getWorshipCharID();
193                                                mob.tell(L("@x1 directs you @x2.",godName,CMLib.directions().getInDirectionName(dir)));
194                                                CMLib.tracking().walk(mob,dir,false,false);
195                                        }
196                                }
197                                if(attack!=null)
198                                        CMLib.combat().postAttack(mob,attack,mob.fetchWieldedItem());
199                        }
200                }
201                return super.tick(ticking,tickID);
202        }
203
204        @Override
205        public boolean invoke(MOB mob, List<String> commands, Physical givenTarget, boolean auto, int asLevel)
206        {
207                MOB target=mob;
208                if((auto)&&(givenTarget!=null)&&(givenTarget instanceof MOB))
209                        target=(MOB)givenTarget;
210
211                if(target.fetchEffect(ID())!=null)
212                {
213                        mob.tell(target,null,null,L("<S-NAME> <S-IS-ARE> already the AVATAR."));
214                        return false;
215                }
216
217                int levels=mob.charStats().getClassLevel("Avatar");
218                if(levels<0)
219                        levels=mob.phyStats().level();
220                else
221                if(!mob.charStats().getCurrentClass().ID().equals("Avatar"))
222                {
223                        mob.tell(L("You have lost this ability for all time."));
224                        return false;
225                }
226                else
227                        levels=adjustedLevel(mob,asLevel);
228
229                if(!super.invoke(mob,commands,givenTarget,auto,asLevel))
230                        return false;
231
232                final boolean success=proficiencyCheck(mob,0,auto);
233
234                if(success)
235                {
236                        final CMMsg msg=CMClass.getMsg(mob,target,this,verbalCastCode(mob,target,auto),auto?"":L("^S<S-NAME> @x1.^?",prayWord(mob)));
237                        if(mob.location().okMessage(mob,msg))
238                        {
239                                mob.location().send(mob,msg);
240                                target.location().show(target,null,CMMsg.MSG_OK_VISUAL,L("<S-NAME> become(s) the AVATAR!"));
241                                beneficialAffect(mob,target,asLevel,levels);
242                        }
243                }
244                else
245                        return beneficialWordsFizzle(mob,target,L("<S-NAME> @x1, but nothing happens.",prayWord(mob)));
246
247                // return whether it worked
248                return success;
249        }
