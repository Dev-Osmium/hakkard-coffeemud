036public class Prayer_PeaceRitual extends Prayer
037{
038        @Override
039        public String ID()
040        {
041                return "Prayer_PeaceRitual";
042        }
043
044        private final static String localizedName = CMLib.lang().L("Peace Ritual");
045
046        @Override
047        public String name()
048        {
049                return localizedName;
050        }
051
052        private final static String localizedStaticDisplay = CMLib.lang().L("(Peace Ritual)");
053
054        @Override
055        public String displayText()
056        {
057                return localizedStaticDisplay;
058        }
059
060        @Override
061        public int classificationCode()
062        {
063                return Ability.ACODE_PRAYER|Ability.DOMAIN_NEUTRALIZATION;
064        }
065
066        @Override
067        public int abstractQuality()
068        {
069                return Ability.QUALITY_OK_SELF;
070        }
071
072        @Override
073        public long flags()
074        {
075                return Ability.FLAG_HOLY;
076        }
077
078        @Override
079        protected int canAffectCode()
080        {
081                return 0;
082        }
083
084        @Override
085        protected int canTargetCode()
086        {
087                return 0;
088        }
089
090        public Clan clan1=null;
091        public Clan clan2=null;
092        public Iterable<Pair<Clan,Integer>> clan2Set=null;
093
094        @Override
095        public boolean tick(final Tickable ticking, final int tickID)
096        {
097                if(!(affected instanceof MOB))
098                        return false;
099
100                if(invoker==null)
101                        return false;
102
103                final MOB mob=(MOB)affected;
104                if(clan2Set==null)
105                {
106                        if(clan2==null)
107                                return super.tick(ticking, tickID);
108                        final Vector<Pair<Clan,Integer>> V=new Vector<Pair<Clan,Integer>>();
109                        V.add(new Pair<Clan,Integer>(clan2,Integer.valueOf(clan2.getGovernment().getAcceptPos())));
110                        clan2Set=V;
111                }
112                final List<String> channels=CMLib.channels().getFlaggedChannelNames(ChannelsLibrary.ChannelFlag.CLANINFO);
113                for(int i=0;i<channels.size();i++)
114                        CMLib.commands().postChannel(channels.get(i),clan2Set,L("@x1 located in '@x2' is performing a peace ritual on behalf of @x3.",mob.name(),mob.location().displayText(mob),clan1.name()),false);
115                return super.tick(ticking,tickID);
116        }
117
118        @Override
119        public boolean okMessage(final Environmental myHost, final CMMsg msg)
120        {
121                if(!super.okMessage(myHost,msg))
122                        return false;
123                if(affected==null)
124                        return true;
125                if(!(affected instanceof MOB))
126                        return true;
127
128                if((msg.target()==affected)
129                &&(msg.source()!=affected)
130                &&(CMath.bset(msg.targetMajor(),CMMsg.MASK_MALICIOUS)))
131                {
132                        msg.source().location().show((MOB)affected,null,CMMsg.MSG_OK_VISUAL,L("The peace ritual is disrupted!"));
133                        clan1=null;
134                        clan2=null;
135                        unInvoke();
136                }
137                else
138                if(msg.amISource((MOB)affected)
139                &&((msg.targetMinor()==CMMsg.TYP_ENTER)||(msg.targetMinor()==CMMsg.TYP_LEAVE)))
140                {
141                        msg.source().location().show((MOB)affected,null,CMMsg.MSG_OK_VISUAL,L("The peace ritual is disrupted!"));
142                        clan1=null;
143                        clan2=null;
144                        unInvoke();
145                }
146                return true;
147        }
148
149        @Override
150        public void unInvoke()
151        {
152                // undo the affects of this spell
153                if(!(affected instanceof MOB))
154                        return;
155                super.unInvoke();
156
157                if((canBeUninvoked())&&(clan1!=null)&&(clan2!=null))
158                {
159                        final Clan C1=clan1;
160                        final Clan C2=clan2;
161                        if((C1!=null)&&(C2!=null))
162                        {
163                                if(C1.getClanRelations(C2.clanID())==Clan.REL_WAR)
164                                {
165                                        C1.setClanRelations(C2.clanID(),Clan.REL_HOSTILE,System.currentTimeMillis());
166                                        C1.update();
167                                }
168                                if(C2.getClanRelations(C1.clanID())==Clan.REL_WAR)
169                                {
170                                        C2.setClanRelations(C1.clanID(),Clan.REL_HOSTILE,System.currentTimeMillis());
171                                        C2.update();
172                                }
173                                final List<String> channels=CMLib.channels().getFlaggedChannelNames(ChannelsLibrary.ChannelFlag.CLANINFO);
174                                for(int i=0;i<channels.size();i++)
175                                        CMLib.commands().postChannel(channels.get(i),CMLib.clans().clanRoles(),L("There is now peace between @x1 and @x2.",C1.name(),C2.name()),false);
176                        }
177                }
178        }
179
180        @Override
181        public boolean invoke(MOB mob, List<String> commands, Physical givenTarget, boolean auto, int asLevel)
182        {
183                Physical target=mob;
184                if((auto)&&(givenTarget!=null))
185                        target=givenTarget;
186                if(target.fetchEffect(this.ID())!=null)
187                {
188                        mob.tell(mob,target,null,L("<T-NAME> <T-IS-ARE> already affected by @x1.",name()));
189                        return false;
190                }
191                clan1=CMLib.clans().findRivalrousClan(mob);
192                if(clan1==null)
193                {
194                        mob.tell(L("You must belong to a clan to use this prayer."));
195                        return false;
196                }
197                if(commands.size()<1)
198                {
199                        mob.tell(L("You must specify the clan you wish to see peace with."));
200                        return false;
201                }
202                final String clan2Name=CMParms.combine(commands,0);
203                clan2=CMLib.clans().findClan(clan2Name);
204                if((clan2==null)
205                ||((clan1.getClanRelations(clan2.clanID())!=Clan.REL_WAR)&&(clan2.getClanRelations(clan1.clanID())!=Clan.REL_WAR)))
206                {
207                        mob.tell(L("Your @x1 is not at war with @x2!",clan1.getGovernmentName(),clan2.name()));
208                        return false;
209                }
210                boolean found=false;
211                for(final Enumeration<MOB> e=CMLib.players().players();e.hasMoreElements();)
212                {
213                        final MOB M=e.nextElement();
214                        if(M.getClanRole(clan2.clanID())!=null)
215                        {
216                                found = true;
217                                break;
218                        }
219                }
220                if(!found)
221                {
222                        mob.tell(L("You must wait until a member of @x1 is online before beginning the ritual.",clan2.name()));
223                        return false;
224                }
225
226                if(!super.invoke(mob,commands,givenTarget,auto,asLevel))
227                        return false;
228
229                final boolean success=proficiencyCheck(mob,0,auto);
230
231                if(success)
232                {
233                        final CMMsg msg=CMClass.getMsg(mob,target,this,verbalCastCode(mob,target,auto),auto?L("<T-NAME> begin(s) a peace ritual."):L("^S<S-NAME> @x1 for peace between @x2 and @x3.^?",prayWord(mob),clan1.name(),clan2.name()));
234                        if(mob.location().okMessage(mob,msg))
235                        {
236                                mob.location().send(mob,msg);
237                                beneficialAffect(mob,target,asLevel,(int)CMProps.getTicksPerMinute()*5);
238                        }
239                }
240                else
241                        return beneficialWordsFizzle(mob,null,L("<S-NAME> @x1 for peace between @x2 and @x3, but there is no answer.",prayWord(mob),clan1.name(),clan2.name()));
242
243                // return whether it worked
244                return success;
245        }
