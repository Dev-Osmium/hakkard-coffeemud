036public class Prayer_Restoration extends Prayer implements MendingSkill
037{
038        @Override
039        public String ID()
040        {
041                return "Prayer_Restoration";
042        }
043
044        private final static String localizedName = CMLib.lang().L("Restoration");
045
046        @Override
047        public String name()
048        {
049                return localizedName;
050        }
051
052        @Override
053        public int abstractQuality()
054        {
055                return Ability.QUALITY_BENEFICIAL_OTHERS;
056        }
057
058        @Override
059        public int classificationCode()
060        {
061                return Ability.ACODE_PRAYER|Ability.DOMAIN_HEALING;
062        }
063
064        @Override
065        public long flags()
066        {
067                return Ability.FLAG_HOLY|Ability.FLAG_HEALINGMAGIC;
068        }
069
070        @Override
071        protected int overrideMana()
072        {
073                return Ability.COST_ALL;
074        }
075
076        @Override
077        public boolean supportsMending(Physical item)
078        {
079                if(!(item instanceof MOB))
080                        return false;
081
082                if(((((MOB)item).curState()).getHitPoints()<(((MOB)item).maxState()).getHitPoints()))
083                        return true;
084                final MOB caster=CMClass.getFactoryMOB();
085                caster.basePhyStats().setLevel(CMProps.getIntVar(CMProps.Int.LASTPLAYERLEVEL));
086                caster.phyStats().setLevel(CMProps.getIntVar(CMProps.Int.LASTPLAYERLEVEL));
087                if(
088                        (item.fetchEffect("Amputation")!=null)
089                ||(item.fetchEffect("Injury")!=null)
090                ||(item.fetchEffect("BrokenLimbs")!=null)
091                ||(item.fetchEffect("Fighter_AtemiStrike")!=null)
092                ||(item.fetchEffect("Undead_EnergyDrain")!=null)
093                ||(item.fetchEffect("Undead_WeakEnergyDrain")!=null)
094                ||(item.fetchEffect("Undead_ColdTouch")!=null)
095                ||((new Prayer_RestoreSmell().returnOffensiveAffects(caster,item)).size()>0)
096                ||((new Prayer_RestoreVoice().returnOffensiveAffects(caster,item)).size()>0)
097                ||((Prayer_RemovePoison.returnOffensiveAffects(item)).size()>0)
098                ||((new Prayer_Freedom().returnOffensiveAffects(caster,item)).size()>0)
099                ||((new Prayer_CureBlindness().returnOffensiveAffects(caster,item)).size()>0)
100                ||((new Prayer_CureDeafness().returnOffensiveAffects(caster,item)).size()>0)
101                )
102                {
103                        caster.destroy();
104                        return true;
105                }
106                caster.destroy();
107                return false;
108        }
109
110        @Override
111        public int castingQuality(MOB mob, Physical target)
112        {
113                if(mob!=null)
114                {
115                        if(target instanceof MOB)
116                        {
117                                if(!supportsMending(target))
118                                        return Ability.QUALITY_INDIFFERENT;
119                        }
120                }
121                return super.castingQuality(mob,target);
122        }
123
124        @Override
125        public boolean invoke(MOB mob, List<String> commands, Physical givenTarget, boolean auto, int asLevel)
126        {
127                final MOB target=this.getTarget(mob,commands,givenTarget);
128                if(target==null)
129                        return false;
130
131                if(!super.invoke(mob,commands,givenTarget,auto,asLevel))
132                        return false;
133
134                final boolean success=proficiencyCheck(mob,0,auto);
135
136                if(success)
137                {
138                        final CMMsg msg=CMClass.getMsg(mob,target,this,verbalCastCode(mob,target,auto),auto?L("<T-NAME> become(s) surrounded by a bright light."):L("^S<S-NAME> @x1 over <T-NAMESELF> for restorative healing.^?",prayWord(mob)));
139                        if(mob.location().okMessage(mob,msg))
140                        {
141                                mob.location().send(mob,msg);
142                                final int healing=target.maxState().getHitPoints()-target.curState().getHitPoints();
143                                if(healing>0)
144                                {
145                                        CMLib.combat().postHealing(mob,target,this,healing,CMMsg.MASK_ALWAYS|CMMsg.TYP_CAST_SPELL,null);
146                                        mob.location().show(target,null,CMMsg.MSG_OK_VISUAL,L("<S-NAME> look(s) much healthier!"));
147                                        target.recoverCharStats();
148                                        target.recoverPhyStats();
149                                        target.recoverMaxState();
150                                }
151                                Ability A;
152                                A=target.fetchEffect("Bleeding");
153                                if(A!=null)
154                                {
155                                        target.delEffect(A);
156                                        mob.location().show(target,null,CMMsg.MSG_OK_VISUAL,L("<S-YOUPOSS> bleeding stops!"));
157                                        A=target.fetchAbility(A.ID());
158                                        if(A!=null)
159                                                target.delAbility(A);
160                                        target.recoverCharStats();
161                                        target.recoverPhyStats();
162                                        target.recoverMaxState();
163                                }
164
165                                A=target.fetchEffect("Amputation");
166                                if(A!=null)
167                                {
168                                        target.delEffect(A);
169                                        mob.location().show(target,null,CMMsg.MSG_OK_VISUAL,L("<S-YOUPOSS> missing parts are restored!"));
170                                        A=target.fetchAbility(A.ID());
171                                        if(A!=null)
172                                                target.delAbility(A);
173                                        target.recoverCharStats();
174                                        target.recoverPhyStats();
175                                        target.recoverMaxState();
176                                }
177
178                                A=target.fetchEffect("BrokenLimbs");
179                                if(A!=null)
180                                {
181                                        target.delEffect(A);
182                                        mob.location().show(target,null,CMMsg.MSG_OK_VISUAL,L("<S-YOUPOSS> broken limbs mend!"));
183                                        A=target.fetchAbility(A.ID());
184                                        if(A!=null)
185                                                target.delAbility(A);
186                                        target.recoverCharStats();
187                                        target.recoverPhyStats();
188                                        target.recoverMaxState();
189                                }
190
191                                A=target.fetchEffect("Injury");
192                                if(A!=null)
193                                {
194                                        target.delEffect(A);
195                                        mob.location().show(target,null,CMMsg.MSG_OK_VISUAL,L("<S-YOUPOSS> injuries are healed!"));
196                                        A=target.fetchAbility(A.ID());
197                                        if(A!=null)
198                                                target.delAbility(A);
199                                        target.recoverCharStats();
200                                        target.recoverPhyStats();
201                                        target.recoverMaxState();
202                                }
203
204                                A=target.fetchEffect("Fighter_AtemiStrike");
205                                if((A!=null)&&(A.canBeUninvoked()))
206                                {
207                                        target.delEffect(A);
208                                        mob.location().show(target,null,CMMsg.MSG_OK_VISUAL,L("<S-YOUPOSS> atemi damage is healed!"));
209                                        target.recoverCharStats();
210                                        target.recoverPhyStats();
211                                        target.recoverMaxState();
212                                }
213
214                                A=target.fetchEffect("Undead_EnergyDrain");
215                                if(A!=null)
216                                {
217                                        A.unInvoke();
218                                        target.delEffect(A);
219                                        mob.location().show(target,null,CMMsg.MSG_OK_VISUAL,L("<S-YOUPOSS> lost levels are restored!"));
220                                        target.recoverCharStats();
221                                        target.recoverPhyStats();
222                                        target.recoverMaxState();
223                                }
224                                A=target.fetchEffect("Undead_WeakEnergyDrain");
225                                if(A!=null)
226                                {
227                                        A.unInvoke();
228                                        target.delEffect(A);
229                                        mob.location().show(target,null,CMMsg.MSG_OK_VISUAL,L("<S-YOUPOSS> lost levels are restored!"));
230                                        target.recoverCharStats();
231                                        target.recoverPhyStats();
232                                        target.recoverMaxState();
233                                }
234                                A=target.fetchEffect("Undead_ColdTouch");
235                                if(A!=null)
236                                {
237                                        A.unInvoke();
238                                        target.delEffect(A);
239                                        mob.location().show(target,null,CMMsg.MSG_OK_VISUAL,L("<S-NAME> <S-IS-ARE> no longer cold and weak!"));
240                                        target.recoverCharStats();
241                                        target.recoverPhyStats();
242                                        target.recoverMaxState();
243                                }
244                                List<Ability> offensiveAffects=new Prayer_RestoreSmell().returnOffensiveAffects(mob,target);
245                                if(offensiveAffects.size()>0)
246                                {
247                                        for(int a=offensiveAffects.size()-1;a>=0;a--)
248                                                offensiveAffects.get(a).unInvoke();
249                                        mob.location().showOthers(target,null,CMMsg.MSG_OK_VISUAL,L("<S-NAME> can smell again!"));
250                                        target.recoverCharStats();
251                                        target.recoverPhyStats();
252                                        target.recoverMaxState();
253                                }
254                                offensiveAffects=new Prayer_RestoreVoice().returnOffensiveAffects(mob,target);
255                                if(offensiveAffects.size()>0)
256                                {
257                                        for(int a=offensiveAffects.size()-1;a>=0;a--)
258                                                offensiveAffects.get(a).unInvoke();
259                                        mob.location().showOthers(target,null,CMMsg.MSG_OK_VISUAL,L("<S-NAME> can speak again!"));
260                                        target.recoverCharStats();
261                                        target.recoverPhyStats();
262                                        target.recoverMaxState();
263                                }
264                                offensiveAffects=Prayer_RemovePoison.returnOffensiveAffects(target);
265                                if(offensiveAffects.size()>0)
266                                {
267                                        for(int a=offensiveAffects.size()-1;a>=0;a--)
268                                                offensiveAffects.get(a).unInvoke();
269                                        mob.location().showOthers(target,null,CMMsg.MSG_OK_VISUAL,L("<S-NAME> is cured of <S-HIS-HER> poisonous afflication!"));
270                                        target.recoverCharStats();
271                                        target.recoverPhyStats();
272                                        target.recoverMaxState();
273                                }
274                                offensiveAffects=new Prayer_Freedom().returnOffensiveAffects(mob,target);
275                                if(offensiveAffects.size()>0)
276                                {
277                                        for(int a=offensiveAffects.size()-1;a>=0;a--)
278                                                offensiveAffects.get(a).unInvoke();
279                                        mob.location().showOthers(target,null,CMMsg.MSG_OK_VISUAL,L("<S-NAME> can move again!"));
280                                        target.recoverCharStats();
281                                        target.recoverPhyStats();
282                                        target.recoverMaxState();
283                                }
284                                offensiveAffects=new Prayer_CureDisease().returnOffensiveAffects(target);
285                                if(offensiveAffects.size()>0)
286                                {
287                                        for(int a=offensiveAffects.size()-1;a>=0;a--)
288                                                offensiveAffects.get(a).unInvoke();
289                                        mob.location().showOthers(target,null,CMMsg.MSG_OK_VISUAL,L("<S-NAME> is cured of <S-HIS-HER> disease!"));
290                                        target.recoverCharStats();
291                                        target.recoverPhyStats();
292                                        target.recoverMaxState();
293                                }
294                                offensiveAffects=new Prayer_CureBlindness().returnOffensiveAffects(mob,target);
295                                if(offensiveAffects.size()>0)
296                                {
297                                        for(int a=offensiveAffects.size()-1;a>=0;a--)
298                                                offensiveAffects.get(a).unInvoke();
299                                        mob.location().showOthers(target,null,CMMsg.MSG_OK_VISUAL,L("<S-NAME> can see again!"));
300                                        target.recoverCharStats();
301                                        target.recoverPhyStats();
302                                        target.recoverMaxState();
303                                }
304                                offensiveAffects=new Prayer_CureDeafness().returnOffensiveAffects(mob,target);
305                                if(offensiveAffects.size()>0)
306                                {
307                                        for(int a=offensiveAffects.size()-1;a>=0;a--)
308                                                offensiveAffects.get(a).unInvoke();
309                                        mob.location().showOthers(target,null,CMMsg.MSG_OK_VISUAL,L("<S-NAME> can hear again!"));
310                                        target.recoverCharStats();
311                                        target.recoverPhyStats();
312                                        target.recoverMaxState();
313                                }
314                                mob.location().recoverRoomStats();
315                        }
316                }
317                else
318                        beneficialWordsFizzle(mob,target,L("<S-NAME> @x1 over <T-NAMESELF>, but @x2 does not heed.",prayWord(mob),hisHerDiety(mob)));
319
320                // return whether it worked
321                return success;
322        }
