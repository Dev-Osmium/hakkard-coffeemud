036public class Prayer_FlameWeapon extends Prayer
037{
038        @Override
039        public String ID()
040        {
041                return "Prayer_FlameWeapon";
042        }
043
044        private final static String localizedName = CMLib.lang().L("Flame Weapon");
045
046        @Override
047        public String name()
048        {
049                return localizedName;
050        }
051
052        private final static String localizedStaticDisplay = CMLib.lang().L("(Enflamed)");
053
054        @Override
055        public String displayText()
056        {
057                return localizedStaticDisplay;
058        }
059
060        @Override
061        public int classificationCode()
062        {
063                return Ability.ACODE_PRAYER|Ability.DOMAIN_CREATION;
064        }
065
066        @Override
067        public int abstractQuality()
068        {
069                return Ability.QUALITY_INDIFFERENT;
070        }
071
072        @Override
073        protected int canAffectCode()
074        {
075                return CAN_ITEMS;
076        }
077
078        @Override
079        protected int canTargetCode()
080        {
081                return CAN_ITEMS;
082        }
083
084        @Override
085        public long flags()
086        {
087                return Ability.FLAG_UNHOLY|Ability.FLAG_HEATING|Ability.FLAG_FIREBASED;
088        }
089
090        protected boolean notAgain=false;
091
092        @Override
093        public void affectPhyStats(final Physical affected, final PhyStats affectableStats)
094        {
095                super.affectPhyStats(affected,affectableStats);
096                if(affected==null)
097                        return;
098                affectableStats.setDisposition(affectableStats.disposition()|PhyStats.IS_BONUS);
099                affectableStats.setDisposition(affectableStats.disposition()|PhyStats.IS_GLOWING);
100                affectableStats.setDisposition(affectableStats.disposition()|PhyStats.IS_LIGHTSOURCE);
101                if(affected instanceof Item)
102                        affectableStats.setAbility(affectableStats.ability()+1);
103        }
104
105        @Override
106        public void executeMsg(final Environmental myHost, final CMMsg msg)
107        {
108                super.executeMsg(myHost,msg);
109
110                if((msg.source().location()!=null)
111                &&(msg.targetMinor()==CMMsg.TYP_DAMAGE)
112                &&((msg.value())>0)
113                &&(msg.tool()==affected)
114                &&(!notAgain)
115                &&(msg.target() instanceof MOB)
116                &&(!((MOB)msg.target()).amDead()))
117                {
118                        try
119                        {
120                                notAgain=true;
121                                final CMMsg msg2=CMClass.getMsg(msg.source(),msg.target(),affected,CMMsg.MSG_OK_ACTION,CMMsg.MSK_MALICIOUS_MOVE|CMMsg.TYP_FIRE,CMMsg.MSG_NOISYMOVEMENT,null);
122                                if(msg.source().location().okMessage(msg.source(),msg2))
123                                {
124                                        msg.source().location().send(msg.source(), msg2);
125                                        if(msg2.value()<=0)
126                                        {
127                                                int flameDamage = (int) Math.round( Math.random() * 6 );
128                                                flameDamage *= (super.getXLEVELLevel(invoker())+(super.getX1Level(invoker())));
129                                                msg.addTrailerMsg(CMClass.getMsg(msg.source(),msg.target(),CMMsg.MSG_OK_ACTION,L("^RThe flame around @x1 @x2 <T-NAME>!^?",affected.name(),CMLib.combat().standardHitWord(Weapon.TYPE_BURNING,flameDamage))));
130                                                final CMMsg msg3=CMClass.getMsg(msg.source(),msg.target(),null,CMMsg.MASK_ALWAYS|CMMsg.TYP_FIRE,CMMsg.MSG_DAMAGE,CMMsg.NO_EFFECT,null);
131                                                msg3.setValue(flameDamage);
132                                                msg.addTrailerMsg(msg3);
133                                        }
134                                }
135                        }finally{notAgain=false;}
136                }
137        }
138
139        @Override
140        public void unInvoke()
141        {
142                Item destroyMe=null;
143                // undo the affects of this spell
144                if(canBeUninvoked())
145                {
146                        if((affected!=null)&&(affected instanceof Item))
147                        {
148                                if(((((Weapon)affected).material()&RawMaterial.MATERIAL_MASK)==RawMaterial.MATERIAL_WOODEN)
149                                ||((((Weapon)affected).material()&RawMaterial.MATERIAL_MASK)==RawMaterial.MATERIAL_VEGETATION))
150                                {
151                                        if((((Item)affected).owner()!=null)&&(((Item)affected).owner() instanceof MOB))
152                                                ((MOB)((Item)affected).owner()).tell(L("The flames around @x1 consume it.",((Item)affected).name()));
153                                        destroyMe=(Item)affected;
154                                }
155                                else
156                                if((((Item)affected).owner()!=null)&&(((Item)affected).owner() instanceof MOB))
157                                        ((MOB)((Item)affected).owner()).tell(L("The flames around @x1 fade.",((Item)affected).name()));
158                        }
159                }
160                super.unInvoke();
161                if(destroyMe!=null)
162                        destroyMe.destroy();
163        }
164
165        @Override
166        public int castingQuality(MOB mob, Physical target)
167        {
168                if(mob!=null)
169                {
170                        if((mob.fetchWieldedItem() instanceof Weapon)
171                        &&(mob.fetchWieldedItem().fetchEffect(ID())==null))
172                                return super.castingQuality(mob, target,Ability.QUALITY_BENEFICIAL_SELF);
173                }
174                return super.castingQuality(mob,target);
175        }
176
177        @Override
178        public boolean invoke(MOB mob, List<String> commands, Physical givenTarget, boolean auto, int asLevel)
179        {
180                Item target=getTarget(mob,mob.location(),givenTarget,commands,Wearable.FILTER_ANY);
181                if(target==null)
182                        return false;
183                if(!(target instanceof Weapon))
184                {
185                        if(auto||mob.isMonster())
186                        {
187                                target=mob.fetchWieldedItem();
188                                if(target==null)
189                                {
190                                        for(int i=0;i<mob.location().numItems();i++)
191                                        {
192                                                final Item I2=mob.location().getItem(i);
193                                                if((I2!=null)&&(I2.container()==null)&&(I2 instanceof Weapon))
194                                                {
195                                                        target=I2;
196                                                        break;
197                                                }
198                                        }
199                                }
200                        }
201                        if(!(target instanceof Weapon))
202                        {
203                                mob.tell(L("You can only enflame weapons."));
204                                return false;
205                        }
206                }
207
208                if(((Weapon)target).fetchEffect(this.ID())!=null)
209                {
210                        mob.tell(L("@x1 is already enflamed.",target.name(mob)));
211                        return false;
212                }
213                if(!super.invoke(mob,commands,givenTarget,auto,asLevel))
214                        return false;
215
216                final boolean success=proficiencyCheck(mob,0,auto);
217
218                if(success)
219                {
220                        final CMMsg msg=CMClass.getMsg(mob,target,this,verbalCastCode(mob,target,auto),auto?L("<T-NAME> appear(s) surrounded by flames!"):L("^S<S-NAME> hold(s) <T-NAMESELF> and @x1.^?",prayWord(mob)));
221                        if(mob.location().okMessage(mob,msg))
222                        {
223                                mob.location().send(mob,msg);
224                                beneficialAffect(mob,target,asLevel,0);
225                                mob.location().show(mob,target,CMMsg.MSG_OK_VISUAL,("<T-NAME> is engulfed in flames!")+CMLib.protocol().msp("fireball.wav",10));
226                                target.recoverPhyStats();
227                                mob.recoverPhyStats();
228                        }
229                }
230                else
231                        return beneficialWordsFizzle(mob,target,L("<S-NAME> hold(s) <T-NAMESELF> and @x1, but nothing happens.",prayWord(mob)));
232                // return whether it worked
233                return success;
234        }
