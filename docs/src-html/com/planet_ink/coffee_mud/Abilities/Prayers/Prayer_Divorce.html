035public class Prayer_Divorce extends Prayer
036{
037        @Override
038        public String ID()
039        {
040                return "Prayer_Divorce";
041        }
042
043        private final static String     localizedName   = CMLib.lang().L("Divorce");
044
045        @Override
046        public String name()
047        {
048                return localizedName;
049        }
050
051        @Override
052        public int classificationCode()
053        {
054                return Ability.ACODE_PRAYER | Ability.DOMAIN_CORRUPTION;
055        }
056
057        @Override
058        public long flags()
059        {
060                return Ability.FLAG_UNHOLY;
061        }
062
063        @Override
064        public int abstractQuality()
065        {
066                return Ability.QUALITY_OK_OTHERS;
067        }
068
069        @Override
070        public boolean invoke(MOB mob, List<String> commands, Physical givenTarget, boolean auto, int asLevel)
071        {
072                final MOB target=getTarget(mob,commands,givenTarget);
073                if(target==null)
074                        return false;
075                if(!target.isMarriedToLiege())
076                {
077                        mob.tell(L("@x1 is not married!",target.name(mob)));
078                        return false;
079                }
080                if(target.fetchItem(null,Wearable.FILTER_WORNONLY,"wedding band")!=null)
081                {
082                        mob.tell(L("@x1 must remove the wedding band first.",target.name(mob)));
083                        return false;
084                }
085
086                if(!super.invoke(mob,commands,givenTarget,auto,asLevel))
087                        return false;
088
089                final boolean success=proficiencyCheck(mob,0,auto);
090                if(success)
091                {
092                        final CMMsg msg=CMClass.getMsg(mob,target,this,verbalCastCode(mob,target,auto),auto?"":L("^S<S-NAME> divorce(s) <T-NAMESELF> from @x1.^?",target.getLiegeID()));
093                        if(mob.location().okMessage(mob,msg))
094                        {
095                                if((!target.isMonster())&&(target.soulMate()==null))
096                                        CMLib.coffeeTables().bump(target,CoffeeTableRow.STAT_DIVORCES);
097                                mob.location().send(mob,msg);
098                                String maleName=target.Name();
099                                String femaleName=target.getLiegeID();
100                                if(target.charStats().getStat(CharStats.STAT_GENDER)=='F')
101                                {
102                                        femaleName=target.Name();
103                                        maleName=target.getLiegeID();
104                                }
105                                final List<String> channels=CMLib.channels().getFlaggedChannelNames(ChannelsLibrary.ChannelFlag.DIVORCES);
106                                for(int i=0;i<channels.size();i++)
107                                        CMLib.commands().postChannel(channels.get(i),mob.clans(),L("@x1 and @x2 are now divorced.",maleName,femaleName),true);
108                                final MOB M=CMLib.players().getPlayerAllHosts(target.getLiegeID());
109                                if(M!=null)
110                                        M.setLiegeID("");
111                                target.setLiegeID("");
112                                try
113                                {
114                                        for(final Enumeration<Room> e=CMLib.map().rooms();e.hasMoreElements();)
115                                        {
116                                                final Room R=e.nextElement();
117                                                final LandTitle T=CMLib.law().getLandTitle(R);
118                                                if((T!=null)&&(T.getOwnerName().equals(maleName)))
119                                                {
120                                                        T.setOwnerName(femaleName);
121                                                        CMLib.database().DBUpdateRoom(R);
122                                                }
123                                                for(int i=0;i<R.numInhabitants();i++)
124                                                {
125                                                        final MOB M2=R.fetchInhabitant(i);
126                                                        if((M2!=null)&&(M2 instanceof Banker))
127                                                        {
128                                                                final Banker B=(Banker)M2;
129                                                                final List<Item> allMaleItems=B.getDepositedItems(maleName);
130                                                                Item coins=B.findDepositInventory(femaleName,""+Integer.MAX_VALUE);
131                                                                for(int v=0;v<allMaleItems.size();v++)
132                                                                {
133                                                                        final Item I=allMaleItems.get(v);
134                                                                        if((I!=null)&&(I.container()==null))
135                                                                        {
136                                                                                final List<Item> items=B.delDepositInventory(maleName,I);
137                                                                                if(I instanceof Coins)
138                                                                                {
139                                                                                        if(coins!=null)
140                                                                                                B.delDepositInventory(femaleName,coins);
141                                                                                        else
142                                                                                        {
143                                                                                                coins=CMClass.getItem("StdCoins");
144                                                                                                ((Coins)coins).setNumberOfCoins(0);
145                                                                                        }
146                                                                                        ((Coins)coins).setNumberOfCoins(((Coins)coins).getNumberOfCoins() + Math.round(((Coins)I).getTotalValue() / ((Coins)coins).getDenomination()));
147                                                                                        B.addDepositInventory(femaleName,coins,null);
148                                                                                }
149                                                                                else
150                                                                                for(Item oI : items)
151                                                                                        B.addDepositInventory(femaleName,oI,oI.container());
152                                                                        }
153                                                                }
154                                                                for(int v=0;v<allMaleItems.size();v++)
155                                                                {
156                                                                        final Item I=allMaleItems.get(v);
157                                                                        if(I!=null)
158                                                                                I.destroy();
159                                                                }
160                                                        }
161                                                }
162                                        }
163                                }
164                                catch (final NoSuchElementException e)
165                                {
166                                }
167                        }
168                }
169                else
170                        beneficialWordsFizzle(mob,target,L("<S-NAME> clear(s) <S-HIS-HER> throat."));
171
172                return success;
173        }
174}
