036public class Prayer_Contagion extends Prayer implements DiseaseAffect
037{
038        @Override
039        public String ID()
040        {
041                return "Prayer_Contagion";
042        }
043
044        private final static String localizedStaticDisplay = CMLib.lang().L("(Contagion)");
045
046        @Override
047        public String displayText()
048        {
049                return localizedStaticDisplay;
050        }
051
052        private final static String localizedName = CMLib.lang().L("Contagion");
053
054        @Override
055        public String name()
056        {
057                return localizedName;
058        }
059
060        @Override
061        public int classificationCode()
062        {
063                return Ability.ACODE_PRAYER|Ability.DOMAIN_CORRUPTION;
064        }
065
066        @Override
067        public int abstractQuality()
068        {
069                return Ability.QUALITY_MALICIOUS;
070        }
071
072        @Override
073        public long flags()
074        {
075                return Ability.FLAG_NEUTRAL;
076        }
077
078        @Override
079        protected int canAffectCode()
080        {
081                return CAN_MOBS;
082        }
083
084        @Override
085        protected int canTargetCode()
086        {
087                return CAN_MOBS;
088        }
089
090        @Override
091        public int difficultyLevel()
092        {
093                return 0;
094        }
095
096        @Override
097        public boolean isMalicious()
098        {
099                return true;
100        }
101
102        @Override
103        public String getHealthConditionDesc()
104        {
105                return ""; // not really a health condition, more of a mystical one
106        }
107
108        @Override
109        public void unInvoke()
110        {
111                if(affected==null)
112                        return;
113                final MOB mob=(MOB)affected;
114                if(canBeUninvoked())
115                {
116                        if(!mob.amDead())
117                                spreadImmunity(mob);
118                        mob.tell(L("The contagion fades."));
119                }
120                super.unInvoke();
121        }
122
123        @Override
124        public int spreadBitmap()
125        {
126                return DiseaseAffect.SPREAD_PROXIMITY;
127        }
128
129        @Override
130        public boolean tick(final Tickable ticking, final int tickID)
131        {
132                if(!super.tick(ticking,tickID))
133                        return false;
134
135                if(affected==null)
136                        return false;
137                if(!(affected instanceof MOB))
138                        return false;
139                final MOB mob=(MOB)affected;
140                if(mob.location().numInhabitants()==1)
141                        return true;
142                final Vector<Ability> choices=new Vector<Ability>();
143                for(final Enumeration<Ability> a=mob.effects();a.hasMoreElements();)
144                {
145                        final Ability A=a.nextElement();
146                        if((A!=null)
147                           &&(A.canBeUninvoked())
148                           &&(!A.ID().equals(ID()))
149                           &&(A.abstractQuality()==Ability.QUALITY_MALICIOUS)
150                           &&(((A.classificationCode()&Ability.ALL_ACODES)==Ability.ACODE_SPELL)
151                                  ||((A.classificationCode()&Ability.ALL_ACODES)==Ability.ACODE_PRAYER))
152                           &&(!A.isAutoInvoked()))
153                                choices.addElement(A);
154                }
155                if(choices.size()==0)
156                        return true;
157                final MOB target=mob.location().fetchRandomInhabitant();
158                final Ability thisOne=choices.elementAt(CMLib.dice().roll(1,choices.size(),-1));
159                if((target==null)||(thisOne==null)||(target.fetchEffect(ID())!=null))
160                        return true;
161                if(CMLib.dice().rollPercentage()>(target.charStats().getSave(CharStats.STAT_SAVE_DISEASE)))
162                {
163                        ((Ability)this.copyOf()).invoke(target,target,true,0);
164                        if(target.fetchEffect(ID())!=null)
165                                ((Ability)thisOne.copyOf()).invoke(target,target,true,0);
166                }
167                else
168                        spreadImmunity(target);
169                return true;
170        }
171
172        @Override
173        public boolean invoke(MOB mob, List<String> commands, Physical givenTarget, boolean auto, int asLevel)
174        {
175                final MOB target=this.getTarget(mob,commands,givenTarget);
176                if(target==null)
177                        return false;
178
179                if(!super.invoke(mob,commands,givenTarget,auto,asLevel))
180                        return false;
181
182                boolean success=proficiencyCheck(mob,0,auto);
183
184                if(success)
185                {
186
187                        final CMMsg msg=CMClass.getMsg(mob,target,this,verbalCastCode(mob,target,auto)|CMMsg.MASK_MALICIOUS,auto?L("<T-NAME> become(s) contagious!"):L("^S<S-NAME> @x1 for a contagion to inflict <T-NAMESELF>.^?",prayWord(mob)));
188                        final CMMsg msg2=CMClass.getMsg(mob,target,this,CMMsg.TYP_DISEASE|CMMsg.MASK_MALICIOUS,null);
189                        if((mob.location().okMessage(mob,msg))&&(mob.location().okMessage(mob,msg2)))
190                        {
191                                mob.location().send(mob,msg);
192                                mob.location().send(mob,msg2);
193                                if((msg.value()<=0)&&(msg2.value()<=0))
194                                        success=maliciousAffect(mob,target,asLevel,0,-1)!=null;
195                                else
196                                        spreadImmunity(target);
197                        }
198                }
199                else
200                        return maliciousFizzle(mob,target,L("<S-NAME> point(s) at <T-NAMESELF> and @x1, but nothing happens.",prayWord(mob)));
201
202                // return whether it worked
203                return success;
204        }
