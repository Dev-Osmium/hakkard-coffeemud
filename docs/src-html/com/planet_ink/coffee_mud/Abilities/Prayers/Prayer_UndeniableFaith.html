036public class Prayer_UndeniableFaith extends Prayer
037{
038        @Override
039        public String ID()
040        {
041                return "Prayer_UndeniableFaith";
042        }
043
044        private final static String localizedName = CMLib.lang().L("Undeniable Faith");
045
046        @Override
047        public String name()
048        {
049                return localizedName;
050        }
051
052        @Override
053        public int classificationCode()
054        {
055                return Ability.ACODE_PRAYER|Ability.DOMAIN_EVANGELISM;
056        }
057
058        @Override
059        public int abstractQuality()
060        {
061                return Ability.QUALITY_OK_OTHERS;
062        }
063
064        @Override
065        protected int canAffectCode()
066        {
067                return CAN_MOBS;
068        }
069
070        @Override
071        protected int canTargetCode()
072        {
073                return CAN_MOBS;
074        }
075
076        @Override
077        public long flags()
078        {
079                return Ability.FLAG_NEUTRAL|Ability.FLAG_CHARMING|Ability.FLAG_MINDALTERING;
080        }
081
082        @Override
083        protected int overrideMana()
084        {
085                return 100;
086        }
087
088        protected String godName="";
089        private static DVector convertStack=new DVector(2);
090
091        @Override
092        public void unInvoke()
093        {
094                final MOB M=(MOB)affected;
095                super.unInvoke();
096                if(canBeUninvoked())
097                        M.tell(L("Your compelled faith is finally subsided."));
098        }
099
100        @Override
101        public boolean tick(final Tickable ticking, final int tickID)
102        {
103                if(!super.tick(ticking,tickID))
104                        return false;
105                if(!(affected instanceof MOB))
106                        return true;
107                final MOB M=(MOB)affected;
108                if(M.location()!=null)
109                {
110                        if((!M.getWorshipCharID().equals(godName))
111                        &&(godName.length()>0))
112                        {
113                                final Deity D=CMLib.map().getDeity(godName);
114                                if(M.getWorshipCharID().length()>0)
115                                {
116                                        final Deity D2=CMLib.map().getDeity(M.getWorshipCharID());
117                                        if(D2!=null)
118                                        {
119                                                final CMMsg msg2=CMClass.getMsg(M,D2,this,CMMsg.MSG_REBUKE,null);
120                                                if(M.location().okMessage(M,msg2))
121                                                        M.location().send(M,msg2);
122                                        }
123                                }
124                                final CMMsg msg2=CMClass.getMsg(M,D,this,CMMsg.MSG_SERVE,null);
125                                if(M.location().okMessage(M,msg2))
126                                {
127                                        M.location().send(M,msg2);
128                                        M.setWorshipCharID(godName);
129                                }
130                        }
131                }
132                return true;
133        }
134
135        @Override
136        public boolean okMessage(final Environmental host, final CMMsg msg)
137        {
138                if((affected instanceof MOB)
139                &&(msg.amISource((MOB)affected))
140                &&(msg.sourceMinor()==CMMsg.TYP_REBUKE)
141                &&(msg.target()!=null)
142                &&((msg.target()==invoker())||(msg.target().Name().equals(godName))))
143                {
144                        msg.source().tell(L("Your faith is too undeniable."));
145                        return false;
146                }
147                return super.okMessage(host,msg);
148        }
149
150        @Override
151        public boolean invoke(final MOB mob, final List<String> commands, final Physical givenTarget, final boolean auto, final int asLevel)
152        {
153                final MOB target=this.getTarget(mob,commands,givenTarget);
154                if(target==null)
155                        return false;
156                if((mob.getWorshipCharID().length()==0)
157                ||(CMLib.map().getDeity(mob.getWorshipCharID())==null))
158                {
159                        if(!auto)
160                                mob.tell(L("You must worship a god to use this prayer."));
161                        return false;
162                }
163                final Deity D=CMLib.map().getDeity(mob.getWorshipCharID());
164                if((target.getWorshipCharID().length()>0)
165                &&(CMLib.map().getDeity(target.getWorshipCharID())!=null))
166                {
167                        if(!auto)
168                                mob.tell(L("@x1 worships @x2, and may not be converted with this prayer.",target.name(mob),target.getWorshipCharID()));
169                        return false;
170                }
171                if((CMLib.flags().isAnimalIntelligence(target)||CMLib.flags().isGolem(target)||(D==null)))
172                {
173                        if(!auto)
174                                mob.tell(L("@x1 can not be converted with this prayer.",target.name(mob)));
175                        return false;
176                }
177                if(!auto)
178                {
179                        if(convertStack.contains(target))
180                        {
181                                final Long L=(Long)convertStack.elementAt(convertStack.indexOf(target),2);
182                                if((System.currentTimeMillis()-L.longValue())>CMProps.getMillisPerMudHour()*5)
183                                        convertStack.removeElement(target);
184                        }
185                        if(convertStack.contains(target))
186                        {
187                                mob.tell(L("@x1 must wait to be undeniably faithful again.",target.name(mob)));
188                                return false;
189                        }
190                }
191
192                if(!super.invoke(mob,commands,givenTarget,auto,asLevel))
193                        return false;
194
195                int levelDiff=target.phyStats().level()-(mob.phyStats().level()+(2*getXLEVELLevel(mob)));
196                if(levelDiff<0)
197                        levelDiff=0;
198                final boolean success=proficiencyCheck(mob,-(levelDiff*25),auto);
199                int type=verbalCastCode(mob,target,auto);
200                int mal=CMMsg.MASK_MALICIOUS;
201                if(auto)
202                {
203                        type=CMath.unsetb(type,CMMsg.MASK_MALICIOUS);
204                        mal=0;
205                }
206                if(success)
207                {
208                        final CMMsg msg=CMClass.getMsg(mob,target,this,type,auto?"":L("^S<S-NAME> @x1 for <T-NAMESELF> to BELIEVE!^?",prayWord(mob)));
209                        final CMMsg msg2=CMClass.getMsg(target,D,this,CMMsg.MSG_SERVE,L("<S-NAME> BELIEVE(S) !!!"));
210                        final CMMsg msg3=CMClass.getMsg(mob,target,this,CMMsg.MSK_CAST_VERBAL|mal|CMMsg.TYP_MIND|(auto?CMMsg.MASK_ALWAYS:0),null);
211                        if((mob.location().okMessage(mob,msg))
212                        &&(mob.location().okMessage(mob,msg3))
213                        &&(mob.location().okMessage(mob,msg2)))
214                        {
215                                mob.location().send(mob,msg);
216                                mob.location().send(mob,msg3);
217                                if((msg.value()<=0)&&(msg3.value()<=0))
218                                {
219                                        target.location().send(target,msg2);
220                                        target.setWorshipCharID(godName);
221                                        if(mob!=target)
222                                                CMLib.leveler().postExperience(mob,target,null,25,false);
223                                        godName=mob.getWorshipCharID();
224                                        beneficialAffect(mob,target,asLevel,CMProps.getIntVar(CMProps.Int.TICKSPERMUDMONTH));
225                                        convertStack.addElement(target,Long.valueOf(System.currentTimeMillis()));
226                                }
227                        }
228                }
229                else
230                        beneficialWordsFizzle(mob,target,auto?"":L("<S-NAME> @x1 for <T-NAMESELF>, but nothing happens.",prayWord(mob)));
231
232                // return whether it worked
233                return success;
234        }
