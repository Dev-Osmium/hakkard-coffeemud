102        private final static int TYP_ICE=0;
103        private final static int TYP_FIRE=1;
104        private final static int TYP_EARTH=2;
105        private final static int TYP_AIR=3;
106
107        protected int wallType=0;
108        protected int amountRemaining=0;
109        protected Item theWall=null;
110
111        @Override
112        public boolean okMessage(final Environmental myHost, final CMMsg msg)
113        {
114                if((affected==null)||(!(affected instanceof Item)))
115                        return true;
116
117                final MOB mob=msg.source();
118                switch(wallType)
119                {
120                case TYP_ICE:
121                        if((invoker!=null)
122                        &&(mob.isInCombat())
123                        &&(mob.getVictim()==invoker)
124                        &&(mob.rangeToTarget()==1))
125                        {
126                                if(msg.sourceMinor()==CMMsg.TYP_ADVANCE)
127                                {
128                                        Item w=mob.fetchWieldedItem();
129                                        if(w==null)
130                                                w=mob.getNaturalWeapon();
131                                        if(w==null)
132                                                return false;
133                                        final Room room=mob.location();
134                                        final CMMsg msg2=CMClass.getMsg(mob,null,CMMsg.MSG_WEAPONATTACK,L("^F^<FIGHT^><S-NAME> hack(s) at the monolith of ice with @x1.^</FIGHT^>^?",w.name()));
135                                        CMLib.color().fixSourceFightColor(msg2);
136                                        if(room.okMessage(mob,msg2))
137                                        {
138                                                room.send(mob,msg2);
139                                                amountRemaining-=mob.phyStats().damage();
140                                                if(amountRemaining<0)
141                                                {
142                                                        for(int i=0;i<room.numInhabitants();i++)
143                                                        {
144                                                                final MOB M=room.fetchInhabitant(i);
145                                                                if((M.isInCombat())
146                                                                &&(M.getVictim()==invoker)
147                                                                &&(M.rangeToTarget()>0)
148                                                                &&(M.rangeToTarget()<3)
149                                                                &&(!M.amDead()))
150                                                                {
151                                                                        final MOB invoker=(invoker()!=null) ? invoker() : M;
152                                                                        CMLib.combat().postDamage(invoker,M,this,CMLib.dice().roll(M.phyStats().level()/2,4,0),CMMsg.MASK_ALWAYS|CMMsg.MASK_MALICIOUS|CMMsg.TYP_COLD,Weapon.TYPE_PIERCING,L("A shard of ice <DAMAGE> <T-NAME>!"));
153                                                                }
154                                                        }
155                                                        mob.location().showHappens(CMMsg.MSG_OK_ACTION,L("The monolith of ice shatters!!!"));
156                                                        ((Item)affected).destroy();
157                                                }
158                                        }
159                                        return false;
160                                }
161                        }
162                        break;
163                case TYP_FIRE:
164                        break;
165                case TYP_AIR:
166                        if((invoker!=null)
167                        &&(mob.isInCombat())
168                        &&(mob.getVictim()==invoker)
169                        &&(mob.rangeToTarget()>=1)
170                        &&(msg.amITarget(invoker))
171                        &&(msg.targetMinor()==CMMsg.TYP_WEAPONATTACK)
172                        &&(msg.tool() instanceof Weapon)
173                        &&(!((Weapon)msg.tool()).amWearingAt(Wearable.IN_INVENTORY))
174                        &&(((Weapon)msg.tool()).weaponClassification()==Weapon.CLASS_RANGED))
175                        {
176                                mob.location().show(mob,invoker,CMMsg.MSG_OK_VISUAL,L("<S-NAME> fire(s) @x1 at <T-NAME>.  The missile enters the monolith of air.",msg.tool().name()));
177                                final MOB M=CMClass.getFactoryMOB();
178                                M.setLocation(mob.location());
179                                M.setName(L("The monolith of air"));
180                                M.setVictim(mob);
181                                M.setRangeToTarget(mob.rangeToTarget());
182                                CMLib.combat().postWeaponAttackResult(M,mob,(Weapon)msg.tool(),true);
183                                M.setLocation(null);
184                                M.setVictim(null);
185                                if(mob.isMonster())
186                                        CMLib.commands().postRemove(mob,(Weapon)msg.tool(),true);
187                                M.destroy();
188                                return false;
189                        }
190                        break;
191                case TYP_EARTH:
192                        if((invoker!=null)
193                        &&(mob.isInCombat())
194                        &&(mob.getVictim()==invoker)
195                        &&(mob.rangeToTarget()==1))
196                        {
197                                if(msg.sourceMinor()==CMMsg.TYP_ADVANCE)
198                                {
199                                        Item w=mob.fetchWieldedItem();
200                                        if(w==null)
201                                                w=mob.getNaturalWeapon();
202                                        if(w==null)
203                                                return false;
204                                        if(mob.location().show(mob,null,w,CMMsg.MSG_WEAPONATTACK,L("^F^<FIGHT^><S-NAME> hack(s) at the monolith of stone with <O-NAME>.^</FIGHT^>^?")))
205                                        {
206                                                amountRemaining-=mob.phyStats().damage();
207                                                if(amountRemaining<0)
208                                                {
209                                                        mob.location().showHappens(CMMsg.MSG_OK_ACTION,L("The monolith of stone is destroyed!"));
210                                                        ((Item)affected).destroy();
211                                                }
212                                        }
213                                        return false;
214                                }
215                        }
216                        break;
217                }
218                return super.okMessage(myHost,msg);
219        }

221        @Override
222        public void unInvoke()
223        {
224                super.unInvoke();
225                if(canBeUninvoked())
226                {
227                        if((theWall!=null)
228                        &&(invoker!=null)
229                        &&(theWall.owner()!=null)
230                        &&(theWall.owner() instanceof Room)
231                        &&(((Room)theWall.owner()).isContent(theWall)))
232                        {
233                                MOB actorM=invoker; if(actorM==null) actorM=CMLib.map().deity();
234                                switch(wallType)
235                                {
236                                case TYP_FIRE:
237                                        ((Room)theWall.owner()).show(actorM,null,CMMsg.MSG_OK_VISUAL,L("The monolith of fire fades."));
238                                        break;
239                                case TYP_AIR:
240                                        ((Room)theWall.owner()).show(actorM,null,CMMsg.MSG_OK_VISUAL,L("The monolith of air dissipates."));
241                                        break;
242                                case TYP_ICE:
243                                        ((Room)theWall.owner()).show(actorM,null,CMMsg.MSG_OK_VISUAL,L("The monolith of ice melts."));
244                                        break;
245                                case TYP_EARTH:
246                                        ((Room)theWall.owner()).show(actorM,null,CMMsg.MSG_OK_VISUAL,L("The monolith of stone crumbles."));
247                                        break;
248                                }
249                                final Item wall=theWall;
250                                theWall=null;
251                                wall.destroy();
252                        }
253                }
254        }

256        @Override
257        public boolean tick(final Tickable ticking, final int tickID)
258        {
259                if(tickID==Tickable.TICKID_MOB)
260                {
261                        switch(wallType)
262                        {
263                        case TYP_ICE:
264                        case TYP_EARTH:
265                        case TYP_AIR:
266                                if((invoker!=null)
267                                   &&(theWall!=null)
268                                   &&(invoker.location()!=null)
269                                   &&(!invoker.location().isContent(theWall)))
270                                        unInvoke();
271                                break;
272                        case TYP_FIRE:
273                                if((invoker!=null)
274                                   &&(theWall!=null)
275                                   &&(invoker.location()!=null))
276                                {
277                                        final Room room=invoker.location();
278                                        if(!invoker.location().isContent(theWall))
279                                                unInvoke();
280                                        else
281                                        for(int m=0;m<room.numInhabitants();m++)
282                                        {
283                                                final MOB mob=room.fetchInhabitant(m);
284                                                if((mob!=null)
285                                                &&(mob!=invoker)
286                                                &&(mob.isInCombat())
287                                                &&(mob.getVictim()==invoker)
288                                                &&(mob.rangeToTarget()==1))
289                                                {
290                                                        final int damage = CMLib.dice().roll((int)Math.round((adjustedLevel(invoker,0)+(2.0*super.getX1Level(invoker())))/4.0),6,1);
291                                                        CMLib.combat().postDamage(invoker,mob,this,damage,CMMsg.MASK_MALICIOUS|CMMsg.MASK_ALWAYS|CMMsg.TYP_FIRE,Weapon.TYPE_BURNING,L("The monolith of fire flares and <DAMAGE> <T-NAME>!"));
292                                                }
293                                        }
294                                }
295                                break;
296                        }
297                }
298                return super.tick(ticking,tickID);
299        }

301        @Override
302        public boolean invoke(MOB mob, List<String> commands, Physical givenTarget, boolean auto, int asLevel)
303        {
304                if((!mob.isInCombat())||(mob.rangeToTarget()<1))
305                {
306                        mob.tell(L("You really should be in ranged combat to cast this."));
307                        return false;
308                }
309                for(int i=0;i<mob.location().numItems();i++)
310                {
311                        final Item I=mob.location().getItem(i);
312                        if((I!=null)&&(I.fetchEffect(ID())!=null))
313                        {
314                                mob.tell(L("There is already a monolith here."));
315                                return false;
316                        }
317                }
318
319                if(!super.invoke(mob,commands,givenTarget,auto,asLevel))
320                        return false;
321
322                final Physical target = mob.location();
323
324                final boolean success=proficiencyCheck(mob,0,auto);
325
326                if(success)
327                {
328
329                        wallType=CMLib.dice().roll(1,4,-1);
330                        final String text=text().toUpperCase().trim();
331                        if((text.indexOf("STONE")>=0)||(text.indexOf("EARTH")>=0))
332                                wallType=TYP_EARTH;
333                        else
334                        if((text.indexOf("ICE")>=0)||(text.indexOf("WATER")>=0))
335                                wallType=TYP_ICE;
336                        else
337                        if(text.indexOf("AIR")>=0)
338                                wallType=TYP_AIR;
339                        else
340                        if(text.indexOf("FIRE")>=0)
341                                wallType=TYP_FIRE;
342                        Item I=null;
343                        switch(wallType)
344                        {
345                        case TYP_EARTH:
346                                amountRemaining=mob.baseState().getHitPoints()/6;
347                                I=CMClass.getItem("GenItem");
348                                I.setName(L("a monolith of stone"));
349                                I.setDisplayText(L("a monolith of stone has been erected here"));
350                                I.setDescription(L("The bricks are sold and sturdy."));
351                                I.setMaterial(RawMaterial.RESOURCE_STONE);
352                                break;
353                        case TYP_ICE:
354                                amountRemaining=20;
355                                I=CMClass.getItem("GenItem");
356                                I.setName(L("a monolith of ice"));
357                                I.setDisplayText(L("a monolith of ice has been erected here"));
358                                I.setDescription(L("The ice is crystal clear."));
359                                I.setMaterial(RawMaterial.RESOURCE_GLASS);
360                                break;
361                        case TYP_AIR:
362                                I=CMClass.getItem("GenItem");
363                                I.setName(L("a monolith of air"));
364                                I.setDisplayText("");
365                                I.setDescription(L("The air is swirling dangerously."));
366                                I.setMaterial(RawMaterial.RESOURCE_NOTHING);
367                                break;
368                        case TYP_FIRE:
369                                I=CMClass.getItem("GenItem");
370                                I.setName(L("a monolith of fire"));
371                                I.setDisplayText(L("a monolith of fire is burning here"));
372                                I.setDescription(L("The flames are high and hot."));
373                                I.setMaterial(RawMaterial.RESOURCE_NOTHING);
374                                I.basePhyStats().setDisposition(I.basePhyStats().disposition()|PhyStats.IS_LIGHTSOURCE);
375                                break;
376                        }
377                        if(I!=null)
378                        {
379                                final CMMsg msg = CMClass.getMsg(mob,target,this,verbalCastCode(mob,target,auto),auto?L("@x1 appears!",I.name()):L("^S<S-NAME> @x1 to construct @x2!^?",prayForWord(mob),I.name()));
380                                if(mob.location().okMessage(mob,msg))
381                                {
382                                        mob.location().send(mob,msg);
383                                        CMLib.flags().setGettable(I,false);
384                                        I.recoverPhyStats();
385                                        mob.location().addItem(I);
386                                        theWall=I;
387                                        beneficialAffect(mob,I,asLevel,0);
388                                }
389                        }
390                }
391                else
392                        return beneficialWordsFizzle(mob,null,L("<S-NAME> @x1, but nothing happens.",prayWord(mob)));
393
394                // return whether it worked
395                return success;
396        }
