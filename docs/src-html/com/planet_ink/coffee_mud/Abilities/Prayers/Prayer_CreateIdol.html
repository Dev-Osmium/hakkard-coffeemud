036public class Prayer_CreateIdol extends Prayer
037{
038        @Override
039        public String ID()
040        {
041                return "Prayer_CreateIdol";
042        }
043
044        private final static String localizedName = CMLib.lang().L("Create Idol");
045
046        @Override
047        public String name()
048        {
049                return localizedName;
050        }
051
052        @Override
053        public int abstractQuality()
054        {
055                return Ability.QUALITY_INDIFFERENT;
056        }
057
058        @Override
059        public int classificationCode()
060        {
061                return Ability.ACODE_PRAYER|Ability.DOMAIN_CURSING;
062        }
063
064        @Override
065        public long flags()
066        {
067                return Ability.FLAG_UNHOLY;
068        }
069
070        @Override
071        protected int canAffectCode()
072        {
073                return CAN_ITEMS;
074        }
075
076        @Override
077        protected int canTargetCode()
078        {
079                return 0;
080        }
081
082        @Override
083        public boolean bubbleAffect()
084        {
085                return true;
086        }
087
088        @Override
089        public void affectPhyStats(Physical aff, PhyStats affectableStats)
090        {
091                super.affectPhyStats(aff,affectableStats);
092                if((affected instanceof Item)&&(((Item)affected).container()==null))
093                {
094                        final int xlvl=super.getXLEVELLevel(invoker());
095                        affectableStats.setArmor(affectableStats.armor()+(20+(4*xlvl)));
096                        affectableStats.setAttackAdjustment(affectableStats.attackAdjustment()-10-(2*xlvl));
097                }
098        }
099
100        @Override
101        public void affectCharStats(MOB aff, CharStats affectableStats)
102        {
103                super.affectCharStats(aff,affectableStats);
104                if((affected instanceof Item)&&(((Item)affected).container()==null))
105                {
106                        if(affectableStats.getStat(CharStats.STAT_STRENGTH)>3)
107                                affectableStats.setStat(CharStats.STAT_STRENGTH,3);
108                        if(affectableStats.getStat(CharStats.STAT_DEXTERITY)>2)
109                                affectableStats.setStat(CharStats.STAT_DEXTERITY,2);
110                        if(affectableStats.getStat(CharStats.STAT_CONSTITUTION)>1)
111                                affectableStats.setStat(CharStats.STAT_CONSTITUTION,1);
112                }
113        }
114
115        @Override
116        public void affectCharState(MOB aff, CharState affectableState)
117        {
118                super.affectCharState(aff,affectableState);
119                if((affected instanceof Item)&&(((Item)affected).container()==null))
120                {
121                        if(aff.maxState().getFatigue()>Long.MIN_VALUE/2)
122                                aff.curState().setFatigue(CharState.FATIGUED_MILLIS+10);
123                        affectableState.setMovement(20);
124                }
125        }
126
127        @Override
128        public boolean okMessage(final Environmental host, final CMMsg msg)
129        {
130                if((msg.targetMinor()==CMMsg.TYP_GIVE)
131                &&(msg.target() instanceof MOB)
132                &&(affected instanceof Item)
133                &&(msg.tool()==affected)
134                &&(!((MOB)msg.target()).willFollowOrdersOf(msg.source())))
135                {
136                        msg.source().tell(L("@x1 won`t accept @x2.",((MOB)msg.target()).name(msg.source()),((Item)msg.tool()).name(msg.source())));
137                        return false;
138                }
139                return super.okMessage(host,msg);
140        }
141
142        @Override
143        public boolean invoke(MOB mob, List<String> commands, Physical givenTarget, boolean auto, int asLevel)
144        {
145                if((mob.getWorshipCharID().length()==0)||(CMLib.map().getDeity(mob.getWorshipCharID())==null))
146                {
147                        mob.tell(L("You must worship a god to use this prayer."));
148                        return false;
149                }
150
151                if(!super.invoke(mob,commands,givenTarget,auto,asLevel))
152                        return false;
153                int material=-1;
154                final Room R=mob.location();
155                if(R!=null)
156                {
157                        if(((R.myResource()&RawMaterial.MATERIAL_MASK)==RawMaterial.MATERIAL_ROCK)
158                                ||((R.myResource()&RawMaterial.MATERIAL_MASK)==RawMaterial.MATERIAL_MITHRIL)
159                                ||((R.myResource()&RawMaterial.MATERIAL_MASK)==RawMaterial.MATERIAL_METAL))
160                                material=R.myResource();
161                        else
162                        {
163                                final List<Integer> V=R.resourceChoices();
164                                if((V!=null)&&(V.size()>0))
165                                for(int v=0;v<V.size()*10;v++)
166                                {
167                                        final int rsc=V.get(CMLib.dice().roll(1,V.size(),-1)).intValue();
168                                        if(((rsc&RawMaterial.MATERIAL_MASK)==RawMaterial.MATERIAL_ROCK)
169                                                ||((rsc&RawMaterial.MATERIAL_MASK)==RawMaterial.MATERIAL_MITHRIL)
170                                                ||((rsc&RawMaterial.MATERIAL_MASK)==RawMaterial.MATERIAL_METAL))
171                                        {
172                                                material=rsc;
173                                                break;
174                                        }
175                                }
176                        }
177                }
178
179                // now see if it worked
180                final boolean success=proficiencyCheck(mob,0,auto);
181                if((success)&&(material>0))
182                {
183                        final CMMsg msg=CMClass.getMsg(mob,null,this,verbalCastCode(mob,null,auto),auto?"":L("^S<S-NAME> @x1 for an idol.^?",prayWord(mob)));
184                        if(mob.location().okMessage(mob,msg))
185                        {
186                                mob.location().send(mob,msg);
187                                final Item newItem=CMClass.getBasicItem("GenItem");
188                                newItem.setBaseValue(1);
189                                final String name=CMLib.english().startWithAorAn(RawMaterial.CODES.NAME(material).toLowerCase()+" idol of "+mob.getWorshipCharID());
190                                newItem.setName(name);
191                                newItem.setDisplayText(L("@x1 sits here.",name));
192                                newItem.basePhyStats().setDisposition(PhyStats.IS_EVIL);
193                                newItem.basePhyStats().setWeight(10);
194                                newItem.setMaterial(material);
195                                newItem.recoverPhyStats();
196                                CMLib.flags().setRemovable(newItem,false);
197                                CMLib.flags().setDroppable(newItem,false);
198                                newItem.addNonUninvokableEffect((Ability)copyOf());
199                                mob.location().addItem(newItem,ItemPossessor.Expire.Resource);
200                                mob.location().showHappens(CMMsg.MSG_OK_ACTION,L("Suddenly, @x1 grows out of the ground.",newItem.name()));
201                                mob.location().recoverPhyStats();
202                        }
203                }
204                else
205                        return beneficialWordsFizzle(mob,null,L("<S-NAME> @x1 for an idol, but there is no answer.",prayWord(mob)));
206
207                // return whether it worked
208                return success;
209        }
