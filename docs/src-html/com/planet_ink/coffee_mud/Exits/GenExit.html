264        @Override
265        public String temporaryDoorLink()
266        {
267                if(keyName.startsWith("{#"))
268                {
269                        final int x=keyName.indexOf("#}");
270                        if(x>=0)
271                                return keyName.substring(2,x);
272                }
273                return "";
274        }
275
276        @Override
277        public void setTemporaryDoorLink(final String link)
278        {
279                if(link.startsWith("{{#"))
280                {
281                        super.setTemporaryDoorLink(link);
282                        return;
283                }
284                if(keyName.startsWith("{#"))
285                {
286                        final int x=keyName.indexOf("#}");
287                        if(x>=0)
288                                keyName=keyName.substring(x+2);
289                }
290                if(link.length()>0)
291                        keyName="{#"+link+"#}"+keyName;
292        }

319        @Override
320        protected int getCodeNum(final String code)
321        {
322                for(int i=0;i<CODES.length;i++)
323                {
324                        if(code.equalsIgnoreCase(CODES[i]))
325                                return i;
326                }
327                return -1;
328        }

330        @Override
331        public String getStat(final String code)
332        {
333                switch(getCodeNum(code))
334                {
335                case 0:
336                        return ID(); // class
337                case 1:
338                        return name(); // name
339                case 2:
340                        return displayText(); // display
341                case 3:
342                        return description(); // description
343                case 4:
344                        return doorName(); // door
345                case 5:
346                        return "" + basePhyStats().level(); // level
347                case 6:
348                        return "" + basePhyStats().ability(); // ability
349                case 7:
350                        return "" + isReadable(); // isreadable
351                case 8:
352                        return CMLib.coffeeMaker().getExtraEnvPropertiesStr(this); // affbehav
353                case 9:
354                        return "" + basePhyStats().disposition(); // disposition
355                case 10:
356                        return "" + readableText(); // readabletext
357                case 11:
358                        return "" + hasADoor(); // hasadoor
359                case 12:
360                        return "" + defaultsClosed(); // defclosed
361                case 13:
362                        return "" + hasALock(); // hasalock
363                case 14:
364                        return "" + defaultsLocked(); // deflocked
365                case 15:
366                        return "" + keyName(); // keyname
367                case 16:
368                        return "" + openDelayTicks(); // open ticks
369                case 17:
370                        return this.closeWord();
371                case 18:
372                        return this.openWord();
373                case 19:
374                        return this.closedText();
375                }
376                return "";
377        }

379        @Override
380        public void setStat(final String code, String val)
381        {
382                switch(getCodeNum(code))
383                {
384                case 0:
385                        return;
386                case 1:
387                        setName(val);
388                        break; // name
389                case 2:
390                        setDisplayText(val);
391                        break; // display
392                case 3:
393                        setDescription(val);
394                        break; // description
395                case 4:
396                        doorName = val;
397                        break; // door
398                case 5:
399                        basePhyStats().setLevel(CMath.parseIntExpression(val));
400                        break; // level
401                case 6:
402                        basePhyStats().setAbility(CMath.parseIntExpression(val));
403                        break; // ability
404                case 7:
405                        setReadable(CMath.s_bool(val));
406                        break; // isreadable
407                case 8:
408                { // affbehav
409                        delAllEffects(true);
410                        delAllBehaviors();
411                        CMLib.coffeeMaker().setExtraEnvProperties(this, CMLib.xml().parseAllXML(val)); // affbehav
412                        break;
413                } // affbehav
414                case 9:
415                { // disposition
416                        if (CMath.isInteger(val) || (val.trim().length() == 0))
417                                basePhyStats().setDisposition(CMath.s_parseIntExpression(val)); // disposition
418                        else
419                        {
420                                basePhyStats().setDisposition(0);
421                                final List<String> V = CMParms.parseCommas(val, true);
422                                for (final Iterator<String> e = V.iterator(); e.hasNext();)
423                                {
424                                        val = e.next();
425                                        final int dispIndex = CMParms.indexOfIgnoreCase(PhyStats.IS_CODES, val);
426                                        if (dispIndex >= 0)
427                                                basePhyStats().setDisposition(basePhyStats().disposition() | (int) CMath.pow(2, dispIndex));
428                                }
429                        }
430                        break;
431                } // disposition
432                case 10:
433                        setReadableText(val);
434                        break; // readabletext
435                case 11:
436                        hasADoor = CMath.s_bool(val);
437                        break; // hasadoor
438                case 12:
439                        doorDefaultsClosed = CMath.s_bool(val);
440                        break; // defclosed
441                case 13:
442                        hasALock = CMath.s_bool(val);
443                        break; // hasalock
444                case 14:
445                        doorDefaultsLocked = CMath.s_bool(val);
446                        break; // deflocked
447                case 15:
448                        setKeyName(val);
449                        break; // keyname
450                case 16:
451                        setOpenDelayTicks(CMath.s_parseIntExpression(val));
452                        break; // openticks
453                case 17:
454                        this.closeName = val;
455                        break;
456                case 18:
457                        this.openName = val;
458                        break;
459                case 19:
460                        this.closedText = val;
461                        break;
462                }
463        }

465        @Override
466        public boolean sameAs(final Environmental E)
467        {
468                if(!(E instanceof GenExit))
469                        return false;
470                final String[] codes=getStatCodes();
471                for(int i=0;i<codes.length;i++)
472                {
473                        if(!E.getStat(codes[i]).equals(getStat(codes[i])))
474                                return false;
475                }
476                return true;
477        }
