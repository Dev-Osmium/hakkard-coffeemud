036public class Withdraw extends StdCommand
037{
038        public Withdraw()
039        {
040        }
041
042        private final String[] access=I(new String[]{"WITHDRAW"});
043        @Override
044        public String[] getAccessWords()
045        {
046                return access;
047        }
048
049        @Override
050        public boolean execute(MOB mob, List<String> commands, int metaFlags)
051                throws java.io.IOException
052        {
053                Vector<String> origCmds=new XVector<String>(commands);
054                final Environmental shopkeeper=CMLib.english().parseShopkeeper(mob,commands,"Withdraw what or how much from whom?");
055                if(shopkeeper==null)
056                        return false;
057                final ShopKeeper SHOP=CMLib.coffeeShops().getShopKeeper(shopkeeper);
058                if((!(SHOP instanceof Banker))&&(!(SHOP instanceof PostOffice))&&(!(SHOP instanceof Librarian)))
059                {
060                        CMLib.commands().postCommandFail(mob,origCmds,L("You can not withdraw anything from @x1.",shopkeeper.name()));
061                        return false;
062                }
063                if(commands.size()==0)
064                {
065                        CMLib.commands().postCommandFail(mob,origCmds,L("Withdraw what or how much?"));
066                        return false;
067                }
068                String str=CMParms.combine(commands,0);
069                if(str.equalsIgnoreCase("all"))
070                        str=""+Integer.MAX_VALUE;
071                final long numCoins=CMLib.english().numPossibleGold(null,str);
072                final String currency=CMLib.english().numPossibleGoldCurrency(shopkeeper,str);
073                final double denomination=CMLib.english().numPossibleGoldDenomination(shopkeeper,currency,str);
074                Item thisThang=null;
075                if(SHOP instanceof Banker)
076                {
077                        final String accountName=((Banker)SHOP).getBankClientName(mob, Clan.Function.WITHDRAW, false);
078                        if(numCoins>0)
079                        {
080                                if(denomination==0.0)
081                                {
082                                        CMLib.commands().postCommandFail(mob,origCmds,L("Withdraw how much?"));
083                                        return false;
084                                }
085                                thisThang=((Banker)SHOP).findDepositInventory(accountName,""+Integer.MAX_VALUE);
086                                if(thisThang instanceof Coins)
087                                        thisThang=CMLib.beanCounter().makeCurrency(currency,denomination,numCoins);
088                        }
089                        else
090                                thisThang=((Banker)SHOP).findDepositInventory(accountName,str);
091
092                        if(((thisThang==null)||((thisThang instanceof Coins)&&(((Coins)thisThang).getNumberOfCoins()<=0)))
093                        &&(!((Banker)SHOP).isSold(ShopKeeper.DEAL_CLANBANKER))
094                        &&(mob.isMarriedToLiege()))
095                        {
096                                final MOB mob2=CMLib.players().getPlayerAllHosts(mob.getLiegeID());
097                                if(mob2!=null)
098                                {
099                                        final String accountName2=((Banker)SHOP).getBankClientName(mob2, Clan.Function.WITHDRAW, false);
100                                        if(numCoins>0)
101                                        {
102                                                thisThang=((Banker)SHOP).findDepositInventory(accountName2,""+Integer.MAX_VALUE);
103                                                if(thisThang instanceof Coins)
104                                                        thisThang=CMLib.beanCounter().makeCurrency(currency,denomination,numCoins);
105                                                else
106                                                {
107                                                        CMLib.commands().postCommandFail(mob,origCmds,L("Withdraw how much?"));
108                                                        return false;
109                                                }
110                                        }
111                                        else
112                                                thisThang=((Banker)SHOP).findDepositInventory(accountName2,str);
113                                }
114                        }
115                }
116                else
117                if(SHOP instanceof PostOffice)
118                {
119                        final String accountName=((PostOffice)SHOP).getSenderName(mob, Clan.Function.WITHDRAW, false);
120                        thisThang=((PostOffice)SHOP).findBoxContents(accountName,str);
121                        if((thisThang==null)
122                        &&(!((PostOffice)SHOP).isSold(ShopKeeper.DEAL_CLANPOSTMAN))
123                        &&(mob.isMarriedToLiege()))
124                        {
125                                final MOB mob2=CMLib.players().getPlayerAllHosts(mob.getLiegeID());
126                                if(mob2!=null)
127                                {
128                                        final String accountName2=((PostOffice)SHOP).getSenderName(mob, Clan.Function.WITHDRAW, false);
129                                        thisThang=((PostOffice)SHOP).findBoxContents(accountName2,str);
130                                }
131                        }
132                }
133
134                if((thisThang==null)||(!CMLib.flags().canBeSeenBy(thisThang,mob)))
135                {
136                        CMLib.commands().postCommandFail(mob,origCmds,L("That doesn't appear to be available.  Try LIST."));
137                        return false;
138                }
139                String str2="<S-NAME> withdraw(s) <O-NAME> from <S-HIS-HER> account with "+shopkeeper.name()+".";
140                if(SHOP instanceof PostOffice)
141                        str2="<S-NAME> withdraw(s) <O-NAME> from <S-HIS-HER> postal box with "+shopkeeper.name()+".";
142                final CMMsg newMsg=CMClass.getMsg(mob,shopkeeper,thisThang,CMMsg.MSG_WITHDRAW,str2);
143                if(!mob.location().okMessage(mob,newMsg))
144                        return false;
145                mob.location().send(mob,newMsg);
146                return false;
147        }
148
149        @Override
150        public double combatActionsCost(final MOB mob, final List<String> cmds)
151        {
152                return CMProps.getCommandCombatActionCost(ID());
153        }
154
155        @Override
156        public double actionsCost(final MOB mob, final List<String> cmds)
157        {
158                return CMProps.getCommandActionCost(ID());
159        }
160
161        @Override
162        public boolean canBeOrdered()
163        {
164                return false;
165        }
