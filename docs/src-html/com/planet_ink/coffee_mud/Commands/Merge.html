053        public static String getStat(final Environmental E, final String stat)
054        {
055                if((stat!=null)
056                &&(stat.length()>0)
057                &&(stat.equalsIgnoreCase("REJUV"))
058                &&(E instanceof Physical))
059                {
060                        if(((Physical)E).basePhyStats().rejuv()==PhyStats.NO_REJUV)
061                                return "0";
062                        return ""+((Physical)E).basePhyStats().rejuv();
063                }
064                return ((Physical)E).getStat(stat);
065        }
066
067        public static void setStat(final Environmental E, final String stat, final String value)
068        {
069                if((stat!=null)
070                &&(stat.length()>0)
071                &&(stat.equalsIgnoreCase("REJUV"))
072                &&(E instanceof Physical))
073                        ((Physical)E).basePhyStats().setRejuv(CMath.s_int(value));
074                else
075                        E.setStat(stat,value);
076        }

085        protected static boolean tryMerge(final MOB mob,
086                                                                          final Room room,
087                                                                          final Environmental E,
088                                                                          final List things,
089                                                                          final List<String> changes,
090                                                                          final List<String> onfields,
091                                                                          final List<String> ignore,
092                                                                          final boolean noisy)
093        {
094                boolean didAnything=false;
095                final List<String> efields=new Vector<String>();
096                List<String> allMyFields=new Vector<String>();
097                final String[] EFIELDS=E.getStatCodes();
098                for(int i=0;i<EFIELDS.length;i++)
099                {
100                        if(!efields.contains(EFIELDS[i]))
101                                efields.add(EFIELDS[i]);
102                }
103                efields.add("REJUV");
104                allMyFields=new XVector<String>(efields);
105                for(int v=0;v<ignore.size();v++)
106                {
107                        if(efields.contains(ignore.get(v)))
108                                efields.remove(ignore.get(v));
109                }
110                for(int v=0;v<changes.size();v++)
111                {
112                        if(efields.contains(changes.get(v)))
113                                efields.remove(changes.get(v));
114                }
115                if(noisy)
116                        mergedebugtell(mob,"AllMy-"+CMParms.toListString(allMyFields));
117                if(noisy)
118                        mergedebugtell(mob,"efields-"+CMParms.toListString(efields));
119                for(int t=0;t<things.size();t++)
120                {
121                        final Environmental E2=(Environmental)things.get(t);
122                        if(noisy)
123                                mergedebugtell(mob,E.name()+"/"+E2.name()+"/"+CMClass.classID(E)+"/"+CMClass.classID(E2));
124                        if(CMClass.classID(E).equals(CMClass.classID(E2)))
125                        {
126                                Vector<String> fieldsToCheck=null;
127                                if(onfields.size()>0)
128                                {
129                                        fieldsToCheck=new Vector<String>();
130                                        for(int v=0;v<onfields.size();v++)
131                                        {
132                                                if(efields.contains(onfields.get(v)))
133                                                        fieldsToCheck.add(onfields.get(v));
134                                        }
135                                }
136                                else
137                                        fieldsToCheck=new XVector<String>(efields);
138
139                                boolean checkedOut=fieldsToCheck.size()>0;
140                                if(noisy)
141                                        mergedebugtell(mob,"fieldsToCheck-"+CMParms.toListString(fieldsToCheck));
142                                if(checkedOut)
143                                for(int i=0;i<fieldsToCheck.size();i++)
144                                {
145                                        final String field=fieldsToCheck.get(i);
146                                        if(noisy)
147                                                mergedebugtell(mob,field+"/"+getStat(E,field)+"/"+getStat(E2,field)+"/"+getStat(E,field).equals(getStat(E2,field)));
148                                        if(!getStat(E,field).equals(getStat(E2,field)))
149                                        {
150                                                checkedOut = false;
151                                                break;
152                                        }
153                                }
154                                if(checkedOut)
155                                {
156                                        List<String> fieldsToChange=null;
157                                        if(changes.size()==0)
158                                                fieldsToChange=new XVector<String>(allMyFields);
159                                        else
160                                        {
161                                                fieldsToChange=new Vector<String>();
162                                                for(int v=0;v<changes.size();v++)
163                                                {
164                                                        if(allMyFields.contains(changes.get(v)))
165                                                                fieldsToChange.add(changes.get(v));
166                                                }
167                                        }
168                                        if(noisy)
169                                                mergedebugtell(mob,"fieldsToChange-"+CMParms.toListString(fieldsToChange));
170                                        for(int i=0;i<fieldsToChange.size();i++)
171                                        {
172                                                final String field=fieldsToChange.get(i);
173                                                if(noisy)
174                                                        mergedebugtell(mob,E.name()+" wants to change "+field+" value "+getStat(E,field)+" to "+getStat(E2,field)+"/"+(!getStat(E,field).equals(getStat(E2,field))));
175                                                if(!getStat(E,field).equals(getStat(E2,field)))
176                                                {
177                                                        setStat(E,field,getStat(E2,field));
178                                                        Log.sysOut("Merge","The "+CMStrings.capitalizeAndLower(field)+" field on "+E.Name()+" in "+room.roomID()+" was changed to "+getStat(E2,field)+".");
179                                                        didAnything=true;
180                                                }
181                                        }
182                                }
183                        }
184                }
185                if(didAnything)
186                {
187                        if(E instanceof Physical)
188                                ((Physical)E).recoverPhyStats();
189                        if(E instanceof MOB)
190                        {
191                                ((MOB)E).recoverCharStats();
192                                ((MOB)E).recoverMaxState();
193                        }
194                        E.text();
195                }
196                return didAnything;
197        }

199        public void sortEnumeratedList(final Enumeration e, final List<String> allKnownFields, final StringBuffer allFieldsMsg)
200        {
201                for(;e.hasMoreElements();)
202                {
203                        final Environmental E=(Environmental)e.nextElement();
204                        final String[] fields=E.getStatCodes();
205                        for(int x=0;x<fields.length;x++)
206                        {
207                                if(!allKnownFields.contains(fields[x]))
208                                {
209                                        allKnownFields.add(fields[x]);
210                                        allFieldsMsg.append(fields[x]+" ");
211                                }
212                        }
213                }
214        }

216        @Override
217        public boolean execute(final MOB mob, final List<String> commands, final int metaFlags)
218                throws java.io.IOException
219        {
220                final boolean noisy=CMSecurity.isDebugging(CMSecurity.DbgFlag.MERGE);
221                Vector<Places> placesToDo=new Vector<Places>();
222                commands.remove(0);
223                if(commands.size()==0)
224                {
225                        mob.tell(L("Merge what? Try DATABASE or a filename"));
226                        return false;
227                }
228                if(mob.isMonster())
229                {
230                        mob.tell(L("No can do."));
231                        return false;
232                }
233                if((commands.size()>0)&&
234                        commands.get(0).equalsIgnoreCase("noprompt"))
235                        commands.remove(0);
236
237                if((commands.size()>0)&&
238                        commands.get(0).equalsIgnoreCase("?"))
239                {
240                        final StringBuffer allFieldsMsg=new StringBuffer("");
241                        final Vector<String> allKnownFields=new Vector<String>();
242                        sortEnumeratedList(CMClass.mobTypes(),allKnownFields,allFieldsMsg);
243                        sortEnumeratedList(CMClass.basicItems(),allKnownFields,allFieldsMsg);
244                        sortEnumeratedList(CMClass.weapons(),allKnownFields,allFieldsMsg);
245                        sortEnumeratedList(CMClass.armor(),allKnownFields,allFieldsMsg);
246                        sortEnumeratedList(CMClass.clanItems(),allKnownFields,allFieldsMsg);
247                        sortEnumeratedList(CMClass.miscMagic(),allKnownFields,allFieldsMsg);
248                        sortEnumeratedList(CMClass.tech(),allKnownFields,allFieldsMsg);
249                        mob.tell(L("Valid field names are @x1",allFieldsMsg.toString()));
250                        return false;
251                }
252                String scope="WORLD";
253                if((commands.size()>0)
254                &&commands.get(0).equalsIgnoreCase("room"))
255                {
256                        if(!CMSecurity.isAllowed(mob,mob.location(),CMSecurity.SecFlag.MERGE))
257                        {
258                                mob.tell(L("You are not allowed to do that here."));
259                                return false;
260                        }
261                        commands.remove(0);
262                        placesToDo.add(mob.location());
263                        scope="ROOM";
264                }
265                if((commands.size()>0)&&
266                        commands.get(0).equalsIgnoreCase("area"))
267                {
268                        if(!CMSecurity.isAllowed(mob,mob.location(),CMSecurity.SecFlag.MERGE))
269                        {
270                                mob.tell(L("You are not allowed to do that here."));
271                                return false;
272                        }
273                        commands.remove(0);
274                        placesToDo.add(mob.location().getArea());
275                        scope="AREA";
276                }
277                if((commands.size()>0)&&
278                        commands.get(0).equalsIgnoreCase("world"))
279                {
280                        if(!CMSecurity.isAllowedEverywhere(mob,CMSecurity.SecFlag.MERGE))
281                        {
282                                mob.tell(L("You are not allowed to do that."));
283                                return false;
284                        }
285                        commands.remove(0);
286                        placesToDo=new Vector<Places>();
287                        scope="WORLD";
288                }
289                if(commands.size()==0)
290                {
291                        mob.tell(L("Merge what? DATABASE or filename"));
292                        return false;
293                }
294                String firstWord=commands.get(0);
295                if(firstWord.equalsIgnoreCase("DATABASE"))
296                {
297                        commands.remove(0);
298                        if(commands.size()==0)
299                        {
300                                mob.tell(L("Merge parameters missing: DBCLASS, DBSERVICE, DBUSER, DBPASS"));
301                                return false;
302                        }
303                        firstWord=commands.get(0);
304                        return doArchonDBCompare(mob, scope, firstWord, commands);
305                }
306                final String filename=commands.get(commands.size()-1);
307                commands.remove(filename);
308                final StringBuffer buf=new CMFile(filename,mob,CMFile.FLAG_LOGERRORS).text();
309                if((buf==null)||(buf.length()==0))
310                {
311                        mob.tell(L("File not found at: '@x1'!",filename));
312                        return false;
313                }
314
315                final List<String> changes=new Vector<String>();
316                final List<String> onfields=new Vector<String>();
317                final List<String> ignore=new Vector<String>();
318                List<String> use=null;
319                final List<String> allKnownFields=new Vector<String>();
320                final List<Physical> things=new Vector<Physical>();
321                boolean aremobs=false;
322                if((buf.length()>20)&&(buf.substring(0,20).indexOf("<MOBS>")>=0))
323                {
324                        if(mob.session()!=null)
325                                mob.session().rawPrint(L("Unpacking mobs from file: '@x1'...",filename));
326                        List<MOB> mobs=new Vector<MOB>();
327                        final String error=CMLib.coffeeMaker().addMOBsFromXML(buf.toString(),mobs,mob.session());
328                        things.addAll(mobs);
329                        mobs.clear();
330                        mobs=null;
331                        if(mob.session()!=null)
332                                mob.session().rawPrintln("!");
333                        if(error.length()>0)
334                        {
335                                mob.tell(L("An error occurred on merge: @x1",error));
336                                mob.tell(L("Please correct the problem and try the import again."));
337                                return false;
338                        }
339                        aremobs=true;
340                }
341                else
342                if((buf.length()>20)&&(buf.substring(0,20).indexOf("<ITEMS>")>=0))
343                {
344                        if(mob.session()!=null)
345                                mob.session().rawPrint(L("Unpacking items from file: '@x1'...",filename));
346                        List<Item> items=new Vector<Item>();
347                        final String error=CMLib.coffeeMaker().addItemsFromXML(buf.toString(),items,mob.session());
348                        things.addAll(items);
349                        items.clear();
350                        items=null;
351                        if(mob.session()!=null)
352                                mob.session().rawPrintln("!");
353                        if(error.length()>0)
354                        {
355                                mob.tell(L("An error occurred on merge: @x1",error));
356                                mob.tell(L("Please correct the problem and try the import again."));
357                                return false;
358                        }
359                }
360                else
361                {
362                        mob.tell(L("Files of this type are not yet supported by MERGE.  You must merge an ITEMS or MOBS file at this time."));
363                        return false;
364                }
365                if(things.size()==0)
366                {
367                        mob.tell(L("Nothing was found in the file to merge!"));
368                        return false;
369                }
370                final StringBuffer allFieldsMsg=new StringBuffer("");
371                if(aremobs)
372                        sortEnumeratedList(CMClass.mobTypes(),allKnownFields,allFieldsMsg);
373                else
374                {
375                        sortEnumeratedList(CMClass.basicItems(),allKnownFields,allFieldsMsg);
376                        sortEnumeratedList(CMClass.weapons(),allKnownFields,allFieldsMsg);
377                        sortEnumeratedList(CMClass.armor(),allKnownFields,allFieldsMsg);
378                        sortEnumeratedList(CMClass.clanItems(),allKnownFields,allFieldsMsg);
379                        sortEnumeratedList(CMClass.miscMagic(),allKnownFields,allFieldsMsg);
380                        sortEnumeratedList(CMClass.tech(),allKnownFields,allFieldsMsg);
381                }
382
383                allKnownFields.add("REJUV");
384                allFieldsMsg.append(L("REJUV "));
385
386                for(int i=0;i<commands.size();i++)
387                {
388                        String str=commands.get(i).toUpperCase();
389                        if(str.startsWith("CHANGE="))
390                        {
391                                use=changes;
392                                str=str.substring(7).trim();
393                        }
394                        if(str.startsWith("ON="))
395                        {
396                                use=onfields;
397                                str=str.substring(3).trim();
398                        }
399                        if(str.startsWith("IGNORE="))
400                        {
401                                use=ignore;
402                                str=str.substring(7).trim();
403                        }
404                        int x=str.indexOf(',');
405                        while(x>=0)
406                        {
407                                final String s=str.substring(0,x).trim();
408                                if(s.length()>0)
409                                {
410                                        if(use==null)
411                                        {
412                                                mob.tell(L("'@x1' is an unknown parameter!",str));
413                                                return false;
414                                        }
415                                        if(allKnownFields.contains(s))
416                                                use.add(s);
417                                        else
418                                        {
419                                                mob.tell(L("'@x1' is an unknown field name.  Valid fields include: @x2",s,allFieldsMsg.toString()));
420                                                return false;
421                                        }
422                                }
423                                str=str.substring(x+1).trim();
424                                x=str.indexOf(',');
425                        }
426                        if(str.length()>0)
427                        {
428                                if(use==null)
429                                {
430                                        mob.tell(L("'@x1' is an unknown parameter!",str));
431                                        return false;
432                                }
433                                if(allKnownFields.contains(str))
434                                        use.add(str);
435                                else
436                                {
437                                        mob.tell(L("'@x1' is an unknown field name.  Valid fields include: @x2",str,allFieldsMsg.toString()));
438                                        return false;
439                                }
440                        }
441                }
442                if((onfields.size()==0)&&(ignore.size()==0)&&(changes.size()==0))
443                {
444                        mob.tell(L("You must specify either an ON, CHANGES, or IGNORE parameter for valid matches to be made."));
445                        return false;
446                }
447                if(placesToDo.size()==0)
448                for(final Enumeration<Area> a=CMLib.map().areas();a.hasMoreElements();)
449                {
450                        final Area A=a.nextElement();
451                        if(A.getCompleteMap().hasMoreElements()
452                        &&CMSecurity.isAllowed(mob,(A.getCompleteMap().nextElement()),CMSecurity.SecFlag.MERGE))
453                                placesToDo.add(A);
454                }
455                if(placesToDo.size()==0)
456                {
457                        mob.tell(L("There are no rooms to merge into!"));
458                        return false;
459                }
460                for(int i=placesToDo.size()-1;i>=0;i--)
461                {
462                        if(placesToDo.get(i) instanceof Area)
463                        {
464                                final Area A=(Area)placesToDo.get(i);
465                                placesToDo.removeElement(A);
466                                for(final Enumeration r=A.getCompleteMap();r.hasMoreElements();)
467                                {
468                                        final Room R=(Room)r.nextElement();
469                                        if(CMSecurity.isAllowed(mob,R,CMSecurity.SecFlag.MERGE))
470                                                placesToDo.add(R);
471                                }
472                        }
473                        else
474                        if(placesToDo.get(i) instanceof Room)
475                        {
476                                if(mob.session()!=null)
477                                        mob.session().rawPrint(".");
478                        }
479                        else
480                                return false;
481                }
482                // now do the merge...
483                if(mob.session()!=null)
484                        mob.session().rawPrint(L("Merging and saving..."));
485                if(noisy)
486                        mergedebugtell(mob,"Rooms to do: "+placesToDo.size());
487                if(noisy)
488                        mergedebugtell(mob,"Things loaded: "+things.size());
489                if(noisy)
490                        mergedebugtell(mob,"On fields="+CMParms.toListString(onfields));
491                if(noisy)
492                        mergedebugtell(mob,"Ignore fields="+CMParms.toListString(ignore));
493                if(noisy)
494                        mergedebugtell(mob,"Change fields="+CMParms.toListString(changes));
495                Log.sysOut("Merge",mob.Name()+" merge '"+filename+"'.");
496                for(int r=0;r<placesToDo.size();r++)
497                {
498                        Room R=(Room)placesToDo.get(r);
499                        if(!CMSecurity.isAllowed(mob,R,CMSecurity.SecFlag.MERGE))
500                                continue;
501                        if(R.roomID().length()==0)
502                                continue;
503                        synchronized(("SYNC"+R.roomID()).intern())
504                        {
505                                R=CMLib.map().getRoom(R);
506                                final Area.State oldFlags=R.getArea().getAreaState();
507                                R.getArea().setAreaState(Area.State.FROZEN);
508                                CMLib.map().resetRoom(R);
509                                boolean savemobs=false;
510                                boolean saveitems=false;
511                                if(aremobs)
512                                {
513                                        for(int m=0;m<R.numInhabitants();m++)
514                                        {
515                                                final MOB M=R.fetchInhabitant(m);
516                                                if((M!=null)&&(M.isSavable()))
517                                                        if(tryMerge(mob,R,M,things,changes,onfields,ignore,noisy))
518                                                                savemobs=true;
519                                        }
520                                }
521                                else
522                                {
523                                        for(int i=0;i<R.numItems();i++)
524                                        {
525                                                final Item I=R.getItem(i);
526                                                if((I!=null)&&(tryMerge(mob,R,I,things,changes,onfields,ignore,noisy)))
527                                                        saveitems=true;
528                                        }
529                                        for(int m=0;m<R.numInhabitants();m++)
530                                        {
531                                                final MOB M=R.fetchInhabitant(m);
532                                                if((M!=null)&&(M.isSavable()))
533                                                {
534                                                        for(int i=0;i<M.numItems();i++)
535                                                        {
536                                                                final Item I=M.getItem(i);
537                                                                if((I!=null)&&(tryMerge(mob,R,I,things,changes,onfields,ignore,noisy)))
538                                                                        savemobs=true;
539                                                        }
540                                                        final ShopKeeper SK=CMLib.coffeeShops().getShopKeeper(M);
541                                                        if(SK!=null)
542                                                        {
543                                                                for(final Iterator<Environmental> i=SK.getShop().getStoreInventory();i.hasNext();)
544                                                                {
545                                                                        final Environmental E=i.next();
546                                                                        if(E instanceof Item)
547                                                                        {
548                                                                                final Item I=(Item)E;
549                                                                                if(tryMerge(mob,R,I,things,changes,onfields,ignore,noisy))
550                                                                                        savemobs=true;
551                                                                        }
552                                                                }
553                                                        }
554                                                }
555                                        }
556                                }
557                                if(saveitems)
558                                        CMLib.database().DBUpdateItems(R);
559                                if(savemobs)
560                                        CMLib.database().DBUpdateMOBs(R);
561                                if(mob.session()!=null)
562                                        mob.session().rawPrint(".");
563                                R.getArea().setAreaState(oldFlags);
564                        }
565                }
566
567                if(mob.session()!=null)
568                        mob.session().safeRawPrintln(L("!\n\rDone!"));
569                Area A=null;
570                for(int i=0;i<placesToDo.size();i++)
571                {
572                        A=((Room)placesToDo.get(i)).getArea();
573                        if((A!=null)&&(A.getAreaState()!=Area.State.ACTIVE))
574                                A.setAreaState(Area.State.ACTIVE);
575                }
576                return false;
577        }

587        private boolean amMergingType(final CMClass.CMObjectType doType, final Environmental E)
588        {
589                if(doType==null)
590                        return true;
591                switch(doType)
592                {
593                case LOCALE:
594                        return true;
595                case MOB:
596                        return E instanceof MOB;
597                case ITEM:
598                        return E instanceof Item;
599                case WEAPON:
600                        return E instanceof Weapon;
601                case ARMOR:
602                        return E instanceof Armor;
603                default:
604                        return false;
605                }
606        }
607
608        private boolean amMerging(final CMClass.CMObjectType doType, final MaskingLibrary.CompiledZMask mask, final Environmental E)
609        {
610                if(amMergingType(doType,E))
611                {
612                        if(mask==null)
613                                return true;
614                        return CMLib.masking().maskCheck(mask, E, true);
615                }
616                return false;
617        }

619        public boolean dbMerge(final MOB mob, final String name, final Modifiable dbM, final Modifiable M, final Set<String> ignores) throws java.io.IOException, CMException
620        {
621                if((M instanceof Physical) && (dbM instanceof Physical))
622                {

663                                if(res.trim().equalsIgnoreCase("N"))
664                                        continue;
665                                if(res.trim().equalsIgnoreCase("Q"))
666                                        throw new CMException("Cancelled by user.");
667                                didSomething=true;
668                                if(res.trim().equalsIgnoreCase("D"))
669                                {
670                                        M.setStat(statCode,dbVal);
671                                        continue;
672                                }
673                                M.setStat(statCode,CMLib.genEd().prompt(mob,M.getStat(statCode),++showNumber,showFlag,promptStr));
674                        }
675                        if(showNumber==0)
676                                return didSomething;
677                        if(showFlag<-900)
678                        {
679                                ok=true;
680                                break;
681                        }
682                        if(showFlag>0)
683                        {
684                                showFlag=-1;
685                                continue;
686                        }
687                        showFlag=CMath.s_int(mob.session().prompt(L("Edit which? "),""));
688                        if(showFlag<=0)
689                        {
690                                showFlag=-1;
691                                ok=true;
692                        }
693                }
694                return didSomething;
