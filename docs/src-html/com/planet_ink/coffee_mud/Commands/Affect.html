047        public String getMOBState(final MOB mob)
048        {
049                final StringBuffer msg=new StringBuffer("");
050                if((mob.playerStats()!=null)
051                &&(mob.soulMate()==null)
052                &&(mob.playerStats().getHygiene()>=PlayerStats.HYGIENE_DELIMIT))
053                {
054                        if(CMSecurity.isASysOp(mob))
055                                mob.playerStats().setHygiene(0);
056                        else
057                        {
058                                final int x=(int)(mob.playerStats().getHygiene()/PlayerStats.HYGIENE_DELIMIT);
059                                if(x<=1)
060                                        msg.append(L("^!You could use a bath.^?\n\r"));
061                                else
062                                if(x<=3)
063                                        msg.append(L("^!You could really use a bath.^?\n\r"));
064                                else
065                                if(x<=7)
066                                        msg.append(L("^!You need to bathe, soon.^?\n\r"));
067                                else
068                                if(x<15)
069                                        msg.append(L("^!You desperately need to bathe.^?\n\r"));
070                                else msg.append(L("^!Your stench is horrendous! Bathe dammit!^?\n\r"));
071                        }
072                }
073
074                if(CMLib.flags().isBound(mob))
075                        msg.append(L("^!You are bound.^?\n\r"));
076
077                // dont do falling -- the flag doubles for drowning/treading water anyway.
078                //if(CMLib.flags().isFalling(mob))
079                //    msg.append(L("^!You are falling!!!^?\n\r"));
080                //else
081                if(CMLib.flags().isSleeping(mob))
082                        msg.append(L("^!You are sleeping.^?\n\r"));
083                else
084                if(CMLib.flags().isSitting(mob))
085                        msg.append(L("^!You are resting.^?\n\r"));
086                else
087                if(CMLib.flags().isSwimmingInWater(mob))
088                        msg.append(L("^!You are swimming.^?\n\r"));
089                else
090                if(CMLib.flags().isClimbing(mob))
091                        msg.append(L("^!You are climbing.^?\n\r"));
092                else
093                if(CMLib.flags().isFlying(mob))
094                        msg.append(L("^!You are flying.^?\n\r"));
095                else
096                        msg.append(L("^!You are standing.^?\n\r"));
097
098                if(mob.riding()!=null)
099                        msg.append(L("^!You are @x1 @x2.^?\n\r",mob.riding().stateString(mob),mob.riding().name()));
100
101                if(mob.isAttributeSet(MOB.Attrib.PLAYERKILL))
102                        msg.append(L("^!Your playerkill flag is on.^?\n\r"));
103
104                if(CMLib.flags().isInvisible(mob))
105                        msg.append(L("^!You are invisible.^?\n\r"));
106                if(CMLib.flags().isHidden(mob))
107                        msg.append(L("^!You are hidden.^?\n\r"));// ("+CMLib.flags().getHideScore(mob)+").^?\n\r");
108                if(CMLib.flags().isSneaking(mob))
109                        msg.append(L("^!You are sneaking.^?\n\r"));
110                if(mob.isAttributeSet(MOB.Attrib.QUIET))
111                        msg.append(L("^!You are in QUIET mode.^?\n\r"));
112
113                if(mob.curState().getFatigue()>CharState.FATIGUED_MILLIS)
114                        msg.append(L("^!You are fatigued.^?\n\r"));
115                if(mob.curState().getHunger()<1)
116                        msg.append(L("^!You are hungry.^?\n\r"));
117                if(mob.curState().getThirst()<1)
118                        msg.append(L("^!You are thirsty.^?\n\r"));
119                return msg.toString();
120        }

122        public String getAffects(Session S, Physical P, boolean xtra, boolean autosAlso)
123        {
124                final StringBuffer msg=new StringBuffer("");
125                final int NUM_COLS=2;
126                final int COL_LEN=CMLib.lister().fixColWidth(36.0,S);
127                int colnum=NUM_COLS;
128                final MOB mob=(S!=null)?S.mob():null;
129                for(final Enumeration<Ability> a=P.effects();a.hasMoreElements();)
130                {
131                        final Ability A=a.nextElement();
132                        if(A==null)
133                                continue;
134                        String disp=A.displayText();
135                        if(autosAlso && disp.length()==0)
136                                disp=A.ID()+"+"+A.proficiency();
137                        if(disp.length()>0)
138                        {
139                                if(disp.startsWith("(")&&disp.endsWith(")"))
140                                {
141                                        long tr=A.expirationDate();
142                                        if(A.invoker()!=null)
143                                                tr=tr-(System.currentTimeMillis()-A.invoker().lastTickedDateTime());
144                                        if(tr<Ability.TICKS_ALMOST_FOREVER)
145                                                disp+=" ^.^N"+CMLib.time().date2EllapsedTime(tr, TimeUnit.SECONDS, true);
146                                }
147                                if(xtra)
148                                        disp+=", BY="+((A.invoker()==null)?"N/A":A.invoker().Name());
149                                String[] disps={disp};
150                                if(CMStrings.lengthMinusColors(disp)>(COL_LEN*NUM_COLS))
151                                {
152                                        String s=CMLib.coffeeFilter().fullOutFilter(S,mob,null,null,null,disp,true);
153                                        s=CMStrings.replaceAll(s,"\r","");
154                                        final List<String> V=CMParms.parseAny(s,'\n',true);
155                                        disps=new String[V.size()];
156                                        for(int d=0;d<V.size();d++)
157                                                disps[d]=V.get(d);
158                                        colnum=NUM_COLS;
159                                }
160                                for (final String disp2 : disps)
161                                {
162                                        disp=disp2;
163                                        if(((++colnum)>=NUM_COLS)||(CMStrings.lengthMinusColors(disp)>COL_LEN))
164                                        {
165                                                msg.append("\n\r");
166                                                colnum=0;
167                                        }
168                                        msg.append("^S"+CMStrings.padRightPreserve("^<HELPNAME NAME='"+CMStrings.removeColors(A.name())+"'^>"+disp+"^.^N^</HELPNAME^>",COL_LEN));
169                                        if(CMStrings.lengthMinusColors(disp)>COL_LEN)
170                                                colnum=99;
171                                }
172                        }
173                }
174                msg.append("^N\n\r");
175                return msg.toString();
176        }

178        @Override
179        public boolean execute(MOB mob, List<String> commands, int metaFlags)
180                throws java.io.IOException
181        {
182                final Session S=mob.session();
183                if(S!=null)
184                {
185                        if(CMSecurity.isAllowed(mob, mob.location(),CMSecurity.SecFlag.CMDMOBS))
186                        {
187                                final String name=CMParms.combine(commands,1);
188                                if(name.length()>0)
189                                {
190                                        Physical P=null;
191                                        if((name.equalsIgnoreCase("here")||(name.equalsIgnoreCase("room"))))
192                                                P=CMLib.map().roomLocation(mob);
193                                        else
194                                        if((name.equalsIgnoreCase("area")||(name.equalsIgnoreCase("zone"))))
195                                                P=CMLib.map().areaLocation(mob);
196                                        else
197                                                P=mob.location().fetchFromMOBRoomFavorsItems(mob,null,name,Wearable.FILTER_ANY);
198                                        if(P==null)
199                                                S.colorOnlyPrint(L("You don't see @x1 here.\n\r^N",name));
200                                        else
201                                        {
202                                                if(S==mob.session())
203                                                        S.colorOnlyPrint(L(" \n\r^!@x1 is affected by: ^?",P.name()));
204                                                final String msg=getAffects(S,P,true,mob.isAttributeSet(MOB.Attrib.SYSOPMSGS));
205                                                if(msg.length()<5)
206                                                        S.colorOnlyPrintln(L("Nothing!\n\r^N"));
207                                                else
208                                                        S.colorOnlyPrintln(msg);
209                                        }
210                                        return false;
211                                }
212
213                        }
214                        if(S==mob.session())
215                                S.colorOnlyPrintln("\n\r"+getMOBState(mob)+"\n\r");
216                        if(S==mob.session())
217                                S.colorOnlyPrint(L("^!You are affected by: ^?"));
218                        final String msg=getAffects(S,mob,mob.isAttributeSet(MOB.Attrib.SYSOPMSGS),mob.isAttributeSet(MOB.Attrib.SYSOPMSGS));
219                        if(msg.length()<5)
220                                S.colorOnlyPrintln(L("Nothing!\n\r^N"));
221                        else
222                                S.colorOnlyPrintln(msg);
223                }
224                return false;
225        }

233        @Override
234        public Object executeInternal(MOB mob, int metaFlags, Object... args) throws java.io.IOException
235        {
236                //if(!super.checkArguments(internalParameters, args)) return Boolean.FALSE.toString();
237
238                Physical target=mob;
239                Session S=(mob!=null)?mob.session():null;
240                for(final Object o : args)
241                {
242                        if(o instanceof Physical)
243                        {
244                                if(o instanceof MOB)
245                                        S=((MOB)o).session();
246                                target=(Physical)o;
247                        }
248                }
249                return getAffects(S,target,false,false);
250        }
