052        public StringBuilder getEquipment(MOB seer, MOB mob, boolean allPlaces)
053        {
054                final StringBuilder msg=new StringBuilder("");
055                if(CMLib.flags().isSleeping(seer))
056                        return new StringBuilder("(nothing you can see right now)");
057
058                final int wrap=CMLib.lister().fixColWidth(74,seer.session());
059                final int shortWrap=CMLib.lister().fixColWidth(50,seer.session());
060                final int headWrap=26;
061
062                long wornCode=0;
063                String header=null;
064                String wornName=null;
065                Item thisItem=null;
066                String tat=null;
067                final boolean paragraphView=(CMProps.getIntVar(CMProps.Int.EQVIEW)>CMProps.Int.EQVIEW_MIXED)
068                                                        ||((seer!=mob)&&(CMProps.getIntVar(CMProps.Int.EQVIEW)>CMProps.Int.EQVIEW_DEFAULT))
069                                                        ||seer.isAttributeSet(MOB.Attrib.COMPRESS);
070                final HashSet<Item> alreadyDone=new HashSet<Item>();
071                final Wearable.CODES codes = Wearable.CODES.instance();
072                for(int l=0;l<codes.all_ordered().length;l++)
073                {
074                        wornCode=codes.all_ordered()[l];
075                        wornName=codes.name(wornCode);
076                        if(paragraphView)
077                                header=" ^!";
078                        else
079                        {
080                                header="^N(^H"+wornName+"^?)";
081                                header+=CMStrings.SPACES.substring(0,headWrap-header.length())+": ^!";
082                        }
083                        final List<Item> wornHere=mob.fetchWornItems(wornCode,(short)(Short.MIN_VALUE+1),(short)0);
084                        int shownThisLoc=0;
085                        int numLocations=mob.getWearPositions(wornCode);
086                        if(numLocations==0)
087                                numLocations=1;
088                        int emptySlots=numLocations;
089                        if(wornHere.size()>0)
090                        {
091                                if(!allPlaces)
092                                {
093                                        final List<List<Item>> sets=new Vector<List<Item>>(numLocations);
094                                        for(int i=0;i<numLocations;i++)
095                                                sets.add(new Vector<Item>());
096                                        Item I=null;
097                                        Item I2=null;
098                                        short layer=Short.MAX_VALUE;
099                                        short layerAtt=0;
100                                        short layer2=Short.MAX_VALUE;
101                                        short layerAtt2=0;
102                                        List<Item> set=null;
103                                        for(int i=0;i<wornHere.size();i++)
104                                        {
105                                                I=wornHere.get(i);
106                                                if(I.container()!=null)
107                                                        continue;
108                                                if(I instanceof Armor)
109                                                {
110                                                        layer=((Armor)I).getClothingLayer();
111                                                        layerAtt=((Armor)I).getLayerAttributes();
112                                                }
113                                                else
114                                                {
115                                                        layer=0;
116                                                        layerAtt=0;
117                                                }
118                                                for(int s=0;s<sets.size();s++)
119                                                {
120                                                        set=sets.get(s);
121                                                        if(set.size()==0)
122                                                        {
123                                                                set.add(I);
124                                                                break;
125                                                        }
126                                                        for(int s2=0;s2<set.size();s2++)
127                                                        {
128                                                                I2=set.get(s2);
129                                                                if(I2 instanceof Armor)
130                                                                {
131                                                                        layer2=((Armor)I2).getClothingLayer();
132                                                                        layerAtt2=((Armor)I2).getLayerAttributes();
133                                                                }
134                                                                else
135                                                                {
136                                                                        layer2=0;
137                                                                        layerAtt2=0;
138                                                                }
139                                                                if(layer2==layer)
140                                                                {
141                                                                        if(((layerAtt&Armor.LAYERMASK_MULTIWEAR)>0)
142                                                                        &&((layerAtt2&Armor.LAYERMASK_MULTIWEAR)>0))
143                                                                                set.add(s2,I);
144                                                                        break;
145                                                                }
146                                                                if(layer2>layer)
147                                                                {
148                                                                        set.add(s2,I);
149                                                                        break;
150                                                                }
151                                                        }
152                                                        if(set.contains(I))
153                                                                break;
154                                                        if(layer2<layer)
155                                                        {
156                                                                set.add(I);
157                                                                break;
158                                                        }
159                                                }
160                                        }
161                                        wornHere.clear();
162                                        for(int s=0;s<sets.size();s++)
163                                        {
164                                                set=sets.get(s);
165                                                int s2=set.size()-1;
166                                                for(;s2>=0;s2--)
167                                                {
168                                                        I2=set.get(s2);
169                                                        wornHere.add(I2);
170                                                        if((!(I2 instanceof Armor))
171                                                        ||(!CMath.bset(((Armor)I2).getLayerAttributes(),Armor.LAYERMASK_SEETHROUGH)))
172                                                        {
173                                                                emptySlots--;
174                                                                break;
175                                                        }
176                                                }
177                                        }
178                                }
179                                for(int i=0;i<wornHere.size();i++)
180                                {
181                                        thisItem=wornHere.get(i);
182                                        if((thisItem.container()==null)&&(thisItem.amWearingAt(wornCode)))
183                                        {
184                                                if(paragraphView)
185                                                {
186                                                        if(alreadyDone.contains(thisItem))
187                                                                continue;
188                                                        alreadyDone.add(thisItem);
189                                                }
190                                                if(CMLib.flags().canBeSeenBy(thisItem,seer))
191                                                {
192                                                        if(paragraphView)
193                                                        {
194                                                                String name=thisItem.name();
195                                                                if(!allPlaces)
196                                                                        name=CMStrings.ellipse(name,wrap);
197                                                                if(wornCode==Wearable.WORN_HELD)
198                                                                {
199                                                                        if(msg.length()==0)
200                                                                                msg.append(L("nothing."));
201                                                                        if(mob==seer)
202                                                                                msg.append(L("\n\rHolding ^<EItem^>@x1^</EItem^>@x2^N",name,CMLib.flags().getDispositionBlurbs(thisItem,seer).toString().trim()));
203                                                                        else
204                                                                                msg.append(L("\n\r@x1 is holding @x2@x3^N.",mob.charStats().HeShe(),name.trim(),CMLib.flags().getDispositionBlurbs(thisItem, seer).toString().trim()));
205                                                                }
206                                                                else
207                                                                if(wornCode==Wearable.WORN_WIELD)
208                                                                {
209                                                                        if(msg.length()==0)
210                                                                                msg.append(L("nothing."));
211                                                                        if(mob==seer)
212                                                                                msg.append(L("\n\rWielding ^<EItem^>@x1^</EItem^>@x2^N.",name,CMLib.flags().getDispositionBlurbs(thisItem,seer).toString().trim()));
213                                                                        else
214                                                                                msg.append(L("\n\r@x1 is wielding @x2@x3^N.",mob.charStats().HeShe(),name.trim(),CMLib.flags().getDispositionBlurbs(thisItem, seer).toString().trim()));
215                                                                }
216                                                                else
217                                                                {
218                                                                        if(mob==seer)
219                                                                                msg.append(header+"^<EItem^>"+name+"^</EItem^>"+CMLib.flags().getDispositionBlurbs(thisItem,seer).toString().trim()+"^N,");
220                                                                        else
221                                                                                msg.append(header+name.trim()+CMLib.flags().getDispositionBlurbs(thisItem,seer).toString().trim()+"^N,");
222                                                                }
223                                                        }
224                                                        else
225                                                        {
226                                                                String name=thisItem.name();
227                                                                if((name.length()>shortWrap)&&(!allPlaces))
228                                                                        name=name.substring(0,shortWrap)+"...";
229                                                                if(mob==seer)
230                                                                        msg.append(header+"^<EItem^>"+name+"^</EItem^>"+CMLib.flags().getDispositionBlurbs(thisItem,seer).toString().trim()+"^?\n\r");
231                                                                else
232                                                                        msg.append(header+name.trim()+CMLib.flags().getDispositionBlurbs(thisItem,seer).toString().trim()+"^?\n\r");
233                                                        }
234                                                        shownThisLoc++;
235                                                }
236                                                else
237                                                if(seer==mob)
238                                                {
239                                                        msg.append(L("@x1(something you can`t see)@x2^?\n\r",header,CMLib.flags().getDispositionBlurbs(thisItem,seer).toString().trim()));
240                                                        shownThisLoc++;
241                                                }
242                                        }
243                                }
244                        }
245                        if(emptySlots>0)
246                        {
247                                double numTattoosTotal=0;
248                                wornName=wornName.toUpperCase();
249                                for(final Enumeration<Tattoo> e=mob.tattoos();e.hasMoreElements();)
250                                {
251                                        final Tattoo T = e.nextElement();
252                                        if(T.getTattooName().startsWith(wornName+":"))
253                                                numTattoosTotal+=1.0;
254                                }
255                                int numTattoosToShow=(int)Math.round(Math.ceil(CMath.mul(numTattoosTotal,CMath.div(emptySlots,numLocations))));
256                                for(final Enumeration<Tattoo> e=mob.tattoos();e.hasMoreElements();)
257                                {
258                                        final Tattoo T = e.nextElement();
259                                        if((T.getTattooName().startsWith(wornName+":"))
260                                        &&((--numTattoosToShow)>=0))
261                                        {
262                                                tat=T.getTattooName();
263                                                if(paragraphView)
264                                                {
265                                                        tat=tat.substring(wornName.length()+1).toLowerCase();
266                                                        if(tat.length()>wrap)
267                                                                tat=tat.substring(0,wrap)+"...";
268                                                        msg.append(header+tat+"^?,");
269                                                }
270                                                else
271                                                {
272                                                        tat=CMStrings.capitalizeAndLower(tat.substring(wornName.length()+1).toLowerCase());
273                                                        if(tat.length()>shortWrap)
274                                                                tat=tat.substring(0,shortWrap)+"...";
275                                                        msg.append(header+tat+"^?\n\r");
276                                                }
277                                                shownThisLoc++;
278                                        }
279                                }
280                        }
281                        if((allPlaces)&&(shownThisLoc==0))
282                        {
283                                if(((!paragraphView)&&(wornCode!=Wearable.WORN_FLOATING_NEARBY))
284                                ||((paragraphView)&&(wornCode!=Wearable.WORN_WIELD)))
285                                        for(int i=mob.getWearPositions(wornCode)-1;i>=0;i--)
286                                                msg.append(header+"^?\n\r");
287                        }
288                }
289                if(msg.length()==0)
290                {
291                        msg.append(L("^!(nothing)^?\n\r"));
292                }
293                else
294                {
295                        int commaDex=(paragraphView)?msg.lastIndexOf(","):-1;
296                        if((paragraphView)&&(commaDex > -1))
297                        {
298                                msg.insert(commaDex + 1, ".");
299                                msg.deleteCharAt(commaDex);
300                                commaDex=msg.lastIndexOf(",");
301                                if(commaDex > -1)
302                                        msg.insert(commaDex + 1, " and");
303                        }
304                }
305                return msg;
306        }

308        private void viewEquipment(MOB mob, boolean longView)
309        {
310                final Session session=mob.session();
311                if(session!=null)
312                {
313                        final boolean paragraphView=(CMProps.getIntVar(CMProps.Int.EQVIEW)==CMProps.Int.EQVIEW_PARAGRAPH);
314                        if(paragraphView)
315                        {
316                                if(longView)
317                                        session.wraplessPrintln(L("You are wearing @x1",getEquipment(mob,mob,true).toString()));
318                                else
319                                        session.wraplessPrintln(L("You are wearing @x1",getEquipment(mob,mob,false).toString()));
320                        }
321                        else
322                        if(longView)
323                                session.wraplessPrintln(L("You are wearing:\n\r@x1",getEquipment(mob,mob,true).toString()));
324                        else
325                                session.wraplessPrintln(L("You are wearing:\n\r@x1",getEquipment(mob,mob,false).toString()));
326                }
327        }

329        @Override
330        public boolean execute(MOB mob, List<String> commands, int metaFlags)
331                throws java.io.IOException
332        {
333                viewEquipment(mob,(commands.size()>1)&&(CMParms.combine(commands,1).equalsIgnoreCase("long")));
334                return false;
335        }
336
337        @Override
338        public boolean canBeOrdered()
339        {
340                return true;
341        }
342
343        @Override
344        public Object executeInternal(MOB mob, int metaFlags, Object... args) throws java.io.IOException
345        {
346                if(!super.checkArguments(internalParameters, args))
347                        return null;
348
349                if((args.length>0)&&(args[0] instanceof MOB))
350                        return getEquipment((MOB)args[0],mob,false);
351                else
352                if((args.length>0)&&(args[0] instanceof Boolean))
353                        viewEquipment(mob,((Boolean)args[0]).booleanValue());
354                else
355                        viewEquipment(mob,false);
356                return null;
357        }
