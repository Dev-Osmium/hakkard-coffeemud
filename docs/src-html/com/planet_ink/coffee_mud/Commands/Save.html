049        public enum SaveTask
050        {
051                ALL,
052                ITEMS,
053                MOBS
054        }
055
056        private int numSavableInhabitants(final Room room)
057        {
058                int ct=0;
059                for(int i=0;i<room.numInhabitants();i++)
060                {
061                        final MOB M=room.fetchInhabitant(i);
062                        if((M!=null)
063                        &&(M.isSavable())
064                        &&((M.amFollowing()==null)||(!M.amFollowing().isPlayer())))
065                                ct++;
066                }
067                return ct;
068        }
069
070        private int numSavableItems(final Room room)
071        {
072                int ct=0;
073                for(int i=0;i<room.numItems();i++)
074                {
075                        final Item I=room.getItem(i);
076                        if((I!=null)&&(I.isSavable()))
077                                ct++;
078                }
079                return ct;
080        }

082        public boolean clearSaveAndRestart(final MOB mob, Room room, final SaveTask taskCode, final boolean noPrompt) throws IOException
083        {
084                synchronized(("SYNC"+room.roomID()).intern())
085                {
086                        room=CMLib.map().getRoom(room);
087                        CMLib.threads().clearDebri(room,0);
088                        if((!noPrompt)&&(mob!=null)&&(mob.session()!=null))
089                        {
090                                final int[] counts = CMLib.database().DBCountRoomMobsItems(room.roomID());
091                                int mobCountDiff = 0;
092                                if((taskCode == SaveTask.ALL) || (taskCode == SaveTask.MOBS))
093                                        mobCountDiff= counts[0] - this.numSavableInhabitants(room);
094                                int itemCountDiff = 0;
095                                if((taskCode == SaveTask.ALL) || (taskCode == SaveTask.ITEMS))
096                                        itemCountDiff= counts[1] - this.numSavableItems(room);
097                                final StringBuilder msg = new StringBuilder();
098                                if(mobCountDiff < 0)
099                                        msg.append(L("add @x1 mob(s) ",""+(-mobCountDiff)));
100                                else
101                                if(mobCountDiff > 0)
102                                        msg.append(L("delete @x1 mob(s) ",""+(mobCountDiff)));
103
104                                if((itemCountDiff != 0) && (mobCountDiff != 0))
105                                        msg.append(L("and "));
106
107                                if(itemCountDiff < 0)
108                                        msg.append(L("add @x1 item(s) ",""+(-itemCountDiff)));
109                                else
110                                if(itemCountDiff > 0)
111                                        msg.append(L("delete @x1 item(s) ",""+(itemCountDiff)));
112
113                                if(msg.length() > 0)
114                                {
115                                        if((!mob.session().confirm(L("Saving @x1 will @x2. Are you sure (Y/n)?",
116                                                CMLib.map().getExtendedRoomID(room),msg.toString()), "Y"))
117                                        ||(mob.session().isStopped()))
118                                        {
119                                                return false;
120                                        }
121                                }
122                        }
123
124                        if((taskCode == SaveTask.ALL) || (taskCode == SaveTask.ITEMS))
125                        {
126                                CMLib.database().DBUpdateItems(room);
127                                room.startItemRejuv();
128                        }
129                        if((taskCode == SaveTask.ALL) || (taskCode == SaveTask.MOBS))
130                                CMLib.database().DBUpdateMOBs(room);
131                        return true;
132                }
133        }

135        @Override
136        public boolean execute(final MOB mob, final List<String> commands, final int metaFlags)
137                throws java.io.IOException
138        {
139                if((commands.size()==1)&&CMSecurity.isSaveFlag(CMSecurity.SaveFlag.NOPLAYERS))
140                {
141                        if(!mob.isMonster())
142                        {
143                                CMLib.database().DBUpdatePlayer(mob);
144                                CMLib.database().DBUpdateFollowers(mob);
145                                mob.tell(L("Your player record has been updated."));
146                        }
147                        return false;
148                }
149
150                mob.location().show(mob,null,CMMsg.MSG_OK_VISUAL,L("^S<S-NAME> wave(s) <S-HIS-HER> arms...^?"));
151                String firstCommand="";
152                String lastCommand = "";
153                if(commands.size()>1)
154                {
155                        firstCommand=commands.get(1).toUpperCase();
156                        lastCommand=commands.get(commands.size()-1).toUpperCase();
157                }
158
159                if(lastCommand.equals("USERS")||lastCommand.equals("PLAYERS")||lastCommand.equals("CHARACTERS"))
160                {
161                        if(!CMSecurity.isAllowed(mob,mob.location(),CMSecurity.SecFlag.CMDROOMS))
162                        {
163                                mob.tell(L("You are not allowed to save players."));
164                                return false;
165                        }
166                        for(final Session S : CMLib.sessions().allIterable())
167                        {
168                                final MOB M=S.mob();
169                                if(M!=null)
170                                {
171                                        CMLib.database().DBUpdatePlayer(M);
172                                        CMLib.database().DBUpdateFollowers(M);
173                                }
174                        }
175                        mob.location().showHappens(CMMsg.MSG_OK_ACTION,L("A feeling of permanency envelopes everyone.\n\r"));
176                }
177                else
178                if(lastCommand.equals("ITEMS"))
179                {
180                        if(!CMSecurity.isAllowed(mob,mob.location(),CMSecurity.SecFlag.CMDROOMS))
181                        {
182                                mob.tell(L("You are not allowed to save the mobs here."));
183                                return false;
184                        }
185                        if(firstCommand.equals("AREA"))
186                        {
187                                if((mob.session()!=null)&&(mob.session().confirm(L("Doing this assumes every item in every room in this area is correctly placed.  Are you sure (N/y)?"),"N")))
188                                {
189                                        final Area A=mob.location().getArea();
190                                        boolean saved = false;
191                                        for(final Enumeration<Room> e=A.getProperMap();e.hasMoreElements();)
192                                                saved = clearSaveAndRestart(mob,e.nextElement(),SaveTask.ITEMS, true) || saved;
193                                        if(saved)
194                                                mob.location().showHappens(CMMsg.MSG_OK_ACTION,L("A feeling of permanency envelopes the area.\n\r"));
195                                }
196                                else
197                                        return false;
198                        }
199                        else
200                        if(firstCommand.equalsIgnoreCase("REWORLD"))
201                        {
202                                if((mob.session()!=null)
203                                &&(mob.session().confirm(L("This will load, and then re-save ever item in the world, for no reason.  Are you sure (N/y)?"),"N")))
204                                {
205                                        for(final Enumeration<Room> r = CMLib.map().rooms();r.hasMoreElements();)
206                                        {
207                                                Room R=r.nextElement();
208                                                if((R==null)||(R.roomID()==null)||(R.roomID().length()==0))
209                                                        continue;
210                                                if(!CMSecurity.isAllowed(mob,R,CMSecurity.SecFlag.GMODIFY))
211                                                        continue;
212                                                synchronized(("SYNC"+R.roomID()).intern())
213                                                {
214                                                        R=CMLib.map().getRoom(R);
215                                                        if((mob.session()==null)
216                                                        ||(mob.session().isStopped())
217                                                        ||(R==null)
218                                                        ||(R.getArea()==null))
219                                                                continue;
220                                                        final Area A=R.getArea();
221                                                        R=CMLib.coffeeMaker().makeNewRoomContent(R,false);
222                                                        if(R==null)
223                                                                continue;
224                                                        final Area.State oldFlag=A.getAreaState();
225                                                        A.setAreaState(Area.State.FROZEN);
226                                                        A.setAreaState(oldFlag);
227                                                        if(R.numItems()>0)
228                                                        {
229                                                                final ArrayList<Item> items=new ArrayList<Item>(R.numItems());
230                                                                for(final Enumeration<Item> i=R.items();i.hasMoreElements();)
231                                                                        items.add(i.nextElement());
232                                                                CMLib.database().DBUpdateTheseItems(R, items);
233                                                        }
234                                                        R.destroy();
235                                                }
236                                        }
237                                }
238                        }
239                        else
240                        {
241                                if(clearSaveAndRestart(mob,mob.location(),SaveTask.ITEMS, false))
242                                        mob.location().showHappens(CMMsg.MSG_OK_ACTION,L("A feeling of permanency envelopes the room.\n\r"));
243                        }
244                        Resources.removeResource("HELP_"+mob.location().getArea().Name().toUpperCase());
245                }
246                else
247                if(lastCommand.equals("ROOM"))
248                {
249                        if(!CMSecurity.isAllowed(mob,mob.location(),CMSecurity.SecFlag.CMDROOMS))
250                        {
251                                mob.tell(L("You are not allowed to save the contents here."));
252                                return false;
253                        }
254                        if(firstCommand.equals("AREA"))
255                        {
256                                if((mob.session()!=null)
257                                &&(mob.session().confirm(L("Doing this assumes every mob and item in every room in this area is correctly placed.  Are you sure (N/y)?"),"N")))
258                                {
259                                        final Area A=mob.location().getArea();
260                                        boolean saved = false;
261                                        for(final Enumeration<Room> e=A.getProperMap();e.hasMoreElements();)
262                                                saved = clearSaveAndRestart(mob,e.nextElement(),SaveTask.ALL, true) || saved;
263                                        if(saved)
264                                                mob.location().showHappens(CMMsg.MSG_OK_ACTION,L("A feeling of permanency envelopes the area.\n\r"));
265                                }
266                                else
267                                        return false;
268                        }
269                        else
270                        {
271                                if(clearSaveAndRestart(mob,mob.location(),SaveTask.ALL, false))
272                                        mob.location().show(mob,null,CMMsg.MSG_OK_ACTION,L("A feeling of permanency envelopes the room.\n\r"));
273                        }
274                        Resources.removeResource("HELP_"+mob.location().getArea().Name().toUpperCase());
275                }
276                else
277                if(lastCommand.equals("MOBS"))
278                {
279                        if(!CMSecurity.isAllowed(mob,mob.location(),CMSecurity.SecFlag.CMDROOMS))
280                        {
281                                mob.tell(L("You are not allowed to save the mobs here."));
282                                return false;
283                        }
284                        if(firstCommand.equals("AREA"))
285                        {
286                                if((mob.session()!=null)&&(mob.session().confirm(L("Doing this assumes every mob in every room in this area is correctly placed.  Are you sure (N/y)?"),"N")))
287                                {
288                                        final Area A=mob.location().getArea();
289                                        boolean saved = false;
290                                        for(final Enumeration<Room> e=A.getProperMap();e.hasMoreElements();)
291                                                saved = clearSaveAndRestart(mob,e.nextElement(),SaveTask.MOBS, true) || saved;
292                                        if(saved)
293                                                mob.location().showHappens(CMMsg.MSG_OK_ACTION,L("A feeling of permanency envelopes the area.\n\r"));
294                                }
295                                else
296                                        return false;
297
298                        }
299                        else
300                        {
301                                if(clearSaveAndRestart(mob,mob.location(),SaveTask.MOBS, false))
302                                        mob.location().showHappens(CMMsg.MSG_OK_ACTION,L("A feeling of permanency envelopes the room.\n\r"));
303                        }
304                        Resources.removeResource("HELP_"+mob.location().getArea().Name().toUpperCase());
305                }
306                else
307                if(firstCommand.equals("QUESTS")||lastCommand.equals("QUESTS"))
308                {
309                        if(!CMSecurity.isAllowed(mob,mob.location(),CMSecurity.SecFlag.CMDQUESTS))
310                        {
311                                mob.tell(L("You are not allowed to save the contents here."));
312                                return false;
313                        }
314                        CMLib.quests().save();
315                        mob.tell(L("Quest list saved."));
316                }
317                else
318                if(firstCommand.equals("FACTION"))
319                {
320                        if(!CMSecurity.isAllowed(mob,mob.location(),CMSecurity.SecFlag.CMDFACTIONS))
321                        {
322                                mob.tell(L("You are not allowed to save factions."));
323                                return false;
324                        }
325                        final String factionID = (commands.size()>2) ? CMParms.combine(commands,2) : "";
326                        final Faction F=CMLib.factions().getFaction(factionID);
327                        if(F==null)
328                        {
329                                mob.tell(L("No such faction '@x1'.",factionID));
330                                return false;
331                        }
332                        else
333                        {
334                                F.setInternalFlags(CMath.unsetb(F.getInternalFlags(), Faction.IFLAG_NEVERSAVE));
335                                CMLib.factions().resaveFaction(F);
336                                mob.tell(L("Faction @x1 saved.",F.factionID()));
337                        }
338                }
339                else
340                if(firstCommand.equals("USER")||firstCommand.equals("PLAYER")||firstCommand.equals("CHARACTER")||firstCommand.equals("CHAR"))
341                {
342                        final MOB M=CMLib.players().getPlayer(lastCommand); //omg stay this t-group
343                        if(M==null)
344                        {
345                                mob.tell(L("No user named @x1",lastCommand));
346                                return false;
347                        }
348                        CMLib.database().DBUpdatePlayer(M);
349                        if(CMLib.flags().isInTheGame(M,true))
350                                CMLib.database().DBUpdateFollowers(M);
351                        mob.location().showHappens(CMMsg.MSG_OK_ACTION,L("A feeling of permanency envelopes '@x1'.\n\r",M.name()));
352                }
353                else
354                if(CMLib.players().getPlayer(firstCommand)!=null) //omg stay this t-group
355                {
356                        final MOB M=CMLib.players().getPlayer(firstCommand);
357                        CMLib.database().DBUpdatePlayer(M);
358                        if(CMLib.flags().isInTheGame(M,true))
359                                CMLib.database().DBUpdateFollowers(M);
360                        mob.location().showHappens(CMMsg.MSG_OK_ACTION,L("A feeling of permanency envelopes '@x1'.\n\r",M.name()));
361                }
362                else
363                if(CMLib.players().getPlayer(lastCommand)!=null) //omg stay this t-group
364                {
365                        final MOB M=CMLib.players().getPlayer(lastCommand);
366                        CMLib.database().DBUpdatePlayer(M);
367                        if(CMLib.flags().isInTheGame(M,true))
368                                CMLib.database().DBUpdateFollowers(M);
369                        mob.location().showHappens(CMMsg.MSG_OK_ACTION,L("A feeling of permanency envelopes '@x1'.\n\r",M.name()));
370                }
371                else
372                {
373                        mob.tell(
374                                L("\n\rYou cannot save '@x1'. However, you might try ITEMS, USERS, [PLAYERNAME], FACTION, QUESTS, MOBS, or ROOM.",firstCommand));
375                }
376                return false;
377        }

391                         ||CMSecurity.isAllowed(mob,mob.location(),CMSecurity.SecFlag.CMDQUESTS)
392                         ||CMSecurity.isSaveFlag(CMSecurity.SaveFlag.NOPLAYERS);
393        }
