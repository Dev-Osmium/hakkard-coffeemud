055        public StringBuffer getQualifiedAbilities(final MOB viewerM,
056                                                                                          final MOB ableM,
057                                                                                          final int ofType,
058                                                                                          final int ofDomain,
059                                                                                          final String prefix,
060                                                                                          final boolean shortOnly)
061        {
062                /*
063                final HashSet<Integer> V=new HashSet<Integer>();
064                int mask=Ability.ALL_ACODES;
065                if(ofDomain>=0)
066                {
067                        mask=Ability.ALL_ACODES|Ability.ALL_DOMAINS;
068                        ofType=ofType|ofDomain;
069                }
070                V.add(Integer.valueOf(ofType));
071                &&(ofTypes.contains(Integer.valueOf(A.classificationCode()&mask)))
072                */
073
074                final int checkCode = ofType;
075                final int badDomain;
076                final int checkDomain;
077                switch(ofDomain)
078                {
079                case SKILL_CRAFTING_ONLY:
080                        badDomain = -1;
081                        checkDomain = Ability.DOMAIN_CRAFTINGSKILL;
082                        break;
083                case SKILL_EPICUREAN_ONLY:
084                        badDomain = -1;
085                        checkDomain = Ability.DOMAIN_EPICUREAN;
086                        break;
087                case SKILL_GATHERING_ONLY:
088                        badDomain = -1;
089                        checkDomain = Ability.DOMAIN_GATHERINGSKILL;
090                        break;
091                case SKILL_BUILDING_ONLY:
092                        badDomain = -1;
093                        checkDomain = Ability.DOMAIN_BUILDINGSKILL;
094                        break;
095                case SKILL_ANY:
096                case 0:
097                        badDomain = -1;
098                        checkDomain = -1;
099                        break;
100                default:
101                        badDomain = -1;
102                        checkDomain = ofDomain;
103                        break;
104                }
105
106                final Filterer<Ability> newFilter=new Filterer<Ability>()
107                {
108                        @Override
109                        public boolean passesFilter(final Ability A)
110                        {
111                                if(((checkDomain < 0) && (A.classificationCode() & Ability.ALL_ACODES) != checkCode))
112                                        return false;
113                                if((A.classificationCode() & Ability.ALL_DOMAINS) == badDomain)
114                                        return false;
115                                if((checkDomain > 0) && ((A.classificationCode() & Ability.ALL_DOMAINS) != checkDomain))
116                                        return false;
117                                return true;
118                        }
119                };
120                return getQualifiedAbilities(viewerM,ableM,newFilter,prefix,shortOnly);
121        }

123        protected Set<Integer> getQualifiedTypes(final MOB ableM)
124        {
125                final Set<Integer> set=new TreeSet<Integer>();
126                final boolean checkUnMet=ableM.charStats().getCurrentClass().showThinQualifyList();
127                for(final Enumeration<Ability> a=CMClass.abilities();a.hasMoreElements();)
128                {
129                        final Ability A=a.nextElement();
130                        final int level=CMLib.ableMapper().qualifyingLevel(ableM,A);
131                        if((CMLib.ableMapper().qualifiesByLevel(ableM,A))
132                        &&(level<(CMLib.ableMapper().qualifyingClassLevel(ableM,A)+1))
133                        &&(CMLib.ableComponents().getSpecialSkillRemainder(ableM, A).specificSkillLimit() > 0)
134                        &&(!checkUnMet || CMLib.ableMapper().getUnmetPreRequisites(ableM,A).size()==0))
135                        {
136                                final Integer acode =Integer.valueOf(A.classificationCode() & Ability.ALL_ACODES);
137                                final Integer dcode =Integer.valueOf(A.classificationCode() & Ability.ALL_DOMAINS);
138                                if(!set.contains(acode))
139                                        set.add(acode);
140                                if(!set.contains(dcode))
141                                        set.add(dcode);
142                        }
143                }
144                return set;
145        }

147        public StringBuffer getQualifiedAbilities(final MOB viewerM,
148                                                                                          final MOB ableM,
149                                                                                          final Filterer<Ability> filter,
150                                                                                          final String prefix,
151                                                                                          final boolean shortOnly)
152        {
153                int highestLevel=0;
154                final StringBuffer msg=new StringBuffer("");
155                final boolean checkUnMet=ableM.charStats().getCurrentClass().showThinQualifyList();
156                for(final Enumeration<Ability> a=CMClass.abilities();a.hasMoreElements();)
157                {
158                        final Ability A=a.nextElement();
159                        final int level=CMLib.ableMapper().qualifyingLevel(ableM,A);
160                        if((CMLib.ableMapper().qualifiesByLevel(ableM,A))
161                        &&(!CMLib.ableMapper().getSecretSkill(ableM,A.ID()))
162                        &&(level>highestLevel)
163                        &&(level<(CMLib.ableMapper().qualifyingClassLevel(ableM,A)+1))
164                        &&(filter.passesFilter(A))
165                        &&(CMLib.ableComponents().getSpecialSkillRemainder(ableM, A).specificSkillLimit() > 0)
166                        &&(ableM.fetchAbility(A.ID())==null)
167                        &&(!checkUnMet || CMLib.ableMapper().getUnmetPreRequisites(ableM,A).size()==0))
168                                highestLevel=level;
169                }
170                int col=1;
171                final int COL_LEN1=CMLib.lister().fixColWidth(3.0,viewerM);
172                final int COL_LEN2=CMLib.lister().fixColWidth(19.0,viewerM);
173                final int COL_LEN3=CMLib.lister().fixColWidth(12.0,viewerM);
174                for(int l=0;l<=highestLevel;l++)
175                {
176                        final StringBuffer thisLine=new StringBuffer("");
177                        for(final Enumeration<Ability> a=CMClass.abilities();a.hasMoreElements();)
178                        {
179                                final Ability A=a.nextElement();
180                                if((CMLib.ableMapper().qualifiesByLevel(ableM,A))
181                                   &&(CMLib.ableMapper().qualifyingLevel(ableM,A)==l)
182                                   &&(!CMLib.ableMapper().getSecretSkill(ableM,A.ID()))
183                                   &&(ableM.fetchAbility(A.ID())==null)
184                                   &&(filter.passesFilter(A))
185                                   &&(CMLib.ableComponents().getSpecialSkillRemainder(ableM, A).specificSkillLimit() > 0)
186                                   &&(!checkUnMet || CMLib.ableMapper().getUnmetPreRequisites(ableM,A).size()==0))
187                                {
188                                        thisLine.append("^N[^H"+CMStrings.padRight(""+l,COL_LEN1)+"^?] "
189                                                                                   +CMStrings.padRight("^<HELP^>"+A.name()+"^</HELP^>",COL_LEN2)+" "
190                                                                                   +CMStrings.padRight(A.requirements(viewerM),COL_LEN3));
191                                        if((++col)>2)
192                                        {
193                                                thisLine.append("\n\r");
194                                                col=1;
195                                        }
196                                        else
197                                                thisLine.append(" ");
198                                }
199                        }
200                        if(thisLine.length()>0)
201                        {
202                                if(msg.length()==0)
203                                        msg.append("\n\r^w[^H"+CMStrings.padRight(L("Lvl"),COL_LEN1)+"^?] ")
204                                                .append(CMStrings.padRight(L("Name"),COL_LEN2)+" ")
205                                                .append(CMStrings.padRight(L("Requires"),COL_LEN3)+" ")
206                                                .append("[^H"+CMStrings.padRight(L("Lvl"),COL_LEN1)+"^?] ")
207                                                .append(CMStrings.padRight(L("Name"),COL_LEN2)+" ")
208                                                .append(L("Requires")+"^N\n\r");
209                                msg.append(thisLine);
210                        }
211                }
212                if(msg.length()==0)
213                        return msg;
214                else
215                if(!msg.toString().endsWith("\n\r"))
216                        msg.append("\n\r");
217                msg.insert(0,prefix);
218                return msg;
219        }
