036public class Report extends Skills
037{
038        public Report()
039        {
040        }
041
042        private final String[] access=I(new String[]{"REPORT"});
043        @Override
044        public String[] getAccessWords()
045        {
046                return access;
047        }
048
049        @Override
050        public boolean execute(MOB mob, List<String> commands, int metaFlags)
051                throws java.io.IOException
052        {
053                if(commands.size()<2)
054                {
055                        final StringBuffer buf=new StringBuffer(
056                                                                L("say \"I have @x1/@x2 hit points, @x3/@x4 mana, @x5/@x6 move",
057                                                                        ""+mob.curState().getHitPoints(),
058                                                                        ""+mob.maxState().getHitPoints(),
059                                                                        ""+mob.curState().getMana(),
060                                                                        ""+mob.maxState().getMana(),
061                                                                        ""+mob.curState().getMovement(),
062                                                                        ""+mob.maxState().getMovement()));
063                        if((!CMSecurity.isDisabled(CMSecurity.DisFlag.EXPERIENCE))
064                        &&(!CMSecurity.isDisabled(CMSecurity.DisFlag.SHOWXP))
065                        &&!mob.charStats().getCurrentClass().expless()
066                        &&!mob.charStats().getMyRace().expless()
067                        &&(mob.getExpNeededLevel()<Integer.MAX_VALUE))
068                           buf.append(L(", and need @x1 to level",""+mob.getExpNeededLevel()));
069                        buf.append(".\"");
070                        final Command C=CMClass.getCommand("Say");
071                        if(C!=null)
072                                C.execute(mob,CMParms.parse(buf.toString()),metaFlags);
073                }
074                else
075                {
076                        final int level=parseOutLevel(commands);
077                        final String s=CMParms.combine(commands,1).toUpperCase();
078                        final StringBuffer say=new StringBuffer("");
079                        if("AFFECTS".startsWith(s)||(s.equalsIgnoreCase("ALL")))
080                        {
081
082                                final StringBuffer aff=new StringBuffer("\n\r^!I am affected by:^? ");
083                                final Command C=CMClass.getCommand("Affect");
084                                if(C!=null)
085                                        aff.append(C.executeInternal(mob,metaFlags,mob).toString());
086                                say.append(aff.toString());
087                        }
088                        if("EXPERTISES".startsWith(s)||(s.equalsIgnoreCase("ALL")))
089                        {
090
091                                final StringBuffer aff=new StringBuffer("\n\r^!My expertises are:^?\n\r");
092                                final Command C=CMClass.getCommand("Expertises");
093                                if(C!=null)
094                                        aff.append(C.executeInternal(mob,metaFlags,mob).toString());
095                                say.append(aff.toString());
096                        }
097                        if("STATS".startsWith(s)||(s.equalsIgnoreCase("ALL")))
098                        {
099                                final StringBuffer stats=new StringBuffer("");
100                                final int max=CMProps.getIntVar(CMProps.Int.BASEMAXSTAT);
101                                final CharStats CT=mob.charStats();
102                                for(final int i : CharStats.CODES.BASECODES())
103                                {
104                                        stats.append("^c" + CMStrings.capitalizeAndLower(CMStrings.limit(CharStats.CODES.NAME(i),3))+": ^w"
105                                                        +CMStrings.padRight(Integer.toString(CT.getStat(i)),2)
106                                                        +"/"+(max+CT.getStat(CharStats.CODES.toMAXBASE(i)))+", ");
107                                }
108                                say.append("\n\r^NMy stats:^? "+stats.toString());
109                        }
110                        if(s.equalsIgnoreCase("ALL"))
111                        {
112
113                                final Vector<Integer> V=new Vector<Integer>();
114                                V.add(Integer.valueOf(Ability.ACODE_THIEF_SKILL));
115                                V.add(Integer.valueOf(Ability.ACODE_SKILL));
116                                V.add(Integer.valueOf(Ability.ACODE_COMMON_SKILL));
117                                V.add(Integer.valueOf(Ability.ACODE_SPELL));
118                                V.add(Integer.valueOf(Ability.ACODE_PRAYER));
119                                V.add(Integer.valueOf(Ability.ACODE_SUPERPOWER));
120                                V.add(Integer.valueOf(Ability.ACODE_LANGUAGE));
121                                V.add(Integer.valueOf(Ability.ACODE_TECH));
122                                V.add(Integer.valueOf(Ability.ACODE_CHANT));
123                                V.add(Integer.valueOf(Ability.ACODE_SONG));
124                                say.append("\n\r^NMy skills:^? "+getAbilities(null,mob,V,Ability.ALL_ACODES,false,level));
125                        }
126                        else
127                        if("SPELLS".startsWith(s))
128                                say.append("\n\r^NMy spells:^? "+getAbilities(null,mob,Ability.ACODE_SPELL,-1,false,level));
129                        else
130                        if("SKILLS".startsWith(s))
131                        {
132                                final Vector<Integer> V=new Vector<Integer>();
133                                V.add(Integer.valueOf(Ability.ACODE_THIEF_SKILL));
134                                V.add(Integer.valueOf(Ability.ACODE_SKILL));
135                                V.add(Integer.valueOf(Ability.ACODE_TECH));
136                                V.add(Integer.valueOf(Ability.ACODE_COMMON_SKILL));
137                                say.append("\n\r^NMy skills:^? "+getAbilities(null,mob,V,Ability.ALL_ACODES,false,level));
138                        }
139                        else
140                        if("PRAYERS".startsWith(s))
141                                say.append("\n\r^NMy prayers:^? "+getAbilities(null,mob,Ability.ACODE_PRAYER,-1,false,level));
142                        else
143                        if(("POWERS".startsWith(s))||("SUPER POWERS".startsWith(s)))
144                                say.append("\n\r^NMy super powers:^? "+getAbilities(null,mob,Ability.ACODE_SUPERPOWER,-1,false,level));
145                        else
146                        if("CHANTS".startsWith(s))
147                                say.append("\n\r^NMy chants:^? "+getAbilities(null,mob,Ability.ACODE_CHANT,-1,false,level));
148                        else
149                        if("SONGS".startsWith(s))
150                                say.append("\n\r^NMy songs:^? "+getAbilities(null,mob,Ability.ACODE_SONG,-1,false,level));
151                        else
152                        if("LANGS".startsWith(s)||"LANGUAGES".startsWith(s))
153                                say.append("\n\r^NMy languages:^? "+getAbilities(null,mob,Ability.ACODE_LANGUAGE,-1,false,level));
154                        else
155                        if("TECH SKILLS".startsWith(s))
156                                say.append("\n\r^NMy tech skills:^? "+getAbilities(null,mob,Ability.ACODE_TECH,-1,false,level));
157
158                        if(say.length()==0)
159                                mob.tell(L("'@x1' is unknown.  Try SPELLS, SKILLS, PRAYERS, CHANTS, SONGS, LANGUAGES, EXPERTISES, STATS, or ALL.",s));
160                        else
161                                CMLib.commands().postSay(mob,null,say.toString(),false,false);
162                }
163                return false;
164        }
165
166        @Override
167        public double combatActionsCost(final MOB mob, final List<String> cmds)
168        {
169                return CMProps.getCommandCombatActionCost(ID());
170        }
171
172        @Override
173        public double actionsCost(final MOB mob, final List<String> cmds)
174        {
175                return CMProps.getCommandActionCost(ID());
176        }
177
178        @Override
179        public boolean canBeOrdered()
180        {
181                return true;
182        }
