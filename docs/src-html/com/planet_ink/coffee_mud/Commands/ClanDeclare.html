036public class ClanDeclare extends StdCommand
037{
038        public ClanDeclare()
039        {
040        }
041
042        private final String[]  access  = I(new String[] { "CLANDECLARE" });
043
044        @Override
045        public String[] getAccessWords()
046        {
047                return access;
048        }
049
050        @Override
051        public boolean execute(MOB mob, List<String> commands, int metaFlags)
052                throws java.io.IOException
053        {
054                if(commands.size()<3)
055                {
056                        mob.tell(L("You must specify the clans name, and a new relationship."));
057                        return false;
058                }
059                commands.set(0,getAccessWords()[0]);
060                final String rel=commands.get(commands.size()-1).toUpperCase();
061                Clan C=null;
062                Clan C2=null;
063                String clanName="";
064
065                final boolean skipChecks=mob.getClanRole(mob.Name())!=null;
066                if(skipChecks)
067                        C=mob.getClanRole(mob.Name()).first;
068
069                String clan2Name=CMParms.combine(commands,1,commands.size()-1);
070                C2=CMLib.clans().findClan(clan2Name);
071                if((C2==null)&&(C==null)&&(commands.size()>3))
072                {
073                        clanName=commands.get(1);
074                        clan2Name=CMParms.combine(commands,2,commands.size()-1);
075                }
076
077                if(C==null)
078                {
079                        for(final Pair<Clan,Integer> c : mob.clans())
080                        {
081                                if((clanName.length()==0)||(CMLib.english().containsString(c.first.getName(), clanName))
082                                &&(c.first.getAuthority(c.second.intValue(), Clan.Function.DECLARE)!=Authority.CAN_NOT_DO))
083                                {
084                                        C = c.first;
085                                        break;
086                                }
087                        }
088                }
089                if(C2==null)
090                {
091                        for(final Pair<Clan,Integer> c : mob.clans())
092                        {
093                                if(CMLib.english().containsString(c.first.getName(), clan2Name))
094                                {
095                                        C2 = c.first;
096                                        break;
097                                }
098                        }
099                }
100
101                if(C2==null)
102                {
103                        mob.tell(L("@x1 is an unknown clan.",clan2Name));
104                        return false;
105                }
106
107                if(C==null)
108                {
109                        mob.tell(L("You aren't allowed to declare @x1 on behalf of @x2.",rel.toLowerCase(),((clanName.length()==0)?"anything":clanName)));
110                        return false;
111                }
112
113                if((!C2.isRivalrous())||(!C.isRivalrous()))
114                {
115                        mob.tell(L("Relations between @x1 and @x2 are impossible.",C.getName(),C2.getName()));
116                        return false;
117                }
118
119                final StringBuffer msg=new StringBuffer("");
120                if(rel.length()>0)
121                {
122                        if(skipChecks||CMLib.clans().goForward(mob,C,commands,Clan.Function.DECLARE,false))
123                        {
124                                int newRole=-1;
125                                for(int i=0;i<Clan.REL_DESCS.length;i++)
126                                {
127                                        if(rel.equalsIgnoreCase(Clan.REL_DESCS[i]))
128                                                newRole=i;
129                                }
130                                if(newRole<0)
131                                {
132                                        mob.tell(L("'@x1' is not a valid relationship. Try WAR, HOSTILE, NEUTRAL, FRIENDLY, or ALLY.",rel));
133                                        return false;
134                                }
135                                if(C2==C)
136                                {
137                                        mob.tell(L("You can't do that."));
138                                        return false;
139                                }
140                                if(C.getClanRelations(C2.clanID())==newRole)
141                                {
142                                        mob.tell(L("@x1 is already in that state with @x2.",C.getName(),C2.getName()));
143                                        return false;
144
145                                }
146                                long last=C.getLastRelationChange(C2.clanID());
147                                if(last>(CMProps.getIntVar(CMProps.Int.TICKSPERMUDMONTH)*CMProps.getTickMillis()))
148                                {
149                                        last=last+(CMProps.getIntVar(CMProps.Int.TICKSPERMUDMONTH)*CMProps.getTickMillis());
150                                        if(System.currentTimeMillis()<last)
151                                        {
152                                                mob.tell(L("You must wait at least 1 mud month between relation changes."));
153                                                return false;
154                                        }
155                                }
156                                commands.clear();
157                                commands.add(getAccessWords()[0]);
158                                commands.add(clan2Name);
159                                commands.add(rel);
160                                if(skipChecks||CMLib.clans().goForward(mob,C,commands,Clan.Function.DECLARE,true))
161                                {
162                                        CMLib.clans().clanAnnounce(mob,L("The @x1 @x2 has declared @x3 @x4.",C.getGovernmentName(),C.clanID(),CMStrings.capitalizeAndLower(Clan.REL_STATES[newRole].toLowerCase()),C2.name()));
163                                        C.setClanRelations(C2.clanID(),newRole,System.currentTimeMillis());
164                                        C.update();
165                                        return false;
166                                }
167                        }
168                        else
169                        {
170                                msg.append(L("You aren't in the right position to declare relationships with your @x1.",C.getGovernmentName()));
171                        }
172                }
173                else
174                {
175                        mob.tell(L("You must specify the clans name, and a new relationship."));
176                        return false;
177                }
178                mob.tell(msg.toString());
179                return false;
180        }
181
182        @Override
183        public boolean canBeOrdered()
184        {
185                return false;
186        }
187
