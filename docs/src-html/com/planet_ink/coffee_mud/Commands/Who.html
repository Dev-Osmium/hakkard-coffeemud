052        public int[] getShortColWidths(MOB seer)
053        {
054                return new int[]{
055                        CMLib.lister().fixColWidth(12,seer.session()),
056                        CMLib.lister().fixColWidth(12,seer.session()),
057                        CMLib.lister().fixColWidth(7,seer.session()),
058                        CMLib.lister().fixColWidth(40,seer.session())
059                };
060        }

062        public String getHead(int[] colWidths)
063        {
064                final StringBuilder head=new StringBuilder("");
065                head.append("^x[");
066                if(!CMSecurity.isDisabled(CMSecurity.DisFlag.RACES))
067                        head.append(CMStrings.padRight(L("Race"),colWidths[0])+" ");
068                if(!CMSecurity.isDisabled(CMSecurity.DisFlag.CLASSES))
069                        head.append(CMStrings.padRight(L("Class"),colWidths[1])+" ");
070                if(!CMSecurity.isDisabled(CMSecurity.DisFlag.LEVELS))
071                        head.append(CMStrings.padRight(L("Level"),colWidths[2]));
072                head.append("] Character name^.^N\n\r");
073                return head.toString();
074        }
075
076        public StringBuffer showWhoShort(MOB who, int[] colWidths)
077        {
078                final StringBuffer msg=new StringBuffer("");
079                msg.append("[");
080                if(!CMSecurity.isDisabled(CMSecurity.DisFlag.RACES))
081                {
082                        if(who.charStats().getCurrentClass().raceless())
083                                msg.append(CMStrings.padRight(" ",colWidths[0])+" ");
084                        else
085                                msg.append(CMStrings.padRight(who.charStats().raceName(),colWidths[0])+" ");
086                }
087                String levelStr=who.charStats().displayClassLevel(who,true).trim();
088                final int x=levelStr.lastIndexOf(' ');
089                if(x>=0)
090                        levelStr=levelStr.substring(x).trim();
091                if(!CMSecurity.isDisabled(CMSecurity.DisFlag.CLASSES))
092                {
093                        if(who.charStats().getMyRace().classless())
094                                msg.append(CMStrings.padRight(" ",colWidths[1])+" ");
095                        else
096                                msg.append(CMStrings.padRight(who.charStats().displayClassName(),colWidths[1])+" ");
097                }
098                if(!CMSecurity.isDisabled(CMSecurity.DisFlag.LEVELS))
099                {
100                        if(who.charStats().getMyRace().leveless()
101                        ||who.charStats().getCurrentClass().leveless())
102                                msg.append(CMStrings.padRight(" ",colWidths[2]));
103                        else
104                                msg.append(CMStrings.padRight(levelStr,colWidths[2]));
105                }
106                String name=getWhoName(who);
107                msg.append("] "+CMStrings.padRight(name,colWidths[3]));
108                msg.append("\n\r");
109                return msg;
110        }

112        public String getWhoName(MOB seenM)
113        {
114                String name;
115                if(CMLib.flags().isCloaked(seenM))
116                        name="("+(seenM.Name().equals(seenM.name())?seenM.titledName():seenM.name())+")^N";
117                else
118                        name=((seenM.Name().equals(seenM.name())?seenM.titledName():seenM.name()))+"^N";
119                if((seenM.session()!=null)&&(seenM.session().isAfk()))
120                        name=name+(" (idle: "+CMLib.time().date2BestShortEllapsedTime(seenM.session().getIdleMillis())+")");
121                return name;
122        }
123
124        public String getPlainWhoName(MOB seenM)
125        {
126                String name;
127                if(CMLib.flags().isCloaked(seenM))
128                        name="("+seenM.Name()+")^N";
129                else
130                        name=seenM.Name()+"^N";
131                if((seenM.session()!=null)&&(seenM.session().isAfk()))
132                        name+=(" (idle: "+CMLib.time().date2BestShortEllapsedTime(seenM.session().getIdleMillis())+")");
133                return name;
134        }

136        public boolean checkWho(MOB seerM, MOB seenM, Set<String> friends, Filterer<MOB> mobFilter)
137        {
138                if((seenM!=null)
139                &&(((!CMLib.flags().isCloaked(seenM))
140                        ||((CMSecurity.isAllowedAnywhere(seerM,CMSecurity.SecFlag.CLOAK)||CMSecurity.isAllowedAnywhere(seerM,CMSecurity.SecFlag.WIZINV))&&(seerM.phyStats().level()>=seenM.phyStats().level()))))
141                &&((friends==null)||(friends.contains(seenM.Name())||(friends.contains("All"))))
142                &&((mobFilter==null)||(mobFilter.passesFilter(seenM)))
143                &&(seenM.basePhyStats().level()>0))
144                        return true;
145                return false;
146        }
147
148        public String getWho(MOB mob, Set<String> friends, boolean emptyOnNone, Filterer<MOB> mobFilter)
149        {
150                final StringBuffer msg=new StringBuffer("");
151                final int[] colWidths=getShortColWidths(mob);
152                for(final Session S : CMLib.sessions().localOnlineIterable())
153                {
154                        MOB mob2=S.mob();
155                        if((mob2!=null)&&(mob2.soulMate()!=null))
156                                mob2=mob2.soulMate();
157
158                        if(checkWho(mob,mob2,friends,mobFilter))
159                                msg.append(showWhoShort(mob2,colWidths));
160                }
161                if((emptyOnNone)&&(msg.length()==0))
162                        return "";
163                else
164                {
165                        final StringBuffer head=new StringBuffer(getHead(colWidths));
166                        head.append(msg.toString());
167                        return head.toString();
168                }
169        }

171        @Override
172        public boolean execute(MOB mob, List<String> commands, int metaFlags)
173                throws java.io.IOException
174        {
175                String mobName=CMParms.combine(commands,1);
176                if((mobName!=null)
177                &&(mob!=null)
178                &&(mobName.startsWith("@")))
179                {
180                        if((!(CMLib.intermud().i3online()))
181                        &&(!CMLib.intermud().imc2online()))
182                                mob.tell(L("Intermud is unavailable."));
183                        else
184                                CMLib.intermud().i3who(mob,mobName.substring(1));
185                        return false;
186                }
187                Set<String> friends=null;
188                if((mobName!=null)
189                &&(mob!=null)
190                &&(mobName.equalsIgnoreCase("friends"))
191                &&(mob.playerStats()!=null))
192                {
193                        friends=mob.playerStats().getFriends();
194                        mobName=null;
195                }
196
197                if((mobName!=null)
198                &&(mob!=null)
199                &&(mobName.equalsIgnoreCase("pk")
200                ||mobName.equalsIgnoreCase("pkill")
201                ||mobName.equalsIgnoreCase("playerkill")))
202                {
203                        friends=new HashSet<String>();
204                        for(final Session S : CMLib.sessions().allIterable())
205                        {
206                                final MOB mob2=S.mob();
207                                if((mob2!=null)&&(mob2.isAttributeSet(MOB.Attrib.PLAYERKILL)))
208                                        friends.add(mob2.Name());
209                        }
210                }
211                
212                if((mobName!=null)
213                &&(mob!=null)
214                &&(mobName.equalsIgnoreCase("acct")
215                        ||mobName.equalsIgnoreCase("accounts")
216                        ||mobName.equalsIgnoreCase("account"))
217                &&(CMSecurity.isAllowed(mob, mob.location(), CMSecurity.SecFlag.CMDPLAYERS))
218                &&(CMProps.isUsingAccountSystem()))
219                {
220                        int[] colWidths = new int[]{
221                                CMLib.lister().fixColWidth(20,mob.session()),
222                                CMLib.lister().fixColWidth(40,mob.session())
223                        };
224                        final StringBuilder msg=new StringBuilder("");
225                        msg.append("^x[");
226                        msg.append(CMStrings.padRight(L("Account"),colWidths[0]));
227                        msg.append(L("] Character name^.^N\n\r"));
228                        for(final Session S : CMLib.sessions().localOnlineIterable())
229                        {
230                                MOB mob2=S.mob();
231                                if((mob2!=null)&&(mob2.soulMate()!=null))
232                                        mob2=mob2.soulMate();
233
234                                if(checkWho(mob,mob2,friends,null) && (mob2!=null))
235                                {
236                                        final PlayerStats pStats2=mob2.playerStats();
237                                        final String accountName = (pStats2 != null) && (pStats2.getAccount() != null) ? pStats2.getAccount().getAccountName() : "?!?";
238                                        msg.append("["+CMStrings.padRight(accountName,colWidths[0]));
239                                        final String name=getPlainWhoName(mob2);
240                                        msg.append("] "+CMStrings.padRight(name,colWidths[1]));
241                                        msg.append("\n\r");
242                                }
243                        }
244                        mob.tell(msg.toString());
245                        return false;
246                }
247
248                final String msg = getWho(mob,friends,mobName!=null,null);
249                if((mobName!=null)&&(msg.length()==0))
250                        mob.tell(L("That person doesn't appear to be online.\n\r"));
251                else
252                        mob.tell(msg);
253                return false;
254        }

256        @Override
257        public Object executeInternal(MOB mob, int metaFlags, Object... args) throws java.io.IOException
258        {
259                if(args.length==0)
260                        return getWho(mob,null,false,null);
261                else
262                if(super.checkArguments(filterParameters, args))
263                        return getWho(mob,null,((Boolean)args[0]).booleanValue(),(Filterer)args[1]);
264                return Boolean.FALSE;
265        }

267        @Override
268        public boolean canBeOrdered()
269        {
270                return true;
271        }
