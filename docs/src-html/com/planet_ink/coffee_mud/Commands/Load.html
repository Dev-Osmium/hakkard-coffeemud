051        public final String[] combine(final String[] set1, final CMClass.CMObjectType[] set2)
052        {
053                final String[] fset=new String[set1.length+set2.length];
054                for(int x=0;x<set1.length;x++)
055                        fset[x]=set1[x];
056                for(int x=0;x<set2.length;x++)
057                        fset[set1.length+x]=set2[x].toString();
058                return fset;
059        }

061        public final String ARCHON_LIST[]=combine(new String[]{"RESOURCE","FACTION","CLASS"},CMClass.CMObjectType.values());
062
063        public final Ammunition getNextAmmunition(String type, List<Ammunition> ammos)
064        {
065                for(final Ammunition ammo : ammos)
066                {
067                        if((!ammo.amDestroyed())&&(ammo.ammunitionRemaining() > 0)&&(ammo.ammunitionType().equalsIgnoreCase(type)))
068                                return ammo;
069                }
070                return null;
071        }

073        @Override
074        public boolean execute(MOB mob, List<String> commands, int metaFlags)
075                throws java.io.IOException
076        {
077                if(mob==null)
078                        return true;
079                boolean tryArchon=CMSecurity.isAllowed(mob,mob.location(),CMSecurity.SecFlag.LOADUNLOAD);
080                if(commands.size()<3)
081                {
082                        if(tryArchon)
083                                mob.tell(L("LOAD what? Try @x1 [CLASSNAME]",CMParms.toListString(ARCHON_LIST)));
084                        else
085                                mob.tell(L("Load what where?"));
086                        return false;
087                }
088                String what=commands.get(1);
089                String name=CMParms.combine(commands,2);
090                if(tryArchon)
091                {
092                        Item I=mob.fetchWieldedItem();
093                        if((I instanceof AmmunitionWeapon)&&((AmmunitionWeapon)I).requiresAmmunition())
094                                tryArchon=false;
095                        else
096                        {
097                                I=mob.location().findItem(null, name);
098                                if((I instanceof AmmunitionWeapon)
099                                &&(((AmmunitionWeapon)I).requiresAmmunition())
100                                &&((AmmunitionWeapon)I).isFreeStanding())
101                                        tryArchon=false;
102                        }
103                        for(final String aList : ARCHON_LIST)
104                        {
105                                if(what.equalsIgnoreCase(aList))
106                                        tryArchon=true;
107                        }
108                }
109                if(!tryArchon)
110                {
111                        commands.remove(0);
112                        final XVector<String> ammoV=new XVector<String>(what);
113                        final List<Item> baseAmmoItems=CMLib.english().fetchItemList(mob,mob,null,ammoV,Wearable.FILTER_UNWORNONLY,false);
114                        final List<Ammunition> ammos=new XVector<Ammunition>();
115                        for (Item I : baseAmmoItems)
116                        {
117                                if(I instanceof Ammunition)
118                                {
119                                        ammos.add((Ammunition)I);
120                                }
121                        }
122                        if(baseAmmoItems.size()==0)
123                                mob.tell(L("You don't seem to have any ammunition like that."));
124                        else
125                        if((ammos.size()==0)&&(!what.equalsIgnoreCase("all")))
126                                mob.tell(L("You can't seem to use that as ammunition."));
127                        else
128                        {
129                                commands.remove(0);
130                                final List<Item> baseItems=CMLib.english().fetchItemList(mob,mob,null,commands,Wearable.FILTER_ANY,false);
131                                baseItems.addAll(mob.location().findItems(null,CMParms.combine(commands,0)));
132                                final List<AmmunitionWeapon> items=new XVector<AmmunitionWeapon>();
133                                for (Item I : baseItems)
134                                {
135                                        if((I instanceof AmmunitionWeapon)
136                                        &&((AmmunitionWeapon)I).requiresAmmunition())
137                                        {
138                                                if(mob.isMine(I))
139                                                        items.add((AmmunitionWeapon)I);
140                                                else
141                                                if(((AmmunitionWeapon)I).isFreeStanding())
142                                                        items.add((AmmunitionWeapon)I);
143                                        }
144                                }
145                                boolean doneOne=false;
146                                if(baseItems.size()==0)
147                                        mob.tell(L("You don't seem to have that."));
148                                else
149                                if(items.size()==0)
150                                        mob.tell(L("You can't seem to load that."));
151                                else
152                                for(final AmmunitionWeapon W : items)
153                                {
154                                        Ammunition ammunition = getNextAmmunition(W.ammunitionType(),ammos);
155                                        if(ammunition==null)
156                                                mob.tell(L("You are all out of @x1.",W.ammunitionType()));
157                                        else
158                                        {
159                                                while((ammunition != null)
160                                                &&((W.ammunitionRemaining() < W.ammunitionCapacity())||(!doneOne)))
161                                                {
162                                                        final CMMsg newMsg=CMClass.getMsg(mob,W,ammunition,CMMsg.MSG_RELOAD,L("<S-NAME> reload(s) <T-NAME> with <O-NAME>."));
163                                                        if(mob.location().okMessage(mob,newMsg))
164                                                        {
165                                                                doneOne=true;
166                                                                mob.location().send(mob,newMsg);
167                                                                ammunition = getNextAmmunition(W.ammunitionType(),ammos);
168                                                        }
169                                                        else
170                                                                break;
171                                                }
172                                        }
173                                }
174                        }
175                }
176                else
177                {
178                        if((what.equalsIgnoreCase("FACTION"))
179                        &&(CMSecurity.isAllowed(mob,mob.location(),CMSecurity.SecFlag.CMDFACTIONS)))
180                        {
181                                final Faction F=CMLib.factions().getFaction(name);
182                                if(F==null)
183                                        mob.tell(L("Faction file '@x1' was not found.",name));
184                                else
185                                        mob.tell(L("Faction '@x1' from file '@x2' was loaded.",F.name(),name));
186                                return false;
187                        }
188                        else
189                        if(what.equalsIgnoreCase("RESOURCE"))
190                        {
191                                final CMFile F=new CMFile(name,mob,CMFile.FLAG_LOGERRORS);
192                                if((!F.exists())||(!F.canRead()))
193                                        mob.tell(L("File '@x1' could not be accessed.",name));
194                                else
195                                {
196                                        final StringBuffer buf=Resources.getFileResource(name,true); // enforces its own security
197                                        if((buf==null)||(buf.length()==0))
198                                                mob.tell(L("Resource '@x1' was not found.",name));
199                                        else
200                                                mob.tell(L("Resource '@x1' was loaded.",name));
201                                }
202                        }
203                        else
204                        if(what.equalsIgnoreCase("AREA")
205                        &&(CMLib.database().DBIsAreaName(name)!=null))
206                        {
207                                final String areaName = CMLib.database().DBIsAreaName(name);
208                                if(CMLib.map().getArea(areaName)!=null)
209                                        mob.tell(L("Area '@x1' already loaded.  Either unload it first, or be happy with what you have.",name));
210                                else
211                                {
212                                        mob.tell(L("Loading area '@x1'...",areaName));
213                                        if(CMLib.database().DBReadAreaFull(areaName))
214                                        {
215                                                final Area newAreaA=CMLib.map().getArea(areaName);
216                                                for(final Enumeration<Room> r=CMLib.map().rooms();r.hasMoreElements();)
217                                                {
218                                                        final Room R=r.nextElement();
219                                                        final Area randomRoomArea = R.getArea();
220                                                        final int numDirs=Directions.NUM_DIRECTIONS();
221                                                        if(!R.getArea().Name().equals(areaName))
222                                                        {
223                                                                for(int d=0;d<numDirs;d++)
224                                                                {
225                                                                        final Room R1=R.rawDoors()[d];
226                                                                        if((R1!=null)
227                                                                        &&(R1.roomID().length()>0)
228                                                                        &&(R1.getArea()!=randomRoomArea)
229                                                                        &&(R1.getArea()!=newAreaA)
230                                                                        &&(R1.getArea().Name().equals(areaName)))
231                                                                                R.rawDoors()[d]=newAreaA.getRoom(R1.roomID());
232                                                                }
233                                                        }
234                                                }
235                                                mob.tell(L("Done."));
236                                        }
237                                        else
238                                                mob.tell(L("Fail."));
239                                }
240                                return false;
241                        }
242                        else
243                        if(CMSecurity.isASysOp(mob))
244                        {
245                                try
246                                {
247                                        if(name.toUpperCase().endsWith(".JAVA"))
248                                        {
249                                                while(name.startsWith("/"))
250                                                        name=name.substring(1);
251                                                Class<?> C=null;
252                                                Object CO=null;
253                                                try
254                                                {
255                                                        C=Class.forName("com.sun.tools.javac.Main", true, CMClass.instance());
256                                                        if(C!=null)
257                                                                CO=C.newInstance();
258                                                }
259                                                catch(final Exception e)
260                                                {
261                                                        Log.errOut("Load",e.getMessage());
262                                                }
263                                                final ByteArrayOutputStream bout=new ByteArrayOutputStream();
264                                                final PrintWriter pout=new PrintWriter(new OutputStreamWriter(bout));
265                                                if(CO==null)
266                                                {
267                                                        mob.tell(L("Unable to instantiate compiler.  You might try including your Java JDK's lib/tools.jar in your classpath next time you boot the mud."));
268                                                        return false;
269                                                }
270                                                final String[] args=new String[]{name};
271                                                if(C!=null)
272                                                {
273                                                        final java.lang.reflect.Method M=C.getMethod("compile",new Class[]{args.getClass(),PrintWriter.class});
274                                                        final Object returnVal=M.invoke(CO,new Object[]{args,pout});
275                                                        if((returnVal instanceof Integer)&&(((Integer)returnVal).intValue()!=0))
276                                                        {
277                                                                pout.flush();
278                                                                mob.tell(L("Compile failed:"));
279                                                                if(mob.session()!=null)
280                                                                        mob.session().rawOut(bout.toString());
281                                                                return false;
282                                                        }
283                                                }
284                                                name=name.substring(0,name.length()-5)+".class";
285                                        }
286
287                                        String unloadClassName=name;
288                                        if(unloadClassName.toUpperCase().endsWith(".CLASS"))
289                                                unloadClassName=unloadClassName.substring(0,unloadClassName.length()-6);
290                                        unloadClassName=unloadClassName.replace('\\','.');
291                                        unloadClassName=unloadClassName.replace('/','.');
292
293                                        if(what.equalsIgnoreCase("CLASS"))
294                                        {
295                                                final Object O=CMClass.getObjectOrPrototype(unloadClassName);
296                                                if(O!=null)
297                                                {
298                                                        final CMClass.CMObjectType x=CMClass.getObjectType(O);
299                                                        if(x!=null)
300                                                                what=x.toString();
301                                                }
302                                        }
303                                        final CMObjectType whatType=CMClass.findObjectType(what);
304                                        if(whatType==null)
305                                                mob.tell(L("Don't know how to load a '@x1'.  Try one of the following: @x2",what,CMParms.toListString(ARCHON_LIST)));
306                                        else
307                                        {
308                                                final Object O=CMClass.getObjectOrPrototype(unloadClassName);
309                                                if((O instanceof CMObject)
310                                                &&(name.toUpperCase().endsWith(".CLASS"))
311                                                &&(CMClass.delClass(whatType,(CMObject)O)))
312                                                        mob.tell(L("@x1 was unloaded.",unloadClassName));
313                                                if(CMClass.loadClass(whatType,name,false))
314                                                {
315                                                        mob.tell(L("@x1 @x2 was successfully loaded.",CMStrings.capitalizeAndLower(what),name));
316                                                        return true;
317                                                }
318                                        }
319                                }
320                                catch(final java.lang.Error err)
321                                {
322                                        mob.tell(err.getMessage());
323                                }
324                                catch(final Exception t)
325                                {
326                                        Log.errOut("Load",t.getClass().getName()+": "+t.getMessage());
327                                }
328                                mob.tell(L("@x1 @x2 was not loaded.",CMStrings.capitalizeAndLower(what),name));
329                        }
330                }
331                return false;
332        }
