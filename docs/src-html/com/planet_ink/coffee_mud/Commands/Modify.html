054        public void items(final MOB mob, final List<String> commands)
055                throws IOException
056        {
057                if(commands.size()<3)
058                {
059                        mob.tell(L("You have failed to specify the proper fields.\n\rThe format is MODIFY ITEM [ITEM NAME](@ room/[MOB NAME]) "
060                                        + "[LEVEL, ABILITY, REJUV, USES, MISC, ?] [NUMBER, TEXT]\n\r"));
061                        mob.location().showOthers(mob,null,CMMsg.MSG_OK_ACTION,L("<S-NAME> flub(s) a spell.."));
062                        return;
063                }
064
065                String itemID=(commands.get(2));
066                MOB srchMob=mob;
067                Item srchContainer=null;
068                Room srchRoom=mob.location();
069                final int x=itemID.indexOf('@');
070                if(x>0)
071                {
072                        final String rest=itemID.substring(x+1).trim();
073                        itemID=itemID.substring(0,x).trim();
074                        if(rest.equalsIgnoreCase("room"))
075                                srchMob=null;
076                        else
077                        if(rest.length()>0)
078                        {
079                                final MOB M=srchRoom.fetchInhabitant(rest);
080                                if(M==null)
081                                {
082                                        final Item I = srchRoom.findItem(null, rest);
083                                        if(I instanceof Container)
084                                                srchContainer=I;
085                                        else
086                                        {
087                                                mob.tell(L("MOB or Container '@x1' not found.",rest));
088                                                mob.location().showOthers(mob,null,CMMsg.MSG_OK_ACTION,L("<S-NAME> flub(s) a spell.."));
089                                                return;
090                                        }
091                                }
092                                else
093                                {
094                                        srchMob=M;
095                                        srchRoom=null;
096                                }
097                        }
098                }
099                String command="";
100                if(commands.size()>3)
101                        command=commands.get(3).toUpperCase();
102                String restStr="";
103                if(commands.size()>4)
104                        restStr=CMParms.combine(commands,4);
105
106                Item modItem=null;
107                if((srchMob!=null)&&(srchRoom!=null))
108                        modItem=(Item)srchRoom.fetchFromMOBRoomFavorsItems(srchMob,srchContainer,itemID,Wearable.FILTER_ANY);
109                else
110                if(srchMob!=null)
111                        modItem=srchMob.findItem(itemID);
112                else
113                if(srchRoom!=null)
114                {
115                        modItem=srchRoom.findItem(srchContainer, itemID);
116                        if(modItem==null)
117                                modItem=srchRoom.findItem(itemID);
118                }
119                if(modItem==null)
120                {
121                        Environmental E=CMLib.map().findSpaceObject(itemID,true);
122                        if(!(E instanceof Item))
123                                E=CMLib.map().findSpaceObject(itemID,false);
124                        if(E instanceof Item)
125                                modItem=(Item)E;
126                }
127                if(modItem == null)
128                {
129                        ShopKeeper SK=CMLib.coffeeShops().getShopKeeper(srchMob);
130                        if(SK!=null)
131                        {
132                                final CoffeeShop shop=(SK instanceof Librarian)?((Librarian)SK).getBaseLibrary():SK.getShop();
133                                final Environmental E=shop.getStock(itemID,mob);
134                                if(E instanceof Item)
135                                        modItem=(Item)E;
136                        }
137                        if(modItem == null)
138                        {
139                                SK=CMLib.coffeeShops().getShopKeeper(srchRoom);
140                                if(SK!=null)
141                                {
142                                        final CoffeeShop shop=(SK instanceof Librarian)?((Librarian)SK).getBaseLibrary():SK.getShop();
143                                        final Environmental E=shop.getStock(itemID,mob);
144                                        if(E instanceof Item)
145                                                modItem=(Item)E;
146                                }
147                        }
148                }
149                if(modItem==null)
150                {
151                        mob.tell(L("I don't see '@x1 here.\n\r",itemID));
152                        mob.location().showOthers(mob,null,CMMsg.MSG_OK_ACTION,L("<S-NAME> flub(s) a spell.."));
153                        return;
154                }
155                mob.location().showOthers(mob,modItem,CMMsg.MSG_OK_ACTION,L("<S-NAME> wave(s) <S-HIS-HER> hands around <T-NAMESELF>."));
156
157                final Item copyItem=(Item)modItem.copyOf();
158                if(command.equals("LEVEL"))
159                {
160                        final int newLevel=CMath.s_int(restStr);
161                        if(newLevel>=0)
162                        {
163                                modItem.basePhyStats().setLevel(newLevel);
164                                modItem.recoverPhyStats();
165                                mob.location().show(mob,null,CMMsg.MSG_OK_ACTION,L("@x1 shake(s) under the transforming power.",modItem.name()));
166                        }
167                }
168                else
169                if(command.equals("ABILITY"))
170                {
171                        final int newAbility=CMath.s_int(restStr);
172                        modItem.basePhyStats().setAbility(newAbility);
173                        modItem.recoverPhyStats();
174                        mob.location().show(mob,null,CMMsg.MSG_OK_ACTION,L("@x1 shake(s) under the transforming power.",modItem.name()));
175                }
176                else
177                if(command.equals("HEIGHT"))
178                {
179                        final int newAbility=CMath.s_int(restStr);
180                        modItem.basePhyStats().setHeight(newAbility);
181                        modItem.recoverPhyStats();
182                        mob.location().show(mob,null,CMMsg.MSG_OK_ACTION,L("@x1 shake(s) under the transforming power.",modItem.name()));
183                }
184                else
185                if(command.equals("REJUV"))
186                {
187                        final int newRejuv=CMath.s_int(restStr);
188                        if(newRejuv>0)
189                        {
190                                modItem.basePhyStats().setRejuv(newRejuv);
191                                modItem.recoverPhyStats();
192                                mob.location().show(mob,null,CMMsg.MSG_OK_ACTION,L("@x1 shake(s) under the transforming power.",modItem.name()));
193                        }
194                        else
195                        {
196                                modItem.basePhyStats().setRejuv(PhyStats.NO_REJUV);
197                                modItem.recoverPhyStats();
198                                mob.tell(L("@x1 will now never rejuvenate.",modItem.name()));
199                                mob.location().show(mob,null,CMMsg.MSG_OK_ACTION,L("@x1 shake(s) under the transforming power.",modItem.name()));
200                        }
201                }
202                else
203                if(command.equals("USES"))
204                {
205                        final int newUses=CMath.s_int(restStr);
206                        if(newUses>=0)
207                        {
208                                modItem.setUsesRemaining(newUses);
209                                mob.location().show(mob,null,CMMsg.MSG_OK_ACTION,L("@x1 shake(s) under the transforming power.",modItem.name()));
210                        }
211                }
212                else
213                if(command.equals("MISC"))
214                {
215                        if(modItem.isGeneric())
216                                CMLib.genEd().genMiscSet(mob,modItem,-1);
217                        else
218                                modItem.setMiscText(restStr);
219                        mob.location().show(mob,null,CMMsg.MSG_OK_ACTION,L("@x1 shake(s) under the transforming power.",modItem.name()));
220                }
221                else
222                if(CMLib.coffeeMaker().isAnyGenStat(modItem, command))
223                {
224                        CMLib.coffeeMaker().setAnyGenStat(modItem,command, restStr);
225                        mob.location().show(mob,null,CMMsg.MSG_OK_ACTION,L("@x1 shake(s) under the transforming power.",modItem.name()));
226                }
227                else
228                if((command.length()==0)&&(modItem.isGeneric()))
229                {
230                        CMLib.genEd().genMiscSet(mob,modItem,-1);
231                        mob.location().show(mob,null,CMMsg.MSG_OK_ACTION,L("@x1 shake(s) under the transforming power.",modItem.name()));
232                }
233                else
234                {
235                        final STreeSet<String> set=new STreeSet<String>();
236                        set.addAll(CMParms.parseCommas("LEVEL,ABILITY,HEIGHT,REJUV,USES,MISC",true));
237                        set.addAll(CMLib.coffeeMaker().getAllGenStats(modItem));
238                        mob.tell(L("...but failed to specify an aspect.  Try one of: @x1",CMParms.toListString(set)));
239                        mob.location().showOthers(mob,null,CMMsg.MSG_OK_ACTION,L("<S-NAME> flub(s) a spell.."));
240                }
241                if(!copyItem.sameAs(modItem))
242                {
243                        Log.sysOut("Items",mob.Name()+" modified item "+modItem.ID()+".");
244                        if(modItem instanceof SpaceObject)
245                        {
246                                CMLib.database().DBUpdateItem("SPACE", modItem);
247                        }
248                }
249                copyItem.destroy();
250        }

266        public void rooms(final MOB mob, final List<String> commands)
267                throws IOException
268        {
269                if(mob.location().roomID().equals(""))
270                {
271                        mob.tell(L("This command is invalid from within a GridLocaleChild room."));
272                        mob.location().showOthers(mob,null,CMMsg.MSG_OK_ACTION,L("<S-NAME> flub(s) a powerful spell."));
273                        return;
274                }
275                mob.location().showOthers(mob,null,CMMsg.MSG_OK_ACTION,L("<S-NAME> wave(s) <S-HIS-HER> hands around the room."));
276                if(commands.size()==2)
277                {
278                        final Room oldRoom=(Room)mob.location().copyOf();
279                        final Room newRoom=CMLib.genEd().modifyRoom(mob,mob.location(),-1);
280                        if((!oldRoom.sameAs(newRoom))&&(!newRoom.amDestroyed()))
281                        {
282                                CMLib.database().DBUpdateRoom(newRoom);
283                                newRoom.showHappens(CMMsg.MSG_OK_ACTION,L("There is something different about this place...\n\r"));
284                                Log.sysOut("Rooms",mob.Name()+" modified room "+newRoom.roomID()+".");
285                        }
286                        oldRoom.destroy();
287                        newRoom.getArea().fillInAreaRoom(newRoom);
288                        return;
289                }
290                if (commands.size() < 3)
291                {
292                        flunkRoomCmd(mob);
293                        return;
294                }
295
296                final String command=commands.get(2).toUpperCase();
297                String restStr="";
298                if(commands.size()>=3)
299                        restStr=CMParms.combine(commands,3);
300
301                if(command.equalsIgnoreCase("AREA"))
302                {
303                        if (commands.size() < 4)
304                        {
305                                flunkRoomCmd(mob);
306                                return;
307                        }
308                        boolean confirmed=false;
309                        String areaType="";
310                        if((commands.size()>4)&&(CMClass.getAreaType(commands.get(commands.size()-1).toString())!=null))
311                        {
312                                areaType=commands.remove(commands.size()-1).toString();
313                                restStr=CMParms.combine(commands,3);
314                                confirmed=true;
315                        }
316                        Area A=CMLib.map().getArea(restStr);
317                        boolean reid=false;
318                        if((A==null)&&(!CMSecurity.isAllowed(mob, mob.location(), CMSecurity.SecFlag.CMDAREAS)))
319                        {
320                                mob.session().println(L("Not permitted."));
321                                mob.location().showOthers(mob,null,CMMsg.MSG_OK_ACTION,L("<S-NAME> flub(s) a powerful spell."));
322                                return;
323                        }
324                        else
325                        if(A==null)
326                        {
327                                if((!mob.isMonster())&&(CMLib.map().getShip(restStr)==null))
328                                {
329                                        if(confirmed||mob.session().confirm(L("\n\rThis command will create a BRAND NEW AREA\n\r with Area code '@x1'.  Are you SURE (y/N)?",restStr),"N"))
330                                        {
331                                                int tries=0;
332                                                while((areaType.length()==0)&&((++tries)<10)&&(mob.session()!=null)&&(!mob.session().isStopped()))
333                                                {
334                                                        areaType=mob.session().prompt(L("Enter an area type to create (default=StdArea): "),L("StdArea"));
335                                                        if(CMClass.getAreaType(areaType)==null)
336                                                        {
337                                                                mob.session().println(L("Invalid area type! Valid ones are:"));
338                                                                mob.session().println(CMLib.lister().reallyList(mob,CMClass.areaTypes()).toString());
339                                                                areaType="";
340                                                        }
341                                                }
342                                                if(areaType.length()==0)
343                                                        areaType="StdArea";
344                                                A=CMClass.getAreaType(areaType);
345                                                A.setName(restStr);
346                                                CMLib.map().addArea(A);
347                                                CMLib.database().DBCreateArea(A);
348                                                mob.location().setArea(A);
349                                                CMLib.coffeeMaker().addAutoPropsToAreaIfNecessary(A);
350                                                reid=true;
351                                                mob.location().showHappens(CMMsg.MSG_OK_ACTION,L("This entire area twitches.\n\r"));
352                                        }
353                                }
354                                else
355                                {
356                                        mob.tell(L("Sorry Charlie!"));
357                                        mob.location().showOthers(mob,null,CMMsg.MSG_OK_ACTION,L("<S-NAME> flub(s) a powerful spell."));
358                                }
359                        }
360                        else
361                        {
362                                mob.location().setArea(A);
363                                if(A.getRandomProperRoom()!=null)
364                                        reid=true;
365                                else
366                                        CMLib.database().DBUpdateRoom(mob.location());
367                                mob.location().showHappens(CMMsg.MSG_OK_ACTION,L("This area twitches.\n\r"));
368                        }
369
370                        if(reid)
371                        {
372                                Room R=mob.location();
373                                final String oldID=R.roomID();
374                                synchronized(("SYNC"+R.roomID()).intern())
375                                {
376                                        R=CMLib.map().getRoom(R);
377                                        final Room reference=CMLib.map().findConnectingRoom(R);
378                                        String checkID=null;
379                                        if(A!=null)
380                                        {
381                                                if(reference!=null)
382                                                        checkID=A.getNewRoomID(reference,CMLib.map().getRoomDir(reference,R));
383                                                else
384                                                        checkID=A.getNewRoomID(R,-1);
385                                                mob.location().setRoomID(checkID);
386                                                CMLib.database().DBReCreate(R,oldID);
387                                        }
388                                }
389                        }
390                }
391                else
392                if(command.equalsIgnoreCase("NAME"))
393                {
394                        if (commands.size() < 4)
395                        {
396                                flunkRoomCmd(mob);
397                                return;
398                        }
399                        mob.location().setDisplayText(restStr);
400                        CMLib.database().DBUpdateRoom(mob.location());
401                        mob.location().showHappens(CMMsg.MSG_OK_ACTION,L("There is something different about this place...\n\r"));
402                }
403                else
404                if(command.equalsIgnoreCase("CLASS"))
405                {
406                        if (commands.size() < 4)
407                        {
408                                flunkRoomCmd(mob);
409                                return;
410                        }
411                        final Room newRoom=CMClass.getLocale(restStr);
412                        if(newRoom==null)
413                        {
414                                mob.tell(L("'@x1' is not a valid room locale.",restStr));
415                                return;
416                        }
417                        CMLib.genEd().changeRoomType(mob.location(),newRoom);
418                        mob.location().showHappens(CMMsg.MSG_OK_ACTION,L("There is something different about this place...\n\r"));
419                }
420                else
421                if((command.equalsIgnoreCase("XGRID"))&&(mob.location() instanceof GridLocale))
422                {
423                        if (commands.size() < 4)
424                        {
425                                flunkRoomCmd(mob);
426                                return;
427                        }
428                        ((GridLocale)mob.location()).setXGridSize(CMath.s_int(restStr));
429                        ((GridLocale)mob.location()).buildGrid();
430                        CMLib.database().DBUpdateRoom(mob.location());
431                        mob.location().showHappens(CMMsg.MSG_OK_ACTION,L("There is something different about this place...\n\r"));
432                }
433                else
434                if((command.equalsIgnoreCase("YGRID"))&&(mob.location() instanceof GridLocale))
435                {
436                        if (commands.size() < 4)
437                        {
438                                flunkRoomCmd(mob);
439                                return;
440                        }
441                        ((GridLocale)mob.location()).setYGridSize(CMath.s_int(restStr));
442                        ((GridLocale)mob.location()).buildGrid();
443                        CMLib.database().DBUpdateRoom(mob.location());
444                        mob.location().showHappens(CMMsg.MSG_OK_ACTION,L("There is something different about this place...\n\r"));
445                }
446                else
447                if(command.equalsIgnoreCase("DESCRIPTION"))
448                {
449                        if (commands.size() < 4)
450                        {
451                                flunkRoomCmd(mob);
452                                return;
453                        }
454                        mob.location().setDescription(restStr);
455                        CMLib.database().DBUpdateRoom(mob.location());
456                        mob.location().showHappens(CMMsg.MSG_OK_ACTION,L("The very nature of reality changes.\n\r"));
457                }
458                else
459                if(command.equalsIgnoreCase("AFFECTS"))
460                {
461                        CMLib.genEd().genAffects(mob,mob.location(),1,1);
462                        mob.location().recoverPhyStats();
463                        CMLib.database().DBUpdateRoom(mob.location());
464                        mob.location().showHappens(CMMsg.MSG_OK_ACTION,L("The very nature of reality changes.\n\r"));
465                }
466                else
467                if(command.equalsIgnoreCase("BEHAVIORS"))
468                {
469                        CMLib.genEd().genBehaviors(mob,mob.location(),1,1);
470                        mob.location().recoverPhyStats();
471                        CMLib.database().DBUpdateRoom(mob.location());
472                        mob.location().showHappens(CMMsg.MSG_OK_ACTION,L("The very nature of reality changes.\n\r"));
473                }
474                else
475                if(CMLib.coffeeMaker().isAnyGenStat(mob.location(), command))
476                {
477                        CMLib.coffeeMaker().setAnyGenStat(mob.location(),command, restStr);
478                        mob.location().recoverPhyStats();
479                        CMLib.database().DBUpdateRoom(mob.location());
480                        mob.location().showHappens(CMMsg.MSG_OK_ACTION,L("The very nature of reality changes.\n\r"));
481                }
482                else
483                {
484                        final STreeSet<String> set=new STreeSet<String>();
485                        set.addAll(CMParms.parseCommas("NAME,AREA,DESCRIPTION,AFFECTS,BEHAVIORS,CLASS,XGRID,YGRID",true));
486                        set.addAll(CMLib.coffeeMaker().getAllGenStats(mob.location()));
487                        mob.tell(L("...but failed to specify an aspect.  Try one of: @x1",CMParms.toListString(set)));
488                        mob.location().showOthers(mob,null,CMMsg.MSG_OK_ACTION,L("<S-NAME> flub(s) a spell.."));
489                        return;
490                }
491                mob.location().recoverRoomStats();
492                Log.sysOut("Rooms",mob.Name()+" modified room "+mob.location().roomID()+".");
493        }

495        public void accounts(final MOB mob, final List<String> commands)
496                throws IOException
497        {
498                mob.location().showOthers(mob,null,CMMsg.MSG_OK_ACTION,L("<S-NAME> wave(s) <S-HIS-HER> hands around the heavens."));
499                PlayerAccount theAccount = null;
500                String oldName = null;
501                if(commands.size()==2)
502                {
503                        theAccount=mob.playerStats().getAccount();
504                        oldName=theAccount.getAccountName();
505                        CMLib.genEd().modifyAccount(mob,theAccount,-1);
506                }
507                else
508                if(commands.size()<3)
509                {
510                        mob.tell(L("You have failed to specify the proper fields.\n\rThe format is MODIFY ACCOUNT ([NAME])\n\r"));
511                        mob.location().showOthers(mob,null,CMMsg.MSG_OK_ACTION,L("<S-NAME> flub(s) a powerful spell."));
512                        return;
513                }
514                else
515                {
516                        final String accountName=CMStrings.capitalizeAndLower(CMParms.combine(commands, 2));
517                        theAccount = CMLib.players().getLoadAccount(accountName);
518                        if(theAccount==null)
519                        {
520                                mob.tell(L("There is no account called '@x1'!\n\r",accountName));
521                                mob.location().showOthers(mob,null,CMMsg.MSG_OK_ACTION,L("<S-NAME> flub(s) a powerful spell."));
522                                return;
523                        }
524                        oldName=theAccount.getAccountName();
525                        CMLib.genEd().modifyAccount(mob,theAccount,-1);
526                        mob.location().recoverRoomStats();
527                }
528                Log.sysOut("Modify",mob.Name()+" modified account "+theAccount.getAccountName()+".");
529                if(!oldName.equals(theAccount.getAccountName()))
530                {
531                        final Vector<MOB> V=new Vector<MOB>();
532                        for(final Enumeration<String> es=theAccount.getPlayers();es.hasMoreElements();)
533                        {
534                                final String playerName=es.nextElement();
535                                final MOB playerM=CMLib.players().getLoadPlayer(playerName);
536                                if((playerM!=null)&&(!CMLib.flags().isInTheGame(playerM,true)))
537                                        V.add(playerM);
538                        }
539                        final PlayerAccount acc = (PlayerAccount)CMClass.getCommon("DefaultPlayerAccount");
540                        acc.setAccountName(oldName);
541                        CMLib.database().DBDeleteAccount(acc);
542                        CMLib.database().DBCreateAccount(theAccount);
543                        for(final MOB playerM : V)
544                                CMLib.database().DBUpdatePlayerPlayerStats(playerM);
545                }
546                CMLib.database().DBUpdateAccount(theAccount);
547        }

549        public void areas(final MOB mob, final List<String> commands)
550                throws IOException
551        {
552                if(mob.location()==null)
553                        return;
554                if(mob.location().getArea()==null)
555                        return;
556                Area myArea=mob.location().getArea();
557
558                String oldName=myArea.Name();
559                final Vector<Room> allMyDamnRooms=new Vector<Room>();
560                for(final Enumeration<Room> e=myArea.getCompleteMap();e.hasMoreElements();)
561                        allMyDamnRooms.add(e.nextElement());
562
563                mob.location().showOthers(mob,null,CMMsg.MSG_OK_ACTION,L("<S-NAME> wave(s) <S-HIS-HER> hands around wildly."));
564                Resources.removeResource("HELP_"+myArea.Name().toUpperCase());
