036public class Alias extends StdCommand
037{
038        private final String[]  access  = I(new String[] { "ALIAS" });
039
040        @Override
041        public String[] getAccessWords()
042        {
043                return access;
044        }
045
046        @Override
047        public boolean execute(final MOB mob, List<String> commands, int metaFlags)
048                throws java.io.IOException
049        {
050                final PlayerStats pStats=mob.playerStats();
051                final Session session=mob.session();
052                if((pStats==null)||(session==null))
053                        return false;
054                final InputCallback IC[]=new InputCallback[1];
055                IC[0]=new InputCallback(InputCallback.Type.PROMPT,"",0)
056                {
057                        @Override
058                        public void showPrompt()
059                        {
060                                final StringBuffer menu=new StringBuffer("^xAlias definitions:^.^?\n\r");
061                                final String[] aliasNames=pStats.getAliasNames();
062                                for(int i=0;i<aliasNames.length;i++)
063                                        menu.append(CMStrings.padRight((i+1)+". "+aliasNames[i],15)+": "+pStats.getAlias(aliasNames[i])+"\n\r");
064                                menu.append((aliasNames.length+1)+". Add a new alias\n\r");
065                                mob.tell(menu.toString());
066                                session.promptPrint(L("Enter a selection: "));
067                        }
068
069                        @Override
070                        public void timedOut()
071                        {
072                        }
073
074                        @Override
075                        public void callBack()
076                        {
077                                if(this.input.length()==0)
078                                        return;
079                                final int num=CMath.s_int(this.input);
080                                if(num<=0)
081                                        return;
082                                if(num<=(pStats.getAliasNames().length))
083                                {
084                                        final String selection=pStats.getAliasNames()[num-1];
085                                        session.prompt(new InputCallback(InputCallback.Type.CHOOSE,"","MD\n",0)
086                                        {
087                                                @Override
088                                                public void showPrompt()
089                                                {
090                                                        session.promptPrint(L("\n\rAlias selected '@x1'.\n\rWould you like to D)elete or M)odify this alias (d/M)? ", selection));
091                                                }
092
093                                                @Override
094                                                public void timedOut()
095                                                {
096                                                }
097
098                                                @Override
099                                                public void callBack()
100                                                {
101                                                        final String check=this.input;
102                                                        if(check.trim().length()==0)
103                                                        {
104                                                                session.prompt(IC[0].reset());
105                                                                return;
106                                                        }
107                                                        if(check.equals("D"))
108                                                        {
109                                                                pStats.delAliasName(selection);
110                                                                mob.tell(L("Alias deleted."));
111                                                                session.prompt(IC[0].reset());
112                                                                return;
113                                                        }
114                                                        modifyAlias(mob,session,pStats,selection,IC);
115                                                }
116                                        });
117                                }
118                                else
119                                {
120                                        session.prompt(new InputCallback(InputCallback.Type.PROMPT,"",0)
121                                        {
122                                                @Override
123                                                public void showPrompt()
124                                                {
125                                                        session.promptPrint(L("\n\rEnter a new alias string consisting of letters and numbers only.\n\r: "));
126                                                }
127
128                                                @Override
129                                                public void timedOut()
130                                                {
131                                                }
132
133                                                @Override
134                                                public void callBack()
135                                                {
136                                                        if(this.input.trim().length()==0)
137                                                        {
138                                                                session.prompt(IC[0].reset());
139                                                                return;
140                                                        }
141                                                        final String commandStr=this.input.toUpperCase().trim();
142                                                        if(pStats.getAlias(commandStr).length()>0)
143                                                        {
144                                                                mob.tell(L("That alias already exists.  Select it from the menu to delete or modify."));
145                                                                session.prompt(IC[0].reset());
146                                                                return;
147                                                        }
148                                                        if(CMParms.contains(access,commandStr))
149                                                        {
150                                                                mob.tell(L("You may not alias alias."));
151                                                                session.prompt(IC[0].reset());
152                                                                return;
153                                                        }
154                                                        for(int i=0;i<commandStr.length();i++)
155                                                        {
156                                                                if(!Character.isLetterOrDigit(commandStr.charAt(i)))
157                                                                {
158                                                                        mob.tell(L("Your alias name may only contain letters and numbers without spaces. "));
159                                                                        session.prompt(IC[0].reset());
160                                                                        return;
161                                                                }
162                                                        }
163                                                        pStats.addAliasName(commandStr);
164                                                        modifyAlias(mob,session,pStats,commandStr,IC);
165                                                }
166                                        });
167                                }
168                        }
169                };
170                session.prompt(IC[0]);
171                return true;
172        }
173
174        public void modifyAlias(final MOB mob, final Session session, final PlayerStats pStats, final String aliasName, final InputCallback[] IC)
175        {
176                session.prompt(new InputCallback(InputCallback.Type.PROMPT,"",0)
177                {
178                        @Override
179                        public void showPrompt()
180                        {
181                                session.safeRawPrintln(L("\n\rEnter a value for alias '@x1'.  Use ~ to separate commands. Prefix with noecho to turn off command echo.", aliasName));
182                                session.promptPrint(": ");
183                        }
184
185                        @Override
186                        public void timedOut()
187                        {
188                        }
189
190                        @Override
191                        public void callBack()
192                        {
193                                String value=this.input;
194                                value=CMStrings.replaceAll(value,"<","");
195                                value=CMStrings.replaceAll(value,"&","");
196                                if((value.length()==0)&&(pStats.getAlias(aliasName).length()>0))
197                                        mob.tell(L("(No change)"));
198                                else
199                                if(value.length()==0)
200                                {
201                                        mob.tell(L("Aborted."));
202                                        pStats.delAliasName(aliasName);
203                                }
204                                else
205                                {
206                                        pStats.setAlias(aliasName,value);
207                                        mob.tell(L("The alias was successfully changed."));
208                                }
209                                session.prompt(IC[0].reset());
210                        }
211                });
212        }
213
214        @Override
215        public boolean canBeOrdered()
216        {
217                return false;
218        }
219
220}
