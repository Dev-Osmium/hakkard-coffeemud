036public class Knock extends StdCommand
037{
038        public Knock()
039        {
040        }
041
042        private final String[] access=I(new String[]{"KNOCK"});
043        @Override
044        public String[] getAccessWords()
045        {
046                return access;
047        }
048
049        @Override
050        public boolean execute(MOB mob, List<String> commands, int metaFlags)
051                throws java.io.IOException
052        {
053                Vector<String> origCmds=new XVector<String>(commands);
054                if(commands.size()<=1)
055                {
056                        CMLib.commands().postCommandFail(mob,origCmds,L("Knock on what?"));
057                        return false;
058                }
059                final String knockWhat=CMParms.combine(commands,1).toUpperCase();
060                final int dir=CMLib.tracking().findExitDir(mob,mob.location(),knockWhat);
061                if(dir<0)
062                {
063                        final Environmental getThis=mob.location().fetchFromMOBRoomItemExit(mob,null,knockWhat,Wearable.FILTER_UNWORNONLY);
064                        if(getThis==null)
065                        {
066                                CMLib.commands().postCommandFail(mob,origCmds,L("You don't see '@x1' here.",knockWhat.toLowerCase()));
067                                return false;
068                        }
069                        final CMMsg msg=CMClass.getMsg(mob,getThis,null,CMMsg.MSG_KNOCK,CMMsg.MSG_KNOCK,CMMsg.MSG_KNOCK,L("<S-NAME> knock(s) on <T-NAMESELF>.@x1",CMLib.protocol().msp("knock.wav",50)));
070                        if(mob.location().okMessage(mob,msg))
071                                mob.location().send(mob,msg);
072
073                }
074                else
075                {
076                        Exit E=mob.location().getExitInDir(dir);
077                        if(E==null)
078                        {
079                                CMLib.commands().postCommandFail(mob,origCmds,L("Knock on what?"));
080                                return false;
081                        }
082                        if(!E.hasADoor())
083                        {
084                                CMLib.commands().postCommandFail(mob,origCmds,L("You can't knock on @x1!",E.name()));
085                                return false;
086                        }
087                        final CMMsg msg=CMClass.getMsg(mob,E,null,CMMsg.MSG_KNOCK,CMMsg.MSG_KNOCK,CMMsg.MSG_KNOCK,L("<S-NAME> knock(s) on <T-NAMESELF>.@x1",CMLib.protocol().msp("knock.wav",50)));
088                        if(mob.location().okMessage(mob,msg))
089                        {
090                                mob.location().send(mob,msg);
091                                E=mob.location().getPairedExit(dir);
092                                final Room R=mob.location().getRoomInDir(dir);
093                                if((R!=null)&&(E!=null)&&(E.hasADoor())
094                                &&(R.showOthers(mob,E,null,CMMsg.MSG_KNOCK,L("You hear a knock on <T-NAMESELF>.@x1",CMLib.protocol().msp("knock.wav",50))))
095                                &&((R.domainType()&Room.INDOORS)==Room.INDOORS))
096                                {
097                                        final Vector<Room> V=new Vector<Room>();
098                                        V.add(mob.location());
099                                        TrackingLibrary.TrackingFlags flags;
100                                        flags = CMLib.tracking().newFlags()
101                                                        .plus(TrackingLibrary.TrackingFlag.OPENONLY);
102                                        CMLib.tracking().getRadiantRooms(R,V,flags,null,5,null);
103                                        V.removeElement(mob.location());
104                                        for(int v=0;v<V.size();v++)
105                                        {
106                                                final Room R2=V.get(v);
107                                                final int dir2=CMLib.tracking().radiatesFromDir(R2,V);
108                                                if((dir2>=0)&&((R2.domainType()&Room.INDOORS)==Room.INDOORS))
109                                                {
110                                                        final Room R3=R2.getRoomInDir(dir2);
111                                                        if(((R3!=null)&&(R3.domainType()&Room.INDOORS)==Room.INDOORS))
112                                                        {
113                                                                final boolean useShipDirs=(R2 instanceof BoardableShip)||(R2.getArea() instanceof BoardableShip);
114                                                                final String inDirName=useShipDirs?CMLib.directions().getShipInDirectionName(dir2):CMLib.directions().getInDirectionName(dir2);
115                                                                R2.showHappens(CMMsg.MASK_SOUND|CMMsg.TYP_KNOCK,L("You hear a knock @x1.@x2",inDirName,CMLib.protocol().msp("knock.wav",50)));
116                                                        }
117                                                }
118                                        }
119                                }
120                        }
121                }
122                return false;
123        }
124
125        @Override
126        public double combatActionsCost(final MOB mob, final List<String> cmds)
127        {
128                return CMProps.getCommandCombatActionCost(ID());
129        }
130
131        @Override
132        public double actionsCost(final MOB mob, final List<String> cmds)
133        {
134                return CMProps.getCommandActionCost(ID());
135        }
136
137        @Override
138        public boolean canBeOrdered()
139        {
140                return true;
141        }
