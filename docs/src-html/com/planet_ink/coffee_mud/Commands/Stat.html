082        public MOB getTarget(MOB mob, String targetName, boolean quiet)
083        {
084                MOB target=null;
085                if(targetName.length()>0)
086                {
087                        target=mob.location().fetchInhabitant(targetName);
088                        if(target==null)
089                        {
090                                final Environmental t=mob.location().fetchFromRoomFavorItems(null,targetName);
091                                if((t!=null)&&(!(t instanceof MOB)))
092                                {
093                                        if(!quiet)
094                                                mob.tell(mob,t,null,L("You can't do that to <T-NAMESELF>."));
095                                        return null;
096                                }
097                        }
098                }
099                return target;
100        }

476        public int averageDamage(MOB M)
477        {
478                double total=0;
479                final double num=5000;
480                for(int i=0;i<num;i++)
481                {
482                        total+=CMLib.combat().adjustedDamage(M,null,null,0,true, false);
483                }
484                return (int)Math.round(Math.floor(total / num));
485        }
