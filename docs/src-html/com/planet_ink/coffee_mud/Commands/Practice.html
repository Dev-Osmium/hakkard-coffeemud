036public class Practice extends StdCommand
037{
038        public Practice()
039        {
040        }
041
042        private final String[] access=I(new String[]{"PRACTICE","PRAC"});
043        @Override
044        public String[] getAccessWords()
045        {
046                return access;
047        }
048
049        @Override
050        public boolean execute(MOB mob, List<String> commands, int metaFlags)
051                throws java.io.IOException
052        {
053                Vector<String> origCmds=new XVector<String>(commands);
054                if(commands.size()<2)
055                {
056                        CMLib.commands().postCommandFail(mob,origCmds,L("You have @x1 practice points.  Enter HELP PRACTICE for more information.",""+mob.getPractices()));
057                        return false;
058                }
059                commands.remove(0);
060
061                MOB teacher=null;
062                boolean triedTeacher=false;
063                String abilityName= CMParms.combine(commands,0);
064                if(commands.size()>1)
065                {
066                        for(int i=0;i<mob.location().numInhabitants();i++)
067                        {
068                                final MOB possTeach=mob.location().fetchInhabitant(i);
069                                if((possTeach!=null)&&(possTeach.findAbility(abilityName)!=null)&&(possTeach!=mob))
070                                {
071                                        teacher=possTeach;
072                                        break;
073                                }
074                        }
075                        if(teacher == null)
076                        {
077                                teacher=mob.location().fetchInhabitant(commands.get(commands.size()-1));
078                                if(teacher!=null)
079                                {
080                                        triedTeacher=true;
081                                        commands.remove(commands.size()-1);
082                                        abilityName=CMParms.combine(commands,0);
083                                }
084                        }
085                }
086
087                if(teacher==null)
088                {
089                        for(int i=0;i<mob.location().numInhabitants();i++)
090                        {
091                                final MOB possTeach=mob.location().fetchInhabitant(i);
092                                if((possTeach!=null)&&(possTeach.findAbility(abilityName)!=null)&&(possTeach!=mob))
093                                {
094                                        teacher=possTeach;
095                                        break;
096                                }
097                        }
098                }
099
100                final Ability myAbility=mob.findAbility(abilityName);
101                if(myAbility==null)
102                {
103                        CMLib.commands().postCommandFail(mob,origCmds,L("You don't seem to know @x1.",abilityName));
104                        return false;
105                }
106                
107                if((teacher==null)||(!CMLib.flags().canBeSeenBy(teacher,mob)))
108                {
109                        if(triedTeacher)
110                                CMLib.commands().postCommandFail(mob,origCmds,L("That person doesn't seem to be here."));
111                        else
112                                CMLib.commands().postCommandFail(mob,origCmds,L("There doesn't seem to be a teacher to practice with here."));
113                        return false;
114                }
115
116                if(!myAbility.isSavable())
117                {
118                        CMLib.commands().postCommandFail(mob,origCmds,L("@x1 cannot be practiced, as it is a native skill.",myAbility.name()));
119                        return false;
120                }
121
122                final Ability teacherAbility=mob.findAbility(abilityName);
123                if(teacherAbility==null)
124                {
125                        CMLib.commands().postCommandFail(mob,origCmds,L("@x1 doesn't seem to know @x2.",teacher.name(),abilityName));
126                        return false;
127                }
128
129                if(!teacherAbility.canBeTaughtBy(teacher,mob))
130                        return false;
131                if(!teacherAbility.canBePracticedBy(teacher,mob))
132                        return false;
133                CMMsg msg=CMClass.getMsg(teacher,mob,null,CMMsg.MSG_SPEAK,null);
134                if(!mob.location().okMessage(mob,msg))
135                        return false;
136                msg=CMClass.getMsg(teacher,mob,null,CMMsg.MSG_NOISYMOVEMENT,L("<S-NAME> practice(s) '@x1' with <T-NAMESELF>.",myAbility.name()));
137                if(!mob.location().okMessage(mob,msg))
138                        return false;
139                teacherAbility.practice(teacher,mob);
140                mob.location().send(mob,msg);
141                return false;
142        }
143
144        @Override
145        public double combatActionsCost(final MOB mob, final List<String> cmds)
146        {
147                return CMProps.getCommandCombatActionCost(ID());
148        }
149
150        @Override
151        public double actionsCost(final MOB mob, final List<String> cmds)
152        {
153                return CMProps.getCommandActionCost(ID());
154        }
155
156        @Override
157        public boolean canBeOrdered()
158        {
159                return false;
160        }
