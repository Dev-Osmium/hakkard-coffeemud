062        public boolean mobs(MOB mob, List<String> commands)
063        {

067                        mob.location().showOthers(mob,null,CMMsg.MSG_OK_ACTION,L("<S-NAME> flub(s) a powerful spell."));
068                        return false;
069                }
070
071                String mobID=CMParms.combine(commands,2);
072                boolean allFlag=commands.get(2).equalsIgnoreCase("all");
073                if(mobID.toUpperCase().startsWith("ALL."))
074                {
075                        allFlag=true;
076                        mobID="ALL "+mobID.substring(4);
077                }
078                if(mobID.toUpperCase().endsWith(".ALL"))
079                {
080                        allFlag=true;
081                        mobID="ALL "+mobID.substring(0,mobID.length()-4);
082                }
083                MOB deadMOB=mob.location().fetchInhabitant(mobID);
084                boolean doneSomething=false;
085                while(deadMOB!=null)
086                {
087                        if(!deadMOB.isMonster())
088                        {
089                                mob.tell(L("@x1 is a PLAYER!!\n\r",deadMOB.name()));
090                                if(!doneSomething)
091                                        mob.location().showOthers(mob,null,CMMsg.MSG_OK_ACTION,L("<S-NAME> flub(s) a powerful spell."));
092                                return false;
093                        }
094                        doneSomething=true;
095                        mob.location().showHappens(CMMsg.MSG_OK_VISUAL,L("@x1 vanishes in a puff of smoke.",deadMOB.name()));
096                        Log.sysOut("Mobs",mob.Name()+" destroyed mob "+deadMOB.Name()+".");
097                        deadMOB.destroy();
098                        mob.location().delInhabitant(deadMOB);
099                        deadMOB=mob.location().fetchInhabitant(mobID);
100                        if(!allFlag)
101                                break;
102                }
103                if(!doneSomething)
104                {
105                        mob.tell(L("I don't see '@x1 here.\n\r",mobID));
106                        mob.location().showOthers(mob,null,CMMsg.MSG_OK_ACTION,L("<S-NAME> flub(s) a powerful spell."));
107                        return false;
108                }
109                return true;
110        }

208        public Thread findThreadGroup(String threadName,ThreadGroup tGroup)
209        {
210                final int ac = tGroup.activeCount();
211                final int agc = tGroup.activeGroupCount();
212                final Thread tArray[] = new Thread [ac+1];
213                final ThreadGroup tgArray[] = new ThreadGroup [agc+1];
214
215                tGroup.enumerate(tArray,false);
216                tGroup.enumerate(tgArray,false);
217
218                for (int i = 0; i<ac; ++i)
219                {
220                        if (tArray[i] != null)
221                        {
222                                if(tArray[i].getName().equalsIgnoreCase(threadName))
223                                        return tArray[i];
224                                final Runnable R=CMLib.threads().findRunnableByThread(tArray[i]);
225                                if(R instanceof TickableGroup)
226                                {
227                                        if(((TickableGroup)R).getName().equalsIgnoreCase(threadName))
228                                                return tArray[i];
229                                        final TickClient T=((TickableGroup)R).getLastTicked();
230                                        if((T!=null)&&(T.getName().equalsIgnoreCase(threadName)))
231                                                return tArray[i];
232                                }
233                        }
234                }
235
236                if (agc > 0)
237                {
238                        for (int i = 0; i<agc; ++i)
239                        {
240                                if (tgArray[i] != null)
241                                {
242                                        final Thread t=findThreadGroup(threadName,tgArray[i]);
243                                        if(t!=null)
244                                                return t;
245                                }
246                        }
247                }
248                return null;
249        }

251        public Thread findThread(String threadName)
252        {
253                Thread t=null;
254                try
255                {
256                        ThreadGroup topTG = Thread.currentThread().getThreadGroup();
257                        while (topTG != null && topTG.getParent() != null)
258                                topTG = topTG.getParent();
259                        if (topTG != null)
260                                t=findThreadGroup(threadName,topTG);
261
262                }
263                catch (final Exception e)
264                {
265                }
266                return t;
267

270        public void rooms(MOB mob, List<String> commands)
271                throws IOException
272        {
273                final String thecmd=commands.get(0).toLowerCase();
274                if(commands.size()<3)
275                {
276                        if(thecmd.equalsIgnoreCase("UNLINK"))
277                                mob.tell(L("You have failed to specify the proper fields.\n\rThe format is UNLINK (N,S,E,W,U, or D)\n\r"));
278                        else
279                                mob.tell(L("You have failed to specify the proper fields.\n\rThe format is DESTROY ROOM ([DIRECTION],[ROOM ID])\n\r"));
280                        mob.location().showOthers(mob,null,CMMsg.MSG_OK_ACTION,L("<S-NAME> flub(s) a powerful spell."));
281                        return;
282                }
283                boolean confirmed=false;
284                if((commands.size()>3))
285                {
286                        if(commands.get(commands.size()-1).equalsIgnoreCase("CONFIRMED"))
287                        {
288                                commands.remove(commands.size()-1);
289                                confirmed=true;
290                        }
291                }
292                final String roomdir=CMParms.combine(commands,2);
293                final int direction=CMLib.directions().getGoodDirectionCode(roomdir);
294                Room deadRoom=null;
295                if(!thecmd.equalsIgnoreCase("UNLINK"))
296                        deadRoom=CMLib.map().getRoom(roomdir);
297                if((deadRoom==null)&&(direction<0))
298                {
299                        if(thecmd.equalsIgnoreCase("UNLINK"))
300                                mob.tell(L("You have failed to specify a direction.  Try (@x1).\n\r",Directions.LETTERS()));
301                        else
302                                mob.tell(L("You have failed to specify a direction.  Try a VALID ROOM ID, or (@x1).\n\r",Directions.LETTERS()));
303                        mob.location().showOthers(mob,null,CMMsg.MSG_OK_ACTION,L("<S-NAME> flub(s) a powerful spell."));
304                        return;
305                }
306                else
307                if(mob.isMonster())
308                {
309                        mob.tell(L("Sorry Charlie!"));
310                        mob.location().showOthers(mob,null,CMMsg.MSG_OK_ACTION,L("<S-NAME> flub(s) a powerful spell."));
311                        return;
312
313                }
314                if(deadRoom!=null)
315                {
316                        if(!CMSecurity.isAllowed(mob,deadRoom,CMSecurity.SecFlag.CMDROOMS))
317                        {
318                                mob.tell(L("Sorry Charlie! Not your room!"));
319                                mob.location().showOthers(mob,null,CMMsg.MSG_OK_ACTION,L("<S-NAME> flub(s) a powerful spell."));
320                                return;
321                        }
322                        if(mob.location()==deadRoom)
323                        {
324                                mob.tell(L("You dip! You have to leave this room first!"));
325                                mob.location().showOthers(mob,null,CMMsg.MSG_OK_ACTION,L("<S-NAME> flub(s) a powerful spell."));
326                                return;
327                        }
328
329                        if(!confirmed)
330                        {
331                                if(!mob.session().confirm(L("You are fixing to permanantly destroy Room \"@x1\".  Are you ABSOLUTELY SURE (y/N)",deadRoom.roomID()),"N"))
332                                        return;
333                        }
334                        CMLib.map().obliterateMapRoom(deadRoom);
335                        mob.tell(L("The sound of massive destruction rings in your ears."));
336                        mob.location().showOthers(mob,null,CMMsg.MSG_NOISE,L("The sound of massive destruction rings in your ears."));
337                        Log.sysOut("Rooms",mob.Name()+" destroyed room "+deadRoom.roomID()+".");
338                }
339                else
340                {
341                        if(!CMSecurity.isAllowed(mob,mob.location(),CMSecurity.SecFlag.CMDEXITS))
342                        {
343                                errorOut(mob);
344                                return;
345                        }
346                        Room unRoom=mob.location().rawDoors()[direction];
347                        if((unRoom!=null)&&(unRoom.getGridParent()!=null))
348                                unRoom=unRoom.getGridParent();
349                        if((mob.location().getGridParent()!=null)
350                        &&(!(mob.location() instanceof GridLocale)))
351                        {
352                                final GridLocale GL=mob.location().getGridParent();
353                                final int myX=GL.getGridChildX(mob.location());
354                                final int myY=GL.getGridChildY(mob.location());
355                                for(final Iterator<GridLocale.CrossExit> i=GL.outerExits();i.hasNext();)
356                                {
357                                        final GridLocale.CrossExit CE=i.next();
358                                        if((CE.out)
359                                        &&(CE.x==myX)
360                                        &&(CE.y==myY)
361                                        &&(CE.dir==direction))
362                                           GL.delOuterExit(CE);
363                                }
364                                CMLib.database().DBUpdateExits(GL);
365                                mob.location().rawDoors()[direction]=null;
366                                mob.location().setRawExit(direction,null);
367                        }
368                        else
369                        {
370                                mob.location().rawDoors()[direction]=null;
371                                mob.location().setRawExit(direction,null);
372                                CMLib.database().DBUpdateExits(mob.location());
373                        }
374                        if(unRoom instanceof GridLocale)
375                        {
376                                final GridLocale GL=(GridLocale)unRoom;
377                                for(final Iterator<GridLocale.CrossExit> i=GL.outerExits();i.hasNext();)
378                                {
379                                        final GridLocale.CrossExit CE=i.next();
380                                        if((!CE.out)
381                                        &&(CE.dir==direction)
382                                        &&(CE.destRoomID.equalsIgnoreCase(CMLib.map().getExtendedRoomID(mob.location()))))
383                                           GL.delOuterExit(CE);
384                                }
385                                CMLib.database().DBUpdateExits(GL);
386                        }
387                        mob.location().getArea().fillInAreaRoom(mob.location());
388                        final boolean useShipDirs=(mob.location() instanceof BoardableShip)||(mob.location().getArea() instanceof BoardableShip);
389                        final String inDirName=useShipDirs?CMLib.directions().getShipDirectionName(direction):CMLib.directions().getDirectionName(direction);
390                        mob.location().showHappens(CMMsg.MSG_OK_ACTION,L("A wall of inhibition falls @x1.",inDirName));
391                        final String dirName=useShipDirs?CMLib.directions().getShipDirectionName(direction):CMLib.directions().getDirectionName(direction);
392                        Log.sysOut("Rooms",mob.Name()+" unlinked direction "+dirName+" from room "+mob.location().roomID()+".");
393                }

460                }

463                MOB srchMob=mob;
464                Item srchContainer=null;
465                Room srchRoom=mob.location();
466                final int x=itemID.indexOf('@');
467                if(x>0)
468                {
469                        final String rest=itemID.substring(x+1).trim();
470                        itemID=itemID.substring(0,x).trim();
471                        if(rest.equalsIgnoreCase("room"))
472                                srchMob=null;
473                        else
474                        if(rest.length()>0)
475                        {
476                                final MOB M=srchRoom.fetchInhabitant(rest);
477                                if(M==null)
478                                {
479                                        final Item I = srchRoom.findItem(null, rest);
480                                        if(I instanceof Container)
481                                                srchContainer=I;
482                                        else
483                                        {
484                                                mob.tell(L("MOB or Container '@x1' not found.",rest));
485                                                mob.location().showOthers(mob,null,CMMsg.MSG_OK_ACTION,L("<S-NAME> flub(s) a spell.."));
486                                                return false;
487                                        }
488                                }
489                                else
490                                {
491                                        srchMob=M;
492                                        srchRoom=null;
493                                }
494                        }
495                }
496
497                int max=Integer.MAX_VALUE;
498                boolean allFlag=commands.get(2).equalsIgnoreCase("all");
499                if(itemID.toUpperCase().startsWith("ALL."))
500                {
501                        allFlag=true;
502                        itemID="ALL "+itemID.substring(4);
503                }
504                if(itemID.toUpperCase().endsWith(".ALL"))
505                {
506                        allFlag=true;
507                        itemID="ALL "+itemID.substring(0,itemID.length()-4);
508                }
509                boolean doneSomething=false;
510                Item deadItem=getItem(allFlag,srchRoom,srchContainer,srchMob,itemID);
511                
512                int spaceDex=itemID.indexOf(' ');
513                if((deadItem == null)&&(spaceDex>0)&&(CMath.isInteger(itemID.substring(0,spaceDex))))
514                {
515                        max=CMath.s_int(itemID.substring(0,spaceDex));
516                        itemID=itemID.substring(spaceDex+1).trim();
517                        deadItem=getItem(allFlag,srchRoom,srchContainer,srchMob,itemID);
518                        if(max>1)
519                                allFlag=true;
520                }
521                int num=0;
522                while((deadItem!=null)&&(max-->0))
523                {
524                        num++;
525                        if((num==100)||(num==1000)||(num==10000)||(num==100000)||(num==1000000))
526                                mob.location().show(mob,null,CMMsg.MSG_OK_ACTION,L("Things are disintegrating..."));
527                        else
528                        if((num<100)
529                        ||((num<1000)&&((num%100)==0))
530                        ||((num<10000)&&((num%1000)==0))
531                        ||((num<100000)&&((num%10000)==0))
532                        ||((num<1000000)&&((num%100000)==0)))
533                        {
534                                mob.location().recoverRoomStats();
535                                mob.location().show(mob,null,CMMsg.MSG_OK_ACTION,L("@x1 disintegrates!",deadItem.name()));
536                                Log.sysOut("Items",mob.Name()+" destroyed item "+deadItem.name()+".");
537                        }
538                        doneSomething=true;
539                        if(deadItem instanceof SpaceObject)
540                        {
541                                CMLib.database().DBDeleteItem("SPACE", deadItem);
542                                deadItem.destroy();
543                                CMLib.map().delObjectInSpace((SpaceObject)deadItem);
544                                deadItem=null;
545                        }
546                        else
547                        {
548                                //if(srchMob!=null)
549                                //      deadItem.setOwner(srchMob);
550                                //else
551                                //if(srchRoom!=null)
552                                //      deadItem.setOwner(srchRoom);
553                                deadItem.destroy();
554                                if(srchMob!=null)
555                                        srchMob.delItem(deadItem);
556                                mob.location().delItem(deadItem);
557                                deadItem=null;
558                                if(!allFlag)
