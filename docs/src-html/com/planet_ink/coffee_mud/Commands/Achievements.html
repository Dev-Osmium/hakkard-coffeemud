066        private Tattooable getTattooable(Agent agent, MOB mob)
067        {
068                switch(agent)
069                {
070                case PLAYER:
071                        return mob;
072                case ACCOUNT:
073                        if(mob.playerStats()!=null)
074                                return mob.playerStats().getAccount();
075                        return null;
076                }
077                return null;
078        }
079        
080        private AccountStats getStatter(Agent agent, MOB mob)
081        {
082                switch(agent)
083                {
084                case PLAYER:
085                        return mob.playerStats();
086                case ACCOUNT:
087                        if(mob.playerStats()!=null)
088                                return mob.playerStats().getAccount();
089                        return null;
090                }
091                return null;
092        }
093        
094        private List<Achievement> getLowestNumberedTattoos(final Agent agent, final Set<String> WonList)
095        {
096                List<Achievement> useList = new LinkedList<Achievement>();
097                HashSet<String> ignoredStarters = new HashSet<String>();
098                for(final Enumeration<Achievement> a=CMLib.achievements().achievements(agent);a.hasMoreElements();)
099                {
100                        final Achievement A=a.nextElement();
101                        final String tattooName = A.getTattoo();
102                        if(WonList.contains(A.getTattoo()))
103                                useList.add(A);
104                        else
105                        if((tattooName.length()>1)
106                        && Character.isDigit(tattooName.charAt(tattooName.length()-1)))
107                        {
108                                int x=tattooName.length()-1;
109                                while((x>0) && Character.isDigit(tattooName.charAt(x)))
110                                        x--;
111                                final String starter = tattooName.substring(0,x+1);
112                                if(!ignoredStarters.contains(starter))
113                                {
114                                        ignoredStarters.add(starter);
115                                        useList.add(A);
116                                }
117                        }
118                        else
119                                useList.add(A);
120                }
121                return useList;
122        }

124        public List<Achievement> getAccountAwards(final PlayerAccount account)
125        {
126                if(account != null)
127                {
128                        List<Achievement> awards=new Vector<Achievement>(1);
129                        for(final Enumeration<Tattoo> t=account.tattoos();t.hasMoreElements();)
130                        {
131                                final Tattoo T = t.nextElement();
132                                final Achievement A=CMLib.achievements().getAchievement(T.getTattooName());
133                                if(A != null)
134                                {
135                                        awards.add(A);
136                                }
137                        }
138                        return awards;
139                }
140                return new ArrayList<Achievement>(0);
141        }
142        
143        public List<Achievement> getAccountAwards(final MOB mob)
144        {
145                final PlayerStats pStats = (mob==null) ? null : mob.playerStats();
146                final PlayerAccount account = (pStats == null) ? null : pStats.getAccount();
147                if(account != null)
148                        return getAccountAwards(account);
149                return new ArrayList<Achievement>(0);
150        }

152        @Override
153        public boolean execute(final MOB mob, List<String> commands, int metaFlags) throws java.io.IOException
154        {
155                String rest = CMParms.combine(commands,1);
156                final PlayerStats pStats = mob.playerStats();
157                if (pStats == null)
158                {
159                        mob.tell(L("You don't have any achievements."));
160                        return false;
161                }
162                
163                if(CMLib.achievements().evaluateAchievements(mob))
164                {
165                        CMLib.s_sleep(4000); 
166                        // yes, I know, but if I call tick down, or some other
167                        // async method, I'll get a freaking prompt.
168                }
169                
170                MOB whoM=mob;
171                final List<String> parms = CMParms.parseSpaces(rest.toUpperCase(), true);
172                if(parms.size()>0)
173                {
174                        final String lastParm=CMStrings.capitalizeAndLower(parms.get(parms.size()-1));
175                        if(CMLib.players().playerExists(lastParm))
176                        {
177                                whoM=CMLib.players().getLoadPlayer(lastParm);
178                                parms.remove(parms.size()-1);
179                        }
180                        else
181                        {
182                                if((CMath.s_valueOf(ValidParms.class,lastParm.toUpperCase().trim()) == null)
183                                &&(CMath.s_valueOf(NonAgentList.class, lastParm.toUpperCase().trim())==null))
184                                {
185                                        mob.tell(L("There is no such player as '@x1'.",lastParm));
186                                        return false;
187                                }
188                        }
189                }
190                
191                String cmd = CMParms.combine(parms);
192                NonAgentList noAgent = (NonAgentList)CMath.s_valueOf(NonAgentList.class, cmd.toUpperCase().trim());
193                if(noAgent != null)
194                {
195                        switch(noAgent)
196                        {
197                        case RETIRE:
198                        case REMORT:
199                        {
200                                final Event E=(noAgent == NonAgentList.REMORT) ? Event.REMORT : Event.RETIRE;
201                                List<Achievement> awards = CMLib.achievements().fakeBumpAchievement(whoM, E, 1);
202                                int numAwards=0;
203                                for(Achievement A : awards)
204                                        numAwards+=A.getRewards().length;
205                                if(numAwards==0)
206                                {
207                                        mob.tell(whoM,null,null,L("<S-YOUPOSS> next "+noAgent.toString().toLowerCase()+" would grant <S-NAME> no new awards."));
208                                }
209                                else
210                                {
211                                        StringBuilder str=new StringBuilder(L("^H<S-YOUPOSS> next "+noAgent.toString().toLowerCase()+" will get the following awards:^?"));
212                                        int i=1;
213                                        for(Achievement A : awards)
214                                        {
215                                                if(A.getRewards().length>0)
216                                                        str.append(L("\n\rFrom the achievement '@x1':",A.getDisplayStr()));
217                                                for(final Award award : A.getRewards())
218                                                        str.append("\n\r"+(i++)+") "+award.getDescription());
219                                        }
220                                        str.append("\n\r");
221                                        mob.tell(mob,whoM,null,str.toString());
222                                }
223                                if(CMProps.getIntVar(CMProps.Int.COMMONACCOUNTSYSTEM)<=1)
224                                {
225                                        return false;
226                                }
227                                List<Achievement> futureAwards = getAccountAwards(whoM);
228                                if(futureAwards.size()==0)
229                                {
230                                        return false;
231                                }
232                                mob.tell(whoM,null,null,L("^HFrom <S-YOUPOSS> previous achievements. ^?"));
233                        }
234                        //$FALL-THROUGH$
235                        case FUTURE:
236                        {
237                                if(CMProps.getIntVar(CMProps.Int.COMMONACCOUNTSYSTEM)<=1)
238                                {
239                                        mob.tell(L("This system does not support new character achievement awards."));
240                                        return false;
241                                }
242                                List<Achievement> awards = this.getAccountAwards(whoM);
243                                int numAwards=0;
244                                for(Achievement A : awards)
245                                        numAwards+=A.getRewards().length;
246                                if(numAwards==0)
247                                {
248                                        mob.tell(whoM,null,null,L("<S-NAME> <S-HAS-HAVE> not won any account awards for new/future characters or remorting."));
249                                        return false;
250                                }
251                                StringBuilder str=new StringBuilder(L("<S-YOUPOSS> next character will get the following awards:"));
252                                int i=1;
253                                for(Achievement A : awards)
254                                {
255                                        if(A.getRewards().length>0)
256                                                str.append(L("\n\rFrom the achievement '@x1':",A.getDisplayStr()));
257                                        for(final Award award : A.getRewards())
258                                                str.append("\n\r"+(i++)+") "+award.getDescription());
259                                }
260                                str.append("\n\r");
261                                mob.tell(whoM,null,null,str.toString());
262                                return true;
263                        }
264                        case AWARDS:
265                        {
266                                final StringBuilder str=new StringBuilder("");
267                                int i=1;
268                                for(final Enumeration<Tattoo> t = whoM.tattoos();t.hasMoreElements();)
269                                {
270                                        final Achievement A=CMLib.achievements().getAchievement(t.nextElement().getTattooName());
271                                        if(A != null)
272                                        {
273                                                if(A.getRewards().length>0)
274                                                        str.append(L("\n\rFrom the achievement '@x1':",A.getDisplayStr()));
275                                                for(final Award award : A.getRewards())
276                                                        str.append("\n\r"+(i++)+") "+award.getDescription());
277                                        }
278                                }
279                                if(str.length()==0)
280                                {
281                                        mob.tell(whoM,null,null,L("<S-NAME> <S-HAS-HAVE> not won any achievement awards."));
282                                        return false;
283                                }
284                                else
285                                {
286                                        mob.tell(whoM,null,null,L("<S-NAME> <S-HAS-HAVE> have been granted the following achievement awards:")+str.toString());
287                                        return true;
288                                }
289                        }
290                        }
291                }
292                
293                List<AccountStats.Agent> agents = new LinkedList<AccountStats.Agent>();
294                boolean announce=false;
295                ValidLists list = ValidLists.WON;
296                for(int p=parms.size()-1;p>=0;p--)
297                {
298                        ValidParms V = (ValidParms)CMath.s_valueOf(ValidParms.class,parms.get(p).toUpperCase().trim());
299                        if(V!=null)
300                        {
301                                switch(V)
302                                {
303                                case ANNOUNCE:
304                                        announce=true;
305                                        break;
306                                case NOW:
307                                case WON:
308                                case ALL:
309                                        list = ValidLists.valueOf(V.name());
310                                        break;
311                                case CHARACTER:
312                                case PLAYER:
313                                        if(!agents.contains(AccountStats.Agent.PLAYER))
314                                                agents.add(AccountStats.Agent.PLAYER);
315                                        break;
316                                case ACCOUNT: 
317                                        if(!agents.contains(AccountStats.Agent.ACCOUNT))
318                                                agents.add(AccountStats.Agent.ACCOUNT);
319                                        break;
320                                }
321                        }
322                }
323                if(agents.size()==0)
324                {
325                        agents.add(AccountStats.Agent.ACCOUNT);
326                        agents.add(AccountStats.Agent.PLAYER);
327                }
328                
329                String prefix = "";
330                if(whoM != mob)
331                {
332                        prefix=whoM.Name()+L("'s ");
333                }
334
335                Set<String> WonList = new HashSet<String>();
336                for(Agent agent : agents)
337                {
338                        final Tattooable T = getTattooable(agent, whoM);
339                        if(T!=null)
340                        {
341                                for(final Enumeration<Achievement> a=CMLib.achievements().achievements(agent);a.hasMoreElements();)
342                                {
343                                        final Achievement A=a.nextElement();
344                                        if(T.findTattoo(A.getTattoo())!=null)
345                                                WonList.add(A.getTattoo());
346                                }
347                        }
348                }
349                final String done=L("DONE!");
350                
351                switch(list)
352                {
353                case ALL:
354                        prefix += prefix=L("All ");
355                        break;
356                case NOW:
357                        prefix += L("Progress in ");
358                        break;
359                case WON:
360                        break;
361                }
362                StringBuilder finalResponse = new StringBuilder();
363                for(Agent agent : agents)
364                {
365                        AccountStats stat = getStatter(agent,whoM);
366                        if(stat != null)
367                        {
368                                List<String> AchievedList = new Vector<String>();
369                                switch(list)
370                                {
371                                case ALL:
372                                {
373                                        final List<Achievement> useList = getLowestNumberedTattoos(agent,WonList);
374                                        int padding=done.length()+1;
375                                        for(final Iterator<Achievement> a=useList.iterator();a.hasNext();)
376                                        {
377                                                final Achievement A=a.next();
378                                                if(!WonList.contains(A.getTattoo()))
379                                                {
380                                                        AchievementLibrary.Tracker T=stat.getAchievementTracker(A, whoM);
381                                                        final int score = (T==null) ? 0 : T.getCount(whoM);
382                                                        int targetScore = A.getTargetCount();
383                                                        if(A.getEvent()==Event.STATVALUE)
384                                                                targetScore=A.isTargetFloor()?targetScore+1:targetScore-1;
385                                                        if(targetScore != Integer.MIN_VALUE)
386                                                        {
387                                                                final int len = (""+score+"/"+targetScore).length(); 
388                                                                if(len >= padding)
389                                                                        padding = len+1;
390                                                        }
391                                                }
392                                        }
393                                        for(final Iterator<Achievement> a=useList.iterator();a.hasNext();)
394                                        {
395                                                final Achievement A=a.next();
396                                                if(WonList.contains(A.getTattoo()))
397                                                        AchievedList.add(CMStrings.padRight("^H"+done+"^?", padding)+": "+A.getDisplayStr());
398                                                else
399                                                {
400                                                        AchievementLibrary.Tracker T=stat.getAchievementTracker(A, whoM);
401                                                        int score = (T==null) ? 0 : T.getCount(whoM);
402                                                        int targetScore = A.getTargetCount();
403                                                        if(A.getEvent()==Event.STATVALUE)
404                                                                targetScore=A.isTargetFloor()?targetScore+1:targetScore-1;
405                                                        if(targetScore == Integer.MIN_VALUE)
406                                                                AchievedList.add(CMStrings.padRight("^w", padding)+"^?: "+A.getDisplayStr());
407                                                        else
408                                                                AchievedList.add(CMStrings.padRight("^w"+score+"/"+targetScore, padding)+"^?: "+A.getDisplayStr());
409                                                }
410                                        }
411                                        break;
412                                }
413                                case NOW:
414                                {
415                                        int padding=done.length()+1;
416                                        final List<Achievement> useList = getLowestNumberedTattoos(agent,WonList);
417                                        for(final Iterator<Achievement> a=useList.iterator();a.hasNext();)
418                                        {
419                                                final Achievement A=a.next();
420                                                if(!WonList.contains(A.getTattoo()))
421                                                {
422                                                        AchievementLibrary.Tracker T=stat.getAchievementTracker(A, whoM);
423                                                        final int score = (T==null) ? 0 : T.getCount(whoM);
424                                                        if(score != 0)
425                                                        {
426                                                                int targetScore = A.getTargetCount();
427                                                                if(A.getEvent()==Event.STATVALUE)
428                                                                        targetScore=A.isTargetFloor()?targetScore+1:targetScore-1;
429                                                                if(targetScore != Integer.MIN_VALUE)
430                                                                {
431                                                                        final int len = (""+score+"/"+targetScore).length(); 
432                                                                        if(len >= padding)
433                                                                                padding = len+1;
434                                                                }
435                                                        }
436                                                }
437                                        }
438                                        for(final Iterator<Achievement> a=useList.iterator();a.hasNext();)
439                                        {
440                                                final Achievement A=a.next();
441                                                if(WonList.contains(A.getTattoo()))
442                                                        AchievedList.add(CMStrings.padRight("^H"+done+"^?", padding)+": "+A.getDisplayStr());
443                                                else
444                                                {
445                                                        AchievementLibrary.Tracker T=stat.getAchievementTracker(A, whoM);
446                                                        int score = (T==null) ? 0 : T.getCount(whoM);
447                                                        if(score != 0)
448                                                        {
449                                                                int targetScore = A.getTargetCount();
450                                                                if(A.getEvent()==Event.STATVALUE)
451                                                                        targetScore=A.isTargetFloor()?targetScore+1:targetScore-1;
452                                                                if(targetScore != Integer.MIN_VALUE)
453                                                                        AchievedList.add(CMStrings.padRight("^w"+score+"/"+targetScore, padding)+"^?: "+A.getDisplayStr());
454                                                                else
455                                                                        AchievedList.add(CMStrings.padRight("^w", padding)+"^?: "+A.getDisplayStr());
456                                                        }
457                                                }
458                                        }
459                                        break;
460                                }
461                                case WON:
462                                {
463                                        for(final Enumeration<Achievement> a=CMLib.achievements().achievements(agent);a.hasMoreElements();)
464                                        {
465                                                final Achievement A=a.nextElement();
466                                                if(WonList.contains(A.getTattoo()))
467                                                {
468                                                        AchievedList.add(A.getDisplayStr());
469                                                }
470                                        }
471                                        break;
472                                }
473                                }
474                                if(AchievedList.size()==0)
475                                        finalResponse .append("^H"+prefix+L(CMStrings.capitalizeAndLower(agent.name())+" Achievements: ^NNone!")+"^w\n\r\n\r");
476                                else
477                                {
478                                        finalResponse.append("^H"+prefix+L(CMStrings.capitalizeAndLower(agent.name())+" Achievements:")+"^w\n\r");
479                                        finalResponse.append(CMLib.lister().makeColumns(mob, AchievedList, null, 2).toString()+"^w\n\r\n\r");
480                                }
481                        }
482                }
483                if(finalResponse.length()==0)
484                        finalResponse.append("^H"+prefix+L("Achievements: ^NNone!")+"^w\n\r");
485                if(announce)
486                        CMLib.commands().postSay(mob, finalResponse.toString());
487                else
488                        mob.tell(finalResponse.toString());
489                
490                return false;
491        }
