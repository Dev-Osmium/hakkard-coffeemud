057        @Override
058        public boolean execute(final MOB mob, List<String> commands, int metaFlags)
059                throws java.io.IOException
060        {
061                boolean pause=false;
062                String what="MOTD";
063                if((commands!=null)&&(commands.size()>0))
064                {
065                        final String firstWord=commands.get(0).toUpperCase();
066                        if(CMParms.indexOf(this.getAccessWords(), firstWord)>0)
067                                what=firstWord;
068                        if((commands.get(commands.size()-1).equalsIgnoreCase("PAUSE")))
069                        {
070                                pause = true;
071                                commands.remove(commands.size()-1);
072                        }
073                        if(commands.size()==1)
074                        {
075                                commands.add("AGAIN");
076                        }
077                }
078                else
079                        commands=DEFAULT_CMD;
080                int max=5;
081                if((commands.size()>2)&&(CMath.isInteger(commands.get(commands.size()-1))))
082                {
083                        max=CMath.s_int(commands.get(commands.size()-1));
084                        commands.remove(commands.size()-1);
085                }
086                final String parm=CMParms.combine(commands,1).toUpperCase();
087                final boolean oldOk = "PREVIOUS".startsWith(parm) || parm.equals("OLD"); 
088                if((mob.playerStats()!=null)
089                &&(parm.equals("AGAIN")||parm.equals("NEW")||oldOk))
090                {
091                        final StringBuffer buf=new StringBuffer("");
092                        try
093                        {
094                                String msg = new CMFile(Resources.buildResourcePath("text")+"motd.txt",null).text().toString();
095                                if(msg.length()>0)
096                                {
097                                        if(msg.startsWith("<cmvp>"))
098                                                msg=new String(CMLib.webMacroFilter().virtualPageFilter(msg.substring(6).getBytes()));
099                                        buf.append(msg+"\n\r--------------------------------------\n\r");
100                                }
101
102                                final MultiList<JournalEntry> multiJournal=new MultiList<JournalEntry>();
103                                multiJournal.addAll(CMLib.database().DBReadJournalMsgsByUpdateDate("SYSTEM_NEWS", false, max));
104                                if(max>multiJournal.size())
105                                        multiJournal.addAll(CMLib.database().DBReadJournalMsgsByUpdateDate("CoffeeMud News", false, max-multiJournal.size())); // deprecated
106                                final ReverseFakeIterator<JournalEntry> entries = new ReverseFakeIterator<JournalEntry>(multiJournal);
107                                
108                                //should read these descending, trim to the max, then iterate.
109                                for(;entries.hasNext() && (max>=0); max--)
110                                {
111                                        final JournalEntry entry=entries.next();
112                                        final String from=entry.from();
113                                        final long last=entry.date();
114                                        String to=entry.to();
115                                        final String subject=entry.subj();
116                                        String message=entry.msg();
117                                        final long compdate=entry.update();
118                                        if((compdate>mob.playerStats().getLastDateTime())||(oldOk))
119                                        {
120                                                boolean allMine=to.equalsIgnoreCase(mob.Name())
121                                                                                ||from.equalsIgnoreCase(mob.Name());
122                                                if(to.toUpperCase().trim().startsWith("MASK=")&&CMLib.masking().maskCheck(to.trim().substring(5),mob,true))
123                                                {
124                                                        allMine=true;
125                                                        to=CMLib.masking().maskDesc(to.trim().substring(5),true);
126                                                }
127                                                if(to.equalsIgnoreCase("ALL")||allMine)
128                                                {
129                                                        if(message.startsWith("<cmvp>"))
130                                                                message=new String(CMLib.webMacroFilter().virtualPageFilter(message.substring(6).getBytes()));
131                                                        buf.append("\n\rNews: "+CMLib.time().date2String(last)+"\n\rFROM: "+CMStrings.padRight(from,15)+"\n\rTO  : "+CMStrings.padRight(to,15)+"\n\rSUBJ: "+subject+"\n\r"+message);
132                                                        buf.append("\n\r--------------------------------------\n\r");
133                                                }
134                                        }
135                                }
136                                final Vector<String> postalChains=new Vector<String>();
137                                final Vector<String> postalBranches=new Vector<String>();
138                                PostOffice P=null;
139                                for(final Enumeration<PostOffice> e=CMLib.map().postOffices();e.hasMoreElements();)
140                                {
141                                        P=e.nextElement();
142                                        if(!postalChains.contains(P.postalChain()))
143                                                postalChains.add(P.postalChain());
144                                        if(!postalBranches.contains(P.postalBranch()))
145                                                postalBranches.add(P.postalBranch());
146                                }
147                                if((postalChains.size()>0)&&(P!=null))
148                                {
149                                        List<PlayerData> V=CMLib.database().DBReadPlayerData(mob.Name(),postalChains);
150                                        final Map<PostOffice,int[]> res=getPostalResults(V,mob.playerStats().getLastDateTime());
151                                        for(final Iterator<PostOffice> e=res.keySet().iterator();e.hasNext();)
152                                        {
153                                                P=e.next();
154                                                final int[] ct=res.get(P);
155                                                buf.append("\n\r"+report("You have",P,ct));
156                                        }
157                                        final Map<PostOffice,int[]> res2=new Hashtable<PostOffice,int[]>();
158                                        for(final Pair<Clan,Integer> clanPair : CMLib.clans().findPrivilegedClans(mob, Clan.Function.WITHDRAW))
159                                        {
160                                                if(clanPair!=null)
161                                                {
162                                                        final Clan C=clanPair.first;
163                                                        if(C.getAuthority(clanPair.second.intValue(),Clan.Function.WITHDRAW)!=Clan.Authority.CAN_NOT_DO)
164                                                        {
165                                                                V=CMLib.database().DBReadPlayerData(C.name(),postalChains);
166                                                                if(V.size()>0)
167                                                                {
168                                                                        res2.putAll(getPostalResults(V,mob.playerStats().getLastDateTime()));
169                                                                }
170                                                        }
171                                                        for(final Iterator<PostOffice> e=res2.keySet().iterator();e.hasNext();)
172                                                        {
173                                                                P=e.next();
174                                                                final int[] ct=res2.get(P);
175                                                                buf.append("\n\r"+report("Your "+C.getGovernmentName()+" "+C.getName()+" has",P,ct));
176                                                        }
177                                                }
178                                        }
179                                        if((res.size()>0)||(res2.size()>0))
180                                                buf.append("\n\r--------------------------------------\n\r");
181                                }
182
183                                final Vector<JournalsLibrary.CommandJournal> myEchoableCommandJournals=new Vector<JournalsLibrary.CommandJournal>();
184                                for(final Enumeration<JournalsLibrary.CommandJournal> e=CMLib.journals().commandJournals();e.hasMoreElements();)
185                                {
186                                        final JournalsLibrary.CommandJournal CMJ=e.nextElement();
187                                        if((CMJ.getFlag(JournalsLibrary.CommandJournalFlags.ADMINECHO)!=null)
188                                        &&((CMSecurity.isJournalAccessAllowed(mob,CMJ.NAME()))
189                                                ||CMSecurity.isAllowed(mob,mob.location(),CMSecurity.SecFlag.LISTADMIN)))
190                                                        myEchoableCommandJournals.add(CMJ);
191                                }
192                                boolean CJseparator=false;
193                                for(int cj=0;cj<myEchoableCommandJournals.size();cj++)
194                                {
195                                        final JournalsLibrary.CommandJournal CMJ=myEchoableCommandJournals.get(cj);
196                                        final List<JournalEntry> items=CMLib.database().DBReadJournalMsgsNewerThan("SYSTEM_"+CMJ.NAME()+"S", "ALL", mob.playerStats().getLastDateTime());
197                                        if(items!=null)
198                                        {
199                                                for(int i=0;i<items.size();i++)
200                                                {
201                                                        final JournalEntry entry=items.get(i);
202                                                        final String from=entry.from();
203                                                        final String message=entry.msg();
204                                                        buf.append("\n\rNEW "+CMJ.NAME()+" from "+from+": "+message+"\n\r");
205                                                        CJseparator=true;
206                                                }
207                                        }
208                                }
209                                if(CJseparator)
210                                        buf.append("\n\r--------------------------------------\n\r");
211
212                                final boolean canReceiveRealEmail = 
213                                                (mob.playerStats()!=null)
214                                          &&(mob.playerStats().getEmail().length()>0)
215                                          &&(mob.isAttributeSet(MOB.Attrib.AUTOFORWARD))
216                                          &&((mob.playerStats().getAccount()==null)
217                                                ||(!mob.playerStats().getAccount().isSet(AccountFlag.NOAUTOFORWARD)));
218                                
219                                if(canReceiveRealEmail

222                                        final String[] queries=new String[] { mob.Name(),"ALL","MASK=%" };
223                                        final List<JournalEntry> msgs=CMLib.database().DBReadJournalMsgsByUpdateDate(CMProps.getVar(CMProps.Str.MAILBOX), false, max, queries);
224                                        for(int num=0;num<msgs.size();num++)
225                                        {
226                                                final JournalEntry thismsg=msgs.get(num);
227                                                final String to=thismsg.to();
228                                                if(to.equalsIgnoreCase("all")
229                                                ||to.equalsIgnoreCase(mob.Name())
230                                                ||(to.toUpperCase().trim().startsWith("MASK=")&&CMLib.masking().maskCheck(to.trim().substring(5),mob,true)))
231                                                {
232                                                        buf.append(L("\n\r^ZYou have mail waiting. Enter 'EMAIL BOX' to read.^?^.\n\r"));
233                                                        break;
234                                                }
235                                        }
236                                }
237
238                                if((CMSecurity.isAllowed(mob, mob.location(), CMSecurity.SecFlag.CMDPLAYERS))
239                                &&(CMProps.getBoolVar(CMProps.Bool.ACCOUNTEXPIRATION)))
240                                {
241                                        final List<String> l=CMLib.login().getExpiredAcctOrCharsList();
242                                        if(l.size()>0)
243                                        {
244                                                buf.append(L("\n\r^XThere are currently @x1 expired "+((CMProps.isUsingAccountSystem())?"accounts":"characters"),""+l.size()));
245                                                buf.append(L(".  Enter LIST EXPIRED to view them.^?^.\n\r"));
246                                        }
247                                }
248
249                                final List<Quest> qQVec=CMLib.quests().getPlayerPersistentQuests(mob);
250                                if(mob.session()!=null)
251                                {
252                                        if(buf.length()>0)
253                                        {
254                                                if(qQVec.size()>0)
255                                                        buf.append(L("\n\r^HYou are on @x1 quest(s).  Enter QUESTS to see them!.^?^.\n\r",""+qQVec.size()));
256                                                mob.session().wraplessPrintln("\n\r--------------------------------------\n\r"+buf.toString());
257                                                if (pause)
258                                                {
259                                                        mob.session().prompt(L("\n\rPress ENTER: "), 10000);
260                                                        mob.session().println("\n\r");
261                                                }
262                                        }
263                                        else
264                                        if(qQVec.size()>0)
265                                                buf.append(L("\n\r^HYou are on @x1 quest(s).  Enter QUESTS to see them!.^?^.\n\r",""+qQVec.size()));
266                                        else
267                                        if(parm.equals("AGAIN"))
268                                                mob.session().println(L("No @x1 to re-read.",what));
269                                        
270                                        // check for new commandjournal postings that require a reply-to-self...
271                                        for(final Enumeration<JournalsLibrary.CommandJournal> e=CMLib.journals().commandJournals();e.hasMoreElements();)
272                                        {
273                                                final JournalsLibrary.CommandJournal CMJ=e.nextElement();
274                                                if(CMJ.getFlag(CommandJournalFlags.MOTD)==null)
275                                                        continue;
276                                                final List<JournalEntry> items=CMLib.database().DBReadJournalMsgsNewerThan(CMJ.JOURNAL_NAME(), mob.Name(), -1);
277                                                if((items!=null)&&(items.size()>0))
278                                                {
279                                                        final Session session=mob.session();
280                                                        if(session.confirm(L("You have messages waiting response in @x1. Read now (y/N)? ", CMJ.NAME()),"N",5000))
281                                                        {
282                                                                int count=1;
283                                                                final Item journalItem=CMClass.getItem("StdJournal");
284                                                                journalItem.setName(CMJ.JOURNAL_NAME());
285                                                                journalItem.setReadableText("FILTER="+mob.Name());
286                                                                while(count<=items.size())
287                                                                {
288                                                                        final CMMsg msg2=CMClass.getMsg(mob,journalItem,null,CMMsg.MSG_READ,null,CMMsg.MSG_READ,""+count,CMMsg.MSG_READ,null);
289                                                                        msg2.setValue(1);
290                                                                        mob.phyStats().setSensesMask(mob.phyStats().sensesMask()|PhyStats.CAN_SEE_DARK);
291                                                                        journalItem.executeMsg(mob,msg2);
292                                                                        mob.recoverPhyStats();
293                                                                        if(msg2.value()==0)
294                                                                                break;
295                                                                        else
296                                                                        if(msg2.value()<0)
297                                                                                items.remove(count-1);
298                                                                        else
299                                                                        if(msg2.value()>0)
300                                                                                count++;
301                                                                }
302                                                        }
303                                                }
304                                        }
305                                }
306                        }
307                        catch(final HTTPRedirectException e)
308                        {
309                        }
310                        return false;
311                }
312                if(parm.equals("ON"))
313                {
314                        if(mob.isAttributeSet(MOB.Attrib.DAILYMESSAGE))
315                        {
316                                mob.setAttribute(MOB.Attrib.DAILYMESSAGE,false);
317                                mob.tell(L("The daily messages have been turned on."));
318                        }
319                        else
320                        {
321                                mob.tell(L("The daily messages are already on."));
322                        }
323                }
324                else
325                if(parm.equals("OFF"))
326                {
327                        if(!mob.isAttributeSet(MOB.Attrib.DAILYMESSAGE))
328                        {
329                                mob.setAttribute(MOB.Attrib.DAILYMESSAGE,true);
330                                mob.tell(L("The daily messages have been turned off."));
331                        }
332                        else
333                        {
334                                mob.tell(L("The daily messages are already off."));
335                        }
336                }
337                else
338                {
339                        mob.tell(L("'@x1' is not a valid parameter.  Try ON, OFF, PREVIOUS, or AGAIN.",parm));
340                }
341                return false;
342        }

344        private String report(String whom, PostOffice P, int[] ct)
345        {
346                String branchName=P.postalBranch();
347                if((P instanceof MOB)&&(((MOB)P).getStartRoom()!=null))
348                        branchName=((MOB)P).getStartRoom().getArea().Name();
349                else
350                {
351                        final int x=branchName.indexOf('#');
352                        if(x>=0)
353                                branchName=branchName.substring(0,x);
354                }
355                if(ct[0]>0)
356                        return L("@x1  @x2 new of @x3 items at the @x4 branch of the @x5 post office.",whom,""+ct[0],""+ct[1],branchName,P.postalChain());
357                return L("@x1 @x2 items still waiting at the @x3 branch of the @x4 post office.",whom,""+ct[1],branchName,P.postalChain());
358        }

360        private Map<PostOffice,int[]> getPostalResults(List<PlayerData> mailData, long newTimeDate)
361        {
362                final Hashtable<PostOffice,int[]> results=new Hashtable<PostOffice,int[]>();
363                PostOffice P=null;
364                for(int i=0;i<mailData.size();i++)
365                {
366                        final DatabaseEngine.PlayerData letter=mailData.get(i);
367                        final String chain=letter.section();
368                        String branch=letter.key();
369                        final int x=branch.indexOf(';');
370                        if(x<0)
371                                continue;
372                        branch=branch.substring(0,x);
373                        P=CMLib.map().getPostOffice(chain,branch);
374                        if(P==null)
375                                continue;
376                        final PostOffice.MailPiece pieces=P.parsePostalItemData(letter.xml());
377                        int[] ct=results.get(P);
378                        if(ct==null)
379                        {
380                                ct=new int[2];
381                                results.put(P,ct);
382                        }
383                        ct[1]++;
384                        if(CMath.s_long(pieces.time)>newTimeDate)
385                                ct[0]++;
386                }
387                return results;
388        }
