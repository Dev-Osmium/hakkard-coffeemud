036public class Fill extends StdCommand
037{
038        public Fill()
039        {
040        }
041
042        private final String[] access=I(new String[]{"FILL"});
043        @Override
044        public String[] getAccessWords()
045        {
046                return access;
047        }
048
049        @Override
050        public boolean execute(MOB mob, List<String> commands, int metaFlags)
051                throws java.io.IOException
052        {
053                Vector<String> origCmds=new XVector<String>(commands);
054                if(commands.size()<2)
055                {
056                        CMLib.commands().postCommandFail(mob,origCmds,L("Fill what, from what?"));
057                        return false;
058                }
059                commands.remove(0);
060                final String testFill=CMParms.combine(commands,0);
061                final Environmental fillThisItem=mob.location().fetchFromRoomFavorItems(null,testFill);
062                if((fillThisItem instanceof Container)
063                &&(!CMLib.flags().isGettable((Container)fillThisItem))
064                &&(((Container)fillThisItem).material()==RawMaterial.RESOURCE_DUST))
065                {
066                        final String fillMsg="<S-NAME> fill(s) in <T-NAMESELF>.";
067                        final CMMsg msg=CMClass.getMsg(mob,fillThisItem,null,CMMsg.MSG_CLOSE,fillMsg,testFill,fillMsg);
068                        if(mob.location().okMessage(msg.source(),msg))
069                                mob.location().send(msg.source(),msg);
070                        return false;
071                }
072
073                if((commands.size()<2)&&(!(mob.location() instanceof Drink)))
074                {
075                        CMLib.commands().postCommandFail(mob,origCmds,L("From what should I fill the @x1?",commands.get(0)));
076                        return false;
077                }
078                Environmental fillFromThis=null;
079                if((commands.size()==1)&&(mob.location() instanceof Drink))
080                        fillFromThis=mob.location();
081                else
082                {
083                        int fromDex=commands.size()-1;
084                        for(int i=commands.size()-2;i>=1;i--)
085                        {
086                                if(commands.get(i).equalsIgnoreCase("from"))
087                                {
088                                        fromDex=i;
089                                        commands.remove(i);
090                                }
091                        }
092                        final String thingToFillFrom=CMParms.combine(commands,fromDex);
093                        fillFromThis=mob.location().fetchFromMOBRoomFavorsItems(mob,null,thingToFillFrom,Wearable.FILTER_ANY);
094                        if((fillFromThis==null)||(!CMLib.flags().canBeSeenBy(fillFromThis,mob)))
095                        {
096                                CMLib.commands().postCommandFail(mob,origCmds,L("I don't see @x1 here.",thingToFillFrom));
097                                return false;
098                        }
099                        while(commands.size()>=(fromDex+1))
100                                commands.remove(commands.size()-1);
101                }
102
103                final int maxToFill=CMLib.english().calculateMaxToGive(mob,commands,true,mob,false);
104                if(maxToFill<0)
105                        return false;
106
107                String thingToFill=CMParms.combine(commands,0);
108                int addendum=1;
109                final Vector<Environmental> V=new Vector<Environmental>();
110                boolean allFlag=(commands.size()>0)?commands.get(0).equalsIgnoreCase("all"):false;
111                if(thingToFill.toUpperCase().startsWith("ALL."))
112                {
113                        allFlag=true;
114                        thingToFill="ALL "+thingToFill.substring(4);
115                }
116                if(thingToFill.toUpperCase().endsWith(".ALL"))
117                {
118                        allFlag=true;
119                        thingToFill="ALL "+thingToFill.substring(0,thingToFill.length()-4);
120                }
121                boolean doBugFix = true;
122                String addendumStr="";
123                while(doBugFix || ((allFlag)&&(maxToFill<addendum)))
124                {
125                        doBugFix=false;
126                        final Item fillThis=mob.findItem(null,thingToFill+addendumStr);
127                        if(fillThis==null)
128                                break;
129                        if((CMLib.flags().canBeSeenBy(fillThis,mob))
130                        &&(!V.contains(fillThis)))
131                                V.add(fillThis);
132                        addendumStr="."+(++addendum);
133                }
134
135                if(V.size()==0)
136                        CMLib.commands().postCommandFail(mob,origCmds,L("You don't seem to have '@x1'.",thingToFill));
137                else
138                for(int i=0;i<V.size();i++)
139                {
140                        final Environmental fillThis=V.get(i);
141                        final CMMsg fillMsg=CMClass.getMsg(mob,fillThis,fillFromThis,CMMsg.MSG_FILL,L("<S-NAME> fill(s) <T-NAME> from <O-NAME>."));
142                        if((!mob.isMine(fillThis))&&(fillThis instanceof Item))
143                        {
144                                if(CMLib.commands().postGet(mob,null,(Item)fillThis,false))
145                                {
146                                        if(mob.location().okMessage(mob,fillMsg))
147                                                mob.location().send(mob,fillMsg);
148                                        else
149                                                CMLib.commands().postCommandRejection(mob, fillThis, fillFromThis, origCmds);
150                                }
151                        }
152                        else
153                        if(mob.location().okMessage(mob,fillMsg))
154                                mob.location().send(mob,fillMsg);
155                        else
156                                CMLib.commands().postCommandRejection(mob, fillThis, fillFromThis, origCmds);
157                }
158                return false;
159        }
160
161        @Override
162        public double combatActionsCost(final MOB mob, final List<String> cmds)
163        {
164                return CMProps.getCommandCombatActionCost(ID());
165        }
166
167        @Override
168        public double actionsCost(final MOB mob, final List<String> cmds)
169        {
170                return CMProps.getCommandActionCost(ID());
171        }
172
173        @Override
174        public boolean canBeOrdered()
175        {
176                return true;
177        }
