051        protected void whereAdd(DVector V, Area area, int i)
052        {
053                if(V.contains(area))
054                        return;
055
056                for(int v=0;v<V.size();v++)
057                {
058                        if(((Integer)V.get(v,2)).intValue()>i)
059                        {
060                                V.insertElementAt(v,area,Integer.valueOf(i));
061                                return;
062                        }
063                }
064                V.add(area,Integer.valueOf(i));
065        }

067        public boolean canShowTo(MOB showTo, MOB show)
068        {
069                if((show!=null)
070                &&(show.session()!=null)
071                &&(showTo!=null)
072                &&(((show.phyStats().disposition()&PhyStats.IS_CLOAKED)==0)
073                        ||((CMSecurity.isAllowedAnywhere(showTo,CMSecurity.SecFlag.CLOAK)||CMSecurity.isAllowedAnywhere(showTo,CMSecurity.SecFlag.WIZINV))
074                                &&(showTo.phyStats().level()>=show.phyStats().level()))))
075                        return true;
076                return false;
077        }

503                                        whereAdd(mobsVec,A,A.getAreaIStats()[Area.Stats.POPULATION.ordinal()]);
504
505                                        final int align=A.getAreaIStats()[Area.Stats.MED_ALIGNMENT.ordinal()];
506                                        final int alignDiff=((int)Math.abs((double)(alignment-align)));
507                                        whereAdd(alignVec,A,alignDiff);
508                                }
509                        }
510                        final StringBuffer msg=new StringBuffer(L("You are currently in: ^H@x1^?\n\r",mob.location().getArea().name()));
511                        if((!CMSecurity.isDisabled(CMSecurity.DisFlag.ROOMVISITS))
512                        &&(mob.playerStats()!=null))
513                        {
514                                msg.append(L("You have explored @x1% of this area and @x2% of the world.\n\r",
515                                                ""+mob.playerStats().percentVisited(mob,mob.location().getArea()),
516                                                ""+mob.playerStats().percentVisited(mob,null)));
517                        }
518                        final DVector scores=new DVector(2);
519                        for(final Enumeration<Area> a=CMLib.map().areas();a.hasMoreElements();)
520                        {
521                                final Area A=a.nextElement();
522                                if(CMLib.flags().canAccess(mob,A))
523                                {
524                                        int index=levelsVec.indexOf(A);
525                                        if(index>=0)
526                                        {
527                                                final Integer I=(Integer)levelsVec.get(index,2);
528                                                if((I!=null)&&(I.intValue()!=0))
529                                                {
530                                                        int score=(index+1);
531                                                        index=mobsVec.indexOf(A);
532                                                        if(index>=0)
533                                                                score+=(index+1);
534
535                                                        index=alignVec.indexOf(A);
536                                                        if(index>=0)
537                                                                score+=(index+1);
538                                                        whereAdd(scores,A,score);
539                                                }
540                                        }
541                                }
542                        }
543                        if(scores.size()==0)
544                                msg.append(L("There appear to be no other areas on this world.\n\r"));
545                        else
546                        {
547                                msg.append(L("\n\r^HThe best areas for you to try appear to be: ^?\n\r\n\r"));
548                                msg.append("^x"+CMStrings.padRight(L("Area Name"),35)+CMStrings.padRight(L("Level"),6)+CMStrings.padRight(L("Alignment"),20)+CMStrings.padRight(L("Pop"),10)+"^.^?\n\r");
549                                final List<Area> finalScoreList = new ArrayList<Area>();
550                                for(int i=scores.size()-1;((i>=0)&&(i>=(scores.size()-15)));i--)
551                                        finalScoreList.add((Area)scores.get(i,1));
552                                final int mobLevel=mob.phyStats().level();
553                                Collections.sort(finalScoreList,new Comparator<Area>()
554                                {
555                                        @Override
556                                        public int compare(Area o1, Area o2)
557                                        {
558                                                int median1=o1.getPlayerLevel();
559                                                if(median1==0)
560                                                        median1=o1.getAreaIStats()[Area.Stats.MED_LEVEL.ordinal()];
561                                                int median2=o2.getPlayerLevel();
562                                                if(median2==0)
563                                                        median2=o2.getAreaIStats()[Area.Stats.MED_LEVEL.ordinal()];
564                                                final int lvlDiff1=Math.abs(mobLevel - median1);
565                                                final int lvlDiff2=Math.abs(mobLevel - median2);
566                                                return lvlDiff1==lvlDiff2?0:(lvlDiff1>lvlDiff2)?1:-1;
567                                        }
568                                        
569                                });
570                                for(Area A : finalScoreList)
571                                {
572                                        int lvl=A.getPlayerLevel();
573                                        if(lvl==0)
574                                                lvl=A.getAreaIStats()[Area.Stats.MED_LEVEL.ordinal()];
575                                        final int align=A.getAreaIStats()[Area.Stats.MED_ALIGNMENT.ordinal()];
576        
577                                        msg.append(CMStrings.padRight(A.name(),35))
578                                           .append(CMStrings.padRight(Integer.toString(lvl),6))
579                                           .append(CMStrings.padRight(CMLib.factions().getRange(CMLib.factions().AlignID(), align).name(),20))
580                                           .append(CMStrings.padRight(Integer.toString(A.getAreaIStats()[Area.Stats.POPULATION.ordinal()]),10))
581                                           .append("\n\r");
582                                }
583                                msg.append(L("\n\r\n\r^HEnter 'HELP (AREA NAME) for more information.^?"));
584                        }
585                        if(!mob.isMonster())
586                                mob.session().colorOnlyPrintln(msg.toString()+"\n\r");
587                }
588                return false;
589        }
