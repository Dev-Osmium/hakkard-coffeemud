038public class Config extends StdCommand
039{
040        public Config()
041        {
042        }
043
044        private final String[] access=I(new String[]{"CONFIG","AUTO"});
045        @Override
046        public String[] getAccessWords()
047        {
048                return access;
049        }
050
051        @Override
052        public boolean execute(MOB mob, List<String> commands, int metaFlags)
053                throws java.io.IOException
054        {
055                String postStr="";
056                if((commands!=null)&&(commands.size()>1))
057                {
058                        final String name=commands.get(1);
059                        MOB.Attrib finalA=null;
060                        for(MOB.Attrib a : MOB.Attrib.values())
061                        {
062                                if(name.equalsIgnoreCase(a.getName()))
063                                        finalA=a;
064                        }
065                        if(finalA==null)
066                        {
067                                if(name.equalsIgnoreCase("LINEWRAP"))
068                                {
069                                        final String newWrap=(commands.size()>2)?CMParms.combine(commands,2):"";
070                                        int newVal=mob.playerStats().getWrap();
071                                        if((CMath.isInteger(newWrap))&&(CMath.s_int(newWrap)>10))
072                                                newVal=CMath.s_int(newWrap);
073                                        else
074                                        if("DISABLED".startsWith(newWrap.toUpperCase())&&(newWrap.length()>0))
075                                                newVal=0;
076                                        else
077                                        {
078                                                mob.tell(L("'@x1' is not a valid linewrap setting. Enter a number larger than 10 or 'disable'.",newWrap));
079                                                return false;
080                                        }
081                                        mob.playerStats().setWrap(newVal);
082                                        postStr=L("Configuration option change: LINEWRAP");
083                                }
084                                else
085                                if(name.equalsIgnoreCase("PAGEBREAK"))
086                                {
087                                        final String newBreak=(commands.size()>2)?CMParms.combine(commands,2):"";
088                                        int newVal=mob.playerStats().getWrap();
089                                        if((CMath.isInteger(newBreak))&&(CMath.s_int(newBreak)>0))
090                                                newVal=CMath.s_int(newBreak);
091                                        else
092                                        if("DISABLED".startsWith(newBreak.toUpperCase())&&(newBreak.length()>0))
093                                                newVal=0;
094                                        else
095                                        {
096                                                mob.tell(L("'@x1' is not a valid pagebreak setting. Enter a number larger than 0 or 'disable'.",newBreak));
097                                                return false;
098                                        }
099                                        mob.playerStats().setPageBreak(newVal);
100                                        postStr=L("Configuration option change: PAGEBREAK");
101                                }
102                                else
103                                        postStr=L("Unknown configuration flag '@x1'.",name);
104                        }
105                        else
106                        {
107                                postStr=L("Configuration flag toggled: "+finalA.getName());
108                                final boolean newSet = !mob.isAttributeSet(finalA);
109                                if(finalA == Attrib.AUTOFORWARD)
110                                {
111                                        final PlayerStats pStats = mob.playerStats();
112                                        if((pStats != null)
113                                        &&(pStats.getAccount() != null))
114                                        {
115                                                pStats.getAccount().setFlag(AccountFlag.NOAUTOFORWARD, newSet);
116                                                CMLib.database().DBUpdateAccount(pStats.getAccount());
117                                        }
118                                }
119                                mob.setAttribute(finalA, newSet);
120                        }
121                        mob.tell(postStr);
122                }
123                
124                final StringBuffer msg=new StringBuffer(L("^HYour configuration flags:^?\n\r"));
125                final List<MOB.Attrib> sorted = new XVector<MOB.Attrib>(MOB.Attrib.values());
126                Collections.sort(sorted,new Comparator<MOB.Attrib>()
127                {
128                        @Override
129                        public int compare(Attrib o1, Attrib o2)
130                        {
131                                return o1.getName().compareTo(o2.getName());
132                        }
133                });
134                for(final MOB.Attrib a : sorted)
135                {
136                        if((a==MOB.Attrib.SYSOPMSGS)&&(!(CMSecurity.isAllowed(mob,mob.location(),CMSecurity.SecFlag.SYSMSGS))))
137                                continue;
138                        if((a==MOB.Attrib.AUTOMAP)&&(CMProps.getIntVar(CMProps.Int.AWARERANGE)<=0))
139                                continue;
140
141                        msg.append("^W"+CMStrings.padRight(a.getName(),15)+"^N: ");
142                        boolean set=mob.isAttributeSet(a);
143                        if(a.isAutoReversed()) 
144                                set=!set;
145                        msg.append(set?L("^gON"):L("^rOFF"));
146                        msg.append("\n\r");
147                }
148                msg.append("^N");
149                if(mob.playerStats()!=null)
150                {
151                        final String wrap=(mob.playerStats().getWrap()!=0)?(""+mob.playerStats().getWrap()):"Disabled";
152                        msg.append("^W"+CMStrings.padRight(L("LINEWRAP"),15)+"^N: ^w"+wrap);
153                        if((mob.session()!=null)&&(mob.playerStats().getWrap() != mob.session().getWrap()))
154                                msg.append(" ("+mob.session().getWrap()+")");
155                        msg.append("\n\r");
156                        final String pageBreak=(mob.playerStats().getPageBreak()!=0)?(""+mob.playerStats().getPageBreak()):"^rDisabled";
157                        msg.append("^w"+CMStrings.padRight(L("PAGEBREAK"),15)+"^N: ^w"+pageBreak);
158                        msg.append("\n\r");
159                }
160                msg.append("^N");
161                mob.tell(msg.toString());
162                mob.tell(postStr);
163                return false;
164        }
165
166        @Override
167        public boolean canBeOrdered()
168        {
169                return false;
170        }
171
172}
