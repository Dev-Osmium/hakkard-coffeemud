039public class ClanAssign extends StdCommand
040{
041        public ClanAssign()
042        {
043        }
044
045        private final String[]  access  = I(new String[] { "CLANASSIGN" });
046
047        @Override
048        public String[] getAccessWords()
049        {
050                return access;
051        }
052
053        @Override
054        public boolean execute(MOB mob, List<String> commands, int metaFlags)
055                throws java.io.IOException
056        {
057                if(commands.size()<3)
058                {
059                        mob.tell(L("You must specify the members name, and a new role."));
060                        return false;
061                }
062                String memberStr=(commands.size()>2)?(String)commands.get(commands.size()-2):"";
063                final String pos=(commands.size()>1)?(String)commands.get(commands.size()-1):"";
064                final String clanName=(commands.size()>3)?CMParms.combine(commands,1,commands.size()-2):"";
065
066                Clan C=null;
067                final boolean skipChecks=mob.getClanRole(mob.Name())!=null;
068                if(skipChecks)
069                        C=mob.getClanRole(mob.Name()).first;
070
071                if(C==null)
072                {
073                        for(final Pair<Clan,Integer> c : mob.clans())
074                        {
075                                if((clanName.length()==0)||(CMLib.english().containsString(c.first.getName(), clanName))
076                                &&(c.first.getAuthority(c.second.intValue(), Clan.Function.ASSIGN)!=Authority.CAN_NOT_DO))
077                                {
078                                        C = c.first;
079                                        break;
080                                }
081                        }
082                }
083
084                commands.clear();
085                commands.add(getAccessWords()[0]);
086                commands.add(memberStr);
087                commands.add(pos);
088                final StringBuffer msg=new StringBuffer("");
089                boolean found=false;
090                if(memberStr.length()>0)
091                {
092                        if(C==null)
093                        {
094                                mob.tell(L("You aren't allowed to assign anyone from @x1.",((clanName.length()==0)?"anything":clanName)));
095                                return false;
096                        }
097                        if(skipChecks||CMLib.clans().goForward(mob,C,commands,Clan.Function.ASSIGN,false))
098                        {
099                                final List<MemberRecord> members=C.getMemberList();
100                                if(members.size()<1)
101                                {
102                                        mob.tell(L("There are no members in your @x1",C.getGovernmentName()));
103                                        return false;
104                                }
105                                final int newPos=C.getRoleFromName(pos);
106                                if(newPos<0)
107                                {
108                                        mob.tell(L("'@x1' is not a valid role.",pos));
109                                        return false;
110                                }
111                                memberStr=CMStrings.capitalizeAndLower(memberStr);
112                                for(final MemberRecord member : members)
113                                {
114                                        if(member.name.equalsIgnoreCase(memberStr))
115                                        {
116                                                found=true;
117                                        }
118                                }
119                                if(found)
120                                {
121                                        final MOB M=CMLib.players().getLoadPlayer(memberStr);
122                                        final Pair<Clan,Integer> oldRole=(M!=null)?M.getClanRole(C.clanID()):null;
123                                        if((M==null)||(oldRole==null))
124                                        {
125                                                mob.tell(L("@x1 was not found.  Could not change @x2 role.",memberStr,C.getGovernmentName()));
126                                                return false;
127                                        }
128                                        if(!C.canBeAssigned(M, newPos))
129                                        {
130                                                mob.tell(L("@x1 may not be assigned to @x2.",M.name(mob),C.getRoleName(newPos,true,false)));
131                                                return false;
132                                        }
133                                        if(skipChecks||CMLib.clans().goForward(mob,C,commands,Clan.Function.ASSIGN,true))
134                                        {
135                                                final int oldPos=oldRole.second.intValue();
136                                                final int maxInNewPos=C.getMostInRole(newPos);
137                                                final Vector<String> currentMembersInNewPosV=new Vector<String>();
138                                                for(final MemberRecord member : members)
139                                                {
140                                                        if(member.role==newPos)
141                                                                currentMembersInNewPosV.add(member.name);
142                                                }
143                                                final List<Integer> topRoleIDs=C.getTopRankedRoles(Function.ASSIGN);
144                                                if(topRoleIDs.contains(Integer.valueOf(oldPos)))
145                                                { // If you WERE already the highest order.. you must be being demoted.
146                                                        // so we check to see if there will be any other high officers left
147                                                        int numMembers=0;
148                                                        for(final MemberRecord member : members)
149                                                        {
150                                                                if(!M.Name().equalsIgnoreCase(member.name))
151                                                                {
152                                                                        if(topRoleIDs.contains(Integer.valueOf(member.role)))
153                                                                                numMembers++;
154                                                                }
155                                                        }
156                                                        if(numMembers==0)
157                                                        {
158                                                                mob.tell(L("@x1 is the last @x2 and must be replaced before being reassigned.",M.Name(),C.getRoleName(oldPos,true,false)));
159                                                                return false;
160                                                        }
161                                                }
162                                                if((currentMembersInNewPosV.size()>0)&&(maxInNewPos<Integer.MAX_VALUE))
163                                                {
164                                                        // if there are too many in the new position, demote some of them.
165                                                        while(currentMembersInNewPosV.size()>=maxInNewPos)
166                                                        {
167                                                                final String s=currentMembersInNewPosV.get(0);
168                                                                currentMembersInNewPosV.remove(0);
169                                                                CMLib.clans().clanAnnounce(mob,L(" @x1 of the @x2 @x3 is now a @x4.",s,C.getGovernmentName(),C.clanID(),C.getRoleName(C.getGovernment().getAcceptPos(),true,false)));
170                                                                final MOB M2=CMLib.players().getPlayer(s);
171                                                                if(M2!=null)
172                                                                        M2.setClan(C.clanID(),C.getGovernment().getAcceptPos());
173                                                                CMLib.database().DBUpdateClanMembership(s, C.clanID(), C.getGovernment().getAcceptPos());
174                                                                C.updateClanPrivileges(M2);
175                                                        }
176                                                }
177                                                // finally, promote
178                                                CMLib.clans().clanAnnounce(mob,L("@x1 of the @x2 @x3 changed from @x4 to @x5.",M.name(),C.getGovernmentName(),C.clanID(),C.getRoleName(oldRole.second.intValue(),true,false),C.getRoleName(newPos,true,false)));
179                                                C.addMember(M,newPos);
180                                                mob.tell(L("@x1 of the @x2 @x3 has been assigned to be @x4. ",M.Name(),C.getGovernmentName(),C.clanID(),CMLib.english().startWithAorAn(C.getRoleName(newPos,false,false))));
181                                                if((M.session()!=null)&&(M.session().mob()==M))
182                                                        M.tell(L("You have been assigned to be @x1 of @x2 @x3.",CMLib.english().startWithAorAn(C.getRoleName(newPos,false,false)),C.getGovernmentName(),C.clanID()));
183                                                return false;
184                                        }
185                                }
186                                else
187                                {
188                                        msg.append(L("@x1 isn't a member of your @x2.",memberStr,C.getGovernmentName()));
189                                }
190                        }
191                        else
192                        {
193                                msg.append(L("You aren't in the right position to assign anyone in your @x1.",C.getGovernmentName()));
194                        }
195                }
196                else
197                {
198                        msg.append(L("You haven't specified which member you are assigning a new role to."));
199                }
200                mob.tell(msg.toString());
201                return false;
202        }
203
204        @Override
205        public boolean canBeOrdered()
206        {
207                return false;
208        }
