071        protected void gmcpSaySend(final String sayName, final MOB mob, final Environmental target, final CMMsg msg)
072        {
073                if((mob.session()!=null)&&(mob.session().getClientTelnetMode(Session.TELNET_GMCP)))
074                {
075                        mob.session().sendGMCPEvent("comm.channel", "{\"chan\":\""+sayName+"\",\"msg\":\""+
076                                        MiniJSON.toJSONString(CMLib.coffeeFilter().fullOutFilter(null, mob, mob, target, null, CMStrings.removeColors(msg.sourceMessage()), false))
077                                        +"\",\"player\":\""+mob.name()+"\"}");
078                }
079                final Room R=mob.location();
080                if(R!=null)
081                for(int i=0;i<R.numInhabitants();i++)
082                {
083                        final MOB M=R.fetchInhabitant(i);
084                        if((M!=null)&&(M!=msg.source())&&(M.session()!=null)&&(M.session().getClientTelnetMode(Session.TELNET_GMCP)))
085                        {
086                                M.session().sendGMCPEvent("comm.channel", "{\"chan\":\""+sayName+"\",\"msg\":\""+
087                                                MiniJSON.toJSONString(CMLib.coffeeFilter().fullOutFilter(null, M, mob, target, null, CMStrings.removeColors(msg.othersMessage()), false))
088                                                +"\",\"player\":\""+mob.name()+"\"}");
089                        }
090                }
091        }

093        @Override
094        public boolean execute(final MOB mob, final List<String> commands, final int metaFlags)
095                throws java.io.IOException
096        {
097                final Vector<String> origCmds=new XVector<String>(commands);
098                String theWord="Say";
099                boolean toFlag=false;
100                final String theCommand=commands.get(0).toUpperCase();
101                if(theCommand.equals("ASK"))
102                        theWord="Ask";
103                else
104                if(theCommand.equals("YELL"))
105                        theWord="Yell";
106                else
107                if(theCommand.equals("SAYTO")
108                ||theCommand.equals("SAYT"))
109                {
110                        theWord="Say";
111                        toFlag=true;
112                }
113
114                final Room R=mob.location();
115                if((commands.size()==1)||(R==null))
116                {
117                        CMLib.commands().postCommandFail(mob,origCmds,L("@x1 what?",theWord));
118                        return false;
119                }
120
121                Vector<Room> yellRooms=new Vector<Room>();
122                if(theCommand.equals("YELL"))
123                {
124                        for(int d=Directions.NUM_DIRECTIONS()-1;d>=0;d--)
125                        {
126                                final Room R2=R.getRoomInDir(d);
127                                final Exit E2=R.getExitInDir(d);
128                                if((R2!=null)&&(E2!=null)&&(E2.isOpen()))
129                                        yellRooms.add(R2);
130                        }
131                }
132
133                String whom="";
134                String theWordSuffix="";
135                Environmental target=null;
136                Physical langTarget=null;
137                if(commands.size()>2)
138                {
139                        whom=commands.get(1).toUpperCase();
140                        if(!toFlag)
141                        {
142                                for (final String impossibleTarget : impossibleTargets)
143                                {
144                                        if(impossibleTarget.startsWith(whom))
145                                        {
146                                                whom = "";
147                                                break;
148                                        }
149                                }
150                        }
151                        if(whom.equalsIgnoreCase("self"))
152                        {
153                                target=mob;
154                                commands.remove(1);
155                        }
156                        else
157                        if(toFlag && (whom.equalsIgnoreCase("noone")||whom.equalsIgnoreCase("nobody")))
158                        {
159                                target=null;
160                                commands.remove(1);
161                        }
162                        else
163                        if(whom.length()>0)
164                        {
165                                target=R.fetchFromRoomFavorMOBs(null,whom);
166                                if((toFlag)&&(target==null))
167                                        target=mob.findItem(null,whom);
168
169                                if((!toFlag)&&(target!=null))
170                                {
171                                        if(!(target instanceof MOB))
172                                                target=null;
173                                        else
174                                        if(target.name().toUpperCase().indexOf(whom.toUpperCase())<0)
175                                                target=null;
176                                        else
177                                        if((!target.name().equalsIgnoreCase(whom))&&(whom.length()<4))
178                                                target=null;
179                                }
180
181                                if(target!=null)
182                                {
183                                        commands.remove(1);
184                                        if(target instanceof Physical)
185                                                langTarget=(Physical)target;
186                                }
187                                else
188                                {
189                                        if(theCommand.equals("YELL"))
190                                        {
191                                                final int dir=CMLib.directions().getGoodCompassDirectionCode(whom);
192                                                if(dir >=0)
193                                                {
194                                                        commands.remove(1);
195                                                        yellRooms=new Vector<Room>();
196                                                        if(theCommand.equals("YELL"))
197                                                        {
198                                                                final Room R2=R.getRoomInDir(dir);
199                                                                final Exit E2=R.getExitInDir(dir);
200                                                                if(R2!=null)
201                                                                {
202                                                                        theWordSuffix=" "+CMLib.directions().getDirectionName(dir);
203                                                                        yellRooms.add(R2);
204                                                                        if((E2!=null)&&(E2.isOpen()))
205                                                                        {
206                                                                                for(int d=Directions.NUM_DIRECTIONS()-1;d>=0;d--)
207                                                                                {
208                                                                                        final Room R3=R2.getRoomInDir(d);
209                                                                                        final Exit E3=R2.getExitInDir(d);
210                                                                                        if((R3!=null)&&(E3!=null)&&(E3.isOpen()))
211                                                                                                yellRooms.add(R3);
212                                                                                }
213                                                                        }
214                                                                }
215                                                        }
216                                                }
217                                        }
218                                }
219                                if(toFlag
220                                &&((target==null)||(!CMLib.flags().canBeSeenBy(target,mob))))
221                                {
222                                        CMLib.commands().postCommandFail(mob,origCmds,L("you don't see @x1 here to speak to.",whom));
223                                        return false;
224                                }
225                        }
226                }
227
228                Language[] langSwap=null;
229                if(!CMSecurity.isDisabled(CMSecurity.DisFlag.AUTOLANGUAGE))
230                {
231                        // if you are the only one in the room to talk to
232                        // then grab a random mob and assume that's who
233                        // you are addressing.
234                        if(!toFlag)
235                        if((langTarget==null)
236                        &&(target==null)
237                        &&(R.numInhabitants()==2))
238                        {
239                                for(int r=0;r<R.numInhabitants();r++)
240                                {
241                                        final MOB M=R.fetchInhabitant(r);
242                                        if(M!=mob)
243                                        {
244                                                langTarget=M;
245                                                target=M;
246                                                break;
247                                        }
248                                }
249                        }
250
251                        // if you are addressing someone speaking a language that you
252                        // can speak, then speak it.
253                        if((langTarget!=null)&&(!mob.isMonster())&&(!CMLib.flags().isAnimalIntelligence(mob)))
254                        {
255                                final Language hisL=CMLib.utensils().getLanguageSpoken(langTarget);
256                                final Language myL=CMLib.utensils().getLanguageSpoken(mob);
257                                if((hisL==null)&&(myL!=null)&&(mob.fetchAbility("Common")!=null))
258                                        langSwap=new Language[]{null,myL};
259                                else
260                                if((hisL!=null)&&((myL==null)||(!hisL.ID().equals(myL.ID()))))
261                                {
262                                        final Language myTargetL = (Language)mob.fetchEffect(hisL.ID());
263                                        if(myTargetL!=null)
264                                                langSwap=new Language[]{myTargetL,myL};
265                                }
266                        }
267                }
268
269                String combinedCommands;
270                if(commands.size()==2)
271                        combinedCommands=commands.get(1);
272                else
273                        combinedCommands=CMParms.combineQuoted(commands,1);
274                if(combinedCommands.equals(""))
275                {
276                        CMLib.commands().postCommandFail(mob,origCmds,L("@x1  what?",theWord));
277                        return false;
278                }
279                combinedCommands=CMProps.applyINIFilter(combinedCommands,CMProps.Str.SAYFILTER);
280                if(mob.isPlayer()
281                || CMath.bset(metaFlags, MUDCmdProcessor.METAFLAG_FORCED)
282                || CMath.bset(metaFlags, MUDCmdProcessor.METAFLAG_ORDER))
283                        combinedCommands=CMLib.coffeeFilter().secondaryUserInputFilter(combinedCommands);
284                CMMsg msg=null;
285                if((!theCommand.equals("ASK"))&&(target!=null))
286                        theWord=L(theWord+"(s) to");
287                else
288                        theWord=L(theWord+"(s)");
289                if(CMLib.flags().isAnimalIntelligence(mob))
290                        msg=CMClass.getMsg(mob,null,null,CMMsg.MSG_SPEAK,"^T^<SAY \""+CMStrings.removeColors(mob.name())+"\"^><S-NAME> "+L("go(es)")+theWordSuffix+" '"+combinedCommands+"'^</SAY^>^?");
291                else
292                if(target==null)
293                        msg=CMClass.getMsg(mob,null,null,CMMsg.MSG_SPEAK,"^T^<SAY \""+CMStrings.removeColors(mob.name())+"\"^><S-NAME> "+theWord.toLowerCase()+theWordSuffix+" '"+combinedCommands+"'^</SAY^>^?");
294                else
295                {
296                        final String fromSelf="^T^<SAY \""+CMStrings.removeColors(target.name())+"\"^><S-NAME> "+theWord.toLowerCase()+theWordSuffix+" <T-NAMESELF> '"+combinedCommands+"'^</SAY^>^?";
297                        final String toTarget="^T^<SAY \""+CMStrings.removeColors(mob.name())+"\"^><S-NAME> "+theWord.toLowerCase()+theWordSuffix+" <T-NAMESELF> '"+combinedCommands+"'^</SAY^>^?";
298                        msg=CMClass.getMsg(mob,target,null,CMMsg.MSG_SPEAK,fromSelf,toTarget,fromSelf);
299                }
300
301                gmcpSaySend("say", mob, target, msg);
302
303                if(langSwap!=null)
304                {
305                        if(langSwap[1]!=null)
306                                langSwap[1].setBeingSpoken(langSwap[1].ID(), false);
307                        if(langSwap[0]!=null)
308                                langSwap[0].setBeingSpoken(langSwap[0].ID(), true);
309                }
310                final boolean useShipDirs=(R instanceof BoardableShip)||(R.getArea() instanceof BoardableShip);
311                if(R.okMessage(mob,msg))
312                {
313                        R.send(mob,msg);
314                        if(theCommand.equals("YELL"))
315                        {
316                                int dirCode=-1;
317                                Room R3=R;
318                                for(final Room R2 : yellRooms)
319                                {
320                                        int newDirCode=CMLib.map().getRoomDir(R, R2);
321                                        if(newDirCode<0)
322                                                newDirCode=CMLib.map().getRoomDir(R3, R2);
323                                        else
324                                                R3=R2;
325                                        if(newDirCode>=0)
326                                                dirCode=newDirCode;
327                                        final Environmental tool=msg.tool();
328                                        int opDirCode=-1;
329                                        if(dirCode>=0)
330                                                opDirCode=Directions.getOpDirectionCode(dirCode);
331                                        final String inDirName=(dirCode<0)?"":(useShipDirs?CMLib.directions().getShipInDirectionName(opDirCode):CMLib.directions().getInDirectionName(opDirCode));
332                                        msg=CMClass.getMsg(mob,null,null,CMMsg.MSG_SPEAK,L("^TYou hear someone yell ")+"'"+combinedCommands+"' "+inDirName+"^?");
333                                        if((R2.okMessage(mob,msg))
334                                        &&((tool==null)||(tool.okMessage(mob,msg))))
335                                        {
336                                                R2.sendOthers(mob,msg);
337                                        }
338                                }
339                        }
340                }
341                if(langSwap!=null)
342                {
343                        if(langSwap[0]!=null)
344                                langSwap[0].setBeingSpoken(langSwap[0].ID(), false);
345                        if(langSwap[1]!=null)
346                                langSwap[1].setBeingSpoken(langSwap[1].ID(), true);
347                }
348                return false;
349        }
