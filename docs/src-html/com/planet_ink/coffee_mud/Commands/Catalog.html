054        public boolean catalog(Room R, MOB mob, Physical P)
055                throws java.io.IOException
056        {
057                final Physical origP=P;
058                final Physical cataP=CMLib.catalog().getCatalogObj(P);
059                if((!(P instanceof DBIdentifiable))
060                ||(!((DBIdentifiable)P).canSaveDatabaseID()))
061                {
062                        mob.tell(L("The object '@x1' can not be cataloged.",P.Name()));
063                        return false;
064                }
065                String newCat=currentCats.get(mob.Name());
066                if(newCat==null)
067                        newCat="";
068                String msg="<S-NAME> catalog(s) <T-NAMESELF> into category '"+newCat+"'.";
069                final CataData data=CMLib.catalog().getCatalogData(cataP);
070                final String oldCat=(data!=null)?data.category():"";
071                final String catagory=(data!=null)?" in category '"+oldCat+"'":"";
072                /*
073                if(CMLib.flags().isCataloged(P))
074                {
075                        mob.tell(L("The object '@x1' is already cataloged@x2.",P.Name(),catagory));
076                        return false;
077                }
078                */
079                if(cataP!=null)
080                {
081                        CMLib.catalog().changeCatalogUsage(P,true);
082                        final StringBuffer diffs=CMLib.catalog().checkCatalogIntegrity(P);
083                        if((diffs==null)||(diffs.length()==0))
084                        {
085                                if((data!=null)&&(!data.category().equals(newCat)))
086                                {
087                                        if((mob.session()==null)
088                                        ||(!mob.session().confirm(L("The object '@x1' already exists in the catalog@x2 , exactly as it is.  Would you like to change it to category '@x3'(y/N)?",cataP.Name(),catagory,newCat),"N")))
089                                        {
090                                                return false;
091                                        }
092                                        else
093                                        {
094                                                CMLib.catalog().updateCatalogCategory(cataP,newCat);
095                                                return true;
096                                        }
097                                }
098                                else
099                                {
100                                        mob.tell(L("The object '@x1' already exists in the catalog@x2 , exactly as it is.",cataP.Name(),catagory));
101                                        return true;
102                                }
103                        }
104                        if((data!=null)&&(!data.category().equals(newCat)))
105                                diffs.insert(0,"New category: '"+newCat+"', ");
106                        if((mob.session()==null)
107                        ||(!mob.session().confirm(L("Cataloging that object will change the existing cataloged '@x1'@x2 by altering the following properties: @x3.  Please confirm (y/N)?",P.Name(),catagory,diffs.toString()),"Y")))
108                        {
109                                CMLib.catalog().changeCatalogUsage(origP,false);
110                                return false;
111                        }
112                        msg="<S-NAME> modif(ys) the cataloged version of <T-NAMESELF>.";
113                        if((data!=null)&&(!data.category().equals(newCat)))
114                                CMLib.catalog().updateCatalogCategory(cataP,newCat);
115                }
116                else
117                        CMLib.catalog().addCatalog(newCat,P);
118                CMLib.catalog().updateCatalog(P);
119                R.show(mob,P,CMMsg.MSG_OK_VISUAL,msg);
120                return true;
121        }

123        public Physical findCatalog(CatalogKind whatKind, String ID, boolean exactOnly)
124        {
125                final Object[] data=new Object[]{null,null};
126                if((data[0]==null)&&((whatKind==CatalogKind.OBJECT)||(whatKind==CatalogKind.MOB)))
127                {
128                        data[0] = CMLib.catalog().getCatalogMob(ID);
129                        if (data[0] != null)
130                                data[1] = Integer.valueOf(1);
131                }
132                if((data[0]==null)&&((whatKind==CatalogKind.OBJECT)||(whatKind==CatalogKind.ITEM)))
133                {
134                        data[0] = CMLib.catalog().getCatalogItem(ID);
135                        if (data[0] != null)
136                                data[1] = Integer.valueOf(2);
137                }
138                if(exactOnly)
139                        return (Physical)data[0];
140                if((data[0]==null)&&((whatKind==CatalogKind.OBJECT)||(whatKind==CatalogKind.MOB)))
141                {
142                        final String[] names=CMLib.catalog().getCatalogMobNames().clone();
143                        for (final String name : names)
144                        {
145                                if(CMLib.english().containsString(name, ID))
146                                {
147                                        data[0] = CMLib.catalog().getCatalogMob(name);
148                                        data[1] = Integer.valueOf(1);
149                                        break;
150                                }
151                        }
152                        if(data[0] == null)
153                        {
154                                for(int s=0;s<names.length;s++)
155                                        names[s] = CMStrings.removeColors(names[s]);
156                                for (final String name : names)
157                                {
158                                        if(ID.equalsIgnoreCase(name))
159                                        {
160                                                data[0] = CMLib.catalog().getCatalogMob(name);
161                                                data[1] = Integer.valueOf(1);
162                                                break;
163                                        }
164                                }
165                                if(data[0] == null)
166                                {
167                                        for (final String name : names)
168                                        {
169                                                if(CMLib.english().containsString(name, ID))
170                                                {
171                                                        data[0] = CMLib.catalog().getCatalogMob(name);
172                                                        data[1] = Integer.valueOf(1);
173                                                        break;
174                                                }
175                                        }
176                                }
177                        }
178                }
179                if((data[0]==null)&&((whatKind==CatalogKind.OBJECT)||(whatKind==CatalogKind.ITEM)))
180                {
181                        final String[] names=CMLib.catalog().getCatalogItemNames().clone();
182                        for (final String name : names)
183                        {
184                                if(CMLib.english().containsString(name, ID))
185                                {
186                                        data[0] = CMLib.catalog().getCatalogItem(name);
187                                        data[1] = Integer.valueOf(2);
188                                        break;
189                                }
190                        }
191                        if(data[0] == null)
192                        {
193                                for(int s=0;s<names.length;s++)
194                                        names[s] = CMStrings.removeColors(names[s]);
195                                for (final String name : names)
196                                {
197                                        if(ID.equalsIgnoreCase(name))
198                                        {
199                                                data[0] = CMLib.catalog().getCatalogItem(name);
200                                                data[1] = Integer.valueOf(2);
201                                                break;
202                                        }
203                                }
204                                if(data[0] == null)
205                                {
206                                        for (final String name : names)
207                                        {
208                                                if(CMLib.english().containsString(name, ID))
209                                                {
210                                                        data[0] = CMLib.catalog().getCatalogItem(name);
211                                                        data[1] = Integer.valueOf(2);
212                                                        break;
213                                                }
214                                        }
215                                }
216                        }
217                }
218                return (Physical)data[0];
219        }

221        public Enumeration<String> getRoomSet(MOB mob, String which)
222        {
223                Enumeration<String> rooms=null;
224                if(which.equalsIgnoreCase("ROOM"))
225                        rooms=new XVector<String>(CMLib.map().getExtendedRoomID(mob.location())).elements();
226                else
227                if(which.equalsIgnoreCase("AREA"))
228                        rooms=mob.location().getArea().getProperRoomnumbers().getRoomIDs();
229                else
230                if(which.equalsIgnoreCase("WORLD"))
231                        rooms=CMLib.map().roomIDs();
232                return rooms;
233        }

235        public CatalogKind getObjectType(List<String> commands)
236        {
237                CatalogKind whatKind=CatalogKind.OBJECT;
238                if((commands.size()>0)&&("MOBS".startsWith(commands.get(0).toUpperCase().trim())))
239                {
240                        commands.remove(0);
241                        whatKind = CatalogKind.MOB;
242                }
243                if((commands.size()>0)&&("ITEMS".startsWith(commands.get(0).toUpperCase().trim())))
244                {
245                        commands.remove(0);
246                        whatKind = CatalogKind.ITEM;
247                }
248                return whatKind;
249        }

252        {

256                ||(commands.get(0).equalsIgnoreCase("AREA"))
257                ||(commands.get(0).equalsIgnoreCase("WORLD")))
258                        return true;
259                return false;

263        protected String listCatagories(CatalogKind whatKind)
264        {
265                StringBuilder list = new StringBuilder("");
266                List<CatalogKind> types = new ArrayList<CatalogKind>();
267                if((whatKind==CatalogKind.OBJECT)||(whatKind==CatalogKind.MOB))
268                        types.add(CatalogKind.MOB);
269                if((whatKind==CatalogKind.OBJECT)||(whatKind==CatalogKind.MOB))
270                        types.add(CatalogKind.ITEM);
271                for(CatalogKind kind : types)
272                {
273                        list.append(L("\n\r^H"+CMStrings.capitalizeAndLower(kind.name()).trim()+" categories:"));
274                        list.append("\n\r"+CMStrings.repeat('-',78)+"^?\n\r");
275                        List<String> cats;
276                        switch(kind)
277                        {
278                        case MOB:
279                                cats=Arrays.asList(CMLib.catalog().getMobCatalogCatagories());
280                                break;
281                        case ITEM:
282                        default:
283                                cats=Arrays.asList(CMLib.catalog().getItemCatalogCatagories());
284                                break;
285                        }
286                        for(String c : cats)
287                                list.append(c+"\n\r");
288                        list.append("\n\r");
289                }
290                return list.toString();
291        }

293        @Override
294        public boolean execute(final MOB mob, List<String> commands, int metaFlags)
295                throws java.io.IOException
296        {
297                Room R=mob.location();
298                if(R==null)
299                        return false;
300                if((commands!=null)&&(commands.size()>1))
301                {
302                        commands.remove(0);
303                        if(checkUserRoomSetEntry(commands))
304                        {
305                                final String which=commands.get(0).toLowerCase();
306                                final Enumeration<String> rooms=getRoomSet(mob,which);
307                                commands.remove(0);
308                                final CatalogKind whatKind=getObjectType(commands);
309                                final String type=whatKind.name().toLowerCase();
310
311                                if((mob.session()!=null)
312                                &&(mob.session().confirm(L("You are about to auto-catalog (room-reset and save) all @x1 @x2.\n\rThis command, if used improperly, "
313                                                + "may alter @x3 in this @x4.\n\rAre you absolutely sure (y/N)?",which,type,type,which),"N"))
314                                &&((which.equalsIgnoreCase("ROOM"))||mob.session().confirm(L("I'm serious now.  You can't abort this, and it WILL modify stuff.\n\r"
315                                                + "Have you tested this command on small areas and know what you're doing?.\n\rAre you absolutely POSITIVELY sure (y/N)?"),"N")))
316                                {
317                                        Physical P=null;
318                                        String roomID=null;
319                                        for(;rooms.hasMoreElements();)
320                                        {
321                                                roomID=rooms.nextElement();
322                                                R=CMLib.coffeeMaker().makeNewRoomContent(CMLib.map().getRoom(roomID),false);
323                                                final List<CatalogLibrary.RoomContent> contents=CMLib.catalog().roomContent(R);
324                                                boolean dirty=false;
325                                                for(final CatalogLibrary.RoomContent content : contents)
326                                                {
327                                                        P=content.P();
328                                                        if(P instanceof Coins)
329                                                                continue;
330                                                        if((P instanceof MOB)&&(whatKind!=CatalogKind.ITEM))
331                                                        {
332                                                                if(catalog(R,mob,P))
333                                                                {
334                                                                        content.flagDirty();
335                                                                        dirty=true;
336                                                                }
337                                                        }
338                                                        else
339                                                        if((P instanceof Item)&&(whatKind!=CatalogKind.MOB))
340                                                        {
341                                                                if(catalog(R,mob,P))
342                                                                {
343                                                                        content.flagDirty();
344                                                                        dirty=true;
345                                                                }
346                                                        }
347                                                }
348                                                if(dirty)
349                                                {
350                                                        CMLib.catalog().updateRoomContent(roomID, contents);
351                                                        CMLib.map().resetRoom(CMLib.map().getRoom(roomID), true);
352                                                }
353                                                R.destroy();
354                                        }
355                                }
356                        }
357                        else
358                        if(commands.get(0).equalsIgnoreCase("CATAGORY")||commands.get(0).equalsIgnoreCase("CATEGORY"))
359                        {
360                                commands.remove(0);
361                                String ID=CMParms.combine(commands,0);
362                                if(ID.equalsIgnoreCase("none"))
363                                {
364                                        commands.add("");
365                                        ID="";
366                                }
367                                String oldCat=currentCats.get(mob.Name());
368                                if(oldCat == null)
369                                        oldCat="";
370                                mob.tell(L("Your current category is '@x1'.",oldCat));
371                                if(commands.size()>0)
372                                {
373                                        ID=ID.toUpperCase().trim();
374                                        if(ID.length()>0)
375                                        {
376                                                if(ID.equalsIgnoreCase("LIST"))
377                                                {
378                                                        mob.tell(listCatagories(CatalogKind.OBJECT));
379                                                        return false;
380                                                }
381                                                else
382                                                if((!CMParms.contains(CMLib.catalog().getItemCatalogCatagories(),ID))
383                                                &&(!CMParms.contains(CMLib.catalog().getMobCatalogCatagories(),ID)))
384                                                {
385                                                        final Session session=mob.session();
386                                                        final String newCat=ID;
387                                                        if(newCat.equalsIgnoreCase("GLOBAL")||newCat.equalsIgnoreCase("NONE")||newCat.equalsIgnoreCase("UNCATEGORIZED"))
388                                                                mob.tell(L("That is not a valid new catagory to create."));
389                                                        else
390                                                        if(session!=null)
391                                                        {
392                                                                session.prompt(new InputCallback(InputCallback.Type.CONFIRM,"N")
393                                                                {
394                                                                        @Override
395                                                                        public void showPrompt()
396                                                                        {
397                                                                                session.promptPrint(L("Create new category '@x1' (y/N)?", newCat));
398                                                                        }
399
400                                                                        @Override
401                                                                        public void timedOut()
402                                                                        {
403                                                                        }
404
405                                                                        @Override
406                                                                        public void callBack()
407                                                                        {
408                                                                                if(this.confirmed)
409                                                                                {
410                                                                                        currentCats.put(mob.Name(), newCat);
411                                                                                        mob.tell(L("Your category is now '@x1' for new mob/item catalog additions.",newCat));
412                                                                                        mob.tell(L("To change back to the global category, enter CATALOG CATAGORY NONE"));

416                                                        }

430                                }

494                                        list.append(CMStrings.padRight(L("Rate"),6)+" ");
495                                        list.append(CMStrings.padRight(L("Mask"),31)+" ");
496                                        list.append("\n\r"+CMStrings.repeat('-',78)+"\n\r");

498                                        for (final String name : names)
499                                        {

501                                                if(I!=null)
502                                                {
503                                                        data=CMLib.catalog().getCatalogItemData(name);
504                                                        if((ID.length()==0)||(CMLib.english().containsString(I.Name(),ID)))
505                                                        {
506                                                                list.append(CMStrings.padRight(I.Name(),34)+" ");
507                                                                list.append(CMStrings.padRight(Integer.toString(data.numReferences()),3)+" ");
508                                                                if(data.getRate()<=0.0)
509                                                                {
510                                                                        list.append("N/A   ");
511                                                                        list.append(CMStrings.padRight(" ",31));
512                                                                }
513                                                                else
514                                                                {
515                                                                        list.append(CMStrings.padRight(CMath.toPct(data.getRate()),6)+" ");

519                                                                list.append("\n\r");

521                                                }

524                                }
525                                if(mob.session()!=null)
526                                        mob.session().wraplessPrintln(list.toString());
527                        }
528                        else
529                        if(commands.get(0).equalsIgnoreCase("DELETE"))
530                        {
531                                commands.remove(0);
532                                final CatalogKind whatKind=getObjectType(commands);
533                                final String ID=CMParms.combine(commands,0);
534                                Physical[] del=null;
535                                if(ID.equalsIgnoreCase("everydamnmob"))
536                                        del=CMLib.catalog().getCatalogMobs();
537                                else
538                                if(ID.equalsIgnoreCase("everydamnitem"))
539                                        del=CMLib.catalog().getCatalogItems();
540                                else
541                                if(ID.equalsIgnoreCase("everydamnthing"))
542                                {
543                                        final java.util.List<Physical> V=new Vector<Physical>();
544                                        V.addAll(Arrays.asList(CMLib.catalog().getCatalogItems()));
545                                        V.addAll(Arrays.asList(CMLib.catalog().getCatalogMobs()));

547                                }
548                                else
549                                {
550                                        final Physical P=findCatalog(whatKind,ID,false);
551                                        if(P==null)
552                                        {
553                                                mob.tell(L("'@x1' not found in catalog! Try CATALOG LIST",ID));
554                                                return false;
555                                        }
556                                        del=new Physical[]{P};
557                                }
558                                for (final Physical P : del)
559                                {
560                                        final CatalogLibrary.CataData data=CMLib.catalog().getCatalogData(P);
561                                        if(P instanceof MOB)
562                                        {

570                                                        mob.tell(L("MOB '@x1 has been permanently removed from the catalog.",((MOB)P).Name()));
571                                                }
572                                        }
573                                        else
574                                        if(P instanceof Item)
575                                        {
576                                                String prefix="";
577                                                if((data!=null)&&(data.numReferences()>0))
578                                                        prefix="Catalog Item '"+((Item)P).Name()+"' is currently listed as being in use '"+data.numReferences()+" times.  ";
579                                                if((mob.session()!=null)
580                                                &&((del.length>10)||(mob.session().confirm(L("@x1This will permanently delete item '@x2' from the catalog.  Are you sure (y/N)?",prefix,((Item)P).Name()),"N"))))
581                                                {
582                                                        CMLib.catalog().delCatalog(P);
583                                                        mob.tell(L("Item '@x1 has been permanently removed from the catalog.",P.Name()));
584                                                }
585                                        }
586                                }
587                        }
588                        else
589                        if(commands.get(0).equalsIgnoreCase("EDIT"))
590                        {
591                                commands.remove(0);
592                                final CatalogKind whatKind=getObjectType(commands);
593                                final String ID=CMParms.combine(commands,0);
594                                final Physical P=findCatalog(whatKind,ID,false);
595                                if(P==null)
596                                {
597                                        mob.tell(L("'@x1' not found in catalog! Try CATALOG LIST",ID));
598                                        return false;
599                                }
600                                final CatalogLibrary.CataData data=CMLib.catalog().getCatalogData(P);
601                                if(P instanceof MOB)
602                                {
603                                        mob.tell(L("There is no extra mob data to edit. See help on CATALOG."));
604                                }
605                                else
606                                if(P instanceof Item)
607                                {
608                                        if(mob.session()!=null)
609                                        {
610                                                Double newPct=null;
611                                                while(newPct == null)
612                                                {
613                                                        final String newRate=mob.session().prompt(L("Enter a new Drop Rate or 0% to disable (@x1): ",CMath.toPct(data.getRate())), CMath.toPct(data.getRate()));
614                                                        if(newRate.trim().length()==0)
615                                                                return false;
616                                                        else
617                                                        if(CMath.isPct(newRate))
618                                                                newPct=Double.valueOf(CMath.s_pct(newRate));
619                                                        else
620                                                                mob.tell(L("'@x1' is not a valid percentage value.  Try something like 10%",newRate));
621                                                }
622                                                data.setRate(newPct.doubleValue());
623                                                if(data.getRate()<=0.0)
624                                                {
625                                                        data.setMaskStr("");
626                                                        data.setRate(0.0);
627                                                        CMLib.database().DBUpdateItem("CATALOG_ITEMS",(Item)P);
628                                                        Log.sysOut("Catalog",mob.Name()+" modified catalog item "+P.Name());
629                                                        mob.tell(L("No drop item."));
630                                                        return false;
631                                                }
632                                                final String choice=mob.session().choose(L("Is this for L)ive mobs or D)ead ones (@x1): ",(data.getWhenLive()?"L":"D")),L("LD"), (data.getWhenLive()?"L":"D"));
633                                                data.setWhenLive(choice.equalsIgnoreCase("L"));
634                                                String newMask="?";
635                                                while(newMask.equalsIgnoreCase("?"))
636                                                {
637                                                        newMask=mob.session().prompt(L("Enter new MOB selection mask, or NULL (@x1)\n\r: ",data.getMaskStr()),data.getMaskStr());
638                                                        if(newMask.equalsIgnoreCase("?"))
639                                                                mob.tell(CMLib.masking().maskHelp("\n","disallow"));
640                                                }
641                                                if((newMask.length()==0)||(newMask.equalsIgnoreCase("null")))
642                                                {
643                                                        data.setMaskStr("");
644                                                        data.setRate(0.0);
645                                                        mob.tell(L("Mask removed."));
646                                                }
647                                                else
648                                                {
649                                                        data.setMaskStr(newMask);
650                                                }
651                                                CMLib.database().DBUpdateItem("CATALOG_ITEMS",(Item)P);
652                                                Log.sysOut("Catalog",mob.Name()+" modified catalog item "+P.Name());
653                                                mob.tell(L("Item '@x1 has been updated.",P.Name()));
654                                        }
655                                }
656                        }
657                        else
658                        if((commands.get(0).equalsIgnoreCase("SCAN"))
659                        ||(commands.get(0).equalsIgnoreCase("DBSCAN")))
660                        {
661                                final boolean db=commands.get(0).toUpperCase().startsWith("DB");
662                                commands.remove(0);
663                                if(checkUserRoomSetEntry(commands))
664                                {
665                                        final String which=commands.get(0).toLowerCase();
666                                        final Enumeration<String> rooms=getRoomSet(mob,which);
667                                        commands.remove(0);
668
669                                        final CatalogKind whatKind=getObjectType(commands);
670                                        Physical P=null;
671                                        String roomID=null;
672                                        for(;rooms.hasMoreElements();)
673                                        {
674                                                roomID=rooms.nextElement();
675                                                R=CMLib.map().getRoom(roomID);
676                                                if(db)
677                                                        R=CMLib.coffeeMaker().makeNewRoomContent(R,false);
678                                                final List<CatalogLibrary.RoomContent> contents=CMLib.catalog().roomContent(R);
679                                                for(final CatalogLibrary.RoomContent content : contents)
680                                                {
681                                                        P=content.P();
682                                                        if(P instanceof Coins)
683                                                                continue;
684                                                        if((P instanceof MOB)
685                                                        &&(whatKind!=CatalogKind.ITEM)
686                                                        &&(CMLib.flags().isCataloged(P)))
687                                                        {
688                                                                if(CMLib.catalog().getCatalogObj(P)!=null)
689                                                                        mob.tell(L("Check: MOB @x1 in @x2 is cataloged.",P.Name(),roomID));
690                                                                else
691                                                                        mob.tell(L("Error: MOB @x1 in @x2 is falsely cataloged.",P.Name(),roomID));
692                                                        }
693
694                                                        if((P instanceof Item)
695                                                        &&(whatKind!=CatalogKind.MOB)
696                                                        &&(CMLib.flags().isCataloged(P)))
697                                                        {
698                                                                if(CMLib.catalog().getCatalogObj(P)!=null)
699                                                                        mob.tell(L("Check: Item @x1 in @x2 is cataloged.",P.Name(),roomID));
700                                                                else
701                                                                        mob.tell(L("Error: Item @x1 in @x2 is falsely cataloged.",P.Name(),roomID));
702                                                        }
703                                                }
704                                                if(db && (R!=null))
705                                                        R.destroy();
706                                        }
707                                }
708                                else
709                                        mob.tell(L("Scan what?"));
710                        }
711                        else
712                        if((commands.get(0).equalsIgnoreCase("OVERLOOK"))
713                        ||(commands.get(0).equalsIgnoreCase("DBOVERLOOK")))
714                        {
715                                final boolean db=commands.get(0).toUpperCase().startsWith("DB");
716                                commands.remove(0);
717                                if(checkUserRoomSetEntry(commands))
718                                {
719                                        final String which=commands.get(0).toLowerCase();
720                                        final Enumeration<String> rooms=getRoomSet(mob,which);
721                                        commands.remove(0);
722
723                                        final CatalogKind whatKind=getObjectType(commands);
724                                        Environmental E=null;
725                                        String roomID=null;
726                                        for(;rooms.hasMoreElements();)
727                                        {
728                                                roomID=rooms.nextElement();
729                                                R=CMLib.map().getRoom(roomID);
730                                                if(R == null)
731                                                        continue;
732                                                if(db)
733                                                        R=CMLib.coffeeMaker().makeNewRoomContent(R,false);
734                                                final List<CatalogLibrary.RoomContent> contents=CMLib.catalog().roomContent(R);
735                                                for(final CatalogLibrary.RoomContent content : contents)
736                                                {
737                                                        E=content.P();
738                                                        if(E instanceof Coins)
739                                                                continue;
740                                                        if((E instanceof MOB)
741                                                        &&(whatKind!=CatalogKind.ITEM)
742                                                        &&(!CMLib.flags().isCataloged(E)))
743                                                        {
744                                                                if(CMLib.catalog().getCatalogMob(E.Name())!=null)
745                                                                        mob.tell(L("MOB @x1 in @x2 should be tied to the catalog.",E.Name(),roomID));
746                                                                else
747                                                                        mob.tell(L("MOB @x1 in @x2 is not cataloged.",E.Name(),roomID));
748                                                        }
749
750                                                        if((E instanceof Item)
751                                                        &&(whatKind!=CatalogKind.MOB)
752                                                        &&(!CMLib.flags().isCataloged(E)))
753                                                        {
754                                                                if(CMLib.catalog().getCatalogItem(E.Name())!=null)
755                                                                        mob.tell(L("Item @x1 in @x2 should be tied to the catalog.",E.Name(),roomID));
756                                                                else
757                                                                        mob.tell(L("Item @x1 in @x2 is not cataloged.",E.Name(),roomID));
758                                                        }
759                                                }
760                                                if(db && (R!=null))
761                                                        R.destroy();
762                                        }
763                                }
764                                else
765                                        mob.tell(L("Scan what?"));
766                        }
767                        else
768                        if(commands.get(0).equalsIgnoreCase("CLEAN"))
769                        {
770                                commands.remove(0);
771                                if(checkUserRoomSetEntry(commands))
772                                {
773                                        final String which=commands.get(0).toLowerCase();
774                                        final Enumeration<String> rooms=getRoomSet(mob,which);
775                                        commands.remove(0);
776                                        final CatalogKind whatKind=getObjectType(commands);
777                                        final String type=whatKind.name().toLowerCase();
778                                        if((mob.session()!=null)
779                                        &&(mob.session().confirm(L("You are about to auto-clean (and auto-save) all @x1 @x2.\n\rThis command, if used improperly, may alter @x3 in this @x4.\n\r"
780                                                        + "Are you absolutely sure (y/N)?",which,type,type,which),"N")))
781                                        {
782                                                Physical P=null;
783                                                String roomID=null;
784                                                for(;rooms.hasMoreElements();)
785                                                {
786                                                        roomID=rooms.nextElement();
787                                                        R=CMLib.coffeeMaker().makeNewRoomContent(CMLib.map().getRoom(roomID),false);
788                                                        final List<CatalogLibrary.RoomContent> contents=CMLib.catalog().roomContent(R);
789                                                        boolean dirty=false;
790                                                        for(final CatalogLibrary.RoomContent content : contents)
791                                                        {
792                                                                P=content.P();
793                                                                if(P instanceof Coins)
794                                                                        continue;
795                                                                if((P instanceof MOB)
796                                                                &&(whatKind!=CatalogKind.ITEM))
797                                                                {
798                                                                        if((CMLib.flags().isCataloged(P))
799                                                                        &&(CMLib.catalog().getCatalogObj(P)==null))
800                                                                        {
801                                                                                CMLib.catalog().changeCatalogUsage(P,false);
802                                                                                content.flagDirty();
803                                                                                dirty=true;
804                                                                                mob.tell(L("MOB @x1 in @x2 was cleaned.",P.Name(),roomID));
805                                                                        }
806                                                                }
807                                                                else
808                                                                if((P instanceof Item)
809                                                                &&(whatKind!=CatalogKind.MOB))
810                                                                {
811                                                                        if((CMLib.flags().isCataloged(P))
812                                                                        &&(CMLib.catalog().getCatalogObj(P)==null))
813                                                                        {
814                                                                                CMLib.catalog().changeCatalogUsage(P,false);
815                                                                                content.flagDirty();
816                                                                                dirty=true;
817                                                                                mob.tell(L("Item @x1 in @x2 was cleaned.",P.Name(),roomID));
818                                                                        }
819                                                                }
820                                                        }
821                                                        if(dirty)
822                                                        {
823                                                                CMLib.catalog().updateRoomContent(roomID, contents);
824                                                                CMLib.map().resetRoom(CMLib.map().getRoom(roomID), true);
825                                                        }
826                                                        R.destroy();
827                                                }
828                                        }
829                                }
830                                else
831                                        mob.tell(L("Clean what?"));
832                        }
833                        else
834                        {
835                                Physical P=null;
836                                final String ID=CMParms.combine(commands,0);
837                                if(ID.equalsIgnoreCase("SELF")||ID.equalsIgnoreCase("ME"))
838                                        P=mob;
839                                if(P==null)
840                                        P=R.fetchFromRoomFavorMOBs(null,ID);
841                                if(P!=null)
842                                {
843                                        if(!catalog(R,mob,P))
844                                                return false;
845                                        if((P instanceof DBIdentifiable)
846                                        &&(((DBIdentifiable)P).canSaveDatabaseID())
847                                        &&(((DBIdentifiable)P).databaseID().length()>0))
848                                        {
849                                                final Room startRoom=CMLib.map().getStartRoom(P);
850                                                if(startRoom !=null)
851                                                {
852                                                        if(P instanceof MOB)
853                                                                CMLib.database().DBUpdateMOB(startRoom.roomID(),(MOB)P);
854                                                        else
855                                                                CMLib.database().DBUpdateItem(startRoom.roomID(),(Item)P);
856                                                }
857                                        }
858                                }
859                                else
860                                        mob.tell(L("You don't see '@x1' here!",ID));
861                        }
862                }
863                else
864                        mob.tell(L("Catalog huh? Try CATALOG LIST (MOBS/ITEMS/CATEGORIES) (MASK), CATALOG [mob/item name], CATALOG DELETE [mob/item name], "
865                                        + "CATALOG EDIT [item name], CATALOG CATEGORY LIST/[category name]."));
866                return false;
867        }
