035public class As extends StdCommand
036{
037        public As()
038        {
039        }
040
041        private final String[] access = I(new String[] { "AS" });
042
043        @Override
044        public String[] getAccessWords()
045        {
046                return access;
047        }
048
049        @Override
050        public boolean execute(MOB mob, List<String> commands, int metaFlags)
051                throws java.io.IOException
052        {
053                commands.remove(0);
054                if(commands.size()<2)
055                {
056                        mob.tell(L("As whom do what?"));
057                        return false;
058                }
059                final String cmd=commands.get(0);
060                commands.remove(0);
061                if((!CMSecurity.isAllowed(mob,mob.location(),CMSecurity.SecFlag.AS))||(mob.isMonster()))
062                {
063                        mob.tell(L("You aren't powerful enough to do that."));
064                        return false;
065                }
066                final Session mySession=mob.session();
067                MOB M=CMLib.players().getLoadPlayer(cmd);
068                if(M==null)
069                        M=mob.location().fetchInhabitant(cmd);
070                if(M==null)
071                {
072                        try
073                        {
074                                final List<MOB> targets=CMLib.map().findInhabitantsFavorExact(CMLib.map().rooms(), mob, cmd, false, 50);
075                                if(targets.size()>0)
076                                        M=targets.get(CMLib.dice().roll(1,targets.size(),-1));
077                        }
078                        catch (final NoSuchElementException e)
079                        {
080                        }
081                }
082                if(M==null)
083                {
084                        mob.tell(L("You don't know of anyone by that name."));
085                        return false;
086                }
087                if(M.soulMate()!=null)
088                {
089                        mob.tell(L("@x1 is being possessed at the moment.",M.Name()));
090                        return false;
091                }
092                if((CMSecurity.isASysOp(M))&&(!CMSecurity.isASysOp(mob)))
093                {
094                        mob.tell(L("You aren't powerful enough to do that."));
095                        return false;
096                }
097                if(!M.isMonster())
098                {
099                        if(!CMSecurity.isAllowedEverywhere(mob,CMSecurity.SecFlag.ORDER))
100                        {
101                                mob.tell(L("You can't do things as players if you can't order them."));
102                                return false;
103                        }
104                }
105                if(M==mob)
106                {
107                        if(commands.get(0).equalsIgnoreCase("here")
108                           ||commands.get(0).equalsIgnoreCase("."))
109                        {
110                                commands.remove(0);
111                        }
112                        M.doCommand(new XVector<String>(CMParms.toStringArray(commands)),metaFlags|MUDCmdProcessor.METAFLAG_AS);
113                        return false;
114                }
115                final Room oldRoom=M.location();
116                boolean inside=(oldRoom!=null)?oldRoom.isInhabitant(M):false;
117                final boolean dead=M.amDead();
118                final Session hisSession=M.session();
119                synchronized(mySession)
120                {
121                        //int myBitmap=mob.getBitmap();
122                        //int oldBitmap=M.getBitmap();
123                        M.setSession(mySession);
124                        mySession.setMob(M);
125                        M.setSoulMate(mob);
126                        //mySession.initTelnetMode(oldBitmap);
127                        if(commands.get(0).equalsIgnoreCase("here")
128                           ||commands.get(0).equalsIgnoreCase("."))
129                        {
130                                if((M.location()!=mob.location())&&(!mob.location().isInhabitant(M)))
131                                        mob.location().bringMobHere(M,false);
132                                commands.remove(0);
133                        }
134                        if(dead)
135                                M.bringToLife();
136                        if((M.location()==null)&&(oldRoom==null)&&(mob.location()!=null))
137                        {
138                                inside=false;
139                                mob.location().bringMobHere(M,false);
140                        }
141                }
142                CMLib.s_sleep(100);
143                M.doCommand(new XVector<String>(CMParms.toStringArray(commands)),metaFlags|MUDCmdProcessor.METAFLAG_AS);
144                synchronized(mySession)
145                {
146                        if(M.playerStats()!=null)
147                                M.playerStats().setLastUpdated(0);
148                        if((oldRoom!=null)&&(inside)&&(!oldRoom.isInhabitant(M)))
149                                oldRoom.bringMobHere(M,false);
150                        else
151                        if((oldRoom==null)||(!inside))
152                        {
153                                if(M.location()!=null)
154                                        M.location().delInhabitant(M);
155                                M.setLocation(oldRoom);
156                        }
157                        M.setSoulMate(null);
158                        M.setSession(hisSession);
159                        mySession.setMob(mob);
160                }
161                CMLib.s_sleep(100);
162                //mySession.initTelnetMode(myBitmap);
163                if(dead)
164                        M.removeFromGame(true,true);
165                return false;
166        }
167
168        @Override
169        public boolean canBeOrdered()
170        {
171                return false;
172        }
173
174        @Override
175        public boolean securityCheck(MOB mob)
176        {
177                return CMSecurity.isAllowedAnywhere(mob, CMSecurity.SecFlag.AS);
178        }
