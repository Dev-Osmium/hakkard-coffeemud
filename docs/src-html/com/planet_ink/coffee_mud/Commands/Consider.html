051        public int relativeLevelDiff(MOB mob1, MOB mob2)
052        {
053                if((mob1==null)||(mob2==null))
054                        return 0;
055                final int mob2Armor=CMLib.combat().adjustedArmor(mob2);
056                final int mob1Armor=CMLib.combat().adjustedArmor(mob1);
057                final double mob1Attack=CMLib.combat().adjustedAttackBonus(mob1,mob2);
058                final double mob2Attack=CMLib.combat().adjustedAttackBonus(mob2,mob1);
059                final int mob2Dmg=mob2.phyStats().damage();
060                final int mob1Dmg=mob1.phyStats().damage();
061                final int mob2Hp=mob2.baseState().getHitPoints();
062                final int mob1Hp=mob1.baseState().getHitPoints();
063
064                final double mob2HitRound=(((CMath.div(CMLib.dice().normalizeBy5((int)Math.round(50.0*mob2Attack/mob1Armor)),100.0))*CMath.div(mob2Dmg,2.0))+1.0)*CMath.mul(mob2.phyStats().speed(),1.0);
065                final double mob1HitRound=(((CMath.div(CMLib.dice().normalizeBy5((int)Math.round(50.0*mob1Attack/mob2Armor)),100.0))*CMath.div(mob1Dmg,2.0))+1.0)*CMath.mul(mob1.phyStats().speed(),1.0);
066                final double mob2SurvivalRounds=CMath.div(mob2Hp,mob1HitRound);
067                final double mob1SurvivalRounds=CMath.div(mob1Hp,mob2HitRound);
068
069                //int levelDiff=(int)Math.round(CMath.div((mob1SurvivalRounds-mob2SurvivalRounds),1));
070                final double levelDiff=(mob1SurvivalRounds-mob2SurvivalRounds)/2;
071                final int levelDiffed=(int)Math.round(Math.sqrt(Math.abs(levelDiff)));
072
073                return levelDiffed*(levelDiff<0.0?-1:1);
074        }

076        public int doConsider(MOB mob, Physical target)
077        {
078                final Room R=mob.location();
079                if(R==null)
080                        return 0;
081                final CMMsg msg=CMClass.getMsg(mob,target,null,CMMsg.MASK_EYES|CMMsg.TYP_OK_VISUAL,null,L("<S-NAME> consider(s) <T-NAMESELF>."),L("<S-NAME> consider(s) <T-NAMESELF>."));
082                if(R.okMessage(mob,msg))
083                        R.send(mob,msg);
084                int lvlDiff=0;
085                if(target instanceof MOB)
086                {
087                        final MOB targetMOB=(MOB)target;
088                        final int relDiff=relativeLevelDiff(targetMOB,mob);
089                        lvlDiff=(target.phyStats().level()-mob.phyStats().level());
090                        final int realDiff=relDiff;//(relDiff+lvlDiff)/2;
091
092                        int theDiff=2;
093                        if(mob.phyStats().level()>20)
094                                theDiff=3;
095                        if(mob.phyStats().level()>40)
096                                theDiff=4;
097                        if(mob.phyStats().level()>60)
098                                theDiff=5;
099                        if(mob.phyStats().level()>80)
100                                theDiff=6;
101
102                        StringBuilder levelMsg=new StringBuilder("");
103                        if(lvlDiff==0)
104                                levelMsg.append(L(targetMOB.charStats().HeShe()+" is your equal"));
105                        else
106                        if(lvlDiff<-CMProps.getIntVar(CMProps.Int.EXPRATE))
107                                levelMsg.append(L(targetMOB.charStats().HeShe()+" is vastly inferior to you"));
108                        else
109                        if(lvlDiff>CMProps.getIntVar(CMProps.Int.EXPRATE))
110                                levelMsg.append(L(targetMOB.charStats().HeShe()+" is far superior to you"));
111                        else
112                        if(CMProps.getIntVar(CMProps.Int.EXPRATE)!=0)
113                        {
114                                final int relLvlDiff=(lvlDiff<0)?-lvlDiff:lvlDiff;
115                                final double pct=CMath.div(relLvlDiff,CMProps.getIntVar(CMProps.Int.EXPRATE));
116                                if((lvlDiff<0)&&(pct<0.5))
117                                        levelMsg.append(L(targetMOB.charStats().HeShe()+" is almost your equal"));
118                                else
119                                if((lvlDiff<0)&&(pct<=1.0))
120                                        levelMsg.append(L(targetMOB.charStats().HeShe()+" is somewhat inferior to you"));
121                                else
122                                if((lvlDiff<0))
123                                        levelMsg.append(L(targetMOB.charStats().HeShe()+" is inferior to you"));
124                                else
125                                if((lvlDiff>0)&&(pct<0.5))
126                                        levelMsg.append(L("You are almost "+targetMOB.charStats().hisher()+" equal"));
127                                else
128                                if((lvlDiff>0)&&(pct<0.8))
129                                        levelMsg.append(L(targetMOB.charStats().HeShe()+" is somewhat superior to you"));
130                                else
131                                        levelMsg.append(L(targetMOB.charStats().HeShe()+" is superior to you"));
132                        }
133
134                        final int levelDiff=Math.abs(realDiff);
135                        if(levelDiff<theDiff)
136                        {
137                                levelMsg.append(L((lvlDiff!=0)?" but ":" and "));
138                                levelMsg.append(L("the perfect match!"));
139                        }
140                        else
141                        if(realDiff<0)
142                        {
143                                levelMsg.append(L((lvlDiff<0)?" and ":" but "));
144                                if(realDiff>-(2*theDiff))
145                                        levelMsg.append(L(targetMOB.charStats().heshe()+" might actually give you a fight."));
146                                else
147                                if(realDiff>-(3*theDiff))
148                                        levelMsg.append(L(targetMOB.charStats().heshe()+" won't put up a big fight."));
149                                else
150                                if(realDiff>-(4*theDiff))
151                                        levelMsg.append(L(targetMOB.charStats().heshe()+" is basically a pushover."));
152                                else
153                                        levelMsg.append(L(targetMOB.charStats().heshe()+" is an easy kill."));
154                        }
155                        else
156                        {
157                                levelMsg.append(L((lvlDiff>0)?" and ":" but "));
158                                if(realDiff<(2*theDiff))
159                                        levelMsg.append(L(targetMOB.charStats().heshe()+" looks a little tough."));
160                                else
161                                if(realDiff<(3*theDiff))
162                                        levelMsg.append(L(targetMOB.charStats().heshe()+" is a serious threat."));
163                                else
164                                if(realDiff<(4*theDiff))
165                                        levelMsg.append(L(targetMOB.charStats().heshe()+" will clean your clock."));
166                                else
167                                        levelMsg.append(L(targetMOB.charStats().heshe()+" WILL KILL YOU DEAD!"));
168                        }
169                        mob.tell(levelMsg.toString());
170                }
171                final StringBuffer withWhat=new StringBuffer("");
172                final Vector<Ability> mendors=new Vector<Ability>();
173                for(final Enumeration<Ability> a=mob.allAbilities();a.hasMoreElements();)
174                {
175                        final Ability A=a.nextElement();
176                        if((A instanceof MendingSkill)&&(((MendingSkill)A).supportsMending((target))))
177                                mendors.add(A);
178                }
179                for(int m=0;m<mendors.size();m++)
180                {
181                        final Ability A=mendors.get(m);
182                        if(m==0)
183                                withWhat.append(L("You could probably help @x1 out with your @x2 skill",target.name(mob),A.name()));
184                        else
185                        if(m<mendors.size()-1)
186                                withWhat.append(L(", your @x1 skill",A.name()));
187                        else
188                                withWhat.append(L(" or your @x1 skill",A.name()));
189                }
190
191                if(withWhat.length()>0)
192                        mob.tell(withWhat.toString()+".");
193                else
194                if(!(target instanceof MOB))
195                        mob.tell(L("You don't have any particular thoughts about that."));
196                return lvlDiff;
197        }

199        @Override
200        public boolean execute(MOB mob, List<String> commands, int metaFlags)
201                throws java.io.IOException
202        {
203                Vector<String> origCmds=new XVector<String>(commands);
204                Physical target=null;
205                if(commands.size()<2)
206                {
207                        CMLib.commands().postCommandFail(mob,origCmds,L("Consider whom or what?"));
208                        return false;
209                }
210                commands.remove(0);
211                final String targetName=CMParms.combine(commands,0);
212                if(ID.equalsIgnoreCase("SELF")||ID.equalsIgnoreCase("ME"))
213                        target=mob;
214                if(target==null)
215                        target=mob.location().fetchFromMOBRoomFavorsMOBs(mob,null,targetName,Wearable.FILTER_ANY);
216                if((target==null)||(!CMLib.flags().canBeSeenBy(target,mob)))
217                {
218                        CMLib.commands().postCommandFail(mob,origCmds,L("I don't see '@x1' here.",targetName));
219                        return false;
220                }
221                doConsider(mob,target);
222                return true;
223        }
224
225        @Override
226        public Object executeInternal(MOB mob, int metaFlags, Object... args) throws java.io.IOException
227        {
228                if(!super.checkArguments(internalParameters, args))
229                        return Integer.valueOf(0);
230                return Integer.valueOf(doConsider(mob, (MOB)args[0]));
231        }
