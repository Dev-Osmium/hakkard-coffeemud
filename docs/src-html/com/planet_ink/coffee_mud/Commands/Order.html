035public class Order extends StdCommand
036{
037        public Order()
038        {
039        }
040
041        private final String[]  access  = I(new String[] { "ORDER" });
042
043        @Override
044        public String[] getAccessWords()
045        {
046                return access;
047        }
048
049        @Override
050        public boolean execute(MOB mob, List<String> commands, int metaFlags)
051                throws java.io.IOException
052        {
053                Vector<String> origCmds=new XVector<String>(commands);
054                if(commands.size()<3)
055                {
056                        CMLib.commands().postCommandFail(mob,origCmds,L("Order who do to what?"));
057                        return false;
058                }
059                commands.remove(0);
060                if(commands.size()<2)
061                {
062                        CMLib.commands().postCommandFail(mob,origCmds,L("Order them to do what?"));
063                        return false;
064                }
065                if((!CMSecurity.isAllowed(mob,mob.location(),CMSecurity.SecFlag.ORDER))
066                &&(!mob.isMonster())
067                &&(mob.isAttributeSet(MOB.Attrib.AUTOASSIST)))
068                {
069                        CMLib.commands().postCommandFail(mob,origCmds,L("You may not order someone around with your AUTOASSIST flag off."));
070                        return false;
071                }
072
073                String whomToOrder=commands.get(0);
074                final Vector<MOB> V=new Vector<MOB>();
075                boolean allFlag=whomToOrder.equalsIgnoreCase("all");
076                if (whomToOrder.toUpperCase().startsWith("ALL."))
077                {
078                        allFlag = true;
079                        whomToOrder = "ALL " + whomToOrder.substring(4);
080                }
081                if (whomToOrder.toUpperCase().endsWith(".ALL"))
082                {
083                        allFlag = true;
084                        whomToOrder = "ALL " + whomToOrder.substring(0, whomToOrder.length() - 4);
085                }
086                int addendum=1;
087                String addendumStr="";
088                boolean doBugFix = true;
089                while(doBugFix || allFlag)
090                {
091                        doBugFix=false;
092                        final MOB target=mob.location().fetchInhabitant(whomToOrder+addendumStr);
093                        if(target==null)
094                                break;
095                        if((CMLib.flags().canBeSeenBy(target,mob))
096                        &&(target!=mob)
097                        &&(!V.contains(target)))
098                                V.add(target);
099                        addendumStr="."+(++addendum);
100                }
101
102                if(V.size()==0)
103                {
104                        if(whomToOrder.equalsIgnoreCase("ALL"))
105                                CMLib.commands().postCommandFail(mob,origCmds,L("You don't see anyone called '@x1' here.",whomToOrder));
106                        else
107                                CMLib.commands().postCommandFail(mob,origCmds,L("You don't see anyone here."));
108                        return false;
109                }
110
111                MOB target=null;
112                if(V.size()==1)
113                {
114                        target=V.get(0);
115                        if((!CMLib.flags().canBeSeenBy(target,mob))
116                        ||(!CMLib.flags().canBeHeardSpeakingBy(mob,target))
117                        ||(target.location()!=mob.location()))
118                        {
119                                CMLib.commands().postCommandFail(mob,origCmds,L("'@x1' doesn't seem to be listening.",whomToOrder));
120                                return false;
121                        }
122                        if(!target.willFollowOrdersOf(mob))
123                        {
124                                CMLib.commands().postCommandFail(mob,origCmds,L("You can't order '@x1' around.",target.name(mob)));
125                                return false;
126                        }
127                }
128
129                commands.remove(0);
130
131                CMObject O=CMLib.english().findCommand(mob,commands);
132                final String order=CMParms.combine(commands,0);
133                if(!CMSecurity.isAllowed(mob,mob.location(),CMSecurity.SecFlag.ORDER))
134                {
135                        if((O instanceof Command)&&(!((Command)O).canBeOrdered()))
136                        {
137                                CMLib.commands().postCommandFail(mob,origCmds,L("You can't order anyone to '@x1'.",order));
138                                return false;
139                        }
140                }
141
142                final Vector<MOB> doV=new Vector<MOB>();
143                for(int v=0;v<V.size();v++)
144                {
145                        target=V.get(v);
146                        O=CMLib.english().findCommand(target,new XVector<String>(commands));
147                        if(!CMSecurity.isAllowed(mob,mob.location(),CMSecurity.SecFlag.ORDER))
148                        {
149                                if((O instanceof Command)
150                                &&((!((Command)O).canBeOrdered())||(!((Command)O).securityCheck(mob))))
151                                {
152                                        CMLib.commands().postCommandFail(mob,origCmds,L("You can't order @x1 to '@x2'.",target.name(mob),order));
153                                        continue;
154                                }
155                                if(O instanceof Ability)
156                                        O=CMLib.english().getToEvoke(target,new XVector<String>(commands));
157                                if(O instanceof Ability)
158                                {
159                                        if(CMath.bset(((Ability)O).flags(),Ability.FLAG_NOORDERING))
160                                        {
161                                                CMLib.commands().postCommandFail(mob,origCmds,L("You can't order @x1 to '@x2'.",target.name(mob),order));
162                                                continue;
163                                        }
164                                }
165                        }
166                        if((!CMLib.flags().canBeSeenBy(target,mob))
167                        ||(!CMLib.flags().canBeHeardSpeakingBy(mob,target))
168                        ||(target.location()!=mob.location()))
169                                CMLib.commands().postCommandFail(mob,origCmds,L("'@x1' doesn't seem to be listening.",whomToOrder));
170                        else
171                        if(!target.willFollowOrdersOf(mob))
172                                CMLib.commands().postCommandFail(mob,origCmds,L("You can't order '@x1' around.",target.name(mob)));
173                        else
174                        {
175                                final CMMsg msg=CMClass.getMsg(mob,target,null,CMMsg.MSG_SPEAK,CMMsg.MSG_ORDER,CMMsg.MSG_SPEAK,L("^T<S-NAME> order(s) <T-NAMESELF> to '@x1'^?.",order));
176                                if((mob.location().okMessage(mob,msg)))
177                                {
178                                        mob.location().send(mob,msg);
179                                        if((msg.targetMinor()==CMMsg.TYP_ORDER)&&(msg.target()==target))
180                                                doV.add(target);
181                                }
182                        }
183                }
184                for(int v=0;v<doV.size();v++)
185                {
186                        target=doV.get(v);
187                        target.enqueCommand(new XVector<String>(commands),metaFlags|MUDCmdProcessor.METAFLAG_ORDER,0);
188                }
189                return false;
190        }
191
192        @Override
193        public double combatActionsCost(final MOB mob, final List<String> cmds)
194        {
195                return CMProps.getCommandCombatActionCost(ID());
196        }
197
198        @Override
199        public double actionsCost(final MOB mob, final List<String> cmds)
200        {
201                return CMProps.getCommandActionCost(ID());
202        }
203
204        @Override
205        public boolean canBeOrdered()
206        {
207                return true;
208        }
