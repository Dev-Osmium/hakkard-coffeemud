035public class Train extends StdCommand
036{
037        public Train()
038        {
039        }
040
041        private final String[] access=I(new String[]{"TRAIN","TR","TRA"});
042        @Override
043        public String[] getAccessWords()
044        {
045                return access;
046        }
047
048        private static final int        TRAIN_HITPOINTS = 101;
049        private static final int        TRAIN_MANA              = 102;
050        private static final int        TRAIN_MOVE              = 103;
051        private static final int        TRAIN_GAIN              = 104;
052        private static final int        TRAIN_PRACTICES = 105;
053        private static final int        TRAIN_CCLASS    = 106;
054        
055        public static List<String> getAllPossibleThingsToTrainFor()
056        {
057                final List<String> V=new Vector<String>();
058                V.add("HIT POINTS");
059                V.add("MANA");
060                V.add("MOVEMENT");
061                V.add("GAIN");
062                V.add("PRACTICES");
063                for(final int i: CharStats.CODES.BASECODES())
064                        V.add(CharStats.CODES.DESC(i));
065                for(final Enumeration<CharClass> c=CMClass.charClasses();c.hasMoreElements();)
066                {
067                        final CharClass C=c.nextElement();
068                        if(CMLib.login().isAvailableCharClass(C))
069                                V.add(C.name().toUpperCase().trim());
070                }
071                return V;
072        }
073
074        @Override
075        public boolean execute(MOB mob, List<String> commands, int metaFlags)
076                throws java.io.IOException
077        {
078                List<String> origCmds=new StringXVector(commands);
079                if(commands.size()<2)
080                {
081                        CMLib.commands().postCommandFail(mob,origCmds,L("You have @x1 training sessions. Enter HELP TRAIN for more information.",""+mob.getTrains()));
082                        return false;
083                }
084                commands.remove(0);
085                String teacherName=null;
086                if(commands.size()>1)
087                {
088                        teacherName=commands.get(commands.size()-1);
089                        if(teacherName.length()>1)
090                                commands.remove(commands.size()-1);
091                        else
092                                teacherName=null;
093                }
094
095                final String abilityName=CMParms.combine(commands,0).toUpperCase();
096                final StringBuffer thingsToTrainFor=new StringBuffer("");
097                for(final int i: CharStats.CODES.BASECODES())
098                        thingsToTrainFor.append(CharStats.CODES.DESC(i)+", ");
099
100                int trainsRequired=1;
101                int abilityCode=mob.baseCharStats().getCode(abilityName);
102                int curStat=-1;
103                if((abilityCode>=0)&&(CharStats.CODES.isBASE(abilityCode)))
104                {
105                        curStat=mob.baseCharStats().getRacialStat(mob, abilityCode);
106                        trainsRequired=CMLib.login().getTrainingCost(mob, abilityCode, false);
107                        if(trainsRequired<0)
108                                return false;
109                }
110                else
111                        abilityCode=-1;
112                CharClass theClass=null;
113                if((!CMSecurity.isDisabled(CMSecurity.DisFlag.CLASSTRAINING))&&(abilityCode<0))
114                {
115                        for(final Enumeration<CharClass> c=CMClass.charClasses();c.hasMoreElements();)
116                        {
117                                final CharClass C=c.nextElement();
118                                int classLevel=mob.charStats().getClassLevel(C);
119                                if(classLevel<0)
120                                        classLevel=0;
121                                if((C.name().toUpperCase().startsWith(abilityName.toUpperCase()))
122                                ||(C.name(classLevel).toUpperCase().startsWith(abilityName.toUpperCase())))
123                                {
124                                        if((C.qualifiesForThisClass(mob,false))
125                                        &&(CMLib.login().isAvailableCharClass(C)))
126                                        {
127                                                abilityCode=TRAIN_CCLASS;
128                                                theClass=C;
129                                        }
130                                        break;
131                                }
132                                else
133                                if((C.qualifiesForThisClass(mob,true))
134                                &&(CMLib.login().isAvailableCharClass(C)))
135                                        thingsToTrainFor.append(C.name()+", ");
136                        }
137                }
138
139                if(abilityCode<0)
140                {
141                        if("HIT POINTS".startsWith(abilityName.toUpperCase()))
142                                abilityCode=TRAIN_HITPOINTS;
143                        else
144                        if("MANA".startsWith(abilityName.toUpperCase()))
145                                abilityCode=TRAIN_MANA;
146                        else
147                        if("MOVE".startsWith(abilityName.toUpperCase()))
148                                abilityCode=TRAIN_MOVE;
149                        else
150                        if("GAIN".startsWith(abilityName.toUpperCase()))
151                                abilityCode=TRAIN_GAIN;
152                        else
153                        if("PRACTICES".startsWith(abilityName.toUpperCase()))
154                                abilityCode=TRAIN_PRACTICES;
155                        else
156                        {
157                                if(abilityCode<0)
158                                {
159                                        CMLib.commands().postCommandFail(mob,origCmds,L("You can't train for '@x1'. Try @x2HIT POINTS, MANA, MOVE, GAIN, or PRACTICES.",abilityName,thingsToTrainFor.toString()));
160                                        return false;
161                                }
162                        }
163                }
164
165                if(abilityCode==TRAIN_GAIN)
166                {
167                        if(mob.getPractices()<7)
168                        {
169                                CMLib.commands().postCommandFail(mob,origCmds,L("You don't seem to have enough practices to do that."));
170                                return false;
171                        }
172                }
173                else
174                if(mob.getTrains()<=0)
175                {
176                        CMLib.commands().postCommandFail(mob,origCmds,L("You don't seem to have enough training sessions to do that."));
177                        return false;
178                }
179                else
180                if(mob.getTrains()<trainsRequired)
181                {
182                        if(trainsRequired>1)
183                        {
184                                CMLib.commands().postCommandFail(mob,origCmds,L("Training that ability further will require @x1 training points.",""+trainsRequired));
185                                return false;
186                        }
187                        else
188                        if(trainsRequired==1)
189                        {
190                                CMLib.commands().postCommandFail(mob,origCmds,L("Training that ability further will require @x1 training points.",""+trainsRequired));
191                                return false;
192                        }
193                }
194
195                MOB teacher=null;
196                if(teacherName!=null)
197                        teacher=mob.location().fetchInhabitant(teacherName);
198                if(teacher==null)
199                for(int i=0;i<mob.location().numInhabitants();i++)
200                {
201                        final MOB possTeach=mob.location().fetchInhabitant(i);
202                        if((possTeach!=null)&&(possTeach!=mob))
203                        {
204                                teacher=possTeach;
205                                break;
206                        }
207                }
208                if((teacher==null)||(!CMLib.flags().canBeSeenBy(teacher,mob)))
209                {
210                        mob.tell(teacher,null,null,L("You can't see <S-NAME>!"));
211                        return false;
212                }
213                if(!CMLib.flags().canBeSeenBy(mob,teacher))
214                {
215                        mob.tell(teacher,null,null,L("<S-NAME> can't see you!"));
216                        return false;
217                }
218                if(teacher==mob)
219                {
220                        CMLib.commands().postCommandFail(mob,origCmds,L("You cannot train with yourself!"));
221                        return false;
222                }
223                if(teacher.isAttributeSet(MOB.Attrib.NOTEACH))
224                {
225                        CMLib.commands().postCommandFail(mob,origCmds,L("@x1 is refusing to teach right now.",teacher.name()));
226                        return false;
227                }
228                if(mob.isAttributeSet(MOB.Attrib.NOTEACH))
229                {
230                        CMLib.commands().postCommandFail(mob,origCmds,L("You are refusing training at this time."));
231                        return false;
232                }
233                if(CMLib.flags().isSleeping(mob)||CMLib.flags().isSitting(mob))
234                {
235                        CMLib.commands().postCommandFail(mob,origCmds,L("You need to stand up for your training."));
236                        return false;
237                }
238                if(CMLib.flags().isSleeping(teacher)||CMLib.flags().isSitting(teacher))
239                {
240                        if(teacher.isMonster())
241                                CMLib.commands().postStand(teacher,true, false);
242                        if(CMLib.flags().isSleeping(teacher)||CMLib.flags().isSitting(teacher))
243                        {
244                                CMLib.commands().postCommandFail(mob,origCmds,L("@x1 looks a bit too relaxed to train with you.",teacher.name()));
245                                return false;
246                        }
247                }
248                if(mob.isInCombat())
249                {
250                        CMLib.commands().postCommandFail(mob,origCmds,L("Not while you are fighting!"));
251                        return false;
252                }
253                if(teacher.isInCombat())
254                {
255                        CMLib.commands().postCommandFail(mob,origCmds,L("Your teacher seems busy right now."));
256                        return false;
257                }
258
259                if(abilityCode==TRAIN_CCLASS)
260                {
261                        boolean canTeach=false;
262                        for(int c=0;c<teacher.charStats().numClasses();c++)
263                        {
264                                if(teacher.charStats().getMyClass(c).baseClass().equals(mob.charStats().getCurrentClass().baseClass()))
265                                        canTeach=true;
266                        }
267                        if((!canTeach)
268                        &&(teacher.charStats().getClassLevel(theClass)<1))
269                        {
270                                if((!CMProps.getVar(CMProps.Str.MULTICLASS).startsWith("MULTI"))
271                                &&(!CMProps.getVar(CMProps.Str.MULTICLASS).endsWith("MULTI")))
272                                {
273                                        final CharClass C=CMClass.getCharClass(mob.charStats().getCurrentClass().baseClass());
274                                        final String baseClassName=(C!=null)?C.name():mob.charStats().getCurrentClass().baseClass();
275                                        CMLib.commands().postCommandFail(mob,origCmds,L("You can only learn that from another @x1.",baseClassName));
276                                }
277                                else
278                                if(theClass!=null)
279                                {
280                                        int classLevel=mob.charStats().getClassLevel(theClass);
281                                        if(classLevel<0)
282                                                classLevel=0;
283                                        CMLib.commands().postCommandFail(mob,origCmds,L("You can only learn that from another @x1.",theClass.name(classLevel)));
284                                }
285                                return false;
286                        }
287                }
288
289                if(abilityCode<100)
290                {
291                        final int teachStat=teacher.charStats().getStat(abilityCode);
292                        if(curStat>=teachStat)
293                        {
294                                CMLib.commands().postCommandFail(mob,origCmds,L("You can only train with someone whose score is higher than yours."));
295                                return false;
296                        }
297                        curStat=mob.baseCharStats().getStat(abilityCode);
298                }
299
300                final CMMsg msg=CMClass.getMsg(teacher,mob,null,CMMsg.MSG_NOISYMOVEMENT,L("<S-NAME> train(s) with <T-NAMESELF>."));
301                if(!mob.location().okMessage(mob,msg))
302                        return false;
303                mob.location().send(mob,msg);
304                switch(abilityCode)
305                {
306                case CharStats.STAT_STRENGTH:
307                        mob.tell(L("You feel stronger!"));
308                        mob.baseCharStats().setStat(CharStats.STAT_STRENGTH,curStat+1);
309                        mob.recoverCharStats();
310                        mob.setTrains(mob.getTrains()-trainsRequired);
311                        break;
312                case CharStats.STAT_INTELLIGENCE:
313                        mob.tell(L("You feel smarter!"));
314                        mob.baseCharStats().setStat(CharStats.STAT_INTELLIGENCE,curStat+1);
315                        mob.recoverCharStats();
316                        mob.setTrains(mob.getTrains()-trainsRequired);
317                        break;
318                case CharStats.STAT_DEXTERITY:
319                        mob.tell(L("You feel more dextrous!"));
320                        mob.baseCharStats().setStat(CharStats.STAT_DEXTERITY,curStat+1);
321                        mob.recoverCharStats();
322                        mob.setTrains(mob.getTrains()-trainsRequired);
323                        break;
324                case CharStats.STAT_CONSTITUTION:
325                        mob.tell(L("You feel healthier!"));
326                        mob.baseCharStats().setStat(CharStats.STAT_CONSTITUTION,curStat+1);
327                        mob.recoverCharStats();
328                        mob.setTrains(mob.getTrains()-trainsRequired);
329                        break;
330                case CharStats.STAT_CHARISMA:
331                        mob.tell(L("You feel more charismatic!"));
332                        mob.baseCharStats().setStat(CharStats.STAT_CHARISMA,curStat+1);
333                        mob.recoverCharStats();
334                        mob.setTrains(mob.getTrains()-trainsRequired);
335                        break;
336                case CharStats.STAT_WISDOM:
337                        mob.tell(L("You feel wiser!"));
338                        mob.baseCharStats().setStat(CharStats.STAT_WISDOM,curStat+1);
339                        mob.recoverCharStats();
340                        mob.setTrains(mob.getTrains()-trainsRequired);
341                        break;
342                case TRAIN_HITPOINTS:
343                        mob.tell(L("You feel even healthier!"));
344                        mob.baseState().setHitPoints(mob.baseState().getHitPoints()+10);
345                        mob.maxState().setHitPoints(mob.maxState().getHitPoints()+10);
346                        mob.curState().setHitPoints(mob.curState().getHitPoints()+10);
347                        mob.setTrains(mob.getTrains()-1);
348                        break;
349                case TRAIN_MANA:
350                        mob.tell(L("You feel more powerful!"));
351                        mob.baseState().setMana(mob.baseState().getMana()+20);
352                        mob.maxState().setMana(mob.maxState().getMana()+20);
353                        mob.curState().setMana(mob.curState().getMana()+20);
354                        mob.setTrains(mob.getTrains()-1);
355                        break;
356                case TRAIN_MOVE:
357                        mob.tell(L("You feel more rested!"));
358                        mob.baseState().setMovement(mob.baseState().getMovement()+20);
359                        mob.maxState().setMovement(mob.maxState().getMovement()+20);
360                        mob.curState().setMovement(mob.curState().getMovement()+20);
361                        mob.setTrains(mob.getTrains()-1);
362                        break;
363                case TRAIN_GAIN:
364                        mob.tell(L("You feel more trainable!"));
365                        mob.setTrains(mob.getTrains()+1);
366                        mob.setPractices(mob.getPractices()-7);
367                        break;
368                case TRAIN_PRACTICES:
369                        mob.tell(L("You feel more educatable!"));
370                        mob.setTrains(mob.getTrains()-1);
371                        mob.setPractices(mob.getPractices()+5);
372                        break;
373                default:
374                        if(CMParms.contains(CharStats.CODES.BASECODES(), abilityCode))
375                        {
376                                mob.tell(L("You feel more @x1!",CharStats.CODES.NAME(abilityCode)));
377                                mob.baseCharStats().setStat(abilityCode,curStat+1);
378                                mob.recoverCharStats();
379                                mob.setTrains(mob.getTrains()-trainsRequired);
380                        }
381                        break;
382                case TRAIN_CCLASS:
383                        if(theClass!=null)
384                        {
385                                int classLevel=mob.charStats().getClassLevel(theClass);
386                                if(classLevel<0)
387                                        classLevel=0;
388                                mob.tell(L("You have undergone @x1 training!",theClass.name(classLevel)));
389                                mob.setTrains(mob.getTrains()-1);
390                                mob.baseCharStats().getCurrentClass().endCharacter(mob);
391                                mob.baseCharStats().setCurrentClass(theClass);
392                                if((!mob.isMonster())&&(mob.soulMate()==null))
393                                        CMLib.coffeeTables().bump(mob,CoffeeTableRow.STAT_CLASSCHANGE);
394                                mob.recoverCharStats();
395                                mob.charStats().getCurrentClass().startCharacter(mob,false,true);
396                        }
397                        break;
398                }
399                return false;
400        }
