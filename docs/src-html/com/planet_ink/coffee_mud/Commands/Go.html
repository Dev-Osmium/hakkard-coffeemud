037{
038        public Go()
039        {
040        }
041
042        private final String[]  access  = I(new String[] { "GO", "WALK" });
043
044        @Override
045        public String[] getAccessWords()
046        {
047                return access;
048        }
049
050        protected Command               stander         = null;
051        protected List<String>  ifneccvec       = null;
052
053        public boolean standIfNecessary(final MOB mob, final List<String> commands, final int metaFlags, final boolean giveMsg)
054                throws java.io.IOException
055        {
056                if(CMLib.flags().isFlying(mob))
057                        return true;
058                final boolean wasStanding = CMLib.flags().isStanding(mob) && (!CMLib.flags().isSleeping(mob));
059                if((ifneccvec==null)||(ifneccvec.size()!=2))
060                {
061                        ifneccvec=new Vector<String>();
062                        ifneccvec.add("STAND");
063                        ifneccvec.add("IFNECESSARY");
064                }
065                if(stander==null)
066                        stander=CMClass.getCommand("Stand");
067                if((stander!=null)&&(ifneccvec!=null))
068                        stander.execute(mob,ifneccvec,metaFlags);
069                final boolean isStanding = CMLib.flags().isStanding(mob) && (!CMLib.flags().isSleeping(mob));
070                if(giveMsg && (!isStanding))
071                {
072                        if(CMLib.flags().isSleeping(mob))
073                                CMLib.commands().postCommandFail(mob, commands, L("You need to wake up first."));
074                        else
075                        if(!wasStanding)
076                                CMLib.commands().postCommandFail(mob, commands, L("You failed to stand up. You might try crawling."));
077                        else
078                                CMLib.commands().postCommandFail(mob, commands, L("You need to stand up."));
079                }
080                return isStanding;
081        }
082
083        @Override
084        public boolean execute(final MOB mob, final List<String> commands, final int metaFlags)
085                throws java.io.IOException
086        {
087                final Vector<String> origCmds=new XVector<String>(commands);
088                if(!standIfNecessary(mob,commands, metaFlags, true))
089                        return false;
090
091                final String whereStr=CMParms.combine(commands,1);
092                final Room R=mob.location();
093                if(R==null)
094                        return false;
095
096                final boolean inAShip =(R instanceof BoardableShip)||(R.getArea() instanceof BoardableShip);
097                final String validDirs = inAShip?Directions.SHIP_NAMES_LIST() : Directions.NAMES_LIST();
098                final boolean running = mob.isAttributeSet(MOB.Attrib.AUTORUN);
099
100                int direction=-1;
101                if(whereStr.equalsIgnoreCase("OUT"))
102                {
103                        if(!CMath.bset(R.domainType(),Room.INDOORS))
104                        {
105                                CMLib.commands().postCommandFail(mob,origCmds,L("You aren't indoors."));
106                                return false;
107                        }
108
109                        for(int d=Directions.NUM_DIRECTIONS()-1;d>=0;d--)
110                        {
111                                if((R.getExitInDir(d)!=null)
112                                &&(R.getRoomInDir(d)!=null)
113                                &&(!CMath.bset(R.getRoomInDir(d).domainType(),Room.INDOORS)))
114                                {
115                                        if(direction>=0)
116                                        {
117                                                CMLib.commands().postCommandFail(mob,origCmds,L("Which way out?  Try @x1.",validDirs));
118                                                return false;
119                                        }
120                                        direction=d;
121                                }
122                        }
123                        if(direction<0)
124                        {
125                                CMLib.commands().postCommandFail(mob,origCmds,L("There is no direct way out of this place.  Try a direction."));
126                                return false;
127                        }
128                }
129                if(direction<0)
130                {
131                        if(mob.isMonster())
132                                direction=CMLib.directions().getGoodDirectionCode(whereStr);
133                        else
134                                direction=(inAShip)?CMLib.directions().getGoodShipDirectionCode(whereStr):CMLib.directions().getGoodCompassDirectionCode(whereStr);
135                }
136                if(direction<0)
137                {
138                        final Environmental E=R.fetchFromRoomFavorItems(null,whereStr);
139                        if(E instanceof Rideable)
140                        {
141                                final Command C=CMClass.getCommand("Enter");
142                                return C.execute(mob,commands,metaFlags);
143                        }
144                        if(E instanceof Exit)
145                        {
146                                for(int d=Directions.NUM_DIRECTIONS()-1;d>=0;d--)
147                                {
148                                        if(R.getExitInDir(d)==E)
149                                        {
150                                                direction=d;
151                                                break;
152                                        }
153                                }
154                        }
155                }
156                final String doing=commands.get(0);
157                if(direction>=0)
158                {
159                        final boolean success;
160                        if(running)
161                                success=CMLib.tracking().run(mob,direction,false,false,false,false);
162                        else
163                                success=CMLib.tracking().walk(mob,direction,false,false,false,false);
164                        if(!success)
165                                mob.clearCommandQueue();
166                }
167                else
168                {
169                        Exit E=R.fetchExit(whereStr);
170                        if(E != null)
171                                return CMLib.commands().forceStandardCommand(mob, "Enter", commands);
172
173                        boolean doneAnything=false;
174                        final List<List<String>> prequeCommands=new ArrayList<List<String>>();
175                        for(int v=1;v<commands.size();v++)
176                        {
177                                int num=1;
178                                String s=commands.get(v);
179                                if(CMath.s_int(s)>0)
180                                {
181                                        num=CMath.s_int(s);
182                                        v++;
183                                        if(v<commands.size())
184                                                s=commands.get(v);
185                                }
186                                else
187                                if((s.length()>0) && (Character.isDigit(s.charAt(0))))
188                                {
189                                        int x=1;
190                                        while((x<s.length()-1)&&(Character.isDigit(s.charAt(x))))
191                                                x++;
192                                        num=CMath.s_int(s.substring(0,x));
193                                        s=s.substring(x);
194                                }
195
196                                if(mob.isMonster())
197                                        direction=CMLib.directions().getGoodDirectionCode(s);
198                                else
199                                        direction=(inAShip)?CMLib.directions().getGoodShipDirectionCode(s):CMLib.directions().getGoodCompassDirectionCode(s);
200                                if(direction>=0)
201                                {
202                                        doneAnything=true;
203                                        for(int i=0;i<num;i++)
204                                        {
205                                                if(mob.isMonster())
206                                                {
207                                                        if(running)
208                                                        {
209                                                                if(!CMLib.tracking().run(mob,direction,false,false,false,false))
210                                                                        return false;
211                                                        }
212                                                        else
213                                                        if(!CMLib.tracking().walk(mob,direction,false,false,false,false))
214                                                                return false;
215                                                }
216                                                else
217                                                {
218                                                        final Vector<String> V=new Vector<String>();
219                                                        V.add(doing);
220                                                        V.add(inAShip?CMLib.directions().getShipDirectionName(direction):CMLib.directions().getDirectionName(direction));
221                                                        prequeCommands.add(V);
222                                                }
223                                        }
224                                }
225                                else
226                                {
227                                        E=R.fetchExit(s);
228                                        if(E!=null)
229                                        {
230                                                if(mob.isMonster())
231                                                        CMLib.commands().forceStandardCommand(mob, "Enter", new XVector<String>("ENTER",s));
232                                                else
233                                                        prequeCommands.add(new XVector<String>("ENTER",s));
234                                        }
235                                        else
236                                                break;
237                                }
238                        }
239                        if(!doneAnything)
240                                mob.tell(L("@x1 which direction?\n\rTry @x2.",CMStrings.capitalizeAndLower(doing),validDirs.toLowerCase()));
241                        else
242                        if(prequeCommands.size()>0)
243                                mob.prequeCommands(prequeCommands, metaFlags);
244                }
245                return false;
246        }

248        @Override
249        public double actionsCost(final MOB mob, final List<String> cmds)
250        {
251                double cost=CMath.div(CMProps.getIntVar(CMProps.Int.DEFCMDTIME),100.0);
252                if((mob!=null)&&(mob.isAttributeSet(MOB.Attrib.AUTORUN)))
253                        cost /= 4.0;
254                return CMProps.getCommandActionCost(ID(), cost);
255        }

257        @Override
258        public boolean canBeOrdered()
259        {
260                return true;
261        }
