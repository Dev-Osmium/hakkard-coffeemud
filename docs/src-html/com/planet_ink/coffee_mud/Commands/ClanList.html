036public class ClanList extends StdCommand
037{
038        public ClanList()
039        {
040        }
041
042        private final String[]  access  = I(new String[] { "CLANLIST", "CLANS" });
043
044        @Override
045        public String[] getAccessWords()
046        {
047                return access;
048        }
049
050        @Override
051        public boolean execute(MOB mob, List<String> commands, int metaFlags)
052                throws java.io.IOException
053        {
054                final boolean trophySystemActive=CMLib.clans().trophySystemActive();
055                final StringBuffer head=new StringBuffer("");
056                head.append("^x[");
057                head.append(CMStrings.padRight(L("Clan Name"),30)+"| ");
058                head.append(CMStrings.padRight(L("Type"),10)+"| ");
059                head.append(CMStrings.padRight(L("Status"),7)+"| ");
060                head.append(CMStrings.padRight(L("Members"),7));
061                if(trophySystemActive)
062                        head.append(" | "+CMStrings.padRight(L("Trophies"),8));
063                head.append("]^.^? \n\r");
064                final StringBuffer msg=new StringBuffer("");
065                for(final Enumeration<Clan> e=CMLib.clans().clans();e.hasMoreElements();)
066                {
067                        final Clan thisClan=e.nextElement();
068                        if(!thisClan.isPubliclyListedFor(mob))
069                                continue;
070
071                        final StringBuffer trophySet = new StringBuffer("");
072                        if(trophySystemActive)
073                        {
074                                for(final Trophy t : Trophy.values())
075                                {
076                                        if(CMath.bset(thisClan.getTrophies(),t.flagNum()))
077                                                trophySet.append(t.codeString.charAt(0));
078                                }
079                        }
080
081                        msg.append(" ");
082                        msg.append("^<CLAN^>"+CMStrings.padRight(CMStrings.removeColors(thisClan.clanID()),30)+"^</CLAN^>  ");
083                        msg.append(CMStrings.padRight(thisClan.getGovernmentName(),10)+"  ");
084                        boolean war=false;
085                        for(final Enumeration<Clan> e2=CMLib.clans().clans();e2.hasMoreElements();)
086                        {
087                                final Clan C=e2.nextElement();
088                                if((C!=thisClan)
089                                &&((thisClan.getClanRelations(C.clanID())==Clan.REL_WAR)
090                                        ||(C.getClanRelations(thisClan.clanID())==Clan.REL_WAR)))
091                                {
092                                        war=true;
093                                        break;
094                                }
095                        }
096                        String status=(war)?"At War":"Active";
097                        switch(thisClan.getStatus())
098                        {
099                        case Clan.CLANSTATUS_FADING:
100                                status="Inactive";
101                                break;
102                        case Clan.CLANSTATUS_PENDING:
103                                status="Pending";
104                                break;
105                        }
106                        msg.append(CMStrings.padRight(status,7)+"  ");
107                        msg.append(CMStrings.padRight(Integer.toString(thisClan.getSize()),7)+"   ");
108                        msg.append(trophySet);
109                        msg.append("\n\r");
110                }
111                mob.tell(head.toString()+msg.toString());
112                return false;
113        }
114
115        @Override
116        public boolean canBeOrdered()
117        {
118                return true;
119        }
120
121}
