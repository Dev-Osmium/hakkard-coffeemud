036public class Group extends StdCommand
037{
038        public Group()
039        {
040        }
041
042        private final String[] access=I(new String[]{"GROUP","GR"});
043        @Override
044        public String[] getAccessWords()
045        {
046                return access;
047        }
048
049        public static StringBuffer showWhoLong(MOB seer, MOB who)
050        {
051                final StringBuffer msg=new StringBuffer("^N");
052                msg.append("[^w");
053                final int[] cols={
054                        CMLib.lister().fixColWidth(7,seer.session()),
055                        CMLib.lister().fixColWidth(7,seer.session()),
056                        CMLib.lister().fixColWidth(5,seer.session()),
057                        CMLib.lister().fixColWidth(13,seer.session()),
058                        CMLib.lister().fixColWidth(3,seer.session()),
059                        CMLib.lister().fixColWidth(12,seer.session())
060                };
061                if(!CMSecurity.isDisabled(CMSecurity.DisFlag.RACES))
062                {
063                        if(who.charStats().getCurrentClass().raceless())
064                                msg.append(CMStrings.padRight(" ",cols[0])+" ");
065                        else
066                                msg.append(CMStrings.padRight(who.charStats().raceName(),cols[0])+" ");
067                }
068
069                String levelStr=who.charStats().displayClassLevel(who,true).trim();
070                final int x=levelStr.lastIndexOf(' ');
071                if(x>=0)
072                        levelStr=levelStr.substring(x).trim();
073                if(!CMSecurity.isDisabled(CMSecurity.DisFlag.CLASSES))
074                {
075                        if(who.charStats().getMyRace().classless())
076                                msg.append(CMStrings.padRight(" ",cols[1])+" ");
077                        else
078                                msg.append(CMStrings.padRight(who.charStats().displayClassName(),cols[1])+" ");
079                }
080                if(!CMSecurity.isDisabled(CMSecurity.DisFlag.LEVELS))
081                {
082                        if(who.charStats().getCurrentClass().leveless()
083                        ||who.charStats().getMyRace().leveless())
084                                msg.append(CMStrings.padRight(" ",cols[2]));
085                        else
086                                msg.append(CMStrings.padRight(levelStr,cols[2]));
087                }
088                final double hpPct = CMath.div(who.curState().getHitPoints(), who.maxState().getHitPoints());
089                final double mnPct = CMath.div(who.curState().getMana(), who.maxState().getMana());
090                final double mvPct = CMath.div(who.curState().getMovement(), who.maxState().getMovement());
091                final String hpColor = (hpPct < .25) ? "^r" : (hpPct < .5) ? "^y" : "^w";
092                final String mnColor = (mnPct < .25) ? "^r" : (mnPct < .5) ? "^y" : "^w";
093                final String mvColor = (mvPct < .25) ? "^r" : (mvPct < .5) ? "^y" : "^w";
094                msg.append("^N] ^H" + CMStrings.padRight(who.name(),cols[3])+"^N ");
095                msg.append(CMStrings.padRightPreserve(CMLib.lang().L("hp(@x1@x2^N/^w@x3^N)",hpColor,CMStrings.padRightPreserve(""+who.curState().getHitPoints(),cols[4]),CMStrings.padRightPreserve(""+who.maxState().getHitPoints(),cols[4])),cols[5]));
096                msg.append(CMStrings.padRightPreserve(CMLib.lang().L("mn(@x1@x2^N/^w@x3^N)",mnColor,CMStrings.padRightPreserve(""+who.curState().getMana(),cols[4]),CMStrings.padRightPreserve(""+who.maxState().getMana(),cols[4])),cols[5]));
097                msg.append(CMStrings.padRightPreserve(CMLib.lang().L("mv(@x1@x2^N/^w@x3^N)",mvColor,CMStrings.padRightPreserve(""+who.curState().getMovement(),cols[4]),CMStrings.padRightPreserve(""+who.maxState().getMovement(),cols[4])),cols[5]));
098                msg.append("\n\r");
099                return msg;
100        }
101
102        @Override
103        public boolean execute(MOB mob, List<String> commands, int metaFlags)
104                throws java.io.IOException
105        {
106                mob.tell(L("@x1's group:\n\r",mob.name()));
107                final Set<MOB> group=mob.getGroupMembers(new HashSet<MOB>());
108                final List<MOB> orderedGroup = new LinkedList<MOB>();
109                final Room R=mob.location();
110                for(final Enumeration<MOB> m=R.inhabitants();m.hasMoreElements();)
111                {
112                        final MOB M=m.nextElement();
113                        if(group.contains(M))
114                        {
115                                orderedGroup.add(M);
116                                group.remove(M);
117                        }
118                }
119                orderedGroup.addAll(group);
120                group.clear();
121                final StringBuffer msg=new StringBuffer("");
122                for (final MOB follower : orderedGroup)
123                {
124                        if((follower!=null)
125                        && (follower!=mob))
126                        {
127                                if(follower.amFollowing()==null)
128                                {
129                                        for (final MOB leader : orderedGroup)
130                                        {
131                                                if((leader!=null)
132                                                &&(leader != follower)
133                                                &&(leader.isFollowedBy(follower)))
134                                                {
135                                                        Log.errOut(follower.Name()+" should be in "+mob.Name()+"'s group, but is in no ones.  Fixing.");
136                                                        follower.setFollowing(leader);
137                                                }
138                                        }
139                                }
140                        }
141                        msg.append(showWhoLong(mob,follower));
142                }
143                mob.tell(msg.toString());
144                return false;
145        }
146
147        @Override
148        public boolean canBeOrdered()
149        {
150                return true;
151        }
