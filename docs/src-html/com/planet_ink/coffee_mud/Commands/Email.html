038public class Email extends StdCommand
039{
040        public Email()
041        {
042        }
043
044        private final String[] access=I(new String[]{"EMAIL"});
045        @Override
046        public String[] getAccessWords()
047        {
048                return access;
049        }
050
051        @Override
052        public boolean execute(MOB mob, List<String> commands, int metaFlags)
053                throws java.io.IOException
054        {
055                if(mob.session()==null)
056                        return true;
057                final PlayerStats pstats=mob.playerStats();
058                if(pstats==null)
059                        return true;
060
061                if((commands!=null)
062                &&(commands.size()>1))
063                {
064
065                        if(CMProps.getVar(CMProps.Str.MAILBOX).length()==0)
066                        {
067                                mob.tell(L("A mailbox has not been defined by this muds administrators, so mail can be neither sent, or received."));
068                                return false;
069                        }
070                        int max=10;
071                        if((commands.get(1).equalsIgnoreCase("BOX") && (commands.size()>2)) && (CMath.isInteger(commands.get(commands.size()-1))))
072                        {
073                                max=CMath.s_int(commands.get(commands.size()-1));
074                                commands.remove(commands.size()-1);
075                        }
076                        final String name=CMParms.combine(commands,1);
077                        if(name.equalsIgnoreCase("BOX"))
078                        {
079                                final String journalName=CMProps.getVar(CMProps.Str.MAILBOX);
080                                final String[] queries=new String[] { mob.Name(),"ALL","MASK=%" };
081                                final List<JournalEntry> msgs=CMLib.database().DBReadJournalMsgsByUpdateDate(journalName, false, max, queries);
082                                for(int num=0;num<msgs.size();num++)
083                                {
084                                        final JournalEntry thismsg=msgs.get(num);
085                                        final String to=thismsg.to();
086                                        if(to.equalsIgnoreCase("ALL")
087                                        ||to.equalsIgnoreCase(mob.Name())
088                                        ||(to.toUpperCase().trim().startsWith("MASK=")&&CMLib.masking().maskCheck(to.trim().substring(5),mob,true)))
089                                        {
090                                                // keep this one
091                                        }
092                                        else
093                                                msgs.remove(num);
094                                }
095                                
096                                final int[] cols={
097                                                CMLib.lister().fixColWidth(48,mob.session()),
098                                                CMLib.lister().fixColWidth(15,mob.session()),
099                                                CMLib.lister().fixColWidth(20,mob.session())
100                                        };
101                                while((mob.session()!=null)&&(!mob.session().isStopped()))
102                                {
103                                        StringBuffer messages=new StringBuffer("^X"+CMStrings.padCenter(mob.Name()+"'s MailBox",cols[0])+"^?^.");
104                                        if(msgs.size()==max)
105                                                messages.append(L(" (Newest @x1 messages)",""+max));
106                                        messages.append("\n\r");
107                                        messages.append("^X### "+CMStrings.padRight(L("From"),cols[1])+" "+CMStrings.padRight(L("Date"),cols[2])+" Subject^?^.\n\r");
108                                        for(int num=0;num<msgs.size();num++)
109                                        {
110                                                final JournalEntry thismsg=msgs.get(num);
111                                                if(thismsg != null)
112                                                {
113                                                        messages.append(CMStrings.padRight(""+(num+1),4)
114                                                                        +CMStrings.padRight((thismsg.from()),cols[1])+" "
115                                                                        +CMStrings.padRight(CMLib.time().date2String(thismsg.date()),cols[2])+" "
116                                                                        +(thismsg.subj())
117                                                                        +"\n\r");
118                                                }
119                                                else
120                                                        Log.errOut("Message "+num+" of "+msgs.size()+" not found.");
121                                        }
122                                        if((msgs.size()==0)
123                                        ||(CMath.bset(metaFlags,MUDCmdProcessor.METAFLAG_POSSESSED))
124                                        ||(CMath.bset(metaFlags,MUDCmdProcessor.METAFLAG_AS)))
125                                        {
126                                                if((!mob.isAttributeSet(MOB.Attrib.AUTOFORWARD))
127                                                &&(mob.playerStats()!=null)
128                                                &&(mob.playerStats().getEmail().length()>0)
129                                                &&((mob.playerStats().getAccount()==null)
130                                                        ||(!mob.playerStats().getAccount().isSet(AccountFlag.NOAUTOFORWARD))))
131                                                        mob.tell(L("You have no email waiting, but then, it's probably been forwarded to you already."));
132                                                else
133                                                        mob.tell(L("You have no email waiting."));
134                                                return false;
135                                        }
136                                        final Session S=mob.session();
137                                        try
138                                        {
139                                                if(S!=null)
140                                                        S.snoopSuspension(1);
141                                                mob.tell(messages.toString());
142                                        }
143                                        finally
144                                        {
145                                                if(S!=null)
146                                                        S.snoopSuspension(-1);
147                                        }
148                                        if(mob.session()==null)
149                                                continue;
150                                        String s=mob.session().prompt(L("Enter a message #"),"");
151                                        if((!CMath.isInteger(s))||(mob.session().isStopped()))
152                                                return false;
153                                        final int num=CMath.s_int(s);
154                                        if((num<=0)||(num>msgs.size()))
155                                                mob.tell(L("That is not a valid number."));
156                                        else
157                                        while((mob.session()!=null)&&(!mob.session().isStopped()))
158                                        {
159                                                final JournalEntry thismsg=msgs.get(num-1);
160                                                final String key=thismsg.key();
161                                                final String from=thismsg.from();
162                                                final String date=CMLib.time().date2String(thismsg.date());
163                                                final String subj=thismsg.subj();
164                                                final String message=thismsg.msg();
165                                                messages=new StringBuffer("");
166                                                messages.append("^XMessage :^?^."+num+"\n\r");
167                                                messages.append("^XFrom    :^?^."+from+"\n\r");
168                                                messages.append("^XDate    :^?^."+date+"\n\r");
169                                                messages.append("^XSubject :^?^."+subj+"\n\r");
170                                                messages.append("^X------------------------------------------------^?^.\n\r");
171                                                messages.append(message+"\n\r\n\r");
172                                                try
173                                                {
174                                                        if(S!=null)
175                                                                S.snoopSuspension(1);
176                                                        mob.tell(messages.toString());
177                                                }
178                                                finally
179                                                {
180                                                        if(S!=null)
181                                                                S.snoopSuspension(-1);
182                                                }
183                                                if(mob.session()==null)
184                                                        continue;
185                                                s=mob.session().choose(L("Would you like to D)elete, H)old, or R)eply (D/H/R)? "),"DHR","H");
186                                                if(s.equalsIgnoreCase("H"))
187                                                        break;
188                                                if(s.equalsIgnoreCase("R"))
189                                                {
190                                                        if((from.length()>0)
191                                                        &&(!from.equals(mob.Name()))
192                                                        &&(!from.equalsIgnoreCase("BOX"))
193                                                        &&(CMLib.players().getLoadPlayer(from)!=null))
194                                                                execute(mob,new XVector<String>(getAccessWords()[0],from),metaFlags);
195                                                        else
196                                                                mob.tell(L("You can not reply to this email."));
197                                                }
198                                                else
199                                                if(s.equalsIgnoreCase("D"))
200                                                {
201                                                        CMLib.database().DBDeleteJournal(journalName,key);
202                                                        msgs.remove(thismsg);
203                                                        mob.tell(L("Deleted."));
204                                                        break;
205                                                }
206                                        }
207                                }
208                        }
209                        else
210                        {
211                                final MOB M=CMLib.players().getLoadPlayer(name);
212                                if(M==null)
213                                {
214                                        mob.tell(L("There is no player called '@x1' to send email to.  If you were trying to read your mail, try EMAIL BOX.  If you were trying to change your email address, just enter EMAIL without any parameters.",name));
215                                        return false;
216                                }
217                                if((!M.isAttributeSet(MOB.Attrib.AUTOFORWARD))
218                                &&((M.playerStats()!=null)
219                                &&(M.playerStats().getEmail().length()>0))              
220                                &&((M.playerStats().getAccount()==null)
221                                        ||(!M.playerStats().getAccount().isSet(AccountFlag.NOAUTOFORWARD))))
222                                {
223                                        if(!mob.session().confirm(L("Send email to '@x1' (Y/n)?",M.Name()),"Y"))
224                                                return false;
225                                }
226                                else
227                                {
228                                        if(!mob.session().confirm(L("Send email to '@x1', even though their AUTOFORWARD is turned off (y/N)?",M.Name()),"N"))
229                                                return false;
230                                }
231                                if(CMProps.getIntVar(CMProps.Int.MAXMAILBOX)>0)
232                                {
233                                        final int count=CMLib.database().DBCountJournal(CMProps.getVar(CMProps.Str.MAILBOX),null,M.Name());
234                                        if(count>=CMProps.getIntVar(CMProps.Int.MAXMAILBOX))
235                                        {
236                                                mob.tell(L("@x1's mailbox is full.",M.Name()));
237                                                return false;
238                                        }
239                                }
240                                if(mob.session()==null)
241                                        return false;
242                                final String subject=mob.session().prompt(L("Email Subject: "),"").trim();
243                                if(subject.length()==0)
244                                {
245                                        mob.tell(L("Aborted"));
246                                        return false;
247                                }
248                                if(mob.session()==null)
249                                        return false;
250                                String message=mob.session().prompt(L("Enter your message\n\r: "),"").trim();
251                                if(message.trim().length()==0)
252                                {
253                                        mob.tell(L("Aborted"));
254                                        return false;
255                                }
256                                if(mob.session()==null)
257                                        return false;
258                                CMLib.smtp().emailOrJournal(mob.Name(), mob.Name(), M.Name(), subject, message);
259                                mob.tell(L("Your email has been sent."));
260                                return true;
261                        }
262                }
263                if((pstats.getEmail()==null)||(pstats.getEmail().length()==0))
264                {
265                        if(CMProps.getVar(CMProps.Str.EMAILREQ).toUpperCase().startsWith("DISABLED"))
266                        {
267                                if(commands!=null)
268                                        mob.session().println(L("\n\rAn email address is not required by this system."));
269                                return true;
270                        }
271                        mob.session().println(L("\n\rYou have no email address on file for this character."));
272                }
273                else
274                {
275                        if(commands==null)
276                                return true;
277                        final String change=mob.session().prompt(L("You currently have '@x1' set as the email address for this character.\n\rChange it (y/N)?",pstats.getEmail()),"N");
278                        if(change.toUpperCase().startsWith("N"))
279                                return false;
280                }
281                if((CMProps.getVar(CMProps.Str.EMAILREQ).toUpperCase().startsWith("PASS"))
282                &&(commands!=null)
283                &&(CMProps.getVar(CMProps.Str.MAILBOX).length()>0))
284                        mob.session().println(L("\n\r** Changing your email address will cause you to be logged off, and a new password to be generated and emailed to the new address. **\n\r"));
285                String newEmail=mob.session().prompt(L("New E-mail Address:"));
286                if(newEmail==null)
287                        return false;
288                newEmail=newEmail.trim();
289                if(!CMProps.getVar(CMProps.Str.EMAILREQ).toUpperCase().startsWith("OPTION"))
290                {
291                        if(newEmail.length()<6)
292                                return false;
293                        if(newEmail.indexOf('@')<0)
294                                return false;
295                        String confirmEmail=mob.session().prompt(L("Confirm that '@x1' is correct by re-entering.\n\rRe-enter:",newEmail));
296                        if(confirmEmail==null)
297                                return false;
298                        confirmEmail=confirmEmail.trim();
299                        if(confirmEmail.length()==0)
300                                return false;
301                        if(!(newEmail.equalsIgnoreCase(confirmEmail)))
302                                return false;
303                }
304                pstats.setEmail(newEmail);
305                CMLib.database().DBUpdateEmail(mob);
306                if((commands!=null)
307                &&(CMProps.getVar(CMProps.Str.EMAILREQ).toUpperCase().startsWith("PASS"))
308                &&(CMProps.getVar(CMProps.Str.MAILBOX).length()>0))
309                {
310                        final String password=CMLib.encoder().generateRandomPassword();
311                        pstats.setPassword(password);
312                        CMLib.database().DBUpdatePassword(mob.Name(),pstats.getPasswordStr());
313                        CMLib.database().DBWriteJournal(CMProps.getVar(CMProps.Str.MAILBOX),
314                                          mob.Name(),
315                                          mob.Name(),
316                                          "Password for "+mob.Name(),
317                                          "Your new password for "+mob.Name()+" is: "+password+"\n\rYou can login by pointing your mud client at "+CMProps.getVar(CMProps.Str.MUDDOMAIN)+" port(s):"+CMProps.getVar(CMProps.Str.MUDPORTS)+".\n\rYou may use the PASSWORD command to change it once you are online.");
318                        mob.tell(L("You will receive an email with your new password shortly.  Goodbye."));
319                        if(mob.session()!=null)
320                        {
321                                CMLib.s_sleep(1000);
322                                mob.session().stopSession(false,false,false);
323                        }
324                }
325                return true;
326        }
327
328        @Override
329        public boolean canBeOrdered()
330        {
331                return false;
332        }
