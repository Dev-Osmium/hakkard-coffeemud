060        public String liveAuctionStatus()
061        {
062                if(getLiveData().getAuctionedItem()!=null)
063                {
064                        String bidWords=CMLib.beanCounter().nameCurrencyShort(getLiveData().getCurrency(),getLiveData().getBid());
065                        if(bidWords.length()==0)
066                                bidWords="0";
067                        return "Up for live auction: "+getLiveData().getAuctionedItem().name()+".  The current bid is "+bidWords+".";
068                }
069                return "";
070        }

085        protected static final int      STATE_ONCE              = 2;
086        protected static final int      STATE_TWICE             = 3;
087        protected static final int      STATE_THREE             = 4;
088        protected static final int      STATE_CLOSED    = 5;

102        @Override
103        public boolean tick(final Tickable ticking, final int tickID)
104        {
105                if(tickID==Tickable.TICKID_LIVEAUCTION)
106                {
107                        final AuctionData AD = getLiveData();
108                        AD.setAuctionTickDown(AD.getAuctionTickDown()-1);
109                        if(AD.getAuctionTickDown()<=0)
110                        {
111                                final MOB auctioneerM=AD.getAuctioningMob();
112                                final MOB winnerM=AD.getHighBidderMob();
113                                if((AD.getAuctionState()==STATE_START)&&((System.currentTimeMillis()-AD.getStartTime())<(5*15000)))
114                                {
115                                        if(((System.currentTimeMillis()-AD.getStartTime())>(3*15000))
116                                        &&((winnerM==null)||(winnerM==auctioneerM)))
117                                                setLiveAuctionState(STATE_RUNOUT);
118                                        else
119                                                setLiveAuctionState(STATE_START);
120                                        return true;
121                                }
122                                setLiveAuctionState(AD.getAuctionState()+1);
123                                final Vector<String> V=new Vector<String>();
124                                V.add("AUCTION");
125                                V.add("CHANNEL");
126                                switch(AD.getAuctionState())
127                                {
128                                case STATE_RUNOUT:
129                                        V.add(L("The live auction for @x1 is almost done. The current bid is @x2.",
130                                                        AD.getAuctionedItem().name(),CMLib.beanCounter().nameCurrencyShort(AD.getCurrency(),AD.getBid())));
131                                        break;
132                                case STATE_ONCE:
133                                        V.add(L("@x1 for @x2 going ONCE!",CMLib.beanCounter().nameCurrencyShort(AD.getCurrency(),AD.getBid()),AD.getAuctionedItem().name()));
134                                        break;
135                                case STATE_TWICE:
136                                        V.add(L("@x1 for @x2 going TWICE!",CMLib.beanCounter().nameCurrencyShort(AD.getCurrency(),AD.getBid()),AD.getAuctionedItem().name()));
137                                        break;
138                                case STATE_THREE:
139                                        V.add(L("@x1 going for @x2! Last chance!",AD.getAuctionedItem().name(),CMLib.beanCounter().nameCurrencyShort(AD.getCurrency(),AD.getBid())));
140                                        break;
141                                case STATE_CLOSED:
142                                        {
143                                                if((winnerM!=null)&&(winnerM!=AD.getAuctioningMob()))
144                                                {
145                                                        V.add(L("@x1 SOLD to @x2 for @x3.",AD.getAuctionedItem().name(),winnerM.name(),
146                                                                        CMLib.beanCounter().nameCurrencyShort(AD.getCurrency(),AD.getBid())));
147                                                        auctioneerM.doCommand(V,MUDCmdProcessor.METAFLAG_FORCED);
148                                                        if(AD.getAuctionedItem() != null)
149                                                        {
150                                                                AD.getAuctionedItem().unWear();
151                                                                final AuctionPolicy aRates=(AuctionPolicy)CMClass.getCommon("DefaultAuctionPolicy");
152                                                                winnerM.location().moveItemTo(AD.getAuctionedItem(),ItemPossessor.Expire.Player_Drop);
153                                                                final double houseCut=Math.floor(AD.getBid()*aRates.liveFinalCutPct());
154                                                                final double finalAmount=AD.getBid()-houseCut;
155                                                                CMLib.coffeeShops().returnMoney(winnerM,AD.getCurrency(),AD.getHighBid()-AD.getBid());
156                                                                CMLib.coffeeShops().returnMoney(auctioneerM,AD.getCurrency(),finalAmount);
157                                                                auctioneerM.tell(L("@x1 has been transferred to you as payment from @x2, after the house took a cut of @x3.  The goods have also been transferred in exchange.",
158                                                                                CMLib.beanCounter().nameCurrencyShort(AD.getCurrency(),finalAmount),
159                                                                                winnerM.name(auctioneerM),
160                                                                                CMLib.beanCounter().nameCurrencyShort(AD.getCurrency(),houseCut)));
161                                                                if(CMLib.commands().postGet(winnerM,null,AD.getAuctionedItem(),false)
162                                                                ||(winnerM.isMine(AD.getAuctionedItem())))
163                                                                {
164                                                                        winnerM.tell(L("@x1 has been transferred to @x2.  You should have received the auctioned goods.  This auction is complete.",
165                                                                                        CMLib.beanCounter().nameCurrencyShort(AD.getCurrency(),AD.getBid()),
166                                                                                        auctioneerM.name(winnerM)));
167                                                                        if(AD.getAuctionedItem() instanceof LandTitle)
168                                                                        {
169                                                                                final CMMsg msg=CMClass.getMsg(auctioneerM,winnerM,AD.getAuctionedItem(),CMMsg.MASK_ALWAYS|CMMsg.TYP_GIVE,null);
170                                                                                AD.getAuctionedItem().executeMsg(winnerM,msg);
171                                                                        }
172                                                                }
173                                                                else
174                                                                {
175                                                                        auctioneerM.moveItemTo(AD.getAuctionedItem());
176                                                                        auctioneerM.tell(L("Your transaction could not be completed because @x1 was unable to collect the item.  Please contact @x2 about receipt of @x3 for @x4.",
177                                                                                        winnerM.name(auctioneerM),winnerM.name(auctioneerM),AD.getAuctionedItem().name(winnerM),
178                                                                                        CMLib.beanCounter().nameCurrencyShort(AD.getCurrency(),AD.getBid())));
179                                                                        winnerM.tell(L("Your transaction could not be completed because you were unable to collect the item.  Please contact @x1 about receipt of @x2 for @x3.",
180                                                                                        auctioneerM.name(winnerM),AD.getAuctionedItem().name(winnerM),
181                                                                                        CMLib.beanCounter().nameCurrencyShort(AD.getCurrency(),AD.getBid())));
182                                                                }
183                                                        }
184                                                }
185                                                else
186                                                {
187                                                        if(!auctioneerM.isMine(AD.getAuctionedItem()))
188                                                                auctioneerM.moveItemTo(AD.getAuctionedItem());
189                                                        V.clear();
190                                                        V.add("AUCTION");
191                                                        V.add("CHANNEL");
192                                                        V.add(L("The auction for @x1 has ended without a winner.",AD.getAuctionedItem().name()));
193                                                        auctioneerM.doCommand(V,MUDCmdProcessor.METAFLAG_FORCED);
194                                                }
195                                                AD.setAuctioningMob(null);
196                                                AD.setAuctionedItem(null);
197                                                AD.setHighBidderMob(null);
198                                                AD.setHighBid(0.0);
199                                                AD.setBid(0.0);
200                                                AD.setAuctionState(0);
201                                                CMLib.threads().deleteTick(this,Tickable.TICKID_LIVEAUCTION);
202                                        }
203                                        return false;
204                                }
205                                auctioneerM.doCommand(V,MUDCmdProcessor.METAFLAG_FORCED);
206                        }
207                }
208                return true;
209        }

211        public boolean doLiveAuction(MOB mob, List<String> commands, Environmental target)
212        {
213                final Vector<String> V=new Vector<String>();
214                V.add("AUCTION");
215                V.add("CHANNEL");
216                if(target!=null)
217                {
218                        if(!(target instanceof Item))
219                                return false;
220                        getLiveData().setAuctioningMob(mob);
221                        getLiveData().setAuctionedItem((Item)target);
222                        final String sb=CMParms.combine(commands,0);
223                        getLiveData().setCurrency(CMLib.english().numPossibleGoldCurrency(mob,sb));
224                        if(getLiveData().getCurrency().length()==0)
225                                getLiveData().setCurrency(CMLib.beanCounter().getCurrency(mob));
226                        final double denomination=CMLib.english().numPossibleGoldDenomination(null,getLiveData().getCurrency(),sb);
227                        final long num=CMLib.english().numPossibleGold(null,sb);
228                        getLiveData().setBid(CMath.mul(denomination,num));
229                        getLiveData().setHighBid(getLiveData().getBid()-1);
230                        getLiveData().setHighBidderMob(null);
231                        getLiveData().setStartTime(System.currentTimeMillis());
232                        setLiveAuctionState(STATE_START);
233                        CMLib.threads().startTickDown(this,Tickable.TICKID_LIVEAUCTION,1);
234                        final String bidWords=CMLib.beanCounter().nameCurrencyShort(getLiveData().getCurrency(),getLiveData().getBid());
235                        if(target instanceof Item)
236                                mob.delItem((Item)target);
237                        V.add(L("New live auction: @x1.  The opening bid is @x2.",getLiveData().getAuctionedItem().name(),bidWords));
238                        if(getLiveData().getAuctioningMob()!=null)
239                                getLiveData().getAuctioningMob().doCommand(V,MUDCmdProcessor.METAFLAG_FORCED);
240                }
241                else
242                {
243                        if(getLiveData().getAuctionState()>0)
244                                setLiveAuctionState(STATE_RUNOUT);
245                        String sb="";
246                        if(commands!=null)
247                                sb=CMParms.combine(commands,0);
248                        final MOB M=getLiveData().getHighBidderMob();
249                        final Triad<String,Double,Long> bidObjs=CMLib.english().parseMoneyStringSDL(mob,sb,getLiveData().getCurrency());
250                        final String currency=bidObjs.first;
251                        final double amt=CMath.mul(bidObjs.second.doubleValue(),bidObjs.third.doubleValue());
252                        final String[] resp=CMLib.coffeeShops().bid(mob,amt,currency,getLiveData(),getLiveData().getAuctionedItem(),V);
253                        if(resp!=null)
254                        {
255                                if(resp[0]!=null)
256                                        mob.tell(resp[0]);
257                                if((resp[1]!=null)&&(M!=null))
258                                        M.tell(resp[1]);
259                        }
260                        if((V.size()>2)
261                        &&(getLiveData().getAuctioningMob()!=null))
262                                getLiveData().getAuctioningMob().doCommand(V,MUDCmdProcessor.METAFLAG_FORCED);
263                }
264                return true;
265        }

267        public void auctionNotify(MOB M, String resp, String regardingItem) throws java.io.IOException
268        {
269                if(CMLib.flags().isInTheGame(M,true))
270                        M.tell(resp);
271                else
272                if(M.playerStats()!=null)
273                {
274                        CMLib.smtp().emailIfPossible(CMProps.getVar(CMProps.Str.SMTPSERVERNAME),
275                                                                                "auction@"+CMProps.getVar(CMProps.Str.MUDDOMAIN).toLowerCase(),
276                                                                                "noreply@"+CMProps.getVar(CMProps.Str.MUDDOMAIN).toLowerCase(),
277                                                                                M.playerStats().getEmail(),
278                                                                                L("Auction Update for item: @x1",regardingItem),
279                                                                                resp);
280                }
281        }

283        @Override
284        public boolean execute(MOB mob, List<String> commands, int metaFlags)
285                throws java.io.IOException
286        {
287                //mob.tell(L("Auctions are currently closed for maintenance.  When it re-opens, this command will continue to remain available for live auctions, and new auctioneer mobs will be placed in the major cities for doing multi-day auctions, so keep your eyes open for that coming soon!"));
288                //if((mob!=null)||(commands!=null)) return false;
289                final PlayerStats pstats=mob.playerStats();
290                if(pstats==null)
291                        return false;
292                final int channelInt=CMLib.channels().getChannelIndex("AUCTION");
293                final int channelNum=CMLib.channels().getChannelCodeNumber("AUCTION");
294
295                if(CMath.isSet(pstats.getChannelMask(),channelInt))
296                {
297                        pstats.setChannelMask(pstats.getChannelMask()&(pstats.getChannelMask()-channelNum));
298                        mob.tell(L("The AUCTION channel has been turned on.  Use `NOAUCTION` to turn it off again."));
299                }
300
301                String cmd=null;
302                commands.remove(0);
303                if(commands.size()<1)
304                        cmd="";
305                else
306                        cmd=commands.get(0).toUpperCase();
307
308                if(cmd.equals("LIST"))
309                {
310                        commands.remove(0);
311                        final StringBuffer buf=new StringBuffer("");
312                        if((getLiveData().getAuctionedItem()!=null)&&(getLiveData().getAuctioningMob()!=null))
313                        {
314                                buf.append(L("\n\r^HCurrent *live* auction: ^N\n\r"));
315                                buf.append(liveAuctionStatus()+"\n\r");
316                        }
317                        else
318                                buf.append(MESSAGE_NOAUCTION());
319                        mob.tell(buf.toString());
320                        return true;
321                }
322                else
323                if(cmd.equals("UP"))
324                {
325                        commands.remove(0);
326                        if((getLiveData().getAuctionedItem()!=null)&&(getLiveData().getAuctioningMob()!=null))
327                        {
328                                mob.tell(L("A live auction is already underway.  Do AUCTION LIST to see it."));
329                                return false;
330                        }
331                        final Vector<String> V=new Vector<String>();
332                        if((commands.size()>=2)
333                        &&((CMLib.english().numPossibleGold(mob,commands.get(commands.size()-1))>0)||(commands.get(commands.size()-1).equals("0"))))
334                        {
335                                V.add(commands.get(commands.size()-1));
336                                commands.remove(commands.size()-1);
337                        }
338                        else
339                                V.add("0");
340
341                        final String s=CMParms.combine(commands,0);
342                        final Environmental E=mob.findItem(null,s);
343                        if((E==null)||(E instanceof MOB))
344                        {
345                                mob.tell(L("@x1 is not an item you can auction.",s));
346                                return false;
347                        }
348                        if((E instanceof Container)&&(((Container)E).hasContent()))
349                        {
350                                mob.tell(L("@x1 will have to be emptied first.",E.name()));
351                                return false;
352                        }
353                        if(!(((Item)E).amWearingAt(Wearable.IN_INVENTORY)))
354                        {
355                                mob.tell(L("@x1 will have to be removed first.",E.name()));
356                                return false;
357                        }
358                        final AuctionPolicy aRates=(AuctionPolicy)CMClass.getCommon("DefaultAuctionPolicy");
359                        final double deposit=aRates.liveListingPrice();
360                        final String depositAmt=CMLib.beanCounter().nameCurrencyLong(mob, deposit);
361
362                        if(deposit>0.0)
363                        {
364                                if((mob.isMonster())
365                                ||(!mob.session().confirm(L("Auctioning @x1 will cost a listing fee of @x2, proceed (Y/n)?",E.name(),depositAmt),"Y")))
366                                        return false;
367                        }
368                        else
369                        if((mob.isMonster())
370                        ||(!mob.session().confirm(L("Auction @x1 live, with a starting bid of @x2 (Y/n)?",E.name(),(V.get(0))),"Y")))
371                                return false;
372                        if(CMLib.beanCounter().getTotalAbsoluteValue(mob,CMLib.beanCounter().getCurrency(mob))<deposit)
373                        {
374                                mob.tell(L("You don't have enough @x1 to cover the listing fee!",CMLib.beanCounter().getDenominationName(CMLib.beanCounter().getCurrency(mob))));
375                                return false;
376                        }
377                        CMLib.beanCounter().subtractMoney(mob, CMLib.beanCounter().getCurrency(mob), deposit);
378                        doLiveAuction(mob,V,E);
379                        if(getLiveData().getAuctionedItem()!=null)
380                                getLiveData().setAuctioningMob(mob);
381                        return true;
382                }
383                else
384                if(cmd.equals("BID"))
385                {
386                        commands.remove(0);
387                        if((getLiveData().getAuctionedItem()==null)||(getLiveData().getAuctioningMob()==null))
388                        {
389                                mob.tell(MESSAGE_NOAUCTION());
390                                return false;
391                        }
392                        if(commands.size()<1)
393                        {
394                                mob.tell(L("Bid how much?"));
395                                return false;
396                        }
397                        final String amount=CMParms.combine(commands,0);
398                        doLiveAuction(mob,new XVector<String>(amount),null);
399                        return true;
400                }
401                else
402                if(cmd.equals("CLOSE"))
403                {
404                        commands.remove(0);
405                        if((getLiveData().getAuctionedItem()==null)||(getLiveData().getAuctioningMob()==null))
406                        {
407                                mob.tell(MESSAGE_NOAUCTION());
408                                return false;
409                        }
410                        if((getLiveData().getAuctionedItem()==null)||(getLiveData().getAuctioningMob()!=mob))
411                        {
412                                mob.tell(L("You are not currently running a live auction."));
413                                return false;
414                        }
415                        final Vector<String> V=new Vector<String>();
416                        V.add("AUCTION");
417                        V.add(L("The auction has been closed."));
418                        CMLib.threads().deleteTick(this,Tickable.TICKID_LIVEAUCTION);
419                        getLiveData().getAuctioningMob().moveItemTo(getLiveData().getAuctionedItem());
420                        if((getLiveData().getHighBid()>0.0)&&(getLiveData().getHighBidderMob()!=null))
421                                CMLib.coffeeShops().returnMoney(getLiveData().getHighBidderMob(),getLiveData().getCurrency(),getLiveData().getHighBid());
422                        getLiveData().setAuctioningMob(null);
423                        getLiveData().setAuctionedItem(null);
424                        super.execute(mob,V,metaFlags);
425                        return true;
426                }
427                else
428                if(cmd.equals("INFO"))
429                {
430                        commands.remove(0);
431                        if((getLiveData().getAuctionedItem()==null)||(getLiveData().getAuctioningMob()==null))
432                        {
433                                mob.tell(MESSAGE_NOAUCTION());
434                                return false;
435                        }
436                        Environmental E=null;
437                        E=getLiveData().getAuctionedItem();
438                        mob.tell(L("Item: @x1",E.name()));
439                        CMLib.commands().handleBeingLookedAt(CMClass.getMsg(mob,CMMsg.MASK_ALWAYS|CMMsg.MSG_EXAMINE,null));
440                        mob.tell(CMLib.coffeeShops().getViewDescription(mob,E));
441                        return true;
442                }
443                else
444                if(cmd.equals("CHANNEL"))
445                {
446                        commands.remove(0);
447                        if(commands.size()==0)
448                        {
449                                mob.tell(L("Channel what?"));
450                                return false;
451                        }
452                        if((getLiveData().getAuctionedItem()==null)||(getLiveData().getAuctioningMob()==null))
453                        {
454                                mob.tell(L("Channeling is only allowed during live auctions."));
455                                return false;
456                        }
457                        commands.add(0,"AUCTION");
458                        super.execute(mob,commands,metaFlags);
459                        return true;
460                }
461                commands.add(0,"AUCTION");
462                super.execute(mob,commands,metaFlags);
463                return false;
464        }
