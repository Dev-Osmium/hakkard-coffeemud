037public class ClanReject extends StdCommand
038{
039        public ClanReject()
040        {
041        }
042
043        private final String[]  access  = I(new String[] { "CLANREJECT" });
044
045        @Override
046        public String[] getAccessWords()
047        {
048                return access;
049        }
050
051        @Override
052        public boolean execute(MOB mob, List<String> commands, int metaFlags)
053                throws java.io.IOException
054        {
055                String memberStr=(commands.size()>1)?(String)commands.get(commands.size()-1):"";
056                final String clanName=(commands.size()>2)?CMParms.combine(commands,1,commands.size()-1):"";
057
058                Clan C=null;
059                final boolean skipChecks=mob.getClanRole(mob.Name())!=null;
060                if(skipChecks)
061                        C=mob.getClanRole(mob.Name()).first;
062
063                if(C==null)
064                {
065                        for(final Pair<Clan,Integer> c : mob.clans())
066                        {
067                                if((clanName.length()==0)||(CMLib.english().containsString(c.first.getName(), clanName))
068                                &&(c.first.getAuthority(c.second.intValue(), Clan.Function.REJECT)!=Authority.CAN_NOT_DO))
069                                {
070                                        C = c.first;
071                                        break;
072                                }
073                        }
074                }
075
076                commands.clear();
077                commands.add(getAccessWords()[0]);
078                commands.add(memberStr);
079
080                final StringBuffer msg=new StringBuffer("");
081                boolean found=false;
082                if(memberStr.length()>0)
083                {
084                        if(C==null)
085                        {
086                                mob.tell(L("You aren't allowed to reject anyone from @x1.",((clanName.length()==0)?"anything":clanName)));
087                                return false;
088                        }
089                        if(C.getGovernment().getAutoRole() == C.getGovernment().getAcceptPos())
090                        {
091                                mob.tell(L("Everyone is already accepted."));
092                                return false;
093                        }
094                        if(skipChecks||CMLib.clans().goForward(mob,C,commands,Clan.Function.REJECT,false))
095                        {
096                                final List<MemberRecord> apps=C.getMemberList(C.getGovernment().getAutoRole());
097                                if(apps.size()<1)
098                                {
099                                        mob.tell(L("There are no applicants to your @x1.",C.getGovernmentName()));
100                                        return false;
101                                }
102                                memberStr=CMStrings.capitalizeAndLower(memberStr);
103                                for(final MemberRecord member : apps)
104                                {
105                                        if(member.name.equalsIgnoreCase(memberStr))
106                                        {
107                                                found=true;
108                                        }
109                                }
110                                if(found)
111                                {
112                                        final MOB M=CMLib.players().getLoadPlayer(memberStr);
113                                        if(M==null)
114                                        {
115                                                mob.tell(L("@x1 was not found.  Could not reject from @x2.",memberStr,C.getGovernmentName()));
116                                                return false;
117                                        }
118                                        if(skipChecks||CMLib.clans().goForward(mob,C,commands,Clan.Function.REJECT,true))
119                                        {
120                                                C.delMember(M);
121                                                mob.tell(L("@x1 has been denied acceptance to @x2 '@x3'.",M.Name(),C.getGovernmentName(),C.clanID()));
122                                                if((M.session()!=null)&&(M.session().mob()==M))
123                                                        M.tell(L("You have been rejected as a member of @x1 '@x2'.",C.getGovernmentName(),C.clanID()));
124                                                return false;
125                                        }
126                                }
127                                else
128                                {
129                                        msg.append(L("@x1 isn't a member of your @x2.",memberStr,C.getGovernmentName()));
130                                }
131                        }
132                        else
133                        {
134                          msg.append(L("You aren't in the right position to reject applicants to your @x1.",C.getGovernmentName()));
135                        }
136                }
137                else
138                {
139                        msg.append(L("You haven't specified which applicant you are rejecting."));
140                }
141                mob.tell(msg.toString());
142                return false;
143        }
144
145        @Override
146        public boolean canBeOrdered()
147        {
148                return false;
149        }
150
151}
