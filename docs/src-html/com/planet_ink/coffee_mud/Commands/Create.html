052        public void exits(MOB mob, List<String> commands)
053                throws IOException
054        {
055                if(mob.location().roomID().equals(""))
056                {
057                        mob.tell(L("This command is invalid from within a GridLocaleChild room."));
058                        mob.location().showOthers(mob,null,CMMsg.MSG_OK_ACTION,L("<S-NAME> flub(s) a powerful spell."));
059                        return;
060                }
061                if(commands.size()<4)
062                {
063                        mob.tell(L("You have failed to specify the proper fields.\n\rThe format is CREATE EXIT [DIRECTION] [EXIT TYPE]\n\r"));
064                        mob.location().showOthers(mob,null,CMMsg.MSG_OK_ACTION,L("<S-NAME> flub(s) a spell.."));
065                        return;
066                }
067
068                final int direction=CMLib.directions().getGoodDirectionCode((commands.get(2)));
069                if(direction<0)
070                {
071                        mob.tell(L("You have failed to specify a direction.  Try @x1.\n\r",Directions.LETTERS()));
072                        mob.location().showOthers(mob,null,CMMsg.MSG_OK_ACTION,L("<S-NAME> flub(s) a spell.."));
073                        return;
074                }
075
076                final String Locale=commands.get(3);
077                Exit thisExit=CMClass.getExit(Locale);
078                if(thisExit==null)
079                {
080                        mob.tell(L("You have failed to specify a valid exit type '@x1'.\n\r",Locale));
081                        mob.location().showOthers(mob,null,CMMsg.MSG_OK_ACTION,L("<S-NAME> flub(s) a spell.."));
082                        return;
083                }
084
085                final Exit opExit=mob.location().getRawExit(direction);
086                final Room opRoom=mob.location().rawDoors()[direction];
087                Exit reverseExit=null;
088                if(opExit != null)
089                {
090                        if(opRoom!=null)
091                                reverseExit=opRoom.getRawExit(Directions.getOpDirectionCode(direction));
092                        if(reverseExit!=null)
093                        {
094                                if((thisExit.isGeneric())&&(reverseExit.isGeneric())&&(thisExit.ID().equals(reverseExit.ID())))
095                                {
096                                        thisExit=(Exit)reverseExit.copyOf();
097                                        CMLib.genEd().modifyGenExit(mob,thisExit,-1);
098                                }
099                        }
100                }
101
102                final boolean useShipDirs=(mob.location() instanceof BoardableShip)||(mob.location().getArea() instanceof BoardableShip);
103                mob.location().setRawExit(direction,thisExit);
104                if(mob.location() instanceof GridLocale)
105                        ((GridLocale)mob.location()).buildGrid();
106                mob.location().showHappens(CMMsg.MSG_OK_ACTION,L("Suddenly a portal opens up @x1.\n\r",
107                                (useShipDirs?CMLib.directions().getShipInDirectionName(direction):CMLib.directions().getInDirectionName(direction))));
108                CMLib.database().DBUpdateExits(mob.location());
109                if((reverseExit!=null)&&(opExit!=null)&&(opRoom!=null))
110                {
111                        final int revDirCode=Directions.getOpDirectionCode(direction);
112                        if(opRoom.getRawExit(revDirCode)==reverseExit)
113                        {
114                                opRoom.setRawExit(revDirCode,(Exit)thisExit.copyOf());
115                                CMLib.database().DBUpdateExits(opRoom);
116                        }
117                }
118                else
119                if((reverseExit==null)&&(opExit==null)&&(opRoom!=null))
120                {
121                        final int revDirCode=Directions.getOpDirectionCode(direction);
122                        if((opRoom.getRawExit(revDirCode)==null)&&(opRoom.rawDoors()[revDirCode]==mob.location()))
123                        {
124                                opRoom.setRawExit(revDirCode,(Exit)thisExit.copyOf());
125                                CMLib.database().DBUpdateExits(opRoom);
126                        }
127                }
128                mob.location().getArea().fillInAreaRoom(mob.location());
129                if(opRoom!=null)
130                        opRoom.getArea().fillInAreaRoom(opRoom);
131                Log.sysOut("Exits",mob.location().roomID()+" exits changed by "+mob.Name()+".");
132        }

134        private long[] makeSpaceLocation(MOB mob, SpaceObject newItem, String rest)
135        {
136                List<String> utokens=CMParms.parseSpaces(rest.toUpperCase(),true);
137                final String distErrorMsg=L("Valid distance units include: @x1.",SpaceObject.Distance.getFullList());
138                int x;
139                if(((x=utokens.indexOf("FROM"))>0)&&(x<utokens.size()-1))
140                {
141                        double[] direction=new double[]{Math.toRadians(CMLib.dice().roll(1, 360, -1)),Math.toRadians(CMLib.dice().roll(1,180,-1))};
142                        String distStr=CMParms.combine(utokens,0,x);
143                        String objName=CMParms.combine(utokens,x+1);
144                        Long dist=CMLib.english().parseSpaceDistance(distStr);
145                        if(dist==null)
146                        {
147                                mob.tell(L("Unknown distance for space object @x1:",newItem.ID())+" '"+distStr+"'. \n\r"+distErrorMsg);
148                                mob.location().showOthers(mob,null,CMMsg.MSG_OK_ACTION,L("<S-NAME> flub(s) a spell.."));
149                                return null;
150                        }
151                        SpaceObject O=null;
152                        if(objName.trim().length()>0)
153                        {
154                                O=CMLib.map().findSpaceObject(objName, true);
155                                if(O==null)
156                                        O=CMLib.map().findSpaceObject(objName, false);
157                        }
158                        if(O==null)
159                        {
160                                mob.tell(L("Unknown relative space object")+" '"+objName+"'.\n\r");
161                                mob.location().showOthers(mob,null,CMMsg.MSG_OK_ACTION,L("<S-NAME> flub(s) a spell.."));
162                                return null;
163                        }
164                        rest=CMParms.toListString(CMLib.map().getLocation(O.coordinates(), direction, dist.longValue()));
165                }
166                List<String> valsL=CMParms.parseCommas(rest,true);
167                if(valsL.size()!=3)
168                {
169                        mob.tell(L("Unknown location for space object @x1:",newItem.ID())+": '"+rest+"'. "
170                                        +L("The format for coordinates is 3 distances from core, comma delimited, or [DISTANCE] FROM [PLACE]\n\r"+distErrorMsg));
171                        mob.location().showOthers(mob,null,CMMsg.MSG_OK_ACTION,L("<S-NAME> flub(s) a spell.."));
172                        return null;
173                }
174                else
175                {
176                        boolean fail=true;
177                        Long[] valL=new Long[3];
178                        for(int i=0;i<3;i++)
179                        {
180                                Long newValue=CMLib.english().parseSpaceDistance(valsL.get(i));
181                                if(newValue==null)
182                                {
183                                        mob.tell(L("Unknown coord: '@x2'. @x3 for space object @x1:",newItem.ID(),valsL.get(i),distErrorMsg));
184                                        break;
185                                }
186                                else
187                                {
188                                        valL[i]=newValue;
189                                        if(i==2)
190                                                fail=false;
191                                }
192                        }
193                        if(!fail)
194                        {
195                                return new long[]{valL[0].longValue(),valL[1].longValue(),valL[2].longValue()};
196                        }
197                        else
198                        {
199                                mob.location().showOthers(mob,null,CMMsg.MSG_OK_ACTION,L("<S-NAME> flub(s) a spell.."));
200                                return null;
201                        }
202                }
203        }

205        public void items(MOB mob, List<String> commands) throws IOException
206        {
207                if(commands.size()<3)
208                {
209                        mob.tell(L("You have failed to specify the proper fields.\n\r"
210                                        + "The format for normal items is CREATE ITEM [ITEM NAME](@ room/[MOB NAME])\n\r"
211                                        + "The format for space items is CREATE ITEM [ITEM NAME](@ [COORDS]/[DISTANCE] FROM [SPACE OBJ])\n\r"));
212                        mob.location().showOthers(mob,null,CMMsg.MSG_OK_ACTION,L("<S-NAME> flub(s) a spell.."));
213                        return;
214                }
215
216                String itemID=CMParms.combine(commands,2);
217                Environmental dest=mob.location();
218                Container setContainer=null;
219                String rest="";
220                long [] coordinates=new long[]{
221                        CMLib.dice().getRandomizer().nextLong(),
222                        CMLib.dice().getRandomizer().nextLong(),
223                        CMLib.dice().getRandomizer().nextLong()
224                };
225                
226                int x=itemID.indexOf('@');
227                if(x>0)
228                {
229                        rest=itemID.substring(x+1).trim();
230                        itemID=itemID.substring(0,x).trim();
231                }
232                
233                Item newItem=CMClass.getItem(itemID);
234                if((newItem==null)&&(CMLib.english().numPossibleGold(null,itemID)>0))
235                {
236                        final long numCoins=CMLib.english().numPossibleGold(null,itemID);
237                        final String currency=CMLib.english().numPossibleGoldCurrency(mob,itemID);
238                        final double denom=CMLib.english().numPossibleGoldDenomination(mob,currency,itemID);
239                        if((numCoins>0)&&(denom>0.0))
240                                newItem=CMLib.beanCounter().makeCurrency(currency,denom,numCoins);
241                }
242
243                boolean doGenerica=true;
244                if(newItem==null)
245                {
246                        newItem=getNewCatalogItem(itemID);
247                        doGenerica=newItem==null;
248                }
249
250                if(newItem==null)
251                {
252                        mob.tell(L("There's no such thing as a '@x1'.\n\r",itemID));
253                        mob.location().showOthers(mob,null,CMMsg.MSG_OK_ACTION,L("<S-NAME> flub(s) a spell.."));
254                        return;
255                }
256                
257                if((newItem instanceof ArchonOnly)
258                &&(!CMSecurity.isASysOp(mob)))
259                {
260                        mob.tell(L("NO!"));
261                        return;
262                }
263                
264                if(x>0)
265                {
266                        if(newItem instanceof SpaceObject)
267                        {
268                                coordinates=makeSpaceLocation(mob,(SpaceObject)newItem,rest);
269                                if(coordinates==null)
270                                {
271                                        return;
272                                }
273                        }
274                        else
275                        if((!rest.equalsIgnoreCase("room"))
276                        &&(rest.length()>0))
277                        {
278                                final MOB M=mob.location().fetchInhabitant(rest);
279                                if(M==null)
280                                {
281                                        final Item I = mob.location().findItem(null, rest);
282                                        if(I instanceof Container)
283                                                setContainer=(Container)I;
284                                        else
285                                        {
286                                                mob.tell(L("MOB or Container '@x1' not found.",rest));
287                                                mob.location().showOthers(mob,null,CMMsg.MSG_OK_ACTION,L("<S-NAME> flub(s) a spell.."));
288                                                return;
289                                        }
290                                }
291                                else
292                                        dest=M;
293                        }
294                }
295                else
296                if(newItem instanceof SpaceObject)
297                {
298                        int i=21;
299                        while((--i>0)&&(CMLib.map().getSpaceObjectsByCenterpointWithin(coordinates, 0, SpaceObject.Distance.SolarSystemDiameter.dm).size()>0))
300                        {
301                                coordinates=new long[]{
302                                        CMLib.dice().getRandomizer().nextLong(),
303                                        CMLib.dice().getRandomizer().nextLong(),
304                                        CMLib.dice().getRandomizer().nextLong()
305                                };
306                        }
307                }
308        
309                if(newItem.subjectToWearAndTear())
310                        newItem.setUsesRemaining(100);
311                if(newItem instanceof SpaceObject)
312                {
313                        CMLib.map().addObjectToSpace(((SpaceObject)newItem), coordinates);
314                        mob.location().showHappens(CMMsg.MSG_OK_ACTION,L("Suddenly, @x1 appears in the sky.",newItem.name()));
315                }
316                else
317                if(dest instanceof Room)
318                {
319                        ((Room)dest).addItem(newItem);
320                        newItem.setContainer(setContainer);
321                        mob.location().showHappens(CMMsg.MSG_OK_ACTION,L("Suddenly, @x1 drops from the sky.",newItem.name()));
322                }
323                else
324                if(dest instanceof MOB)
325                {
326                        ((MOB)dest).addItem(newItem);
327                        newItem.setContainer(setContainer);
328                        mob.location().showHappens(CMMsg.MSG_OK_ACTION,L("Suddenly, @x1 drops into @x2's arms.",newItem.name(),dest.name()));
329                }
330
331                if((newItem.isGeneric())&&(doGenerica))
332                        CMLib.genEd().genMiscSet(mob,newItem,-1);
333                if(newItem instanceof SpaceObject)
334                {
335                        CMLib.database().DBCreateThisItem("SPACE", newItem);
336                }
337                mob.location().recoverRoomStats();
338                Log.sysOut("Items",mob.Name()+" created item "+newItem.ID()+".");
339        }

547                }
548
549                final String mobID=CMParms.combine(commands,2);

551
552                boolean doGenerica=true;
553                if(newMOB==null)
554                {
555                        newMOB=getNewCatalogMob(mobID);
556                        doGenerica=newMOB==null;
557                }
558

561                CharClass classC = null;
562                if(newMOB == null)
563                {
564                        raceR=CMClass.getRace(mobID);
565                        if(raceR==null)
566                        {
567                                classC=CMClass.getCharClass(mobID);
