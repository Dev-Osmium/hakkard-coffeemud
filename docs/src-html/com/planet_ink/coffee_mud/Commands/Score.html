054        protected String getExtremeValue(int extreme)
055        {
056                final StringBuilder str=new StringBuilder("");
057                for(char c : CMath.convertToRoman(extreme).toCharArray())
058                {
059                        switch(c)
060                        {
061                        case 'I': 
062                                str.append(CMProps.getListFileChoiceFromIndexedList(CMProps.ListFile.EXTREME_ADVS,0)).append(" ");
063                                break;
064                        case 'V': 
065                                str.append(CMProps.getListFileChoiceFromIndexedList(CMProps.ListFile.EXTREME_ADVS,1)).append(" ");
066                                break;
067                        case 'X': 
068                                str.append(CMProps.getListFileChoiceFromIndexedList(CMProps.ListFile.EXTREME_ADVS,2)).append(" ");
069                                break;
070                        case 'L': 
071                                str.append(CMProps.getListFileChoiceFromIndexedList(CMProps.ListFile.EXTREME_ADVS,3)).append(" ");
072                                break;
073                        case 'C': 
074                                str.append(CMProps.getListFileChoiceFromIndexedList(CMProps.ListFile.EXTREME_ADVS,4)).append(" ");
075                                break;
076                        case 'D': 
077                                str.append(CMProps.getListFileChoiceFromIndexedList(CMProps.ListFile.EXTREME_ADVS,5)).append(" ");
078                                break;
079                        case 'M': 
080                                str.append(CMProps.getListFileChoiceFromIndexedList(CMProps.ListFile.EXTREME_ADVS,6)).append(" ");
081                                break;
082                        }
083                }
084                return str.toString();
085        }

087        public StringBuilder getScore(MOB mob, String parm)
088        {
089                final int prowessCode = CMProps.getIntVar(CMProps.Int.COMBATPROWESS);
090
091                final StringBuilder msg=new StringBuilder("^N");
092
093                final int classLevel=mob.charStats().getClassLevel(mob.charStats().getCurrentClass());
094                if((!CMSecurity.isDisabled(CMSecurity.DisFlag.CLASSES))
095                &&(!mob.charStats().getMyRace().classless())
096                &&(!CMSecurity.isDisabled(CMSecurity.DisFlag.LEVELS))
097                &&(!mob.charStats().getMyRace().leveless())
098                &&(!mob.charStats().getCurrentClass().leveless()))
099                {
100                        String levelStr=null;
101                        if(classLevel>=mob.phyStats().level())
102                                levelStr=L("level ")+mob.phyStats().level()+" "+mob.charStats().getCurrentClass().name(mob.charStats().getCurrentClassLevel());
103                        else
104                                levelStr=mob.charStats().getCurrentClass().name(mob.charStats().getCurrentClassLevel())+" "+classLevel+"/"+mob.phyStats().level();
105                        msg.append(L("You are ^H@x1^? the ^H@x2^?.\n\r",mob.Name(),levelStr));
106                }
107                else
108                if((!CMSecurity.isDisabled(CMSecurity.DisFlag.LEVELS))
109                &&(!mob.charStats().getCurrentClass().leveless())
110                &&(!mob.charStats().getMyRace().leveless()))
111                {
112                        String levelStr=null;
113                        if(classLevel>=mob.phyStats().level())
114                                levelStr=L(", level ")+mob.phyStats().level();
115                        else
116                                levelStr=L(", level ")+classLevel+"/"+mob.phyStats().level();
117                        msg.append(L("You are ^H@x1^?^H@x2^?.\n\r",mob.Name(),levelStr));
118                }
119                else
120                if((!CMSecurity.isDisabled(CMSecurity.DisFlag.CLASSES))
121                &&(!mob.charStats().getMyRace().classless()))
122                        msg.append(L("You are ^H@x1^? the ^H@x2^?.\n\r",mob.Name(),mob.charStats().getCurrentClass().name(mob.charStats().getCurrentClassLevel())));
123                else
124                        msg.append(L("You are ^H@x1^?.\n\r",mob.Name()));
125
126                if((!CMSecurity.isDisabled(CMSecurity.DisFlag.CLASSES))
127                &&(classLevel<mob.phyStats().level()))
128                {
129                        msg.append(L("You also have levels in: "));
130                        final StringBuilder classList=new StringBuilder("");
131                        for(int c=0;c<mob.charStats().numClasses()-1;c++)
132                        {
133                                final CharClass C=mob.charStats().getMyClass(c);
134                                if(C!=mob.charStats().getCurrentClass())
135                                {
136                                        if(classList.length()>0)
137                                                if(c==mob.charStats().numClasses()-2)
138                                                        classList.append(L(", and "));
139                                                else
140                                                        classList.append(", ");
141                                        classList.append(C.name(mob.charStats().getClassLevel(C))+" ("+mob.charStats().getClassLevel(C)+") ");
142                                }
143                        }
144                        msg.append(classList.toString()+".\n\r");
145                }
146
147                if(CMProps.getBoolVar(CMProps.Bool.ACCOUNTEXPIRATION)&&(mob.playerStats()!=null))
148                        msg.append(L("Your account is Registered and Active until: @x1!\n\r",CMLib.time().date2String(mob.playerStats().getAccountExpiration())));
149
150                String genderName=L("^!neuter");
151                if(mob.charStats().getStat(CharStats.STAT_GENDER)=='M') 
152                        genderName=L("^!male");
153                else
154                if(mob.charStats().getStat(CharStats.STAT_GENDER)=='F') 
155                        genderName=L("^!female");
156                msg.append(L("You are a "));
157                if((mob.baseCharStats().getStat(CharStats.STAT_AGE)>0)&&(!CMSecurity.isDisabled(CMSecurity.DisFlag.ALL_AGEING)))
158                        msg.append(L("^!@x1^? year old ",""+mob.baseCharStats().getStat(CharStats.STAT_AGE)));
159                msg.append(genderName);
160                if((!CMSecurity.isDisabled(CMSecurity.DisFlag.RACES))
161                &&(!mob.charStats().getCurrentClass().raceless()))
162                        msg.append(" "+mob.charStats().getMyRace().name() + "^?");
163                else
164                        msg.append("^?");
165                if(mob.getLiegeID().length()>0)
166                {
167                        if(mob.isMarriedToLiege())
168                                msg.append(L(" who is married to ^H@x1^?",mob.getLiegeID()));
169                        else
170                                msg.append(L(" who serves ^H@x1^?",mob.getLiegeID()));
171                }
172                if(mob.getWorshipCharID().length()>0)
173                        msg.append(L(" worshipping ^H@x1^?",mob.getWorshipCharID()));
174                msg.append(".\n\r");
175                if(mob.clans().iterator().hasNext())
176                {
177                        msg.append(L("You are "));
178                        for(final Iterator<Pair<Clan,Integer>> c = mob.clans().iterator();c.hasNext();)
179                        {
180                                final Pair<Clan,Integer> p=c.next();
181                                final Clan C=p.first;
182                                String role=C.getRoleName(p.second.intValue(),true,false);
183                                role=CMLib.english().startWithAorAn(role);
184                                msg.append(L("@x1 of ^H@x2^?^.",role,C.getName()));
185                                if(c.hasNext())
186                                        msg.append(", ");
187                        }
188                        msg.append("\n\r");
189                }
190                if(!CMSecurity.isDisabled(CMSecurity.DisFlag.ATTRIBS))
191                {
192                        msg.append(L("\n\r^NYour stats are: "));
193                        msg.append(CMLib.protocol().mxpImage(mob," ALIGN=RIGHT H=70 W=70"));
194                        msg.append("\n\r");
195                        CharStats CT=mob.charStats();
196                        if(parm.equalsIgnoreCase("BASE")) 
197                                CT=mob.baseCharStats();
198                        msg.append("^N^!");
199                        final int longest=CharStats.CODES.LONNGEST_BASECODE_NAME();
200                        final boolean useWords=CMProps.Int.Prowesses.STAT_PROFICIENCY.is(prowessCode);
201                        final String[] charStatList = CMProps.getListFileStringList(CMProps.ListFile.CHARSTAT_CHART);
202                        for(final int i : CharStats.CODES.BASECODES())
203                        {
204                                final String statDesc;
205                                if(useWords)
206                                {
207                                        final int val=CT.getStat(i);
208                                        if(val < 0)
209                                                statDesc=CMStrings.capitalizeAndLower(charStatList[0]);
210                                        else
211                                        if(val < charStatList.length)
212                                                statDesc=CMStrings.capitalizeAndLower(charStatList[val]);
213                                        else
214                                                statDesc=CMStrings.capitalizeAndLower(getExtremeValue(val-charStatList.length+1))+" "+CMStrings.capitalizeAndLower(charStatList[charStatList.length-1]);
215                                }
216                                else
217                                        statDesc=CMStrings.padRight(Integer.toString(CT.getStat(i)),2)+"/"+(CT.getMaxStat(i));
218                                msg.append(CMStrings.padRight("^<HELP^>" + CMStrings.capitalizeAndLower(CharStats.CODES.NAME(i))+"^</HELP^>",longest)
219                                                +": ^H"+statDesc+"^?\n\r");
220                        }
221                        msg.append("^N\n\r");
222                }
223                msg.append(L("You have ^H@x1^? ^<HELP^>hit points^</HELP^>, ^H",mob.curState().getHitPoints()+"/"+mob.maxState().getHitPoints()));
224                msg.append(L("@x1^? ^<HELP^>mana^</HELP^>, and ^H",mob.curState().getMana()+"/"+mob.maxState().getMana()));
225                msg.append(L("@x1^? ^<HELP^>movement^</HELP^>.\n\r",mob.curState().getMovement()+"/"+mob.maxState().getMovement()));
226                if(mob.phyStats().height()<0)
227                        msg.append(L("You are incorporeal, but still weigh ^!@x1^? pounds.\n\r",""+mob.baseWeight()));
228                else
229                        msg.append(L("You are ^!@x1^? inches tall and weigh ^!@x2^? pounds.\n\r",""+mob.phyStats().height(),""+mob.baseWeight()));
230                if(CMSecurity.isAllowed(mob,mob.location(),CMSecurity.SecFlag.CARRYALL))
231                        msg.append(L("You are carrying ^!@x1^? items weighing ^!@x2^? pounds.\n\r",""+mob.numItems(),""+mob.phyStats().weight()));
232                else
233                        msg.append(L("You are carrying ^!@x1^?/^!@x2^? items weighing ^!@x3^? pounds.\n\r",""+mob.numItems(),""+mob.maxItems(),mob.phyStats().weight()+"^?/^!"+mob.maxCarry()));
234                msg.append(L("You have ^!@x1^? ^<HELP^>practices^</HELP^>, ^!@x2^? ^<HELP^>training sessions^</HELP^>, and ^!@x3^? ^<HELP^>quest points^</HELP^>.\n\r",""+mob.getPractices(),""+mob.getTrains(),""+mob.getQuestPoint()));
235                if((!CMSecurity.isDisabled(CMSecurity.DisFlag.EXPERIENCE))
236                &&(!CMSecurity.isDisabled(CMSecurity.DisFlag.SHOWXP))
237                &&!mob.charStats().getCurrentClass().expless()
238                &&!mob.charStats().getMyRace().expless())
239                {
240                        if((!CMSecurity.isDisabled(CMSecurity.DisFlag.LEVELS))
241                        &&(!mob.charStats().getCurrentClass().leveless())
242                        &&(!mob.charStats().getMyRace().leveless()))
243                        {
244                                if(((CMProps.getIntVar(CMProps.Int.LASTPLAYERLEVEL)>0)
245                                        &&(mob.basePhyStats().level()>CMProps.getIntVar(CMProps.Int.LASTPLAYERLEVEL)))
246                                ||(mob.getExpNeededLevel()==Integer.MAX_VALUE)
247                                ||(mob.charStats().isLevelCapped(mob.charStats().getCurrentClass())))
248                                        msg.append(L("You have scored ^!@x1^? ^<HELP^>experience points^</HELP^>, ^!@x2^? over your last level.\n\r",""+mob.getExperience(),""+mob.getExpNeededDelevel()));
249                                else
250                                        msg.append(L("You have scored ^!@x1^? ^<HELP^>experience points^</HELP^>, and need ^!@x2^? to advance.\n\r",""+mob.getExperience(),""+mob.getExpNeededLevel()));
251                        }
252                        else
253                                msg.append(L("You have scored ^!@x1^? ^<HELP^>experience points^</HELP^>.\n\r",""+mob.getExperience()));
254                }
255                msg.append(L("You have been online for ^!@x1^? hours.\n\r",""+Math.round(CMath.div(mob.getAgeMinutes(),60.0))));
256                final FactionManager fac=CMLib.factions();
257                final boolean useFactionWords=CMProps.Int.Prowesses.FACTION_RANGE.is(prowessCode);
258                for(final Enumeration<String> e=mob.factions();e.hasMoreElements();)
259                {
260                        final String factionID=e.nextElement();
261                        final Faction F=fac.getFaction(factionID);
262                        if(F!=null)
263                        {
264                                final int factionAmt=mob.fetchFaction(factionID);
265                                final Faction.FRange FR=fac.getRange(factionID,factionAmt);
266                                if((FR!=null)
267                                &&(F.showInScore()))
268                                {
269                                        if(useFactionWords)
270                                                msg.append(L("^NYour ")+CMStrings.padRight(L("^<HELP^>@x1^</HELP^>",F.name()),15)+": ^H"+FR.name()+" ^.\n\r");
271                                        else
272                                                msg.append(L("^NYour ")+CMStrings.padRight(L("^<HELP^>@x1^</HELP^>",F.name()),15)+": ^H"+FR.name()+" ^.("+factionAmt+")\n\r");
273                                }
274                        }
275                }
276                if((CMProps.getIntVar(CMProps.Int.COMBATPROWESS)&CMProps.Int.ANY_ARMOR_PROWESS)!=0)
277                        msg.append(L("Your ^<HELP^>armored defence^</HELP^>: ^H@x1^.^N\n\r",CMLib.combat().armorStr(mob)));
278                if((CMProps.getIntVar(CMProps.Int.COMBATPROWESS)&CMProps.Int.ANY_COMBAT_PROWESS)!=0)
279                        msg.append(L("Your ^<HELP^>combat prowess^</HELP^> : ^H@x1^.^N\n\r",CMLib.combat().fightingProwessStr(mob)));
280                if((CMProps.getIntVar(CMProps.Int.COMBATPROWESS)&CMProps.Int.ANY_DAMAGE_PROWESS)!=0)
281                        msg.append(L("Your ^<HELP^>damage threat^</HELP^>  : ^H@x1^.^N\n\r",CMLib.combat().damageProwessStr(mob)));
282                //if(CMLib.flags().canSeeHidden(mob))
283                //    msg.append(L("Your ^<HELP^>observation score^</HELP^> : ^H@x1^?.\n\r",CMLib.flags().getDetectScore(mob)));
284                msg.append(L("Wimpy is set to ^!@x1^? hit points.\n\r",""+mob.getWimpHitPoint()));
285
286                msg.append(getMOBState(mob));
287                msg.append(getAffects(mob.session(),mob,false,false));
288                return msg;
289        }
290
291        @Override
292        public boolean execute(MOB mob, List<String> commands, int metaFlags)
293                throws java.io.IOException
294        {
295                String parm="";
296                if(commands.size()>1)
297                        parm=CMParms.combine(commands,1);
298                final StringBuilder msg=getScore(mob,parm);
299                if(commands.size()==0)
300                {
301                        commands.add(msg.toString());
302                        return false;
303                }
304                if(!mob.isMonster())
305                        mob.session().wraplessPrintln(msg.toString());
306                return false;
307        }

310        public boolean canBeOrdered()
311        {
312                return true;
313        }
