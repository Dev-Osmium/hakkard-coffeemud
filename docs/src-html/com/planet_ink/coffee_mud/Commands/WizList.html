036public class WizList extends StdCommand
037{
038        public WizList()
039        {
040        }
041
042        private final String[] access=I(new String[]{"WIZLIST"});
043        @Override
044        public String[] getAccessWords()
045        {
046                return access;
047        }
048
049        @Override
050        public boolean execute(MOB mob, List<String> commands, int metaFlags)
051                throws java.io.IOException
052        {
053                final StringBuffer head=new StringBuffer("");
054                final boolean isArchonLooker=CMSecurity.isASysOp(mob);
055                head.append("^x[");
056                head.append(CMStrings.padRight(L("Class"),16)+" ");
057                head.append(CMStrings.padRight(L("Race"),8)+" ");
058                head.append(CMStrings.padRight(L("Lvl"),4)+" ");
059                if(isArchonLooker)
060                        head.append(CMStrings.padRight(L("Last"),18)+" ");
061                head.append("] Character Name^.^?\n\r");
062                mob.tell("^x["+CMStrings.centerPreserve(L("The Administrators of @x1",CMProps.getVar(CMProps.Str.MUDNAME)),head.length()-10)+"]^.^?");
063                final java.util.List<PlayerLibrary.ThinPlayer> allUsers=CMLib.database().getExtendedUserList();
064                String mask=CMProps.getVar(CMProps.Str.WIZLISTMASK);
065                if(mask.length()==0)
066                        mask="-ANYCLASS +Archon";
067                final MaskingLibrary.CompiledZMask compiledMask=CMLib.masking().maskCompile(mask);
068                for(final PlayerLibrary.ThinPlayer U : allUsers)
069                {
070                        CharClass C;
071                        final MOB player = CMLib.players().getPlayer(U.name()); // only called from this t-groups database anyway, so keep it local
072                        if(player != null)
073                                C=player.charStats().getCurrentClass();
074                        else
075                                C=CMClass.getCharClass(U.charClass());
076                        if(C==null)
077                                C=CMClass.findCharClass(U.charClass());
078                        if(((player!=null)&&(CMLib.masking().maskCheck(compiledMask, player, true)))
079                        ||(CMLib.masking().maskCheck(compiledMask, U)))
080                        {
081                                head.append("[");
082                                if(C!=null)
083                                        head.append(CMStrings.padRight(C.name(),16)+" ");
084                                else
085                                        head.append(CMStrings.padRight(L("Unknown"),16)+" ");
086                                head.append(CMStrings.padRight(U.race(),8)+" ");
087                                if((C==null)||(!C.leveless()))
088                                        head.append(CMStrings.padRight(""+U.level(),4)+" ");
089                                else
090                                        head.append(CMStrings.padRight("    ",4)+" ");
091                                if(isArchonLooker)
092                                        head.append(CMStrings.padRight(CMLib.time().date2String(U.last()),18)+" ");
093                                head.append("] "+U.name());
094                                head.append("\n\r");
095                        }
096                }
097                mob.tell(head.toString());
098                return false;
099        }
100
101        @Override
102        public boolean canBeOrdered()
103        {
104                return true;
105        }
106
107}
