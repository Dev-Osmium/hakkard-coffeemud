058        @Override
059        public boolean execute(MOB mob, List<String> commands, int metaFlags)
060                throws java.io.IOException
061        {
062                String rest="";
063                if(commands.size()>1)
064                        rest=CMParms.combine(commands,1);
065                final Session session=mob.session();
066                if(session==null)
067                        return false;
068
069                JScriptEvent scope=(JScriptEvent)Resources.getResource("JCONSOLE_"+mob.Name());
070                final Context cx=Context.enter();
071                try
072                {
073                        if(scope==null)
074                        {
075                                try
076                                {
077                                        scope = new JScriptEvent(mob);
078                                        cx.initStandardObjects(scope);
079                                        final String[] names = { "mob" ,"getVar", "setVar", "toJavaString"};
080                                        scope.defineFunctionProperties(names, JScriptEvent.class, ScriptableObject.DONTENUM);
081                                        Resources.submitResource("JCONSOLE_"+mob.Name(), scope);
082                                }
083                                catch(final Exception e)
084                                {
085                                        mob.tell(L("Error: @x1",e.getMessage()));
086                                        return false;
087                                }
088                        }
089                        if(rest.length()>0)
090                        {
091                                try
092                                {
093                                        mob.tell(cx.evaluateString(scope, rest.toString(),"<cmd>", 1, null).toString());
094                                        return true;
095                                }
096                                catch(final Exception e)
097                                {
098                                        mob.tell(L("Error: @x1",e.getMessage()));
099                                        return false;
100                                }
101                        }
102                        final JScriptEvent myScope=scope;
103                        final InputCallback IC[]=new InputCallback[1];
104                        final boolean[] addMode=new boolean[1];
105                        final StringBuilder inpBuilder=new StringBuilder("");
106                        myScope.c.tick(mob, Tickable.TICKID_MOB); // set lastknownlocation
107                        IC[0]=new InputCallback(InputCallback.Type.PROMPT)
108                        {
109                                @Override
110                                public void showPrompt()
111                                {
112                                        session.print(addMode[0]?".":">");
113                                }
114
115                                @Override
116                                public void timedOut()
117                                {
118                                }
119
120                                @Override 
121                                public void callBack()
122                                {
123                                        if(this.input.equalsIgnoreCase("exit"))
124                                                return;
125                                        if(this.input.equals("<") && !addMode[0])
126                                        {
127                                                addMode[0]=true;
128                                                session.prompt(IC[0].reset());
129                                                return;
130                                        }
131                                        if(this.input.equals(">") && addMode[0])
132                                        {
133                                                addMode[0]=false;
134                                                this.input=inpBuilder.toString();
135                                                inpBuilder.setLength(0);
136                                        }
137                                        if(addMode[0])
138                                                inpBuilder.append(this.input).append("\n");
139                                        else
140                                        if(this.input.length()>0)
141                                        {
142                                                try
143                                                {
144                                                        final Context cx=Context.enter();
145                                                        session.safeRawPrintln(cx.evaluateString(myScope, this.input.replace('`','\''),"<cmd>", 1, null).toString());
146                                                }
147                                                catch(final Exception e)
148                                                {
149                                                        session.println(e.getMessage());
150                                                }
151                                                finally
152                                                {
153                                                        Context.exit();
154                                                }
155                                        }
156                                        session.prompt(IC[0].reset());
157                                }
158                        };
159                        mob.tell(L("JConsole! Enter \"exit\" to exit."));
160                        session.prompt(IC[0]);
161                }
162                finally
163                {
164                        Context.exit();
165                }
166                return false;
167        }

201                @Override
202                public Object get(final String name, Scriptable start)
203                {
204                        if (super.has(name, start))
205                                return super.get(name, start);
206                        if (methH.contains(name) || funcH.contains(name)
207                        || (name.endsWith("$")&&(funcH.contains(name.substring(0,name.length()-1)))))
208                        {
209                                return new Function()
210                                {
211                                        @Override
212                                        public Object call(Context cx, Scriptable scope, Scriptable thisObj, Object[] args)
213                                        {
214                                                if(methH.contains(name))
215                                                {
216                                                        final StringBuilder strb=new StringBuilder(name);
217                                                        if(args.length==1)
218                                                                strb.append(" ").append(String.valueOf(args[0]));
219                                                        else
220                                                        for(int i=0;i<args.length;i++)
221                                                                if(i==args.length-1)
222                                                                        strb.append(" ").append(String.valueOf(args[i]));
223                                                                else
224                                                                        strb.append(" ").append("'"+String.valueOf(args[i])+"'");
225                                                        final DVector DV=new DVector(2);
226                                                        DV.add("JS_PROG",null);
227                                                        DV.add(strb.toString(),null);
228                                                        return c.execute(mob,mob,null,mob,null,null,DV,"",objs);
229                                                }
230                                                if(name.endsWith("$"))
231                                                {
232                                                        final StringBuilder strb=new StringBuilder(name.substring(0,name.length()-1)).append("(");
233                                                        if(args.length==1)
234                                                                strb.append(" ").append(String.valueOf(args[0]));
235                                                        else
236                                                        for(int i=0;i<args.length;i++)
237                                                                if(i==args.length-1)
238                                                                        strb.append(" ").append(String.valueOf(args[i]));
239                                                                else
240                                                                        strb.append(" ").append("'"+String.valueOf(args[i])+"'");
241                                                        strb.append(" ) ");
242                                                        return c.functify(mob,mob,null,mob,null,null,"",objs,strb.toString());
243                                                }
244                                                final String[] sargs=new String[args.length+3];
245                                                sargs[0]=name;
246                                                sargs[1]="(";
247                                                for(int i=0;i<args.length;i++)
248                                                        sargs[i+2]=String.valueOf(args[i]);
249                                                sargs[sargs.length-1]=")";
250                                                final String[][] EVAL={sargs};
251                                                return Boolean.valueOf(c.eval(mob,mob,null,mob,null,null,"",objs,EVAL,0));
252                                        }
