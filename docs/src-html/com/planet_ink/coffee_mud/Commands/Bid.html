036public class Bid extends StdCommand
037{
038        public Bid()
039        {
040        }
041
042        private final String[] access=I(new String[]{"BID"});
043        @Override
044        public String[] getAccessWords()
045        {
046                return access;
047        }
048
049        @Override
050        public boolean execute(MOB mob, List<String> commands, int metaFlags)
051                throws java.io.IOException
052        {
053                Vector<String> origCmds=new XVector<String>(commands);
054                final Environmental shopkeeper=CMLib.english().parseShopkeeper(mob,commands,"Bid how much, on what, with whom?");
055                if(shopkeeper==null)
056                        return false;
057                if(commands.size()<2)
058                {
059                        CMLib.commands().postCommandFail(mob,origCmds,L("Bid how much on what?"));
060                        return false;
061                }
062                if(!(CMLib.coffeeShops().getShopKeeper(shopkeeper) instanceof Auctioneer))
063                {
064                        CMLib.commands().postCommandFail(mob,origCmds,L("@x1 is not an auctioneer!",shopkeeper.name()));
065                        return false;
066                }
067
068                String bidStr=commands.get(0);
069                if(CMLib.english().numPossibleGold(mob,bidStr)<=0)
070                {
071                        CMLib.commands().postCommandFail(mob,origCmds,L("It does not look like '@x1' is enough to offer.",bidStr));
072                        return false;
073                }
074                final Triad<String,Double,Long> bidThang=CMLib.english().parseMoneyStringSDL(mob,bidStr,null);
075                bidStr=CMLib.beanCounter().nameCurrencyShort(bidThang.first,CMath.mul(bidThang.second.doubleValue(),bidThang.third.longValue()));
076                commands.remove(0);
077
078                int maxToDo=Integer.MAX_VALUE;
079                if((commands.size()>1)
080                &&(CMath.s_int(commands.get(0))>0))
081                {
082                        maxToDo=CMath.s_int(commands.get(0));
083                        commands.set(0,"all");
084                }
085
086                String whatName=CMParms.combine(commands,0);
087                final Vector<Environmental> V=new Vector<Environmental>();
088                boolean allFlag=commands.get(0).equalsIgnoreCase("all");
089                if(whatName.toUpperCase().startsWith("ALL."))
090                {
091                        allFlag=true;
092                        whatName="ALL "+whatName.substring(4);
093                }
094                if(whatName.toUpperCase().endsWith(".ALL"))
095                {
096                        allFlag=true;
097                        whatName="ALL "+whatName.substring(0,whatName.length()-4);
098                }
099                int addendum=1;
100                boolean doBugFix = true;
101                while(doBugFix || ((allFlag)&&(addendum<=maxToDo)))
102                {
103                        doBugFix=false;
104                        final ShopKeeper SK=CMLib.coffeeShops().getShopKeeper(shopkeeper);
105                        final Environmental itemToDo=SK.getShop().getStock(whatName,mob);
106                        if(itemToDo==null)
107                                break;
108                        if(CMLib.flags().canBeSeenBy(itemToDo,mob))
109                                V.add(itemToDo);
110                        if(addendum>=CMLib.coffeeShops().getShopKeeper(shopkeeper).getShop().numberInStock(itemToDo))
111                                break;
112                        ++addendum;
113                }
114                if(V.size()==0)
115                        mob.tell(mob,shopkeeper,null,L("<T-NAME> do(es)n't appear to have any '@x1' available for auction.  Try LIST.",whatName));
116                else
117                for(int v=0;v<V.size();v++)
118                {
119                        final Environmental thisThang=V.get(v);
120                        final CMMsg newMsg=CMClass.getMsg(mob,shopkeeper,thisThang,
121                                        CMMsg.MSG_BID,L("<S-NAME> bid(s) @x1 on <O-NAME> with <T-NAMESELF>.",bidStr),
122                                        CMMsg.MSG_BID,L("<S-NAME> bid(s) '@x1' on <O-NAME> with <T-NAMESELF>.",bidStr),
123                                        CMMsg.MSG_BID,L("<S-NAME> place(s) a bid with <T-NAMESELF>."));
124                        if(mob.location().okMessage(mob,newMsg))
125                                mob.location().send(mob,newMsg);
126                }
127                return false;
128        }
129
130        @Override
131        public double combatActionsCost(final MOB mob, final List<String> cmds)
132        {
133                return CMProps.getCommandCombatActionCost(ID());
134        }
135
136        @Override
137        public double actionsCost(final MOB mob, final List<String> cmds)
138        {
139                return CMProps.getCommandActionCost(ID());
140        }
141
142        @Override
143        public boolean canBeOrdered()
144        {
145                return false;
146        }
147}
