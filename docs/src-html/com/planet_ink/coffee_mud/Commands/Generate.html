047        private static final SHashtable<String,CMClass.CMObjectType> OBJECT_TYPES=new SHashtable<String,CMClass.CMObjectType>(new Object[][]{
048                        {"STRING",CMClass.CMObjectType.LIBRARY},
049                        {"AREA",CMClass.CMObjectType.AREA},
050                        {"MOB",CMClass.CMObjectType.MOB},
051                        {"ROOM",CMClass.CMObjectType.LOCALE},

063        public void createNewPlace(MOB mob, Room oldR, Room R, int direction, boolean save)
064        {
065                if(R.roomID().length()==0)
066                {
067                        R.setArea(oldR.getArea());
068                        R.setRoomID(oldR.getArea().getNewRoomID(oldR, direction));
069                }
070                Exit E=R.getExitInDir(Directions.getOpDirectionCode(direction));
071                if(E==null)
072                        E = CMClass.getExit("Open");
073                oldR.setRawExit(direction, E);
074                oldR.rawDoors()[direction]=R;
075                final int opDir=Directions.getOpDirectionCode(direction);
076                if(R.getRoomInDir(opDir)!=null)
077                        mob.tell(L("An error has caused the following exit to be one-way."));
078                else
079                {
080                        R.setRawExit(opDir, E);
081                        R.rawDoors()[opDir]=oldR;
082                }
083                if(save)
084                        CMLib.database().DBUpdateExits(oldR);
085                final String dirName=((R instanceof BoardableShip)||(R.getArea() instanceof BoardableShip))?
086                                CMLib.directions().getShipDirectionName(direction):CMLib.directions().getDirectionName(direction);
087                oldR.showHappens(CMMsg.MSG_OK_VISUAL,L("A new place materializes to the @x1",dirName));
088        }

090        @Override
091        public boolean execute(MOB mob, List<String> commands, int metaFlags)
092                throws java.io.IOException
093        {
094                boolean save=true;
095                if(commands.size()>1)
096                {
097                        if(commands.get(1).equalsIgnoreCase("nosave"))
098                        {
099                                save=false;
100                                commands.remove(1);
101                        }
102                }
103                if(commands.size()<3)
104                {
105                        mob.tell(L("Generate what? Try GENERATE [TYPE] [ID] (FROM [DATA_FILE_PATH]) ([VAR=VALUE]..) [DIRECTION]"));
106                        return false;
107                }
108                final String finalLog = mob.Name()+" called generate command with parms: " + CMParms.combine(commands, 1);
109                CMFile file = null;
110                if((commands.size()>3)&&commands.get(3).equalsIgnoreCase("FROM"))
111                {
112                        file = new CMFile(Resources.buildResourcePath(commands.get(4)),mob);
113                        commands.remove(3);
114                        commands.remove(3);
115                }
116                else
117                        file = new CMFile(Resources.buildResourcePath("randareas/example.xml"),mob);
118                if(!file.canRead())
119                {
120                        mob.tell(L("Random data file '@x1' not found.  Aborting.",file.getCanonicalPath()));
121                        return false;
122                }
123                final StringBuffer xml = file.textUnformatted();
124                final List<XMLLibrary.XMLTag> xmlRoot = CMLib.xml().parseAllXML(xml);
125                final Hashtable<String,Object> definedIDs = new Hashtable<String,Object>();
126                CMLib.percolator().buildDefinedIDSet(xmlRoot,definedIDs);
127                final String typeName = commands.get(1);
128                String objectType = typeName.toUpperCase().trim();
129                CMClass.CMObjectType codeI=OBJECT_TYPES.get(objectType);
130                if(codeI==null)
131                {
132                        for(final Enumeration<String> e=OBJECT_TYPES.keys();e.hasMoreElements();)
133                        {
134                                final String key =e.nextElement();
135                                if(key.startsWith(typeName.toUpperCase().trim()))
136                                {
137                                        objectType = key;
138                                        codeI=OBJECT_TYPES.get(key);
139                                }
140                        }
141                        if(codeI==null)
142                        {
143                                mob.tell(L("'@x1' is an unknown object type.  Try: @x2",typeName,CMParms.toListString(OBJECT_TYPES.keys())));
144                                return false;
145                        }
146                }
147                int direction=-1;
148                if((codeI==CMClass.CMObjectType.AREA)||(codeI==CMClass.CMObjectType.LOCALE))
149                {
150                        final String possDir=commands.get(commands.size()-1);
151                        direction = CMLib.directions().getGoodDirectionCode(possDir);
152                        if(direction<0)
153                        {
154                                mob.tell(L("When creating an area or room, the LAST parameter to this command must be a direction to link to this room by."));
155                                return false;
156                        }
157                        if(mob.location().getRoomInDir(direction)!=null)
158                        {
159                                final String dirName=((mob.location() instanceof BoardableShip)||(mob.location().getArea() instanceof BoardableShip))?
160                                                CMLib.directions().getShipDirectionName(direction):CMLib.directions().getDirectionName(direction);
161                                mob.tell(L("A room already exists in direction @x1. Action aborted.",dirName));
162                                return false;
163                        }
164                }
165                final String idName = commands.get(2).toUpperCase().trim();
166                if((!(definedIDs.get(idName) instanceof XMLTag))
167                ||(!((XMLTag)definedIDs.get(idName)).tag().equalsIgnoreCase(objectType)))
168                {
169                        if(!idName.equalsIgnoreCase("LIST"))
170                                mob.tell(L("The @x1 id '@x2' has not been defined in the data file.",objectType,idName));
171                        final StringBuffer foundIDs=new StringBuffer("");
172                        for(final Enumeration<String> tkeye=OBJECT_TYPES.keys();tkeye.hasMoreElements();)
173                        {
174                                final String tKey=tkeye.nextElement();
175                                foundIDs.append("^H"+tKey+"^N: \n\r");
176                                final Vector<String> xmlTagsV=new Vector<String>();
177                                for(final Enumeration<String> keys=definedIDs.keys();keys.hasMoreElements();)
178                                {
179                                        final String key=keys.nextElement();
180                                        if((definedIDs.get(key) instanceof XMLTag)
181                                        &&(((XMLTag)definedIDs.get(key)).tag().equalsIgnoreCase(tKey)))
182                                                xmlTagsV.add(key.toLowerCase());
183                                }
184                                foundIDs.append(CMParms.toListString(xmlTagsV)+"\n\r");
185                        }
186                        mob.tell(L("Found ids include: \n\r@x1",foundIDs.toString()));
187                        return false;
188                }
189
190                final XMLTag piece=(XMLTag)definedIDs.get(idName);
191                definedIDs.putAll(CMParms.parseEQParms(commands,3,commands.size()));
192                try
193                {
194                        CMLib.percolator().checkRequirements(piece, definedIDs);
195                }
196                catch(final CMException cme)
197                {
198                        mob.tell(L("Required ids for @x1 were missing: @x2",idName,cme.getMessage()));
199                        return false;
200                }
201                final Vector<Object> V = new Vector<Object>();
202                try
203                {
204                        switch(codeI)
205                        {
206                        case LIBRARY:
207                        {
208                                CMLib.percolator().preDefineReward(piece, definedIDs);
209                                CMLib.percolator().defineReward(piece,definedIDs);
210                                final String s=CMLib.percolator().findString("STRING", piece, definedIDs);
211                                if(s!=null)
212                                        V.add(s);
213                                break;
214                        }
215                        case AREA:
216                                CMLib.percolator().preDefineReward(piece, definedIDs);
217                                CMLib.percolator().defineReward(piece,definedIDs);
218                                definedIDs.put("ROOMTAG_NODEGATEEXIT", CMLib.directions().getDirectionName(Directions.getOpDirectionCode(direction)));
219                                definedIDs.put("ROOMTAG_GATEEXITROOM", mob.location());
220                                final Area A=CMLib.percolator().findArea(piece, definedIDs, direction);
221                                if(A!=null)
222                                        V.add(A);
223                                break;
224                        case MOB:
225                                CMLib.percolator().preDefineReward(piece, definedIDs);
226                                CMLib.percolator().defineReward(piece,definedIDs);
227                                V.addAll(CMLib.percolator().findMobs(piece, definedIDs));
228                                break;
229                        case LOCALE:
230                        {
231                                final Exit[] exits=new Exit[Directions.NUM_DIRECTIONS()];
232                                CMLib.percolator().preDefineReward(piece, definedIDs);
233                                CMLib.percolator().defineReward(piece,definedIDs);
234                                definedIDs.put("ROOMTAG_NODEGATEEXIT", CMLib.directions().getDirectionName(Directions.getOpDirectionCode(direction)));
235                                definedIDs.put("ROOMTAG_GATEEXITROOM", mob.location());
236                                final Room R=CMLib.percolator().buildRoom(piece, definedIDs, exits, direction);
237                                if(R!=null)
238                                        V.add(R);
239                                break;
240                        }
241                        case ITEM:
242                                CMLib.percolator().preDefineReward(piece, definedIDs);
243                                CMLib.percolator().defineReward(piece,definedIDs);
244                                V.addAll(CMLib.percolator().findItems(piece, definedIDs));
245                                break;
246                        default:
247                                break;
248                        }
249                        if(V.size()==0)
250                                mob.tell(L("Nothing generated."));
251                        else
252                        for(int v=0;v<V.size();v++)
253                        {
254                                if(V.get(v) instanceof MOB)
255                                {
256                                        ((MOB)V.get(v)).bringToLife(mob.location(),true);
257                                        mob.location().showHappens(CMMsg.MSG_OK_VISUAL,L("@x1 appears.",((MOB)V.get(v)).name()));
258                                        CMLib.percolator().postProcess(definedIDs);
259                                        Log.sysOut("Generate",mob.Name()+" generated mob "+((MOB)V.get(v)).name());
260                                }
261                                else
262                                if(V.get(v) instanceof Item)
263                                {
264                                        mob.location().addItem((Item)V.get(v));
265                                        mob.location().showHappens(CMMsg.MSG_OK_VISUAL,L("@x1 appears.",((Item)V.get(v)).name()));
266                                        CMLib.percolator().postProcess(definedIDs);
267                                        Log.sysOut("Generate",mob.Name()+" generated item "+((Item)V.get(v)).name());
268                                }
269                                else
270                                if(V.get(v) instanceof String)
271                                {
272                                        CMLib.percolator().postProcess(definedIDs);
273                                        mob.tell((String)V.get(v));
274                                }
275                                else
276                                if(V.get(v) instanceof Room)
277                                {
278                                        final Room R=(Room)V.get(v);
279                                        createNewPlace(mob,mob.location(),R,direction,save);
280                                        CMLib.percolator().postProcess(definedIDs);
281                                        if(save)
282                                        {
283                                                CMLib.database().DBCreateRoom(R);
284                                                CMLib.database().DBUpdateExits(R);
285                                                CMLib.database().DBUpdateItems(R);
286                                                CMLib.database().DBUpdateMOBs(R);
287                                        }
288                                        Log.sysOut("Generate",mob.Name()+" generated room "+R.roomID());
289                                }
290                                else
291                                if(V.get(v) instanceof Area)
292                                {
293                                        final Area A=(Area)V.get(v);
294                                        CMLib.map().addArea(A);
295                                        if(save)
296                                                CMLib.database().DBCreateArea(A);
297                                        Room R=A.getRoom(A.Name()+"#0");
298                                        if(R==null)
299                                                R=A.getFilledProperMap().nextElement();
300                                        createNewPlace(mob,mob.location(),R,direction,save);
301                                        CMLib.percolator().postProcess(definedIDs);
302                                        if(save)
303                                        {
304                                                mob.tell(L("Saving remaining rooms for area '@x1'...",A.name()));
305                                                for(final Enumeration<Room> e=A.getFilledProperMap();e.hasMoreElements();)
306                                                {
307                                                        R=e.nextElement();
308                                                        CMLib.database().DBCreateRoom(R);
309                                                        CMLib.database().DBUpdateExits(R);
310                                                        CMLib.database().DBUpdateItems(R);
311                                                        CMLib.database().DBUpdateMOBs(R);
312                                                }
313                                                mob.tell(L("Done saving remaining rooms for area '@x1'",A.name()));
314                                        }
315                                        Log.sysOut("Generate",mob.Name()+" generated area "+A.name());
316                                }
317                        }
318                }
319                catch(final CMException cex)
320                {
321                        mob.tell(L("Unable to fully generate: @x1",cex.getMessage()));
322                        if(CMSecurity.isDebugging(CMSecurity.DbgFlag.MUDPERCOLATOR))
323                                Log.debugOut("Generate",cex);
324                        return false;
325                }
326                Log.sysOut("Generate",finalLog);
327                return true;
328        }
