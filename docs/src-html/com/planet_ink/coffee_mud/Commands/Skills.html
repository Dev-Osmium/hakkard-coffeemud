054        protected boolean parsedOutIndividualSkill(MOB mob, String qual, List<Integer> acodes)
055        {
056                if((qual==null)||(qual.length()==0)||(qual.equalsIgnoreCase("all")))
057                        return false;
058                if(qual.length()>0)
059                {
060                        for(int i=1;i<Ability.DOMAIN_DESCS.length;i++)
061                        {
062                                if(Ability.DOMAIN_DESCS[i].replace('_',' ').equalsIgnoreCase(qual))
063                                        return false;
064                                else
065                                if((Ability.DOMAIN_DESCS[i].replace('_',' ').indexOf('/')>=0)
066                                &&(Ability.DOMAIN_DESCS[i].replace('_',' ').substring(Ability.DOMAIN_DESCS[i].indexOf('/')+1).equalsIgnoreCase(qual)))
067                                        return false;
068                        }
069                }
070                final Ability A=CMClass.findAbility(qual);
071                if((A!=null)
072                &&(acodes.contains(Integer.valueOf(A.classificationCode()&Ability.ALL_ACODES))))
073                {
074                        final Ability A2=mob.fetchAbility(A.ID());
075                        if(A2==null)
076                                mob.tell(L("You don't know '@x1'.",A.name()));
077                        else
078                        {
079                                int level=CMLib.ableMapper().qualifyingLevel(mob,A2);
080                                if(level<0)
081                                        level=0;
082                                final StringBuffer line=new StringBuffer("");
083                                line.append("\n\rLevel ^!"+level+"^?:\n\r");
084                                line.append("^N[^H"+CMStrings.padRight(Integer.toString(A2.proficiency()),3)+"%^?]^N "+CMStrings.padRight("^<HELP^>"+A2.name()+"^</HELP^>",19));
085                                line.append("^?\n\r");
086                                if(mob.session()!=null)
087                                        mob.session().wraplessPrintln(line.toString());
088                        }
089                        return true;
090                }
091                return false;
092        }

094        protected int parseOutLevel(List<String> commands)
095        {
096                if((commands.size()>1)
097                &&(CMath.isNumber(commands.get(commands.size()-1))))
098                {
099                        final int x=CMath.s_int(commands.get(commands.size()-1));
100                        commands.remove(commands.size()-1);
101                        return x;
102                }
103                return -1;
104        }

106        /**
107         * Returns whether there are any crossings between a particular Ability type
108         * and a particular Ability domain.
109         * @see com.planet_ink.coffee_mud.Abilities.interfaces.Ability#abilityCode()
110         * @see com.planet_ink.coffee_mud.Abilities.interfaces.Ability#DOMAIN_DESCS
111         * @see com.planet_ink.coffee_mud.Abilities.interfaces.Ability#ACODE_DESCS
112         * 
113         * @param domain the domain mask 
114         * @param acode the ability code
115         * @return true if they meet somewhere
116         */
117        public boolean isDomainIncludedInAnyAbility(int domain, int acode)
118        {
119                @SuppressWarnings("unchecked")
120                Map<Integer, Set<Integer>> completeDomainMap = (Map<Integer, Set<Integer>>)Resources.getResource("SYSYETM_ABLEDOMAINMAP");
121                if(completeDomainMap == null)
122                {
123                        completeDomainMap = new SHashtable<Integer,Set<Integer>>();
124                        Resources.submitResource("SYSYETM_ABLEDOMAINMAP",completeDomainMap);
125                }
126                STreeSet<Integer> V=(STreeSet<Integer>)completeDomainMap.get(Integer.valueOf(domain));
127                if(V==null)
128                {
129                        Ability A=null;
130                        V=new STreeSet<Integer>();
131                        for(final Enumeration<Ability> e=CMClass.abilities();e.hasMoreElements();)
132                        {
133                                A=e.nextElement();
134                                if(((A.classificationCode()&Ability.ALL_DOMAINS)==domain)
135                                &&(!V.contains(Integer.valueOf((A.classificationCode()&Ability.ALL_ACODES)))))
136                                        V.add(Integer.valueOf((A.classificationCode()&Ability.ALL_ACODES)));
137                        }
138                        completeDomainMap.put(Integer.valueOf(domain),V);
139                }
140                return V.contains(Integer.valueOf(acode));
141        }

143        protected void parseDomainInfo(MOB mob, List<String> commands, Vector<Integer> acodes, int[] level, int[] domain, String[] domainName)
144        {
145                level[0]=parseOutLevel(commands);
146                final String qual=CMParms.combine(commands,1).toUpperCase();
147                domain[0]=-1;
148                if(qual.length()>0)
149                for(int i=1;i<Ability.DOMAIN_DESCS.length;i++)
150                {
151                        if(Ability.DOMAIN_DESCS[i].replace('_',' ').startsWith(qual))
152                        {
153                                domain[0] = i << 5;
154                                break;
155                        }
156                        else
157                        if((Ability.DOMAIN_DESCS[i].replace('_',' ').indexOf('/')>=0)
158                        &&(Ability.DOMAIN_DESCS[i].replace('_',' ').substring(Ability.DOMAIN_DESCS[i].indexOf('/')+1).startsWith(qual)))
159                        {
160                                domain[0] = i << 5;
161                                break;
162                        }
163                }
164                if(domain[0]>0)
165                        domainName[0]=Ability.DOMAIN_DESCS[domain[0]>>5].toLowerCase();
166                if((domain[0]<0)&&(qual.length()>0))
167                {
168                        StringBuffer domains=new StringBuffer("");
169                        domains.append("\n\rValid schools/domains are: ");
170                        for(int i=1;i<Ability.DOMAIN_DESCS.length;i++)
171                        {
172                                boolean found=false;
173                                for(int a=0;a<acodes.size();a++)
174                                        found=found||isDomainIncludedInAnyAbility(i<<5,acodes.get(a).intValue());
175                                if(found)
176                                        domains.append(Ability.DOMAIN_DESCS[i].toLowerCase().replace('_',' ')+", ");
177                        }
178                        if(domains.toString().endsWith(", "))
179                                domains=new StringBuffer(domains.substring(0,domains.length()-2));
180                        if(!mob.isMonster())
181                                mob.session().wraplessPrintln(domains.toString()+"\n\r");
182                }
183                else
184                if(qual.length()>0)
185                        domainName[0]+=" ";
186        }

188        protected StringBuilder getAbilities(MOB viewerM, MOB ableM, int ofType, int ofDomain, boolean addQualLine, int maxLevel)
189        {
190                final ArrayList<Integer> V=new ArrayList<Integer>();
191                int mask=Ability.ALL_ACODES;
192                if(ofDomain>=0)
193                {
194                        mask=Ability.ALL_ACODES|Ability.ALL_DOMAINS;
195                        ofType=ofType|ofDomain;
196                }
197                V.add(Integer.valueOf(ofType));
198                return getAbilities(viewerM,ableM,V,mask,addQualLine,maxLevel);
199        }

201        protected final static Comparator<Ability> nameComparator=new Comparator<Ability>()
202        {
203                @Override
204                public int compare(Ability o1, Ability o2)
205                {
206                        if(o1==null)
207                                return o2==null?0:-1;
208                        if(o2==null)
209                                return 1;
210                        return o1.Name().compareTo(o2.Name());
211                }
212        };

214        protected StringBuilder getAbilities(MOB viewerM, MOB ableM, List<Integer> ofTypes, int mask, boolean addQualLine, int maxLevel)
215        {
216                final int COL_LEN1=CMLib.lister().fixColWidth(3.0,viewerM);
217                final int COL_LEN2=CMLib.lister().fixColWidth(18.0,viewerM);
218                final int COL_LEN3=CMLib.lister().fixColWidth(18.0,viewerM);
219                int highestLevel=0;
220                final int lowestLevel=ableM.phyStats().level()+1;
221                final StringBuilder msg=new StringBuilder("");
222                for(final Enumeration<Ability> a=ableM.allAbilities();a.hasMoreElements();)
223                {
224                        final Ability A=a.nextElement();
225                        int level=CMLib.ableMapper().qualifyingLevel(ableM,A);
226                        if(level<0)
227                                level=0;
228                        if((A!=null)
229                        &&(level>highestLevel)
230                        &&(level<lowestLevel)
231                        &&(ofTypes.contains(Integer.valueOf(A.classificationCode()&mask))))
232                                highestLevel=level;
233                }
234                if((maxLevel>=0)&&(maxLevel<highestLevel))
235                        highestLevel=maxLevel;
236                final int prowessCode = CMProps.getIntVar(CMProps.Int.COMBATPROWESS);
237                final boolean useWords=CMProps.Int.Prowesses.SKILL_PROFICIENCY.is(prowessCode);
238                final int[] proficiencyRanges=new int[]{5,10,20,30,40,50,60,70,75,80,85,90,95,100};
239                int MAX_COLS=useWords?2:3;
240                final List<Ability> sortedAllAbilities = new XVector<Ability>(ableM.allAbilities());
241                Collections.sort(sortedAllAbilities,nameComparator);
242                for(int l=0;l<=highestLevel;l++)
243                {
244                        final StringBuilder thisLine=new StringBuilder("");
245                        int col=0;
246                        
247                        for(final Ability A : sortedAllAbilities)
248                        {
249                                int level=CMLib.ableMapper().qualifyingLevel(ableM,A);
250                                if(level<0)
251                                        level=0;
252                                if((A!=null)
253                                &&(level==l)
254                                &&(ofTypes.contains(Integer.valueOf(A.classificationCode()&mask))))
255                                {
256                                        if(thisLine.length()==0)
257                                                thisLine.append("\n\rLevel ^!"+l+"^?:\n\r");
258                                        col++;
259                                        if(!useWords)
260                                        {
261                                                thisLine.append("^N[^H").append(CMStrings.padRight(Integer.toString(A.proficiency()),COL_LEN1));
262                                                thisLine.append("%^?]^N");
263                                                thisLine.append(" ");//+(A.isAutoInvoked()?"^H.^N":" ")
264                                                if(col < MAX_COLS)
265                                                        thisLine.append(CMStrings.padRight("^<HELP^>",A.name(),"^</HELP^>",COL_LEN2));
266                                                else
267                                                {
268                                                        thisLine.append(CMStrings.limit("^<HELP^>",A.name(),"^</HELP^>\n\r",COL_LEN3));
269                                                        col=0;
270                                                }
271                                        }
272                                        else
273                                        {
274                                                thisLine.append(CMStrings.padRight("^<HELP^>",A.name(),"^</HELP^>",COL_LEN2));
275                                                int ordinal=0;
276                                                for(int i=0;i<proficiencyRanges.length;i++)
277                                                {
278                                                        if(A.proficiency()<=proficiencyRanges[i])
279                                                        {
280                                                                ordinal=i;
281                                                                break;
282                                                        }
283                                                }
284                                                final String message=CMProps.getListFileChoiceFromIndexedList(CMProps.ListFile.SKILL_PROFICIENCY_DESC, ordinal);
285                                                if(col < MAX_COLS)
286                                                        thisLine.append(CMStrings.padRight("^N(^H",message,"^?)^N",COL_LEN2));
287                                                else
288                                                {
289                                                        thisLine.append(CMStrings.limit("^N(^H",message,"^?)^N\n\r",COL_LEN3));
290                                                        col=0;
291                                                }
292                                        }
293                                }
294                        }
295                        if(thisLine.length()>0)
296                                msg.append(thisLine);
297                }
298                if(msg.length()==0)
299                        msg.append(L("^!None!^?"));
300                else
301                if(addQualLine)
302                        msg.append(L("\n\r\n\rUse QUALIFY to see additional skills you can GAIN."));// ^H.^N = passive/auto-invoked."));
303                return msg;
304        }

306        @Override
307        public boolean execute(MOB mob, List<String> commands, int metaFlags)
308                throws java.io.IOException
309        {
310                final StringBuilder msg=new StringBuilder("");
311                final Vector<Integer> V=new Vector<Integer>();
312                V.add(Integer.valueOf(Ability.ACODE_THIEF_SKILL));
313                V.add(Integer.valueOf(Ability.ACODE_SKILL));
314                V.add(Integer.valueOf(Ability.ACODE_COMMON_SKILL));
315                final String qual=CMParms.combine(commands,1).toUpperCase();
316                if(parsedOutIndividualSkill(mob,qual,V))
317                        return true;
318                final int[] level=new int[1];
319                final int[] domain=new int[1];
320                final String[] domainName=new String[1];
321                domainName[0]="";
322                level[0]=-1;
323                parseDomainInfo(mob,commands,V,level,domain,domainName);
324                int mask=Ability.ALL_ACODES;
325                if(domain[0]>=0)
326                {
327                        mask=mask|Ability.ALL_DOMAINS;
328                        for(int v=0;v<V.size();v++)
329                                V.setElementAt(Integer.valueOf(V.get(v).intValue()+domain[0]),v);
330                }
331                if((domain[0]>=0)||(qual.length()==0))
332                        msg.append(L("\n\r^HYour @x1skills:^? @x2",domainName[0].replace('_',' '),getAbilities(mob,mob,V,mask,true,level[0]).toString()));
333                if(!mob.isMonster())
334                        mob.session().wraplessPrintln(msg.toString());
335                return false;
336        }
