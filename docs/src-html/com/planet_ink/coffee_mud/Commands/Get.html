062        public static boolean get(MOB mob, Item container, Item getThis, boolean quiet, String getWord, boolean optimize)
063        {
064                final Room R=mob.location();
065                String theWhat="<T-NAME>";
066                Item target=getThis;
067                Item tool=null;
068                if(container!=null)
069                {
070                        tool=getThis;
071                        target=container;
072                        theWhat="<O-NAME> from <T-NAME>";
073                }
074                if(!getThis.amWearingAt(Wearable.IN_INVENTORY))
075                {
076                        final CMMsg msg=CMClass.getMsg(mob,getThis,null,(optimize?CMMsg.MASK_OPTIMIZE:0)|CMMsg.MSG_REMOVE,null);
077                        if(!R.okMessage(mob,msg))
078                                return false;
079                        R.send(mob,msg);
080                }
081                final CMMsg msg=CMClass.getMsg(mob,target,tool,(optimize?CMMsg.MASK_OPTIMIZE:0)|CMMsg.MSG_GET,quiet?null:CMLib.lang().L("<S-NAME> @x1(s) @x2.",getWord,theWhat));
082                if(!R.okMessage(mob,msg))
083                        return false;
084                // we do this next step because, when a container is involved,
085                // the item deserves to be the target of the GET.
086                if((!mob.isMine(target))&&(target!=getThis))
087                {
088                        final CMMsg msg2=CMClass.getMsg(mob,getThis,null,(optimize?CMMsg.MASK_OPTIMIZE:0)|CMMsg.MSG_GET,null);
089                        if(!R.okMessage(mob,msg2))
090                                return false;
091                        R.send(mob,msg);
092                        R.send(mob,msg2);
093                }
094                else
095                {
096                        R.send(mob,msg);
097                }
098                return true;
099        }

101        @Override
102        public boolean execute(MOB mob, List<String> commands, int metaFlags)
103                throws java.io.IOException
104        {
105                final Room R=mob.location();
106                Vector<String> origCmds=new XVector<String>(commands);
107                if(commands.size()<2)
108                {
109                        CMLib.commands().postCommandFail(mob,origCmds,L("Get what?"));
110                        return false;
111                }
112                commands.remove(0);
113                boolean quiet=false;
114                if((commands.size()>0)&&(commands.get(commands.size()-1).equalsIgnoreCase("UNOBTRUSIVELY")))
115                {
116                        quiet=true;
117                        commands.remove(commands.size()-1);
118                }
119
120                String containerName="";
121                if(commands.size()>0)
122                        containerName=commands.get(commands.size()-1);
123                final Vector<String> containerCommands=new XVector<String>(commands);
124                final java.util.List<Container> containers=CMLib.english().possibleContainers(mob,commands,Wearable.FILTER_ANY,true);
125                int c=0;
126
127                int maxToGet=CMLib.english().calculateMaxToGive(mob,commands,containers.size()==0,R,true);
128                if(maxToGet<0)
129                        return false;
130
131                String whatToGet=CMParms.combine(commands,0);
132                final String unmodifiedWhatToGet=whatToGet;
133                boolean allFlag = (commands.size() > 0) ? commands.get(0).equalsIgnoreCase("all") : false;
134                if (whatToGet.toUpperCase().startsWith("ALL."))
135                {
136                        allFlag = true;
137                        whatToGet = "ALL " + whatToGet.substring(4);
138                }
139                if (whatToGet.toUpperCase().endsWith(".ALL"))
140                {
141                        allFlag = true;
142                        whatToGet = "ALL " + whatToGet.substring(0, whatToGet.length() - 4);
143                }
144                boolean doneSomething=false;
145                while((c<containers.size())||(containers.size()==0))
146                {
147                        final Vector<Item> V=new Vector<Item>();
148                        Container container=null;
149                        if(containers.size()>0)
150                                container=containers.get(c++);
151                        int addendum=1;
152                        String addendumStr="";
153                        boolean doBugFix = true;
154                        while(doBugFix || ((allFlag)&&(addendum<=maxToGet)))
155                        {
156                                doBugFix=false;
157                                Environmental getThis=null;
158                                if((container!=null)&&(mob.isMine(container)))
159                                        getThis=R.fetchFromMOBRoomFavorsItems(mob,container,whatToGet+addendumStr,Wearable.FILTER_UNWORNONLY);
160                                else
161                                {
162                                        if(!allFlag)
163                                                getThis=CMLib.english().possibleRoomGold(mob,R,container,whatToGet);
164                                        if(getThis==null)
165                                                getThis=R.fetchFromRoomFavorItems(container,whatToGet+addendumStr);
166                                }
167                                if(getThis==null)
168                                        break;
169
170                                if((maxToGet>1)&&(getThis instanceof RawMaterial)&&(container!=null)
171                                &&(((RawMaterial)getThis).container()==container))
172                                {
173                                        final int weight=((RawMaterial)getThis).phyStats().weight();
174                                        if((weight>1) &&(weight>=maxToGet) &&(CMStrings.containsWordIgnoreCase(((RawMaterial)getThis).name(), "bundle")))
175                                        {
176                                                if(weight>maxToGet)
177                                                        getThis=CMLib.materials().splitBundle((RawMaterial)getThis, maxToGet,container);
178                                                maxToGet=1;
179                                        }
180                                }
181
182                                if((getThis instanceof Item)
183                                &&((CMLib.flags().canBeSeenBy(getThis,mob)||(getThis instanceof Light)))
184                                &&((!allFlag)||CMLib.flags().isGettable(((Item)getThis))||(getThis.displayText().length()>0))
185                                &&(!V.contains(getThis)))
186                                        V.add((Item)getThis);
187
188                                addendumStr="."+(++addendum);
189                        }
190
191                        final boolean optimize = V.size()>1 || allFlag;
192                        for(int i=0;i<V.size();i++)
193                        {
194                                final Item getThis=V.get(i);
195                                if(!getThis.amDestroyed())
196                                {
197                                        get(mob,container,getThis,quiet,"get",optimize);
198                                        if(getThis instanceof Coins)
199                                                ((Coins)getThis).putCoinsBack();
200                                        if(getThis instanceof RawMaterial)
201                                                ((RawMaterial)getThis).rebundle();
202                                        doneSomething=true;
203                                }
204                        }
205                        R.recoverRoomStats();
206                        if(!optimize)
207                                R.recoverRoomStats();
208
209                        if(containers.size()==0)
210                                break;
211                }
212                if(!doneSomething)
213                {
214                        if(containers.size()>0)
215                        {
216                                final Container container=containers.get(0);
217                                if(container.isOpen())
218                                        CMLib.commands().postCommandFail(mob,container,null,commands,L("You don't see '@x1' in <T-NAME>.",unmodifiedWhatToGet));
219                                else
220                                        CMLib.commands().postCommandFail(mob,origCmds,L("@x1 is closed.",container.name()));
221                        }
222                        else
223                        if(containerName.equalsIgnoreCase("all"))
224                                CMLib.commands().postCommandFail(mob,origCmds,L("You don't see anything here."));
225                        else
226                        {
227                                final java.util.List<Container> V=CMLib.english().possibleContainers(mob,containerCommands,Wearable.FILTER_ANY,false);
228                                if(V.size()==0)
229                                        CMLib.commands().postCommandFail(mob,origCmds,L("You don't see '@x1' here.",containerName));
230                                else
231                                if(V.size()==1)
232                                        CMLib.commands().postCommandFail(mob,V.get(0),null,commands,L("You don't see '@x1' in <T-NAME> here.",unmodifiedWhatToGet));
233                                else
234                                        CMLib.commands().postCommandFail(mob,origCmds,L("You don't see '@x1' in any '@x2'.",unmodifiedWhatToGet,containerName));
235                        }
236                }
237                return false;
238        }

258        @Override
259        public Object executeInternal(MOB mob, int metaFlags, Object... args) throws java.io.IOException
260        {
261                if(!super.checkArguments(internalParameters, args))
262                        return Boolean.FALSE;
263                
264                if(args[0] instanceof Item)
265                {
266                        final Item item=(Item)args[0];
267                        Item container=null;
268                        boolean quiet=false;
269                        for(int i=1;i<args.length;i++)
270                        {
271                                if(args[i] instanceof Container)
272                                        container=(Item)args[1];
273                                else
274                                if(args[i] instanceof Boolean)
275                                        quiet=((Boolean)args[i]).booleanValue();
276                        }
277                        final boolean success=get(mob,container,item,quiet);
278                        if(item instanceof Coins)
279                                ((Coins)item).putCoinsBack();
280                        if(item instanceof RawMaterial)
281                                ((RawMaterial)item).rebundle();
282                        return Boolean.valueOf(success);
283                }
284                return Boolean.FALSE;
285        }
