193        public StringBuilder roomDetails(final Session viewerS, final Enumeration<Room> these, final Room likeRoom, final String rest)
194        {
195                final StringBuilder lines=new StringBuilder("");
196                if(!these.hasMoreElements())
197                        return lines;
198                Room R=null;
199                String thisOne=null;
200                final int COL_LEN1=CMLib.lister().fixColWidth(31.0,viewerS);
201                final int COL_LEN2=CMLib.lister().fixColWidth(43.0,viewerS);
202                final  boolean chkRest = (rest != null) && (rest.length()>0);
203                for(final Enumeration<Room> r=these;r.hasMoreElements();)
204                {
205                        R=r.nextElement();
206                        thisOne=R.roomID();
207                        if((thisOne.length()>0)
208                        &&((likeRoom==null)||(R.getArea().Name().equals(likeRoom.getArea().Name())))
209                        &&((!chkRest)
210                                ||(CMLib.english().containsString(R.displayText(), rest))
211                                ||(R.roomID().indexOf(rest)>=0)))
212                                lines.append(CMStrings.padRightPreserve("^N^<LSTROOMID^>"+thisOne+"^</LSTROOMID^>",COL_LEN1)+": "+CMStrings.limit(R.displayText(),COL_LEN2)+"^N^.\n\r");
213                }
214                lines.append("\n\r");
215                return lines;
216        }

246        {

251                        return roomPropertyDetails(viewerS, A.getMetroMap(), null);

254                        return roomPropertyDetails(viewerS, CMLib.map().rooms(), null);
255                else
256                if(rest.trim().toLowerCase().startsWith("area "))
257                        return roomPropertyDetails(viewerS, A.getMetroMap(), rest.trim().substring(5).trim());
258                else
259                if(rest.trim().toLowerCase().startsWith("world "))
260                        return roomPropertyDetails(viewerS, CMLib.map().rooms(), rest.trim().substring(6).trim());
261                else
262                        return new StringBuilder("Illegal parameters... try LIST REALESTATE AREA/WORLD (USERNAME/CLANNAME)");
263        }

309        private String getShopPrice(final Room R, final MOB buyer, final ShopKeeper SK, final Environmental E)
310        {
311                MOB seller = null;
312                boolean destroySeller = false;
313                try
314                {
315                        if(SK instanceof MOB)
316                                seller=(MOB)SK;
317                        else
318                        if((SK instanceof Item)&&(((Item)SK).owner() instanceof MOB))
319                                seller=(MOB)((Item)SK).owner();
320                        else
321                        if((SK instanceof Ability)&&(((Ability)SK).affecting() instanceof MOB))
322                                seller=(MOB)((Ability)SK).affecting();
323                        else
324                        {
325                                seller=CMClass.getFactoryMOB("the shop", 1, R);
326                                destroySeller=true;
327                        }
328                        final ShopPrice p = CMLib.coffeeShops().sellingPrice(seller, buyer, E, SK, SK.getShop(), false);
329                        if(p!=null)
330                        {
331                                if(p.experiencePrice>0)
332                                        return p.experiencePrice+"xp";
333                                else
334                                if(p.questPointPrice>0)
335                                        return p.questPointPrice+"qp";
336                                else
337                                if(p.absoluteGoldPrice>0)
338                                        return CMLib.beanCounter().abbreviatedPrice(seller, p.absoluteGoldPrice);
339                        }
340                }
341                finally
342                {
343                        if(destroySeller && (seller != null))
344                                seller.destroy();
345                }
346                return "?";
347        }

820        {
821                final int ac = tGroup.activeCount();
