036public class ClanApply extends StdCommand
037{
038        public ClanApply()
039        {
040        }
041
042        private final String[]  access  = I(new String[] { "CLANAPPLY" });
043
044        @Override
045        public String[] getAccessWords()
046        {
047                return access;
048        }
049
050        @Override
051        public boolean execute(MOB mob, List<String> commands, int metaFlags)
052                throws java.io.IOException
053        {
054                commands.set(0,getAccessWords()[0]);
055                final String clanName=CMParms.combine(commands,1);
056                if(mob.isMonster())
057                        return false;
058                final StringBuffer msg=new StringBuffer("");
059                if(clanName.length()>0)
060                {
061                        final Clan C=CMLib.clans().findClan(clanName);
062                        if(C!=null)
063                        {
064                                if(C.isOnlyFamilyApplicants()
065                                &&(!CMLib.clans().isFamilyOfMembership(mob,C.getMemberList())))
066                                {
067                                        mob.tell(L("The clan  @x1 is a family.  You can not join a family, you must be born or married into it.",C.clanID()));
068                                        return false;
069                                }
070
071                                final List<Pair<Clan,Integer>> oldList=CMLib.clans().getClansByCategory(mob, C.getCategory());
072                                if(oldList.size()>=CMProps.getMaxClansThisCategory(C.getCategory()))
073                                {
074                                        if(oldList.size()>0)
075                                        {
076                                                final Pair<Clan,Integer> p=oldList.get(0);
077                                                mob.tell(L("You are already a member of @x1. You need to resign before you can apply to another.",p.first.getName()));
078                                        }
079                                        else
080                                                mob.tell(L("You are not elligible to apply to this clan."));
081                                        return false;
082                                }
083
084                                if(!CMLib.masking().maskCheck(C.getBasicRequirementMask(), mob, true))
085                                {
086                                        mob.tell(L("You are not of the right qualities to join @x1. Use CLANDETAILS \"@x2\" for more information.",C.clanID(),C.clanID()));
087                                        return false;
088                                }
089
090                                final CharClass CC = CMClass.getCharClass(C.getClanClass());
091                                if((CC!=null) && (mob.charStats().getClassLevel(CC)<0) && (!CC.qualifiesForThisClass(mob, false)))
092                                        return false;
093
094                                if(CMLib.masking().maskCheck(C.getAcceptanceSettings(),mob,true))
095                                {
096                                        final List<Clan.MemberRecord> members=C.getMemberList();
097                                        if((CMLib.masking().maskCheck("-<"+CMProps.getIntVar(CMProps.Int.MINCLANLEVEL),mob,true))
098                                        ||(CMLib.clans().isFamilyOfMembership(mob,members)))
099                                        {
100                                                final int maxMembers=CMProps.getIntVar(CMProps.Int.MAXCLANMEMBERS);
101                                                final int numMembers=members.size();
102                                                if((maxMembers<=0)||(numMembers<maxMembers))
103                                                {
104                                                        final int role=C.getAutoPosition();
105                                                        C.addMember(mob,role);
106                                                        final Pair<Clan,Integer> newRole=mob.getClanRole(C.clanID());
107                                                        if((newRole.second.intValue()!=C.getGovernment().getAcceptPos())
108                                                        &&(newRole.second.intValue()==C.getGovernment().getAutoRole()))
109                                                        {
110                                                                CMLib.clans().clanAnnounce(mob,L("The @x1 @x2 has a new Applicant: @x3",C.getGovernmentName(),C.clanID(),mob.Name()));
111                                                                mob.tell(L("You have successfully applied for membership in clan @x1.  Your application will be reviewed by management.  Use SCORE to check for a change in status.",C.clanID()));
112                                                        }
113                                                        else
114                                                        {
115                                                                if(C.getGovernment().getEntryScript().trim().length()>0)
116                                                                {
117                                                                        final ScriptingEngine S=(ScriptingEngine)CMClass.getCommon("DefaultScriptingEngine");
118                                                                        S.setSavable(false);
119                                                                        S.setVarScope("*");
120                                                                        S.setScript(C.getGovernment().getEntryScript());
121                                                                        final CMMsg msg2=CMClass.getMsg(mob,mob,null,CMMsg.MSG_OK_VISUAL,null,null,L("CLANENTRY"));
122                                                                        S.executeMsg(mob, msg2);
123                                                                        S.dequeResponses();
124                                                                        S.tick(mob,Tickable.TICKID_MOB);
125                                                                }
126                                                                CMLib.clans().clanAnnounce(mob,L("The @x1 @x2 has a new member: @x3",C.getGovernmentName(),C.clanID(),mob.Name()));
127                                                                mob.tell(L("You have successfully joined @x1.  Use CLANDETAILS for information.",C.clanID()));
128                                                        }
129                                                }
130                                                else
131                                                {
132                                                        msg.append(L("This @x1 already has the maximum number of members (@x2/@x3) and can not accept new applicants.",C.getGovernmentName(),""+numMembers,""+maxMembers));
133                                                }
134                                        }
135                                        else
136                                        {
137                                                msg.append(L("You must be at least level @x1 to join a clan.",""+CMProps.getIntVar(CMProps.Int.MINCLANLEVEL)));
138                                        }
139                                }
140                                else
141                                {
142                                        msg.append(L("You are not of the right qualities to join @x1. Use CLANDETAILS \"@x2\" for more information.",C.clanID(),C.clanID()));
143                                }
144                        }
145                        else
146                        {
147                                msg.append(L("There is no clan named @x1.",clanName));
148                        }
149                }
150                else
151                {
152                        msg.append(L("You haven't specified which clan you are applying to."));
153                }
154                mob.tell(msg.toString());
155                return false;
156        }
157
158        @Override
159        public boolean canBeOrdered()
160        {
161                return false;
162        }
