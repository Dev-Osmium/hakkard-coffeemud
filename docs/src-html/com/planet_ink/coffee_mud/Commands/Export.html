056        public void reallyExport(MOB mob, Session S, String fileName, String xml)
057        {
058                if(fileName==null)
059                        return;
060                if(mob==null)
061                        return;
062                if(xml==null)
063                        return;
064                if(xml.length()==0)
065                        return;
066
067                if(fileName.equalsIgnoreCase("SCREEN"))
068                {
069                        if(S!=null)
070                                mob.tell(L("Here it is:\n\r\n\r"));
071                        xml=xml.replace('\n',' ');
072                        xml=xml.replace('\r',' ');
073                        if(S!=null)
074                                S.safeRawPrintln(xml+"\n\r\n\r");
075                }
076                else
077                if(fileName.equalsIgnoreCase("EMAIL"))
078                {
079                        if(!CMProps.getBoolVar(CMProps.Bool.EMAILFORWARDING))
080                        {
081                                if(S!=null)
082                                        mob.tell(L("Mail forwarding is not enabled on this mud."));
083                                return;
084                        }
085                        if(CMProps.getVar(CMProps.Str.MAILBOX).length()==0)
086                        {
087                                if(S!=null)
088                                        mob.tell(L("No email box has been defined."));
089                                return;
090                        }
091                        if((mob.playerStats()==null)||(mob.playerStats().getEmail().length()==0))
092                        {
093                                if(S!=null)
094                                        mob.tell(L("No email address has been defined."));
095                                return;
096                        }
097                        xml=xml.replace('\n',' ');
098                        xml=xml.replace('\r',' ');
099                        CMLib.database().DBWriteJournal(
100                                        CMProps.getVar(CMProps.Str.MAILBOX),
101                                        mob.Name(),
102                                        mob.Name(),
103                                        "Exported XML",
104                                        xml);
105                        if(S!=null)
106                                mob.tell(L("XML emailed to @x1",mob.playerStats().getEmail()));
107                }
108                else
109                {
110                        if(S!=null)
111                                mob.tell(L("Writing file..."));
112                        if(fileName.indexOf('.')<0)
113                                fileName=fileName+".cmare";
114                        new CMFile(fileName,mob).saveText(xml);
115                        if(S!=null)
116                                mob.tell(L("File '@x1' written.",fileName));
117                }
118        }

120        @Override
121        public boolean execute(MOB mob, List<String> commands, int metaFlags)
122                throws java.io.IOException
123        {
124                String commandType="";
125                String fileName="";
126                int fileNameCode=-1; // -1=indetermined, 0=screen, 1=file, 2=path, 3=email
127                final Room room=mob.location();
128                final Area area=(room!=null)?room.getArea():null;
129                final Session S=mob.session();
130                if(commands.size()>0)
131                        commands.remove(0);
132                if(commands.size()>0)
133                {
134                        commandType=commands.get(0).toUpperCase();
135                        commands.remove(0);
136                }
137                if(commandType.equalsIgnoreCase("REALESTATE"))
138                {
139                        if((commands.size()==0)||(area==null))
140                        {
141                                if(S!=null)
142                                        mob.tell(L("Export which real estate?  Name a clan or player!"));
143                                return false;
144                        }
145                        else
146                        {
147                                String clanName = commands.remove(0);
148                                List<Room> rooms=new ArrayList<Room>();
149                                for(final Enumeration<Room> r=area.getProperMap();r.hasMoreElements();)
150                                {
151                                        final Room R=CMLib.map().getRoom(r.nextElement());
152                                        if((R!=null)&&(R.roomID().length()>0))
153                                        {
154                                                LandTitle title=CMLib.law().getLandTitle(R);
155                                                if((title != null)&&(title.getOwnerName().equalsIgnoreCase(clanName)))
156                                                        rooms.add(R);
157                                        }
158                                }
159                                if(rooms.size()==0)
160                                {
161                                        if(S!=null)
162                                                mob.tell(L("No rooms in this area match '@x1'",clanName));
163                                        return false;
164                                }
165                                else
166                                {
167                                        for(Room R : rooms)
168                                        {
169                                                R.bringMobHere(mob, false);
170                                                if(R.isInhabitant(mob))
171                                                {
172                                                        List<String> cmds=new XVector<String>(commands);
173                                                        if(cmds.size()==0)
174                                                                cmds.add("EXPORT");
175                                                        else
176                                                                cmds.add(0,"EXPORT");
177                                                        if(cmds.size()==1)
178                                                                cmds.add("ROOM");
179                                                        else
180                                                                cmds.add(1,"ROOM");
181                                                        if(cmds.size()==2)
182                                                                cmds.add(R.roomID().replaceAll("#","_"));
183                                                        execute(mob,cmds,metaFlags);
184                                                }
185                                                else
186                                                if(S!=null)
187                                                        mob.tell(L("Failed room '@x1'",R.roomID()));
188                                        }
189                                        if(room!=null)
190                                                room.bringMobHere(mob, false);
191                                        return true;
192                                }
193                        }
194                }
195                
196                if((!commandType.equalsIgnoreCase("ROOM"))
197                &&(!commandType.equalsIgnoreCase("WORLD"))
198                &&(!commandType.equalsIgnoreCase("CATALOG"))
199                &&(!commandType.equalsIgnoreCase("PLAYER"))
200                &&(!commandType.equalsIgnoreCase("USER"))
201                &&(!(commandType.equalsIgnoreCase("ACCOUNT")&&(CMProps.isUsingAccountSystem())))
202                &&(!commandType.equalsIgnoreCase("AREA")))
203                {
204                        if(S!=null)
205                        {
206                                if(CMProps.isUsingAccountSystem())
207                                        mob.tell(L("Export what?  Room, World, Player, Account, or Area?"));
208                                else
209                                        mob.tell(L("Export what?  Room, World, Player, or Area?"));
210                        }
211                        return false;
212                }
213                if(commandType.equalsIgnoreCase("PLAYER")||commandType.equalsIgnoreCase("ACCOUNT")||commandType.equalsIgnoreCase("USER"))
214                {
215                        if(!CMSecurity.isAllowedEverywhere(mob,CMSecurity.SecFlag.EXPORTPLAYERS))
216                        {
217                                if(S!=null)
218                                        mob.tell(L("You are not allowed to export player data."));
219                                return false;
220                        }
221                }
222                else
223                if(!CMSecurity.isAllowed(mob,room,CMSecurity.SecFlag.EXPORT))
224                {
225                        if(S!=null)
226                                mob.tell(L("You are not allowed to export room, mob, or item data."));
227                        return false;
228                }
229
230                String subType="DATA";
231                if(commands.size()>0)
232                {
233                        final String sub=commands.get(0).toUpperCase().trim();
234                        if((sub.equalsIgnoreCase("ITEMS")
235                                ||sub.equalsIgnoreCase("MOBS")
236                                ||sub.equalsIgnoreCase("WEAPONS")
237                                ||sub.equalsIgnoreCase("ARMOR"))
238                        &&(!commandType.equalsIgnoreCase("PLAYER"))
239                        &&(!commandType.equalsIgnoreCase("USER"))
240                        &&(!commandType.equalsIgnoreCase("CATALOG"))
241                        &&(!commandType.equalsIgnoreCase("ACCOUNT")))
242                        {
243                                subType=sub;
244                                commands.remove(0);
245                        }
246                        else
247                        if(sub.equalsIgnoreCase("data"))
248                                commands.remove(0);
249                        else
250                        if((commandType.equalsIgnoreCase("PLAYER")||commandType.equalsIgnoreCase("USER"))
251                        &&(CMLib.players().getLoadPlayer(sub)!=null))
252                        {
253                                subType=sub;
254                                commands.remove(0);
255                        }
256                        else
257                        if((commandType.equalsIgnoreCase("ACCOUNT"))
258                        &&(CMLib.players().getLoadAccount(sub)!=null))
259                        {
260                                subType=sub;
261                                commands.remove(0);
262                        }
263
264                        if(commands.size()==0)
265                        {
266                                if(S!=null)
267                                        mob.tell(L("You must specify a file name to create, or enter 'SCREEN' to have a screen dump, or 'EMAIL' to send to your email address."));
268                                return false;
269                        }
270                        fileName=CMParms.combine(commands,0);
271                        if(fileName.equalsIgnoreCase("screen"))
272                                fileNameCode=0;
273                        else
274                        if(fileName.equalsIgnoreCase("email"))
275                                fileNameCode=3;
276                        else
277                        if(fileName.equalsIgnoreCase("memory"))
278                        {
279                                if(!CMSecurity.isAllowedAnywhere(mob,CMSecurity.SecFlag.EXPORTFILE))
280                                {
281                                        if(S!=null)
282                                                mob.tell(L("You are not allowed to export to memory."));
283                                        return false;
284                                }
285                                fileNameCode=4;
286                        }
287                        else
288                        {
289                                if(!CMSecurity.isAllowedAnywhere(mob,CMSecurity.SecFlag.EXPORTFILE))
290                                {
291                                        if(S!=null)
292                                                mob.tell(L("You are not allowed to export to a file."));
293                                        return false;
294                                }
295                                final CMFile F=new CMFile(fileName,mob);
296                                if(F.isDirectory())
297                                        fileNameCode=2;
298                        }
299                        if(fileNameCode<0)
300                                fileNameCode=1;
301                }
302                else
303                {
304                        if(S!=null)
305                                mob.tell(L("You must specify a file name to create, or enter 'SCREEN' to have a screen dump or 'EMAIL' to send to an email address."));
306                        return false;
307                }
308                executeInternal(mob,metaFlags,commandType,subType,fileName,Integer.valueOf(fileNameCode),S,area,room);
309                return true;
310        }

312        protected CMObjectType getItemSubType(String subType, int fileNameCode, String[] fileName)
313        {
314                CMObjectType type=CMObjectType.ITEM;
315                if(subType.equalsIgnoreCase("WEAPONS"))
316                {
317                        if(fileNameCode==2)
318                                fileName[0]=fileName[0]+"/weapons";
319                        type=CMObjectType.WEAPON;
320                }
321                else
322                if(subType.equalsIgnoreCase("ARMOR"))
323                {
324                        if(fileNameCode==2)
325                                fileName[0]=fileName[0]+"/armor";
326                        type=CMObjectType.ARMOR;
327                }
328                else
329                if(subType.equalsIgnoreCase("TECH"))
330                {
331                        if(fileNameCode==2)
332                                fileName[0]=fileName[0]+"/tech";
333                        type=CMObjectType.TECH;
334                }
335                else
336                if(subType.equalsIgnoreCase("MISCMAGIC"))
337                {
338                        if(fileNameCode==2)
339                                fileName[0]=fileName[0]+"/miscmagic";
340                        type=CMObjectType.MISCMAGIC;
341                }
342                else
343                if(subType.equalsIgnoreCase("CLANITEM"))
344                {
345                        if(fileNameCode==2)
346                                fileName[0]=fileName[0]+"/clanitem";
347                        type=CMObjectType.CLANITEM;
348                }
349                else
350                if(subType.equalsIgnoreCase("COMPTECH"))
351                {
352                        if(fileNameCode==2)
353                                fileName[0]=fileName[0]+"/comptech";
354                        type=CMObjectType.COMPTECH;
355                }
356                else
357                if(subType.equalsIgnoreCase("SOFTWARE"))
358                {
359                        if(fileNameCode==2)
360                                fileName[0]=fileName[0]+"/software";
361                        type=CMObjectType.SOFTWARE;
362                }
363                else
364                if(fileNameCode==2)
365                {
366                        fileName[0]=fileName[0]+"/items";
367                }
368                return type;
369        }

371        protected String getCatalogData(Map<String,?> found)
372        {
373                StringBuilder str = new StringBuilder("<CATADATAS>");
374                for(String key : found.keySet())
375                {
376                        List<?> foundMs = (List<?>)found.get(key);
377                        for(Object P : foundMs)
378                        {
379                                CatalogLibrary.CataData data = CMLib.catalog().getCatalogData((Physical)P);
380                                str.append(data.data(((Physical)P).name()));
381                        }
382                }
383                str.append("</CATADATAS>");
384                return str.toString();
385        }

387        /**
388         * @see com.planet_ink.coffee_mud.Commands.interfaces.Command#executeInternal(MOB, int, Object...)
389         * args[0] = commandType: AREA, PLAYER, ROOM
390         * args[1] = subType: DATA, PLAYER, MOBS, ITEMS, WEAPONS, ARMOR
391         * args[2] = fileName: MEMORY, SCREEN, EMAIL
392         * args[3] = fileNameType (0=screen, 1=disk, 2=directory, 3=email, 4=memory)
393         * args[4] = Session
394         * args[5] = area
395         * args[6] = room
396         * @return xml document, with filenameType=4, or null
397         */
398        @Override
399        public Object executeInternal(MOB mob, int metaFlags, Object... args) throws java.io.IOException
400        {
401                if(!super.checkArguments(internalParameters, args))
402                        return null;
403
404                final String commandType=(String)args[0];
405                final String subType=(String)args[1];
406                String fileName=(String)args[2];
407                final int fileNameCode = ((Integer)args[3]).intValue();
408                final Session S = (Session)args[4];
409                final Area area = (Area)args[5];
410                final Room room = (Room)args[6];
411
412                final Set<CMObject> custom=new HashSet<CMObject>();
413                final Set<String> files=new HashSet<String>();
414
415                String xml="";
416                if(subType.equalsIgnoreCase("DATA"))
417                {
418                        if(commandType.equalsIgnoreCase("PLAYER")||commandType.equalsIgnoreCase("USER"))
419                        {
420                                final StringBuffer x=new StringBuffer("<PLAYERS>");
421                                if(S!=null)
422                                        S.rawPrint(L("Reading players..."));
423                                final java.util.List<String> V=CMLib.database().getUserList();
424                                for(final String name : V)
425                                {
426                                        //if(S!=null) S.rawPrint(".");
427                                        final MOB M=CMLib.players().getLoadPlayer(name);
428                                        if(M!=null)
429                                        {
430                                                x.append("\r\n<PLAYER>");
431                                                x.append(CMLib.coffeeMaker().getPlayerXML(M,custom,files));
432                                                x.append("</PLAYER>");
433                                        }
434                                }
435                                if(fileNameCode==2)
436                                        fileName=fileName+"/player";
437                                xml=x.toString()+"</PLAYERS>";
438                                if(S!=null)
439                                        S.rawPrintln("!");
440                        }
441                        else
442                        if(commandType.equalsIgnoreCase("ACCOUNT"))
443                        {
444                                final StringBuffer x=new StringBuffer("<ACCOUNTS>");
445                                if(S!=null)
446                                        S.rawPrint(L("Reading accounts and players..."));
447                                for(final Enumeration<PlayerAccount> a=CMLib.players().accounts("",null);a.hasMoreElements();)
448                                {
449                                        final PlayerAccount A=a.nextElement();
450                                        //if(S!=null) S.rawPrint(".");
451                                        x.append("\r\n<ACCOUNT>");
452                                        x.append(CMLib.coffeeMaker().getAccountXML(A,custom,files));
453                                        x.append("</ACCOUNT>");
454                                }
455                                if(fileNameCode==2)
456                                        fileName=fileName+"/account";
457                                xml=x.toString()+"</ACCOUNTS>";
458                                if(S!=null)
459                                        S.rawPrintln("!");
460                        }
461                        else
462                        if(commandType.equalsIgnoreCase("CATALOG"))
463                        {
464                                final StringBuffer x=new StringBuffer("<CATALOG HASBOTH>");
465                                if(S!=null)
466                                        S.rawPrint(L("Reading catalog items and mobs..."));
467                                x.append("\r\n<MOBS>");
468                                Map<String,List<MOB>> foundMobs=new TreeMap<String,List<MOB>>();
469                                x.append(CMLib.coffeeMaker().getMobsXML(Arrays.asList(CMLib.catalog().getCatalogMobs()), custom, files, foundMobs));
470                                x.append("\r\n</MOBS>\r\n");
471                                x.append("\r\n").append(getCatalogData(foundMobs));
472                                foundMobs.clear();
473                                x.append("\r\n<ITEMS>");
474                                Map<String,List<Item>> foundItems=new TreeMap<String,List<Item>>();
475                                x.append(CMLib.coffeeMaker().getItemsXML(Arrays.asList(CMLib.catalog().getCatalogItems()), foundItems, files, null));
476                                x.append("\r\n</ITEMS>\r\n");
477                                x.append("\r\n").append(getCatalogData(foundItems));
478                                foundItems.clear();
479                                if(fileNameCode==2)
480                                        fileName=fileName+"/catalog";
481                                if(S!=null)
482                                        S.rawPrintln("!");
483                                xml=x.toString()+"</CATALOG>";
484                        }
485                        else
486                        if(commandType.equalsIgnoreCase("ROOM"))
487                        {
488                                xml=CMLib.coffeeMaker().getRoomXML(room,custom,files,true).toString();
489                                if(fileNameCode==2)
490                                        fileName=fileName+"/room";
491                        }
492                        else
493                        if(commandType.equalsIgnoreCase("AREA"))
494                        {
495                                if(S!=null)
496                                        S.rawPrint(L("Reading area '@x1'...",area.Name()));
497                                xml=CMLib.coffeeMaker().getAreaXML(area,S,custom,files,true).toString();
498                                if(fileNameCode==2)
499                                {
500                                        if(area.getArchivePath().length()>0)
501                                                fileName=fileName+"/"+area.getArchivePath();
502                                        else
503                                                fileName=fileName+"/"+area.Name();
504                                }
505                                if(S!=null)
506                                        S.rawPrintln("!");
507                        }
508                        else
509                        {
510                                if(!CMSecurity.isAllowedEverywhere(mob,CMSecurity.SecFlag.EXPORT))
511                                {
512                                        if(S!=null)
513                                                mob.tell(L("You are not allowed to export world data."));
514                                        return Boolean.FALSE;
515                                }
516                                StringBuffer buf=new StringBuffer("");
517                                if(fileNameCode!=2)
518                                        buf.append("<AREAS>");
519                                for(final Enumeration<Area> a=CMLib.map().areas();a.hasMoreElements();)
520                                {
521                                        final Area A=a.nextElement();
522                                        if(A!=null)
523                                        {
524                                                if(S!=null)
525                                                        S.rawPrint(L("Reading area '@x1'...",A.name()));
526                                                buf.append(CMLib.coffeeMaker().getAreaXML(A,S,custom,files,true).toString());
527                                                if(S!=null)
528                                                        S.rawPrintln("!");
529                                                if(fileNameCode==2)
530                                                {
531                                                        String name=fileName;
532                                                        if(A.getArchivePath().length()>0)
533                                                                name=fileName+"/"+A.getArchivePath();
534                                                        else
535                                                                name=fileName+"/"+A.Name();
536                                                        reallyExport(mob,S,name,buf.toString());
537                                                        buf=new StringBuffer("");
538                                                }
539                                        }
540                                }
541                                if(fileNameCode!=2)
542                                        xml=buf.toString()+"</AREAS>";
543                        }
544                }
545                else
546                if(commandType.equalsIgnoreCase("PLAYERS")
547                ||commandType.equalsIgnoreCase("USERS")
548                ||commandType.equalsIgnoreCase("PLAYER")
549                ||commandType.equalsIgnoreCase("USER"))
550                {
551                        final StringBuffer x=new StringBuffer("<PLAYERS>");
552                        final MOB M=CMLib.players().getLoadPlayer(subType);
553                        if(M!=null)
554                        {
555                                x.append("\r\n<PLAYER>");
556                                x.append(CMLib.coffeeMaker().getPlayerXML(M,custom,files));
557                                x.append("</PLAYER>");
558                        }
559                        if(fileNameCode==2)
560                                fileName=fileName+"/player";
561                        xml=x.toString()+"</PLAYERS>";
562                }
563                else
564                if(commandType.equalsIgnoreCase("ACCOUNT"))
565                {
566                        final StringBuffer x=new StringBuffer("<ACCOUNTS>");
567                        final PlayerAccount A=CMLib.players().getLoadAccount(subType);
568                        if(A!=null)
569                        {
570                                x.append("\r\n<ACCOUNT>");
571                                x.append(CMLib.coffeeMaker().getAccountXML(A,custom,files));
572                                x.append("</ACCOUNT>");
573                        }
574                        if(fileNameCode==2)
575                                fileName=fileName+"/player";
576                        xml=x.toString()+"</ACCOUNTS>";
577                }
578                else
579                if(commandType.equalsIgnoreCase("CATALOG"))
580                {
581                        String[] fileNameAdj = new String[]{fileName};
582                        CMObjectType type = getItemSubType(subType, fileNameCode,fileNameAdj);
583                        fileName = fileNameAdj[0];
584                        
585                        final StringBuffer x=new StringBuffer("<CATALOG");
586                        if(((type!=null)&&(type != CMObjectType.ITEM))||(subType.equalsIgnoreCase("ITEMS")))
587                                x.append(" HASITEMS>");
588                        else
589                        if(subType.equalsIgnoreCase("MOBS"))
590                                x.append(" HASMOBS>");
591                        else
592                                x.append(" HASBOTH>");
593                        if((((type==null)||(type == CMObjectType.ITEM)))&&(!subType.equalsIgnoreCase("ITEMS")))
594                        {
595                                x.append("\r\n<MOBS>");
596                                final Map<String,List<MOB>> found = new TreeMap<String,List<MOB>>();
597                                x.append(CMLib.coffeeMaker().getMobsXML(Arrays.asList(CMLib.catalog().getCatalogMobs()), custom, files, found));
598                                x.append("\r\n</MOBS>\r\n");
599                                x.append("\r\n").append(getCatalogData(found));
600                        }
601                        if(!subType.equalsIgnoreCase("MOBS"))
602                        {
603                                Map<String,List<Item>> found=new TreeMap<String,List<Item>>();
604                                x.append("\r\n<ITEMS>");
605                                x.append(CMLib.coffeeMaker().getItemsXML(Arrays.asList(CMLib.catalog().getCatalogItems()), found, files, type));
606                                x.append("\r\n</ITEMS>\r\n");
607                                x.append("\r\n").append(getCatalogData(found));
608                        }
609                        if((fileNameCode==2)&&(((type==null)||(type == CMObjectType.ITEM))))
610                                fileName=fileName+"/catalog";
611                        if(S!=null)
612                                S.rawPrintln("!");
613                        xml=x.toString()+"</CATALOG>";
614                }
615                else
616                if(subType.equalsIgnoreCase("MOBS"))
617                {
618                        if(fileNameCode==2)
619                                fileName=fileName+"/mobs";
620                        final Hashtable<String,List<MOB>> found=new Hashtable<String,List<MOB>>();
621                        if(commandType.equalsIgnoreCase("ROOM"))
622                                xml="<MOBS>"+CMLib.coffeeMaker().getRoomMobs(room,custom,files,found).toString()+"</MOBS>";
623                        else
624                        if(commandType.equalsIgnoreCase("AREA"))
625                        {
626                                if(S!=null)
627                                        S.rawPrint(L("Reading area mobs '@x1'...",area.Name()));
628                                final StringBuffer buf=new StringBuffer("<MOBS>");
629                                for(final Enumeration<Room> r=area.getCompleteMap();r.hasMoreElements();)
630                                {
631                                        final Room R=r.nextElement();
632                                        //if(S!=null) S.rawPrint(".");
633                                        buf.append(CMLib.coffeeMaker().getRoomMobs(R,custom,files,found).toString());
634                                }
635                                xml=buf.toString()+"</MOBS>";
636                                if(S!=null)
637                                        S.rawPrintln("!");
638                        }
639                        else
640                        {
641                                if(S!=null)
642                                        S.rawPrint(L("Reading world mobs ..."));
643                                final StringBuffer buf=new StringBuffer("<MOBS>");
644                                try
645                                {
646                                        for(final Enumeration<Room> r=CMLib.map().rooms();r.hasMoreElements();)
647                                        {
648                                                final Room R=r.nextElement();
649                                                //if(S!=null) S.rawPrint(".");
650                                                buf.append(CMLib.coffeeMaker().getRoomMobs(R,custom,files,found).toString());
651                                        }
652                                }
653                                catch(final NoSuchElementException e)
654                                {
655                                }
656                                xml=buf.toString()+"</MOBS>";
657                                if(S!=null)
658                                        S.rawPrintln("!");
659                        }
660                }
661                else
662                if((subType.equalsIgnoreCase("ITEMS"))
663                ||(subType.equalsIgnoreCase("WEAPONS"))
664                ||(subType.equalsIgnoreCase("ARMOR")))
665                {
666
667                        String[] fileNameAdj = new String[]{fileName};
668                        CMObjectType type = getItemSubType(subType, fileNameCode,fileNameAdj);
669                        fileName = fileNameAdj[0];
670                        
671                        final Hashtable<String,List<Item>> found=new Hashtable<String,List<Item>>();
672                        if(commandType.equalsIgnoreCase("ROOM"))
673                                xml="<ITEMS>"+CMLib.coffeeMaker().getRoomItems(room,found,files,type).toString()+"</ITEMS>";
674                        else
675                        if(commandType.equalsIgnoreCase("AREA"))
676                        {
677                                if(S!=null)
678                                        S.rawPrint(L("Reading area @x1 '@x2'...",subType.toLowerCase(),area.Name()));
679                                final StringBuffer buf=new StringBuffer("<ITEMS>");
680                                for(final Enumeration<Room> r=area.getCompleteMap();r.hasMoreElements();)
681                                {
682                                        final Room R=r.nextElement();
683                                        //if(S!=null) S.rawPrint(".");
684                                        buf.append(CMLib.coffeeMaker().getRoomItems(R,found,files,type).toString());
685                                }
686                                xml=buf.toString()+"</ITEMS>";
687                                if(S!=null)
688                                        S.rawPrintln("!");
689                        }
690                        else
691                        {
692                                if(S!=null)
693                                        S.rawPrint(L("Reading world @x1 ...",subType.toLowerCase()));
694                                final StringBuffer buf=new StringBuffer("<ITEMS>");
695                                try
696                                {
697                                        for(final Enumeration<Room> r=CMLib.map().rooms();r.hasMoreElements();)
698                                        {
699                                                final Room R=r.nextElement();
700                                                //if(S!=null) S.rawPrint(".");
701                                                buf.append(CMLib.coffeeMaker().getRoomItems(R,found,files,type).toString());
702                                        }
703                                }
704                                catch (final NoSuchElementException e)
705                                {
706                                }
707                                xml=buf.toString()+"</ITEMS>";
708                                if(S!=null)
709                                        S.rawPrintln("!");
710                        }
711                }
712                if(custom.size()>0)
713                {
714                        final StringBuffer str=new StringBuffer("<CUSTOM>");
715                        for (final Object o : custom)
716                        {
717                                if(o instanceof Race)
718                                        str.append(((Race)o).racialParms());
719                                else
720                                if(o instanceof CharClass)
721                                        str.append(((CharClass)o).classParms());
722                                else
723                                if(o instanceof Ability)
724                                        str.append(CMLib.coffeeMaker().getGenAbilityXML((Ability)o));
725                                else
726                                if(o instanceof Manufacturer)
727                                        str.append("<MANUFACTURER>").append(((Manufacturer)o).getXml()).append("</MANUFACTURER>");
728                        }
729                        str.append("</CUSTOM>");
730                        xml+=str.toString();
731                }
732                if(files.size()>0)
733                {
734                        final StringBuffer str=new StringBuffer("<FILES>");
735                        for (final Object O : files)
736                        {
737                                final String filename=(String)O;
738                                final StringBuffer buf=new CMFile(Resources.makeFileResourceName(filename),null,CMFile.FLAG_LOGERRORS).text();
739                                if((buf!=null)&&(buf.length()>0))
740                                {
741                                        str.append("<FILE NAME=\""+filename+"\">");
742                                        str.append(buf);
743                                        str.append("</FILE>");
744                                }
745                        }
746                        str.append("</FILES>");
747                        xml+=str.toString();
748                }
749                if(fileNameCode==2)
750                        fileName=fileName+"/extras";
751                if(fileNameCode==4)
752                        return xml;
753                reallyExport(mob,S,fileName,xml);
754                return null;
755        }
