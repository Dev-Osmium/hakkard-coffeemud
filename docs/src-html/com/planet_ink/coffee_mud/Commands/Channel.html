036public class Channel extends StdCommand
037{
038        public Channel()
039        {
040        }
041        @Override
042        public String[] getAccessWords()
043        {
044                return CMLib.channels().getChannelNames();
045        }
046
047        private final static Class<?>[][] internalParameters=new Class<?>[][]{{Boolean.class,String.class,String.class}};
048
049        @Override
050        public boolean execute(MOB mob, List<String> commands, int metaFlags)
051                throws java.io.IOException
052        {
053                return channel(mob, commands, false);
054        }
055
056        @Override
057        public Object executeInternal(MOB mob, int metaFlags, Object... args) throws java.io.IOException
058        {
059                if(!super.checkArguments(internalParameters, args))
060                        return Boolean.FALSE;
061                final boolean systemMsg=((Boolean)args[0]).booleanValue();
062                final String channelName=(String)args[1];
063                final String message=(String)args[2];
064                CMLib.channels().createAndSendChannelMessage(mob,channelName,message,systemMsg);
065                return Boolean.TRUE;
066        }
067
068        public boolean channel(MOB mob, List<String> commands, boolean systemMsg)
069        {
070                final PlayerStats pstats=mob.playerStats();
071                final String channelName=commands.get(0).toUpperCase().trim();
072                commands.remove(0);
073                final int channelInt=CMLib.channels().getChannelIndex(channelName);
074                final int channelNum=CMLib.channels().getChannelCodeNumber(channelName);
075
076                if((pstats!=null)&&(CMath.isSet(pstats.getChannelMask(),channelInt)))
077                {
078                        pstats.setChannelMask(pstats.getChannelMask()&(pstats.getChannelMask()-channelNum));
079                        mob.tell(L("@x1 has been turned on.  Use `NO@x2` to turn it off again.",channelName,channelName.toUpperCase()));
080                        return false;
081                }
082
083                if(mob.isAttributeSet(MOB.Attrib.QUIET))
084                {
085                        mob.tell(L("You have QUIET mode on.  You must turn it off first."));
086                        return false;
087                }
088
089                final ChannelsLibrary.CMChannel chan=CMLib.channels().getChannel(channelInt);
090                if((chan == null)||(!CMLib.masking().maskCheck(chan.mask(),mob,true)))
091                {
092                        mob.tell(L("This channel is not available to you."));
093                        return false;
094                }
095                
096                if(commands.size()==0)
097                {
098                        int size = CMLib.channels().getChannelQue(channelInt, 0, 5).size();
099                        if(size>0)
100                        {
101                                if(size>5)
102                                        size=5;
103                                mob.tell(L("@x1 what?  Here's the last @x2 message(s):\n\r",channelName,""+size));
104                                commands.add("LAST");
105                                commands.add(Integer.toString(size));
106                        }
107                        else
108                        {
109                                mob.tell(L("@x1 what?",channelName));
110                                return false;
111                        }
112                }
113
114                for(int i=0;i<commands.size();i++)
115                {
116                        final String s=commands.get(i);
117                        if(s.indexOf(' ')>=0)
118                                commands.set(i,"\""+s+"\"");
119                }
120
121                final Set<ChannelsLibrary.ChannelFlag> flags=chan.flags();
122                if((flags.contains(ChannelsLibrary.ChannelFlag.CLANONLY)||flags.contains(ChannelsLibrary.ChannelFlag.CLANALLYONLY)))
123                {
124                        if(!CMLib.clans().checkClanPrivilege(mob, Clan.Function.CHANNEL))
125                        {
126                                mob.tell(L("You can't talk to your clan - you don't have one that allows you."));
127                                return false;
128                        }
129                }
130
131                if((commands.size()==2)
132                &&(mob.session()!=null)
133                &&(commands.get(0).equalsIgnoreCase("last"))
134                &&(CMath.isNumber(commands.get(commands.size()-1))))
135                {
136                        int num=CMath.s_int(commands.get(commands.size()-1));
137                        final List<ChannelsLibrary.ChannelMsg> que=CMLib.channels().getChannelQue(channelInt, 0, num);
138                        boolean showedAny=false;
139                        if(que.size()>0)
140                        {
141                                if(num>que.size())
142                                        num=que.size();
143                                final boolean areareq=flags.contains(ChannelsLibrary.ChannelFlag.SAMEAREA);
144                                long elapsedTime=0;
145                                final long now=System.currentTimeMillis();
146                                final LinkedList<ChannelsLibrary.ChannelMsg> showThese=new LinkedList<ChannelsLibrary.ChannelMsg>();
147                                for (final ChannelMsg channelMsg : que)
148                                {
149                                        showThese.add(channelMsg);
150                                        if(showThese.size()>num)
151                                                showThese.removeFirst();
152                                }
153                                for(final ChannelsLibrary.ChannelMsg msg : showThese)
154                                {
155                                        final CMMsg modMsg = (CMMsg)msg.msg().copyOf();
156                                        elapsedTime=now-msg.sentTimeMillis();
157                                        elapsedTime=Math.round(elapsedTime/1000L)*1000L;
158                                        if(elapsedTime<0)
159                                        {
160                                                Log.errOut("Channel","Wierd elapsed time: now="+now+", then="+msg.sentTimeMillis());
161                                                elapsedTime=0;
162                                        }
163
164                                        final String timeAgo = "^.^N ("+CMLib.time().date2SmartEllapsedTime(elapsedTime,false)+" ago)";
165                                        if((modMsg.sourceMessage()!=null)&&(modMsg.sourceMessage().length()>0))
166                                                modMsg.setSourceMessage(modMsg.sourceMessage()+timeAgo);
167                                        if((modMsg.targetMessage()!=null)&&(modMsg.targetMessage().length()>0))
168                                                modMsg.setTargetMessage(modMsg.targetMessage()+timeAgo);
169                                        if((modMsg.othersMessage()!=null)&&(modMsg.othersMessage().length()>0))
170                                                modMsg.setOthersMessage(modMsg.othersMessage()+timeAgo);
171                                        if(CMath.bset(modMsg.sourceCode(),CMMsg.MASK_CHANNEL))
172                                                modMsg.setSourceCode(CMMsg.MASK_CHANNEL|(CMMsg.TYP_CHANNEL+channelInt));
173                                        if(CMath.bset(modMsg.targetCode(),CMMsg.MASK_CHANNEL))
174                                                modMsg.setTargetCode(CMMsg.MASK_CHANNEL|(CMMsg.TYP_CHANNEL+channelInt));
175                                        if(CMath.bset(modMsg.othersCode(),CMMsg.MASK_CHANNEL))
176                                                modMsg.setOthersCode(CMMsg.MASK_CHANNEL|(CMMsg.TYP_CHANNEL+channelInt));
177                                        showedAny=CMLib.channels().sendChannelCMMsgTo(mob.session(),areareq,channelInt,modMsg,modMsg.source())||showedAny;
178                                }
179                        }
180                        if(!showedAny)
181                        {
182                                mob.tell(L("There are no previous entries on this channel."));
183                                return false;
184                        }
185                }
186                else
187                if(flags.contains(ChannelsLibrary.ChannelFlag.READONLY))
188                {
189                        mob.tell(L("This channel is read-only."));
190                        return false;
191                }
192                else
193                if(flags.contains(ChannelsLibrary.ChannelFlag.PLAYERREADONLY)&&(!mob.isMonster()))
194                {
195                        mob.tell(L("This channel is read-only."));
196                        return false;
197                }
198                else
199                if(flags.contains(ChannelsLibrary.ChannelFlag.ARCHONREADONLY)&&(!CMSecurity.isASysOp(mob)))
200                {
201                        mob.tell(L("This channel is read-only."));
202                        return false;
203                }
204                
205                else
206                        CMLib.channels().createAndSendChannelMessage(mob,channelName,CMParms.combine(commands,0),systemMsg);
207                return false;
208        }

212        {
213                return true;
214        }
