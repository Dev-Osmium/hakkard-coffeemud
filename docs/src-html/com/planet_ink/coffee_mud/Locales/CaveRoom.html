035public class CaveRoom extends StdRoom
036{
037        @Override
038        public String ID()
039        {
040                return "CaveRoom";
041        }
042
043        public CaveRoom()
044        {
045                super();
046                name="the cave";
047                basePhyStats().setDisposition(basePhyStats().disposition()|PhyStats.IS_DARK);
048                basePhyStats.setWeight(2);
049                recoverPhyStats();
050                climask=Places.CLIMASK_NORMAL;
051        }
052
053        @Override
054        public int domainType()
055        {
056                return Room.DOMAIN_INDOORS_CAVE;
057        }
058
059        @Override
060        public int maxRange()
061        {
062                return 5;
063        }
064
065        @Override
066        public void executeMsg(final Environmental myHost, final CMMsg msg)
067        {
068                if((msg.amITarget(this)||(msg.targetMinor()==CMMsg.TYP_ADVANCE)||(msg.targetMinor()==CMMsg.TYP_RETREAT))
069                &&(!msg.source().isMonster())
070                &&(msg.source().curState().getHitPoints()<msg.source().maxState().getHitPoints())
071                &&(CMLib.dice().rollPercentage()==1)
072                &&(CMLib.dice().rollPercentage()==1)
073                &&(isInhabitant(msg.source()))
074                &&(!CMSecurity.isDisabled(CMSecurity.DisFlag.AUTODISEASE)))
075                {
076                        final Ability A=CMClass.getAbility("Disease_Syphilis");
077                        if((A!=null)&&(msg.source().fetchEffect(A.ID())==null)&&(!CMSecurity.isAbilityDisabled(A.ID())))
078                                A.invoke(msg.source(),msg.source(),true,0);
079                }
080                super.executeMsg(myHost,msg);
081        }
