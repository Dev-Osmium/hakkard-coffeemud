039/**
040 * Interface for a Room that virtually contains other rooms, which are 
041 * called grid-child rooms. This is for producing substantially identical 
042 * rooms in grid formations, or mazes, or other groupings where unique 
043 * control of exits and content is not desired.
044 *
045 * Grids have the following general features: They are rectangular with a 
046 * defineable width and length.  All rooms in the grid are of the same
047 * locale type.  Grid children can only have exits to each other, or to
048 * the GridLocale's exiting rooms, unless "outer exits" are defined.
049 * @author Bo Zimmerman
050 */
051public interface GridLocale extends Room, GridZones 
052{
053        /**
054         * Gets the Room ID() for the type of Java room class used to
055         * populate the grid.  For example: StdRoom.
056         * @return the Room ID() for the type of Java room class
057         */
058        public String getGridChildLocaleID();
059
060        /**
061         * Returns the room found in the given direction from the from-room, 
062         * where the stand-in room is the to-room.
063         * 
064         * Grids are not required to actually fill themselves in at boot
065         * time, nor are they required to remain filled in.  Therefore,
066         * rooms can be constructed in real time.
067         * 
068         * This method is called whenever a player wants to move into another
069         * room from a room that is part of a grid to some other room.
070         * @param fromRoom the room moving from, also the grid child
071         * @param toRoom the room ostensibly being moved into
072         * @param direction the direction from the from room moving
073         * @return the new to-room, the actual room to move into
074         */
075        public Room prepareGridLocale(Room fromRoom, Room toRoom, int direction);
076
077        /**
078         * Called whenever structural changes are made to either the grid
079         * room, or one of the adjacent connecting rooms, this method will
080         * "rebuild" the grids internal structure as necessary.
081         */
082        public void buildGrid();
083
084        /**
085         * Empties the grid-children rooms of all contents, mob and item,
086         * and then destroys the internal grid-children rooms themselves.
087         * This puts the grid back into a default state.
088         * @param bringBackHere a room to teleport *ALL* mobs and items to
089         */
090        public void clearGrid(Room bringBackHere);
091
092        /**
093         * Returns a read-only list of grid-child rooms.  You are guaranteed
094         * at least one room, even from grids that create them as-needed.
095         * @return a read-only list of grid-child rooms.
096         */
097        public List<Room> getAllRooms();
098
099        /**
100         * Returns a read-only list of grid-child rooms, and any skys or seas
101         * attached to them.
102         * @return a read-only list of all rooms around here.
103         */
104        public List<Room> getAllRoomsFilled();
105        
106        /**
107         * Returns a read-only iterator over the existing grid-child rooms.
108         * This iterator may be empty if the grid creates children as-needed.
109         * @return a read-only iterator over the existing grid-child rooms.
110         */
111        public Iterator<Room> getExistingRooms();
112
113        /**
114         * Normally the grid-child rooms can only exit to each other, or to
115         * the same places as the gridlocale host.  Outer Exits are a way around
116         * this by pre-defining exits from grid children to elsewhere on the 
117         * map, including into other gridlocales.
118         * @see GridLocale.CrossExit
119         * @see com.planet_ink.coffee_mud.Locales.interfaces.GridLocale#addOuterExit(GridLocale.CrossExit)
120         * @see com.planet_ink.coffee_mud.Locales.interfaces.GridLocale#delOuterExit(GridLocale.CrossExit)
121         * @return an iterator of cross ("outer") exits.
122         */
123        public Iterator<CrossExit> outerExits();
124
125        /**
126         * Normally the grid-child rooms can only exit to each other, or to
127         * the same places as the gridlocale host.  Outer Exits are a way around
128         * this by pre-defining exits from grid children to elsewhere on the 
129         * map, including into other gridlocales.
130         * This method will add a new one.
131         * @see GridLocale.CrossExit
132         * @see GridLocale#outerExits()
133         * @see com.planet_ink.coffee_mud.Locales.interfaces.GridLocale#delOuterExit(GridLocale.CrossExit)
134         * @param x the new cross ("outer") exit
135         */
136        public void addOuterExit(CrossExit x);
137
138        /**
139         * Normally the grid-child rooms can only exit to each other, or to
140         * the same places as the gridlocale host.  Outer Exits are a way around
141         * this by pre-defining exits from grid children to elsewhere on the 
142         * map, including into other gridlocales.
143         * This method will remove an existing one
144         * @see GridLocale.CrossExit
145         * @see GridLocale#outerExits()
146         * @see com.planet_ink.coffee_mud.Locales.interfaces.GridLocale#addOuterExit(GridLocale.CrossExit)
147         * @param x the existing cross ("outer") exit to delete
148         */
149        public void delOuterExit(CrossExit x);
150        
151        /**
152         * class definition for an exit that goes from inside a grid locale child to a place 
153         * outside the parent gridlocale room
154         * @author Bo Zimmerman
155         */
156        public static class CrossExit
157        {
158                public int x;
159                public int y;
160                public int dir;
161                public String destRoomID="";
162                public boolean out=false;
163        
164                public static CrossExit make(int xx, int xy, int xdir, String xdestRoomID, boolean xout)
165                {
166                        final CrossExit EX = new CrossExit();
167                        EX.x = xx;
168                        EX.y = xy;
169                        EX.dir = xdir;
170                        EX.destRoomID = xdestRoomID;
171                        EX.out = xout;
172                        return EX;
173                }
174        }
175}
