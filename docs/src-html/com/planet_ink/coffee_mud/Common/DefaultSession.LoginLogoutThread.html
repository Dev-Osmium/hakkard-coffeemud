554        private void negotiateTelnetMode(final OutputStream out, final int optionCode)
555        throws IOException
556        {
557                if(CMSecurity.isDebugging(CMSecurity.DbgFlag.TELNET))
558                        Log.debugOut("Sent sub-option: "+Session.TELNET_DESCS[optionCode]);
559                if(optionCode==TELNET_TERMTYPE)
560                {
561                        final byte[] stream={(byte)TELNET_IAC,(byte)TELNET_SB,(byte)optionCode,(byte)1,(byte)TELNET_IAC,(byte)TELNET_SE};
562                        rawBytesOut(out, stream);
563                }
564                else
565                {
566                        final byte[] stream={(byte)TELNET_IAC,(byte)TELNET_SB,(byte)optionCode,(byte)TELNET_IAC,(byte)TELNET_SE};
567                        rawBytesOut(out, stream);
568                }
569                out.flush();
570        }

617        private void changeTelnetMode(final OutputStream out, final int telnetCode, final boolean onOff) throws IOException
618        {
619                final byte[] command;
620                if(telnetCode == TELNET_TERMTYPE)
621                        command=new byte[]{(byte)TELNET_IAC,onOff?(byte)TELNET_DO:(byte)TELNET_DONT,(byte)telnetCode};
622                else
623                        command=new byte[]{(byte)TELNET_IAC,onOff?(byte)TELNET_WILL:(byte)TELNET_WONT,(byte)telnetCode};
624                rawBytesOut(out, command);
625                out.flush();
626                if(CMSecurity.isDebugging(CMSecurity.DbgFlag.TELNET))
627                        Log.debugOut("Sent: "+(onOff?"Will":"Won't")+" "+Session.TELNET_DESCS[telnetCode]);
628                setServerTelnetMode(telnetCode,onOff);
629        }

1673        public void handleEscape() throws IOException, InterruptedIOException
1674        {
1675                if((in==null)||(out==null))
1676                        return;
1677                int c=readByte();
1678                if((char)c!='[')
1679                        return;
1680
1681                boolean quote=false;
1682                final StringBuffer escapeStr=new StringBuffer("");
1683                while(((c=readByte())!=-1)
1684                &&(!killFlag)
1685                &&((quote)||(!Character.isLetter((char)c))))
1686                {
1687                        escapeStr.append((char)c);
1688                        if(c=='"')
1689                                quote=!quote;
1690                }
1691                if(c==-1)
1692                        return;
1693                escapeStr.append((char)c);
1694                String esc=escapeStr.toString().trim();
1695                // at the moment, we only handle mxp escapes
1696                // everything else is effectively EATEN
1697                if(!esc.endsWith("z"))
1698                        return;
1699                esc=esc.substring(0,esc.length()-1);
1700                if(!CMath.isNumber(esc))
1701                        return;
1702                final int escNum=CMath.s_int(esc);
1703                // only LINE-based mxp escape sequences are respected
1704                if(escNum>3)
1705                        return;
1706                sock.setSoTimeout(30000);
1707                final StringBuffer line=new StringBuffer("");
1708                while(((c=readByte())!=-1)&&(!killFlag))
1709                {
1710                        if(c=='\n')
1711                                break;
1712                        line.append((char)c);
1713                }
1714                sock.setSoTimeout(SOTIMEOUT);
1715                String l=line.toString().toUpperCase().trim();
1716                // now we have the line, so parse out tags -- only tags matter!
1717                while(l.length()>0)
1718                {
1719                        final int tagStart=l.indexOf('<');
1720                        if(tagStart<0)
1721                                return;
1722                        final int tagEnd=l.indexOf('>');
1723                        if(tagEnd<0)
1724                                return;
1725                        String tag=l.substring(tagStart+1,tagEnd).trim();
1726                        l=l.substring(tagEnd+1).trim();
1727                        // now we have a tag, and its parameters (space delimited)
1728                        final List<String> parts=CMParms.parseSpaces(tag,true);
1729                        if(CMSecurity.isDebugging(CMSecurity.DbgFlag.TELNET))
1730                                Log.debugOut("Got secure MXP tag: "+tag);
1731                        if(parts.size()>1)
1732                        {
1733                                tag=parts.get(0);
1734                                if(tag.equals("VERSION"))
1735                                {
1736                                        for(int p=1;p<parts.size();p++)
1737                                        {
1738                                                final String pp=parts.get(p);
1739                                                final int x=pp.indexOf('=');
1740                                                if(x<0)
1741                                                        continue;
1742                                                mxpVersionInfo.remove(pp.substring(0,x).trim());
1743                                                mxpVersionInfo.put(pp.substring(0,x).trim(),pp.substring(x+1).trim());
1744                                        }
1745                                }
1746                                else
1747                                if(tag.equals("SUPPORTS"))
1748                                {
1749                                        for(int p=1;p<parts.size();p++)
1750                                        {
1751                                                final String s=parts.get(p).toUpperCase();
1752                                                final int x=s.indexOf('.');
1753                                                if(s.startsWith("+"))
1754                                                {
1755                                                        if((terminalType == null)
1756                                                        ||(!terminalType.toLowerCase().startsWith("mushclient"))
1757                                                        ||(s.indexOf("IMAGE")<0))
1758                                                        {
1759                                                                mxpSupportSet.add(s);
1760                                                                if(x>0)
1761                                                                        mxpSupportSet.add(s.substring(0, x));
1762                                                        }
1763                                                }
1764                                                else
1765                                                if(s.startsWith("-"))
1766                                                {
1767                                                        mxpSupportSet.add(s);
1768                                                        mxpSupportSet.remove("+"+s.substring(1));
1769                                                }
1770                                        }
1771                                }
1772                                else
1773                                if(tag.equals("SHUTDOWN"))
1774                                {
1775                                        final MOB M=CMLib.players().getLoadPlayer(parts.get(1));
1776                                        if((M!=null)
1777                                        &&(M.playerStats().matchesPassword(parts.get(2)))
1778                                        &&(CMSecurity.isASysOp(M)))
1779                                        {
1780                                                final boolean keepDown=parts.size()>3?CMath.s_bool(parts.get(3)):true;
1781                                                final String externalCmd=(parts.size()>4)?CMParms.combine(parts,4):null;
1782                                                final Vector<String> cmd=new XVector<String>("SHUTDOWN","NOPROMPT");
1783                                                if(!keepDown)
1784                                                {
1785                                                        cmd.add("RESTART");
1786                                                        if((externalCmd!=null)&&(externalCmd.length()>0))
1787                                                                cmd.add(externalCmd);
1788                                                }
1789                                                final Command C=CMClass.getCommand("Shutdown");
1790                                                l="";
1791                                                setKillFlag(true);
1792                                                rawCharsOut(out,"\n\n\033[1z<Executing Shutdown...\n\n".toCharArray());
1793                                                M.setSession(this);
1794                                                if(C!=null)
1795                                                        C.execute(M,cmd,0);
1796                                        }
1797                                }
1798                        }
1799                }
