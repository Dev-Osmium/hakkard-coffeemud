285        @Override
286        public void changeStates(final LegalWarrant W, final int state)
287        {
288                if((W==null)||(W.criminal()==null))
289                        return;
290                if(warrants.contains(W))
291                {
292                        for(int w=0;w<warrants.size();w++)
293                        {
294                                final LegalWarrant W2=warrants.elementAt(w);
295                                if(W2.criminal()==W.criminal())
296                                        W2.setState(state);
297                        }
298                }
299        }

301        @Override
302        public TreasurySet getTreasuryNSafe(final Area A)
303        {
304                Room treasuryR=null;
305                Container container=null;
306                final String tres=(String)taxLaws().get("TREASURY");
307                Item I=null;
308                if((tres!=null)&&(tres.length()>0))
309                {
310                        final List<String> V=CMParms.parseSemicolons(tres,false);
311                        if(V.size()>0)
312                        {
313                                Room R=null;
314                                final String room=V.get(0);
315                                String item="";
316                                if(V.size()>1)
317                                        item=CMParms.combine(V,1);
318                                if(!room.equalsIgnoreCase("*"))
319                                {
320                                        treasuryR=CMLib.map().getRoom(room);
321                                        if(treasuryR!=null)
322                                        {
323                                                I=treasuryR.findItem(item);
324                                                if(I instanceof Container)
325                                                container=(Container)I;
326                                        }
327                                }
328                                else
329                                if(item.length()>0)
330                                for(final Enumeration<Room> e=A.getMetroMap();e.hasMoreElements();)
331                                {
332                                        R=e.nextElement();
333                                        I=R.findItem(item);
334                                        if(I instanceof Container)
335                                        {
336                                                container=(Container)I;
337                                                treasuryR=R;
338                                                break;
339                                        }
340                                }
341                                if((room.length()>0)&&(treasuryR==null))
342                                        treasuryR=A.getRandomMetroRoom();
343                        }
344                }
345                return new Law.TreasurySet(treasuryR,container);
346        }

354        protected boolean notifyPlayer(final String ownerName, String owerName, final double owed, final String fourWord, final String subject, final String message)
355        {
356                MOB M=CMLib.players().getPlayerAllHosts(ownerName);
357                if((M!=null)&&(CMLib.flags().isInTheGame(M, true)))
358                {
359                        final String amountOwed = CMLib.beanCounter().nameCurrencyLong(M, owed);
360                        if(owerName.length()==0)
361                                owerName=M.Name();
362                        M.tell(CMLib.lang().L(message,owerName,amountOwed,CMProps.getVar(CMProps.Str.MUDNAME),fourWord));
363                }
364                else
365                {
366                        M=CMLib.players().getLoadPlayer(ownerName);
367                        if(M!=null)
368                        {
369                                if(owerName.length()==0)
370                                        owerName=M.Name();
371                                final String amountOwed = CMLib.beanCounter().nameCurrencyLong(M, owed);
372                                final String subj = CMLib.lang().L(subject,owerName,amountOwed,CMProps.getVar(CMProps.Str.MUDNAME),fourWord);
373                                final String msg = CMLib.lang().L(message,owerName,amountOwed,CMProps.getVar(CMProps.Str.MUDNAME),fourWord);
374                                return sendGameMail(M.Name(), subj, msg);
375                        }
376                        return false;
377                }
378                return true;
379        }

381        @Override
382        public void propertyTaxTick(final Area A, final boolean debugging)
383        {
384                if(!Resources.isResource("SYSTEM_TAXES_LASTCHECK"))
385                        Resources.submitResource("SYSTEM_TAXES_LASTCHECK", Integer.valueOf(A.getTimeObj().getMonth()));
386                final int lastMonthChecked = ((Integer)Resources.getResource("SYSTEM_TAXES_LASTCHECK")).intValue();
387                if(lastMonthChecked!=A.getTimeObj().getMonth())
388                {
389                        Resources.submitResource("SYSTEM_TAXES_LASTCHECK", Integer.valueOf(A.getTimeObj().getMonth()));
390                        double tax=CMath.s_double((String)taxLaws.get("PROPERTYTAX"));
391                        if(tax==0.0)
392                                return;
393                        tax=CMath.div(tax,100.0);
394                        List<LandTitle> titles=CMLib.law().getAllUniqueLandTitles(A.getMetroMap(),"*",false);
395                        final Map<String,List<LandTitle>> owners=new Hashtable<String,List<LandTitle>>();
396                        for(final LandTitle T : titles)
397                        {
398                                List<LandTitle> D=owners.get(T.getOwnerName());
399                                if(D==null)
400                                {
401                                        D=new Vector<LandTitle>();
402                                        owners.put(T.getOwnerName(),D);
403                                }
404                                if(T.getPrice()>1)
405                                        D.add(T);
406                        }
407                        titles=null;
408                        final Law.TreasurySet treas=getTreasuryNSafe(A);
409                        final Room treasuryR=treas.room;
410                        final Container container=treas.container;
411                        final String[] evasionBits=(String[])taxLaws().get("TAXEVASION");
412
413                        for(final String owner : owners.keySet())
414                        {
415                                MOB responsibleMob=null;
416                                final Clan C=CMLib.clans().getClan(owner);
417                                if(C!=null)
418                                        responsibleMob=C.getResponsibleMember();
419                                else
420                                        responsibleMob=CMLib.players().getLoadPlayer(owner);
421                                final List<LandTitle> particulars=owners.get(owner);
422
423                                double totalValue=0;
424                                double paid=0;
425                                double owed=0;
426                                final StringBuffer properties=new StringBuffer("");
427                                LandTitle T=null;
428
429                                for(int p=0;p<particulars.size();p++)
430                                {
431                                        if(p>0)
432                                                properties.append(", ");
433                                        T=(particulars.get(p));
434                                        final List<Room> propertyRooms=T.getAllTitledRooms();
435                                        if((propertyRooms.size()<2)
436                                        ||(CMLib.map().getArea(T.landPropertyID())!=null))
437                                                properties.append(T.landPropertyID());
438                                        else
439                                                properties.append("around "+CMLib.map().getExtendedRoomID(propertyRooms.get(0)));
440                                        totalValue+=T.getPrice();
441                                        if(T.backTaxes()>0)
442                                        {
443                                                totalValue+=T.backTaxes();
444                                                owed+=T.backTaxes();
445                                        }
446                                }
447                                owed+=CMath.mul(totalValue,tax);
448
449                                if(owed>0)
450                                {
451                                        for(int p=0;p<particulars.size();p++)
452                                        {
453                                                T=(particulars.get(p));
454                                                if(T.backTaxes()<0)
455                                                {
456                                                        if((-T.backTaxes())>=owed)
457                                                        {
458                                                                paid+=owed;
459                                                                T.setBackTaxes((int)Math.round((T.backTaxes())+owed));
460                                                                T.updateTitle();
461                                                                break;
462                                                        }
463                                                        paid+=(-T.backTaxes());
464                                                        T.setBackTaxes(0);
465                                                        T.updateTitle();
466                                                }
467                                        }
468                                }
469                                if(owed>0)
470                                {
471                                        owed-=paid;
472                                        if((owed>0)
473                                        &&(!CMLib.beanCounter().modifyLocalBankGold(A,
474                                                owner,
475                                                CMLib.utensils().getFormattedDate(A)+": Withdrawal of "+owed+": Taxes on property: "+properties.toString(),
476                                                CMLib.beanCounter().getCurrency(A),
477                                                -owed)))
478                                        {
479                                                boolean owesButNotConfiscated=false;
480                                                for(int p=0;p<particulars.size();p++)
481                                                {
482                                                        T=particulars.get(p);
483                                                        double owedOnThisLand=CMath.mul(T.getPrice(),tax);
484                                                        owedOnThisLand-=(paid/particulars.size());
485                                                        if(owedOnThisLand>0)
486                                                        {
487                                                                final int oldBackTaxes = T.backTaxes();
488                                                                T.setBackTaxes((int)Math.round(oldBackTaxes+owedOnThisLand));
489                                                                if(CMath.div(T.getPrice(),T.backTaxes())<2.0)
490                                                                {
491                                                                        final Clan clanC=CMLib.clans().getClan(T.getOwnerName());
492                                                                        if(clanC!=null)
493                                                                        {
494                                                                                final MOB M=clanC.getResponsibleMember();
495                                                                                final List<Pair<Clan,Integer>> clanSet=new Vector<Pair<Clan,Integer>>();
496                                                                                clanSet.add(new Pair<Clan,Integer>(C,Integer.valueOf(0)));
497                                                                                final List<String> channels=CMLib.channels().getFlaggedChannelNames(ChannelsLibrary.ChannelFlag.CLANINFO);
498                                                                                for(int i=0;i<channels.size();i++)
499                                                                                {
500                                                                                        CMLib.commands().postChannel(channels.get(i),clanSet,
501                                                                                                CMLib.lang().L("@x1 has lost the title to @x2 in "+A.Name()+" due to failure to pay "
502                                                                                                                        + "property taxes.",T.getOwnerName(),T.landPropertyID()),false);
503                                                                                }
504                                                                                if(M!=null)
505                                                                                {
506                                                                                        notifyPlayer(M.Name(),clanC.name(),owed,"","@x1 lost property on @x3.",
507                                                                                                        "@x1 has lost the title to @x4 in "+A.Name()+" due to failure to pay property taxes.");
508                                                                                }
509                                                                        }
510                                                                        else
511                                                                        {
512                                                                                notifyPlayer(T.getOwnerName(),"",owed,T.landPropertyID(),"@x1 property lost on @x3.",
513                                                                                                "@x1 has lost the title to @x4 in "+A.Name()+" due to failure to pay property taxes.");
514                                                                        }
515                                                                        T.setBackTaxes(0);
516                                                                        T.setOwnerName("");
517                                                                        T.updateTitle();
518                                                                }
519                                                                else
520                                                                if(T.backTaxes() > oldBackTaxes)
521                                                                {
522                                                                        owesButNotConfiscated=true;
523                                                                        T.updateTitle();
524                                                                }
525                                                        }
526                                                }
527                                                if(owesButNotConfiscated)
528                                                {
529                                                        final Clan clanC=CMLib.clans().getClan(owner);
530                                                        if(clanC!=null)
531                                                        {
532                                                                final MOB M=clanC.getResponsibleMember();
533                                                                final String amountOwed = CMLib.beanCounter().nameCurrencyLong(M, owed);
534                                                                final List<Pair<Clan,Integer>> clanSet=new Vector<Pair<Clan,Integer>>();
535                                                                clanSet.add(new Pair<Clan,Integer>(C,Integer.valueOf(0)));
536                                                                final List<String> channels=CMLib.channels().getFlaggedChannelNames(ChannelsLibrary.ChannelFlag.CLANINFO);
537                                                                for(int i=0;i<channels.size();i++)
538                                                                {
539                                                                        CMLib.commands().postChannel(channels.get(i),clanSet,CMLib.lang().L("@x1 owes @x2 in back taxes to "+A.Name()+".  "
540                                                                                        + "Sufficient funds were not found in a local bank account."
541                                                                                        + "  Failure to pay could result in loss of property. ",clanC.name(),amountOwed),false);
542                                                                }
543                                                                if(M!=null)
544                                                                {
545                                                                        notifyPlayer(M.Name(),clanC.name(),owed,"",A.Name()+" Taxes Owed by @x1 on @x3.",
546                                                                                        "@x1 owes @x2 in back taxes to "+A.Name()+".  Sufficient funds were not found in a local bank account.  "
547                                                                                        + "Failure to pay could result in loss of property.");
548                                                                }
549                                                        }
550                                                        else
551                                                        {
552                                                                notifyPlayer(owner,"",owed,"",A.Name()+"Taxes Owed by @x1 on @x3.",
553                                                                                "@x1 owes @x2 in back taxes to "+A.Name()+".  Sufficient were not found in a local bank account.  "
554                                                                                                + "Failure to pay could result in loss of property.");
555                                                        }
556                                                        if((evasionBits!=null)
557                                                        &&(evasionBits[Law.BIT_CRIMENAME].length()>0)
558                                                        &&(responsibleMob!=null))
559                                                        {
560                                                                legalDetails.fillOutWarrant(responsibleMob,
561                                                                                                                        this,
562                                                                                                                        A,
563                                                                                                                        null,
564                                                                                                                        evasionBits[Law.BIT_CRIMELOCS],
565                                                                                                                        evasionBits[Law.BIT_CRIMEFLAGS],
566                                                                                                                        evasionBits[Law.BIT_CRIMENAME],
567                                                                                                                        evasionBits[Law.BIT_SENTENCE],
568                                                                                                                        evasionBits[Law.BIT_WARNMSG]);
569                                                        }
570                                                }
571                                        }
572                                        else
573                                        {
574                                                for(int p=0;p<particulars.size();p++)
575                                                {
576                                                        T=particulars.get(p);
577                                                        if(T.backTaxes()>0)
578                                                        {
579                                                                T.setBackTaxes(0);
580                                                                T.updateTitle();
581                                                        }
582                                                }
583                                                if(owed<0)
584                                                        owed=0;
585                                                if((treasuryR!=null)&&((owed+paid)>0))
586                                                {
587                                                        final List<Coins> V=CMLib.beanCounter().makeAllCurrency(CMLib.beanCounter().getCurrency(A),owed+paid);
588                                                        for(int v=0;v<V.size();v++)
589                                                        {
590                                                                final Coins COIN=V.get(v);
591                                                                COIN.setContainer(container);
592                                                                treasuryR.addItem(COIN);
593                                                                COIN.putCoinsBack();
594                                                        }
595                                                }
596                                                if((evasionBits!=null)
597                                                &&(evasionBits[Law.BIT_CRIMENAME].length()>0)
598                                                &&(responsibleMob!=null))
599                                                {
600                                                        while(getWarrant(responsibleMob,evasionBits[Law.BIT_CRIMENAME],true,debugging)!=null)
601                                                        {
602                                                        }
603                                                }
604                                        }
605                                }
606                        }
607
608                }
609        }

699        private void resetLaw(final Properties laws)
700        {
701                theLaws=laws;
702                activated=(!getInternalStr("ACTIVATED").equalsIgnoreCase("FALSE"));
703                officerNames=CMParms.parse(getInternalStr("OFFICERS"));
