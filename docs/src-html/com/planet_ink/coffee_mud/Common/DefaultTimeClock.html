202        {
203                final StringBuffer timeDesc=new StringBuffer("");
204                timeDesc.append("hour "+getHourOfDay()+" on ");
205                if(getDaysInWeek()>0)
206                {
207                        long x=((long)getYear())*((long)getMonthsInYear())*getDaysInMonth();
208                        x=x+((long)(getMonth()-1))*((long)getDaysInMonth());
209                        x=x+getDayOfMonth();
210                        timeDesc.append(getWeekNames()[(int)(x%getDaysInWeek())]+", ");
211                }
212                timeDesc.append("the "+getDayOfMonth()+CMath.numAppendage(getDayOfMonth()));
213                timeDesc.append(" day of "+getMonthNames()[getMonth()-1]);
214                if(getYearNames().length>0)
215                        timeDesc.append(", "+CMStrings.replaceAll(getYearNames()[getYear()%getYearNames().length],"#",""+getYear()));
216                return timeDesc.toString();

254        public String timeDescription(MOB mob, Room room)
255        {
256                final StringBuffer timeDesc=new StringBuffer("");
257
258                if(CMLib.flags().canSee(mob))
259                        timeDesc.append(getTODCode().getDesc());
260                timeDesc.append("(Hour: "+getHourOfDay()+"/"+(getHoursInDay()-1)+")");
261                timeDesc.append("\n\rIt is ");
262                if(getDaysInWeek()>0)
263                {
264                        long x=((long)getYear())*((long)getMonthsInYear())*getDaysInMonth();
265                        x=x+((long)(getMonth()-1))*((long)getDaysInMonth());
266                        x=x+getDayOfMonth();
267                        timeDesc.append(getWeekNames()[(int)(x%getDaysInWeek())]+", ");
268                }
269                timeDesc.append("the "+getDayOfMonth()+CMath.numAppendage(getDayOfMonth()));
270                timeDesc.append(" day of "+getMonthNames()[getMonth()-1]);
271                if(getYearNames().length>0)
272                        timeDesc.append(", "+CMStrings.replaceAll(getYearNames()[getYear()%getYearNames().length],"#",""+getYear()));
273                timeDesc.append(L(".\n\rIt is "+getSeasonCode().toString().toLowerCase()+"."));
274                if((CMLib.flags().canSee(mob))
275                &&(getTODCode()==TimeClock.TimeOfDay.NIGHT)
276                &&(CMLib.map().hasASky(room)))
277                {
278                        switch(room.getArea().getClimateObj().weatherType(room))
279                        {
280                        case Climate.WEATHER_BLIZZARD:
281                        case Climate.WEATHER_HAIL:
282                        case Climate.WEATHER_SLEET:
283                        case Climate.WEATHER_SNOW:
284                        case Climate.WEATHER_RAIN:
285                        case Climate.WEATHER_THUNDERSTORM:
286                                timeDesc.append("\n\r"+room.getArea().getClimateObj().weatherDescription(room)+L(" You can't see the moon.")); break;
287                        case Climate.WEATHER_CLOUDY:
288                                timeDesc.append(L("\n\rThe clouds obscure the moon."));
289                                break;
290                        case Climate.WEATHER_DUSTSTORM:
291                                timeDesc.append(L("\n\rThe dust obscures the moon."));
292                                break;
293                        default:
294                                timeDesc.append(L("\n\r"+getMoonPhase(room).getDesc()));
295                                break;
296                        }
297                }
298                return timeDesc.toString();
299        }

338        @Override
339        public MoonPhase getMoonPhase(Room room)
340        {
341                int moonDex = (int)Math.round(Math.floor(CMath.mul(CMath.div(getDayOfMonth(),getDaysInMonth()+1),8.0)));
342                if(room != null)
343                {
344                        final Area area = room.getArea();
345                        if((room.numEffects()>0) || ((area != null)&&(area.numEffects()>0)))
346                        {
347                                final List<Ability> moonEffects = new ArrayList<Ability>(1);
348                                if(room.numEffects()>0)
349                                        moonEffects.addAll(CMLib.flags().domainAffects(room, Ability.DOMAIN_MOONALTERING));
350                                if((area != null)&&(area.numEffects()>0))
351                                        moonEffects.addAll(CMLib.flags().domainAffects(area, Ability.DOMAIN_MOONALTERING));
352                                for(Ability A : moonEffects)
353                                        moonDex = (moonDex + A.abilityCode()) % 8;
354                        }
355                }
356                return TimeClock.MoonPhase.values()[moonDex];
357        }

359        @Override
360        public TidePhase getTidePhase(Room room)
361        {
362                final MoonPhase moonPhase = getMoonPhase(room);
363                TidePhase tidePhase;
364                if(getHourOfDay() == dawnToDusk[1])
365                        tidePhase = moonPhase.getLowTide();
366                else 
367                if(getHourOfDay() == dawnToDusk[2])
368                        tidePhase = moonPhase.getHighTide();
369                else
370                        tidePhase = TidePhase.NO_TIDE;
371                if(room != null)
372                {
373                        final Area area = room.getArea();
374                        if((room.numEffects()>0) || ((area != null)&&(area.numEffects()>0)))
375                        {
376                                final List<Ability> moonEffects = new ArrayList<Ability>(1);
377                                if(room.numEffects()>0)
378                                        moonEffects.addAll(CMLib.flags().flaggedAffects(room, Ability.FLAG_TIDEALTERING));
379                                if((area != null)&&(area.numEffects()>0))
380                                        moonEffects.addAll(CMLib.flags().flaggedAffects(area, Ability.FLAG_TIDEALTERING));
381                                if(moonEffects.size()>0)
382                                {
383                                        int tideDex = CMParms.indexOf(TidePhase.values(), tidePhase);
384                                        for(Ability A : moonEffects)
385                                                tideDex = (tideDex + A.abilityCode()) % TidePhase.values().length;
386                                        tidePhase = TidePhase.values()[tideDex];
387                                }
388                        }
389                }
390                return tidePhase;
391        }

411        @Override
412        public TimeOfDay getTODCode()
413        {
414                if((time>=getDawnToDusk()[TimeClock.TimeOfDay.NIGHT.ordinal()])&&(getDawnToDusk()[TimeClock.TimeOfDay.NIGHT.ordinal()]>=0))
415                        return TimeClock.TimeOfDay.NIGHT;
416                if((time>=getDawnToDusk()[TimeClock.TimeOfDay.DUSK.ordinal()])&&(getDawnToDusk()[TimeClock.TimeOfDay.DUSK.ordinal()]>=0))
417                        return TimeClock.TimeOfDay.DUSK;
418                if((time>=getDawnToDusk()[TimeClock.TimeOfDay.DAY.ordinal()])&&(getDawnToDusk()[TimeClock.TimeOfDay.DAY.ordinal()]>=0))
419                        return TimeClock.TimeOfDay.DAY;
420                if((time>=getDawnToDusk()[TimeClock.TimeOfDay.DAWN.ordinal()])&&(getDawnToDusk()[TimeClock.TimeOfDay.DAWN.ordinal()]>=0))
421                        return TimeClock.TimeOfDay.DAWN;
422                // it's before night, dusk, day, and dawn... before dawn is still night.
423                if(getDawnToDusk()[TimeClock.TimeOfDay.NIGHT.ordinal()]>=0)
424                        return TimeClock.TimeOfDay.NIGHT;
425                return TimeClock.TimeOfDay.DAY;
426        }

467        @Override
468        public String deriveEllapsedTimeString(long millis)
469        {
470                int hours=(int)(millis/CMProps.getMillisPerMudHour());
471                int days=0;
472                int months=0;
473                int years=0;
474                if(hours>getHoursInDay())
475                {
476                        days=(int)Math.round(Math.floor(CMath.div(hours,getHoursInDay())));
477                        hours=hours-(days*getHoursInDay());
478                }
479                if(days>getDaysInMonth())
480                {
481                        months=(int)Math.round(Math.floor(CMath.div(days,getDaysInMonth())));
482                        days=days-(months*getDaysInMonth());
483                }
484                if(months>getMonthsInYear())
485                {
486                        years=(int)Math.round(Math.floor(CMath.div(months,getMonthsInYear())));
487                        months=months-(years*getMonthsInYear());
488                }
489                final StringBuffer buf=new StringBuffer("");
490                if(years>0)
491                        buf.append(years+" years");
492                if(months>0)
493                {
494                        if(buf.length()>0)
495                                buf.append(", ");
496                        buf.append(months+" months");
497                }
498                if(days>0)
499                {
500                        if(buf.length()>0)
501                                buf.append(", ");
502                        buf.append(days+" days");
503                }
504                if(hours>0)
505                {
506                        if(buf.length()>0)
507                                buf.append(", ");
508                        buf.append(hours+" hours");
509                }
510                if(buf.length()==0)
511                        return "under an hour";
512                return buf.toString();
513        }

521        @Override
522        public long deriveMudHoursAfter(TimeClock C)
523        {
524                long numMudHours=0;
525                if(C.getYear()>getYear())
526                        return -1;
527                else
528                if(C.getYear()==getYear())
529                {
530                        if(C.getMonth()>getMonth())
531                                return -1;
532                        else
533                        if(C.getMonth()==getMonth())
534                        {
535                                if(C.getDayOfMonth()>getDayOfMonth())
536                                        return -1;
537                                else
538                                if(C.getDayOfMonth()==getDayOfMonth())
539                                {
540                                        if(C.getHourOfDay()>getHourOfDay())
541                                                return -1;
542                                }
543                        }
544                }
545                numMudHours+=(getYear()-C.getYear())*(getHoursInDay()*getDaysInMonth()*getMonthsInYear());
546                numMudHours+=(getMonth()-C.getMonth())*(getHoursInDay()*getDaysInMonth());
547                numMudHours+=(getDayOfMonth()-C.getDayOfMonth())*getHoursInDay();
548                numMudHours+=(getHourOfDay()-C.getHourOfDay());
549                return numMudHours;
550        }

552        @Override
553        public void handleTimeChange()
554        {
555                try
556                {
557                        for(final Enumeration<Area> a=CMLib.map().areas();a.hasMoreElements();)
558                        {
559                                final Area A=a.nextElement();
560                                if(A.getTimeObj()==this)
561                                for(final Enumeration<Room> r=A.getProperMap();r.hasMoreElements();)
562                                {
563                                        final Room R=r.nextElement();
564                                        if((R!=null)&&((R.numInhabitants()>0)||(R.numItems()>0)))
565                                        {
566                                                R.recoverPhyStats();
567                                                for(int m=0;m<R.numInhabitants();m++)
568                                                {
569                                                        final MOB mob=R.fetchInhabitant(m);
570                                                        if((mob!=null)
571                                                        &&(!mob.isMonster()))
572                                                        {
573                                                                if(CMLib.map().hasASky(R)
574                                                                &&(!CMLib.flags().isSleeping(mob))
575                                                                &&(CMLib.flags().canSee(mob)))
576                                                                {
577                                                                        final String message = CMProps.getListFileChoiceFromIndexedList(CMProps.ListFile.TOD_CHANGE_OUTSIDE, getTODCode().ordinal());
578                                                                        if(message.trim().length()>0)
579                                                                                mob.tell(message);
580                                                                }
581                                                                else
582                                                                {
583                                                                        final String message = CMProps.getListFileChoiceFromIndexedList(CMProps.ListFile.TOD_CHANGE_INSIDE, getTODCode().ordinal());
584                                                                        if(message.trim().length()>0)
585                                                                                mob.tell(message);
586                                                                }
587                                                        }
588                                                }
589                                        }
590                                        if(R!=null)
591                                                R.recoverRoomStats();
592                                }
593                        }
594                }
595                catch (final java.util.NoSuchElementException x)
596                {
597                }
598        }

600        protected void tickTock(int howManyHours, boolean moveTheSky)
601        {
602                final TimeOfDay todCode=getTODCode();
603                if(howManyHours!=0)
604                {
605                        setHourOfDay(getHourOfDay()+howManyHours);
606                        lastTicked=System.currentTimeMillis();
607                        while(getHourOfDay()>=getHoursInDay())
608                        {
609                                setHourOfDay(getHourOfDay()-getHoursInDay());
610                                setDayOfMonth(getDayOfMonth()+1);
611                                if(getDayOfMonth()>getDaysInMonth())
612                                {
613                                        setDayOfMonth(1);
614                                        setMonth(getMonth()+1);
615                                        if(getMonth()>getMonthsInYear())
616                                        {
617                                                setMonth(1);
618                                                setYear(getYear()+1);
619                                        }
620                                }
621                        }
622                        while(getHourOfDay()<0)
623                        {
624                                setHourOfDay(getHoursInDay()+getHourOfDay());
625                                setDayOfMonth(getDayOfMonth()-1);
626                                if(getDayOfMonth()<1)
627                                {
628                                        setDayOfMonth(getDaysInMonth());
629                                        setMonth(getMonth()-1);
630                                        if(getMonth()<1)
631                                        {
632                                                setMonth(getMonthsInYear());
633                                                setYear(getYear()-1);
634                                        }
635                                }
636                        }
637                }
638                if((moveTheSky)&&(getTODCode()!=todCode))
639                        handleTimeChange();
640        }

678        @Override
679        public long toHoursSinceEpoc()
680        {
681                final long hoursInDay = this.getHoursInDay();
682                final long hoursInMonth = this.getDaysInMonth() * hoursInDay;
683                final long hoursInYear = this.getMonthsInYear() * hoursInMonth;
684                long total = this.getHourOfDay() + 
685                                ((day-1) * hoursInDay) +
686                                ((month-1) * hoursInMonth) + 
687                                (year * hoursInYear);
688                return total;
689        }

691        @Override
692        public void setFromHoursSinceEpoc(long num)
693        {
694                final long hoursInDay = this.getHoursInDay();
695                final long hoursInMonth = this.getDaysInMonth() * hoursInDay;
696                final long hoursInYear = this.getMonthsInYear() * hoursInMonth;
697                year=0;
698                while(num > hoursInYear)
699                {
700                        year++;
701                        num -= hoursInYear;
702                }
703                month = 1;
704                while(num > hoursInMonth)
705                {
706                        month++;
707                        num -= hoursInMonth;
708                }
709                day = 1;
710                while(num > hoursInDay)
711                {
712                        day++;
713                        num -= hoursInDay;
714                }
715                time = (int)num;
716        }

737        @Override
738        public boolean tick(final Tickable ticking, final int tickID)
739        {
740                tickStatus=Tickable.STATUS_NOT;
741                if(((loadName==null)||(loaded))
742                &&(((System.currentTimeMillis()-lastTicked)<=CMProps.getMillisPerMudHour())))
743                        return true;
744                boolean process = false;
745                boolean timeToTick=false;
746                synchronized(this)
747                {
748                        timeToTick = ((System.currentTimeMillis()-lastTicked)>CMProps.getMillisPerMudHour());
749                        lastTicked=System.currentTimeMillis();
750                        if((loadName!=null)&&(!loaded))
751                        {
752                                process=true;
753                        }
754                }
755                if(process)
756                {
757                        loaded=true;
758                        final List<PlayerData> bitV=CMLib.database().DBReadPlayerData(loadName,"TIMECLOCK");
759                        String timeRsc=null;
760                        if((bitV==null)||(bitV.size()==0))
761                                timeRsc="<TIME>-1</TIME><DAY>1</DAY><MONTH>1</MONTH><YEAR>1</YEAR>";
762                        else
763                                timeRsc=bitV.get(0).xml();
764                        final List<XMLLibrary.XMLTag> V=CMLib.xml().parseAllXML(timeRsc);
765                        setHourOfDay(CMLib.xml().getIntFromPieces(V,"TIME"));
766                        setDayOfMonth(CMLib.xml().getIntFromPieces(V,"DAY"));
767                        setMonth(CMLib.xml().getIntFromPieces(V,"MONTH"));
768                        setYear(CMLib.xml().getIntFromPieces(V,"YEAR"));
769                        if(this!=CMLib.time().globalClock())
770                        {
771                                if((CMLib.xml().getValFromPieces(V,"HOURS").length()==0)
772                                ||(CMLib.xml().getValFromPieces(V,"DAYS").length()==0)
773                                ||(CMLib.xml().getValFromPieces(V,"MONTHS").length()==0))
774                                {
775                                        setHoursInDay(CMLib.time().globalClock().getHoursInDay());
776                                        setDaysInMonth(CMLib.time().globalClock().getDaysInMonth());
777                                        setMonthsInYear(CMLib.time().globalClock().getMonthNames());
778                                        setDawnToDusk(CMLib.time().globalClock().getDawnToDusk()[TimeOfDay.DAWN.ordinal()],
779                                                                  CMLib.time().globalClock().getDawnToDusk()[TimeOfDay.DAY.ordinal()],
780                                                                  CMLib.time().globalClock().getDawnToDusk()[TimeOfDay.DUSK.ordinal()],
781                                                                  CMLib.time().globalClock().getDawnToDusk()[TimeOfDay.NIGHT.ordinal()]);
782                                        setDaysInWeek(CMLib.time().globalClock().getWeekNames());
783                                        setYearNames(CMLib.time().globalClock().getYearNames());
784                                }
785                                else
786                                {
787                                        setHoursInDay(CMLib.xml().getIntFromPieces(V,"HOURS"));
788                                        setDaysInMonth(CMLib.xml().getIntFromPieces(V,"DAYS"));
789                                        setMonthsInYear(CMParms.toStringArray(CMParms.parseCommas(CMLib.xml().getValFromPieces(V,"MONTHS"),true)));
790                                        setDawnToDusk(CMLib.xml().getIntFromPieces(V,"DAWNHR"),
791                                                                  CMLib.xml().getIntFromPieces(V,"DAYHR"),
792                                                                  CMLib.xml().getIntFromPieces(V,"DUSKHR"),
793                                                                  CMLib.xml().getIntFromPieces(V,"NIGHTHR"));
794                                        setDaysInWeek(CMParms.toStringArray(CMParms.parseCommas(CMLib.xml().getValFromPieces(V,"WEEK"),true)));
795                                        setYearNames(CMParms.toStringArray(CMParms.parseCommas(CMLib.xml().getValFromPieces(V,"YEARS"),true)));
796                                }
797                        }
798                }
799                if(timeToTick)
800                        tickTock(1);
801                return true;
802        }

804        @Override
805        public int compareTo(final CMObject o)
806        {
807                if(o instanceof TimeClock)
808                {
809                        final TimeClock c2=(TimeClock)o;
810                        long myHrs = (getHourOfDay() 
811                                                + (getDayOfMonth() * getHoursInDay())
812                                                + (getMonth() * getDaysInMonth() * getHoursInDay())
813                                                + (getYear() * getMonthsInYear() * getDaysInMonth() * getHoursInDay()));
814                        long hisHrs = (c2.getHourOfDay() 
815                                        + (c2.getDayOfMonth() * c2.getHoursInDay())
816                                        + (c2.getMonth() * c2.getDaysInMonth() * c2.getHoursInDay())
817                                        + (c2.getYear() * c2.getMonthsInYear() * c2.getDaysInMonth() * c2.getHoursInDay()));
818                        if (myHrs == hisHrs)
819                                return 0;
820                        if (myHrs > hisHrs)
821                                return 1;
822                        return -1;
823                }
824                else
825                        return CMClass.classID(this).compareToIgnoreCase(CMClass.classID(o));
826        }
