107        @Override
108        public String data()
109        {
110                final StringBuffer data=new StringBuffer("");
111                final XMLLibrary xml=CMLib.xml();
112                if(xml != null)
113                {
114                        data.append(xml.convertXMLtoTag("HIGH",highestOnline));
115                        data.append(xml.convertXMLtoTag("NUMONLINE",numberOnlineTotal));
116                        data.append(xml.convertXMLtoTag("NUMCOUNT",numberOnlineCounter));
117                        data.append("<STATS>");
118                        final Map<String,long[]> stats=this.stats;
119                        if(stats == null)
120                                return "";
121                        for(final Iterator<String> e=stats.keySet().iterator();e.hasNext();)
122                        {
123                                final String s=e.next();
124                                final long[] l=stats.get(s);
125                                data.append(xml.convertXMLtoTag(s,CMParms.toListString(l)));
126                        }
127                        data.append("</STATS>");
128                }
129                return data.toString();
130        }
131
132        @Override
133        public void bumpVal(final String s, final int type)
134        {
135                long[] stat=null;
136                synchronized(stats)
137                {
138                        if(stats.containsKey(s))
139                                stat=stats.get(s);
140                        else
141                        {
142                                stat=new long[STAT_TOTAL];
143                                stats.put(s,stat);
144                        }
145                }
146                stat[type]++;
147        }
148
149        @Override
150        public void totalUp(String code, final long[] tot)
151        {
152                code=tagFix(code);
153                for(final Iterator<String> e=stats.keySet().iterator();e.hasNext();)
154                {
155                        final String s=e.next();
156                        if(s.startsWith(code)
157                        ||(s.startsWith("C")&&code.startsWith("*")))
158                        {
159                                final long[] theseStats=stats.get(s);
160                                for(int t=0;t<theseStats.length;t++)
161                                        tot[t]+=theseStats[t];
162                        }
163                }
164        }
165
166        @Override
167        public String tagFix(final String s)
168        {
169                return s.trim().replaceAll(" ","_").toUpperCase();
170        }
171
172        @Override
173        public void bumpVal(final CMObject E, final int type)
174        {
175                if((E instanceof MOB)&&(((MOB)E).isMonster()))
176                        return;
177
178                if(type==STAT_SPECIAL_NUMONLINE)
179                {
180                        int ct=0;
181                        for(final Session S : CMLib.sessions().localOnlineIterable())
182                        {
183                                if(S!=null)
184                                        ct++;
185                        }
186                        numberOnlineCounter++;
187                        numberOnlineTotal+=ct;
188                        if(ct>highestOnline)
189                                highestOnline=ct;
190                        return;
191                }
192                // classes, races, levels, genders, faiths, clanned, grouped
193                if(E instanceof MOB)
194                {
195                        final MOB mob=(MOB)E;
196                        final Room R=mob.location();
197                        Area A=(R==null) ? null : R.getArea();
198                        if((A!=null) && (CMath.bset(A.flags(),Area.FLAG_INSTANCE_CHILD)))
199                                A=CMLib.map().getModelArea(A);
200                        if(A!=null)
201                                bumpVal("X"+tagFix(A.Name()),type);
202                        bumpVal("B"+tagFix(mob.baseCharStats().getCurrentClass().baseClass()),type);
203                        bumpVal("C"+tagFix(mob.baseCharStats().getCurrentClass().ID()),type);
204                        bumpVal("R"+tagFix(mob.baseCharStats().getMyRace().ID()),type);
205                        bumpVal("L"+mob.basePhyStats().level(),type);
206                        bumpVal("G"+((char)mob.baseCharStats().getStat(CharStats.STAT_GENDER)),type);
207                        bumpVal("F"+tagFix(mob.getWorshipCharID()),type);
208                        for(final Pair<Clan,Integer> p : mob.clans())
209                                bumpVal("Q"+tagFix(p.first.clanID()),type);
210                        final Set<MOB> H=mob.getGroupMembers(new HashSet<MOB>());
211                        bumpVal("J"+H.size(),type);
212                        int pct=0;
213                        for (final MOB mob2 : H)
214                        {
215                                if(!mob2.isMonster())
216                                        pct++;
217                        }
218                        if(pct==0)
219                                pct=1;
220                        bumpVal("P"+pct,type);
221                }
222                else
223                if(E instanceof Ability)
224                        bumpVal("A"+tagFix(E.ID()),type);
225                else
226                if(E instanceof Quest)
227                        bumpVal("U"+tagFix(((Quest)E).name()),type);
228        }
229
230        @Override
231        public void populate(final long start, final long end, final String data)
232        {
233                synchronized(stats)
234                {
235                        startTime=start;
236                        endTime=end;
237                        final List<XMLLibrary.XMLTag> all=CMLib.xml().parseAllXML(data);
238                        if((all==null)||(all.size()==0))
239                                return;
240                        highestOnline=CMLib.xml().getIntFromPieces(all,"HIGH");
241                        numberOnlineTotal=CMLib.xml().getIntFromPieces(all,"NUMONLINE");
242                        numberOnlineCounter=CMLib.xml().getIntFromPieces(all,"NUMCOUNT");
243                        final XMLTag X=CMLib.xml().getPieceFromPieces(all,"STATS");
244                        if((X==null)||(X.contents()==null)||(X.contents().size()==0)||(!X.tag().equals("STATS")))
245                                return;
246                        stats.clear();
247                        for(int s=0;s<X.contents().size();s++)
248                        {
249                                final XMLTag S=X.contents().get(s);
250                                long[] l=CMParms.toLongArray(CMParms.parseCommas(S.value(),true));
251                                if(l.length<STAT_TOTAL)
252                                {
253                                        final long[] l2=new long[STAT_TOTAL];
254                                        for(int i=0;i<l.length;i++)
255                                                l2[i]=l[i];
256                                        l=l2;
257                                }
258                                final long[] l2=stats.get(S.tag());
259                                if(l2!=null)
260                                {
261                                        for(int i=0;i<l2.length;i++)
262                                                l[i]+=l2[i];
263                                        stats.remove(S.tag());
264                                }
265                                stats.put(S.tag(),l);
266                        }
267                }
268        }
