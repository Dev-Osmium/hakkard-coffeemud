159        public Object getDesignatedObject(final String named)
160        {
161                int code=-1;
162                for(int i=0;i<QCODES.length;i++)
163                {
164                        if(named.equalsIgnoreCase(QCODES[i]))
165                        {
166                                code = i;
167                                break;
168                        }
169                }
170                switch(code)
171                {
172                case 0:
173                        return ID();
174                case 1:
175                        return name();
176                case 2:
177                        return "" + duration();
178                case 3:
179                        return "" + minWait();
180                case 4:
181                        return "" + minPlayers();
182                case 5:
183                        return "" + playerMask();
184                case 6:
185                        return "" + runLevel();
186                case 7:
187                        return "" + startDate();
188                case 8:
189                        return "" + startDate();
190                case 9:
191                        return "" + waitInterval();
192                case 10:
193                        return SPAWN_DESCS[getSpawn()];
194                case 11:
195                        return displayName();
196                case 12:
197                        break; // instructions should fall through
198                case 13:
199                        return Boolean.toString(durable);
200                case 14:
201                        return "" + author();
202                }
203                return questState.getStat(named);
204        }

208        {
209                if(isCopy())
210                        return;
211                if((internalFiles!=null)&&(internalFiles.size()>0))
212                {
213                        for(int i=0;i<internalFiles.size();i++)
214                        {
215                                final String filename=((String)internalFiles.elementAt(i,1)).toUpperCase();
216                                final Vector<String> delThese=new Vector<String>();
217                                boolean foundKey=false;
218                                for(final Iterator<String> k=Resources.findResourceKeys(filename);k.hasNext();)
219                                {
220                                        final String key=k.next();
221                                        if(key.startsWith("PARSEDPRG: ")&&(key.toUpperCase().endsWith(filename)))
222                                        {
223                                                foundKey = true;
224                                                delThese.addElement(key);
225                                        }
226                                }
227                                if(foundKey)
228                                {
229                                        for(int d=0;d<delThese.size();d++)
230                                                Resources.removeResource(delThese.elementAt(d));

481                else
482                if(CMLib.english().containsString(mname,str))
483                {
484                        if((choices==null)||(choices==choices2)||(choices==choices3))
485                                choices=choices1;

487                }

492                                choices=choices2;
493                        choices2.add(E);

497                {
498                        if(choices==null)
499                                choices=choices3;
500                        choices3.add(E);
501                }

526        public void parseQuestScriptWArgs(final Vector script, List args)
527        {
528                if(args==null)
529                        args=new Vector();
530                if(args.size()==0)
531                        parseQuestScript(script, args, -1);
532                else
533                {
534                        final Vector allArgs=new Vector();
535                        for(int i=0;i<args.size();i++)
536                        {
537                                final Object O=args.get(i);
538                                if(O instanceof List)
539                                {
540                                        final List V=(List)O;
541                                        if(allArgs.size()==0)
542                                        {
543                                                for(int v=0;v<V.size();v++)
544                                                        allArgs.addElement(new XVector(V.get(v)));
545                                        }
546                                        else
547                                        {
548                                                final List allArgsCopy=(List)allArgs.clone();
549                                                allArgs.clear();
550                                                for(int aa=0;aa<allArgsCopy.size();aa++)
551                                                {
552                                                        final List argSet=(List)allArgsCopy.get(aa);
553                                                        for(int v=0;v<V.size();v++)
554                                                        {
555                                                                final List V2=new XVector(argSet);
556                                                                V2.add(V.get(v));
557                                                                allArgs.addElement(V2);
558                                                        }
559                                                }
560                                        }
561                                }
562                                else
563                                if(allArgs.size()==0)
564                                        allArgs.addElement(new XVector(O));
565                                else
566                                for(int aa=0;aa<allArgs.size();aa++)
567                                        ((List)allArgs.elementAt(aa)).add(O);
568                        }
569                        for(int a=0;a<allArgs.size();a++)
570                                parseQuestScript(script, (List)allArgs.elementAt(a),-1);
571                }
572        }

594                if((choices!=null)&&(choices.size()>0))

597                        for(int c=choices.size()-1;c>=0;c--)

614                                errorOccurred(q,isQuiet,"Quest '"+name()+"', all choices were taken: '"+choicesStr+"' by: "+CMParms.toListString(inUseByWhom)+".");
615                }

618        public void parseQuestScript(Vector script, final List args, final int startLine)
619        {
620                final Vector<String> finalScript=new Vector<String>();
621                for(int v=0;v<script.size();v++)
622                {
623                        if(script.elementAt(v) instanceof String)
624                                finalScript.addElement((String)script.elementAt(v));
625                        else
626                        if(script.elementAt(v) instanceof List)
627                        {
628                                final int vs=v;
629                                while((v<script.size())&&(script.elementAt(v) instanceof List))
630                                        v++;
631                                final int rnum=vs+CMLib.dice().roll(1,v-vs,-1);
632                                if(rnum<script.size())
633                                {
634                                        final List V=(List)script.elementAt(rnum);
635                                        for(int v2=0;v2<V.size();v2++)
636                                        {
637                                                if(V.get(v2) instanceof String)
638                                                        finalScript.addElement((String)V.get(v2));
639                                        }
640                                }
641                        }
642                }
643
644                script=finalScript;
645                final QuestState q=questState;
646                int vStart=startLine;
647                if(vStart<0)
648                        vStart=0;
649                q.done=false;
650                if(vStart>=script.size())
651                        return;
652                q.startLine=vStart;
653                for(int v=vStart;v<script.size();v++)
654                {
655                        if(startLine>=0)
656                                q.lastLine=v;
657                        final String s=modifyStringFromArgs((String)script.elementAt(v),args);
658                        final Vector<String> p=CMParms.parse(s);
659                        boolean isQuiet=q.beQuiet;
660                        if(p.size()>0)
661                        {
662                                String cmd=p.elementAt(0).toUpperCase();
663                                if(cmd.equals("<SCRIPT>"))
664                                {
665                                        final StringBuffer jscript=new StringBuffer("");
666                                        while(((++v)<script.size())
667                                        &&(!((String)script.elementAt(v)).trim().toUpperCase().startsWith("</SCRIPT>")))
668                                                jscript.append(((String)script.elementAt(v))+"\n");
669                                        if(v>=script.size())
670                                        {
671                                                errorOccurred(q,false,"Quest '"+name()+"', <SCRIPT> command without </SCRIPT> found.");
672                                                break;
673                                        }
674                                        if(!CMSecurity.isApprovedJScript(jscript))
675                                        {
676                                                errorOccurred(q,false,"Quest '"+name()+"', <SCRIPT> not approved.  Use MODIFY JSCRIPT to approve.");
677                                                break;
678                                        }
679                                        final Context cx = Context.enter();
680                                        try
681                                        {
682                                                final JScriptQuest scope = new JScriptQuest(this,q);
683                                                cx.initStandardObjects(scope);
684                                                scope.defineFunctionProperties(JScriptQuest.functions,
685                                                                                                           JScriptQuest.class,
686                                                                                                           ScriptableObject.DONTENUM);
687                                                cx.evaluateString(scope, jscript.toString(),"<cmd>", 1, null);
688                                        }
689                                        catch(final Exception e)
690                                        {
691                                                errorOccurred(q,false,"Quest '"+name()+"', JScript q.error: "+e.getMessage()+".");
692                                                Context.exit();
693                                                break;
694                                        }
695                                        Context.exit();
696                                        continue;
697                                }
698                                if(cmd.equals("QUIET"))
699                                {
700                                        if(p.size()<2)
701                                        {
702                                                q.beQuiet=true;
703                                                continue;
704                                        }
705                                        isQuiet=true;
706                                        p.removeElementAt(0);
707                                        cmd=p.elementAt(0).toUpperCase();
708                                }
709                                if(cmd.equals("STEP"))
710                                {
711                                        q.autoStepAfterDuration=false;
712                                        if((p.size()>1)&&(p.elementAt(1).equalsIgnoreCase("BREAK")))
713                                        {
714                                                q.lastLine=script.size();
715                                                q.done=true;
716                                        }
717                                        else
718                                        if((p.size()>1)&&(p.elementAt(1).equalsIgnoreCase("BACK")))
719                                        {
720                                                if(startLine>=0)
721                                                        q.lastLine=q.startLine;
722                                        }
723                                        else
724                                        if((p.size()>1)&&(p.elementAt(1).equalsIgnoreCase("AUTO")))
725                                        {
726                                                if(startLine>=0)
727                                                        q.lastLine=v+1;
728                                                q.autoStepAfterDuration=true;
729                                        }
730                                        else
731                                        {
732                                                if(startLine>=0)
733                                                        q.lastLine=v+1;
734                                        }
735                                        return;
736                                }
737                                if(cmd.equals("RESET"))
738                                {
739                                        if(q.room!=null)
740                                                CMLib.map().resetRoom(q.room, true);
741                                        else
742                                        if(q.roomGroup!=null)
743                                        {
744                                                for(int r=0;r<q.roomGroup.size();r++)
745                                                        CMLib.map().resetRoom(q.roomGroup.get(r), true);
746                                        }
747                                        else
748                                        if(q.area!=null)
749                                                CMLib.map().resetArea(q.area);
