424        public SHashSet<String> getHashFrom(String str)
425        {
426                final SHashSet<String> h=new SHashSet<String>();
427                if((str==null)||(str.length()==0))
428                        return h;
429                str=CMStrings.replaceAll(str,"<FRIENDS>","");
430                str=CMStrings.replaceAll(str,"<IGNORED>","");
431                str=CMStrings.replaceAll(str,"<INTROS>","");
432                str=CMStrings.replaceAll(str,"</INTROS>","");
433                str=CMStrings.replaceAll(str,"</FRIENDS>","");
434                str=CMStrings.replaceAll(str,"</IGNORED>","");
435                int x=str.indexOf(';');
436                while(x>=0)
437                {
438                        final String fi=str.substring(0,x).trim();
439                        if(fi.length()>0)
440                                h.add(fi);
441                        str=str.substring(x+1);
442                        x=str.indexOf(';');
443                }
444                if(str.trim().length()>0)
445                        h.add(str.trim());
446                return h;
447        }

502        {
503                if(mob==null)
504                        return false;
505                if(account != null)
506                        return account.isIgnored(mob);

508                        mob=mob.soulMate();

510                        return true;
511                final PlayerStats stats=mob.playerStats();
512                if(stats ==null)
513                        return false;
514                final PlayerAccount account=stats.getAccount();
515                if(account == null)
516                        return false;
517                return ignored.contains(account.getAccountName()+"*");

675        @Override
676        public int initializeBirthday(TimeClock clock, int ageHours, final Race R)
677        {
678                if(clock == null)
679                        clock=CMLib.time().globalClock();
680                birthday=new int[4];
681                birthday[0]=clock.getDayOfMonth();
682                birthday[1]=clock.getMonth();
683                birthday[2]=clock.getYear();
684                birthday[3]=clock.getYear();
685                while(ageHours>15)
686                {
687                        birthday[2]-=1;
688                        ageHours-=15;
689                }
690                if(ageHours>0)
691                {
692                        birthday[1]=CMLib.dice().roll(1,clock.getMonthsInYear(),0);
693                        birthday[0]=CMLib.dice().roll(1,clock.getDaysInMonth(),0);
694                }
695                final int month=clock.getMonth();
696                final int day=clock.getDayOfMonth();
697                if((month<birthday[1])||((month==birthday[1])&&(birthday[0]<day)))
698                        return (R.getAgingChart()[Race.AGE_YOUNGADULT]+clock.getYear()-birthday[2])-1;
699                return (R.getAgingChart()[Race.AGE_YOUNGADULT]+clock.getYear()-birthday[2]);
700        }

712        @Override
713        public void bumpPrideStat(final PrideStat stat, final int amt)
714        {
715                final long now=System.currentTimeMillis();
716                if(stat!=null)
717                for(final TimeClock.TimePeriod period : TimeClock.TimePeriod.values())
718                {
719                        if(period==TimeClock.TimePeriod.ALLTIME)
720                                prideStats[period.ordinal()][stat.ordinal()]+=amt;
721                        else
722                        {
723                                if(now>prideExpireTime[period.ordinal()])
724                                {
725                                        for(final AccountStats.PrideStat stat2 : AccountStats.PrideStat.values())
726                                                prideStats[period.ordinal()][stat2.ordinal()]=0;
727                                        prideExpireTime[period.ordinal()]=period.nextPeriod();
728                                }
729                                prideStats[period.ordinal()][stat.ordinal()]+=amt;
730                        }
731                }
732        }
