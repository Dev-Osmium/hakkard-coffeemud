1019                private int[]                   baseStatCodes                           = new int[0];
1020                private int[]                   maxStatCodes                            = new int[0];

1027                private String[]                statNames                                       = new String[0];
1028                private String[]                baseStatNames                           = new String[0];
1029                private String[]                shortNames                                      = new String[0];
1030                private String[]                statAttributionDescriptions     = new String[0];
1031                private int[]                   statCMMsgMapping                        = new int[0];
1032                private int                             longestBaseCodeName                     = -1;
1033                private final Map<Integer,Integer> rvsStatCMMsgMapping  = new Hashtable<Integer,Integer>();
1034
1035                public CODES()
1036                {
1037                        super();
1038                        final char c=Thread.currentThread().getThreadGroup().getName().charAt(0);
1039                        if(insts==null)
1040                                insts=new CODES[256];
1041                        if(insts[c]==null)
1042                                insts[c]=this;
1043
1044                        final String[][] addExtra = CMProps.instance().getStrsStarting("ADDCHARSTAT_");
1045                        final String[][] repExtra = CMProps.instance().getStrsStarting("REPLACECHARSTAT_");
1046                        for(int i=STAT_STRENGTH;i<=STAT_WISDOM;i++)
1047                                addBaseStat(DEFAULT_STAT_ABBR[i],DEFAULT_STAT_DESCS[i],DEFAULT_STAT_NAMES[i],DEFAULT_STAT_DESC_ATTS[i],DEFAULT_STAT_MSG_MAP[i]);
1048                        addAllStat(DEFAULT_STAT_ABBR[STAT_GENDER],DEFAULT_STAT_DESCS[STAT_GENDER],DEFAULT_STAT_NAMES[STAT_GENDER],
1049                                                DEFAULT_STAT_DESC_ATTS[STAT_GENDER],DEFAULT_STAT_MSG_MAP[STAT_GENDER],false);
1050                        for(int i=STAT_SAVE_PARALYSIS;i<=STAT_SAVE_TRAPS;i++)
1051                                addSavingThrow(DEFAULT_STAT_ABBR[i],DEFAULT_STAT_DESCS[i],DEFAULT_STAT_NAMES[i],DEFAULT_STAT_DESC_ATTS[i],DEFAULT_STAT_MSG_MAP[i]);
1052                        int baseCtr=0;
1053                        for(int i=STAT_MAX_STRENGTH_ADJ;i<=STAT_MAX_WISDOM_ADJ;i++)
1054                                addMaxStat(baseStatCodes[baseCtr++],DEFAULT_STAT_ABBR[i],DEFAULT_STAT_DESCS[i],DEFAULT_STAT_NAMES[i],DEFAULT_STAT_DESC_ATTS[i],DEFAULT_STAT_MSG_MAP[i]);
1055                        addAllStat(DEFAULT_STAT_ABBR[STAT_AGE],DEFAULT_STAT_DESCS[STAT_AGE],DEFAULT_STAT_NAMES[STAT_AGE],
1056                                                DEFAULT_STAT_DESC_ATTS[STAT_AGE],DEFAULT_STAT_MSG_MAP[STAT_AGE],false);
1057                        for(int i=STAT_SAVE_DETECTION;i<STAT_FAITH;i++)
1058                                addSavingThrow(DEFAULT_STAT_ABBR[i],DEFAULT_STAT_DESCS[i],DEFAULT_STAT_NAMES[i],DEFAULT_STAT_DESC_ATTS[i],DEFAULT_STAT_MSG_MAP[i]);
1059                        for(int i=STAT_FAITH;i<DEFAULT_NUM_STATS;i++)
1060                                addAllStat(DEFAULT_STAT_ABBR[i],DEFAULT_STAT_DESCS[i],DEFAULT_STAT_NAMES[i],DEFAULT_STAT_DESC_ATTS[i],DEFAULT_STAT_MSG_MAP[i],false);
1061                        for(int i=0;i<addExtra.length+repExtra.length;i++)
1062                        {
1063                                final String[] array = (i>=addExtra.length)?repExtra[i-addExtra.length]:addExtra[i];
1064                                final boolean replace = i>=addExtra.length;
1065                                String stat = array[0].toUpperCase().trim();
1066                                final String p=array[1];
1067                                final List<String> V=CMParms.parseCommas(p, false);
1068                                if(V.size()!=4)
1069                                {
1070                                        Log.errOut("CharStats","Bad coffeemud.ini charstat row, requires 4 ; separated entries: "+p);
1071                                        continue;
1072                                }
1073                                String type=V.get(0).toUpperCase().trim();
1074                                int oldStatCode=-1;
1075                                if(replace)
1076                                {
1077                                        final String repStat=stat;
1078                                        stat=type;
1079                                        oldStatCode=CMParms.indexOf(DEFAULT_STAT_NAMES, repStat);
1080                                        if(oldStatCode<0)
1081                                                oldStatCode=CMParms.indexOf(DEFAULT_STAT_DESCS, repStat);
1082                                        if(oldStatCode>=0)
1083                                                type="REPLACE";
1084                                        else
1085                                        {
1086                                                Log.errOut("CharStats","Bad coffeemud.ini charstat row, bad stat name: "+repStat);
1087                                                continue;
1088                                        }
1089                                }
1090                                final String abbr=V.get(1);
1091                                final String desc=V.get(2).toUpperCase();
1092                                final String adj=V.get(3).toUpperCase();
1093                                if(type.equalsIgnoreCase("BASE"))
1094                                {
1095                                        addBaseStat(abbr, desc, stat, adj, -1);
1096                                        final int baseStatCode=allStatCodes.length-1;
1097                                        addMaxStat(baseStatCode, "m"+abbr, "MAX "+stat+" ADJ.", "MAX"+stat, "POTENTIALLY "+adj, -1);
1098                                }
1099                                else
1100                                if(type.equalsIgnoreCase("SAVE"))
1101                                        addSavingThrow(abbr,desc,stat,adj,-1);
1102                                else
1103                                if(type.equalsIgnoreCase("OTHER"))
1104                                        addAllStat(abbr,desc,stat,adj,-1,false);
1105                                else
1106                                if(replace&&(oldStatCode>=0))
1107                                {
1108                                        statAbbreviations[oldStatCode]=abbr;
1109                                        statDescriptions[oldStatCode]=desc;
1110                                        statNames[oldStatCode]=stat;
1111                                        if(stat.length()>3)
1112                                                shortNames[oldStatCode]=CMStrings.capitalizeAndLower(stat).substring(0,3);
1113                                        else
1114                                                shortNames[oldStatCode]=CMStrings.capitalizeAndLower(stat);
1115                                        statAttributionDescriptions[oldStatCode]=adj;
1116                                        statCMMsgMapping[oldStatCode]=-1;
1117                                }
1118                                else
1119                                        Log.errOut("CharStats","Bad coffeemud.ini charstat row, bad type: "+type);
1120                        }
1121                }
