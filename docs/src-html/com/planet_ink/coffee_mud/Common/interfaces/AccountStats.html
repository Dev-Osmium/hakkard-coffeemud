037/**
038 * An interface for a base player account. Shared by PlayerAccount
039 * and PlayerStats (since the account system is optional)
040 */
041public interface AccountStats extends CMCommon
042{
043        /**
044         * Returns the players email address, if available.
045         *
046         * @see com.planet_ink.coffee_mud.Common.interfaces.AccountStats#setEmail(String)
047         *
048         * @return the players email address, if available.
049         */
050        public String getEmail();
051
052        /**
053         * Sets the players email address, if available.
054         *
055         * @see com.planet_ink.coffee_mud.Common.interfaces.AccountStats#getEmail()
056         *
057         * @param newAdd the players email address, if available.
058         */
059        public void setEmail(String newAdd);
060
061        /**
062         * The time, in milis since 1970, that the player last logged off.
063         *
064         * @see com.planet_ink.coffee_mud.Common.interfaces.AccountStats#setLastDateTime(long)
065         *
066         * @return time, in milis since 1970, that the player last logged off.
067         */
068        public long getLastDateTime();
069
070        /**
071         * Sets the time, in milis since 1970, that the player last logged off.
072         *
073         * @see com.planet_ink.coffee_mud.Common.interfaces.AccountStats#getLastDateTime()
074         *
075         * @param C the time, in milis since 1970, that the player last logged off.
076         */
077        public void setLastDateTime(long C);
078
079        /**
080         * The time, in milis since 1970, that the player was last saved.
081         *
082         * @see com.planet_ink.coffee_mud.Common.interfaces.AccountStats#setLastUpdated(long)
083         *
084         * @return the time, in milis since 1970, that the player was last saved.
085         */
086        public long getLastUpdated();
087
088        /**
089         * Sets the time, in milis since 1970, that the player was last saved.
090         *
091         * @see com.planet_ink.coffee_mud.Common.interfaces.AccountStats#getLastUpdated()
092         *
093         * @param time the time, in milis since 1970, that the player was last saved.
094         */
095        public void setLastUpdated(long time);
096
097        /**
098         * Returns the players password, perhaps encoded, perhaps plain text.
099         * Use matchesPassword(String) to do actual checks.
100         *
101         * @see com.planet_ink.coffee_mud.Common.interfaces.AccountStats#setPassword(String)
102         * @see com.planet_ink.coffee_mud.Common.interfaces.AccountStats#matchesPassword(String)
103         *
104         * @return the players password.
105         */
106        public String getPasswordStr();
107
108        /**
109         * Returns the players password, perhaps encoded, perhaps plain text.
110         * Use matchesPassword(String) to do actual checks.
111         *
112         * @see com.planet_ink.coffee_mud.Common.interfaces.AccountStats#setPassword(String)
113         * @see com.planet_ink.coffee_mud.Common.interfaces.AccountStats#getPasswordStr()
114         * @param str the password to compare reality to
115         * @return the players password.
116         */
117        public boolean matchesPassword(String str);
118
119        /**
120         * Sets the players password.
121         *
122         * @see com.planet_ink.coffee_mud.Common.interfaces.AccountStats#getPasswordStr()
123         * @see com.planet_ink.coffee_mud.Common.interfaces.AccountStats#matchesPassword(String)
124         *
125         * @param newPassword the players password.
126         */
127        public void setPassword(String newPassword);
128
129        /**
130         * The last IP address this player logged in from.
131         *
132         * @see com.planet_ink.coffee_mud.Common.interfaces.AccountStats#setLastIP(String)
133         *
134         * @return the last IP address this player logged in from.
135         */
136        public String getLastIP();
137
138        /**
139         * Sets the last IP address this player logged in from.
140         *
141         * @see com.planet_ink.coffee_mud.Common.interfaces.AccountStats#getLastIP()
142         *
143         * @param ip the last IP address this player logged in from.
144         */
145        public void setLastIP(String ip);
146
147        /**
148         * If the ACCOUNT system is used, this returns the time, in milis since
149         * 1970, that this account will expire (meaning the player will no longer
150         * be able to log in)
151         *
152         * @see com.planet_ink.coffee_mud.Common.interfaces.AccountStats#setAccountExpiration(long)
153         *
154         * @return the time, in milis, that this player expires.
155         */
156        public long getAccountExpiration();
157
158        /**
159         * If the ACCOUNT system is used, this sets the time, in milis since
160         * 1970, that this account will expire (meaning the player will no longer
161         * be able to log in)
162         *
163         * @see com.planet_ink.coffee_mud.Common.interfaces.AccountStats#getAccountExpiration()
164         *
165         * @param newVal the time, in milis, that this player expires.
166         */
167        public void setAccountExpiration(long newVal);
168
169        /**
170         * Returns the administrative notes entered about this player.
171         *
172         * @see com.planet_ink.coffee_mud.Common.interfaces.AccountStats#setNotes(String)
173         *
174         * @return the administrative notes entered about this player.
175         */
176        public String getNotes();
177
178        /**
179         * Sets the administrative notes entered about this player.
180         *
181         * @see com.planet_ink.coffee_mud.Common.interfaces.AccountStats#getNotes()
182         *
183         * @param newnotes the administrative notes entered about this player.
184         */
185        public void setNotes(String newnotes);
186
187        /**
188         * Returns the number of bonus char stat points to assign to stats
189         * during character creation.  This is independent of the account
190         * bonus.
191         * @see AccountStats#setBonusCharStatPoints(int)
192         *
193         * @return the number of bonus charstat points
194         */
195        public int getBonusCharStatPoints();
196
197        /**
198         * Sets the number of bonus char stat points to assign to stats
199         * during character creation.  This is independent of the account
200         * bonus.
201         * @see AccountStats#getBonusCharStatPoints()
202         *
203         * @param bonus the number of bonus charstat points
204         */
205        public void setBonusCharStatPoints(int bonus);
206
207        /**
208         * Returns the number of bonus common skills available to
209         * this character.  This is independent of the account
210         * bonus.
211         *
212         * @see AccountStats#setBonusCommonSkillLimits(int)
213         *
214         * @return the number of bonus Common points
215         */
216        public int getBonusCommonSkillLimits();
217
218        /**
219         * Sets the number of bonus common skills available to
220         * this character.  This is independent of the account
221         * bonus.
222         *
223         * @see AccountStats#getBonusCommonSkillLimits()
224         *
225         * @param bonus the number of bonus Common points
226         */
227        public void setBonusCommonSkillLimits(int bonus);
228
229        /**
230         * Returns the number of bonus Crafting skills available to
231         * this character.  This is independent of the account
232         * bonus.
233         *
234         * @see AccountStats#setBonusCraftingSkillLimits(int)
235         *
236         * @return the number of bonus Crafting points
237         */
238        public int getBonusCraftingSkillLimits();
239
240        /**
241         * Sets the number of bonus Crafting skills available to
242         * this character.  This is independent of the account
243         * bonus.
244         *
245         * @see AccountStats#getBonusCraftingSkillLimits()
246         *
247         * @param bonus the number of bonus Crafting points
248         */
249        public void setBonusCraftingSkillLimits(int bonus);
250
251        /**
252         * Returns the number of bonus Gathering skills available to
253         * this character.  This is independent of the account
254         * bonus.
255         *
256         * @see AccountStats#setBonusNonCraftingSkillLimits(int)
257         *
258         * @return the number of bonus Gathering points
259         */
260        public int getBonusNonCraftingSkillLimits();
261
262        /**
263         * Sets the number of bonus Gathering skills available to
264         * this character.  This is independent of the account
265         * bonus.
266         *
267         * @see AccountStats#getBonusNonCraftingSkillLimits()
268         *
269         * @param bonus the number of bonus Gathering points
270         */
271        public void setBonusNonCraftingSkillLimits(int bonus);
272
273        /**
274         * Returns the number of bonus Language skills available to
275         * this character.  This is independent of the account
276         * bonus.
277         *
278         * @see AccountStats#setBonusLanguageLimits(int)
279         *
280         * @return the number of bonus Language points
281         */
282        public int getBonusLanguageLimits();
283
284        /**
285         * Sets the number of bonus Language skills available to
286         * this character.  This is independent of the account
287         * bonus.
288         *
289         * @see AccountStats#getBonusLanguageLimits()
290         *
291         * @param bonus the number of bonus Language points
292         */
293        public void setBonusLanguageLimits(int bonus);
294
295        /**
296         * Returns a modifiable Set that contains the set of player
297         * Names that constitutes this players friends.
298         *
299         * @return a set of player friend names
300         */
301        public Set<String> getFriends();
302
303        /**
304         * Returns a modifiable Set that contains the set of player
305         * Names that constitutes this players ignored player list.
306         * @see AccountStats#isIgnored(String)
307         * @see AccountStats#isIgnored(MOB)
308         *
309         * @return a set of player ignored player list Names
310         */
311        public Set<String> getIgnored();
312
313        /**
314         * Returns whether the given player is being ignored.
315         * @see AccountStats#getIgnored()
316         * @see AccountStats#isIgnored(String)
317         *
318         * @param mob the mob to check
319         * @return true if the given mob is ignored
320         */
321        public boolean isIgnored(final MOB mob);
322
323        /**
324         * Returns whether the given player name is being ignored.
325         * @see AccountStats#getIgnored()
326         * @see AccountStats#isIgnored(MOB)
327         *
328         * @param name the name to check
329         * @return true if the given name is ignored
330         */
331        public boolean isIgnored(final String name);
332
333        /**
334         * Add to one of the pride stats for this player or account
335         * @see PrideStat
336         * @param stat which pride stat to add to
337         * @param amt the amount to add
338         */
339        public void bumpPrideStat(PrideStat stat, int amt);
340
341        /**
342         * Get one of the pride stats for this player or account
343         * @see PrideStat
344         * @param period the time period to get the number for
345         * @param stat which pride stat to get
346         * @return the pride stat value/count/whatever
347         */
348        public int getPrideStat(TimeClock.TimePeriod period, PrideStat stat);
349
350        /**
351         * Returns the tracker for the given achievement, for the given mob, or
352         * creates it if it does not exist.
353         * @see AccountStats#rebuildAchievementTracker(MOB, String)
354         * see #killAchievementTracker(Achievement, MOB)
355         * @see com.planet_ink.coffee_mud.Libraries.interfaces.AchievementLibrary.Tracker
356         * @see com.planet_ink.coffee_mud.Libraries.interfaces.AchievementLibrary.Achievement
357         * @param A the achievement to get the tracker for
358         * @param mob the mob to create a tracker for
359         * @return the Tracker object that handles this achievement/mob
360         */
361        public Tracker getAchievementTracker(Achievement A, MOB mob);
362
363        /**
364         * Deletes the tracker for the given achievement, for the given mob.
365         * @see AccountStats#rebuildAchievementTracker(MOB, String)
366         * @see com.planet_ink.coffee_mud.Libraries.interfaces.AchievementLibrary.Tracker
367         * @see com.planet_ink.coffee_mud.Libraries.interfaces.AchievementLibrary.Achievement
368         * @param A the achievement to kill the tracker for
369         * @param mob the mob to delete a tracker for
370         */
371        public void killAchievementTracker(Achievement A, MOB mob);
372
373        /**
374         * If an Achievement is modified or removed, this method will update the
375         * internal player tracker for that achievement.  It does not delete old
376         * achievements per se, just their trackers!
377         * @see AccountStats#getAchievementTracker(com.planet_ink.coffee_mud.Libraries.interfaces.AchievementLibrary.Achievement, MOB)
378         * @param mob the mob to modify the tracker for.
379         * @param achievementTattoo the tattoo/id of the achievement
380         */
381        public void rebuildAchievementTracker(final MOB mob, String achievementTattoo);
382
383        /**
384         * Returns an XML representation of all the data in this object, for
385         * persistent storage.
386         *
387         * @see com.planet_ink.coffee_mud.Common.interfaces.AccountStats#setXML(String)
388         *
389         * @return an XML representation of all the data in this object
390         */
391        public String getXML();
392
393        /**
394         * Restores the data in this object from an XML document.
395         *
396         * @see com.planet_ink.coffee_mud.Common.interfaces.AccountStats#getXML()
397         *
398         * @param str an XML representation of all the data in this object
399         */
400        public void setXML(String str);
401
402        /**
403         * The recorded player and account statistics.
404         * @author Bo Zimmerman
405         *
406         */
407        public enum PrideStat
408        {
409                PVPKILLS,
410                AREAS_EXPLORED,
411                ROOMS_EXPLORED,
412                EXPERIENCE_GAINED,
413                MINUTES_ON,
414                QUESTS_COMPLETED,
415                QUESTPOINTS_EARNED
416        }
417
418        /**
419         * A simple enum for picking between a player and an account
420         * @author Bo Zimmerman
421         *
422         */
423        public enum Agent
424        {
425                PLAYER,
426                ACCOUNT
427        }
