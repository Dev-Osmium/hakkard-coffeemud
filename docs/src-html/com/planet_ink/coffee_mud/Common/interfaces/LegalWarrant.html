033/**
034 * A LegalWarrant is an important object in the triad of LegalBehavior
035 * and Law objects, as a part of the CoffeeMud legal system.  A LegalWarrant
036 * is created by the LegalBehavior in response to the Law definitions, when
037 * it has determined that a mob or player has broken one of the Laws.  The
038 * Warrant is then used as a state object following the specific circumstances
039 * around the violated law, and its adjudication.  After adjudication, the
040 * object is transferred into an Old Warrants cue under the Law to use as
041 * a reference for repeat offenders.
042 *
043 * @see com.planet_ink.coffee_mud.Behaviors.interfaces.LegalBehavior
044 * @see com.planet_ink.coffee_mud.Common.interfaces.Law
045 */
046public interface LegalWarrant extends CMCommon
047{
048        /**
049         * If the criminal has been identified by an otherwise unoccupied
050         * officer, this will be that officer.
051         *
052         * @see com.planet_ink.coffee_mud.Common.interfaces.LegalWarrant#setArrestingOfficer(Area, MOB)
053         *
054         * @return the actual living mob officer who will enforce the warrant
055         */
056        public MOB arrestingOfficer();
057
058        /**
059         * If the criminal has been identified by an otherwise unoccupied
060         * officer, this will set that officer.
061         *
062         * @see com.planet_ink.coffee_mud.Common.interfaces.LegalWarrant#arrestingOfficer()
063         *
064         * @param legalArea the Area governing the law
065         * @param mob the actual living mob officer who will enforce the warrant
066         */
067        public void setArrestingOfficer(Area legalArea, MOB mob);
068
069        /**
070         * The actual player/mob criminal accused of the crime.
071         *
072         * @see com.planet_ink.coffee_mud.Common.interfaces.LegalWarrant#setCriminal(MOB)
073         *
074         * @return the actual player/mob criminal accused of the crime.
075         */
076        public MOB criminal();
077
078        /**
079         * Sets the actual player/mob criminal accused of the crime.
080         *
081         * @see com.planet_ink.coffee_mud.Common.interfaces.LegalWarrant#criminal()
082         *
083         * @param mob the actual player/mob criminal accused of the crime.
084         */
085        public void setCriminal(MOB mob);
086
087        /**
088         * If applicable, the victim of the crime (or null)
089         *
090         * @see com.planet_ink.coffee_mud.Common.interfaces.LegalWarrant#setVictim(MOB)
091         *
092         * @return the victim of the crime, or null
093         */
094        public MOB victim();
095
096        /**
097         * If applicable, sets the victim of the crime (or null)
098         *
099         * @see com.planet_ink.coffee_mud.Common.interfaces.LegalWarrant#victim()
100         *
101         * @param mob the victim of the crime, or null
102         */
103        public void setVictim(MOB mob);
104
105        /**
106         * If applicable, gets the witness of the crime (or null)
107         *
108         * @see com.planet_ink.coffee_mud.Common.interfaces.LegalWarrant#setWitness(MOB)
109         *
110         * @return the witness of the crime or null
111         */
112        public MOB witness();
113
114        /**
115         * If applicable, sets the witness of the crime (or null)
116         *
117         * @see com.planet_ink.coffee_mud.Common.interfaces.LegalWarrant#witness()
118         *
119         * @param mob the witness of the crime (or null)
120         */
121        public void setWitness(MOB mob);
122
123        /**
124         * The crime name
125         *
126         * @see com.planet_ink.coffee_mud.Common.interfaces.LegalWarrant#setCrime(String)
127         * @see com.planet_ink.coffee_mud.Common.interfaces.Law#BIT_CRIMENAME
128         *
129         * @return the crime name
130         */
131        public String crime();
132
133        /**
134         * Sets the crime name
135         *
136         * @see com.planet_ink.coffee_mud.Common.interfaces.LegalWarrant#crime()
137         * @see com.planet_ink.coffee_mud.Common.interfaces.Law#BIT_CRIMENAME
138         *
139         * @param crime the crime name
140         */
141        public void setCrime(String crime);
142
143        /**
144         * Gets the full punishment code for the crime
145         *
146         * @see com.planet_ink.coffee_mud.Common.interfaces.LegalWarrant#setPunishment(int)
147         * @see com.planet_ink.coffee_mud.Common.interfaces.LegalWarrant#getPunishmentParm(int)
148         * @see com.planet_ink.coffee_mud.Common.interfaces.LegalWarrant#addPunishmentParm(int, String)
149         * @see com.planet_ink.coffee_mud.Common.interfaces.Law#PUNISHMENT_DESCS
150         * @see com.planet_ink.coffee_mud.Common.interfaces.Law#PUNISHMENTMASK_DESCS
151         *
152         * @return the full punishment code for the crime
153         */
154        public int punishment();
155
156        /**
157         * Sets the full punishment code for the crime
158         *
159         * @see com.planet_ink.coffee_mud.Common.interfaces.LegalWarrant#punishment()
160         * @see com.planet_ink.coffee_mud.Common.interfaces.LegalWarrant#getPunishmentParm(int)
161         * @see com.planet_ink.coffee_mud.Common.interfaces.LegalWarrant#addPunishmentParm(int, String)
162         * @see com.planet_ink.coffee_mud.Common.interfaces.Law#PUNISHMENT_DESCS
163         * @see com.planet_ink.coffee_mud.Common.interfaces.Law#PUNISHMENTMASK_DESCS
164         *
165         * @param code the full punishment code for the crime
166         */
167        public void setPunishment(int code);
168
169        /**
170         * If applicable, returns any parameters associated with a particular
171         * bitmask on the punishment.
172         *
173         * @see com.planet_ink.coffee_mud.Common.interfaces.LegalWarrant#addPunishmentParm(int, String)
174         * @see com.planet_ink.coffee_mud.Common.interfaces.LegalWarrant#punishment()
175         * @see com.planet_ink.coffee_mud.Common.interfaces.LegalWarrant#setPunishment(int)
176         * @see com.planet_ink.coffee_mud.Common.interfaces.Law#PUNISHMENTMASK_DESCS
177         *
178         * @param code the punishment mask bitmap code
179         * @return any parameters needed, or null
180         */
181        public String getPunishmentParm(int code);
182
183        /**
184         * If applicable, sets any parameters associated with a particular
185         * bitmask on the punishment.
186         *
187         * @see com.planet_ink.coffee_mud.Common.interfaces.LegalWarrant#getPunishmentParm(int)
188         * @see com.planet_ink.coffee_mud.Common.interfaces.LegalWarrant#punishment()
189         * @see com.planet_ink.coffee_mud.Common.interfaces.LegalWarrant#setPunishment(int)
190         * @see com.planet_ink.coffee_mud.Common.interfaces.Law#PUNISHMENTMASK_DESCS
191         *
192         * @param code the punishment mask bitmap code
193         * @param parm the parameter to set
194         */
195        public void addPunishmentParm(int code, String parm);
196
197        /**
198         * If applicable, returns the number of ticks the punishment calls
199         * for the criminal to remain in jail.
200         *
201         * @see com.planet_ink.coffee_mud.Common.interfaces.LegalWarrant#setJailTime(int)
202         *
203         * @return the number of ticks to be jailed for
204         */

206
207        /**
208         * If applicable, sets the number of ticks the punishment calls
209         * for the criminal to remain in jail.
210         *
211         * @see com.planet_ink.coffee_mud.Common.interfaces.LegalWarrant#jailTime()
212         *
213         * @param time the number of ticks to be jailed for
214         */
215        public void setJailTime(int time);
216
217        /**
218         * Returns the current state of adjudication.
219         *
220         * @see com.planet_ink.coffee_mud.Common.interfaces.LegalWarrant#setState(int)

222         * @see com.planet_ink.coffee_mud.Common.interfaces.Law#STATE_SEEKING
223         *
224         * @return the current state of adjudication
225         */

227
228        /**
229         * Sets the current state of adjudication.
230         *
231         * @see com.planet_ink.coffee_mud.Common.interfaces.LegalWarrant#state()
232         * @see com.planet_ink.coffee_mud.Common.interfaces.LegalWarrant#getLastStateChangeTime()
233         * @see com.planet_ink.coffee_mud.Common.interfaces.Law#STATE_DETAINING
234         * @see com.planet_ink.coffee_mud.Common.interfaces.Law#STATE_EXECUTING
235         *
236         * @param state the current state of adjudication.
237         */
238        public void setState(int state);
239
240        /**
241         * Returns the number of times this criminal has done this crime.  This
242         * is usually 0 unless it also exists in the old warrants list.
243         *
244         * @see com.planet_ink.coffee_mud.Common.interfaces.LegalWarrant#setOffenses(int)
245         * @see com.planet_ink.coffee_mud.Common.interfaces.Law#oldWarrants()
246         *
247         * @return the number of times this criminal has committed this crime.
248         */
249        public int offenses();
250
251        /**
252         * Sets the number of times this criminal has done this crime.  This
253         * is usually 0 unless it also exists in the old warrants list.
254         *
255         * @see com.planet_ink.coffee_mud.Common.interfaces.LegalWarrant#offenses()
256         * @see com.planet_ink.coffee_mud.Common.interfaces.Law#oldWarrants()
257         *
258         * @param num the number of times this criminal has committed this crime.
259         */
260        public void setOffenses(int num);
261
262        /**
263         * Returns the last time, in miliseconds since 1970, the criminal has
264         * committed this crime.  Very useful in oldWarrants
265         *
266         * @see com.planet_ink.coffee_mud.Common.interfaces.LegalWarrant#setLastOffense(long)
267         * @see com.planet_ink.coffee_mud.Common.interfaces.Law#oldWarrants()
268         *
269         * @return the last time this crime was done, in milis
270         */
271        public long lastOffense();
272
273        /**
274         * Sets the last time, in miliseconds since 1970, the criminal has
275         * committed this crime.  Very useful in oldWarrants
276         *
277         * @see com.planet_ink.coffee_mud.Common.interfaces.LegalWarrant#lastOffense()
278         * @see com.planet_ink.coffee_mud.Common.interfaces.Law#oldWarrants()
279         *
280         * @param last the last time this crime was done, in milis
281         */
282        public void setLastOffense(long last);
283
284        /**
285         * Returns the time, in milis since 1970, that a trip to the judge
286         * or to the jail was started.  For housekeeping purposes.
287         *
288         * @see com.planet_ink.coffee_mud.Common.interfaces.LegalWarrant#setTravelAttemptTime(long)
289         *
290         * @return the milis time since the trip began
291         */

293
294        /**
295         * Sets the time, in milis since 1970, that a trip to the judge

297         *
298         * @see com.planet_ink.coffee_mud.Common.interfaces.LegalWarrant#travelAttemptTime()
299         *
300         * @param time the milis time since the trip began
301         */
302        public void setTravelAttemptTime(long time);
303
304        /**
305         * Returns the warning message given to the criminal by the officer.
306         *
307         * @see com.planet_ink.coffee_mud.Common.interfaces.LegalWarrant#setWarnMsg(String)
308         *
309         * @return the warning msg given
310         */
311        public String warnMsg();
312
313        /**
314         * Sets the warning message given to the criminal by the officer.
315         *
316         * @see com.planet_ink.coffee_mud.Common.interfaces.LegalWarrant#warnMsg()
317         *
318         * @param msg the warning msg given
319         */
320        public void setWarnMsg(String msg);
321
322        /**
323         * Sets the room into which this criminal will be jailed.
324         *
325         * @see com.planet_ink.coffee_mud.Common.interfaces.LegalWarrant#jail()
326         *
327         * @param R the room into which this criminal will be jailed.
328         */
329        public void setJail(Room R);
330
331        /**
332         * Returns the room into which this criminal will be jailed.
333         *
334         * @see com.planet_ink.coffee_mud.Common.interfaces.LegalWarrant#setJail(Room)
335         *
336         * @return the room into which this criminal will be jailed.
337         */
338        public Room jail();
339
340        /**
341         * Returns the room into which this criminal will be released after jail.
342         *
343         * @see com.planet_ink.coffee_mud.Common.interfaces.LegalWarrant#setReleaseRoom(Room)
344         *
345         * @return the room into which this criminal will be released after jail.
346         */
347        public Room releaseRoom();
348
349        /**
350         * Sets the room into which this criminal will be released after jail.
351         *
352         * @see com.planet_ink.coffee_mud.Common.interfaces.LegalWarrant#releaseRoom()
353         *
354         * @param R the room into which this criminal will be released after jail.
355         */
356        public void setReleaseRoom(Room R);
357        
358        /**
359         * Returns the real time in ms when the state last changed.
360         * 
361         * @see com.planet_ink.coffee_mud.Common.interfaces.LegalWarrant#setState(int)
362         *
363         * @return the real time in ms when the state last changed.
364         */
365        public long getLastStateChangeTime();
366
367        /**
368         * Returns the real time in ms when the warrant can no longer be ignored
369         * 
370         * @see com.planet_ink.coffee_mud.Common.interfaces.LegalWarrant#setIgnoreUntilTime(long)
371         *
372         * @return the real time in ms when the warrant can no longer be ignored
373         */
374        public long getIgnoreUntilTime();
375        
376        /**
377         * Sets the real time in ms when the warrant can no longer be ignored
378         * 
379         * @see com.planet_ink.coffee_mud.Common.interfaces.LegalWarrant#getIgnoreUntilTime()
380         *
381         * @param time the real time in ms when the warrant can no longer be ignored
382         */
383        public void setIgnoreUntilTime(long time);
