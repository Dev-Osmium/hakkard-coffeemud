020import com.planet_ink.coffee_mud.Items.interfaces.Item;
021import com.planet_ink.coffee_mud.MOBS.interfaces.Auctioneer;
022import com.planet_ink.coffee_mud.MOBS.interfaces.MOB;
023
024/**
025 * Class for storing basic data about a specific auction in progress.
026 * Handles both Live and Timed auction data.
027 * 
028 * @author Bo Zimmerman
029 */
030public interface AuctionData extends CMCommon
031{
032        /**
033         * Returns the number of days remaining in this
034         * auction.  The mobs are only for determining
035         * which calendar to use.  The first mobs'
036         * local calendar gets priority.
037         * @param mob the mob whose calendar to use
038         * @param mob2 the second mob whose calendar to use
039         * @return the number of days remaining in the auction
040         */
041        public int daysRemaining(MOB mob, MOB mob2);
042
043        /**
044         * Returns the number of days ellapsed in this
045         * auction.  The mobs are only for determining
046         * which calendar to use.  The first mobs'
047         * local calendar gets priority.
048         * @param mob the mob whose calendar to use
049         * @param mob2 the second mob whose calendar to use
050         * @return the number of days ellapsed in the auction
051         */
052        public int daysEllapsed(MOB mob, MOB mob2);
053
054        /**
055         * Get the item being auctioned.
056         * @see AuctionData#setAuctionedItem(Item)
057         * @return the item being auctioned.
058         */
059        public Item getAuctionedItem();
060
061        /**
062         * Set the item being auctioned.
063         * @see AuctionData#getAuctionedItem()
064         * @param auctionedItem the item being auctioned.
065         */
066        public void setAuctionedItem(Item auctionedItem);
067
068        /**
069         * Get the mob auctioning the item
070         * @see AuctionData#setAuctioningMob(MOB)
071         * @return the mob auctioning the item
072         */
073        public MOB getAuctioningMob();
074
075        /**
076         * Set the mob auctioning the item
077         * @see AuctionData#getAuctioningMob()
078         * @param auctioningM the mob auctioning the item
079         */
080        public void setAuctioningMob(MOB auctioningM);
081
082        /**
083         * Get the mob who is the high bidder in the auction.
084         * @see AuctionData#setHighBidderMob(MOB)
085         * @return the mob who is the high bidder in the auction.
086         */
087        public MOB getHighBidderMob();
088
089        /**
090         * Set the mob who is the high bidder in the auction.
091         * @see AuctionData#getHighBidderMob()
092         * @param highBidderM the mob who is the high bidder in the auction.
093         */
094        public void setHighBidderMob(MOB highBidderM);
095
096        /**
097         * Get the currency the auction is being held in.
098         * @see AuctionData#setCurrency(String)
099         * @return the currency the auction is being held in.
100         */
101        public String getCurrency();
102
103        /**
104         * Set the currency the auction is being held in.
105         * @see AuctionData#getCurrency()
106         * @param currency the currency the auction is being held in.
107         */
108        public void setCurrency(String currency);
109
110        /**
111         * Get the current high bid in the auction
112         * @see AuctionData#setHighBid(double)
113         * @return the current high bid in the auction
114         */
115        public double getHighBid();
116        
117        /**
118         * Set the current high bid in the auction
119         * @see AuctionData#getHighBid()
120         * @param highBid the current high bid in the auction
121         */
122        public void setHighBid(double highBid);
123
124        /**
125         * Get the current bid in the auction
126         * @see AuctionData#setBid(double)
127         * @return the current bid in the auction
128         */
129        public double getBid();
130
131        /**
132         * Set the current bid in the auction
133         * @see AuctionData#getBid()
134         * @param bid the current bid in the auction
135         */
136        public void setBid(double bid);
137
138        /**
139         * Get the current buy-out price for this auction. 
140         * @see AuctionData#setBuyOutPrice(double)
141         * @return the current buy-out price for this auction.
142         */
143        public double getBuyOutPrice();
144
145        /**
146         * Set the current buy-out price for this auction. 
147         * @see AuctionData#getBuyOutPrice()
148         * @param buyOutPrice the current buy-out price for this auction.
149         */
150        public void setBuyOutPrice(double buyOutPrice);
151
152        /**
153         * Get the current auction State
154         * @see AuctionData#setAuctionState(int)
155         * @return the current auction State
156         */
157        public int getAuctionState();
158
159        /**
160         * Set the current auction State
161         * @see AuctionData#getAuctionState()
162         * @param state the current auction State
163         */
164        public void setAuctionState(int state);
165
166        /**
167         * Get the tick down timer to check stuff
168         * @see AuctionData#setAuctionTickDown(long)
169         * @return the tick down timer to check stuff
170         */
171        public long getAuctionTickDown();
172
173        /**
174         * Set the tick down timer to check stuff
175         * @see AuctionData#getAuctionTickDown()
176         * @param tickDown the tick down timer to check stuff
177         */
178        public void setAuctionTickDown(long tickDown);
179
180        /**
181         * Get the start time of this auction
182         * @see AuctionData#setStartTime(long)
183         * @return the start time of this auction
184         */
185        public long getStartTime();
186
187        /**
188         * Set the start time of this auction
189         * @see AuctionData#getStartTime()
190         * @param start the start time of this auction
191         */
192        public void setStartTime(long start);
193
194        /**
195         * Get the auction database key for this auction
196         * @see AuctionData#setAuctionDBKey(String)
197         * @return the auction database key for this auction
198         */
199        public String getAuctionDBKey();
200
201        /**
202         * Set the auction database key for this auction
203         * @see AuctionData#getAuctionDBKey()
204         * @param auctionDBKey the auction database key for this auction
205         */
206        public void setAuctionDBKey(String auctionDBKey);
207}
