033/**
034 * A climate represents the entire weather pattern of an area
035 * It does most of the mundane things related to weather
036 * changes based on season, area, and other factors.
037 */
038public interface Climate extends Tickable, CMCommon
039{
040        /**
041         * Returns the specific weather situation for
042         * the given room.  Returns one of the Climate.WEATHER_*
043         * constants.  It always returns WEATHER_CLEAR for
044         * indoor locales.
045         * @see Climate
046         * @see Climate#setCurrentWeatherType(int)
047         * @param room the room to evaluate.
048         * @return the Climate.WEATHER_* constant
049         */
050        public int weatherType(Room room);
051        /**
052         * Returns the upcoming specific weather situation for
053         * the given room.  Returns one of the Climate.WEATHER_*
054         * constants.  It always returns WEATHER_CLEAR for
055         * indoor locales.  Climate objects always calculate
056         * weather 1 step ahead for predictive purposes.
057         * @see Climate
058         * @see Climate#setNextWeatherType(int)
059         * @param room the room to evaluate.
060         * @return the Climate.WEATHER_* constant
061         */
062        public int nextWeatherType(Room room);
063        /**
064         * Sets the upcoming specific weather situation for
065         * this area.  Takes one of the Climate.WEATHER_*
066         * constants. Climate objects always calculate
067         * weather 1 step ahead for predictive purposes.
068         * @see Climate
069         * @see Climate#nextWeatherType(Room)
070         * @param weatherCode the Climate.WEATHER_* constant
071         */
072        public void setNextWeatherType(int weatherCode);
073        /**
074         * Sets the current specific weather situation for
075         * this area.  Takes one of the Climate.WEATHER_*
076         * constants. Climate objects always calculate
077         * weather 1 step ahead for predictive purposes.
078         * @see Climate
079         * @see Climate#weatherType(Room)
080         * @param weatherCode the Climate.WEATHER_* constant
081         */
082        public void setCurrentWeatherType(int weatherCode);
083        /**
084         * Returns a readable description of the weather
085         * for the given room.
086         * @param room the room to evaluate
087         * @return a readable string.
088         */
089        public String weatherDescription(Room room);
090        /**
091         * Returns a readable description of the weather
092         * that will be upcoming for the given room.
093         * @param room the room to evaluate
094         * @return a readable string.
095         */
096        public String nextWeatherDescription(Room room);
097        /**
098         * Returns whether a player can see the stars from
099         * the current room.
100         * @param room the room vantage point
101         * @return whether the stars can be seen.
102         */
103        public boolean canSeeTheStars(Room room);
104        /**
105         * Returns whether a player can see the moon from
106         * the current room.
107         * @param room the room vantage point
108         * @param butNotA an ability to exempt from moon flags
109         * @return whether the moon can be seen.
110         */
111        public boolean canSeeTheMoon(Room room, Ability butNotA);
112        /**
113         * Returns whether a player can see the sun from
114         * the current room.
115         * @param room the room vantage point
116         * @return whether the sun can be seen.
117         */
118        public boolean canSeeTheSun(Room room);
119        /**
120         * Returns a readable string describing the
121         * weather conditions in the given area, assuming
122         * this climate is the correct one.
123         * @param A the area to evaluate
124         * @return a description of the weather.
125         */
126        public String getWeatherDescription(Area A);
127        /**
128         * Returns a readable string describing the upcoming
129         * weather conditions in the given area, assuming
130         * this climate is the correct one.
131         * @param A the area to evaluate
132         * @return a description of the weather coming up.
133         */
134        public String getNextWeatherDescription(Area A);
135        /**
136         * This method forces the weather to change to what
137         * is designated by the nextWeatherType().  The Area
138         * is required to calculate a new nextWeatherType.
139         * @param A the area to calculate next weather for.
140         */
141        public void forceWeatherTick(Area A);
142        /**
143         * If applicable, thies method will return the base water
144         * thirstiness of the given mob, modified up or down based
145         * on the room given.
146         * @param base the water thirst gain to start with
147         * @param room the room the mob is in
148         * @return the modified base thirst, after gain or loss
149         */
150        public int adjustWaterConsumption(int base, Room room);
151        /**
152         * If applicable, thies method will return the base movement
153         * cost of the given mob, modified up or down based
154         * on the room given.
155         * @param base the movement  loss to start with
156         * @param room the room the mob is moving through
157         * @return the modified movement, after gain or loss
158         */
159        public int adjustMovement(int base, Room room);
160        /** constant describing clear or no weather. @see Climate#weatherType(Room) */
161        public final static int WEATHER_CLEAR=0;
162        /** constant describing  cloudy weather. @see Climate#weatherType(Room) */
163        public final static int WEATHER_CLOUDY=1;
164        /** constant describing windy weather. @see Climate#weatherType(Room) */
165        public final static int WEATHER_WINDY=2;
166        /** constant describing rainy weather. @see Climate#weatherType(Room) */
167        public final static int WEATHER_RAIN=3;
168        /** constant describing thunderstorm weather. @see Climate#weatherType(Room) */
169        public final static int WEATHER_THUNDERSTORM=4;
170        /** constant describing snowy weather. @see Climate#weatherType(Room) */
171        public final static int WEATHER_SNOW=5;
172        /** constant describing haily weather. @see Climate#weatherType(Room) */
173        public final static int WEATHER_HAIL=6;
174        /** constant describing heat wave weather. @see Climate#weatherType(Room) */
175        public final static int WEATHER_HEAT_WAVE=7;
176        /** constant describing sleety weather. @see Climate#weatherType(Room) */
177        public final static int WEATHER_SLEET=8;
178        /** constant describing blizzardous weather. @see Climate#weatherType(Room) */
179        public final static int WEATHER_BLIZZARD=9;
180        /** constant describing duststorming weather. @see Climate#weatherType(Room) */
181        public final static int WEATHER_DUSTSTORM=10;
182        /** constant describing drought condition weather. @see Climate#weatherType(Room) */
183        public final static int WEATHER_DROUGHT=11;
184        /** constant describing harsh cold weather. @see Climate#weatherType(Room) */
185        public final static int WEATHER_WINTER_COLD=12;
186        /** constant describing the number of weather types. @see Climate#weatherType(Room) */
187        public final static int NUM_WEATHER=13;
188        /** constant defining how often weather changes in the climates */
189        public static final int WEATHER_TICK_DOWN=150; // 150 = 10 minutes * 60 seconds / 4
190        /**  descriptive strings for the Climate.WEATHER_* constants, ordered by their value. @see Climate */
191        public final static String[] WEATHER_DESCS=
192        { 
193                "CLEAR","CLOUDY","WINDY","RAIN","THUNDERSTORM","SNOW","HAIL","HEAT","SLEET","BLIZZARD","DUST","DROUGHT","COLD"
194        };
