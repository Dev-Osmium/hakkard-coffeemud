020/**
021 * Although at one time a simple public variable object, the Social became an environmental
022 * when it became import to be able to pass it as a tool of a message for various purposes.
023 * It still retains much of its original awkwardness as can be seen in the method names.
024 * Today the social is still mostly a variable storage unit, with a few methods to make use
025 * of those variables.
026 *
027 * A social object exists not simply as one for each social command (smile, wiggy) but one
028 * for each social variation (smile, smile self, smile bob).  The Name() field holds the
029 * full unique name of each social object, with ID only having "DefaultSocial".
030 *
031 * @see com.planet_ink.coffee_mud.core.interfaces.Environmental#Name()
032 * @see com.planet_ink.coffee_mud.core.interfaces.Environmental#setName(String)
033 *
034 * Managed by the SocialsList library.
035 *
036 * @see com.planet_ink.coffee_mud.Common.interfaces.Social
037 * @see com.planet_ink.coffee_mud.Libraries.interfaces.SocialsList
038 */
039
040public interface Social extends Environmental, CMCommon
041{
042
043        /**
044         * Base name is the name of the social's command word.  It is NOT unique to a social
045         * object however, since a socials name usually includes its target extension.
046         *
047         * @return the base name
048         */
049        public String baseName();
050
051        /**
052         * Tail name is the full remainder of the social's target AND optional argument(s).  
053         * It is NOT unique to a social object however, since a socials name usually 
054         * includes its target extension.
055         *
056         * @return the full tail name
057         */
058        public String tailName();
059
060        /**
061         * Target name is the social's target argument only, such as <T-NAME>.  
062         * It is NOT unique to a social object however, since a socials name usually 
063         * includes their target extension.
064         *
065         * @return the target name, if any
066         */
067        public String targetName();
068
069        /**
070         * Returns whether targetName is an -NAME> code, meaning it is targettable 
071         * at a person or mob or something.
072         *
073         * @return true if this social is targetable at someone other than the user
074         */
075        public boolean isTargetable();
076
077        /**
078         * Argument name is the social's optional remaining argument.  This requires
079         * that the social be targetable, and further specifies something about the
080         * target, or a qualifier.  
081         * It is NOT unique to a social object however, since a socials name usually 
082         * includes their target extension and name and stuff.
083         *
084         * @return the socials optional remaining argument.
085         */
086        public String argumentName();
087
088        /**
089         * You_see: what the player themself sees when they use this social.
090         *
091         * @see com.planet_ink.coffee_mud.Common.interfaces.Social#setSourceMessage(String)
092         * @return  what the player themself sees when they use this social.
093         */
094        public String getSourceMessage();
095
096        /**
097         * You_see: what the player themself sees when they use this social.
098         *
099         * @see com.planet_ink.coffee_mud.Common.interfaces.Social#getSourceMessage()
100         * @param str  what the player themself sees when they use this social.
101         */
102        public void setSourceMessage(String str);
103
104        /**
105         * Third_party_sees:  what everyone but the player and their target
106         * sees when a player uses this social.
107         *
108         * @see com.planet_ink.coffee_mud.Common.interfaces.Social#setOthersMessage(String)
109         * @return the string everyone else sees
110         */
111        public String getOthersMessage();
112
113        /**
114         * Third_party_sees:  what everyone but the player and their target
115         * sees when a player uses this social.
116         *
117         * @see com.planet_ink.coffee_mud.Common.interfaces.Social#getOthersMessage()
118         * @param str the string everyone else sees
119         */
120        public void setOthersMessage(String str);
121
122        /**
123         * Target_sees: what the target of this social sees when social is used on them.
124         *
125         * @see com.planet_ink.coffee_mud.Common.interfaces.Social#setTargetMessage(String)
126         * @return  what the target of this social sees when social is used on them
127         */
128        public String getTargetMessage();
129
130        /**
131         * Target_sees: what the target of this social sees when social is used on them.
132         *
133         * @see com.planet_ink.coffee_mud.Common.interfaces.Social#getTargetMessage()
134         * @param str what the target of this social sees when social is used on them
135         */
136        public void setTargetMessage(String str);
137
138        /**
139         * See_when_no_target: what the player sees when this social is used targeted, but
140         * the target is not available.
141         *
142         * @see com.planet_ink.coffee_mud.Common.interfaces.Social#setFailedMessage(String)
143         * @return the string seen by the player when the target has left
144         */
145        public String getFailedTargetMessage();
146
147        /**
148         * See_when_no_target: what the player sees when this social is used targeted, but
149         * the target is not available.
150         *
151         * @see com.planet_ink.coffee_mud.Common.interfaces.Social#getFailedTargetMessage()
152         * @param str the string seen by the player when the target has left
153         */
154        public void setFailedMessage(String str);
155
156        /**
157         * The official CoffeeMud message code used as the source code for this social,
158         * which reflects what the player is doing.  Only certain codes allowed.
159         *
160         * @see com.planet_ink.coffee_mud.Common.interfaces.Social#setSourceCode(int)
161         * @see com.planet_ink.coffee_mud.Common.interfaces.CMMsg#MSG_SPEAK
162         * @see com.planet_ink.coffee_mud.Common.interfaces.CMMsg#MSG_HANDS
163         * @see com.planet_ink.coffee_mud.Common.interfaces.CMMsg#MSG_NOISE
164         * @see com.planet_ink.coffee_mud.Common.interfaces.CMMsg#MSG_NOISYMOVEMENT
165         * @return the CMMsg social code, fully formed
166         */
167        public int getSourceCode();
168
169        /**
170         * Sets the official CoffeeMud message code used as the source code for this social,
171         * which reflects what the player is doing.  Only certain codes allowed.
172         *
173         * @see com.planet_ink.coffee_mud.Common.interfaces.Social#getSourceCode()
174         * @see com.planet_ink.coffee_mud.Common.interfaces.CMMsg#MSG_SPEAK
175         * @see com.planet_ink.coffee_mud.Common.interfaces.CMMsg#MSG_HANDS
176         * @see com.planet_ink.coffee_mud.Common.interfaces.CMMsg#MSG_NOISE
177         * @see com.planet_ink.coffee_mud.Common.interfaces.CMMsg#MSG_NOISYMOVEMENT
178         * @param code the CMMsg social code, fully formed
179         */
180        public void setSourceCode(int code);
181
182        /**
183         * The official CoffeeMud message code used as the others code for this social,
184         * which reflects what the player appears to be doing.  Only certain codes allowed.
185         *
186         * @see com.planet_ink.coffee_mud.Common.interfaces.Social#setOthersCode(int)
187         * @see com.planet_ink.coffee_mud.Common.interfaces.CMMsg#MSG_SPEAK
188         * @see com.planet_ink.coffee_mud.Common.interfaces.CMMsg#MSG_HANDS
189         * @see com.planet_ink.coffee_mud.Common.interfaces.CMMsg#MSG_NOISE
190         * @see com.planet_ink.coffee_mud.Common.interfaces.CMMsg#MSG_NOISYMOVEMENT
191         * @see com.planet_ink.coffee_mud.Common.interfaces.CMMsg#MSG_OK_VISUAL
192         * @return the CMMsg social code, fully formed
193         */
194        public int getOthersCode();
195
196        /**
197         * Sets the official CoffeeMud message code used as the others code for this social,
198         * which reflects what the player appears to be doing.  Only certain codes allowed.
199         *
200         * @see com.planet_ink.coffee_mud.Common.interfaces.Social#getOthersCode()
201         * @see com.planet_ink.coffee_mud.Common.interfaces.CMMsg#MSG_SPEAK
202         * @see com.planet_ink.coffee_mud.Common.interfaces.CMMsg#MSG_HANDS
203         * @see com.planet_ink.coffee_mud.Common.interfaces.CMMsg#MSG_NOISE
204         * @see com.planet_ink.coffee_mud.Common.interfaces.CMMsg#MSG_NOISYMOVEMENT
205         * @see com.planet_ink.coffee_mud.Common.interfaces.CMMsg#MSG_OK_VISUAL
206         *
207         * @param code the CMMsg social code, fully formed
208         */
209        public void setOthersCode(int code);
210
211        /**
212         * The official CoffeeMud message code used as the target code for this social,
213         * which reflects what is being done to the target.  Only certain codes allowed.
214         *
215         * @see com.planet_ink.coffee_mud.Common.interfaces.Social#setTargetCode(int)
216         * @see com.planet_ink.coffee_mud.Common.interfaces.CMMsg#MSG_SPEAK
217         * @see com.planet_ink.coffee_mud.Common.interfaces.CMMsg#MSG_HANDS
218         * @see com.planet_ink.coffee_mud.Common.interfaces.CMMsg#MSG_NOISE
219         * @see com.planet_ink.coffee_mud.Common.interfaces.CMMsg#MSG_NOISYMOVEMENT

221         * @return the CMMsg social code, fully formed
222         */
223        public int getTargetCode();
224
225        /**
226         * Sets the official CoffeeMud message code used as the target code for this social,
227         * which reflects what is being done to the target.  Only certain codes allowed.

229         * @see com.planet_ink.coffee_mud.Common.interfaces.Social#getTargetCode()

231         * @see com.planet_ink.coffee_mud.Common.interfaces.CMMsg#MSG_HANDS

236         */
237        public void setTargetCode(int code);
238
239        /**
240         * Whether this social variation can be targeted.  Is determined by whether
241         * there is a target parameter in the name.
242         *
243         * @see com.planet_ink.coffee_mud.core.interfaces.Environmental#Name()
244         * @param E The object to check to see if this is targetable against.
245         * @return true, if it can be targeted
246         */
247        public boolean targetable(Environmental E);
248
249        /**
250         * Returns the name of the MSP sound file associated with this social.
251         *
252         * @see com.planet_ink.coffee_mud.Common.interfaces.Social#setSoundFile(String)
253         * @return the msp sound file name
254         */
255        public String getSoundFile();
256
257        /**
258         * Sets the name of the MSP sound file associated with this social.
259         *
260         * @see com.planet_ink.coffee_mud.Common.interfaces.Social#getSoundFile()
261         * @param newFile the msp sound file name
262         */
263        public void setSoundFile(String newFile);
264
265        /**
266         * Executes this social by the given mob, using the target provided, with
267         * help from the provided command line strings in a vector,
268         * and with override message code flag.  It will generate a proper message
269         * and send it to the same room as the mob.
270         *
271         * @param mob the source of the social action
272         * @param commands the vector of strings
273         * @param target the target of the social
274         * @param auto true to override physical constraints of the source or target
275         *
276         * @return true, if successful
277         */
278        public boolean invoke(MOB mob,
279                                                  List<String> commands,
280                                                  Physical target,
281                                                  boolean auto);
282
283        /**
284         * Manufactures a proper channel message that represents executing this
285         * social over a public chat channel.
286         *
287         * @see com.planet_ink.coffee_mud.Common.interfaces.CMMsg
288         * @see com.planet_ink.coffee_mud.Common.interfaces.Social#makeMessage(MOB, String, String, int, int, List, String, boolean)
289         *
290         * @param mob the mob doing the social
291         * @param channelInt the channel int representing which chat channel
292         * @param channelName the channel name of the chat channel
293         * @param commands the commands as a string vector entered by the user
294         * @param makeTarget true to create a fake target, or false to find a real one.
295         *
296         * @return the CMMsg that can now be sent to the world as a chat channel message
297         */
298        public CMMsg makeChannelMsg(MOB mob,
299                                                                int channelInt,
300                                                                String channelName,
301                                                                List<String> commands,
302                                                                boolean makeTarget);
303
304        /**
305         * Manufactures a proper message that represents executing this social as an action
306         *
307         * @see com.planet_ink.coffee_mud.Common.interfaces.CMMsg
308         * @see com.planet_ink.coffee_mud.Common.interfaces.Social#makeChannelMsg(MOB, int, String, List, boolean)
309         *
310         * @param mob the mob doing the social
311         * @param str the str an optional "header" string for the msg, usually mxp related
312         * @param end the end an optional "trailer" string for the msg, usually mxp related
313         * @param srcMask the src mask to logically OR with the generated message source code
314         * @param fullCode the override target and others code from the one in this social
315         * @param commands the commands as a string vector entered by the user
316         * @param I3channelName the i3channel name or null if n/a
317         * @param makeTarget true to manufacture a target, or false to use a real one
318         *
319         * @return the CMMsg that can now be sent to the world as a chat channel message
320         */
321        public CMMsg makeMessage(MOB mob,
322                                                         String str,
323                                                         String end,
324                                                         int srcMask,
325                                                         int fullCode,
326                                                         List<String> commands,
327                                                         String I3channelName,
328                                                         boolean makeTarget);
329
330        /**
331         * Returns the number of actions required to completely
332         * activate this social. A value of 0.0 means perform
333         * instantly.  This method only applies when the user
334         * is not in combat.
335         * @see Social#combatActionsCost(MOB, List)
336         * @see Social#checkedActionsCost(MOB, List)
337         * @param mob the mob executing the social, if any
338         * @param cmds the parameters to be passed to the social, if any
339         * @return the number of player free actions required to do this
340         */
341        public double actionsCost(final MOB mob, final List<String> cmds);
342        /**
343         * Returns the number of actions required to completely
344         * activate this social. A value of 0.0 means perform
345         * instantly.  This method only applies when the user
346         * is fighting in social.
347         * @see Social#actionsCost(MOB, List)
348         * @see Social#checkedActionsCost(MOB, List)
349         * @param mob the mob executing the social, if any
350         * @param cmds the parameters to be passed to the social, if any
351         * @return the number of player free actions required to do this
352         */
353        public double combatActionsCost(final MOB mob, final List<String> cmds);
354        /**
355         * Returns the number of actions required to completely
356         * activate this social. A value of 0.0 means perform
357         * instantly.  This method only should check whether the
358         * user is in combat and return a proper value.
359         * @see Social#combatActionsCost(MOB, List)
360         * @see Social#actionsCost(MOB, List)
361         * @param mob the mob executing the social, if any
362         * @param cmds the parameters to be passed to the social, if any
363         * @return the number of player free actions required to do this
364         */
365        public double checkedActionsCost(final MOB mob, final List<String> cmds);
