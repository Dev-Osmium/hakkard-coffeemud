034/**
035 * A class for holding CoffeeMud-style room IDs of the
036 * form AreaName#[NUMBER], for instance, Midgaard#3001
037 * This class is totally awesome because it stores them
038 * in an efficient way (holding room ids 5-10 not as
039 * 5,6,7,8,9,10, but as 5,10), but sorts them for quick
040 * reads.
041 *
042 * Stores the internal numbers using LongSet
043 * @see com.planet_ink.coffee_mud.core.collections.LongSet
044 */
045public interface RoomnumberSet extends CMCommon
046{
047        /**
048         * Returns the number of room ids stored for the
049         * given Area name.
050         * @param areaName the Area to count the rooms of
051         * @return the number of room ids in the area
052         */
053        public int roomCount(String areaName);
054
055        /**
056         * Returns the total number of room ids stored here
057         * @return the total number of room ids stored here
058         */
059        public int roomCountAllAreas();
060
061        /**
062         * Returns whether any rooms at all are defined.
063         * @return true if none are defined, false otherwise
064         */
065        public boolean isEmpty();
066
067        /**
068         * Returns whether the given room id is stored here
069         * @param str the room id to look for
070         * @return true if the given room id is stored here
071         */
072        public boolean contains(String str);
073
074        /**
075         * Converts the contents of this object into an xml
076         * document.
077         * @return the contents of this object as an xml
078         */
079        public String xml();
080
081        /**
082         * Restores this object from an xml document
083         * @param xml contents for this object as an xml
084         */
085        public void parseXML(String xml);
086
087        /**
088         * Adds the given room id to this object
089         * @param str a room id
090         */
091        public void add(String str);
092
093        /**
094         * Adds a set of room ids to this object
095         * @param set the room ids to add
096         */
097        public void add(RoomnumberSet set);
098
099        /**
100         * Removes a single room id from this object
101         * @param str the room id to remove from this object
102         */
103        public void remove(String str);
104
105        /**
106         * Returns a random, fully qualified room id from those
107         * stored in here.  Includes Area name.
108         * @return a random full room id
109         */
110        public String random();
111
112        /**
113         * Returns an enumerator for all room ids in this object
114         * @return an enumerator for all room ids in this object
115         */
116        public Enumeration<String> getRoomIDs();
117
118        /**
119         * Returns an enumerator for all area names in this object
120         * @return an enumerator for all area names in this object
121         */
122        public Iterator<String> getAreaNames();
123
124        /**
125         * Returns the number parts of the room ids stored in this
126         * object for a given area.
127         * @see com.planet_ink.coffee_mud.core.collections.LongSet
128         * @param areaName the area name to look for
129         * @return a set of numbers.
130         */
131        public LongSet getGrouper(String areaName);
132}
