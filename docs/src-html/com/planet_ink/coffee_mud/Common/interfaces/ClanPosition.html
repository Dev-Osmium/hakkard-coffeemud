035/**
036 * A class for the characteristics of a position within a
037 * clan government.
038 *
039 * @author Bo Zimmerman
040 */
041public interface ClanPosition extends Modifiable, CMCommon
042{
043
044        /**
045         * Gets the iD.
046         *
047         * @see ClanPosition#setID(String)
048         * @return the iD
049         */
050        public String getID();
051
052        /**
053         * Sets the iD.
054         *
055         * @see ClanPosition#getID()
056         * @param iD the new iD
057         */
058        public void setID(String iD);
059
060        /**
061         * Gets the role id.
062         *
063         * @see ClanPosition#setRoleID(int)
064         * @return the role id
065         */
066        public int getRoleID();
067
068        /**
069         * Sets the role id.
070         *
071         * @see ClanPosition#getRoleID()
072         * @param roleID the new role id
073         */
074        public void setRoleID(int roleID);
075
076        /**
077         * Gets the rank.
078         *
079         * @see ClanPosition#setRank(int)
080         * @return the rank
081         */
082        public int getRank();
083
084        /**
085         * Sets the rank.
086         *
087         * @see ClanPosition#getRank()
088         * @param rank the new rank
089         */
090        public void setRank(int rank);
091
092        /**
093         * Gets the name.
094         *
095         * @see ClanPosition#setName(String)
096         * @return the name
097         */
098        public String getName();
099
100        /**
101         * Sets the name.
102         *
103         * @see ClanPosition#getName()
104         * @param name the new name
105         */
106        public void setName(String name);
107
108        /**
109         * Gets the plural name.
110         *
111         * @see ClanPosition#setPluralName(String)
112         * @return the plural name
113         */
114        public String getPluralName();
115
116        /**
117         * Sets the plural name.
118         *
119         * @see ClanPosition#getPluralName()
120         * @param pluralName the new plural name
121         */
122        public void setPluralName(String pluralName);
123
124        /**
125         * Gets the max.
126         *
127         * @see ClanPosition#setMax(int)
128         * @return the max
129         */
130        public int getMax();
131
132        /**
133         * Sets the max.
134         *
135         * @see ClanPosition#getMax()
136         * @param max the new max
137         */
138        public void setMax(int max);
139
140        /**
141         * Gets the inner mask str.
142         *
143         * @see ClanPosition#setInnerMaskStr(String)
144         * @return the inner mask str
145         */
146        public String getInnerMaskStr();
147
148        /**
149         * Sets the inner mask str.
150         *
151         * @see ClanPosition#getInnerMaskStr()
152         * @param innerMaskStr the new inner mask str
153         */
154        public void setInnerMaskStr(String innerMaskStr);
155
156        /**
157         * Checks if is public.
158         *
159         * @see ClanPosition#setPublic(boolean)
160         * @return true, if is public
161         */
162        public boolean isPublic();
163
164        /**
165         * Sets the public.
166         *
167         * @see ClanPosition#isPublic()
168         * @param isPublic the new public
169         */
170        public void setPublic(boolean isPublic);
171
172        /**
173         * Gets the function chart.
174         *
175         * @see ClanPosition#setFunctionChart(com.planet_ink.coffee_mud.Common.interfaces.Clan.Authority[])
176         * @return the function chart
177         */
178        public Clan.Authority[] getFunctionChart();
179
180        /**
181         * Sets the function chart.
182         *
183         * @see ClanPosition#getFunctionChart()
184         * @param functionChart the new function chart
185         */
186        public void setFunctionChart(Clan.Authority[] functionChart);
187}
