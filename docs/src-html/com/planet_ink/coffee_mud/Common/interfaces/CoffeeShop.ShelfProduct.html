036/**
037 * A CoffeeShop is an object for storing the inventory of a shopkeeper, banker,
038 * auctionhouse, merchant, or other object that implements the ShopKeeper interface
039 * for the purpose of selling goods and services.
040 *
041 * ShopKeepers maintain two types of inventory, the base inventory, and the stock
042 * inventory. The stock or store inventory is the list of items the shopkeeper
043 * currently has for sale, the amounts, base prices, etc.
044 * The base inventory is used only for shopkeepers who only buy things like
045 * they originally had in stock, and so the base inventory is always populated with
046 * a single copy of the original store inventory, to be used as a base of comparison
047 * for situations where the stock is empty, but someone is wanting to sell.
048 *
049 * @see com.planet_ink.coffee_mud.core.interfaces.ShopKeeper
050 * @see com.planet_ink.coffee_mud.core.interfaces.ShopKeeper#isSold(int)
051 * @see com.planet_ink.coffee_mud.core.interfaces.ShopKeeper#DEAL_INVENTORYONLY
052 */
053public interface CoffeeShop extends CMCommon
054{
055        /**
056         * Returns whether an item sufficiently like the given item originally
057         * existed in this shops inventory when it was created.  Applies only
058         * to shops where their whatIsSold method returns ONLY_INVENTORY
059         * @see com.planet_ink.coffee_mud.core.interfaces.ShopKeeper#isSold(int)
060         * @see com.planet_ink.coffee_mud.core.interfaces.ShopKeeper#DEAL_INVENTORYONLY
061         * @param thisThang the thing to compare against the base inventory
062         * @return whether the item, or one just like it, is in the base inventory
063         */
064        public boolean inEnumerableInventory(Environmental thisThang);
065
066        /**
067         * Adds a new item to the store inventory.  Use this method when an item is sold
068         * to the store, as pricing and other information will have to be derived.
069         * @see com.planet_ink.coffee_mud.Common.interfaces.CoffeeShop#addStoreInventory(Environmental, int, int)
070         * @param thisThang the thing to sell
071         * @return the core store inventory item added
072         */
073        public Environmental addStoreInventory(Environmental thisThang);
074
075        /**
076         * Returns the number of items in the stores base inventory.  Only really useful
077         * for historical reasons, or if the shop sells inventory only.
078         * @see com.planet_ink.coffee_mud.core.interfaces.ShopKeeper#isSold(int)
079         * @see com.planet_ink.coffee_mud.core.interfaces.ShopKeeper#DEAL_INVENTORYONLY
080         * @return the number of items in the base inventory
081         */
082        public int enumerableStockSize();
083
084        /**
085         * Returns the number of items this shop currently has for sale.  Does not
086         * take number of duplicates into account.  For that call totalStockSizeIncludingDuplicates
087         * @see com.planet_ink.coffee_mud.Common.interfaces.CoffeeShop#totalStockSizeIncludingDuplicates()
088         * @return the number of items for sale.
089         */
090        public int totalStockSize();
091
092        /**
093         * Destroys all the items in this shop.
094         */
095        public void destroyStoreInventory();
096
097        /**
098         * Remove the specific give shelf product from the official stores.
099         * @param P the shelf product to remove permanently
100         */
101        public void deleteShelfProduct(final ShelfProduct P);
102
103        /**
104         * Returns a iterator of all the Environmental objects this shop has for sale.
105         * Will only return one of each item, even if multiple are available.
106         * @return a iterator of objects for sale.
107         */
108        public Iterator<Environmental> getStoreInventory();
109
110        /**
111         * Returns a iterator of all the ShelfProduct objects this shop has for sale.
112         * @return a iterator of all the shelves at the shop.
113         */
114        public Iterator<ShelfProduct> getStoreShelves();
115
116        /**
117         * Returns a iterator of all the Environmental objects this shop has for sale
118         * which match the given search string.
119         * Will only return one of each item, even if multiple are available.
120         * @param srchStr the item to hunt for.
121         * @return a iterator of objects for sale.
122         */
123        public Iterator<Environmental> getStoreInventory(String srchStr);
124
125        /**
126         * Returns a iterator of all the Environmental objects this shop has in its base
127         * inventory.  Only useful for historical reasons, or if the shop sells inventory
128         * only.
129         * @see com.planet_ink.coffee_mud.core.interfaces.ShopKeeper#isSold(int)
130         * @see com.planet_ink.coffee_mud.core.interfaces.ShopKeeper#DEAL_INVENTORYONLY
131         * @return a iterator of objects in base inventory
132         */
133        public Iterator<Environmental> getEnumerableInventory();
134
135        /**
136         * Clears both the base and stock/store inventories.
137         */
138        public void emptyAllShelves();
139
140        /**
141         * Adds a new item to the store inventory so the shopkeeper can sell it.  All items
142         * added go cumulatively into the store inventory, and one copy is kept in the
143         * base inventory for historical reasons.  The method is called when multiple items
144         * need to be added, or if the price is available.  This method is usually used to
145         * build an original shop inventory.
146         * @param thisThang the item/mob/ability to sell
147         * @param number the number of items to sell
148         * @param price the price of the item (in base currency) or -1 to have it determined
149         * @return the actual object stored in the inventory
150         */
151        public Environmental addStoreInventory(Environmental thisThang, int number, int price);
152
153        /**
154         * Total weight, in pounds, of all items in the store inventory, taking number in
155         * stock into account.
156         * @return the total weight in pounds
157         */
158        public int totalStockWeight();
159
160        /**
161         * The number of items in the store inventory, taking number in stock into account.
162         * Call this method to see how crowded the shop really is, as opposed to totalStockSize.
163         * @see com.planet_ink.coffee_mud.Common.interfaces.CoffeeShop#totalStockSize()
164         * @return the total number of all items in stock
165         */
166        public int totalStockSizeIncludingDuplicates();
167
168        /**
169         * Removes all items like the given item from the base and store inventory.
170         * @see com.planet_ink.coffee_mud.core.interfaces.ShopKeeper#isSold(int)
171         * @param thisThang the item like which to remove
172         */
173        public void delAllStoreInventory(Environmental thisThang);
174
175        /**
176         * Returns whether an item with the given name is presently in this stores
177         * stock inventory, and available for sale.
178         * @see com.planet_ink.coffee_mud.core.interfaces.ShopKeeper#isSold(int)
179         * @param name the name of the item to search for
180         * @param mob the mob who is interested (stock can differ depending on customer)
181         * @return whether the item is available
182         */
183        public boolean doIHaveThisInStock(String name, MOB mob);
184
185        /**
186         * Returns the base stock price (not the final price by any means) that the shop
187         * will use as a foundation for determining the given items price.  -1 would mean
188         * that the shopkeeper uses the valuation of the item as a basis, whereas another
189         * value is in base gold.  Best to get likeThis item from the getStoreInventory()
190         * @see com.planet_ink.coffee_mud.Common.interfaces.CoffeeShop#getStoreInventory()
191         * @param likeThis the item like which to compare
192         * @return the stock price of the item given.
193         */
194        public int stockPrice(Environmental likeThis);
195
196        /**
197         * Returns the number of items like the one given that the shopkeeper presently
198         * has in stock and available for sale.
199         * @see com.planet_ink.coffee_mud.Common.interfaces.CoffeeShop#getStoreInventory()
200         * @param likeThis the item like which to compare
201         * @return the number currently in stock.
202         */
203        public int numberInStock(Environmental likeThis);
204
205        /**
206         * Searches this shops stock of items for sale for one matching the given name.
207         * @see com.planet_ink.coffee_mud.Common.interfaces.CoffeeShop#getStoreInventory()
208         * @param name the name of the item to search for
209         * @param mob the mob who is interested (stock can differ depending on customer)
210         * @return the available item, if found
211         */
212        public Environmental getStock(String name, MOB mob);
213
214        /**
215         * Searches this shops stock of items for sale for one matching the given name.
216         * If one is found, it copies the item, removes one from the available stock, and
217         * returns the copy.
218         * @see com.planet_ink.coffee_mud.Common.interfaces.CoffeeShop#getStoreInventory()
219         * @param name the name of the item to search for
220         * @param mob the mob who is interested (stock can differ depending on customer)
221         * @return the available item, if found
222         */
223        public Environmental removeStock(String name, MOB mob);
224
225        /**
226         * Searches this shops stock of items for sale for one matching the given name.
227         * If one is found, then one unit is removed from one of the available stock and
228         * true is returned.
229         * @param name the name of the item to lower the stock of
230         * @return true if it was lowered
231         */
232        public boolean lowerStock(String name);
233
234        /**
235         * Searches this shops stock of items for sale for one matching the given name.
236         * If one is found, it copies the item, removes one from the available stock, and
237         * prepares it for sale by adding it to a list along with any necessary accessories,
238         * such as necessary keys, or if a container, any contents of the container.
239         * @see com.planet_ink.coffee_mud.Common.interfaces.CoffeeShop#getStoreInventory()
240         * @param named the name of the item to search for
241         * @param mob the mob who is interested (stock can differ depending on customer)
242         * @return the available items, if found, as a list of Environmental objects
243         */
244        public List<Environmental> removeSellableProduct(String named, MOB mob);
245
246        /**
247         * Generates an XML document of all available shop inventory, prices, and availability.
248         * @see com.planet_ink.coffee_mud.Common.interfaces.CoffeeShop#getStoreInventory()
249         * @see com.planet_ink.coffee_mud.Common.interfaces.CoffeeShop#buildShopFromXML(String)
250         * @return an XML document.
251         */
252        public String makeXML();
253
254        /**
255         * Repopulates this shop inventory from a given xml document, restoring store inventory,
256         * base inventory, prices, and availability.
257         * @see com.planet_ink.coffee_mud.Common.interfaces.CoffeeShop#makeXML()
258         * @param text the xml document to restore from
259         */
260        public void buildShopFromXML(String text);
261
262        /**
263         * A method for quickly making wholesale changes to a shopkeepers inventory.
264         * getStoreInventory should be called to get the list of items.  The items can
265         * then be modified, and this method called to properly "resubmit" them to
266         * the shopkeeper.
267         * @param shopItems the items for inventory
268         */
269        public void resubmitInventory(List<Environmental> shopItems);
270
271        /**
272         * Initializes this shop object with its host ShopKeeper
273         * @param SK the shopkeeper that hosts this object
274         * @return always this
275         */
276        public CoffeeShop build(ShopKeeper SK);
277
278        /**
279         * Returns the shopKeeper that is hosting this shop
280         * @return the shopKeeper that is hosting this shop
281         */
282        public ShopKeeper shopKeeper();
283
284        /**
285         * Returns a thin copy with independent lists, but the
286         * same items as the original host.
287         * @return the thin copy
288         */
289        public CoffeeShop weakCopyOf();
290
291        /**
292         * Returns whether the whatIsSold code applies to the shopkeeper hosting this shop.
293         * @see ShopKeeper#DEAL_DESCS
294         * @param code the whatIsSold code
295         * @return whether the whatIsSold code applies to the shopkeeper hosting this shop.
296         */
297        public boolean isSold(int code);
298
299        /**
300         * Class for representing a shelf product, holding
301         * an item prototype, the number in stock, and the
302         * price.  A price of -1 means to use the items
303         * calculated value (common).
304         * @author Bo Zimmerman
305         */
306        public static class ShelfProduct
307        {
308                public Environmental product;
309                public int number;
310                public int price;
311                public ShelfProduct(Environmental E, int number, int price)
312                {
313                        this.product=E;this.number=number;this.price=price;
314                }
315                @Override
316                public int hashCode()
317                {
318                        return product.hashCode() ^ number ^ price;
319                }
320        }
321        
322        /**
323         * Returns a hash of all the items currently in the
324         * shop, for help in determining when things change.
325         * @return the hash of the contents.
326         */
327        public long contentHash();
328}
