038/**
039 * The interface for the main CoffeeMud scripting engine, which implements
040 * a scripting engine descended from the old mud codebases of the 90's
041 * usually called MOBPROG.  Its main features include easy to understand
042 * event-oriented triggers, making all mud commands implicit scripting
043 * commands.  It also includes methods for embedding javascript.
044 *
045 * @see com.planet_ink.coffee_mud.Behaviors.Scriptable
046 */
047public interface ScriptingEngine extends CMCommon, Tickable, MsgListener
048{
049        /**
050         * Executes a script in response to an event
051         * The scripts are formatted as a 2 dimensional DVector
052         * with the first row being the trigger information.  Each
053         * row consists of the String command, and a parsed String[]
054         * array as dimension 2.
055         * @see com.planet_ink.coffee_mud.Common.interfaces.ScriptingEngine.ScriptableResponse
056         * @param scripted the object that is scripted
057         * @param source the source of the event
058         * @param target the target of the event
059         * @param monster a mob representation of the scripted object
060         * @param primaryItem an item involved in the event
061         * @param secondaryItem a second item involved in the event
062         * @param script 2 dimensional DVector, the script to execute
063         * @param msg a string message associated with the event
064         * @param tmp miscellaneous local variables
065         * @return N/A
066         */
067        public String execute(PhysicalAgent scripted,
068                                                  MOB source,
069                                                  Environmental target,
070                                                  MOB monster,
071                                                  Item primaryItem,
072                                                  Item secondaryItem,
073                                                  DVector script,
074                                                  String msg,
075                                                  Object[] tmp);
076
077        /**
078         * Uses this scripting engines variable parsing system to replace
079         * any script variables $XXXX with their script determined values.
080         * This is a powerful mechanism for getting at the script functions
081         * in order to access stat data about specific objects, do math, etc.
082         *
083         * @param source the source of the event
084         * @param target the target of the event
085         * @param scripted the object that is scripted
086         * @param monster a mob representation of the scripted object
087         * @param primaryItem an item involved in the event
088         * @param secondaryItem a second item involved in the event
089         * @param msg a string message associated with the event
090         * @param tmp miscellaneous local variables
091         * @param varifyable the string to parse
092         * @return N/A
093         */
094        public String varify(MOB source,
095                                                 Environmental target,
096                                                 PhysicalAgent scripted,
097                                                 MOB monster,
098                                                 Item primaryItem,
099                                                 Item secondaryItem,
100                                                 String msg,
101                                                 Object[] tmp,
102                                                 String varifyable);
103        /**
104         * Forces any queued event responses to be immediately
105         * executed.
106         */
107        public void dequeResponses();
108
109        /**
110         * Creates a mob from the Tickable object sent, possibly saving it
111         * locally to this object for use later.  If the object is a mob,
112         * it returns the mob.  Otherwise, it makes a fake one.
113         * @param ticking the scripted object to make a fake mob out of
114         * @return a mob from a tickable
115         */
116        public MOB getMakeMOB(Tickable ticking);

118        /**
119         * Receives a string for evaluation by the eval function, and stores
120         * it as the first element in the given 2 dimensional string array.
121         * @param evaluable the eval expression
122         * @return EVAL the 1 dimensional array to hold the compiled eval
123         * @throws ScriptParseException a parse error
124         */
125        public String[] parseEval(String evaluable) throws ScriptParseException;
126
127        /**
128         * Evaluates a scripting function.  Is called by the execute command
129         * to resolve IF, WHILE, and similar expressions that utilize the MOBPROG
130         * functions.  The expressions are passed in as a String array stored
131         * in a single string array entry (for replacement) in element 0.
132         * @param scripted the object that is scripted
133         * @param source the source of the event
134         * @param target the target of the event
135         * @param monster a mob representation of the scripted object
136         * @param primaryItem an item involved in the event
137         * @param secondaryItem a second item involved in the event
138         * @param msg a string message associated with the event
139         * @param tmp miscellaneous local variables
140         * @param eval the pre-parsed expression
141         * @param startEval while line to start evaluating on.
142         * @return true if the expression is true, false otherwise.
143         */
144        public boolean eval(PhysicalAgent scripted,
145                                                MOB source,
146                                                Environmental target,
147                                                MOB monster,
148                                                Item primaryItem,
149                                                Item secondaryItem,
150                                                String msg,
151                                                Object[] tmp,
152                                                String[][] eval,
153                                                int startEval);
154

156         * Evaluates one of the boolean functions as a string
157         * variable expression, which gives different and
158         * informative results.  See the Green Table in the
159         * Scripting Guide.
160         * @param scripted the object that is scripted
161         * @param source the source of the event
162         * @param target the target of the event
163         * @param monster a mob representation of the scripted object
164         * @param primaryItem an item involved in the event
165         * @param secondaryItem a second item involved in the event
166         * @param msg a string message associated with the event
167         * @param tmp miscellaneous local variables
168         * @param evaluable the function expression
169         * @return the results of the function expression
170         */

172                                                        MOB source,
173                                                        Environmental target,
174                                                        MOB monster,
175                                                        Item primaryItem,
176                                                        Item secondaryItem,
177                                                        String msg,
178                                                        Object[] tmp,
179                                                        String evaluable);

341        /**
342         * An object that holds the information about an event until it is
343         * time to execute its associated script.
344         * @author Bo Zimmerman
345         */
346        public static class ScriptableResponse
347        {
348                private int tickDelay=0;
349                /** (host) the object being scripted */
350                public PhysicalAgent h=null;
351                /** (source) the source of the event */
352                public MOB s=null;
353                /** (target) the target of the event */
354                public Environmental t=null;
355                /** (host mob) a mob representation of the host (h)*/
356                public MOB m=null;
357                /** (primary item) an item associated with this event */
358                public Item pi=null;
359                /** (second item) a second item associated with this event */
360                public Item si=null;
361                /** (script) the actual script to execute for this event */
362                public DVector scr;
363                /** a string associated with this event */
364                public String message=null;
365
366                /**
367                 * Create an event response object
368                 * @param host the object being scripted
369                 * @param source the source of the event
370                 * @param target the target of the event
371                 * @param monster a mob representation of the host
372                 * @param primaryItem an item associated with this event
373                 * @param secondaryItem a second item associated with this event
374                 * @param script the actual script to execute for this event
375                 * @param ticks how many ticks to wait before executing the script
376                 * @param msg a string associated with this event
377                 */
378                public ScriptableResponse(PhysicalAgent host,
379                                                                  MOB source,
380                                                                  Environmental target,
381                                                                  MOB monster,
382                                                                  Item primaryItem,
383                                                                  Item secondaryItem,
384                                                                  DVector script,
385                                                                  int ticks,
386                                                                  String msg)
387                {
388                        h=host;
389                        s=source;
390                        t=target;
391                        m=monster;
392                        pi=primaryItem;
393                        si=secondaryItem;
394                        scr=script;
395                        tickDelay=ticks;
396                        message=msg;
397                }
398
399                /**
400                 * Decrements the internal tick counter and returns true if
401                 * the tick counter has dropped to or below 0
402                 * @return true if its time to execute
403                 */
404                public boolean checkTimeToExecute()
405                {
406                        return ((--tickDelay) <= 0);
407                }
408        }
