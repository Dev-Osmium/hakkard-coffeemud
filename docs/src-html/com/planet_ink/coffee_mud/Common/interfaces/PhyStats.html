033/**
034 * The PhyStats interface is a state object that holds some basic information about
035 * just about every Physical object in the game.  Not all stats are relevant for
036 * ALL Physicals, but most, especially the big ones.
037 *
038 * Physicals always keep two instances of this object, a base one, representing
039 * his base unmodified state, and current one, representing his state after spells
040 * and other affects have had their say.
041 * @see com.planet_ink.coffee_mud.core.interfaces.Modifiable
042 * @see com.planet_ink.coffee_mud.core.interfaces.Affectable#basePhyStats()
043 * @see com.planet_ink.coffee_mud.core.interfaces.Affectable#phyStats()
044 */
045public interface PhyStats extends CMCommon, Modifiable
046{
047        /** Constant shortcut for setting the rejuvenation rate on a mob so that it will NOT rejuv. */
048        public static final int NO_REJUV=Integer.MAX_VALUE;
049
050        /**
051         * Returns a bitmask for sense related flags of mobs (CAN_ constants), or
052         * miscellaneous runtime flags for items and other Physicals (SENSE_ constants).
053         * @see com.planet_ink.coffee_mud.Common.interfaces.PhyStats#SENSE_ITEMNOREMOVE
054         * @see com.planet_ink.coffee_mud.Common.interfaces.PhyStats#CAN_NOT_SEE
055         * @see com.planet_ink.coffee_mud.Common.interfaces.PhyStats#setSensesMask(int)
056         * @return a bitmask made up of SENSE_ constants or CAN_ constants
057         */
058        public int sensesMask(); // mobs, run-time items
059
060        /**
061         * Sets a bitmask for sense related flags of mobs (CAN_ constants), or
062         * miscellaneous runtime flags for items and other Physicals (SENSE_ constants).
063         * @see com.planet_ink.coffee_mud.Common.interfaces.PhyStats#SENSE_ITEMNOREMOVE
064         * @see com.planet_ink.coffee_mud.Common.interfaces.PhyStats#CAN_NOT_SEE
065         * @see com.planet_ink.coffee_mud.Common.interfaces.PhyStats#setSensesMask(int)
066         * @param newMask a bitmask made up of SENSE_ constants or CAN_ constants
067         */
068        public void setSensesMask(int newMask);
069
070        /**
071         * Returns a bitmask for disposition related flags of Physicals (IS_ constants).
072         * @see com.planet_ink.coffee_mud.Common.interfaces.PhyStats#IS_BONUS
073         * @see com.planet_ink.coffee_mud.Common.interfaces.PhyStats#setDisposition(int)
074         * @return a bitmask of disposition related flags
075         */
076        public int disposition(); // items, mobs
077
078        /**
079         * Sets a bitmask for disposition related flags of Physicals (IS_ constants).
080         * @see com.planet_ink.coffee_mud.Common.interfaces.PhyStats#IS_BONUS

083         */
084        public void setDisposition(int newDisposition);
085

089         * @return the experience level
090         */
091        public int level(); // items, exits, mobs
092
093        /**
094         * Sets the experience level of the mob, item, exit, Physical.
095         * @see com.planet_ink.coffee_mud.Common.interfaces.PhyStats#level()
096         * @param newLevel the new experience level
097         */
098        public void setLevel(int newLevel);
099
100        /**
101         * Returns the ability level (a secondary level, e.g. if magical, how much?)

103         * it designates a hit point multiplier.

106         */
107        public int ability(); // items, mobs
108

113         * @see com.planet_ink.coffee_mud.Common.interfaces.PhyStats#ability()
114         * @param newAdjustment the new ability level (or misc integer)
115         */
116        public void setAbility(int newAdjustment);
117
118        /**
119         * Returns the number of ticks before a Physical removed from the game
120         * (due to death, destruction, or just removal from home) is restored.
121         * @see com.planet_ink.coffee_mud.Common.interfaces.PhyStats#setRejuv(int)

123         */

125

127         * Sets the number of ticks before a Physical removed from the game

250         * Returns a list of ambiances (extra words, visible fields) that are tacked

258

260         * Check to see if the given ambiance exists in the list.
261         * @see com.planet_ink.coffee_mud.Common.interfaces.PhyStats#addAmbiance(String)
262         * @see com.planet_ink.coffee_mud.Common.interfaces.PhyStats#delAmbiance(String)
263         * @see com.planet_ink.coffee_mud.Common.interfaces.PhyStats#ambiances()
264         * @param ambiance the ambiance to look for
265         * @return true if its in there, false otherwise
266         */
267        public boolean isAmbiance(String ambiance);
268
269        /**
270         * Adds an ambiance (extra word, visible field) to the list that are tacked

272         * @see com.planet_ink.coffee_mud.Common.interfaces.PhyStats#ambiances()
273         * @see com.planet_ink.coffee_mud.Common.interfaces.PhyStats#delAmbiance(String)
274         * @see com.planet_ink.coffee_mud.Common.interfaces.PhyStats#isAmbiance(String)
275         * @param ambiance a new ambiance string
276         */
277        public void addAmbiance(String ambiance);
278
279        /**
280         * Removes an ambiance (extra word, visible field) from the list that are tacked
281         * onto the display text of this Physical.
282         * @see com.planet_ink.coffee_mud.Common.interfaces.PhyStats#ambiances()
283         * @see com.planet_ink.coffee_mud.Common.interfaces.PhyStats#addAmbiance(String)
284         * @see com.planet_ink.coffee_mud.Common.interfaces.PhyStats#isAmbiance(String)
285         * @param ambiance the old ambiance string
286         */
287        public void delAmbiance(String ambiance);
288
289        /**
290         * Returns a single string summary of several important stats from this object.
291         * @return a single string summary of several important stats from this object.
292         */
293        public String getCombatStats();
294
295        /**
296         * Returns whether the given object is substantially the same as this one
297         * @param E the PhyStats to compare to
298         * @return whether or not they are the same
299         */
300        public boolean sameAs(PhyStats E);
301
302        /**
303         * Copies the internal data of this object into another of kind.
304         * @param intoStats another PhyStats object.
305         */
306        public void copyInto(PhyStats intoStats);
307
308        /**
309         * Sets all the stats in this object to the given value
310         * @param def a value to set all the stats to
311         */
312        public void setAllValues(int def);
313
314        /**
315         * Resets all the stats in this object to their factory defaults.
316         */
317        public void reset();
