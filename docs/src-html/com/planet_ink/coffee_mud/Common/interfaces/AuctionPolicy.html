020import com.planet_ink.coffee_mud.MOBS.interfaces.Auctioneer;
021
022/**
023 * Class for storing basic Auction House policies.  
024 * @author Bo Zimmerman
025 */
026public interface AuctionPolicy extends CMCommon
027{
028        /**
029         * Gets the flat fee in base currency to list an 
030         * item of any sort at the auction house for a timed
031         * auction.
032         * @see AuctionPolicy#setTimedListingPrice(double)
033         * @return the flat fee in base currency to list
034         */
035        public double timedListingPrice();
036
037        /**
038         * Sets the flat fee in base currency to list an 
039         * item of any sort at the auction house for a timed
040         * auction.
041         * @see AuctionPolicy#timedListingPrice()
042         * @param d the flat fee in base currency to list
043         */
044        public void setTimedListingPrice(double d);
045
046        /**
047         * Gets the percent of an items value, per day,
048         * to charge to list an item. 0.0-1.0 for a timed
049         * auction.
050         * @see AuctionPolicy#setTimedListingPct(double)
051         * @return  the percent of an items value, per day
052         */
053        public double timedListingPct();
054        
055        /**
056         * Sets the percent of an items value, per day,
057         * to charge to list an item. 0.0-1.0for a timed
058         * auction.
059         * @see AuctionPolicy#timedListingPct()
060         * @param d the percent of an items value, per day
061         */
062        public void setTimedListingPct(double d);
063
064        /**
065         * Gets the percent of an items final value
066         * to take off the winning bid for the house
067         * for a timed auction. 0.0-1.0
068         * @see AuctionPolicy#setTimedFinalCutPct(double) 
069         * @return the percent of an items final value
070         */
071        public double timedFinalCutPct();
072        
073        /**
074         * Gets the percent of an items final value
075         * to take off the winning bid for the house.
076         * for a timed auction. 0.0-1.0
077         * @see AuctionPolicy#timedFinalCutPct() 
078         * @param d the percent of an items final value
079         */
080        public void setTimedFinalCutPct(double d);
081
082        /**
083         * Gets the maximum number of game-days that 
084         * an auction can continue.
085         * @see AuctionPolicy#setMaxTimedAuctionDays(int)
086         * @return the maximum number of game-days
087         */
088        public int maxTimedAuctionDays();
089        
090        /**
091         * Sets the maximum number of game-days that 
092         * an auction can continue.
093         * @see AuctionPolicy#maxTimedAuctionDays()
094         * @param d the maximum number of game-days
095         */
096        public void setMaxTimedAuctionDays(int d);
097
098        /**
099         * Gets the minimum number of game-days that 
100         * an auction can continue.
101         * @see AuctionPolicy#setMinTimedAuctionDays(int)
102         * @return the minimum number of game-days
103         */
104        public int minTimedAuctionDays();
105        
106        /**
107         * Sets the minimum number of game-days that 
108         * an auction can continue.
109         * @see AuctionPolicy#minTimedAuctionDays()
110         * @param d the minimum number of game-days
111         */
112        public void setMinTimedAuctionDays(int d);
113
114        /**
115         * Gets the flat fee in base currency to list an 
116         * item of any sort at the auction house for a live
117         * auction.
118         * @see AuctionPolicy#setLiveListingPrice(double)
119         * @return the flat fee in base currency to list
120         */
121        public double liveListingPrice();
122
123        /**
124         * Sets the flat fee in base currency to list an 
125         * item of any sort at the auction house for a live
126         * auction.
127         * @see AuctionPolicy#liveListingPrice()
128         * @param d the flat fee in base currency to list
129         */
130        public void setLiveListingPrice(double d);
131
132        /**
133         * Gets the percent of an items final value
134         * to take off the winning bid for the house
135         * for a live auction. 0.0-1.0
136         * @see AuctionPolicy#setLiveFinalCutPct(double) 
137         * @return the percent of an items final value
138         */
139        public double liveFinalCutPct();
140        
141        /**
142         * Gets the percent of an items final value
143         * to take off the winning bid for the house.
144         * for a live auction. 0.0-1.0
145         * @see AuctionPolicy#liveFinalCutPct() 
146         * @param d the percent of an items final value
147         */
148        public void setLiveFinalCutPct(double d);
149        
150        /**
151         * Alters this policy to take account of the policies
152         * of the policies of a full timed auctioneer.
153         * @see com.planet_ink.coffee_mud.MOBS.interfaces.Auctioneer 
154         * @param auction the auctioneer
155         */
156        public void mergeAuctioneerPolicy(Auctioneer auction);
157}
