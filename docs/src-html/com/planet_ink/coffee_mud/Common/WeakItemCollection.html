044/**
045 * Abstract collection of item objects, complete with some
046 * finders and various accessors.  Also, the copyOf method
047 * does a deep copy.  Also, this is a weak item collection,
048 * so as items get destroyed, they disappear from here.
049 */
050public class WeakItemCollection implements ItemCollection, CMCommon
051{
052        private final SVector<WeakReference<Item>> innerContents=new SVector<WeakReference<Item>>(0);
053        private final ConvertingList<WeakReference<Item>,Item> contents
054                                = new ConvertingList<WeakReference<Item>,Item>(innerContents,
055                                                new Converter<WeakReference<Item>,Item>()
056                                                {
057                                        @Override
058                                        public Item convert(WeakReference<Item> obj)
059                                        {
060                                                return obj.get();
061                                        }
062                                });
063
064        @Override
065        public String ID()
066        {
067                return "WeakItemCollection";
068        }
069
070        @Override
071        public String name()
072        {
073                return ID();
074        }
075
076        @Override
077        public CMObject copyOf()
078        {
079                final WeakItemCollection c=(WeakItemCollection)newInstance();
080                for(int i=0;i<contents.size();i++)
081                {
082                        final Item I=contents.get(i);
083                        if(I!=null)
084                        {
085                                final Item I2=(Item)I.copyOf();
086                                I2.setOwner(I.owner());
087                                c.innerContents.add(new WeakReference<Item>(I2));
088                        }
089                }
090                for(int i=0;i<contents.size();i++)
091                {
092                        final Item I=contents.get(i);
093                        final Item I2=c.contents.get(i);
094                        if((I!=null)&&(I2!=null))
095                        {
096                                if(I.container() != null)
097                                {
098                                        for(int i2=0;i2<contents.size();i2++)
099                                        {
100                                                if(I.container() == contents.get(i2))
101                                                {
102                                                        I2.setContainer((Container)c.contents.get(i2));
103                                                        break;
104                                                }
105                                        }
106                                }
107                        }
108                }
109                return c;
110        }
111
112        @Override
113        public void initializeClass()
114        {
115        }
116
117        @Override
118        public CMObject newInstance()
119        {
120                return new WeakItemCollection();
121        }
122
123        @Override
124        public int compareTo(final CMObject o)
125        {
126                return o==this?0:1;
127        }
128
129        @Override
130        public Item findItem(String itemID)
131        {
132                Item item=(Item)CMLib.english().fetchEnvironmental(contents,itemID,true);
133                if(item==null)
134                        item=(Item)CMLib.english().fetchEnvironmental(contents,itemID,false);
135                return item;
136        }
137
138        @Override
139        public Enumeration<Item> items()
140        {
141                return new IteratorEnumeration<Item>(contents.iterator());
142        }
143
144        @Override
145        public Item findItem(Item goodLocation, String itemID)
146        {
147                Item item=CMLib.english().fetchAvailableItem(contents,itemID,goodLocation,Wearable.FILTER_ANY,true);
148                if(item==null)
149                        item=CMLib.english().fetchAvailableItem(contents,itemID,goodLocation,Wearable.FILTER_ANY,false);
150                return item;
151        }
152
153        @Override
154        public List<Item> findItems(Item goodLocation, String itemID)
155        {
156                List<Item> items=CMLib.english().fetchAvailableItems(contents,itemID,goodLocation,Wearable.FILTER_ANY,true);
157                if(items.size()==0)
158                        items=CMLib.english().fetchAvailableItems(contents,itemID,goodLocation,Wearable.FILTER_ANY,false);
159                return items;
160        }
161
162        @Override @SuppressWarnings({"unchecked","rawtypes"})
163        public List<Item> findItems(String itemID)
164        {
165                List items=CMLib.english().fetchEnvironmentals(contents,itemID,true);
166                if(items.size()==0)
167                        items=CMLib.english().fetchEnvironmentals(contents,itemID, false);
168                return items;
169        }
170
171        @Override
172        public void addItem(Item item)
173        {
174                if((item!=null)&&(!item.amDestroyed()))
175                        innerContents.addElement(new WeakReference<Item>(item));
176        }
177
178        @Override
179        public void delItem(Item item)
180        {
181                for(int i=contents.size()-1;i>=0;i--)
182                {
183                        final Item I=getItem(i);
184                        if(I==item)
185                        {
186                                innerContents.remove(i);
187                                break;
188                        }
189                        else
190                        if(I==null)
191                        {
192                                try
193                                {
194                                        innerContents.remove(i);
195                                }
196                                catch (final java.lang.ArrayIndexOutOfBoundsException x)
197                                {
198                                }
199                        }
200                }
201        }
202
203        @Override
204        public void delAllItems(boolean destroy)
205        {
206                if(destroy)
207                {
208                        for(int i=numItems()-1;i>=0;i--)
209                        {
210                                final Item I=getItem(i);
211                                if(I!=null)
212                                        I.destroy();
213                        }
214                }
215                innerContents.clear();
216        }
217
218        @Override
219        public int numItems()
220        {
221                return contents.size();
222        }
223
224        @Override
225        public boolean isContent(Item item)
226        {
227                for(int i=contents.size()-1;i>=0;i--)
228                {
229                        final Item I=getItem(i);
230                        if(I==item)
231                                return true;
232                        if(I==null)
233                        {
234                                try
235                                {
236                                         innerContents.remove(i); 
237                                }
238                                catch(final java.lang.ArrayIndexOutOfBoundsException x)
239                                {
240                                }
241                        }
242                }
243                return false;
244        }
245
246        @Override
247        public Item getItem(int i)
248        {
249                try
250                {
251                        final Item I=contents.get(i);
252                        if(I==null)
253                                innerContents.remove(i);
254                        else
255                                return I;
256                }
257                catch(final java.lang.ArrayIndexOutOfBoundsException x)
258                {
259                }
260                return null;
261        }
262
263        @Override
264        public void eachItem(final EachApplicable<Item> applier)
265        {
266                final List<Item> contents=this.contents;
267                if(contents!=null)
268                try
269                {
270                        for(int a=0;a<contents.size();a++)
271                        {
272                                final Item I=contents.get(a);
273                                if(I!=null)
274                                        applier.apply(I);
275                                else
276                                {
277                                        try
278                                        {
279                                                 innerContents.remove(a); 
280                                        }
281                                        catch(final java.lang.ArrayIndexOutOfBoundsException x)
282                                        {
283                                        }
284                                }
285                        }
286                }
287                catch(final ArrayIndexOutOfBoundsException e)
288                {
289                }
290        }
291
292        @Override
293        public Item getRandomItem()
294        {
295                if(numItems()==0)
296                        return null;
297                return getItem(CMLib.dice().roll(1,numItems(),-1));
298        }
